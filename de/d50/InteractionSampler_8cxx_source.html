17 void InteractionSampler::init()
18 {
19  // (re-)initialize sample and check parameters consistency
20 
21  int nBCSet = mBCFilling.getNBunches();
22  if (!nBCSet) {
23  LOG(WARNING) << "No bunch filling provided, impose default one";
24  mBCFilling.setDefault();
25  nBCSet = mBCFilling.getNBunches();
26  }
27 
28  if (mMuBC < 0. && mIntRate < 0.) {
29  LOG(WARNING) << "No IR or muBC is provided, setting default IR";
30  mIntRate = DefIntRate;
31  }
32 
33  if (mMuBC > 0.) {
34  mIntRate = mMuBC * nBCSet * o2::constants::lhc::LHCRevFreq;
35  LOG(INFO) << "Deducing IR=" << mIntRate << "Hz from " << nBCSet << " BCs at mu=" << mMuBC;
36  } else {
37  mMuBC = mIntRate / (nBCSet * o2::constants::lhc::LHCRevFreq);
38  LOG(INFO) << "Deducing mu=" << mMuBC << " per BC from IR=" << mIntRate << " with " << nBCSet << " BCs";
39  }
40 
41  mBCMin = 0;
42  mBCMax = -1;
43  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
44  if (mBCFilling.testBC(i)) {
45  if (mBCMin > i) {
46  mBCMin = i;
47  }
48  if (mBCMax < i) {
49  mBCMax = i;
50  }
51  }
52  }
53  double muexp = TMath::Exp(-mMuBC);
54  mProbInteraction = 1. - muexp;
55  mMuBCZTRed = mMuBC * muexp / mProbInteraction;
56  mIntBCCache = 0;
57  mIR.bc = mBCMin;
58 }

73 const o2::InteractionTimeRecord& InteractionSampler::generateCollisionTime()
74 {
75  // generate single interaction record
76  if (mIntRate < 0) {
77  init();
78  }
79 
80  if (mIntBCCache < 1) { // do we still have interaction in current BC?
81  mIntBCCache = simulateInteractingBC(); // decide which BC interacts and N collisions
82  }
83  mIR.timeNS = mTimeInBC.back() + mIR.bc2ns();
84  mTimeInBC.pop_back();
85  mIntBCCache--;
86 
87  return mIR;
88 }

91 int InteractionSampler::simulateInteractingBC()
92 {
93  // Returns number of collisions assigned to selected BC
94 
95  do {
96  nextCollidingBC(); // pick next interacting bunch
97  } while (gRandom->Rndm() > mProbInteraction); // skip BCs w/o collisions
98 
99  // once BC is decided, enforce at least one interaction
100  int ncoll = genPoissonZT();
101  // assign random time withing a bunch
102  for (int i = ncoll; i--;) {
103  double tInBC = 0; // tInBC should be in the vicinity of the BC
104  do {
105  tInBC = gRandom->Gaus(0., mBCTimeRMS);
106  } while (std::abs(tInBC) > o2::constants::lhc::LHCBunchSpacingNS / 2.1);
107  mTimeInBC.push_back(tInBC);
108  }
109  if (ncoll > 1) { // sort in DECREASING time order (we are reading vector from the end)
110  std::sort(mTimeInBC.begin(), mTimeInBC.end(), [](const float a, const float b) { return a > b; });
111  }
112  return ncoll;
113 }
