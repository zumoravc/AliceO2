54  };
55  auto tester = [FoodMap](const char* arg) {
56  // use a custom mapper function, this evetually throws an exception if the token is not in the map
57  return std::move(RangeTokenizer::tokenize<Food>(arg, [FoodMap](auto const& token) { return FoodMap.at(token); }));
58  };
59 
60  auto tokens = tester("apple,tomato");
