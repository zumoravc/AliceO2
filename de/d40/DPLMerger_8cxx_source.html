54 // This is a possible implementation of a DPL compliant and generic gatherer whit trivial messages concatenation
55 o2f::DataProcessorSpec defineMerger(std::string devName, o2f::Inputs usrInputs, o2f::OutputSpec usrOutput)
56 {
57  // This lambda retrieves the payload size through the API and back-inserts it on the output buffer
58  auto funcMerge = [](OutputBuffer& buf, const o2f::DataRef d) {
59  auto msgSize = (o2::header::get<o2::header::DataHeader*>(d.header))->payloadSize;
60  buf.resize(buf.size() + msgSize);
61  std::copy(&(d.payload[0]), &(d.payload[msgSize - 1]), std::back_inserter(buf));
62  };
63  // Callling complete implementation
64  return defineMerger(devName, usrInputs, usrOutput, funcMerge);
