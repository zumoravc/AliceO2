40 double ChamberResponse::getFiredProbability(double distance, int cathode, int deId, double theta) const
41 {
43 
44  // Need to convert the distance from cm to mm
45  double distMM = distance * 10.;
46  double parA = mParams.getParA(mHV.getHV(deId));
47  double parB = mParams.getParB(cathode, deId);
48  double parC = mParams.getParC(mHV.getHV(deId));
49  double costheta = std::cos(theta);
50  return (parC + parA / (parA + costheta * std::pow(distMM, parB))) / (1 + parC);
51 }
