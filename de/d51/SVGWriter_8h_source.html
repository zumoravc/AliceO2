71  template <typename T>
72  void polygon(const o2::mch::contour::Polygon<T>& p)
73  {
74  mSVGBuffer << R"(<polygon points=")";
75  auto vertices = getVertices(p);
76  for (auto j = 0; j < vertices.size(); ++j) {
77  auto v = vertices[j];
