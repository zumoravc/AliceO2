57  static constexpr int scaleTimePacked = 64; //< ~50 is needed for 0.1mm precision, but leads to float rounding artifacts around 20ms
58  static constexpr int scalePadPacked = 64; //< ~60 is needed for 0.1mm precision, but power of two avoids rounding

169 inline void ClusterNativeAccess::setOffsetPtrs()
170 {
171  int offset = 0;
172  for (unsigned int i = 0; i < Constants::MAXSECTOR; i++) {
173  nClustersSector[i] = 0;
174  for (unsigned int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
175  clusterOffset[i][j] = offset;
176  clusters[i][j] = &clustersLinear[offset];
177  nClustersSector[i] += nClusters[i][j];
178  offset += nClusters[i][j];
179  }
180  }
181  nClustersTotal = offset;
