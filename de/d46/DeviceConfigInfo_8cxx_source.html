26 // Parses a config entry in the form
27 //
28 // [CONFIG] <key>=<vaue> <timestamp> <provenance>
29 bool DeviceConfigHelper::parseConfig(std::string_view s, ParsedConfigMatch& match)
30 {
31  const char* begin = s.begin();
32  const char* end = s.end();
33  if (s.size() > 17 && (strncmp("[CONFIG] ", begin + 17, 9) != 0)) {
34  return false;
35  }
36  match.beginKey = s.data() + 9 + 17;
37  match.endKey = (char const*)memchr(match.beginKey, '=', s.size() - 9);
38  if (match.endKey == nullptr) {
39  return false;
40  }
41  match.beginValue = match.endKey + 1;
42  match.endValue = (char const*)memchr(match.beginValue, ' ', end - match.beginValue);
43  if (match.endValue == nullptr) {
44  return false;
45  }
46  char* next = nullptr;
47  match.timestamp = strtoll(match.endValue, &next, 10);
48 
49  if (!next || *next != ' ') {
50  return false;
51  }
52 
53  match.beginProvenance = next + 1;
54  match.endProvenance = end;
55 
56  return true;
57 }

59 bool DeviceConfigHelper::processConfig(ParsedConfigMatch& match,
60  DeviceInfo& info)
61 {
62  if (match.beginKey == nullptr || match.endKey == nullptr ||
63  match.beginValue == nullptr || match.endValue == nullptr ||
64  match.beginProvenance == nullptr || match.endProvenance == nullptr) {
65  return false;
66  }
67  info.currentConfig.put(std::string(match.beginKey, match.endKey - match.beginKey),
68  std::string(match.beginValue, match.endValue - match.beginValue));
69  info.currentProvenance.put(std::string(match.beginKey, match.endKey - match.beginKey),
70  std::string(match.beginProvenance, match.endProvenance - match.beginProvenance));
71  return true;
72 }
