136 inline std::vector<Vertex> Vertexer::exportVertices()

143  }

147 template <typename... T>
148 float Vertexer::evaluateTask(void (Vertexer::*task)(T...), const char* taskName, std::ostream& ostream,
149  T&&... args)
150 {
151  float diff{0.f};
152 
153  if (constants::DoTimeBenchmarks) {
154  auto start = std::chrono::high_resolution_clock::now();
155  (this->*task)(std::forward<T>(args)...);
156  auto end = std::chrono::high_resolution_clock::now();
157 
158  std::chrono::duration<double, std::milli> diff_t{end - start};
159  diff = diff_t.count();
160 
161  if (taskName == nullptr) {
162  ostream << diff << "\t";
163  } else {
164  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
165  }
166  } else {
167  (this->*task)(std::forward<T>(args)...);
168  }
169 
170  return diff;
171 }
