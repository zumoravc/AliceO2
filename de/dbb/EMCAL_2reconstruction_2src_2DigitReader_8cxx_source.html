23 template <class InputType>
24 void DigitReader<InputType>::openInput(const std::string fileName)
25 {
26  clear();
27  if (!(mInputTree = o2::utils::RootChain::load("o2sim", fileName))) {
28  LOG(FATAL) << "Failed to load cells/digits tree from " << fileName;
29  }
30 
31  if constexpr (std::is_same<InputType, Digit>::value) {
32  mInputTree->SetBranchAddress("EMCALDigit", &mInputArray);
33  if (!mInputArray) {
34  LOG(FATAL) << "Failed to find EMCALDigit branch in the " << mInputTree->GetName()
35  << " from file " << fileName;
36  }
37  mInputTree->SetBranchAddress("EMCALDigitTRGR", &mTriggerArray);
38  if (!mTriggerArray) {
39  LOG(FATAL) << "Failed to find TriggerRecords branch in the " << mInputTree->GetName()
40  << " from file " << fileName;
41  }
42  } else if constexpr (std::is_same<InputType, Cell>::value) {
43  mInputTree->SetBranchAddress("EMCALCell", &mInputArray);
44  if (!mInputArray) {
45  LOG(FATAL) << "Failed to find EMCALCell branch in the " << mInputTree->GetName()
46  << " from file " << fileName;
47  }
48  mInputTree->SetBranchAddress("EMCALCellTRGR", &mTriggerArray);
49  if (!mTriggerArray) {
50  LOG(FATAL) << "Failed to find TriggerRecords branch in the " << mInputTree->GetName()
51  << " from file " << fileName;
52  }
53  }
54 }

61 
62  if (mCurrentEntry >= mInputTree->GetEntriesFast()) {
63  return false;
64  }
65 
66  mInputTree->GetEntry(mCurrentEntry);
