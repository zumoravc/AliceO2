76 {
77  using decayed = std::decay_t<H>;
78  if constexpr (sizeof...(T) == 0) {
79  if constexpr (is_type_with_originals_v<decayed>) {
80  return typename decayed::originals{};
81  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {

112 template <typename T, typename ChunkingPolicy = Chunked>
113 class ColumnIterator : ChunkingPolicy
114 {
115  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
116 
117  public:
122  ColumnIterator(arrow::ChunkedArray const* column)
123  : mColumn{column},
124  mCurrentPos{nullptr},
125  mFirstIndex{0},
126  mCurrentChunk{0}
127  {
128  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
129  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
130  mLast = mCurrent + array->length();
131  }
132 
133  ColumnIterator() = default;
134  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
135  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
136 
137  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
138  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
139 
141  void nextChunk() const
142  {
143  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
144  mFirstIndex += previousArray->length();
145  mCurrentChunk++;
146  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
147  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
148  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
149  }
150 
151  void prevChunk() const
152  {
153  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
154  mFirstIndex -= previousArray->length();
155  mCurrentChunk--;
156  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
157  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
158  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
159  }
160 
161  void moveToChunk(int chunk)
162  {
163  if (mCurrentChunk < chunk) {
164  while (mCurrentChunk != chunk) {
165  nextChunk();
166  }
167  } else {
168  while (mCurrentChunk != chunk) {
169  prevChunk();
170  }
171  }
172  }
173 
175  void moveToEnd()
176  {
177  mCurrentChunk = mColumn->num_chunks() - 1;
178  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(mColumn->chunk(mCurrentChunk));
179  assert(array.get());
180  mFirstIndex = mColumn->length() - array->length();
181  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
182  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
183  }
184 
185  decltype(auto) operator*() const
186  {
187  if constexpr (ChunkingPolicy::chunked) {
188  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
189  nextChunk();
190  }
191  }
192  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
193  // FIXME: notice that due to the type punning we cannot simply convert the
194  // masked char to a bool, because it's undefined behavior.
195  // FIXME: check if shifting the masked bit to the first position is better than != 0
196  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
197  } else {
198  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
199  }
200  }
201 
202  // Move to the chunk which containts element pos
203  ColumnIterator<T>& moveToPos()
204  {
205  // If we get outside range of the current chunk, go to the next.
206  if constexpr (ChunkingPolicy::chunked) {
207  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
208  nextChunk();
209  }
210  }
211  return *this;
212  }
213 
214  // Move to the chunk which containts element pos
215  ColumnIterator<T>& checkNextChunk()
216  {
217  if constexpr (ChunkingPolicy::chunked) {
218  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
219  return *this;
220  }
221  nextChunk();
222  }
223  return *this;
224  }
225 
226  mutable T const* mCurrent;
227  int64_t const* mCurrentPos;
228  mutable T const* mLast;
229  arrow::ChunkedArray const* mColumn;
230  mutable int mFirstIndex;
231  mutable int mCurrentChunk;
232 };
