93  // info on the smallest block of data to be read when fetching the HBF
94  struct LinkBlock {
95  enum { StartTF = 0x1,
96  StartHB = 0x1 << 1,
97  StartSP = 0x1 << 2,
98  EndHB = 0x1 << 3 };
99  size_t offset = 0; // where data of the block starts
100  uint32_t size = 0; // block size
101  uint32_t tfID = 0; // tf counter (from 0)
102  uint32_t orbit = 0; // orbit starting the block
103  uint16_t fileID = 0; // file id where the block is located
104  uint8_t flags = 0; // different flags
105  LinkBlock() = default;
106  LinkBlock(int fid, size_t offs) : offset(offs), fileID(fid) {}
107  void setFlag(uint8_t fl, bool v = true)
108  {
109  if (v)
110  flags |= fl;
111  else
112  flags &= ~fl;
113  }
114  bool testFlag(uint8_t fl) const { return (flags & fl) == fl; }
115  void print(const std::string& pref = "") const;
116  };
