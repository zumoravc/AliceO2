225 void someDataProducerAlgorithm(ProcessingContext& ctx)
226 {
227  std::this_thread::sleep_for(std::chrono::seconds(1));
228  // Creates a new message of size collectionChunkSize which
229  // has "TPC" as data origin and "CLUSTERS" as data description.
230  auto& tpcClusters = ctx.outputs().make<FakeCluster>(Output{"TPC", "CLUSTERS", 0}, collectionChunkSize);
231  int i = 0;
232 
233  for (auto& cluster : tpcClusters) {
234  assert(i < collectionChunkSize);
235  cluster.x = i;
236  cluster.y = i;
237  cluster.z = i;
238  cluster.q = rand() % 1000;
239  i++;
240  }
241 
242  auto& itsClusters = ctx.outputs().make<FakeCluster>(Output{"ITS", "CLUSTERS", 0}, collectionChunkSize);
243  i = 0;
244  for (auto& cluster : itsClusters) {
245  assert(i < collectionChunkSize);
246  cluster.x = i;
247  cluster.y = i;
248  cluster.z = i;
249  cluster.q = rand() % 10;
250  i++;

254 void someProcessingStageAlgorithm(ProcessingContext& ctx)
255 {
256  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
257  const FakeCluster* inputDataIts = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataITS").payload);
258 
259  auto processedTpcClusters =
260  ctx.outputs().make<FakeCluster>(Output{"TPC", "CLUSTERS_P", 0}, collectionChunkSize);
261  auto processedItsClusters =
262  ctx.outputs().make<FakeCluster>(Output{"ITS", "CLUSTERS_P", 0}, collectionChunkSize);
263 
264  int i = 0;
265  for (auto& cluster : processedTpcClusters) {
266  assert(i < collectionChunkSize);
267  cluster.x = -inputDataTpc[i].x;
268  cluster.y = 2 * inputDataTpc[i].y;
269  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
270  cluster.q = inputDataTpc[i].q;
271  i++;
272  }
273 
274  i = 0;
275  for (auto& cluster : processedItsClusters) {
276  assert(i < collectionChunkSize);
277  cluster.x = -inputDataIts[i].x;
278  cluster.y = 2 * inputDataIts[i].y;
279  cluster.z = inputDataIts[i].z * inputDataIts[i].q;
280  cluster.q = inputDataIts[i].q;
281  i++;
282  }
283 };
