86 class RCombinedDSColumnJoinIndex : public RCombinedDSIndex

90  : fIndexColumnName{indexColumnName}
91  {

94  std::pair<ULong64_t, ULong64_t> GetAssociatedEntries(ULong64_t entry) final
95  {
96  auto left = fAssociations[entry];
97  return std::pair<ULong64_t, ULong64_t>(left, entry);
98  }
99 
100  std::vector<std::pair<ULong64_t, ULong64_t>>
101  BuildIndex(std::unique_ptr<RDataFrame>& left,
102  std::unique_ptr<RDataFrame>& right) final
103  {
104  std::vector<std::pair<ULong64_t, ULong64_t>> ranges;
105  auto nEntries = *right->Count();
106  fAssociations.reserve(nEntries);
107  // Fill the index with the associations
108  auto filler = [& assoc = fAssociations](INDEX_TYPE ri) { assoc.push_back(ri); };
109  right->Foreach(filler, std::vector<std::string>{fIndexColumnName});
110 
111  // Create the ranges by processing 64 entries per range
112  auto deltaRange = 64;
113  ranges.reserve(nEntries / deltaRange + 1);
114  ULong64_t i = 0;
115  while (deltaRange * (i + 1) < nEntries) {
116  ranges.emplace_back(std::pair<ULong64_t, ULong64_t>(deltaRange * i, deltaRange * (i + 1)));
117  }
118  ranges.emplace_back(std::pair<ULong64_t, ULong64_t>(deltaRange * i, nEntries)); // Last entry
119  return ranges;
120  }
121 
122  private:
123  std::string fIndexColumnName;
124  std::vector<INDEX_TYPE> fAssociations;
125 };

149 class RCombinedDSBlockJoinIndex : public RCombinedDSIndex
150 {
151 
152  using Association = std::pair<ULong64_t, ULong64_t>;
153 
154  public:
155  RCombinedDSBlockJoinIndex(std::string const& leftCategoryColumn,
156  bool self = true,
157  BlockCombinationRule combinationType = BlockCombinationRule::Anti,
158  std::string const& rightCategoryColumn = "")
159  : fLeftCategoryColumn{leftCategoryColumn},
160  fRightCategoryColumn{rightCategoryColumn.empty() ? leftCategoryColumn : rightCategoryColumn},
161  fSelf{self},
162  fCombinationType{combinationType}
163  {
164  }
165 
166  std::pair<ULong64_t, ULong64_t> GetAssociatedEntries(ULong64_t entry) final
167  {
168  return fAssociations[entry];
169  }
170 
171  std::vector<std::pair<ULong64_t, ULong64_t>>
172  BuildIndex(std::unique_ptr<RDataFrame>& left,
173  std::unique_ptr<RDataFrame>& right) final
174  {
175  std::vector<std::pair<ULong64_t, ULong64_t>> ranges;
176  std::vector<INDEX_TYPE> leftCategories;
177  std::vector<INDEX_TYPE> rightCategories;
178  std::vector<Association> leftPairs;
179  std::vector<Association> rightPairs;
180 
181  computePairsAndCategories(left, leftCategories, leftPairs, fLeftCategoryColumn);
183  if (fSelf) {
184  rightCategories = leftCategories;
185  rightPairs = leftPairs;
186  } else {
187  computePairsAndCategories(right, rightCategories, rightPairs, fRightCategoryColumn);
188  }
189 
190  auto same = [](std::pair<ULong64_t, ULong64_t> const& a, std::pair<ULong64_t, ULong64_t> const& b) {
191  return a.first < b.first;
192  };
193 
197  int startSize = fAssociations.size();
198  for (auto categoryValue : leftCategories) {
199  std::pair<ULong64_t, ULong64_t> p{categoryValue, 0};
200  auto outerRange = std::equal_range(leftPairs.begin(), leftPairs.end(), p, same);
201  decltype(outerRange) innerRange;
202  if (fSelf) {
203  innerRange = outerRange;
204  } else {
205  innerRange = std::equal_range(rightPairs.begin(), rightPairs.end(), p, same);
206  }
207  int offset = 0;
208  switch (fCombinationType) {
209  case BlockCombinationRule::Full:
210  for (auto out = outerRange.first; out != outerRange.second; ++out) {
211  for (auto in = innerRange.first; in != innerRange.second; ++in) {
212  fAssociations.emplace_back(Association{out->second, in->second});
213  }
214  }
215  break;
216  case BlockCombinationRule::Upper:
217  offset = 0;
218  for (auto out = outerRange.first; out != outerRange.second; ++out) {
219  if (innerRange.first == innerRange.second) {
220  break;
221  }
222  for (auto in = innerRange.first + offset; in != innerRange.second; ++in) {
223  fAssociations.emplace_back(Association{out->second, in->second});
224  }
225  offset++;
226  }
227  break;
228  case BlockCombinationRule::StrictlyUpper:
229  offset = 1;
230  for (auto out = outerRange.first; out != outerRange.second; ++out) {
231  if (innerRange.first == innerRange.second || innerRange.first + 1 == innerRange.second) {
232  break;
233  }
234  for (auto in = innerRange.first + offset; in != innerRange.second; ++in) {
235  fAssociations.emplace_back(Association{out->second, in->second});
236  }
237  offset++;
238  }
239  break;
240  case BlockCombinationRule::Anti:
241  for (auto out = outerRange.first; out != outerRange.second; ++out) {
242  for (auto in = innerRange.first; in != innerRange.second; ++in) {
243  if (std::distance(innerRange.first, in) == std::distance(outerRange.first, out)) {
244  continue;
245  }
246  fAssociations.emplace_back(Association{out->second, in->second});
247  }
248  offset++;
249  }
250  break;
251  case BlockCombinationRule::Diagonal:
252  auto sizeRow = std::distance(outerRange.first, outerRange.second);
253  auto sizeCol = std::distance(innerRange.first, innerRange.second);
254  for (size_t i = 0, e = std::min(sizeRow, sizeCol); i < e; ++i) {
255  fAssociations.emplace_back(Association{(outerRange.first + i)->second, (innerRange.first + i)->second});
256  }
257  break;
258  }
259  auto rangeFirst = startSize;
260  auto rangeSecond = fAssociations.size();
261  startSize = fAssociations.size();
262  ranges.emplace_back(std::make_pair<ULong64_t, ULong64_t>(rangeFirst, rangeSecond));
263  }
264  return ranges;
265  }
266 
267  private:
268  std::string fLeftCategoryColumn;
269  std::string fRightCategoryColumn;
270  bool fSelf;
271  BlockCombinationRule fCombinationType;
272  std::vector<Association> fAssociations;
273  void computePairsAndCategories(std::unique_ptr<RDataFrame>& df,
274  std::vector<INDEX_TYPE>& categories,
275  std::vector<Association>& pairs,
276  std::string const& column)
277  {
278  categories = *df->template Take<INDEX_TYPE>(column);
279  pairs.reserve(categories.size());
280  // Fill the pairs according tho the actual category
281  for (size_t i = 0; i < categories.size(); ++i) {
282  pairs.emplace_back(categories[i], i);
283  }
284  // Do a stable sort so that same categories entries are
285  // grouped together.
286  std::stable_sort(pairs.begin(), pairs.end());
287  // Keep only the categories.
288  std::stable_sort(categories.begin(), categories.end());
289  auto last = std::unique(categories.begin(), categories.end());
290  categories.erase(last, categories.end());
291  }
