26 void SymbolStatistics::rescaleToNBits(size_t bits)
27 {
28  const size_t newCumulatedFrequency = bitsToRange(bits);
29  assert(newCumulatedFrequency >= mFrequencyTable.size());
30 
31  size_t cumulatedFrequencies = mCumulativeFrequencyTable.back();
32 
33  // resample distribution based on cumulative frequencies_
34  for (size_t i = 1; i <= mFrequencyTable.size(); i++)
35  mCumulativeFrequencyTable[i] =
36  (static_cast<uint64_t>(newCumulatedFrequency) *
37  mCumulativeFrequencyTable[i]) /
38  cumulatedFrequencies;
39 
40  // if we nuked any non-0 frequency symbol to 0, we need to steal
41  // the range to make the frequency nonzero from elsewhere.
42  //
43  // this is not at all optimal, i'm just doing the first thing that comes to
44  // mind.
45  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
46  if (mFrequencyTable[i] &&
47  mCumulativeFrequencyTable[i + 1] == mCumulativeFrequencyTable[i]) {
48  // symbol i was set to zero freq
49 
50  // find best symbol to steal frequency from (try to steal from low-freq
51  // ones)
52  std::pair<size_t, size_t> stealFromEntry{mFrequencyTable.size(), ~0u};
53  for (size_t j = 0; j < mFrequencyTable.size(); j++) {
54  uint32_t frequency =
55  mCumulativeFrequencyTable[j + 1] - mCumulativeFrequencyTable[j];
56  if (frequency > 1 && frequency < stealFromEntry.second) {
57  stealFromEntry.second = frequency;
58  stealFromEntry.first = j;
59  }
60  }
61  assert(stealFromEntry.first != mFrequencyTable.size());
62 
63  // and steal from it!
64  if (stealFromEntry.first < i) {
65  for (size_t j = stealFromEntry.first + 1; j <= i; j++)
66  mCumulativeFrequencyTable[j]--;
67  } else {
68  assert(stealFromEntry.first > i);
69  for (size_t j = i + 1; j <= stealFromEntry.first; j++)
70  mCumulativeFrequencyTable[j]++;
71  }
72  }
73  }
74 
75  // calculate updated freqs and make sure we didn't screw anything up
76  assert(mCumulativeFrequencyTable.front() == 0 &&
77  mCumulativeFrequencyTable.back() == newCumulatedFrequency);
78  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
79  if (mFrequencyTable[i] == 0)
80  assert(mCumulativeFrequencyTable[i + 1] == mCumulativeFrequencyTable[i]);
81  else
82  assert(mCumulativeFrequencyTable[i + 1] > mCumulativeFrequencyTable[i]);
83 
84  // calc updated freq
85  mFrequencyTable[i] =
86  mCumulativeFrequencyTable[i + 1] - mCumulativeFrequencyTable[i];
87  }
88  // for(int i = 0; i<static_cast<int>(freqs.getNumSymbols()); i++){
89  // std::cout << i << ": " << i + min_ << " " << freqs[i] << " " <<
90  // cummulatedFrequencies_[i] << std::endl;
91  // }
92  // std::cout << cummulatedFrequencies_.back() << std::endl;
93 }

126 void SymbolStatistics::buildCumulativeFrequencyTable()
127 {
128  mCumulativeFrequencyTable.resize(mFrequencyTable.size() + 1);
129  mCumulativeFrequencyTable[0] = 0;
130  std::partial_sum(mFrequencyTable.begin(), mFrequencyTable.end(),
131  mCumulativeFrequencyTable.begin() + 1);
132 }
