35 void Digitizer::init()
36 {
37  mSimParam = &(o2::emcal::SimParam::Instance());
38  mLiveTime = mSimParam->getLiveTime();
39  mBusyTime = mSimParam->getBusyTime();
40  mRandomGenerator = new TRandom3(std::chrono::high_resolution_clock::now().time_since_epoch().count());
41 
42  float tau = mSimParam->getTimeResponseTau();
43  float N = mSimParam->getTimeResponsePower();
44  float delay = std::fmod(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE, 1);
45  mDelay = ((int)(std::floor(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE)));
46 
47  mSmearEnergy = mSimParam->doSmearEnergy();
48  mSimulateTimeResponse = mSimParam->doSimulateTimeResponse();
49  mRemoveDigitsBelowThreshold = mSimParam->doRemoveDigitsBelowThreshold();
50  mSimulateNoiseDigits = mSimParam->doSimulateNoiseDigits();
51 
52  mTimeBinOffset.clear();
53  mAmplitudeInTimeBins.clear();
54 
55  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
56  RawResponse.SetParameters(1., 0., tau, N, 0.);
57 
58  for (int i = 0; i < 4; i++) {
59  int offset = ((int)(std::floor(tau - delay - 0.25 * i)));
60  mTimeBinOffset.push_back(offset);
61 
62  std::vector<double> sf;
63  RawResponse.SetParameter(1, 0.25 * i + delay);
64 
65  for (int j = 0; j < constants::EMCAL_MAXTIMEBINS; j++) {
66  sf.push_back(RawResponse.Eval(j - offset));
67  }
68 
69  mAmplitudeInTimeBins.push_back(sf);
70  }
71 }

74 double Digitizer::rawResponseFunction(double* x, double* par)
75 {
76  double signal = 0.;
77  double tau = par[2];
78  double n = par[3];
79  double ped = par[4];
80  double xx = (x[0] - par[1] + tau) / tau;
81 
82  if (xx <= 0) {
83  signal = ped;
84  } else {
85  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
86  }
87 
88  return signal;
89 }

109 void Digitizer::process(const std::vector<Hit>& hits)
110 {
111  for (auto hit : hits) {
112  try {
113  hitToDigits(hit);
114 
115  for (auto digit : mTempDigitVector) {
116  Int_t id = digit.getTower();
117 
118  if (id < 0 || id > mGeometry->GetNCells()) {
119  LOG(WARNING) << "tower index out of range: " << id;
120  continue;
121  }
122 
123  MCLabel label(hit.GetTrackID(), mCurrEvID, mCurrSrcID, false, 1.0);
124  if (digit.getAmplitude() == 0)
125  label.setAmplitudeFraction(0);
126  LabeledDigit d(digit, label);
127  mDigits[id].push_back(d);
128  }
129  } catch (InvalidPositionException& e) {
130  LOG(ERROR) << "Error in creating the digit: " << e.what();
131  }
132  }
133 
134  mEmpty = false;
135 }

137 //_______________________________________________________________________
138 void Digitizer::hitToDigits(const Hit& hit)
139 {
140  mTempDigitVector.clear();
141  Int_t tower = hit.GetDetectorID();
142  Double_t energy = hit.GetEnergyLoss();
143 
144  if (mSmearEnergy) {
145  energy = smearEnergy(energy);
146  }
147 
148  if (mSimulateTimeResponse && (energy != 0)) {
149  for (int j = 0; j < mAmplitudeInTimeBins.at(mPhase).size(); j++) {
150  double val = energy * (mAmplitudeInTimeBins.at(mPhase).at(j));
151 
152  Digit digit(tower, val, (mEventTimeOffset + j - mTimeBinOffset.at(mPhase) + mDelay) * constants::EMCAL_TIMESAMPLE);
153  mTempDigitVector.push_back(digit);
154  }
155  } else {
156  Digit digit(tower, energy, mEventTime + mDelay * constants::EMCAL_TIMESAMPLE);
157  mTempDigitVector.push_back(digit);
158  }
159 }

161 //_______________________________________________________________________
162 double Digitizer::smearEnergy(double energy)
163 {
164  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
165  energy *= mRandomGenerator->Poisson(fluct) / fluct;
166  return energy;
167 }

170 void Digitizer::setEventTime(double t)
171 {
172  // assign event time, it should be in a strictly increasing order
173  // convert to ns
174  t *= mCoeffToNanoSecond;
175 
176  if (t < mEventTime && mContinuous) {
177  LOG(FATAL) << "New event time (" << t << ") is < previous event time (" << mEventTime << ")";
178  }
179 
180  if (t - mTriggerTime >= mLiveTime + mBusyTime) {
181  mTriggerTime = t;
182  }
183 
184  mEventTime = t - mTriggerTime;
185 
186  mPhase = ((int)((std::fmod(mEventTime, 100) + 12.5) / 25));
187  mEventTimeOffset = ((int)((mEventTime - std::fmod(mEventTime, 100) + 0.1) / 100));
188  if (mPhase == 4) {
189  mPhase = 0;
190  mEventTimeOffset++;
191  }
192 }

195 void Digitizer::addNoiseDigits(LabeledDigit& d1)
196 {
197  double amplitude = d1.getAmplitude();
198  double sigma = mSimParam->getPinNoise();
199  if (amplitude > constants::EMCAL_HGLGTRANSITION * constants::EMCAL_ADCENERGY) {
200  sigma = mSimParam->getPinNoiseLG();
201  }
202 
203  double noise = std::abs(mRandomGenerator->Gaus(0, sigma));
204  MCLabel label(true, 1.0);
205  LabeledDigit d(d1.getTower(), noise, d1.getTimeStamp(), label);
206  d1 += d;
207 }

210 void Digitizer::fillOutputContainer(std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::emcal::MCLabel>& labelsout)
211 {
212  std::list<LabeledDigit> l;
213 
214  for (auto t : mDigits) {
215  std::list<LabeledDigit> tower = t.second;
216  tower.sort();
217 
218  while (!tower.empty()) {
219  LabeledDigit ld1 = tower.front();
220  tower.pop_front();
221 
222  // loop over all other entries in the container, check if we can add the digits
223  std::vector<decltype(tower.begin())> toDelete;
224  for (auto ld2 = tower.begin(); ld2 != tower.end(); ++ld2) { // must be iterator in order to know the position in the container for erasing
225  if (ld1.canAdd(*ld2)) {
226  ld1 += *ld2;
227  toDelete.push_back(ld2);
228  }
229  }
230  for (auto del : toDelete) {
231  tower.erase(del);
232  }
233 
234  if (mSimulateNoiseDigits) {
235  addNoiseDigits(ld1);
236  }
237 
238  if (mRemoveDigitsBelowThreshold && (ld1.getAmplitude() < mSimParam->getDigitThreshold() * (constants::EMCAL_ADCENERGY)))
239  continue;
240  if (ld1.getAmplitude() < 0)
241  continue;
242  if (ld1.getTimeStamp() >= mSimParam->getLiveTime())
243  continue;
244 
245  l.push_back(ld1);
246  }
247  }
248 
249  l.sort();
250 
251  for (auto d : l) {
252  Digit digit = d.getDigit();
253  std::vector<MCLabel> labels = d.getLabels();
254  digits.push_back(digit);
255 
256  Int_t LabelIndex = labelsout.getIndexedSize();
257  for (auto label : labels) {
258  labelsout.addElementRandomAccess(LabelIndex, label);
259  }
260  }
261 
262  mDigits.clear();
263  mEmpty = true;
264 }
