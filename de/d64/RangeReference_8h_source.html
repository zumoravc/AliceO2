57 // Compact (32bit long) range reference
58 template <int NBitsN>
59 class RangeRefComp
60 {
61  using Base = std::uint32_t;
62 
63  private:
64  static constexpr int NBitsTotal = sizeof(Base) * 8;
65  static constexpr Base MaskN = ((0x1 << NBitsN) - 1);
66  static constexpr Base MaskR = (~Base(0)) & (~MaskN);
67  Base mData = 0;
68  void sanityCheck()
69  {
70  static_assert(NBitsN < NBitsTotal, "NBitsN too large");
71  }
72 
73  public:
74  RangeRefComp(int ent, int n) { set(ent, n); }
75  RangeRefComp() = default;
76  RangeRefComp(const RangeRefComp& src) = default;
77  void set(int ent, int n)
78  {
79  mData = (Base(ent) << NBitsN) + (Base(n) & MaskN);
80  }
81  static constexpr Base getMaxFirstEntry() { return MaskR >> NBitsN; }
82  static constexpr Base getMaxEntries() { return MaskN; }
83  int getFirstEntry() const { return mData >> NBitsN; }
84  int getEntries() const { return mData & ((0x1 << NBitsN) - 1); }
85  void setFirstEntry(int ent) { mData = (Base(ent) << NBitsN) | (mData & MaskN); }
86  void setEntries(int n) { mData = (mData & MaskR) | (Base(n) & MaskN); }
87  void changeEntriesBy(int inc) { setEntries(getEntries() + inc); }
88  bool operator==(const RangeRefComp& other) const
89  {
90  return mData == other.mData;
91  }
92 
93  ClassDefNV(RangeRefComp, 1);
94 };
