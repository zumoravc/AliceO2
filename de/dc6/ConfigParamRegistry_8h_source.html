50  template <typename T>
51  T get(const char* key) const
52  {
53  assert(mStore.get());
54  try {
55  if constexpr (std::is_same_v<T, int> ||
56  std::is_same_v<T, int64_t> ||
57  std::is_same_v<T, float> ||
58  std::is_same_v<T, double> ||
59  std::is_same_v<T, bool>) {
60  return mStore->store().get<T>(key);
61  } else if constexpr (std::is_same_v<T, std::string>) {
62  return mStore->store().get<std::string>(key);
63  } else if constexpr (std::is_same_v<T, std::string_view>) {
64  return std::string_view{mStore->store().get<std::string>(key)};
65  } else if constexpr (std::is_same_v<T, boost::property_tree::ptree>) {
66  return mStore->store().get_child(key);
67  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree>) {
68  return T{mStore->store().get_child(key)};
69  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree> == false) {
70  static_assert(std::is_constructible_v<T, boost::property_tree::ptree> == false,
71  "Not a basic type and no constructor from ptree provided");
72  }
73  } catch (std::exception& e) {
74  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
75  } catch (...) {
76  throw std::invalid_argument(std::string("error parsing option: ") + key);
77  }
78  }
79  private:
