123 void CalibTOF::run(int flag, int sector)
124 {
126 
127  Int_t currTOFInfoTreeEntry = -1;
128 
129  std::vector<o2::dataformats::CalibInfoTOFshort>* localCalibInfoTOF = nullptr;
130  TFile fOpenLocally(mTreeCollectedCalibInfoTOF->GetCurrentFile()->GetName());
131  TTree* localTree = (TTree*)fOpenLocally.Get(mTreeCollectedCalibInfoTOF->GetName());
132 
133  if (!localTree) {
134  LOG(FATAL) << "tree " << mTreeCollectedCalibInfoTOF->GetName() << " not found in " << mTreeCollectedCalibInfoTOF->GetCurrentFile()->GetName();
135  }
136 
137  localTree->SetBranchAddress(mCollectedCalibInfoTOFBranchName.data(), &localCalibInfoTOF);
138 
139  if (!mInitDone) {
140  LOG(FATAL) << "init() was not done yet";
141  }
142 
143  TStopwatch timerTot;
144  timerTot.Start();
145 
146  if (flag & kLHCphase) { // LHC phase --> we will use all the entries in the tree
147  while (loadTOFCollectedCalibInfo(localTree, currTOFInfoTreeEntry)) { // fill here all histos you need
148  fillLHCphaseCalibInput(localCalibInfoTOF); // we will fill the input for the LHC phase calibration
149  }
150  doLHCPhaseCalib();
151  }
152  // channel offset + problematic (flag = 2), or time slewing (flag = 4)
153  if ((flag & kChannelOffset) || (flag & kChannelTimeSlewing)) { // for the moment compute everything idependetly of the flag
154  TH1F* histoChOffsetTemp[NPADSPERSTEP];
155  std::vector<o2::dataformats::CalibInfoTOFshort>* calibTimePad[NPADSPERSTEP];
156  for (int ipad = 0; ipad < NPADSPERSTEP; ipad++) {
157  histoChOffsetTemp[ipad] = new TH1F(Form("OffsetTemp_Sec%02d_Pad%04d", sector, ipad), Form("Sector %02d (pad = %04d);channel offset (ps)", sector, ipad), 1000, -24400, 24400);
158  if (flag & kChannelTimeSlewing)
159  calibTimePad[ipad] = new std::vector<o2::dataformats::CalibInfoTOFshort>; // temporary array containing [time, tot] for every pad that we process; this will be the input for the 2D histo for timeSlewing calibration (to be filled after we get the channel offset)
160  else
161  calibTimePad[ipad] = nullptr;
162  }
163 
164  TF1* funcChOffset = new TF1(Form("fTOFchOffset_%02d", sector), "[0]*TMath::Gaus((x-[1])*(x-[1] < 12500 && x-[1] > -12500) + (x-[1]+25000)*(x-[1] < -12500) + (x-[1]-25000)*(x-[1] > 12500),0,[2])*(x > -12500 && x < 12500)", -12500, 12500);
165  funcChOffset->SetParLimits(1, -12500, 12500);
166  funcChOffset->SetParLimits(2, 50, 2000);
167 
168  TH2F* histoChTimeSlewingTemp = new TH2F(Form("hTOFchTimeSlewingTemp_%02d", sector), Form("Sector %02d;tot (ns);t - t_{exp} - t_{offset} (ps)", sector), 5000, 0., 250., 1000, -24400., 24400.);
169 
170  int startLoop = 0; // first pad that we will process in this process (we are processing a sector, unless sector = -1)
171  int endLoop = o2::tof::Geo::NCHANNELS; // last pad that we will process in this process (we are processing a sector)
172  if (sector > -1) {
173  startLoop = sector * o2::tof::Geo::NPADSXSECTOR; // first pad that we will process in this process (we are processing a sector, unless sector = -1)
174  endLoop = startLoop + o2::tof::Geo::NPADSXSECTOR; // last pad that we will process in this process (we are processing a sector, unless sector = -1)
175  }
176  for (int ich = startLoop; ich < endLoop; ich += NPADSPERSTEP) {
177  sector = ich / o2::tof::Geo::NPADSXSECTOR; // we change the value of sector which is needed when it is "-1" to put
178  // in the output histograms a meaningful name; this is not needed in
179  // case we run with sector != -1, but it will not hurt :)
180  resetChannelLevelHistos(histoChOffsetTemp, histoChTimeSlewingTemp, calibTimePad);
181  printf("strip %i\n", ich / 96);
182  currTOFInfoTreeEntry = ich - 1;
183  int ipad = 0;
184  int entryNext = currTOFInfoTreeEntry + o2::tof::Geo::NCHANNELS;
185 
186  while (loadTOFCollectedCalibInfo(localTree, currTOFInfoTreeEntry)) { // fill here all histos you need
187 
188  histoChOffsetTemp[ipad]->SetName(Form("OffsetTemp_Sec%02d_Pad%04d", sector, (ipad + ich) % o2::tof::Geo::NPADSXSECTOR));
189  fillChannelCalibInput(localCalibInfoTOF, mInitialCalibChannelOffset[ich + ipad], ipad, histoChOffsetTemp[ipad], calibTimePad[ipad]); // we will fill the input for the channel-level calibration
190  ipad++;
191 
192  if (ipad == NPADSPERSTEP) {
193  ipad = 0;
194  currTOFInfoTreeEntry = entryNext;
195  entryNext += o2::tof::Geo::NCHANNELS;
196  }
197  }
198  TFile* fout = nullptr;
199  if (flag & kChannelTimeSlewing && mDebugMode)
200  fout = new TFile(Form("timeslewingTOF%06i.root", ich / 96), "RECREATE");
201 
202  for (ipad = 0; ipad < NPADSPERSTEP; ipad++) {
203  if (histoChOffsetTemp[ipad]->GetEntries() > 30) {
204  float fractionUnderPeak = doChannelCalibration(ipad, histoChOffsetTemp[ipad], funcChOffset);
205  mCalibChannelOffset[ich + ipad] = funcChOffset->GetParameter(1) + mInitialCalibChannelOffset[ich + ipad];
206 
207  int channelInSector = (ipad + ich) % o2::tof::Geo::NPADSXSECTOR;
208 
209  mTimeSlewingObj->setFractionUnderPeak(sector, channelInSector, fractionUnderPeak);
210  mTimeSlewingObj->setSigmaPeak(sector, channelInSector, abs(funcChOffset->GetParameter(2)));
211 
212  // now fill 2D histo for time-slewing using current channel offset
213 
214  if (flag & kChannelTimeSlewing) {
215  histoChTimeSlewingTemp->Reset();
216  fillChannelTimeSlewingCalib(mCalibChannelOffset[ich + ipad], ipad, histoChTimeSlewingTemp, calibTimePad[ipad]); // we will fill the input for the channel-time-slewing calibration
217 
218  histoChTimeSlewingTemp->SetName(Form("TimeSlewing_Sec%02d_Pad%04d", sector, channelInSector));
219  histoChTimeSlewingTemp->SetTitle(Form("Sector %02d (pad = %04d)", sector, channelInSector));
220  TGraphErrors* gTimeVsTot = processSlewing(histoChTimeSlewingTemp, 1, funcChOffset);
221 
222  if (gTimeVsTot && gTimeVsTot->GetN()) {
223  for (int itot = 0; itot < gTimeVsTot->GetN(); itot++)
224  mTimeSlewingObj->addTimeSlewingInfo(ich + ipad, gTimeVsTot->GetX()[itot], gTimeVsTot->GetY()[itot] + mCalibChannelOffset[ich + ipad]);
225  } else { // just add the channel offset
226  mTimeSlewingObj->addTimeSlewingInfo(ich + ipad, 0, mCalibChannelOffset[ich + ipad]);
227  }
228 
229  if (mDebugMode && gTimeVsTot && gTimeVsTot->GetN() && fout) {
230  fout->cd();
231  int istrip = ((ich + ipad) / o2::tof::Geo::NPADS) % o2::tof::Geo::NSTRIPXSECTOR;
232  gTimeVsTot->SetName(Form("pad_%02d_%02d_%02d", sector, istrip, ipad % o2::tof::Geo::NPADS));
233  gTimeVsTot->Write();
234  // histoChTimeSlewingTemp->Write(Form("histoChTimeSlewingTemp_%02d_%02d_%02d", sector, istrip, ipad%o2::tof::Geo::NPADS)); // no longer written since it produces a very large output
235  }
236  } else if (flag & kChannelOffset) {
237  mTimeSlewingObj->addTimeSlewingInfo(ich + ipad, 0, mCalibChannelOffset[ich + ipad]);
238  }
239  }
240  }
241  if (fout) {
242  fout->Close();
243  delete fout;
244  }
245  }
246 
247  for (int ipad = 0; ipad < NPADSPERSTEP; ipad++) {
248  delete histoChOffsetTemp[ipad];
249  if (calibTimePad[ipad])
250  delete calibTimePad[ipad];
251  }
252  delete histoChTimeSlewingTemp;
253  delete funcChOffset;
254  }
255 
256  fOpenLocally.Close();
257 
258  timerTot.Stop();
259  printf("Timing (%i):\n", sector);
260  printf("Total: ");
261  timerTot.Print();
262 }

317 void CalibTOF::fillLHCphaseCalibInput(std::vector<o2::dataformats::CalibInfoTOFshort>* calibinfotof)
318 {
319 
320  // we will fill the input for the LHC phase calibration
321 
322  static double bc = 1.e13 / o2::constants::lhc::LHCRFFreq; // bunch crossing period (ps)
323  static double bc_inv = 1. / bc;
324 
325  for (auto infotof = calibinfotof->begin(); infotof != calibinfotof->end(); infotof++) {
326  double dtime = infotof->getDeltaTimePi();
327  dtime -= (int(dtime * bc_inv + 5.5) - 5) * bc; // do truncation far (by 5 units) from zero to avoid truncation of negative numbers
328 
329  mHistoLHCphase->Fill(dtime, infotof->getTimestamp());
330  }
331 }

334 void CalibTOF::doLHCPhaseCalib()
335 {
336 
337  // calibrate with respect LHC phase
338 
339  if (!mFuncLHCphase) {
340  mFuncLHCphase = new TF1("fLHCphase", "gaus");
341  }
342 
343  int ifit0 = 1;
344  for (int ifit = ifit0; ifit <= mHistoLHCphase->GetNbinsY(); ifit++) {
345  TH1D* htemp = mHistoLHCphase->ProjectionX("htemp", ifit0, ifit);
346  if (htemp->GetEntries() < 300) {
347  // we cannot fit the histogram, we will merge with the next bin
348  // Printf("We don't have enough entries to fit");
349  continue;
350  }
351 
352  int res = FitPeak(mFuncLHCphase, htemp, 500., 3., 2., "LHCphase");
353  if (res)
354  continue;
355 
356  mLHCphaseObj->addLHCphase(mHistoLHCphase->GetYaxis()->GetBinLowEdge(ifit0), mFuncLHCphase->GetParameter(1));
357  ifit0 = ifit + 1; // starting point for the next LHC interval
358  }
359 }

362 void CalibTOF::fillChannelCalibInput(std::vector<o2::dataformats::CalibInfoTOFshort>* calibinfotof, float offset, int ipad, TH1F* histo, std::vector<o2::dataformats::CalibInfoTOFshort>* calibTimePad)
363 {
364 
365  // we will fill the input for the channel-level calibration
366 
367  static double bc = 1.e13 / o2::constants::lhc::LHCRFFreq; // bunch crossing period (ps)
368  static double bc_inv = 1. / bc;
369 
370  for (auto infotof = calibinfotof->begin(); infotof != calibinfotof->end(); infotof++) {
371  double dtime = infotof->getDeltaTimePi() - offset; // removing existing offset

373 
374  histo->Fill(dtime);
375  if (calibTimePad)
376  calibTimePad->push_back(*infotof);
377  }

400 float CalibTOF::doChannelCalibration(int ipad, TH1F* histo, TF1* funcChOffset)
401 {
402  // calibrate single channel from histos - offsets
403 
404  float integral = histo->Integral();
405  if (!integral)
406  return -1; // we skip directly the channels that were switched off online, the PHOS holes...
407 
408  int resfit = FitPeak(funcChOffset, histo, 500., 3., 2., "ChannelOffset");
409 
410  // return a number greater than zero to distinguish bad fit from empty channels(fraction=0)
411  if (resfit)
412  return 0.0001; // fit was not good
413 
414  float mean = funcChOffset->GetParameter(1);
415  float sigma = funcChOffset->GetParameter(2);
416  float intmin = mean - 5 * sigma;
417  float intmax = mean + 5 * sigma;
418  float intmin2;
419  float intmax2;
420 
421  // if peak is at the border of our bunch-crossing window (-12.5:12.5 ns)
422  // continue to extrapolate gaussian integral from the other border
423  float addduetoperiodicity = 0;
424  if (intmin < -12500) { // at left border
425  intmin2 = intmin + 25000;
426  intmin = -12500;
427  intmax2 = 12500;
428  if (intmin2 > intmax) {
429  int binmin2 = histo->FindBin(intmin2);
430  int binmax2 = histo->FindBin(intmax2);
431  addduetoperiodicity = histo->Integral(binmin2, binmax2);
432  }
433  } else if (intmax > 12500) { // at right border
434  intmax2 = intmax - 25000;
435  intmax = 12500;
436  intmin2 = -12500;
437  if (intmax2 < intmin) {
438  int binmin2 = histo->FindBin(intmin2);
439  int binmax2 = histo->FindBin(intmax2);
440  addduetoperiodicity = histo->Integral(binmin2, binmax2);
441  }
442  }
443 
444  int binmin = histo->FindBin(intmin);
445  int binmax = histo->FindBin(intmax);
446 
447  if (binmin < 1)
448  binmin = 1; // avoid to take the underflow bin (can happen in case the sigma is too large)
449  if (binmax > histo->GetNbinsX())
450  binmax = histo->GetNbinsX(); // avoid to take the overflow bin (can happen in case the sigma is too large)
451 
452  return (histo->Integral(binmin, binmax) + addduetoperiodicity) / integral;
453 }

471 TGraphErrors* CalibTOF::processSlewing(TH2F* histo, Bool_t forceZero, TF1* fitFunc)
472 {
473  /* projection-x */
474  TH1D* hpx = histo->ProjectionX("hpx");
475 
476  /* define mix and max TOT bin */
477  Int_t minBin = hpx->FindFirstBinAbove(0);
478  Int_t maxBin = hpx->FindLastBinAbove(0);
479  Float_t minTOT = hpx->GetBinLowEdge(minBin);
480  Float_t maxTOT = hpx->GetBinLowEdge(maxBin + 1);
481  // printf("min/max TOT defined: %f < TOT < %f ns [%d, %d]\n", minTOT, maxTOT, minBin, maxBin);
482 
483  /* loop over TOT bins */
484  Int_t nPoints = 0;
485  Float_t tot[10000], toterr[10000];
486  Float_t mean[10000], meanerr[10000];
487  Float_t sigma[10000], vertexSigmaerr[10000];
488  for (Int_t ibin = minBin; ibin <= maxBin; ibin++) {
489 
490  /* define TOT window */
491  Int_t startBin = ibin;
492  Int_t endBin = ibin;
493  while (hpx->Integral(startBin, endBin) < 300) {
494  if (startBin == 1 && forceZero)
495  break;
496  if (endBin < maxBin)
497  endBin++;
498  else if (startBin > minBin)
499  startBin--;
500  else
501  break;
502  }
503  if (hpx->Integral(startBin, endBin) <= 0)
504  continue;
505  // printf("TOT window defined: %f < TOT < %f ns [%d, %d], %d tracks\n", hpx->GetBinLowEdge(startBin), hpx->GetBinLowEdge(endBin + 1), startBin, endBin, (Int_t)hpx->Integral(startBin, endBin));
506 
507  /* projection-y */
508  TH1D* hpy = histo->ProjectionY("hpy", startBin, endBin);
509 
510  /* average TOT */
511  hpx->GetXaxis()->SetRange(startBin, endBin);
512  tot[nPoints] = hpx->GetMean();
513  toterr[nPoints] = hpx->GetMeanError();
514 
515  /* fit peak in slices of tot */
516  if (FitPeak(fitFunc, hpy, 500., 3., 2., Form("TotBins%04d_%04d", startBin, endBin), histo) != 0) {
517  // printf("troubles fitting time-zero TRACKS, skip\n");
518  delete hpy;
519  continue;
520  }
521  mean[nPoints] = fitFunc->GetParameter(1);

661 void CalibTOF::flagProblematics()
662 {
663 
664  // method to flag problematic channels: Fraction, Sigma -> all negative if channel is bad (otherwise all positive)
665 
666  TH1F* hsigmapeak = new TH1F("hsigmapeak", ";#sigma_{peak} (ps)", 1000, 0, 1000);
667  TH1F* hfractionpeak = new TH1F("hfractionpeak", ";fraction under peak", 1001, 0, 1.01);
668 
669  int ipad;
670  float sigmaMin, sigmaMax, fractionMin;
671 
672  int nActiveChannels = 0;
673  int nGoodChannels = 0;
674 
675  TF1* fFuncSigma = new TF1("fFuncSigma", "TMath::Gaus(x,[1],[2])*[0]*(x<[1]) + TMath::Gaus(x,[1],[3])*[0]*(x>[1])");
676  fFuncSigma->SetParameter(0, 1000);
677  fFuncSigma->SetParameter(1, 200);
678  fFuncSigma->SetParameter(2, 200);
679  fFuncSigma->SetParameter(3, 200);
680 
681  TF1* fFuncFraction = new TF1("fFuncFraction", "TMath::Gaus(x,[1],[2])*[0]*(x<[1]) + TMath::Gaus(x,[1],[3])*[0]*(x>[1])");
682  fFuncFraction->SetParameter(0, 1000);
683  fFuncFraction->SetParameter(1, 0.8);
684  fFuncFraction->SetParameter(2, 0.1);
685  fFuncFraction->SetParameter(3, 0.1);
686 
687  // we group pads according to the z-coordinate since we noted a z-dependence in sigmas and fraction-under-peak
688  for (int iz = 0; iz < o2::tof::Geo::NSTRIPXSECTOR * 2; iz++) {
689  hsigmapeak->Reset();
690  hfractionpeak->Reset();
691 
692  for (int k = 0; k < o2::tof::Geo::NPADX; k++) {
693  ipad = 48 * iz + k;
694 
695  for (int i = 0; i < 18; i++) {
696  // exclude channel without entries
697  if (mTimeSlewingObj->getFractionUnderPeak(i, ipad) < 0)
698  continue;
699 
700  nActiveChannels++;
701 
702  hsigmapeak->Fill(mTimeSlewingObj->getSigmaPeak(i, ipad));
703  hfractionpeak->Fill(mTimeSlewingObj->getFractionUnderPeak(i, ipad));
704  }
705  }
706 
707  hsigmapeak->Fit(fFuncSigma, "WWq0");
708  hfractionpeak->Fit(fFuncFraction, "WWq0");
709 
710  sigmaMin = fFuncSigma->GetParameter(1) - mNsigmaSigmaProblematicCut * abs(fFuncSigma->GetParameter(2));
711  sigmaMax = fFuncSigma->GetParameter(1) + mNsigmaSigmaProblematicCut * abs(fFuncSigma->GetParameter(3));
712  fractionMin = fFuncFraction->GetParameter(1) - mNsigmaFractionProblematicCut * abs(fFuncFraction->GetParameter(2));
713 
714  for (int k = 0; k < o2::tof::Geo::NPADX; k++) {
715  ipad = 48 * iz + k;
716 
717  for (int i = 0; i < 18; i++) {
718  // exclude channel without entries
719  if (mTimeSlewingObj->getFractionUnderPeak(i, ipad) < 0)
720  continue;
721 
722  if (mTimeSlewingObj->getSigmaPeak(i, ipad) < sigmaMin ||
723  mTimeSlewingObj->getSigmaPeak(i, ipad) > sigmaMax ||
724  mTimeSlewingObj->getFractionUnderPeak(i, ipad) < fractionMin) {
725  mTimeSlewingObj->setFractionUnderPeak(i, ipad, -mTimeSlewingObj->getFractionUnderPeak(i, ipad));
726  mTimeSlewingObj->setSigmaPeak(i, ipad, -mTimeSlewingObj->getSigmaPeak(i, ipad));
727  } else
728  nGoodChannels++;
729  }
730  }
731  }
732 
733  Printf("Check for TOF problematics: nActiveChannels=%d - nGoodChannels=%d - fractionGood = %f", int(nActiveChannels), int(nGoodChannels), nGoodChannels * 1. / nActiveChannels);
734 }
