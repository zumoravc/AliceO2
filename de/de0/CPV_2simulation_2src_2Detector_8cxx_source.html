115 Bool_t Detector::ProcessHits(FairVolume* v)
116 {
117  // ------------------------------------------------------------------------
118  // Digitize one CPV hit:
119  // use exact 4-momentum p and position zxhit of the hit,
120  // find the pad response around this hit and
121  // put the amplitudes in the pads into array digits
122  //
123  // Author: Yuri Kharlov (after Serguei Sadovsky)
124  // 2 October 2000
125  // ported from AliRoot 2019
126  // ------------------------------------------------------------------------
127 
128  //analyze only charged
129  if (fMC->TrackCharge() == 0) {
130  fMC->SetMaxStep(1.e10);
131  return false;
132  }
133 
134  if (!fMC->IsTrackEntering()) { //simulate once per track
135  return false;
136  }
137 
138  int moduleNumber;
139  fMC->CurrentVolOffID(2, moduleNumber);
140 
141  double time = fMC->TrackTime() * 1.e+9; // time in ns?? To be consistent with PHOS/EMCAL
142 
143  float xyzm[3], xyzd[3];
144  fMC->TrackPosition(xyzm[0], xyzm[1], xyzm[2]);
145 
146  fMC->Gmtod(xyzm, xyzd, 1); // transform coordinate from master to daughter system
147 
148  // Current momentum of the hit's track in the local ref. system
149  float pm[3], pd[3], energy = 0.;
150  fMC->TrackMomentum(pm[0], pm[1], pm[2], energy);
151  fMC->Gmtod(pm, pd, 2); // transform 3-momentum from master to daughter system
152 
153  // Just a reminder on axes notation in the CPV module:
154  // axis Z goes along the beam
155  // axis X goes across the beam in the module plane
156  // axis Y is a normal to the module plane showing from the IP
157 
158  auto& cpvparam = o2::cpv::CPVSimParams::Instance();
159  // Digitize the current CPV hit:
160  // find pad response
161  float hitX = xyzd[0]; //hit coordinate in daugter frame
162  float hitZ = xyzd[2]; // z-coordinate of track in daugter frame
163  float pX = pd[0];
164  float pZ = pd[2];
165  float pNorm = -pd[1];
166  float dZY = pZ / pNorm * cpvparam.mCPVGasThickness;
167  float dXY = pX / pNorm * cpvparam.mCPVGasThickness;
168  float rnor1 = 0., rnor2 = 0.;
169  gRandom->Rannor(rnor1, rnor2);
170  float eloss = cpvparam.mdEdx * (1 + cpvparam.mDetR * rnor1) *
171  TMath::Sqrt((1. + (pow(dZY, 2) + pow(dXY, 2)) / pow(cpvparam.mCPVGasThickness, 2)));
172  float zhit1 = hitZ + cpvparam.mPadSizeZ * cpvparam.mnCellZ / 2 - dZY / 2;
173  float xhit1 = hitX + cpvparam.mPadSizeX * cpvparam.mnCellX / 2 - dXY / 2;
174  float zhit2 = zhit1 + dZY;
175  float xhit2 = xhit1 + dXY;
176 
177  int iwht1 = (int)(xhit1 / cpvparam.CellWr()); // wire (x) coordinate "in"
178  int iwht2 = (int)(xhit2 / cpvparam.CellWr()); // wire (x) coordinate "out"
179 
180  int nIter;
181  float zxe[3][5];
182  if (iwht1 == iwht2) { // incline 1-wire hit
183  nIter = 2;
184  zxe[0][0] = (zhit1 + zhit2 - dZY * 0.57735) / 2;
185  zxe[1][0] = (iwht1 + 0.5) * cpvparam.CellWr();
186  zxe[2][0] = eloss / 2;
187  zxe[0][1] = (zhit1 + zhit2 + dZY * 0.57735) / 2;
188  zxe[1][1] = (iwht1 + 0.5) * cpvparam.CellWr();
189  zxe[2][1] = eloss / 2;
190  } else if (TMath::Abs(iwht1 - iwht2) != 1) { // incline 3-wire hit
191  nIter = 3;
192  int iwht3 = (iwht1 + iwht2) / 2;
193  float xwht1 = (iwht1 + 0.5) * cpvparam.CellWr(); // wire 1
194  float xwht2 = (iwht2 + 0.5) * cpvparam.CellWr(); // wire 2
195  float xwht3 = (iwht3 + 0.5) * cpvparam.CellWr(); // wire 3
196  float xwr13 = (xwht1 + xwht3) / 2; // center 13
197  float xwr23 = (xwht2 + xwht3) / 2; // center 23
198  float dxw1 = xhit1 - xwr13;
199  float dxw2 = xhit2 - xwr23;
200  float egm1 = TMath::Abs(dxw1) / (TMath::Abs(dxw1) + TMath::Abs(dxw2) + cpvparam.CellWr());
201  float egm2 = TMath::Abs(dxw2) / (TMath::Abs(dxw1) + TMath::Abs(dxw2) + cpvparam.CellWr());
202  float egm3 = cpvparam.CellWr() / (TMath::Abs(dxw1) + TMath::Abs(dxw2) + cpvparam.CellWr());
203  zxe[0][0] = (dXY * (xwr13 - xwht1) / dXY + zhit1 + zhit1) / 2;
204  zxe[1][0] = xwht1;
205  zxe[2][0] = eloss * egm1;
206  zxe[0][1] = (dXY * (xwr23 - xwht1) / dXY + zhit1 + zhit2) / 2;
207  zxe[1][1] = xwht2;
208  zxe[2][1] = eloss * egm2;
209  zxe[0][2] = dXY * (xwht3 - xwht1) / dXY + zhit1;
210  zxe[1][2] = xwht3;
211  zxe[2][2] = eloss * egm3;
212  } else { // incline 2-wire hit
213  nIter = 2;
214  float xwht1 = (iwht1 + 0.5) * cpvparam.CellWr();
215  float xwht2 = (iwht2 + 0.5) * cpvparam.CellWr();
216  float xwr12 = (xwht1 + xwht2) / 2;
217  float dxw1 = xhit1 - xwr12;
218  float dxw2 = xhit2 - xwr12;
219  float egm1 = TMath::Abs(dxw1) / (TMath::Abs(dxw1) + TMath::Abs(dxw2));
220  float egm2 = TMath::Abs(dxw2) / (TMath::Abs(dxw1) + TMath::Abs(dxw2));
221  zxe[0][0] = (zhit1 + zhit2 - dZY * egm1) / 2;
222  zxe[1][0] = xwht1;
223  zxe[2][0] = eloss * egm1;
224  zxe[0][1] = (zhit1 + zhit2 + dZY * egm2) / 2;
225  zxe[1][1] = xwht2;
226  zxe[2][1] = eloss * egm2;
227  }
228 
229  // Finite size of ionization region
230 
231  int nz3 = (cpvparam.mNgamz + 1) / 2;
232  int nx3 = (cpvparam.mNgamx + 1) / 2;
233 
234  TVirtualMCStack* stack = fMC->GetStack();
235  const Int_t partID = stack->GetCurrentTrackNumber();
236 
237  for (int iter = 0; iter < nIter; iter++) {
238 
239  float zhit = zxe[0][iter];
240  float xhit = zxe[1][iter];
241  float qhit = zxe[2][iter];
242  float zcell = zhit / cpvparam.mPadSizeZ;
243  float xcell = xhit / cpvparam.mPadSizeX;
244  if (zcell <= 0 || xcell <= 0 ||
245  zcell >= cpvparam.mnCellZ || xcell >= cpvparam.mnCellX) {
246  return true; //beyond CPV
247  }
248  int izcell = (int)zcell;
249  int ixcell = (int)xcell;
250  float zc = zcell - izcell - 0.5;
251  float xc = xcell - ixcell - 0.5;
252  for (int iz = 1; iz <= cpvparam.mNgamz; iz++) {
253  int kzg = izcell + iz - nz3;
254  if (kzg <= 0 || kzg > cpvparam.mnCellZ) {
255  continue;
256  }
257  float zg = (float)(iz - nz3) - zc;
258  for (int ix = 1; ix <= cpvparam.mNgamx; ix++) {
259  int kxg = ixcell + ix - nx3;
260  if (kxg <= 0 || kxg > cpvparam.mnCellX) {
261  continue;
262  }
263  float xg = (float)(ix - nx3) - xc;
264 
265  // Now calculate pad response
266  float qpad = padResponseFunction(qhit, zg, xg);
267  qpad += cpvparam.mNoise * rnor2;
268  if (qpad < 0) {
269  continue;
270  }
271  // Fill hit with pad response ID and amplitude
272  // hist will be sorted and merged later if necessary
273  short detID = Geometry::relToAbsId(moduleNumber, kxg, kzg);
274  addHit(partID, detID, Point3D<float>(xyzm[0], xyzm[1], xyzm[2]), time, qpad);
275  }
276  }
277  }
278 
279  return true;
280 }

282 float Detector::padResponseFunction(float qhit, float zhit, float xhit)
283 {
284  // ------------------------------------------------------------------------
285  // Calculate the amplitude in one CPV pad using the
286  // cumulative pad response function
287  // Author: Yuri Kharlov (after Serguei Sadovski)
288  // 3 October 2000
289  // ------------------------------------------------------------------------
290 
291  auto& cpvparam = o2::cpv::CPVSimParams::Instance();
292  float dz = cpvparam.mPadSizeZ / 2;
293  float dx = cpvparam.mPadSizeX / 2;
294  float z = zhit * cpvparam.mPadSizeZ;
295  float x = xhit * cpvparam.mPadSizeX;
296  float amplitude = qhit *
297  (CPVCumulPadResponse(z + dz, x + dx) - CPVCumulPadResponse(z + dz, x - dx) -
298  CPVCumulPadResponse(z - dz, x + dx) + CPVCumulPadResponse(z - dz, x - dx));
299  return amplitude;
300 }

302 float Detector::CPVCumulPadResponse(float x, float y)
303 {
304  // ------------------------------------------------------------------------
305  // Cumulative pad response function
306  // It includes several terms from the CF decomposition in electrostatics
307  // Note: this cumulative function is wrong since omits some terms
308  // but the cell amplitude obtained with it is correct because
309  // these omitting terms cancel
310  // Author: Yuri Kharlov (after Serguei Sadovski)
311  // 3 October 2000
312  // ------------------------------------------------------------------------
313 
314  auto& cpvparam = o2::cpv::CPVSimParams::Instance();
315  float r2 = x * x + y * y;
316  float xy = x * y;
317  float cumulPRF = 0;
318  for (Int_t i = 0; i <= 4; i++) {
319  float b1 = (2 * i + 1) * cpvparam.mB;
320  cumulPRF += TMath::Power(-1, i) * TMath::ATan(xy / (b1 * TMath::Sqrt(b1 * b1 + r2)));
321  }
322  cumulPRF *= cpvparam.mA / (2 * TMath::Pi());
323  return cumulPRF;
324 }

335 void Detector::ConstructGeometry()
336 {
337  // Create geometry description of CPV depector for Geant simulations.
338 
339  using boost::algorithm::contains;
340 
341  cpv::GeometryParams* geomParams = cpv::GeometryParams::GetInstance("CPVRun3Params");
342 
343  if (!geomParams) {
344  LOG(ERROR) << "ConstructGeometry: CPV Geometry class has not been set up.\n";
345  }
346 
347  if (!fMC) {
348  fMC = TVirtualMC::GetMC();
349  }
350 
351  // Configure geometry So far we have only one: Run3
352  {
353  mActiveModule[0] = kFALSE;
354  mActiveModule[1] = kTRUE;
355  mActiveModule[2] = kTRUE;
356  mActiveModule[3] = kTRUE;
357  mActiveModule[4] = kFALSE;
358  mActiveModule[5] = kFALSE;
359  }
360 
361  // First create necessary materials
362  CreateMaterials();
363 
364  // Create a CPV modules-containers which will be filled with the stuff later.
365  float par[3], x, y, z;
366 
367  // The box containing all CPV filled with air
368  par[0] = geomParams->GetCPVBoxSize(0) / 2.0;
369  par[1] = geomParams->GetCPVBoxSize(1) / 2.0;
370  par[2] = geomParams->GetCPVBoxSize(2) / 2.0;
371  fMC->Gsvolu("CPV", "BOX ", getMediumID(ID_AIR), par, 3);
372 
373  // --- Position CPV modules in ALICE setup ---
374  int idrotm[5];
375  int iXYZ, iAngle;
376  char im[5];
377  for (int iModule = 0; iModule < 5; iModule++) {
378  if (!mActiveModule[iModule + 1]) {
379  continue;
380  }
381  float angle[3][2] = {0};
382  geomParams->GetModuleAngle(iModule, angle);
383  Matrix(idrotm[iModule], angle[0][0], angle[0][1], angle[1][0], angle[1][1], angle[2][0], angle[2][1]);
384 
385  float pos[3] = {0};
386  geomParams->GetModuleCenter(iModule, pos);
387 
388  fMC->Gspos("CPV", iModule + 1, "cave", pos[0], pos[1], pos[2], idrotm[iModule], "ONLY");
389  }
390 
391  //start filling CPV moodules
392  // Gassiplex board
393  par[0] = geomParams->GetGassiplexChipSize(0) / 2.;
394  par[1] = geomParams->GetGassiplexChipSize(1) / 2.;
395  par[2] = geomParams->GetGassiplexChipSize(2) / 2.;
396  fMC->Gsvolu("CPVG", "BOX ", ID_TEXTOLIT, par, 3);
397 
398  // Cu+Ni foil covers Gassiplex board
399  par[1] = geomParams->GetCPVCuNiFoilThickness() / 2;
400  fMC->Gsvolu("CPVC", "BOX ", ID_CU, par, 3);
401  y = -(geomParams->GetGassiplexChipSize(1) / 2 - par[1]);
402  fMC->Gspos("CPVC", 1, "CPVG", 0, y, 0, 0, "ONLY");
403 
404  // Position of the chip inside CPV
405  float xStep = geomParams->GetCPVActiveSize(0) / (geomParams->GetNumberOfCPVChipsPhi() + 1);
406  float zStep = geomParams->GetCPVActiveSize(1) / (geomParams->GetNumberOfCPVChipsZ() + 1);
407  int copy = 0;
408  y = geomParams->GetCPVFrameSize(1) / 2 - geomParams->GetFTPosition(0) +
409  geomParams->GetCPVTextoliteThickness() / 2 + geomParams->GetGassiplexChipSize(1) / 2 + 0.1;
410  for (int ix = 0; ix < geomParams->GetNumberOfCPVChipsPhi(); ix++) {
411  x = xStep * (ix + 1) - geomParams->GetCPVActiveSize(0) / 2;
412  for (int iz = 0; iz < geomParams->GetNumberOfCPVChipsZ(); iz++) {
413  copy++;
414  z = zStep * (iz + 1) - geomParams->GetCPVActiveSize(1) / 2;
415  fMC->Gspos("CPVG", copy, "CPV", x, y, z, 0, "ONLY");
416  }
417  }
418 
419  // Foiled textolite (1 mm of textolite + 50 mkm of Cu + 6 mkm of Ni)
420  par[0] = geomParams->GetCPVActiveSize(0) / 2;
421  par[1] = geomParams->GetCPVTextoliteThickness() / 2;
422  par[2] = geomParams->GetCPVActiveSize(1) / 2;
423  fMC->Gsvolu("CPVF", "BOX ", ID_TEXTOLIT, par, 3);
424 
425  // Argon gas volume
426  par[1] = (geomParams->GetFTPosition(2) - geomParams->GetFTPosition(1) - geomParams->GetCPVTextoliteThickness()) / 2;
427  fMC->Gsvolu("CPVAr", "BOX ", ID_AR, par, 3);
428 
429  for (int i = 0; i < 4; i++) {
430  y = geomParams->GetCPVFrameSize(1) / 2 - geomParams->GetFTPosition(i) + geomParams->GetCPVTextoliteThickness() / 2;
431  fMC->Gspos("CPVF", i + 1, "CPV", 0, y, 0, 0, "ONLY");
432  if (i == 1) {
433  y -= (geomParams->GetFTPosition(2) - geomParams->GetFTPosition(1)) / 2;
434  fMC->Gspos("CPVAr", 1, "CPV ", 0, y, 0, 0, "ONLY");
435  }
436  }
437 
438  // Dummy sensitive plane in the middle of argone gas volume
439  par[1] = 0.001;
440  fMC->Gsvolu("CPVQ", "BOX ", ID_AR, par, 3);
441  fMC->Gspos("CPVQ", 1, "CPVAr", 0, 0, 0, 0, "ONLY");
442 
443  // Cu+Ni foil covers textolite
444  par[1] = geomParams->GetCPVCuNiFoilThickness() / 2;
445  fMC->Gsvolu("CPVP1", "BOX ", ID_CU, par, 3);
446  y = geomParams->GetCPVTextoliteThickness() / 2 - par[1];
447  fMC->Gspos("CPVP1", 1, "CPVF", 0, y, 0, 0, "ONLY");
448 
449  // Aluminum frame around CPV
450  par[0] = geomParams->GetCPVFrameSize(0) / 2;
451  par[1] = geomParams->GetCPVFrameSize(1) / 2;
452  par[2] = geomParams->GetCPVBoxSize(2) / 2;
453  fMC->Gsvolu("CPVF1", "BOX ", ID_AL, par, 3);
454 
455  par[0] = geomParams->GetCPVBoxSize(0) / 2 - geomParams->GetCPVFrameSize(0);
456  par[1] = geomParams->GetCPVFrameSize(1) / 2;
457  par[2] = geomParams->GetCPVFrameSize(2) / 2;
458  fMC->Gsvolu("CPVF2", "BOX ", ID_AL, par, 3);
459 
460  for (int j = 0; j <= 1; j++) {
461  x = TMath::Sign(1, 2 * j - 1) * (geomParams->GetCPVBoxSize(0) - geomParams->GetCPVFrameSize(0)) / 2;
462  fMC->Gspos("CPVF1", j + 1, "CPV", x, 0, 0, 0, "ONLY");
463  z = TMath::Sign(1, 2 * j - 1) * (geomParams->GetCPVBoxSize(2) - geomParams->GetCPVFrameSize(2)) / 2;
464  fMC->Gspos("CPVF2", j + 1, "CPV", 0, 0, z, 0, "ONLY");
465  }
466 
467  gGeoManager->CheckGeometry();
468 }

470 void Detector::CreateMaterials()
471 {
472  // Definitions of materials to build CPV and associated tracking media.
473 
474  int isxfld = 2;
475  float sxmgmx = 10.0;
476  o2::base::Detector::initFieldTrackingParams(isxfld, sxmgmx);
477 
478  // --- Air ---
479  float aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
480  float zAir[4] = {6., 7., 8., 18.};
481  float wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
482  float dAir = 1.20479E-3;
483 
484  Mixture(ID_AIR, "Air", aAir, zAir, dAir, 4, wAir);
485  Medium(ID_AIR, "Air", ID_AIR, 0, isxfld, sxmgmx, 10.0, 1.0, 0.1, 0.1, 10.0, nullptr, 0);
486 
487  // The Textolit which makes up the box which contains the calorimeter module -> idtmed[707]
488  float aTX[4] = {16.0, 28.09, 12.011, 1.00794};
489  float zTX[4] = {8.0, 14.0, 6.0, 1.0};
490  float wTX[4] = {292.0, 68.0, 462.0, 736.0};
491  float dTX = 1.75;
492 
493  Mixture(ID_TEXTOLIT, "Textolit", aTX, zTX, dTX, -4, wTX);
494  Medium(ID_TEXTOLIT, "Textolit", ID_TEXTOLIT, 0, isxfld, sxmgmx, 10.0, 0.1, 0.1, 0.1, 0.1, nullptr, 0);
495 
496  // Copper -> idtmed[710]
497  Material(ID_CU, "Cupr", 63.546, 29, 8.96, 1.43, 14.8, nullptr, 0);
498  Medium(ID_CU, "Cupr", ID_CU, 0, isxfld, sxmgmx, 10.0, 0.1, 0.1, 0.1, 0.0001, nullptr, 0);
499 
500  // The gas mixture: ArCo2 -> idtmed[715]
501  // Ar+CO2 Mixture (80% / 20%)
502  // --- The gas mixture ---
503  // Co2
504  // float aCO[2] = {12.0, 16.0} ;
505  // float zCO[2] = {6.0, 8.0} ;
506  // float wCO[2] = {1.0, 2.0} ;
507  float dCO = 0.001977; //Co2 density
508  float dAr = 0.001782; //Argon density
509 
510  float arContent = 0.80; // Ar-content of the ArCO2-mixture
511  float aArCO[3] = {39.948, 12.0, 16.0};
512  float zArCO[3] = {18.0, 6.0, 8.0};
