418 void MagneticWrapperChebyshev::Field(const Double_t* xyz, Double_t* b) const
419 {
420  Double_t rphiz[3];
421 
422 #ifndef _BRING_TO_BOUNDARY_ // exact matching to fitted volume is requested
423  b[0] = b[1] = b[2] = 0;
424 #endif
425 
426  if (xyz[2] > mMinZSolenoid) {
427  cartesianToCylindrical(xyz, rphiz);
428  fieldCylindricalSolenoid(rphiz, b);
429  // convert field to cartesian system
430  cylindricalToCartesianCylB(rphiz, b, b);
431  return;
432  }
433 
434  int iddip = findDipoleSegment(xyz);
435  if (iddip < 0) {
436  return;
437  }
438  Chebyshev3D* par = getParameterDipole(iddip);
439 #ifndef _BRING_TO_BOUNDARY_
440  if (!par->isInside(xyz)) {
441  return;
442  }
443 #endif
444  par->Eval(xyz, b);
445 }

447 Double_t MagneticWrapperChebyshev::getBz(const Double_t* xyz) const
448 {
449  Double_t rphiz[3];
450 
451  if (xyz[2] > mMinZSolenoid) {
452  cartesianToCylindrical(xyz, rphiz);
453  return fieldCylindricalSolenoidBz(rphiz);
454  }
455 
456  int iddip = findDipoleSegment(xyz);
457  if (iddip < 0) {
458  return 0.;
459  }
460  Chebyshev3D* par = getParameterDipole(iddip);
461 #ifndef _BRING_TO_BOUNDARY_
462  if (!par->isInside(xyz)) {
463  return 0.;
464  }
465 #endif
466  return par->Eval(xyz, 2);
467 }

510 Int_t MagneticWrapperChebyshev::findDipoleSegment(const Double_t* xyz) const
511 {
512  if (!mNumberOfParameterizationDipole) {
513  return -1;
514  }
515  int xid, yid, zid = TMath::BinarySearch(mNumberOfDistinctZSegmentsDipole, mCoordinatesSegmentsZDipole,
516  (Float_t)xyz[2]); // find zsegment
517 
518  Bool_t reCheck = kFALSE;
519  while (true) {
520  int ysegBeg = mBeginningOfSegmentsYDipole[zid];
521 
522  for (yid = 0; yid < mNumberOfSegmentsYDipole[zid]; yid++) {
523  if (xyz[1] < mCoordinatesSegmentsYDipole[ysegBeg + yid]) {
524  break;
525  }
526  }
527  if (--yid < 0) {
528  yid = 0;
529  }
530  yid += ysegBeg;
531 
532  int xsegBeg = mBeginningOfSegmentsXDipole[yid];
533  for (xid = 0; xid < mNumberOfSegmentsXDipole[yid]; xid++) {
534  if (xyz[0] < mCoordinatesSegmentsXDipole[xsegBeg + xid]) {
535  break;
536  }
537  }
538 
539  if (--xid < 0) {
540  xid = 0;
541  }
542  xid += xsegBeg;
543 
544  // to make sure that due to the precision problems we did not pick the next Zbin
545  if (!reCheck && (xyz[2] - mCoordinatesSegmentsZDipole[zid] < 3.e-5) && zid &&
546  !getParameterDipole(mSegmentIdDipole[xid])->isInside(xyz)) { // check the previous Z bin
547  zid--;
548  reCheck = kTRUE;
549  continue;
550  }
551  break;
552  }
