280 void Geometry::DefineSamplingFraction(const std::string_view mcname, const std::string_view mctitle)
281 {
282  // Jun 05,2006
283  // Look http://rhic.physics.wayne.edu/~pavlinov/ALICE/SHISHKEBAB/RES/linearityAndResolutionForTRD1.html
284  // Keep for compatibility
285  //
286  using boost::algorithm::contains;
287 
288  // Sampling factor for G3
289  mSampling = 10.87; // Default value - Nov 25,2010
290  if (mNECLayers == 69) { // 10% layer reduction
291  mSampling = 12.55;
292  } else if (mNECLayers == 61) { // 20% layer reduction
293  mSampling = 12.80;
294  } else if (mNECLayers == 77) {
295  if (contains(mGeoName, "V1")) {
296  mSampling = 10.87; // Adding paper sheets and cover plate; Nov 25,2010
297  } else if (mECScintThick > 0.159 && mECScintThick < 0.161) { // original sampling fraction, equal layers
298  mSampling = 12.327; // fECScintThick = fECPbRadThickness = 0.160;
299  } else if (mECScintThick > 0.175 && mECScintThick < 0.177) { // 10% Pb thicknes reduction
300  mSampling = 10.5; // fECScintThick = 0.176, fECPbRadThickness=0.144;
301  } else if (mECScintThick > 0.191 && mECScintThick < 0.193) { // 20% Pb thicknes reduction
302  mSampling = 8.93; // fECScintThick = 0.192, fECPbRadThickness=0.128;
303  }
304  }
305 
306  Float_t samplingFactorTranportModel = 1.;
307  if (contains(mcname, "Geant3"))
308  samplingFactorTranportModel = 1.; // 0.988 // Do nothing
309  else if (contains(mcname, "Fluka"))
310  samplingFactorTranportModel = 1.; // To be set
311  else if (contains(mcname, "Geant4")) {
312  if (contains(mctitle, "EMV-EMCAL"))
313  samplingFactorTranportModel = 0.821; // EMC list but for EMCal, before 0.86
314  else if (contains(mctitle, "EMV"))
315  samplingFactorTranportModel = 1.096; // 0.906, 0.896 (OPT)
316  else
317  samplingFactorTranportModel = 0.821; // 1.15 (CHIPS), 1.149 (BERT), 1.147 (BERT_CHIPS)
318  }
319 
320  LOG(INFO) << "MC modeler <" << mcname << ">, Title <" << mctitle << ">: Sampling " << std::setw(2)
321  << std::setprecision(3) << mSampling << ", model fraction with respect to G3 "
322  << samplingFactorTranportModel << ", final sampling " << mSampling * samplingFactorTranportModel;
323 
324  mSampling *= samplingFactorTranportModel;
325 }

327 void Geometry::DefineEMC(std::string_view mcname, std::string_view mctitle)
328 {
329  using boost::algorithm::contains;
330 
331  // geometry
332  std::transform(mGeoName.begin(), mGeoName.end(), mGeoName.begin(), ::toupper);
333 
334  // Convert old geometry names to new ones
335  if (contains(mGeoName, "SHISH_77_TRD1_2X2_FINAL_110DEG")) {
336  if (contains(mGeoName, "PBTH=0.144") && contains(mGeoName, "SCTH=0.176")) {
337  mGeoName = "EMCAL_COMPLETE";
338  } else {
339  mGeoName = "EMCAL_PDC06";
340  }
341  }
342 
343  if (contains(mGeoName, "WSUC"))
344  mGeoName = "EMCAL_WSUC";
345 
346  // check that we have a valid geometry name
347  if (!(contains(mGeoName, "EMCAL_PDC06") || contains(mGeoName, "EMCAL_WSUC") || contains(mGeoName, "EMCAL_COMPLETE") ||
348  contains(mGeoName, "EMCAL_COMPLETEV1") || contains(mGeoName, "EMCAL_COMPLETE12SMV1") ||
349  contains(mGeoName, "EMCAL_FIRSTYEAR") || contains(mGeoName, "EMCAL_FIRSTYEARV1"))) {
350  LOG(FATAL) << "Init, " << mGeoName << " is an undefined geometry!\n";
351  }
352 
353  // Option to know whether we have the "half" supermodule(s) or not
354  mKey110DEG = 0;
355  if (contains(mGeoName, "COMPLETE") || contains(mGeoName, "PDC06") || contains(mGeoName, "12SM"))
356  mKey110DEG = 1; // for GetAbsCellId
357  if (contains(mGeoName, "COMPLETEV1"))
358  mKey110DEG = 0;
359 
360  mnSupModInDCAL = 0;
361  if (contains(mGeoName, "DCAL_DEV")) {
362  mnSupModInDCAL = 10;
363  } else if (contains(mGeoName, "DCAL_8SM")) {
364  mnSupModInDCAL = 8;
365  } else if (contains(mGeoName, "DCAL")) {
366  mnSupModInDCAL = 6;
367  }
368 
369  // JLK 13-Apr-2008
370  // default parameters are those of EMCAL_COMPLETE geometry
371  // all others render variations from these at the end of
372  // geometry-name specific options
373 
374  mNumberOfSuperModules = 12; // 12 = 6 * 2 (6 in phi, 2 in Z)
375  mNPhi = 12; // module granularity in phi within smod (azimuth)
376  mNZ = 24; // module granularity along Z within smod (eta)
377  mNPHIdiv = mNETAdiv = 2; // tower granularity within module
378  mArm1PhiMin = 80.0; // degrees, Starting EMCAL Phi position
379  mArm1PhiMax = 200.0; // degrees, Ending EMCAL Phi position
380  mArm1EtaMin = -0.7; // pseudorapidity, Starting EMCAL Eta position
381  mArm1EtaMax = +0.7; // pseudorapidity, Ending EMCAL Eta position
382  mIPDistance = 428.0; // cm, radial distance to front face from nominal vertex point
383  mPhiGapForSM = 2.; // cm, only for final TRD1 geometry
384  mFrontSteelStrip = 0.025; // 0.025cm = 0.25mm (13-may-05 from V.Petrov)
385  mPassiveScintThick = 0.8; // 0.8cm = 8mm (13-may-05 from V.Petrov)
386  mLateralSteelStrip = 0.01; // 0.01cm = 0.1mm (13-may-05 from V.Petrov) - was 0.025
387  mTrd1Angle = 1.5; // in degrees
388 
389  mSampling = 1.; // should be calculated with call to DefineSamplingFraction()
390  mNECLayers = 77; // (13-may-05 from V.Petrov) - can be changed with additional options
391  mECScintThick = 0.176; // scintillator layer thickness
392  mECPbRadThickness = 0.144; // lead layer thickness
393 
394  mPhiModuleSize = 12.26 - mPhiGapForSM / Float_t(mNPhi); // first assumption
395  mEtaModuleSize = mPhiModuleSize;
396 
397  mZLength = 700.; // Z coverage (cm)
398  mPhiSuperModule = 20.; // phi in degree
399  mDCALInnerEdge = mIPDistance * TMath::Tan(mTrd1Angle * 8. * TMath::DegToRad());
400 
401  // modifications to the above for PDC06 geometry
402  if (contains(mGeoName, "PDC06")) { // 18-may-05 - about common structure
403  mECScintThick = mECPbRadThickness = 0.16; // (13-may-05 from V.Petrov)
404  }
405 
406  // modifications to the above for WSUC geometry
407  if (contains(mGeoName, "WSUC")) { // 18-may-05 - about common structure
408  mNumberOfSuperModules = 2; // 27-may-05; Nov 24,2010 for TB
409  mNPhi = mNZ = 4;
410  mTrd1AlFrontThick = 1.0; // one cm
411  // Bond paper - two sheets around Sc tile
412  mTrd1BondPaperThick = 0.01; // 0.01cm = 0.1 mm
413 
414  mPhiModuleSize = 12.0;
415  mEtaModuleSize = mPhiModuleSize;
416  mLateralSteelStrip = 0.015; // 0.015cm = 0.15mm
417  }
418 
419  // In 2009-2010 data taking runs only 4 SM, in the upper position.
420  if (contains(mGeoName, "FIRSTYEAR")) {
421  mNumberOfSuperModules = 4;
422  mArm1PhiMax = 120.0;
423  }
424 
425  if (contains(mGeoName, "FIRSTYEARV1") || contains(mGeoName, "COMPLETEV1") || contains(mGeoName, "COMPLETE12SMV1")) {
426  // Oct 26,2010 : First module has tilt = 0.75 degree :
427  // look to AliEMCALShishKebabTrd1Module::DefineFirstModule(key)
428  // New sizes from production drawing, added Al front plate.
429  // The thickness of sampling is change due to existing two sheets of paper.
430 
431  // Will replace fFrontSteelStrip
432  mTrd1AlFrontThick = 1.0; // one cm
433  // Bond paper - two sheets around Sc tile
434  mTrd1BondPaperThick = 0.01; // 0.01cm = 0.1 mm
435 
436  mPhiModuleSize = 12.0;
437  mEtaModuleSize = mPhiModuleSize;
438  mLateralSteelStrip = 0.015; // 0.015cm = 0.15mm
439 
440  if (contains(mGeoName, "COMPLETEV1")) {
441  mNumberOfSuperModules = 10;
442  mArm1PhiMax = 180.0;
443  } else if (contains(mGeoName, "COMPLETE12SMV1")) {
444  mNumberOfSuperModules = 12;
445  mArm1PhiMax = 200.0;
446  }
447  if (contains(mGeoName, "DCAL")) {
448  mNumberOfSuperModules = 12 + mnSupModInDCAL;
449  mArm1PhiMax = 320.0;
450  if (contains(mGeoName, "DCAL_8SM"))
451  mArm1PhiMax = 340.0; // degrees, End of DCAL Phi position
452  else if (contains(mGeoName, "DCAL_DEV"))
453  mArm1PhiMin = 40.0; // degrees, Starting EMCAL(shifted) Phi position
454  mDCALPhiMin = mArm1PhiMax - 10. * mnSupModInDCAL;
455  }
456  }
457 
458  //
459  // Init EMCal/DCal SMs type array
460  mEMCSMSystem.clear();
461  mEMCSMSystem.resize(mNumberOfSuperModules);
462 
463  for (Int_t i = 0; i < mNumberOfSuperModules; i++)
464  mEMCSMSystem[i] = NOT_EXISTENT;
465 
466  Int_t iSM = 0;
467 
468  //
469  // BASIC EMCAL SM
470  if (contains(mGeoName, "WSUC")) {
471  for (int i = 0; i < 2; i++) {
472  mEMCSMSystem[iSM] = EMCAL_STANDARD;
473  iSM++;
474  }
475  } else if (contains(mGeoName, "FIRSTYEAR")) {
476  for (int i = 0; i < 4; i++) {
477  mEMCSMSystem[iSM] = EMCAL_STANDARD;
478  iSM++;
479  }
480  } else if (contains(mGeoName, "PDC06") || contains(mGeoName, "COMPLETE")) {
481  for (int i = 0; i < 10; i++) {
482  mEMCSMSystem[iSM] = EMCAL_STANDARD;
483  iSM++;
484  }
485  }
486 
487  //
488  // EMCAL 110SM
489  if (mKey110DEG && contains(mGeoName, "12SM")) {
490  for (int i = 0; i < 2; i++) {
491  mEMCSMSystem[iSM] = EMCAL_HALF;
492  if (contains(mGeoName, "12SMV1")) {
493  mEMCSMSystem[iSM] = EMCAL_THIRD;
494  }
495  iSM++;
496  }
497  }
498 
499  //
500  // DCAL SM
501  if (mnSupModInDCAL && contains(mGeoName, "DCAL")) {
502  if (contains(mGeoName, "8SM")) {
503  for (int i = 0; i < mnSupModInDCAL - 2; i++) {
504  mEMCSMSystem[iSM] = DCAL_STANDARD;
505  iSM++;
506  }
507  for (int i = 0; i < 2; i++) {
508  mEMCSMSystem[iSM] = DCAL_EXT;
509  iSM++;
510  }
511  } else {
512  for (int i = 0; i < mnSupModInDCAL; i++) {
513  mEMCSMSystem[iSM] = DCAL_STANDARD;
514  iSM++;
515  }
516  }
517  }
518 
519  // constant for transition absid <--> indexes
520  mNCellsInModule = mNPHIdiv * mNETAdiv;
521  mNCellsInSupMod = mNCellsInModule * mNPhi * mNZ;
