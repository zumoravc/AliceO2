37 GPUdii() void GPUTPCCFClusterizer::computeClustersImpl(int nBlocks, int nThreads, int iBlock, int iThread,
38  const CfFragment& fragment,
39  GPUSharedMemory& smem,
40  const Array2D<PackedCharge>& chargeMap,
41  const ChargePos* filteredPeakPositions,
42  MCLabelAccumulator* labelAcc,
43  uint clusternum,
44  uint maxClusterPerRow,
45  uint* clusterInRow,
46  tpc::ClusterNative* clusterByRow)
47 {
48  uint idx = get_global_id(0);
49 
50  // For certain configurations dummy work items are added, so the total
51  // number of work items is dividable by 64.
52  // These dummy items also compute the last cluster but discard the result.
53  ChargePos pos = filteredPeakPositions[CAMath::Min(idx, clusternum - 1)];
54  Charge charge = chargeMap[pos].unpack();
55 
56  ClusterAccumulator pc;
57  CPU_ONLY(labelAcc->collect(pos, charge));
58 
59  buildClusterScratchPad(
60  chargeMap,
61  pos,
62  smem.posBcast,
63  smem.buf,
64  smem.innerAboveThreshold,
65  &pc,
66  labelAcc);
67 
68  if (idx >= clusternum || fragment.isOverlap(pos.time())) {
69  return;
70  }
71  pc.finalize(pos, charge, fragment.start);
72 
73  tpc::ClusterNative myCluster;
74  pc.toNative(pos, charge, myCluster);
75 
76 #if defined(CUT_QTOT)
77  bool aboveQTotCutoff = (pc.Q > QTOT_CUTOFF);
78 #else
79  bool aboveQTotCutoff = true;
80 #endif
81 
82  if (!aboveQTotCutoff) {
83  return;
84  }
85 
86  uint rowIndex = sortIntoBuckets(
87  myCluster,
88  pos.row(),
89  maxClusterPerRow,
90  clusterInRow,
91  clusterByRow);
92  static_cast<void>(rowIndex); // Avoid unused varible warning on GPU.
93 
94  CPU_ONLY(labelAcc->commit(pos.row(), rowIndex, maxClusterPerRow));
95 }

97 GPUdii() void GPUTPCCFClusterizer::updateClusterScratchpadInner(
98  ushort lid,
99  ushort N,
100  const PackedCharge* buf,
101  const ChargePos& pos,
102  ClusterAccumulator* cluster,
103  MCLabelAccumulator* labelAcc,
104  uchar* innerAboveThreshold)
105 {
106  uchar aboveThreshold = 0;
107 
108  GPUCA_UNROLL(U(), U())
109  for (ushort i = 0; i < N; i++) {
110  Delta2 d = CfConsts::InnerNeighbors[i];
111 
112  PackedCharge p = buf[N * lid + i];
113 
114  Charge q = cluster->updateInner(p, d);
115 
116  CPU_ONLY(
117  labelAcc->collect(pos.delta(d), q));
118 
119  aboveThreshold |= (uchar(q > CHARGE_THRESHOLD) << i);
120  }
121 
122  innerAboveThreshold[lid] = aboveThreshold;
123 
124  GPUbarrier();
125 }

151 GPUdii() void GPUTPCCFClusterizer::buildClusterScratchPad(
152  const Array2D<PackedCharge>& chargeMap,
153  ChargePos pos,
154  ChargePos* posBcast,
155  PackedCharge* buf,
156  uchar* innerAboveThreshold,
157  ClusterAccumulator* myCluster,
158  MCLabelAccumulator* labelAcc)
159 {
160  ushort ll = get_local_id(0);
161 
162  posBcast[ll] = pos;
163  GPUbarrier();
164 
165  CfUtils::blockLoad<PackedCharge>(
166  chargeMap,
167  SCRATCH_PAD_WORK_GROUP_SIZE,
168  SCRATCH_PAD_WORK_GROUP_SIZE,
169  ll,
170  0,
171  8,
172  CfConsts::InnerNeighbors,
173  posBcast,
174  buf);
175  updateClusterScratchpadInner(
176  ll,
177  8,
178  buf,
179  pos,
180  myCluster,
181  labelAcc,
182  innerAboveThreshold);
183 
184  ushort wgSizeHalf = (SCRATCH_PAD_WORK_GROUP_SIZE + 1) / 2;
185 
186  bool inGroup1 = ll < wgSizeHalf;
187 
188  ushort llhalf = (inGroup1) ? ll : (ll - wgSizeHalf);
189 
190  CfUtils::condBlockLoad(
191  chargeMap,
192  wgSizeHalf,
193  SCRATCH_PAD_WORK_GROUP_SIZE,
194  ll,
195  0,
196  16,
197  CfConsts::OuterNeighbors,
198  posBcast,
199  innerAboveThreshold,
200  buf);
201 
202  if (inGroup1) {
203  updateClusterScratchpadOuter(
204  llhalf,
205  16,
206  16,
207  0,
208  buf,
209  pos,
210  myCluster,
211  labelAcc);
212  }
213 
214 #if defined(GPUCA_GPUCODE)
215  CfUtils::condBlockLoad(
216  chargeMap,
217  wgSizeHalf,
218  SCRATCH_PAD_WORK_GROUP_SIZE,
219  ll,
220  0,
221  16,
222  CfConsts::OuterNeighbors,
223  posBcast + wgSizeHalf,
224  innerAboveThreshold + wgSizeHalf,
225  buf);
226  if (!inGroup1) {
227  updateClusterScratchpadOuter(
228  llhalf,
229  16,
230  16,
231  0,
232  buf,
233  pos,
234  myCluster,
235  labelAcc);
236  }
237 #endif
238 }
