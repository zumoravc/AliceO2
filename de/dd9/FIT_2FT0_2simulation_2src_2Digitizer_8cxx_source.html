29 Digitizer::CFDOutput Digitizer::get_time(const std::vector<float>& times, float deadTime)
30 {
31  assert(std::is_sorted(std::begin(times), std::end(times)));
32 
33  // get a new batch of random values
34  for (float& n : mNoiseSamples) {
35  n = mRndGaus.getNextValue();
36  }
37 
38  // this lambda function evaluates the signal shape at a give time
39  auto value_at = [&](float time) {
40  float val = 0;
41  // (1) sum over individual hits
42  Vc::float_v acc(0);
43  Vc::float_v tableVal(0);
44  const float* tp = times.data();
45  size_t m = times.size() / Vc::float_v::size();
46  for (size_t i = 0; i < m; ++i) {
47  tableVal.load(tp);
48  tp += Vc::float_v::size();
49  Vc::prefetchForOneRead(tp);
50  acc += signalFormVc(time - tableVal);
51  }
52  val += acc.sum();
53  // non-SIMD tail
54  for (size_t i = Vc::float_v::size() * m; i < times.size(); ++i, ++tp) {
55  val += signalForm(time - (*tp));
56  }
57  // (2) add noise
58  // find the right indices into the sinc table
59  int timeIndex = std::lround(time / mParameters.mNoisePeriod * mSincTable.size());
60  int timeOffset = timeIndex / mSincTable.size();
61  timeIndex %= mSincTable.size();
62  if (timeOffset >= mNumNoiseSamples) { // this happens when time >= 25 ns
63  timeOffset = mNumNoiseSamples - 1;
64  LOG(DEBUG) << "timeOffset >= mNumNoiseSamples";
65  }
66  if (timeOffset <= -mNumNoiseSamples) { // this happens when time <= -25 ns
67  timeOffset = -mNumNoiseSamples + 1;
68  LOG(DEBUG) << "timeOffset <= -mNumNoiseSamples";
69  }
70  Vc::float_v noiseVal(0);
71  const float* np = mNoiseSamples.data();
72  tp = mSincTable[timeIndex].data() + mNumNoiseSamples - timeOffset;
73  acc = 0.0f;
74  m = mNumNoiseSamples / Vc::float_v::Size;
75  for (size_t i = 0; i < m; ++i) {
76  tableVal.load(tp);
77  tp += Vc::float_v::Size;
78  Vc::prefetchForOneRead(tp);
79  noiseVal.load(np);
80  np += Vc::float_v::Size;
81  Vc::prefetchForOneRead(np);
82  acc += noiseVal * tableVal;
83  }
84  val += acc.sum(); // horizontal sum
85  // non-SIMD tail
86  for (size_t i = Vc::float_v::Size * m; i < mNumNoiseSamples; ++i, ++tp, ++np) {
87  val += (*np) * (*tp);
88  }
89  return val;
90  };
91  auto const min_time = std::max(deadTime, *std::min_element(std::begin(times),
92  std::end(times)));
93  CFDOutput result{std::nullopt, -0.5f * mParameters.bunchWidth};
94  bool is_positive = true;
95 
96  // reset the chache
97  std::fill_n(std::begin(mSignalCache), std::size(mSignalCache), -1.0f);
98 
99  // we need double precision for time in order to match previous behaviour
100  for (double time = min_time; time < 0.5 * mParameters.bunchWidth; time += DP::SIGNAL_CACHE_DT) {
101  float const val = value_at(time);
102  int const index = std::lround((time + 0.5 * mParameters.bunchWidth) / DP::SIGNAL_CACHE_DT);
103  if (index >= 0 && index < mSignalCache.size()) { // save the value for later use
104  mSignalCache[index] = val;
105  }
106  // look up the time-shifted signal value from the past
107  float val_prev = 0.0f;
108  int const index_prev = std::lround((time - mParameters.mCFDShiftPos + 0.5f * mParameters.bunchWidth) / DP::SIGNAL_CACHE_DT);
109  val_prev = ((index_prev < 0 || index_prev >= mSignalCache.size() || mSignalCache[index_prev] < 0.0f)
110  ? value_at(time - mParameters.mCFDShiftPos) // was not computed before
111  : mSignalCache[index_prev]); // is available in the cache
112  float const cfd_val = 5.0f * val_prev - val;
113  if (std::abs(val) > mParameters.mCFD_trsh && !is_positive && cfd_val > 0.0f) {
114  if (!result.particle) {
115  result.particle = time;
116  }
117  result.deadTime = time + mParameters.mCFDdeadTime;
118  time += mParameters.mCFDdeadTime - DP::SIGNAL_CACHE_DT;
119  is_positive = true;
120  } else {
121  is_positive = cfd_val > 0.0f;
122  }
123  }
124  if (!result.particle) {
125  LOG(DEBUG) << "CFD failed to find peak ";
126  for (auto t : times) {
127  LOG(DEBUG) << t << ", dead time " << deadTime;
128  }
129  }
130  return result;
131 }

133 double Digitizer::measure_amplitude(const std::vector<float>& times) const
134 {
135  float const from = mParameters.mAmpRecordLow;
136  float const to = from + mParameters.mAmpRecordUp;
137  // SIMD version has a negligible effect on the total wall time
138  Vc::float_v acc(0);
139  Vc::float_v tv(0);
140  const float* tp = times.data();
141  size_t const m = times.size() / Vc::float_v::Size;
142  for (size_t i = 0; i < m; ++i) {
143  tv.load(tp);
144  tp += Vc::float_v::Size;
145  Vc::prefetchForOneRead(tp);
146  acc += signalForm_integralVc(to - tv) - signalForm_integralVc(from - tv);
147  }
148  float result = acc.sum(); // horizontal sum
149  // non-SIMD tail
150  for (size_t i = Vc::float_v::Size * m; i < times.size(); ++i, ++tp) {
151  result += signalForm_integral(to - (*tp)) - signalForm_integral(from - (*tp));
152  }
153  return result;
154 }

156 void Digitizer::process(const std::vector<o2::ft0::HitType>* hits,
157  std::vector<o2::ft0::Digit>& digitsBC,
158  std::vector<o2::ft0::ChannelData>& digitsCh,
159  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>& label)
160 {
161  ;
162  //Calculating signal time, amplitude in mean_time +- time_gate --------------
163  flush(digitsBC, digitsCh, label);
164  Int_t parent = -10;
165  for (auto const& hit : *hits) {
166  if (hit.GetEnergyLoss() > 0)
167  continue;
168 
169  Int_t hit_ch = hit.GetDetectorID();
170  Bool_t is_A_side = (hit_ch < 4 * mParameters.nCellsA);
171  Float_t time_compensate = is_A_side ? mParameters.A_side_cable_cmps : mParameters.C_side_cable_cmps;
172  Double_t hit_time = hit.GetTime() - time_compensate;
173  auto relBC = o2::InteractionRecord{hit_time};
174  if (mCache.size() <= relBC.bc) {
175  mCache.resize(relBC.bc + 1);
176  }
177  mCache[relBC.bc].hits.emplace_back(BCCache::particle{hit_ch, hit_time - relBC.bc2ns()});
178 
179  //charge particles in MCLabel
180  Int_t parentID = hit.GetTrackID();
181  if (parentID != parent)
182  mCache[relBC.bc].labels.emplace(parentID, mEventID, mSrcID, hit_ch);
183  parent = parentID;
184  }
185 }

187 void Digitizer::storeBC(BCCache& bc,
188  std::vector<o2::ft0::Digit>& digitsBC,
189  std::vector<o2::ft0::ChannelData>& digitsCh,
190  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>& labels)
191 {
192  if (bc.hits.empty())
193  return;
194  int n_hit_A = 0, n_hit_C = 0, mean_time_A = 0, mean_time_C = 0;
195  int summ_ampl_A = 0, summ_ampl_C = 0;
196  int vertex_time;
197 
198  int first = digitsCh.size(), nStored = 0;
199  auto& particles = bc.hits;
200  std::sort(std::begin(particles), std::end(particles));
201  auto channel_end = particles.begin();
202  std::vector<float> channel_times;
203  for (Int_t ipmt = 0; ipmt < mParameters.mMCPs; ++ipmt) {
204  auto channel_begin = channel_end;
205  channel_end = std::find_if(channel_begin, particles.end(),
206  [ipmt](BCCache::particle const& p) { return p.hit_ch != ipmt; });
207  if (channel_end - channel_begin < mParameters.mAmp_trsh)
208  continue;
209  channel_times.resize(channel_end - channel_begin);
210  std::transform(channel_begin, channel_end, channel_times.begin(), [](BCCache::particle const& p) { return p.hit_time; });
211  int chain = (std::rand() % 2) ? 1 : 0;
212 
213  auto cfd = get_time(channel_times, mDeadTimes[ipmt].intrec.bc2ns() -
214  firstBCinDeque.bc2ns() +
215  mDeadTimes[ipmt].deadTime);
216  mDeadTimes[ipmt].intrec = firstBCinDeque;
217  mDeadTimes[ipmt].deadTime = cfd.deadTime;
218 
219  if (!cfd.particle)
220  continue;
221  int smeared_time = 1000. * (*cfd.particle - mParameters.mCfdShift) * mParameters.ChannelWidthInverse;
222  bool is_time_in_signal_gate = (smeared_time > -mParameters.mTime_trg_gate && smeared_time < mParameters.mTime_trg_gate);
223  float charge = measure_amplitude(channel_times) * mParameters.charge2amp;
224  float amp = is_time_in_signal_gate ? mParameters.mV_2_Nchannels * charge : 0;
225  if (amp > 4095)
226  amp = 4095;
227  LOG(DEBUG) << "bc " << firstBCinDeque.bc << ", ipmt " << ipmt << ", smeared_time " << smeared_time << " nStored " << nStored;
228  digitsCh.emplace_back(ipmt, smeared_time, int(amp), chain);
229  nStored++;
230 
231  // fill triggers
232 
233  Bool_t is_A_side = (ipmt <= 4 * mParameters.nCellsA);
234  if (smeared_time > mParameters.mTime_trg_gate || smeared_time < -mParameters.mTime_trg_gate)
235  continue;
236 
237  if (is_A_side) {
238  n_hit_A++;
239  summ_ampl_A += amp;
240  mean_time_A += smeared_time;
241  } else {
242  n_hit_C++;
243  summ_ampl_C += amp;
244  mean_time_C += smeared_time;
245  }
246  }
247  Bool_t is_A, is_C, isVertex, is_Central, is_SemiCentral = 0;
248  is_A = n_hit_A > 0;
249  is_C = n_hit_C > 0;
250  is_Central = summ_ampl_A + summ_ampl_C >= mParameters.mtrg_central_trh;
251  is_SemiCentral = summ_ampl_A + summ_ampl_C >= mParameters.mtrg_semicentral_trh;
252  vertex_time = (mean_time_A - mean_time_C) * 0.5;
253  isVertex = is_A && is_C && (vertex_time > -mParameters.mTime_trg_gate && vertex_time < mParameters.mTime_trg_gate);
254  uint32_t amplA = is_A ? summ_ampl_A : 0; // sum amplitude A side
255  uint32_t amplC = is_C ? summ_ampl_C : 0; // sum amplitude C side
256  uint16_t timeA = is_A ? mean_time_A / n_hit_A : 0; // average time A side
257  uint16_t timeC = is_C ? mean_time_C / n_hit_C : 0; // average time C side
258 
259  Triggers triggers;
260  triggers.setTriggers(is_A, is_C, isVertex, is_Central, is_SemiCentral, int8_t(n_hit_A), int8_t(n_hit_C),
261  amplA, amplC, timeA, timeC);
262 
263  digitsBC.emplace_back(first, nStored, firstBCinDeque, triggers, mEventID);
264  size_t const nBC = digitsBC.size();
265  for (auto const& lbl : bc.labels)
266  labels.addElement(nBC - 1, lbl);
267 
268  // Debug output -------------------------------------------------------------
269 
270  LOG(INFO) << "Event ID: " << mEventID << ", bc " << firstBCinDeque.bc << ", N hit " << bc.hits.size();
271  LOG(INFO) << "N hit A: " << int(triggers.nChanA) << " N hit C: " << int(triggers.nChanC) << " summ ampl A: " << int(triggers.amplA)
272  << " summ ampl C: " << int(triggers.amplC) << " mean time A: " << triggers.timeA
273  << " mean time C: " << triggers.timeC;
274 
275  LOG(INFO) << "IS A " << triggers.getOrA() << " IsC " << triggers.getOrC() << " vertex " << triggers.getVertex() << " is Central " << triggers.getCen() << " is SemiCentral " << triggers.getSCen();
276 }

306 void Digitizer::initParameters()
307 {
308  auto const sinc = [](double x) { x *= TMath::Pi(); return (std::abs(x) < 1e-12) ? 1.0 : std::sin(x) / x; };
309 
310  // number of noise samples in one BC
311  mNumNoiseSamples = std::ceil(mParameters.bunchWidth / mParameters.mNoisePeriod);
312  mNoiseSamples.resize(mNumNoiseSamples);
313 
314  // set up tables with sinc function values (times noiseVar)
315  for (size_t i = 0, n = mSincTable.size(); i < n; ++i) {
316  float const time = i / float(n) * mParameters.mNoisePeriod; // [0 .. 1/mParameters.mNoisePeriod)
317  std::cout << "initParameters " << i << "/" << n << " " << time << std::endl;
318  // we make a table of sinc values between -num_noise_samples and 2*num_noise_samples
319  mSincTable[i].resize(3 * mNumNoiseSamples);
320  for (int j = -mNumNoiseSamples; j < 2 * mNumNoiseSamples; ++j) {
321  mSincTable[i][mNumNoiseSamples + j] = mParameters.mNoiseVar * sinc((time + 0.5f * mParameters.bunchWidth) / mParameters.mNoisePeriod - j);
322  }
323  }
324  // set up the lookup table for the signal form
325  for (size_t i = 0; i < DP::SIGNAL_TABLE_SIZE; ++i) {
326  float const x = float(i) / float(DP::SIGNAL_TABLE_SIZE) * mParameters.bunchWidth;
327  mSignalTable[i] = signalForm_i(x);
328  }
329 
330  // cache for signal time series used by the CFD -BC/2 .. +3BC/2
331  mSignalCache.resize(std::lround(mParameters.bunchWidth / DP::SIGNAL_CACHE_DT));
332 }
