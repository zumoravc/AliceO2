49 void IrregularSpline1D::construct(int numberOfKnots, const float inputKnots[], int numberOfAxisBins)
50 {
68 
69  numberOfAxisBins -= 1;
70 
71  FlatObject::startConstruction();
72 
73  if (numberOfAxisBins < 4) {
74  numberOfAxisBins = 4;
75  }
76 
77  std::vector<int> vKnotBins;
78 
79  { // reorganize knots
80 
81  int lastBin = numberOfAxisBins; // last bin starts with U value 1.f, therefore it is outside of the [0.,1.] interval
82 
83  vKnotBins.push_back(0); // obligatory knot at 0.0
84 
85  for (int i = 0; i < numberOfKnots; ++i) {
86  int bin = (int)roundf(inputKnots[i] * numberOfAxisBins);
87  if (bin <= vKnotBins.back() || bin >= lastBin) {
88  continue; // same knot
89  }
90  vKnotBins.push_back(bin);
91  }
92 
93  vKnotBins.push_back(lastBin); // obligatory knot at 1.0
94 
95  if (vKnotBins.size() < 5) { // too less knots, make a grid with 5 knots
96  vKnotBins.clear();
97  vKnotBins.push_back(0);
98  vKnotBins.push_back((int)roundf(0.25 * numberOfAxisBins));
99  vKnotBins.push_back((int)roundf(0.50 * numberOfAxisBins));
100  vKnotBins.push_back((int)roundf(0.75 * numberOfAxisBins));
101  vKnotBins.push_back(lastBin);
102  }
103  }
104 
105  mNumberOfKnots = vKnotBins.size();
106  mNumberOfAxisBins = numberOfAxisBins;
107  mBin2KnotMapOffset = mNumberOfKnots * sizeof(IrregularSpline1D::Knot);
108 
109  FlatObject::finishConstruction(mBin2KnotMapOffset + (numberOfAxisBins + 1) * sizeof(int));
110 
111  IrregularSpline1D::Knot* s = getKnotsNonConst();
112 
113  for (int i = 0; i < mNumberOfKnots; i++) {
114  s[i].u = vKnotBins[i] / ((double)mNumberOfAxisBins); // do division in double
115  }
116 
117  { // values will not be used, we define them for consistency
118  int i = 0;
119  double du = (s[i + 1].u - s[i].u);
120  double x3 = (s[i + 2].u - s[i].u) / du;
121  s[i].scale = 1. / du;
122  s[i].scaleL0 = 0.; // undefined
123  s[i].scaleL2 = 0.; // undefined
124  s[i].scaleR2 = (x3 - 2.) / (x3 - 1.);
125  s[i].scaleR3 = 1. / (x3 * (x3 - 1.));
126  }
127 
128  for (int i = 1; i < mNumberOfKnots - 2; i++) {
129  double du = (s[i + 1].u - s[i].u);
130  double x0 = (s[i - 1].u - s[i].u) / du;
131  double x3 = (s[i + 2].u - s[i].u) / du;
132  s[i].scale = 1. / du;
133  s[i].scaleL0 = -1. / (x0 * (x0 - 1.));
134  s[i].scaleL2 = x0 / (x0 - 1.);
135  s[i].scaleR2 = (x3 - 2.) / (x3 - 1.);
136  s[i].scaleR3 = 1. / (x3 * (x3 - 1.));
137  }
138 
139  { // values will not be used, we define them for consistency
140  int i = mNumberOfKnots - 2;
141  double du = (s[i + 1].u - s[i].u);
142  double x0 = (s[i - 1].u - s[i].u) / du;
143  s[i].scale = 1. / du;
144  s[i].scaleL0 = -1. / (x0 * (x0 - 1.));
145  s[i].scaleL2 = x0 / (x0 - 1.);
146  s[i].scaleR2 = 0; // undefined
147  s[i].scaleR3 = 0; // undefined
148  }
149 
150  { // values will not be used, we define them for consistency
151  int i = mNumberOfKnots - 1;
152  s[i].scale = 0; // undefined
153  s[i].scaleL0 = 0; // undefined
154  s[i].scaleL2 = 0; // undefined
155  s[i].scaleR2 = 0; // undefined
156  s[i].scaleR3 = 0; // undefined
157  }
158 
159  // Set up map (U bin) -> (knot index)
160 
161  int* map = getBin2KnotMapNonConst();
162 
163  int iKnotMin = 1;
164  int iKnotMax = mNumberOfKnots - 3;
165 
166  //
167  // With iKnotMin=1, iKnotMax=nKnots-3 we release edge intervals:
168  //
169  // Map U coordinates from the first segment [knot0,knot1] to the second segment [knot1,knot2].
170  // Map coordinates from segment [knot{n-2},knot{n-1}] to the previous segment [knot{n-3},knot{n-4}]
171  //
172  // This trick allows one to use splines without special conditions for edge cases.
173  // Any U from [0,1] is mapped to some knote i, where i-1, i, i+1, and i+2 knots always exist
174  //
175 
176  for (int iBin = 0, iKnot = iKnotMin; iBin <= mNumberOfAxisBins; iBin++) {
177  if ((iKnot < iKnotMax) && vKnotBins[iKnot + 1] == iBin) {
178  iKnot = iKnot + 1;
179  }
180  map[iBin] = iKnot;
181  }
182 }

184 void IrregularSpline1D::constructRegular(int numberOfKnots)
185 {
189 
190  if (numberOfKnots < 5)
191  numberOfKnots = 5;
192 
193  std::vector<float> knots(numberOfKnots);
194  double du = 1. / (numberOfKnots - 1.);
195  for (int i = 1; i < numberOfKnots - 1; i++) {
196  knots[i] = i * du;
197  }
198  knots[0] = 0.f;
199  knots[numberOfKnots - 1] = 1.f;
200  construct(numberOfKnots, knots.data(), numberOfKnots);
201 }
