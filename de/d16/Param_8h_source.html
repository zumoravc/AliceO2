179  void Lors2Mars(Int_t c, float x, float y, double* m, Int_t pl = kPc) const
180  {

182  switch (pl) {

193  double l[3] = {x - mX, y - mY, z};
194  mM[c]->LocalToMaster(l, m);
195  }
