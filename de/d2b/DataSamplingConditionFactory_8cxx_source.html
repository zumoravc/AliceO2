23 std::unique_ptr<DataSamplingCondition> DataSamplingConditionFactory::create(std::string name)
24 {
25  if (name == "random" || name == "DataSamplingConditionRandom") {
26  return createDataSamplingConditionRandom();
27  } else if (name == "payloadSize" || name == "DataSamplingConditionPayloadSize") {
28  return createDataSamplingConditionPayloadSize();
29  } else if (name == "nConsecutive" || name == "DataSamplingConditionNConsecutive") {
30  return createDataSamplingConditionNConsecutive();
31  } else if (name == "custom" || name == "DataSamplingConditionCustom") {
32  return createDataSamplingConditionCustom();
33  }
34  throw std::runtime_error("DataSamplingCondition '" + name + "' unknown.");
35 }
