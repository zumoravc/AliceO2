21 // TRD pad plane class //
22 // //
23 // Contains the information on ideal pad positions, pad dimensions, //
24 // tilting angle, etc. //
25 // It also provides methods to identify the current pad number from //
26 // local tracking coordinates. //
27 // //

74  GPUd() double getTiltOffset(double rowOffset) const { return mTiltingTan * (rowOffset - 0.5 * mLengthIPad); };
75  GPUd() double getPadRowOffset(int row, double z) const
76  {
77  if ((row < 0) || (row >= mNrows))
78  return -1.0;
79  else
80  return mPadRow[row] + mPadRowSMOffset - z;
81  };
82  GPUd() double getPadRowOffsetROC(int row, double z) const
83  {

85  return -1.0;

88  };
89 
90  GPUd() double getPadColOffset(int col, double rphi) const
91  {

93  return -1.0;
94  else
95  return rphi - mPadCol[col];
96  };

105  GPUd() double getRowEndROC() const { return mPadRow[mNrows - 1] - mLengthOPad; };
106  GPUd() double getColEnd() const { return mPadCol[mNcols - 1] + mWidthOPad; };
107  GPUd() double getRowPos(int row) const { return mPadRow[row] + mPadRowSMOffset; };
108  GPUd() double getRowPosROC(int row) const { return mPadRow[row]; };
109  GPUd() double getColPos(int col) const { return mPadCol[col]; };
110  GPUd() double getRowSize(int row) const
111  {
112  if ((row == 0) || (row == mNrows - 1))
113  return mLengthOPad;
114  else
115  return mLengthIPad;
116  };

164  double mPadCol[MAXCOLS]; // Pad border positions in column direction

168  double mAnodeWireOffset; // Distance of first anode wire from pad edge
