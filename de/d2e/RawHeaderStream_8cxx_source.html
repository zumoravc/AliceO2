14 std::istream& o2::emcal::operator>>(std::istream& stream, o2::header::RAWDataHeaderV4& header)
15 {
16  //std::cout << "called, 10 words" << std::endl;
17  using wordtype = uint64_t;
18  constexpr int RAWHEADERWORDS = sizeof(header) / sizeof(wordtype);
19  wordtype message[RAWHEADERWORDS];
20  auto address = reinterpret_cast<char*>(message);
21  for (int i = 0; i < RAWHEADERWORDS; i++) {
22  stream.read(address + i * sizeof(wordtype) / sizeof(char), sizeof(message[i]));
23  //std::cout << "Word " << i << ": " << std::hex << message[i] << std::dec << std::endl;
24  }
25  header.word0 = message[0];
26  header.word1 = message[1];
27  header.word2 = message[2];
28  header.word3 = message[3];
29  header.word4 = message[4];
30  header.word5 = message[5];
31  header.word6 = message[6];
