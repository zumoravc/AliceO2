153 // Poor man variant class. Does not take ownership of anything passed to it.
154 // FIXME: we should really use C++17 std::variant when it
155 // comes about
156 class Variant
157 {
158  using storage_t = std::aligned_union<8, int, int64_t, const char*, float, double, bool>::type;
159 
160  public:
161  Variant(VariantType type = VariantType::Unknown) : mType{type} {}
162 
163  template <typename T>
164  Variant(T value) : mType{variant_trait_v<T>}
165  {
166  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
167  }
168 
169  template <typename T>
170  Variant(std::initializer_list<T>) : mType{VariantType::Unknown}
171  {
172  static_assert(sizeof(T) == 0,
173  "brace-enclosed initializer list forbidden for Variant"
174  "\n did you accidentally put braces around the default value?");
175  }
176 
177  Variant(const Variant& other) : mType(other.mType)
178  {
179  // In case this is a string we need to duplicate it to avoid
180  // double deletion.
181  if (mType == variant_trait_v<const char*>) {
182  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
183  } else {
184  mStore = other.mStore;
185  }
186  }
187 
188  Variant(Variant&& other) : mType(other.mType)
189  {
190  // In case this is a string we need to duplicate it to avoid
191  // double deletion.
192  if (mType == variant_trait_v<const char*>) {
193  mStore = other.mStore;
194  *reinterpret_cast<char**>(&(other.mStore)) = nullptr;
195  } else {
196  mStore = other.mStore;
197  }
198  }
199 
200  ~Variant()
201  {
202  // In case we allocated a string out of bound, we
203  // should delete it.
204  if (mType == variant_trait_v<const char*> || mType == variant_trait_v<char*>) {
205  free(*reinterpret_cast<void**>(&mStore));
206  }
207  }
208 
209  void operator=(const Variant& other)
210  {
211  if (mType == variant_trait_v<const char*>) {
212  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
213  } else {
214  mStore = other.mStore;
215  }
216  }
217 
218  template <typename T>
219  T get() const
220  {
221  auto v = variant_trait_v<T>;
222  if (mType != variant_trait_v<T>) {
223  throw std::runtime_error("Mismatch between types");
224  }
225  return variant_helper<storage_t, T>::get(&mStore);
226  }
227 
228  template <typename T>
229  void set(T value)
230  {
231  return variant_helper<storage_t, T>::set(mStore, value);
232  }
233 
234  VariantType type() const { return mType; }
235 
236  private:
237  friend std::ostream& operator<<(std::ostream& oss, Variant const& val);
238  storage_t mStore;
239  VariantType mType;
240 };
