55 void TrackFitter::fit(Track& track, bool smooth, bool finalize,
56  std::list<TrackParam>::reverse_iterator* itStartingParam)
57 {
63 
64  // initialize the starting track parameters and cluster
65  auto itParam(track.rbegin());
66  if (itStartingParam != nullptr) {
67  // use the ones pointed to by itStartingParam
68  if (*itStartingParam == track.rend()) {
69  throw runtime_error("invalid starting parameters");
70  }
71  itParam = *itStartingParam;
72  } else {
73  // or start from the last cluster and compute the track parameters from its position
74  // and the one of the first previous cluster found on a different chamber
75  auto itPreviousParam(itParam);
76  do {
77  ++itPreviousParam;
78  if (itPreviousParam == track.rend()) {
79  throw runtime_error("A track is made of at least 2 clusters in 2 different chambers");
80  }
81  } while (itPreviousParam->getClusterPtr()->getChamberId() == itParam->getClusterPtr()->getChamberId());
82  initTrack(*itPreviousParam->getClusterPtr(), *itParam->getClusterPtr(), *itParam);
83  }
84 
85  // recusively add the upstream clusters and update the track parameters
86  TrackParam* startingParam = &*itParam;
87  while (++itParam != track.rend()) {
88  addCluster(*startingParam, *itParam->getClusterPtr(), *itParam);
89  startingParam = &*itParam;
90  }
91 
92  // smooth the track if requested and the smoother enabled
93  if (smooth && mSmooth) {
94  smoothTrack(track, finalize);
95  }
96 }

99 void TrackFitter::initTrack(const Cluster& cl1, const Cluster& cl2, TrackParam& param)
100 {
105 
106  // compute the track parameters at the last cluster
107  double dZ = cl1.getZ() - cl2.getZ();
108  param.setNonBendingCoor(cl2.getX());
109  param.setBendingCoor(cl2.getY());
110  param.setZ(cl2.getZ());
111  param.setNonBendingSlope((cl1.getX() - cl2.getX()) / dZ);
112  param.setBendingSlope((cl1.getY() - cl2.getY()) / dZ);
113  double bendingImpact = cl2.getY() - cl2.getZ() * param.getBendingSlope();
114  double inverseBendingMomentum = 1. / TrackExtrap::getBendingMomentumFromImpactParam(bendingImpact);
115  param.setInverseBendingMomentum(inverseBendingMomentum);
116 
117  // compute the track parameter covariances at the last cluster (as if the other clusters did not exist)
118  TMatrixD lastParamCov(5, 5);
119  lastParamCov.Zero();
120  // Non bending plane
121  lastParamCov(0, 0) = cl2.getEx2();
122  lastParamCov(0, 1) = -lastParamCov(0, 0) / dZ;
123  lastParamCov(1, 0) = lastParamCov(0, 1);
124  lastParamCov(1, 1) = (1000. * cl1.getEx2() + lastParamCov(0, 0)) / dZ / dZ;
125  // Bending plane
126  double cl1Ey2 = cl1.getEy2();
127  lastParamCov(2, 2) = cl2.getEy2();
128  lastParamCov(2, 3) = -lastParamCov(2, 2) / dZ;
129  lastParamCov(3, 2) = lastParamCov(2, 3);
130  lastParamCov(3, 3) = (1000. * cl1Ey2 + lastParamCov(2, 2)) / dZ / dZ;
131  // Inverse bending momentum (vertex resolution + bending slope resolution + 10% error on dipole parameters+field)
132  if (TrackExtrap::isFieldON()) {
133  lastParamCov(4, 4) =
134  ((SBendingVertexDispersion * SBendingVertexDispersion +
135  (cl1.getZ() * cl1.getZ() * lastParamCov(2, 2) + cl2.getZ() * cl2.getZ() * 1000. * cl1Ey2) / dZ / dZ) /
136  bendingImpact / bendingImpact +
137  0.1 * 0.1) *
138  inverseBendingMomentum * inverseBendingMomentum;
139  lastParamCov(2, 4) = cl1.getZ() * lastParamCov(2, 2) * inverseBendingMomentum / bendingImpact / dZ;
140  lastParamCov(4, 2) = lastParamCov(2, 4);
141  lastParamCov(3, 4) = -(cl1.getZ() * lastParamCov(2, 2) + cl2.getZ() * 1000. * cl1Ey2) * inverseBendingMomentum /
142  bendingImpact / dZ / dZ;
143  lastParamCov(4, 3) = lastParamCov(3, 4);
144  } else {
145  lastParamCov(4, 4) = inverseBendingMomentum * inverseBendingMomentum;
146  }
147  param.setCovariances(lastParamCov);
148 
149  // set other parameters
150  param.setClusterPtr(&cl2);
151  param.setTrackChi2(0.);
152 }

155 void TrackFitter::addCluster(const TrackParam& startingParam, const Cluster& cl, TrackParam& param)
156 {
161 
162  if (cl.getZ() <= startingParam.getZ()) {
163  throw runtime_error("The new cluster must be upstream");
164  }
165 
166  // copy the current parameters into the new ones
167  param.setParameters(startingParam.getParameters());
168  param.setZ(startingParam.getZ());
169  param.setCovariances(startingParam.getCovariances());
170  param.setTrackChi2(startingParam.getTrackChi2());
171 
172  // add MCS effect in the current chamber
173  int currentChamber(startingParam.getClusterPtr()->getChamberId());
174  TrackExtrap::addMCSEffect(&param, SChamberThicknessInX0[currentChamber], -1.);
175 
176  // reset propagator for smoother
177  if (mSmooth) {
178  param.resetPropagator();
179  }
180 
181  // add MCS in missing chambers if any
182  int expectedChamber(currentChamber - 1);
183  currentChamber = cl.getChamberId();
184  while (currentChamber < expectedChamber) {
185  if (!TrackExtrap::extrapToZCov(&param, SDefaultChamberZ[expectedChamber], mSmooth)) {
186  throw runtime_error("Track extrapolation failed");
187  }
188  TrackExtrap::addMCSEffect(&param, SChamberThicknessInX0[expectedChamber], -1.);
189  expectedChamber--;
190  }
191 
192  // extrapolate to the z position of the new cluster
193  if (!TrackExtrap::extrapToZCov(&param, cl.getZ(), mSmooth)) {
194  throw runtime_error("Track extrapolation failed");
195  }
196 
197  // save extrapolated parameters and covariances for smoother
198  if (mSmooth) {
199  param.setExtrapParameters(param.getParameters());
200  param.setExtrapCovariances(param.getCovariances());
201  }
202 
203  // recompute the parameters
204  param.setClusterPtr(&cl);
205  runKalmanFilter(param);
206 }

209 void TrackFitter::smoothTrack(Track& track, bool finalize)
210 {
215 
216  auto itCurrentParam(track.begin());
217  auto itPreviousParam(itCurrentParam);
218  ++itCurrentParam;
219 
220  // smoothed parameters and covariances at first cluster = filtered parameters and covariances
221  itPreviousParam->setSmoothParameters(itPreviousParam->getParameters());
222  itPreviousParam->setSmoothCovariances(itPreviousParam->getCovariances());
223 
224  // local chi2 at first cluster = last additional chi2 provided by Kalman
225  itPreviousParam->setLocalChi2(itPreviousParam->getTrackChi2() - itCurrentParam->getTrackChi2());
226 
227  // recursively smooth the next parameters and covariances
228  do {
229  runSmoother(*itPreviousParam, *itCurrentParam);
230  ++itPreviousParam;
231  } while (++itCurrentParam != track.end());
232 
233  // update the regular parameters and covariances if requested
234  if (finalize) {
235  for (auto& param : track) {
236  param.setParameters(param.getSmoothParameters());
237  param.setCovariances(param.getSmoothCovariances());
238  }
239  }

243 void TrackFitter::runKalmanFilter(TrackParam& trackParam)
244 {
248 
249  // get actual track parameters (p)
250  TMatrixD param(trackParam.getParameters());
251 
252  // get new cluster parameters (m)
253  const Cluster* cluster = trackParam.getClusterPtr();
254  TMatrixD clusterParam(5, 1);
255  clusterParam.Zero();
256  clusterParam(0, 0) = cluster->getX();
257  clusterParam(2, 0) = cluster->getY();
258 
259  // compute the actual parameter weight (W)
260  TMatrixD paramWeight(trackParam.getCovariances());
261  if (paramWeight.Determinant() != 0) {
262  paramWeight.Invert();
263  } else {
264  throw runtime_error("Determinant = 0");
265  }
266 
267  // compute the new cluster weight (U)
268  TMatrixD clusterWeight(5, 5);
269  clusterWeight.Zero();
270  clusterWeight(0, 0) = 1. / cluster->getEx2();
271  clusterWeight(2, 2) = 1. / cluster->getEy2();
272 
273  // compute the new parameters covariance matrix ((W+U)^-1)
274  TMatrixD newParamCov(paramWeight, TMatrixD::kPlus, clusterWeight);
275  if (newParamCov.Determinant() != 0) {
276  newParamCov.Invert();
277  } else {
278  throw runtime_error("Determinant = 0");
279  }
280  trackParam.setCovariances(newParamCov);
281 
282  // compute the new parameters (p' = ((W+U)^-1)U(m-p) + p)
283  TMatrixD tmp(clusterParam, TMatrixD::kMinus, param); // m-p
284  TMatrixD tmp2(clusterWeight, TMatrixD::kMult, tmp); // U(m-p)
285  TMatrixD newParam(newParamCov, TMatrixD::kMult, tmp2); // ((W+U)^-1)U(m-p)
286  newParam += param; // ((W+U)^-1)U(m-p) + p
287  trackParam.setParameters(newParam);
288 
289  // compute the additional chi2 (= ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m))
290  tmp = newParam; // p'
291  tmp -= param; // (p'-p)
292  TMatrixD tmp3(paramWeight, TMatrixD::kMult, tmp); // W(p'-p)
293  TMatrixD addChi2Track(tmp, TMatrixD::kTransposeMult, tmp3); // ((p'-p)^-1)W(p'-p)
294  tmp = newParam; // p'
295  tmp -= clusterParam; // (p'-m)
296  TMatrixD tmp4(clusterWeight, TMatrixD::kMult, tmp); // U(p'-m)
297  addChi2Track += TMatrixD(tmp, TMatrixD::kTransposeMult, tmp4); // ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m)
298  trackParam.setTrackChi2(trackParam.getTrackChi2() + addChi2Track(0, 0));
299 }

302 void TrackFitter::runSmoother(const TrackParam& previousParam, TrackParam& param)
303 {
306 
307  // get variables
308  const TMatrixD& extrapParameters = previousParam.getExtrapParameters(); // X(k+1 k)
309  const TMatrixD& filteredParameters = param.getParameters(); // X(k k)
310  const TMatrixD& previousSmoothParameters = previousParam.getSmoothParameters(); // X(k+1 n)
311  const TMatrixD& propagator = previousParam.getPropagator(); // F(k)
312  const TMatrixD& extrapCovariances = previousParam.getExtrapCovariances(); // C(k+1 k)
313  const TMatrixD& filteredCovariances = param.getCovariances(); // C(k k)
314  const TMatrixD& previousSmoothCovariances = previousParam.getSmoothCovariances(); // C(k+1 n)
315 
316  // compute smoother gain: A(k) = C(kk) * F(k)^t * (C(k+1 k))^-1
317  TMatrixD extrapWeight(extrapCovariances);
318  if (extrapWeight.Determinant() != 0) {
319  extrapWeight.Invert(); // (C(k+1 k))^-1
320  } else {
321  throw runtime_error("Determinant = 0");
322  }
323  TMatrixD smootherGain(filteredCovariances, TMatrixD::kMultTranspose, propagator); // C(kk) * F(k)^t
324  smootherGain *= extrapWeight; // C(kk) * F(k)^t * (C(k+1 k))^-1
325 
326  // compute smoothed parameters: X(k n) = X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
327  TMatrixD tmpParam(previousSmoothParameters, TMatrixD::kMinus, extrapParameters); // X(k+1 n) - X(k+1 k)
328  TMatrixD smoothParameters(smootherGain, TMatrixD::kMult, tmpParam); // A(k) * (X(k+1 n) - X(k+1 k))
329  smoothParameters += filteredParameters; // X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
330  param.setSmoothParameters(smoothParameters);
331 
332  // compute smoothed covariances: C(k n) = C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
333  TMatrixD tmpCov(previousSmoothCovariances, TMatrixD::kMinus, extrapCovariances); // C(k+1 n) - C(k+1 k)
334  TMatrixD tmpCov2(tmpCov, TMatrixD::kMultTranspose, smootherGain); // (C(k+1 n) - C(k+1 k)) * (A(k))^t
335  TMatrixD smoothCovariances(smootherGain, TMatrixD::kMult, tmpCov2); // A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
336  smoothCovariances += filteredCovariances; // C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
337  param.setSmoothCovariances(smoothCovariances);
338 
339  // compute smoothed residual: r(k n) = cluster - X(k n)
340  const Cluster* cluster = param.getClusterPtr();
341  TMatrixD smoothResidual(2, 1);
342  smoothResidual.Zero();
343  smoothResidual(0, 0) = cluster->getX() - smoothParameters(0, 0);
344  smoothResidual(1, 0) = cluster->getY() - smoothParameters(2, 0);
345 
346  // compute weight of smoothed residual: W(k n) = (clusterCov - C(k n))^-1
347  TMatrixD smoothResidualWeight(2, 2);
348  smoothResidualWeight(0, 0) = cluster->getEx2() - smoothCovariances(0, 0);
349  smoothResidualWeight(0, 1) = -smoothCovariances(0, 2);
350  smoothResidualWeight(1, 0) = -smoothCovariances(2, 0);
351  smoothResidualWeight(1, 1) = cluster->getEy2() - smoothCovariances(2, 2);
352  if (smoothResidualWeight.Determinant() != 0) {
353  smoothResidualWeight.Invert();
354  } else {
355  throw runtime_error("Determinant = 0");
356  }
357 
358  // compute local chi2 = (r(k n))^t * W(k n) * r(k n)
359  TMatrixD tmpChi2(smoothResidual, TMatrixD::kTransposeMult, smoothResidualWeight); // (r(k n))^t * W(k n)
360  TMatrixD localChi2(tmpChi2, TMatrixD::kMult, smoothResidual); // (r(k n))^t * W(k n) * r(k n)
361  param.setLocalChi2(localChi2(0, 0));
362 }
