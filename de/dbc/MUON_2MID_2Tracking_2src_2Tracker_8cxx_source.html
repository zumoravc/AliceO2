99 void Tracker::process(gsl::span<const Cluster2D> clusters, bool accumulate)
100 {
103 
104  // Reset cluster and tracks information
105  for (auto& clIdx : mClusterIndexes) {

129 bool Tracker::processSide(bool isRight, bool isInward)
130 {
132  int firstCh = (isInward) ? 3 : 0;
133  int secondCh = (isInward) ? 2 : 1;
134  int rpcOffset1 = detparams::getDEId(isRight, firstCh, 0);
135  int rpcOffset2 = detparams::getDEId(isRight, secondCh, 0);
136 
137  // loop on RPCs in first plane
138  Track track;
139  for (int irpc = 0; irpc < 9; ++irpc) {
140  int deId1 = rpcOffset1 + irpc;
141  for (size_t icl1 = 0; icl1 < mClusterIndexes[deId1].size(); ++icl1) {
142  // loop on clusters of the RPC in the first plane
143  auto& cl1 = mClusters[mClusterIndexes[deId1][icl1].first];
144  int firstRpc = getFirstNeighbourRPC(irpc);
145  int lastRpc = getLastNeighbourRPC(irpc);
146  for (int irpc2 = firstRpc; irpc2 <= lastRpc; ++irpc2) {
147  // loop on (neighbour) RPCs in second plane
148  int deId2 = rpcOffset2 + irpc2;
149  for (size_t icl2 = 0; icl2 < mClusterIndexes[deId2].size(); ++icl2) {
150  // loop on clusters of the RPC in the second plane
151  auto& cl2 = mClusters[mClusterIndexes[deId2][icl2].first];
152 
153  if (!makeTrackSeed(track, cl1, cl2)) {
154  continue;
155  }
156 
157  track.setClusterMatched(firstCh, mClusterIndexes[deId1][icl1].first);
158  track.setClusterMatched(secondCh, mClusterIndexes[deId2][icl2].first);
159  track.setClusterMatched(3 - firstCh, -1);
160  track.setClusterMatched(3 - secondCh, -1);
161  std::invoke(mFollowTrack, this, track, isRight, isInward);
162  } // loop on clusters in second plane
163  } // loop on RPCs in second plane
164  } // loop on clusters in first plane
165  } // loop on RPCs in first plane
166  return true;
167 }

169 //______________________________________________________________________________
170 bool Tracker::makeTrackSeed(Track& track, const Cluster3D& cl1, const Cluster3D& cl2) const
171 {
173 
174  // First check if the delta_x between the two clusters is not too large
175  double dZ = cl2.zCoor - cl1.zCoor;
176  double dZ2 = dZ * dZ;
177  double nonBendingSlope = (cl2.xCoor - cl1.xCoor) / dZ;
178  double nonBendingImpactParam = std::abs(cl2.xCoor - cl2.zCoor * nonBendingSlope);
179  double nonBendingImpactParamErr = std::sqrt(
180  (cl1.zCoor * cl1.zCoor * cl2.sigmaX2 + cl2.zCoor * cl2.zCoor * cl1.sigmaX2) / dZ2);
181  if ((nonBendingImpactParam - mSigmaCut * nonBendingImpactParamErr) > mImpactParamCut) {
182  return false;
183  }
184 
185  // Then start making the track (from 2 points)
186  track.setPosition(cl2.xCoor, cl2.yCoor, cl2.zCoor);
187  track.setDirection(nonBendingSlope, (cl2.yCoor - cl1.yCoor) / dZ, 1.);
188  track.setCovarianceParameters(cl2.sigmaX2, // x-x
189  cl2.sigmaY2, // y-y
190  (cl1.sigmaX2 + cl2.sigmaX2) / dZ2, // slopeX-slopeX
191  (cl1.sigmaY2 + cl2.sigmaY2) / dZ2, // slopeY-slopeY
192  cl2.sigmaX2 / dZ, // x-slopeX
193  cl2.sigmaY2 / dZ); // y-slopeY
194 
195  return true;
196 }

199 bool Tracker::followTrackKeepAll(const Track& track, bool isRight, bool isInward)
200 {
210  std::array<int, 2> chamberOrder;
211  chamberOrder[0] = isInward ? 1 : 2;
212  chamberOrder[1] = isInward ? 0 : 3;
213 
214  Track copyTrack;
215 
216  bool isAdded = false;
217 
218  // loop on next two chambers
219  for (int ich = 0; ich < 2; ++ich) {
220  int rpcOffset = detparams::getDEId(isRight, chamberOrder[ich], 0);
221  for (int irpc = 0; irpc <= 8; ++irpc) {
222  int deId = rpcOffset + irpc;
223  for (auto& clIdx : mClusterIndexes[deId]) {
224  if (ich == 1 && clIdx.second) {
225  // If the cluster in the last chamber was already used as a fourth cluster of a track
226  // do not try to attach it standalone
227  continue;
228  }
229  copyTrack = track;
230  if (findAllClusters(copyTrack, clIdx.first, isRight, isInward, chamberOrder[ich], irpc)) {
231  // Extrapolate to first cluster in MT11 and compute the chi2
232  finalizeTrack(copyTrack);
233  // Add the track if it is not compatible or better than the ones we already have
234  tryAddTrack(copyTrack);
235  isAdded = true;
236  }
237  }
238  }
239  }
240  return isAdded;
241 }

243 //______________________________________________________________________________
244 bool Tracker::followTrackKeepBest(const Track& track, bool isRight, bool isInward)
245 {
248 
249  // This algorithm is works generally well,
250  // but tests show that it can fail reconstructing the generated track
251  // in case of two tracks close in space.
252  // The main issue is that we do not have charge distribution information, and the strips are large,
253  // so it can happen that the track obtained by combining the hits of the first track in MT1
254  // and of the second track in MT2 can sometimes have a better chi2 of the real tracks.
255  // Since this algorithm only keeps the best track, it might chose the fake instead of the real track.
256  // The algorithm is however fast, so it should be preferred in case of low tracks multiplicity
257 
258  std::array<int, 2> chamberOrder;
259  chamberOrder[0] = isInward ? 1 : 2;
260  chamberOrder[1] = isInward ? 0 : 3;
261 
262  Track bestTrack;
263  bestTrack.setChi2(2. * mSigmaCut * mSigmaCut);
264 
265  // loop on next two chambers
266  for (int ich = 0; ich < 2; ++ich) {
267  findNextCluster(track, isRight, isInward, chamberOrder[ich], 0, 8, bestTrack);
268  if (bestTrack.getNDF() == 2) {
269  // We already have a track with 4 clusters: no need to search for a track with only one cluster
270  // in the next chamber
271  break;
272  }
273  }
274 
275  if (bestTrack.getNDF() > 0) {
276  // Extrapolate to first cluster in MT11 and compute the chi2
277  finalizeTrack(bestTrack);
278 
279  // Add the track if it is not compatible or better than the ones we already have
280  if (tryAddTrack(bestTrack)) {
281  return true;
282  }
283  }
284 
285  return false;
286 }

289 bool Tracker::findNextCluster(const Track& track, bool isRight, bool isInward, int chamber, int firstRPC, int lastRPC,
290  Track& bestTrack) const
291 {
293  int nextChamber = (isInward) ? chamber - 1 : chamber + 1;
294  int rpcOffset = detparams::getDEId(isRight, chamber, 0);
295  Track newTrack;
296  for (int irpc = firstRPC; irpc <= lastRPC; ++irpc) {
297  int deId = rpcOffset + irpc;
298  for (size_t icl = 0; icl < mClusterIndexes[deId].size(); ++icl) {
299  auto& cl = mClusters[mClusterIndexes[deId][icl].first];
300  double addChi2AtCluster = tryOneCluster(track, cl, newTrack);
301  newTrack.setChi2(track.getChi2() + addChi2AtCluster);
302  if (newTrack.getChi2() > bestTrack.getChi2()) {
303  continue;
304  }
305  newTrack.setClusterMatched(detparams::getChamber(cl.deId), mClusterIndexes[deId][icl].first);
306  newTrack.setNDF(track.getNDF() + 1);
307  if (nextChamber >= 0 && nextChamber <= 3) {
308  // We found a cluster in the first chamber of the station
309  // We search for a cluster in the last chamber, this time limiting to the RPC above and below this one
310  findNextCluster(newTrack, isRight, isInward, nextChamber, getFirstNeighbourRPC(irpc), getLastNeighbourRPC(irpc), bestTrack);
311  }
312  if (newTrack.getNDF() >= bestTrack.getNDF() && newTrack.getChi2OverNDF() < bestTrack.getChi2OverNDF()) {
313  // Prefer tracks with a larger number of attached clusters, even if the chi2 is worse
314  bestTrack = newTrack;
315  }
316  } // loop on clusters
317  } // loop on RPC
318 
319  return (bestTrack.getNDF() > 2);
320 }

323 bool Tracker::findAllClusters(Track& track, int clIdx, bool isRight, bool isInward, int chamber, int irpc)
324 {
326  Track newTrack;
327  double addChi2AtCluster = tryOneCluster(track, mClusters[clIdx], newTrack);
328  if (addChi2AtCluster > mMaxChi2) {
329  return false;
330  }
331 
332  newTrack.setChi2(track.getChi2() + addChi2AtCluster);
333  newTrack.setClusterMatched(chamber, clIdx);
334  newTrack.setNDF(track.getNDF() + 1);
335  track = newTrack;
336 
337  // We found a cluster in the first chamber of the station
338  // We search for a cluster in the last chamber, limiting to the RPC above and below this one
339  int nextChamber = (isInward) ? chamber - 1 : chamber + 1;
340  if (nextChamber >= 0 && nextChamber <= 3) {
341  double bestChi2AtCluster = mMaxChi2;
342  int bestClusterIdx = -1;
343  int bestClusterDE = -1;
344  int rpcOffset = detparams::getDEId(isRight, nextChamber, 0);
345  Track bestTrack;
346  for (int jrpc = getFirstNeighbourRPC(irpc); jrpc <= getLastNeighbourRPC(irpc); ++jrpc) {
347  int deId2 = rpcOffset + jrpc;
348  for (size_t idx = 0; idx < mClusterIndexes[deId2].size(); ++idx) {
349  addChi2AtCluster = tryOneCluster(track, mClusters[mClusterIndexes[deId2][idx].first], newTrack);
350  if (addChi2AtCluster < bestChi2AtCluster) {
351  bestChi2AtCluster = addChi2AtCluster;
352  bestClusterIdx = idx;
353  bestClusterDE = deId2;
354  bestTrack = newTrack;
355  }
356  }
357  }
358 
359  if (bestClusterIdx >= 0) {
360  bestTrack.setChi2(track.getChi2() + bestChi2AtCluster);
361  bestTrack.setClusterMatched(nextChamber, mClusterIndexes[bestClusterDE][bestClusterIdx].first);
362  bestTrack.setNDF(track.getNDF() + 1);
363  mClusterIndexes[bestClusterDE][bestClusterIdx].second = true;
364  track = bestTrack;
365  }
366  }
367 
368  return true;
369 }

372 double Tracker::tryOneCluster(const Track& track, const Cluster3D& cluster, Track& newTrack) const
373 {
379  double dZ = cluster.zCoor - track.getPositionZ();
380  double dZ2 = dZ * dZ;
381  double cpos[2] = {cluster.xCoor, cluster.yCoor};
382  double cerr2[2] = {cluster.sigmaX2, cluster.sigmaY2};
383  double pos[2] = {track.getPositionX(), track.getPositionY()};
384  double newPos[2] = {0., 0.};
385  double dist[2] = {0., 0.};
386  double dir[2] = {track.getDirectionX(), track.getDirectionY()};
387  const std::array<float, 6> covParams = track.getCovarianceParameters();
388  for (int icoor = 0; icoor < 2; ++icoor) {
389  newPos[icoor] = pos[icoor] + dir[icoor] * dZ;
390  dist[icoor] = cpos[icoor] - newPos[icoor];
391  double err2 = covParams[icoor] + dZ2 * covParams[icoor + 2] + 2. * dZ * covParams[icoor + 4] + cerr2[icoor];
392  double distMax = mSigmaCut * std::sqrt(2. * err2) + 4.;
393  if (std::abs(dist[icoor]) > distMax) {
394  return 2. * mMaxChi2;
395  }
396  }
397 
398  newTrack = track;
399  newTrack.propagateToZ(cluster.zCoor);
400 
401  return runKalmanFilter(newTrack, cluster);
402 }

404 //__________________________________________________________________________
405 double Tracker::runKalmanFilter(Track& track, const Cluster3D& cluster) const
406 {
409 
410  double pos[2] = {track.getPositionX(), track.getPositionY()};
411  double dir[2] = {track.getDirectionX(), track.getDirectionY()};
412  double clusPos[2] = {cluster.xCoor, cluster.yCoor};
413 
414  std::array<float, 6> newCovParams;
415  const std::array<float, 6> covParams = track.getCovarianceParameters();
416  double newPos[2], newDir[2];
417  double clusterSigma[2] = {cluster.sigmaX2, cluster.sigmaY2};
418  double chi2 = 0.;
419  for (int idx = 0; idx < 2; ++idx) {
420  int slopeIdx = idx + 2;
421  int covIdx = idx + 4;
422 
423  double den = clusterSigma[idx] + covParams[idx];
424  assert(den != 0.);
425  // if ( den == 0. ) return 2.*mMaxChi2;
426 
427  // Compute the new covariance
428  // cov -> (W+U)^-1
429  // where W = (old_cov)^-1
430  // and U = inverse of uncertainties of cluster
431  // s_x^2 -> s_x^2 * cl_s_x^2 / (s_x^2 + cl_s_x^2)
432  newCovParams[idx] = covParams[idx] * clusterSigma[idx] / den;
433  // cov(x,slopeX) -> cov(x,slopeX) * cl_s_x^2 / (s_x^2 + cl_s_x^2)
434  newCovParams[covIdx] = covParams[covIdx] * clusterSigma[idx] / den;
435  // s_slopeX^2 -> s_slopeX^2 - cov(x,slopeX)^2 / (s_x^2 + cl_s_x^2)
436  newCovParams[slopeIdx] = covParams[slopeIdx] - covParams[covIdx] * covParams[covIdx] / den;
437 
438  double diff = clusPos[idx] - pos[idx];
439 
440  // New parameters: p' = ((W+U)^-1)U(m-p) + p
441  // x -> x + ( cl_x - x ) * s_x^2 / (s_x^2 + cl_s_x^2)
442  newPos[idx] = pos[idx] + diff * covParams[idx] / den;
443  // slopeX -> slopeX + ( cl_x - x ) * s_slopeX^2 / (s_x^2 + cl_s_x^2)
444  newDir[idx] = dir[idx] + diff * covParams[covIdx] / den;
445 
446  chi2 += diff * diff / den;
447  }
448 
449  // Save the new parameters
450  track.setPosition(newPos[0], newPos[1], cluster.zCoor);
451  track.setDirection(newDir[0], newDir[1], 1.);
452 
453  // Save the new parameters covariance matrix
454  track.setCovarianceParameters(newCovParams);
455 
456  return chi2;
457 }

459 //______________________________________________________________________________
460 void Tracker::finalizeTrack(Track& track)
461 {
464  int ndf = 0;
465  double chi2 = 0.;
466  for (int ich = 3; ich >= 0; --ich) {
467  int matchedClusterIdx = track.getClusterMatched(ich);
468  if (matchedClusterIdx < 0) {
469  continue;
470  }
471  ++ndf;
472  Cluster3D& cl(mClusters[matchedClusterIdx]);
473  track.propagateToZ(cl.zCoor);
474  double clPos[2] = {cl.xCoor, cl.yCoor};
475  double clErr2[2] = {cl.sigmaX2, cl.sigmaY2};
476  double trackPos[2] = {track.getPositionX(), track.getPositionY()};
477  double trackCov[2] = {track.getCovarianceParameter(Track::CovarianceParamIndex::VarX),
478  track.getCovarianceParameter(Track::CovarianceParamIndex::VarY)};
479  for (int icoor = 0; icoor < 2; ++icoor) {
480  double diff = trackPos[icoor] - clPos[icoor];
481  chi2 += diff * diff / (trackCov[icoor] + clErr2[icoor]);
482  }
483  }
484  track.setChi2(chi2);
485  track.setNDF(ndf);
486 }

488 //______________________________________________________________________________
489 bool Tracker::tryAddTrack(const Track& track)
490 {
496 
497  float chi2OverNDF = track.getChi2OverNDF();
498  // We divide the chi2 by two since we want to consider only the uncertainty
499  // on one of the two tracks. We further reduce to 0.4 since we want to account
500  // for the case where one of the two reconstructed tracks has a much better precision
501  // of the other
502  float chi2Cut = 0.4 * mSigmaCut * mSigmaCut;
503  for (auto checkTrack = mTracks.begin() + mTrackOffset; checkTrack != mTracks.end(); ++checkTrack) {
504  int nCommonClusters = 0;
505  for (int ich = 0; ich < 4; ++ich) {
506  if (track.getClusterMatched(ich) == checkTrack->getClusterMatched(ich)) {
507  ++nCommonClusters;
508  }
509  }
510  if (nCommonClusters == 4) {
511  return false;
512  }
513  if (nCommonClusters == 3 && track.isCompatible(*checkTrack, chi2Cut)) {
514  // The new track is compatible with an existing one
515  if (chi2OverNDF < checkTrack->getChi2OverNDF()) {
516  // The new track is more precise than the old one: replace it!
517  *checkTrack = track;
518  }
519  return false;
520  }
521  }
522 
523  // The new track is not compatible with the previous ones: keep it
524  mTracks.emplace_back(track);
525  return true;
526 }
