110  // common function for hit creation (can be called from multiple interfaces)
111  bool createOrAddHit(int detector,
112  int sector,
113  int currentMediumid,
114  bool issecondary,
115  int nphe,
116  int trackn,
117  int parent,
118  float tof,
119  float trackenergy,
120  Vector3D<float> const& xImp,
121  float eDep, float x, float y, float z, float px, float py, float pz)
122  {
123  // A new hit is created when there is nothing yet for this det + sector
124  if (mCurrentHitsIndices[detector - 1][sector] == -1) {
125  mTotLightPMC = mTotLightPMQ = 0;
126  if (currentMediumid == mMediumPMCid) {
127  mTotLightPMC = nphe;
128  } else if (currentMediumid == mMediumPMQid) {
129  mTotLightPMQ = nphe;
130  }
131 
132  Vector3D<float> pos(x, y, z);
133  Vector3D<float> mom(px, py, pz);
134  addHit(trackn, parent, issecondary, trackenergy, detector, sector,
135  pos, mom, tof, xImp, eDep, mTotLightPMC, mTotLightPMQ);
136  // stack->addHit(GetDetId());
137  mCurrentHitsIndices[detector - 1][sector] = mHits->size() - 1;
138 
139  mXImpact = xImp;
140  return true;
141  } else {
142  auto& curHit = (*mHits)[mCurrentHitsIndices[detector - 1][sector]];
143  // summing variables that needs to be updated (Eloss and light yield)
144  curHit.setNoNumContributingSteps(curHit.getNumContributingSteps() + 1);
145  int nPMC{0}, nPMQ{0};
146  if (currentMediumid == mMediumPMCid) {
147  mTotLightPMC += nphe;
148  nPMC = nphe;
149  } else if (currentMediumid == mMediumPMQid) {
150  mTotLightPMQ += nphe;
151  nPMQ = nphe;
152  }
153  if (nphe > 0) {
154  curHit.SetEnergyLoss(curHit.GetEnergyLoss() + eDep);
155  curHit.setPMCLightYield(curHit.getPMCLightYield() + nPMC);
156  curHit.setPMQLightYield(curHit.getPMQLightYield() + nPMQ);
157  }
158  return true;
159  }
160  }
