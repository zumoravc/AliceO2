33  OutputObj<TH1F> htgl_nocuts{TH1F("htgl_nocuts", "tgl tracks (#GeV)", 100, 0., 10.)};

49  void process(aod::Collision const& collision,
50  aod::BCs const& bcs,
51  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
52  {
53  //LOGP(error, "Trigger selection {}", std::string{triggersel});
54  int trigindex = int{triggerindex};
55  if (trigindex != -1) {
56  LOGF(info, "Selecting on trigger bit %d", trigindex);
57  uint64_t triggerMask = collision.bc().triggerMask();
58  bool isTriggerClassFired = triggerMask & 1ul << (trigindex - 1);
59  if (!isTriggerClassFired)
60  return;
61  }
62 
63  LOGF(info, "N. of Tracks for collision: %d", tracks.size());
64  o2::vertexing::DCAFitterN<2> df;
65  df.setBz(5.0);
66  // After finding the vertex, propagate tracks to the DCA. This is default anyway
67  df.setPropagateToPCA(true);
68  // do not consider V0 seeds with 2D circles crossing above this R. This is default anyway
69  df.setMaxR(200);
70  // do not consider V0 seeds with tracks Z-distance exceeding this. This is default anyway
71  df.setMaxDZIni(4);
72  // stop iterations if max correction is below this value. This is default anyway
73  df.setMinParamChange(1e-3);
74  // stop iterations if chi2 improves by less that this factor
75  df.setMinRelChi2Change(0.9);
76 
77  hvtxp_x_out->Fill(collision.posX());
78  hvtxp_y_out->Fill(collision.posY());
79  hvtxp_z_out->Fill(collision.posZ());
80  for (auto it0 = tracks.begin(); it0 != tracks.end(); ++it0) {
81  auto& track_0 = *it0;
82 
83  UChar_t clustermap_0 = track_0.itsClusterMap();
84  //fill track distribution before selection
85  hitsmap_nocuts->Fill(clustermap_0);
86  hpt_nocuts->Fill(track_0.pt());
87  htgl_nocuts->Fill(track_0.tgl());
88  bool isselected_0 = track_0.tpcNClsFound() > 70 && track_0.flags() & 0x4;
89  isselected_0 = isselected_0 && (TESTBIT(clustermap_0, 0) || TESTBIT(clustermap_0, 1));
90  if (!isselected_0)
91  continue;
92  //fill track distribution after selection
93  hitsmap_cuts->Fill(clustermap_0);
94  hpt_cuts->Fill(track_0.pt());
95  htgl_cuts->Fill(track_0.tgl());
96 
97  float x0_ = track_0.x();
98  float alpha0_ = track_0.alpha();
99  std::array<float, 5> arraypar0 = {track_0.y(), track_0.z(), track_0.snp(),
100  track_0.tgl(), track_0.signed1Pt()};
101  std::array<float, 15> covpar0 = {track_0.cYY(), track_0.cZY(), track_0.cZZ(),
102  track_0.cSnpY(), track_0.cSnpZ(),
103  track_0.cSnpSnp(), track_0.cTglY(), track_0.cTglZ(),
104  track_0.cTglSnp(), track_0.cTglTgl(),
105  track_0.c1PtY(), track_0.c1PtZ(), track_0.c1PtSnp(),
106  track_0.c1PtTgl(), track_0.c1Pt21Pt2()};
107  o2::track::TrackParCov trackparvar0(x0_, alpha0_, arraypar0, covpar0);
108  for (auto it1 = it0 + 1; it1 != tracks.end(); ++it1) {
109  auto& track_1 = *it1;
110  UChar_t clustermap_1 = track_1.itsClusterMap();
111  bool isselected_1 = track_1.tpcNClsFound() > 70 && track_1.flags() & 0x4;
112  isselected_1 = isselected_1 && (TESTBIT(clustermap_1, 0) || TESTBIT(clustermap_1, 1));
113  if (!isselected_1)
114  continue;
115  if (track_0.signed1Pt() * track_1.signed1Pt() > 0)
116  continue;
117  float x1_ = track_1.x();
118  float alpha1_ = track_1.alpha();
119  std::array<float, 5> arraypar1 = {track_1.y(), track_1.z(), track_1.snp(),
120  track_1.tgl(), track_1.signed1Pt()};
121  std::array<float, 15> covpar1 = {track_1.cYY(), track_1.cZY(), track_1.cZZ(),
122  track_1.cSnpY(), track_1.cSnpZ(),
123  track_1.cSnpSnp(), track_1.cTglY(), track_1.cTglZ(),
124  track_1.cTglSnp(), track_1.cTglTgl(),
125  track_1.c1PtY(), track_1.c1PtZ(), track_1.c1PtSnp(),
126  track_1.c1PtTgl(), track_1.c1Pt21Pt2()};
127  o2::track::TrackParCov trackparvar1(x1_, alpha1_, arraypar1, covpar1);
128  df.setUseAbsDCA(true);
129  int nCand = df.process(trackparvar0, trackparvar1);
130  if (nCand == 0)
131  continue;
132  const auto& vtx = df.getPCACandidate();
133  LOGF(info, "vertex x %f", vtx[0]);
134  hvtx_x_out->Fill(vtx[0]);
135  hvtx_y_out->Fill(vtx[1]);
136  hvtx_z_out->Fill(vtx[2]);
137  o2::track::TrackParCov trackdec0 = df.getTrack(0);
138  o2::track::TrackParCov trackdec1 = df.getTrack(1);
139  std::array<float, 3> pvec0;
140  std::array<float, 3> pvec1;
141  trackdec0.getPxPyPzGlo(pvec0);
142  trackdec1.getPxPyPzGlo(pvec1);
143  float masspion = 0.140;
144  float masskaon = 0.494;
145  float mass_ = invmass2prongs(pvec0[0], pvec0[1], pvec0[2], masspion,
146  pvec1[0], pvec1[1], pvec1[2], masskaon);
147  float masssw_ = invmass2prongs(pvec0[0], pvec0[1], pvec0[2], masskaon,
148  pvec1[0], pvec1[1], pvec1[2], masspion);
149  secvtx2prong(track_0.collisionId(),
150  collision.posX(), collision.posY(), collision.posZ(),
151  vtx[0], vtx[1], vtx[2], track_0.globalIndex(),
152  pvec0[0], pvec0[1], pvec0[2], track_0.y(),
153  track_1.globalIndex(), pvec1[0], pvec1[1], pvec1[2], track_1.y(),
154  mass_, masssw_);
155  hchi2dca->Fill(df.getChi2AtPCACandidate());
156  }
157  }
158  }
159 };

179  df.setMinRelChi2Change(0.9);

181  for (auto& secVtx2prong : secVtx2Prongs) {
182  LOGF(INFO, " ------- new event ---------");

186  LOGF(INFO, " track1 y from track %f", secVtx2prong.index1().y());
187 
188  float x0_ = secVtx2prong.index0().x();
189  float alpha0_ = secVtx2prong.index0().alpha();
190  std::array<float, 5> arraypar0 = {secVtx2prong.index0().y(), secVtx2prong.index0().z(),
191  secVtx2prong.index0().snp(), secVtx2prong.index0().tgl(),
192  secVtx2prong.index0().signed1Pt()};

194  secVtx2prong.index0().cZZ(), secVtx2prong.index0().cSnpY(),
195  secVtx2prong.index0().cSnpZ(), secVtx2prong.index0().cSnpSnp(),

202 
203  float x1_ = secVtx2prong.index1().x();
204  float alpha1_ = secVtx2prong.index1().alpha();
205  std::array<float, 5> arraypar1 = {secVtx2prong.index1().y(), secVtx2prong.index1().z(),
206  secVtx2prong.index1().snp(), secVtx2prong.index1().tgl(),
207  secVtx2prong.index1().signed1Pt()};
208  std::array<float, 15> covpar1 = {secVtx2prong.index1().cYY(), secVtx2prong.index1().cZY(),
209  secVtx2prong.index1().cZZ(), secVtx2prong.index1().cSnpY(),
210  secVtx2prong.index1().cSnpZ(), secVtx2prong.index1().cSnpSnp(),
211  secVtx2prong.index1().cTglY(), secVtx2prong.index1().cTglZ(),
212  secVtx2prong.index1().cTglSnp(), secVtx2prong.index1().cTglTgl(),
213  secVtx2prong.index1().c1PtY(), secVtx2prong.index1().c1PtZ(),
214  secVtx2prong.index1().c1PtSnp(), secVtx2prong.index1().c1PtTgl(),
215  secVtx2prong.index1().c1Pt21Pt2()};
216  o2::track::TrackParCov trackparvar1(x1_, alpha1_, arraypar1, covpar1);
217 
218  df.setUseAbsDCA(true);
219  //FIXME: currently I rebuild the vertex for each track-track pair and
220  //select the candidate via its index. It is redundant cause the secondary
221  //vertex recostruction is performed more than once for each dca candidate
222  int nCand = df.process(trackparvar0, trackparvar1);
223  if (nCand == 0) {
224  LOGF(error, " DCAFitter failing in the candidate building: it should not happen");
225  }
226  const auto& secvtx = df.getPCACandidate();
227  float masspion = 0.140;
228  float masskaon = 0.494;

234  secVtx2prong.pz0(), masskaon,
235  secVtx2prong.px1(), secVtx2prong.py1(),
236  secVtx2prong.pz1(), masspion);
237  cand2prong(mass_, masssw_);
238  o2::track::TrackParCov trackdec0 = df.getTrack(0);
239  o2::track::TrackParCov trackdec1 = df.getTrack(1);
240  std::array<float, 3> pvec0;
241  std::array<float, 3> pvec1;
242  trackdec0.getPxPyPzGlo(pvec0);
243  trackdec1.getPxPyPzGlo(pvec1);
244  LOGF(info, "Pt track 0 from table %f and from calc %f", secVtx2prong.px0(), pvec0[0]);
245  if (abs(secVtx2prong.px0() - pvec0[0]) > 0.000000001)
246  LOGF(info, "BIG ERRROR");
247  }
248  }
