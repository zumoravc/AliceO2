54 bool Encoder::open(const std::string name, std::string path)
55 {
56  bool status = false;
57 
58  // register links
59  o2::header::RAWDataHeader rdh;
60  for (int feeid = 0; feeid < 72; feeid++) {
61  // cru=0 --> FLP 0, endpoint 0 --> 18 links -> fees 0-17
62  // cru=1 --> FLP 0, endpoint 1 --> 18 links -> fees 18-35
63  // cru=2 --> FLP 1, endpoint 0 --> 18 links -> fees 36-53
64  // cru=3 --> FLP 1, endpoint 1 --> 18 links -> fees 54-71
65  rdh.feeId = feeid;
66  rdh.cruID = feeid / NLINKSPERCRU;
67  rdh.linkID = feeid % NLINKSPERCRU;
68  rdh.endPointID = rdh.cruID % 2;
69  // currently storing each CRU in a separate file
70  if (mCrateOn[feeid])
71  mFileWriter.registerLink(rdh, Form("%s/cru%02d%s", path.c_str(), rdh.cruID, name.c_str()));
72  }
73 
74  return status;
75 }

78 {

80  int cru = icrate / NLINKSPERCRU;
81  if (nbyte) {
82  if (mCrateOn[icrate]) {
83  //printf("flush crate %d -- byte = %d -- orbit = %d, bc = %d\n",icrate, nbyte, mIR.orbit, mIR.bc);
84  mFileWriter.addData(icrate, cru, icrate % NLINKSPERCRU, cru % 2, mIR, gsl::span(mBuffer[icrate], nbyte));
85  }
86  mIntegratedAllBytes += nbyte;
87  }
88  mUnion[icrate] = mStart[icrate];

110  mUnion[0] = mStart[0]; // rewind
111

120 void Encoder::encodeTRM(const std::vector<Digit>& summary, Int_t icrate, Int_t itrm, int& istart) // encode one TRM assuming digit vector sorted by electronic index
121 // return next TRM index (-1 if not in the same crate)
122 // start to convert digiti from istart --> then update istart to the starting position of the new TRM
123 {
124  if (mVerbose)
125  printf("Crate %d: encode TRM %d \n", icrate, itrm);
126 
127  // TRM HEADER
128  mUnion[icrate]->trmDataHeader.slotId = itrm;
129  mUnion[icrate]->trmDataHeader.eventWords = 0; // to be filled at the end
130  mUnion[icrate]->trmDataHeader.eventCnt = mEventCounter;
131  mUnion[icrate]->trmDataHeader.emptyBit = 0;
132  mUnion[icrate]->trmDataHeader.dataId = 4;
133  nextWord(icrate);
134 
135  // LOOP OVER CHAINS
136  for (int ichain = 0; ichain < 2; ichain++) {
137  // CHAIN HEADER
138  mUnion[icrate]->trmChainHeader.slotId = itrm;
139  mUnion[icrate]->trmChainHeader.bunchCnt = mIR.bc;
140  mUnion[icrate]->trmChainHeader.mbz = 0;
141  mUnion[icrate]->trmChainHeader.dataId = 2 * ichain;
142  nextWord(icrate);
143 
144  while (istart < summary.size()) { // fill hits
146  int whatChain = summary[istart].getElChainIndex();
147  if (whatChain != ichain)
148  break;
149  int whatTRM = summary[istart].getElTRMIndex();
150  if (whatTRM != itrm)
151  break;
152  int whatCrate = summary[istart].getElCrateIndex();
153  if (whatCrate != icrate)
154  break;
155 
156  int hittimeTDC = (summary[istart].getBC() - mEventCounter * Geo::BC_IN_WINDOW) * 1024 + summary[istart].getTDC(); // time in TDC bin within the TOF WINDOW
157 
158  if (hittimeTDC < 0) {
159  LOG(ERROR) << "Negative hit encoded " << hittimeTDC << ", something went wrong in filling readout window";
160  printf("%d %d %d\n", summary[istart].getBC(), mEventCounter * Geo::BC_IN_WINDOW, summary[istart].getTDC());
161  }
162  // leading time
163  mUnion[icrate]->trmDataHit.time = hittimeTDC;
164  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
165  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
166  mUnion[icrate]->trmDataHit.dataId = 0xa;
167  nextWord(icrate);
168 
169  // trailing time
170  mUnion[icrate]->trmDataHit.time = hittimeTDC + summary[istart].getTOT() * Geo::RATIO_TOT_TDC_BIN;
171  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
172  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
173  mUnion[icrate]->trmDataHit.dataId = 0xc;
174  nextWord(icrate);
175 
176  istart++;
177  }
178 
179  // CHAIN TRAILER
180  mUnion[icrate]->trmChainTrailer.status = 0;
181  mUnion[icrate]->trmChainTrailer.mbz = 0;
182  mUnion[icrate]->trmChainTrailer.eventCnt = mEventCounter;
183  mUnion[icrate]->trmChainTrailer.dataId = 1 + 2 * ichain;
184  nextWord(icrate);
185  }
186 
187  // TRM TRAILER
188  mUnion[icrate]->trmDataTrailer.trailerMark = 3;
189  mUnion[icrate]->trmDataTrailer.eventCRC = 0; // to be implemented
190  mUnion[icrate]->trmDataTrailer.tempValue = 0;
191  mUnion[icrate]->trmDataTrailer.tempAddress = 0;
192  mUnion[icrate]->trmDataTrailer.tempChain = 0;
193  mUnion[icrate]->trmDataTrailer.tempAck = 0;
194  mUnion[icrate]->trmDataTrailer.lutErrorBit = 0;
195  mUnion[icrate]->trmDataTrailer.dataId = 5;
196  nextWord(icrate);
197 }

199 bool Encoder::encode(std::vector<std::vector<o2::tof::Digit>> digitWindow, int tofwindow) // pass a vector of digits in a TOF readout window, tof window is the entry in the vector-of-vector of digits needed to extract bunch id and orbit
200 {
201  if (digitWindow.size() != Geo::NWINDOW_IN_ORBIT) {
202  printf("Something went wrong in encoding: digitWindow=%lu different from %d\n", digitWindow.size(), Geo::NWINDOW_IN_ORBIT);
203  return 999;
204  }
205 
206  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
207  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
208  // caching electronic indexes in digit array
209  for (auto dig = summary.begin(); dig != summary.end(); dig++) {
210  int digitchannel = dig->getChannel();
211  dig->setElectronicIndex(Geo::getECHFromCH(digitchannel));
212  }
213 
214  // sorting by electroni indexes
215  std::sort(summary.begin(), summary.end(),
216  [](Digit a, Digit b) { return a.getElectronicIndex() < b.getElectronicIndex(); });
217  }
218 
219 #ifdef VERBOSE
220  if (mVerbose)
221  std::cout << "-------- START ENCODE EVENT ----------------------------------------" << std::endl;
222 #endif
223  auto start = std::chrono::high_resolution_clock::now();
224 
225  mEventCounter = tofwindow; // tof window index
226  mIR.orbit = mEventCounter / Geo::NWINDOW_IN_ORBIT;
227 
228  for (int i = 0; i < 72; i++)
229  mNextWordStatus[i] = false;
230 
231  int bcFirstWin;
232 
233  // encode all windows
234  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
235  mEventCounter = tofwindow + iwin; // tof window index
236 
237  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
238 
239  mIR.bc = ((mEventCounter % Geo::NWINDOW_IN_ORBIT) * Geo::BC_IN_ORBIT) / Geo::NWINDOW_IN_ORBIT + mFirstBC; // bunch crossing in the current orbit at the beginning of the window.
240 
241  if (iwin == 0)
242  bcFirstWin = mIR.bc;
243 
244  int icurrentdigit = 0;
245  // TOF data header
246  for (int i = 0; i < 72; i++) {
247  mTOFDataHeader[i] = reinterpret_cast<TOFDataHeader_t*>(mUnion[i]);
248  mTOFDataHeader[i]->bytePayload = 0; // event length in byte (to be filled later)
249  mTOFDataHeader[i]->mbz = 0;
250  mTOFDataHeader[i]->dataId = 4;
251  nextWord(i);
252 
253  mUnion[i]->tofOrbit.orbit = mIR.orbit;
254  nextWord(i);
255 
256  mDRMDataHeader[i] = reinterpret_cast<DRMDataHeader_t*>(mUnion[i]);
257  mDRMDataHeader[i]->slotId = 1;
258  mDRMDataHeader[i]->eventWords = 0; // event length in word (to be filled later) --> word = byte/4
259  mDRMDataHeader[i]->drmId = i;
260  mDRMDataHeader[i]->dataId = 4;
261  nextWord(i);
262 
263  mUnion[i]->drmHeadW1.slotId = 1;
264  mUnion[i]->drmHeadW1.partSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
265  mUnion[i]->drmHeadW1.clockStatus = 2;
266  mUnion[i]->drmHeadW1.drmhVersion = 0x12;
267  mUnion[i]->drmHeadW1.drmHSize = 5;
268  mUnion[i]->drmHeadW1.mbza = 0;
269  mUnion[i]->drmHeadW1.mbzb = 0;
270  mUnion[i]->drmHeadW1.dataId = 4;
271  nextWord(i);
272 
273  mUnion[i]->drmHeadW2.slotId = 1;
274  mUnion[i]->drmHeadW2.enaSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
275  mUnion[i]->drmHeadW2.mbz = 0;
276  mUnion[i]->drmHeadW2.faultSlotMask = 0;
277  mUnion[i]->drmHeadW2.readoutTimeOut = 0;
278  mUnion[i]->drmHeadW2.dataId = 4;
279  nextWord(i);
280 
281  mUnion[i]->drmHeadW3.slotId = 1;
282  mUnion[i]->drmHeadW3.gbtBunchCnt = mIR.bc;
283  mUnion[i]->drmHeadW3.locBunchCnt = 0;
284  mUnion[i]->drmHeadW3.dataId = 4;
285  nextWord(i);
286 
287  mUnion[i]->drmHeadW4.slotId = 1;
288  mUnion[i]->drmHeadW4.tempValue = 0;
289  mUnion[i]->drmHeadW4.mbza = 0;
290  mUnion[i]->drmHeadW4.tempAddress = 0;
291  mUnion[i]->drmHeadW4.mbzb = 0;
292  mUnion[i]->drmHeadW4.dataId = 4;
293  nextWord(i);
294 
295  mUnion[i]->drmHeadW5.slotId = 1;
296  mUnion[i]->drmHeadW5.eventCRC = 0;
297  mUnion[i]->drmHeadW5.irq = 0;
298  mUnion[i]->drmHeadW5.mbz = 0;
299  mUnion[i]->drmHeadW5.dataId = 4;
300  nextWord(i);
301 
302  int trm0 = 4 - (i % 2);
303  for (int itrm = trm0; itrm < 13; itrm++) {
304  encodeTRM(summary, i, itrm, icurrentdigit);
305  }
306 
307  mUnion[i]->drmDataTrailer.slotId = 1;
308  mUnion[i]->drmDataTrailer.locEvCnt = mEventCounter;
309  mUnion[i]->drmDataTrailer.mbz = 0;
310  mUnion[i]->drmDataTrailer.dataId = 5;
311  nextWord(i);
312  mUnion[i]->data = 0x70000000;
313  nextWord(i);
314 
315  mTOFDataHeader[i]->bytePayload = getSize(mTOFDataHeader[i], mUnion[i]);
316  mDRMDataHeader[i]->eventWords = mTOFDataHeader[i]->bytePayload / 4;
317  }
318 
319  // check that all digits were used
320  if (icurrentdigit < summary.size()) {
321  LOG(ERROR) << "Not all digits are been used : only " << icurrentdigit << " of " << summary.size();
322  }
323  }
324 
325  mIR.bc = bcFirstWin;
326 
327  for (int i = 0; i < 72; i++)
328  flush(i);
329 
330  mStartRun = false;
331 
332  auto finish = std::chrono::high_resolution_clock::now();
333  std::chrono::duration<double> elapsed = finish - start;
334  mIntegratedTime = elapsed.count();
335 
336  return false;
337 }
