375 template <typename DataT>
376 GPUhd() DataT Spline1D<DataT>::interpolate(DataT x) const
377 {
379  DataT u = convXtoU(x);
380  int iknot = getKnotIndexU(u);
381  const DataT* d = mFparameters + (2 * iknot) * mFdimensions;
382  DataT S = 0.;
383  interpolateU(1, getKnotNonSafe(iknot), &(d[0]), &(d[mFdimensions]),
384  &(d[2 * mFdimensions]), &(d[3 * mFdimensions]), u, &S);
385  return S;
386 }

388 template <typename DataT>
389 template <typename T>
390 GPUhd() void Spline1D<DataT>::interpolateU(int nFdim, const Spline1D<DataT>::Knot& knotL,
391  GPUgeneric() const T Sl[/*nFdim*/], GPUgeneric() const T Dl[/*nFdim*/],
392  GPUgeneric() const T Sr[/*nFdim*/], GPUgeneric() const T Dr[/*nFdim*/],
393  DataT u, GPUgeneric() T Su[/*nFdim*/])
394 {
399 
400  T uu = T(u - knotL.u);
401  T li = T(knotL.Li);
402  T v = uu * li; // scaled u
403  for (int dim = 0; dim < nFdim; ++dim) {
404  T df = (Sr[dim] - Sl[dim]) * li;
405  T a = Dl[dim] + Dr[dim] - df - df;
406  T b = df - Dl[dim] - a;
407  Su[dim] = ((a * v + b) * v + Dl[dim]) * uu + Sl[dim];
408  }
409 
410  /* another way to calculate f(u):
411  T uu = T(u - knotL.u);
412  T v = uu * T(knotL.Li); // scaled u
413  T vm1 = v-1;
414  T v2 = v * v;
415  float cSr = v2*(3-2*v);
416  float cSl = 1-cSr;
417  float cDl = v*vm1*vm1*knotL.L;
418  float cDr = v2*vm1*knotL.L;
419  return cSl*Sl + cSr*Sr + cDl*Dl + cDr*Dr;
420  */
421 }
