45 void ROframe::sortClusters()
46 {
47  Int_t nClsInLayer, binPrevIndex, clsMinIndex, clsMaxIndex, jClsLayer;
48  // sort the clusters in R-Phi
49  for (Int_t iLayer = 0; iLayer < constants::mft::LayersNumber; ++iLayer) {
50  if (mClusters[iLayer].size() == 0)
51  continue;
52  // sort clusters in layer according to the bin index
53  sort(mClusters[iLayer].begin(), mClusters[iLayer].end(),
54  [](Cluster& c1, Cluster& c2) { return c1.indexTableBin < c2.indexTableBin; });
55  // find the cluster local index range in each bin
56  // index = element position vector in the vector
57  nClsInLayer = mClusters[iLayer].size();
58  binPrevIndex = mClusters[iLayer].at(0).indexTableBin;
59  clsMinIndex = 0;
60  for (jClsLayer = 1; jClsLayer < nClsInLayer; ++jClsLayer) {
61  if (mClusters[iLayer].at(jClsLayer).indexTableBin == binPrevIndex)
62  continue;
63  clsMaxIndex = jClsLayer - 1;
64  std::pair<Int_t, Int_t> pair1(clsMinIndex, clsMaxIndex);
65  mClusterBinIndexRange[iLayer].insert(std::pair<Int_t, std::pair<Int_t, Int_t>>(binPrevIndex, pair1));
66  binPrevIndex = mClusters[iLayer].at(jClsLayer).indexTableBin;
67  clsMinIndex = jClsLayer;
68  } // clusters
69  // last cluster
70  clsMaxIndex = jClsLayer - 1;
71  std::pair<Int_t, Int_t> pair1(clsMinIndex, clsMaxIndex);
72  mClusterBinIndexRange[iLayer].insert(std::pair<Int_t, std::pair<Int_t, Int_t>>(binPrevIndex, pair1));
73  } // layers
74 }
