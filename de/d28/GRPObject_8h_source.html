151 inline float GRPObject::getBeamZ2A(beamDirection b) const
152 {
153  // Z/A of beam 0 or 1
154  int a = getBeamA(b);
155  return a ? getBeamZ(b) / static_cast<float>(a) : 0.f;
156 }
