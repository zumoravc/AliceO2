117 void DataInterpreterVSD::LoadEsdTracks(VisualisationEvent& /*event*/)
118 {
119  // Read reconstructed tracks from current event.
120 
121  if (mTrackList == nullptr) {
122  mTrackList = new TEveTrackList("ESD Tracks");
123  mTrackList->SetMainColor(6);
124  mTrackList->SetMarkerColor(kYellow);
125  mTrackList->SetMarkerStyle(4);
126  mTrackList->SetMarkerSize(0.5);
127  mTrackList->SetLineWidth(1);
128 
129  mTrackList->IncDenyDestroy();
130  } else {
131  mTrackList->DestroyElements();
132  }
133 
134  TEveTrackPropagator* trkProp = mTrackList->GetPropagator();
135  // !!!! Need to store field on file !!!!
136  // Can store TEveMagField ?
137  trkProp->SetMagField(0.5);
138  trkProp->SetStepper(TEveTrackPropagator::kRungeKutta);
139 
140  Int_t nTracks = mVSD->fTreeR->GetEntries();
141 
142  for (Int_t n = 0; n < nTracks; n++) {
143  mVSD->fTreeR->GetEntry(n);
144 
145  auto* track = new TEveTrack(&mVSD->fR, trkProp);
146 
147  track->SetAttLineAttMarker(mTrackList);
148  track->SetName(Form("ESD Track %d", mVSD->fR.fIndex));
149  track->SetStdTitle();
150  track->SetAttLineAttMarker(mTrackList);
151  mTrackList->AddElement(track);
152  }
153 
154  mTrackList->MakeTracks();
155 }
