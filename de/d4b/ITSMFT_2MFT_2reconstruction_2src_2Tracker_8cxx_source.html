32 void Tracker::process(const std::vector<Cluster>& clusters, std::vector<TrackMFT>& tracks, const std::vector<o2::itsmft::ROFRecord>& rofs)
33 {
34 
35  static int entry = 0;
36 
37  LOG(INFO);
38  LOG(INFO) << "Tracker::process() entry " << entry++ << ", number of threads: " << mNumOfThreads;
39 
40  Int_t nClFrame = 0;
41  Int_t numOfClustersLeft = clusters.size(); // total number of clusters
42  if (numOfClustersLeft == 0) {
43  LOG(WARNING) << "No clusters to process !";
44  return;
45  }
46 
47  auto start = std::chrono::system_clock::now();
48 
49  while (numOfClustersLeft > 0) {
50 
51  for (const auto& rof : rofs) {
52  nClFrame = loadClusters(clusters, rof);
53  if (!nClFrame) {
54  LOG(FATAL) << "Failed to select any cluster out of " << numOfClustersLeft << " check if cont/trig mode is correct";
55  }
56  numOfClustersLeft -= nClFrame;
57  auto end = std::chrono::system_clock::now();
58  std::chrono::duration<double> diff = end - start;
59  LOG(INFO) << "Loading clusters: " << nClFrame << " in single frame " << mROFrame << " : " << diff.count() << " s";
60 
61  start = end;
62 
63  processFrame(tracks);
64 
65  unloadClusters();
66  end = std::chrono::system_clock::now();
67  diff = end - start;
68  LOG(INFO) << "Processing time for single frame " << mROFrame << " : " << diff.count() << " s";
69 
70  start = end;
71  if (mContinuousMode) {
72  mROFrame++; // expect incremented frame in following clusters
73  }
74  } // end ROFs
75  }
76 }

80 {
81 
82  LOG(INFO) << "Tracker::processFrame(), number of threads: " << mNumOfThreads;
83 
84  std::vector<std::future<std::vector<TrackMFT>>> futures(mNumOfThreads);
85  Int_t numOfClusters = sLayers[0].getNumberOfClusters();
86  for (Int_t t = 0, first = 0; t < mNumOfThreads; t++) {
87  Int_t rem = t < (numOfClusters % mNumOfThreads) ? 1 : 0;
88  Int_t last = first + (numOfClusters / mNumOfThreads) + rem;
89  futures[t] = std::async(std::launch::async, &Tracker::trackInThread, this, first, last);
90  first = last;
91  }
92 
93  std::vector<std::vector<TrackMFT>> trackArray(mNumOfThreads);
94  for (Int_t t = 0; t < mNumOfThreads; t++) {
95  trackArray[t] = futures[t].get();
96  }
97 }

128 Int_t Tracker::loadClusters(const std::vector<Cluster>& clusters, const o2::itsmft::ROFRecord& rof)
129 {
130  auto first = rof.getFirstEntry();
131  auto number = rof.getNEntries();
132 
133  auto clusters_in_frame = gsl::make_span(&clusters[first], number);
134  for (const auto& c : clusters_in_frame) {
135  Int_t layer = mGeom->getLayer(c.getSensorID());
136  sLayers[layer].insertCluster(&c);
137  }
138 
139  if (number) {
140  std::vector<std::future<void>> fut;
141  for (Int_t l = 0; l < constants::LayersNumber; l += mNumOfThreads) {
142  for (Int_t t = 0; t < mNumOfThreads; t++) {
143  if (l + t >= constants::LayersNumber)
144  break;
145  auto f = std::async(std::launch::async, &Tracker::Layer::init, sLayers + (l + t));
146  fut.push_back(std::move(f));
147  }
148  for (Int_t t = 0; t < fut.size(); t++)
149  fut[t].wait();
150  }
151  }
152 
153  return number;
154 }
