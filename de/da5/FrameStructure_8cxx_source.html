114  vmc->Gsvolu(t4name, "TUBE", kAlu, thshT, 3);

116  int sig = 1;
117  int ipo = 1;
118  for (int i = 0; i < 5; i++) {
119  sig *= -1;
120  dx += 8.00;
121  dy = 4. * sig;
122  float dy1 = -(thshM[1] - 15.5) * sig;
123  float dy2 = -(thshM[1] - 7.5) * sig;
124 
125  vmc->Gspos(t2name, ipo++, mname, dx, dy, 0.025, rot1);
126  dx += 6.9;
127  vmc->Gspos(t2name, ipo++, mname, dx, dy, 0.025, rot1);
128

130  vmc->Gspos(t4name, i + 1, mname, dx - 3.45, dy2, 0.025, rot2);
131  }
132  dx += 8.;
133  vmc->Gspos(t1name, 2, mname, dx, 0., 0.025, rot1);
134  vmc->Gspos(t3name, 6, mname, dx - 3.45, -(thshM[1] - 7.5), 0.025, rot2);
135 }

137 void FrameStructure::createWebFrame(const char* name, float dHz, float theta0, float phi0)
138 {
139  //
140  // Create a web frame element
141  //
142  auto vmc = TVirtualMC::GetMC();
143 
144  phi0 = 0.;
145  const float krad2deg = 180. / TMath::Pi();
146  const float kdeg2rad = 1. / krad2deg;
147  const int kAir = mAirMedID;
148  const int kSteel = mSteelMedID;
149 
150  float ptrap[11];
151  char nameA[16];
152  snprintf(nameA, 16, "%sA", name);
153 
154  char nameI[16];
155  snprintf(nameI, 16, "%sI", name);
156 
157  theta0 *= kdeg2rad;
158  phi0 *= kdeg2rad;
159  float theta = TMath::Pi() / 2.;
160  float phi = TMath::ACos(TMath::Cos(theta0) * TMath::Cos(phi0));
161 
162  if (phi0 < 0)
163  phi = -phi;
164 
165  phi *= krad2deg;
166  theta *= krad2deg;
167 
168  ptrap[0] = dHz / 2;
169  ptrap[2] = theta;
170  ptrap[1] = phi;
171  ptrap[3] = 6. / cos(theta0 * kdeg2rad) / 2.;
172  ptrap[4] = 1.;
173  ptrap[5] = ptrap[4];
174  ptrap[6] = 0;
175  ptrap[7] = ptrap[3];
176  ptrap[8] = ptrap[4];
177  ptrap[9] = ptrap[4];
178  ptrap[10] = 0;
179  vmc->Gsvolu(name, "TRAP", kSteel, ptrap, 11);
180  vmc->Gsvolu(nameI, "TRAP", kSteel, ptrap, 11);
181  ptrap[3] = (6. - 1.) / cos(theta0 * kdeg2rad) / 2.;
182  ptrap[4] = 0.75;
183  ptrap[5] = ptrap[4];
184  ptrap[7] = ptrap[3];
185  ptrap[8] = ptrap[4];
186  ptrap[9] = ptrap[4];
187 
188  vmc->Gsvolu(nameA, "TRAP", kAir, ptrap, 11);
189  vmc->Gspos(nameA, 1, name, -0.25, 0.0, 0., 0, "ONLY");
190  vmc->Gspos(nameA, 2, nameI, +0.25, 0.0, 0., 0, "ONLY");
191  gGeoManager->GetVolume(name)->SetVisContainers();
192  gGeoManager->GetVolume(nameI)->SetVisContainers();
193 }
