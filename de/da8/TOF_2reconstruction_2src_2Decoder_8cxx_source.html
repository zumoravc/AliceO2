35 bool Decoder::open(const std::string name)
36 {
37  int nfileopened = NCRU;
38 
39  int fullsize = 0;
40  for (int i = 0; i < NCRU; i++) {
41  std::string nametmp;
42  nametmp.append(Form("cru%02d", i));
43  nametmp.append(name);
44 
45  if (mFile[i].is_open()) {
46  std::cout << "Warning: a file (" << i << ") was already open, closing" << std::endl;
47  mFile[i].close();
48  }
49  mFile[i].open(nametmp.c_str(), std::fstream::in | std::fstream::binary);
50  if (!mFile[i].is_open()) {
51  std::cout << "Cannot open " << nametmp << std::endl;
52  nfileopened--;
53  mSize[i] = 0;
54  mCruIn[i] = false;
55  } else {
56  mFile[i].seekg(0, mFile[i].end);
57  mSize[i] = mFile[i].tellg();
58  mFile[i].seekg(0);
59 
60  fullsize += mSize[i];
61 
62  mCruIn[i] = true;
63  }
64  }
65 
66  if (!nfileopened) {
67  std::cerr << "No streams available" << std::endl;
68  return true;
69  }
70 
71  // mBufferLocal.resize(fullsize);
72 
73  printf("Full input buffer size = %d byte\n", fullsize);
74 
75  char* pos = new char[fullsize]; //mBufferLocal.data();
76  for (int i = 0; i < NCRU; i++) {
77  if (!mCruIn[i])
78  continue;
79 
80  mBuffer[i] = pos;
81 
82  // read content of infile
83  mFile[i].read(mBuffer[i], mSize[i]);
84  mUnion[i] = reinterpret_cast<Union_t*>(mBuffer[i]);
85  mUnionEnd[i] = reinterpret_cast<Union_t*>(mBuffer[i] + mSize[i] - 1);
86 
87  pos += mSize[i];
88  }
89 
90  printf("Number of TOF compressed streamers = %d/%d\n", nfileopened, NCRU);
91 
92  close();
93 
94  return false;
95 }

112 {

114 
115  fromRawHit2Digit(icrate, itrm, itdc, ichain, channel, orbit, bunchid, time_ext + tdc, tot, digitInfo);
116 
117  mHitDecoded++;
118 
119  int isnext = digitInfo[3] * Geo::BC_IN_WINDOW_INV;
120 
121  if (isnext >= MAXWINDOWS) { // accumulate all digits which are not in the first windows
122 
123  insertDigitInFuture(digitInfo[0], digitInfo[1], digitInfo[2], digitInfo[3], 0, digitInfo[4], digitInfo[5]);
124  } else {

130 
131  // add digit
132  fillDigitsInStrip(cstrip, digitInfo[0], digitInfo[1], digitInfo[2], digitInfo[3], istrip);

136 void Decoder::readTRM(int icru, int icrate, int orbit, int bunchid)
137 {
138 
139  if (orbit < mFirstOrbit || (orbit == mFirstOrbit && bunchid < mFirstBunch)) {
140  mFirstOrbit = orbit;
141  mFirstBunch = bunchid;
142  }
143 
144  if (mVerbose)
145  printTRMInfo(icru);
146  int nhits = mUnion[icru]->frameHeader.numberOfHits;
147  int time_ext = mUnion[icru]->frameHeader.frameID << 13;
148  int itrm = mUnion[icru]->frameHeader.trmID;
149  int deltaBC = mUnion[icru]->frameHeader.deltaBC;
150 
151  if (deltaBC != 0)
152  printf("DeltaBC = %d\n", deltaBC);
153  mUnion[icru]++;
154  mIntegratedBytes[icru] += 4;
155 
156  // read hits
157  Int_t channel, echannel;
158  Int_t tdc;
159  Int_t tot;
160  Int_t bc;
161  Int_t time;
162 
163  std::array<int, 6> digitInfo;
164 
165  for (int i = 0; i < nhits; i++) {
166  fromRawHit2Digit(icrate, itrm, mUnion[icru]->packedHit.tdcID, mUnion[icru]->packedHit.chain, mUnion[icru]->packedHit.channel, orbit, bunchid, time_ext + mUnion[icru]->packedHit.time, mUnion[icru]->packedHit.tot, digitInfo);
167 
168  mHitDecoded++;
169 
170  if (mVerbose)
171  printHitInfo(icru);
172 
173  int isnext = digitInfo[3] * Geo::BC_IN_WINDOW_INV;
174 
175  if (isnext >= MAXWINDOWS) { // accumulate all digits which are not in the first windows
176 
177  insertDigitInFuture(digitInfo[0], digitInfo[1], digitInfo[2], digitInfo[3], 0, digitInfo[4], digitInfo[5]);
178  } else {
179  std::vector<Strip>* cstrip = mStripsCurrent; // first window
180  if (isnext)
181  cstrip = mStripsNext[isnext - 1]; // next window
182 
183  UInt_t istrip = digitInfo[0] / Geo::NPADS;
184 
185  // add digit
186  fillDigitsInStrip(cstrip, digitInfo[0], digitInfo[1], digitInfo[2], digitInfo[3], istrip);
187  }
188 
189  mUnion[icru]++;
190  mIntegratedBytes[icru] += 4;
191  }

194 void Decoder::fromRawHit2Digit(int icrate, int itrm, int itdc, int ichain, int channel, int orbit, int bunchid, int tdc, int tot, std::array<int, 6>& digitInfo)
195 {
196  // convert raw info in digit info (channel, tdc, tot, bc)
197  // tdc = packetHit.time + (frameHeader.frameID << 13)
198  int echannel = Geo::getECHFromIndexes(icrate, itrm, ichain, itdc, channel);
199  digitInfo[0] = Geo::getCHFromECH(echannel);
200  digitInfo[2] = tot;
201 
202  digitInfo[3] = int(orbit * o2::tof::Geo::BC_IN_ORBIT);
203  digitInfo[3] += bunchid;
204  digitInfo[3] += tdc / 1024;
205  digitInfo[1] = tdc % 1024;
206 
207  digitInfo[4] = orbit;
208  digitInfo[5] = bunchid;
209 }

219 bool Decoder::decode() // return a vector of digits in a TOF readout window
220 {
221  mReadoutWindowCurrent = 0;
222  mFirstOrbit = 0;
223  mFirstBunch = 0;
224 
225 #ifdef VERBOSE
226  if (mVerbose)
227  std::cout << "-------- START DECODE EVENTS IN THE HB ----------------------------------------" << std::endl;
228 #endif
229  auto start = std::chrono::high_resolution_clock::now();
230 
231  // start from the beginning of the timeframe
232  mEventTime = 0;
233 
234  // loop over CRUs
235  for (int icru = 0; icru < NCRU; icru++) {
236  if (!mCruIn[icru])
237  continue; // no data stream available for this cru
238 
239  printf("decoding cru %d\n", icru);
240 
241  while (mUnion[icru] < mUnionEnd[icru]) { // read all the buffer
242  // read open RDH
243  mRDH = reinterpret_cast<o2::header::RAWDataHeader*>(mUnion[icru]);
244  if (mVerbose)
245  printRDH();
246 
247  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
248  // note that RDH continue is not yet considered as option (to be added)
249  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
250 
251  // move after RDH
252  char* shift = reinterpret_cast<char*>(mRDH);
253  mUnion[icru] = reinterpret_cast<Union_t*>(shift + mRDH->headerSize);
254  mIntegratedBytes[icru] += mRDH->headerSize;
255 
256  if (mUnion[icru] >= mUnionEnd[icru])
257  continue; // end of data stream reac
258  for (int window = 0; window < Geo::NWINDOW_IN_ORBIT; window++) {
259  // read Crate Header
260  int bunchid = mUnion[icru]->crateHeader.bunchID;
261  int icrate = mUnion[icru]->crateHeader.drmID;
262  if (mVerbose)
263  printCrateInfo(icru);
264  mUnion[icru]++;
265  mIntegratedBytes[icru] += 4;
266 
267  //read Orbit
268  int orbit = mUnion[icru]->crateOrbit.orbitID;
269  if (mVerbose)
270  printf("orbit ID = %d\n", orbit);
271  mUnion[icru]++;
272  mIntegratedBytes[icru] += 4;
273 
274  while (!mUnion[icru]->frameHeader.mustBeZero) {
275  readTRM(icru, icrate, orbit, bunchid);
276  }
277 
278  // read Crate Trailer
279  if (mVerbose)
280  printCrateTrailerInfo(icru);
281  auto ndw = mUnion[icru]->crateTrailer.numberOfDiagnostics;
282  mUnion[icru]++;
283  mIntegratedBytes[icru] += 4;
284 
285  // loop over number of diagnostic words
286  for (int idw = 0; idw < ndw; ++idw) {
287  mUnion[icru]++;
288  mIntegratedBytes[icru] += 4;
289  }
290  }
291 
292  // read close RDH
293  mRDH = reinterpret_cast<o2::header::RAWDataHeader*>(nextPage(mRDH, mRDH->memorySize));
294  if (mVerbose)
295  printRDH();
296  mIntegratedBytes[icru] += mRDH->headerSize;
297 
298  // go to next page
299  mUnion[icru] = reinterpret_cast<Union_t*>(nextPage(mRDH, mRDH->memorySize));
300  }
301  }
302 
303  // since digits are not yet divided in tof readout window we need to do it
304  // flushOutputContainer does the job
305  FillWindows();
306 
307  return false;
308 }
