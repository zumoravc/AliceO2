33 static int GPUFailedMsgAI(const long long int error, const char* file, int line)
34 {
35  // Check for CUDA Error and in the case of an error display the corresponding error string
36  if (error == cudaSuccess) {
37  return (0);
38  }
39  GPUError("CUDA Error: %lld / %s (%s:%d)", error, cudaGetErrorString((cudaError_t)error), file, line);
