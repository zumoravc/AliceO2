38 void DataSamplingPolicy::configure(const ptree& config)
39 {
40  mName = config.get<std::string>("id");
41  if (mName.size() > 14) {
42  LOG(WARNING) << "DataSamplingPolicy name '" << mName << "' is longer than 14 characters, we have to trim it. "
43  << "Use a shorter policy name to avoid potential output name conflicts.";
44  mName.resize(14);
45  }
46 
47  auto subSpecString = config.get_optional<std::string>("subSpec").value_or("*");
48  auto subSpec = subSpecString.find_first_of("-*") != std::string::npos ? -1 : std::strtoull(subSpecString.c_str(), nullptr, 10);
49 
50  mPaths.clear();
51  size_t outputId = 0;
52  std::vector<InputSpec> inputSpecs = DataDescriptorQueryBuilder::parse(config.get<std::string>("query").c_str());
53 
54  for (const auto& inputSpec : inputSpecs) {
55 
56  if (DataSpecUtils::getOptionalSubSpec(inputSpec).has_value()) {
57  OutputSpec outputSpec{
58  {inputSpec.binding},
59  createPolicyDataOrigin(),
60  createPolicyDataDescription(mName, outputId++),
61  DataSpecUtils::getOptionalSubSpec(inputSpec).value(),
62  inputSpec.lifetime};
63 
64  mPaths.push_back({inputSpec, outputSpec});
65 
66  } else {
67  OutputSpec outputSpec{
68  {inputSpec.binding},
69  {createPolicyDataOrigin(), createPolicyDataDescription(mName, outputId++)},
70  inputSpec.lifetime};
71 
72  mPaths.push_back({inputSpec, outputSpec});
73  }
74 
75  if (outputId > 9) {
76  LOG(ERROR) << "Maximum 10 inputs in DataSamplingPolicy are supported. Call the developers if you really need more.";
77  break;
78  }
79  }
80 
81  mConditions.clear();
82  for (const auto& conditionConfig : config.get_child("samplingConditions")) {
83  mConditions.push_back(DataSamplingConditionFactory::create(conditionConfig.second.get<std::string>("condition")));
84  mConditions.back()->configure(conditionConfig.second);
85  }
86 
87  mFairMQOutputChannel = config.get_optional<std::string>("fairMQOutput").value_or("");
88 }

95 bool DataSamplingPolicy::decide(const o2::framework::DataRef& dataRef)
96 {
97  bool decision = std::all_of(mConditions.begin(), mConditions.end(),
98  [dataRef](std::unique_ptr<DataSamplingCondition>& condition) {
99  return condition->decide(dataRef);
100  });
101 
102  mTotalAcceptedMessages += decision;
103  mTotalEvaluatedMessages++;
104 
105  return decision;
106 }

108 const Output DataSamplingPolicy::prepareOutput(const ConcreteDataMatcher& input, Lifetime lifetime) const
109 {
110  auto result = mPaths.find(input);
111  if (result != mPaths.end()) {
112  auto dataType = DataSpecUtils::asConcreteDataTypeMatcher(result->second);
113  return Output{dataType.origin, dataType.description, input.subSpec, lifetime};
114  } else {
115  return Output{header::gDataOriginInvalid, header::gDataDescriptionInvalid};
116  }
117 }
