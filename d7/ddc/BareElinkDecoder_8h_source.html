137 std::string bitBufferString(const std::bitset<50>& bs, int imax)
138 {
139  std::string s;
140  for (int i = 0; i < 64; i++) {
141  if ((static_cast<uint64_t>(1) << i) > imax) {
142  break;
143  }
144  if (bs.test(i)) {
145  s += "1";
146  } else {
147  s += "0";
148  }
149  }
150  return s;
151 }

177 template <typename CHARGESUM>
178 void BareElinkDecoder<CHARGESUM>::appendOneBit(bool bit)
179 {
180  mNofBitSeen++;

182  mBitBuffer += bit * mMask;
183  mMask *= 2;

217 template <typename CHARGESUM>
218 void BareElinkDecoder<CHARGESUM>::findSync()
219 {
220  assert(mState == State::LookingForSync);
221  if (mBitBuffer != sampaSyncWord) {
222  mBitBuffer >>= 1;
223  mMask /= 2;
224  return;
225  }
226  changeState(State::LookingForHeader, HEADERSIZE);
227  mNofSync++;
228 }

230 template <typename CHARGESUM>
231 void BareElinkDecoder<CHARGESUM>::handleHeader()
232 {
233  assert(mState == State::LookingForHeader);
234 
235  mSampaHeader.uint64(mBitBuffer);
236 
237  ++mNofHeaderSeen;
238 
239  if (mSampaHeader.hasError()) {
240  ++mNofHammingErrors;
241  }
242 
243  switch (mSampaHeader.packetType()) {
244  case SampaPacketType::DataTruncated:
245  case SampaPacketType::DataTruncatedTriggerTooEarly:
246  case SampaPacketType::DataTriggerTooEarly:
247  case SampaPacketType::DataTriggerTooEarlyNumWords:
248  case SampaPacketType::DataNumWords:
249  // data with a problem is still data, i.e. there will
250  // probably be some data words to read in...
251  // so we fallthrough the simple Data case
252  case SampaPacketType::Data:
253  mNof10BitsWordsToRead = mSampaHeader.nof10BitWords();
254  changeState(State::ReadingNofSamples, 10);
255  break;
256  case SampaPacketType::Sync:
257  mNofSync++;
258  softReset();
259  break;
260  case SampaPacketType::HeartBeat:
261  fmt::printf("BareElinkDecoder %d: HEARTBEAT found. Should be doing sth about it ?\n", mDsId);
262  softReset();
263  break;
264  default:
265  throw std::logic_error("that should not be possible");
266  break;
267  }
268 }

295 template <typename CHARGESUM>
296 void BareElinkDecoder<CHARGESUM>::handleReadSample()
297 {
298  mSamples.push_back(mBitBuffer);
299  if (mNofSamples > 0) {
300  --mNofSamples;
301  }
302  oneLess10BitWord();
303  if (mNofSamples) {
304  changeToReadingData();
305  } else {
306  sendCluster();
307  if (mNof10BitsWordsToRead) {
308  changeState(State::ReadingNofSamples, 10);
309  } else {
310  changeState(State::LookingForHeader, HEADERSIZE);
311  }
312  }
313 }

364 template <typename CHARGESUM>
365 void BareElinkDecoder<CHARGESUM>::process()
366 {
367  switch (mState) {
368  case State::LookingForSync:
369  findSync();
370  break;
371  case State::LookingForHeader:
372  handleHeader();
373  break;
374  case State::ReadingNofSamples:
375  handleReadTimestamp();
376  break;
377  case State::ReadingTimestamp:
378  handleReadData();
379  break;
380  case State::ReadingSample:
381  handleReadSample();
382  break;
383  case State::ReadingClusterSum:
384  handleReadClusterSum();
385  break;
386  }
