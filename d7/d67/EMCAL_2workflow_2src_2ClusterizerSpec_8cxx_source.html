51 template <class InputType>
52 void ClusterizerSpec<InputType>::run(framework::ProcessingContext& ctx)
53 {
54  LOG(DEBUG) << "[EMCALClusterizer - run] called";
55  mTimer.Start(false);
56  std::string inputname;
57  std::string TrigName;
58 
59  if constexpr (std::is_same<InputType, o2::emcal::Digit>::value) {
60  inputname = "digits";
61  TrigName = "digitstrgr";
62  } else if constexpr (std::is_same<InputType, o2::emcal::Cell>::value) {
63  inputname = "cells";
64  TrigName = "cellstrgr";
65  }
66 
67  auto Inputs = ctx.inputs().get<gsl::span<InputType>>(inputname.c_str());
68  LOG(DEBUG) << "[EMCALClusterizer - run] Received " << Inputs.size() << " Cells/digits, running clusterizer ...";
69 
70  auto InputTriggerRecord = ctx.inputs().get<gsl::span<TriggerRecord>>(TrigName.c_str());
71  LOG(DEBUG) << "[EMCALClusterizer - run] Received " << InputTriggerRecord.size() << " Trigger Records, running clusterizer ...";
72 
73  mOutputClusters->clear();
74  mOutputCellDigitIndices->clear();
75  mOutputTriggerRecord->clear();
76  mOutputTriggerRecordIndices->clear();
77 
78  int currentStartClusters = mOutputClusters->size();
79  int currentStartIndices = mOutputCellDigitIndices->size();
80  for (auto iTrgRcrd : InputTriggerRecord) {
81 
82  mClusterizer.findClusters(gsl::span<const InputType>(&Inputs[iTrgRcrd.getFirstEntry()], iTrgRcrd.getNumberOfObjects())); // Find clusters on cells/digits (pass by ref)
83 
84  // Get found clusters + cell/digit indices for output
85  // * A cluster contains a range that correspond to the vector of cell/digit indices
86  // * The cell/digit index vector contains the indices of the clusterized cells/digits wrt to the original cell/digit array
87 
88  auto outputClustersTemp = mClusterizer.getFoundClusters();
89  auto outputCellDigitIndicesTemp = mClusterizer.getFoundClustersInputIndices();
90 
91  std::copy(outputClustersTemp->begin(), outputClustersTemp->end(), std::back_inserter(*mOutputClusters));
92  std::copy(outputCellDigitIndicesTemp->begin(), outputCellDigitIndicesTemp->end(), std::back_inserter(*mOutputCellDigitIndices));
93 
94  mOutputTriggerRecord->emplace_back(iTrgRcrd.getBCData(), currentStartClusters, outputClustersTemp->size());
95  mOutputTriggerRecordIndices->emplace_back(iTrgRcrd.getBCData(), currentStartIndices, outputCellDigitIndicesTemp->size());
96 
97  currentStartClusters = mOutputClusters->size();
98  currentStartIndices = mOutputCellDigitIndices->size();
99  }
100  LOG(DEBUG) << "[EMCALClusterizer - run] Writing " << mOutputClusters->size() << " clusters ...";
101  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, *mOutputClusters);
102  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "INDICES", 0, o2::framework::Lifetime::Timeframe}, *mOutputCellDigitIndices);
103 
104  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "CLUSTERSTRGR", 0, o2::framework::Lifetime::Timeframe}, *mOutputTriggerRecord);
105  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "INDICESTRGR", 0, o2::framework::Lifetime::Timeframe}, *mOutputTriggerRecordIndices);
106  mTimer.Stop();
107 }

115 o2::framework::DataProcessorSpec o2::emcal::reco_workflow::getClusterizerSpec(bool useDigits)
116 {
117  std::vector<o2::framework::InputSpec> inputs;
118  std::vector<o2::framework::OutputSpec> outputs;
119 
120  if (useDigits) {
121  inputs.emplace_back("digits", o2::header::gDataOriginEMC, "DIGITS", 0, o2::framework::Lifetime::Timeframe);
122  inputs.emplace_back("digitstrgr", o2::header::gDataOriginEMC, "DIGITSTRGR", 0, o2::framework::Lifetime::Timeframe);
123  } else {
124  inputs.emplace_back("cells", o2::header::gDataOriginEMC, "CELLS", 0, o2::framework::Lifetime::Timeframe);
125  inputs.emplace_back("cellstrgr", o2::header::gDataOriginEMC, "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe);
126  }
127 
128  outputs.emplace_back(o2::header::gDataOriginEMC, "CLUSTERS", 0, o2::framework::Lifetime::Timeframe);
129  outputs.emplace_back(o2::header::gDataOriginEMC, "INDICES", 0, o2::framework::Lifetime::Timeframe);
130  outputs.emplace_back(o2::header::gDataOriginEMC, "CLUSTERSTRGR", 0, o2::framework::Lifetime::Timeframe);
131  outputs.emplace_back(o2::header::gDataOriginEMC, "INDICESTRGR", 0, o2::framework::Lifetime::Timeframe);
132 
133  if (useDigits) {
134  return o2::framework::DataProcessorSpec{"EMCALClusterizerSpec",
135  inputs,
136  outputs,
137  o2::framework::adaptFromTask<o2::emcal::reco_workflow::ClusterizerSpec<o2::emcal::Digit>>()};
138  } else {
139  return o2::framework::DataProcessorSpec{"EMCALClusterizerSpec",
140  inputs,
141  outputs,
142  o2::framework::adaptFromTask<o2::emcal::reco_workflow::ClusterizerSpec<o2::emcal::Cell>>()};
143  }
