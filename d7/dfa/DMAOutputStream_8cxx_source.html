45 int DMAOutputStream::writeData(RawHeader header, gsl::span<const char> buffer)
46 {
47  if (!mInitialized)
48  open();
49 
50  constexpr int PAGESIZE = 8192;
51  // Handling of the termination word (0x001d3082): The termination word is added to the payload
52  // but not included in the payload size (as done in the hardware). Therefore it has to be subtracted
53  // from the maximum possible payload size
54  constexpr int MAXNWORDS = PAGESIZE - sizeof(header) - sizeof(uint32_t);
55  bool writeNext = true;
56  int pagecounter = header.pageCnt, currentindex = 0;
57  while (writeNext) {
58  int sizeRemain = buffer.size() - currentindex;
59  int nwordspage = MAXNWORDS;
60  if (sizeRemain < MAXNWORDS) {
61  // Last page
62  nwordspage = sizeRemain;
63  writeNext = false;
64  }
65  header.pageCnt = pagecounter;
66  header.memorySize = nwordspage + sizeof(RawHeader);
67  header.offsetToNext = 8192;
68 
69  writeDMAPage(header, gsl::span(buffer.data() + currentindex, nwordspage), PAGESIZE);
70 
71  if (writeNext) {
72  currentindex += nwordspage;
73  }
74  pagecounter++;
75  }
76  return pagecounter;
77 }

79 void DMAOutputStream::writeDMAPage(const RawHeader& header, gsl::span<const char> payload, int pagesize)
80 {
81  std::vector<char> dmapage(pagesize);
82  RawHeader* outheader = reinterpret_cast<RawHeader*>(dmapage.data());
83  *outheader = header;
84  char* outpayload = dmapage.data() + sizeof(header);
85  memcpy(outpayload, payload.data(), payload.size());

91 }
