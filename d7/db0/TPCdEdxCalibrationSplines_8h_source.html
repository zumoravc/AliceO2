187 inline void TPCdEdxCalibrationSplines::setSplinesFromFile(TFile& inpf)
188 {
189  FlatObject::startConstruction();
190 
191  int buffSize = 0;
192  int offsets1[FSplines];
193  int offsets2[FSplines];
194 
195  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
196  std::string splinename = fmt::format("spline_qMax_region{}", ireg);
197  o2::gpu::Spline2D<float, 1>* splineTmpqMax = o2::gpu::Spline2D<float, 1>::readFromFile(inpf, splinename.data());
198  mCalibSplinesqMax[ireg] = *splineTmpqMax;
199  buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
200  offsets1[ireg] = buffSize;
201  buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
202  }
203 
204  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
205  std::string splinename = fmt::format("spline_qTot_region{}", ireg);
206  o2::gpu::Spline2D<float, 1>* splineTmpqTot = o2::gpu::Spline2D<float, 1>::readFromFile(inpf, splinename.data());
207  mCalibSplinesqTot[ireg] = *splineTmpqTot;
208  buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
209  offsets2[ireg] = buffSize;
210  buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
211  }
212 
213  FlatObject::finishConstruction(buffSize);
214 
215  for (unsigned int i = 0; i < FSplines; i++) {
216  mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
217  }
218  for (unsigned int i = 0; i < FSplines; i++) {
219  mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
220  }
221 }
