316 inline void TrackPar::getLineParams(o2::utils::IntervalXY& ln, float& sna, float& csa) const
317 {
318  // get line parameterization as { x = x0 + xSlp*t, y = y0 + ySlp*t }
319  o2::utils::sincosf(getAlpha(), sna, csa);
320  o2::utils::rotateZ(getX(), getY(), ln.xP, ln.yP, sna, csa); // reference point in global frame
321  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp));
322  ln.dxP = csp * csa - snp * sna;
323  ln.dyP = snp * csa + csp * sna;
324 }

327 inline void TrackPar::getCircleParams(float bz, o2::utils::CircleXY& c, float& sna, float& csa) const
328 {
329  // get circle params in loc and lab frame, for straight line just set to global coordinates
330  getCircleParamsLoc(bz, c);
331  o2::utils::sincosf(getAlpha(), sna, csa);
332  o2::utils::rotateZ(c.xC, c.yC, c.xC, c.yC, sna, csa); // center in global frame
333 }

335 //_______________________________________________________
336 inline void TrackPar::getCircleParamsLoc(float bz, o2::utils::CircleXY& c) const
337 {
338  // get circle params in track local frame, for straight line just set to local coordinates
339  c.rC = getCurvature(bz);
340  if (std::abs(c.rC) > o2::constants::math::Almost0) {
341  c.rC = 1.f / getCurvature(bz);
342  float sn = getSnp(), cs = sqrtf((1. - sn) * (1. + sn));
343  c.xC = getX() - sn * c.rC; // center in tracking
344  c.yC = getY() + cs * c.rC; // frame. Note: r is signed!!!
345  c.rC = fabs(c.rC);
346  } else {
347  c.rC = 0.f; // signal straight line
348  c.xC = getX();
349  c.yC = getY();
350  }
351 }
