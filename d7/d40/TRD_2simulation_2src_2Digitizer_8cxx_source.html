96 void Digitizer::flush(DigitContainer& digits, o2::dataformats::MCTruthContainer<MCLabel>& labels)
97 {
98  // Convert Signals to Digits over all TRD detectors (in a parallel fashion)
99 #ifdef WITH_OPENMP
100  omp_set_num_threads(mNumThreads);
101 #pragma omp parallel for schedule(dynamic)
102 #endif
103  for (int det = 0; det < kNdet; ++det) {
104 #ifdef WITH_OPENMP
105  const int threadid = omp_get_thread_num();
106 #else
107  const int threadid = 0;
108 #endif
109  if (mSignalsMapCollection[det].size() == 0) {
110  continue;
111  }
112  bool status = convertSignalsToADC(det, mSignalsMapCollection[det], mDigitsCollection[det], threadid);
113  if (!status) {
114  LOG(WARN) << "TRD conversion of signals to digits failed for detector " << det;
115  continue; // go to the next chamber
116  }
117  }
118 
119  // Finalize
120  for (int det = 0; det < kNdet; ++det) {
121  std::move(mDigitsCollection[det].begin(), mDigitsCollection[det].end(), std::back_inserter(digits));
122  for (const auto& it : mSignalsMapCollection[det]) {
123  std::move(it.second.labels.begin(), it.second.labels.end(), std::back_inserter(mMergedLabels));
124  }
125  }
126  labels.addElements(labels.getIndexedSize(), mMergedLabels);
127  clearCollections();
128 }

130 void Digitizer::process(std::vector<HitType> const& hits, DigitContainer& digits, o2::dataformats::MCTruthContainer<MCLabel>& labels)
131 {
132  if (!mCalib) {
133  LOG(FATAL) << "TRD Calibration database not available";
134  }
135 
136  // if there are digits for longer than 3 microseconds, flush them
137  // todo: check for pileup
138  // constexpr double totalSamplingTime = kTimeBins / mCommonParam->GetSamplingFrequency();
139  if ((mTime - mLastTime) > 0) {
140  flush(digits, labels);
141  if (digits.size() == 0) {
142  LOG(WARN) << "Digit container is empty after flushing signals";
143  }
144  }
145  mLastTime = mTime; // always update the last time to the curret time for the next iteration
146 
147  // Get the a hit container for all the hits in a given detector then call convertHits for a given detector (0 - 539)
148  std::array<std::vector<HitType>, kNdet> hitsPerDetector;
149  getHitContainerPerDetector(hits, hitsPerDetector);
150 
151 #ifdef WITH_OPENMP
152  omp_set_num_threads(mNumThreads);
153 // Loop over all TRD detectors (in a parallel fashion)
154 #pragma omp parallel for schedule(dynamic)
155 #endif
156  for (int det = 0; det < kNdet; ++det) {
157 #ifdef WITH_OPENMP
158  const int threadid = omp_get_thread_num();
159 #else
160  const int threadid = 0;
161 #endif
162  auto& signalsMap = mSignalsMapCollection[det];
163  // Jump to the next detector if the detector is
164  // switched off, not installed, etc
165  if (mCalib->isChamberNoData(det)) {
166  continue;
167  }
168  if (!mGeo->chamberInGeometry(det)) {
169  continue;
170  }
171 
172  // Go to the next detector if there are no hits
173  if (hitsPerDetector[det].size() == 0) {
174  continue;
175  }
176 
177  if (!convertHits(det, hitsPerDetector[det], signalsMap, threadid)) {
178  LOG(WARN) << "TRD conversion of hits failed for detector " << det;
179  continue; // go to the next chamber
180  }
181  }
182 }

196 bool Digitizer::convertHits(const int det, const std::vector<HitType>& hits, SignalContainer& signalMapCont, int thread)
197 {
198  //
199  // Convert the detector-wise sorted hits to detector signals
200  //
201 
202  double padSignal[mNpad];
203 
204  const double calExBDetValue = mCalib->getExB(det); // T * V/cm (check units)
205  const TRDPadPlane* padPlane = mGeo->getPadPlane(det);
206  const int layer = mGeo->getLayer(det);
207  const float rowEndROC = padPlane->getRowEndROC();
208  const float row0 = padPlane->getRow0ROC();
209  const int nRowMax = padPlane->getNrows();
210  const int nColMax = padPlane->getNcols();
211 
212  // Loop over hits
213  for (const auto& hit : hits) {
214  const int qTotal = hit.GetCharge();
215  /*
216  Now the real local coordinate system of the ROC
217  column direction: locC
218  row direction: locR
219  time direction: locT
220  locR and locC are identical to the coordinates of the corresponding
221  volumina of the drift or amplification region.
222  locT is defined relative to the wire plane (i.e. middle of amplification
223  region), meaning locT = 0, and is negative for hits coming from the
224  drift region.
225  */
226  double locC = hit.getLocalC(); // col direction in amplification or drift volume
227  double locR = hit.getLocalR(); // row direction in amplification or drift volume
228  double locT = hit.getLocalT(); // time direction in amplification or drift volume
229  const double driftLength = -1 * locT; // The drift length in cm without diffusion
230  // Patch to take care of TR photons that are absorbed
231  // outside the chamber volume. A real fix would actually need
232  // a more clever implementation of the TR hit generation
233  if (qTotal < 0) {
234  if ((locR < rowEndROC) || (locR > row0)) {
235  continue;
236  }
237  if ((driftLength < DrMin) || (driftLength > DrMax)) {
238  continue;
239  }
240  }
241 
242  int rowE = padPlane->getPadRowNumberROC(locR);
243  if (rowE < 0) {
244  continue;
245  }
246 
247  double rowOffset = padPlane->getPadRowOffsetROC(rowE, locR);
248  double offsetTilt = padPlane->getTiltOffset(rowOffset);
249  int colE = padPlane->getPadColNumber(locC + offsetTilt);
250  if (colE < 0) {
251  continue;
252  }
253 
254  double absDriftLength = std::fabs(driftLength); // Normalized drift length
255  if (mCommonParam->ExBOn()) {
256  absDriftLength /= std::sqrt(1 / (1 + calExBDetValue * calExBDetValue));
257  }
258 
259  float driftVelocity = mCalib->getVDrift(det, colE, rowE); // The drift velocity
260  float t0 = mCalib->getT0(det, colE, rowE); // The T0 velocity
261 
262  // Loop over all created electrons
263  const int nElectrons = std::fabs(qTotal);
264  for (int el = 0; el < nElectrons; ++el) {
265  // Electron attachment
266  if (mSimParam->ElAttachOn()) {
267  if (mFlatRandomRings[thread].getNextValue() < absDriftLength * mElAttachProp) {
268  continue;
269  }
270  }
271  // scoped diffused coordinates for each electron
272  double locRd{locR}, locCd{locC}, locTd{locT};
273 
274  // Apply diffusion smearing
275  if (mSimParam->DiffusionOn()) {
276  if (!diffusion(driftVelocity, absDriftLength, calExBDetValue, locR, locC, locT, locRd, locCd, locTd, thread)) {
277  continue;
278  }
279  }
280 
281  // Apply E x B effects
282  if (mCommonParam->ExBOn()) {
283  locCd = locCd + calExBDetValue * driftLength;
284  }
285  // The electron position after diffusion and ExB in pad coordinates.
286  rowE = padPlane->getPadRowNumberROC(locRd);
287  if (rowE < 0) {
288  continue;
289  }
290  rowOffset = padPlane->getPadRowOffsetROC(rowE, locRd);
291  // The pad column (rphi-direction)
292  offsetTilt = padPlane->getTiltOffset(rowOffset);
293  colE = padPlane->getPadColNumber(locCd + offsetTilt);
294  if (colE < 0) {
295  continue;
296  }
297  const double colOffset = padPlane->getPadColOffset(colE, locCd + offsetTilt);
298  driftVelocity = mCalib->getVDrift(det, colE, rowE); // The drift velocity for the updated col and row
299  t0 = mCalib->getT0(det, colE, rowE); // The T0 velocity for the updated col and row
300  // Convert the position to drift time [mus], using either constant drift velocity or
301  // time structure of drift cells (non-isochronity, GARFIELD calculation).
302  // Also add absolute time of hits to take pile-up events into account properly
303  double driftTime;
304  if (mSimParam->TimeStructOn()) {
305  // Get z-position with respect to anode wire
306  double zz = row0 - locRd + padPlane->getAnodeWireOffset();
307  zz -= ((int)(2 * zz)) * 0.5;
308  if (zz > 0.25) {
309  zz = 0.5 - zz;
310  }
311  // Use drift time map (GARFIELD)
312  driftTime = mDriftEstimators[thread].TimeStruct(driftVelocity, 0.5 * AmWidth - 1.0 * locTd, zz) + hit.GetTime();
313  } else {
314  // Use constant drift velocity
315  driftTime = std::fabs(locTd) / driftVelocity + hit.GetTime(); // drift time in microseconds
316  }
317 
318  // Apply the gas gain including fluctuations
319  const double signal = -(mSimParam->GetGasGain()) * mLogRandomRings[thread].getNextValue();
320 
321  // Apply the pad response
322  if (mSimParam->PRFOn()) {
323  // The distance of the electron to the center of the pad in units of pad width
324  double dist = (colOffset - 0.5 * padPlane->getColSize(colE)) / padPlane->getColSize(colE);
325  // ********************************************************************************
326  // This is a fixed parametrization, i.e. not dependent on calibration values !
327  // ********************************************************************************
328  if (!(mPRF->getPRF(signal, dist, layer, padSignal))) {
329  continue;
330  }
331  } else {
332  padSignal[0] = 0;
333  padSignal[1] = signal;
334  padSignal[2] = 0;
335  }
336 
337  // The time bin (always positive), with t0 distortion
338  double timeBinIdeal = driftTime * mSamplingRate + t0;
339  // Protection
340  if (std::fabs(timeBinIdeal) > (2 * mMaxTimeBins)) {
341  timeBinIdeal = 2 * mMaxTimeBins;
342  }
343  int timeBinTruncated = ((int)timeBinIdeal);
344  // The distance of the position to the middle of the timebin
345  double timeOffset = ((float)timeBinTruncated + 0.5 - timeBinIdeal) / mSamplingRate;
346  // Sample the time response inside the drift region + additional time bins before and after.
347  // The sampling is done always in the middle of the time bin
348  const int firstTimeBin = std::max(timeBinTruncated, 0);
349  const int lastTimeBin = std::min(timeBinTruncated + mTimeBinTRFend, mMaxTimeBins);
350 
351  // loop over pads first then over timebins for better cache friendliness
352  // and less access to signalMapCont
353  for (int pad = 0; pad < mNpad; ++pad) {
354  int colPos = colE + pad - 1;
355  if (colPos < 0) {
356  continue;
357  }
358  if (colPos >= nColMax) {
359  break;
360  }
361 
362  const int key = calculateKey(det, rowE, colPos);
363  auto& currentSignalData = signalMapCont[key]; // Get the old signal or make a new one if it doesn't exist
364  auto& currentSignal = currentSignalData.signals;
365  auto& trackIds = currentSignalData.trackIds;
366  auto& labels = currentSignalData.labels;
367 
368  // add a label record
369  addLabel(hit, labels, trackIds);
370 
371  // add signal with crosstalk for the non-central pads only
372  if (colPos != colE) {
373  for (int tb = firstTimeBin; tb < lastTimeBin; ++tb) {
374  const double t = (tb - timeBinTruncated) / mSamplingRate + timeOffset;
375  const double timeResponse = mSimParam->TRFOn() ? mSimParam->TimeResponse(t) : 1;
376  const double crossTalk = mSimParam->CTOn() ? mSimParam->CrossTalk(t) : 0;
377  currentSignal[tb] += padSignal[pad] * (timeResponse + crossTalk);
378  } // end of loop time bins
379  } else {
380  for (int tb = firstTimeBin; tb < lastTimeBin; ++tb) {
381  const double t = (tb - timeBinTruncated) / mSamplingRate + timeOffset;
382  const double timeResponse = mSimParam->TRFOn() ? mSimParam->TimeResponse(t) : 1;
383  currentSignal[tb] += padSignal[pad] * timeResponse;
384  } // end of loop time bins
385  }
386  } // end of loop over pads
387  } // end of loop over electrons
388  } // end of loop over hits
389  return true;
390 }

408 bool Digitizer::convertSignalsToADC(const int det, SignalContainer& signalMapCont, DigitContainer& digits, int thread)
409 {
410  //
411  // Converts the sampled electron signals to ADC values for a given chamber
412  //
413  if (signalMapCont.size() == 0) {
414  return false;
415  }
416 
417  constexpr double kEl2fC = 1.602e-19 * 1.0e15; // Converts number of electrons to fC
418  double coupling = mSimParam->GetPadCoupling() * mSimParam->GetTimeCoupling(); // Coupling factor
419  double convert = kEl2fC * mSimParam->GetChipGain(); // Electronics conversion factor
420  double adcConvert = mSimParam->GetADCoutRange() / mSimParam->GetADCinRange(); // ADC conversion factor
421  double baseline = mSimParam->GetADCbaseline() / adcConvert; // The electronics baseline in mV
422  double baselineEl = baseline / convert; // The electronics baseline in electrons
423 
424  for (auto& signalMapIter : signalMapCont) {
425  const auto key = signalMapIter.first;
426  const int det = getDetectorFromKey(key);
427  const int row = getRowFromKey(key);
428  const int col = getColFromKey(key);
429  // halfchamber masking
430  int mcm = (int)(col / 18); // current group of 18 col pads
431  int halfchamberside = (mcm > 3 ? 1 : 0); // 0=Aside, 1=Bside
432 
433  // Halfchambers that are switched off, masked by mCalib
434  /* Something is wrong with isHalfChamberNoData - deactivated for now
435  if (mCalib->isHalfChamberNoData(det, halfchamberside)) {
436  continue;
437  }
438  */
439 
440  // Check whether pad is masked
441  // Bridged pads are not considered yet!!!
442  if (mCalib->isPadMasked(det, col, row) || mCalib->isPadNotConnected(det, col, row)) {
443  continue;
444  }
445 
446  float padgain = mCalib->getPadGainFactor(det, row, col); // The gain factor
447  if (padgain <= 0) {
448  LOG(FATAL) << "Not a valid gain " << padgain << ", " << det << ", " << col << ", " << row;
449  }
450 
451  // Loop over the all timebins in the ADC array
452  SignalArray& signalData = signalMapIter.second;
453  auto& signalArray = signalData.signals;
454  ArrayADC adcs{};
455  for (int tb = 0; tb < mMaxTimeBinsTRAP; ++tb) {
456  float signalAmp = (float)signalArray[tb]; // The signal amplitude
457  signalAmp *= coupling; // Pad and time coupling
458  signalAmp *= padgain; // Gain factors
459  // Add the noise, starting from minus ADC baseline in electrons
460  signalAmp = std::max((double)drawGaus(mGausRandomRings[thread], signalAmp, mSimParam->GetNoise()), -baselineEl);
461  signalAmp *= convert; // Convert to mV
462  signalAmp += baseline; // Add ADC baseline in mV
463  // Convert to ADC counts
464  // Set the overflow-bit fADCoutRange if the signal is larger than fADCinRange
465  ADC_t adc = 0;
466  if (signalAmp >= mSimParam->GetADCinRange()) {
467  adc = ((ADC_t)mSimParam->GetADCoutRange());
468  } else {
469  adc = std::lround(signalAmp * adcConvert);
470  }
471  // update the adc array value
472  adcs[tb] = adc;
473  } // loop over timebins
474  // Convert the map to digits here, and push them to the container
475  digits.emplace_back(det, row, col, adcs, getEventTime());
476  } // loop over digits
477  return true;
478 }

480 bool Digitizer::diffusion(float vdrift, float absdriftlength, float exbvalue,
481  float lRow0, float lCol0, float lTime0,
482  double& lRow, double& lCol, double& lTime, int thread)
483 {
484  //
485  // Applies the diffusion smearing to the position of a single electron.
486  // Depends on absolute drift length.
487  //
488  float diffL = 0.0;
489  float diffT = 0.0;
490  if (mDriftEstimators[thread].GetDiffCoeff(diffL, diffT, vdrift)) {
491  float driftSqrt = std::sqrt(absdriftlength);
492  float sigmaT = driftSqrt * diffT;
493  float sigmaL = driftSqrt * diffL;
494  lRow = drawGaus(mGausRandomRings[thread], lRow0, sigmaT);
495  if (mCommonParam->ExBOn()) {
496  const float exbfactor = 1.f / (1.f + exbvalue * exbvalue);
497  lCol = drawGaus(mGausRandomRings[thread], lCol0, sigmaT * exbfactor);
498  lTime = drawGaus(mGausRandomRings[thread], lTime0, sigmaL * exbfactor);
499  } else {
500  lCol = drawGaus(mGausRandomRings[thread], lCol0, sigmaT);
501  lTime = drawGaus(mGausRandomRings[thread], lTime0, sigmaL);
502  }
503  return true;
504  } else {
505  return false;
506  }
