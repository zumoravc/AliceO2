50 void StandaloneDebugger::fillCombinatoricsTree(std::array<std::vector<Cluster>, constants::its::LayersNumberVertexer>& clusters,
51  std::vector<Tracklet> comb01,
52  std::vector<Tracklet> comb12,
53  const ROframe* event)
54 {
55  assert(event != nullptr);
56  assert(mTreeStream != nullptr);
57 
58  for (auto& combination : comb01) {
59  o2::MCCompLabel lblClus0 = event->getClusterLabels(0, clusters[0][combination.firstClusterIndex].clusterId);
60  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][combination.secondClusterIndex].clusterId);
61  float c0z{clusters[0][combination.firstClusterIndex].zCoordinate};
62  float c1z{clusters[1][combination.secondClusterIndex].zCoordinate};
63  unsigned char isValidated{lblClus0.compare(lblClus1) == 1};
64  (*mTreeStream)
65  << "combinatorics01"
66  << "tanLambda=" << combination.tanLambda
67  << "phi=" << combination.phiCoordinate
68  << "c0z=" << c0z
69  << "c1z=" << c1z
70  << "isValidated=" << isValidated
71  << "lblClus0=" << lblClus0
72  << "lblClus1=" << lblClus1
73  << "\n";
74  }
75 
76  for (auto& combination : comb12) {
77  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][combination.secondClusterIndex].clusterId);
78  o2::MCCompLabel lblClus2 = event->getClusterLabels(2, clusters[2][combination.secondClusterIndex].clusterId);
79  float c1z{clusters[1][combination.firstClusterIndex].zCoordinate};
80  float c2z{clusters[2][combination.secondClusterIndex].zCoordinate};
81  unsigned char isValidated{lblClus1.compare(lblClus2) == 1};
82  (*mTreeStream)
83  << "combinatorics12"
84  << "tanLambda=" << combination.tanLambda
85  << "phi=" << combination.phiCoordinate
86  << "c1z=" << c1z
87  << "c2z=" << c2z
88  << "isValidated=" << isValidated
89  << "lblClus1=" << lblClus1
90  << "lblClus2=" << lblClus2
91  << "\n";

95 void StandaloneDebugger::fillTrackletSelectionTree(std::array<std::vector<Cluster>, constants::its::LayersNumberVertexer>& clusters,
96  std::vector<Tracklet> comb01,
97  std::vector<Tracklet> comb12,
98  std::vector<std::array<int, 2>> allowedTracklets,
99  const ROframe* event)
100 {
101  assert(event != nullptr);
102  assert(mTreeStream != nullptr);
103  int id = event->getROFrameId();
104  for (auto& trackletPair : allowedTracklets) {
105  o2::MCCompLabel lblClus0 = event->getClusterLabels(0, clusters[0][comb01[trackletPair[0]].firstClusterIndex].clusterId);
106  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][comb01[trackletPair[0]].secondClusterIndex].clusterId);
107  o2::MCCompLabel lblClus2 = event->getClusterLabels(2, clusters[2][comb12[trackletPair[1]].secondClusterIndex].clusterId);
108  unsigned char isValidated{lblClus0.compare(lblClus1) == 1 && lblClus0.compare(lblClus2) == 1};
109  float deltaPhi{comb01[trackletPair[0]].phiCoordinate - comb12[trackletPair[1]].phiCoordinate};
110  float deltaTanLambda{comb01[trackletPair[0]].tanLambda - comb12[trackletPair[1]].tanLambda};
111  mTreeStream->GetDirectory()->cd(); // in case of existing other open files
112  (*mTreeStream)
113  << "selectedTracklets"
114  << "ROframeId=" << id
115  << "deltaTanlambda=" << deltaTanLambda
116  << "deltaPhi=" << deltaPhi
117  << "isValidated=" << isValidated
118  << "cluster0z=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].zCoordinate
119  << "cluster0r=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].rCoordinate
120  << "cluster0phi=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].phiCoordinate
121  << "cluster1z=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].zCoordinate
122  << "cluster1r=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].rCoordinate
123  << "cluster1phi=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].phiCoordinate
124  << "cluster2z=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].zCoordinate
125  << "cluster2r=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].rCoordinate
126  << "lblClus0=" << lblClus0
127  << "lblClus1=" << lblClus1
128  << "lblClus2=" << lblClus2
129  << "\n";
130  }

158 void StandaloneDebugger::fillPairsInfoTree(std::vector<Line> lines, const ROframe* event)
159 {
160  assert(event != nullptr);
161  int id = event->getROFrameId();
162  for (unsigned int iLine1{0}; iLine1 < lines.size(); ++iLine1) { // compute centroids for every line pair
163  auto line1 = lines[iLine1];
164  for (unsigned int iLine2{iLine1 + 1}; iLine2 < lines.size(); ++iLine2) {
165  auto line2 = lines[iLine2];
166  ClusterLines cluster{-1, line1, -1, line2};
167  auto vtx = cluster.getVertex();
168  if (std::hypot(vtx[0], vtx[1]) < 1.98) {
169  float dcaPair = Line::getDCA(line1, line2);
170  if (dcaPair < 0.02) {
171  (*mTreeStream)
172  << "pairInfo"
173  << "centroids"
174  << "ROframeId=" << id
175  << "xCoord=" << vtx[0]
176  << "yCoord=" << vtx[1]
177  << "zCoord=" << vtx[2]
178  << "DCApair=" << dcaPair
179  << "\n";
180  }
181  }
182  }

249 int StandaloneDebugger::getBinIndex(const float value, const int size, const float min, const float max)
250 {
251  std::vector<double> divisions;
252  const double binsize = (max - min) / size;
253  divisions.resize(size + 1);
254  for (int i{0}; i <= size; ++i) {
255  divisions[i] = min + i * binsize;
256  }
257  return std::distance(divisions.begin(), TMath::BinarySearch(divisions.begin(), divisions.end(), value));
258 }
