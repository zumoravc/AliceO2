45 int genEvents::GetSlice(double GlobalPhi)
46 {
47  double phi = GlobalPhi;
48  // std::cout<<" GetSlice: phi = "<<phi<<std::endl;
49 
50  if (phi >= mTwoPi) {
51  phi -= mTwoPi;
52  }
53  if (phi < 0) {
54  phi += mTwoPi;
55  }
56  return (int)(phi / mSliceDAngle);
57 }

63 int genEvents::RecalculateSlice(GPUTPCGMPhysicalTrackModel& t, int& iSlice)
64 {
65  double phi = atan2(t.GetY(), t.GetX());
66  // std::cout<<" recalculate: phi = "<<phi<<std::endl;
67  int dSlice = GetDSlice(phi);
68 
69  if (dSlice == 0) {
70  return 0; // nothing to do
71  }
72  // std::cout<<" dSlice = "<<dSlice<<std::endl;
73  double dAlpha = dSlice * mSliceDAngle;
74  // rotate track on angle dAlpha
75 
76  t.Rotate(dAlpha);
77 
78  iSlice += dSlice;
79  if (iSlice >= 18) {
80  iSlice -= 18;
81  }
82  return 1;
83 }

147 int genEvents::GenerateEvent(const GPUParam& param, char* filename)
148 {
149  mRec->ClearIOPointers();
150  static int iEvent = -1;
151  iEvent++;
152  if (iEvent == 0) {
153  gRandom->SetSeed(configStandalone.seed);
154  }
155 
156  int nTracks = configStandalone.configEG.numberOfTracks; // Number of MC tracks, must be at least as large as the largest fMCID assigned above
157  cout << "NTracks " << nTracks << endl;
158  std::vector<GPUTPCMCInfo> mcInfo(nTracks);
159  memset(mcInfo.data(), 0, nTracks * sizeof(mcInfo[0]));
160 
161  // double Bz = param.ConstBz();
162  // std::cout<<"Bz[kG] = "<<param.BzkG()<<std::endl;
163 
164  GPUTPCGMPropagator prop;
165  {
166  prop.SetToyMCEventsFlag(kTRUE);
167  const GPUTPCGMMerger& merger = mRec->GetTPCMerger();
168  prop.SetPolynomialField(&merger.Param().polynomialField);
169  }
170 
171  // const double kCLight = 0.000299792458;
172  // Bz*=kCLight;
173 
174  std::vector<GenCluster> vClusters;
175  int clusterId = 0; // Here we count up the cluster ids we fill (must be unique).
176  // gRandom->SetSeed(0);
177  // unsigned int seed = gRandom->GetSeed();
178 
179  for (int itr = 0; itr < nTracks; itr++) {
180  // std::cout<<"Track "<<itr<<":"<<std::endl;
181  // gRandom->SetSeed(seed);
182 
183  mcInfo[itr].pid = -100; //-100: Unknown / other, 0: Electron, 1, Muon, 2: Pion, 3: Kaon, 4: Proton
184  mcInfo[itr].charge = 1;
185  mcInfo[itr].prim = 1; // Primary particle
186  mcInfo[itr].primDaughters = 0; // Primary particle with daughters in the TPC
187  mcInfo[itr].x = 0; // Position of MC track at entry of TPC / first hit in the TPC
188  mcInfo[itr].y = 0;
189  mcInfo[itr].z = 0;
190  mcInfo[itr].pX = 0; // Momentum of MC track at that position
191  mcInfo[itr].pY = 0;
192  mcInfo[itr].pZ = 0;
193 
194  GPUTPCGMPhysicalTrackModel t;
195  double dphi = mTwoPi / nTracks;
196  double phi = mSliceAngleOffset + dphi * itr;
197  double eta = gRandom->Uniform(-1.5, 1.5);
198 
199  double theta = 2 * std::atan(1. / exp(eta));
200  double lambda = theta - M_PI / 2;
201  // double theta = gRandom->Uniform(-60,60)*M_PI/180.;
202  double pt = .08 * std::pow(10, gRandom->Uniform(0, 2.2));
203 
204  double q = 1.;
205  int iSlice = GetSlice(phi);
206  phi = phi - GetSliceAngle(iSlice);
207 
208  // std::cout<<"phi = "<<phi<<std::endl;
209  double x0 = cosf(phi);
210  double y0 = sinf(phi);
211  double z0 = tanf(lambda);
212  t.Set(x0, y0, z0, pt * x0, pt * y0, pt * z0, q);
213 
214  if (RecalculateSlice(t, iSlice) != 0) {
215  std::cout << "Initial slice wrong!!!" << std::endl;
216  // exit(0);
217  }
218 
219  for (int iRow = 0; iRow < GPUCA_ROW_COUNT; iRow++) {
220  // if( iRow>=50 ) break; //SG!!!
221  float xRow = param.tpcGeometry.Row2X(iRow);
222  // transport to row
223  int err = 0;
224  for (int itry = 0; itry < 1; itry++) {
225  float B[3];
226  prop.GetBxByBz(GetSliceAngle(iSlice), t.GetX(), t.GetY(), t.GetZ(), B);
227  float dLp = 0;
228  err = t.PropagateToXBxByBz(xRow, B[0], B[1], B[2], dLp);
229  if (err) {
230  std::cout << "Can not propagate to x = " << xRow << std::endl;
231  t.Print();
232  break;
233  }
234  if (fabsf(t.GetZ()) >= 250.) {
235  std::cout << "Can not propagate to x = " << xRow << ": Z outside the volume" << std::endl;
236  t.Print();
237  err = -1;
238  break;
239  }
240  // rotate track coordinate system to current sector
241  int isNewSlice = RecalculateSlice(t, iSlice);
242  if (!isNewSlice) {
243  break;
244  } else {
245  std::cout << "track " << itr << ": new slice " << iSlice << " at row " << iRow << std::endl;
246  }
247  }
248  if (err) {
249  break;
250  }
251  // std::cout<<" track "<<itr<<": Slice "<<iSlice<<" row "<<iRow<<" params :"<<std::endl;
252  // t.Print();
253  // track at row iRow, slice iSlice
254  if (iRow == 0) { // store MC track at first row
255  // std::cout<<std::setprecision( 20 );
256  // std::cout<<"track "<<itr<<": x "<<t.X()<<" y "<<t.Y()<<" z "<<t.Z()<<std::endl;
257  GPUTPCGMPhysicalTrackModel tg(t); // global coordinates
258  tg.Rotate(-GetSliceAngle(iSlice));
259 
260  mcInfo[itr].pid = 2; // pion
261  mcInfo[itr].charge = 3 * q;
262  mcInfo[itr].x = tg.GetX(); // Position of MC track at entry of TPC / first hit in the TPC
263  mcInfo[itr].y = tg.GetY();
264  mcInfo[itr].z = tg.GetZ();
265  mcInfo[itr].pX = tg.GetPx(); // Momentum of MC track at that position
266  mcInfo[itr].pY = tg.GetPy();
267  mcInfo[itr].pZ = tg.GetPz();
268  // std::cout<<" mc Z = "<<tg.GetZ()<<std::endl;
269  }
270 
271  // create cluster
272  GenCluster c;
273  float sigmaY = 0.3;
274  float sigmaZ = 0.5;
275  const int rowType = iRow < 64 ? 0 : iRow < 128 ? 2 : 1;
276  t.UpdateValues();
277  param.GetClusterErrors2(rowType, t.GetZ(), t.GetSinPhi(), t.GetDzDs(), sigmaY, sigmaZ);
278  sigmaY = std::sqrt(sigmaY);
279  sigmaZ = std::sqrt(sigmaZ);
280  mClusterError[rowType][0]->Fill(sigmaY);
281  mClusterError[rowType][1]->Fill(sigmaZ);
282  // std::cout<<sigmaY<<" "<<sigmaY<<std::endl;
283  // if( sigmaY > 0.5 ) sigmaY = 0.5;
284  // if( sigmaZ > 0.5 ) sigmaZ = 0.5;
285  c.sector = (t.GetZ() >= 0.) ? iSlice : iSlice + 18;
286  c.row = iRow;
287  c.mcID = itr;
288  c.x = t.GetX();
289  c.y = t.GetY() + GetGaus(sigmaY);
290  c.z = t.GetZ() + GetGaus(sigmaZ);
291  c.id = clusterId++;
292  vClusters.push_back(c);
293  } // iRow
294  } // itr
295 
296  std::vector<AliHLTTPCClusterMCLabel> labels;
297 
298  std::unique_ptr<GPUTPCClusterData> clSlices[GPUChainTracking::NSLICES];
299 
300  for (int iSector = 0; iSector < (int)GPUChainTracking::NSLICES; iSector++) // HLT Sector numbering, sectors go from 0 to 35, all spanning all rows from 0 to 158.
301  {
302  int nNumberOfHits = 0;
303  for (unsigned int i = 0; i < vClusters.size(); i++) {
304  if (vClusters[i].sector == iSector) {
305  nNumberOfHits++;
306  }
307  }
308  // For every sector we first have to fill the number of hits in this sector to the file
309  mRec->mIOPtrs.nClusterData[iSector] = nNumberOfHits;
310 
311  GPUTPCClusterData* clusters = new GPUTPCClusterData[nNumberOfHits];
312  clSlices[iSector].reset(clusters);
313  int icl = 0;
314  for (unsigned int i = 0; i < vClusters.size(); i++) {
315  GenCluster& c = vClusters[i];
316  if (c.sector == iSector) {
317  clusters[icl].id = c.id;
318  clusters[icl].row = c.row; // We fill one hit per TPC row
319  clusters[icl].x = c.x;
320  clusters[icl].y = c.y;
321  clusters[icl].z = c.z;
322  clusters[icl].amp = 100; // Arbitrary amplitude
323  icl++;
324  AliHLTTPCClusterMCLabel clusterLabel;
325  for (int j = 0; j < 3; j++) {
326  clusterLabel.fClusterID[j].fMCID = -1;
327  clusterLabel.fClusterID[j].fWeight = 0;
328  }
329  clusterLabel.fClusterID[0].fMCID = c.mcID;
330  clusterLabel.fClusterID[0].fWeight = 1;
331  labels.push_back(clusterLabel);
332  }
333  }
334  mRec->mIOPtrs.clusterData[iSector] = clusters;
335  }
336 
337  // Create vector with cluster MC labels, clusters are counter from 0 to clusterId in the order they have been written above. No separation in slices.
338 
339  mRec->mIOPtrs.nMCLabelsTPC = labels.size();
340  mRec->mIOPtrs.mcLabelsTPC = labels.data();
341 
342  mRec->mIOPtrs.nMCInfosTPC = mcInfo.size();
343  mRec->mIOPtrs.mcInfosTPC = mcInfo.data();
344 
345  mRec->DumpData(filename);
346  labels.clear();
347  mcInfo.clear();
348  return (0);
349 }
