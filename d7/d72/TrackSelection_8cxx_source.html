25 bool TrackSelection::FulfillsITSHitRequirements(uint8_t itsClusterMap)
26 {
27  constexpr uint8_t bit = 1;
28  for (auto& itsRequirement : mRequiredITSHits) {
29  auto hits = std::count_if(itsRequirement.second.begin(), itsRequirement.second.end(), [&](auto&& requiredLayer) { return itsClusterMap & (bit << requiredLayer); });
30  if ((itsRequirement.first == -1) && (hits > 0)) {
31  return false; // no hits were required in specified layers
32  } else if (hits < itsRequirement.first) {
33  return false; // not enough hits found in specified layers
34  }
35  }
36  return true;
37 };
