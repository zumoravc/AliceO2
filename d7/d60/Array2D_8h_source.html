46 template <typename Grid>
47 class TilingLayout
48 {
49  public:
50  enum {
51  Height = Grid::Height,
52  Width = Grid::Width,
53  WidthInTiles = (TPC_NUM_OF_PADS + Width - 1) / Width,
54  };
55 
56  GPUdi() static tpccf::SizeT idx(const ChargePos& p)
57  {
58  const tpccf::SizeT tilePad = p.gpad / Width;
59  const tpccf::SizeT tileTime = p.timePadded / Height;
60 
61  const tpccf::SizeT inTilePad = p.gpad % Width;
62  const tpccf::SizeT inTileTime = p.timePadded % Height;
63 
64  return (tileTime * WidthInTiles + tilePad) * (Width * Height) + inTileTime * Width + inTilePad;
65  }
66 
67  GPUd() static size_t items()
68  {
69  return (TPC_NUM_OF_PADS + Width - 1) / Width * Width * (TPC_MAX_FRAGMENT_LEN_PADDED + Height - 1) / Height * Height;
70  }
71 };
