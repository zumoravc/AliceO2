23 size_t copyBuffer(const std::vector<uint64_t>& b64,
24  std::vector<std::byte>& b8,
25  uint64_t prefix = 0)
26 {
27  constexpr uint64_t m = 0xFF;
28  auto s8 = b8.size();
29  b8.reserve(s8 + b64.size() / 8);
30  for (auto& b : b64) {
31  uint64_t g = b | prefix;
32  for (uint64_t i = 0; i < 64; i += 8) {
33  uint64_t w = m << i;
34  b8.emplace_back(std::byte{static_cast<uint8_t>((g & w) >> i)});
35  }
36  }
37  return b8.size() - s8;
38 }

65 size_t copyBuffer(gsl::span<const std::byte> b8,
66  std::vector<uint64_t>& b64,
67  uint64_t prefix = 0)
68 {

74  uint64_t w = b8to64(b8, i);
