40 GPUd() void MEM_LG(GPUTPCGrid)::Create(float yMin, float yMax, float zMin, float zMax, int ny, int nz)
41 {
42  //* Create the grid
43  mYMin = yMin;
44  mZMin = zMin;
45 
46  float sy = CAMath::Max((yMax + 0.1f - yMin) / ny, GPUCA_MIN_BIN_SIZE);
47  float sz = CAMath::Max((zMax + 0.1f - zMin) / nz, GPUCA_MIN_BIN_SIZE);
48 
49  mStepYInv = 1.f / sy;
50  mStepZInv = 1.f / sz;
51 
52  mNy = ny;
53  mNz = nz;
54 
55  mN = mNy * mNz;
56 
57  mYMax = mYMin + mNy * sy;
58  mZMax = mZMin + mNz * sz;
59 }

117 GPUd() void MEM_LG(GPUTPCGrid)::GetBinArea(float Y, float Z, float dy, float dz, int& bin, int& ny, int& nz) const
118 {
119  Y -= mYMin;
120  int by = (int)((Y - dy) * mStepYInv);
121  ny = (int)((Y + dy) * mStepYInv) - by;
122  Z -= mZMin;
123  int bz = (int)((Z - dz) * mStepZInv);
124  nz = (int)((Z + dz) * mStepZInv) - bz;
125  if (by >= (int)mNy) {
126  by = mNy - 1;
127  }
128  if (by < 0) {
129  by = 0;
130  }
131  if (bz >= (int)mNz) {
132  bz = mNz - 1;
133  }

136  }

139  }

142  }
