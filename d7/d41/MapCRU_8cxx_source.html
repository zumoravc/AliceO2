29 MapCRU::MapCRU(std::string_view content)
30 {
31  mFeeLink2Solar.fill(std::numeric_limits<uint16_t>::max());
32  std::istringstream input(content.data());
33  std::string s;
34  while (std::getline(input, s)) {
35  if (s.empty()) {
36  continue;
37  }
38  std::istringstream line(s);
39  int f, l, link_id;
40  line >> link_id >> f >> l;
41  auto ix = indexFeeLink(f, l);
42  if (ix < 0) {
43  continue;
44  }
45  mFeeLink2Solar.at(ix) = link_id;
46  }
47 }

49 int MapCRU::indexFeeLink(int feeid, int linkid) const
50 {
51  if (feeid < 0 || feeid >= sMaxFeeId) {
52  return -1;
53  }
54  if (linkid < 0 || linkid >= sMaxLinkId) {
55  return -1;
56  }
57  return feeid * sMaxLinkId + linkid;
58 }

65 std::optional<uint16_t> MapCRU::operator()(const o2::mch::raw::FeeLinkId& feeLinkId) const
66 {
