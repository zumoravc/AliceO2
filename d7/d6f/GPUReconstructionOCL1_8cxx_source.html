85 bool GPUReconstructionOCL1Backend::CheckPlatform(unsigned int i)
86 {
87  char platform_version[64] = {}, platform_vendor[64] = {};
88  clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
89  clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
90  if (strcmp(platform_vendor, "Advanced Micro Devices, Inc.") == 0 && strstr(platform_version, "OpenCL 2.0 AMD-APP (") != nullptr) {
91  float ver = 0;
92  sscanf(platform_version, "OpenCL 2.0 AMD-APP (%f)", &ver);
93  if (ver < 2000.f) {
94  if (mDeviceProcessingSettings.debugLevel >= 2) {
95  GPUInfo("AMD APP OpenCL Platform found");
96  }
97  return true;
98  }
99  }
100  return false;
101 }
