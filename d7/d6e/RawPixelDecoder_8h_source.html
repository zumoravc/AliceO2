123 template <class Mapping>
124 template <class DigitContainer, class ROFContainer>
125 int RawPixelDecoder<Mapping>::fillDecodedDigits(DigitContainer& digits, ROFContainer& rofs)
126 {
127  if (mInteractionRecord.isDummy()) {
128  return 0; // nothing was decoded
129  }
130  mTimerFetchData.Start(false);
131  int ref = digits.size();
132  for (unsigned int iru = 0; iru < mRUDecodeVec.size(); iru++) {
133  for (int ic = 0; ic < mRUDecodeVec[iru].nChipsFired; ic++) {
134  const auto& chip = mRUDecodeVec[iru].chipsData[ic];
135  for (const auto& hit : mRUDecodeVec[iru].chipsData[ic].getData()) {
136  digits.emplace_back(chip.getChipID(), hit.getRow(), hit.getCol());
137  }
138  }
139  }
140  int nFilled = digits.size() - ref;
141  rofs.emplace_back(mInteractionRecord, mROFCounter, ref, nFilled);
142  mTimerFetchData.Stop();
143  return nFilled;
144 }
