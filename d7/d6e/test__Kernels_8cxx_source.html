26 BOOST_AUTO_TEST_CASE(TestHashByColumnKernel)
27 {
28  TableBuilder builder;
29  auto rowWriter = builder.persist<uint64_t, uint64_t>({"x", "y"});
30  rowWriter(0, 0, 0);
31  rowWriter(0, 0, 1);
32  rowWriter(0, 0, 2);
33  rowWriter(0, 0, 3);
34  rowWriter(0, 1, 4);
35  rowWriter(0, 1, 5);
36  rowWriter(0, 1, 6);
37  rowWriter(0, 1, 7);
38  rowWriter(0, 2, 8);
39  auto table = builder.finalize();
40 
41  arrow::compute::FunctionContext ctx;
42  HashByColumnKernel kernel{{"x"}};
43  std::shared_ptr<arrow::Array> out;
44  auto outDatum = arrow::compute::Datum(out);
45  BOOST_CHECK_EQUAL(kernel.Call(&ctx, arrow::compute::Datum(table), &outDatum).ok(), true);
46  auto indices = arrow::util::get<std::shared_ptr<arrow::ChunkedArray>>(outDatum.value);
47  BOOST_CHECK_EQUAL(indices->length(), table->num_rows());
48 
49  std::shared_ptr<arrow::Array> uniqueValues;
50  BOOST_CHECK_EQUAL(Unique(&ctx, arrow::compute::Datum(indices), &uniqueValues).ok(), true);
51  BOOST_REQUIRE(uniqueValues.get() != nullptr);
52  BOOST_CHECK_EQUAL(uniqueValues->length(), 3);
53 
54  arrow::compute::Datum outRanges;
55  SortedGroupByKernel<uint64_t, arrow::UInt64Array> groupBy{{"x", 3}};
56  BOOST_CHECK_EQUAL(groupBy.Call(&ctx, arrow::compute::Datum(table), &outRanges).ok(), true);
57  auto result = arrow::util::get<std::shared_ptr<arrow::Table>>(outRanges.value);
58  BOOST_REQUIRE(result.get() != nullptr);
59  BOOST_CHECK_EQUAL(result->num_rows(), 3);
60 
61  std::vector<Datum> splitted;
62  BOOST_CHECK_EQUAL(sliceByColumn(&ctx, "x", static_cast<uint64_t>(3), arrow::compute::Datum(table), &splitted).ok(), true);
63  BOOST_REQUIRE_EQUAL(splitted.size(), 3);
64  BOOST_CHECK_EQUAL(util::get<std::shared_ptr<Table>>(splitted[0].value)->num_rows(), 4);
65  BOOST_CHECK_EQUAL(util::get<std::shared_ptr<Table>>(splitted[1].value)->num_rows(), 4);
66  BOOST_CHECK_EQUAL(util::get<std::shared_ptr<Table>>(splitted[2].value)->num_rows(), 1);
67 }
