35 template <typename CallbackId, typename... Args>
36 class CallbackRegistry
37 {
38  public:
39  // FIXME:
40  // - add more checks
41  // - recursive check of the argument pack
42  // - required to be of type RegistryPair
43  // - callback type is specialization of std::function
44  // - extend to variable return type
45 
46  static constexpr std::size_t size = sizeof...(Args);
47 
48  // set callback for slot
49  template <typename U>
50  void set(CallbackId id, U&& cb)
51  {
52  set<size>(id, cb);
53  }
54 
55  // execute callback at specified slot with argument pack
56  template <typename... TArgs>
57  auto operator()(CallbackId id, TArgs&&... args)
58  {
59  exec<size>(id, std::forward<TArgs>(args)...);
60  }
61 
62  private:
63  // helper trait to check whether class has a constructor taking callback as argument
64  template <class T, typename CB>
65  struct has_matching_callback {
66  template <class U, typename V>
67  static int check(decltype(U(std::declval<V>()))*);
68 
69  template <typename U, typename V>
70  static char check(...);
71 
72  static const bool value = sizeof(check<T, CB>(nullptr)) == sizeof(int);
73  };
74 
75  // set the callback function of the specified id
76  // this iterates over defined slots and sets the matching callback
77  template <std::size_t pos, typename U>
78  typename std::enable_if<pos != 0>::type set(CallbackId id, U&& cb)
79  {
80  if (std::tuple_element<pos - 1, decltype(mStore)>::type::id::value != id) {
81  return set<pos - 1>(id, cb);
82  }
83  // note: there are two substitutions, the one for callback matching the slot type sets the
84  // callback, while the other substitution should never be called
85  setAt<pos - 1, typename std::tuple_element<pos - 1, decltype(mStore)>::type::type>(cb);
86  }
87  // termination of the recursive loop
88  template <std::size_t pos, typename U>
89  typename std::enable_if<pos == 0>::type set(CallbackId id, U&& cb)
90  {
91  }
92 
93  // set the callback at specified slot position
94  template <std::size_t pos, typename U, typename F>
95  typename std::enable_if<has_matching_callback<U, F>::value == true>::type setAt(F&& cb)
96  {
97  // create a new std::function object and init with the callback function
98  std::get<pos>(mStore).callback = (U)(cb);
99  }
100  // substitution for not matching callback
101  template <std::size_t pos, typename U, typename F>
102  typename std::enable_if<has_matching_callback<U, F>::value == false>::type setAt(F&& cb)
103  {
104  throw std::runtime_error("mismatch in function substitution at position " + std::to_string(pos));
105  }
106 
107  // exec callback of specified id
108  template <std::size_t pos, typename... TArgs>
109  typename std::enable_if<pos != 0>::type exec(CallbackId id, TArgs&&... args)
110  {
111  if (std::tuple_element<pos - 1, decltype(mStore)>::type::id::value != id) {
112  return exec<pos - 1>(id, std::forward<TArgs>(args)...);
113  }
114  // build a callable function type from the result type of the slot and the
115  // argument pack, this is used to selcet the matching substitution
116  using FunT = typename std::tuple_element<pos - 1, decltype(mStore)>::type::type;
117  using ResT = typename FunT::result_type;
118  using CheckT = std::function<ResT(TArgs...)>;
119  FunT& fct = std::get<pos - 1>(mStore).callback;
120  auto& storedTypeId = fct.target_type();
121  execAt<pos - 1, FunT, CheckT>(std::forward<TArgs>(args)...);
122  }
123  // termination of the recursive loop
124  template <std::size_t pos, typename... TArgs>
125  typename std::enable_if<pos == 0>::type exec(CallbackId id, TArgs&&... args)
126  {
127  }
128  // exec callback at specified slot
129  template <std::size_t pos, typename U, typename V, typename... TArgs>
130  typename std::enable_if<std::is_same<U, V>::value == true>::type execAt(TArgs&&... args)
131  {
132  if (std::get<pos>(mStore).callback) {
133  std::get<pos>(mStore).callback(std::forward<TArgs>(args)...);
134  }
135  }
136  // substitution for not matching argument pack
137  template <std::size_t pos, typename U, typename V, typename... TArgs>
138  typename std::enable_if<std::is_same<U, V>::value == false>::type execAt(TArgs&&... args)
139  {
140  }
141 
142  // store of RegistryPairs
143  std::tuple<Args...> mStore;
144 };
