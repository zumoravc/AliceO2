39 GPUd() bool GPUTPCTrackletConstructor::CheckCov(MEM_LG2(GPUTPCTrackParam) & GPUrestrict() tParam)
40 {
41  bool ok = 1;
42  const float* c = tParam.Cov();
43  for (int i = 0; i < 15; i++) {
44  ok = ok && CAMath::Finite(c[i]);
45  }
46  for (int i = 0; i < 5; i++) {
47  ok = ok && CAMath::Finite(tParam.Par()[i]);
48  }
49  ok = ok && (tParam.X() > 50);
50  if (c[0] <= 0 || c[2] <= 0 || c[5] <= 0 || c[9] <= 0 || c[14] <= 0) {
51  ok = 0;
52  }
53  return (ok);
54 }

57 GPUd() void GPUTPCTrackletConstructor::StoreTracklet(int /*nBlocks*/, int /*nThreads*/, int /*iBlock*/, int /*iThread*/, GPUsharedref() MEM_LOCAL(GPUSharedMemory) & GPUrestrict() s, GPUTPCThreadMemory& GPUrestrict() r, GPUconstantref() MEM_LG2(GPUTPCTracker) & GPUrestrict() tracker, MEM_LG3(GPUTPCTrackParam) & GPUrestrict() tParam, calink* rowHits)
58 {
59  // reconstruction of tracklets, tracklet store step
60  if (r.mNHits == 0 || (r.mNHits < GPUCA_TRACKLET_SELECTOR_MIN_HITS(tParam.QPt()) || !CheckCov(tParam) || CAMath::Abs(tParam.GetQPt()) > tracker.Param().rec.MaxTrackQPt)) {
61  return;
62  }
63 
64  /*GPUInfo("Tracklet %d: Hits %3d NDF %3d Chi %8.4f Sign %f Cov: %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f", r.mISH, r.mNHits, tParam.GetNDF(), tParam.GetChi2(), tParam.GetSignCosPhi(),
65  tParam.Cov()[0], tParam.Cov()[1], tParam.Cov()[2], tParam.Cov()[3], tParam.Cov()[4], tParam.Cov()[5], tParam.Cov()[6], tParam.Cov()[7], tParam.Cov()[8], tParam.Cov()[9],
66  tParam.Cov()[10], tParam.Cov()[11], tParam.Cov()[12], tParam.Cov()[13], tParam.Cov()[14]);*/
67 
68  unsigned int itrout = CAMath::AtomicAdd(tracker.NTracklets(), 1u);
69  const unsigned int nHits = r.mLastRow + 1 - r.mFirstRow;
70  unsigned int hitout = CAMath::AtomicAdd(tracker.NRowHits(), nHits);
71  if (itrout >= tracker.NMaxTracklets() || hitout + nHits > tracker.NMaxRowHits()) {
72  tracker.CommonMemory()->kernelError = itrout >= tracker.NMaxTracklets() ? GPUCA_ERROR_TRACKLET_OVERFLOW : GPUCA_ERROR_TRACKLET_HIT_OVERFLOW;
73  CAMath::AtomicExch(tracker.NTracklets(), 0u);
74  if (hitout + nHits > tracker.NMaxRowHits()) {
75  CAMath::AtomicExch(tracker.NRowHits(), tracker.NMaxRowHits());
76  }
77  return;
78  }
79 
80  GPUglobalref() MEM_GLOBAL(GPUTPCTracklet) & GPUrestrict() tracklet = tracker.Tracklets()[itrout];
81 
82  tracklet.SetNHits(r.mNHits);
83  CADEBUG(GPUInfo(" DONE %d hits", r.mNHits));
84 
85  tracklet.SetFirstRow(r.mFirstRow);
86  tracklet.SetLastRow(r.mLastRow);
87  tracklet.SetFirstHit(hitout);
88  tracklet.SetParam(tParam.GetParam());
89  int w = tracker.CalculateHitWeight(r.mNHits, tParam.GetChi2(), r.mISH);
90  tracklet.SetHitWeight(w);
91 #ifdef __HIPCC__ // Todo: fixme!
92  for (int iRow = r.mFirstRow - 1; ++iRow <= r.mLastRow; /*iRow++*/) {
93 #else
94  for (int iRow = r.mFirstRow; iRow <= r.mLastRow; iRow++) {
95 #endif
96  calink ih = rowHits[iRow];
97  tracker.TrackletRowHits()[hitout + (iRow - r.mFirstRow)] = ih;
98  if (ih != CALINK_INVAL) {
99  CA_MAKE_SHARED_REF(GPUTPCRow, row, tracker.Row(iRow), s.mRows[iRow]);
100  tracker.MaximizeHitWeight(row, ih, w);
101  }
102  }
103 }

106 GPUd() void GPUTPCTrackletConstructor::UpdateTracklet(int /*nBlocks*/, int /*nThreads*/, int /*iBlock*/, int /*iThread*/, GPUsharedref() T& GPUrestrict() s, GPUTPCThreadMemory& GPUrestrict() r, GPUconstantref() MEM_GLOBAL(GPUTPCTracker) & GPUrestrict() tracker, MEM_LG2(GPUTPCTrackParam) & GPUrestrict() tParam, int iRow, calink& rowHit)
107 {
108  // reconstruction of tracklets, tracklets update step
109  CA_MAKE_SHARED_REF(GPUTPCRow, row, tracker.Row(iRow), s.mRows[iRow]);
110 
111  float y0 = row.Grid().YMin();
112  float stepY = row.HstepY();
113  float z0 = row.Grid().ZMin() - tParam.ZOffset();
114  float stepZ = row.HstepZ();
115 
116  if (r.mStage == 0) { // fitting part
117  do {
118  if (iRow < r.mStartRow || r.mCurrIH == CALINK_INVAL) {
119  break;
120  }
121  if ((iRow - r.mStartRow) & 1) {
122  rowHit = CALINK_INVAL;
123  break; // SG!!! - jump over the row
124  }
125 
126  cahit2 hh = CA_TEXTURE_FETCH(cahit22, gAliTexRefu2, tracker.HitData(row), r.mCurrIH);
127 
128  int oldIH = r.mCurrIH;
129  r.mCurrIH = CA_TEXTURE_FETCH(calink, gAliTexRefs, tracker.HitLinkUpData(row), r.mCurrIH);
130 
131  float x = row.X();
132  float y = y0 + hh.x * stepY;
133  float z = z0 + hh.y * stepZ;
134 
135  if (iRow == r.mStartRow) {
136  tParam.SetX(x);
137  tParam.SetY(y);
138  r.mLastY = y;
139  if (tracker.Param().ContinuousTracking) {
140  tParam.SetZ(0.f);
141  r.mLastZ = 0.f;
142  tParam.SetZOffset(z);
143  } else {
144  tParam.SetZ(z);
145  r.mLastZ = z;
146  tParam.SetZOffset(0.f);
147  }
148  CADEBUG(
149  printf("Tracklet %5d: FIT INIT ROW %3d X %8.3f -", r.mISH, iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
150  } else {
151  float err2Y, err2Z;
152  float dx = x - tParam.X();
153  float dy, dz;
154  if (r.mNHits >= 10) {
155  dy = y - tParam.Y();
156  dz = z - tParam.Z();
157  } else {
158  dy = y - r.mLastY;
159  dz = z - r.mLastZ;
160  }
161  r.mLastY = y;
162  r.mLastZ = z;
163 
164  float ri = 1.f / CAMath::Sqrt(dx * dx + dy * dy);
165  if (iRow == r.mStartRow + 2) { // SG!!! important - thanks to Matthias
166  tParam.SetSinPhi(dy * ri);
167  tParam.SetSignCosPhi(dx);
168  tParam.SetDzDs(dz * ri);
169  // std::cout << "Init. errors... " << r.mISH << std::endl;
170  tracker.GetErrors2Seeding(iRow, tParam, err2Y, err2Z);
171  // std::cout << "Init. errors = " << err2Y << " " << err2Z << std::endl;
172  tParam.SetCov(0, err2Y);
173  tParam.SetCov(2, err2Z);
174  }
175  float sinPhi, cosPhi;
176  if (r.mNHits >= 10 && CAMath::Abs(tParam.SinPhi()) < GPUCA_MAX_SIN_PHI_LOW) {
177  sinPhi = tParam.SinPhi();
178  cosPhi = CAMath::Sqrt(1 - sinPhi * sinPhi);
179  } else {
180  sinPhi = dy * ri;
181  cosPhi = dx * ri;
182  }
183  CADEBUG(
184  printf("%14s: FIT TRACK ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
185  if (!tParam.TransportToX(x, sinPhi, cosPhi, tracker.Param().ConstBz, GPUCA_MAX_SIN_PHI)) {
186  rowHit = CALINK_INVAL;
187  break;
188  }
189  CADEBUG(
190  printf("%15s hits %3d: FIT PROP ROW %3d X %8.3f -", "", r.mNHits, iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
191  tracker.GetErrors2Seeding(iRow, tParam.GetZ(), sinPhi, tParam.GetDzDs(), err2Y, err2Z);
192 
193  if (r.mNHits >= 10) {
194  const float kFactor = tracker.Param().rec.HitPickUpFactor * tracker.Param().rec.HitPickUpFactor * 3.5f * 3.5f;
195  float sy2 = kFactor * (tParam.Err2Y() + err2Y);
196  float sz2 = kFactor * (tParam.Err2Z() + err2Z);
197  if (sy2 > 2.f) {
198  sy2 = 2.f;
199  }
200  if (sz2 > 2.f) {
201  sz2 = 2.f;
202  }
203  dy = y - tParam.Y();
204  dz = z - tParam.Z();
205  if (dy * dy > sy2 || dz * dz > sz2) {
206  if (++r.mNMissed >= GPUCA_TRACKLET_CONSTRUCTOR_MAX_ROW_GAP_SEED) {
207  r.mCurrIH = CALINK_INVAL;
208  }
209  rowHit = CALINK_INVAL;
210  break;
211  }
212  }
213 
214  if (!tParam.Filter(y, z, err2Y, err2Z, GPUCA_MAX_SIN_PHI_LOW)) {
215  rowHit = CALINK_INVAL;
216  break;
217  }
218  CADEBUG(
219  printf("%14s: FIT FILT ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
220  }
221  rowHit = oldIH;
222  r.mNHitsEndRow = ++r.mNHits;
223  r.mLastRow = iRow;
224  r.mEndRow = iRow;
225  r.mNMissed = 0;
226  } while (0);
227 
228  /*QQQQprintf("Extrapolate Row %d X %f Y %f Z %f SinPhi %f DzDs %f QPt %f", iRow, tParam.X(), tParam.Y(), tParam.Z(), tParam.SinPhi(), tParam.DzDs(), tParam.QPt());
229  for (int i = 0;i < 15;i++) printf(" C%d=%6.2f", i, tParam.GetCov(i));
230  printf("\n");*/
231 
232  if (r.mCurrIH == CALINK_INVAL) {
233  r.mStage = 1;
234  r.mLastY = tParam.Y(); // Store last spatial position here to start inward following from here
235  r.mLastZ = tParam.Z();
236  if (CAMath::Abs(tParam.SinPhi()) > GPUCA_MAX_SIN_PHI) {
237  r.mGo = 0;
238  }
239  }
240  } else { // forward/backward searching part
241  do {
242  if (r.mStage == 2 && iRow > r.mEndRow) {
243  break;
244  }
245  if (r.mNMissed > GPUCA_TRACKLET_CONSTRUCTOR_MAX_ROW_GAP) {
246  r.mGo = 0;
247  break;
248  }
249 
250  r.mNMissed++;
251 
252  float x = row.X();
253  float err2Y, err2Z;
254  CADEBUG(
255  printf("%14s: SEA TRACK ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
256  if (!tParam.TransportToX(x, tParam.SinPhi(), tParam.GetCosPhi(), tracker.Param().ConstBz, GPUCA_MAX_SIN_PHI_LOW)) {
257  r.mGo = 0;
258  rowHit = CALINK_INVAL;
259  break;
260  }
261  CADEBUG(
262  printf("%14s: SEA PROP ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
263  if (row.NHits() < 1) {
264  rowHit = CALINK_INVAL;
265  break;
266  }
267 
268 #ifndef GPUCA_TEXTURE_FETCH_CONSTRUCTOR
269  GPUglobalref() const cahit2* hits = tracker.HitData(row);
270  GPUglobalref() const calink* firsthit = tracker.FirstHitInBin(row);
271 #endif
272  float fY = tParam.GetY();
273  float fZ = tParam.GetZ();
274  calink best = CALINK_INVAL;
275 
276  { // search for the closest hit
277  tracker.GetErrors2Seeding(iRow, *((MEM_LG2(GPUTPCTrackParam)*)&tParam), err2Y, err2Z);
278  const float kFactor = tracker.Param().rec.HitPickUpFactor * tracker.Param().rec.HitPickUpFactor * 3.5f * 3.5f;
279  float sy2 = kFactor * (tParam.Err2Y() + err2Y);
280  float sz2 = kFactor * (tParam.Err2Z() + err2Z);
281  if (sy2 > 2.f) {
282  sy2 = 2.f;
283  }
284  if (sz2 > 2.f) {
285  sz2 = 2.f;
286  }
287 
288  int bin, ny, nz;
289  row.Grid().GetBinArea(fY, fZ + tParam.ZOffset(), 1.5f, 1.5f, bin, ny, nz);
290  float ds = 1e6f;
291 
292 #ifdef __HIPCC__ // Todo: fixme!
293  for (int k = -1; ++k <= nz; /*k++*/) {
294 #else
295  for (int k = 0; k <= nz; k++) {
296 #endif
297  int nBinsY = row.Grid().Ny();
298  int mybin = bin + k * nBinsY;
299  unsigned int hitFst = CA_TEXTURE_FETCH(calink, gAliTexRefu, firsthit, mybin);
300  unsigned int hitLst = CA_TEXTURE_FETCH(calink, gAliTexRefu, firsthit, mybin + ny + 1);
301 #ifdef __HIPCC__ // Todo: fixme!
302  for (unsigned int ih = hitFst - 1; ++ih < hitLst; /*ih++*/) {
303 #else
304  for (unsigned int ih = hitFst; ih < hitLst; ih++) {
305 #endif
306  cahit2 hh = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits, ih);
307  float y = y0 + hh.x * stepY;
308  float z = z0 + hh.y * stepZ;
309  float dy = y - fY;
310  float dz = z - fZ;
311  if (dy * dy < sy2 && dz * dz < sz2) {
312  float dds = GPUCA_Y_FACTOR * CAMath::Abs(dy) + CAMath::Abs(dz);
313  if (dds < ds) {
314  ds = dds;
315  best = ih;
316  }
317  }
318  }
319  }
320  } // end of search for the closest hit
321 
322  if (best == CALINK_INVAL) {
323  rowHit = CALINK_INVAL;
324  break;
325  }
326 
327  cahit2 hh = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits, best);
328  float y = y0 + hh.x * stepY;
329  float z = z0 + hh.y * stepZ;
330 
331  CADEBUG(GPUInfo("%14s: SEA Hit %5d, Res %f %f", "", best, tParam.Y() - y, tParam.Z() - z));
332 
333  calink oldHit = (r.mStage == 2 && iRow >= r.mStartRow) ? rowHit : CALINK_INVAL;
334  if (oldHit != best && !tParam.Filter(y, z, err2Y, err2Z, GPUCA_MAX_SIN_PHI_LOW, oldHit != CALINK_INVAL)) {
335  rowHit = CALINK_INVAL;
336  break;
337  }
338  rowHit = best;
339  r.mNHits++;
340  r.mNMissed = 0;
341  CADEBUG(
342  printf("%5s hits %3d: SEA FILT ROW %3d X %8.3f -", "", r.mNHits, iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
343  if (r.mStage == 1) {
344  r.mLastRow = iRow;
345  } else {
346  r.mFirstRow = iRow;
347  }
348  } while (0);
349  }
350 }

352 GPUd() void GPUTPCTrackletConstructor::DoTracklet(GPUconstantref() MEM_GLOBAL(GPUTPCTracker) & GPUrestrict() tracker, GPUsharedref() GPUTPCTrackletConstructor::MEM_LOCAL(GPUSharedMemory) & GPUrestrict() s, GPUTPCThreadMemory& GPUrestrict() r)
353 {
354  int iRow = 0, iRowEnd = GPUCA_ROW_COUNT;
355  MEM_PLAIN(GPUTPCTrackParam)
356  tParam;
357  calink rowHits[GPUCA_ROW_COUNT];
358  if (r.mGo) {
359  GPUTPCHitId id = tracker.TrackletStartHits()[r.mISH];
360 
361  r.mStartRow = r.mEndRow = r.mFirstRow = r.mLastRow = id.RowIndex();
362  r.mCurrIH = id.HitIndex();
363  r.mNMissed = 0;
364  iRow = r.mStartRow;
365  GPUTPCTrackletConstructor::InitTracklet(tParam);
366  }
367  r.mStage = 0;
368  r.mNHits = 0;
369  // if (tracker.Param().ISlice() != 35 && tracker.Param().ISlice() != 34 || r.mISH == CALINK_INVAL) {StoreTracklet( 0, 0, 0, 0, s, r, tracker, tParam );return;}
370 
371 #ifdef __HIPCC__ // Todo: fixme!
372  for (int k = -1; ++k < 2; /*k++*/) {
373 #else
374  for (int k = 0; k < 2; k++) {
375 #endif
376  for (; iRow != iRowEnd; iRow += r.mStage == 2 ? -1 : 1) {
377  if (!r.mGo) {
378  break;
379  }
380  UpdateTracklet(0, 0, 0, 0, s, r, tracker, tParam, iRow, rowHits[iRow]);
381  }
382  if (!r.mGo && r.mStage == 2) {
383  for (; iRow >= r.mStartRow; iRow--) {
384  rowHits[iRow] = CALINK_INVAL;
385  }
386  }
387  if (r.mStage == 2) {
388  StoreTracklet(0, 0, 0, 0, s, r, tracker, tParam, rowHits);
389  } else {
390  r.mNMissed = 0;
391  if ((r.mGo = (tParam.TransportToX(tracker.Row(r.mEndRow).X(), tracker.Param().ConstBz, GPUCA_MAX_SIN_PHI) && tParam.Filter(r.mLastY, r.mLastZ, tParam.Err2Y() * 0.5f, tParam.Err2Z() * 0.5f, GPUCA_MAX_SIN_PHI_LOW, true)))) {
392  CADEBUG(
393  printf("%14s: SEA BACK ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
394  float err2Y, err2Z;
395  tracker.GetErrors2Seeding(r.mEndRow, tParam, err2Y, err2Z);
396  if (tParam.GetCov(0) < err2Y) {
397  tParam.SetCov(0, err2Y);
398  }
399  if (tParam.GetCov(2) < err2Z) {
400  tParam.SetCov(2, err2Z);
401  }
402  CADEBUG(
403  printf("%14s: SEA ADJUS ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
404  }
405  r.mNHits -= r.mNHitsEndRow;
406  r.mStage = 2;
407  iRow = r.mEndRow;
408  iRowEnd = -1;
409  }
410  }

433 #ifdef GPUCA_GPUCODE
434  int mySlice = get_group_id(0) % GPUCA_NSLICES;

436 
437  if (get_local_id(0) == 0) {

439  }
