95  size_t addRow(RowDescType rowData, byte* seqData, size_t seqSize)
96  {
97  unsigned nFrames = mFrames.size();
98  unsigned currentRow = mRowData.size();
99  ParserType p;
100  p.parse(
101  seqData, seqSize,
102  [](const typename ParserT::HeaderType& h) { return (h); },
103  [](const typename ParserT::TrailerType& t) { return (t); },
104  [](const typename ParserT::TrailerType& t) {
105  return t.dataLength + ParserT::totalOffset;
106  },
107  [this, currentRow](typename ParserT::FrameInfo entry) {
108  // insert the header as column index in ascending order
109  auto position = mColumns.begin();
110  while (position != mColumns.end() && *position < *entry.header) {
111  position++;
112  }
113  if (position == mColumns.end() || *entry.header < *position) {
114  mColumns.emplace(position, *entry.header);
115  }
116 
117  // insert frame descriptor under key composed from header and row
118  auto result = mFrames.emplace(FrameIndex{*entry.header, currentRow},
119  FrameData{entry.payload, entry.length});
120  return result.second;
121  });
122  auto insertedFrames = mFrames.size() - nFrames;
123  if (insertedFrames > 0) {
124  mRowData.emplace_back(rowData);
125  }
126  return insertedFrames;
127  }

165  class iterator
166  { // TODO: derive from forward_iterator
167  public:
168  struct value_type : public FrameData {
169  RowDescType desc;
170  };

173  enum IteratorDirections {
174  kAlongRow,
175  kAlongColumn
176  };
177 
178  iterator() = delete;
179  ~iterator() = default;
180  iterator(IteratorDirections direction, TableView* parent, unsigned row = 0, unsigned column = 0)
181  : mDirection(direction), mRow(row), mColumn(column), mEnd(direction == kAlongRow ? parent->getNColumns() : parent->getNRows()), mParent(parent), mCache(), mIsCached(false)
182  {
183  while (!isValid() && !isEnd())
184  operator++();
185  }
186 
187  self_type& operator++()
188  {
189  mIsCached = false;
190  if (mDirection == kAlongRow) {
191  if (mColumn < mEnd)
192  mColumn++;
193  } else {
194  if (mRow < mEnd)
195  mRow++;
196  }
197  while (!isEnd() && !isValid())
198  operator++();
199  return *this;
200  }
201 
202  value_type operator*() const
203  {
204  if (!mIsCached) {
205  self_type* ncthis = const_cast<self_type*>(this);
206  mParent->get(mRow, mColumn, ncthis->mCache);
207  ncthis->mCache.desc = mParent->getRowData(mRow);
208  ncthis->mIsCached = true;
209  }
210  return mCache;
211  }
212 
213  bool operator==(const self_type& other) const
214  {
215  return mDirection == kAlongRow ? (mColumn == other.mColumn) : (mRow == other.mRow);
216  }
217 
218  bool operator!=(const self_type& other) const
219  {
220  return mDirection == kAlongRow ? (mColumn != other.mColumn) : (mRow != other.mRow);
221  }
222 
223  bool isEnd() const
224  {
225  return (mDirection == kAlongRow) ? (mColumn >= mEnd) : (mRow >= mEnd);
226  }
227 
228  bool isValid() const
229  {
230  if (!mIsCached) {
231  self_type* ncthis = const_cast<self_type*>(this);
232  ncthis->mIsCached = mParent->get(mRow, mColumn, ncthis->mCache);
233  ncthis->mCache.desc = mParent->getRowData(mRow);
234  }
235  return mIsCached;
236  }

240  unsigned mRow;

313  private:
315  bool get(unsigned row, unsigned column, FrameData& data)
316  {
317  if (this->mColumns.size() == 0)
318  return false;
319  auto element = this->mFrames.find(FrameIndex{this->mColumns[column], row});
320  if (element != this->mFrames.end()) {
321  data = element->second;
322  return true;
323  }
324  return false;
325  }
