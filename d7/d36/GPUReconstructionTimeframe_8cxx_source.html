36 GPUReconstructionTimeframe::GPUReconstructionTimeframe(GPUChainTracking* chain, int (*read)(int), int nEvents) : mChain(chain), mReadEvent(read), mNEventsInDirectory(nEvents), mDisUniReal(0., 1.), mRndGen1(configStandalone.seed), mRndGen2(mDisUniInt(mRndGen1))
37 {
38  mMaxBunchesFull = TIME_ORBIT / config.bunchSpacing;
39  mMaxBunches = (TIME_ORBIT - config.abortGapTime) / config.bunchSpacing;
40 
41  if (config.overlayRaw && chain->GetTPCTransform() == nullptr) {
42  GPUInfo("Overlay Raw Events requires TPC Fast Transform");
43  throw std::exception();
44  }
45  if (config.bunchSim) {
46  if (config.bunchCount * config.bunchTrainCount > mMaxBunches) {
47  GPUInfo("Invalid timeframe settings: too many colliding bunches requested!");
48  throw std::exception();
49  }
50  mTrainDist = mMaxBunches / config.bunchTrainCount;
51  mCollisionProbability = (float)config.interactionRate * (float)(mMaxBunchesFull * config.bunchSpacing / 1e9f) / (float)(config.bunchCount * config.bunchTrainCount);
52  GPUInfo("Timeframe settings: %d trains of %d bunches, bunch spacing: %d, train spacing: %dx%d, filled bunches %d / %d (%d), collision probability %f, mixing %d events", config.bunchTrainCount, config.bunchCount, config.bunchSpacing, mTrainDist, config.bunchSpacing,
53  config.bunchCount * config.bunchTrainCount, mMaxBunches, mMaxBunchesFull, mCollisionProbability, mNEventsInDirectory);
54  }
55 
56  mEventStride = configStandalone.seed;
57  mSimBunchNoRepeatEvent = configStandalone.StartEvent;
58  mEventUsed.resize(mNEventsInDirectory);
59  if (config.noEventRepeat == 2) {
60  memset(mEventUsed.data(), 0, mNEventsInDirectory * sizeof(mEventUsed[0]));
61  }

64 int GPUReconstructionTimeframe::ReadEventShifted(int iEvent, float shiftZ, float minZ, float maxZ, bool silent)
65 {
66  mReadEvent(iEvent);
67  if (config.overlayRaw) {
68  float shiftTTotal = (((double)config.timeFrameLen - DRIFT_TIME) * ((double)TPCZ / (double)DRIFT_TIME) - shiftZ) / mChain->GetTPCTransform()->getVDrift();
69  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
70  for (unsigned int j = 0; j < mChain->mIOPtrs.nRawClusters[iSlice]; j++) {
71  auto& tmp = mChain->mIOMem.rawClusters[iSlice][j];
72  tmp.fTime += shiftTTotal;
73  }
74  }
75  }
76  if (shiftZ != 0.) {
77  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
78  for (unsigned int j = 0; j < mChain->mIOPtrs.nClusterData[iSlice]; j++) {
79  auto& tmp = mChain->mIOMem.clusterData[iSlice][j];
80  tmp.z += iSlice < NSLICES / 2 ? shiftZ : -shiftZ;
81  }
82  }
83  for (unsigned int i = 0; i < mChain->mIOPtrs.nMCInfosTPC; i++) {
84  auto& tmp = mChain->mIOMem.mcInfosTPC[i];
85  tmp.z += i < NSLICES / 2 ? shiftZ : -shiftZ;
86  }
87  }
88 
89  // Remove clusters outside boundaries
90  unsigned int nClusters = 0;
91  unsigned int removed = 0;
92  if (minZ > -1e6 || maxZ > -1e6) {
93  unsigned int currentClusterTotal = 0;
94  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
95  unsigned int currentClusterSlice = 0;
96  bool doRaw = config.overlayRaw && mChain->mIOPtrs.nClusterData[iSlice] == mChain->mIOPtrs.nRawClusters[iSlice];
97  for (unsigned int i = 0; i < mChain->mIOPtrs.nClusterData[iSlice]; i++) {
98  float sign = iSlice < NSLICES / 2 ? 1 : -1;
99  if (sign * mChain->mIOMem.clusterData[iSlice][i].z >= minZ && sign * mChain->mIOMem.clusterData[iSlice][i].z <= maxZ) {
100  if (currentClusterSlice != i) {
101  mChain->mIOMem.clusterData[iSlice][currentClusterSlice] = mChain->mIOMem.clusterData[iSlice][i];
102  if (doRaw) {
103  mChain->mIOMem.rawClusters[iSlice][currentClusterSlice] = mChain->mIOMem.rawClusters[iSlice][i];
104  }
105  }
106  if (mChain->mIOPtrs.nMCLabelsTPC > currentClusterTotal && nClusters != currentClusterTotal) {
107  mChain->mIOMem.mcLabelsTPC[nClusters] = mChain->mIOMem.mcLabelsTPC[currentClusterTotal];
108  }
109  // GPUInfo("Keeping Cluster ID %d (ID in slice %d) Z=%f (sector %d) --> %d (slice %d)", currentClusterTotal, i, mChain->mIOMem.clusterData[iSlice][i].fZ, iSlice, nClusters, currentClusterSlice);
110  currentClusterSlice++;
111  nClusters++;
112  } else {
113  // GPUInfo("Removing Cluster ID %d (ID in slice %d) Z=%f (sector %d)", currentClusterTotal, i, mChain->mIOMem.clusterData[iSlice][i].fZ, iSlice);
114  removed++;
115  }
116  currentClusterTotal++;
117  }
118  mChain->mIOPtrs.nClusterData[iSlice] = currentClusterSlice;
119  if (doRaw) {
120  mChain->mIOPtrs.nRawClusters[iSlice] = currentClusterSlice;
121  }
122  }
123  if (mChain->mIOPtrs.nMCLabelsTPC) {
124  mChain->mIOPtrs.nMCLabelsTPC = nClusters;
125  }
126  } else {
127  for (unsigned int i = 0; i < NSLICES; i++) {
128  nClusters += mChain->mIOPtrs.nClusterData[i];
129  }
130  }
131 
132  if (!silent) {
133  GPUInfo("Read %u Clusters with %d MC labels and %d MC tracks", nClusters, (int)mChain->mIOPtrs.nMCLabelsTPC, (int)mChain->mIOPtrs.nMCInfosTPC);
134  if (minZ > -1e6 || maxZ > 1e6) {
135  GPUInfo("\tRemoved %u / %u clusters", removed, nClusters + removed);
136  }
137  }
138 
139  mShiftedEvents.emplace_back(mChain->mIOPtrs, std::move(mChain->mIOMem), mChain->mIOPtrs.clustersNative ? *mChain->mIOPtrs.clustersNative : o2::tpc::ClusterNativeAccess());
140  return nClusters;
141 }

143 void GPUReconstructionTimeframe::MergeShiftedEvents()
144 {
145  mChain->ClearIOPointers();
146  for (unsigned int i = 0; i < mShiftedEvents.size(); i++) {
147  auto& ptr = std::get<0>(mShiftedEvents[i]);
148  for (unsigned int j = 0; j < NSLICES; j++) {
149  mChain->mIOPtrs.nClusterData[j] += ptr.nClusterData[j];
150  if (config.overlayRaw) {
151  mChain->mIOPtrs.nRawClusters[j] += ptr.nRawClusters[j];
152  }
153  }
154  mChain->mIOPtrs.nMCLabelsTPC += ptr.nMCLabelsTPC;
155  mChain->mIOPtrs.nMCInfosTPC += ptr.nMCInfosTPC;
156  SetDisplayInformation(i);
157  }
158  unsigned int nClustersTotal = 0;
159  unsigned int nClustersTotalRaw = 0;
160  unsigned int nClustersSliceOffset[NSLICES] = {0};
161  for (unsigned int i = 0; i < NSLICES; i++) {
162  nClustersSliceOffset[i] = nClustersTotal;
163  nClustersTotal += mChain->mIOPtrs.nClusterData[i];
164  nClustersTotalRaw += mChain->mIOPtrs.nRawClusters[i];
165  }
166  if (nClustersTotalRaw && nClustersTotalRaw != nClustersTotal) {
167  GPUError("Inconsitency between raw clusters and cluster data");
168  throw std::exception();
169  }
170  if (mChain->mIOPtrs.nMCLabelsTPC && nClustersTotal != mChain->mIOPtrs.nMCLabelsTPC) {
171  GPUError("Inconsitency between TPC clusters and MC labels");
172  throw std::exception();
173  }
174  mChain->AllocateIOMemory();
175  mChain->mIOPtrs.clustersNative = nullptr;
176 
177  unsigned int nTrackOffset = 0;
178  unsigned int nClustersEventOffset[NSLICES] = {0};
179  for (unsigned int i = 0; i < mShiftedEvents.size(); i++) {
180  auto& ptr = std::get<0>(mShiftedEvents[i]);
181  unsigned int inEventOffset = 0;
182  for (unsigned int j = 0; j < NSLICES; j++) {
183  memcpy((void*)&mChain->mIOMem.clusterData[j][nClustersEventOffset[j]], (void*)ptr.clusterData[j], ptr.nClusterData[j] * sizeof(ptr.clusterData[j][0]));
184  if (nClustersTotalRaw) {
185  memcpy((void*)&mChain->mIOMem.rawClusters[j][nClustersEventOffset[j]], (void*)ptr.rawClusters[j], ptr.nRawClusters[j] * sizeof(ptr.rawClusters[j][0]));
186  }
187  if (mChain->mIOPtrs.nMCLabelsTPC) {
188  memcpy((void*)&mChain->mIOMem.mcLabelsTPC[nClustersSliceOffset[j] + nClustersEventOffset[j]], (void*)&ptr.mcLabelsTPC[inEventOffset], ptr.nClusterData[j] * sizeof(ptr.mcLabelsTPC[0]));
189  }
190  for (unsigned int k = 0; k < ptr.nClusterData[j]; k++) {
191  mChain->mIOMem.clusterData[j][nClustersEventOffset[j] + k].id = nClustersSliceOffset[j] + nClustersEventOffset[j] + k;
192  if (mChain->mIOPtrs.nMCLabelsTPC) {
193  for (int l = 0; l < 3; l++) {
194  auto& label = mChain->mIOMem.mcLabelsTPC[nClustersSliceOffset[j] + nClustersEventOffset[j] + k].fClusterID[l];
195  if (label.fMCID >= 0) {
196  label.fMCID += nTrackOffset;
197  }
198  }
199  }
200  }
201 
202  nClustersEventOffset[j] += ptr.nClusterData[j];
203  inEventOffset += ptr.nClusterData[j];
204  }
205 
206  memcpy((void*)&mChain->mIOMem.mcInfosTPC[nTrackOffset], (void*)ptr.mcInfosTPC, ptr.nMCInfosTPC * sizeof(ptr.mcInfosTPC[0]));
207  nTrackOffset += ptr.nMCInfosTPC;
208  }
209 
210  GPUInfo("Merged %d events, %u clusters total", (int)mShiftedEvents.size(), nClustersTotal);
211 
212  mShiftedEvents.clear();
213 }

215 int GPUReconstructionTimeframe::LoadCreateTimeFrame(int iEvent)
216 {
217  if (config.nTotalInTFEvents && mNTotalCollisions >= config.nTotalInTFEvents) {
218  return (2);
219  }
220 
221  long long int nBunch = -DRIFT_TIME / config.bunchSpacing;
222  long long int lastBunch = config.timeFrameLen / config.bunchSpacing;
223  long long int lastTFBunch = lastBunch - DRIFT_TIME / config.bunchSpacing;
224  int nCollisions = 0, nBorderCollisions = 0, nTrainCollissions = 0, nMultipleCollisions = 0, nTrainMultipleCollisions = 0;
225  int nTrain = 0;
226  int mcMin = -1, mcMax = -1;
227  unsigned int nTotalClusters = 0;
228  while (nBunch < lastBunch) {
229  for (int iTrain = 0; iTrain < config.bunchTrainCount && nBunch < lastBunch; iTrain++) {
230  int nCollisionsInTrain = 0;
231  for (int iBunch = 0; iBunch < config.bunchCount && nBunch < lastBunch; iBunch++) {
232  const bool inTF = nBunch >= 0 && nBunch < lastTFBunch && (config.nTotalInTFEvents == 0 || nCollisions < mNTotalCollisions + config.nTotalInTFEvents);
233  if (mcMin == -1 && inTF) {
234  mcMin = mChain->mIOPtrs.nMCInfosTPC;
235  }
236  if (mcMax == -1 && nBunch >= 0 && !inTF) {
237  mcMax = mChain->mIOPtrs.nMCInfosTPC;
238  }
239  int nInBunchPileUp = 0;
240  double randVal = mDisUniReal(inTF ? mRndGen2 : mRndGen1);
241  double p = exp(-mCollisionProbability);
242  double p2 = p;
243  while (randVal > p) {
244  if (config.noBorder && (nBunch < 0 || nBunch >= lastTFBunch)) {
245  break;
246  }
247  if (nCollisionsInTrain >= mNEventsInDirectory) {
248  GPUError("Error: insuffient events for mixing!");
249  return (1);
250  }
251  if (nCollisionsInTrain == 0 && config.noEventRepeat == 0) {
252  memset(mEventUsed.data(), 0, mNEventsInDirectory * sizeof(mEventUsed[0]));
253  }
254  if (inTF) {
255  nCollisions++;
256  } else {
257  nBorderCollisions++;
258  }
259  int useEvent;
260  if (config.noEventRepeat == 1) {
261  useEvent = mSimBunchNoRepeatEvent;
262  } else {
263  while (mEventUsed[useEvent = (inTF && config.eventStride ? (mEventStride += config.eventStride) : mDisUniInt(inTF ? mRndGen2 : mRndGen1)) % mNEventsInDirectory]) {
264  ;
265  }
266  }
267  if (config.noEventRepeat) {
268  mSimBunchNoRepeatEvent++;
269  }
270  mEventUsed[useEvent] = 1;
271  double shift = (double)nBunch * (double)config.bunchSpacing * (double)TPCZ / (double)DRIFT_TIME;
272  int nClusters = ReadEventShifted(useEvent, shift, 0, (double)config.timeFrameLen * (double)TPCZ / (double)DRIFT_TIME, true);
273  if (nClusters < 0) {
274  GPUError("Unexpected error");
275  return (1);
276  }
277  nTotalClusters += nClusters;
278  printf("Placing event %4d+%d (ID %4d) at z %7.3f (time %'dns) %s(collisions %4d, bunch %6lld, train %3d) (%'10d clusters, %'10d MC labels, %'10d track MC info)\n", nCollisions, nBorderCollisions, useEvent, shift, (int)(nBunch * config.bunchSpacing), inTF ? " inside" : "outside",
279  nCollisions, nBunch, nTrain, nClusters, mChain->mIOPtrs.nMCLabelsTPC, mChain->mIOPtrs.nMCInfosTPC);
280  nInBunchPileUp++;
281  nCollisionsInTrain++;
282  p2 *= mCollisionProbability / nInBunchPileUp;
283  p += p2;
284  if (config.noEventRepeat && mSimBunchNoRepeatEvent >= mNEventsInDirectory) {
285  nBunch = lastBunch;
286  }
287  }
288  if (nInBunchPileUp > 1) {
289  nMultipleCollisions++;
290  }
291  nBunch++;
292  }
293  nBunch += mTrainDist - config.bunchCount;
294  if (nCollisionsInTrain) {
295  nTrainCollissions++;
296  }
297  if (nCollisionsInTrain > 1) {
298  nTrainMultipleCollisions++;
299  }
300  nTrain++;
301  }
302  nBunch += mMaxBunchesFull - mTrainDist * config.bunchTrainCount;
303  }
304  mNTotalCollisions += nCollisions;
305  GPUInfo("Timeframe statistics: collisions: %d+%d in %d trains (inside / outside), average rate %f (pile up: in bunch %d, in train %d)", nCollisions, nBorderCollisions, nTrainCollissions, (float)nCollisions / (float)(config.timeFrameLen - DRIFT_TIME) * 1e9, nMultipleCollisions,
306  nTrainMultipleCollisions);
307  MergeShiftedEvents();
308  GPUInfo("\tTotal clusters: %u, MC Labels %d, MC Infos %d", nTotalClusters, (int)mChain->mIOPtrs.nMCLabelsTPC, (int)mChain->mIOPtrs.nMCInfosTPC);
309 
310  if (!config.noBorder && mChain->GetQA()) {
311  mChain->GetQA()->SetMCTrackRange(mcMin, mcMax);
312  }
313  return (0);
314 }

316 int GPUReconstructionTimeframe::LoadMergedEvents(int iEvent)
317 {
318  for (int iEventInTimeframe = 0; iEventInTimeframe < config.nMerge; iEventInTimeframe++) {
319  float shift;
320  if (config.shiftFirstEvent || iEventInTimeframe) {
321  if (config.randomizeDistance) {
322  shift = mDisUniReal(mRndGen2);
323  if (config.shiftFirstEvent) {
324  shift = (iEventInTimeframe + shift) * config.averageDistance;
325  } else {
326  if (iEventInTimeframe == 0) {
327  shift = 0;
328  } else {
329  shift = (iEventInTimeframe - 0.5 + shift) * config.averageDistance;
330  }
331  }
332  } else {
333  if (config.shiftFirstEvent) {
334  shift = config.averageDistance * (iEventInTimeframe + 0.5);
335  } else {
336  shift = config.averageDistance * (iEventInTimeframe);
337  }
338  }
339  } else {
340  shift = 0.;
341  }
342 
343  if (ReadEventShifted(iEvent * config.nMerge + iEventInTimeframe, shift) < 0) {
344  return (1);
345  }
346  }
347  MergeShiftedEvents();
348  return (0);
349 }
