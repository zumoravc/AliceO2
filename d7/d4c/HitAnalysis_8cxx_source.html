67 InitStatus HitAnalysis::Init()
68 {
69  // Get the FairRootManager
70  FairRootManager* mgr = FairRootManager::Instance();
71  if (!mgr) {
72  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ...";
73  return kERROR;
74  }
75 
76  mHits = mgr->InitObjectAs<const std::vector<o2::itsmft::Hit>*>("ITSHit");
77  if (!mHits) {
78  LOG(ERROR) << "ITS points not registered in the FairRootManager. Exiting ...";
79  return kERROR;
80  }
81 
82  // Create geometry, initialize chip array
83  GeometryTGeo* geom = GeometryTGeo::Instance();
84  if (!geom->isBuilt())
85  geom->Build(true);
86  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::L2G)); // make sure T2L matrices are loaded
87 
88  mGeometry = geom;
89 
90  // Create histograms
91  // Ranges to be adjusted
92  Double_t maxLengthX(Segmentation::SensorSizeRows), maxLengthY(Segmentation::SensorLayerThickness),
93  maxLengthZ(Segmentation::SensorSizeCols);
94 
95  mLineSegment = new TH1D("lineSegment", "Length of the line segment within the chip", 500, 0.0, 0.01);
96  mLocalX0 = new TH1D("localX0", "X position in local (chip) coordinates at the start of a hit", 5000, -2 * maxLengthX,
97  2 * maxLengthX);
98  mLocalX1 = new TH1D("localX1", "X position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
99  mLocalY0 = new TH1D("localY0", "Y position in local (chip) coordinates at the start of a hit", 5000, -2 * maxLengthY,
100  2. * maxLengthY);
101  mLocalY1 = new TH1D("localY1", "Y position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
102  mLocalZ0 = new TH1D("localZ0", "Z position in local (chip) coordinates at the start of a hit", 5000, 2 * maxLengthZ,
103  -2 * maxLengthZ);
104  mLocalZ1 = new TH1D("localZ1", "Z position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
105  mHitCounter = new TH1F("hitcounter", "Simple hit counter", 1, 0.5, 1.5);
106 
107  mIsInitialized = kTRUE;
108  return kSUCCESS;
109 }

127 void HitAnalysis::ProcessHits()
128 {
129  for (auto& hit : *mHits) {
130  auto loc = mGeometry->getMatrixL2G(hit.GetDetectorID()) ^ (hit.GetPos()); // global->local end position
131  auto locS = mGeometry->getMatrixL2G(hit.GetDetectorID()) ^ (hit.GetPosStart()); // global->local start position
132  auto glo = mGeometry->getMatrixL2G(hit.GetDetectorID())(loc);
133  auto gloS = mGeometry->getMatrixL2G(hit.GetDetectorID())(locS);
134  mLocalX0->Fill(locS.X());
135  mLocalY0->Fill(locS.Y());
136  mLocalZ0->Fill(locS.Z());
137  loc.SetXYZ(loc.X() - locS.X(), loc.Y() - locS.Y(), loc.Z() - locS.Z());
138  mLocalX1->Fill(loc.X());
139  mLocalY1->Fill(loc.Y());
140  mLocalZ1->Fill(loc.Z());
141  }
142 }
