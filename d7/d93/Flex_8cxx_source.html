55 TGeoVolumeAssembly* Flex::makeFlex(Int_t nbsensors, Double_t length)
56 {
57 
58  // Informations from the technical report mft_flex_proto_5chip_v08_laz50p.docx on MFT twiki and private communications
59 
60  // For the naming
61  Geometry* mftGeom = Geometry::instance();
62  Int_t idHalfMFT = mftGeom->getHalfID(mLadderSeg->GetUniqueID());
63  Int_t idHalfDisk = mftGeom->getDiskID(mLadderSeg->GetUniqueID());
64  Int_t idLadder = mftGeom->getLadderID(mLadderSeg->GetUniqueID());
65 
66  // First a global pointer for the flex
67  TGeoMedium* kMedAir = gGeoManager->GetMedium("MFT_Air$");
68  auto* flex = new TGeoVolumeAssembly(Form("flex_%d_%d_%d", idHalfMFT, idHalfDisk, idLadder));
69 
70  // Defining one single layer for the strips and the AVDD and DVDD
71  TGeoVolume* lines = makeLines(nbsensors, length - Geometry::sClearance, Geometry::sFlexHeight - Geometry::sClearance,
72  Geometry::sAluThickness);
73 
74  // AGND and DGND layers
75  TGeoVolume* agnd_dgnd = makeAGNDandDGND(length - Geometry::sClearance, Geometry::sFlexHeight - Geometry::sClearance,
76  Geometry::sAluThickness);
77 
78  // The others layers
79  TGeoVolume* kaptonlayer = makeKapton(length, Geometry::sFlexHeight, Geometry::sKaptonThickness);
80  TGeoVolume* varnishlayerIn = makeVarnish(length, Geometry::sFlexHeight, Geometry::sVarnishThickness, 0);
81  TGeoVolume* varnishlayerOut = makeVarnish(length, Geometry::sFlexHeight, Geometry::sVarnishThickness, 1);
82 
83  // Final flex building
84  Double_t zvarnishIn = Geometry::sKaptonThickness / 2 + Geometry::sAluThickness + Geometry::sVarnishThickness / 2 -
85  Geometry::sGlueThickness;
86  Double_t zgnd = Geometry::sKaptonThickness / 2 + Geometry::sAluThickness / 2 - Geometry::sGlueThickness;
87  Double_t zkaptonlayer = -Geometry::sGlueThickness;
88  Double_t zlines = -Geometry::sKaptonThickness / 2 - Geometry::sAluThickness / 2 - Geometry::sGlueThickness;
89  Double_t zvarnishOut = -Geometry::sKaptonThickness / 2 - Geometry::sAluThickness - Geometry::sVarnishThickness / 2 -
90  Geometry::sGlueThickness;
91 
92  //-----------------------------------------------------------------------------------------
93  //-------------------------- Adding all layers of the FPC ----------------------------------
94  //-----------------------------------------------------------------------------------------
95 
96  flex->AddNode(varnishlayerIn, 1, new TGeoTranslation(0., 0., zvarnishIn)); // inside, in front of the cold plate
97  flex->AddNode(agnd_dgnd, 1, new TGeoTranslation(0., 0., zgnd));
98  flex->AddNode(kaptonlayer, 1, new TGeoTranslation(0., 0., zkaptonlayer));
99  flex->AddNode(lines, 1, new TGeoTranslation(0., 0., zlines));
100  flex->AddNode(varnishlayerOut, 1, new TGeoTranslation(0., 0., zvarnishOut)); // outside
101 
102  makeElectricComponents(flex, nbsensors, length, zvarnishOut);
103 
104  return flex;
105 }

108 void Flex::makeElectricComponents(TGeoVolumeAssembly* flex, Int_t nbsensors, Double_t length, Double_t zvarnish)
109 {
110 
111  // Making and adding all the electric components
112  TGeoVolumeAssembly* electric[200];
113 
114  // 2 components on the connector side
115  Int_t total;
116 
117  auto* rotation = new TGeoRotation("rotation", 90., 0., 0.);
118  auto* rotationpi = new TGeoRotation("rotationpi", 180., 0., 0.);
119  auto* transformation0 =
120  new TGeoCombiTrans(length / 2 - 0.1, Geometry::sFlexHeight / 2 - 0.2,
121  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sCapacitorDz / 2, rotation);
122  auto* transformation1 =
123  new TGeoCombiTrans(length / 2 - 0.1, Geometry::sFlexHeight / 2 - 0.6,
124  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sCapacitorDz / 2, rotation);
125 
126  for (Int_t id = 0; id < 2; id++) {
127  electric[id] = makeElectricComponent(Geometry::sCapacitorDy, Geometry::sCapacitorDx, Geometry::sCapacitorDz, id);
128  }
129  flex->AddNode(electric[0], 1, transformation0);
130  flex->AddNode(electric[1], 2, transformation1);
131  total = 2;
132 
133  // 2 lines of electric components along the FPC in the middle (4 per sensor)
134  for (Int_t id = 0; id < 4 * nbsensors; id++) {
135  electric[id + total] =
136  makeElectricComponent(Geometry::sCapacitorDy, Geometry::sCapacitorDx, Geometry::sCapacitorDz, id + total);
137  }
138  for (Int_t id = 0; id < 2 * nbsensors; id++) {
139  flex->AddNode(electric[id + total], id + 1000,
140  new TGeoTranslation(-length / 2 + (id + 0.5) * SegmentationAlpide::SensorSizeCols / 2,
141  Geometry::sFlexHeight / 2 - 0.35,
142  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sCapacitorDz / 2));
143  flex->AddNode(electric[id + total + 2 * nbsensors], id + 2000,
144  new TGeoTranslation(-length / 2 + (id + 0.5) * SegmentationAlpide::SensorSizeCols / 2, 0.,
145  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sCapacitorDz / 2));
146  }
147  total = total + 4 * nbsensors;
148 
149  // ------- 3 components on the FPC side --------
150  for (Int_t id = 0; id < 3; id++) {
151  electric[id + total] =
152  makeElectricComponent(Geometry::sCapacitorDy, Geometry::sCapacitorDx, Geometry::sCapacitorDz, id + total);
153  }
154  for (Int_t id = 0; id < 3; id++) {
155  flex->AddNode(electric[id + total], id + 3000,
156  new TGeoTranslation(-length / 2 + SegmentationAlpide::SensorSizeCols + (id + 1) * 0.3 - 0.6,
157  -Geometry::sFlexHeight / 2 + 0.2,
158  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sCapacitorDz / 2));
159  }
160  total = total + 3;
161 
162  /*
163  // The connector of the FPC
164  for(Int_t id=0; id < 74; id++)electric[id+total] = makeElectricComponent(Geometry::sConnectorLength,
165  Geometry::sConnectorWidth,
166  Geometry::sConnectorThickness, id+total);
167  for(Int_t id=0; id < 37; id++){
168  flex->AddNode(electric[id+total], id+100, new TGeoTranslation(length/2+0.15-Geometry::sConnectorOffset,
169  id*0.04-Geometry::sFlexHeight/2 + 0.1,
170  zvarnish-Geometry::sVarnishThickness/2-Geometry::sCapacitorDz/2));
171  flex->AddNode(electric[id+total+37], id+200, new TGeoTranslation(length/2-0.15-Geometry::sConnectorOffset,
172  id*0.04-Geometry::sFlexHeight/2 + 0.1,
173  zvarnish - Geometry::sVarnishThickness/2 -
174  Geometry::sCapacitorDz/2));
175  }
176  total=total+74;
177  */
178 
179  //-------------------------- New Connector ----------------------
180  TGeoMedium* kMedAlu = gGeoManager->GetMedium("MFT_Alu$");
181  TGeoMedium* kMedPeek = gGeoManager->GetMedium("MFT_PEEK$");
182 
183  auto* connect = new TGeoBBox("connect", Geometry::sConnectorLength / 2, Geometry::sConnectorWidth / 2,
184  Geometry::sConnectorHeight / 2);
185  auto* remov =
186  new TGeoBBox("remov", Geometry::sConnectorLength / 2, Geometry::sConnectorWidth / 2 + Geometry::sEpsilon,
187  Geometry::sConnectorHeight / 2 + Geometry::sEpsilon);
188 
189  auto* t1 = new TGeoTranslation("t1", Geometry::sConnectorThickness, 0., -0.01);
190  auto* connecto = new TGeoSubtraction(connect, remov, nullptr, t1);
191  auto* connector = new TGeoCompositeShape("connector", connecto);
192  auto* connectord = new TGeoVolume("connectord", connector, kMedAlu);
193  connectord->SetVisibility(kTRUE);
194  connectord->SetLineColor(kRed);
195  connectord->SetLineWidth(1);
196  connectord->SetFillColor(connectord->GetLineColor());
197  connectord->SetFillStyle(4000); // 0% transparent
198 
199  Double_t interspace = 0.1; // interspace inside the 2 ranges of connector pads
200  Double_t step = 0.04; // interspace between each pad inside the connector
201  for (Int_t id = 0; id < 37; id++) {
202  flex->AddNode(
203  connectord, id + total,
204  new TGeoTranslation(length / 2 + interspace / 2 + Geometry::sConnectorLength / 2 - Geometry::sConnectorOffset,
205  id * step - Geometry::sFlexHeight / 2 + 0.1,
206  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sConnectorHeight / 2));
207  auto* transformationpi =
208  new TGeoCombiTrans(length / 2 - interspace / 2 - Geometry::sConnectorLength / 2 - Geometry::sConnectorOffset,
209  id * step - Geometry::sFlexHeight / 2 + 0.1,
210  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sConnectorHeight / 2, rotationpi);
211  flex->AddNode(connectord, id + total + 37, transformationpi);
212  }
213 
214  Double_t boxthickness = 0.05;
215  auto* boxconnect = new TGeoBBox("boxconnect", (2 * Geometry::sConnectorThickness + interspace + boxthickness) / 2,
216  Geometry::sFlexHeight / 2 - 0.04, Geometry::sConnectorHeight / 2);
217  auto* boxremov = new TGeoBBox("boxremov", (2 * Geometry::sConnectorThickness + interspace) / 2,
218  (Geometry::sFlexHeight - 0.1 - step) / 2, Geometry::sConnectorHeight / 2 + 0.001);
219  auto* boxconnecto = new TGeoSubtraction(boxconnect, boxremov, nullptr, nullptr);
220  auto* boxconnector = new TGeoCompositeShape("boxconnector", boxconnecto);
221  auto* boxconnectord = new TGeoVolume("boxconnectord", boxconnector, kMedPeek);
222  flex->AddNode(boxconnectord, 1,
223  new TGeoTranslation(length / 2 - Geometry::sConnectorOffset, -step / 2,
224  zvarnish - Geometry::sVarnishThickness / 2 - Geometry::sConnectorHeight / 2 -
225  Geometry::sConnectorThickness));
226 }

294 TGeoVolume* Flex::makeLines(Int_t nbsensors, Double_t length, Double_t widthflex, Double_t thickness)
295 {
296 
297  // One line is built by removing 3 lines of aluminium in the TGeoBBox *layer_def layer. Then one line is made by the 2
298  // remaining aluminium strips.
299 
300  // the initial layer of aluminium
301  auto* layer_def = new TGeoBBox("layer_def", length / 2, widthflex / 2, thickness / 2);
302 
303  // Two holes for fixing and positionning of the FPC on the cold plate
304  auto* hole1 = new TGeoTube("hole1", 0., Geometry::sRadiusHole1, thickness / 2 + Geometry::sEpsilon);
305  auto* hole2 = new TGeoTube("hole2", 0., Geometry::sRadiusHole2, thickness / 2 + Geometry::sEpsilon);
306 
307  auto* t1 = new TGeoTranslation("t1", length / 2 - Geometry::sHoleShift1, 0., 0.);
308  auto* layerholesub1 = new TGeoSubtraction(layer_def, hole1, nullptr, t1);
309  auto* layerhole1 = new TGeoCompositeShape("layerhole1", layerholesub1);
310 
311  auto* t2 = new TGeoTranslation("t2", length / 2 - Geometry::sHoleShift2, 0., 0.);
312  auto* layerholesub2 = new TGeoSubtraction(layerhole1, hole2, nullptr, t2);
313  auto* layer = new TGeoCompositeShape("layerhole2", layerholesub2);
314 
315  TGeoBBox* line[25];
316  TGeoTranslation *t[6], *ts[15], *tvdd, *tl[2];
317  TGeoSubtraction* layerl[25];
318  TGeoCompositeShape* layern[25];
319  Int_t istart, istop;
320  Int_t kTotalLinesNb = 0;
321  Int_t kTotalLinesNb1, kTotalLinesNb2;
322  Double_t length_line;
323 
324  // ----------- two lines along the FPC digital side --------------
325  t[0] = new TGeoTranslation("t0", SegmentationAlpide::SensorSizeCols / 2 - Geometry::sConnectorOffset / 2,
326  -widthflex / 2 + 2 * Geometry::sLineWidth, 0.);
327  line[0] = new TGeoBBox("line0", length / 2 - Geometry::sConnectorOffset / 2 - SegmentationAlpide::SensorSizeCols / 2,
328  Geometry::sLineWidth / 2, thickness / 2 + Geometry::sEpsilon);
329  layerl[0] = new TGeoSubtraction(layer, line[0], nullptr, t[0]);
330  layern[0] = new TGeoCompositeShape(Form("layer%d", 0), layerl[0]);
331 
332  istart = 1;
333  istop = 6;
334  for (int iline = istart; iline < istop; iline++) {
335  t[iline] =
336  new TGeoTranslation(Form("t%d", iline), SegmentationAlpide::SensorSizeCols / 2 - Geometry::sConnectorOffset / 2,
337  -widthflex / 2 + 2 * (iline + 1) * Geometry::sLineWidth, 0.);
338  line[iline] = new TGeoBBox(Form("line%d", iline),
339  length / 2 - Geometry::sConnectorOffset / 2 - SegmentationAlpide::SensorSizeCols / 2,
340  Geometry::sLineWidth / 2, thickness / 2 + Geometry::sEpsilon);
341  layerl[iline] = new TGeoSubtraction(layern[iline - 1], line[iline], nullptr, t[iline]);
342  layern[iline] = new TGeoCompositeShape(Form("layer%d", iline), layerl[iline]);
343  kTotalLinesNb++;
344  }
345 
346  // --------- lines for the sensors, one line/sensor -------------
347  istart = kTotalLinesNb + 1;
348  istop = 6 + 3 * nbsensors;
349  for (int iline = istart; iline < istop; iline++) {
350  length_line = length - Geometry::sConnectorOffset -
351  TMath::Nint((iline - 6) / 3) * SegmentationAlpide::SensorSizeCols -
352  SegmentationAlpide::SensorSizeCols / 2;
353  ts[iline] = new TGeoTranslation(Form("t%d", iline), length / 2 - length_line / 2 - Geometry::sConnectorOffset,
354  -2 * (iline - 6) * Geometry::sLineWidth + 0.5 - widthflex / 2, 0.);
355  line[iline] = new TGeoBBox(Form("line%d", iline), length_line / 2, Geometry::sLineWidth / 2,
356  thickness / 2 + Geometry::sEpsilon);
357  layerl[iline] = new TGeoSubtraction(layern[iline - 1], line[iline], nullptr, ts[iline]);
358  layern[iline] = new TGeoCompositeShape(Form("layer%d", iline), layerl[iline]);
359  kTotalLinesNb++;
360  }
361 
362  // --------- an interspace to separate AVDD and DVDD -------------
363  kTotalLinesNb++;
364  tvdd = new TGeoTranslation("tvdd", 0., widthflex / 2 - Geometry::sShiftDDGNDline, 0.);
365  line[kTotalLinesNb] = new TGeoBBox(Form("line%d", kTotalLinesNb), length / 2, 2 * Geometry::sLineWidth / 2,
366  thickness / 2 + Geometry::sEpsilon);
367  layerl[kTotalLinesNb] = new TGeoSubtraction(layern[kTotalLinesNb - 1], line[kTotalLinesNb], nullptr, tvdd);
368  layern[kTotalLinesNb] = new TGeoCompositeShape(Form("layer%d", kTotalLinesNb), layerl[kTotalLinesNb]);
369  kTotalLinesNb++;
370 
371  // --------- one line along the FPC analog side -------------
372  istart = kTotalLinesNb;
373  istop = kTotalLinesNb + 2;
374  for (int iline = istart; iline < istop; iline++) {
375  length_line = length - Geometry::sConnectorOffset;
376  tl[iline - istart] =
377  new TGeoTranslation(Form("tl%d", iline), length / 2 - length_line / 2 - Geometry::sConnectorOffset,
378  widthflex / 2 - Geometry::sShiftline - 2. * (iline - istart) * Geometry::sLineWidth, 0.);
379  line[iline] = new TGeoBBox(Form("line%d", iline), length_line / 2, Geometry::sLineWidth / 2,
380  thickness / 2 + Geometry::sEpsilon);
381  layerl[iline] = new TGeoSubtraction(layern[iline - 1], line[iline], nullptr, tl[iline - istart]);
382  layern[iline] = new TGeoCompositeShape(Form("layer%d", iline), layerl[iline]);
383  kTotalLinesNb++;
384  }
385 
386  Geometry* mftGeom = Geometry::instance();
387  Int_t idHalfMFT = mftGeom->getHalfID(mLadderSeg->GetUniqueID());
388  Int_t idHalfDisk = mftGeom->getDiskID(mLadderSeg->GetUniqueID());
389  Int_t idLadder = mftGeom->getLadderID(mLadderSeg->GetUniqueID());
390 
391  TGeoMedium* kMedAlu = gGeoManager->GetMedium("MFT_Alu$");
392 
393  auto* lineslayer =
394  new TGeoVolume(Form("lineslayer_%d_%d_%d", idHalfMFT, idHalfDisk, idLadder), layern[kTotalLinesNb - 1], kMedAlu);
395  lineslayer->SetVisibility(true);
396  lineslayer->SetLineColor(kBlue);
397 
398  return lineslayer;
399 }

402 TGeoVolume* Flex::makeAGNDandDGND(Double_t length, Double_t widthflex, Double_t thickness)
403 {
404 
405  // AGND and DGND layers
406  auto* layer = new TGeoBBox("layer", length / 2, widthflex / 2, thickness / 2);
407  auto* hole1 = new TGeoTube("hole1", 0., Geometry::sRadiusHole1, thickness / 2 + Geometry::sEpsilon);
408  auto* hole2 = new TGeoTube("hole2", 0., Geometry::sRadiusHole2, thickness / 2 + Geometry::sEpsilon);
409 
410  auto* t1 = new TGeoTranslation("t1", length / 2 - Geometry::sHoleShift1, 0., 0.);
411  auto* layerholesub1 = new TGeoSubtraction(layer, hole1, nullptr, t1);
412  auto* layerhole1 = new TGeoCompositeShape("layerhole1", layerholesub1);
413 
414  auto* t2 = new TGeoTranslation("t2", length / 2 - Geometry::sHoleShift2, 0., 0.);
415  auto* layerholesub2 = new TGeoSubtraction(layerhole1, hole2, nullptr, t2);
416  auto* layerhole2 = new TGeoCompositeShape("layerhole2", layerholesub2);
417 
418  //--------------
419  TGeoBBox* line[3];
420  TGeoTranslation* t[3];
421  TGeoCompositeShape* layern[3];
422  TGeoSubtraction* layerl[3];
423  Double_t length_line;
424  length_line = length - Geometry::sConnectorOffset;
425 
426  // First, the two lines along the FPC side
427  t[0] = new TGeoTranslation("t0", length / 2 - length_line / 2 - Geometry::sConnectorOffset,
428  widthflex / 2 - Geometry::sShiftline, 0.);
429  line[0] = new TGeoBBox("line0", length / 2 - Geometry::sConnectorOffset / 2, Geometry::sLineWidth / 2,
430  thickness / 2 + Geometry::sEpsilon);
431  layerl[0] = new TGeoSubtraction(layerhole2, line[0], nullptr, t[0]);
432  layern[0] = new TGeoCompositeShape(Form("layer%d", 0), layerl[0]);
433 
434  t[1] = new TGeoTranslation("t1", length / 2 - length_line / 2 - Geometry::sConnectorOffset,
435  widthflex / 2 - Geometry::sShiftline - 2 * Geometry::sLineWidth, 0.);
436  line[1] = new TGeoBBox("line1", length / 2 - Geometry::sConnectorOffset / 2, Geometry::sLineWidth / 2,
437  thickness / 2 + Geometry::sEpsilon);
438  layerl[1] = new TGeoSubtraction(layern[0], line[1], nullptr, t[1]);
439  layern[1] = new TGeoCompositeShape(Form("layer%d", 1), layerl[1]);
440 
441  // Now the interspace to separate the AGND et DGND --> same interspace compare the AVDD et DVDD
