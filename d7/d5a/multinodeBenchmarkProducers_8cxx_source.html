43 WorkflowSpec defineDataProcessing(ConfigContext const& config)
44 {
45  int objectsBins = config.options().get<int>("obj-bins");
46  double objectsRate = config.options().get<double>("obj-rate");
47  int objectsProducers = config.options().get<int>("obj-producers");
48  int objectsPerMessage = config.options().get<int>("obj-per-message");
49  std::string outputChannelConfig = config.options().get<std::string>("output-channel-config");
50  SubSpec subSpec = static_cast<SubSpec>(config.options().get<int>("first-subspec"));
51  WorkflowSpec specs;
52  // clang-format off
53  // one 1D histo
54  for (size_t p = 0; p < objectsProducers; p++, subSpec++) {
55  DataProcessorSpec producer{
56  "producer-histo" + std::to_string(subSpec),
57  Inputs{},
58  Outputs{ { { "histo" }, "TST", "HISTO", subSpec} },
59  AlgorithmSpec{
60  (AlgorithmSpec::InitCallback)[=](InitContext& ictx) {
61  const size_t randoms = 10000;
62  int periodus = static_cast<int>(1000000 / objectsRate);
63  TRandom gen;
64  gen.SetSeed(p);
65 
66  double randomsArray[randoms];
67  TObjArray* collection = new TObjArray();
68  collection->SetOwner(true);
69 
70  for (size_t i = 0; i < objectsPerMessage; i++) {
71  TH1I* h = new TH1I(std::to_string(i).c_str(), "uni", objectsBins, 0, 1);
72  collection->Add(h);
73  }
74 
75  return (AlgorithmSpec::ProcessCallback)[=](ProcessingContext& pctx) mutable {
76 
77  static auto lastTime = steady_clock::now() - std::chrono::microseconds(periodus * p / objectsProducers);
78  auto now = steady_clock::now();
79  if (duration_cast<microseconds>(now - lastTime).count() > periodus) {
80  lastTime += microseconds(periodus);
81 
82  if (objectsPerMessage > 1) {
83  for (auto o : *collection) {
84  gen.RndmArray(randoms, randomsArray);
85 
86  TH1I* h = dynamic_cast<TH1I*>(o);
87  h->Reset();
88  h->FillN(randoms, randomsArray, nullptr);
89  }
90 
91  collection->SetOwner(false);
92  pctx.outputs().snapshot({"TST", "HISTO", subSpec}, *collection);
93  collection->SetOwner(true);
94  } else {
95  gen.RndmArray(randoms, randomsArray);
96 
97  TH1I* h = dynamic_cast<TH1I*>(collection->At(0));
98  h->Reset();
99  h->FillN(randoms, randomsArray, nullptr);
100 
101  pctx.outputs().snapshot({"TST", "HISTO", subSpec}, *h);
102  }
103  }
104  };
105  }
106  }
107  };
108  specs.push_back(producer);
109 
110  // We spawn one proxy per each producer to simulate the real scenario.
111  specs.emplace_back(
112  std::move(
113  specifyFairMQDeviceOutputProxy(
114  ("histo-proxy-" + std::to_string(p)).c_str(),
115  {{"histo", "TST", "HISTO", subSpec }}, outputChannelConfig.c_str())));
116  }
117 
118  return specs;
119 }
