49 NodeColor
50  decideColorForNode(const DeviceInfo& info)
51 {
52  NodeColor result;
53  if (info.active == false) {
54  result.normal = PaletteHelpers::RED;
55  result.hovered = PaletteHelpers::SHADED_RED;
56  return result;
57  }
58  switch (info.streamingState) {
59  case StreamingState::EndOfStreaming:
60  result.normal = PaletteHelpers::SHADED_YELLOW;
61  result.hovered = PaletteHelpers::YELLOW;
62  result.title = PaletteHelpers::YELLOW;
63  result.title_hovered = PaletteHelpers::DARK_YELLOW;
64  break;
65  case StreamingState::Idle:
66  result.normal = PaletteHelpers::SHADED_GREEN;
67  result.hovered = PaletteHelpers::GREEN;
68  result.title = PaletteHelpers::GREEN;
69  result.title_hovered = PaletteHelpers::DARK_GREEN;
70  break;
71  case StreamingState::Streaming:
72  default:
73  result.normal = PaletteHelpers::GRAY;
74  result.hovered = PaletteHelpers::LIGHT_GRAY;
75  result.title = PaletteHelpers::GRAY;
76  result.title_hovered = PaletteHelpers::BLACK;
77  break;
78  }
79  return result;
80 }

83 void displayGrid(bool show_grid, ImVec2 offset, ImDrawList* draw_list)
84 {

139 // Private helper struct for the layout of the graph
140 struct NodePos {
141  ImVec2 pos;
142  static ImVec2 GetInputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
143  {
144  ImVec2 const& pos = positions[nodeId].pos;
145  ImVec2 const& size = infos[nodeId].Size;
146  float inputsCount = infos[nodeId].InputsCount;
147  return ImVec2(pos.x, pos.y + size.y * ((float)slot_no + 1) / (inputsCount + 1));
148  }
149  static ImVec2 GetOutputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
150  {
151  ImVec2 const& pos = positions[nodeId].pos;
152  ImVec2 const& size = infos[nodeId].Size;
153  float outputsCount = infos[nodeId].OutputsCount;
154  return ImVec2(pos.x + size.x, pos.y + size.y * ((float)slot_no + 1) / (outputsCount + 1));
155  }
156 };

197  auto prepareChannelView = [&specs, &metadata](ImVector<Node>& nodeList, ImVector<Group>& groupList) {
198  struct LinkInfo {
199  int specId;
200  int outputId;
201  };
202  std::map<std::string, LinkInfo> linkToIndex;
203  for (int si = 0; si < specs.size(); ++si) {
204  int oi = 0;
205  for (auto&& output : specs[si].outputChannels) {
206  linkToIndex.insert(std::make_pair(output.name, LinkInfo{si, oi}));
207  oi += 1;
208  }
209  }
210  // Prepare the list of groups.
211  std::string workflow = "Ungrouped";
212  int groupId = 0;
213  for (size_t mi = 0; mi < metadata.size(); ++mi) {
214  auto const& metadatum = metadata[mi];
215  if (metadatum.executable == workflow) {
216  continue;
217  }
218  workflow = metadatum.executable;
219  char* groupBasename = strrchr(workflow.data(), '/');
220  char const* groupName = groupBasename ? groupBasename + 1 : workflow.data();
221  bool hasDuplicate = false;
222  for (size_t gi = 0; gi < groupList.Size; ++gi) {
223  if (strncmp(groupName, groupList[gi].name, MAX_GROUP_NAME_SIZE - 1) == 0) {
224  hasDuplicate = true;
225  break;
226  }
227  }
228  if (hasDuplicate == false) {
229  groupList.push_back(Group(groupId++, groupName, mi));
230  }
231  }
232  // Do matching between inputs and outputs
233  for (int si = 0; si < specs.size(); ++si) {
234  auto& spec = specs[si];
235  int groupId = 0;
236 
237  auto metadatum = std::find_if(metadata.begin(), metadata.end(),
238  [& name = spec.name](DataProcessorInfo const& info) { return info.name == name; });
239 
240  for (size_t gi = 0; gi < groupList.Size; ++gi) {
241  if (metadatum == metadata.end()) {
242  break;
243  }
244  const char* groupName = strrchr(metadatum->executable.data(), '/');
245  if (strncmp(groupList[gi].name, groupName ? groupName + 1 : metadatum->executable.data(), 127) == 0) {
246  groupId = gi;
247  break;
248  }
249  }
250  nodeList.push_back(Node(si, groupId, spec.id.c_str(), 0.5f,
251  ImColor(255, 100, 100),
252  spec.inputChannels.size(),
253  spec.outputChannels.size()));
254  int ii = 0;
255  for (auto& input : spec.inputChannels) {
256  auto const& outName = input.name;
257  auto const& out = linkToIndex.find(input.name);
258  if (out == linkToIndex.end()) {
259  LOG(ERROR) << "Could not find suitable node for " << outName;
260  continue;
261  }
262  links.push_back(NodeLink{out->second.specId, out->second.outputId, si, ii});
263  ii += 1;
264  }
265  }
266 
267  // ImVector does boudary checks, so I bypass the case there is no
268  // edges.
269  std::vector<TopoIndexInfo> sortedNodes = {{0, 0}};
270  if (links.size()) {
271  sortedNodes = WorkflowHelpers::topologicalSort(specs.size(), &(links[0].InputIdx), &(links[0].OutputIdx), sizeof(links[0]), links.size());
272  }
273  // This is to protect for the cases in which there is a loop in the
274  // definition of the inputs and of the outputs due to the
275  // way the forwarding creates hidden dependencies between processes.
276  // This should not happen, but apparently it does.
277  for (auto di = 0; di < specs.size(); ++di) {
278  auto fn = std::find_if(sortedNodes.begin(), sortedNodes.end(), [di](TopoIndexInfo const& info) {
279  return di == info.index;
280  });
281  if (fn == sortedNodes.end()) {
282  sortedNodes.push_back({(int)di, 0});
283  }
284  }
285  assert(specs.size() == sortedNodes.size());
287  std::sort(sortedNodes.begin(), sortedNodes.end());
288 
289  std::vector<int> layerEntries(1024, 0);
290  std::vector<int> layerMax(1024, 0);
291  for (auto& node : sortedNodes) {
292  layerMax[node.layer < 1023 ? node.layer : 1023] += 1;
293  }
294 
295  // FIXME: display nodes using topological sort

297  for (int si = 0; si < specs.size(); ++si) {
298  auto& node = sortedNodes[si];
299  assert(node.index == si);
300  int xpos = 40 + 240 * node.layer;
301  int ypos = 300 + (600 / (layerMax[node.layer] + 1)) * (layerEntries[node.layer] - layerMax[node.layer] / 2);
302  positions.push_back(NodePos{ImVec2(xpos, ypos)});
303  layerEntries[node.layer] += 1;
304  }
305  };
