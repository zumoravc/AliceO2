34 BOOST_AUTO_TEST_CASE(RootTree2Table)
35 {
36  using namespace o2::framework;
38  TTree t1("t1", "a simple Tree with simple variables");
39  Float_t xyz[3];
40  Int_t ij[2];
41  Float_t px, py, pz;
42  Double_t random;
43  Int_t ev;
44  t1.Branch("px", &px, "px/F");
45  t1.Branch("py", &py, "py/F");
46  t1.Branch("pz", &pz, "pz/F");
47  t1.Branch("random", &random, "random/D");
48  t1.Branch("ev", &ev, "ev/I");
49  t1.Branch("xyz", xyz, "xyz[3]/F");
50  t1.Branch("ij", ij, "ij[2]/I");
51  //fill the tree
52  for (Int_t i = 0; i < 1000; i++) {
53  //gRandom->Rannor(xyz[0], xyz[1]);
54  xyz[0] = 1;
55  xyz[1] = 2;
56  xyz[2] = 3;
57  gRandom->Rannor(px, py);
58  pz = px * px + py * py;
59  xyz[2] = i + 1;
60  ij[0] = i;
61  ij[1] = i + 1;
62  random = gRandom->Rndm();
63  ev = i + 1;
64  t1.Fill();
65  }
66 
67  // Create an arrow table from this.
68  TableBuilder builder;
69  TTreeReader reader(&t1);
70  auto&& xyzReader = HolderMaker<float[3]>::make(reader, "xyz");
71  auto&& ijkReader = HolderMaker<int[2]>::make(reader, "ij");
72  auto&& pxReader = HolderMaker<float>::make(reader, "px");
73  auto&& pyReader = HolderMaker<float>::make(reader, "py");
74  auto&& pzReader = HolderMaker<float>::make(reader, "pz");
75  auto&& randomReader = HolderMaker<double>::make(reader, "random");
76  auto&& evReader = HolderMaker<int>::make(reader, "ev");
77 
78  RootTableBuilderHelpers::convertTTree(builder, reader, std::move(xyzReader), std::move(ijkReader), std::move(pxReader), std::move(pyReader), std::move(pzReader), std::move(randomReader), std::move(evReader));
79  auto table = builder.finalize();
80  BOOST_REQUIRE_EQUAL(table->num_rows(), 1000);
81  BOOST_REQUIRE_EQUAL(table->num_columns(), 7);
82  BOOST_REQUIRE_EQUAL(table->schema()->field(0)->type()->id(), arrow::fixed_size_binary(sizeof(float[3]))->id());
83  BOOST_REQUIRE_EQUAL(table->schema()->field(1)->type()->id(), arrow::fixed_size_binary(sizeof(int[2]))->id());
84  BOOST_REQUIRE_EQUAL(table->schema()->field(2)->type()->id(), arrow::float32()->id());
85  BOOST_REQUIRE_EQUAL(table->schema()->field(3)->type()->id(), arrow::float32()->id());
86  BOOST_REQUIRE_EQUAL(table->schema()->field(4)->type()->id(), arrow::float32()->id());
87  BOOST_REQUIRE_EQUAL(table->schema()->field(5)->type()->id(), arrow::float64()->id());

125 BOOST_AUTO_TEST_CASE(RootTree2TableViaASoA)
126 {
127  using namespace o2::framework;
129  TTree t2("t2", "a simple Tree with simple variables");
130  Float_t xyz[3];
131  Int_t ij[2];
132  Float_t px, py, pz;
133  Double_t random;
134  Int_t ev;
135  t2.Branch("px", &px, "px/F");
136  t2.Branch("py", &py, "py/F");
137  t2.Branch("pz", &pz, "pz/F");
138  t2.Branch("random", &random, "random/D");
139  t2.Branch("ev", &ev, "ev/I");
140  t2.Branch("xyz", xyz, "xyz[3]/F");
141  t2.Branch("ij", ij, "ij[2]/I");
142  //fill the tree
143  for (Int_t i = 0; i < 1000; i++) {
144  gRandom->Rannor(xyz[0], xyz[1]);
145  gRandom->Rannor(px, py);
146  pz = px * px + py * py;
147  xyz[2] = i + 1;
148  ij[0] = i;
149  ij[1] = i + 1;
150  random = gRandom->Rndm();
151  ev = i + 1;
152  t2.Fill();
153  }
154

156  TableBuilder builder;
157  TTreeReader reader(&t2);
158  BOOST_REQUIRE_EQUAL(t2.GetEntries(), 1000);
159 
160  RootTableBuilderHelpers::convertASoA<o2::aod::Test>(builder, reader);
161  auto table = builder.finalize();
162  BOOST_REQUIRE_EQUAL(table->num_rows(), 1000);
163  BOOST_REQUIRE_EQUAL(table->num_columns(), 7);
164  BOOST_REQUIRE_EQUAL(table->column(0)->type()->id(), arrow::float32()->id());
165  BOOST_REQUIRE_EQUAL(table->column(1)->type()->id(), arrow::float32()->id());
166  BOOST_REQUIRE_EQUAL(table->column(2)->type()->id(), arrow::float32()->id());
167  BOOST_REQUIRE_EQUAL(table->column(3)->type()->id(), arrow::fixed_size_binary(sizeof(float[3]))->id());
168  BOOST_REQUIRE_EQUAL(table->column(4)->type()->id(), arrow::fixed_size_binary(sizeof(int[2]))->id());
169  BOOST_REQUIRE_EQUAL(table->column(5)->type()->id(), arrow::float64()->id());
170  BOOST_REQUIRE_EQUAL(table->column(6)->type()->id(), arrow::int32()->id());
171 
172  o2::aod::Test testTable{table};
173  for (auto& row : testTable) {
174  BOOST_REQUIRE_EQUAL(row.ij()[0], row.ij()[1] - 1);
