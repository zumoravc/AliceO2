21 bool IntervalXY::seenByCircle(const CircleXY& circle, float eps) const
22 {
23  // check if XY interval is seen by the circle.
24  // The tolerance parameter eps is interpreted as a fraction of the interval
25  // lenght to be added to the edges along the interval (i.e. while the points
26  // within the interval are spanning
27  // x = xc + dx*t
28  // y = yc + dy*t
29  // with 0<t<1., we acctually check the interval for -eps<t<1+eps
30  auto dx0 = xP - circle.xC, dy0 = yP - circle.yC, dx1 = dx0 + dxP, dy1 = dy0 + dyP;
31  if (eps > 0.) {
32  auto dxeps = eps * dxP, dyeps = eps * dyP;
33  dx0 -= dxeps;
34  dx1 += dxeps;
35  dy0 -= dyeps;
36  dy1 += dyeps;
37  }
38  auto d02 = dx0 * dx0 + dy0 * dy0, d12 = dx1 * dx1 + dy1 * dy1, rC2 = circle.rC * circle.rC; // distance^2 from circle center to edges
39  return (d02 - rC2) * (d12 - rC2) < 0;
40 }

43 bool IntervalXY::seenByLine(const IntervalXY& other, float eps) const
44 {
45  // check if XY interval is seen by the line defined by other interval
46  // The tolerance parameter eps is interpreted as a fraction of the interval
47  // lenght to be added to the edges along the interval (i.e. while the points
48  // within the interval are spanning
49  // x = xc + dx*t
50  // y = yc + dy*t
51  // with 0<t<1., we acctually check the interval for -eps<t<1+eps
52  float a, b, c, x0, y0, x1, y1; // find equation of the line a*x+b*y+c = 0
53  other.getLineCoefs(a, b, c);
54  eval(-eps, x0, y0);
55  eval(1.f + eps, x1, y1);
56  return (a * x0 + b * y0 + c) * (a * x1 + b * y1 + c) < 0;
57 }

60 bool IntervalXY::lineCrossParam(const IntervalXY& other, float& t) const
61 {
62  // get crossing parameter of 2 intervals
63  float dx = other.xP - xP, dy = other.yP - yP;
64  float det = -dxP * other.dyP + dyP * other.dxP;
65  if (fabs(det) < 1.e-9) {
66  return false; // parallel
67  }
68  t = (-dx * other.dyP + dy * other.dxP) / det;
69  return true;
70 }

73 bool IntervalXY::circleCrossParam(const CircleXY& circle, float& t) const
74 {
75  // get crossing point of circle with the interval
76  // solution of the system (wrt t)
77  // (x-xC)^2+(y-yC)^2 = rC2
78  // x = xEdge + xProj * t;
79  // y = yEdge + yProj * t;
80  float dx = xP - circle.xC, dy = yP - circle.yC, del2I = 1. / (dxP * dxP + dyP * dyP),
81  b = (dx * dxP + dy * dyP) * del2I, c = del2I * (dx * dx + dy * dy - circle.rC * circle.rC);
82  float det = b * b - c;
83  if (det < 0.) {
84  return false;
85  }
86  det = sqrtf(det);
87  float t0 = -b - det, t1 = -b + det;
88  // select the one closer to [0:1] interval
89  t = (fabs(t0 - 0.5) < fabs(t1 - 0.5)) ? t0 : t1;
90  return true;
91 }

94 void IntervalXY::getLineCoefs(float& a, float& b, float& c) const
95 {
96  // convert to line parameters in canonical form: a*x+b*y+c = 0
97  c = xP * dyP - yP * dxP;
98  if (c) {
99  a = -dyP;
100  b = dxP;
101  } else if (fabs(dxP) > fabs(dyP)) {
102  a = 0.;
103  b = -1.;
104  c = yP;
105  } else {
106  a = -1.;
107  b = 0.;
108  c = xP;
109  }
110 }
