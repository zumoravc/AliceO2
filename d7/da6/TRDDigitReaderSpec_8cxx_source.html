79 void TRDDigitReaderSpec::run(ProcessingContext& pc)
80 {
81  if (mState != 1) {
82  LOG(info) << "mState is not 1";
83  return;
84  }
85  LOG(info) << "in TRDDigitReadSpec run method ";
86  TTree* DPLTree = ((TTree*)mFile->Get(mDigitTreeName.c_str()));
87  if (DPLTree) {
88  DPLTree->SetBranchAddress(mDigitBranchName.c_str(), &mPDigits);
89  DPLTree->SetBranchAddress(mTriggerRecordBranchName.c_str(), &mPTriggerRecords);
90  DPLTree->SetBranchAddress(mMCLabelsBranchName.c_str(), &mPMCLabels);
91  DPLTree->GetEntry(0);
92  LOG(info) << "TRDDigitReader digits size=" << mDigits.size() << " triggerrecords size=" << mTriggerRecords.size() << " mc labels size=" << mMCLabels.getNElements();
93  pc.outputs().snapshot(Output{"TRD", "DIGITS", 0, Lifetime::Timeframe}, mDigits);
94  pc.outputs().snapshot(Output{"TRD", "TRGRDIG", 0, Lifetime::Timeframe}, mTriggerRecords);
95  pc.outputs().snapshot(Output{"TRD", "LABELS", 0, Lifetime::Timeframe}, mMCLabels);
96  }
97  //delete DPLTree; // next line will delete the pointer as well.
98  mFile->Close();
99  LOG(info) << "returning from run method of TRDDigitReader";
100 
101  mState = 2; // prevent coming in here again.
102  // send endOfData control event and mark the reader as ready to finish
103  pc.services().get<ControlService>().endOfStream();
104  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
105  LOG(info) << "returning from run method of TRDDigitReader finally";
106 }
