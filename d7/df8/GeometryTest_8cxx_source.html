40 TGeoVolume* createAirVacuumCave(const char* name)
41 {
42  // create the air medium (only used for the geometry test)
43  auto& mgr = o2::base::MaterialManager::Instance();
44 
45  const int nAir = 4;
46  Float_t aAir[nAir] = {12.0107, 14.0067, 15.9994, 39.948};
47  Float_t zAir[nAir] = {6., 7., 8., 18.};
48  Float_t wAir[nAir] = {0.000124, 0.755267, 0.231781, 0.012827};
49  Float_t dAirVacuum = 1.20479E-10;
50  const int kID = 90; // to avoid conflicts with definitions of other MCH materials
51 
52  mgr.Mixture("MCH", kID, "Air", aAir, zAir, dAirVacuum, nAir, wAir);
53  mgr.Medium("MCH", kID, "Air", kID,
54  false, /* isvol */
55  0, /* ifield */
56  -1.0, /* fieldm */
57  -1.0, /* tmaxfd */
58  -1.0, /* stemax */
59  -1.0, /* deemax */
60  -1.0, /* epsil */
61  -1.0 /* stmin */);
62  return gGeoManager->MakeBox(name, gGeoManager->GetMedium("MCH_Air"), 2000.0, 2000.0, 3000.0);
63 }

65 void dump(std::ostream& out, const TGeoNode& n, int level, int maxdepth, std::string prefix)
66 {
67  if (level >= maxdepth) {
68  return;
69  }
70 
71  if (level == 0) {
72  out << n.GetName() << "\n";
73  }
74 
75  if (level < maxdepth) {
76  for (int i = 0; i < n.GetNdaughters(); i++) {
77  TGeoNode* d = n.GetDaughter(i);
78  if (i == n.GetNdaughters() - 1) {
79  out << prefix + "└──" << d->GetName()
80  << "\n";
81  dump(out, *d, level + 1, maxdepth, prefix + " ");
82  } else {
83  out << prefix + "├──" << d->GetName()
84  << "\n";
85  dump(out, *d, level + 1, maxdepth, prefix + "│ ");
86  }
87  }
88  }
89 }

228 Vector3D<double> getNormalVector(const o2::Transform3D& t)
229 {
230  Point3D<double> px, py, po;
231  t.LocalToMaster(Point3D<double>{0, 1, 0}, py);
232  t.LocalToMaster(Point3D<double>{1, 0, 0}, px);
233  t.LocalToMaster(Point3D<double>{0, 0, 0}, po);
234  Vector3D<double> a{px - po};
235  Vector3D<double> b{py - po};
236  return a.Cross(b).Unit();
237 }

239 TH2* getRadio(int detElemId, float xmin, float ymin, float xmax, float ymax, float xstep, float ystep, float thickness)
240 {
241  if (xmin >= xmax || ymin >= ymax) {
242  std::cerr << "incorrect limits\n";
243  return nullptr;
244  }
245  TH2* hmatb = new TH2F("hmatb", "hmatb", (int)((xmax - xmin) / xstep), xmin, xmax, (int)((ymax - ymin) / ystep), ymin, ymax);
246 
247  auto t = o2::mch::getTransformation(detElemId, *gGeoManager);
248 
249  auto normal = getNormalVector(t);
250 
251  for (auto x = xmin; x < xmax; x += xstep) {
252  for (auto y = ymin; y < ymax; y += ystep) {
253  auto matb = getMatBudgetExt(t, normal, x, y, thickness);
254  if (std::isfinite(matb.meanX2X0)) {
255  hmatb->Fill(x, y, matb.meanX2X0);
256  }
257  }
258  }
259  return hmatb;
260 }
