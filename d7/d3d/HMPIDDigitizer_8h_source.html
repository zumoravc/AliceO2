27  public:
28  // set event time and return true if it is within the active hold/data-taking time
29  // or false otherwise (in which case we don't need to do digitization)
30  bool setEventTime(double timeNS)
31  {
32  if ((timeNS - mCurrentTriggerTime) > TRACKHOLDTIME) {
33  return false;
34  } else {
35  return true;
36  }
37  }
38 
39  // set a trigger; returns true if accepted or false if busy
40  // (assuming some extern decision on this time)
41  bool setTriggerTime(double timeNS)
42  {
43  if (mReadoutCounter == -1) {
44  // for the first trigger no busy check necessary
45  mCurrentTriggerTime = timeNS;
46  mReadoutCounter++;
47  return true;
48  } else {
49  if ((timeNS - mCurrentTriggerTime) > BUSYTIME) {
50  mCurrentTriggerTime = timeNS;
51  mReadoutCounter++;
52  return true;
53  } else {
54  return false;
55  }
56  }
57  }
