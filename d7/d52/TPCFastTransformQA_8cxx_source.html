36 int TPCFastTransformQA::doQA(const TPCFastTransform& fastTransform)
37 {
38  const char* fileName = "fastTransformQA.root";
39 
40  AliTPCcalibDB* pCalib = AliTPCcalibDB::Instance();
41  if (!pCalib) {
42  return storeError(-1, "TPCFastTransformQA: No TPC calibration instance found");
43  }
44 
45  AliTPCParam* tpcParam = pCalib->GetParameters();
46  if (!tpcParam) {
47  return storeError(-2, "TPCFastTransformQA: No TPCParam object found");
48  }
49 
50  AliTPCTransform* origTransform = pCalib->GetTransform();
51  if (!origTransform) {
52  return storeError(-3, "TPCFastTransformQA: No TPC transformation found");
53  }
54 
55  const AliTPCRecoParam* rec = origTransform->GetCurrentRecoParam();
56  if (!rec) {
57  return storeError(-5, "TPCFastTransformQA: No TPC Reco Param set in transformation");
58  }
59  rec->Print();
60 
61  int lastTimeBin = rec->GetLastBin();
62 
63  // measure execution time
64  {
65  TStopwatch timer1;
66  double nCalls1 = 0;
67  double sum1 = 0;
68  for (Int_t iSec = 0; iSec < 1; iSec++) {
69  cout << "Measure original transformation time for TPC sector " << iSec << " .." << endl;
70  int nRows = tpcParam->GetNRow(iSec);
71  for (int iRow = 0; iRow < nRows; iRow++) {
72  Int_t nPads = tpcParam->GetNPads(iSec, iRow);
73  for (float pad = 0.5; pad < nPads; pad += 1.) {
74  for (float time = 0; time < lastTimeBin; time++) {
75  Int_t is[] = {iSec};
76  double orig[3] = {static_cast<Double_t>(iRow), pad, time};
77  origTransform->Transform(orig, is, 0, 1);
78  nCalls1++;
79  sum1 += orig[0] + orig[1] + orig[2];
80  }
81  }
82  }
83  }
84  timer1.Stop();
85 
86  TStopwatch timer2;
87  double nCalls2 = 0;
88  double sum2 = 0;
89  for (Int_t iSec = 0; iSec < 1; iSec++) {
90  cout << "Measure fast transformation time for TPC sector " << iSec << " .." << endl;
91  int nRows = tpcParam->GetNRow(iSec);
92  for (int iRow = 0; iRow < nRows; iRow++) {
93  Int_t nPads = tpcParam->GetNPads(iSec, iRow);
94  int slice = 0, slicerow = 0;
95  AliHLTTPCGeometry::Sector2Slice(slice, slicerow, iSec, iRow);
96  for (float pad = 0.5; pad < nPads; pad += 1.) {
97  for (float time = 0; time < lastTimeBin; time++) {
98  float fast[3];
99  fastTransform.Transform(slice, slicerow, pad, time, fast[0], fast[1], fast[2]);
100  nCalls2++;
101  sum2 += fast[0] + fast[1] + fast[2];
102  }
103  }
104  }
105  }
106  timer2.Stop();
107  cout << "nCalls1 = " << nCalls1 << endl;
108  cout << "nCalls2 = " << nCalls2 << endl;
109  cout << "Orig transformation : " << timer1.RealTime() * 1.e9 / nCalls1 << " ns / call" << endl;
110  cout << "Fast transformation : " << timer2.RealTime() * 1.e9 / nCalls2 << " ns / call" << endl;
111 
112  cout << "Fast Transformation speedup: " << 1. * timer1.RealTime() / timer2.RealTime() * nCalls2 / nCalls1 << endl;
113 
114  int size = sizeof(fastTransform) + fastTransform.getFlatBufferSize();
115  cout << "Fast Transformation memory usage: " << size / 1000. / 1000. << " MB" << endl;
116  cout << "ignore this " << sum1 << " " << sum2 << endl;
117  }
118 
119  if (1) {
120  TFile* file = new TFile(fileName, "RECREATE");
121  if (!file || !file->IsOpen()) {
122  return storeError(-1, "Can't recreate QA file !");
123  }
124  file->cd();
125  TNtuple* nt = new TNtuple("fastTransformQA", "fastTransformQA", "sec:row:pad:time:x:y:z:fx:fy:fz");
126 
127  for (Int_t iSec = 0; iSec < 1; iSec++) {
128  int nRows = tpcParam->GetNRow(iSec);
129  for (int iRow = 0; iRow < nRows; iRow++) {
130  cout << "Write fastTransform QA for TPC sector " << iSec << ", row " << iRow << " .." << endl;
131  Int_t nPads = tpcParam->GetNPads(iSec, iRow);
132  int slice = 0, slicerow = 0;
133  AliHLTTPCGeometry::Sector2Slice(slice, slicerow, iSec, iRow);
134  for (float pad = 0.5; pad < nPads; pad += 1.) {
135  for (float time = 0; time < lastTimeBin; time++) {
136  Int_t is[] = {iSec};
137  double orig[3] = {static_cast<Double_t>(iRow), pad, time};
138  float fast[3];
139  origTransform->Transform(orig, is, 0, 1);
140  fastTransform.Transform(slice, slicerow, pad, time, fast[0], fast[1], fast[2]);
141  float entry[] = {(float)iSec, (float)iRow, pad, time, (float)orig[0], (float)orig[1], (float)orig[2], fast[0], fast[1], fast[2]};
142  nt->Fill(entry);
143  }
144  }
145  }
146  }
147  file->Write();
148  file->Close();
149  delete file;
150  }
151  return 0;
152 }
