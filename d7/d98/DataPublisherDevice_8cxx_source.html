30 template <typename T>
31 void fakePayload(std::vector<o2::byte>& buffer, std::function<void(T&, int)> filler, int numOfElements)
32 {
33  auto payloadSize = sizeof(T) * numOfElements;
34  LOG(INFO) << "Payload size " << payloadSize << "\n";
35  buffer.resize(buffer.size() + payloadSize);
36 
37  T* payload = reinterpret_cast<T*>(buffer.data() + sizeof(HeartbeatHeader));
38  for (int i = 0; i < numOfElements; ++i) {
39  new (payload + i) T();
40  // put some random toy time stamp to each cluster
41  filler(payload[i], i);
42  }
43 }

57 void DataPublisherDevice::InitTask()
58 {
59  mInputChannelName = GetConfig()->GetValue<std::string>(OptionKeyInputChannelName);
60  mOutputChannelName = GetConfig()->GetValue<std::string>(OptionKeyOutputChannelName);
61  // TODO: this turned out to be a missing feature in the description fields of
62  // the different headers. The idea is to use a field of n bytes as an integer
63  // but create the integers from a char sequence. This was thought to be done
64  // at compile time, but here we want to have it configurable at runtime.
65  //
66  // GetConfig()->GetValue<std::string>(OptionKeyDataDescription).c_str()
67  // GetConfig()->GetValue<std::string>(OptionKeyDataOrigin).c_str()
68  //
69  // Still needed:
70  // * a registry for descriptors (probably for different descriptor types of
71  // even the same size) to ensure uniqueness
72  // * create the unsigned integer value once from the configurable string and
73  // check in the registry
74  // * constructors and assignment operators taking the integer type as argument
75  if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "TPCCLUSTER") {
76  mDataDescription = DataDescription("CLUSTERS");
77  mDataOrigin = DataOrigin("TPC");
78  } else if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "ITSRAW") {
79  mDataDescription = DataDescription("CLUSTERS");
80  mDataOrigin = DataOrigin("ITS");
81  }
82  mSubSpecification = GetConfig()->GetValue<SubSpecificationT>(OptionKeySubspecification);
83  mFileName = GetConfig()->GetValue<std::string>(OptionKeyFileName);
84 
85  OnData(mInputChannelName.c_str(), &DataPublisherDevice::HandleData);
86 
87  // reserve space for the HBH at the beginning
88  mFileBuffer.resize(sizeof(o2::header::HeartbeatHeader));
89 
90  if (!mFileName.empty()) {
91  AppendFile(mFileName.c_str(), mFileBuffer);
92  } else if (mDataDescription == DataDescription("CLUSTERS") && mDataOrigin == DataOrigin("TPC")) {
93  auto f = [](TPCTestCluster& cluster, int idx) { cluster.timeStamp = idx; };
94  fakePayload<TPCTestCluster>(mFileBuffer, f, 1000);
95  LOG(INFO) << "Payload size (after) " << mFileBuffer.size() << "\n";
96  // For the moment, add the data as another part to this message
97  } else if (mDataDescription == DataDescription("CLUSTERS") && mDataOrigin == DataOrigin("ITS")) {
98  auto f = [](ITSRawData& cluster, int idx) { cluster.timeStamp = idx; };
99  fakePayload<ITSRawData>(mFileBuffer, f, 500);
100  }
101 
102  mFileBuffer.resize(mFileBuffer.size() + sizeof(o2::header::HeartbeatTrailer));
103  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
104  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
105  *hbhOut = o2::header::HeartbeatHeader();
106  *hbtOut = o2::header::HeartbeatTrailer();
107 }

116 bool DataPublisherDevice::HandleO2LogicalBlock(const byte* headerBuffer,
117  size_t headerBufferSize,
118  const byte* dataBuffer,
119  size_t dataBufferSize)
120 {
121  // AliceO2::header::hexDump("data buffer", dataBuffer, dataBufferSize);
122  const auto* dataHeader = o2::header::get<o2::header::DataHeader*>(headerBuffer);
123  const auto* hbfEnvelope = o2::header::get<o2::header::HeartbeatFrameEnvelope*>(headerBuffer);
124 
125  // TODO: not sure what the return value is supposed to indicate, it's
126  // not handled in O2Device::ForEach at the moment
127  // indicate that the block has not been processed by a 'false'
128  if (!dataHeader ||
129  (dataHeader->dataDescription) != o2::header::gDataDescriptionHeartbeatFrame)
130  return false;
131 
132  if (!hbfEnvelope) {
133  LOG(ERROR) << "no heartbeat frame envelope header found";
134  return false;
135  }
136 
137  // TODO: consistency checks
138  // hbfEnvelope->header;
139  // hbfEnvelope->trailer;
140  // - block type in both HBH and HBT
141  // - HBH size + payload size (specified in HBT) + HBT size == dataBufferSize
142  // - dynamically adjust start of the trailer (if this contains more than one
143  // 64 bit word
144 
145  // TODO: make tool for reading and manipulation of the HeartbeatFrame/Envelop
146 
147  // assume everything valid
148  // write the HBH and HBT as envelop to the buffer of the file data
149  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
150  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
151 
152  // copy HBH and HBT, but set the length explicitely to 1
153  // TODO: handle all kinds of corner cases, or add an assert
154  *hbhOut = hbfEnvelope->header;
155  hbhOut->headerLength = 1;
156  *hbtOut = hbfEnvelope->trailer;
157  hbtOut->dataLength = mFileBuffer.size() - sizeof(o2::header::HeartbeatFrameEnvelope);
158 
159  // top level subframe header, the DataHeader is going to be used with
160  // configured description, origin and sub specification
161  o2::header::DataHeader dh;
162  dh.dataDescription = mDataDescription;
163  dh.dataOrigin = mDataOrigin;
164  dh.subSpecification = mSubSpecification;
165  dh.payloadSize = mFileBuffer.size();
166 
167  O2Message outgoing;
168 
169  LOG(DEBUG) << "Sending buffer of size " << mFileBuffer.size() << "\n";
170  LOG(DEBUG) << "Orbit number " << hbhOut->orbit << "\n";
171  // build multipart message from header and payload
172  // TODO: obviously there is a lot to do here, avoid copying etc, this
173  // is just a proof of principle
174  // NewSimpleMessage(mFileBuffer) does not work with the vector
175 
176  // TODO: fix payload size in dh
177  auto* buffer = new char[mFileBuffer.size()];
178  memcpy(buffer, mFileBuffer.data(), mFileBuffer.size());
179  o2::base::addDataBlock(outgoing, dh, NewMessage(
180  buffer, mFileBuffer.size(), [](void* data, void* hint) { delete[] reinterpret_cast<char*>(data); }, nullptr));
181 
182  // send message
183  Send(outgoing, mOutputChannelName.c_str());
184  outgoing.fParts.clear();
185 
186  return true;
187 }

189 bool DataPublisherDevice::AppendFile(const char* name, std::vector<o2::byte>& buffer)
190 {
191  bool result = true;
192  std::ifstream ifile(name, std::ifstream::binary);
193  if (ifile.bad())
194  return false;
195 
196  // get length of file:
197  ifile.seekg(0, ifile.end);
198  int length = ifile.tellg();
199  ifile.seekg(0, ifile.beg);
200 
201  // allocate memory:
202  int position = buffer.size();
203  buffer.resize(buffer.size() + length);
204 
205  // read data as a block:
206  ifile.read(reinterpret_cast<char*>(&buffer[position]), length);
207  if (!(result = ifile.good())) {
208  LOG(ERROR) << "failed to read " << length << " byte(s) from file " << name << std::endl;
209  }
210 
211  ifile.close();
212 
213  return result;
214 }
