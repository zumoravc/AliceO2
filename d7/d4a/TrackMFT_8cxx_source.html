57 void TrackMFT::extrapHelixToZ(double zEnd, double Field)
58 {
60 
61  if (getZ() == zEnd) {
62  return; // nothing to be done if same z
63  }
64 
65  // Extrapolate track parameters
66  double dZ = (zEnd - getZ()); // Propagate in meters
67  double tanl0 = getTanl();
68  double invtanl0 = 1.0 / tanl0;
69  double px0 = getPx();
70  double py0 = getPy();
71  double invqpt0 = getInvQPt();
72  auto q = getCharge();
73  auto Hz = std::copysign(1.0, Field);
74  double k = TMath::Abs(o2::constants::math::B2C * Field);
75  auto invk = 1.0 / k;
76  double theta = -invqpt0 * dZ * k * invtanl0;
77  double costheta, sintheta;
78  o2::utils::sincos(theta, sintheta, costheta);
79 
80  double deltax = Hz * py0 * invk * (1.0 - costheta) - px0 * q * invk * sintheta;
81  double deltay = -Hz * px0 * invk * (1.0 - costheta) - py0 * q * invk * sintheta;
82  double x = getX() + deltax;
83  double y = getY() + deltay;
84  double deltaphi = +dZ * k * invqpt0 * invtanl0;
85  double phi = getPhi() + theta;
86  setX(x);
87  setY(y);
88  setZ(zEnd);
89  setPhi(phi);
90 }
