51 bool TrackFitter::fit(FitterTrackMFT& track, bool smooth, bool finalize,
52  std::list<TrackParamMFT>::reverse_iterator* itStartingParam)
53 {
59 
60  if (mVerbose)
61  std::cout << "\n ***************************** Start Fitting new track ***************************** \n";
62 
63  // initialize the starting track parameters and cluster
64  double chi2invqptquad;
65  double invpqtquad;
66 
67  auto itParam(track.rbegin());
68  if (itStartingParam != nullptr) {
69  // use the ones pointed to by itStartingParam
70  if (*itStartingParam == track.rend()) {
71  LOG(INFO) << "MFT Fit ERROR: invalid track starting parameters";
72  track.removable();
73  return false;
74  }
75  itParam = *itStartingParam;
76  } else {
77  // or start from the last cluster and compute the track parameters from its position
78  // and the one of the first previous cluster found on a different layer
79  auto itPreviousParam(itParam);
80  ++itPreviousParam;
81  invpqtquad = invQPtFromParabola(track, mBZField, chi2invqptquad);
82  track.setInvQPtQuadtratic(invpqtquad);
83  track.setChi2QPtQuadtratic(chi2invqptquad);
84  (*itParam).setInvQPt(track.getInvQPtQuadtratic()); // Initial momentum estimate
85  initTrack(*itParam->getClusterPtr(), *itParam);
86  }
87 
88  if (mVerbose) {
89  std::cout << "Seed covariances:";
90  itParam->getCovariances().Print();
91  }
92 
93  // recusively add the upstream clusters and update the track parameters
94  TrackParamMFT* startingParam = &*itParam;
95  while (++itParam != track.rend()) {
96  if (!addCluster(*startingParam, *itParam->getClusterPtr(), *itParam)) {
97  track.removable();
98  return false;
99  }
100  startingParam = &*itParam;
101  }
102 
103  itParam--;
104  if (mVerbose) {
105  std::cout << "Track covariances:";
106  itParam->getCovariances().Print();
107  std::cout << "Track Chi2 = " << itParam->getTrackChi2() << std::endl;
108  std::cout << " ***************************** Done fitting *****************************\n";
109  }
110 
111  // smooth the track if requested and the smoother enabled
112  if (smooth && mSmooth) {
113  if (!smoothTrack(track, finalize)) {
114  track.removable();
115  return false;
116  }
117  }
118  return true;
119 }

122 void TrackFitter::initTrack(const o2::itsmft::Cluster& cl, TrackParamMFT& param)
123 {
124 
129 
130  auto& mftTrackingParam = MFTTrackingParam::Instance();
131 
132  // compute the track parameters at the last cluster
133  double x0 = cl.getX();
134  double y0 = cl.getY();
135  double z0 = cl.getZ();
136  double pt = TMath::Sqrt(x0 * x0 + y0 * y0);
137  double pz = z0;
138  double phi0 = TMath::ATan2(cl.getY(), cl.getX());
139  double tanl = pz / pt;
140  double r0sq = cl.getX() * cl.getX() + cl.getY() * cl.getY();
141  double r0cu = r0sq * TMath::Sqrt(r0sq);
142  double invr0sq = 1.0 / r0sq;
143  double invr0cu = 1.0 / r0cu;
144  double sigmax0sq = cl.getSigmaZ2(); // FIXME: from cluster
145  double sigmay0sq = cl.getSigmaY2(); // FIXME: from cluster
146  double sigmaDeltaZsq = 5.0; // Primary vertex distribution: beam interaction diamond
147  double sigmaboost = mftTrackingParam.sigmaboost; // Boost q/pt seed covariances
148  double seedH_k = mftTrackingParam.sigmaboost; // SeedH constant
149 
150  param.setX(cl.getX());
151  param.setY(cl.getY());
152  param.setZ(cl.getZ());
153  param.setPhi(phi0);
154  param.setTanl(tanl);
155 
156  // Configure the track seed
157  switch (mftTrackingParam.seed) {
158  case AB:
159  if (mVerbose)
160  std::cout << " Init track with Seed A / B; sigmaboost = " << sigmaboost << ".\n";
161  param.setInvQPt(1.0 / pt); // Seeds A & B
162  break;
163  case CE:
164  if (mVerbose)
165  std::cout << " Init track with Seed C / E; sigmaboost = " << sigmaboost << ".\n";
166  param.setInvQPt(std::copysign(1.0, param.getInvQPt()) / pt); // Seeds C & E
167  break;
168  case DH:
169  if (mVerbose)
170  std::cout << " Init track with Seed H; (k = " << seedH_k << "); sigmaboost = " << sigmaboost << ".\n";
171  param.setInvQPt(param.getInvQPt() / seedH_k); // SeedH
172  break;
173  default:
174  if (mVerbose)
175  std::cout << " Init track with Seed D.\n";
176  break;
177  }
178 
179  if (mVerbose) {
180  auto model = mftTrackingParam.trackmodel == Helix ? "Helix" : "Quadratic";
181  std::cout << "Track Model: " << model << std::endl;
182  std::cout << "initTrack Cluster X = " << cl.getX() << " Y = " << cl.getY() << " Z = " << cl.getZ() << std::endl;
183  std::cout << " seed Phi, Tanl, InvQpt = " << param.getPhi() << " " << param.getTanl() << " " << param.getInvQPt() << std::endl;
184  }
185 
186  // compute the track parameter covariances at the last cluster (as if the other clusters did not exist)
187  TMatrixD lastParamCov(5, 5);
188  lastParamCov.Zero();
189  lastParamCov(0, 0) = sigmaboost * sigmax0sq; // <X,X>
190  lastParamCov(0, 1) = 0; // <Y,X>
191  lastParamCov(0, 2) = sigmaboost * -sigmax0sq * y0 * invr0sq; // <PHI,X>
192  lastParamCov(0, 3) = sigmaboost * -z0 * sigmax0sq * x0 * invr0cu; // <TANL,X>
193  lastParamCov(0, 4) = sigmaboost * -x0 * sigmax0sq * invr0cu; // <INVQPT,X>
194 
195  lastParamCov(1, 1) = sigmaboost * sigmay0sq; // <Y,Y>
196  lastParamCov(1, 2) = sigmaboost * sigmay0sq * x0 * invr0sq; // <PHI,Y>
197  lastParamCov(1, 3) = sigmaboost * -z0 * sigmay0sq * y0 * invr0cu; // <TANL,Y>
198  lastParamCov(1, 4) = sigmaboost * y0 * sigmay0sq * invr0cu; //1e-2; // <INVQPT,Y>
199 
200  lastParamCov(2, 2) = sigmaboost * (sigmax0sq * y0 * y0 + sigmay0sq * x0 * x0) * invr0sq * invr0sq; // <PHI,PHI>
201  lastParamCov(2, 3) = sigmaboost * z0 * x0 * y0 * (sigmax0sq - sigmay0sq) * invr0sq * invr0cu; // <TANL,PHI>
202  lastParamCov(2, 4) = sigmaboost * y0 * x0 * invr0cu * invr0sq * (sigmax0sq - sigmay0sq); // <INVQPT,PHI>
203 
204  lastParamCov(3, 3) = sigmaboost * z0 * z0 * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0) * invr0cu * invr0cu + sigmaDeltaZsq * invr0sq; // <TANL,TANL>
205  lastParamCov(3, 4) = sigmaboost * z0 * invr0cu * invr0cu * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0); // <INVQPT,TANL>
206 
207  lastParamCov(4, 4) = sigmaboost * sigmaboost * (sigmax0sq * x0 * x0 + sigmay0sq * y0 * y0) * invr0cu * invr0cu; // <INVQPT,INVQPT>
208 
209  lastParamCov(1, 0) = lastParamCov(0, 1); //
210  lastParamCov(2, 0) = lastParamCov(0, 2); //
211  lastParamCov(2, 1) = lastParamCov(1, 2); //
212  lastParamCov(3, 0) = lastParamCov(0, 3); //
213  lastParamCov(3, 1) = lastParamCov(1, 3); //
214  lastParamCov(3, 2) = lastParamCov(2, 3); //
215  lastParamCov(4, 0) = lastParamCov(0, 4); //
216  lastParamCov(4, 1) = lastParamCov(1, 4); //
217  lastParamCov(4, 2) = lastParamCov(2, 4); //
218  lastParamCov(4, 3) = lastParamCov(3, 4); //
219 
220  param.setCovariances(lastParamCov);
221 
222  // set other parameters
223  param.setClusterPtr(&cl);
224  param.setTrackChi2(0.);
225 }

302 bool TrackFitter::smoothTrack(FitterTrackMFT& track, bool finalize)
303 {
308 
309  auto itCurrentParam(track.begin());
310  auto itPreviousParam(itCurrentParam);
311  ++itCurrentParam;
312 
313  // smoothed parameters and covariances at first cluster = filtered parameters and covariances
314  itPreviousParam->setSmoothParameters(itPreviousParam->getParameters());
315  itPreviousParam->setSmoothCovariances(itPreviousParam->getCovariances());
316 
317  // local chi2 at first cluster = last additional chi2 provided by Kalman
318  itPreviousParam->setLocalChi2(itPreviousParam->getTrackChi2() - itCurrentParam->getTrackChi2());
319 
320  // recursively smooth the next parameters and covariances
321  do {
322  if (!runSmoother(*itPreviousParam, *itCurrentParam)) {
323  return false;
324  }
325  ++itPreviousParam;
326  } while (++itCurrentParam != track.end());
327 
328  // update the regular parameters and covariances if requested
329  if (finalize) {
330  for (auto& param : track) {
331  param.setParameters(param.getSmoothParameters());
332  param.setCovariances(param.getSmoothCovariances());
333  }
334  }
335  return true;
336 }

338 //_________________________________________________________________________________________________
339 bool TrackFitter::runKalmanFilter(TrackParamMFT& trackParam)
340 {
344 
345  // get actual track parameters (p)
346  TMatrixD param(trackParam.getParameters());
347 
348  // get new cluster parameters (m)
349  const o2::itsmft::Cluster* cluster = trackParam.getClusterPtr();
350  TMatrixD clusterParam(5, 1);
351  clusterParam.Zero();
352  clusterParam(0, 0) = cluster->getX();
353  clusterParam(1, 0) = cluster->getY();
354 
355  // compute the actual parameter weight (W)
356  TMatrixD paramWeight(trackParam.getCovariances());
357  if (paramWeight.Determinant() != 0) {
358  paramWeight.Invert();
359  } else {
360  LOG(INFO) << "runKalmanFilter ERROR: Determinant = 0";
361  return false;
362  }
363 
364  // compute the new cluster weight (U)
365  TMatrixD clusterWeight(5, 5);
366  clusterWeight.Zero();
367  clusterWeight(0, 0) = 1. / cluster->getSigmaZ2(); // 1. / cluster->getEx2();
368  clusterWeight(1, 1) = 1. / cluster->getSigmaY2(); // 1. / cluster->getEy2();
369 
370  // compute the new parameters covariance matrix ((W+U)^-1)
371  TMatrixD newParamCov(paramWeight, TMatrixD::kPlus, clusterWeight);
372  if (newParamCov.Determinant() != 0) {
373  newParamCov.Invert();
374  } else {
375  LOG(INFO) << "runKalmanFilter ERROR: Determinant = 0";
376  return false;
377  }
378  trackParam.setCovariances(newParamCov);
379 
380  // compute the new parameters (p' = ((W+U)^-1)U(m-p) + p)
381  TMatrixD tmp(clusterParam, TMatrixD::kMinus, param); // m-p
382  TMatrixD tmp2(clusterWeight, TMatrixD::kMult, tmp); // U(m-p)
383  TMatrixD newParam(newParamCov, TMatrixD::kMult, tmp2); // ((W+U)^-1)U(m-p)
384  newParam += param; // ((W+U)^-1)U(m-p) + p
385  trackParam.setParameters(newParam);
386 
387  // compute the additional chi2 (= ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m))
388  tmp = newParam; // p'
389  tmp -= param; // (p'-p)
390  TMatrixD tmp3(paramWeight, TMatrixD::kMult, tmp); // W(p'-p)
391  TMatrixD addChi2Track(tmp, TMatrixD::kTransposeMult, tmp3); // ((p'-p)^-1)W(p'-p)
392  tmp = newParam; // p'
393  tmp -= clusterParam; // (p'-m)
394  TMatrixD tmp4(clusterWeight, TMatrixD::kMult, tmp); // U(p'-m)
395  addChi2Track += TMatrixD(tmp, TMatrixD::kTransposeMult, tmp4); // ((p'-p)^-1)W(p'-p) + ((p'-m)^-1)U(p'-m)
396  trackParam.setTrackChi2(trackParam.getTrackChi2() + addChi2Track(0, 0));
397 
398  return true;
399 }

401 //_________________________________________________________________________________________________
402 bool TrackFitter::runSmoother(const TrackParamMFT& previousParam, TrackParamMFT& param)
403 {
406 
407  // get variables
408  const TMatrixD& extrapParameters = previousParam.getExtrapParameters(); // X(k+1 k)
409  const TMatrixD& filteredParameters = param.getParameters(); // X(k k)
410  const TMatrixD& previousSmoothParameters = previousParam.getSmoothParameters(); // X(k+1 n)
411  const TMatrixD& propagator = previousParam.getPropagator(); // F(k)
412  const TMatrixD& extrapCovariances = previousParam.getExtrapCovariances(); // C(k+1 k)
413  const TMatrixD& filteredCovariances = param.getCovariances(); // C(k k)
414  const TMatrixD& previousSmoothCovariances = previousParam.getSmoothCovariances(); // C(k+1 n)
415 
416  // compute smoother gain: A(k) = C(kk) * F(k)^t * (C(k+1 k))^-1
417  TMatrixD extrapWeight(extrapCovariances);
418  if (extrapWeight.Determinant() != 0) {
419  extrapWeight.Invert(); // (C(k+1 k))^-1
420  } else {
421  LOG(INFO) << "Smoother ERROR: Determinant = 0";
422  return false;
423  }
424  TMatrixD smootherGain(filteredCovariances, TMatrixD::kMultTranspose, propagator); // C(kk) * F(k)^t
425  smootherGain *= extrapWeight; // C(kk) * F(k)^t * (C(k+1 k))^-1
426 
427  // compute smoothed parameters: X(k n) = X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
428  TMatrixD tmpParam(previousSmoothParameters, TMatrixD::kMinus, extrapParameters); // X(k+1 n) - X(k+1 k)
429  TMatrixD smoothParameters(smootherGain, TMatrixD::kMult, tmpParam); // A(k) * (X(k+1 n) - X(k+1 k))
430  smoothParameters += filteredParameters; // X(k k) + A(k) * (X(k+1 n) - X(k+1 k))
431  param.setSmoothParameters(smoothParameters);
432 
433  // compute smoothed covariances: C(k n) = C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
434  TMatrixD tmpCov(previousSmoothCovariances, TMatrixD::kMinus, extrapCovariances); // C(k+1 n) - C(k+1 k)
435  TMatrixD tmpCov2(tmpCov, TMatrixD::kMultTranspose, smootherGain); // (C(k+1 n) - C(k+1 k)) * (A(k))^t
436  TMatrixD smoothCovariances(smootherGain, TMatrixD::kMult, tmpCov2); // A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
437  smoothCovariances += filteredCovariances; // C(k k) + A(k) * (C(k+1 n) - C(k+1 k)) * (A(k))^t
438  param.setSmoothCovariances(smoothCovariances);
439 
440  // compute smoothed residual: r(k n) = cluster - X(k n)
441  const o2::itsmft::Cluster* cluster = param.getClusterPtr();
442  TMatrixD smoothResidual(2, 1);
443  smoothResidual.Zero();
444  smoothResidual(0, 0) = cluster->getX() - smoothParameters(0, 0);
445  smoothResidual(1, 0) = cluster->getY() - smoothParameters(1, 0);
446 
447  // compute weight of smoothed residual: W(k n) = (clusterCov - C(k n))^-1
448  TMatrixD smoothResidualWeight(2, 2);
449  smoothResidualWeight(0, 0) = cluster->getSigmaZ2() - smoothCovariances(0, 0); // cluster->getEx2() - smoothCovariances(0, 0);
450  smoothResidualWeight(0, 1) = -smoothCovariances(0, 2);
451  smoothResidualWeight(1, 0) = -smoothCovariances(2, 0);
452  smoothResidualWeight(1, 1) = cluster->getSigmaY2() - smoothCovariances(2, 2); // cluster->getEy2() - smoothCovariances(2, 2);
453  if (smoothResidualWeight.Determinant() != 0) {
454  smoothResidualWeight.Invert();
455  } else {
456  LOG(INFO) << "Smoother ERROR: Determinant = 0";
457  return false;
458  }
459 
460  // compute local chi2 = (r(k n))^t * W(k n) * r(k n)
461  TMatrixD tmpChi2(smoothResidual, TMatrixD::kTransposeMult, smoothResidualWeight); // (r(k n))^t * W(k n)
462  TMatrixD localChi2(tmpChi2, TMatrixD::kMult, smoothResidual); // (r(k n))^t * W(k n) * r(k n)
463  param.setLocalChi2(localChi2(0, 0));
464  return true;
465 }

468 Double_t invQPtFromParabola(const FitterTrackMFT& track, Double_t bFieldZ, Double_t& chi2)
469 {
470  //rotate track to stabilize quadratic fitting
471  auto deltax = track.rbegin()->getX() - track.first().getX();
472  auto deltay = track.rbegin()->getY() - track.first().getY();
473  auto x_m = (track.rbegin()->getX() + track.first().getX()) / 2;
474  auto y_m = (track.rbegin()->getY() + track.first().getY()) / 2;
475  auto theta = -TMath::ATan2(deltay, deltax);
476  auto costheta = TMath::Cos(theta), sintheta = TMath::Sin(theta);
477 
478  bool verbose = false;
479  if (verbose) {
480  std::cout << "First and last cluster X,Y => " << track.first().getX() << " , " << track.first().getY() << " / " << track.rbegin()->getX() << " , " << track.rbegin()->getY() << std::endl;
481  std::cout << " Angle to rotate: " << theta << " ( " << theta * TMath::RadToDeg() << " deg ) " << std::endl;
482  }
483 
484  auto nPoints = track.getNClusters();
485  Double_t* x = new Double_t[nPoints];
486  Double_t* y = new Double_t[nPoints];
487  int n = 0;
488  for (auto trackparam = track.begin(); trackparam != track.end(); trackparam++) {
489  auto x_0 = trackparam->getClusterPtr()->getX() - x_m;
490  auto y_0 = trackparam->getClusterPtr()->getY() - y_m;
491  x[n] = x_0 * costheta - y_0 * sintheta;
492  y[n] = x_0 * sintheta + y_0 * costheta;
493  //std::cout << " adding rotated point to fit at z = " << trackparam->getClusterPtr()->getZ() << " (" << x[n] << "," << y[n] << ") "<< std::endl;
494  n++;
495  }
496 
497  Double_t q0, q1, q2;

515 Double_t QuadraticRegression(Int_t nVal, Double_t* xVal, Double_t* yVal, Double_t& p0, Double_t& p1, Double_t& p2)
516 {
520 
521  TMatrixD y(nVal, 1);
522  TMatrixD x(nVal, 3);
523  TMatrixD xtrans(3, nVal);
524 
525  for (int i = 0; i < nVal; i++) {
526  y(i, 0) = yVal[i];
527  x(i, 0) = 1.;
528  x(i, 1) = xVal[i];
529  x(i, 2) = xVal[i] * xVal[i];
530  xtrans(0, i) = 1.;
531  xtrans(1, i) = xVal[i];
532  xtrans(2, i) = xVal[i] * xVal[i];
533  }
534  TMatrixD tmp(xtrans, TMatrixD::kMult, x);
535  tmp.Invert();
536 
537  TMatrixD tmp2(xtrans, TMatrixD::kMult, y);
538  TMatrixD b(tmp, TMatrixD::kMult, tmp2);
539 
540  p0 = b(0, 0);
541  p1 = b(1, 0);
542  p2 = b(2, 0);
543 
544  // chi2 = (y-xb)^t . W . (y-xb)
545  TMatrixD tmp3(x, TMatrixD::kMult, b);
546  TMatrixD tmp4(y, TMatrixD::kMinus, tmp3);
547  TMatrixD chi2(tmp4, TMatrixD::kTransposeMult, tmp4);
548 
549  return chi2(0, 0);
550 }
