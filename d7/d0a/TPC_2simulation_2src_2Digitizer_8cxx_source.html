47 void Digitizer::process(const std::vector<o2::tpc::HitGroup>& hits,
48  const int eventID, const int sourceID)
49 {
50  const static Mapper& mapper = Mapper::instance();
51  auto& detParam = ParameterDetector::Instance();
52  auto& eleParam = ParameterElectronics::Instance();
53  auto& gemParam = ParameterGEM::Instance();
54 
55  static GEMAmplification& gemAmplification = GEMAmplification::instance();
56  gemAmplification.updateParameters();
57  static ElectronTransport& electronTransport = ElectronTransport::instance();
58  electronTransport.updateParameters();
59  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
60  sampaProcessing.updateParameters();
61 
62  const int nShapedPoints = eleParam.NShapedPoints;
63  const auto amplificationMode = gemParam.AmplMode;
64  static std::vector<float> signalArray;
65  signalArray.resize(nShapedPoints);
66 
68  mDigitContainer.reserve(sampaProcessing.getTimeBinFromTime(mEventTime));
69 
71  float maxEleTime = (int(mDigitContainer.size()) - nShapedPoints) * eleParam.ZbinWidth;
72 
73  for (auto& hitGroup : hits) {
74  const int MCTrackID = hitGroup.GetTrackID();
75  for (size_t hitindex = 0; hitindex < hitGroup.getSize(); ++hitindex) {
76  const auto& eh = hitGroup.getHit(hitindex);
77 
78  GlobalPosition3D posEle(eh.GetX(), eh.GetY(), eh.GetZ());
79 
80  // Distort the electron position in case space-charge distortions are used
81  if (mUseSCDistortions) {
82  mSpaceCharge->distortElectron(posEle);
83  }
84 
87  if (electronTransport.isCompletelyOutOfSectorCoarseElectronDrift(posEle, mSector)) {
88  continue;
89  }
90 
92  const int nPrimaryElectrons = static_cast<int>(eh.GetEnergyLoss());
93  const float hitTime = eh.GetTime() * 0.001;
94  float driftTime = 0.f;
95 
97 
99  for (int iEle = 0; iEle < nPrimaryElectrons; ++iEle) {
100 
102  const GlobalPosition3D posEleDiff = electronTransport.getElectronDrift(posEle, driftTime);
103  const float eleTime = driftTime + hitTime;
104  if (eleTime > maxEleTime) {
105  LOG(WARNING) << "Skipping electron with driftTime " << driftTime << " from hit at time " << hitTime;
106  continue;
107  }
108  const float absoluteTime = eleTime + mEventTime;
109 
111  if (electronTransport.isElectronAttachment(driftTime)) {
112  continue;
113  }
114 
116  if (std::abs(posEleDiff.Z()) > detParam.TPClength) {
117  continue;
118  }
119 
121  if (mapper.isOutOfSector(posEleDiff, mSector)) {
122  continue;
123  }
124 
126  const DigitPos digiPadPos = mapper.findDigitPosFromGlobalPosition(posEleDiff, mSector);
127  if (!digiPadPos.isValid()) {
128  continue;
129  }
130 
132  if (digiPadPos.getCRU().sector() != mSector) {
133  continue;
134  }
135 
137  const int nElectronsGEM = gemAmplification.getStackAmplification(digiPadPos.getCRU(), digiPadPos.getPadPos(), amplificationMode);
138  if (nElectronsGEM == 0) {
139  continue;
140  }
141 
142  const GlobalPadNumber globalPad = mapper.globalPadNumber(digiPadPos.getGlobalPadPos());
143  const float ADCsignal = sampaProcessing.getADCvalue(static_cast<float>(nElectronsGEM));
144  const MCCompLabel label(MCTrackID, eventID, sourceID, false);
145  sampaProcessing.getShapedSignal(ADCsignal, absoluteTime, signalArray);
146  for (float i = 0; i < nShapedPoints; ++i) {
147  const float time = absoluteTime + i * eleParam.ZbinWidth;
148  mDigitContainer.addDigit(label, digiPadPos.getCRU(), sampaProcessing.getTimeBinFromTime(time), globalPad,
149  signalArray[i]);
150  }
152  }
154  }
155  }
156 }
