240 int createPipes(int maxFd, int* pipes)
241 {
242  auto p = pipe(pipes);
243  maxFd = maxFd > pipes[0] ? maxFd : pipes[0];
244  maxFd = maxFd > pipes[1] ? maxFd : pipes[1];
245 
246  if (p == -1) {
247  std::cerr << "Unable to create PIPE: ";
248  switch (errno) {
249  case EFAULT:
250  assert(false && "EFAULT while reading from pipe");
251  break;
252  case EMFILE:
253  std::cerr << "Too many active descriptors";
254  break;
255  case ENFILE:
256  std::cerr << "System file table is full";
257  break;
258  default:
259  std::cerr << "Unknown PIPE" << std::endl;
260  };
261  // Kill immediately both the parent and all the children
262  kill(-1 * getpid(), SIGKILL);
263  }
264  return maxFd;
265 }

324 void spawnDevice(std::string const& forwardedStdin,
325  DeviceSpec const& spec,
326  std::map<int, size_t>& socket2DeviceInfo,
327  DeviceControl& control,
328  DeviceExecution& execution,
329  std::vector<DeviceInfo>& deviceInfos,
330  int& maxFd, fd_set& childFdset)
331 {
332  int childstdin[2];
333  int childstdout[2];
334  int childstderr[2];
335 
336  maxFd = createPipes(maxFd, childstdin);
337  maxFd = createPipes(maxFd, childstdout);
338  maxFd = createPipes(maxFd, childstderr);
339 
340  // If we have a framework id, it means we have already been respawned
341  // and that we are in a child. If not, we need to fork and re-exec, adding
342  // the framework-id as one of the options.
343  pid_t id = 0;
344  id = fork();
345  // We are the child: prepare options and reexec.
346  if (id == 0) {
347  // We allow being debugged and do not terminate on SIGTRAP
348  signal(SIGTRAP, SIG_IGN);
349 
350  // This is the child.
351  // For stdout / stderr, we close the read part of the pipe, the
352  // old descriptor, and then replace it with the write part of the pipe.
353  // For stdin, we close the write part of the pipe, the old descriptor,
354  // and then we replace it with the read part of the pipe.
355  close(childstdin[1]);
356  close(childstdout[0]);
357  close(childstderr[0]);
358  close(STDIN_FILENO);
359  close(STDOUT_FILENO);
360  close(STDERR_FILENO);
361  dup2(childstdin[0], STDIN_FILENO);
362  dup2(childstdout[1], STDOUT_FILENO);
363  dup2(childstderr[1], STDERR_FILENO);
364  execvp(execution.args[0], execution.args.data());
365  }
366 
367  // This is the parent. We close the write end of
368  // the child pipe and and keep track of the fd so
369  // that we can later select on it.
370  struct sigaction sa_handle_int;
371  sa_handle_int.sa_handler = handle_sigint;
372  sigemptyset(&sa_handle_int.sa_mask);
373  sa_handle_int.sa_flags = SA_RESTART;
374  if (sigaction(SIGINT, &sa_handle_int, nullptr) == -1) {
375  perror("Unable to install signal handler");
376  exit(1);
377  }
378 
379  LOG(INFO) << "Starting " << spec.id << " on pid " << id;
380  DeviceInfo info;
381  info.pid = id;
382  info.active = true;
383  info.readyToQuit = false;
384  info.historySize = 1000;
385  info.historyPos = 0;
386  info.maxLogLevel = LogParsingHelpers::LogLevel::Debug;
387  info.dataRelayerViewIndex = Metric2DViewIndex{"data_relayer", 0, 0, {}};
388  info.variablesViewIndex = Metric2DViewIndex{"matcher_variables", 0, 0, {}};
389  info.queriesViewIndex = Metric2DViewIndex{"data_queries", 0, 0, {}};
390 
391  socket2DeviceInfo.insert(std::make_pair(childstdout[0], deviceInfos.size()));
392  socket2DeviceInfo.insert(std::make_pair(childstderr[0], deviceInfos.size()));
393  deviceInfos.emplace_back(info);
394  // Let's add also metrics information for the given device
395  gDeviceMetricsInfos.emplace_back(DeviceMetricsInfo{});
396 
397  close(childstdin[0]);
398  close(childstdout[1]);
399  close(childstderr[1]);
400  size_t result = write(childstdin[1], forwardedStdin.data(), forwardedStdin.size());
401  if (result != forwardedStdin.size()) {
402  LOG(ERROR) << "Unable to pass configuration to children";
403  }
404  close(childstdin[1]); // Not allowing further communication...
405 
406  // Setting them to non-blocking to avoid haing the driver hang when
407  // reading from child.
408  int resultCode = fcntl(childstdout[0], F_SETFL, O_NONBLOCK);
409  if (resultCode == -1) {
410  LOGP(ERROR, "Error while setting the socket to non-blocking: {}", strerror(errno));
411  }
412  resultCode = fcntl(childstderr[0], F_SETFL, O_NONBLOCK);
413  if (resultCode == -1) {
414  LOGP(ERROR, "Error while setting the socket to non-blocking: {}", strerror(errno));
415  }
416  FD_SET(childstdout[0], &childFdset);
417  FD_SET(childstderr[0], &childFdset);
418 }

434 void processChildrenOutput(DriverInfo& driverInfo, DeviceInfos& infos, DeviceSpecs const& specs,
435  DeviceControls& controls, std::vector<DeviceMetricsInfo>& metricsInfos)
436 {
437  // Wait for children to say something. When they do
438  // print it.
439  fd_set fdset;
440  timeval timeout;
441  timeout.tv_sec = 0;
442  timeout.tv_usec = 16666; // This should be enough to allow 60 HZ redrawing.
443  memcpy(&fdset, &driverInfo.childFdset, sizeof(fd_set));
444  int numFd = select(driverInfo.maxFd, &fdset, nullptr, nullptr, &timeout);
445  if (numFd == 0) {
446  return;
447  }
448  for (int si = 0; si < driverInfo.maxFd; ++si) {
449  if (FD_ISSET(si, &fdset)) {
450  assert(driverInfo.socket2DeviceInfo.find(si) != driverInfo.socket2DeviceInfo.end());
451  auto& info = infos[driverInfo.socket2DeviceInfo[si]];
452 
453  bool fdActive = getChildData(si, info);
454  // If the pipe was closed due to the process exiting, we
455  // can avoid the select.
456  if (!fdActive) {
457  info.active = false;
458  close(si);
459  FD_CLR(si, &driverInfo.childFdset);
460  }
461  --numFd;
462  }
463  // FIXME: no need to check after numFd gets to 0.
464  }
465  // Display part. All you need to display should actually be in
466  // `infos`.
467  // TODO: split at \n
468  // TODO: update this only once per 1/60 of a second or
469  // things like this.
470  // TODO: have multiple display modes
471  // TODO: graphical view of the processing?
472  assert(infos.size() == controls.size());
473  std::smatch match;
474  ParsedMetricMatch metricMatch;
475  ParsedConfigMatch configMatch;
476  const std::string delimiter("\n");
477  bool hasNewMetric = false;
478  for (size_t di = 0, de = infos.size(); di < de; ++di) {
479  DeviceInfo& info = infos[di];
480  DeviceControl& control = controls[di];
481  DeviceMetricsInfo& metrics = metricsInfos[di];
482  assert(specs.size() == infos.size());
483  DeviceSpec const& spec = specs[di];
484 
485  if (info.unprinted.empty()) {
486  continue;
487  }
488 
489  O2_SIGNPOST_START(DriverStatus::ID, DriverStatus::BYTES_PROCESSED, info.pid, 0, 0);
490 
491  std::string_view s = info.unprinted;
492  size_t pos = 0;
493  info.history.resize(info.historySize);
494  info.historyLevel.resize(info.historySize);
495 
496  auto updateMetricsViews =
497  Metric2DViewIndex::getUpdater({&info.dataRelayerViewIndex,
498  &info.variablesViewIndex,
499  &info.queriesViewIndex});
500 
501  auto newMetricCallback = [&updateMetricsViews, &driverInfo, &metricsInfos, &hasNewMetric](std::string const& name, MetricInfo const& metric, int value, size_t metricIndex) {
502  updateMetricsViews(name, metric, value, metricIndex);
503  hasNewMetric = true;
504  };
505 
506  auto doToMatchingPid = [&infos](int pid, auto lambda) {
507  for (auto& deviceInfo : infos) {
508  if (deviceInfo.pid == pid) {
509  lambda(deviceInfo);
510  break;
511  }
512  }
513  };
514 
515  while ((pos = s.find(delimiter)) != std::string::npos) {
516  std::string token{s.substr(0, pos)};
517  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
518 
519  // Check if the token is a metric from SimpleMetricsService
520  // if yes, we do not print it out and simply store it to be displayed
521  // in the GUI.
522  // Then we check if it is part of our Poor man control system
523  // if yes, we execute the associated command.
524  if (DeviceMetricsHelper::parseMetric(token, metricMatch)) {
525  // We use this callback to cache which metrics are needed to provide a
526  // the DataRelayer view.
527  DeviceMetricsHelper::processMetric(metricMatch, metrics, newMetricCallback);
528  } else if (logLevel == LogParsingHelpers::LogLevel::Info && parseControl(token, match)) {
529  auto command = match[1];
530  auto arg = match[2];
531  LOGP(debug, "Found control command {} from pid {} with argument {}.", command, info.pid, arg);
532  if (command == "QUIT" && arg == "ALL") {
533  for (auto& deviceInfo : infos) {
534  deviceInfo.readyToQuit = true;
535  }
536  } else if (command == "QUIT" && arg == "ME") {
537  doToMatchingPid(info.pid, [](DeviceInfo& info) { info.readyToQuit = true; });
538  } else if (command == "NOTIFY_STREAMING_STATE" && arg == "IDLE") {
539  // FIXME: this should really be a policy...
540  doToMatchingPid(info.pid, [](DeviceInfo& info) { info.readyToQuit = true; info.streamingState = StreamingState::Idle; });
541  } else if (command == "NOTIFY_STREAMING_STATE" && arg == "STREAMING") {
542  // FIXME: this should really be a policy...
543  doToMatchingPid(info.pid, [](DeviceInfo& info) { info.streamingState = StreamingState::Streaming; });
544  } else if (command == "NOTIFY_STREAMING_STATE" && arg == "EOS") {
545  // FIXME: this should really be a policy...
546  doToMatchingPid(info.pid, [](DeviceInfo& info) { info.streamingState = StreamingState::EndOfStreaming; });
547  }
548  } else if (logLevel == LogParsingHelpers::LogLevel::Info && DeviceConfigHelper::parseConfig(token, configMatch)) {
549  DeviceConfigHelper::processConfig(configMatch, info);
550  } else if (!control.quiet && (token.find(control.logFilter) != std::string::npos) &&
551  logLevel >= control.logLevel) {
552  assert(info.historyPos >= 0);
553  assert(info.historyPos < info.history.size());
554  info.history[info.historyPos] = token;
555  info.historyLevel[info.historyPos] = logLevel;
556  info.historyPos = (info.historyPos + 1) % info.history.size();
557  std::cout << "[" << info.pid << ":" << spec.name << "]: " << token << std::endl;
558  }
559  // We keep track of the maximum log error a
560  // device has seen.
561  if (logLevel > info.maxLogLevel && logLevel > LogParsingHelpers::LogLevel::Info &&
562  logLevel != LogParsingHelpers::LogLevel::Unknown) {
563  info.maxLogLevel = logLevel;
564  }
565  if (logLevel == LogParsingHelpers::LogLevel::Error) {
566  info.lastError = token;
567  }
568  s.remove_prefix(pos + delimiter.length());
569  }
570  size_t oldSize = info.unprinted.size();
571  info.unprinted = s;
572  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_PROCESSED, oldSize - info.unprinted.size(), 0, 0);
573  }
574  if (hasNewMetric) {
575  hasNewMetric = false;
576  updateMetricsNames(driverInfo, metricsInfos);
577  }
578  // FIXME: for the gui to work correctly I would actually need to
579  // run the loop more often and update whenever enough time has
580  // passed.
581 }

786 // This is the handler for the parent inner loop.
787 int runStateMachine(DataProcessorSpecs const& workflow,
788  WorkflowInfo const& workflowInfo,
789  DataProcessorInfos const& previousDataProcessorInfos,
790  DriverControl& driverControl,
791  DriverInfo& driverInfo,
792  std::vector<DeviceMetricsInfo>& metricsInfos,
793  std::string frameworkId)
794 {
795  DeviceSpecs deviceSpecs;
796  DeviceInfos infos;
797  DeviceControls controls;
798  DeviceExecutions deviceExecutions;
799  DataProcessorInfos dataProcessorInfos = previousDataProcessorInfos;
800 
801  std::vector<ComputingResource> resources;
802 
803  if (driverInfo.resources != "") {
804  resources = ComputingResourceHelpers::parseResources(driverInfo.resources);
805  } else {
806  resources = {ComputingResourceHelpers::getLocalhostResource()};
807  }
808 
809  auto resourceManager = std::make_unique<SimpleResourceManager>(resources);
810 
811  void* window = nullptr;
812  decltype(gui::getGUIDebugger(infos, deviceSpecs, dataProcessorInfos, metricsInfos, driverInfo, controls, driverControl)) debugGUICallback;
813 
814  // An empty frameworkId means this is the driver, so we initialise the GUI
815  if (driverInfo.batch == false && frameworkId.empty()) {
816  window = initGUI("O2 Framework debug GUI");
817  }
818  if (driverInfo.batch == false && window == nullptr && frameworkId.empty()) {
819  LOG(WARN) << "Could not create GUI. Switching to batch mode. Do you have GLFW on your system?";
820  driverInfo.batch = true;
821  }
822  bool guiQuitRequested = false;
823 
824  auto frameLast = std::chrono::high_resolution_clock::now();
825  auto inputProcessingLast = frameLast;
826  // FIXME: I should really have some way of exiting the
827  // parent..
828  DriverState current;
829  DriverState previous;
830  while (true) {
831  // If control forced some transition on us, we push it to the queue.
832  if (driverControl.forcedTransitions.empty() == false) {
833  for (auto transition : driverControl.forcedTransitions) {
834  driverInfo.states.push_back(transition);
835  }
836  driverControl.forcedTransitions.resize(0);
837  }
838  // In case a timeout was requested, we check if we are running
839  // for more than the timeout duration and exit in case that's the case.
840  {
841  auto currentTime = std::chrono::steady_clock::now();
842  std::chrono::duration<double> diff = currentTime - driverInfo.startTime;
843  if ((graceful_exit == false) && (driverInfo.timeout > 0) && (diff.count() > driverInfo.timeout)) {
844  LOG(INFO) << "Timout ellapsed. Requesting to quit.";
845  graceful_exit = true;
846  }
847  }
848  // Move to exit loop if sigint was sent we execute this only once.
849  if (graceful_exit == true && driverInfo.sigintRequested == false) {
850  driverInfo.sigintRequested = true;
851  driverInfo.states.resize(0);
852  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
853  driverInfo.states.push_back(DriverState::GUI);
854  }
855  // If one of the children dies and sigint was not requested
856  // we should decide what to do.
857  if (sigchld_requested == true && driverInfo.sigchldRequested == false) {
858  driverInfo.sigchldRequested = true;
859  pruneGUI(driverInfo.states);
860  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
861  driverInfo.states.push_back(DriverState::GUI);
862  }
863  if (driverInfo.states.empty() == false) {
864  previous = current;
865  current = driverInfo.states.back();
866  } else {
867  current = DriverState::UNKNOWN;
868  }
869  driverInfo.states.pop_back();
870  switch (current) {
871  case DriverState::INIT:
872  LOG(INFO) << "Initialising O2 Data Processing Layer";
873 
874  // Install signal handler for quitting children.
875  driverInfo.sa_handle_child.sa_handler = &handle_sigchld;
876  sigemptyset(&driverInfo.sa_handle_child.sa_mask);
877  driverInfo.sa_handle_child.sa_flags = SA_RESTART | SA_NOCLDSTOP;
878  if (sigaction(SIGCHLD, &driverInfo.sa_handle_child, nullptr) == -1) {
879  perror(nullptr);
880  exit(1);
881  }
882  FD_ZERO(&(driverInfo.childFdset));
883 
886  cleanupSHM(driverInfo.uniqueWorkflowId);
891  driverInfo.states.push_back(DriverState::RUNNING);
892  driverInfo.states.push_back(DriverState::GUI);
893  // driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
894  LOG(INFO) << "O2 Data Processing Layer initialised. We brake for nobody.";
895 #ifdef NDEBUG
896  LOGF(info, "Optimised build. O2DEBUG / LOG(DEBUG) / LOGF(DEBUG) / assert statement will not be shown.");
897 #endif
898  break;
899  case DriverState::IMPORT_CURRENT_WORKFLOW:
900  // This state is needed to fill the metadata structure
901  // which contains how to run the current workflow
902  dataProcessorInfos = previousDataProcessorInfos;
903  for (auto const& device : deviceSpecs) {
904  auto exists = std::find_if(dataProcessorInfos.begin(),
905  dataProcessorInfos.end(),
906  [id = device.id](DataProcessorInfo const& info) -> bool { return info.name == id; });
907  if (exists != dataProcessorInfos.end()) {
908  continue;
909  }
910  dataProcessorInfos.push_back(
911  DataProcessorInfo{
912  device.id,
913  workflowInfo.executable,
914  workflowInfo.args,
915  workflowInfo.options});
916  }
917  break;
918  case DriverState::MATERIALISE_WORKFLOW:
919  try {
920  DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(workflow,
921  driverInfo.channelPolicies,
922  driverInfo.completionPolicies,
923  driverInfo.dispatchPolicies,
924  deviceSpecs,
925  *resourceManager,
926  driverInfo.uniqueWorkflowId);
927  // This should expand nodes so that we can build a consistent DAG.
928  } catch (std::runtime_error& e) {
929  std::cerr << "Invalid workflow: " << e.what() << std::endl;
930  return 1;
931  } catch (...) {
932  std::cerr << "Unknown error while materialising workflow";
933  return 1;
934  }
935  break;
936  case DriverState::DO_CHILD:
937  // We do not start the process if by default we are stopped.
938  if (driverControl.defaultStopped) {
939  kill(getpid(), SIGSTOP);
940  }
941  for (auto& spec : deviceSpecs) {
942  if (spec.id == frameworkId) {
943  return doChild(driverInfo.argc, driverInfo.argv, spec);
944  }
945  }
946  {
947  std::ostringstream ss;
948  for (auto& processor : workflow) {
949  ss << " - " << processor.name << "\n";
950  }
951  for (auto& spec : deviceSpecs) {
952  ss << " - " << spec.name << "(" << spec.id << ")"
953  << "\n";
954  }
955  LOG(ERROR) << "Unable to find component with id "
956  << frameworkId << ". Available options:\n"
957  << ss.str();
958  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
959  }
960  break;
961  case DriverState::REDEPLOY_GUI:
962  // The callback for the GUI needs to be recalculated every time
963  // the deployed configuration changes, e.g. a new device
964  // has been added to the topology.
965  // We need to recreate the GUI callback every time we reschedule
966  // because getGUIDebugger actually recreates the GUI state.
967  if (window) {
968  debugGUICallback = gui::getGUIDebugger(infos, deviceSpecs, dataProcessorInfos, metricsInfos, driverInfo, controls, driverControl);
969  }
970  break;
971  case DriverState::SCHEDULE: {
972  // FIXME: for the moment modifying the topology means we rebuild completely
973  // all the devices and we restart them. This is also what DDS does at
974  // a larger scale. In principle one could try to do a delta and only
975  // restart the data processors which need to be restarted.
976  LOG(INFO) << "Redeployment of configuration asked.";
977  controls.resize(deviceSpecs.size());
978  deviceExecutions.resize(deviceSpecs.size());
979 
980  DeviceSpecHelpers::reworkShmSegmentSize(dataProcessorInfos);
981  DeviceSpecHelpers::prepareArguments(driverControl.defaultQuiet,
982  driverControl.defaultStopped,
983  dataProcessorInfos,
984  deviceSpecs,
985  deviceExecutions, controls,
986  driverInfo.uniqueWorkflowId);
987 
988  std::ostringstream forwardedStdin;
989  WorkflowSerializationHelpers::dump(forwardedStdin, workflow, dataProcessorInfos);
990  for (size_t di = 0; di < deviceSpecs.size(); ++di) {
991  if (deviceSpecs[di].resource.hostname != driverInfo.deployHostname) {
992  spawnRemoteDevice(forwardedStdin.str(),
993  deviceSpecs[di], driverInfo.socket2DeviceInfo, controls[di], deviceExecutions[di], infos,
994  driverInfo.maxFd, driverInfo.childFdset);
995  } else {
996  spawnDevice(forwardedStdin.str(),
997  deviceSpecs[di], driverInfo.socket2DeviceInfo, controls[di], deviceExecutions[di], infos,
998  driverInfo.maxFd, driverInfo.childFdset);
999  }
1000  }
1001  driverInfo.maxFd += 1;
1002  assert(infos.empty() == false);
1003  LOG(INFO) << "Redeployment of configuration done.";
1004  } break;
1005  case DriverState::RUNNING:
1006  // Calculate what we should do next and eventually
1007  // show the GUI
1008  if (guiQuitRequested ||
1009  (driverInfo.terminationPolicy == TerminationPolicy::QUIT && (checkIfCanExit(infos) == true))) {
1010  // Something requested to quit. This can be a user
1011  // interaction with the GUI or (if --completion-policy=quit)
1012  // it could mean that the workflow does not have anything else to do.
1013  // Let's update the GUI one more time and then EXIT.
1014  LOG(INFO) << "Quitting";
1015  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1016  driverInfo.states.push_back(DriverState::GUI);
1017  } else if (infos.size() != deviceSpecs.size()) {
1018  // If the number of deviceSpecs is different from
1019  // the DeviceInfos it means the speicification
1020  // does not match what is running, so we need to do
1021  // further scheduling.
1022  driverInfo.states.push_back(DriverState::RUNNING);
1023  driverInfo.states.push_back(DriverState::GUI);
1024  driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
1025  driverInfo.states.push_back(DriverState::GUI);
1026  driverInfo.states.push_back(DriverState::SCHEDULE);
1027  driverInfo.states.push_back(DriverState::GUI);
1028  } else if (deviceSpecs.size() == 0) {
1029  LOG(INFO) << "No device resulting from the workflow. Quitting.";
1030  // If there are no deviceSpecs, we exit.
1031  driverInfo.states.push_back(DriverState::EXIT);
1032  } else {
1033  driverInfo.states.push_back(DriverState::RUNNING);
1034  driverInfo.states.push_back(DriverState::GUI);
1035  }
1036  {
1037  usleep(1000); // We wait for 1 millisecond between one processing
1038  // and the other.
1039  auto inputProcessingStart = std::chrono::high_resolution_clock::now();
1040  auto inputProcessingLatency = inputProcessingStart - inputProcessingLast;
1041  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
1042  auto inputProcessingEnd = std::chrono::high_resolution_clock::now();
1043  driverInfo.inputProcessingCost = std::chrono::duration_cast<std::chrono::milliseconds>(inputProcessingEnd - inputProcessingStart).count();
1044  driverInfo.inputProcessingLatency = std::chrono::duration_cast<std::chrono::milliseconds>(inputProcessingLatency).count();
1045  inputProcessingLast = inputProcessingStart;
1046  }
1047  break;
1048  case DriverState::GUI:
1049  if (window) {
1050  auto frameStart = std::chrono::high_resolution_clock::now();
1051  auto frameLatency = frameStart - frameLast;
1052  // We want to render at ~60 frames per second, so latency needs to be ~16ms
1053  if (std::chrono::duration_cast<std::chrono::milliseconds>(frameLatency).count() > 20) {
1054  guiQuitRequested = (pollGUI(window, debugGUICallback) == false);
1055  auto frameEnd = std::chrono::high_resolution_clock::now();
1056  driverInfo.frameCost = std::chrono::duration_cast<std::chrono::milliseconds>(frameEnd - frameStart).count();
1057  driverInfo.frameLatency = std::chrono::duration_cast<std::chrono::milliseconds>(frameLatency).count();
1058  frameLast = frameStart;
1059  }
1060  }
1061  break;
1062  case DriverState::QUIT_REQUESTED:
1063  LOG(INFO) << "QUIT_REQUESTED";
1064  guiQuitRequested = true;
1065  // We send SIGCONT to make sure stopped children are resumed
1066  killChildren(infos, SIGCONT);
1067  killChildren(infos, SIGTERM);
1068  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
1069  driverInfo.states.push_back(DriverState::GUI);
1070  break;
1071  case DriverState::HANDLE_CHILDREN:
1072  // I allow queueing of more sigchld only when
1073  // I process the previous call
1074  if (forceful_exit == true) {
1075  LOG(INFO) << "Forceful exit requested.";
1076  killChildren(infos, SIGCONT);
1077  killChildren(infos, SIGKILL);
1078  }
1079  sigchld_requested = false;
1080  driverInfo.sigchldRequested = false;
1081  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
1082  processSigChild(infos);
1083  if (areAllChildrenGone(infos) == true &&
1084  (guiQuitRequested || (checkIfCanExit(infos) == true) || graceful_exit)) {
1085  // We move to the exit, regardless of where we were
1086  driverInfo.states.resize(0);
1087  driverInfo.states.push_back(DriverState::EXIT);
1088  driverInfo.states.push_back(DriverState::GUI);
1089  } else if (areAllChildrenGone(infos) == false &&
1090  (guiQuitRequested || checkIfCanExit(infos) == true || graceful_exit)) {
1091  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
1092  driverInfo.states.push_back(DriverState::GUI);
1093  } else {
1094  driverInfo.states.push_back(DriverState::GUI);
1095  }
1096  break;
1097  case DriverState::EXIT: {
1098  // This is a clean exit. Before we do so, if required,
1099  // we dump the configuration of all the devices so that
1100  // we can reuse it
1101  boost::property_tree::ptree finalConfig;
1102  assert(infos.size() == deviceSpecs.size());
1103  for (size_t di = 0; di < infos.size(); ++di) {
1104  auto info = infos[di];
1105  auto spec = deviceSpecs[di];
1106  PropertyTreeHelpers::merge(finalConfig, info.currentConfig, spec.name);
1107  }
1108  LOG(INFO) << "Dumping used configuration in dpl-config.json";
1109  boost::property_tree::write_json("dpl-config.json", finalConfig);
1110  cleanupSHM(driverInfo.uniqueWorkflowId);
1111  return calculateExitCode(infos);
1112  }
1113  case DriverState::PERFORM_CALLBACKS:
1114  for (auto& callback : driverControl.callbacks) {
1115  callback(workflow, deviceSpecs, deviceExecutions, dataProcessorInfos);
1116  }
1117  driverControl.callbacks.clear();
1118  driverInfo.states.push_back(DriverState::GUI);
1119  break;
1120  default:
1121  LOG(ERROR) << "Driver transitioned in an unknown state("
1122  << "current: " << (int)current
1123  << ", previous: " << (int)previous
1124  << "). Shutting down.";
1125  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1126  }
1127  }
1128 }
