90 TGeoVolume* createRPC(geoparams::RPCtype type, int iChamber)
91 {
93 
94  auto sname = getRPCVolumeName(type, iChamber);
95  auto name = sname.c_str();
96 
97  auto rpc = new TGeoVolumeAssembly(name);
98 
99  // get the dimensions from MIDBase/Constants
100  double halfLength = (type == geoparams::RPCtype::Short) ? geoparams::RPCShortHalfLength : geoparams::RPCHalfLength;
101  halfLength *= geoparams::ChamberScaleFactors[iChamber];
102  double halfHeight = geoparams::getRPCHalfHeight(iChamber);
103 
105 
107  // trigger gas
108  auto gas = new TGeoVolume(Form("Gas %s", name),
109  new TGeoBBox(Form("%sGasBox", name), halfLength, halfHeight, kGasHalfThickness),
110  assertMedium(Medium::Gas));
111 
112  // resisitive electrod plate
113  auto electrod = new TGeoVolume(Form("Electrod %s", name),
114  new TGeoBBox(Form("%sElecBox", name), halfLength, halfHeight, kElectrodHalfThickness),
115  assertMedium(Medium::Bakelite));
116 
118  auto insu = new TGeoVolume(Form("Insulator %s", name),
119  new TGeoBBox(Form("%sInsuBox", name), halfLength, halfHeight, kInsulatorHalfThickness),
120  assertMedium(Medium::Mylar));
121 
123  // cooper foil
124  auto copper = new TGeoVolume(Form("Copper %s", name),
125  new TGeoBBox(Form("%sCopperBox", name), halfLength, halfHeight, kCopperHalfThickness),
126  assertMedium(Medium::Copper));
127 
128  // mylar foil
129  auto mylar = new TGeoVolume(Form("Mylar %s", name),
130  new TGeoBBox(Form("%sMylarBox", name), halfLength, halfHeight, kMylarHalfThickness),
131  assertMedium(Medium::Mylar));
132 
133  // styrofoam plane
134  auto styro = new TGeoVolume(Form("Styrofoam %s", name),
135  new TGeoBBox(Form("%sStyroBox", name), halfLength, halfHeight, kStyrofoamHalfThickness),
136  assertMedium(Medium::Styrofoam));
137 
139  // aluminium foil
140  auto alu = new TGeoVolume(Form("Aluminium %s", name),
141  new TGeoBBox(Form("%sAluBox", name), halfLength, halfHeight, kAluminiumHalfThickness),
142  assertMedium(Medium::Aluminium));
143 
144  // nomex
145  auto nomex = new TGeoVolume(Form("Nomex %s", name),
146  new TGeoBBox(Form("%sNomexBox", name), halfLength, halfHeight, kNomexHalfThickness),
147  assertMedium(Medium::Nomex));
148 
149  // change the volume shape if we are creating a "cut" RPC
150  if (type == geoparams::RPCtype::TopCut || type == geoparams::RPCtype::BottomCut) {
151  // dimensions of the cut
152  double cutHalfLength = geoparams::getLocalBoardWidth(iChamber) / 2.;
153  double cutHalfHeight = geoparams::getLocalBoardHeight(iChamber) / 2.;
154 
155  bool isTopCut = (type == geoparams::RPCtype::TopCut);
156  const char* cutName = Form("%sCut%s", (isTopCut) ? "top" : "bottom", name);
157 
158  // position of the cut w.r.t the center of the RPC
159  auto cutPos = new TGeoTranslation(Form("%sPos", cutName), cutHalfLength - halfLength, (isTopCut) ? halfHeight - cutHalfHeight : cutHalfHeight - halfHeight, 0.);
160  cutPos->RegisterYourself();
161 
162  // for each volume, create a box and change the volume shape by extracting the cut shape
163  new TGeoBBox(Form("%sGasCut", name), cutHalfLength, cutHalfHeight, 2 * kGasHalfThickness);
164  gas->SetShape(new TGeoCompositeShape(Form("%sGasShape", name), Form("%sGasBox-%sGasCut:%sPos", name, name, cutName)));
165 
166  new TGeoBBox(Form("%sElecCut", name), cutHalfLength, cutHalfHeight, 2 * kElectrodHalfThickness);
167  electrod->SetShape(new TGeoCompositeShape(Form("%sElecShape", name), Form("%sElecBox-%sElecCut:%sPos", name, name, cutName)));
168 
169  new TGeoBBox(Form("%sInsuCut", name), cutHalfLength, cutHalfHeight, 2 * kInsulatorHalfThickness);
170  insu->SetShape(new TGeoCompositeShape(Form("%sInsuShape", name), Form("%sInsuBox-%sInsuCut:%sPos", name, name, cutName)));
171 
172  new TGeoBBox(Form("%sCopperCut", name), cutHalfLength, cutHalfHeight, 2 * kCopperHalfThickness);
173  copper->SetShape(new TGeoCompositeShape(Form("%sCopperShape", name), Form("%sCopperBox-%sCopperCut:%sPos", name, name, cutName)));
174 
175  new TGeoBBox(Form("%sMylarCut", name), cutHalfLength, cutHalfHeight, 2 * kMylarHalfThickness);
176  mylar->SetShape(new TGeoCompositeShape(Form("%sMylarShape", name), Form("%sMylarBox-%sMylarCut:%sPos", name, name, cutName)));
177 
178  new TGeoBBox(Form("%sStyroCut", name), cutHalfLength, cutHalfHeight, 2 * kStyrofoamHalfThickness);
179  styro->SetShape(new TGeoCompositeShape(Form("%sStyroShape", name), Form("%sStyroBox-%sStyroCut:%sPos", name, name, cutName)));
180 
181  new TGeoBBox(Form("%sAluCut", name), cutHalfLength, cutHalfHeight, 2 * kAluminiumHalfThickness);
182  alu->SetShape(new TGeoCompositeShape(Form("%sAluShape", name), Form("%sAluBox-%sAluCut:%sPos", name, name, cutName)));
183 
184  new TGeoBBox(Form("%sNomexCut", name), cutHalfLength, cutHalfHeight, 2 * kNomexHalfThickness);
185  nomex->SetShape(new TGeoCompositeShape(Form("%sNomexShape", name), Form("%sNomexBox-%sNomexCut:%sPos", name, name, cutName)));
186  }
187 
189  double halfThickness = kGasHalfThickness;
190  rpc->AddNode(gas, 1);
191  double z = halfThickness; // increment this value when adding a new layer
192 
193  halfThickness = kElectrodHalfThickness;
194  z += halfThickness;
195  rpc->AddNode(electrod, 1, new TGeoTranslation(0., 0., z));
196  rpc->AddNode(electrod, 2, new TGeoTranslation(0., 0., -z));
197  z += halfThickness;
198 
199  halfThickness = kInsulatorHalfThickness;
200  z += halfThickness;
201  rpc->AddNode(insu, 1, new TGeoTranslation(0., 0., z));
202  rpc->AddNode(insu, 2, new TGeoTranslation(0., 0., -z));
203  z += halfThickness;
204 
205  halfThickness = kCopperHalfThickness;
206  z += halfThickness;
207  rpc->AddNode(copper, 1, new TGeoTranslation(0., 0., z));
208  rpc->AddNode(copper, 2, new TGeoTranslation(0., 0., -z));
209  z += halfThickness;
210 
211  halfThickness = kMylarHalfThickness;
212  z += halfThickness;
213  rpc->AddNode(mylar, 1, new TGeoTranslation(0., 0., z));
214  rpc->AddNode(mylar, 2, new TGeoTranslation(0., 0., -z));
215  z += halfThickness;
216 
217  halfThickness = kStyrofoamHalfThickness;
218  z += halfThickness;
219  rpc->AddNode(styro, 1, new TGeoTranslation(0., 0., z));
220  rpc->AddNode(styro, 2, new TGeoTranslation(0., 0., -z));
221  z += halfThickness;
222 
223  halfThickness = kMylarHalfThickness;
224  z += halfThickness;
225  rpc->AddNode(mylar, 3, new TGeoTranslation(0., 0., z));
226  rpc->AddNode(mylar, 4, new TGeoTranslation(0., 0., -z));
227  z += halfThickness;
228 
229  halfThickness = kCopperHalfThickness;
230  z += halfThickness;
231  rpc->AddNode(copper, 3, new TGeoTranslation(0., 0., z));
232  rpc->AddNode(copper, 4, new TGeoTranslation(0., 0., -z));
233  z += halfThickness;
234 
235  halfThickness = kAluminiumHalfThickness;
236  z += halfThickness;
237  rpc->AddNode(alu, 1, new TGeoTranslation(0., 0., z));
238  rpc->AddNode(alu, 2, new TGeoTranslation(0., 0., -z));
239  z += halfThickness;
240 
241  halfThickness = kNomexHalfThickness;
242  z += halfThickness;
243  rpc->AddNode(nomex, 1, new TGeoTranslation(0., 0., z));
244  rpc->AddNode(nomex, 2, new TGeoTranslation(0., 0., -z));
245  z += halfThickness;
246

248  z += halfThickness;
249  rpc->AddNode(alu, 3, new TGeoTranslation(0., 0., z));
250  rpc->AddNode(alu, 4, new TGeoTranslation(0., 0., -z));
251 
252  return rpc;

255 TGeoMatrix* getTransformation(const ROOT::Math::Transform3D& matrix)
256 {
258  double xx, xy, xz, dx, yx, yy, yz, dy, zx, zy, zz, dz;
259  matrix.GetComponents(xx, xy, xz, dx, yx, yy, yz, dy, zx, zy, zz, dz);
260  double vect[3] = {dx, dy, dz};
261  double rotMatrix[9] = {xx, xy, xz, yx, yy, yz, zx, zy, zz};
262  TGeoHMatrix* geoMatrix = new TGeoHMatrix("Transformation");
263  geoMatrix->SetTranslation(vect);
264  geoMatrix->SetRotation(rotMatrix);
265  return geoMatrix;
266 }

268 TGeoVolume* createChamber(int iChamber)
269 {
271 
272  auto chamber = new TGeoVolumeAssembly(geoparams::getChamberVolumeName(iChamber).c_str());
273 
274  double scale = geoparams::ChamberScaleFactors[iChamber];
275 
276  // create the service volumes
277  auto vertSupp = createVerticalSupport(iChamber);
278  auto horiSupp = createHorizontalSupport(iChamber);
279 
280  // create the 4 types of RPC
281  auto longRPC = createRPC(geoparams::RPCtype::Long, iChamber);
282  auto bottomCutRPC = createRPC(geoparams::RPCtype::BottomCut, iChamber);
283  auto topCutRPC = createRPC(geoparams::RPCtype::TopCut, iChamber);
284  auto shortRPC = createRPC(geoparams::RPCtype::Short, iChamber);
285 
286  // for node counting
287  int iHoriSuppNode = 0, iVertSuppNode = 0;
288 
289  // place the volumes on both side of the chamber
290  for (int iside = 0; iside < 2; iside++) {
291 
292  bool isRight = (iside == 0);
293  double xSign = (isRight) ? 1. : -1.;
294 
295  // place 4 vertical supports per side
296  for (int i = 0; i < 4; i++) {
297  chamber->AddNode(vertSupp, iVertSuppNode++, new TGeoTranslation(xSign * kVerticalSupportXPos[i] * scale, 0., 0.));
298  }
299 
300  // place the RPCs
301  for (int iRPC = 0; iRPC < detparams::NRPCLines; iRPC++) {
302 
303  double x = xSign * geoparams::getRPCCenterPosX(iChamber, iRPC);
304  double zSign = (iRPC % 2 == 0) ? 1. : -1.;
305 
306  if (!isRight) {
307  zSign *= -1.;
308  }
309  double z = zSign * geoparams::RPCZShift;
310  double y = 2 * geoparams::getRPCHalfHeight(iChamber) * (iRPC - 4) / (1 - (z / geoparams::DefaultChamberZ[0]));
311 
312  // ID convention (from bottom to top of the chamber) : long, long, long, cut, short, cut, long, long, long
313  TGeoVolume* rpc = nullptr;
314  switch (iRPC) {
315  case 4: // short
316  rpc = shortRPC;
317  break;
318  case 5: // cut (bottom)
319  rpc = bottomCutRPC;
320  break;
321  case 3: // cut (top)
322  rpc = topCutRPC;
323  break;
324  default: // long
325  rpc = longRPC;
326  break;
327  }
328 
329  int deId = detparams::getDEId(isRight, iChamber, iRPC);
330  chamber->AddNode(rpc, deId, getTransformation(getDefaultRPCTransform(isRight, iChamber, iRPC)));
331 
332  // place 3 horizontal supports behind the RPC (and the vertical rods)
333  x = xSign * kHorizontalSupportPos[0] * scale;
334  z = -zSign * kHorizontalSupportPos[2];
335  for (int i = 0; i < 3; i++) {
336  chamber->AddNode(horiSupp, iHoriSuppNode++, new TGeoTranslation(x, y + (i - 1) * kHorizontalSupportPos[1] * scale, z));
337  }
338 
339  } // end of the loop over the number of RPC lines
340 
341  } // end of the side loop
342 
343  return chamber;
344 }
