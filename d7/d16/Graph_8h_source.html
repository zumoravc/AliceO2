99 template <typename T>
100 void Graph<T>::computeEdges(std::function<bool(const T& v1, const T& v2)> linkFunction)
101 {
102  mLinkFunction = linkFunction;
103  int tot_nedges = 0;
104  const size_t size = {mVertices->size()};
105  if (!mIsMultiThread) {
106  for (size_t iVertex{0}; iVertex < size; ++iVertex) {
107  findVertexEdges(mEdges[iVertex], (*mVertices)[iVertex], iVertex, size);
108  tot_nedges += static_cast<int>(mEdges[iVertex].size());
109  }
110  } else {
111  mNThreads = std::min(static_cast<const size_t>(std::thread::hardware_concurrency()), mNThreads);
112  mExecutors.resize(mNThreads);
113  const size_t stride{static_cast<size_t>(std::ceil(mVertices->size() / static_cast<size_t>(mExecutors.size())))};
114  for (size_t iExecutor{0}; iExecutor < mExecutors.size(); ++iExecutor) {
115  // We cannot pass a template function to std::thread(), using lambda instead
116  mExecutors[iExecutor] = std::thread(
117  [iExecutor, stride, this](const auto& linkFunction) {
118  for (size_t iVertex1{iExecutor * stride}; iVertex1 < stride * (iExecutor + 1) && iVertex1 < mVertices->size(); ++iVertex1) {
119  for (size_t iVertex2{0}; iVertex2 < mVertices->size(); ++iVertex2) {
120  if (iVertex1 != iVertex2 && linkFunction((*mVertices)[iVertex1], (*mVertices)[iVertex2])) {
121  mEdges[iVertex1].emplace_back(iVertex2);
122  }
123  }
124  }
125  },
126  mLinkFunction);
127  }
128  }
129  for (auto&& thread : mExecutors) {
130  thread.join();
131  }
132 }

143 template <typename T>
144 std::vector<unsigned char> Graph<T>::getCluster(const int vertexId)
145 {
146  // This method uses a BFS algorithm to return all the graph
147  // vertex ids belonging to a graph
148  std::vector<int> indices;
149  std::vector<unsigned char> visited(mVertices->size(), false);
150 
151  if (!mIsMultiThread) {
152  std::queue<int> idQueue;
153  idQueue.emplace(vertexId);
154  visited[vertexId] = true;
155 
156  // Consume the queue
157  while (!idQueue.empty()) {
158  const int id = idQueue.front();
159  idQueue.pop();
160  for (Edge edge : mEdges[id]) {
161  if (!visited[edge]) {
162  idQueue.emplace(edge);
163  indices.emplace_back(edge);
164  visited[edge] = true;
165  }
166  }
167  }
168  } else {
169  const size_t stride{static_cast<size_t>(std::ceil(static_cast<float>(this->mVertices->size()) / static_cast<size_t>(this->mExecutors.size())))};
170  std::vector<unsigned char> frontier(mVertices->size(), false);
171  std::vector<unsigned char> flags(mVertices->size(), false);
172 
173  frontier[vertexId] = true;
174  int counter{0};
175  while (std::any_of(frontier.begin(), frontier.end(), [](const char t) { return t; })) {
176  flags.resize(mVertices->size(), false);
177  Barrier barrier(mExecutors.size());
178  for (size_t iExecutor{0}; iExecutor < this->mExecutors.size(); ++iExecutor) {
179  mExecutors[iExecutor] = std::thread(
180  [&stride, &frontier, &visited, &barrier, &flags, this](const int executorId) {
181  for (size_t iVertex{executorId * stride}; iVertex < stride * (executorId + 1) && iVertex < this->mVertices->size(); ++iVertex) {
182  if (frontier[iVertex]) {
183  flags[iVertex] = true;
184  frontier[iVertex] = false;
185  visited[iVertex] = true;
186  }
187  }
188  barrier.Wait();
189  for (size_t iVertex{executorId * stride}; iVertex < stride * (executorId + 1) && iVertex < this->mVertices->size(); ++iVertex) {
190  if (flags[iVertex]) {
191  for (auto& edge : mEdges[iVertex]) {
192  if (!visited[edge]) {
193  frontier[edge] = true;
194  }
195  }
196  }
197  }
198  },
199  iExecutor);
200  }
201  for (auto&& thread : mExecutors) {
202  thread.join();
203  }
204  }
205  }
206  return visited;
207 }
