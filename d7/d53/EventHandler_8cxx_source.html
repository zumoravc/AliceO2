52 template <class CellInputType>
53 const o2::InteractionRecord& EventHandler<CellInputType>::getInteractionRecordForEvent(int eventID) const
54 {
55  const InteractionRecord *irClusters(nullptr), *irCells(nullptr);
56  if (mTriggerRecordsClusters.size()) {
57  if (eventID >= mTriggerRecordsClusters.size())
58  throw RangeException(eventID, mTriggerRecordsClusters.size());
59  irClusters = &(mTriggerRecordsClusters[eventID].getBCData());
60  }
61  if (mTriggerRecordsCells.size()) {
62  if (eventID >= mTriggerRecordsCells.size())
63  throw RangeException(eventID, mTriggerRecordsCells.size());
64  irCells = &(mTriggerRecordsCellIndices[eventID].getBCData());
65  }
66  if (irClusters && irCells) {
67  if (compareInteractionRecords(*irClusters, *irCells))
68  return *irClusters;
69  else
70  throw InteractionRecordInvalidException(*irClusters, *irCells);
71  } else if (irClusters) {
72  return *irClusters;
73  } else if (irCells) {
74  return *irCells;
75  }
76  throw NotInitializedException();
77 }

126 template <class CellInputType>
127 EventData<CellInputType> EventHandler<CellInputType>::buildEvent(int eventID) const
128 {
129  EventData<CellInputType> outputEvent;
130  outputEvent.mInteractionRecord = getInteractionRecordForEvent(eventID);
131  if (hasClusters())
132  outputEvent.mClusters = getClustersForEvent(eventID);
133  if (hasClusterIndices())
134  outputEvent.mCellIndices = getClusterCellIndicesForEvent(eventID);
135  if (hasCells())
136  outputEvent.mCells = getCellsForEvent(eventID);
137 
138  return outputEvent;
139 }
