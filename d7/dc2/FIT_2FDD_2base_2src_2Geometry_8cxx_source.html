38 void Geometry::buildGeometry()
39 {
40 
41  // Top volume of FIT FDD detector
42  TGeoVolumeAssembly* vFDD = new TGeoVolumeAssembly("FDD");
43 
44  LOG(INFO) << "Geometry::buildGeometry()::Volume name = " << vFDD->GetName();
45 
46  //Rotations used
47  //TGeoRotation* Rx90m = new TGeoRotation("Rx90m", 0., -90., 0.);
48  //TGeoRotation* Rx90 = new TGeoRotation("Rx90", 0., 90., 0.);
49  TGeoRotation* Rx180 = new TGeoRotation("Rx180", 0., 180., 0.); // 4 | 1
50  TGeoRotation* Rz180 = new TGeoRotation("Rz180", 180., 0., 0.); // ---------------> x
51  TGeoRotation* Ry180 = new TGeoRotation("Ry180", 180., 180., 0.); // 3 | 2
52  //TGeoRotation* Ry90m = new TGeoRotation("Ry90m", 90., -90., -90.);
53  //TGeoRotation* Ry90 = new TGeoRotation("Ry90", 90., 90., -90.);
54  //TGeoRotation* Rz90 = new TGeoRotation("Rz90", 90., 0., 0.);
55 
56  const Double_t kZbegFrontBar = 1949.1; // Begining of Front Bar
57 
58  //Medium for FDA
59  TGeoMedium* medFDASci = gGeoManager->GetMedium("FDD_BC420");
60  //Medium for FDC
61  TGeoMedium* medFDCSci = gGeoManager->GetMedium("FDD_BC420");
62 
63  // FDA Scintillator Pad
64  const Double_t kFDACellSideY = 21.6;
65  const Double_t kFDACellSideX = 18.1;
66  // FDC Scintillator Pad
67  const Double_t kFDCCellSideY = 21.6;
68  const Double_t kFDCCellSideX = 18.1;
69  // WLS bar : 0.40 cm ( 4.0 mm )
70  // Wrapping : 0.20 cm ( 2.0 mm )
71  // Aluminnized Mylar: 0.01 cm ( 0.1 mm )
72  // Fishing line : 0.04 cm ( 0.4 mm )
73  // total shift on X : 0.65 cm
74  // total shift on Y : 0.21 cm
75  const Double_t kShiftX = 0.54;
76  const Double_t kShiftY = 0.10;
77  const Double_t kFDACelldz = 2.54;
78  const Double_t kFDCCelldz = 2.54;
79  const Double_t kFDABeamPipeR = 6.20; // Radius of beam pipe hole for FDD_A (Diameter 12.4 cm)
80  const Double_t kFDCBeamPipeR = 3.70; // Radius of beam pipe hole for FDD_C (Diameter 7.4 cm)
81  const Int_t kColorFDA = kGreen;
82  const Int_t kColorFDC = kGreen;
83  Double_t X = kShiftX + kFDACellSideX * 0.5;
84  Double_t Y = kShiftY + kFDACellSideY * 0.5;
85  Double_t WLS_dx = 0.4;
86  Double_t WLS_dz = 2.5;
87  Double_t WLS_SideA_Long_dy = 24.20; // 24.2;
88  Double_t WLS_SideC_Long_dy = 24.20; // 24.2;
89  Double_t WLS_SideA_Short_dy = 18.20; // 18.41;
90  Double_t WLS_SideC_Short_dy = 20.70; // 20.91;
91 
92  // Creating FDA WLS bars
93  TGeoVolume* vFDA_WLS_s = new TGeoVolume("FDAWLSshort", new TGeoBBox("shFDAWLSbarShort", WLS_dx / 2.0, WLS_SideA_Short_dy / 2.0, WLS_dz / 2.0), medFDASci);
94  TGeoVolume* vFDA_WLS_l = new TGeoVolume("FDAWLSlong", new TGeoBBox("shFDAWLSbarLong", WLS_dx / 2.0, WLS_SideA_Long_dy / 2.0, WLS_dz / 2.0), medFDASci);
95  vFDA_WLS_l->SetLineColor(kRed);
96  vFDA_WLS_s->SetLineColor(kRed);
97 
98  // Creating FDC WLS bars
99  TGeoVolume* vFDC_WLS_s = new TGeoVolume("FDCWLSshort", new TGeoBBox("shFDCWLSbarShort", WLS_dx / 2.0, WLS_SideC_Short_dy / 2.0, WLS_dz / 2.0), medFDCSci);
100  TGeoVolume* vFDC_WLS_l = new TGeoVolume("FDCWLSlong", new TGeoBBox("shFDCWLSbarLong", WLS_dx / 2.0, WLS_SideC_Long_dy / 2.0, WLS_dz / 2.0), medFDCSci);
101  vFDC_WLS_l->SetLineColor(kRed);
102  vFDC_WLS_s->SetLineColor(kRed);
103 
104  // Make FDA scintillator pad
105  new TGeoBBox("shFDAbox", kFDACellSideX / 2.0, kFDACellSideY / 2.0, kFDACelldz / 2.0);
106  new TGeoTube("shFDAHole", 0., kFDABeamPipeR, kFDACelldz);
107  (new TGeoTranslation("trFDAbox", X, Y, 0.))->RegisterYourself();
108 
109  TGeoVolume* vFDApad = new TGeoVolume("FDApad", new TGeoCompositeShape("shFDApad", "shFDAbox:trFDAbox-shFDAHole"), medFDASci);
110  vFDApad->SetLineColor(kColorFDA);
111 
112  TGeoVolume* secFDA = new TGeoVolumeAssembly("FDAsec");
113  // Add PAD
114  Double_t fX_FDA_WLS_s = 0.1 + WLS_dx / 2.0;
115  Double_t fX_FDA_WLS_l = kShiftX + WLS_dx / 2.0 + kFDACellSideX + 0.04;
116  secFDA->AddNode(vFDApad, 1);
117  secFDA->AddNode(vFDA_WLS_s, 1, new TGeoTranslation(fX_FDA_WLS_s, kFDABeamPipeR + WLS_SideA_Short_dy / 2.0, 0.0));
118  secFDA->AddNode(vFDA_WLS_l, 1, new TGeoTranslation(fX_FDA_WLS_l, kShiftY + WLS_SideA_Long_dy / 2.0, 0.0));
119 
121  TGeoVolume* vFDAarray = new TGeoVolumeAssembly("FDA"); // ^ y
122  vFDAarray->AddNode(secFDA, 1); // |
123  vFDAarray->AddNode(secFDA, 2, Ry180); // 2 | 1
124  vFDAarray->AddNode(secFDA, 3, Rz180); // ---------------> x
125  vFDAarray->AddNode(secFDA, 4, Rx180); // 3 | 4
126 
127  const Float_t kPosFDA = 1696.67; // z-center of assembly (cm)
128 
129  vFDD->AddNode(vFDAarray, 1, new TGeoTranslation(0., 0., kPosFDA - kFDACelldz / 2. - 0.1));
130  vFDD->AddNode(vFDAarray, 2, new TGeoTranslation(0., 0., kPosFDA + kFDACelldz / 2. + 0.1));
131 
133 
134  new TGeoBBox("shFDCbox", kFDCCellSideX / 2.0, kFDCCellSideY / 2.0, kFDCCelldz / 2.0);
135  new TGeoTube("shFDCHole", 0., kFDCBeamPipeR, kFDCCelldz);
136  X = kShiftX + kFDCCellSideX * 0.5;
137  Y = kShiftY + kFDCCellSideY * 0.5;
138  (new TGeoTranslation("trFDCbox", X, Y, 0.))->RegisterYourself();
139  TGeoVolume* vFDCpad = new TGeoVolume("FDCpad", new TGeoCompositeShape("shFDCpad", "shFDCbox:trFDCbox-shFDCHole"), medFDCSci);
140  vFDCpad->SetLineColor(kColorFDC);
141 
143  TGeoVolume* voFDC = new TGeoVolumeAssembly("FDCsec");
144  // Add PAD
145  voFDC->AddNode(vFDCpad, 1);
146  // Add FDC WLS Short bar
147  voFDC->AddNode(vFDC_WLS_s, 1, new TGeoTranslation(0.1 + WLS_dx / 2.0, kFDCBeamPipeR + WLS_SideC_Short_dy / 2.0, 0.0));
148  // Add FDC WLS Long bar
149  voFDC->AddNode(vFDC_WLS_l, 1, new TGeoTranslation(0.04 + WLS_dx / 2.0 + kFDCCellSideX + kShiftX, kShiftY + WLS_SideC_Long_dy / 2.0, 0.0));
150 
152  TGeoVolume* vFDCarray = new TGeoVolumeAssembly("FDC"); // ^ y
153  vFDCarray->AddNode(voFDC, 1); // |
154  vFDCarray->AddNode(voFDC, 2, Ry180); // 2 | 1
155  vFDCarray->AddNode(voFDC, 3, Rz180); // ---------------> x
156  vFDCarray->AddNode(voFDC, 4, Rx180); // 3 | 4
157  // |
158 
159  const Float_t kPosFDC = -kZbegFrontBar - 2. - 3.0 - 0.3; // 3.0 = (5.6 + 0.2 + 0.2)/2.
160  vFDD->AddNode(vFDCarray, 1, new TGeoTranslation(0., 0., kPosFDC - kFDCCelldz / 2. - 0.23));
161  vFDD->AddNode(vFDCarray, 2, new TGeoTranslation(0., 0., kPosFDC + kFDCCelldz / 2. + 0.23));
162 
163  TGeoVolume* vALIC = gGeoManager->GetVolume("cave");
164  if (!vALIC) {
165  LOG(FATAL) << "Could not find the top volume";
166  }
167  vALIC->AddNode(vFDD, 1);
168 }
