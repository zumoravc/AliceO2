63

68  return kERROR;
69  }
70 
71  // Register output container
72  if (mTracksArray) {
73  mgr->RegisterAny("MFTTrack", mTracksArray, kTRUE);
74  }
75

77  if (mTrkLabels) {
78  mgr->RegisterAny("MFTTrackMCTruth", mTrkLabels, kTRUE);

81  LOG(ERROR) << "MFT cluster labels not registered in the FairRootManager. Exiting ...";
82  return kERROR;
83  }
84  }
85 
86  GeometryTGeo* geom = GeometryTGeo::Instance();

89  mTracker.setGeometry(geom);
90  mTracker.setMCTruthContainers(mClsLabels, mTrkLabels);
91  mTracker.setContinuousMode(mContinuousMode);
