29  // Temporary function to check if track passes selection criteria. To be
30  // replaced by framework filters
31  template <typename T>
32  bool IsSelected(T const& track)
33  {
34  if (track.trackType() == o2::aod::track::TrackTypeEnum::GlobalTrack &&
35  track.pt() >= mMinPt && track.pt() < mMaxPt && track.eta() >= mMinEta &&
36  track.eta() < mMaxEta && track.tpcNClsFound() >= mMinNClustersTPC &&
37  track.tpcNClsCrossedRows() >= mMinNCrossedRowsTPC &&
38  track.tpcCrossedRowsOverFindableCls() >=
39  mMinNCrossedRowsOverFindableClustersTPC &&
40  (track.itsNCls() >= mMinNClustersITS) &&
41  (track.itsChi2NCl() < mMaxChi2PerClusterITS) &&
42  (track.tpcChi2NCl() < mMaxChi2PerClusterTPC) &&
43  (mRequireITSRefit && (track.flags() & 0x4)) &&
44  (mRequireTPCRefit && (track.flags() & 0x40)) &&
45  FulfillsITSHitRequirements(track.itsClusterMap()) &&
46  abs(track.dcaXY()) < mMaxDcaXY && abs(track.dcaZ()) < mMaxDcaZ) {
47  return true;
48  } else {
49  return false;
50  }
51  }
