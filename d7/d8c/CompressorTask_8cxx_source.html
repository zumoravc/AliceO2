54 template <typename RAWDataHeader, bool verbose>
55 void CompressorTask<RAWDataHeader, verbose>::run(ProcessingContext& pc)
56 {
57  LOG(DEBUG) << "Compressor run";
58 
60  char bufferOut[1048576];
61  mCompressor.setEncoderBuffer(bufferOut);
62  mCompressor.setEncoderBufferSize(1048576);
63 
64  auto device = pc.services().get<o2::framework::RawDeviceService>().device();
65  auto outputRoutes = pc.services().get<o2::framework::RawDeviceService>().spec().outputs;
66  auto fairMQChannel = outputRoutes.at(0).channel;
67 
69  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
70  if (!iit.isValid())
71  continue;
72 
74  FairMQParts parts;
75 
77  for (auto const& ref : iit) {
78 
79  auto headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
80  auto dataProcessingHeaderIn = DataRefUtils::getHeader<o2::framework::DataProcessingHeader*>(ref);
81  auto payloadIn = ref.payload;
82  auto payloadInSize = headerIn->payloadSize;
83  mCompressor.setDecoderBuffer(payloadIn);
84  mCompressor.setDecoderBufferSize(payloadInSize);
85 
87  mCompressor.run();
88  auto payloadOutSize = mCompressor.getEncoderByteCounter();
89  auto payloadMessage = device->NewMessage(payloadOutSize);
90  std::memcpy(payloadMessage->GetData(), bufferOut, payloadOutSize);
91 
93  auto headerOut = *headerIn;
94  auto dataProcessingHeaderOut = *dataProcessingHeaderIn;
95  headerOut.dataDescription = "CRAWDATA";
96  headerOut.payloadSize = payloadOutSize;
97  o2::header::Stack headerStack{headerOut, dataProcessingHeaderOut};
98  auto headerMessage = device->NewMessage(headerStack.size());
99  std::memcpy(headerMessage->GetData(), headerStack.data(), headerStack.size());
100 
102  parts.AddPart(std::move(headerMessage));
103  parts.AddPart(std::move(payloadMessage));
104  }
105 
107  device->Send(parts, fairMQChannel);
108  }
109 }
