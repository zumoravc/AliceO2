84 void Digitizer::process(const std::vector<Hit> hits, std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& mcContainer)
85 {
86  digits.clear();
87  mDigits.clear();
88  mTrackLabels.clear();
89  mcContainer.clear();
90  //array of MCH hits for a given simulated event
91  for (auto& hit : hits) {
92  int detID = hit.GetDetectorID();
93  int ndigits = processHit(hit, detID, mEventTime);
94  MCCompLabel label(hit.GetTrackID(), mEventID, mSrcID, false);
95  for (int i = 0; i < ndigits; ++i) {
96  int digitIndex = mDigits.size() - ndigits + i;
97  mMCTruthOutputContainer.addElement(digitIndex, label);
98  } //loop over digits to generate MCdigits
99  } //loop over hits
100  fillOutputContainer(digits);
101  provideMC(mcContainer);
102 }

104 int Digitizer::processHit(const Hit& hit, int detID, double event_time)
105 {
106  Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
107 
108  Response& resp = response(isStation1(detID));
109 
110  //convert energy to charge
111  auto charge = resp.etocharge(hit.GetEnergyLoss());
112  auto time = event_time + hit.GetTime();
113 
114  //transformation from global to local
115  auto t = o2::mch::getTransformation(detID, *gGeoManager);
116  Point3D<float> lpos;
117  t.MasterToLocal(pos, lpos);
118 
119  auto anodpos = resp.getAnod(lpos.X());
120  auto fracplane = resp.chargeCorr();
121  auto chargebend = fracplane * charge;
122  auto chargenon = charge / fracplane;
123 
124  //borders of charge gen.
125  auto xMin = anodpos - resp.getQspreadX() * resp.getSigmaIntegration() * 0.5;
126  auto xMax = anodpos + resp.getQspreadX() * resp.getSigmaIntegration() * 0.5;
127  auto yMin = lpos.Y() - resp.getQspreadY() * resp.getSigmaIntegration() * 0.5;
128  auto yMax = lpos.Y() + resp.getQspreadY() * resp.getSigmaIntegration() * 0.5;
129 
130  //get segmentation for detector element
131  auto& seg = segmentation(detID);
132  auto localX = anodpos;
133  auto localY = lpos.Y();
134 
135  //get area for signal induction from segmentation
136  //single pad as check
137  int padidbendcent = 0;
138  int padidnoncent = 0;
139  int ndigits = 0;
140 
141  bool padexists = seg.findPadPairByPosition(localX, localY, padidbendcent, padidnoncent);
142  if (!padexists) {
143  LOG(WARNING) << "Did not find _any_ pad for localX,Y=" << localX << "," << localY << ", for detID: " << detID;
144  return 0;
145  }
146 
147  seg.forEachPadInArea(xMin, yMin, xMax, yMax, [&resp, &digits = this->mDigits, chargebend, chargenon, localX, localY, &seg, &ndigits, detID, time](int padid) {
148  auto dx = seg.padSizeX(padid) * 0.5;
149  auto dy = seg.padSizeY(padid) * 0.5;
150  auto xmin = (localX - seg.padPositionX(padid)) - dx;
151  auto xmax = xmin + 2 * dx;
152  auto ymin = (localY - seg.padPositionY(padid)) - dy;
153  auto ymax = ymin + 2 * dy;
154  auto q = resp.chargePadfraction(xmin, xmax, ymin, ymax);
155  if (resp.aboveThreshold(q)) {
156  if (seg.isBendingPad(padid)) {
157  q *= chargebend;
158  } else {
159  q *= chargenon;
160  }
161  auto signal = (unsigned long)q * resp.getInverseChargeThreshold();
162  if (signal > 0) {
163  Digit::Time dtime;
164  dtime.sampaTime = static_cast<uint16_t>(time) & 0x3FF;
165  digits.emplace_back(detID, padid, signal, dtime);
166  ++ndigits;
167  }
168  }
169  });
170  return ndigits;
171 }

173 void Digitizer::mergeDigits()
174 {
175  std::vector<int> indices(mDigits.size());
176  std::iota(begin(indices), end(indices), 0);
177  std::sort(indices.begin(), indices.end(), [& digits = this->mDigits, this](int a, int b) {
178  return (getGlobalDigit(digits[a].getDetID(), digits[a].getPadID()) < getGlobalDigit(digits[b].getDetID(), digits[b].getPadID()));
179  });
180 
181  auto sortedDigits = [digits = this->mDigits, &indices](int i) {
182  return digits[indices[i]];
183  };
184 
185  auto sortedLabels = [labels = this->mTrackLabels, &indices](int i) {
186  return labels[indices[i]];
187  };
188 
189  auto sizedigits = mDigits.size();
190 
191  mDigits.clear();
192  mDigits.reserve(sizedigits);
193  mMCTruthOutputContainer.clear();
194 
195  int count = 0;
196 
197  int i = 0;
198  while (i < indices.size()) {
199  int j = i + 1;
200  while (j < indices.size() && (getGlobalDigit(sortedDigits(i).getDetID(), sortedDigits(i).getPadID())) == (getGlobalDigit(sortedDigits(j).getDetID(), sortedDigits(j).getPadID())) && (std::fabs(sortedDigits(i).getTime().sampaTime - sortedDigits(j).getTime().sampaTime) < mDeltat)) {
201  j++;
202  }
203  unsigned long adc{0};
204  float padc{0};
205  Response& resp = response(isStation1(sortedDigits(i).getDetID()));
206 
207  for (int k = i; k < j; k++) {
208  adc += sortedDigits(k).getADC();
209  if (k == i) {
210  MCCompLabel label(sortedLabels(i).getTrackID(), sortedLabels(i).getEventID(), sortedLabels(i).getSourceID(), false);
211  mMCTruthOutputContainer.addElement(count, label);
212  } else {
213  if ((sortedLabels(k).getTrackID() != sortedLabels(k - 1).getTrackID()) || (sortedLabels(k).getSourceID() != sortedLabels(k - 1).getSourceID())) {
214  MCCompLabel label(sortedLabels(k).getTrackID(), sortedLabels(k).getEventID(), sortedLabels(k).getSourceID(), false);
215  mMCTruthOutputContainer.addElement(count, label);
216  }
217  }
218  }
219  padc = adc * resp.getChargeThreshold();
220  adc = TMath::Nint(padc);
221  adc = resp.response(adc);
222  mDigits.emplace_back(sortedDigits(i).getDetID(), sortedDigits(i).getPadID(), adc, sortedDigits(i).getTime());
223  i = j;
224  ++count;
225  }
226  mDigits.resize(mDigits.size());
227 }

229 void Digitizer::mergeDigits(std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& mcContainer)
230 {
231  mDigits.clear();
232  mDigits.reserve(digits.size());
233  mTrackLabels.clear();
234 
235  for (int index = 0; index < digits.size(); ++index) {
236  auto digit = digits.at(index);
237  mDigits.emplace_back(digit.getDetID(), digit.getPadID(), digit.getADC(), digit.getTime());
238  }
239  for (int index = 0; index < mcContainer.getNElements(); ++index) {
240  auto label = mcContainer.getElement(index);
241  mTrackLabels.emplace_back(label.getTrackID(), label.getEventID(), label.getSourceID(), false);
242  }
243 
244  mergeDigits();
245  fillOutputContainer(digits);
246  provideMC(mcContainer);
247 }

249 void Digitizer::fillOutputContainer(std::vector<Digit>& digits)
250 {
251  // filling the digit container
252  if (mDigits.empty())
253  return;
254 
255  digits.clear();
256  digits.reserve(mDigits.size());
257 
258  auto itBeg = mDigits.begin();
259  auto iter = itBeg;
260  for (; iter != mDigits.end(); ++iter) {
261  digits.emplace_back(*iter);
262  }
263  mDigits.erase(itBeg, iter);
