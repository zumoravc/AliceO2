42 bool PadResponse::importPRF(std::string file, std::unique_ptr<TGraph2D>& grPRF) const
43 {
44  std::string inputDir;
45  const char* aliceO2env = std::getenv("O2_ROOT");
46  if (aliceO2env)
47  inputDir = aliceO2env;
48  inputDir += "/share/Detectors/TPC/files/";
49 
50  float x, y, normalizedPadResponse;
51  int i = 0;
52  std::ifstream prfFile(inputDir + file, std::ifstream::in);
53  if (!prfFile) {
54  LOG(FATAL) << "tpc::PadResponse - Input file '" << inputDir + file << "' does not exist! No PRF loaded!";
55  return false;
56  }
57  for (std::string line; std::getline(prfFile, line);) {
58  std::istringstream is(line);
59  while (is >> x >> y >> normalizedPadResponse) {
60  grPRF->SetPoint(i++, x, y, normalizedPadResponse);
61  }
62  }
63  return true;
64 }

66 float PadResponse::getPadResponse(GlobalPosition3D posEle, DigitPos digiPadPos) const
67 {
69  const Mapper& mapper = Mapper::instance();

73  const LocalPosition3D padCentreLocal(padCentre.X(), padCentre.Y(), posEle.Z());
74  const GlobalPosition3D padCentrePos = mapper.LocalToGlobal(padCentreLocal, sector);
75 
77 
78  const int gemStack = int(cru.gemStack());
79  const float offsetX = std::abs(posEle.X() - padCentre.X()) * 10.f;
80  const float offsetY = std::abs(posEle.Y() - padCentre.Y()) * 10.f;
81  float normalizedPadResponse = 0;
82  if (gemStack == 0) {
83  normalizedPadResponse = mIROC->Interpolate(offsetX, offsetY);
84  } else if (gemStack == 1 || gemStack == 2) {
85  normalizedPadResponse = mOROC12->Interpolate(offsetX, offsetY);
86  } else {
87  normalizedPadResponse = mOROC3->Interpolate(offsetX, offsetY);
88  }
89  return normalizedPadResponse;
