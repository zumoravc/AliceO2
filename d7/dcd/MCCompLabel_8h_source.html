16 namespace o2

18 // Composed Label to encode MC track id, event it comes from and the source (file)
19 
20 class MCCompLabel
21 {
22  private:
23  static constexpr ULong64_t ul0x1 = 0x1;
24  static constexpr ULong64_t NotSet = 0xffffffffffffffff;
25  static constexpr ULong64_t Noise = 0xfffffffffffffffe;
26  static constexpr ULong64_t Fake = ul0x1 << 63;
27  static constexpr int NReservedBits = 1;
28 
29  ULong64_t mLabel = NotSet;
30 
31  public:
32  // number of bits reserved for MC track ID, DON'T modify this, since the
33  // track ID might be negative
34  static constexpr int nbitsTrackID = 31; // number of bits reserved for MC track ID
35  static constexpr int nbitsEvID = 19; // number of bits reserved for MC event ID
36  static constexpr int nbitsSrcID = 8; // number of bits reserved for MC source ID
37  // the rest of the bits is reserved at the moment
38 
39  // check if the fields are defined consistently
40  static_assert(nbitsTrackID + nbitsEvID + nbitsSrcID <= sizeof(ULong64_t) * 8 - NReservedBits,
41  "Fields cannot be stored in 64 bits");
42 
43  // mask to extract MC track ID
44  static constexpr ULong64_t maskTrackID = (ul0x1 << nbitsTrackID) - 1;
45  // mask to extract MC track ID
46  static constexpr ULong64_t maskEvID = (ul0x1 << nbitsEvID) - 1;
47  // mask to extract MC track ID
48  static constexpr ULong64_t maskSrcID = (ul0x1 << nbitsSrcID) - 1;
49  // mask for all used fields
50  static constexpr ULong64_t maskFull = (ul0x1 << (nbitsTrackID + nbitsEvID + nbitsSrcID)) - 1;
51 
52  MCCompLabel(int trackID, int evID, int srcID, bool fake = false) { set(trackID, evID, srcID, fake); }
53  MCCompLabel(bool noise = false)
54  {
55  if (noise) {
56  mLabel = Noise;
57  } else {
58  mLabel = NotSet;
59  }
60  }
61  ~MCCompLabel() = default;
62 
63  // check if label was assigned
64  bool isSet() const { return mLabel != NotSet; }
65  // check if label was not assigned
66  bool isEmpty() const { return mLabel == NotSet; }
67  // check if label corresponds to real particle
68  bool isNoise() const { return mLabel == Noise; }
69  // check if label was assigned as for correctly identified particle
70  bool isValid() const { return isSet() && !isNoise(); }
71 
72  // check if label was assigned as for incorrectly identified particle or not set or noise
73  bool isFake() const { return mLabel & Fake; }
74  // check if label was assigned as for correctly identified particle
75  bool isCorrect() const { return !isFake(); }
76 
77  // return 1 if the tracks are the same and correctly identified
78  // 0 if the tracks are the same but at least one of them is fake
79  // -1 otherwhise
80  int compare(const MCCompLabel& other) const
81  {
82  if (getEventID() != other.getEventID() || getSourceID() != other.getSourceID()) {
83  return -1;
84  }
85  int tr1 = getTrackID(), tr2 = other.getTrackID();
86  return (tr1 == tr2) ? ((isCorrect() && other.isCorrect()) ? 1 : 0) : -1;
87  }
88 
89  // conversion operator
90  operator ULong64_t() const { return mLabel; }
91  // allow to retrieve bare label
92  ULong64_t getRawValue() const { return mLabel; }
93 
94  // comparison operator, compares only label, not eventual weight or correctness info
95  bool operator==(const MCCompLabel& other) const { return (mLabel & maskFull) == (other.mLabel & maskFull); }
96  // invalidate
97  void unset() { mLabel = NotSet; }
98  void setNoise() { mLabel = Noise; }
99  void setFakeFlag(bool v = true)
100  {
101  if (v) {
102  mLabel |= Fake;
103  } else {
104  mLabel &= ~Fake;
105  }
106  }
107 
108  void set(unsigned int trackID, int evID, int srcID, bool fake)
109  {
111  mLabel = (maskTrackID & static_cast<ULong64_t>(trackID)) |
112  (maskEvID & static_cast<ULong64_t>(evID)) << nbitsTrackID |
113  (maskSrcID & static_cast<ULong64_t>(srcID)) << (nbitsTrackID + nbitsEvID);
114  if (fake) {
115  setFakeFlag();
116  }
117  }
118 
119  int getTrackID() const { return static_cast<int>(mLabel & maskTrackID); }
120  int getTrackIDSigned() const { return isFake() ? -getTrackID() : getTrackID(); }
121  int getEventID() const { return (mLabel >> nbitsTrackID) & maskEvID; }
122  int getSourceID() const { return (mLabel >> (nbitsTrackID + nbitsEvID)) & maskSrcID; }
123  void get(int& trackID, int& evID, int& srcID, bool& fake)
124  {
126  trackID = getTrackID();
127  evID = getEventID();
128  srcID = getSourceID();
129  fake = isFake();
130  }
131 
132  void print() const;
133 
134  static constexpr int maxSourceID() { return maskSrcID; }
135  static constexpr int maxEventID() { return maskEvID; }
136  static constexpr int maxTrackID() { return maskTrackID; }
137  ClassDefNV(MCCompLabel, 1);
138 };
