77 template <typename FORMAT, typename CHARGESUM>
78 void PayloadEncoderImpl<FORMAT, CHARGESUM>::addChannelData(DsElecId dsId, uint8_t chId, const std::vector<SampaCluster>& data)
79 {
80  auto solarId = dsId.solarId();
81  auto gbt = mGBTs.find(solarId);
82  if (gbt == mGBTs.end()) {
83  auto f = mSolar2FeeLink(solarId);
84  if (!f.has_value()) {
85  throw std::invalid_argument(fmt::format("Could not get fee,link for solarId={}\n", solarId));
86  }
87  mGBTs.emplace(solarId, std::make_unique<GBTEncoder<FORMAT, CHARGESUM>>(f->linkId()));
88  gbt = mGBTs.find(solarId);
89  }
90  gbt->second->addChannelData(dsId.elinkGroupId(), dsId.elinkIndexInGroup(), chId, data);
91 }

93 template <typename FORMAT, typename CHARGESUM>
94 void PayloadEncoderImpl<FORMAT, CHARGESUM>::gbts2buffer(uint32_t orbit, uint16_t bunchCrossing)
95 {
96  // append to our own buffer all the words buffers from all our gbts,
97  // prepending each one with a corresponding payload header
98 
99  for (auto& p : mGBTs) {
100  auto& gbt = p.second;
101  std::vector<std::byte> gbtBuffer;
102  gbt->moveToBuffer(gbtBuffer);
103  if (gbtBuffer.empty()) {
104  continue;
105  }
106  assert(gbtBuffer.size() % 4 == 0);
107  DataBlockHeader header{orbit, bunchCrossing, p.first, gbtBuffer.size()};
108  appendDataBlockHeader(mBuffer, header);
109  mBuffer.insert(mBuffer.end(), gbtBuffer.begin(), gbtBuffer.end());
110  }
111 }

113 template <typename FORMAT, typename CHARGESUM>
114 size_t PayloadEncoderImpl<FORMAT, CHARGESUM>::moveToBuffer(std::vector<std::byte>& buffer)
115 {
116  closeHeartbeatFrame(mOrbit, mBunchCrossing);
117  buffer.insert(buffer.end(), mBuffer.begin(), mBuffer.end());
118  auto s = mBuffer.size();
119  mBuffer.clear();
120  return s;
121 }

129 template <typename FORMAT, typename CHARGESUM>
130 void PayloadEncoderImpl<FORMAT, CHARGESUM>::startHeartbeatFrame(uint32_t orbit, uint16_t bunchCrossing)
131 {
132  impl::assertIsInRange("bunchCrossing", bunchCrossing, 0, 0xFFF);
133  // build a buffer with the _previous_ (orbit,bx)
134  if (!mFirstHBFrame) {
135  closeHeartbeatFrame(mOrbit, mBunchCrossing);
136  }
137  mFirstHBFrame = false;
138  // then save the (orbit,bx) for next time
139  mOrbit = orbit;
140  mBunchCrossing = bunchCrossing;
141 }
