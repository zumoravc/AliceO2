51 void TimeframeWriterDevice::Run()
52 {
53  boost::filesystem::path p(mOutFileName);
54  size_t streamedTimeframes = 0;
55  bool needsNewFile = true;
56  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this) && mFileCount < mMaxFiles) {
57  // In case we need to process more than one file,
58  // the filename is split in basename and extension
59  // and we call the files `<basename><count>.<extension>`.
60  if (needsNewFile) {
61  std::string filename = mOutFileName;
62  if (mMaxFiles > 1) {
63  std::string base_path(mOutFileName, 0, mOutFileName.find_last_of("."));
64  std::string extension(mOutFileName, mOutFileName.find_last_of("."));
65  filename = base_path + std::to_string(mFileCount) + extension;
66  }
67  LOG(INFO) << "Opening " << filename << " for output\n";
68  mFile.open(filename.c_str(), std::ofstream::out | std::ofstream::binary);
69  needsNewFile = false;
70  }
71 
72  FairMQParts timeframeParts;
73  if (Receive(timeframeParts, mInChannelName, 0, 100) <= 0)
74  continue;
75 
76  streamTimeframe(mFile, timeframeParts);
77  if ((mFile.tellp() > mMaxFileSize) || (streamedTimeframes++ > mMaxTimeframes)) {
78  mFile.flush();
79  mFile.close();
80  mFileCount++;
81  needsNewFile = true;
82  }
83  }
84 }
