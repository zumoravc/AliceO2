49 void Tracker::clustersToTracks(const ROframe& event, std::ostream& timeBenchmarkOutputStream)
50 {
51  const int verticesNum = event.getPrimaryVerticesNum();
52  mTracks.clear();
53  mTrackLabels.clear();
54 
55  for (int iVertex = 0; iVertex < verticesNum; ++iVertex) {
56 
57  float total{0.f};
58 
59  for (int iteration = 0; iteration < mTrkParams.size(); ++iteration) {
60  mTraits->UpdateTrackingParameters(mTrkParams[iteration]);
62  int pass = iteration + iVertex;
63  std::array<float, 3> pV = {event.getPrimaryVertex(iVertex).x, event.getPrimaryVertex(iVertex).y, event.getPrimaryVertex(iVertex).z};
64  total += evaluateTask(&Tracker::initialisePrimaryVertexContext, "Context initialisation",
65  timeBenchmarkOutputStream, mMemParams[iteration], event.getClusters(), pV, pass);
66  total += evaluateTask(&Tracker::computeTracklets, "Tracklet finding", timeBenchmarkOutputStream);
67  total += evaluateTask(&Tracker::computeCells, "Cell finding", timeBenchmarkOutputStream);
68  total += evaluateTask(&Tracker::findCellsNeighbours, "Neighbour finding", timeBenchmarkOutputStream, iteration);
69  total += evaluateTask(&Tracker::findRoads, "Road finding", timeBenchmarkOutputStream, iteration);
70  total += evaluateTask(&Tracker::findTracks, "Track finding", timeBenchmarkOutputStream, event);
71  }
72 
73  if (constants::DoTimeBenchmarks)
74  timeBenchmarkOutputStream << std::setw(2) << " - "
75  << "Vertex processing completed in: " << total << "ms" << std::endl;
76  }
77  if (event.hasMCinformation()) {
78  computeTracksMClabels(event);
79  } else {
80  rectifyClusterIndices(event);
81  }
82 }

94 void Tracker::findCellsNeighbours(int& iteration)
95 {
96  for (int iLayer{0}; iLayer < constants::its::CellsPerRoad - 1; ++iLayer) {
97 
98  if (mPrimaryVertexContext->getCells()[iLayer + 1].empty() ||
99  mPrimaryVertexContext->getCellsLookupTable()[iLayer].empty()) {
100  continue;
101  }
102 
103  int layerCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer].size())};
104 
105  for (int iCell{0}; iCell < layerCellsNum; ++iCell) {
106 
107  const Cell& currentCell{mPrimaryVertexContext->getCells()[iLayer][iCell]};
108  const int nextLayerTrackletIndex{currentCell.getSecondTrackletIndex()};
109  const int nextLayerFirstCellIndex{mPrimaryVertexContext->getCellsLookupTable()[iLayer][nextLayerTrackletIndex]};
110  if (nextLayerFirstCellIndex != constants::its::UnusedIndex &&
111  mPrimaryVertexContext->getCells()[iLayer + 1][nextLayerFirstCellIndex].getFirstTrackletIndex() ==
112  nextLayerTrackletIndex) {
113 
114  const int nextLayerCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer + 1].size())};
115  mPrimaryVertexContext->getCellsNeighbours()[iLayer].resize(nextLayerCellsNum);
116 
117  for (int iNextLayerCell{nextLayerFirstCellIndex};
118  iNextLayerCell < nextLayerCellsNum &&
119  mPrimaryVertexContext->getCells()[iLayer + 1][iNextLayerCell].getFirstTrackletIndex() ==
120  nextLayerTrackletIndex;
121  ++iNextLayerCell) {
122 
123  Cell& nextCell{mPrimaryVertexContext->getCells()[iLayer + 1][iNextLayerCell]};
124  const float3 currentCellNormalVector{currentCell.getNormalVectorCoordinates()};
125  const float3 nextCellNormalVector{nextCell.getNormalVectorCoordinates()};
126  const float3 normalVectorsDeltaVector{currentCellNormalVector.x - nextCellNormalVector.x,
127  currentCellNormalVector.y - nextCellNormalVector.y,
128  currentCellNormalVector.z - nextCellNormalVector.z};
129 
130  const float deltaNormalVectorsModulus{(normalVectorsDeltaVector.x * normalVectorsDeltaVector.x) +
131  (normalVectorsDeltaVector.y * normalVectorsDeltaVector.y) +
132  (normalVectorsDeltaVector.z * normalVectorsDeltaVector.z)};
133  const float deltaCurvature{std::abs(currentCell.getCurvature() - nextCell.getCurvature())};
134 
135  if (deltaNormalVectorsModulus < mTrkParams[iteration].NeighbourMaxDeltaN[iLayer] &&
136  deltaCurvature < mTrkParams[iteration].NeighbourMaxDeltaCurvature[iLayer]) {
137 
138  mPrimaryVertexContext->getCellsNeighbours()[iLayer][iNextLayerCell].push_back(iCell);
139 
140  const int currentCellLevel{currentCell.getLevel()};
141 
142  if (currentCellLevel >= nextCell.getLevel()) {
143 
144  nextCell.setLevel(currentCellLevel + 1);
145  }
146  }
147  }
148  }
149  }
150  }

153 void Tracker::findRoads(int& iteration)
154 {
155  for (int iLevel{constants::its::CellsPerRoad}; iLevel >= mTrkParams[iteration].CellMinimumLevel(); --iLevel) {
156  CA_DEBUGGER(int nRoads = -mPrimaryVertexContext->getRoads().size());
157  const int minimumLevel{iLevel - 1};
158 
159  for (int iLayer{constants::its::CellsPerRoad - 1}; iLayer >= minimumLevel; --iLayer) {
160 
161  const int levelCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer].size())};
162 
163  for (int iCell{0}; iCell < levelCellsNum; ++iCell) {
164 
165  Cell& currentCell{mPrimaryVertexContext->getCells()[iLayer][iCell]};
166 
167  if (currentCell.getLevel() != iLevel) {
168  continue;
169  }
170 
171  mPrimaryVertexContext->getRoads().emplace_back(iLayer, iCell);
172 
175  if (iLevel == 1) {
176  continue;
177  }
178 
179  const int cellNeighboursNum{static_cast<int>(
180  mPrimaryVertexContext->getCellsNeighbours()[iLayer - 1][iCell].size())};
181  bool isFirstValidNeighbour = true;
182 
183  for (int iNeighbourCell{0}; iNeighbourCell < cellNeighboursNum; ++iNeighbourCell) {
184 
185  const int neighbourCellId = mPrimaryVertexContext->getCellsNeighbours()[iLayer - 1][iCell][iNeighbourCell];
186  const Cell& neighbourCell = mPrimaryVertexContext->getCells()[iLayer - 1][neighbourCellId];
187 
188  if (iLevel - 1 != neighbourCell.getLevel()) {
189  continue;
190  }
191 
192  if (isFirstValidNeighbour) {
193 
194  isFirstValidNeighbour = false;
195 
196  } else {
197 
198  mPrimaryVertexContext->getRoads().emplace_back(iLayer, iCell);
199  }
200 
201  traverseCellsTree(neighbourCellId, iLayer - 1);
202  }
203 
204  // TODO: crosscheck for short track iterations
205  // currentCell.setLevel(0);
206  }
207  }
208 #ifdef CA_DEBUG
209  nRoads += mPrimaryVertexContext->getRoads().size();
210  std::cout << "+++ Roads with " << iLevel + 2 << " clusters: " << nRoads << " / " << mPrimaryVertexContext->getRoads().size() << std::endl;
211 #endif
212  }

215 void Tracker::findTracks(const ROframe& event)
216 {
217  mTracks.reserve(mTracks.capacity() + mPrimaryVertexContext->getRoads().size());
218  std::vector<TrackITSExt> tracks;
219  tracks.reserve(mPrimaryVertexContext->getRoads().size());
220 #ifdef CA_DEBUG
221  std::array<int, 4> roadCounters{0, 0, 0, 0};
222  std::array<int, 4> fitCounters{0, 0, 0, 0};
223  std::array<int, 4> backpropagatedCounters{0, 0, 0, 0};
224  std::array<int, 4> refitCounters{0, 0, 0, 0};
225  std::array<int, 4> nonsharingCounters{0, 0, 0, 0};
226 #endif
227 
228  for (auto& road : mPrimaryVertexContext->getRoads()) {
229  std::array<int, 7> clusters{constants::its::UnusedIndex, constants::its::UnusedIndex, constants::its::UnusedIndex, constants::its::UnusedIndex, constants::its::UnusedIndex, constants::its::UnusedIndex, constants::its::UnusedIndex};
230  int lastCellLevel = constants::its::UnusedIndex;
231  CA_DEBUGGER(int nClusters = 2);
232 
233  for (int iCell{0}; iCell < constants::its::CellsPerRoad; ++iCell) {
234  const int cellIndex = road[iCell];
235  if (cellIndex == constants::its::UnusedIndex) {
236  continue;
237  } else {
238  clusters[iCell] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getFirstClusterIndex();
239  clusters[iCell + 1] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getSecondClusterIndex();
240  clusters[iCell + 2] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getThirdClusterIndex();
241  assert(clusters[iCell] != constants::its::UnusedIndex &&
242  clusters[iCell + 1] != constants::its::UnusedIndex &&
243  clusters[iCell + 2] != constants::its::UnusedIndex);
244  lastCellLevel = iCell;
245  CA_DEBUGGER(nClusters++);
246  }
247  }
248 
249  assert(nClusters >= mTrkParams[0].MinTrackLength);
250  CA_DEBUGGER(roadCounters[nClusters - 4]++);
251 
252  if (lastCellLevel == constants::its::UnusedIndex)
253  continue;
254 
256  for (int iC{0}; iC < clusters.size(); iC++) {
257  if (clusters[iC] != constants::its::UnusedIndex) {
258  clusters[iC] = mPrimaryVertexContext->getClusters()[iC][clusters[iC]].clusterId;
259  }
260  }
262  const auto& cluster1_glo = event.getClustersOnLayer(lastCellLevel + 2).at(clusters[lastCellLevel + 2]);
263  const auto& cluster2_glo = event.getClustersOnLayer(lastCellLevel + 1).at(clusters[lastCellLevel + 1]);
264  const auto& cluster3_glo = event.getClustersOnLayer(lastCellLevel).at(clusters[lastCellLevel]);
265 
266  const auto& cluster3_tf = event.getTrackingFrameInfoOnLayer(lastCellLevel).at(clusters[lastCellLevel]);
267 
269  TrackITSExt temporaryTrack{buildTrackSeed(cluster1_glo, cluster2_glo, cluster3_glo, cluster3_tf)};
270  for (size_t iC = 0; iC < clusters.size(); ++iC) {
271  temporaryTrack.setExternalClusterIndex(iC, clusters[iC], clusters[iC] != constants::its::UnusedIndex);
272  }
273  bool fitSuccess = fitTrack(event, temporaryTrack, constants::its::LayersNumber - 4, -1, -1);
274  if (!fitSuccess)
275  continue;
276  CA_DEBUGGER(fitCounters[nClusters - 4]++);
277  temporaryTrack.resetCovariance();
278  fitSuccess = fitTrack(event, temporaryTrack, 0, constants::its::LayersNumber, 1);
279  if (!fitSuccess)
280  continue;
281  CA_DEBUGGER(backpropagatedCounters[nClusters - 4]++);
282  temporaryTrack.getParamOut() = temporaryTrack;
283  temporaryTrack.resetCovariance();
284  fitSuccess = fitTrack(event, temporaryTrack, constants::its::LayersNumber - 1, -1, -1);
285  if (!fitSuccess)
286  continue;
287  CA_DEBUGGER(refitCounters[nClusters - 4]++);
288  temporaryTrack.setROFrame(mROFrame);
289  tracks.emplace_back(temporaryTrack);
290  assert(nClusters == temporaryTrack.getNumberOfClusters());
291  }
292  //mTraits->refitTracks(event.getTrackingFrameInfo(), tracks);
293 
294  std::sort(tracks.begin(), tracks.end(),
295  [](TrackITSExt& track1, TrackITSExt& track2) { return track1.isBetter(track2, 1.e6f); });
296 
297 #ifdef CA_DEBUG
298  std::array<int, 26> sharingMatrix{0};
299  int prevNclusters = 7;
300  auto cumulativeIndex = [](int ncl) -> int {
301  constexpr int idx[5] = {0, 5, 11, 18, 26};
302  return idx[ncl - 4];
303  };
304  std::array<int, 4> xcheckCounters{0};
305 #endif
306 
307  for (auto& track : tracks) {
308  CA_DEBUGGER(int nClusters = 0);
309  int nShared = 0;
310  for (int iLayer{0}; iLayer < constants::its::LayersNumber; ++iLayer) {
311  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
312  continue;
313  }
314  nShared += int(mPrimaryVertexContext->isClusterUsed(iLayer, track.getClusterIndex(iLayer)));
315  CA_DEBUGGER(nClusters++);
316  }
317 
318 #ifdef CA_DEBUG
319  assert(nClusters == track.getNumberOfClusters());
320  xcheckCounters[nClusters - 4]++;
321  assert(nShared <= nClusters);
322  sharingMatrix[cumulativeIndex(nClusters) + nShared]++;
323 #endif
324 
325  if (nShared > mTrkParams[0].ClusterSharing) {
326  continue;
327  }
328 
329 #ifdef CA_DEBUG
330  nonsharingCounters[nClusters - 4]++;
331  assert(nClusters <= prevNclusters);
332  prevNclusters = nClusters;
333 #endif
334 
335  for (int iLayer{0}; iLayer < constants::its::LayersNumber; ++iLayer) {
336  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
337  continue;
338  }
339  mPrimaryVertexContext->markUsedCluster(iLayer, track.getClusterIndex(iLayer));
340  }
341  mTracks.emplace_back(track);
342  }
343 
344 #ifdef CA_DEBUG
345  std::cout << "+++ Found candidates with 4, 5, 6 and 7 clusters:\t";
346  for (int count : roadCounters)
347  std::cout << count << "\t";
348  std::cout << std::endl;
349 
350  std::cout << "+++ Fitted candidates with 4, 5, 6 and 7 clusters:\t";
351  for (int count : fitCounters)
352  std::cout << count << "\t";
353  std::cout << std::endl;
354 
355  std::cout << "+++ Backprop candidates with 4, 5, 6 and 7 clusters:\t";
356  for (int count : backpropagatedCounters)
357  std::cout << count << "\t";
358  std::cout << std::endl;
359 
360  std::cout << "+++ Refitted candidates with 4, 5, 6 and 7 clusters:\t";
361  for (int count : refitCounters)
362  std::cout << count << "\t";
363  std::cout << std::endl;
364 
365  std::cout << "+++ Cross check counters for 4, 5, 6 and 7 clusters:\t";
366  for (size_t iCount = 0; iCount < refitCounters.size(); ++iCount) {
367  std::cout << xcheckCounters[iCount] << "\t";
368  //assert(refitCounters[iCount] == xcheckCounters[iCount]);
369  }
370  std::cout << std::endl;
371 
372  std::cout << "+++ Nonsharing candidates with 4, 5, 6 and 7 clusters:\t";
373  for (int count : nonsharingCounters)
374  std::cout << count << "\t";
375  std::cout << std::endl;
376 
377  std::cout << "+++ Sharing matrix:\n";
378  for (int iCl = 4; iCl <= 7; ++iCl) {
379  std::cout << "+++ ";
380  for (int iSh = cumulativeIndex(iCl); iSh < cumulativeIndex(iCl + 1); ++iSh) {
381  std::cout << sharingMatrix[iSh] << "\t";
382  }
383  std::cout << std::endl;
384  }
385 #endif
386 }

388 bool Tracker::fitTrack(const ROframe& event, TrackITSExt& track, int start, int end, int step)
389 {
390  track.setChi2(0);
391  for (int iLayer{start}; iLayer != end; iLayer += step) {
392  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
393  continue;
394  }
395  const TrackingFrameInfo& trackingHit = event.getTrackingFrameInfoOnLayer(iLayer).at(track.getClusterIndex(iLayer));
396 
397  if (!track.rotate(trackingHit.alphaTrackingFrame))
398  return false;
399 
400  if (!track.propagateTo(trackingHit.xTrackingFrame, getBz()))
401  return false;
402 
403  track.setChi2(track.getChi2() +
404  track.getPredictedChi2(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame));
405  if (!track.TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame))
406  return false;
407 
408  const float xx0 = (iLayer > 2) ? 0.008f : 0.003f; // Rough layer thickness
409  constexpr float radiationLength = 9.36f; // Radiation length of Si [cm]
410  constexpr float density = 2.33f; // Density of Si [g/cm^3]
411  if (!track.correctForMaterial(xx0, xx0 * radiationLength * density, true))
412  return false;
413  }
414  return true;

421 
422  mPrimaryVertexContext->getRoads().back().addCell(currentLayerId, currentCellId);

454 void Tracker::computeRoadsMClabels(const ROframe& event)
455 {
457  if (!event.hasMCinformation()) {
458  return;
459  }
460 
461  mPrimaryVertexContext->initialiseRoadLabels();
462 
463  int roadsNum{static_cast<int>(mPrimaryVertexContext->getRoads().size())};
464 
465  for (int iRoad{0}; iRoad < roadsNum; ++iRoad) {
466 
467  Road& currentRoad{mPrimaryVertexContext->getRoads()[iRoad]};
468  MCCompLabel maxOccurrencesValue{constants::its::UnusedIndex, constants::its::UnusedIndex,
469  constants::its::UnusedIndex, false};
470  int count{0};
471  bool isFakeRoad{false};
472  bool isFirstRoadCell{true};
473 
474  for (int iCell{0}; iCell < constants::its::CellsPerRoad; ++iCell) {
475  const int currentCellIndex{currentRoad[iCell]};
476 
477  if (currentCellIndex == constants::its::UnusedIndex) {
478  if (isFirstRoadCell) {
479  continue;
480  } else {
481  break;
482  }
483  }
484 
485  const Cell& currentCell{mPrimaryVertexContext->getCells()[iCell][currentCellIndex]};
486 
487  if (isFirstRoadCell) {
488 
489  const int cl0index{mPrimaryVertexContext->getClusters()[iCell][currentCell.getFirstClusterIndex()].clusterId};
490  auto& cl0labs{event.getClusterLabels(iCell, cl0index)};
491  maxOccurrencesValue = cl0labs;
492  count = 1;
493 
494  const int cl1index{mPrimaryVertexContext->getClusters()[iCell + 1][currentCell.getSecondClusterIndex()].clusterId};
495  auto& cl1labs{event.getClusterLabels(iCell + 1, cl1index)};
496  const int secondMonteCarlo{cl1labs.getTrackID()};
497 
498  if (secondMonteCarlo == maxOccurrencesValue) {
499  ++count;
500  } else {
501  maxOccurrencesValue = secondMonteCarlo;
502  count = 1;
503  isFakeRoad = true;
504  }
505 
506  isFirstRoadCell = false;
507  }
508 
509  const int cl2index{mPrimaryVertexContext->getClusters()[iCell + 2][currentCell.getThirdClusterIndex()].clusterId};
510  auto& cl2labs{event.getClusterLabels(iCell + 2, cl2index)};
511  const int currentMonteCarlo = {cl2labs.getTrackID()};
512 
513  if (currentMonteCarlo == maxOccurrencesValue) {
514  ++count;
515  } else {
516  --count;
517  isFakeRoad = true;
518  }
519 
520  if (count == 0) {
521  maxOccurrencesValue = currentMonteCarlo;
522  count = 1;
523  }
524  }
525 
526  mPrimaryVertexContext->setRoadLabel(iRoad, maxOccurrencesValue, isFakeRoad);
527  }

530 void Tracker::computeTracksMClabels(const ROframe& event)
531 {
533  if (!event.hasMCinformation()) {
534  return;
535  }
536 
537  int tracksNum{static_cast<int>(mTracks.size())};
538 
539  for (auto& track : mTracks) {
540 
541  MCCompLabel maxOccurrencesValue{constants::its::UnusedIndex, constants::its::UnusedIndex,
542  constants::its::UnusedIndex, false};
543  int count{0};
544  bool isFakeTrack{false};
545 
546  for (int iCluster = 0; iCluster < TrackITSExt::MaxClusters; ++iCluster) {
547  const int index = track.getClusterIndex(iCluster);
548  if (index == constants::its::UnusedIndex) {
549  continue;
550  }
551 
552  const MCCompLabel& currentLabel = event.getClusterLabels(iCluster, index);
553  if (currentLabel == maxOccurrencesValue) {
554  ++count;
555  } else {
556  if (count != 0) { // only in the first iteration count can be 0 at this point
557  isFakeTrack = true;
558  --count;
559  }
560  if (count == 0) {
561  maxOccurrencesValue = currentLabel;
562  count = 1;
563  }
564  }
565  track.setExternalClusterIndex(iCluster, event.getClusterExternalIndex(iCluster, index));
566  }
567 
568  if (isFakeTrack) {
569  maxOccurrencesValue.setFakeFlag();
570  }
571  mTrackLabels.addElement(mTrackLabels.getIndexedSize(), maxOccurrencesValue);
572  }

592 track::TrackParCov Tracker::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
593  const Cluster& cluster3, const TrackingFrameInfo& tf3)
594 {
595  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
596  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
597  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
598  const float z1 = cluster1.zCoordinate;
599  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
600  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
601  const float z2 = cluster2.zCoordinate;
602  const float x3 = tf3.xTrackingFrame;
603  const float y3 = tf3.positionTrackingFrame[0];
604  const float z3 = tf3.positionTrackingFrame[1];
605 
606  const float crv = math_utils::computeCurvature(x1, y1, x2, y2, x3, y3);
