57 bool MagFieldFast::LoadData(const string inpFName)
58 {
59  // load field from text file
60 
61  std::ifstream in(gSystem->ExpandPathName(inpFName.data()), std::ifstream::in);
62  if (in.fail()) {
63  LOG(FATAL) << "Failed to open file " << inpFName;
64  return false;
65  }
66  std::string line;
67  int valI, component = -1, nParams = 0, header[4] = {-1, -1, -1, -1}; // iR, iZ, iQuadrant, nVal
68  SolParam* curParam = nullptr;
69 
70  while (std::getline(in, line)) {
71  if (line.empty() || line[0] == '#')
72  continue; // empy or comment
73  std::stringstream ss(line);
74  int cnt = 0;
75 
76  if (component < 0) {
77  while (cnt < 4 && (ss >> header[cnt++]))
78  ;
79  if (cnt != 4) {
80  LOG(FATAL) << "Wrong header " << line;
81  return false;
82  }
83  curParam = &mSolPar[header[0]][header[1]][header[2]];
84  } else {
85  while (cnt < header[3] && (ss >> curParam->parBxyz[component][cnt++]))
86  ;
87  if (cnt != header[3]) {
88  LOG(FATAL) << "Wrong data (npar=" << cnt << ") for param " << header[0] << " " << header[1] << " " << header[2]
89  << " " << header[3] << " " << line;
90  return false;
91  }
92  }
93  component++;
94  if (component > 2) {
95  component = -1; // next header expected
96  nParams++;
97  }
98  }
99  //
100  LOG(INFO) << "Loaded " << nParams << " params from " << inpFName;
101  if (nParams != kNSolRRanges * kNSolZRanges * kNQuadrants) {
102  LOG(FATAL) << "Was expecting " << kNSolRRanges * kNSolZRanges * kNQuadrants << " params";
103  }
104  return true;
105 }

109 {

117  bxyz[kY] = CalcPol(par->parBxyz[kY], xyz[kX], xyz[kY], xyz[kZ]) * mFactorSol;
118  bxyz[kZ] = CalcPol(par->parBxyz[kZ], xyz[kX], xyz[kY], xyz[kZ]) * mFactorSol;
119  //
120  return true;
121 }

212 bool MagFieldFast::GetSegment(float x, float y, float z, int& zSeg, int& rSeg, int& quadrant) const
213 {
214  // get segment of point location
215  const float zGridSpaceInv = 1.f / (kSolZMax * 2 / kNSolZRanges);
216  zSeg = -1;
217  if (z < kSolZMax) {
218  if (z > -kSolZMax)
219  zSeg = (z + kSolZMax) * zGridSpaceInv; // solenoid params
220  else { // need to check dipole params
221  return false;
222  }
223  } else
224  return false;
225  // R segment
226  float xx = x * x, yy = y * y, rr = xx + yy;
227  for (rSeg = 0; rSeg < kNSolRRanges; rSeg++)
228  if (rr < kSolR2Max[rSeg])
229  break;
230  if (rSeg == kNSolRRanges)
231  return kFALSE;
232  quadrant = GetQuadrant(x, y);
233  return true;
234 }
