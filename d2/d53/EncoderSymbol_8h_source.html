29 // Encoder symbol description
30 // This (admittedly odd) selection of parameters was chosen to make
31 // RansEncPutSymbol as cheap as possible.
32 template <typename T>
33 
34 struct EncoderSymbol {
35  EncoderSymbol(uint32_t start, uint32_t freq, uint32_t scale_bits)
36  {
37  //TODO(lettrich): a check should be definitely done here.
38  // RansAssert(scale_bits <= 16);
39  assert(start <= (1u << scale_bits));
40  assert(freq <= (1u << scale_bits) - start);
41 
42  // Say M := 1 << scale_bits.
43  //
44  // The original encoder does:
45  // x_new = (x/freq)*M + start + (x%freq)
46  //
47  // The fast encoder does (schematically):
48  // q = mul_hi(x, rcp_freq) >> rcp_shift (division)
49  // r = x - q*freq (remainder)
50  // x_new = q*M + bias + r (new x)
51  // plugging in r into x_new yields:
52  // x_new = bias + x + q*(M - freq)
53  // =: bias + x + q*cmpl_freq (*)
54  //
55  // and we can just precompute cmpl_freq. Now we just need to
56  // set up our parameters such that the original encoder and
57  // the fast encoder agree.
58 
59  this->freq = freq;
60  this->cmpl_freq = static_cast<T>((1 << scale_bits) - freq);
61  if (freq < 2) {
62  // freq=0 symbols are never valid to encode, so it doesn't matter what
63  // we set our values to.
64  //
65  // freq=1 is tricky, since the reciprocal of 1 is 1; unfortunately,
66  // our fixed-point reciprocal approximation can only multiply by values
67  // smaller than 1.
68  //
69  // So we use the "next best thing": rcp_freq=0xffffffff, rcp_shift=0.
70  // This gives:
71  // q = mul_hi(x, rcp_freq) >> rcp_shift
72  // = mul_hi(x, (1<<32) - 1)) >> 0
73  // = floor(x - x/(2^32))
74  // = x - 1 if 1 <= x < 2^32
75  // and we know that x>0 (x=0 is never in a valid normalization interval).
76  //
77  // So we now need to choose the other parameters such that
78  // x_new = x*M + start
79  // plug it in:
80  // x*M + start (desired result)
81  // = bias + x + q*cmpl_freq (*)
82  // = bias + x + (x - 1)*(M - 1) (plug in q=x-1, cmpl_freq)
83  // = bias + 1 + (x - 1)*M
84  // = x*M + (bias + 1 - M)
85  //
86  // so we have start = bias + 1 - M, or equivalently
87  // bias = start + M - 1.
88  this->rcp_freq = static_cast<T>(~0ul);
89  this->rcp_shift = 0;
90  this->bias = start + (1 << scale_bits) - 1;
91  } else {
92  // Alverson, "Integer Division using reciprocals"
93  // shift=ceil(log2(freq))
94  uint32_t shift = 0;
95  while (freq > (1u << shift))
96  shift++;
97 
98  if constexpr (needs64Bit<T>()) {
99  uint64_t x0, x1, t0, t1;
100  // long divide ((uint128) (1 << (shift + 63)) + freq-1) / freq
101  // by splitting it into two 64:64 bit divides (this works because
102  // the dividend has a simple form.)
103  x0 = freq - 1;
104  x1 = 1ull << (shift + 31);
105 
106  t1 = x1 / freq;
107  x0 += (x1 % freq) << 32;
108  t0 = x0 / freq;
109 
110  this->rcp_freq = t0 + (t1 << 32);
111  } else {
112  this->rcp_freq = static_cast<uint32_t>(((1ull << (shift + 31)) + freq - 1) / freq);
113  }
114  this->rcp_shift = shift - 1;
115 
116  // With these values, 'q' is the correct quotient, so we
117  // have bias=start.
118  this->bias = start;
119  }
120  };
121 
122  T rcp_freq; // Fixed-point reciprocal frequency
123  uint32_t freq; // (Exclusive) upper bound of pre-normalization interval
124  uint32_t bias; // Bias
125  uint32_t cmpl_freq; // Complement of frequency: (1 << scale_bits) - freq
126  uint32_t rcp_shift; // Reciprocal shift
127 };
