61 o2::framework::WorkflowSpec getWorkflow(bool propagateMC,
62  bool enableDigitsPrinter,
63  std::string const& cfgInput,
64  std::string const& cfgOutput)
65 {
66  InputType inputType;
67 
68  try {
69  inputType = InputMap.at(cfgInput);
70  } catch (std::out_of_range&) {
71  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
72  }
73  std::vector<OutputType> outputTypes;
74  try {
75  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
76  } catch (std::out_of_range&) {
77  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
78  }
79  auto isEnabled = [&outputTypes](OutputType type) {
80  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
81  };
82 
83  o2::framework::WorkflowSpec specs;
84 
85  if (inputType == InputType::Digits) {
86  using digitInputType = std::vector<o2::emcal::Digit>;
87  specs.emplace_back(o2::emcal::getPublisherSpec<digitInputType>(PublisherConf{
88  "emcal-digit-reader",
89  "o2sim",
90  {"digitbranch", "EMCALDigit", "Digit branch"},
91  {"digittriggerbranch", "EMCALDigitTRGR", "Trigger record branch"},
92  {"mcbranch", "EMCALDigitMCTruth", "MC label branch"},
93  o2::framework::OutputSpec{"EMC", "DIGITS"},
94  o2::framework::OutputSpec{"EMC", "DIGITSTRGR"},
95  o2::framework::OutputSpec{"EMC", "DIGITSMCTR"}},
96  propagateMC));
97 
98  if (enableDigitsPrinter) {
99  try {
100  specs.emplace_back(o2::emcal::reco_workflow::getEmcalDigitsPrinterSpec("digits"));
101  } catch (std::runtime_error& e) {
102  LOG(ERROR) << "Cannot create digits printer spec: " << e.what();
103  }
104  }
105 
106  if (isEnabled(OutputType::Cells)) {
107  // add converter for cells
108  specs.emplace_back(o2::emcal::reco_workflow::getCellConverterSpec(propagateMC));
109  }
110 
111  if (isEnabled(OutputType::Clusters)) {
112  // add clusterizer
113  specs.emplace_back(o2::emcal::reco_workflow::getClusterizerSpec(true));
114  }
115  if (isEnabled(OutputType::AnalysisClusters)) {
116  // add clusters from cells
117  specs.emplace_back(o2::emcal::reco_workflow::getAnalysisClusterSpec(true));
118  }
119  } else if (inputType == InputType::Cells) {
120  using cellInputType = std::vector<o2::emcal::Cell>;
121  specs.emplace_back(o2::emcal::getPublisherSpec<cellInputType>(PublisherConf{
122  "emcal-cell-reader",
123  "o2sim",
124  {"cellbranch", "EMCALCell", "Cell branch"},
125  {"celltriggerbranch", "EMCALCellTRGR", "Trigger record branch"},
126  {"mcbranch", "EMCALCellMCTruth", "MC label branch"},
127  o2::framework::OutputSpec{"EMC", "CELLS"},
128  o2::framework::OutputSpec{"EMC", "CELLSTRGR"},
129  o2::framework::OutputSpec{"EMC", "CELLSMCTR"}},
130  propagateMC));
131  if (enableDigitsPrinter) {
132  try {
133  specs.emplace_back(o2::emcal::reco_workflow::getEmcalDigitsPrinterSpec("cells"));
134  } catch (std::runtime_error& e) {
135  LOG(ERROR) << "Cannot create digits printer spec: " << e.what();
136  }
137  }
138 
139  if (isEnabled(OutputType::Clusters)) {
140  // add clusterizer from cells
141  specs.emplace_back(o2::emcal::reco_workflow::getClusterizerSpec(false));
142  }
143  if (isEnabled(OutputType::AnalysisClusters)) {
144  // add clusters from cells
145  specs.emplace_back(o2::emcal::reco_workflow::getAnalysisClusterSpec(false));
146  }
147  } else if (inputType == InputType::Raw) {
148 
149  //Subcscribe to Readout data using o2-readout-proxy
150  if (isEnabled(OutputType::Cells)) {
151  // add converter for cells
152  specs.emplace_back(o2::emcal::reco_workflow::getRawToCellConverterSpec());
153  }
154  }
155 
156  // check if the process is ready to quit
157  // this is decided upon the meta information in the EMCAL block header, the operation is set
158  // value kNoPayload in case of no data or no operation
159  // see also PublisherSpec.cxx
160  // in this workflow, the EOD is sent after the last real data, and all inputs will receive EOD,
161  // so it is enough to check on the first occurence
162  // FIXME: this will be changed once DPL can propagate control events like EOD
163  auto checkReady = [](o2::framework::DataRef const& ref) {
164  auto const* emcalheader = o2::framework::DataRefUtils::getHeader<o2::emcal::EMCALBlockHeader*>(ref);
165  // sector number -1 indicates end-of-data
166  if (emcalheader != nullptr) {
167  // indicate normal processing if not ready and skip if ready
168  if (!emcalheader->mHasPayload) {
169  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::SkipProcessing, true);
170  }
171  }
172  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::DoProcessing, false);
173  };

175  auto makeWriterSpec = [propagateMC, checkReady](const char* processName, const char* defaultFileName, const char* defaultTreeName,
176  auto&& databranch, auto&& triggerbranch, auto&& mcbranch) {
177  // depending on the MC propagation flag, the RootTreeWriter spec is created with two
178  // or one branch definition
179  if (propagateMC) {
180  return std::move(o2::framework::MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
181  o2::framework::MakeRootTreeWriterSpec::TerminationCondition{checkReady},
182  std::move(databranch),
183  std::move(triggerbranch),
184  std::move(mcbranch)));
185  }
186  return std::move(o2::framework::MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
187  o2::framework::MakeRootTreeWriterSpec::TerminationCondition{checkReady},
188  std::move(databranch),
189  std::move(triggerbranch)));
190  };
