47 void trackleterKernelSerial(
48  const std::vector<Cluster>& clustersNextLayer, // 0 2
49  const std::vector<Cluster>& clustersCurrentLayer, // 1 1
50  const std::array<int, ZBins * PhiBins + 1>& indexTableNext,
51  const unsigned char pairOfLayers,
52  const float phiCut,
53  std::vector<Tracklet>& Tracklets,
54  std::vector<int>& foundTracklets,
55  // const ROframe* evt = nullptr,
56  const int maxTrackletsPerCluster = static_cast<int>(2e3))
57 {
58  foundTracklets.resize(clustersCurrentLayer.size(), 0);
59  // loop on layer1 clusters
60  for (unsigned int iCurrentLayerClusterIndex{0}; iCurrentLayerClusterIndex < clustersCurrentLayer.size(); ++iCurrentLayerClusterIndex) {
61  int storedTracklets{0};
62  const Cluster currentCluster{clustersCurrentLayer[iCurrentLayerClusterIndex]};
63  const int layerIndex{pairOfLayers == LAYER0_TO_LAYER1 ? 0 : 2};
64  const int4 selectedBinsRect{VertexerTraits::getBinsRect(currentCluster, layerIndex, 0.f, 50.f, phiCut / 2)};
65  if (selectedBinsRect.x != 0 || selectedBinsRect.y != 0 || selectedBinsRect.z != 0 || selectedBinsRect.w != 0) {
66  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
67  if (phiBinsNum < 0) {
68  phiBinsNum += PhiBins;
69  }
70  // loop on phi bins next layer
71  for (int iPhiBin{selectedBinsRect.y}, iPhiCount{0}; iPhiCount < phiBinsNum; iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
72  const int firstBinIndex{index_table_utils::getBinIndex(selectedBinsRect.x, iPhiBin)};
73  const int firstRowClusterIndex{indexTableNext[firstBinIndex]};
74  const int maxRowClusterIndex{indexTableNext[firstBinIndex + ZBins]};
75  // loop on clusters next layer
76  for (int iNextLayerClusterIndex{firstRowClusterIndex}; iNextLayerClusterIndex < maxRowClusterIndex && iNextLayerClusterIndex < static_cast<int>(clustersNextLayer.size()); ++iNextLayerClusterIndex) {
77  const Cluster& nextCluster{clustersNextLayer[iNextLayerClusterIndex]};
78  if (gpu::GPUCommonMath::Abs(currentCluster.phiCoordinate - nextCluster.phiCoordinate) < phiCut) {
79  if (storedTracklets < maxTrackletsPerCluster) {
80  if (pairOfLayers == LAYER0_TO_LAYER1) {
81  Tracklets.emplace_back(iNextLayerClusterIndex, iCurrentLayerClusterIndex, nextCluster, currentCluster);
82  } else {
83  Tracklets.emplace_back(iCurrentLayerClusterIndex, iNextLayerClusterIndex, currentCluster, nextCluster);
84  }
85  ++storedTracklets;
86  }
87  }
88  }
89  }
90  }
91  foundTracklets[iCurrentLayerClusterIndex] = storedTracklets;
92  }
93 }

95 void trackletSelectionKernelSerial(
96  const std::vector<Cluster>& clustersNextLayer, //0
97  const std::vector<Cluster>& clustersCurrentLayer, //1
98  const std::vector<Tracklet>& tracklets01,
99  const std::vector<Tracklet>& tracklets12,
100  const std::vector<int>& foundTracklets01,
101  const std::vector<int>& foundTracklets12,
102  std::vector<Line>& destTracklets,
103 #ifdef _ALLOW_DEBUG_TREES_ITS_
104  std::vector<std::array<int, 2>>& allowedTrackletPairs,
105  StandaloneDebugger* debugger,
106  ROframe* event,
107 #endif
108  const float tanLambdaCut = 0.025f,
109  const float phiCut = 0.005f,
110  const int maxTracklets = static_cast<int>(1e2))
111 {
112  int offset01{0};
113  int offset12{0};
114  for (unsigned int iCurrentLayerClusterIndex{0}; iCurrentLayerClusterIndex < clustersCurrentLayer.size(); ++iCurrentLayerClusterIndex) {
115  int validTracklets{0};
116  for (int iTracklet12{offset12}; iTracklet12 < offset12 + foundTracklets12[iCurrentLayerClusterIndex]; ++iTracklet12) {
117  for (int iTracklet01{offset01}; iTracklet01 < offset01 + foundTracklets01[iCurrentLayerClusterIndex]; ++iTracklet01) {
118  const float deltaTanLambda{gpu::GPUCommonMath::Abs(tracklets01[iTracklet01].tanLambda - tracklets12[iTracklet12].tanLambda)};
119  const float deltaPhi{gpu::GPUCommonMath::Abs(tracklets01[iTracklet01].phiCoordinate - tracklets12[iTracklet12].phiCoordinate)};
120  if (deltaTanLambda < tanLambdaCut && deltaPhi < phiCut && validTracklets != maxTracklets) {
121  assert(tracklets01[iTracklet01].secondClusterIndex == tracklets12[iTracklet12].firstClusterIndex);
122 #ifdef _ALLOW_DEBUG_TREES_ITS_
123  allowedTrackletPairs.push_back(std::array<int, 2>{iTracklet01, iTracklet12});
124  destTracklets.emplace_back(tracklets01[iTracklet01], clustersNextLayer.data(), clustersCurrentLayer.data(), debugger->getEventId(clustersNextLayer[tracklets01[iTracklet01].firstClusterIndex].clusterId, clustersCurrentLayer[tracklets01[iTracklet01].secondClusterIndex].clusterId, event));
125 #else
126  destTracklets.emplace_back(tracklets01[iTracklet01], clustersNextLayer.data(), clustersCurrentLayer.data());
127 #endif
128  ++validTracklets;
129  }
130  }
131  }
132  offset01 += foundTracklets01[iCurrentLayerClusterIndex];
133  offset12 += foundTracklets12[iCurrentLayerClusterIndex];
134  }
135 }

194 void VertexerTraits::arrangeClusters(ROframe* event)
195 {
196  mEvent = event;
197  for (int iLayer{0}; iLayer < constants::its::LayersNumberVertexer; ++iLayer) {
198  const auto& currentLayer{event->getClustersOnLayer(iLayer)};
199  const size_t clustersNum{currentLayer.size()};
200  if (clustersNum > 0) {
201  if (clustersNum > mClusters[iLayer].capacity()) {
202  mClusters[iLayer].reserve(clustersNum);
203  }
204  for (unsigned int iCluster{0}; iCluster < clustersNum; ++iCluster) {
205  mClusters[iLayer].emplace_back(iLayer, currentLayer.at(iCluster));
206  mAverageClustersRadii[iLayer] += mClusters[iLayer].back().rCoordinate;
207  }
208  mAverageClustersRadii[iLayer] *= 1.f / clustersNum;
209 
210  std::sort(mClusters[iLayer].begin(), mClusters[iLayer].end(), [](Cluster& cluster1, Cluster& cluster2) {
211  return cluster1.indexTableBinIndex < cluster2.indexTableBinIndex;
212  });
213  int previousBinIndex{0};
214  mIndexTables[iLayer][0] = 0;
215  for (unsigned int iCluster{0}; iCluster < clustersNum; ++iCluster) {
216  const int currentBinIndex{mClusters[iLayer][iCluster].indexTableBinIndex};
217  if (currentBinIndex > previousBinIndex) {
218  for (int iBin{previousBinIndex + 1}; iBin <= currentBinIndex; ++iBin) {
219  mIndexTables[iLayer][iBin] = iCluster;
220  }
221  previousBinIndex = currentBinIndex;
222  }
223  }
224  for (int iBin{previousBinIndex + 1}; iBin <= ZBins * PhiBins; iBin++) {
225  mIndexTables[iLayer][iBin] = static_cast<int>(clustersNum);
226  }
227  }
228  }
229  mDeltaRadii10 = mAverageClustersRadii[1] - mAverageClustersRadii[0];
230  mDeltaRadii21 = mAverageClustersRadii[2] - mAverageClustersRadii[1];
231  mMaxDirectorCosine3 =
232  LayersZCoordinate()[2] / std::sqrt(LayersZCoordinate()[2] * LayersZCoordinate()[2] +
233  (mDeltaRadii10 + mDeltaRadii21) * (mDeltaRadii10 + mDeltaRadii21));
234 }

236 const std::vector<std::pair<int, int>> VertexerTraits::selectClusters(const std::array<int, ZBins * PhiBins + 1>& indexTable,
237  const std::array<int, 4>& selectedBinsRect)
238 {
239  std::vector<std::pair<int, int>> filteredBins{};
240  int phiBinsNum{selectedBinsRect[3] - selectedBinsRect[1] + 1};
241  if (phiBinsNum < 0)
242  phiBinsNum += PhiBins;
243  filteredBins.reserve(phiBinsNum);
244  for (int iPhiBin{selectedBinsRect[1]}, iPhiCount{0}; iPhiCount < phiBinsNum;
245  iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
246  const int firstBinIndex{index_table_utils::getBinIndex(selectedBinsRect[0], iPhiBin)};
247  filteredBins.emplace_back(
248  indexTable[firstBinIndex],
249  index_table_utils::countRowSelectedBins(indexTable, iPhiBin, selectedBinsRect[0], selectedBinsRect[2]));
250  }
251  return filteredBins;
252 }

274  auto& currentCluster{mClusters[2][iCurrentLayerClusterIndex]};

276  const Cluster& nextCluster{mClusters[1][iNextLayerClusterIndex]};

393 void VertexerTraits::computeVertices()
394 {
395  const int numTracklets{static_cast<int>(mTracklets.size())};
396  std::vector<bool> usedTracklets{};
397  usedTracklets.resize(mTracklets.size(), false);
398  for (int tracklet1{0}; tracklet1 < numTracklets; ++tracklet1) {
399  if (usedTracklets[tracklet1])
400  continue;
401  for (int tracklet2{tracklet1 + 1}; tracklet2 < numTracklets; ++tracklet2) {
402  if (usedTracklets[tracklet2])
403  continue;
404  if (Line::getDCA(mTracklets[tracklet1], mTracklets[tracklet2]) <= mVrtParams.pairCut) {
405  mTrackletClusters.emplace_back(tracklet1, mTracklets[tracklet1], tracklet2, mTracklets[tracklet2]);
406  std::array<float, 3> tmpVertex{mTrackletClusters.back().getVertex()};
407  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
408  mTrackletClusters.pop_back();
409  break;
410  }
411  usedTracklets[tracklet1] = true;
412  usedTracklets[tracklet2] = true;
413  for (int tracklet3{0}; tracklet3 < numTracklets; ++tracklet3) {
414  if (usedTracklets[tracklet3])
415  continue;
416  if (Line::getDistanceFromPoint(mTracklets[tracklet3], tmpVertex) < mVrtParams.pairCut) {
417  mTrackletClusters.back().add(tracklet3, mTracklets[tracklet3]);
418  usedTracklets[tracklet3] = true;
419  tmpVertex = mTrackletClusters.back().getVertex();
420  }
421  }
422  break;
423  }
424  }
425  }
426 #ifdef _ALLOW_DEBUG_TREES_ITS_
427  if (isDebugFlag(VertexerDebug::LineSummaryAll)) {
428  mDebugger->fillLineClustersTree(mTrackletClusters, mEvent);
429  }
430 #endif
431  std::sort(mTrackletClusters.begin(), mTrackletClusters.end(),
432  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
433  int noClusters{static_cast<int>(mTrackletClusters.size())};
434  for (int iCluster1{0}; iCluster1 < noClusters; ++iCluster1) {
435  std::array<float, 3> vertex1{mTrackletClusters[iCluster1].getVertex()};
436  std::array<float, 3> vertex2{};
437  for (int iCluster2{iCluster1 + 1}; iCluster2 < noClusters; ++iCluster2) {
438  vertex2 = mTrackletClusters[iCluster2].getVertex();
439  if (std::abs(vertex1[2] - vertex2[2]) < mVrtParams.clusterCut) {
440  float distance{(vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
441  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
442  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2])};
443  if (distance <= mVrtParams.pairCut * mVrtParams.pairCut) {
444  for (auto label : mTrackletClusters[iCluster2].getLabels()) {
445  mTrackletClusters[iCluster1].add(label, mTracklets[label]);
446  vertex1 = mTrackletClusters[iCluster1].getVertex();
447  }
448  }
449  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster2);
450  --iCluster2;
451  --noClusters;
452  }
453  }
454  }
455  for (int iCluster{0}; iCluster < noClusters; ++iCluster) {
456  if (mTrackletClusters[iCluster].getSize() < mVrtParams.clusterContributorsCut && noClusters > 1) {
457  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster);
458  noClusters--;
459  continue;
460  }
461  if (mTrackletClusters[iCluster].getVertex()[0] * mTrackletClusters[iCluster].getVertex()[0] +
462  mTrackletClusters[iCluster].getVertex()[1] * mTrackletClusters[iCluster].getVertex()[1] <
463  1.98 * 1.98) {
464  mVertices.emplace_back(mTrackletClusters[iCluster].getVertex()[0],
465  mTrackletClusters[iCluster].getVertex()[1],
466  mTrackletClusters[iCluster].getVertex()[2],
467  mTrackletClusters[iCluster].getRMS2(), // Symm matrix. Diagonal: RMS2 components,
468  // off-diagonal: square mean of projections on planes.
469  mTrackletClusters[iCluster].getSize(), // Contributors
470  mTrackletClusters[iCluster].getAvgDistance2(), // In place of chi2
471  mEvent->getROFrameId()
472 #ifdef _ALLOW_DEBUG_TREES_ITS_
473  ,
474  mTrackletClusters[iCluster].getEventId(),
475  mTrackletClusters[iCluster].getPurity()
476 #endif
477  );
478  mEvent->addPrimaryVertex(mVertices.back().mX, mVertices.back().mY, mVertices.back().mZ);
479  }
480  }
481 #ifdef _ALLOW_DEBUG_TREES_ITS_
482  for (auto& vertex : mVertices) {
483  mDebugger->fillVerticesInfoTree(vertex.mX, vertex.mY, vertex.mZ, vertex.mContributors, mEvent->getROFrameId(), vertex.mEventId, vertex.mPurity);
484  }
485 #endif
486 }

488 void VertexerTraits::computeHistVertices()
489 {
490  o2::its::VertexerHistogramsConfiguration histConf;
491 #ifdef _ALLOW_DEBUG_TREES_ITS_
492  std::array<std::vector<int>, 4001> indicesBookKeeper; // HARD_TO_CODE
493  for (auto v : indicesBookKeeper) {
494  v.clear();
495  }
496  const float binSize{histConf.highHistBoundariesXYZ[2] - histConf.lowHistBoundariesXYZ[2]};
497 #endif
498  std::vector<boost::histogram::axis::regular<float>> axes;
499  axes.reserve(3);
500  for (size_t iAxis{0}; iAxis < 3; ++iAxis) {
501  axes.emplace_back(histConf.nBinsXYZ[iAxis] - 1, histConf.lowHistBoundariesXYZ[iAxis], histConf.highHistBoundariesXYZ[iAxis]);
502  }
503 
504  auto histX = boost::histogram::make_histogram(axes[0]);
505  auto histY = boost::histogram::make_histogram(axes[1]);
506  auto histZ = boost::histogram::make_histogram(axes[2]);
507 
508  // Loop over lines, calculate transverse vertices within beampipe and fill XY histogram to find pseudobeam projection
509  for (size_t iTracklet1{0}; iTracklet1 < mTracklets.size(); ++iTracklet1) {
510  for (size_t iTracklet2{iTracklet1 + 1}; iTracklet2 < mTracklets.size(); ++iTracklet2) {
511  if (Line::getDCA(mTracklets[iTracklet1], mTracklets[iTracklet2]) < mVrtParams.histPairCut) {
512  ClusterLines cluster{mTracklets[iTracklet1], mTracklets[iTracklet2]};
513  if (cluster.getVertex()[0] * cluster.getVertex()[0] + cluster.getVertex()[1] * cluster.getVertex()[1] < 1.98f * 1.98f) {
514  histX(cluster.getVertex()[0]);
515  histY(cluster.getVertex()[1]);
516  }
517  }
518  }
519  }
520 
521  // Try again to use std::max_element as soon as boost is upgraded to 1.71...
522  // atm you can iterate over histograms, not really possible to get bin index. Need to use iterate(histogram)
523  int maxXBinContent{0};
524  int maxYBinContent{0};
525  int maxXIndex{0};
526  int maxYIndex{0};
527  for (auto x : indexed(histX)) {
528  if (x.get() > maxXBinContent) {
529  maxXBinContent = x.get();
530  maxXIndex = x.index();
531  }
532  }
533 
534  for (auto y : indexed(histY)) {
535  if (y.get() > maxYBinContent) {
536  maxYBinContent = y.get();
537  maxYIndex = y.index();
538  }
539  }
540 
541  // Compute weighted average around XY to smooth the position
542  if (maxXBinContent || maxYBinContent) {
543  float tmpX{histConf.lowHistBoundariesXYZ[0] + histConf.binSizeHistX * maxXIndex + histConf.binSizeHistX / 2};
544  float tmpY{histConf.lowHistBoundariesXYZ[1] + histConf.binSizeHistY * maxYIndex + histConf.binSizeHistY / 2};
545  int sumX{maxXBinContent};
546  int sumY{maxYBinContent};
547  float wX{tmpX * static_cast<float>(maxXBinContent)};
548  float wY{tmpY * static_cast<float>(maxYBinContent)};
549  for (int iBinX{std::max(0, maxXIndex - histConf.binSpanXYZ[0])}; iBinX < std::min(maxXIndex + histConf.binSpanXYZ[0] + 1, histConf.nBinsXYZ[0] - 1); ++iBinX) {
550  if (iBinX != maxXIndex) {
551  wX += (histConf.lowHistBoundariesXYZ[0] + histConf.binSizeHistX * iBinX + histConf.binSizeHistX / 2) * histX.at(iBinX);
552  sumX += histX.at(iBinX);
553  }
554  }
555  for (int iBinY{std::max(0, maxYIndex - histConf.binSpanXYZ[1])}; iBinY < std::min(maxYIndex + histConf.binSpanXYZ[1] + 1, histConf.nBinsXYZ[1] - 1); ++iBinY) {
556  if (iBinY != maxYIndex) {
557  wY += (histConf.lowHistBoundariesXYZ[1] + histConf.binSizeHistY * iBinY + histConf.binSizeHistY / 2) * histY.at(iBinY);
558  sumY += histY.at(iBinY);
559  }
560  }
561 
562  const float beamCoordinateX{wX / sumX};
563  const float beamCoordinateY{wY / sumY};
564 
565  // create actual pseudobeam line
566  Line pseudoBeam{std::array<float, 3>{beamCoordinateX, beamCoordinateY, 1}, std::array<float, 3>{beamCoordinateX, beamCoordinateY, -1}};
567 
568  // Fill z coordinate histogram
569  for (auto& line : mTracklets) {
570  if (Line::getDCA(line, pseudoBeam) < mVrtParams.histPairCut) {
571  ClusterLines cluster{line, pseudoBeam};
572 #ifdef _ALLOW_DEBUG_TREES_ITS_
573  // here needs to store ID and bin
574  int b_index = StandaloneDebugger::getBinIndex(cluster.getVertex()[2],
575  histConf.nBinsXYZ[2] - 1,
576  histConf.lowHistBoundariesXYZ[2],
577  histConf.highHistBoundariesXYZ[2]);
578  const int evId = cluster.getEventId();
579  if (!(evId < 0)) {
580  indicesBookKeeper[b_index].push_back(evId);
581  }
582 #endif
583  histZ(cluster.getVertex()[2]);
584  }
585  }
586  for (int iVertex{0};; ++iVertex) {
587 #ifdef _ALLOW_DEBUG_TREES_ITS_
588  int mVotes{0};
589  int mNVotes{0};
590  int mPoll{-1};
591  int mTotVotes{0};
592  int mSwitches{0};
593  std::unordered_map<int, int> mMap;
594 #endif
595  int maxZBinContent{0};
596  int maxZIndex{0};
597  // find maximum
598  for (auto z : indexed(histZ)) {
599  if (z.get() > maxZBinContent) {
600  maxZBinContent = z.get();
601  maxZIndex = z.index();
602  }
603  }
604  float tmpZ{histConf.lowHistBoundariesXYZ[2] + histConf.binSizeHistZ * maxZIndex + histConf.binSizeHistZ / 2};
605  int sumZ{maxZBinContent};
606  float wZ{tmpZ * static_cast<float>(maxZBinContent)};
607  for (int iBinZ{std::max(0, maxZIndex - histConf.binSpanXYZ[2])}; iBinZ < std::min(maxZIndex + histConf.binSpanXYZ[2] + 1, histConf.nBinsXYZ[2] - 1); ++iBinZ) {
608  if (iBinZ != maxZIndex) {
609  wZ += (histConf.lowHistBoundariesXYZ[2] + histConf.binSizeHistZ * iBinZ + histConf.binSizeHistZ / 2) * histZ.at(iBinZ);
610  sumZ += histZ.at(iBinZ);
611  histZ.at(iBinZ) = 0;
612 #ifdef _ALLOW_DEBUG_TREES_ITS_
613  indicesBookKeeper[iBinZ].clear();
614 #endif
615  }
616  }
617  if ((sumZ < mVrtParams.clusterContributorsCut) && (iVertex != 0)) {
618  break;
619  }
620 #ifdef _ALLOW_DEBUG_TREES_ITS_
621  for (int iBinZ{std::max(0, maxZIndex - histConf.binSpanXYZ[2])};
622  iBinZ < std::min(maxZIndex + histConf.binSpanXYZ[2] + 1, histConf.nBinsXYZ[2] - 1);
623  ++iBinZ) {
624  // Boyer-Moore-ish
625  for (auto& evId : indicesBookKeeper[iBinZ]) {
626  if (mNVotes == 0) {
627  mPoll = evId;
628  ++mNVotes;
629  } else {
630  if (evId == mPoll) {
631  ++mNVotes;
632  } else {
633  ++mSwitches;
634  --mNVotes;
635  }
636  }
637  auto it = mMap.find(evId);
638  if (it == mMap.end()) {
639  mMap.emplace(evId, 1);
640  } else {
641  it->second += 1;
642  }
643  ++mTotVotes;
644  }
645  // std::cout<<"\n";
646  }
647  // Compute purity
648  auto it = mMap.find(mPoll);
649  assert(it != mMap.end());
650  float purity{(float)it->second / (float)mTotVotes};
651 #endif
652  histZ.at(maxZIndex) = 0;
653 #ifdef _ALLOW_DEBUG_TREES_ITS_
654  indicesBookKeeper[maxZIndex].clear();
655 #endif
656  const float vertexZCoordinate{wZ / sumZ};
657  mVertices.emplace_back(beamCoordinateX,
658  beamCoordinateY,
659  vertexZCoordinate,
660  std::array<float, 6>{0., 0., 0., 0., 0., 0.},
661  sumZ,
662  0.,
663  mEvent->getROFrameId()
664 #ifdef _ALLOW_DEBUG_TREES_ITS_
665  ,
666  mPoll,
667  purity
668 #endif
669  );
670  }
671  }
672 #ifdef _ALLOW_DEBUG_TREES_ITS_
673  for (auto& vertex : mVertices) {
674  mDebugger->fillVerticesInfoTree(vertex.mX, vertex.mY, vertex.mZ, vertex.mContributors, mEvent->getROFrameId(), vertex.mEventId, vertex.mPurity);
675  }
676 #endif
677 }
