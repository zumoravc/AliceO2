39 void o2::data_flow::SubframeBuilderDevice::InitTask()
40 {
41  mOrbitDuration = GetConfig()->GetValue<uint32_t>(OptionKeyOrbitDuration);
42  mOrbitsPerTimeframe = GetConfig()->GetValue<uint32_t>(OptionKeyOrbitsPerTimeframe);
43  mInputChannelName = GetConfig()->GetValue<std::string>(OptionKeyInputChannelName);
44  mOutputChannelName = GetConfig()->GetValue<std::string>(OptionKeyOutputChannelName);
45  mFLPId = GetConfig()->GetValue<size_t>(OptionKeyFLPId);
46  mStripHBF = GetConfig()->GetValue<bool>(OptionKeyStripHBF);
47 
48  LOG(INFO) << "Obtaining data from DataPublisher\n";
49  // Now that we have all the information lets create the policies to do the
50  // payload extraction and merging and create the actual PayloadMerger.
51 
52  // We extract the timeframeId from the number of orbits.
53  // FIXME: handle multiple socket ids
54  Merger::IdExtractor makeId = [this](std::unique_ptr<FairMQMessage>& msg) -> SubframeId {
55  HeartbeatHeader* hbh = reinterpret_cast<HeartbeatHeader*>(msg->GetData());
56  SubframeId id = {.timeframeId = hbh->orbit / this->mOrbitsPerTimeframe,
57  .socketId = 0};
58  return id;
59  };
60 
61  // We extract the payload differently depending on wether we want to strip
62  // the header or not.
63  Merger::PayloadExtractor payloadExtractor = [this](char** out, char* in, size_t inSize) -> size_t {
64  if (!this->mStripHBF) {
65  return Merger::fullPayloadExtractor(out, in, inSize);
66  }
67  return o2::dataflow::extractDetectorPayloadStrip(out, in, inSize);
68  };
69 
70  // Whether a given timeframe is complete depends on how many orbits per
71  // timeframe we want.
72  Merger::MergeCompletionCheker checkIfComplete =
73  [this](Merger::MergeableId id, Merger::MessageMap& map) {
74  return map.count(id) < this->mOrbitsPerTimeframe;
75  };
76 
77  mMerger.reset(new Merger(makeId, checkIfComplete, payloadExtractor));
78  OnData(mInputChannelName.c_str(), &o2::data_flow::SubframeBuilderDevice::HandleData);
79 }

81 bool o2::data_flow::SubframeBuilderDevice::BuildAndSendFrame(FairMQParts& inParts)
82 {
83  auto id = mMerger->aggregate(inParts.At(1));
84 
85  char** outBuffer;
86  size_t outSize = mMerger->finalise(outBuffer, id);
87  // In this case we do not have enough subtimeframes for id,
88  // so we simply return.
89  if (outSize == 0)
90  return true;
91  // If we reach here, it means we do have enough subtimeframes.
92 
93  // top level subframe header, the DataHeader is going to be used with
94  // description "SUBTIMEFRAMEMD"
95  // this should be defined in a common place, and also the origin
96  // the origin can probably name a detector identifier, but not sure if
97  // all CRUs of a FLP in all cases serve a single detector
98  o2::header::DataHeader dh;
99  dh.dataDescription = o2::header::DataDescription("SUBTIMEFRAMEMD");
100  dh.dataOrigin = o2::header::DataOrigin("FLP");
101  dh.subSpecification = mFLPId;
102  dh.payloadSize = sizeof(SubframeMetadata);
103 
104  DataHeader payloadheader(*o2::header::get<DataHeader*>((byte*)inParts.At(0)->GetData()));
105 
106  // subframe meta information as payload
107  SubframeMetadata md;
108  // id is really the first orbit in the timeframe.
109  md.startTime = id.timeframeId * mOrbitsPerTimeframe * static_cast<uint64_t>(mOrbitDuration);
110  md.duration = mOrbitDuration * mOrbitsPerTimeframe;
111  LOG(INFO) << "Start time for subframe (" << md.startTime << ", "
112  << md.duration
113  << ")";
114 
115  // Add the metadata about the merged subtimeframes
116  // FIXME: do we really need this?
117  O2Message outgoing;
118  o2::base::addDataBlock(outgoing, dh, NewSimpleMessage(md));
119 
120  // Add the actual merged payload.
121  o2::base::addDataBlock(outgoing, payloadheader,
122  NewMessage(
123  *outBuffer, outSize,
124  [](void* data, void* hint) { delete[] reinterpret_cast<char*>(hint); }, *outBuffer));
125  // send message
126  Send(outgoing, mOutputChannelName.c_str());
127  // FIXME: do we actually need this? outgoing should go out of scope
128  outgoing.fParts.clear();
129 
130  return true;
131 }

133 bool o2::data_flow::SubframeBuilderDevice::HandleData(FairMQParts& msgParts, int /*index*/)
134 {
135  // loop over header payload pairs in the incoming multimessage
136  // for each pair
137  // - check timestamp
138  // - create new subtimeframe if none existing where the timestamp of the data fits
139  // - add pair to the corresponding subtimeframe
140 
141  // check for completed subtimeframes and send all completed frames
142  // the builder does not implement the routing to the EPN, this is done in the
143  // specific FLP-EPN setup
144  // to fit into the simple emulation of event/frame ids in the flpSender the order of
145  // subtimeframes needs to be preserved
146  BuildAndSendFrame(msgParts);
147  return true;
148 }
