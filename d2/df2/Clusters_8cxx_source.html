28 bool Clusters::processCluster(const o2::tpc::ClusterNative& cluster, const o2::tpc::Sector sector, const int row)
29 {
30  const int nROC = row < 63 ? int(sector) : int(sector) + 36;
31  const int rocRow = row < 63 ? row : row - 63;
32 
33  const float pad = cluster.getPad();
34 
35  const uint16_t qMax = cluster.qMax;
36  const uint16_t qTot = cluster.qTot;
37  const float sigmaPad = cluster.getSigmaPad();
38  const float sigmaTime = cluster.getSigmaTime();
39  const float timeBin = cluster.getTime();
40 
41  float count = mNClusters.getCalArray(nROC).getValue(rocRow, pad);
42  mNClusters.getCalArray(nROC).setValue(rocRow, pad, count + 1);
43 
44  float charge = mQMax.getCalArray(nROC).getValue(rocRow, pad);
45  mQMax.getCalArray(nROC).setValue(rocRow, pad, charge + qMax);
46 
47  charge = mQTot.getCalArray(nROC).getValue(rocRow, pad);
48  mQTot.getCalArray(nROC).setValue(rocRow, pad, charge + qTot);
49 
50  count = mSigmaTime.getCalArray(nROC).getValue(rocRow, pad);
51  mSigmaTime.getCalArray(nROC).setValue(rocRow, pad, count + sigmaTime);
52 
53  count = mSigmaPad.getCalArray(nROC).getValue(rocRow, pad);
54  mSigmaPad.getCalArray(nROC).setValue(rocRow, pad, count + sigmaPad);
55 
56  count = mTimeBin.getCalArray(nROC).getValue(rocRow, pad);
57  mTimeBin.getCalArray(nROC).setValue(rocRow, pad, count + timeBin);
58 
59  return true;
60 }
