23 bool Geometry::absToRelNumbering(short absId, short* relid)
24 {
25  // Converts the absolute numbering into the following array
26  // relid[0] = CPV Module number 1:fNModules
27  // relid[1] = Column number inside a CPV module (Phi coordinate)
28  // relid[2] = Row number inside a CPV module (Z coordinate)
29 
30  short nCPV = kNumberOfCPVPadsPhi * kNumberOfCPVPadsZ;
31  relid[0] = (absId - 1) / nCPV + 1;
32  absId -= (relid[0] - 1) * nCPV;
33  relid[2] = absId / kNumberOfCPVPadsZ + 1;
34  relid[1] = absId - (relid[2] - 1) * kNumberOfCPVPadsZ;
35 
36  return true;
37 }

44 int Geometry::areNeighbours(short absId1, short absId2)
45 {
46 
47  // Gives the neighbourness of two digits = 0 are not neighbour but continue searching
48  // = 1 are neighbour
49  // = 2 are not neighbour but do not continue searching
50  // =-1 are not neighbour, continue searching, but do not look before d2 next
51  // time
52  // neighbours are defined as digits having at least a common vertex
53  // The order of d1 and d2 is important: first (d1) should be a digit already in a cluster
54  // which is compared to a digit (d2) not yet in a cluster
55 
56  short relid1[3];
57  absToRelNumbering(absId1, relid1);
58 
59  short relid2[3];
60  absToRelNumbering(absId2, relid2);
61 
62  if (relid1[0] == relid2[0]) { // inside the same CPV module
63  short rowdiff = TMath::Abs(relid1[1] - relid2[1]);
64  short coldiff = TMath::Abs(relid1[2] - relid2[2]);
65 
66  if ((coldiff <= 1) && (rowdiff <= 1)) { // At least common vertex
67  return 1;
68  } else {
69  if ((relid2[1] > relid1[1]) && (relid2[2] > relid1[2] + 1)) {
70  return 2; // Difference in row numbers is too large to look further
71  }
72  }
73  return 0;
74 
75  } else {
76  if (relid1[0] > relid2[0]) { // we switched to the next module
77  return -1;
78  }
79  return 2;
80  }
81  return 0;
82 }
