91 template <typename T>
92 inline T RegularSpline1D::getSpline(const int iknot1, T f0, T f1, T f2, T f3, float u) const
93 {
98 
104 
105  f0 -= f1;
106  f2 -= f1;
107  f3 -= f1;
108 
109  const T half(0.5f);
110 
111  // Scale u so that the knot1 is at 0.0 and the knot2 is at 1.0
112  T x = T((mNumberOfKnots - 1) * u - iknot1);
113  T x2 = x * x;
114 
115  T z1 = half * (f2 - f0); // scaled u derivative at the knot 1
116  T z2 = half * f3; // scaled u derivative at the knot 2
117 
118  // f(u) = a*u^3 + b*u^2 + c*u + d
119  //
120  // f(0) = f1
121  // f(1) = f2
122  // f'(0) = z1 = 0.5 f2 - 0.5 f0
123  // f'(1) = z2 = 0.5 f3 - 0.5 f1
124  //
125  // T d = f1;
126  // T c = z1;
127 
128  T a = -f2 - f2 + z1 + z2;
129  T b = f2 - z1 - a;
130  return (a * x + b) * x2 + z1 * x + f1;
131 }

151 inline int RegularSpline1D::getKnotIndex(float u) const
152 {
153  //index is just u elem [0, 1] * numberOfKnots and then floored. (so the "left" coordinate beside u gets chosen)
154  int index = (int)(u * (mNumberOfKnots - 1));
155  if (index <= 1)
156  index = 1;
157  else if (index >= mNumberOfKnots - 3)
158  index = mNumberOfKnots - 3;
159  return index;
160 }

162 template <typename T>
163 inline void RegularSpline1D::correctEdges(T* data) const
164 {
165  // data[i] is the i-th f-value
166  constexpr T c0(0.5), c1(1.5);
167  const int i = mNumberOfKnots - 1;
168  data[0] = c0 * (data[0] + data[3]) + c1 * (data[1] - data[2]);
169  data[i] = c0 * (data[i - 0] + data[i - 3]) + c1 * (data[i - 1] - data[i - 2]);
170 }
