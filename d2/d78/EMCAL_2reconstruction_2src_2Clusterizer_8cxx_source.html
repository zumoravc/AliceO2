54 template <class InputType>
55 void Clusterizer<InputType>::getClusterFromNeighbours(std::vector<InputwithIndex>& clusterInputs, int row, int column)
56 {
57  // Recursion 0, add seed cell/digit to cluster
58  if (!clusterInputs.size()) {
59  clusterInputs.emplace_back(mInputMap[row][column]);
60  }
61 
62  // Mark the current cell as clustered
63  mCellMask[row][column] = kTRUE;
64 
65  // Now go recursively to the next 4 neighbours and add them to the cluster if they fulfill the conditions
66  constexpr int rowDiffs[4] = {-1, 0, 0, 1};
67  constexpr int colDiffs[4] = {0, -1, 1, 0};
68  for (int dir = 0; dir < 4; dir++) {
69  if ((row + rowDiffs[dir] < 0) || (row + rowDiffs[dir] >= NROWS)) {
70  continue;
71  }
72  if ((column + colDiffs[dir] < 0) || (column + colDiffs[dir] >= NCOLS)) {
73  continue;
74  }
75 
76  if (mInputMap[row + rowDiffs[dir]][column + colDiffs[dir]].mInput) {
77  if (!mCellMask[row + rowDiffs[dir]][column + colDiffs[dir]]) {
78  if (mDoEnergyGradientCut && not(mInputMap[row + rowDiffs[dir]][column + colDiffs[dir]].mInput->getEnergy() > mInputMap[row][column].mInput->getEnergy() + mGradientCut)) {
79  if (not(TMath::Abs(mInputMap[row + rowDiffs[dir]][column + colDiffs[dir]].mInput->getTimeStamp() - mInputMap[row][column].mInput->getTimeStamp()) > mTimeCut)) {
80  getClusterFromNeighbours(clusterInputs, row + rowDiffs[dir], column + colDiffs[dir]);
81  // Add the cell/digit to the current cluster -- if we end up here, the selected cluster fulfills the condition
82  clusterInputs.emplace_back(mInputMap[row + rowDiffs[dir]][column + colDiffs[dir]]);
83  }
84  }
85  }
86  }
87  }
88 }

94 template <class InputType>
95 void Clusterizer<InputType>::getTopologicalRowColumn(const InputType& input, int& row, int& column)
96 {
97  // Get SM number and relative row/column for SM
98  auto cellIndex = mEMCALGeometry->GetCellIndex(input.getTower());
99  int nSupMod = std::get<0>(cellIndex);
100 
101  auto phiEtaIndex = mEMCALGeometry->GetCellPhiEtaIndexInSModule(nSupMod, std::get<1>(cellIndex), std::get<2>(cellIndex), std::get<3>(cellIndex));
102  row = std::get<0>(phiEtaIndex);
103  column = std::get<1>(phiEtaIndex);
104 
105  // Add shifts wrt. supermodule and type of calorimeter
106  // NOTE:
107  // * Rows (phi) are arranged that one space is left empty between supermodules in phi
108  // This is due to the physical gap that forbids clustering
109  // * For DCAL, there is an additional empty column between two supermodules in eta
110  // Again, this is to account for the gap in DCAL
111 
112  row += nSupMod / 2 * (24 + 1);
113  // In DCAL, leave a gap between two SMs with same phi
114  if (!mEMCALGeometry->IsDCALSM(nSupMod)) { // EMCAL
115  column += nSupMod % 2 * 48;
116  } else {
117  column += nSupMod % 2 * (48 + 1);
118  }
119 }

124 template <class InputType>
125 void Clusterizer<InputType>::findClusters(const gsl::span<InputType const>& inputArray)
126 {
127  mFoundClusters.clear();
128  mInputIndices.clear();
129 
130  // Algorithm
131  // - Fill cells/digits in 2D topological map
132  // - Fill struct arrays (energy,x,y) (to get mapping energy -> (x,y))
133  // - Create 2D bitmap (cell/digit is already clustered or not)
134  // - Sort struct arrays with descending energy
135  //
136  // - Loop over arrays:
137  // --> Check 2D bitmap (don't use cell/digit which are already clustered)
138  // --> Take valid cell/digit with highest energy as seed (they are already sorted)
139  // --> Recursive to neighboughs and create cluster
140  // --> Seed cell and all neighbours belonging to cluster will be put in 2D bitmap
141 
142  // Reset cell/digit maps and cell masks
143  // Loop over one array dim, then reset each array
144  for (auto iArr = 0; iArr < NROWS; iArr++) {
145  mCellMask[iArr].fill(kFALSE);
146  mInputMap[iArr].fill({nullptr, -1});
147  }
148 
149  // Calibrate cells/digits and fill the maps/arrays
150  int nCells = 0;
151  double ehs = 0.0;
152  //for (auto dig : inputArray) {
153  for (int iIndex = 0; iIndex < inputArray.size(); iIndex++) {
154 
155  auto dig = inputArray[iIndex];
156 
157  Float_t inputEnergy = dig.getEnergy();
158  Float_t time = dig.getTimeStamp();
159 
160  if (inputEnergy < mThresholdCellEnergy || time > mTimeMax || time < mTimeMin) {
161  continue;
162  }
163  if (!mEMCALGeometry->CheckAbsCellId(dig.getTower())) {
164  continue;
165  }
166  ehs += inputEnergy;
167 
168  // Put cell/digit to 2D map
169  int row = 0, column = 0;
170  getTopologicalRowColumn(dig, row, column);
171  mInputMap[row][column].mInput = &dig; // mInputMap saves pointers to cells/digits, therefore use addr operator here
172  mInputMap[row][column].mIndex = iIndex; // mInputMap saves the position of cells/digits in the input array
173  mSeedList[nCells].energy = inputEnergy;
174  mSeedList[nCells].row = row;
175  mSeedList[nCells].column = column;
176  nCells++;
177  }
178 
179  // Sort struct arrays with ascending energy
180  std::sort(mSeedList.begin(), std::next(std::begin(mSeedList), nCells));
181 
182  // Take next valid cell/digit in calorimeter as seed (in descending energy order)
183  for (int i = nCells; i--;) {
184  int row = mSeedList[i].row, column = mSeedList[i].column;
185  // Continue if the cell is already masked (i.e. was already clustered)
186  if (mCellMask[row][column]) {
187  continue;
188  }
189  // Continue if energy constraints are not fulfilled
190  if (mSeedList[i].energy <= mThresholdSeedEnergy) {
191  continue;
192  }
193 
194  // Seed is found, form cluster recursively
195  std::vector<InputwithIndex> clusterInputs;
196  getClusterFromNeighbours(clusterInputs, row, column);
197 
198  // Add cells/digits for current cluster to cell/digit index vector
199  int inputIndexStart = mInputIndices.size();
200  for (auto dig : clusterInputs) {
201  mInputIndices.emplace_back(dig.mIndex);
202  }
203  int inputIndexSize = mInputIndices.size() - inputIndexStart;
204 
205  // Now form cluster object from cells/digits
206  mFoundClusters.emplace_back(mInputMap[row][column].mInput->getTimeStamp(), inputIndexStart, inputIndexSize); // Cluster object initialized w/ time of seed cell, start + size of associated cells
207  }
208  LOG(DEBUG) << mFoundClusters.size() << "clusters found from " << nCells << " cells/digits (total=" << inputArray.size() << ")-> ehs " << ehs << " (minE " << mThresholdCellEnergy << ")";
209 }
