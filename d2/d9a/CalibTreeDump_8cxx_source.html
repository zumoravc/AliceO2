37 void CalibTreeDump::dumpToFile(const std::string filename)
38 {
39  // ===| open file and crate tree |============================================
40  std::unique_ptr<TFile> file(TFile::Open(filename.c_str(), "recreate"));
41  auto tree = new TTree("calibTree", "Calibration data object tree");
42 
43  // ===| add default mappings |================================================
44  addDefaultMapping(tree);
45 
46  // ===| add FEE mapping |=====================================================
47  if (mAddFEEInfo) {
48  addFEEMapping(tree);
49  }
50 
51  // ===| fill calDet objects |=================================================
52  addCalDetObjects(tree);
53 
54  // ===| default aliases |=====================================================
55  setDefaultAliases(tree);
56 
57  file->Write();
58 }

61 void CalibTreeDump::addDefaultMapping(TTree* tree)
62 {
63  // loop over ROCs
64  //
65 
66  // ===| mapper |==============================================================
67  const auto& mapper = Mapper::instance();
68 
69  // ===| default mapping objects |=============================================
70  uint16_t rocNumber = 0;
71  // positions
72  std::vector<float> gx;
73  std::vector<float> gy;
74  std::vector<float> lx;
75  std::vector<float> ly;
76  // row and pad
77  std::vector<unsigned char> row;
78  std::vector<unsigned char> pad;
79  std::vector<short> cpad;
80 
81  // ===| add branches with default mappings |==================================
82  tree->Branch("roc", &rocNumber);
83  tree->Branch("gx", &gx);
84  tree->Branch("gy", &gy);
85  tree->Branch("lx", &lx);
86  tree->Branch("ly", &ly);
87  tree->Branch("row", &row);
88  tree->Branch("pad", &pad);
89  tree->Branch("cpad", &cpad);
90 
91  // ===| loop over readout chambers |==========================================
92  for (ROC roc; !roc.looped(); ++roc) {
93  rocNumber = roc;
94 
95  // ===| clear position vectors |============================================
96  gx.clear();
97  gy.clear();
98  lx.clear();
99  ly.clear();
100  row.clear();
101  pad.clear();
102  cpad.clear();
103 
104  // ===| loop over pad rows |================================================
105  const int numberOfRows = mapper.getNumberOfRowsROC(roc);
106  for (int irow = 0; irow < numberOfRows; ++irow) {
107 
108  // ===| loop over pads in row |===========================================
109  const int numberOfPadsInRow = mapper.getNumberOfPadsInRowROC(roc, irow);
110  for (int ipad = 0; ipad < numberOfPadsInRow; ++ipad) {
111  const PadROCPos padROCPos(rocNumber, irow, ipad);
112  const PadPos padPos = mapper.getGlobalPadPos(padROCPos); // pad and row in sector
113  const PadCentre& localPadXY = mapper.getPadCentre(padPos);
114  const LocalPosition2D globalPadXY = mapper.getPadCentre(padROCPos);
115 
116  gx.emplace_back(globalPadXY.X());
117  gy.emplace_back(globalPadXY.Y());
118  lx.emplace_back(localPadXY.X());
119  ly.emplace_back(localPadXY.Y());
120 
121  row.emplace_back(irow);
122  pad.emplace_back(ipad);
123  cpad.emplace_back(ipad - numberOfPadsInRow / 2);
124  }
125  }
126 
127  tree->Fill();
128  }
129 }

132 void CalibTreeDump::addFEEMapping(TTree* tree)
133 {
134  // loop over ROCs
135  //
136 
137  // ===| mapper |==============================================================
138  const auto& mapper = Mapper::instance();
139 
140  readTraceLengths();
141 
142  // ===| default mapping objects |=============================================
143  // FEC
144  std::vector<unsigned char> fecInSector;
145  std::vector<unsigned char> sampaOnFEC;
146  std::vector<unsigned char> channelOnSampa;
147  // trace length
148  std::vector<float>* traceLength = nullptr;
149 
150  // ===| add branches with default mappings |==================================
151  auto brFecInSector = tree->Branch("fecInSector", &fecInSector);
152  auto brSampaOnFEC = tree->Branch("sampaOnFEC", &sampaOnFEC);
153  auto brChannelOnSampa = tree->Branch("channelOnSampa", &channelOnSampa);
154  auto brTraceLength = tree->Branch("traceLength", &traceLength);
155 
156  // ===| loop over readout chambers |==========================================
157  for (ROC roc; !roc.looped(); ++roc) {
158  int rocNumber = roc;
159  //tree->GetEntry(rocNumber);
160  traceLength = ((roc.rocType() == RocType::IROC) ? &mTraceLengthIROC : &mTraceLengthOROC);
161 
162  // ===| clear position vectors |============================================
163  fecInSector.clear();
164  sampaOnFEC.clear();
165  channelOnSampa.clear();
166 
167  const int rowOffset = (roc.rocType() == RocType::OROC) ? mapper.getPadsInIROC() : 0;
168  // ===| loop over pad rows |================================================
169  const int numberOfRows = mapper.getNumberOfRowsROC(roc);
170  for (int irow = 0; irow < numberOfRows; ++irow) {
171 
172  // ===| loop over pads in row |===========================================
173  const int numberOfPadsInRow = mapper.getNumberOfPadsInRowROC(roc, irow);
174  for (int ipad = 0; ipad < numberOfPadsInRow; ++ipad) {
175  const PadROCPos padROCPos(rocNumber, irow, ipad);
176  const auto& fecInfo = mapper.getFECInfo(padROCPos);
177  const auto globalPadNumber = mapper.globalPadNumber(fecInfo);
178 
179  const CRU cru(mapper.getCRU(roc.getSector(), globalPadNumber));
180 
181  fecInSector.emplace_back(fecInfo.getIndex());
182  sampaOnFEC.emplace_back(fecInfo.getSampaChip());
183  channelOnSampa.emplace_back(fecInfo.getSampaChannel());
184  }
185  }
186  brFecInSector->Fill();
187  brSampaOnFEC->Fill();
188  brChannelOnSampa->Fill();
189  brTraceLength->Fill();
190  }
191 }

194 void CalibTreeDump::addCalDetObjects(TTree* tree)
195 {
196 
197  int iter = 0;
198  for (auto& calDet : mCalDetObjects) {
199  // ===| branch names |===
200  std::string name = calDet.getName();
201 
202  if (name == "PadCalibrationObject" || name.size() == 0) {
203  name = fmt::format("calDet_{%02d}", iter);
204  }
205 
206  std::string meanName = fmt::format("{}_mean", name);
207  std::string stdDevName = fmt::format("{}_stdDev", name);
208  std::string medianName = fmt::format("{}_median", name);
209 
210  // ===| branch variables |===
211  std::vector<float>* data = nullptr;
212  float mean{};
213  float stdDev{};
214  float median{};
215 
216  // ===| branch definitions |===
217  TBranch* brMean = tree->Branch(meanName.data(), &mean);
218  TBranch* brStdDev = tree->Branch(stdDevName.data(), &stdDev);
219  TBranch* brMedian = tree->Branch(medianName.data(), &median);
220  TBranch* brData = tree->Branch(name.data(), &data);
221 
222  // ===| loop over ROCs and fill |===
223  for (auto& calArray : calDet.getData()) {
224  // ---| set data |---
225  data = &calArray.getData();
226 
227  // ---| statistics |---
228  mean = TMath::Mean(data->begin(), data->end());
229  stdDev = TMath::StdDev(data->begin(), data->end());
230  median = TMath::Median(data->size(), data->data());
231 
232  // ---| filling |---
233  brData->Fill();
234  brMean->Fill();
235  brStdDev->Fill();
236  brMedian->Fill();
237  }
238  }
239 }

242 void CalibTreeDump::readTraceLengths(std::string_view mappingDir)
243 {
244  std::string inputDir = mappingDir.data();
245  if (!inputDir.size()) {
246  const char* aliceO2env = std::getenv("O2_ROOT");
247  if (aliceO2env)
248  inputDir = aliceO2env;
249  inputDir += "/share/Detectors/TPC/files";
250  }
251 
252  mTraceLengthIROC.reserve(Mapper::getPadsInIROC());
253  mTraceLengthOROC.reserve(Mapper::getPadsInOROC());
254  setTraceLengths(inputDir + "/LENGTH-IROC.txt", mTraceLengthIROC);
255  setTraceLengths(inputDir + "/LENGTH-OROC1.txt", mTraceLengthOROC);
256  setTraceLengths(inputDir + "/LENGTH-OROC2.txt", mTraceLengthOROC);
257  setTraceLengths(inputDir + "/LENGTH-OROC3.txt", mTraceLengthOROC);
258 
259  assert(mTraceLengthIROC.size() == Mapper::getPadsInIROC());
260  assert(mTraceLengthOROC.size() == Mapper::getPadsInOROC());
261 }

264 void CalibTreeDump::setTraceLengths(std::string_view inputFile, std::vector<float>& length)
265 {
266 
267  std::ifstream infile(inputFile.data(), std::ifstream::in);
268  if (!infile.is_open()) {
269  std::cout << "could not open file " << inputFile.data() << "\n";
270  exit(1);
271  }
272 
273  // e.g. IROC file
274  // Col 0 -> INDEX (0 - 5279)
275  // Col 1 -> PADROW (0 - 62)
276  // Col 2 -> PAD (0 - (Np-1))
277  // Col 3 -> Connector (1 - 132)
278  // Col 4 -> Pin (1 - 40)
279  // Col 5 -> Trace length (mm)
280  // Col 6 -> Number of vias
281 
282  unsigned int index{};
283  unsigned int padrow{};
284  unsigned int pad{};
285  unsigned int connector{};
286  unsigned int pin{};
287  float traceLength{};
288  unsigned int numberOfVias{};
289 
290  std::string line;
291  while (std::getline(infile, line)) {
292  std::stringstream streamLine(line);
293  streamLine >> index >> padrow >> pad >> connector >> pin >> traceLength >> numberOfVias;
294 
295  traceLength /= 10.f;
296 
297  length.emplace_back(traceLength);
298  }
299 }
