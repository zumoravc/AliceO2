49 // Create a vector of SampaCluster from a string d
50 // where d is of the form ts-#-bs-#-q-# or
51 // ts-#-bs-#-q-#-#-# ...
52 // d is expected to be a valid SampaCluster string representation
53 // see the asString function in SampaCluster
54 std::vector<SampaCluster> getSampaClusters(const std::string& d)
55 {
56  std::vector<SampaCluster> clusters;
57 
58  auto index = d.find("ts-");
59  auto ts = std::stoi(d.substr(index + 3));
60  index = d.find("bc-");
61  auto bc = std::stoi(d.substr(index + 3));
62  index = d.find("q-");
63  auto q = d.substr(index + 2);
64  index = q.find("-");
65  if (index != std::string::npos) {
66  std::vector<uint10_t> charges;
67  std::istringstream ss(q);
68  std::string adc;
69  while (std::getline(ss, adc, '-')) {
70  charges.emplace_back(std::stoi(adc) & 0x3FF);
71  }
72  clusters.emplace_back(SampaCluster(ts, bc, charges));
73  } else {
74  clusters.emplace_back(SampaCluster(ts, bc, std::stoi(q)));
75  }
76  return clusters;
77 }

79 // create a raw data buffer from a list of strings
80 // where each string is of the form
81 // S#-J#-DS#-CH#-ts-#-bc-#-q-#
82 //
83 // it's a two steps process :
84 //
85 // - first create a buffer of payloads using a PayloadEncoder
86 // - then create the raw data itself (with proper RDHs) from the payload buffer
87 //
88 template <typename FORMAT, typename CHARGESUM>
89 std::vector<std::byte> createBuffer(gsl::span<std::string> data,
90  uint32_t orbit = 12345, uint16_t bc = 678)
91 {
92  auto encoder = createPayloadEncoder<FORMAT, CHARGESUM, true>();
93  encoder->startHeartbeatFrame(orbit, bc);
94  for (auto d : data) {
95  auto dsElecId = decodeDsElecId(d);
96  if (!dsElecId) {
97  std::cout << "Could not get dsElecId for " << d << "\n";
98  }
99  auto channel = decodeChannelId(d);
100  if (!channel) {
101  std::cout << "Could not get channel for " << d << "\n";
102  }
103  auto sampaClusters = getSampaClusters(d);
104  encoder->addChannelData(dsElecId.value(), channel.value(), sampaClusters);
105  }
106  std::vector<std::byte> buffer;
107  encoder->moveToBuffer(buffer);
108 
109  const o2::raw::HBFUtils& hbfutils = o2::raw::HBFUtils::Instance();
110  o2::conf::ConfigurableParam::setValue<uint32_t>("HBFUtils", "orbitFirst", orbit);
111  o2::conf::ConfigurableParam::setValue<uint16_t>("HBFUtils", "bcFirst", bc);
112  std::vector<std::byte> out = o2::mch::raw::paginate(buffer,
113  isUserLogicFormat<FORMAT>::value,
114  isChargeSumMode<CHARGESUM>::value,
115  fmt::format("mch-closure-codec-{}-{}.raw",
116  orbit, bc));
117  return out;
118 }

129 bool testDecode(gsl::span<const std::byte> testBuffer, gsl::span<std::string> expected)
130 {
131  std::vector<std::string> result;
132 
133  auto pageDecoder = createPageDecoder(testBuffer, handlePacketStoreAsVec(result));
134 
135  auto parser = createPageParser();
136 
137  parser(testBuffer, pageDecoder);
138 
139  bool sameSize = result.size() == expected.size();
140  bool permutation = std::is_permutation(begin(result), end(result), begin(expected));
141  BOOST_CHECK_EQUAL(sameSize, true);
142  BOOST_CHECK_EQUAL(permutation, true);
143  if (!permutation || !sameSize) {
144  std::cout << "Got " << result.size() << " results:\n";
145  for (auto s : result) {
146  std::cout << s << "\n";
147  }
148  std::cout << "Expected " << expected.size() << ":\n";
149  for (auto s : expected) {
150  std::cout << s << "\n";
151  }
152  return false;
153  }
154  return true;
155 }
