72 DataProcessorSpec getSourceSpec()
73 {
74  auto processingFct = [](ProcessingContext& pc) {
75  static int counter = 0;
76  o2::test::TriviallyCopyable a(42, 23, 0xdead);
77  o2::test::Polymorphic b(0xbeef);
78  std::vector<o2::test::Polymorphic> c{{0xaffe}, {0xd00f}};
79  // class TriviallyCopyable is both messageable and has a dictionary, the default
80  // picked by the framework is no serialization
81  test::MetaHeader meta1{42};
82  test::MetaHeader meta2{23};
83  pc.outputs().snapshot(Output{"TST", "MESSAGEABLE", 0, Lifetime::Timeframe, {meta1, meta2}}, a);
84  pc.outputs().snapshot(Output{"TST", "MSGBLEROOTSRLZ", 0, Lifetime::Timeframe},
85  o2::framework::ROOTSerialized<decltype(a)>(a));
86  // class Polymorphic is not messageable, so the serialization type is deduced
87  // from the fact that the type has a dictionary and can be ROOT-serialized.
88  pc.outputs().snapshot(Output{"TST", "ROOTNONTOBJECT", 0, Lifetime::Timeframe}, b);
89  // vector of ROOT serializable class
90  pc.outputs().snapshot(Output{"TST", "ROOTVECTOR", 0, Lifetime::Timeframe}, c);
91  // likewise, passed anonymously with char type and class name
92  o2::framework::ROOTSerialized<char, const char> d(*((char*)&c), "vector<o2::test::Polymorphic>");
93  pc.outputs().snapshot(Output{"TST", "ROOTSERLZDVEC", 0, Lifetime::Timeframe}, d);
94  // vector of ROOT serializable class wrapped with TClass info as hint
95  auto* cl = TClass::GetClass(typeid(decltype(c)));
96  ASSERT_ERROR(cl != nullptr);
97  o2::framework::ROOTSerialized<char, TClass> e(*((char*)&c), cl);
98  pc.outputs().snapshot(Output{"TST", "ROOTSERLZDVEC2", 0, Lifetime::Timeframe}, e);
99  // test the 'make' methods
100  pc.outputs().make<o2::test::TriviallyCopyable>(OutputRef{"makesingle", 0}) = a;
101  auto& multi = pc.outputs().make<o2::test::TriviallyCopyable>(OutputRef{"makespan", 0}, 3);
102  ASSERT_ERROR(multi.size() == 3);
103  for (auto& object : multi) {
104  object = a;
105  }
106  // test the adopt method
107  auto freefct = [](void* data, void* hint) {}; // simply ignore the cleanup for the test
108  static std::string teststring = "adoptchunk";
109  pc.outputs().adoptChunk(Output{"TST", "ADOPTCHUNK", 0, Lifetime::Timeframe}, teststring.data(), teststring.length(), freefct, nullptr);
110  // test resizable data chunk, initial size 0 and grow
111  auto& growchunk = pc.outputs().newChunk(OutputRef{"growchunk", 0}, 0);
112  growchunk.resize(sizeof(o2::test::TriviallyCopyable));
113  memcpy(growchunk.data(), &a, sizeof(o2::test::TriviallyCopyable));
114  // test resizable data chunk, large initial size and shrink
115  auto& shrinkchunk = pc.outputs().newChunk(OutputRef{"shrinkchunk", 0}, 1000000);
116  shrinkchunk.resize(sizeof(o2::test::TriviallyCopyable));
117  memcpy(shrinkchunk.data(), &a, sizeof(o2::test::TriviallyCopyable));
118  // make Root-serializable object derived from TObject
119  auto& rootobject = pc.outputs().make<TNamed>(OutputRef{"maketobject", 0}, "a_name", "a_title");
120  // make Root-serializable object Non-TObject
121  auto& rootpolymorphic = pc.outputs().make<o2::test::Polymorphic>(OutputRef{"makerootserlzblobj", 0}, b);
122  // make vector of Root-serializable objects
123  auto& rootserlzblvector = pc.outputs().make<std::vector<o2::test::Polymorphic>>(OutputRef{"rootserlzblvector", 0});
124  rootserlzblvector.emplace_back(0xacdc);
125  rootserlzblvector.emplace_back(0xbeef);
126  // make vector of messagable objects
127  auto& messageablevector = pc.outputs().make<std::vector<o2::test::TriviallyCopyable>>(OutputRef{"messageablevector", 0});
128  ASSERT_ERROR(messageablevector.size() == 0);
129  messageablevector.push_back(a);
130  messageablevector.emplace_back(10, 20, 0xacdc);
131 
132  // create multiple parts matching the same output spec with subspec wildcard
133  // we are using ConcreteDataTypeMatcher to define the output spec matcher independent
134  // of subspec (i.a. a wildcard), all data blcks will go on the same channel regardless
135  // of sebspec.
136  pc.outputs().make<int>(OutputRef{"multiparts", 0}) = 10;
137  pc.outputs().make<int>(OutputRef{"multiparts", 1}) = 20;
138  pc.outputs().make<int>(OutputRef{"multiparts", 2}) = 30;
139 
140  // make a PMR std::vector, make it large to test the auto transport buffer resize funtionality as well
141  Output pmrOutputSpec{"TST", "PMRTESTVECTOR", 0};
142  auto pmrvec = o2::vector<o2::test::TriviallyCopyable>(pc.outputs().getMemoryResource(pmrOutputSpec));
143  pmrvec.reserve(100);
144  pmrvec.emplace_back(o2::test::TriviallyCopyable{1, 2, 3});
145  pc.outputs().adoptContainer(pmrOutputSpec, std::move(pmrvec));
146 
147  // make a vector of POD and set some data
148  pc.outputs().make<std::vector<int>>(OutputRef{"podvector"}) = {10, 21, 42};
149 
150  // now we are done and signal this downstream
151  pc.services().get<ControlService>().endOfStream();
152  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
153 
154  ASSERT_ERROR(pc.outputs().isAllowed({"TST", "MESSAGEABLE", 0}) == true);
155  ASSERT_ERROR(pc.outputs().isAllowed({"TST", "MESSAGEABLE", 1}) == false);
156  ASSERT_ERROR(pc.outputs().isAllowed({"TST", "NOWAY", 0}) == false);
157  };

159  return DataProcessorSpec{"source", // name of the processor
160  {},
161  {OutputSpec{"TST", "MESSAGEABLE", 0, Lifetime::Timeframe},
162  OutputSpec{{"makesingle"}, "TST", "MAKESINGLE", 0, Lifetime::Timeframe},

184 DataProcessorSpec getSinkSpec()
185 {
186  auto processingFct = [](ProcessingContext& pc) {
187  using DataHeader = o2::header::DataHeader;
188  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
189  auto const& input = *iit;
190  LOG(INFO) << (*iit).spec->binding << " " << (iit.isValid() ? "is valid" : "is not valid");
191  if (iit.isValid() == false) {
192  continue;
193  }
194  auto* dh = o2::header::get<const DataHeader*>(input.header);
195  LOG(INFO) << "{" << dh->dataOrigin.str << ":" << dh->dataDescription.str << ":" << dh->subSpecification << "}"
196  << " payload size " << dh->payloadSize;
197 
198  using DumpStackFctType = std::function<void(const o2::header::BaseHeader*)>;
199  DumpStackFctType dumpStack = [&](const o2::header::BaseHeader* h) {
200  o2::header::hexDump("", h, h->size());
201  if (h->flagsNextHeader) {
202  auto next = reinterpret_cast<const o2::byte*>(h) + h->size();
203  dumpStack(reinterpret_cast<const o2::header::BaseHeader*>(next));
204  }
205  };
206 
207  dumpStack(dh);
208 
209  if ((*iit).spec->binding == "inputMP") {
210  LOG(INFO) << "inputMP with " << iit.size() << " part(s)";
211  int nPart = 0;
212  for (auto const& ref : iit) {
213  LOG(INFO) << "accessing part " << nPart++ << " of input slot 'inputMP':"
214  << pc.inputs().get<int>(ref);
215  ASSERT_ERROR(pc.inputs().get<int>(ref) == nPart * 10);
216  }
217  ASSERT_ERROR(nPart == 3);
218  }
219  }
220  // plain, unserialized object in input1 channel
221  LOG(INFO) << "extracting o2::test::TriviallyCopyable from input1";
222  auto object1 = pc.inputs().get<o2::test::TriviallyCopyable>("input1");
223  ASSERT_ERROR(object1 == o2::test::TriviallyCopyable(42, 23, 0xdead));
224  LOG(INFO) << "extracting span of o2::test::TriviallyCopyable from input1";
225  auto object1span = pc.inputs().get<gsl::span<o2::test::TriviallyCopyable>>("input1");
226  ASSERT_ERROR(object1span.size() == 1);
227  ASSERT_ERROR(sizeof(typename decltype(object1span)::value_type) == sizeof(o2::test::TriviallyCopyable));
228  // check the additional header on the stack
229  auto* metaHeader1 = DataRefUtils::getHeader<test::MetaHeader*>(pc.inputs().get("input1"));
230  // check if there are more of the same type
231  auto* metaHeader2 = metaHeader1 ? o2::header::get<test::MetaHeader*>(metaHeader1->next()) : nullptr;
232  ASSERT_ERROR(metaHeader1 != nullptr);
233  ASSERT_ERROR(metaHeader1->secret == 42);
234  ASSERT_ERROR(metaHeader2 != nullptr && metaHeader2->secret == 23);
235 
236  // ROOT-serialized messageable object in input2 channel
237  LOG(INFO) << "extracting o2::test::TriviallyCopyable pointer from input2";
238  auto object2 = pc.inputs().get<o2::test::TriviallyCopyable*>("input2");
239  ASSERT_ERROR(object2 != nullptr);
240  ASSERT_ERROR(*object2 == o2::test::TriviallyCopyable(42, 23, 0xdead));
241 
242  // ROOT-serialized, non-messageable object in input3 channel
243  LOG(INFO) << "extracting o2::test::Polymorphic pointer from input3";
244  auto object3 = pc.inputs().get<o2::test::Polymorphic*>("input3");
245  ASSERT_ERROR(object3 != nullptr);
246  ASSERT_ERROR(*object3 == o2::test::Polymorphic(0xbeef));
247 
248  // container of objects
249  LOG(INFO) << "extracting vector of o2::test::Polymorphic from input4";
250  auto object4 = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input4");
251  ASSERT_ERROR(object4.size() == 2);
252  ASSERT_ERROR(object4[0] == o2::test::Polymorphic(0xaffe));
253  ASSERT_ERROR(object4[1] == o2::test::Polymorphic(0xd00f));
254 
255  // container of objects
256  LOG(INFO) << "extracting vector of o2::test::Polymorphic from input5";
257  auto object5 = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input5");
258  ASSERT_ERROR(object5.size() == 2);
259  ASSERT_ERROR(object5[0] == o2::test::Polymorphic(0xaffe));
260  ASSERT_ERROR(object5[1] == o2::test::Polymorphic(0xd00f));
261 
262  // container of objects
263  LOG(INFO) << "extracting vector of o2::test::Polymorphic from input6";
264  auto object6 = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input6");
265  ASSERT_ERROR(object6.size() == 2);
266  ASSERT_ERROR(object6[0] == o2::test::Polymorphic(0xaffe));
267  ASSERT_ERROR(object6[1] == o2::test::Polymorphic(0xd00f));
268 
269  // checking retrieving buffer as raw char*, and checking content by cast
270  LOG(INFO) << "extracting raw char* from input1";
271  auto rawchar = pc.inputs().get<const char*>("input1");
272  const auto& data1 = *reinterpret_cast<const o2::test::TriviallyCopyable*>(rawchar);
273  ASSERT_ERROR(data1 == o2::test::TriviallyCopyable(42, 23, 0xdead));
274 
275  LOG(INFO) << "extracting o2::test::TriviallyCopyable from input7";
276  auto object7 = pc.inputs().get<o2::test::TriviallyCopyable>("input7");
277  ASSERT_ERROR(object1 == o2::test::TriviallyCopyable(42, 23, 0xdead));
278 
279  LOG(INFO) << "extracting span of o2::test::TriviallyCopyable from input8";
280  auto objectspan8 = DataRefUtils::as<o2::test::TriviallyCopyable>(pc.inputs().get("input8"));
281  ASSERT_ERROR(objectspan8.size() == 3);
282  for (auto const& object8 : objectspan8) {
283  ASSERT_ERROR(object8 == o2::test::TriviallyCopyable(42, 23, 0xdead));
284  }
285 
286  LOG(INFO) << "extracting std::string from input9";
287  auto object9 = pc.inputs().get<std::string>("input9");
288  ASSERT_ERROR(object9 == "adoptchunk");
289 
290  LOG(INFO) << "extracting o2::test::TriviallyCopyable from input10";
291  auto object10 = pc.inputs().get<o2::test::TriviallyCopyable>("input10");
292  ASSERT_ERROR(object10 == o2::test::TriviallyCopyable(42, 23, 0xdead));
293 
294  LOG(INFO) << "extracting o2::test::TriviallyCopyable from input11";
295  auto object11 = pc.inputs().get<o2::test::TriviallyCopyable>("input11");
296  ASSERT_ERROR(object11 == o2::test::TriviallyCopyable(42, 23, 0xdead));
297 
298  LOG(INFO) << "extracting the original std::vector<o2::test::TriviallyCopyable> as span from input12";
299  auto object12 = pc.inputs().get<gsl::span<o2::test::TriviallyCopyable>>("input12");
300  ASSERT_ERROR(object12.size() == 2);
301  ASSERT_ERROR((object12[0] == o2::test::TriviallyCopyable{42, 23, 0xdead}));
302  ASSERT_ERROR((object12[1] == o2::test::TriviallyCopyable{10, 20, 0xacdc}));
303  // forward the read-only span on a different route
304  pc.outputs().snapshot(Output{"TST", "MSGABLVECTORCPY", 0, Lifetime::Timeframe}, object12);
305 
306  LOG(INFO) << "extracting TNamed object from input13";
307  auto object13 = pc.inputs().get<TNamed*>("input13");
308  ASSERT_ERROR(strcmp(object13->GetName(), "a_name") == 0);
309  ASSERT_ERROR(strcmp(object13->GetTitle(), "a_title") == 0);
310 
311  LOG(INFO) << "extracting Root-serialized Non-TObject from input14";
312  auto object14 = pc.inputs().get<o2::test::Polymorphic*>("input14");
313  ASSERT_ERROR(*object14 == o2::test::Polymorphic{0xbeef});
314 
315  LOG(INFO) << "extracting Root-serialized vector from input15";
316  auto object15 = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input15");
317  ASSERT_ERROR(object15[0] == o2::test::Polymorphic{0xacdc});
318  ASSERT_ERROR(object15[1] == o2::test::Polymorphic{0xbeef});
319 
320  LOG(INFO) << "extracting PMR vector";
321  auto pmrspan = pc.inputs().get<gsl::span<o2::test::TriviallyCopyable>>("inputPMR");
322  ASSERT_ERROR((pmrspan[0] == o2::test::TriviallyCopyable{1, 2, 3}));

324  auto header = o2::header::get<const o2::header::DataHeader*>(dataref.header);
