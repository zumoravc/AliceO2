36 void Clusterer::process(gsl::span<const Digit> digits, gsl::span<const TriggerRecord> dtr,
37  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* dmc,
38  std::vector<Cluster>* clusters, std::vector<TriggerRecord>* trigRec,
39  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* cluMC)
40 {
41  clusters->clear(); //final out list of clusters
42  trigRec->clear();
43  if (cluMC)
44  cluMC->clear();
45 
46  for (const auto& tr : dtr) {
47  mFirstDigitInEvent = tr.getFirstEntry();
48  mLastDigitInEvent = mFirstDigitInEvent + tr.getNumberOfObjects();
49  int indexStart = clusters->size();
50  mClusters.clear(); // internal list of FullClusters
51 
52  LOG(DEBUG) << "Starting clusteriztion digits from " << mFirstDigitInEvent << " to " << mLastDigitInEvent;
53 
54  if (!mBadMap) {
55  if (o2::cpv::CPVSimParams::Instance().mCCDBPath.compare("localtest") == 0) {
56  mBadMap = new BadChannelMap(1); // test default map
57  mCalibParams = new CalibParams(1); //test calibration map
58  LOG(INFO) << "No reading BadMap/Calibration from ccdb requested, set default";
59  } else {
60  LOG(INFO) << "Getting BadMap object from ccdb";
61  o2::ccdb::CcdbApi ccdb;
62  std::map<std::string, std::string> metadata; // do we want to store any meta data?
63  ccdb.init("http://ccdb-test.cern.ch:8080"); // or http://localhost:8080 for a local installation
64  long bcTime = 1; //TODO!!! Convert BC time to time o2::InteractionRecord bcTime = digitsTR.front().getBCData() ;
65  mBadMap = ccdb.retrieveFromTFileAny<o2::cpv::BadChannelMap>("CPV/BadMap", metadata, bcTime);
66  mCalibParams = ccdb.retrieveFromTFileAny<o2::cpv::CalibParams>("CPV/Calib", metadata, bcTime);
67  if (!mBadMap) {
68  LOG(FATAL) << "[CPVClusterer - run] can not get Bad Map";
69  }
70  if (!mCalibParams) {
71  LOG(FATAL) << "[CPVClusterer - run] can not get CalibParams";
72  }
73  }
74  }
75 
76  // Collect digits to clusters
77  makeClusters(digits);
78 
79  // Unfold overlapped clusters
80  // Split clusters with several local maxima if necessary
81  if (o2::cpv::CPVSimParams::Instance().mUnfoldClusters) {
82  makeUnfoldings(digits);
83  }
84 
85  // Calculate properties of collected clusters (Local position, energy, disp etc.)
86  evalCluProperties(digits, clusters, dmc, cluMC);
87 
88  LOG(DEBUG) << "Found clusters from " << indexStart << " to " << clusters->size();
89 
90  trigRec->emplace_back(tr.getBCData(), indexStart, clusters->size());
91  }
92 }

94 void Clusterer::makeClusters(gsl::span<const Digit> digits)
95 {
96  // A cluster is defined as a list of neighbour digits
97 
98  // Mark all digits as unused yet
99  const int maxNDigits = 12546; // There is no digits more than in CPV modules ;)
100  bool digitsUsed[maxNDigits];
101  memset(digitsUsed, 0, sizeof(bool) * maxNDigits);
102 
103  int iFirst = mFirstDigitInEvent; // first index of digit which potentially can be a part of cluster
104 
105  for (int i = iFirst; i < mLastDigitInEvent; i++) {
106  if (digitsUsed[i - mFirstDigitInEvent])
107  continue;
108 
109  const Digit& digitSeed = digits[i];
110  float digitSeedEnergy = calibrate(digitSeed.getAmplitude(), digitSeed.getAbsId());
111  if (isBadChannel(digitSeed.getAbsId())) {
112  digitSeedEnergy = 0.;
113  }
114  if (digitSeedEnergy < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
115  continue;
116  }
117 
118  // is this digit so energetic that start cluster?
119  FullCluster* clu = nullptr;
120  int iDigitInCluster = 0;
121  if (digitSeedEnergy > o2::cpv::CPVSimParams::Instance().mClusteringThreshold) {
122  // start new cluster
123  mClusters.emplace_back(digitSeed.getAbsId(), digitSeedEnergy, digitSeed.getLabel());
124  clu = &(mClusters.back());
125 
126  digitsUsed[i - mFirstDigitInEvent] = true;
127  iDigitInCluster = 1;
128  } else {
129  continue;
130  }
131  // Now scan remaining digits in list to find neigbours of our seed
132  int index = 0;
133  while (index < iDigitInCluster) { // scan over digits already in cluster
134  short digitSeedAbsId = clu->getDigitAbsId(index);
135  index++;
136  for (Int_t j = iFirst; j < mLastDigitInEvent; j++) {
137  if (digitsUsed[j - mFirstDigitInEvent])
138  continue; // look through remaining digits
139  const Digit* digitN = &(digits[j]);
140  float digitNEnergy = calibrate(digitN->getAmplitude(), digitN->getAbsId());
141  if (isBadChannel(digitN->getAbsId())) { //remove digit
142  digitNEnergy = 0.;
143  }
144  if (digitNEnergy < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
145  continue;
146  }
147 
148  // call (digit,digitN) in THAT oder !!!!!
149  Int_t ineb = Geometry::areNeighbours(digitSeedAbsId, digitN->getAbsId());
150  switch (ineb) {
151  case -1: // too early (e.g. previous module), do not look before j at subsequent passes
152  iFirst = j;
153  break;
154  case 0: // not a neighbour
155  break;
156  case 1: // are neighbours
157  clu->addDigit(digitN->getAbsId(), digitNEnergy, digitN->getLabel());
158  iDigitInCluster++;
159  digitsUsed[j - mFirstDigitInEvent] = true;
160  break;
161  case 2: // too far from each other
162  default:
163  break;
164  } // switch
165  }
166  } // loop over cluster
167  } // energy theshold
168 }

170 void Clusterer::makeUnfoldings(gsl::span<const Digit> digits)
171 {
172  //Split cluster if several local maxima are found
173 
174  std::array<int, NLMMax> maxAt; // NLMMax:Maximal number of local maxima
175 
176  int numberOfNotUnfolded = mClusters.size();
177 
178  for (int i = 0; i < numberOfNotUnfolded; i++) { //can not use iterator here as list can expand
179  FullCluster& clu = mClusters[i];
180  if (clu.getNExMax() > -1) { //already unfolded
181  continue;
182  }
183  char nMultipl = clu.getMultiplicity();
184  char nMax = clu.getNumberOfLocalMax(maxAt);
185  if (nMax > 1) {
186  unfoldOneCluster(clu, nMax, maxAt, digits);
187  clu.setEnergy(0); // will be skipped later
188  } else {
189  clu.setNExMax(nMax); // Only one local maximum
190  }
191  }

194 void Clusterer::unfoldOneCluster(FullCluster& iniClu, char nMax, gsl::span<int> digitId, gsl::span<const Digit> digits)
195 {
196  // Performs the unfolding of a cluster with nMax overlapping showers
197  // Parameters: iniClu cluster to be unfolded
198  // nMax number of local maxima found (this is the number of new clusters)
199  // digitId: index of digits, corresponding to local maxima
200  // maxAtEnergy: energies of digits, corresponding to local maxima
201 
202  // Take initial cluster and calculate local coordinates of digits
203  // To avoid multiple re-calculation of same parameters
204  char mult = iniClu.getMultiplicity();
205  if (meInClusters.capacity() < mult) {
206  meInClusters.reserve(mult);
207  mfij.reserve(mult);
208  }
209 
210  const std::vector<FullCluster::CluElement>* cluElist = iniClu.getElementList();
211 
212  // Coordinates of centers of clusters
213  std::array<float, NLMMax> xMax;
214  std::array<float, NLMMax> zMax;
215  std::array<float, NLMMax> eMax;
216  std::array<float, NLMMax> deNew;
217 
218  //transient variables
219  std::array<float, NLMMax> a;
220  std::array<float, NLMMax> b;
221  std::array<float, NLMMax> c;
222 
223  for (int iclu = 0; iclu < nMax; iclu++) {
224  xMax[iclu] = (*cluElist)[digitId[iclu]].localX;
225  zMax[iclu] = (*cluElist)[digitId[iclu]].localZ;
226  eMax[iclu] = 2. * (*cluElist)[digitId[iclu]].energy;
227  }
228 
229  std::array<float, NLMMax> prop; // proportion of clusters in the current digit
230 
231  // Try to decompose cluster to contributions
232  int nIterations = 0;
233  bool insuficientAccuracy = true;
234  while (insuficientAccuracy && nIterations < o2::cpv::CPVSimParams::Instance().mNMaxIterations) {
235  insuficientAccuracy = false; // will be true if at least one parameter changed too much
236  std::memset(&a, 0, sizeof a);
237  std::memset(&b, 0, sizeof b);
238  std::memset(&c, 0, sizeof c);
239  //First calculate shower shapes
240  for (int idig = 0; idig < mult; idig++) {
241  auto it = (*cluElist)[idig];
242  for (int iclu = 0; iclu < nMax; iclu++) {
243  mfij[idig][iclu] = responseShape(it.localX - xMax[iclu], it.localZ - zMax[iclu]);
244  }
245  }
246 
247  //Fit energies
248  for (int idig = 0; idig < mult; idig++) {
249  auto it = (*cluElist)[idig];
250  for (int iclu = 0; iclu < nMax; iclu++) {
251  a[iclu] += mfij[idig][iclu] * mfij[idig][iclu];
252  b[iclu] += it.energy * mfij[idig][iclu];
253  for (int kclu = 0; kclu < nMax; kclu++) {
254  if (iclu == kclu)
255  continue;
256  c[iclu] += eMax[kclu] * mfij[idig][iclu] * mfij[idig][kclu];
257  }
258  }
259  }
260  //Evaluate new maximal energies
261  for (int iclu = 0; iclu < nMax; iclu++) {
262  if (a[iclu] != 0.) {
263  float eNew = (b[iclu] - c[iclu]) / a[iclu];
264  insuficientAccuracy += (std::abs(eMax[iclu] - eNew) > eNew * o2::cpv::CPVSimParams::Instance().mUnfogingEAccuracy);
265  eMax[iclu] = eNew;
266  }
267  } // otherwise keep old value
268 
269  // Loop over all digits of parent cluster and split their energies between daughter clusters
270  // according to shower shape
271  // then re-evaluate local position of clusters
272  for (int idig = 0; idig < mult; idig++) {
273  float eEstimated = 0;
274  for (int iclu = 0; iclu < nMax; iclu++) {
275  prop[iclu] = eMax[iclu] * mfij[idig][iclu];
276  eEstimated += prop[iclu];
277  }
278  if (eEstimated == 0.) { // numerical accuracy
279  continue;
280  }
281  // Split energy of digit according to contributions
282  for (int iclu = 0; iclu < nMax; iclu++) {
283  meInClusters[idig][iclu] = (*cluElist)[idig].energy * prop[iclu] / eEstimated;
284  }
285  }
286 
287  // Recalculate parameters of clusters and check relative variation of energy and absolute of position
288  for (int iclu = 0; iclu < nMax; iclu++) {
289  float oldX = xMax[iclu];
290  float oldZ = zMax[iclu];
291  // full energy, need for weight
292  float eTotNew = 0;
293  for (int idig = 0; idig < mult; idig++) {
294  eTotNew += meInClusters[idig][iclu];
295  }
296  xMax[iclu] = 0;
297  zMax[iclu] = 0.;
298  float wtot = 0.;
299  for (int idig = 0; idig < mult; idig++) {
300  if (meInClusters[idig][iclu] > 0) {
301  // In unfolding it is better to use linear weight to reduce contribution of unfolded tails
302  float w = meInClusters[idig][iclu] / eTotNew;
303  // float w = std::max(std::log(eInClusters[idig][iclu] / eTotNew) + o2::cpv::CPVSimParams::Instance().mLogWeight, float(0.));
304  xMax[iclu] += (*cluElist)[idig].localX * w;
305  zMax[iclu] += (*cluElist)[idig].localZ * w;
306  wtot += w;
307  }
308  }
309  if (wtot > 0.) {
310  wtot = 1. / wtot;
311  xMax[iclu] *= wtot;
312  zMax[iclu] *= wtot;
313  }
314  // Compare variation of parameters
315  insuficientAccuracy += (std::abs(xMax[iclu] - oldX) > o2::cpv::CPVSimParams::Instance().mUnfogingXZAccuracy);
316  insuficientAccuracy += (std::abs(zMax[iclu] - oldZ) > o2::cpv::CPVSimParams::Instance().mUnfogingXZAccuracy);
317  }
318  nIterations++;
319  }
320  // Iterations finished, add new clusters
321  for (int iclu = 0; iclu < nMax; iclu++) {
322  mClusters.emplace_back();
323  FullCluster& clu = mClusters.back();
324  clu.setNExMax(nMax);
325  int idig = 0;
326  for (int idig = 0; idig < mult; idig++) {
327  float eDigit = meInClusters[idig][iclu];
328  idig++;
329  if (eDigit < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
330  continue;
331  }
332  clu.addDigit((*cluElist)[idig].absId, eDigit, (*cluElist)[idig].label);
333  }
334  }
335 }

338 void Clusterer::evalCluProperties(gsl::span<const Digit> digits, std::vector<Cluster>* clusters,
339  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* dmc,
340  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* cluMC)
341 {
342 
343  if (clusters->capacity() - clusters->size() < mClusters.size()) { //avoid expanding vector per element
344  clusters->reserve(clusters->size() + mClusters.size());
345  }
346 
347  int labelIndex = 0;
348  if (cluMC) {
349  labelIndex = cluMC->getIndexedSize();
350  }
351  auto clu = mClusters.begin();
352 
353  while (clu != mClusters.end()) {
354 
355  if (clu->getEnergy() < 1.e-4) { //Marked earlier for removal
356  ++clu;
357  continue;
358  }
359 
360  // may be soft digits remain after unfolding
361  clu->purify();
362 
363  // LOG(DEBUG) << "Purify done";
364  clu->evalAll();
365 
366  if (clu->getEnergy() > 1.e-4) { //Non-empty cluster
367  clusters->emplace_back(*clu);
368 
369  if (cluMC) { //Handle labels
370  //Calculate list of primaries
371  //loop over entries in digit MCTruthContainer
372  const std::vector<FullCluster::CluElement>* vl = clu->getElementList();
373  auto ll = vl->begin();
374  while (ll != vl->end()) {
375  int i = (*ll).label; //index
376  if (i < 0) {
377  ++ll;
378  continue;
379  }
380  gsl::span<const o2::MCCompLabel> spDigList = dmc->getLabels(i);
381  gsl::span<o2::MCCompLabel> spCluList = cluMC->getLabels(labelIndex); //get updated list
382  auto digL = spDigList.begin();
383  while (digL != spDigList.end()) {
384  bool exist = false;
385  auto cluL = spCluList.begin();
386  while (cluL != spCluList.end()) {
387  if (*digL == *cluL) { //exist
388  exist = true;
389  break;
390  }
391  ++cluL;
392  }
393  if (!exist) { //just add label
394  cluMC->addElement(labelIndex, (*digL));
395  }
396  ++digL;
397  }
398  ++ll;
399  }
400  clusters->back().setLabel(labelIndex);
401  labelIndex++;
402  } // Work with MC
403  }
404 
405  ++clu;
406  }
407 }
