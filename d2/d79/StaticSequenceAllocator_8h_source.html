66  template <typename BufferT>
67  constexpr size_t insert(BufferT* buffer) const noexcept
68  {
69  static_assert(sizeof(BufferT) == 1, "buffer required to be of byte-type");
70  size_t length = 0;
71  memcpy(buffer + length, &header, sizeof(HeaderType));
72  length += sizeof(HeaderType);
73  memcpy(buffer + length, data, dataLength);
74  length += dataLength;
75  if (trailerLength > 0) {
76  memcpy(buffer + length, &trailer, trailerLength);
77  length += trailerLength;
78  }
79  return length;
80  }
