78 DECLARE_SOA_COLUMN(Signed1Pt, signed1Pt, float);
79 DECLARE_SOA_DYNAMIC_COLUMN(Phi, phi, [](float snp, float alpha) -> float { return asinf(snp) + alpha + static_cast<float>(M_PI); });
80 DECLARE_SOA_DYNAMIC_COLUMN(Eta, eta, [](float tgl) -> float { return logf(tanf(0.25f * static_cast<float>(M_PI) - 0.5f * atanf(tgl))); });
81 DECLARE_SOA_DYNAMIC_COLUMN(Pt, pt, [](float signed1Pt) -> float { return std::abs(1.0f / signed1Pt); });
82 DECLARE_SOA_DYNAMIC_COLUMN(Charge, charge, [](float signed1Pt) -> short { return (signed1Pt > 0) ? 1 : -1; });
83 DECLARE_SOA_DYNAMIC_COLUMN(Px, px, [](float signed1Pt, float snp, float alpha) -> float {
84  auto pt = 1.f / std::abs(signed1Pt);
85  float cs, sn;
86  utils::sincosf(alpha, sn, cs);
87  auto r = std::sqrt((1.f - snp) * (1.f + snp));
88  return pt * (r * cs - snp * sn);
89 });
90 DECLARE_SOA_DYNAMIC_COLUMN(Py, py, [](float signed1Pt, float snp, float alpha) -> float {
91  auto pt = 1.f / std::abs(signed1Pt);
92  float cs, sn;
93  utils::sincosf(alpha, sn, cs);
94  auto r = std::sqrt((1.f - snp) * (1.f + snp));
95  return pt * (snp * cs + r * sn);
96 });
97 DECLARE_SOA_DYNAMIC_COLUMN(Pz, pz, [](float signed1Pt, float tgl) -> float {
98  auto pt = 1.f / std::abs(signed1Pt);
99  return pt * tgl;
100 });
101 DECLARE_SOA_DYNAMIC_COLUMN(P, p, [](float signed1Pt, float tgl) -> float {
102  return std::sqrt(1.f + tgl * tgl) / std::abs(signed1Pt);
103 });
104 DECLARE_SOA_DYNAMIC_COLUMN(P2, p2, [](float signed1Pt, float tgl) -> float {
105  return (1.f + tgl * tgl) / (signed1Pt * signed1Pt);
106 });
