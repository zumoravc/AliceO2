31 bool CanTypeFitValue(const U value)
32 {
33  const intmax_t botT = intmax_t(std::numeric_limits<T>::lowest());
34  const intmax_t botU = intmax_t(std::numeric_limits<U>::lowest());
35  const uintmax_t topT = uintmax_t(std::numeric_limits<T>::max());
36  const uintmax_t topU = uintmax_t(std::numeric_limits<U>::max());
37  return !((botT > botU && value < static_cast<U>(botT)) || (topT < topU && value > static_cast<U>(topT)));
