50 std::unique_ptr<VisualisationEvent> DataInterpreterITS::interpretDataForType(TObject* data, EVisualisationDataType type)
51 {
52  TList* list = (TList*)data;
53  Int_t event = ((TVector2*)list->At(2))->X();
54 
55  auto ret_event = std::make_unique<VisualisationEvent>(0, 0, 0, 0, "", 0);
56 
57  if (type == Clusters) {
58  its::GeometryTGeo* gman = its::GeometryTGeo::Instance();
59 
60  TFile* clustFile = (TFile*)list->At(1);
61  TTree* clusters = (TTree*)clustFile->Get("o2sim");
62 
63  //Read all clusters to a buffer
64  std::vector<itsmft::Cluster>* clusArr = nullptr;
65  clusters->SetBranchAddress("ITSCluster", &clusArr);
66 
67  std::vector<itsmft::ROFRecord>* clusterROFrames = nullptr;
68  clusters->SetBranchAddress("ITSClustersROF", &clusterROFrames);
69  clusters->GetEntry(0);
70 
71  auto currentClusterROF = clusterROFrames->at(event);
72 
73  int first, last;
74  first = currentClusterROF.getFirstEntry();
75  last = first + currentClusterROF.getNEntries();
76 
77  gsl::span<itsmft::Cluster> mClusters = gsl::make_span(&(*clusArr)[first], last - first);
78 
79  for (const auto& c : mClusters) {
80  const auto& gloC = c.getXYZGloRot(*gman);
81  double xyz[3] = {gloC.X(), gloC.Y(), gloC.Z()};
82  VisualisationCluster cluster(xyz);
83 
84  ret_event->addCluster(cluster);
85  }
86  } else if (type == ESD) {
87  TFile* trackFile = (TFile*)list->At(0);
88  TFile* clustFile = (TFile*)list->At(1);
89 
90  TTree* tracks = (TTree*)trackFile->Get("o2sim");
91 
92  TTree* clusters = (TTree*)clustFile->Get("o2sim");
93 
94  //Read all tracks to a buffer
95  std::vector<its::TrackITS>* trkArr = nullptr;
96  tracks->SetBranchAddress("ITSTrack", &trkArr);
97 
98  //Read all track RO frames to a buffer
99  std::vector<itsmft::ROFRecord>* trackROFrames = nullptr;
100  tracks->SetBranchAddress("ITSTracksROF", &trackROFrames);
101 
102  tracks->GetEntry(0);
103 
104  //Read all clusters to a buffer
105  std::vector<itsmft::Cluster>* clusArr = nullptr;
106  clusters->SetBranchAddress("ITSCluster", &clusArr);
107 
108  std::vector<itsmft::ROFRecord>* clusterROFrames = nullptr;
109  clusters->SetBranchAddress("ITSClustersROF", &clusterROFrames);
110  clusters->GetEntry(0);
111 
112  TEveTrackList* trackList = new TEveTrackList("tracks");
113  trackList->IncDenyDestroy();
114  auto prop = trackList->GetPropagator();
115  prop->SetMagField(0.5);
116  prop->SetMaxR(50.);
117 
118  auto currentTrackROF = trackROFrames->at(event);
119 
120  int first, last;
121  first = currentTrackROF.getFirstEntry();
122  last = first + currentTrackROF.getNEntries();
123 
124  gsl::span<its::TrackITS> mTracks = gsl::make_span(&(*trkArr)[first], last - first);
125 
126  for (const auto& rec : mTracks) {
127  std::array<float, 3> p;
128  rec.getPxPyPzGlo(p);
129  TEveRecTrackD t;
130  t.fP = {p[0], p[1], p[2]};
131  t.fSign = (rec.getSign() < 0) ? -1 : 1;
132  TEveTrack* eve_track = new TEveTrack(&t, prop);
133  eve_track->MakeTrack();
134 
135  auto start = eve_track->GetLineStart();
136  auto end = eve_track->GetLineEnd();
137  double track_start[3] = {start.fX, start.fY, start.fZ};
138  double track_end[3] = {end.fX, end.fY, end.fZ};
139  double track_p[3] = {p[0], p[1], p[2]};
140 
141  VisualisationTrack track(rec.getSign(), 0.0, 0, 0, 0.0, 0.0, track_start, track_end, track_p, 0, 0.0, 0.0, 0.0, 0);
142 
143  for (Int_t i = 0; i < eve_track->GetN(); ++i) {
144  Float_t x, y, z;
145  eve_track->GetPoint(i, x, y, z);
146  track.addPolyPoint(x, y, z);
147  }
148  delete eve_track;
149 
150  // TEvePointSet* tpoints = new TEvePointSet("tclusters");
151  // int nc = rec.getNumberOfClusters();
152  // while (nc--) {
153  // Int_t idx = rec.getClusterEntry(nc);
154  // itsmft::Cluster& c = (*clusArr)[idx];
155  // const auto& gloC = c.getXYZGloRot(*gman);
156  // tpoints->SetNextPoint(gloC.X(), gloC.Y(), gloC.Z());
157  // }
158 
159  ret_event->addTrack(track);
160  }
161  delete trackList;
162  }
163  return ret_event;
164 }
