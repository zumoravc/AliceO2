25 void ELinkDecoder::add(uint8_t byte)
26 {
28  mBytes.emplace_back(byte);
29  if (mBytes.size() == sMinimumSize) {
30  if (raw::isLoc(mBytes[0])) {
31  // This is a local card
32  uint8_t mask = getInputs();
33  for (int ich = 0; ich < 4; ++ich) {
34  if ((mask >> ich) & 0x1) {
35  // We expect 2 bytes for the BP and 2 for the NBP
36  mTotalSize += 4;
37  }
38  }
39  }
40  }
41 }

60 uint16_t ELinkDecoder::getPattern(int cathode, int chamber) const
61 {
63  uint8_t mask = getInputs();
64  if (((mask >> chamber) & 0x1) == 0) {
65  return 0;
66  }
67 
68  int idx = mBytes.size();
69  for (int ich = 0; ich <= chamber; ++ich) {
70  if ((mask >> ich) & 0x1) {
71  idx -= 4;
72  }
73  }
74 
75  idx += 2 * cathode;
76 
77  return joinBytes(idx);
78 }
