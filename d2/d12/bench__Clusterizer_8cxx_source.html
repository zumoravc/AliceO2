37 bool addStrip(o2::mid::ColumnData& column, int cathode, int line, int strip)
38 {
39  uint16_t pattern = column.getPattern(cathode, line);
40  uint16_t currStrip = (1 << strip);
41  if (pattern & currStrip) {
42  return false;
43  }
44  column.addStrip(strip, cathode, line);
45  return true;
46 }

48 void addNeighbour(std::vector<o2::mid::ColumnData>& patterns, o2::mid::Mapping::MpStripIndex stripIndex, int cathode,
49  int deId, const o2::mid::Mapping& midMapping, int& nAdded, int maxAdded)
50 {
51  std::vector<o2::mid::Mapping::MpStripIndex> neighbours = midMapping.getNeighbours(stripIndex, cathode, deId);
52  for (auto& neigh : neighbours) {
53  o2::mid::ColumnData& column = getColumn(patterns, static_cast<uint8_t>(neigh.column), static_cast<uint8_t>(deId));
54  if (!addStrip(column, cathode, neigh.line, neigh.strip)) {
55  continue;
56  }
57  ++nAdded;
58  if (nAdded >= maxAdded) {
59  return;
60  }
61  addNeighbour(patterns, neigh, cathode, deId, midMapping, nAdded, maxAdded);
62  if (nAdded >= maxAdded) {
63  return;
64  }
65  }
66 }

68 std::vector<o2::mid::ColumnData> generateTestData(int deId, int nClusters, int clusterSize,
69  const o2::mid::Mapping& midMapping)
70 {
71  int firstColumnInDE = midMapping.getFirstColumn(deId);
72 
73  std::uniform_int_distribution<int> distColumn(firstColumnInDE, 6);
74 
75  std::random_device rd;
76  std::mt19937 mt(rd());
77 
78  std::vector<o2::mid::ColumnData> patterns;
79  o2::mid::Mapping::MpStripIndex stripIndex;
80  std::vector<o2::mid::Mapping::MpStripIndex> neighbours;
81 
82  for (int icl = 0; icl < nClusters; ++icl) {
83  int icolumn = distColumn(mt);
84  for (int cathode = 0; cathode < 2; ++cathode) {
85  int iline = (cathode == 1) ? 0 : midMapping.getFirstBoardBP(icolumn, deId);
86  int nStrips = (cathode == 0) ? 16 : midMapping.getNStripsNBP(icolumn, deId);
87  std::uniform_int_distribution<int> distStrip(0, nStrips - 1);
88  stripIndex.column = icolumn;
89  stripIndex.line = iline;
90  stripIndex.strip = distStrip(mt);
91  o2::mid::ColumnData& column = getColumn(patterns, static_cast<uint8_t>(icolumn), static_cast<uint8_t>(deId));
92  addStrip(column, cathode, iline, stripIndex.strip);
93  int nAdded = 1;
94  if (nAdded < clusterSize) {
95  addNeighbour(patterns, stripIndex, cathode, deId, midMapping, nAdded, clusterSize);
96  }
97  }
98  }
99  return patterns;
100 }
