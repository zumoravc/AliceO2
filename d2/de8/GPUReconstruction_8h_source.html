410 template <class T>
411 inline short GPUReconstruction::RegisterMemoryAllocation(T* proc, void* (T::*setPtr)(void*), int type, const char* name, const GPUMemoryReuse& re)
412 {
413  if (!(type & (GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU))) {
414  if ((type & GPUMemoryResource::MEMORY_SCRATCH) && !mDeviceProcessingSettings.keepDisplayMemory) { // keepAllMemory --> keepDisplayMemory
415  type |= (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU ? GPUMemoryResource::MEMORY_HOST : GPUMemoryResource::MEMORY_GPU);
416  } else {
417  type |= GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU;
418  }
419  }
420  if (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU) {
421  type &= ~GPUMemoryResource::MEMORY_GPU;
422  }
423  mMemoryResources.emplace_back(proc, static_cast<void* (GPUProcessor::*)(void*)>(setPtr), (GPUMemoryResource::MemoryType)type, name);
424  if (mMemoryResources.size() >= 32768) {
425  throw std::bad_alloc();
426  }
427  unsigned short retVal = mMemoryResources.size() - 1;
428  if (re.type != GPUMemoryReuse::NONE) {
429  const auto& it = mMemoryReuse1to1.find(re.id);
430  if (it == mMemoryReuse1to1.end()) {
431  mMemoryReuse1to1[re.id] = {proc, {retVal}};
432  } else {
433  mMemoryResources[retVal].mReuse = it->second.res[0];
434  it->second.res.emplace_back(retVal);
435  }
436  }
437  return retVal;
438 }

466 template <class T, class S>
467 inline void GPUReconstruction::DumpData(FILE* fp, const T* const* entries, const S* num, InOutPointerType type)
468 {
469  int count = getNIOTypeMultiplicity(type);
470  unsigned int numTotal = 0;
471  for (int i = 0; i < count; i++) {
472  numTotal += num[i];
473  }
474  if (numTotal == 0) {
475  return;
476  }
477  fwrite(&type, sizeof(type), 1, fp);
478  for (int i = 0; i < count; i++) {
479  fwrite(&num[i], sizeof(num[i]), 1, fp);
480  if (num[i]) {
481  fwrite(entries[i], sizeof(*entries[i]), num[i], fp);
482  }
483  }
484 }

486 template <class T, class S>
487 inline size_t GPUReconstruction::ReadData(FILE* fp, const T** entries, S* num, std::unique_ptr<T[]>* mem, InOutPointerType type)
488 {
489  if (feof(fp)) {
490  return 0;
491  }
492  InOutPointerType inType;
493  size_t r, pos = ftell(fp);
494  r = fread(&inType, sizeof(inType), 1, fp);
495  if (r != 1 || inType != type) {
496  fseek(fp, pos, SEEK_SET);
497  return 0;
498  }
499 
500  int count = getNIOTypeMultiplicity(type);
501  size_t numTotal = 0;
502  for (int i = 0; i < count; i++) {
503  r = fread(&num[i], sizeof(num[i]), 1, fp);
504  AllocateIOMemoryHelper(num[i], entries[i], mem[i]);
505  if (num[i]) {
506  r = fread(mem[i].get(), sizeof(*entries[i]), num[i], fp);
507  }
508  numTotal += num[i];
509  }
510  (void)r;
511  if (mDeviceProcessingSettings.debugLevel >= 2) {
512  GPUInfo("Read %d %s", (int)numTotal, IOTYPENAMES[type]);
513  }
514  return numTotal;
515 }

531 template <class T>
532 inline std::unique_ptr<T> GPUReconstruction::ReadFlatObjectFromFile(const char* file)
533 {
534  FILE* fp = fopen(file, "rb");
535  if (fp == nullptr) {
536  return nullptr;
537  }
538  size_t size[2] = {0}, r;
539  r = fread(size, sizeof(size[0]), 2, fp);
540  if (r == 0 || size[0] != sizeof(T)) {
541  fclose(fp);
542  GPUError("ERROR reading %s, invalid size: %lld (%lld expected)", file, (long long int)size[0], (long long int)sizeof(T));
543  throw std::runtime_error("invalid size");
544  }
545  std::unique_ptr<T> retVal(new T);
546  retVal->destroy();
547  char* buf = new char[size[1]]; // Not deleted as ownership is transferred to FlatObject
548  r = fread((void*)retVal.get(), 1, size[0], fp);
