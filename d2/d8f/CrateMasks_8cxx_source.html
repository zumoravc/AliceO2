72 bool CrateMasks::load(const char* filename)
73 {
79  mActiveBoards.fill(0);
80  std::ifstream inFile(filename);
81  if (!inFile.is_open()) {
82  return false;
83  }
84  std::string line, token;
85  while (std::getline(inFile, line)) {
86  if (std::count(line.begin(), line.end(), ' ') < 1) {
87  continue;
88  }
89  if (line.find('#') < line.find(' ')) {
90  continue;
91  }
92  std::stringstream ss;
93  ss << line;
94  std::getline(ss, token, ' ');
95  uint16_t feeId = std::atoi(token.c_str());
96  std::getline(ss, token, ' ');
97  uint8_t mask = std::atoi(token.c_str());
98  mActiveBoards[feeId] = mask;
99  }
100  inFile.close();
101  return true;
102 }
