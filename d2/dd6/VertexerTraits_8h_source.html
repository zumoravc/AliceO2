200 GPUhdi() const int4 VertexerTraits::getBinsRect(const Cluster& currentCluster, const int layerIndex,
201  const float directionZIntersection, float maxdeltaz, float maxdeltaphi)
202 {
203  const float zRangeMin = directionZIntersection - 2 * maxdeltaz;
204  const float phiRangeMin = currentCluster.phiCoordinate - maxdeltaphi;
205  const float zRangeMax = directionZIntersection + 2 * maxdeltaz;
206  const float phiRangeMax = currentCluster.phiCoordinate + maxdeltaphi;
207 
208  if (zRangeMax < -constants::its::LayersZCoordinate()[layerIndex + 1] ||
209  zRangeMin > constants::its::LayersZCoordinate()[layerIndex + 1] || zRangeMin > zRangeMax) {
210 
211  return getEmptyBinsRect();
212  }
213 
214  return int4{gpu::GPUCommonMath::Max(0, index_table_utils::getZBinIndex(layerIndex + 1, zRangeMin)),
215  index_table_utils::getPhiBinIndex(math_utils::getNormalizedPhiCoordinate(phiRangeMin)),
216  gpu::GPUCommonMath::Min(constants::index_table::ZBins - 1, index_table_utils::getZBinIndex(layerIndex + 1, zRangeMax)),
217  index_table_utils::getPhiBinIndex(math_utils::getNormalizedPhiCoordinate(phiRangeMax))};
218 }
