101 Bool_t Detector::ProcessHits(FairVolume* vol)
102 {
103  mStepCounter++;
104  auto& gasParam = ParameterGas::Instance();
105 
106  /* This method is called from the MC stepping for the sensitive volume only */
107  // LOG(INFO) << "tpc::ProcessHits";
108  const double trackCharge = fMC->TrackCharge();
109  if (static_cast<int>(trackCharge) == 0) {
110 
111  // set a very large step size for neutral particles
112  fMC->SetMaxStep(1.e10);
113  return kFALSE; // take only charged particles
114  }
115 
116  // ===| SET THE LENGTH OF THE NEXT ENERGY LOSS STEP |=========================
117  // (We do this first so we can skip out of the method in the following
118  // Eloss->Ionization part.)
119  //
120  // In all cases we have multiple collisions and we use 2mm (+ some
121  // random shift to avoid binning effects), which was tuned for GEANT4, see
122  // https://indico.cern.ch/event/316891/contributions/732168/
123 
124  const double rnd = fMC->GetRandom()->Rndm();
125  fMC->SetMaxStep(0.2 + (2. * rnd - 1.) * 0.05); // 2 mm +- rndm*0.5mm step
126 
127  // ===| check active sector |=================================================
128  //
129  // Get the sector ID and check if the sector is active
130  static thread_local TLorentzVector position;
131  fMC->TrackPosition(position);
132  // for processing reasons in the digitizer, the sectors are shifted by -10deg, so sector 0 will be
133  // from -10 - +10 deg instead of 0-20 and so on
134  const int sectorID = static_cast<int>(Sector::ToShiftedSector(position.X(), position.Y(), position.Z()));
135  // const int sectorID = static_cast<int>(Sector::ToSector(position.X(), position.Y(), position.Z()));
136  // TODO: Temporary hack to process only one sector
137  // if (sectorID != 0) return kFALSE;
138 
139  // ---| momentum and beta gamma |---
140  static TLorentzVector momentum; // static to make avoid creation/deletion of this expensive object
141  fMC->TrackMomentum(momentum);
142 
143  const float time = fMC->TrackTime() * 1.0e9;
144  const int trackID = fMC->GetStack()->GetCurrentTrackNumber();
145  const int detID = vol->getMCid();
146  o2::data::Stack* stack = (o2::data::Stack*)fMC->GetStack();
147  if (fMC->IsTrackEntering() || fMC->IsTrackExiting()) {
148  stack->addTrackReference(o2::TrackReference(position.X(), position.Y(), position.Z(), momentum.X(), momentum.Y(),
149  momentum.Z(), fMC->TrackLength(), time, trackID, GetDetId()));
150  }
151 
152  // ===| CONVERT THE ENERGY LOSS TO IONIZATION ELECTRONS |=====================
153  //
154  // The energy loss is implemented directly below and taken GEANT3,
155  // ILOSS model 5 (in gfluct.F), which gives
156  // the energy loss in a single collision (NA49 model).
157  // TODO: Add discussion about drawback
158 
159  Int_t numberOfElectrons = 0;
160  // I.H. - the type expected in addHit is short
161 
162  // ---| Stepsize in cm |---
163  const double stepSize = fMC->TrackStep();
164 
165  double betaGamma = momentum.P() / fMC->TrackMass();
166  betaGamma = TMath::Max(betaGamma, 7.e-3); // protection against too small bg
167 
168  // ---| number of primary ionisations per cm |---
169  const double primaryElectronsPerCM =
170  gasParam.Nprim * BetheBlochAleph(static_cast<float>(betaGamma), gasParam.BetheBlochParam[0],
171  gasParam.BetheBlochParam[1], gasParam.BetheBlochParam[2],
172  gasParam.BetheBlochParam[3], gasParam.BetheBlochParam[4]);
173 
174  // ---| mean number of collisions and random for this event |---
175  const double meanNcoll = stepSize * trackCharge * trackCharge * primaryElectronsPerCM;
176  const int nColl = static_cast<int>(fMC->GetRandom()->Poisson(meanNcoll));
177 
178  // Variables needed to generate random powerlaw distributed energy loss
179  const double alpha_p1 = 1. - gasParam.Exp; // NA49/G3 value
180  const double oneOverAlpha_p1 = 1. / alpha_p1;
181  const double eMin = gasParam.Ipot;
182  const double eMax = gasParam.Eend;
183  const double kMin = TMath::Power(eMin, alpha_p1);
184  const double kMax = TMath::Power(eMax, alpha_p1);
185  const double wIon = gasParam.Wion;
186 
187  for (Int_t n = 0; n < nColl; n++) {
188  // Use GEANT3 / NA49 expression:
189  // P(eDep) ~ k * edep^-gasParam.getExp()
190  // eMin(~I) < eDep < eMax(300 electrons)
191  // k fixed so that Int_Emin^EMax P(Edep) = 1.
192  const double rndm = fMC->GetRandom()->Rndm();
193  const double eDep = TMath::Power((kMax - kMin) * rndm + kMin, oneOverAlpha_p1);
194  int nel_step = static_cast<int>(((eDep - eMin) / wIon) + 1);
195  nel_step = TMath::Min(nel_step, 300); // 300 electrons corresponds to 10 keV
196  numberOfElectrons += nel_step;
197  }
198 
199  // LOG(INFO) << "tpc::AddHit" << FairLogger::endl << "Eloss: "
200  //<< fMC->Edep() << ", Nelectrons: "
201  //<< numberOfElectrons;
202 
203  if (numberOfElectrons <= 0) // Could maybe be smaller than 0 due to the Gamma function
204  return kFALSE;
205 
206  // ADD HIT
207  static thread_local int oldTrackId = trackID;
208  static thread_local int oldDetId = detID;
209  static thread_local int groupCounter = 0;
210  static thread_local int oldSectorId = sectorID;
211 
212  // a new group is starting -> put it into the container
213  static thread_local HitGroup* currentgroup = nullptr;
214  if (groupCounter == 0) {
215  mHitsPerSectorCollection[sectorID]->emplace_back(trackID);
216  currentgroup = &(mHitsPerSectorCollection[sectorID]->back());
217  }
218  if (trackID == oldTrackId && oldSectorId == sectorID) {
219  groupCounter++;
220  mHitCounter++;
221  mElectronCounter += numberOfElectrons;
222  currentgroup->addHit(position.X(), position.Y(), position.Z(), time, numberOfElectrons);
223  }
224  // finish group
225  else {
226  oldTrackId = trackID;
227  oldSectorId = sectorID;
228  groupCounter = 0;
229  }
230 
231  // LOG(INFO) << "tpc::AddHit" << FairLogger::endl
232  //<< " -- " << trackNumberID <<"," << volumeID << " " << vol->GetName()
233  //<< ", Pos: (" << position.X() << ", " << position.Y() <<", "<< position.Z()<< ", " << r << ") "
234  //<< ", Mom: (" << momentum.Px() << ", " << momentum.Py() << ", " << momentum.Pz() << ") "
235  //<< " Time: "<< time <<", Len: " << length << ", Nelectrons: " <<
236  // numberOfElectrons;
237  // I.H. - the code above does not compile if uncommented
238 
239  // Increment number of Detector det points in TParticle
240  stack->addHit(GetDetId());
241 
242  return kTRUE;
243 }
