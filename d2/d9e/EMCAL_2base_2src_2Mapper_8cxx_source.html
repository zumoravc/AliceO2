33 void Mapper::init(const std::string_view filename)
34 {
35  std::ifstream reader(filename.data());
36  std::string tmp;
37 
38  int maxHardwareAddress(-1);
39  try {
40  std::getline(reader, tmp); // max. number of entries - no longer needed
41  std::getline(reader, tmp);
42  maxHardwareAddress = std::stoi(tmp);
43  } catch (std::invalid_argument& e) {
44  throw FileFormatException(e.what());
45  } catch (std::out_of_range& e) {
46  throw FileFormatException(e.what());
47  } catch (std::iostream::failure& e) {
48  throw FileFormatException(e.what());
49  }
50 
51  // loop over channels
52  int address, row, col, caloflag;
53  while (getline(reader, tmp)) {
54  std::stringstream addressmapping(tmp);
55 
56  try {
57  addressmapping >> address >> row >> col >> caloflag;
58  } catch (std::iostream::failure& e) {
59  throw FileFormatException(e.what());
60  }
61 
62  if (address > maxHardwareAddress)
63  throw AddressRangeException(address, maxHardwareAddress);
64 
65  auto chantype = o2::emcal::intToChannelType(caloflag);
66 
67  mMapping.insert(std::pair<int, ChannelID>(address, {uint8_t(row), uint8_t(col), chantype}));
68  mInverseMapping.insert(std::pair<ChannelID, int>({uint8_t(row), uint8_t(col), chantype}, address));
69  }
70 
71  mInitStatus = true;
72 }
