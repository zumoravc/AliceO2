29 void DigitReader::readNextEntry()
30 {
31  assert(mDigitsPerIR.empty());
32 
33  mDigitBranch->GetEntry(mEntry);
34  mEntry++;
35 
36  // builds a list of digits per HBF
37  const o2::raw::HBFUtils& hbfutils = o2::raw::HBFUtils::Instance();
38 
39  for (auto d : (*mDigits)) {
40  o2::InteractionTimeRecord ir(d.getTime().sampaTime);
41  // get the closest (previous) HBF from this IR
42  // and assign the digits to that one
43  auto hbf = hbfutils.getIRHBF(hbfutils.getHBF(ir));
44  mDigitsPerIR[hbf].push_back(d);
45  }
46 }

48 bool DigitReader::nextDigits(InteractionRecord& ir, std::vector<o2::mch::Digit>& digits)
49 {
50  if (mIterator == mDigitsPerIR.end()) {
51  if (mEntry >= mDigitBranch->GetEntries()) {
52  return false;
53  }
54  readNextEntry();
55  mIterator = mDigitsPerIR.begin();
56  }
57  ir = mIterator->first;
58  digits = mIterator->second;
59  mIterator++;
60  return true;
61 }
