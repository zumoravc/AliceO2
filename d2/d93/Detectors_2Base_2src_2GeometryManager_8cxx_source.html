48 Bool_t GeometryManager::getOriginalMatrix(const char* symname, TGeoHMatrix& m)
49 {
50  m.Clear();
51 
52  if (!gGeoManager || !gGeoManager->IsClosed()) {
53  LOG(ERROR) << "No active geometry or geometry not yet closed!";
54  ;
55  return kFALSE;
56  }
57 
58  if (!gGeoManager->GetListOfPhysicalNodes()) {
59  LOG(WARNING) << "gGeoManager doesn't contain any aligned nodes!";
60 
61  if (!gGeoManager->cd(symname)) {
62  LOG(ERROR) << "Volume path " << symname << " not valid!";
63  return kFALSE;
64  } else {
65  m = *gGeoManager->GetCurrentMatrix();
66  return kTRUE;
67  }
68  }
69 
70  TGeoPNEntry* pne = gGeoManager->GetAlignableEntry(symname);
71  const char* path = nullptr;
72 
73  if (pne) {
74  m = *pne->GetGlobalOrig();
75  return kTRUE;
76  } else {
77  LOG(WARNING) << "The symbolic volume name " << symname
78  << "does not correspond to a physical entry. Using it as a volume path!";
79  path = symname;
80  }
81 
82  return getOriginalMatrixFromPath(path, m);
83 }

86 Bool_t GeometryManager::getOriginalMatrixFromPath(const char* path, TGeoHMatrix& m)
87 {
88  m.Clear();
89 
90  if (!gGeoManager || !gGeoManager->IsClosed()) {
91  LOG(ERROR) << "Can't get the original global matrix! gGeoManager doesn't exist or it is still opened!";
92  return kFALSE;
93  }
94 
95  if (!gGeoManager->CheckPath(path)) {
96  LOG(ERROR) << "Volume path " << path << " not valid!";
97  return kFALSE;
98  }
99 
100  TIter next(gGeoManager->GetListOfPhysicalNodes());
101  gGeoManager->cd(path);
102 
103  while (gGeoManager->GetLevel()) {
104  TGeoPhysicalNode* physNode = nullptr;
105  next.Reset();
106  TGeoNode* node = gGeoManager->GetCurrentNode();
107 
108  while ((physNode = (TGeoPhysicalNode*)next())) {
109  if (physNode->GetNode() == node) {
110  break;
111  }
112  }
113 
114  TGeoMatrix* lm = nullptr;
115  if (physNode) {
116  lm = physNode->GetOriginalMatrix();
117  if (!lm) {
118  lm = node->GetMatrix();
119  }
120  } else {
121  lm = node->GetMatrix();
122  }
123 
124  m.MultiplyLeft(lm);
125 
126  gGeoManager->CdUp();
127  }
128  return kTRUE;
129 }

170

175 
176  const char* path = pne->GetTitle();
177  gGeoManager->PushPath(); // Preserve the modeler state.
178  if (!gGeoManager->cd(path)) {
179  gGeoManager->PopPath();
180  LOG(ERROR) << "Volume path " << path << " not valid!";
181  return nullptr;
182  }
183  matTmp = *gGeoManager->GetCurrentMatrix();
184  gGeoManager->PopPath();
185  return &matTmp;

205 bool GeometryManager::applyAlignment(TObjArray& algParArray, bool ovlpcheck, double ovlToler)
206 {
208 
209  algParArray.Sort(); // sort to apply alignment in correct hierarchy
210 
211  int nvols = algParArray.GetEntriesFast();
212  bool res = true;
213  for (int i = 0; i < nvols; i++) {
214  AlignParam* alg = dynamic_cast<AlignParam*>(algParArray[i]);
215  if (alg) {
216  if (!alg->applyToGeometry(ovlpcheck, ovlToler)) {
217  res = false;
218  LOG(ERROR) << "Error applying alignment object for volume" << alg->getSymName();
219  }
220  }
221  }
222  return res;
223 }

241 void GeometryManager::accountMaterial(const TGeoMaterial* material, GeometryManager::MatBudgetExt& bd)
242 {
243  bd.meanRho = material->GetDensity();
244  bd.meanX2X0 = material->GetRadLen();
245  bd.meanA = material->GetA();
246  bd.meanZ = material->GetZ();
247  if (material->IsMixture()) {
248  TGeoMixture* mixture = (TGeoMixture*)material;
249  Double_t norm = 0.;
250  bd.meanZ2A = 0.;
251  for (Int_t iel = 0; iel < mixture->GetNelements(); iel++) {
252  norm += mixture->GetWmixt()[iel];
253  bd.meanZ2A += mixture->GetZmixt()[iel] * mixture->GetWmixt()[iel] / mixture->GetAmixt()[iel];
254  }
255  bd.meanZ2A /= norm;
256  } else {
257  bd.meanZ2A = bd.meanZ / bd.meanA;
258  }
259 }

262 GeometryManager::MatBudgetExt GeometryManager::meanMaterialBudgetExt(float x0, float y0, float z0, float x1, float y1, float z1)
263 {
264  //
265  // TODO? It seems there is no real nead for extended material budget, consider eliminating it
266  //
267  // Calculate mean material budget and material properties (extended version) between
268  // the points "0" and "1".
269  //
270  // see MatBudgetExt data members for provided information
271  //
272  // Origin: Marian Ivanov, Marian.Ivanov@cern.ch
273  //
274  // Corrections and improvements by
275  // Andrea Dainese, Andrea.Dainese@lnl.infn.it,
276  // Andrei Gheata, Andrei.Gheata@cern.ch
277  //
278  // Ported to O2: ruben.shahoyan@cern.ch
279  //
280 
281  double length, startD[3] = {x0, y0, z0};
282  double dir[3] = {x1 - x0, y1 - y0, z1 - z0};
283  if ((length = dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]) < TGeoShape::Tolerance() * TGeoShape::Tolerance()) {
284  return MatBudgetExt(); // return empty struct
285  }
286  length = TMath::Sqrt(length);
287  double invlen = 1. / length;
288  for (int i = 3; i--;) {
289  dir[i] *= invlen;
290  }
291 
292  // Initialize start point and direction
293  TGeoNode* currentnode = gGeoManager->InitTrack(startD, dir);
294  if (!currentnode) {
295  LOG(ERROR) << "start point out of geometry: " << x0 << ':' << y0 << ':' << z0;
296  return MatBudgetExt(); // return empty struct
297  }
298 
299  MatBudgetExt budTotal, budStep;
300  accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial(), budStep);
301  budStep.length = length;
302 
303  // Locate next boundary within length without computing safety.
304  // Propagate either with length (if no boundary found) or just cross boundary
305  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
306  Double_t stepTot = 0.0; // Step made
307  Double_t step = gGeoManager->GetStep();
308  // If no boundary within proposed length, return current step data
309  if (!gGeoManager->IsOnBoundary()) {
310  budStep.meanX2X0 = budStep.length / budStep.meanX2X0;
311  return MatBudgetExt(budStep);
312  }
313  // Try to cross the boundary and see what is next
314  Int_t nzero = 0;
315  while (length > TGeoShape::Tolerance()) {
316  if (step < 2. * TGeoShape::Tolerance()) {
317  nzero++;
318  } else {
319  nzero = 0;
320  }
321  if (nzero > 3) {
322  // This means navigation has problems on one boundary
323  // Try to cross by making a small step
324  const double* curPos = gGeoManager->GetCurrentPoint();
325  LOG(ERROR) << "Cannot cross boundary at (" << curPos[0] << ',' << curPos[1] << ',' << curPos[2] << ')';
326  budTotal.normalize(stepTot);
327  budTotal.nCross = -1; // flag failed navigation
328  return MatBudgetExt(budTotal);
329  }
330  stepTot += step;
331 
332  budTotal.meanRho += step * budStep.meanRho;
333  budTotal.meanX2X0 += step / budStep.meanX2X0;
334  budTotal.meanA += step * budStep.meanA;
335  budTotal.meanZ += step * budStep.meanZ;
336  budTotal.meanZ2A += step * budStep.meanZ2A;
337  budTotal.nCross++;
338 
339  if (step >= length) {
340  break;
341  }
342  currentnode = gGeoManager->GetCurrentNode();
343  if (!currentnode) {
344  break;
345  }
346  length -= step;
347  accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial(), budStep);
348  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
349  step = gGeoManager->GetStep();
350  }
351  budTotal.normalize(stepTot);
352  return MatBudgetExt(budTotal);
353 }
