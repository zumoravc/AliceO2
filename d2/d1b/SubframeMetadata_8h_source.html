34 // Helper function to derive the timeframe id from the actual timestamp.
35 // Timestamp is in nanoseconds. Each Timeframe is ~22ms i.e. 2^17 nanoseconds,
36 // so we can get a unique id by dividing by the timeframe period and masking
37 // the lower 16 bits. Overlaps will only happen every ~ 22 minutes.
38 constexpr uint16_t
39  timeframeIdFromTimestamp(uint64_t timestamp, uint64_t timeFrameDuration)
40 {
41  return (timestamp / timeFrameDuration) & 0xffff;
42 }
