24 BOOST_AUTO_TEST_CASE(Cartesian3D_test)
25 {
26  // we create Transform3D by conversion from TGeoHMatrix
27  TGeoRotation rotg("r", 10., 20., 30.);
28  TGeoTranslation trag("g", 100., 200., 300.);
29  TGeoHMatrix hmat = trag;
30  hmat *= rotg;
31 
32  Transform3D tr(hmat);
33  Point3D<double> pd(10., 20., 30.);
34  Point3D<float> pf(10.f, 20.f, 30.f);
35  //
36  // local to master
37  auto pdt = tr(pd); // operator form
38  Point3D<float> pft;
39  tr.LocalToMaster(pf, pft); // TGeoHMatrix form
40 
41  std::cout << "Create Transform3D " << std::endl
42  << tr << std::endl
43  << "from" << std::endl;
44  hmat.Print();
45 
46  std::cout << " Transforming " << pd << " to master" << std::endl;
47  // compare difference between float and double vector transform
48  std::cout << "Float: " << pft << std::endl;
49  std::cout << "Double: " << pdt << std::endl;
50  std::cout << "Diff: " << pdt.X() - pft.X() << "," << pdt.Y() - pft.Y() << "," << pdt.Z() - pft.Z() << std::endl;
51 
52  const double toler = 1e-4;
53  BOOST_CHECK(std::abs(pdt.X() - pft.X()) < toler);
54  BOOST_CHECK(std::abs(pdt.Y() - pft.Y()) < toler);
55  BOOST_CHECK(std::abs(pdt.Z() - pft.Z()) < toler);
56 
57  // inverse transform
58  auto pfti = tr ^ (pft); // operator form
59  Point3D<double> pdti;
60  tr.MasterToLocal(pdt, pdti); // TGeoHMatrix form
61 
62  std::cout << " Transforming back to local" << std::endl;
63  std::cout << "Float: " << pfti << std::endl;
64  std::cout << "Double: " << pdti << std::endl;
65  std::cout << "Diff: " << pd.X() - pfti.X() << ", " << pd.Y() - pfti.Y() << ", " << pd.Z() - pfti.Z() << std::endl;
66 
67  BOOST_CHECK(std::abs(pd.X() - pfti.X()) < toler);
68  BOOST_CHECK(std::abs(pd.Y() - pfti.Y()) < toler);
69  BOOST_CHECK(std::abs(pd.Z() - pfti.Z()) < toler);
70 
71  BOOST_CHECK(std::abs(pdti.X() - pfti.X()) < toler);
72  BOOST_CHECK(std::abs(pdti.Y() - pfti.Y()) < toler);
73  BOOST_CHECK(std::abs(pdti.Z() - pfti.Z()) < toler);
