36 void SpatialPhotonResponse::addPhoton(double x, double y, int nphotons)
37 {
38  const int xpixel = (int)(std::floor((x - mLowerX) * mInvLxOfCell));
39  const int ypixel = (int)(std::floor((y - mLowerY) * mInvLyOfCell));
40  if (nphotons < 0) {
41  std::cerr << "negative photon number\n";
42  return;
43  }
44  if (xpixel < 0 || xpixel >= mNx) {
45  std::cerr << "X-PIXEL OUT OF RANGE " << xpixel << " " << x << " , " << y << "\n";
46  return;
47  }
48  if (ypixel < 0 || ypixel >= mNy) {
49  std::cerr << "Y-PIXEL OUT OF RANGE " << ypixel << " " << x << " , " << y << "\n";
50  return;
51  }
52  mImageData[xpixel][ypixel] += nphotons;
53  mPhotonSum += nphotons;
54 }

89 std::array<int, 5> SpatialPhotonResponse::getPhotonsPerChannel() const
90 {
91  std::array<int, 5> photonsum = {0, 0, 0, 0, 0};
92  if (mPhotonSum == 0) {
93  return photonsum;
94  }
95 
96  if (mDetectorID == -1) {
97  std::cerr << "SpatialPhotonResponse has no detectorID ";
98  return photonsum;
99  }
100 
101  auto determineChannel = [](int detector, int x, int y, int Nx, int Ny) {
102  if ((x + y) % 2 == 0) {
103  return (int)ChannelTypeZNP::Common;
104  }
105 
106  if (detector == DetectorID::ZNA || detector == DetectorID::ZNC) {
107  if (x < Nx / 2) {
108  if (y < Ny / 2) {
109  return (int)ChannelTypeZNP::Ch1;
110  } else {
111  return (int)ChannelTypeZNP::Ch3;
112  }
113  } else {
114  if (y >= Ny / 2) {
115  return (int)ChannelTypeZNP::Ch4;
116  } else {
117  return (int)ChannelTypeZNP::Ch2;
118  }
119  }
120  }
121 
122  if (detector == DetectorID::ZPA || detector == DetectorID::ZPC) {
123  auto i = (int)(4.f * x / Nx);
124  return (int)(i + 1);
125  }
126  return -1;
127  };
128 
129  int sum = 0;
130  for (int x = 0; x < mNx; ++x) {
131  // loop over y = rows
132  for (int y = 0; y < mNy; ++y) {
133  // get channel
134  int channel = determineChannel(mDetectorID, x, y, mNx, mNy);
135  photonsum[channel] += mImageData[x][y];
136  sum += 0;
137  }
138  }
139  // assert (mPhotonSum == photonsum[0] + photonsum[1] + photonsum[2] + photonsum[3] + photonsum[4]);
140 
141  return photonsum;
142 }
