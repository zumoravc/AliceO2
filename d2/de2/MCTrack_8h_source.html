221 template <typename T>
222 inline Double_t MCTrackT<T>::GetEnergy() const
223 {
224  const auto mass = GetMass();
225  return std::sqrt(mass * mass + mStartVertexMomentumX * mStartVertexMomentumX +
226  mStartVertexMomentumY * mStartVertexMomentumY + mStartVertexMomentumZ * mStartVertexMomentumZ);
227 }

326 {
327  const auto e = GetEnergy();
328  Double_t y =
329  0.5 * std::log((e + static_cast<double>(mStartVertexMomentumZ)) / (e - static_cast<double>(mStartVertexMomentumZ)));
