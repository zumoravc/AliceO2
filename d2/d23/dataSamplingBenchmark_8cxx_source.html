84 std::function<size_t(void)> createFreeMemoryGetter(InitContext& ictx)
85 {
86  std::string sessionID;
87  std::string channelConfig;
88  if (FairMQDevice* device = ictx.services().get<RawDeviceService>().device()) {
89  if (auto options = device->GetConfig()) {
90  sessionID = options->GetPropertyAsString("session", "");
91  channelConfig = options->GetPropertyAsString("channel-config", "");
92  }
93  }
94  if (!sessionID.empty() && sessionID != "default" && channelConfig.find("transport=shmem") != std::string::npos) {
95  LOG(INFO) << "The benchmark is running with shared memory,"
96  " producing messages will be throttled by looking at available memory in the segment: "
97  << sessionID;
98  std::string segmentID = "fmq_" + buildShmIdFromSessionIdAndUserId(sessionID) + "_main";
99  auto segment = std::make_shared<bipc::managed_shared_memory>(bipc::open_only, segmentID.c_str());
100 
101  return [segment]() {
102  return segment->get_free_memory();
103  };
104  } else {
105 #if defined(__APPLE__)
106  LOG(WARNING) << "The benchmark is running without shared memory. "
107  "The throttling mechanism is not supported on MacOS for the ZeroMQ transport, "
108  "the results might be incorrect for larger payload sizes.";
109 #else
110  LOG(INFO) << "The benchmark is running without shared memory,"
111  " producing messages will be throttled by looking at the global free RAM";
112 #endif
113 
114  auto sysInfo = std::make_shared<struct sysinfo>();
115 
116  return [sysInfo]() {
117  sysinfo(sysInfo.get());
118  return sysInfo->freeram;
119  };
120  }
121 }

123 WorkflowSpec defineDataProcessing(ConfigContext const& config)
124 {
125  double samplingFraction = config.options().get<double>("sampling-fraction");
126  size_t payloadSize = config.options().get<int>("payload-size");
127  size_t producers = config.options().get<int>("producers");
128  size_t dispatchers = config.options().get<int>("dispatchers");
129  size_t usleepTime = config.options().get<int>("usleep");
130  size_t testDuration = config.options().get<int>("test-duration");
131  size_t throttlingMB = config.options().get<int>("throttling");
132  bool fill = config.options().get<bool>("fill");
133 
134  std::string configurationPath = "/tmp/dataSamplingBenchmark-" + std::to_string(samplingFraction) + ".json";
135  std::string configuration =
136  "{\n"
137  " \"dataSamplingPolicies\": [\n"
138  " {\n"
139  " \"id\": \"benchmark\",\n"
140  " \"active\": \"true\",\n"
141  " \"machines\": [],\n"
142  " \"query\": \"TST:TST/RAWDATA\",\n"
143  " \"samplingConditions\": [\n"
144  " {\n"
145  " \"condition\": \"random\",\n"
146  " \"fraction\": \"" + std::to_string(samplingFraction) + "\",\n"
147  " \"seed\": \"22222\"\n"
148  " }\n"
149  " ],\n"
150  " \"blocking\": \"false\"\n"
151  " }\n"
152  " ]\n"
153  "}";
154 
155  if (!boost::filesystem::exists(configurationPath)) {
156  std::ofstream configurationFile(configurationPath);
157  configurationFile << configuration;
158  configurationFile.close();
159  }
160 
161  WorkflowSpec specs;
162 
163  for (size_t p = 0; p < producers; p++) {
164  specs.push_back(DataProcessorSpec{
165  "dataProducer" + std::to_string(p),
166  Inputs{},
167  Outputs{
168  OutputSpec{ "TST", "RAWDATA", static_cast<SubSpec>(p) }
169  },
170  AlgorithmSpec{
171  (AlgorithmSpec::InitCallback) [=](InitContext& ictx) {
172 
173  sleep(5); // wait a few seconds before trying to open a shmem segment to make sure it is there
174 
175  std::function<size_t(void)> getFreeMemory = createFreeMemoryGetter(ictx);
176  const size_t maxFreeMemory = getFreeMemory(); // that may vary on the process sync
177  size_t maximumAllowedMessages = (maxFreeMemory - throttlingMB * 1000000) / payloadSize / producers;
178  LOG(INFO) << "First cycle, this producer will send " << maximumAllowedMessages << " messages";
179 
180  auto messagesProducedSinceLastCycle = std::make_shared<size_t>(0);
181  std::shared_ptr<bool> mightSaturate = nullptr;
182 
183  return (AlgorithmSpec::ProcessCallback) [=](ProcessingContext& pctx) mutable {
184  usleep(usleepTime);
185 
186  // This is a mechanism which protects the benchmark from reaching the maximum of available memory.
187  // In that case it is likely that we get killed by oom-killer or trapped inside a deadlock.
188  // It works well for shmem tests, but not for zeromq - the latter increases the memory usage
189  // muuuch more inertia (slower and harder to control) and we cannot observe it this way.
190  // Then, it is recommended to use the '--fill' parameter, which prevents Linux from overcommitting
191  // the memory by writing on the produced messages before sending (which unfortunately slows down the
192  // message production rate).
193  if (*messagesProducedSinceLastCycle < maximumAllowedMessages) {
194  auto data = pctx.outputs().make<char>(Output{ "TST", "RAWDATA", static_cast<SubSpec>(p) }, payloadSize);
195  *messagesProducedSinceLastCycle += 1;
196  if (fill) {
197  memset(data.data(), 0x00, payloadSize);
198  }
199  } else if (mightSaturate == nullptr) {
200  // maximumAllowedMessages has been reached, so we check if we should protect the benchmark from asking
201  // for too much memory.
202  sleep(1);
203  // 4 is a magic number - an arbitrary high limit of free memory is 4 times larger than the low limit.
204  mightSaturate = std::make_shared<bool>(getFreeMemory() < 4 * throttlingMB * 1000000);
205  } else if (*mightSaturate == false) {
206  // there is no risk of reaching the maximum available memory,
207  // so we allow for a continuous message production.
208  *messagesProducedSinceLastCycle = 0;
209  maximumAllowedMessages = -1;
210  LOG(INFO) << "The memory usage should not reach the limits, we allow to produce as much messages as possible";
211  } else if (size_t freeMemory = getFreeMemory(); freeMemory > 4 * throttlingMB * 1000000) {
212  // if we are here, then the maximumAllowedMessages has been reached and we have waited until
213  // the memory usage dropped to the safe level again.
214  *messagesProducedSinceLastCycle = 0;
215  maximumAllowedMessages = (freeMemory - throttlingMB * 1000000) / payloadSize / producers;
216  LOG(INFO) << "New cycle, this producer will send " << maximumAllowedMessages << " messages";
217  }
218  };
219  }
220  }
221  });
222  }
223 
224  DataSampling::GenerateInfrastructure(specs, "json:/" + configurationPath, dispatchers);
225 
226  DataProcessorSpec podDataSink{
227  "dataSink",
228  Inputs{{"test-data", {DataSamplingPolicy::createPolicyDataOrigin(), DataSamplingPolicy::createPolicyDataDescription("benchmark", 0)}},
229  {"test-timer", "TST", "TIMER", 0, Lifetime::Timer}},
230  Outputs{},
231  AlgorithmSpec{
232  (AlgorithmSpec::ProcessCallback)[](ProcessingContext & ctx){
233  // todo: maybe add a check
234  if (ctx.inputs().isValid("test-timer")){
235  ctx.services().get<ControlService>().readyToQuit(QuitRequest::All);
236  }
237  }
238  },
239  Options{
240  { "period-test-timer", VariantType::Int, static_cast<int>(testDuration * 1000000), { "timer period" }}
241  }
242  };
243 
244  specs.push_back(podDataSink);
245  return specs;
246 }
