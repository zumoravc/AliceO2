22 void BadChannelMap::addBadChannel(unsigned short channelID, MaskType_t mask)
23 {

48 BadChannelMap::MaskType_t BadChannelMap::getChannelStatus(unsigned short channelID) const
49 {
50  auto status = MaskType_t::GOOD_CELL;
51  if (mDeadCells.test(channelID))
52  status = MaskType_t::DEAD_CELL;
53  else if (mBadCells.test(channelID))
54  status = MaskType_t::BAD_CELL;
55  else if (mWarmCells.test(channelID))
56  status = MaskType_t::WARM_CELL;
57  return status;
58 }

60 TH2* BadChannelMap::getHistogramRepresentation() const
61 {
62  const int MAXROWS = 208,
63  MAXCOLS = 96;
64  auto hist = new TH2S("badchannelmap", "Bad Channel Map", MAXCOLS, -0.5, double(MAXCOLS) - 0.5, MAXROWS, -0.5, double(MAXROWS) - 0.5);
65  hist->SetDirectory(nullptr);
66  auto geo = Geometry::GetInstance();
67  if (!geo) {
68  LOG(ERROR) << "Geometry needs to be initialized";
69  return hist;
70  }
71 
72  for (size_t cellID = 0; cellID < mBadCells.size(); cellID++) {
73  int value(0);
74  if (mBadCells.test(cellID))
75  value = 1;
76  else if (mDeadCells.test(cellID))
77  value = 2;
78  else if (mWarmCells.test(cellID))
79  value = 3;
80  if (value) {
81  auto position = geo->GlobalRowColFromIndex(cellID);
82  hist->Fill(std::get<1>(position), std::get<0>(position), value);
83  }
84  }
85  return hist;
86 }

88 BadChannelMap& BadChannelMap::operator+=(const BadChannelMap& rhs)
89 {
90  for (size_t cellID = 0; cellID < mDeadCells.size(); cellID++) {
91  if (rhs.mDeadCells.test(cellID)) {
92  mDeadCells.set(cellID);
93  if (mBadCells.test(cellID))
94  mBadCells.reset(cellID);
95  if (mWarmCells.test(cellID))
96  mWarmCells.reset(cellID);
97  }
98  if (rhs.mBadCells.test(cellID) && !mDeadCells.test(cellID)) {
99  mBadCells.set(cellID);
100  if (mWarmCells.test(cellID))
101  mWarmCells.reset(cellID);
102  }
103  if (rhs.mWarmCells.test(cellID) && !(mBadCells.test(cellID) || mDeadCells.test(cellID)))
104  mWarmCells.set(cellID);
105  }
106  return *this;
107 }
