35 // general remark:
36 // at the moment its not clear how the heartbeat frame is transmitted in AliceO2.
37 // Current understanding is that the heartbeat header and trailer embed some detector
38 // data and form the heartbeat frame. The detector data is the payload which is going
39 // to be sent as the payload of the O2 data packet. The HBH and HBT must be stripped
40 // from the payload and probably be added to the header stack and later on be stripped
41 // from there as well during the accumulation of subtimeframes and timeframes as the
42 // HB information should be identical in the (sub)timeframes

50 struct HeartbeatHeader {
51  union {
52  // the complete 64 bit header word, initialize with blockType 1 and size 1
53  uint64_t headerWord = 0x1100000000000000;
54  struct {
55  // bit 0 to 31: orbit number
56  uint32_t orbit;
57  // bit 32 to 43: bunch crossing id
58  uint16_t bcid : 12;
59  // bit 44 to 47: reserved
60  uint16_t reserved : 4;
61  // bit 48 to 51: trigger type
62  uint8_t triggerType : 4;
63  // bit 52 to 55: reserved
64  uint8_t reservedTriggerType : 4;
65  // bit 56 to 59: header length
66  uint8_t headerLength : 4;
67  // bit 60 to 63: block type (=1 HBF/trigger Header)
68  uint8_t blockType : 4;
69  };
70  }; // end union
71 
72  operator bool() const { return headerWord != 0 && blockType == 0x1; }
73 
74  bool operator<(const HeartbeatHeader& other) const
75  {
76  return this->orbit < other.orbit;
77  }
78 
79  operator uint64_t() const { return headerWord; }
80 };
81 static_assert(sizeof(HeartbeatHeader) == 8, "Heartbeat header must be 64bit");
82 
83 struct HeartbeatTrailer {
84  union {
85  // the complete 64 bit trailer word, initialize with blockType 5 and size 1
86  uint64_t trailerWord = 0x5100000000000000;
87  struct {
88  // bit 0 to 31: data length in words
89  uint32_t dataLength;
90  // bit 32 to 52: detector specific status words
91  uint32_t status : 21;
92  // bit 53: =1 in case a new physics trigger arrived within read-out period
93  uint16_t hbfTruncated : 1;
94  // bit 54: =0 HBF correctly transmitted
95  uint16_t hbfStatus : 1;
96  // bit 55: =1 HBa/0 HBr received
97  uint16_t hbAccept : 1;
98  // bit 56 to 59: trailer length
99  uint16_t trailerLength : 4;
100  // bit 60 to 63: block type (=5 HBF Trailer)
101  uint8_t blockType : 4;
102  };
103  }; // end union
104 
105  operator bool() const { return trailerWord != 0 && blockType == 0x5; }
106 
107  operator uint64_t() const { return trailerWord; }
108 };
109 static_assert(sizeof(HeartbeatTrailer) == 8, "Heartbeat trailer must be 64bit");
