54 // clang-format off
55 WorkflowSpec defineDataProcessing(ConfigContext const& config)
56 {
57  size_t producers = config.options().get<int>("producers");
58  size_t testDuration = config.options().get<int>("test-duration");
59 
60  WorkflowSpec specs;
61  for (size_t p = 0; p < producers; p++) {
62  specs.push_back(DataProcessorSpec{
63  "dataProducer" + std::to_string(p),
64  Inputs{},
65  Outputs{
66  OutputSpec{ "TST", "NODATA", static_cast<SubSpec>(p) }
67  },
68  AlgorithmSpec{
69  (AlgorithmSpec::ProcessCallback) [=](ProcessingContext& pctx) mutable {
70  usleep(1000000);
71  }
72  }
73  });
74  }
75 
76  DataProcessorSpec sink{
77  "sink",
78  Inputs{{ "test-data", { "TST", "NODATA" }},
79  { "sink-timer", "TST", "TIMER", 0, Lifetime::Timer }},
80  Outputs{{{ "output" }, "TST", "ALSONODATA" }},
81  AlgorithmSpec{
82  (AlgorithmSpec::InitCallback) [](InitContext& ictx) {
83  auto timer = std::make_shared<Timer>();
84  timer->reset(10 * 1000000);
85  uint64_t loopCounter = 0;
86 
87  return (AlgorithmSpec::ProcessCallback) [=](ProcessingContext& ctx) mutable {
88  loopCounter++;
89  if (timer->isTimeout()) {
90  timer->increment();
91  auto& monitoring = ctx.services().get<Monitoring>();
92  monitoring.send({ loopCounter, "loop_counter" });
93  }
94  };
95  }
96  },
97  Options{{"period-sink-timer", VariantType::Int, 0, { "timer period" }}}
98  };
99  specs.push_back(sink);
100 
101  DataProcessorSpec heWhoRequestsExit{
102  "heWhoRequestsExit",
103  Inputs{{ "input", "TST", "ALSONODATA" },
104  { "test-timer", "TST", "TIMER2", 0, Lifetime::Timer }},
105  Outputs{},
106  AlgorithmSpec{
107  (AlgorithmSpec::ProcessCallback) [](ProcessingContext& ctx) {
108  if (ctx.inputs().isValid("test-timer")) {
109  LOG(INFO) << "Planned exit";
110  ctx.services().get<ControlService>().readyToQuit(QuitRequest::All);
111  }
112  }
113  },
114  Options{{"period-test-timer", VariantType::Int, static_cast<int>(testDuration * 1000000), { "timer period" }}}
115  };
116 
117  specs.push_back(heWhoRequestsExit);
118  return specs;
119 }
