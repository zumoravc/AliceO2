53 GPUd() float GPUdEdx::GetSortTruncMean(float* GPUrestrict() array, int count, int trunclow, int trunchigh)
54 {
55  trunclow = count * trunclow / 128;
56  trunchigh = count * trunchigh / 128;
57  if (trunclow >= trunchigh) {
58  return (0.);
59  }
60  CAAlgo::sort(array, array + count);
61  float mean = 0;
62  for (int i = trunclow; i < trunchigh; i++) {
63  mean += array[i];
64  }
65  return (mean / (trunchigh - trunclow));
66 }
