58 template <typename source_T>
59 void TPCEntropyEncoder::compress(const std::string& name, // name
60  const source_T* const sourceBegin, // begin of source message
61  const source_T* const sourceEnd, // end of source message
62  uint8_t probabilityBits, // encoding into
63  EncodedClusters& ec) // structure to hold encoded data
64 {
65 
66  // find which array we are dealing with
67  auto it = std::find(ec.NAMES.begin(), ec.NAMES.end(), name);
68  assert(it != ec.NAMES.end());
69  auto i = std::distance(ec.NAMES.begin(), it);
70 
71  // get references to the right data
72  auto& dicts = ec.dicts;
73  auto& buffers = ec.buffers;
74  auto& md = *ec.metadata;
75 
76  // symbol statistics and encoding
77  using stream_t = typename rans::Encoder64<source_T>::stream_t;
78  rans::SymbolStatistics stats{sourceBegin, sourceEnd};
79  stats.rescaleToNBits(probabilityBits);
80  const auto buffSize = calculateMaxBufferSize(stats.getMessageLength(),
81  stats.getAlphabetRangeBits(),
82  sizeof(source_T));
83 
84  std::vector<stream_t> encoderBuffer(buffSize);
85  const rans::Encoder64<source_T> encoder{stats, probabilityBits};
86  const auto encodedMessageStart = encoder.process(encoderBuffer.begin(),
87  encoderBuffer.end(),
88  sourceBegin, sourceEnd);
89 
90  // write metadata
91  md.emplace_back(EncodedClusters::Metadata{sizeof(uint64_t), sizeof(stream_t), probabilityBits, stats.getMinSymbol(), stats.getMaxSymbol()});
92 
93  //write dict
94  dicts[i] = new std::vector<uint32_t>();
95  for (const auto& item : stats) {
96  dicts[i]->push_back(item.first);
97  }
98 
99  // write buffer
100  buffers[i] = new std::vector<uint32_t>(encodedMessageStart, encoderBuffer.end());
101 
102  LOG(INFO) << "finished entropy coding of " << name;
103 }
