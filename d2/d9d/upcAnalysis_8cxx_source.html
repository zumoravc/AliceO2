25 struct UPCAnalysis {
26  OutputObj<TH1D> hMass{TH1D("hMass", ";#it{m_{#pi#pi}}, GeV/c^{2};", 500, 0.55, 1.4)};
27  OutputObj<TH1D> hPt{TH1D("hPt", ";#it{p_{T}}, GeV/c;", 500, 0., 1.)};
28  OutputObj<TH2D> hSignalTPC1vsSignalTPC2{TH2D("hSignalTPC1vsSignalTPC2", ";TPC Signal 1;TPC Signal 2", 1000, 0., 200., 1000, 0., 200.)};
29  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra> const& tracks)
30  {
31  std::vector<soa::Join<aod::Tracks, aod::TracksExtra>::iterator> selTracks;
32  for (auto track : tracks) {
33  UChar_t clustermap = track.itsClusterMap();
34  bool isselected = track.tpcNClsFound() > 70 && track.flags() & 0x4;
35  isselected = isselected && (TESTBIT(clustermap, 0) && TESTBIT(clustermap, 1));
36  if (!isselected)
37  continue;
38  selTracks.push_back(track);
39  if (selTracks.size() == 2)
40  break;
41  }
42  if (selTracks.size() == 2 && selTracks[0].charge() * selTracks[1].charge() < 0) {
43  TLorentzVector p1, p2, p;
44  p1.SetXYZM(selTracks[0].px(), selTracks[0].py(), selTracks[0].pz(), mpion);
45  p2.SetXYZM(selTracks[1].px(), selTracks[1].py(), selTracks[1].pz(), mpion);
46  p = p1 + p2;
47  hPt->Fill(p.Pt());
48  double signalTPC1 = selTracks[0].tpcSignal();
49  double signalTPC2 = selTracks[1].tpcSignal();
50  hSignalTPC1vsSignalTPC2->Fill(signalTPC1, signalTPC2);
51  if ((p.Pt() < 0.1) && (signalTPC1 + signalTPC2 < 140)) {
52  hMass->Fill(p.M());
53  }
54  }
55  }
56 };
