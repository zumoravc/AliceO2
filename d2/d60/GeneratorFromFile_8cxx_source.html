25 GeneratorFromFile::GeneratorFromFile(const char* name)
26 {
27  mEventFile = TFile::Open(name);
28  if (mEventFile == nullptr) {
29  LOG(FATAL) << "EventFile " << name << " not found \n";
30  return;
31  }
32  // the kinematics will be stored inside a Tree "TreeK" with branch "Particles"
33  // different events are stored inside TDirectories
34 
35  // we need to probe for the number of events
36  TObject* object = nullptr;
37  do {
38  std::stringstream eventstringstr;
39  eventstringstr << "Event" << mEventsAvailable;
40  // std::cout << "probing for " << eventstring << "\n";
41  object = mEventFile->Get(eventstringstr.str().c_str());
42  // std::cout << "got " << object << "\n";
43  if (object != nullptr)
44  mEventsAvailable++;
45  } while (object != nullptr);
46  LOG(INFO) << "Found " << mEventsAvailable << " events in this file \n";
47 }

58 bool GeneratorFromFile::rejectOrFixKinematics(TParticle& p)
59 {
60  const auto nominalmass = p.GetMass();
61  auto mom2 = p.Px() * p.Px() + p.Py() * p.Py() + p.Pz() * p.Pz();
62  auto calculatedmass = p.Energy() * p.Energy() - mom2;
63  calculatedmass = (calculatedmass >= 0.) ? std::sqrt(calculatedmass) : -std::sqrt(-calculatedmass);
64  const double tol = 1.E-4;
65  auto difference = std::abs(nominalmass - calculatedmass);
66  if (std::abs(nominalmass - calculatedmass) > tol) {
67  const auto asgmass = p.GetCalcMass();
68  bool fix = mFixOffShell && std::abs(nominalmass - asgmass) < tol;
69  LOG(WARN) << "Particle " << p.GetPdgCode() << " has off-shell mass: M_PDG= " << nominalmass << " (assigned= " << asgmass
70  << ") calculated= " << calculatedmass << " -> diff= " << difference << " | " << (fix ? "fixing" : "skipping");
71  if (fix) {
72  double e = std::sqrt(nominalmass * nominalmass + mom2);
73  p.SetMomentum(p.Px(), p.Py(), p.Pz(), e);
74  p.SetCalcMass(nominalmass);
75  } else {
76  return false;
77  }
78  }
79  return true;
80 }

82 Bool_t GeneratorFromFile::ReadEvent(FairPrimaryGenerator* primGen)
83 {
84  if (mEventCounter < mEventsAvailable) {
85  int particlecounter = 0;
86 
87  // get the tree and the branch
88  std::stringstream treestringstr;
89  treestringstr << "Event" << mEventCounter << "/TreeK";
90  TTree* tree = (TTree*)mEventFile->Get(treestringstr.str().c_str());
91  if (tree == nullptr) {
92  return kFALSE;
93  }
94 
95  auto branch = tree->GetBranch("Particles");
96  TParticle* particle = nullptr;
97  branch->SetAddress(&particle);
98  LOG(INFO) << "Reading " << branch->GetEntries() << " particles from Kinematics file";
99 
100  // read the whole kinematics initially
101  std::vector<TParticle> particles;
102  for (int i = 0; i < branch->GetEntries(); ++i) {
103  branch->GetEntry(i);
104  particles.push_back(*particle);
105  }
106 
107  // filter the particles from Kinematics.root originally put by a generator
108  // and which are trackable
109  auto isFirstTrackableDescendant = [](TParticle const& p) {
110  const int kDoneBit = 4;
111  // The particle should have not set kDone bit and its status should not exceed 1
112  if (p.GetStatusCode() > 1 || p.TestBit(kDoneBit)) {
113  return false;
114  }
115  return true;
116  };
117 
118  for (int i = 0; i < branch->GetEntries(); ++i) {
119  auto& p = particles[i];
120 
121  if (!isFirstTrackableDescendant(p)) {
122  continue;
123  }
124 
125  bool wanttracking = true; // RS as far as I understand, if it reached this point, it is trackable
126  if (wanttracking || !mSkipNonTrackable) {
127  if (!rejectOrFixKinematics(p)) {
128  continue;
129  }
130  auto pdgid = p.GetPdgCode();
131  auto px = p.Px();
132  auto py = p.Py();
133  auto pz = p.Pz();
134  auto vx = p.Vx();
135  auto vy = p.Vy();
136  auto vz = p.Vz();
137  auto parent = -1;
138  auto e = p.Energy();
139  auto tof = p.T();
140  auto weight = p.GetWeight();
141  LOG(DEBUG) << "Putting primary " << pdgid << " " << p.GetStatusCode() << " " << p.GetUniqueID();
142  primGen->AddTrack(pdgid, px, py, pz, vx, vy, vz, parent, wanttracking, e, tof, weight);
143  particlecounter++;
144  }
145  }
146  mEventCounter++;
147 
148  LOG(INFO) << "Event generator put " << particlecounter << " on stack";
149  return kTRUE;
150  } else {
151  LOG(ERROR) << "GeneratorFromFile: Ran out of events\n";
152  }
153  return kFALSE;
154 }
