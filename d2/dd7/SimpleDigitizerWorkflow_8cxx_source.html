183 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)
184 // by default this will be std::max(the number of physical cores, numberofsectors)
185 // as a temporary means to fully use a machine and as a way to play with different topologies
186 int getNumTPCLanes(std::vector<int> const& sectors, ConfigContext const& configcontext)
187 {
188  auto lanes = configcontext.options().get<int>("tpc-lanes");
189  if (lanes < 0) {
190  if (gIsMaster) {
191  LOG(FATAL) << "tpc-lanes needs to be positive\n";
192  }
193  return 0;
194  }
195  // crosscheck with sectors
196  return std::min(lanes, (int)sectors.size());
197 }

201 void initTPC()
202 {
203  // We only want to do this for the DPL master
204  // I am not aware of an easy way to query if "I am DPL master" so
205  // using for the moment a mechanism defining/setting an environment variable
206  // with the parent ID and query inside forks if this environment variable exists
207  // (it assumes fundamentally that the master executes this function first)
208  std::stringstream streamthis;
209  std::stringstream streamparent;
210 
211  streamthis << "TPCGEMINIT_PID" << getpid();
212  streamparent << "TPCGEMINIT_PID" << getppid();
213  if (getenv(streamparent.str().c_str())) {
214  LOG(DEBUG) << "GEM ALREADY INITIALIZED ... SKIPPING HERE";
215  return;
216  }
217 
218  LOG(DEBUG) << "INITIALIZING TPC GEMAmplification";
219  setenv(streamthis.str().c_str(), "ON", 1);
220 
221  auto& cdb = o2::tpc::CDBInterface::instance();
222  cdb.setUseDefaults();
223  // by invoking this constructor we make sure that a common file will be created
224  // in future we should take this from OCDB and just forward per message
225  const static auto& ampl = o2::tpc::GEMAmplification::instance();
226 }

245 // Split a given string on a separator character
246 std::vector<std::string> splitString(std::string src, char sep)
247 {
248  std::vector<std::string> fields;
249  std::string token;
250  std::istringstream ss(src);
251 
252  while (std::getline(ss, token, sep)) {
253  if (!token.empty()) {
254  fields.push_back(token);
255  }
256  }
257 
258  return fields;
259 }

262 // Filters detectors based on a white/black list provided via the onlyDet/skipDet CLI args
263 struct DetFilterer {
264  // detlist: A character-separated list of detectors
265  // unsetVal: The value when the option is unset
266  // separator: The character that separates the list of detectors defined in option
267  // mustContain: The nature of this DetFilterer. If true, it is a white lister
268  // i.e. option defines the list of allowed detectors. If false
269  // it is a black lister i.e defines the list of disallowed detectors.
270  DetFilterer(std::string detlist, std::string unsetVal, char separator, bool doWhiteListing)
271  {
272  // option is not set, nothing to do
273  if (detlist.compare(unsetVal) == 0) {
274  return;
275  }
276 
277  std::vector<std::string> tokens = splitString(detlist, separator);
278 
279  // Convert a vector of strings to one of o2::detectors::DetID
280  for (auto token : tokens) {
281  ids.emplace_back(token.c_str());
282  }
283 
284  isWhiteLister = doWhiteListing;
285  }
286 
287  // isSet determines if a detector list was provided
288  // against which to filter
289  bool isSet()
290  {
291  return ids.size() > 0;
292  }
293 
294  // accept determines if a given detector should be accepted
295  bool accept(o2::detectors::DetID id)
296  {
297  bool found = std::find(ids.begin(), ids.end(), id) != ids.end();
298  return found == isWhiteLister;
299  }
300 
301  private:
302  std::vector<o2::detectors::DetID> ids;
303  bool isWhiteLister; // true = accept only detectors in the ids vector
304 };
305 
306 // Helper function to define a white listing DetFilterer
307 DetFilterer whitelister(std::string optionVal, std::string unsetValue, char separator)
308 {
309  return DetFilterer(optionVal, unsetValue, separator, true);
310 }
311 
312 // Helper function to define a black listing DetFilterer
313 DetFilterer blacklister(std::string optionVal, std::string unsetValue, char separator)
314 {
315  return DetFilterer(optionVal, unsetValue, separator, false);
316 }

318 // Finding out if the current process is the master DPL driver process,
319 // first setting up the topology. Might be important to know when we write
320 // files (to prevent that multiple processes write the same file)
321 bool isMasterWorkflowDefinition(ConfigContext const& configcontext)
322 {
323  int argc = configcontext.argc();
324  auto argv = configcontext.argv();
325  bool ismaster = true;
326  for (int argi = 0; argi < argc; ++argi) {
327  // when channel-config is present it means that this is started as
328  // as FairMQDevice which means it is already a forked process
329  if (strcmp(argv[argi], "--channel-config") == 0) {
330  ismaster = false;
331  break;
332  }
333  }
334  return ismaster;
335 }

341 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
342 {
343  // check if we merely construct the topology to create help options
344  // if this is the case we don't need to read from GRP
345  bool helpasked = configcontext.helpOnCommandLine();
346  bool ismaster = isMasterWorkflowDefinition(configcontext);
347  gIsMaster = ismaster;
348 
349  // Reserve one entry which fill be filled with the SimReaderSpec
350  // at the end. This places the processor at the beginning of the
351  // workflow in the upper left corner of the GUI.
352  WorkflowSpec specs(1);
353 
354  using namespace o2::conf;
355  ConfigurableParam::updateFromFile(configcontext.options().get<std::string>("configFile"));
356 
357  // Update the (declared) parameters if changed from the command line
358  // Note: In the future this should be done only on a dedicated processor managing
359  // the parameters and then propagated automatically to all devices
360  ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
361 
362  // which sim productions to overlay and digitize
363  auto simPrefixes = splitString(configcontext.options().get<std::string>("sims"), ',');
364 
365  // First, read the GRP to detect which components need instantiations
366  auto grpfile = o2::base::NameConf::getGRPFileName(simPrefixes[0]);
367  std::shared_ptr<o2::parameters::GRPObject const> grp(nullptr);
368  if (!helpasked) {
369  grp = readGRP(grpfile.c_str());
370  if (!grp) {
371  return WorkflowSpec{};
372  }
373  }
374 
375  // update the digitization configuration with the right geometry file
376  // we take the geometry from the first simPrefix (could actually check if they are
377  // all compatible)
378  auto geomfilename = o2::base::NameConf::getGeomFileName(simPrefixes[0]);
379  ConfigurableParam::setValue("DigiParams.digitizationgeometry", geomfilename);
380  ConfigurableParam::setValue("DigiParams.grpfile", grpfile);
381  LOG(INFO) << "MC-TRUTH " << !configcontext.options().get<bool>("disable-mc");
382  bool mctruth = !configcontext.options().get<bool>("disable-mc");
383  ConfigurableParam::setValue("DigiParams", "mctruth", mctruth);
384 
385  // write the configuration used for the digitizer workflow
386  if (ismaster) {
387  o2::conf::ConfigurableParam::writeINI(std::string(o2::base::NameConf::DIGITIZATIONCONFIGFILE));
388  }
389 
390  // onlyDet takes precedence on skipDet
391  DetFilterer filterers[2] = {
392  whitelister(configcontext.options().get<std::string>("onlyDet"), "none", ','),
393  blacklister(configcontext.options().get<std::string>("skipDet"), "none", ',')};
394 
395  auto accept = [&configcontext, &filterers](o2::detectors::DetID id) {
396  for (auto& f : filterers) {
397  if (f.isSet()) {
398  return f.accept(id);
399  }
400  }
401 
402  // accept all if neither onlyDet/skipDet are provided
403  return true;
404  };
405 
406  // lambda to extract detectors which are enabled in the workflow
407  // will complain if user gave wrong input in construction of DetID
408  auto isEnabled = [&configcontext, &filterers, accept, grp, helpasked](o2::detectors::DetID id) {
409  if (helpasked) {
410  return true;
411  }
412  auto accepted = accept(id);
413  bool is_ingrp = grp->isDetReadOut(id);
414  if (gIsMaster) {
415  LOG(INFO) << id.getName()
416  << " is in grp? " << (is_ingrp ? "yes" : "no") << ";"
417  << " is skipped? " << (!accepted ? "yes" : "no");
418  }
419  return accepted && is_ingrp;
420  };
421 
422  std::vector<o2::detectors::DetID> detList; // list of participating detectors
423 
424  // the TPC part
425  // we need to init this anyway since TPC is treated a bit special (for the moment)
426  if (!helpasked && ismaster) {
427  initTPC();
428  }
429 
430  // keeps track of which tpc sectors to process
431  std::vector<int> tpcsectors;
432 
433  if (isEnabled(o2::detectors::DetID::TPC)) {
434  tpcsectors = o2::RangeTokenizer::tokenize<int>(configcontext.options().get<std::string>("tpc-sectors"));
435  // only one lane for the help printout
436  auto lanes = helpasked ? 1 : getNumTPCLanes(tpcsectors, configcontext);
437  detList.emplace_back(o2::detectors::DetID::TPC);
438 
439  WorkflowSpec tpcPipelines = o2::tpc::getTPCDigitizerSpec(lanes, tpcsectors, mctruth);
440  specs.insert(specs.end(), tpcPipelines.begin(), tpcPipelines.end());
441 
442  auto tpcRecoOutputType = configcontext.options().get<std::string>("tpc-reco-type");
443  if (tpcRecoOutputType.empty()) {
444  // for writing digits to disc
445  specs.emplace_back(o2::tpc::getTPCDigitRootWriterSpec(tpcsectors, mctruth));
446  } else {
447  // attach the TPC reco workflow
448  auto tpcRecoWorkflow = o2::tpc::reco_workflow::getWorkflow(tpcsectors, tpcsectors, true, lanes, "digitizer", tpcRecoOutputType.c_str());
449  specs.insert(specs.end(), tpcRecoWorkflow.begin(), tpcRecoWorkflow.end());
450  }
451  }
452

454  const int firstOtherChannel = 36;
455  int fanoutsize = firstOtherChannel;

458  if (isEnabled(o2::detectors::DetID::ITS)) {
