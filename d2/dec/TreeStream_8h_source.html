175 template <class T>
176 Int_t TreeStream::CheckIn(T* obj)
177 {
178  // check in arbitrary class having dictionary
179  TClass* pClass = nullptr;
180  if (obj)
181  pClass = obj->IsA();
182 
183  if (mCurrentIndex >= static_cast<int>(mElements.size())) {
184  mElements.emplace_back();
185  auto& element = mElements.back();
186  element.cls = pClass;
187  TString name = mNextName;
188  if (name.Length()) {
189  if (mNextNameCounter > 0) {
190  name += mNextNameCounter;
191  }
192  } else {
193  name = TString::Format("B%d", static_cast<int>(mElements.size()));
194  }
195  element.name = name.Data();
196  element.ptr = obj;
197  } else {
198  auto& element = mElements[mCurrentIndex];
199  if (!element.cls) {
200  element.cls = pClass;
201  } else {
202  if (element.cls != pClass && pClass) {
203  mStatus++;
204  return 1; // mismatched data element
205  }
206  }
207  element.ptr = obj;
208  }
209  mCurrentIndex++;
210  return 0;
211 }
