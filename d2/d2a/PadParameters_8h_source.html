81 template <class T>
82 int PadParameters<T>::init(int chamberindex)
83 {
84  mPlane = TRDGeometry::getLayer(chamberindex);
85  mChamber = TRDGeometry::getStack(chamberindex);
86  if (mChamber == 2)
87  mNrows = FeeParam::mgkNrowC0;
88  else
89  mNrows = FeeParam::mgkNrowC1;
90  // the FeeParam variables need to be unprotected, and dont want to change FeeParam in this PR.
91  mNchannels = mNrows * mNcols;
92  mData.resize(mNchannels);
93  if (mData.size() != mNchannels || mData.size() == 0)
94  LOG(FATAL) << "PadParamaters initialised with a size of " << mData.size() << " and mNchannels of " << mNchannels;
95  return 0;
96 }

98 template <class T>
99 void PadParameters<T>::reset(int chamberindex, int cols, int rows, std::vector<T>& data)
100 {
101  mPlane = TRDGeometry::getLayer(chamberindex);
102  mChamber = TRDGeometry::getStack(chamberindex);
103  mNrows = rows;
104  mNcols = cols;
105  // the FeeParam variables need to be unprotected, and dont want to change FeeParam in this PR.
106  mNchannels = mNrows * mNcols;
107  if (mData.size() != mNchannels)
108  LOG(info) << "reseting a readoutchamber size which should not be needed given the detectors have not changed in run2 to run3";
109  mData.resize(mNchannels);
110  if (mData.size() != mNchannels || mData.size() == 0)
111  LOG(FATAL) << "PadParamaters initialised with a size of " << mData.size() << " and mNchannels of " << mNchannels;
112 
113  // now reset the data of the pads.
114  int counter = 0;
115  for (auto pad : mData) {
116  pad = data[counter];
117  counter++;
118  }
