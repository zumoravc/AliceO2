41 void CRUBareDecoder::process(gsl::span<const uint8_t> bytes)
42 {
44  reset();
45 
46  mBuffer.setBuffer(bytes);
47 
48  // Each CRU word consists of 128 bits, i.e 16 bytes
49  while (mBuffer.hasNext(16)) {
50  processGBT(0);
51  processGBT(5);
52  // The GBT word consists of 10 bytes
53  // but the CRU word is 16 bytes
54  // So we have 6 empty bytes
55  for (int iword = 0; iword < 6; ++iword) {
56  mBuffer.next();
57  }
58  }
59 }

61 void CRUBareDecoder::processGBT(size_t offset)
62 {
64 
65  // Regional link
66  // loc#0 loc#1 loc#2 loc#3 reg
67  std::array<uint8_t, 5> bytes{mBuffer.next(), mBuffer.next(), mBuffer.next(), mBuffer.next(), mBuffer.next()};
68 
69  // Byte corresponding to regional
70  size_t ibyte = 4;
71  size_t ilink = offset + ibyte;
72 
73  if (mELinkDecoders[ilink].add(bytes[ibyte], 0x80) && mELinkDecoders[ilink].isComplete()) {
74  // In principle, in the same HB we should have the info of only 1 crate
75  // So it should be safe to store this information
76  // and apply it to all subsequent e-links, without checking the clock
77  mCrateId = mELinkDecoders[ilink].getId();
78  mAddReg(ilink);
79  mELinkDecoders[ilink].reset();
80  }
81 
82  // local links
83  for (ibyte = 0; ibyte < 4; ++ibyte) {
84  ilink = offset + ibyte;
85  if (mELinkDecoders[ilink].add(bytes[ibyte], 0xc0) && mELinkDecoders[ilink].isComplete()) {
86  if (mCheckBoard(ilink)) {
87  addLoc(ilink);
88  }
89  mELinkDecoders[ilink].reset();
90  }
91  }
92 }

101 void CRUBareDecoder::addBoard(size_t ilink)
102 {
104  uint16_t localClock = mELinkDecoders[ilink].getCounter();
105  EventType eventType = EventType::Standard;
106  if (mELinkDecoders[ilink].getTriggerWord() & (1 << 4)) {
107  mCalibClocks[ilink] = localClock;
108  eventType = EventType::Noise;
109  } else if (localClock == mCalibClocks[ilink] + sDelayCalibToFET) {
110  eventType = EventType::Dead;
111  }
112  auto firstEntry = mData.size();
113  InteractionRecord intRec(localClock - sDelayBCToLocal, mBuffer.getRDH()->triggerOrbit);
114  mData.push_back({mELinkDecoders[ilink].getStatusWord(), mELinkDecoders[ilink].getTriggerWord(), crateparams::makeUniqueLocID(mCrateId, mELinkDecoders[ilink].getId()), mELinkDecoders[ilink].getInputs()});
115  mROFRecords.emplace_back(intRec, eventType, firstEntry, 1);
116 }

118 void CRUBareDecoder::addLoc(size_t ilink)
119 {
121  addBoard(ilink);
122  bool invert = (mROFRecords.back().eventType == EventType::Dead);
123  for (int ich = 0; ich < 4; ++ich) {
124  if ((mData.back().firedChambers & (1 << ich))) {
125  mData.back().patternsBP[ich] = getPattern(mELinkDecoders[ilink].getPattern(0, ich), invert);
126  mData.back().patternsNBP[ich] = getPattern(mELinkDecoders[ilink].getPattern(1, ich), invert);
127  }
128  }
129  mELinkDecoders[ilink].reset();
130 }
