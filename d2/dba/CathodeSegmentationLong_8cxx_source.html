56 std::vector<Point> checkGaps(const CathodeSegmentation& seg, double xstep = 1.0, double ystep = 1.0)
57 {
58  std::vector<Point> gaps;
59  auto bbox = o2::mch::mapping::getBBox(seg);
60  auto env = o2::mch::mapping::getEnvelop(seg);
61 
62  if (env.size() != 1) {
63  throw std::runtime_error("assumption env contour = one polygon is not verified");
64  }
65 
66  for (double x = bbox.xmin() - xstep; x <= bbox.xmax() + xstep; x += xstep) {
67  for (double y = bbox.ymin() - ystep; y <= bbox.ymax() + ystep; y += ystep) {
68  double distanceToEnveloppe =
69  std::sqrt(o2::mch::contour::squaredDistancePointToPolygon(o2::mch::contour::Vertex<double>{x, y}, env[0]));
70  bool withinEnveloppe = env.contains(x, y) && (distanceToEnveloppe > 1E-5);
71  if (withinEnveloppe && !seg.isValid(seg.findPadByPosition(x, y))) {
72  gaps.emplace_back(x, y);
73  }
74  }
75  }
76  return gaps;
77 }

102 int testOne(const o2::mch::mapping::CathodeSegmentation& seg, rapidjson::Value& tp,
103  Comparator comp)
104 {
105  double x = tp["x"].GetDouble();
106  double y = tp["y"].GetDouble();
107  int catPadIndex = seg.findPadByPosition(x, y);
108 
109  bool isOutside{false};
110 
111  if (tp.HasMember("isoutside")) {
112  isOutside = (tp["isoutside"].GetString() == std::string("true"));
113  }
114 
115  if (seg.isValid(catPadIndex) && isOutside) {
116  std::cerr << "found a pad where I was not expecting one" << std::endl;
117  return 1;
118  }
119  if (!seg.isValid(catPadIndex) && !isOutside) {
120  std::cerr << "did not find a pad where I was expecting one" << std::endl;
121  return 1;
122  }
123 
124  if (seg.isValid(catPadIndex)) {
125  return comp(seg, catPadIndex, tp);
126  }
127  return 0;
128 }
