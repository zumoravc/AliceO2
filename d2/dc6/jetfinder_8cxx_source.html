33 struct JetFinderTask {
34  Produces<o2::aod::Jets> jets;
35  Produces<o2::aod::JetConstituents> constituents;
36 
37  // options for jet finding
38  Configurable<float> rParam{"rParam", 0.4, "jet radius"};
39  Configurable<float> ghostEtamax{"ghostEtamax", 1.0, "eta max for ghosts"};
40  Configurable<float> minJetPt{"minJetPt", 0., "minimum jet pt"};
41  // TODO: initialize with 0.9 - rParam (requires lazy evaluation)
42  Configurable<float> maxJetEta{"maxJetEta", 0.5, "eta range for jets"};
43  // TODO: use configurables also for enums
44  fastjet::JetAlgorithm algorithm{fastjet::antikt_algorithm};
45  fastjet::RecombinationScheme recombScheme{fastjet::E_scheme};
46  fastjet::Strategy strategy{fastjet::Best};
47  fastjet::AreaType areaType{fastjet::passive_area};
48  fastjet::GhostedAreaSpec ghostSpec{ghostEtamax};
49  fastjet::AreaDefinition areaDef{areaType, ghostSpec};
50  fastjet::JetDefinition jetDef{algorithm, rParam, recombScheme, strategy};
51  fastjet::Selector selJet = fastjet::SelectorPtMin(minJetPt) &&
52  fastjet::SelectorAbsRapMax(maxJetEta);
53 
54  // options for background subtraction
55  enum class BkgMode { none,
56  rhoArea };
57  BkgMode bkgMode = BkgMode::none;
58  Configurable<double> rParamBkg{"rParamBkg", 0.2, "jet radius for background"};
59  Configurable<double> rapBkg{"rapBkg", .9, "rapidity range for background"};
60  // TODO: use configurables also for enums
61  fastjet::JetAlgorithm algorithmBkg{fastjet::kt_algorithm};
62  fastjet::RecombinationScheme recombSchemeBkg{fastjet::E_scheme};
63  fastjet::JetDefinition jetDefBkg{algorithmBkg, rParamBkg, recombSchemeBkg, strategy};
64  fastjet::AreaDefinition areaDefBkg{areaType, ghostSpec};
65  fastjet::Selector selBkg = fastjet::SelectorAbsRapMax(rapBkg);
66 
67  // TODO: use abs, eventually use pt
68  // TODO: use values from configurables
69  // TODO: add eta cuts
70  Filter trackCuts = (aod::track::signed1Pt < 10.f) &&
71  (aod::track::signed1Pt > -10.f);
72 
73  std::unique_ptr<fastjet::BackgroundEstimatorBase> bge;
74  std::unique_ptr<fastjet::Subtractor> sub;
75 
76  std::vector<fastjet::PseudoJet> pJets;
77 
78  void init(InitContext const&)
79  {
80  if (bkgMode == BkgMode::none) {
81  } else if (bkgMode == BkgMode::rhoArea) {
82  bge = decltype(bge)(new fastjet::JetMedianBackgroundEstimator(selBkg, jetDefBkg, areaDefBkg));
83  sub = decltype(sub){new fastjet::Subtractor{bge.get()}};
84  } else {
85  LOGF(ERROR, "requested subtraction mode not implemented!");
86  }
87  }
88 
89  void process(aod::Collision const& collision,
90  soa::Filtered<aod::Tracks> const& fullTracks)
91  {
92  // TODO: retrieve pion mass from somewhere
93  const float mPionSquared = 0.139 * 0.139;
94 
95  pJets.clear();
96 
97  std::vector<fastjet::PseudoJet> inputParticles;
98  for (auto& track : fullTracks) {
99  auto energy = std::sqrt(track.p2() + mPionSquared);
100  inputParticles.emplace_back(track.px(), track.py(), track.pz(), energy);
101  inputParticles.back().set_user_index(track.globalIndex());
102  }
103 
104  fastjet::ClusterSequenceArea clust_seq(inputParticles, jetDef, areaDef);
105  if (bge)
106  bge->set_particles(inputParticles);
107  pJets = sub ? (*sub)(clust_seq.inclusive_jets()) : clust_seq.inclusive_jets();
108 
109  pJets = selJet(pJets);
110 
111  for (const auto& pjet : pJets) {
112  jets(collision, pjet.eta(), pjet.phi(), pjet.pt(),
113  pjet.area(), pjet.Et(), pjet.m());
114  for (const auto& track : pjet.constituents()) {
115  LOGF(DEBUG, "jet %d constituent %d: %f %f %f", jets.lastIndex(),
116  track.user_index(), track.eta(), track.phi(), track.pt());
117  constituents(jets.lastIndex(), track.user_index());
118  }
119  }
120  }
121 };
