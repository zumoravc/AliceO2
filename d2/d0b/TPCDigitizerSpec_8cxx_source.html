69  void initDigitizerTask(framework::InitContext& ic) override
70  {
71  LOG(INFO) << "Initializing TPC digitization";
72 
73  mWithMCTruth = o2::conf::DigiParams::Instance().mctruth;
74  auto useDistortions = ic.options().get<int>("distortionType");
75  auto triggeredMode = ic.options().get<bool>("TPCtriggered");
76 
77  if (useDistortions > 0) {
78  if (useDistortions == 1) {
79  LOG(INFO) << "Using realistic space-charge distortions.";
80  } else {
81  LOG(INFO) << "Using constant space-charge distortions.";
82  }
83  auto readSpaceChargeString = ic.options().get<std::string>("readSpaceCharge");
84  std::vector<std::string> readSpaceCharge;
85  std::stringstream ssSpaceCharge(readSpaceChargeString);
86  while (ssSpaceCharge.good()) {
87  std::string substr;
88  getline(ssSpaceCharge, substr, ',');
89  readSpaceCharge.push_back(substr);
90  }
91  if (readSpaceCharge[0].size() != 0) { // use pre-calculated space-charge object
92  std::unique_ptr<o2::tpc::SpaceCharge> spaceCharge;
93  if (!gSystem->AccessPathName(readSpaceCharge[0].data())) {
94  auto fileSC = std::unique_ptr<TFile>(TFile::Open(readSpaceCharge[0].data()));
95  if (fileSC->FindKey(readSpaceCharge[1].data())) {
96  spaceCharge.reset((o2::tpc::SpaceCharge*)fileSC->Get(readSpaceCharge[1].data()));
97  }
98  }
99  if (spaceCharge.get() != nullptr) {
100  LOG(INFO) << "Using pre-calculated space-charge object: " << readSpaceCharge[1].data();
101  mDigitizer.setUseSCDistortions(spaceCharge.release());
102  } else {
103  LOG(ERROR) << "Space-charge object or file not found!";
104  }
105  } else { // create new space-charge object either with empty TPC or an initial space-charge density provided by histogram
106  o2::tpc::SpaceCharge::SCDistortionType distortionType = useDistortions == 2 ? o2::tpc::SpaceCharge::SCDistortionType::SCDistortionsConstant : o2::tpc::SpaceCharge::SCDistortionType::SCDistortionsRealistic;
107  auto gridSizeString = ic.options().get<std::string>("gridSize");
108  std::vector<int> gridSize;
109  std::stringstream ss(gridSizeString);
110  while (ss.good()) {
111  std::string substr;
112  getline(ss, substr, ',');
113  gridSize.push_back(std::stoi(substr));
114  }
115  auto inputHistoString = ic.options().get<std::string>("initialSpaceChargeDensity");
116  std::vector<std::string> inputHisto;
117  std::stringstream ssHisto(inputHistoString);
118  while (ssHisto.good()) {
119  std::string substr;
120  getline(ssHisto, substr, ',');
121  inputHisto.push_back(substr);
122  }
123  std::unique_ptr<TH3> hisSCDensity;
124  if (!gSystem->AccessPathName(inputHisto[0].data())) {
125  auto fileSCInput = std::unique_ptr<TFile>(TFile::Open(inputHisto[0].data()));
126  if (fileSCInput->FindKey(inputHisto[1].data())) {
127  hisSCDensity.reset((TH3*)fileSCInput->Get(inputHisto[1].data()));
128  hisSCDensity->SetDirectory(nullptr);
129  }
130  }
131  if (hisSCDensity.get() != nullptr) {
132  LOG(INFO) << "TPC: Providing initial space-charge density histogram: " << hisSCDensity->GetName();
133  mDigitizer.setUseSCDistortions(distortionType, hisSCDensity.get(), gridSize[0], gridSize[1], gridSize[2]);
134  } else {
135  if (distortionType == SpaceCharge::SCDistortionType::SCDistortionsConstant) {
136  LOG(ERROR) << "Input space-charge density histogram or file not found!";
137  }
138  }
139  }
140  }
141  mDigitizer.setContinuousReadout(!triggeredMode);
142 
143  // we send the GRP data once if the corresponding output channel is available
144  // and set the flag to false after
145  mWriteGRP = true;
146  }

148  void run(framework::ProcessingContext& pc)
149  {
150  LOG(INFO) << "Processing TPC digitization";
151 
153  auto& cdb = o2::tpc::CDBInterface::instance();
154  cdb.setUseDefaults();
155  if (!gSystem->AccessPathName("GainMap.root")) {
156  LOG(INFO) << "TPC: Using gain map from 'GainMap.root'";
157  cdb.setGainMapFromFile("GainMap.root");
158  }
159 
160  for (auto it = pc.inputs().begin(), end = pc.inputs().end(); it != end; ++it) {
161  for (auto const& inputref : it) {
162  process(pc, inputref);
163  }
164  }

167  // process one sector
168  void process(framework::ProcessingContext& pc, framework::DataRef const& inputref)
169  {
170  // read collision context from input
171  auto context = pc.inputs().get<o2::steer::DigitizationContext*>(inputref);
172  context->initSimChains(o2::detectors::DetID::TPC, mSimChains);
173  auto& irecords = context->getEventRecords();
174  LOG(INFO) << "TPC: Processing " << irecords.size() << " collisions";
175  if (irecords.size() == 0) {
176  return;
177  }
178  auto const* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(inputref);
179 
180  bool isContinuous = mDigitizer.isContinuousReadout();
181  // we publish the GRP data once if the output channel is there
182  if (mWriteGRP && pc.outputs().isAllowed({"TPC", "ROMode", 0})) {
183  auto roMode = isContinuous ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
184  LOG(INFO) << "TPC: Sending ROMode= " << (mDigitizer.isContinuousReadout() ? "Continuous" : "Triggered")
185  << " to GRPUpdater from channel " << dh->subSpecification;
186  pc.outputs().snapshot(Output{"TPC", "ROMode", 0, Lifetime::Timeframe}, roMode);
187  }
188  mWriteGRP = false;
189 
190  // extract which sector to treat
191  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(inputref);
192  if (sectorHeader == nullptr) {
193  LOG(ERROR) << "TPC: Sector header missing, skipping processing";
194  return;
195  }
196  auto sector = sectorHeader->sector();
197  LOG(INFO) << "TPC: Processing sector " << sector;
198  // the active sectors need to be propagated
199  uint64_t activeSectors = 0;
200  activeSectors = sectorHeader->activeSectors;
201 
202  // lambda that snapshots digits to be sent out; prepares and attaches header with sector information
203  auto snapshotDigits = [this, sector, &pc, activeSectors, &dh](std::vector<o2::tpc::Digit> const& digits) {
204  o2::tpc::TPCSectorHeader header{sector};
205  header.activeSectors = activeSectors;
206  // note that snapshoting only works with non-const references (to be fixed?)
207  pc.outputs().snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe,
208  header},
209  const_cast<std::vector<o2::tpc::Digit>&>(digits));
210  };
211  // lambda that snapshots the common mode vector to be sent out; prepares and attaches header with sector information
212  auto snapshotCommonMode = [this, sector, &pc, activeSectors, &dh](std::vector<o2::tpc::CommonMode> const& commonMode) {
213  o2::tpc::TPCSectorHeader header{sector};
214  header.activeSectors = activeSectors;
215  // note that snapshoting only works with non-const references (to be fixed?)
216  pc.outputs().snapshot(Output{"TPC", "COMMONMODE", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe,
217  header},
218  const_cast<std::vector<o2::tpc::CommonMode>&>(commonMode));
219  };
220  // lambda that snapshots labels to be sent out; prepares and attaches header with sector information
221  auto snapshotLabels = [this, &sector, &pc, activeSectors, &dh](o2::dataformats::MCTruthContainer<o2::MCCompLabel> const& labels) {
222  o2::tpc::TPCSectorHeader header{sector};
223  header.activeSectors = activeSectors;
224  if (mWithMCTruth) {
225  pc.outputs().snapshot(Output{"TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(dh->subSpecification),
226  Lifetime::Timeframe, header},
227  const_cast<o2::dataformats::MCTruthContainer<o2::MCCompLabel>&>(labels));
228  }
229  };
230  // lambda that snapshots digits grouping (triggers) to be sent out; prepares and attaches header with sector information
231  auto snapshotEvents = [this, sector, &pc, activeSectors, &dh](const std::vector<DigiGroupRef>& events) {
232  o2::tpc::TPCSectorHeader header{sector};
233  header.activeSectors = activeSectors;
234  LOG(INFO) << "TPC: Send TRIGGERS for sector " << sector << " channel " << dh->subSpecification << " | size " << events.size();
235  pc.outputs().snapshot(Output{"TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe,
236  header},
237  const_cast<std::vector<DigiGroupRef>&>(events));
238  };
239 
240  std::vector<o2::tpc::Digit> digitsAccum; // accumulator for digits
241  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum; // timeframe accumulator for labels
242  std::vector<CommonMode> commonModeAccum;
243  std::vector<DigiGroupRef> eventAccum;
244 
245  // this should not happen any more, legacy condition when the sector variable was used
246  // to transport control information
247  if (sector < 0) {
248  throw std::runtime_error("Legacy control information is not expected any more");
249  }
250 
251  // the TPCSectorHeader now allows to transport information for more than one sector,
252  // e.g. for transporting clusters in one single data block. The digitization is however
253  // only on sector level
254  if (sector >= TPCSectorHeader::NSectors) {
255  throw std::runtime_error("Digitizer can only work on single sectors");
256  }
257 
258  mDigitizer.setSector(sector);
259  mDigitizer.init();
260 
261  auto& eventParts = context->getEventParts();
262 
263  auto flushDigitsAndLabels = [this, &digitsAccum, &labelAccum, &commonModeAccum](bool finalFlush = false) {
264  // flush previous buffer
265  mDigits.clear();
266  mLabels.clear();
267  mCommonMode.clear();
268  mDigitizer.flush(mDigits, mLabels, mCommonMode, finalFlush);
269  LOG(INFO) << "TPC: Flushed " << mDigits.size() << " digits, " << mLabels.getNElements() << " labels and " << mCommonMode.size() << " common mode entries";
270  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(digitsAccum));
271  if (mWithMCTruth) {
272  labelAccum.mergeAtBack(mLabels);
273  }
274  std::copy(mCommonMode.begin(), mCommonMode.end(), std::back_inserter(commonModeAccum));
275  };
276 
277  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
278  mDigitizer.setStartTime(sampaProcessing.getTimeBinFromTime(irecords[0].timeNS / 1000.f));
279 
280  TStopwatch timer;
281  timer.Start();
282 
283  // loop over all composite collisions given from context
284  // (aka loop over all the interaction records)
285  for (int collID = 0; collID < irecords.size(); ++collID) {
286  const float eventTime = irecords[collID].timeNS / 1000.f;
287  LOG(INFO) << "TPC: Event time " << eventTime << " us";
288  mDigitizer.setEventTime(eventTime);
289  if (!isContinuous) {
290  mDigitizer.setStartTime(sampaProcessing.getTimeBinFromTime(eventTime));
291  }
292  int startSize = digitsAccum.size();
293 
294  // for each collision, loop over the constituents event and source IDs
295  // (background signal merging is basically taking place here)
296  for (auto& part : eventParts[collID]) {
297  const int eventID = part.entryID;
298  const int sourceID = part.sourceID;
299 
300  // get the hits for this event and this source
301  std::vector<o2::tpc::HitGroup> hitsLeft;
302  std::vector<o2::tpc::HitGroup> hitsRight;
303  context->retrieveHits(mSimChains, getBranchNameLeft(sector).c_str(), part.sourceID, part.entryID, &hitsLeft);
304  context->retrieveHits(mSimChains, getBranchNameRight(sector).c_str(), part.sourceID, part.entryID, &hitsRight);
305  LOG(DEBUG) << "TPC: Found " << hitsLeft.size() << " hit groups left and " << hitsRight.size() << " hit groups right in collision " << collID << " eventID " << part.entryID;
306 
307  mDigitizer.process(hitsLeft, eventID, sourceID);
308  mDigitizer.process(hitsRight, eventID, sourceID);
309 
310  flushDigitsAndLabels();
311 
312  if (!isContinuous) {
313  eventAccum.emplace_back(startSize, digitsAccum.size() - startSize);
314  }
315  }
316  }
317 
318  // final flushing step; getting everything not yet written out
319  if (isContinuous) {
320  LOG(INFO) << "TPC: Final flush";
321  flushDigitsAndLabels(true);
322  eventAccum.emplace_back(0, digitsAccum.size()); // all digits are grouped to 1 super-event pseudo-triggered mode
323  }
324 
325  // send out to next stage
326  snapshotEvents(eventAccum);
327  snapshotDigits(digitsAccum);
328  snapshotCommonMode(commonModeAccum);
329  snapshotLabels(labelAccum);
330 
331  timer.Stop();
332  LOG(INFO) << "TPC: Digitization took " << timer.CpuTime() << "s";
333  }
334 
335  private:
336  o2::tpc::Digitizer mDigitizer;
337  std::vector<TChain*> mSimChains;
338  std::vector<o2::tpc::Digit> mDigits;
339  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabels;
340  std::vector<o2::tpc::CommonMode> mCommonMode;
341  bool mWriteGRP = false;
342  bool mWithMCTruth = true;
343 };

345 o2::framework::DataProcessorSpec getTPCDigitizerSpec(int channel, bool writeGRP, bool mctruth)
346 {
347  // create the full data processor spec using
348  // a name identifier
349  // input description
350  // algorithmic description (here a lambda getting called once to setup the actual processing function)
351  // options that can be used for this processor (here: input file names where to take the hits)
352  std::stringstream id;
353  id << "TPCDigitizer" << channel;
354 
355  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
356 
357  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
358  outputs.emplace_back("TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
359  if (mctruth) {
360  outputs.emplace_back("TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
361  }
362  outputs.emplace_back("TPC", "COMMONMODE", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
363  if (writeGRP) {
364  outputs.emplace_back("TPC", "ROMode", 0, Lifetime::Timeframe);
365  LOG(DEBUG) << "TPC: Channel " << channel << " will supply ROMode";
366  }
367 
368  return DataProcessorSpec{
369  id.str().c_str(),
370  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
371  outputs,
372  AlgorithmSpec{adaptFromTask<TPCDPLDigitizerTask>()},
373  Options{{"distortionType", VariantType::Int, 0, {"Distortion type to be used. 0 = no distortions (default), 1 = realistic distortions (not implemented yet), 2 = constant distortions"}},
374  {"gridSize", VariantType::String, "129,144,129", {"Comma separated list of number of bins in (r,phi,z) for distortion lookup tables (r and z can only be 2**N + 1, N=1,2,3,...)"}},
375  {"initialSpaceChargeDensity", VariantType::String, "", {"Path to root file containing TH3 with initial space-charge density and name of the TH3 (comma separated)"}},
376  {"readSpaceCharge", VariantType::String, "", {"Path to root file containing pre-calculated space-charge object and name of the object (comma separated)"}},
377  {"TPCtriggered", VariantType::Bool, false, {"Impose triggered RO mode (default: continuous)"}}}};
378 }

380 o2::framework::WorkflowSpec getTPCDigitizerSpec(int nLanes, std::vector<int> const& sectors, bool mctruth)
381 {
382  // channel parameter is deprecated in the TPCDigitizer processor, all descendants
383  // are initialized not to publish GRP mode, but the channel will be added to the first
384  // processor after the pipelines have been created. The processor will decide upon
385  // the index in the ParallelContext whether to publish
386  WorkflowSpec pipelineTemplate{getTPCDigitizerSpec(0, false, mctruth)};
387  // override the predefined name, index will be added by parallelPipeline method
388  pipelineTemplate[0].name = "TPCDigitizer";
389  WorkflowSpec pipelines = parallelPipeline(
390  pipelineTemplate, nLanes, [size = sectors.size()]() { return size; }, [&sectors](size_t index) { return sectors[index]; });
391  // add the channel for the GRP information to the first processor
392  pipelines[0].outputs.emplace_back("TPC", "ROMode", 0, Lifetime::Timeframe);
393  return pipelines;
394 }
