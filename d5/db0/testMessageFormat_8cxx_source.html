41  "data1",
42  "anotherDataSet"};
43 
44  std::vector<BlockDescriptor> dataDescriptors;
45  for (auto& dataField : dataFields) {
46  dataDescriptors.emplace_back((void*)dataField.c_str(), dataField.size() + 1, AliHLTComponentDataTypeInitializer("TESTDATA", "TEST"), 0);
47  }
48 
49  unsigned totalPayloadSize = 0;
50  for (auto& desc : dataDescriptors) {
51  totalPayloadSize += desc.fSize;
52  hexDump("HLT data descriptor", &desc, sizeof(desc));
53  if (!desc.fPtr)
54  continue;
55  hexDump(" data payload", (char*)desc.fPtr + desc.fOffset, desc.fSize);
56  }
57 
58  // testing without event info
59  auto outputs = handler.createMessages(&dataDescriptors[0], dataDescriptors.size(), totalPayloadSize);
60  BOOST_REQUIRE(outputs.size() == dataDescriptors.size());
61  unsigned dataidx = 0;
62  for (auto& output : outputs) {
63  hexDump("Output block (w/o event info)", output.mP, output.mSize);
64  const char* data = (char*)output.mP + sizeof(AliHLTComponentBlockData);
65  BOOST_CHECK(dataFields[dataidx++] == data);
66  }
67 
68  // testing with dummy event info
69  AliHLTComponentEventData evtData;
70  handler.clear();
71  outputs.clear();
72  outputs = handler.createMessages(&dataDescriptors[0], dataDescriptors.size(), totalPayloadSize, &evtData);
73  dataidx = 0;
74  for (auto& output : outputs) {
75  std::string debugMessage = "Output block (with";
76  debugMessage += (dataidx == 0 ? "" : "out");
77  debugMessage += " event info)";
78  hexDump(debugMessage.c_str(), output.mP, output.mSize);
79  const char* data = (char*)output.mP + sizeof(AliHLTComponentBlockData) + (dataidx == 0 ? sizeof(AliHLTComponentEventData) : 0);
80  BOOST_CHECK(dataFields[dataidx++] == data);
81  }
