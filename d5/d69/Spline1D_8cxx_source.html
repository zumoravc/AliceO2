114 template <typename DataT>
115 void Spline1D<DataT>::recreate(int numberOfKnots, const int inputKnots[], int nFdimensions)
116 {
126 
127  FlatObject::startConstruction();
128 
129  std::vector<int> knotU;
130 
131  { // sort knots
132  std::vector<int> tmp;
133  for (int i = 0; i < numberOfKnots; i++) {
134  tmp.push_back(inputKnots[i]);
135  }
136  std::sort(tmp.begin(), tmp.end());
137 
138  knotU.push_back(0); // first knot at 0
139 
140  for (unsigned int i = 1; i < tmp.size(); ++i) {
141  int u = tmp[i] - tmp[0];
142  if (knotU.back() < u) { // remove duplicated knots
143  knotU.push_back(u);
144  }
145  }
146  if (knotU.back() < 1) { // at least 2 knots
147  knotU.push_back(1);
148  }
149  }
150 
151  mNumberOfKnots = knotU.size();
152  mUmax = knotU.back();
153  mFdimensions = nFdimensions;
154  mXmin = 0.;
155  mXtoUscale = 1.;
156 
157  const int uToKnotMapOffset = mNumberOfKnots * sizeof(Spline1D::Knot);
158  const int parametersOffset = uToKnotMapOffset + (mUmax + 1) * sizeof(int);
159  const int bufferSize = parametersOffset + getSizeOfParameters(mFdimensions);
160 
161  FlatObject::finishConstruction(bufferSize);
162 
163  mUtoKnotMap = reinterpret_cast<int*>(mFlatBufferPtr + uToKnotMapOffset);
164  mFparameters = reinterpret_cast<DataT*>(mFlatBufferPtr + parametersOffset);
165 
166  Knot* s = getKnots();
167 
168  for (int i = 0; i < mNumberOfKnots; i++) {
169  s[i].u = knotU[i];
170  }
171 
172  for (int i = 0; i < mNumberOfKnots - 1; i++) {
173  s[i].Li = 1. / (s[i + 1].u - s[i].u); // do division in double
174  }
175 
176  s[mNumberOfKnots - 1].Li = 0.; // the value will not be used, we define it for consistency
177 
178  // Set up the map (integer U) -> (knot index)
179 
180  int* map = getUtoKnotMap();
181 
182  const int iKnotMax = mNumberOfKnots - 2;
183 
184  //
185  // With iKnotMax=nKnots-2 we map the U==Umax coordinate to the last [nKnots-2, nKnots-1] segment.
186  // This trick allows one to avoid a special condition for this edge case.
187  // Any U from [0,Umax] is mapped to some knot_i such, that the next knot_i+1 always exist
188  //
189 
190  for (int u = 0, iKnot = 0; u <= mUmax; u++) {
191  if ((knotU[iKnot + 1] == u) && (iKnot < iKnotMax)) {
192  iKnot = iKnot + 1;
193  }
194  map[u] = iKnot;
195  }
196 
197  for (int i = 0; i < getNumberOfParameters(mFdimensions); i++) {
198  mFparameters[i] = 0.f;
199  }
200 }

261 template <typename DataT>
262 int Spline1D<DataT>::test(const bool draw, const bool drawDataPoints)
263 {
264  using namespace std;
265 
266  // input function F
267 
268  const int Ndim = 5;
269  const int Fdegree = 4;
270  double Fcoeff[Ndim][2 * (Fdegree + 1)];
271 
272  auto F = [&](DataT x, DataT f[]) -> void {
273  double cosx[Fdegree + 1], sinx[Fdegree + 1];
274  double xi = 0;
275  for (int i = 0; i <= Fdegree; i++, xi += x) {
276  GPUCommonMath::SinCos(xi, sinx[i], cosx[i]);
277  }
278  for (int dim = 0; dim < Ndim; dim++) {
279  f[dim] = 0; // Fcoeff[0]/2;
280  for (int i = 1; i <= Fdegree; i++) {
281  f[dim] += Fcoeff[dim][2 * i] * cosx[i] + Fcoeff[dim][2 * i + 1] * sinx[i];
282  }
283  }
284  };
285 
286  TCanvas* canv = nullptr;
287  TNtuple* nt = nullptr;
288  TNtuple* knots = nullptr;
289 
290  auto ask = [&]() -> bool {
291  if (!canv) {
292  return 0;
293  }
294  canv->Update();
295  cout << "type 'q ' to exit" << endl;
296  std::string str;
297  std::getline(std::cin, str);
298  return (str != "q" && str != ".q");
299  };
300 
301  std::cout << "Test 1D interpolation with the compact spline" << std::endl;
302 
303  int nTries = 100;
304 
305  if (draw) {
306  canv = new TCanvas("cQA", "Spline1D QA", 1000, 600);
307  nTries = 10000;
308  }
309 
310  double statDf1 = 0;
311  double statDf2 = 0;
312  double statDf1D = 0;
313  double statN = 0;
314 
315  int seed = 1;
316 
317  for (int itry = 0; itry < nTries; itry++) {
318 
319  // init random F
320  for (int dim = 0; dim < Ndim; dim++) {
321  gRandom->SetSeed(seed++);
322  for (int i = 0; i < 2 * (Fdegree + 1); i++) {
323  Fcoeff[dim][i] = gRandom->Uniform(-1, 1);
324  }
325  }
326 
327  // spline
328 
329  int nKnots = 4;
330  const int uMax = nKnots * 3;
331 
332  Spline1D spline1;
333  int knotsU[nKnots];
334 
335  do { // set knots randomly
336  knotsU[0] = 0;
337  double du = 1. * uMax / (nKnots - 1);
338  for (int i = 1; i < nKnots; i++) {
339  knotsU[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
340  }
341  knotsU[nKnots - 1] = uMax;
342  spline1.recreate(nKnots, knotsU, Ndim);
343  if (nKnots != spline1.getNumberOfKnots()) {
344  cout << "warning: n knots changed during the initialisation " << nKnots
345  << " -> " << spline1.getNumberOfKnots() << std::endl;
346  continue;
347  }
348  } while (0);
349 
350  std::string err = FlatObject::stressTest(spline1);
351  if (!err.empty()) {
352  cout << "error at FlatObject functionality: " << err << endl;
353  return -1;
354  } else {
355  // cout << "flat object functionality is ok" << endl;
356  }
357 
358  nKnots = spline1.getNumberOfKnots();
359  int nAxiliaryPoints = 1;
360  Spline1D spline2; //(spline1);
361  spline2.cloneFromObject(spline1, nullptr);
362 
363  spline1.approximateFunction(0., TMath::Pi(), F, nAxiliaryPoints);
364 
365  //if (itry == 0)
366  {
367  TFile outf("testSpline1D.root", "recreate");
368  if (outf.IsZombie()) {
369  cout << "Failed to open output file testSpline1D.root " << std::endl;
370  } else {
371  const char* name = "spline1Dtest";
372  spline1.writeToFile(outf, name);
373  Spline1D<DataT>* p = spline1.readFromFile(outf, name);
374  if (p == nullptr) {
375  cout << "Failed to read Spline1D from file testSpline1D.root " << std::endl;
376  } else {
377  spline1 = *p;
378  }
379  outf.Close();
380  }
381  }
382 
383  SplineHelper1D<DataT> helper;
384  helper.setSpline(spline2, Ndim, nAxiliaryPoints);
385  helper.approximateFunctionGradually(spline2, 0., TMath::Pi(), F, nAxiliaryPoints);
386 
387  // 1-D splines for each dimension
388  Spline1D splines3[Ndim];
389  {
390  for (int dim = 0; dim < Ndim; dim++) {
391  auto F3 = [&](DataT u, DataT f[]) -> void {
392  DataT ff[Ndim];
393  F(u, ff);
394  f[0] = ff[dim];
395  };
396  splines3[dim].recreate(nKnots, knotsU, 1);
397  splines3[dim].approximateFunction(0., TMath::Pi(), F3, nAxiliaryPoints);
398  }
399  }
400 
401  double stepX = 1.e-2;
402  for (double x = 0; x < TMath::Pi(); x += stepX) {
403  DataT f[Ndim], s1[Ndim], s2[Ndim];
404  F(x, f);
405  spline1.interpolate(x, s1);
406  spline2.interpolate(x, s2);
407  for (int dim = 0; dim < Ndim; dim++) {
408  statDf1 += (s1[dim] - f[dim]) * (s1[dim] - f[dim]);
409  statDf2 += (s2[dim] - f[dim]) * (s2[dim] - f[dim]);
410  DataT s1D = splines3[dim].interpolate(x);
411  statDf1D += (s1D - s1[dim]) * (s1D - s1[dim]);
412  }
413  statN += Ndim;
414  }
415  // cout << "std dev : " << sqrt(statDf1 / statN) << std::endl;
416 
417  if (draw) {
418  delete nt;
419  delete knots;
420  nt = new TNtuple("nt", "nt", "u:f:s");
421  DataT drawMax = -1.e20;
422  DataT drawMin = 1.e20;
423  DataT stepX = 1.e-4;
424  for (double x = 0; x < TMath::Pi(); x += stepX) {
425  DataT f[Ndim], s[Ndim];
426  F(x, f);
427  spline1.interpolate(x, s);
428  nt->Fill(spline1.convXtoU(x), f[0], s[0]);
429  drawMax = std::max(drawMax, std::max(f[0], s[0]));
430  drawMin = std::min(drawMin, std::min(f[0], s[0]));
431  }
432 
433  nt->SetMarkerStyle(8);
434 
435  {
436  TNtuple* ntRange = new TNtuple("ntRange", "nt", "u:f");
437  drawMin -= 0.1 * (drawMax - drawMin);
438 
439  ntRange->Fill(0, drawMin);
440  ntRange->Fill(0, drawMax);
441  ntRange->Fill(uMax, drawMin);
442  ntRange->Fill(uMax, drawMax);
443  ntRange->SetMarkerColor(kWhite);
444  ntRange->SetMarkerSize(0.1);
445  ntRange->Draw("f:u", "", "");
446  delete ntRange;
447  }
448 
449  nt->SetMarkerColor(kGray);
450  nt->SetMarkerSize(2.);
451  nt->Draw("f:u", "", "P,same");
452 
453  nt->SetMarkerSize(.5);
454  nt->SetMarkerColor(kBlue);
455  nt->Draw("s:u", "", "P,same");
456 
457  knots = new TNtuple("knots", "knots", "type:u:s");
458  for (int i = 0; i < nKnots; i++) {
459  double u = spline1.getKnot(i).u;
460  DataT s[Ndim];
461  spline1.interpolate(spline1.convUtoX(u), s);
462  knots->Fill(1, u, s[0]);
463  }
464 
465  knots->SetMarkerStyle(8);
466  knots->SetMarkerSize(1.5);
467  knots->SetMarkerColor(kRed);
468  knots->SetMarkerSize(1.5);
469  knots->Draw("s:u", "type==1", "same"); // knots
470 
471  if (drawDataPoints) {
472  for (int j = 0; j < helper.getNumberOfDataPoints(); j++) {
473  const typename SplineHelper1D<DataT>::DataPoint& p = helper.getDataPoint(j);
474  if (p.isKnot) {
475  continue;
476  }
477  DataT s[Ndim];
478  spline1.interpolate(spline1.convUtoX(p.u), s);
479  knots->Fill(2, p.u, s[0]);
480  }
481  knots->SetMarkerColor(kBlack);
482  knots->SetMarkerSize(1.);
483  knots->Draw("s:u", "type==2", "same"); // data points
484  }
485 
486  if (!ask()) {
487  break;
488  }
489  } // draw
490  }
491  //delete canv;
492  //delete nt;
493  //delete knots;
494 
495  statDf1 = sqrt(statDf1 / statN);
496  statDf2 = sqrt(statDf2 / statN);
497  statDf1D = sqrt(statDf1D / statN);
498 
499  cout << "\n std dev for Compact Spline : " << statDf1 << " / " << statDf2
500  << std::endl;
501  cout << " mean difference between 1-D and " << Ndim
502  << "-D splines : " << statDf1D << std::endl;
503 
504  if (statDf1 < 0.05 && statDf2 < 0.06 && statDf1D < 1.e-20) {
505  cout << "Everything is fine" << endl;
506  } else {
507  cout << "Something is wrong!!" << endl;
508  return -2;
509  }
510  return 0;
511 }
