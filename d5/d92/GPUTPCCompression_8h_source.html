105 template <class T>
106 GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, unsigned int nBits, unsigned int max)
107 {
108  if (nBits == 0) {
109  return;
110  }
111 
112  unsigned int val = v;
113  unsigned int ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
114  if (val && ldz > nBits) {
115  if (val & (1 << (ldz - nBits - 1))) {
116  val += (1 << (ldz - nBits - 1));
117  ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
118  }
119  val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
120  if (val >= max) {
121  val = max - 1;
122  }
123  // GPUInfo("CHANGING X %x --> %x", (unsigned int) v, val);
124  v = val;
125  }
