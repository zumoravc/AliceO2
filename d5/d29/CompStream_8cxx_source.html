35 void pushDecompressor(T& stream, CompressionMethod method)
36 {
37  switch (method) {
38  case CompressionMethod::Gzip:
39  stream.push(boost::iostreams::gzip_decompressor());
40  break;
41  case CompressionMethod::Zlib:
42  stream.push(boost::iostreams::zlib_decompressor());
43  break;
44  case CompressionMethod::Lzma:
45  throw std::runtime_error("lzma support not enabled");
46  //stream.push(boost::iostreams::lzma_decompressor());
47  break;
48  case CompressionMethod::Bzip2:
49  stream.push(boost::iostreams::bzip2_decompressor());
50  break;
51  case CompressionMethod::None:
52  break;
53  default:

90 auto Method(std::string method)
91 {
92  auto const& i = Mapping.find(method);
93  if (i == Mapping.end()) {
94  throw std::invalid_argument("invalid compression method: " + method);
95  }
96  return i->second;
