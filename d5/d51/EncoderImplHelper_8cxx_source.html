20 uint16_t chipAddress(uint8_t elinkId, uint8_t chId)

23  if (!opt.has_value()) {
24  throw std::invalid_argument(fmt::format("elinkId {} is not valid", elinkId));

33 SampaHeader buildSampaHeader(uint8_t elinkId, uint8_t chId, gsl::span<const SampaCluster> data)
34 {
35  assertIsInRange("chId", chId, 0, 63);
36 
37  SampaHeader header;
38  header.packetType(SampaPacketType::Data);
39 
40  uint16_t n10{0};
41  for (const auto& s : data) {
42  n10 += s.nof10BitWords();
43  }
44  uint32_t bunchCrossingCounter{0};
45  if (data.size()) {
46  bunchCrossingCounter = data[0].bunchCrossing;
47  }
48  assertNofBits("nof10BitWords", n10, 10);
49  header.nof10BitWords(n10);
50  header.chipAddress(chipAddress(elinkId, chId));
51  header.channelAddress(chId % 32);
52  header.hammingCode(computeHammingCode(header.uint64()));
53  header.headerParity(computeHeaderParity(header.uint64()));
54  assertNofBits("bunchCrossingCounter", bunchCrossingCounter, 20);
55  header.bunchCrossingCounter(bunchCrossingCounter);
56 
57  // FIXME: compute payload parity
58  return header;
59 }

61 uint64_t build64(uint16_t a10, uint16_t b10 = 0, uint16_t c10 = 0, uint16_t d10 = 0, uint16_t e10 = 0)
62 {
63  impl::assertIsInRange("a10", a10, 0, 1023);
64  impl::assertIsInRange("b10", a10, 0, 1023);
65  impl::assertIsInRange("c10", a10, 0, 1023);
66  impl::assertIsInRange("d10", a10, 0, 1023);
67  impl::assertIsInRange("e10", a10, 0, 1023);
68  return (static_cast<uint64_t>(a10) << 40) |
69  (static_cast<uint64_t>(b10) << 30) |
70  (static_cast<uint64_t>(c10) << 20) |
71  (static_cast<uint64_t>(d10) << 10) |
72  (static_cast<uint64_t>(e10));
73 }

83 void b10to64(std::vector<uint10_t> b10, std::vector<uint64_t>& b64, uint16_t prefix14)
84 {
85  uint64_t prefix = prefix14;
86  prefix <<= 50;
87 
88  addPadding(b10);
89 
90  for (auto i = 0; i < b10.size(); i += 5) {
91  uint64_t v = build64(b10[i + 4], b10[i + 3], b10[i + 2], b10[i + 1], b10[i + 0]);
92  b64.emplace_back(v | prefix);
93  }
94 }

96 void bufferizeClusters(gsl::span<const SampaCluster> clusters, std::vector<uint10_t>& b10)
97 {
98  for (auto& c : clusters) {
99  b10.emplace_back(c.nofSamples());
100  b10.emplace_back(c.sampaTime);
101  if (c.isClusterSum()) {
102  b10.emplace_back(c.chargeSum & 0x3FF);
103  b10.emplace_back((c.chargeSum & 0xFFC00) >> 10);
104  } else {
105  for (auto& s : c.samples) {
106  b10.emplace_back(s);
107  }
108  }
109  }
110 }

112 void append(std::vector<uint10_t>& b10, uint50_t value)
113 {
114  b10.emplace_back((value & 0x3FF));
115  b10.emplace_back((value & 0xFFC00) >> 10);
116  b10.emplace_back((value & 0x3FF00000) >> 20);
117  b10.emplace_back((value & 0xFFC0000000) >> 30);
118  b10.emplace_back((value & 0x3FF0000000000) >> 40);
