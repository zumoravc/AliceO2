26 void SpaceFrame::CreateGeometry()
27 {
28  LOG(DEBUG) << "Create CalFrame Geometry" << std::endl;
29 
30  /************ Definition of constants *****************************/
31 
32  // space frame parameters from "SINGLE FRAME ASSEMBLY 27D624H.pdf"
33  // provided by Lawrence Berkeley Labs, USA
34  const int NUMCROSS = 12;
35  const int NUMSUBSETS = 3;
36  const double TOTALHALFWIDTH = 152.3; // Half Width of a Half Frame
37  // (CalFrame comes in 2 sections)
38  const double BEGINPHI = 76.8;
39  const double ENDPHI = 193.03;
40  const double BEGINRADIUS = 490.;
41 
42  const double HALFFRAMETRANS = TOTALHALFWIDTH + 57.2 / 2.; // Half Frame Connector is 57.2cm wide,
43  // Supermodule is 340cm wide
44  // Sources: HALF-FRAME-CONNECTOR-27E226A.pdf
45  // provided by LBL
46 
47  const double FLAGEWIDTH = 15.2;
48  const double RIBWIDTH = 1.5;
49  const double CROSSBOTTOMHEIGHT = 15.2;
50  const double CROSSBOTTOMRADTHICK = 1.5;
51  const double CROSSTOPHEIGHT = 1.5;
52  const double CROSSTOPRADTHICK = 35. / 2.;
53 
54  //const double TOTALPHI = ENDPHI - BEGINPHI;
55  const double FLAGEHEIGHT = BEGINRADIUS + 3.;
56  const double RIBHEIGHT = FLAGEHEIGHT + 35;
57  const double CROSSBOTTOMWIDTH = 0.5 / (Double_t)NUMSUBSETS * (2. * TOTALHALFWIDTH - 8. * FLAGEWIDTH);
58  const double CROSSTOPWIDTH = CROSSBOTTOMWIDTH; // mCrossBottomWidth + mFlangeWidth - mRibWidth;
59  // for future release pending
60  // overlap correction - new TGeoVolume creation
61 
62  const double CROSSBEAMARCLENGTH = (112.62597) / (NUMCROSS - 1) - .001; // To account for shape of TGeoBBox
63  const double CROSSBOTTOMSTARTRADIUS = BEGINRADIUS + CROSSBOTTOMRADTHICK;
64  const double CROSSTOPSTART =
65  BEGINRADIUS + 2. * CROSSBOTTOMRADTHICK + CROSSTOPRADTHICK + 0.015; // 0.015 is a
66  // bubblegum and duct tape
67  // fix for an overlap problem
68  // will be worked out in future releases
69  const double ENDRADIUS = RIBHEIGHT + 1.15;
70  const double ENDBEAMRADTHICK = CROSSBOTTOMRADTHICK + CROSSTOPRADTHICK;
71  const double ENDBEAMBEGINRADIUS = BEGINRADIUS + ENDBEAMRADTHICK;
72 
73  /************ End efinition of constants **************************/
74 
76  TGeoVolume* top = gGeoManager->GetVolume("cave");
77  TGeoMedium* steel = gGeoManager->GetMedium("EMC_S steel$");
78  TGeoMedium* air = gGeoManager->GetMedium("EMC_Air$");
79 
81  TGeoVolume* calFrameMO = gGeoManager->MakeTubs("CalFrame", air, BEGINRADIUS - 2.1, ENDRADIUS, TOTALHALFWIDTH * 3,
82  BEGINPHI - 3, ENDPHI + 3); // Mother Volume
83 
84  calFrameMO->SetVisibility(kFALSE);
85 
86  // Half Frame Mother Volume
87  TGeoVolume* calHalfFrameMO = gGeoManager->MakeTubs("HalfFrame", air, BEGINRADIUS - 2, ENDRADIUS, TOTALHALFWIDTH,
88  BEGINPHI - 2.9, ENDPHI + 2.9);
89 
90  calHalfFrameMO->SetVisibility(kFALSE);
91 
92  TGeoVolume* endBeams =
93  gGeoManager->MakeBox("End Beams", steel, ENDBEAMRADTHICK, CROSSTOPHEIGHT, TOTALHALFWIDTH); // End Beams
94 
95  TGeoVolume* skin = gGeoManager->MakeTubs("skin", steel, RIBHEIGHT + 0.15, ENDRADIUS, TOTALHALFWIDTH, BEGINPHI,
96  ENDPHI); // back frame
97 
98  TGeoVolume* flangeVolume = gGeoManager->MakeTubs("supportBottom", steel, BEGINRADIUS, FLAGEHEIGHT, FLAGEWIDTH,
99  BEGINPHI, ENDPHI); // FlangeVolume Beams
100 
101  TGeoVolume* ribVolume =
102  gGeoManager->MakeTubs("RibVolume", steel, FLAGEHEIGHT, RIBHEIGHT, RIBWIDTH, BEGINPHI, ENDPHI);
103 
104  TGeoVolume* subSetCross = gGeoManager->MakeTubs(
105  "subSetCross", air, BEGINRADIUS - 1, BEGINRADIUS + 2 * CROSSBOTTOMRADTHICK + 2 * CROSSTOPRADTHICK + 0.15,
106  CROSSBOTTOMWIDTH, BEGINPHI, ENDPHI); // Cross Beam Containers
107  subSetCross->SetVisibility(kFALSE);
108  /* // Obsolete for now
109  TGeoVolume *subSetCrossTop =
110  gGeoManager->MakeTubs("SubSetCrossTop", air, mBeginRadius+2*mCrossBottomRadThick-1,
111  mBeginRadius+2*mCrossBottomRadThick+ 2*mCrossTopRadThick+1, mCrossTopWidth, mBeginPhi, mEndPhi); // Cross
112  subSetCrossTop->SetVisibility(kFALSE);
113  */
114  TGeoVolume* crossBottomBeams = gGeoManager->MakeBox("crossBottom", steel, CROSSBOTTOMRADTHICK, CROSSBOTTOMHEIGHT,
115  CROSSBOTTOMWIDTH); // Cross Beams
116 
117  TGeoVolume* crossTopBeams =
118  gGeoManager->MakeBox("crossTop", steel, CROSSTOPRADTHICK, CROSSTOPHEIGHT, CROSSTOPWIDTH); // Cross Beams
119 
120  TGeoTranslation* trTEST = new TGeoTranslation();
121  TGeoRotation* rotTEST = new TGeoRotation();
122 
123  Double_t conv = TMath::Pi() / 180.;
124  Double_t radAngle = 0;
125  Double_t endBeamParam = .4;
126  // cout<<"\nmCrossBottomStartRadius: "<<mCrossBottomStartRadius<<"\n";
127 
128  for (Int_t i = 0; i < NUMCROSS; i++) {
129  Double_t loopPhi = BEGINPHI + 1.8;
130 
131  // Cross Bottom Beams
132 
133  radAngle = (loopPhi + i * CROSSBEAMARCLENGTH) * conv;
134 
135  rotTEST->SetAngles(BEGINPHI + i * CROSSBEAMARCLENGTH, 0,
136  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
137  trTEST->SetTranslation(cos(radAngle) * CROSSBOTTOMSTARTRADIUS, sin(radAngle) * CROSSBOTTOMSTARTRADIUS, 0);
138 
139  TGeoCombiTrans* combo = new TGeoCombiTrans(*trTEST, *rotTEST); // TGeoTranslation &tr, const TGeoRotation &rot);
140  combo->RegisterYourself();
141  crossBottomBeams->SetVisibility(1);
142  subSetCross->AddNode(crossBottomBeams, i + 1, combo);
143  if (i != 0 && i != NUMCROSS - 1) {
144  // Cross Bottom Beams
145  rotTEST->SetAngles(BEGINPHI + i * CROSSBEAMARCLENGTH, 0,
146  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
147  trTEST->SetTranslation(cos(radAngle) * CROSSTOPSTART, sin(radAngle) * CROSSTOPSTART, 0);
148  crossTopBeams->SetVisibility(1);
149  subSetCross->AddNode(crossTopBeams, i + 1, new TGeoCombiTrans(*trTEST, *rotTEST));
150  }
151 
152  else if (i == 0) {
153  rotTEST->SetAngles(BEGINPHI + i * CROSSBEAMARCLENGTH, 0,
154  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
155  trTEST->SetTranslation(cos((77 - endBeamParam) * conv) * (ENDBEAMBEGINRADIUS),
156  sin((77 - endBeamParam) * conv) * (ENDBEAMBEGINRADIUS), 0);
157  endBeams->SetVisibility(1);
158  calHalfFrameMO->AddNode(endBeams, 1, new TGeoCombiTrans(*trTEST, *rotTEST));
159  } else {
160  rotTEST->SetAngles(193.03, 0, 0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
161  trTEST->SetTranslation(cos((193.03 + endBeamParam) * conv) * (ENDBEAMBEGINRADIUS) /*more duct tape*/,
162  sin((193.03 + endBeamParam) * conv) * (ENDBEAMBEGINRADIUS), 0);
163  endBeams->SetVisibility(1);
164  calHalfFrameMO->AddNode(endBeams, 2, new TGeoCombiTrans(*trTEST, *rotTEST));
165  }
166  }
167 
168  // Beam Containers
169 
170  // Translations
171 
172  TGeoTranslation* origin1 = new TGeoTranslation(0, 0, 0); // Equivalent to gGeoIdentity
173  TGeoTranslation* origin2 = new TGeoTranslation(0, 0, 2 * (CROSSBOTTOMWIDTH + FLAGEWIDTH));
174  TGeoTranslation* origin3 = new TGeoTranslation(0, 0, -2 * (CROSSBOTTOMWIDTH + FLAGEWIDTH));
175 
176  // FlangeVolume translations
177  TGeoTranslation* str1 = new TGeoTranslation(0, 0, -3 * (CROSSBOTTOMWIDTH + FLAGEWIDTH));
178  TGeoTranslation* str2 = new TGeoTranslation(0, 0, -(CROSSBOTTOMWIDTH + FLAGEWIDTH));
179  TGeoTranslation* str3 = new TGeoTranslation(0, 0, (CROSSBOTTOMWIDTH + FLAGEWIDTH));
180  TGeoTranslation* str4 = new TGeoTranslation(0, 0, 3 * (CROSSBOTTOMWIDTH + FLAGEWIDTH));
181 
182  // Half Frame Translations
183  TGeoTranslation* halfTrans1 = new TGeoTranslation(0, 0, HALFFRAMETRANS);
184  TGeoTranslation* halfTrans2 = new TGeoTranslation(0, 0, -HALFFRAMETRANS);
185 
186  // Beams Volume
187  calHalfFrameMO->AddNode(flangeVolume, 1, str1);
188  calHalfFrameMO->AddNode(flangeVolume, 2, str2);
189  calHalfFrameMO->AddNode(flangeVolume, 3, str3);
190  calHalfFrameMO->AddNode(flangeVolume, 4, str4);
191 
192  calHalfFrameMO->AddNode(ribVolume, 1, str1);
193  calHalfFrameMO->AddNode(ribVolume, 2, str2);
194  calHalfFrameMO->AddNode(ribVolume, 3, str3);
195  calHalfFrameMO->AddNode(ribVolume, 4, str4);
196 
197  // Cross Beams
198  calHalfFrameMO->AddNode(subSetCross, 1, origin1);
199  calHalfFrameMO->AddNode(subSetCross, 2, origin2);
200  calHalfFrameMO->AddNode(subSetCross, 3, origin3);
201  /* // Obsolete for now
202  calHalfFrameMO->AddNode(subSetCrossTop, 1, origin1);
203  calHalfFrameMO->AddNode(subSetCrossTop, 2, origin2);
204  calHalfFrameMO->AddNode(subSetCrossTop, 3, origin3);
205  */
206 
207  calHalfFrameMO->AddNode(skin, 1, gGeoIdentity);
208 
209  calFrameMO->AddNode(calHalfFrameMO, 1, halfTrans1);
210  calFrameMO->AddNode(calHalfFrameMO, 2, halfTrans2);
211 
212  top->AddNode(calFrameMO, 1, gGeoIdentity);
213  LOG(DEBUG) << "**********************************\nmEndRadius:\t" << ENDRADIUS << std::endl;
214 }
