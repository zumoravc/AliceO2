56 void ioutils::convertCompactClusters(gsl::span<const itsmft::CompClusterExt> clusters,
57  gsl::span<const unsigned char>::iterator& pattIt,
58  std::vector<o2::BaseCluster<float>>& output,
59  const itsmft::TopologyDictionary& dict)
60 {
61  GeometryTGeo* geom = GeometryTGeo::Instance();
62  for (auto& c : clusters) {
63  auto pattID = c.getPatternID();
64  Point3D<float> locXYZ;
65  float sigmaY2 = ioutils::DefClusError2Row, sigmaZ2 = ioutils::DefClusError2Col, sigmaYZ = 0; //Dummy COG errors (about half pixel size)
66  if (pattID != itsmft::CompCluster::InvalidPatternID) {
67  sigmaY2 = dict.getErr2X(pattID);
68  sigmaZ2 = dict.getErr2Z(pattID);
69  if (!dict.isGroup(pattID)) {
70  locXYZ = dict.getClusterCoordinates(c);
71  } else {
72  o2::itsmft::ClusterPattern patt(pattIt);
73  locXYZ = dict.getClusterCoordinates(c, patt);
74  }
75  } else {
76  o2::itsmft::ClusterPattern patt(pattIt);
77  locXYZ = dict.getClusterCoordinates(c, patt);
78  }
79  auto& cl3d = output.emplace_back(c.getSensorID(), geom->getMatrixT2L(c.getSensorID()) ^ locXYZ); // local --> tracking
80  cl3d.setErrors(sigmaY2, sigmaZ2, sigmaYZ);
81  }

96 std::vector<ROframe> ioutils::loadEventData(const std::string& fileName)
97 {
98  std::vector<ROframe> events{};
99  std::ifstream inputStream{};
100  std::string line{}, unusedVariable{};
101  int layerId{}, monteCarlo{};
102  int clusterId{EventLabelsSeparator};
103  float xCoordinate{}, yCoordinate{}, zCoordinate{}, alphaAngle{};
104  float varZ{-1.f}, varY{-1.f};
105 
106  inputStream.open(fileName);
107 
109  dataformats::MCTruthContainer<MCCompLabel>* mcLabels = nullptr;
110  while (std::getline(inputStream, line)) {
111 
112  std::istringstream inputStringStream(line);
113  if (inputStringStream >> layerId >> xCoordinate >> yCoordinate >> zCoordinate) {
114 
115  if (layerId == PrimaryVertexLayerId) {
116 
117  if (clusterId != 0) {
118  events.emplace_back(events.size());
119  }
120 
121  events.back().addPrimaryVertex(xCoordinate, yCoordinate, zCoordinate);
122  clusterId = 0;
123 
124  } else {
125 
126  if (inputStringStream >> varY >> varZ >> unusedVariable >> alphaAngle >> monteCarlo) {
127  events.back().addClusterToLayer(layerId, xCoordinate, yCoordinate, zCoordinate,
128  events.back().getClustersOnLayer(layerId).size());
129  const float sinAlpha = std::sin(alphaAngle);
130  const float cosAlpha = std::cos(alphaAngle);
131  const float xTF = xCoordinate * cosAlpha - yCoordinate * sinAlpha;
132  const float yTF = xCoordinate * sinAlpha + yCoordinate * cosAlpha;
133  events.back().addTrackingFrameInfoToLayer(layerId, xCoordinate, yCoordinate, zCoordinate, xTF, alphaAngle,
134  std::array<float, 2>{yTF, zCoordinate}, std::array<float, 3>{varY, 0.f, varZ});
135  events.back().addClusterLabelToLayer(layerId, MCCompLabel(monteCarlo));
136 
137  ++clusterId;
138  }
139  }
140  }
141  }
142 
143  return events;
144 }

146 void ioutils::loadEventData(ROframe& event, gsl::span<const itsmft::Cluster> clusters,
147  const dataformats::MCTruthContainer<MCCompLabel>* clsLabels)
148 {
149  if (clusters.empty()) {
150  std::cerr << "Missing clusters." << std::endl;
151  return;
152  }
153  event.clear();
154  GeometryTGeo* geom = GeometryTGeo::Instance();
155  geom->fillMatrixCache(utils::bit2Mask(TransformType::T2GRot));
156  int clusterId{0};
157 
158  for (auto& c : clusters) {
159  int layer = geom->getLayer(c.getSensorID());
160 
162  auto xyz = c.getXYZGloRot(*geom);
163  event.addTrackingFrameInfoToLayer(layer, xyz.x(), xyz.y(), xyz.z(), c.getX(), geom->getSensorRefAlpha(c.getSensorID()),
164  std::array<float, 2>{c.getY(), c.getZ()},
165  std::array<float, 3>{c.getSigmaY2(), c.getSigmaYZ(), c.getSigmaZ2()});
166 
168  event.addClusterToLayer(layer, xyz.x(), xyz.y(), xyz.z(), event.getClustersOnLayer(layer).size());
169  if (clsLabels) {
170  event.addClusterLabelToLayer(layer, *(clsLabels->getLabels(clusterId).begin()));
171  }
172  event.addClusterExternalIndexToLayer(layer, clusterId);
173  clusterId++;
174  }

309 void ioutils::generateSimpleData(ROframe& event, const int phiDivs, const int zDivs = 1)
310 {
311  const float angleOffset = constants::math::TwoPi / static_cast<float>(phiDivs);
312  // Maximum z allowed on innermost layer should be: ~9,75
313  const float zOffsetFirstLayer = (zDivs == 1) ? 0 : 1.5 * (LayersZCoordinate()[6] * LayersRCoordinate()[0]) / (LayersRCoordinate()[6] * (static_cast<float>(zDivs) - 1));
314  std::vector<float> x, y;
315  std::array<std::vector<float>, 7> z;
316  for (size_t j{0}; j < zDivs; ++j) {
317  for (size_t i{0}; i < phiDivs; ++i) {
318  x.emplace_back(cos(i * angleOffset + 0.001)); // put an epsilon to move from periods (e.g. 20 clusters vs 20 cells)
319  y.emplace_back(sin(i * angleOffset + 0.001));
320  const float zFirstLayer{-static_cast<float>((zDivs - 1.) / 2.) * zOffsetFirstLayer + zOffsetFirstLayer * static_cast<float>(j)};
321  z[0].emplace_back(zFirstLayer);
322  for (size_t iLayer{1}; iLayer < constants::its::LayersNumber; ++iLayer) {
323  z[iLayer].emplace_back(zFirstLayer * LayersRCoordinate()[iLayer] / LayersRCoordinate()[0]);
324  }
325  }
326  }
327 
328  for (int iLayer{0}; iLayer < constants::its::LayersNumber; ++iLayer) {
329  for (int i = 0; i < phiDivs * zDivs; i++) {
330  o2::MCCompLabel label{i, 0, 0, false};
331  event.addClusterLabelToLayer(iLayer, label); //last argument : label, goes into mClustersLabel
332  event.addClusterToLayer(iLayer, LayersRCoordinate()[iLayer] * x[i], LayersRCoordinate()[iLayer] * y[i], z[iLayer][i], i); //uses 1st constructor for clusters
333  }
334  }
335 }

337 std::vector<std::unordered_map<int, Label>> ioutils::loadLabels(const int eventsNum, const std::string& fileName)
338 {
339  std::vector<std::unordered_map<int, Label>> labelsMap{};
340  std::unordered_map<int, Label> currentEventLabelsMap{};
341  std::ifstream inputStream{};
342  std::string line{};
343  int monteCarloId{}, pdgCode{}, numberOfClusters{};
344  float transverseMomentum{}, phiCoordinate{}, pseudorapidity{};
345 
346  labelsMap.reserve(eventsNum);
347 
348  inputStream.open(fileName);
349  std::getline(inputStream, line);
350 
351  while (std::getline(inputStream, line)) {
352 
353  std::istringstream inputStringStream(line);
354 
355  if (inputStringStream >> monteCarloId) {
356 
357  if (monteCarloId == EventLabelsSeparator) {
358 
359  labelsMap.emplace_back(currentEventLabelsMap);
360  currentEventLabelsMap.clear();
361 
362  } else {
363 
364  if (inputStringStream >> transverseMomentum >> phiCoordinate >> pseudorapidity >> pdgCode >> numberOfClusters) {
365 
366  if (std::abs(pdgCode) == constants::pdgcodes::PionCode && numberOfClusters == 7) {
367 
368  currentEventLabelsMap.emplace(std::piecewise_construct, std::forward_as_tuple(monteCarloId),
369  std::forward_as_tuple(monteCarloId, transverseMomentum, phiCoordinate,
370  pseudorapidity, pdgCode, numberOfClusters));
371  }
372  }
373  }
374  }
375  }
376 
377  labelsMap.emplace_back(currentEventLabelsMap);
378 
379  return labelsMap;
380 }

382 void ioutils::writeRoadsReport(std::ofstream& correctRoadsOutputStream, std::ofstream& duplicateRoadsOutputStream,
383  std::ofstream& fakeRoadsOutputStream, const std::vector<std::vector<Road>>& roads,
384  const std::unordered_map<int, Label>& labelsMap)
385 {
386  const int numVertices{static_cast<int>(roads.size())};
387  std::unordered_set<int> foundMonteCarloIds{};
388 
389  correctRoadsOutputStream << EventLabelsSeparator << std::endl;
390  fakeRoadsOutputStream << EventLabelsSeparator << std::endl;
391 
392  for (int iVertex{0}; iVertex < numVertices; ++iVertex) {
393 
394  const std::vector<Road>& currentVertexRoads{roads[iVertex]};
395  const int numRoads{static_cast<int>(currentVertexRoads.size())};
396 
397  for (int iRoad{0}; iRoad < numRoads; ++iRoad) {
398 
399  const Road& currentRoad{currentVertexRoads[iRoad]};
400  const int currentRoadLabel{currentRoad.getLabel()};
401 
402  if (!labelsMap.count(currentRoadLabel)) {
403 
404  continue;
405  }
406 
407  const Label& currentLabel{labelsMap.at(currentRoadLabel)};
408 
409  if (currentRoad.isFakeRoad()) {
410 
411  fakeRoadsOutputStream << currentLabel << std::endl;
412 
413  } else {
414 
415  if (foundMonteCarloIds.count(currentLabel.monteCarloId)) {
416 
417  duplicateRoadsOutputStream << currentLabel << std::endl;
418 
419  } else {
420 
421  correctRoadsOutputStream << currentLabel << std::endl;
422  foundMonteCarloIds.emplace(currentLabel.monteCarloId);
423  }
424  }
425  }
426  }
