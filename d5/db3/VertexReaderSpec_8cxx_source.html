33 void VertexReader::run(ProcessingContext& pc)
34 {
35  auto ent = mTree->GetReadEntry() + 1;
36  assert(ent < mTree->GetEntries()); // this should not happen
37  mTree->GetEntry(ent);
38  LOG(INFO) << "Pushing " << mVerticesPtr->size() << " vertices in " << mVerticesROFRecPtr->size()
39  << " ROFs at entry " << ent;
40  pc.outputs().snapshot(Output{"ITS", "VERTICES", 0, Lifetime::Timeframe}, mVertices);
41  pc.outputs().snapshot(Output{"ITS", "VERTICESROF", 0, Lifetime::Timeframe}, mVerticesROFRec);
42 
43  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
44  pc.services().get<ControlService>().endOfStream();
45  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
46  }
47 }

49 void VertexReader::connectTree(const std::string& filename)
50 {
51  mTree.reset(nullptr); // in case it was already loaded
52  mFile.reset(TFile::Open(filename.c_str()));
53  assert(mFile && !mFile->IsZombie());
54  mTree.reset((TTree*)mFile->Get(mVertexTreeName.c_str()));
55  assert(mTree);
56  assert(mTree->GetBranch(mVertexBranchName.c_str()));
57  assert(mTree->GetBranch(mVertexROFBranchName.c_str()));
58  mTree->SetBranchAddress(mVertexBranchName.c_str(), &mVerticesPtr);
59  mTree->SetBranchAddress(mVertexROFBranchName.c_str(), &mVerticesROFRecPtr);
60  LOG(INFO) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
61 }
