73 // Split a given string on a delim character, return vector of tokens
74 // If trim is true, then also remove leading/trailing whitespace of each token.
75 std::vector<std::string> splitString(const std::string& src, char delim, bool trim = false)
76 {
77  std::stringstream ss(src);
78  std::string token;
79  std::vector<std::string> tokens;
80 
81  while (std::getline(ss, token, delim)) {
82  token = (trim ? trimSpace(token) : token);
83  if (!token.empty()) {
84  tokens.push_back(std::move(token));
85  }
86  }
87 
88  return tokens;
89 }

91 // a simple tool to take an abitrary object in a ROOT file and upload it to CCDB including some
92 // meta information
93 int main(int argc, char* argv[])
94 {
95  bpo::options_description options("Allowed options");
96  bpo::variables_map vm;
97  if (!initOptionsAndParse(options, argc, argv, vm)) {
98  return 1;
99  }
100 
101  o2::ccdb::CcdbApi api;
102  auto host = vm["host"].as<std::string>();
103  api.init(host);
104 
105  std::map<std::string, std::string> filter;
106  long starttimestamp = vm["starttimestamp"].as<long>();
107  if (starttimestamp == -1) {
108  starttimestamp = o2::ccdb::getCurrentTimestamp();
109  }
110  long endtimestamp = vm["endtimestamp"].as<long>();
111  if (endtimestamp == -1) {
112  constexpr long SECONDSPERYEAR = 365 * 24 * 60 * 60;
113  endtimestamp = o2::ccdb::getFutureTimestamp(SECONDSPERYEAR);
114  }
115 
116  auto filename = vm["file"].as<std::string>();
117  auto path = vm["path"].as<std::string>();
118  auto keyname = vm["key"].as<std::string>();
119 
120  // Take a vector of strings with elements of form a=b, and
121  // return a vector of pairs with each pair of form <a, b>
122  auto toKeyValPairs = [](std::vector<std::string> const& tokens) {
123  std::vector<std::pair<std::string, std::string>> pairs;
124 
125  for (auto& token : tokens) {
126  auto keyval = splitString(token, '=');
127  if (keyval.size() != 2) {
128  // LOG(FATAL) << "Illegal command-line key/value string: " << token;
129  continue;
130  }
131 
132  std::pair<std::string, std::string> pair = std::make_pair(keyval[0], trimSpace(keyval[1]));
133  pairs.push_back(pair);
134  }
135 
136  return pairs;
137  };
138  auto metastring = vm["meta"].as<std::string>();
139  auto keyvalues = toKeyValPairs(splitString(metastring, ';', true));
140 
141  // fill meta map
142  std::map<std::string, std::string> meta;
143  for (auto& p : keyvalues) {
144  meta[p.first] = p.second;
145  }
146 
147  TFile f(filename.c_str());
148  auto key = f.GetKey(keyname.c_str());
149  if (key) {
150  // get type of key
151  auto classname = key->GetClassName();
152  auto object = f.Get<void>(keyname.c_str());
153  // convert classname to typeinfo
154  auto tcl = TClass::GetClass(classname);
155  // typeinfo
156  auto ti = tcl->GetTypeInfo();
157 
158  std::cout << " Uploading an object of type " << key->GetClassName()
159  << " to path " << path << " with timestamp validy from " << starttimestamp
160  << " to " << endtimestamp << "\n";
161 
162  api.storeAsTFile_impl(object, *ti, path, meta, starttimestamp, endtimestamp);
163  } else {
164  std::cerr << "Key " << keyname << " does not exist\n";
165  }
166 
167  return 0;
168 }
