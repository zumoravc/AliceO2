31 {
32  framework::WorkflowSpec specs;
33 
34  bool passFullITSClusters = false; // temporarily pass full clusters,
35  bool passCompITSClusters = true; // eventually only compact of recpoints will be passed
36  bool passITSClusPatterns = true;
37 
38  std::vector<int> tpcClusSectors = o2::RangeTokenizer::tokenize<int>("0-35");
39  std::vector<int> tpcClusLanes = tpcClusSectors;
40 
41  if (!disableRootInp) {
42  specs.emplace_back(o2::its::getITSTrackReaderSpec(useMC));
43  specs.emplace_back(o2::itsmft::getITSClusterReaderSpec(useMC, passFullITSClusters, passCompITSClusters, passITSClusPatterns));
44  specs.emplace_back(o2::tpc::getTPCTrackReaderSpec(useMC));
45  specs.emplace_back(o2::tpc::getPublisherSpec(o2::tpc::PublisherConf{
46  "tpc-native-cluster-reader",
47  "tpcrec",
48  {"clusterbranch", "TPCClusterNative", "Branch with TPC native clusters"},
49  {"clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch"},
50  OutputSpec{"TPC", "CLUSTERNATIVE"},
51  OutputSpec{"TPC", "CLNATIVEMCLBL"},
52  tpcClusSectors,
53  tpcClusLanes},
54  useMC));
55 
56  if (o2::globaltracking::MatchITSTPCParams::Instance().runAfterBurner) {
57  specs.emplace_back(o2::ft0::getFT0RecPointReaderSpec(useMC));
58  }
59  }
60  specs.emplace_back(o2::globaltracking::getTPCITSMatchingSpec(useMC, tpcClusLanes));
61 
62  if (!disableRootOut) {
63  specs.emplace_back(o2::globaltracking::getTrackWriterTPCITSSpec(useMC));
64  }

66 }
