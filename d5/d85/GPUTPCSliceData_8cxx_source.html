123 GPUdi() void GPUTPCSliceData::CreateGrid(GPUconstantref() const MEM_CONSTANT(GPUConstantMem) * mem, GPUTPCRow* GPUrestrict() row, float yMin, float yMax, float zMin, float zMax)
124 {
125  float dz = zMax - zMin;
126  float tfFactor = 1.;
127  if (dz > 270.) {
128  tfFactor = dz / 250.;
129  dz = 250.;
130  }
131  const float norm = CAMath::FastInvSqrt(row->mNHits / tfFactor);
132  float sy = CAMath::Min(CAMath::Max((yMax - yMin) * norm, GPUCA_MIN_BIN_SIZE), GPUCA_MAX_BIN_SIZE);
133  float sz = CAMath::Min(CAMath::Max(dz * norm, GPUCA_MIN_BIN_SIZE), GPUCA_MAX_BIN_SIZE);
134  int maxy, maxz;
135  GetMaxNBins(mem, row, maxy, maxz);
136  int ny = CAMath::Max(1, CAMath::Min<int>(maxy, (yMax - yMin) / sy + 1));
137  int nz = CAMath::Max(1, CAMath::Min<int>(maxz, (zMax - zMin) / sz + 1));
138  row->mGrid.Create(yMin, yMax, zMin, zMax, ny, nz);
139 }

197  for (int i = 0; i < mNumberOfHits; i++) {

199  tmp.x = mClusterData[i].y;
200  tmp.y = mClusterData[i].z;
201  int tmpRow = mClusterData[i].row;
202  int newIndex = RowOffsetsA[tmpRow] + (RowsFilled[tmpRow])++;
203  YZData[newIndex] = tmp;
204  tmpHitIndex[newIndex] = i;
205  }
206  } // Other cases below in loop over rows
207 #else
208  float2* YZData = (float2*)mLinkUpData; // TODO: we can do this as well on the CPU, just must make sure that CPU has the scratch memory
209  calink* binMemory = (calink*)mHitWeights;

211  static_assert(sizeof(*binMemory) <= sizeof(*mHitWeights), "Cannot reuse memory");

214  for (int rowIndex = iBlock; rowIndex < GPUCA_ROW_COUNT; rowIndex += nBlocks) {

216  float yMax = -1.e6f;
217  float zMin = 1.e6f;

221  const unsigned int RowOffset = EarlyTransformWithoutClusterNative ? RowOffsets[rowIndex] : (mem->ioPtrs.clustersNative->clusterOffset[iSlice][rowIndex] - mem->ioPtrs.clustersNative->clusterOffset[iSlice][0]);

232  if (iThread == 0) {

235  tmpMinMax[2] = -zMin;
236  tmpMinMax[3] = zMax;

238  }
239  GPUbarrier();
240  GPUAtomic(calink)* c = (GPUAtomic(calink)*)mFirstHitInBin + row.mFirstHitInBinOffset;
241  if (NumberOfClusters == 0) {
242  if (iThread == 0) {
243  row.mGrid.CreateEmpty();
244  row.mNHits = 0;
245  row.mHitNumberOffset = 0;
246  row.mHy0 = 0.f;
247  row.mHz0 = 0.f;
248  row.mHstepY = 1.f;
249  row.mHstepZ = 1.f;
250  row.mHstepYi = 1.f;
251  row.mHstepZi = 1.f;
252  for (int i = 0; i < 4; i++) {
253  c[i] = 0;
254  }
255  }
256  continue;
257  }
258 
259  if (EarlyTransformWithoutClusterNative) {
260  for (unsigned int i = iThread; i < NumberOfClusters; i += nThreads) {
261  UpdateMinMaxYZ(yMin, yMax, zMin, zMax, YZData[RowOffset + i].x, YZData[RowOffset + i].y);
262  }
263  } else {
264  if (mem->param.earlyTpcTransform) { // Early transform case with ClusterNative present
265  for (unsigned int i = iThread; i < NumberOfClusters; i += nThreads) {
266  float2 tmp;
267  tmp.x = mClusterData[RowOffset + i].y;
268  tmp.y = mClusterData[RowOffset + i].z;
269  UpdateMinMaxYZ(yMin, yMax, zMin, zMax, tmp.x, tmp.y);
270  YZData[RowOffset + i] = tmp;
271  }
272  } else {
273  for (unsigned int i = iThread; i < NumberOfClusters; i += nThreads) {
274  float x, y, z;
275  GPUTPCConvertImpl::convert(*mem, iSlice, rowIndex, mem->ioPtrs.clustersNative->clusters[iSlice][rowIndex][i].getPad(), mem->ioPtrs.clustersNative->clusters[iSlice][rowIndex][i].getTime(), x, y, z);
276  UpdateMinMaxYZ(yMin, yMax, zMin, zMax, y, z);
277  YZData[RowOffset + i] = CAMath::MakeFloat2(y, z);
278  }
279  }
280  }
281 
282  if (iThread == 0) {
283  row.mNHits = NumberOfClusters;
284  row.mHitNumberOffset = CAMath::nextMultipleOf<GPUCA_ROWALIGNMENT / sizeof(calink)>(RowOffset + rowIndex * GPUCA_ROWALIGNMENT / sizeof(calink));
285  }
286 
287  /* CAMath::AtomicMaxShared(&tmpMinMax[0], -yMin); // Atomic max not supported for float
288  CAMath::AtomicMaxShared(&tmpMinMax[1], yMax);
289  CAMath::AtomicMaxShared(&tmpMinMax[2], -zMin);

314  const int numberOfBins = grid.N();

317  return 1;
318  }
319  const unsigned int nn = numberOfBins + grid.Ny() + 3;
320  if (nn >= GetGridSize(NumberOfClusters, 1)) {

326

349  }

352  constexpr float maxVal = (((long long int)1 << (sizeof(cahit) < 3 ? sizeof(cahit) * 8 : 24)) - 1); // Stay within float precision in any case!

355  const float z0 = row.mGrid.ZMin();

362  row.mHz0 = z0;

387  mHitData[globalBinsortedIndex].y = (cahit)yy;
388  }
389 
390  if (iThread == 0) {
391  const float maxAbsZ = CAMath::Max(CAMath::Abs(tmpMinMax[2]), CAMath::Abs(tmpMinMax[3]));

395  }
396  }
397  }
