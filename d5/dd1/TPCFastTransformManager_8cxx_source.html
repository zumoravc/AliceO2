30 int TPCFastTransformManager::create(TPCFastTransform& fastTransform,
31  AliTPCTransform* transform,
32  Long_t TimeStamp)
33 {
35 
36  AliTPCcalibDB* pCalib = AliTPCcalibDB::Instance();
37  if (!pCalib) {
38  return storeError(
39  -1, "TPCFastTransformManager::Init: No TPC calibration instance found");
40  }
41 
42  AliTPCParam* tpcParam = pCalib->GetParameters();
43  if (!tpcParam) {
44  return storeError(
45  -2, "TPCFastTransformManager::Init: No TPCParam object found");
46  }
47 
48  if (!transform) {
49  transform = pCalib->GetTransform();
50  }
51  if (!transform) {
52  return storeError(
53  -3, "TPCFastTransformManager::Init: No TPC transformation found");
54  }
55 
56  mOrigTransform = transform;
57 
58  tpcParam->Update();
59  tpcParam->ReadGeoMatrices();
60 
61  const AliTPCRecoParam* rec = transform->GetCurrentRecoParam();
62  if (!rec) {
63  return storeError(-5,
64  "TPCFastTransformManager::Init: No TPC Reco Param "
65  "set in transformation");
66  }
67 
68  bool useCorrectionMap = rec->GetUseCorrectionMap();
69 
70  if (useCorrectionMap) {
71  transform->SetCorrectionMapMode(kTRUE); // If the simulation set this to
72  // false to simulate corrections, we
73  // need to reverse it for the
74  // transformation
75  }
76  // find last calibrated time bin
77 
78  fLastTimeBin = rec->GetLastBin();
79 
80  const int nRows = tpcParam->GetNRowLow() + tpcParam->GetNRowUp();
81 
82  TPCFastTransformGeo geo;
83 
84  { // construct the geometry
85  geo.startConstruction(nRows);
86 
87  float tpcZlengthSideA = tpcParam->GetZLength(0);
88  float tpcZlengthSideC =
89  tpcParam->GetZLength(TPCFastTransformGeo::getNumberOfSlices() / 2);
90 
91  geo.setTPCzLength(tpcZlengthSideA, tpcZlengthSideC);
92  geo.setTPCalignmentZ(-mOrigTransform->GetDeltaZCorrTime());
93 
94  for (int row = 0; row < geo.getNumberOfRows(); row++) {
95  int slice = 0, sector = 0, secrow = 0;
96  AliHLTTPCGeometry::Slice2Sector(slice, row, sector, secrow);
97  Int_t nPads = tpcParam->GetNPads(sector, secrow);
98  float xRow = tpcParam->GetPadRowRadii(sector, secrow);
99  float padWidth = tpcParam->GetInnerPadPitchWidth();
100  if (row >= tpcParam->GetNRowLow()) {
101  padWidth = tpcParam->GetOuterPadPitchWidth();
102  }
103  geo.setTPCrow(row, xRow, nPads, padWidth);
104  }
105  geo.finishConstruction();
106  }
107 
108  TPCFastSpaceChargeCorrection correction;
109 
110  { // create the correction map
111 
112  const int nDistortionScenarios = 1;
113 
114  correction.startConstruction(geo, nDistortionScenarios);
115 
116  TPCFastSpaceChargeCorrection::SplineType spline;
117  spline.recreate(8, 20);
118 
119  int scenario = 0;
120  correction.setSplineScenario(scenario, spline);
121 
122  for (int row = 0; row < geo.getNumberOfRows(); row++) {
123  correction.setRowScenarioID(row, scenario);
124  }
125 
126  correction.finishConstruction();
127  } // .. create the correction map
128 
129  { // create the fast transform object
130 
131  fastTransform.startConstruction(correction);
132 
133  // tell the transformation to apply the space charge corrections
134  fastTransform.setApplyCorrectionOn();
135 
136  // set some initial calibration values, will be reinitialised later int
137  // updateCalibration()
138  const float t0 = 0.;
139  const float vDrift = 0.f;
140  const float vdCorrY = 0.;
141  const float ldCorr = 0.;
142  const float tofCorr = 0.;
143  const float primVtxZ = 0.;
144  const long int initTimeStamp = -1;
145  fastTransform.setCalibration(initTimeStamp, t0, vDrift, vdCorrY, ldCorr,
146  tofCorr, primVtxZ);
147 
148  fastTransform.finishConstruction();
149  }
150 
151  return updateCalibration(fastTransform, TimeStamp);
152 }

154 int TPCFastTransformManager::updateCalibration(TPCFastTransform& fastTransform,
155  Long_t TimeStamp)
156 {
157  // Update the calibration with the new time stamp
158 
159  Long_t lastTS = fastTransform.getTimeStamp();
160 
161  // deinitialize
162 
163  fastTransform.setTimeStamp(-1);
164 
165  if (TimeStamp < 0) {
166  return 0;
167  }
168 
169  // search for the calibration database
170 
171  if (!mOrigTransform) {
172  return storeError(-1,
173  "TPCFastTransformManager::SetCurrentTimeStamp: TPC "
174  "transformation has not been set properly");
175  }
176 
177  AliTPCcalibDB* pCalib = AliTPCcalibDB::Instance();
178  if (!pCalib) {
179  return storeError(-2,
180  "TPCFastTransformManager::SetCurrentTimeStamp: No "
181  "TPC calibration found");
182  }
183 
184  AliTPCParam* tpcParam = pCalib->GetParameters();
185  if (!tpcParam) {
186  return storeError(-3,
187  "TPCFastTransformManager::SetCurrentTimeStamp: No "
188  "TPCParam object found");
189  }
190 
191  AliTPCRecoParam* recoParam = mOrigTransform->GetCurrentRecoParamNonConst();
192  if (!recoParam) {
193  return storeError(-5,
194  "TPCFastTransformManager::Init: No TPC Reco Param "
195  "set in transformation");
196  }
197 
198  // calibration found, set the initialized status back
199 
200  fastTransform.setTimeStamp(lastTS);
201 
202  // less than 60 seconds from the previois time stamp, don't do anything
203 
204  if (lastTS >= 0 && TMath::Abs(lastTS - TimeStamp) < 60) {
205  return 0;
206  }
207 
208  // start the initialization
209 
210  bool useCorrectionMap = recoParam->GetUseCorrectionMap();
211 
212  if (useCorrectionMap) {
213  // If the simulation set this to false to simulate corrections, we need to
214  // reverse it for the transformation This is a design feature. Historically
215  // HLT code runs as a part of simulation, not reconstruction.
216  mOrigTransform->SetCorrectionMapMode(kTRUE);
217  }
218 
219  // set the current time stamp
220 
221  mOrigTransform->SetCurrentTimeStamp(static_cast<UInt_t>(TimeStamp));
222  fastTransform.setTimeStamp(TimeStamp);
223 
224  // find last calibrated time bin
225 
226  fLastTimeBin = recoParam->GetLastBin();
227 
228  double t0 = mOrigTransform->GetTBinOffset();
229  double driftCorrPT = mOrigTransform->GetDriftCorrPT();
230  double vdCorrectionTime = mOrigTransform->GetVDCorrectionTime();
231  double vdCorrectionTimeGY = mOrigTransform->GetVDCorrectionTimeGY();
232  double time0CorrTime = mOrigTransform->GetTime0CorrTime();
233 
234  // original formula:
235  // L = (t-t0)*ZWidth*driftCorrPT*vdCorrectionTime*( 1 +
236  // yLab*vdCorrectionTimeGY ) - time0CorrTime + 3.*tpcParam->GetZSigma(); Z =
237  // Z(L) - fDeltaZCorrTime chebyshev corrections for xyz Time-of-flight
238  // correction: ldrift += dist-to-vtx*tofCorr
239 
240  // fast transform formula:
241  // L = (t-t0)*(mVdrift + mVdriftCorrY*yLab ) + mLdriftCorr
242  // Z = Z(L) + tpcAlignmentZ
243  // spline corrections for xyz
244  // Time-of-flight correction: ldrift += dist-to-vtx*tofCorr
245 
246  double vDrift = tpcParam->GetZWidth() * driftCorrPT * vdCorrectionTime;
247  double vdCorrY = vDrift * vdCorrectionTimeGY;
248  double ldCorr = -time0CorrTime + 3 * tpcParam->GetZSigma();
249 
250  double tofCorr = (0.01 * tpcParam->GetDriftV()) / TMath::C();
251  double primVtxZ = mOrigTransform->GetPrimVertex()[2];
252 
253  bool useTOFcorrection = recoParam->GetUseTOFCorrection();
254 
255  if (!useTOFcorrection) {
256  tofCorr = 0;
257  }
258 
259  fastTransform.setCalibration(TimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr,
260  primVtxZ);
261 
262  // now calculate the correction map: dx,du,dv = ( origTransform() -> x,u,v) -
263  // fastTransformNominal:x,u,v
264 
265  const TPCFastTransformGeo& geo = fastTransform.getGeometry();
266 
267  TPCFastSpaceChargeCorrection& correction =
268  fastTransform.getCorrection();
269 
270  // switch TOF correction off for a while
271 
272  recoParam->SetUseTOFCorrection(kFALSE);
273 
274  for (int slice = 0; slice < geo.getNumberOfSlices(); slice++) {
275 
276  for (int row = 0; row < geo.getNumberOfRows(); row++) {
277 
278  const TPCFastTransformGeo::RowInfo& rowInfo = geo.getRowInfo(row);
279 
280  const TPCFastSpaceChargeCorrection::SplineType& spline = correction.getSpline(slice, row);
281  float* data = correction.getSplineData(slice, row);
282 
283  SplineHelper2D<float> helper;
284  helper.setSpline(spline, 4, 4);
285  auto F = [&](float su, float sv, float dxuv[3]) {
286  float x = rowInfo.x;
287  // x, u, v cordinates of the knot (local cartesian coord. of slice
288  // towards central electrode )
289  float u = 0, v = 0;
290  geo.convScaledUVtoUV(slice, row, su, sv, u, v);
291 
292  // row, pad, time coordinates of the knot
293  float vertexTime = 0.f;
294  float pad = 0.f, time = 0.f;
295  fastTransform.convUVtoPadTime(slice, row, u, v, pad, time, vertexTime);
296 
297  // nominal x,y,z coordinates of the knot (without corrections and
298  // time-of-flight correction)
299  float y = 0, z = 0;
300  geo.convUVtoLocal(slice, u, v, y, z);
301 
302  // original TPC transformation (row,pad,time) -> (x,y,z) without
303  // time-of-flight correction
304  float ox = 0, oy = 0, oz = 0;
305  {
306  int sector = 0, secrow = 0;
307  AliHLTTPCGeometry::Slice2Sector(slice, row, sector, secrow);
308  int is[] = {sector};
309  double xx[] = {static_cast<double>(secrow), pad, time};
310  mOrigTransform->Transform(xx, is, 0, 1);
311  ox = xx[0];
312  oy = xx[1];
313  oz = xx[2];
314  }
315  // convert to u,v
316  float ou = 0, ov = 0;
317  geo.convLocalToUV(slice, oy, oz, ou, ov);
318 
319  // corrections in x,u,v:
320  dxuv[0] = ox - x;
321  dxuv[1] = ou - u;
322  dxuv[2] = ov - v;
323  };
324 
325  helper.approximateFunction(data, 0., 1., 0., 1., F);
326  } // row
327  } // slice
328 
329  correction.initInverse();
330 
331  // set back the time-of-flight correction;
332 
333  recoParam->SetUseTOFCorrection(useTOFcorrection);
334 
335  return 0;
336 }
