41 TEveGeoShape* GeometryManager::getGeometryForDetector(string detectorName)
42 {
43  TEnv settings;
44  ConfigurationManager::getInstance().getConfig(settings);
45 
46  // read geometry path from config file
47  string geomPath = settings.GetValue(this->mR2Geometry ? "simple.geom.R2.path" : "simple.geom.R3.path", "");
48 
49  // load ROOT file with geometry
50  TFile* f = TFile::Open(Form("%s/simple_geom_%s.root", geomPath.c_str(), detectorName.c_str()));
51  if (!f) {
52  LOG(ERROR) << "GeometryManager::GetSimpleGeom -- no file with geometry found for: " << detectorName << "!";
53  return nullptr;
54  }
55  LOG(INFO) << "GeometryManager::GetSimpleGeom for: " << detectorName << " from " << Form("%s/simple_geom_%s.root", geomPath.c_str(), detectorName.c_str());
56 
57  TEveGeoShapeExtract* geomShapreExtract = static_cast<TEveGeoShapeExtract*>(f->Get(detectorName.c_str()));
58  TEveGeoShape* geomShape = TEveGeoShape::ImportShapeExtract(geomShapreExtract);
59  f->Close();
60 
61  geomShape->SetName(detectorName.c_str());
62  // tricks for different R-Phi geom of TPC:
63  if (detectorName == "RPH") { // use all other parameters of regular TPC geom
64  detectorName = "TPC";
65  }
66 
67  // prepare geometry to be drawn including all children
68  drawDeep(geomShape,
69  settings.GetValue(Form("%s.color", detectorName.c_str()), -1),
70  settings.GetValue(Form("%s.trans", detectorName.c_str()), -1),
71  settings.GetValue(Form("%s.line.color", detectorName.c_str()), -1));
72 
73  gEve->GetDefaultGLViewer()->UpdateScene();
74 
75  return geomShape;
76 }

78 void GeometryManager::drawDeep(TEveGeoShape* geomShape, Color_t color, Char_t transparency, Color_t lineColor)
79 {
80  if (geomShape->HasChildren()) {
81  geomShape->SetRnrSelf(false);
82 
83  if (strcmp(geomShape->GetElementName(), "TPC_Drift_1") == 0) { // hack for TPC drift chamber
84  geomShape->SetRnrSelf(kTRUE);
85  if (color >= 0)
86  geomShape->SetMainColor(color);
87  if (lineColor >= 0) {
88  geomShape->SetLineColor(lineColor);
89  geomShape->SetLineWidth(0.1);
90  geomShape->SetDrawFrame(true);
91  } else {
92  geomShape->SetDrawFrame(false);
93  }
94  if (transparency >= 0) {
95  geomShape->SetMainTransparency(transparency);
96  }
97  }
98 
99  for (TEveElement::List_i i = geomShape->BeginChildren(); i != geomShape->EndChildren(); ++i) {
100  drawDeep(static_cast<TEveGeoShape*>(*i), color, transparency, lineColor);
101  }
102  } else {
103  geomShape->SetRnrSelf(true);
104  if (color >= 0)
105  geomShape->SetMainColor(color);
106  if (lineColor >= 0) {
107  geomShape->SetLineColor(lineColor);
108  geomShape->SetLineWidth(0.1);
109  geomShape->SetDrawFrame(true);
110  } else {
111  geomShape->SetDrawFrame(false);
112  }
113  if (transparency >= 0) {
114  geomShape->SetMainTransparency(transparency);
115  }
116 
117  if (strcmp(geomShape->GetElementName(), "PHOS_5") == 0) { // hack for PHOS module which is not installed
118  geomShape->SetRnrSelf(false);
119  }
120  }
121 }
