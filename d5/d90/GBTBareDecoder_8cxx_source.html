34 void GBTBareDecoder::process(gsl::span<const uint8_t> bytes, uint16_t bc, uint32_t orbit, uint16_t pageCnt)
35 {
37  mOutputHandler.setIR(bc, orbit, pageCnt);
38 
39  uint8_t byte = 0;
40  size_t ilink = 0, linkMask = 0, byteOffset = 0;
41 
42  for (size_t idx = 0; idx < bytes.size(); idx += 16) {
43  for (int ireg = 0; ireg < 2; ++ireg) {
44  byteOffset = idx + 5 * ireg;
45  for (int ib = 0; ib < 4; ++ib) {
46  byte = bytes[byteOffset + ib];
47  ilink = ib + 4 * ireg;
48  linkMask = (1 << ilink);
49  if ((mMask & linkMask) && ((mIsFeeding & linkMask) || byte)) {
50  processLoc(ilink, byte);
51  }
52  } // loop on locs
53  byte = bytes[byteOffset + 4];
54  ilink = 8 + ireg;
55  linkMask = (1 << ilink);
56  if ((mIsFeeding & linkMask) || byte) {
57  std::invoke(mProcessReg, this, ilink, byte);
58  }
59  } // loop on half regional
60  } // loop on buffer index
61 }

63 void GBTBareDecoder::processLoc(size_t ilink, uint8_t byte)
64 {
66  if (mELinkDecoders[ilink].getNBytes() > 0) {
67  mELinkDecoders[ilink].add(byte);
68  if (mELinkDecoders[ilink].isComplete()) {
69  std::invoke(mOnDoneLoc, mOutputHandler, ilink, mELinkDecoders[ilink]);
70  mELinkDecoders[ilink].reset();
71  mIsFeeding &= (~(1 << ilink));
72  }
73  } else if ((byte & (raw::sSTARTBIT | raw::sCARDTYPE)) == (raw::sSTARTBIT | raw::sCARDTYPE)) {
74  mELinkDecoders[ilink].add(byte);
75  mIsFeeding |= (1 << ilink);
76  }
77 }
