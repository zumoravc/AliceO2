251 // testNeighbours returns true if the neighbours of dePadIndex, as
252 // returned by the Segmentation::forEachNeighbouringPad, are the same
253 // as the elements of expected vector.
254 bool testNeighbours(const Segmentation& seg, int dePadIndex, std::vector<PadInfo>& expected)
255 {
256  std::vector<int> nei;
257  seg.forEachNeighbouringPad(dePadIndex, [&nei](int depadindex) {
258  nei.push_back(depadindex);
259  });
260 
261  if (nei.size() != expected.size()) {
262  return false;
263  }
264  auto notFound = nei.size();
265  for (auto n : nei) {
266  for (auto e : expected) {
267  if (seg.padDualSampaId(n) == e.fec &&
268  seg.padDualSampaChannel(n) == e.ch &&
269  areEqual(seg.padPositionX(n), e.x) &&
270  areEqual(seg.padPositionY(n), e.y) &&
271  areEqual(seg.padSizeX(n), e.sx) &&
272  areEqual(seg.padSizeY(n), e.sy)) {
273  notFound--;
274  }
275  }
276  }
277  return notFound == 0;
278 }
