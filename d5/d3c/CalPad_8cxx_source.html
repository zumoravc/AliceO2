162 //_____________________________________________________________________________
163 double CalPad::getMeanRMS(double& rms, const CalDet* calDet, int type)
164 {
165  //
166  // Calculate mean an RMS of all rocs
167  // If calDet correct the CalROC from the detector coefficient
168  // type == 0 for gain and vdrift
169  // type == 1 for t0
170  //
171  double factor = 0.0;
172  if (type == 0)
173  factor = 1.0;
174  double sum = 0, sum2 = 0, n = 0, val;
175  for (int idet = 0; idet < kNdet; idet++) {
176  if (calDet)
177  factor = calDet->getValue(idet);
178  CalROC* calRoc = fROC[idet];
179  if (calRoc) {
180  for (int irow = 0; irow < calRoc->getNrows(); irow++) {
181  for (int icol = 0; icol < calRoc->getNcols(); icol++) {
182  if (type == 0)
183  val = calRoc->getValue(icol, irow) * factor;
184  else
185  val = calRoc->getValue(icol, irow) + factor;
186  sum += val;
187  sum2 += val * val;
188  n++;
189  }
190  }
191  }
192  }
193  double n1 = 1. / n;
194  double mean = sum * n1;
195  rms = TMath::Sqrt(TMath::Abs(sum2 * n1 - mean * mean));
196  return mean;
197 }

200 double CalPad::getMean(const CalDet* calDet, int type, CalPad* const outlierPad)
201 {
202  //

296 double CalPad::getLTM(double* sigma, double fraction, const CalDet* calDet, int type, CalPad* const outlierPad)
297 {
298  //
299  // return mean of the LTM and sigma of all ROCs
300  // If calDet correct the CalROC from the detector coefficient
301  // type == 0 for gain and vdrift
302  // type == 1 for t0
303  //
304  double factor = 0.0;
305  if (type == 0)
306  factor = 1.0;
307  double arrm[kNdet];
308  double arrs[kNdet];
309  double* sTemp = 0x0;
310  int n = 0;
311 
312  for (int idet = 0; idet < kNdet; idet++) {
313  if (calDet)
314  factor = calDet->getValue(idet);
315  CalROC* calRoc = fROC[idet];
316  if (calRoc) {
317  if (sigma)
318  sTemp = arrs + n;
319  CalROC* outlierROC = 0;
320  if (outlierPad)
321  outlierROC = outlierPad->getCalROC(idet);
322  if (type == 0)
323  arrm[n] = calRoc->getLTM(sTemp, fraction, outlierROC) * factor;
324  else
325  arrm[n] = calRoc->getLTM(sTemp, fraction, outlierROC) + factor;
326  n++;
327  }
328  }
329  if (sigma)
330  *sigma = TMath::Mean(n, arrs);
331  return TMath::Mean(n, arrm);
332 }

334 //_____________________________________________________________________________
335 TH1F* CalPad::makeHisto1D(const CalDet* calDet, int typedet, float min, float max, int type)
336 {
337  //
338  // make 1D histo
339  // type -1 = user defined range
340  // 0 = nsigma cut nsigma=min
341  // If calDet correct the CalROC from the detector coefficient
342  // typedet == 0 for gain and vdrift
343  // typedet == 1 for t0
344  //
345 
346  double factor = 0.0;
347  if (typedet == 0)
348  factor = 1.0;
349 
350  if (type >= 0) {
351  if (type == 0) {
352  // nsigma range
353  float mean = getMean(calDet, typedet);
354  float sigma = 0.0;
355  float kEpsilonr = 0.005;
356  if (getRMS(calDet, typedet) > kEpsilonr)
357  sigma = getRMS(calDet, typedet);
358  else {
359  double rms = 0.0;
360  sigma = getMeanRMS(rms, calDet, typedet);
361  }
362  float nsigma = TMath::Abs(min);
363  sigma *= nsigma;
364  if (sigma < kEpsilonr)
365  sigma = kEpsilonr;
366  min = mean - sigma;
367  max = mean + sigma;
368  }
369  if (type == 1) {
370  // fixed range
371  float mean = getMedian(calDet, typedet);
372  float delta = min;
373  min = mean - delta;
374  max = mean + delta;
375  }
376  if (type == 2) {
377  //
378  // LTM mean +- nsigma
379  //
380  double sigma;
381  float mean = getLTM(&sigma, max, calDet, typedet);
382  sigma *= min;
383  float kEpsilonr = 0.005;
384  if (sigma < kEpsilonr)
385  sigma = kEpsilonr;
386  min = mean - sigma;
387  max = mean + sigma;
388  }
389  }
390  char name[1000];
391  snprintf(name, 1000, "%s Pad 1D", getTitle());
392  TH1F* his = new TH1F(name, name, 100, min, max);
393  for (int idet = 0; idet < kNdet; idet++) {
394  if (calDet)
395  factor = calDet->getValue(idet);
396  if (fROC[idet]) {
397  for (int irow = 0; irow < fROC[idet]->getNrows(); irow++) {
398  for (int icol = 0; icol < fROC[idet]->getNcols(); icol++) {
399  if (typedet == 0)
400  his->Fill(fROC[idet]->getValue(irow, icol) * factor);
401  else
402  his->Fill(fROC[idet]->getValue(irow, icol) + factor);
403  }
404  }
405  }
406  }
407  return his;
408 }

501  his->SetBinContent(binz, biny, value * factor);
502  else
503  his->SetBinContent(binz, biny, value + factor);
504  }
505  }
506  }
507  }
508  }
509  his->SetXTitle("z (cm)");
510  his->SetYTitle("y (cm)");
511  his->SetStats(0);
512  his->SetMaximum(max);
513  his->SetMinimum(min);
514  delete trdGeo;
515  return his;
516 }

663 //_____________________________________________________________________________
664 bool CalPad::add(const CalPad* pad, double c1, const CalDet* calDet1, const CalDet* calDet2, int type)
665 {
666  //
667  // add calpad channel by channel multiplied by c1 - all ROCs
668  // If calDet1 and calDet2, the correct the CalROC from the detector coefficient
669  // then you have calDet1 and the calPad together
670  // calDet2 and pad together
671  // typedet == 0 for gain and vdrift
672  // typedet == 1 for t0
673  //
674  float kEpsilon = 0.000000000001;
675 
676  double factor1 = 0.0;
677  double factor2 = 0.0;
678  if (type == 0) {
679  factor1 = 1.0;
680  factor2 = 1.0;
681  }
682  bool result = kTRUE;
683  for (int idet = 0; idet < kNdet; idet++) {
684  if (calDet1)
685  factor1 = calDet1->getValue(idet);
686  if (calDet2)
687  factor2 = calDet2->getValue(idet);
688  if (fROC[idet]) {
689  if (type == 0) {
690  if (TMath::Abs(factor1) > kEpsilon) {
691  if (!fROC[idet]->add(pad->getCalROC(idet), c1 * factor2 / factor1))
692  result = kFALSE;
693  } else
694  result = kFALSE;
695  } else {
696  CalROC* croc = new CalROC((const CalROC)*pad->getCalROC(idet));
