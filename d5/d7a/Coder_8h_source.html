33 template <typename T>
34 using State = T;

38 // READ ME FIRST:
39 //
40 // This is designed like a typical arithmetic coder API, but there's three
41 // twists you absolutely should be aware of before you start hacking:
42 //
43 // 1. You need to encode data in *reverse* - last symbol first. rANS works
44 // like a stack: last in, first out.
45 // 2. Likewise, the encoder outputs bytes *in reverse* - that is, you give
46 // it a pointer to the *end* of your buffer (exclusive), and it will
47 // slowly move towards the beginning as more bytes are emitted.
48 // 3. Unlike basically any other entropy coder implementation you might
49 // have used, you can interleave data from multiple independent rANS
50 // encoders into the same bytestream without any extra signaling;
51 // you can also just write some bytes by yourself in the middle if
52 // you want to. This is in addition to the usual arithmetic encoder
53 // property of being able to switch models on the fly. Writing raw
54 // bytes can be useful when you have some data that you know is
55 // incompressible, and is cheaper than going through the rANS encode
56 // function. Using multiple rANS coders on the same byte stream wastes
57 // a few bytes compared to using just one, but execution of two
58 // independent encoders can happen in parallel on superscalar and
59 // Out-of-Order CPUs, so this can be *much* faster in tight decoding
60 // loops.
61 //
62 // This is why all the rANS functions take the write pointer as an
63 // argument instead of just storing it in some context struct.
64 
65 // --------------------------------------------------------------------------
66 template <typename T, typename Stream_t>
67 class Coder
68 {
69  // the Coder works either with a 64Bit state and 32 bit streaming or
70  //a 32 Bit state and 8 Bit streaming We need to make sure it gets initialized with
71  //the right template arguments at compile time.
72  static_assert((sizeof(T) == sizeof(uint32_t) && sizeof(Stream_t) == sizeof(uint8_t)) ||
73  (sizeof(T) == sizeof(uint64_t) && sizeof(Stream_t) == sizeof(uint32_t)),
74  "Coder can either be 32Bit with 8 Bit stream type or 64 Bit Type with 32 Bit stream type");
75 
76  public:
77  Coder() = delete;
78 
79  // Initialize a rANS encoder.
80  static void encInit(State<T>* r)
81  {
82  *r = LOWER_BOUND;
83  };
84 
85  // Encodes a single symbol with range start "start" and frequency "freq".
86  // All frequencies are assumed to sum to "1 << scale_bits", and the
87  // resulting bytes get written to ptr (which is updated).
88  //
89  // NOTE: With rANS, you need to encode symbols in *reverse order*, i.e. from
90  // beginning to end! Likewise, the output bytestream is written *backwards*:
91  // ptr starts pointing at the end of the output buffer and keeps decrementing.
92  static void encPut(State<T>* r, Stream_t** pptr, uint32_t start, uint32_t freq, uint32_t scale_bits)
93  {
94  // renormalize
95  State<T> x = encRenorm(*r, pptr, freq, scale_bits);
96 
97  // x = C(s,x)
98  *r = ((x / freq) << scale_bits) + (x % freq) + start;
99  };
100 
101  // Flushes the rANS encoder.
102  static void encFlush(State<T>* r, Stream_t** pptr)
103  {
104  T x = *r;
105  Stream_t* ptr = *pptr;
106 
107  if constexpr (needs64Bit<T>()) {
108  ptr -= 2;
109  ptr[0] = static_cast<Stream_t>(x >> 0);
110  ptr[1] = static_cast<Stream_t>(x >> 32);
111  } else {
112  ptr -= 4;
113  ptr[0] = static_cast<Stream_t>(x >> 0);
114  ptr[1] = static_cast<Stream_t>(x >> 8);
115  ptr[2] = static_cast<Stream_t>(x >> 16);
116  ptr[3] = static_cast<Stream_t>(x >> 24);
117  }
118 
119  *pptr = ptr;
120  };
121 
122  // Initializes a rANS decoder.
123  // Unlike the encoder, the decoder works forwards as you'd expect.
124  static void decInit(State<T>* r, Stream_t** pptr)
125  {
126  T x;
127  Stream_t* ptr = *pptr;
128 
129  if constexpr (needs64Bit<T>()) {
130  x = static_cast<T>(ptr[0]) << 0;
131  x |= static_cast<T>(ptr[1]) << 32;
132  ptr += 2;
133  } else {
134  x = ptr[0] << 0;
135  x |= ptr[1] << 8;
136  x |= ptr[2] << 16;
137  x |= ptr[3] << 24;
138  ptr += 4;
139  }
140 
141  *pptr = ptr;
142  *r = x;
143  };

145  // Returns the current cumulative frequency (map it to a symbol yourself!)
146  static uint32_t decGet(State<T>* r, uint32_t scale_bits)
147  {
148  return *r & ((1u << scale_bits) - 1);
149  };
150 
151  // Advances in the bit stream by "popping" a single symbol with range start
152  // "start" and frequency "freq". All frequencies are assumed to sum to "1 << scale_bits",
153  // and the resulting bytes get written to ptr (which is updated).
154  static void decAdvance(State<T>* r, Stream_t** pptr, uint32_t start, uint32_t freq, uint32_t scale_bits)
155  {
156  T mask = (1ull << scale_bits) - 1;
157 
158  // s, x = D(x)
159  T x = *r;
160  x = freq * (x >> scale_bits) + (x & mask) - start;
161 
162  // renormalize
163  decRenorm(&x, pptr);
164 
165  *r = x;
166  };
167 
168  // Encodes a given symbol. This is faster than straight RansEnc since we can do
169  // multiplications instead of a divide.
170  //
171  // See Rans32EncSymbolInit for a description of how this works.
172  static void encPutSymbol(State<T>* r, Stream_t** pptr, EncoderSymbol<T> const* sym, uint32_t scale_bits)
173  {
174  assert(sym->freq != 0); // can't encode symbol with freq=0
175 
176  // renormalize
177  T x = encRenorm(*r, pptr, sym->freq, scale_bits);
178 
179  // x = C(s,x)
180  T q;
181 
182  if constexpr (needs64Bit<T>()) {
183  // This code needs support for 64-bit long multiplies with 128-bit result
184  // (or more precisely, the top 64 bits of a 128-bit result).
185  q = static_cast<T>((static_cast<uint128>(x) * sym->rcp_freq) >> 64);
186  } else {
187  q = static_cast<T>((static_cast<uint64_t>(x) * sym->rcp_freq) >> 32);
188  }
189  q = q >> sym->rcp_shift;
190 
191  *r = x + sym->bias + q * sym->cmpl_freq;
192  };
193 
194  // Equivalent to Rans32DecAdvance that takes a symbol.
195  static void decAdvanceSymbol(State<T>* r, Stream_t** pptr, DecoderSymbol const* sym, uint32_t scale_bits)
196  {
197  decAdvance(r, pptr, sym->start, sym->freq, scale_bits);
198  };

200  // Advances in the bit stream by "popping" a single symbol with range start
201  // "start" and frequency "freq". All frequencies are assumed to sum to "1 << scale_bits".
202  // No renormalization or output happens.
203  static void decAdvanceStep(State<T>* r, uint32_t start, uint32_t freq, uint32_t scale_bits)
204  {
205  T mask = (1u << scale_bits) - 1;
206 
207  // s, x = D(x)
208  T x = *r;
209  *r = freq * (x >> scale_bits) + (x & mask) - start;
210  };
211 
212  // Equivalent to Rans32DecAdvanceStep that takes a symbol.
213  static void decAdvanceSymbolStep(State<T>* r, DecoderSymbol const* sym, uint32_t scale_bits)
214  {
215  decAdvanceStep(r, sym->start, sym->freq, scale_bits);
216  };

218  // Renormalize.
219  static inline void decRenorm(State<T>* r, Stream_t** pptr)
220  {
221  // renormalize
222  T x = *r;
223  if (x < LOWER_BOUND) {
224  if constexpr (needs64Bit<T>()) {
225  x = (x << STREAM_BITS) | **pptr;
226  *pptr += 1;
227  assert(x >= LOWER_BOUND);
228  } else {
229  Stream_t* ptr = *pptr;
230  do
231  x = (x << STREAM_BITS) | *ptr++;
232  while (x < LOWER_BOUND);
233  *pptr = ptr;
234  }
235  }
236  *r = x;
237  }
238 
239  private:
240  // Renormalize the encoder.
241  static inline State<T> encRenorm(State<T> x, Stream_t** pptr, uint32_t freq, uint32_t scale_bits)
242  {
243  T x_max = ((LOWER_BOUND >> scale_bits) << STREAM_BITS) * freq; // this turns into a shift.
244  if (x >= x_max) {
245  if constexpr (needs64Bit<T>()) {
246  *pptr -= 1;
247  **pptr = static_cast<Stream_t>(x);
248  x >>= STREAM_BITS;
249  } else {
250  Stream_t* ptr = *pptr;
251  do {
252  *--ptr = static_cast<Stream_t>(x & 0xff);
253  x >>= STREAM_BITS;
254  } while (x >= x_max);
255  *pptr = ptr;
256  }
257  }
258  return x;
259  };
260 
261  // L ('l' in the paper) is the lower bound of our normalization interval.
262  // Between this and our byte-aligned emission, we use 31 (not 32!) bits.
263  // This is done intentionally because exact reciprocals for 31-bit uints
264  // fit in 32-bit uints: this permits some optimizations during encoding.
265  inline static constexpr T LOWER_BOUND = needs64Bit<T>() ? (1u << 31) : (1u << 23); // lower bound of our normalization interval
266 
267  inline static constexpr T STREAM_BITS = sizeof(Stream_t) * 8; // lower bound of our normalization interval
268 };
