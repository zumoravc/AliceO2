22 // class encoding sim track status
23 struct SimTrackStatus {
24  public:
25  enum SimTrackStatus_Enum {
26  kTrackEntering = 0x1,
27  kTrackInside = 0x1 << 1,
28  kTrackExiting = 0x1 << 2,
29  kTrackOut = 0x1 << 3,
30  kTrackStopped = 0x1 << 4,
31  kTrackAlive = 0x1 << 5,
32  kTrackNew = 0x1 << 6
33  };
34  SimTrackStatus() = default;
35  SimTrackStatus(const TVirtualMC& vmc)
36  {
37  // This is quite annoying since every single call
38  // is virtual
39  if (vmc.IsTrackEntering()) {
40  mStatus |= kTrackEntering;
41  }
42  if (vmc.IsTrackExiting()) {
43  mStatus |= kTrackExiting;
44  }
45  if (vmc.IsTrackInside()) {
46  mStatus |= kTrackInside;
47  }
48  if (vmc.IsTrackOut()) {
49  mStatus |= kTrackOut;
50  }
51  if (vmc.IsTrackStop()) {
52  mStatus |= kTrackAlive;
53  }
54  if (vmc.IsNewTrack()) {
55  mStatus |= kTrackNew;
56  }
57  }
58  bool isEntering() const { return mStatus & kTrackEntering; }
59  bool isInside() const { return mStatus & kTrackInside; }
60  bool isExiting() const { return mStatus & kTrackExiting; }
61  bool isOut() const { return mStatus & kTrackOut; }
62  bool isStopped() const { return mStatus & kTrackStopped; }
63  bool isAlive() const { return mStatus & kTrackAlive; }
64  bool isNew() const { return mStatus & kTrackNew; }
65  unsigned char getStatusWord() const { return mStatus; }
66 
67  friend std::ostream& operator<<(std::ostream&, const SimTrackStatus&);
68 
69  private:
70  unsigned char mStatus = 0;
71  ClassDefNV(SimTrackStatus, 1);
72 };

87 class TrackReference
88 {
89  public:
91  TrackReference() = default;

96 
98  ~TrackReference() = default;
99 
100  Int_t getTrackID() const { return mTrackNumber; }
101  void setTrackID(Int_t track) { mTrackNumber = track; }
102  void setLength(float length) { mTrackLength = length; }
103  void setTime(float time) { mTof = time; }
104  float getLength() const { return mTrackLength; }
105  float getTime() const { return mTof; }
106  float R() const { return TMath::Sqrt(mX * mX + mY * mY); }
107 
108  float Pt() const { return TMath::Sqrt(mPX * mPX + mPY * mPY); }
109  float PhiPos() const { return TMath::ATan2(mY, mX); }
110  float Phi() const { return TMath::ATan2(mPY, mPX); }
111  float Theta() const { return TMath::ACos(mPZ / P()); }
112  float X() const { return mX; }
113  float Y() const { return mY; }
114  float Z() const { return mZ; }
115  float Px() const { return mPX; }
116  float Py() const { return mPY; }
117  float Pz() const { return mPZ; }
118  float P() const { return TMath::Sqrt(mPX * mPX + mPY * mPY + mPZ * mPZ); }
119  Int_t getUserId() const { return mUserId; }
120  Int_t getDetectorId() const { return mDetectorId; }
121  void setDetectorId(Int_t id) { mDetectorId = id; }
122 
123  void setPosition(float x, float y, float z)
124  {
125  mX = x;
126  mY = y;
127  mZ = z;
128  }
129 
130  void setMomentum(float px, float py, float pz)
131  {
132  mPX = px;
133  mPY = py;
134  mPZ = pz;
135  }
136 
137  void setUserId(Int_t userId) { mUserId = userId; }
138 
139  // Methods to get position of the track reference in
140  // in the TPC/TRD/TOF Tracking coordinate system
141  float phiPosition() const { return TMath::Pi() + TMath::ATan2(-mY, -mX); }
142 
143  float Alpha() const { return TMath::Pi() * (20 * ((((Int_t)(phiPosition() * 180 / TMath::Pi())) / 20)) + 10) / 180.; }
144 
145  float LocalX() const
146  {
147  auto alpha = Alpha();
148  return mX * TMath::Cos(-alpha) - mY * TMath::Sin(-alpha);
149  }
150 
151  float LocalY() const
152  {
153  auto alpha = Alpha();
154  return mX * TMath::Sin(-alpha) + mY * TMath::Cos(-alpha);
155  }
156 
157  const SimTrackStatus& getTrackStatus() const { return mStatus; }
158 
159  private:
160  Int_t mTrackNumber = 0;
161  float mX = 0;
162  float mY = 0;
163  float mZ = 0;
164  float mPX = 0;
165  float mPY = 0;
166  float mPZ = 0;
167  float mTrackLength = 0;
168  float mTof = 0;
169  Int_t mUserId = 0;
170  Int_t mDetectorId = 0;
171  SimTrackStatus mStatus;
172 
173  friend std::ostream& operator<<(std::ostream&, const TrackReference&);
174 
175  ClassDefNV(TrackReference, 1); // Base class for all Alice track references
176 };
