60 uint16_t SampaCluster::nof10BitWords() const
61 {
62  uint16_t n10{2}; // 10 bits (nsamples) + 10 bits (sampaTime)
63  if (isClusterSum()) {
64  n10 += 2; // 20 bits (chargesum)
65  } else {
66  for (auto s : samples) {
67  ++n10; // 10 bits for each sample
68  }
69  }
70  return n10;
71 }
