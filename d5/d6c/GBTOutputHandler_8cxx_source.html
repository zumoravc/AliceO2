32 void GBTOutputHandler::setIR(uint16_t bc, uint32_t orbit, int pageCnt)
33 {
35  if (mIRs[0].isDummy()) {
36  for (auto& ir : mIRs) {
37  ir.bc = bc;
38  ir.orbit = orbit;
39  }
40  for (auto& clock : mLastClock) {
41  clock = constants::lhc::LHCMaxBunches;
42  }
43  for (auto& clock : mCalibClocks) {
44  clock = constants::lhc::LHCMaxBunches;
45  }
46  }
47 
48  if (pageCnt == 0) {
49  // FIXME: in the tests, the BC counter increases at each RDH
50  // However, the inner clock of the RO is not reset,
51  // so if we want to have the absolute BC,
52  // we need to store only the BC counter of the first page.
53  // Not sure how it will work on data...
54  mIRFirstPage.bc = bc;
55  mIRFirstPage.orbit = orbit;
56  }
57 }

65 void GBTOutputHandler::addBoard(size_t ilink, const ELinkDecoder& decoder)
66 {
68  uint16_t localClock = decoder.getCounter();
69  EventType eventType = EventType::Standard;
70  if (decoder.getTriggerWord() & raw::sCALIBRATE) {
71  mCalibClocks[ilink] = localClock;
72  eventType = EventType::Noise;
73  } else if (localClock == mCalibClocks[ilink] + sDelayCalibToFET) {
74  eventType = EventType::Dead;
75  }
76  auto firstEntry = mData.size();
77  mData.push_back({decoder.getStatusWord(), decoder.getTriggerWord(), crateparams::makeUniqueLocID(crateparams::getCrateIdFromROId(mFeeId), decoder.getId()), decoder.getInputs()});
78  InteractionRecord intRec(mIRs[ilink].bc + localClock - sDelayBCToLocal, mIRs[ilink].orbit);
79  mROFRecords.emplace_back(intRec, eventType, firstEntry, 1);
80 }

82 void GBTOutputHandler::addLoc(size_t ilink, const ELinkDecoder& decoder)
83 {
85  addBoard(ilink, decoder);
86  for (int ich = 0; ich < 4; ++ich) {
87  if ((mData.back().firedChambers & (1 << ich))) {
88  mData.back().patternsBP[ich] = decoder.getPattern(0, ich);
89  mData.back().patternsNBP[ich] = decoder.getPattern(1, ich);
90  }
91  }
92  if (mROFRecords.back().eventType == EventType::Dead) {
93  if (invertPattern(mData.back())) {
94  mData.pop_back();
95  mROFRecords.pop_back();
96  }
97  }
98 }

100 bool GBTOutputHandler::updateIR(size_t ilink, const ELinkDecoder& decoder)
101 {
103  if (decoder.getTriggerWord() & raw::sORB) {
104  // This is the answer to an orbit trigger
105  // The local clock is reset: we are now in synch with the new HB
106  mIRs[ilink] = mIRFirstPage;
107  if (!(decoder.getTriggerWord() & (raw::sSOX | raw::sEOX))) {
108  mLastClock[ilink] = decoder.getCounter();
109  }
110  return true;
111  }
112  return false;
113 }

134 void GBTOutputHandler::onDoneRegDebug(size_t ilink, const ELinkDecoder& decoder)
135 {
137  updateIR(ilink, decoder);
138  addBoard(ilink, decoder);
139  // The board creation is optimized for the local boards, not the regional
140  // (which are transmitted only in debug mode).
141  // So, at this point, for the regional board, the local Id is actually the crate ID.
142  // If we want to distinguish the two regional e-links, we can use the link ID instead
143  mData.back().boardId = crateparams::makeUniqueLocID(crateparams::getCrateIdFromROId(mFeeId), ilink + 8 * (crateparams::getGBTIdInCrate(mFeeId) - 1));
144  if (mData.back().triggerWord == 0) {
145  if (mROFRecords.back().interactionRecord.bc < sDelayRegToLocal) {
146  // In the tests, the HB does not really correspond to a change of orbit
147  // So we need to keep track of the last clock at which the HB was received
148  // and come back to that value
149  // FIXME: Remove this part as well as mLastClock when tests are no more needed
150  mROFRecords.back().interactionRecord -= (constants::lhc::LHCMaxBunches - mLastClock[ilink] - 1);
151  }
152  // This is a self-triggered event.
153  // In this case the regional card needs to wait to receive the tracklet decision of each local
154  // which result in a delay that needs to be subtracted if we want to be able to synchronize
155  // local and regional cards for the checks
156  mROFRecords.back().interactionRecord -= sDelayRegToLocal;
157  }
158 }

160 bool GBTOutputHandler::invertPattern(LocalBoardRO& loc)
161 {
163  for (int ich = 0; ich < 4; ++ich) {
164  loc.patternsBP[ich] = ~loc.patternsBP[ich];
165  loc.patternsNBP[ich] = ~loc.patternsNBP[ich];
166  if (loc.patternsBP[ich] == 0 && loc.patternsNBP[ich] == 0) {
167  loc.firedChambers &= ~(1 << ich);
168  }
169  }
170  return (loc.firedChambers == 0);
171 }
