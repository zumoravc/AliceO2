30 int DigitPixelReader::decodeNextTrigger()
31 {
32  // prerare data of the next trigger, return number of digits
33  while (++mIdROF < mROFRecVec.size()) {
34  if (mROFRecVec[mIdROF].getNEntries() > 0) {
35  mIdDig = 0; // jump to the 1st digit of the trigger
36  mInteractionRecord = mROFRecVec[mIdROF].getBCData();
37  return mROFRecVec[mIdROF].getNEntries();
38  }
39  }
40  return 0;
41 }

44 ChipPixelData* DigitPixelReader::getNextChipData(std::vector<ChipPixelData>& chipDataVec)
45 {
46  // decode data of single chip to corresponding slot of chipDataVec
47  if (mIdROF >= mROFRecVec.size()) {
48  return nullptr; // TF is done
49  }
50  if (mIdROF < 0 || mIdDig >= mROFRecVec[mIdROF].getNEntries()) {
51  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no automatic decoding is asked or we are at the end of the TF
52  return nullptr;
53  }
54  }
55  auto chipID = mDigits[mROFRecVec[mIdROF].getFirstEntry() + mIdDig].getChipIndex();
56  return getNextChipData(chipDataVec[chipID]) ? &chipDataVec[chipID] : nullptr;
57 }

60 bool DigitPixelReader::getNextChipData(ChipPixelData& chipData)
61 {
62  // decode data of single chip to chipData
63  if (mIdROF >= mROFRecVec.size()) {
64  return false; // TF is done
65  }
66  if (mIdROF < 0 || mIdDig >= mROFRecVec[mIdROF].getNEntries()) {
67  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no automatic decoding is asked or we are at the end of the TF
68  return false;
69  }
70  }
71  chipData.clear();
72  int did = mROFRecVec[mIdROF].getFirstEntry() + mIdDig;
73  chipData.setStartID(did); // for the MC references
74  const auto* digit = &mDigits[did];
75  chipData.setChipID(digit->getChipIndex());
76  chipData.setROFrame(mROFRecVec[mIdROF].getROFrame());
77  chipData.setInteractionRecord(mInteractionRecord);
78  chipData.setTrigger(mTrigger);
79  chipData.getData().emplace_back(digit);
80  int lim = mROFRecVec[mIdROF].getFirstEntry() + mROFRecVec[mIdROF].getNEntries();
81  while ((++did < lim) && (digit = &mDigits[did])->getChipIndex() == chipData.getChipID()) {
82  chipData.getData().emplace_back(digit);
83  }
84  mIdDig = did - mROFRecVec[mIdROF].getFirstEntry();
85  return true;
86 }

89 void DigitPixelReader::openInput(const std::string inpName, o2::detectors::DetID det)
90 {
91  // open input file, load digits, MC labels
92  assert(det.getID() == o2::detectors::DetID::ITS || det.getID() == o2::detectors::DetID::MFT);
93 
94  clear();
95  std::string detName = det.getName();
96 
97  if (!(mInputTree = o2::utils::RootChain::load("o2sim", inpName))) {
98  LOG(FATAL) << "Failed to load Digits tree from " << inpName;
99  }
100  mInputTree->SetBranchAddress((detName + "Digit").c_str(), &mDigitsSelf);
101  if (!mDigitsSelf) {
102  LOG(FATAL) << "Failed to find " << (detName + "Digit").c_str() << " branch in the " << mInputTree->GetName()
103  << " from file " << inpName;
104  }
105 
106  mInputTree->SetBranchAddress((detName + "DigitROF").c_str(), &mROFRecVecSelf);
107  if (!mROFRecVecSelf) {
108  LOG(FATAL) << "Failed to find " << (detName + "DigitROF").c_str() << " branch in the " << mInputTree->GetName()
109  << " from file " << inpName;
110  }
111 
112  mInputTree->SetBranchAddress((detName + "DigitMC2ROF").c_str(), &mMC2ROFRecVecSelf);
113  if (!mMC2ROFRecVecSelf) {
114  LOG(FATAL) << "Failed to find " << (detName + "DigitMC2ROF").c_str() << " branch in the " << mInputTree->GetName()
115  << " from file " << inpName;
116  }
117 
118  mInputTree->SetBranchAddress((detName + "DigitMCTruth").data(), &mDigitsMCTruthSelf);
119  setDigitsMCTruth(mDigitsMCTruthSelf); // it will be assigned again at the reading, this is just to signal that the MCtruth is there
120 }

123 bool DigitPixelReader::readNextEntry()
124 {
125  // load next entry from the self-managed input
126  auto nev = mInputTree->GetEntries();
127  auto evID = mInputTree->GetReadEntry();
128  if (evID < -1)
129  evID = -1;
130  if (++evID < nev) {
131  init();
132  mInputTree->GetEntry(evID);
133  setDigits(gsl::span(mDigitsSelf->data(), mDigitsSelf->size()));
134  setROFRecords(gsl::span(mROFRecVecSelf->data(), mROFRecVecSelf->size()));
135  setMC2ROFRecords(gsl::span(mMC2ROFRecVecSelf->data(), mMC2ROFRecVecSelf->size()));
136  setDigitsMCTruth(mDigitsMCTruthSelf);
137  return true;
138  } else {
139  return false;
140  }
141 }
