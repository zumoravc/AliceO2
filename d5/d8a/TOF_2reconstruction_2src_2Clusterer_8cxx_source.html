24 void Clusterer::process(DataReader& reader, std::vector<Cluster>& clusters, MCLabelContainer const* digitMCTruth)
25 {
26  TStopwatch timerProcess;
27  timerProcess.Start();
28 
29  reader.init();
30  int totNumDigits = 0;
31 
32  while (reader.getNextStripData(mStripData)) {
33  LOG(DEBUG) << "TOFClusterer got Strip " << mStripData.stripID << " with Ndigits "
34  << mStripData.digits.size();
35  totNumDigits += mStripData.digits.size();
36 
37  calibrateStrip();
38  processStrip(clusters, digitMCTruth);
39  }
40 
41  LOG(DEBUG) << "We had " << totNumDigits << " digits in this event";
42  timerProcess.Stop();
43  printf("Timing:\n");
44  printf("Clusterer::process: ");
45  timerProcess.Print();
46 }

49 void Clusterer::calibrateStrip()
50 {
51  // method to calibrate the times from the current strip
52 
53  for (int idig = 0; idig < mStripData.digits.size(); idig++) {
54  // LOG(DEBUG) << "Checking digit " << idig;
55  Digit* dig = &mStripData.digits[idig];
56  double calib = mCalibApi->getTimeCalibration(dig->getChannel(), dig->getTOT() * Geo::TOTBIN_NS);
57  //printf("channel %d) isProblematic = %d, fractionUnderPeak = %f\n",dig->getChannel(),mCalibApi->isProblematic(dig->getChannel()),mCalibApi->getFractionUnderPeak(dig->getChannel())); // toberem
58  dig->setIsProblematic(mCalibApi->isProblematic(dig->getChannel()));
59  dig->setCalibratedTime(dig->getTDC() * Geo::TDCBIN + dig->getBC() * o2::constants::lhc::LHCBunchSpacingNS * 1E3 - calib); //TODO: to be checked that "-" is correct, and we did not need "+" instead :-)
60  //printf("calibration correction = %f\n",calib); // toberem
61  }
62 }

65 void Clusterer::processStrip(std::vector<Cluster>& clusters, MCLabelContainer const* digitMCTruth)
66 {
67  // method to clusterize the current strip
68 
69  Int_t detId[5];
70  Int_t chan, chan2, chan3;
71  Int_t strip1, strip2;
72  Int_t iphi, iphi2, iphi3;
73  Int_t ieta, ieta2, ieta3; // it is the number of padz-row increasing along the various strips
74 
75  for (int idig = 0; idig < mStripData.digits.size(); idig++) {
76  // LOG(DEBUG) << "Checking digit " << idig;
77  Digit* dig = &mStripData.digits[idig];
78  //printf("checking digit %d - alreadyUsed=%d - problematic=%d\n",idig,dig->isUsedInCluster(),dig->isProblematic()); // toberem
79  if (dig->isUsedInCluster() || dig->isProblematic())
80  continue; // the digit was already used to build a cluster, or it was declared problematic
81 
82  mNumberOfContributingDigits = 0;
83  dig->getPhiAndEtaIndex(iphi, ieta);
84  if (mStripData.digits.size() > 1)
85  LOG(DEBUG) << "idig = " << idig;
86 
87  // first we make a cluster out of the digit
88  int noc = clusters.size();
89  // LOG(DEBUG) << "noc = " << noc << "\n";
90  clusters.emplace_back();
91  Cluster& c = clusters[noc];
92  addContributingDigit(dig);
93  double timeDig = dig->getCalibratedTime();
94 
95  for (int idigNext = idig + 1; idigNext < mStripData.digits.size(); idigNext++) {
96  Digit* digNext = &mStripData.digits[idigNext];
97  if (digNext->isUsedInCluster() || dig->isProblematic())
98  continue; // the digit was already used to build a cluster, or was problematic
99  // check if the TOF time are close enough to be merged; if not, it means that nothing else will contribute to the cluster (since digits are ordered in time)
100  double timeDigNext = digNext->getCalibratedTime(); // in ps
101  LOG(DEBUG) << "Time difference = " << timeDigNext - timeDig;
102  if (timeDigNext - timeDig > 500 /*in ps*/)
103  break;
104  digNext->getPhiAndEtaIndex(iphi2, ieta2);
105 
106  // check if the fired pad are close in space
107  LOG(DEBUG) << "phi difference = " << iphi - iphi2;
108  LOG(DEBUG) << "eta difference = " << ieta - ieta2;
109  if ((TMath::Abs(iphi - iphi2) > 1) || (TMath::Abs(ieta - ieta2) > 1))
110  continue;
111 
112  // if we are here, the digit contributes to the cluster
113  addContributingDigit(digNext);
114 
115  } // loop on the second digit
116 
117  //printf("build cluster\n");
118  buildCluster(c, digitMCTruth); // toberem
119 
120  } // loop on the first digit
121 }

123 void Clusterer::addContributingDigit(Digit* dig)
124 {
125 
126  // adding a digit to the array that stores the contributing ones
127 
128  if (mNumberOfContributingDigits == 6) {
129  LOG(WARNING) << "The cluster has already 6 digits associated to it, we cannot add more; returning without doing anything";
130 
131  int phi, eta;
132  for (int i = 0; i < mNumberOfContributingDigits; i++) {
133  mContributingDigit[i]->getPhiAndEtaIndex(phi, eta);
134  LOG(WARNING) << "digit already in " << i << ", channel = " << mContributingDigit[i]->getChannel() << ",phi,eta = (" << phi << "," << eta << "), TDC = " << mContributingDigit[i]->getTDC() << ", calibrated time = " << mContributingDigit[i]->getCalibratedTime();
135  }
136 
137  dig->getPhiAndEtaIndex(phi, eta);
138  LOG(WARNING) << "skipped digit"
139  << ", channel = " << dig->getChannel() << ",phi,eta = (" << phi << "," << eta << "), TDC = " << dig->getTDC() << ", calibrated time = " << dig->getCalibratedTime();
140 
141  dig->setIsUsedInCluster(); // flag is at used in any case
142 
143  return;
144  }
145  mContributingDigit[mNumberOfContributingDigits] = dig;
146  mNumberOfContributingDigits++;
147  dig->setIsUsedInCluster();
148 
149  return;
150 }

153 void Clusterer::buildCluster(Cluster& c, MCLabelContainer const* digitMCTruth)
154 {
155  static const float inv12 = 1. / 12.;
156 
157  // here we finally build the cluster from all the digits contributing to it
158 
159  Digit* temp;
160  for (int idig = 1; idig < mNumberOfContributingDigits; idig++) {
161  // the digit[0] will be the main one
162  if (mContributingDigit[idig]->getTOT() > mContributingDigit[0]->getTOT()) {
163  temp = mContributingDigit[0];
164  mContributingDigit[0] = mContributingDigit[idig];
165  mContributingDigit[idig] = temp;
166  }
167  }
168 
169  c.setMainContributingChannel(mContributingDigit[0]->getChannel());
170  c.setTime(mContributingDigit[0]->getCalibratedTime()); // time in ps (for now we assume it calibrated)
171  c.setTimeRaw(mContributingDigit[0]->getTDC() * Geo::TDCBIN + mContributingDigit[0]->getBC() * o2::constants::lhc::LHCBunchSpacingNS * 1E3); // time in ps (for now we assume it calibrated)
172 
173  c.setTot(mContributingDigit[0]->getTOT() * Geo::TOTBIN_NS); // TOT in ns (for now we assume it calibrated)
174  //setL0L1Latency(); // to be filled (maybe)
175  //setDeltaBC(); // to be filled (maybe)
176 
177  int chan1, chan2;
178  int phi1, phi2;
179  int eta1, eta2;
180  int deltaPhi, deltaEta;
181  int mask;
182 
183  mContributingDigit[0]->getPhiAndEtaIndex(phi1, eta1);
184  // now set the mask with the secondary digits
185  for (int idig = 1; idig < mNumberOfContributingDigits; idig++) {
186  mContributingDigit[idig]->getPhiAndEtaIndex(phi2, eta2);
187  deltaPhi = phi1 - phi2;
188  deltaEta = eta1 - eta2;
189  if (deltaPhi == 1) { // the digit is to the LEFT of the cluster; let's check about UP/DOWN/Same Line
190  if (deltaEta == 1) { // the digit is DOWN LEFT wrt the cluster
191  mask = Cluster::kDownLeft;
192  } else if (deltaEta == -1) { // the digit is UP LEFT wrt the cluster
193  mask = Cluster::kUpLeft;
194  } else { // the digit is LEFT wrt the cluster
195  mask = Cluster::kLeft;
196  }
197  } else if (deltaPhi == -1) { // the digit is to the RIGHT of the cluster; let's check about UP/DOWN/Same Line
198  if (deltaEta == 1) { // the digit is DOWN RIGHT wrt the cluster
199  mask = Cluster::kDownRight;
200  } else if (deltaEta == -1) { // the digit is UP RIGHT wrt the cluster
201  mask = Cluster::kUpRight;
202  } else { // the digit is RIGHT wrt the cluster
203  mask = Cluster::kRight;
204  }
205  } else if (deltaPhi == 0) { // the digit is on the same column as the cluster; is it UP or Down?
206  if (deltaEta == 1) { // the digit is DOWN wrt the cluster
207  mask = Cluster::kDown;
208  } else if (deltaEta == -1) { // the digit is UP wrt the cluster
209  mask = Cluster::kUp;
210  } else { // impossible!!
211  LOG(DEBUG) << " Check what is going on, the digit you are trying to merge to the cluster must be in a different channels... ";
212  }
213  } else { // impossible!!! We checked above...
214  LOG(DEBUG) << " Check what is going on, the digit you are trying to merge to the cluster is too far from the cluster, you should have not got here... ";
215  }
216  c.addBitInContributingChannels(mask);
217  }
218 
219  // filling the MC labels of this cluster; the first will be those of the main digit; then the others
220  if (digitMCTruth != nullptr) {
221  int lbl = mClsLabels->getIndexedSize(); // this should correspond to the number of digits also;
222  //printf("lbl = %d\n", lbl);
223  for (int i = 0; i < mNumberOfContributingDigits; i++) {
224  //printf("contributing digit = %d\n", i);
225  int digitLabel = mContributingDigit[i]->getLabel();
226  //printf("digitLabel = %d\n", digitLabel);
227  gsl::span<const o2::MCCompLabel> mcArray = digitMCTruth->getLabels(digitLabel);
228  for (int j = 0; j < static_cast<int>(mcArray.size()); j++) {
229  //printf("checking element %d in the array of labels\n", j);
230  auto label = digitMCTruth->getElement(digitMCTruth->getMCTruthHeader(digitLabel).index + j);
231  //printf("EventID = %d\n", label.getEventID());
232  mClsLabels->addElement(lbl, label);
233  }
234  }
235  }
236 
237  // set geometrical variables
238  int det[5];
239  Geo::getVolumeIndices(c.getMainContributingChannel(), det);
240  float pos[3];
241  Geo::getPos(det, pos);
242  Geo::rotateToSector(pos, c.getSector());
243  c.setXYZ(pos[2], pos[0], pos[1]); // storing coordinates in sector frame: note that the rotation above puts z in pos[1], the radial coordinate in pos[2], and the tangent coordinate in pos[0] (this is to match the TOF residual system, where we don't use the radial component), so we swap their positions.
244 
245  c.setR(TMath::Sqrt(pos[0] * pos[0] + pos[1] * pos[1])); // it is the R in the sector frame
246  c.setPhi(TMath::ATan2(pos[1], pos[0]));
247 
248  float errY2 = Geo::XPAD * Geo::XPAD * inv12;
249  float errZ2 = Geo::ZPAD * Geo::ZPAD * inv12;
250  float errYZ = 0;
251 
252  //if(c.getNumOfContributingChannels() > 1){
253  // set errors according to a model not yet defined!
254  // TO DO
255  //}
256  c.setErrors(errY2, errZ2, errYZ);
257 
258  return;
259 }
