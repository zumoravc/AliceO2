143 TH1F* TRDCalPadStatus::makeHisto1D()
144 {
145  //
146  // Make 1D histo
147  //
148 
149  char name[1000];
150  snprintf(name, 1000, "%s Pad 1D", getTitle().c_str());
151  TH1F* his = new TH1F(name, name, 6, -0.5, 5.5);
152  his->GetXaxis()->SetBinLabel(1, "Good");
153  his->GetXaxis()->SetBinLabel(2, "Masked");
154  his->GetXaxis()->SetBinLabel(3, "PadBridgedLeft");
155  his->GetXaxis()->SetBinLabel(4, "PadBridgedRight");
156  his->GetXaxis()->SetBinLabel(5, "ReadSecond");
157  his->GetXaxis()->SetBinLabel(6, "NotConnected");
158 
159  for (int idet = 0; idet < kNdet; idet++) {
160  if (mROC[idet]) {
161  for (int ichannel = 0; ichannel < mROC[idet]->getNchannels(); ichannel++) {
162  int status = (int)mROC[idet]->getStatus(ichannel);
163  if (status == 2)
164  status = 1;
165  if (status == 4)
166  status = 2;
167  if (status == 8)
168  status = 3;
169  if (status == 16)
170  status = 4;
171  if (status == 32)
172  status = 5;
173  his->Fill(status);
174  }
175  }
176  }
177 
178  return his;
179 }

182 TH2F* TRDCalPadStatus::makeHisto2DSmPl(int sm, int pl)
183 {
184  //
185  // Make 2D graph
186  //
187 
188  gStyle->SetPalette(1);
189  TRDGeometry* trdGeo = new TRDGeometry();
190  const TRDPadPlane* padPlane0 = trdGeo->getPadPlane(pl, 0);
191  Double_t row0 = padPlane0->getRow0();
192  Double_t col0 = padPlane0->getCol0();
193 
194  char name[1000];
195  snprintf(name, 1000, "%s Pad 2D sm %d pl %d", getTitle().c_str(), sm, pl);
196  TH2F* his = new TH2F(name, name, 88, -TMath::Abs(row0), TMath::Abs(row0), 148, -TMath::Abs(col0), TMath::Abs(col0));
197 
198  // Where we begin
199  int offsetsmpl = 30 * sm + pl;
200 
201  for (int k = 0; k < kNcham; k++) {
202  int det = offsetsmpl + k * 6;
203  if (mROC[det]) {
204  TRDCalSingleChamberStatus* calRoc = mROC[det];
205  for (int icol = 0; icol < calRoc->getNcols(); icol++) {
206  for (int irow = 0; irow < calRoc->getNrows(); irow++) {
207  int binz = 0;
208  int kb = kNcham - 1 - k;
209  int krow = calRoc->getNrows() - 1 - irow;
210  int kcol = calRoc->getNcols() - 1 - icol;
211  if (kb > 2)
212  binz = 16 * (kb - 1) + 12 + krow + 1 + 2 * (kb + 1);
213  else
214  binz = 16 * kb + krow + 1 + 2 * (kb + 1);
215  int biny = kcol + 1 + 2;
216  Float_t value = calRoc->getStatus(icol, irow);
217  his->SetBinContent(binz, biny, value);
218  }
219  }
220  for (int icol = 1; icol < 147; icol++) {
221  for (int l = 0; l < 2; l++) {
222  int binz = 0;
223  int kb = kNcham - 1 - k;
224  if (kb > 2)
225  binz = 16 * (kb - 1) + 12 + 1 + 2 * (kb + 1) - (l + 1);
226  else
227  binz = 16 * kb + 1 + 2 * (kb + 1) - (l + 1);
228  his->SetBinContent(binz, icol, 50.0);
229  }
230  }
231  }
232  }
233  for (int icol = 1; icol < 147; icol++) {
234  his->SetBinContent(88, icol, 50.0);
235  his->SetBinContent(87, icol, 50.0);
236  }
237  for (int irow = 1; irow < 89; irow++) {
238  his->SetBinContent(irow, 1, 50.0);
239  his->SetBinContent(irow, 2, 50.0);
240  his->SetBinContent(irow, 147, 50.0);
241  his->SetBinContent(irow, 148, 50.0);
242  }
243 
244  his->SetXTitle("z (cm)");
245  his->SetYTitle("y (cm)");
246  his->SetMaximum(50);
247  his->SetMinimum(0.0);
248  his->SetStats(0);
249 
250  return his;
251 }
