40 std::vector<uint8_t> generateTestData(size_t nTF, size_t nDataInTF, size_t nColDataInEvent, size_t nLinks = o2::mid::crateparams::sNGBTs)
41 {
42  std::vector<o2::mid::ColumnData> colData;
43  colData.reserve(nColDataInEvent);
44  int maxNcols = 7;
45  int nDEs = nColDataInEvent / maxNcols;
46  int nColLast = nColDataInEvent % maxNcols;
47  if (nColLast > 0) {
48  ++nDEs;
49  }
50 
51  // Generate data
52  for (int ide = 0; ide < nDEs; ++ide) {
53  int nCol = (ide == nDEs - 1) ? nColLast : maxNcols;
54  auto rpcLine = o2::mid::detparams::getRPCLine(ide);
55  int firstCol = (rpcLine < 3 || rpcLine > 5) ? 0 : 1;
56  for (int icol = firstCol; icol < nCol; ++icol) {
57  colData.emplace_back(getColData(ide, icol, 0xFF00, 0xFF0));
58  }
59  }
60 
61  auto severity = fair::Logger::GetConsoleSeverity();
62  fair::Logger::SetConsoleSeverity(fair::Severity::WARNING);
63  std::string tmpFilename = "tmp_mid_raw.dat";
64  o2::mid::Encoder encoder;
65  encoder.init(tmpFilename.c_str());
66 
67  // Fill TF
68  for (size_t itf = 0; itf < nTF; ++itf) {
69  for (int ilocal = 0; ilocal < nDataInTF; ++ilocal) {
70  o2::InteractionRecord ir(ilocal, itf);
71  encoder.process(colData, ir, o2::mid::EventType::Standard);
72  }
73  }
74  encoder.finalize();
75 
76  o2::raw::RawFileReader rawReader;
77  rawReader.addFile(tmpFilename.c_str());
78  rawReader.init();
79  size_t nActiveLinks = rawReader.getNLinks() < nLinks ? rawReader.getNLinks() : nLinks;
80  std::vector<char> buffer;
81  for (size_t itf = 0; itf < rawReader.getNTimeFrames(); ++itf) {
82  rawReader.setNextTFToRead(itf);
83  for (size_t ilink = 0; ilink < nActiveLinks; ++ilink) {
84  auto& link = rawReader.getLink(ilink);
85  auto tfsz = link.getNextTFSize();
86  if (!tfsz) {
87  continue;
88  }
89  std::vector<char> linkBuffer(tfsz);
90  link.readNextTF(linkBuffer.data());
91  buffer.insert(buffer.end(), linkBuffer.begin(), linkBuffer.end());
92  }
93  }
94  fair::Logger::SetConsoleSeverity(severity);
95 
96  std::remove(tmpFilename.c_str());
97 
98  std::vector<uint8_t> data(buffer.size());
99  memcpy(data.data(), buffer.data(), buffer.size());
100 
101  return data;
102 }

123 static void BM_GBTDecoder(benchmark::State& state)
124 {
125  o2::mid::GBTUserLogicDecoder decoder;
126  decoder.init(0, false);
127 
128  int nTF = state.range(0);
129  int nEventPerTF = state.range(1);
130  int nFiredPerEvent = state.range(2);
131  double num{0};
132 
133  auto inputData = generateTestData(nTF, nEventPerTF, nFiredPerEvent, 1);
134 
135  for (auto _ : state) {
136  decoder.clear();
137  o2::framework::RawParser parser(inputData.data(), inputData.size());
138  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
139  if (it.size() == 0) {
140  continue;
141  }
142  auto* rdhPtr = it.template get_if<o2::header::RAWDataHeader>();
143  gsl::span<const uint8_t> payload(it.data(), it.size());
144  decoder.process(payload, *rdhPtr);
145  }
146  ++num;
147  }
148 
149  state.counters["num"] = benchmark::Counter(num, benchmark::Counter::kIsRate);
150 }
