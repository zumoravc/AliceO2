46 std::unique_ptr<VisualisationEvent> DataInterpreterTPC::interpretDataForType(TObject* data, EVisualisationDataType type)
47 {
48  TList* list = (TList*)data;
49 
50  // Int_t event = ((TVector2*)list->At(2))->X();
51  auto ret_event = std::make_unique<VisualisationEvent>(0, 0, 0, 0, "", 0);
52 
53  if (type == Clusters) {
54  TFile* clustFile = (TFile*)list->At(1);
55 
56  //Why cannot TPC clusters be read like other clusters?
57  auto reader = new tpc::ClusterNativeHelper::Reader();
58  reader->init(clustFile->GetName());
59  reader->read(0);
60 
61  auto access = std::make_unique<o2::tpc::ClusterNativeAccess>();
62  std::unique_ptr<tpc::ClusterNative[]> clusterBuffer;
63  tpc::MCLabelContainer clusterMCBuffer;
64 
65  reader->fillIndex(*access, clusterBuffer, clusterMCBuffer);
66 
67  const auto& mapper = tpc::Mapper::instance();
68  const auto& clusterRefs = access->clusters;
69 
70  for (int sector = 0; sector < o2::tpc::Constants::MAXSECTOR; sector++) {
71  for (int row = 0; row < o2::tpc::Constants::MAXGLOBALPADROW; row++) {
72  const auto& c = clusterRefs[sector][row];
73 
74  const auto pad = mapper.globalPadNumber(tpc::PadPos(row, c->getPad()));
75  const tpc::LocalPosition3D localXYZ(mapper.padCentre(pad).X(), mapper.padCentre(pad).Y(), c->getTime());
76  const auto globalXYZ = mapper.LocalToGlobal(localXYZ, sector);
77  double xyz[3] = {globalXYZ.X(), globalXYZ.Y(), globalXYZ.Z()};
78 
79  VisualisationCluster cluster(xyz);
80  ret_event->addCluster(cluster);
81  }
82  }
83  } else if (type == ESD) {
84  TFile* trackFile = (TFile*)list->At(0);
85 
86  TTree* tracks = (TTree*)trackFile->Get("tpcrec");
87 
88  //Read all tracks to a buffer
89  std::vector<tpc::TrackTPC>* trkArr = nullptr;
90  tracks->SetBranchAddress("TPCTracks", &trkArr);
91  tracks->GetEntry(0);
92 
93  TEveTrackList* trackList = new TEveTrackList("tracks");
94  trackList->IncDenyDestroy();
95  auto prop = trackList->GetPropagator();
96  prop->SetMagField(0.5);
97 
98  int first, last;
99  first = 0;
100  last = trkArr->size();
101 
102  gsl::span<tpc::TrackTPC> mTracks = gsl::make_span(&(*trkArr)[first], last - first);
103 
104  for (const auto& rec : mTracks) {
105  std::array<float, 3> p;
106  rec.getPxPyPzGlo(p);
107  TEveRecTrackD t;
108  t.fP = {p[0], p[1], p[2]};
109  t.fSign = (rec.getSign() < 0) ? -1 : 1;
110  TEveTrack* eve_track = new TEveTrack(&t, prop);
111  eve_track->MakeTrack();
112 
113  auto start = eve_track->GetLineStart();
114  auto end = eve_track->GetLineEnd();
115  double track_start[3] = {start.fX, start.fY, start.fZ};
116  double track_end[3] = {end.fX, end.fY, end.fZ};
117  double track_p[3] = {p[0], p[1], p[2]};
118 
119  VisualisationTrack track(rec.getSign(), 0.0, 0, 0, 0.0, 0.0, track_start, track_end, track_p, 0, 0.0, 0.0, 0.0, 0);
120 
121  for (Int_t i = 0; i < eve_track->GetN(); ++i) {
122  Float_t x, y, z;
123  eve_track->GetPoint(i, x, y, z);
124  track.addPolyPoint(x, y, z);
125  }
126  delete eve_track;
127 
128  ret_event->addTrack(track);
129  }
130  delete trackList;
131  }
132  return ret_event;
133 }
