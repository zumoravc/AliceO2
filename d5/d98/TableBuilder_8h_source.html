135  template <typename BuilderType, typename ITERATOR>
136  static arrow::Status append(BuilderType& builder, std::pair<ITERATOR, ITERATOR> ip)
137  {
138  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
139  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
140  // FIXME: for the moment we do not fill things.
141  auto status = builder->Append();
142  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(builder->value_builder());
143  return status & valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));

146  // Lists do not have UnsafeAppend so we need to use the slow path in any case.
147  template <typename BuilderType, typename ITERATOR>
148  static void unsafeAppend(BuilderType& builder, std::pair<ITERATOR, ITERATOR> ip)
149  {
150  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
151  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
152  // FIXME: for the moment we do not fill things.
153  auto status = builder->Append();
154  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(builder->value_builder());
155  status &= valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));
156  if (!status.ok()) {
157  throw std::runtime_error("Unable to append values");
158  }
159  return;
160  }

380  template <typename... ARGS>
381  void validate(std::vector<std::string> const& columnNames)
382  {
383  constexpr int nColumns = sizeof...(ARGS);
384  if (nColumns != columnNames.size()) {
385  throw std::runtime_error("Mismatching number of column types and names");
386  }
387  if (mBuilders != nullptr) {
388  throw std::runtime_error("TableBuilder::persist can only be invoked once per instance");
389  }
390  }
391 
392  template <typename... ARGS>
393  auto makeBuilders(std::vector<std::string> const& columnNames, size_t nRows)
394  {
395  mSchema = std::make_shared<arrow::Schema>(TableBuilderHelpers::makeFields<ARGS...>(columnNames));
396 
397  auto builders = new BuildersTuple<ARGS...>(BuilderMaker<ARGS>::make(mMemoryPool)...);
398  if (nRows != -1) {
399  auto seq = std::make_index_sequence<sizeof...(ARGS)>{};
400  TableBuilderHelpers::reserveAll(*builders, nRows, seq);
401  }
402  mBuilders = builders; // We store the builders
403  }
404 
405  template <typename... ARGS>
406  auto makeFinalizer()
407  {
408  mFinalizer = [schema = mSchema, &arrays = mArrays, builders = (BuildersTuple<ARGS...>*)mBuilders]() -> std::shared_ptr<arrow::Table> {
409  auto status = TableBuilderHelpers::finalize(arrays, *builders, std::make_index_sequence<sizeof...(ARGS)>{});
410  if (status == false) {
411  throw std::runtime_error("Unable to finalize");
412  }
413  return arrow::Table::Make(schema, arrays);
414  };
415  }

418  TableBuilder(arrow::MemoryPool* pool = arrow::default_memory_pool())
419  : mBuilders{nullptr},
420  mMemoryPool{pool}
421  {
422  }

426  template <typename... ARGS>
427  auto persist(std::vector<std::string> const& columnNames)
428  {
429  using args_pack_t = framework::pack<ARGS...>;
430  if constexpr (sizeof...(ARGS) == 1 &&
431  is_bounded_array<pack_element_t<0, args_pack_t>>::value == false &&
432  std::is_arithmetic_v<pack_element_t<0, args_pack_t>> == false) {
433  using objType_t = pack_element_t<0, framework::pack<ARGS...>>;
434  using argsPack_t = decltype(tuple_to_pack(framework::to_tuple(std::declval<objType_t>())));
435  auto persister = persistTuple(argsPack_t{}, columnNames);
436  return [persister = persister](unsigned int slot, objType_t const& obj) -> void {
437  auto t = to_tuple(obj);
438  persister(slot, t);
439  };
440  } else if constexpr (sizeof...(ARGS) == 1 &&
441  is_bounded_array<pack_element_t<0, args_pack_t>>::value == true) {
442  using objType_t = pack_element_t<0, framework::pack<ARGS...>>;
443  auto persister = persistTuple(framework::pack<objType_t>{}, columnNames);
444  // Callback used to fill the builders
445  return [persister = persister](unsigned int slot, typename BuilderMaker<objType_t>::FillType const& arg) -> void {
446  persister(slot, std::forward_as_tuple(arg));
447  };
448  } else if constexpr (sizeof...(ARGS) >= 1) {
449  auto persister = persistTuple(framework::pack<ARGS...>{}, columnNames);
450  // Callback used to fill the builders
451  return [persister = persister](unsigned int slot, typename BuilderMaker<ARGS>::FillType... args) -> void {
452  persister(slot, std::forward_as_tuple(args...));
453  };
454  } else {
455  static_assert(o2::framework::always_static_assert_v<ARGS...>, "Unmanaged case");
456  }
457  }
458 
460  template <typename... ARGS>
461  auto persistTuple(framework::pack<ARGS...>, std::vector<std::string> const& columnNames)
462  {
463  constexpr int nColumns = sizeof...(ARGS);
464  validate<ARGS...>(columnNames);
465  mArrays.resize(nColumns);
466  makeBuilders<ARGS...>(columnNames, 1000);
467  makeFinalizer<ARGS...>();
468 
469  // Callback used to fill the builders
470  using FillTuple = std::tuple<typename BuilderMaker<ARGS>::FillType...>;
471  return [builders = (BuildersTuple<ARGS...>*)mBuilders](unsigned int slot, FillTuple const& t) -> void {
472  auto status = TableBuilderHelpers::append(*builders, std::index_sequence_for<ARGS...>{}, t);
473  if (status == false) {
474  throw std::runtime_error("Unable to append");
475  }
476  };
477  }
