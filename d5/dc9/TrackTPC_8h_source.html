79  uint8_t& sectorIndex, uint8_t& rowIndex, uint32_t& clusterIndex) const
80  {
81  // data for given tracks starts at clinfo[ mClustersReference.getFirstEntry() ],
82  // 1st mClustersReference.getEntries() cluster indices are stored as uint32_t
83  // then sector indices as uint8_t, then row indices ar uin8_t
84 
85  // const uint32_t* clIndArr = &clinfo[ mClustersReference.getFirstEntry() ];
86  const uint32_t* clIndArr = reinterpret_cast<const uint32_t*>(&clinfo[mClustersReference.getFirstEntry()]); // TODO remove this trick
87  clusterIndex = clIndArr[nCluster];
88  const uint8_t* srIndexArr = reinterpret_cast<const uint8_t*>(clIndArr + mClustersReference.getEntries());
89  sectorIndex = srIndexArr[nCluster];
90  rowIndex = srIndexArr[nCluster + mClustersReference.getEntries()];
