35 double CaloRawFitterStandard::rawResponseFunction(double* x, double* par)
36 {
37  double signal = 0.;
38  double tau = par[2];
39  double n = par[3];
40  double ped = par[4];
41  double xx = (x[0] - par[1] + tau) / tau;
42 
43  if (xx <= 0)
44  signal = ped;
45  else
46  signal = ped + par[0] * TMath::Power(xx, n) * TMath::Exp(n * (1 - xx));
47 
48  return signal;
49 }

51 CaloFitResults CaloRawFitterStandard::evaluate(const std::vector<Bunch>& bunchlist,
52  std::optional<unsigned int> altrocfg1, std::optional<unsigned int> altrocfg2)
53 {
54 
55  float time = 0;
56  float amp = 0;
57  float chi2 = 0;
58  int ndf = 0;
59  bool fitDone = kFALSE;
60 
61  auto [nsamples, bunchIndex, ampEstimate,
62  maxADC, timeEstimate, pedEstimate, first, last] = preFitEvaluateSamples(bunchlist, altrocfg1, altrocfg2, mAmpCut);
63 
64  if (ampEstimate >= mAmpCut) {
65  time = timeEstimate;
66  int timebinOffset = bunchlist.at(bunchIndex).getStartTime() - (bunchlist.at(bunchIndex).getBunchLength() - 1);
67  amp = ampEstimate;
68 
69  if (nsamples > 1 && maxADC < constants::OVERFLOWCUT) {
70  std::tie(amp, time, chi2, fitDone) = fitRaw(first, last);
71  time += timebinOffset;
72  timeEstimate += timebinOffset;
73  ndf = nsamples - 2;
74  }
75  }
76  if (fitDone) {
77  float ampAsymm = (amp - ampEstimate) / (amp + ampEstimate);
78  float timeDiff = time - timeEstimate;
79 
80  if ((TMath::Abs(ampAsymm) > 0.1) || (TMath::Abs(timeDiff) > 2)) {
81  amp = ampEstimate;
82  time = timeEstimate;
83  fitDone = kFALSE;
84  }
85  }
86  if (amp >= mAmpCut) {
87  if (!fitDone) {
88  std::default_random_engine generator;
89  std::uniform_real_distribution<float> distribution(0.0, 1.0);
90  amp += (0.5 - distribution(generator));
91  }
92  time = time * constants::EMCAL_TIMESAMPLE;
93  time -= mL1Phase;
94 
95  return CaloFitResults(maxADC, pedEstimate, mAlgo, amp, time, (int)time, chi2, ndf);
96  }
97  return CaloFitResults(-1, -1);
98 }

100 std::tuple<float, float, float, bool> CaloRawFitterStandard::fitRaw(int firstTimeBin, int lastTimeBin) const
101 {
102 
103  float amp(0), time(0), chi2(0);
104  bool fitDone(false);
105 
106  int nsamples = lastTimeBin - firstTimeBin + 1;
107  fitDone = kFALSE;
108  if (nsamples < 3)
109  return std::make_tuple(amp, time, chi2, fitDone);
110 
111  TGraph gSig(nsamples);
112 
113  for (int i = 0; i < nsamples; i++) {
114  int timebin = firstTimeBin + i;
115  gSig.SetPoint(i, timebin, getReversed(timebin));
116  }
117 
118  TF1 signalF("signal", CaloRawFitterStandard::rawResponseFunction, 0, constants::EMCAL_MAXTIMEBINS, 5);
119 
120  signalF.SetParameters(10., 5., constants::TAU, constants::ORDER, 0.); //set all defaults once, just to be safe
121  signalF.SetParNames("amp", "t0", "tau", "N", "ped");
122  signalF.FixParameter(2, constants::TAU);
123  signalF.FixParameter(3, constants::ORDER);
124  signalF.FixParameter(4, 0);
125  signalF.SetParameter(1, time);
126  signalF.SetParameter(0, amp);
127  signalF.SetParLimits(0, 0.5 * amp, 2 * amp);
128  signalF.SetParLimits(1, time - 4, time + 4);
129 
130  int status = gSig.Fit(&signalF, "QROW"); // Note option 'W': equal errors on all points
131  if (status == 0) {
132  amp = signalF.GetParameter(0);
133  time = signalF.GetParameter(1);
134  chi2 = signalF.GetChisquare();
135  fitDone = kTRUE;
136  } else
137  fitDone = kFALSE;
138 
139  return std::make_tuple(amp, time, chi2, fitDone);
140 }
