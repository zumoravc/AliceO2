48 void InvertPcon(TGeoPcon* pcon)
49 {
50  //
51  // z -> -z
52  //
53  Int_t nz = pcon->GetNz();
54  Double_t* z = new Double_t[nz];
55  Double_t* rmin = new Double_t[nz];
56  Double_t* rmax = new Double_t[nz];
57 
58  Double_t* z0 = pcon->GetZ();
59  Double_t* rmin0 = pcon->GetRmin();
60  Double_t* rmax0 = pcon->GetRmax();
61 
62  for (Int_t i = 0; i < nz; i++) {
63  z[i] = z0[i];
64  rmin[i] = rmin0[i];
65  rmax[i] = rmax0[i];
66  }
67 
68  for (Int_t i = 0; i < nz; i++) {
69  Int_t j = nz - i - 1;
70  pcon->DefineSection(i, -z[j], rmin[j], rmax[j]);
71  }
72 
73  delete[] z;
74  delete[] rmin;
75  delete[] rmax;
76 }

86  Int_t nz = pcon->GetNz();
