25 void RawBuffer::readFromStream(std::istream& in, uint32_t payloadsize)
26 {
27  flush();
28  uint32_t word(0);
29  auto address = reinterpret_cast<char*>(&word);
30  int nbyte = 0;
31  while (nbyte < payloadsize) {
32  in.read(address, sizeof(word));
33  nbyte += sizeof(word);
34  if ((word & 0xFFFFFF) == 0x1d3082) {
35  LOG(ERROR) << "Found stop word" << std::endl;
36  // Termination word
37  // should normally not be decoded in case the payload size
38  // is determined correctly
39  break;
40  }
41  mDataWords[mNDataWords++] = word;
42  }
43 }

45 void RawBuffer::readFromMemoryBuffer(const gsl::span<const char> rawmemory)
46 {
47  flush();
48  auto address = reinterpret_cast<const uint32_t*>(rawmemory.data());
49  for (auto iword = 0; iword < rawmemory.size() / sizeof(uint32_t); iword++) {
50  if ((address[iword] & 0xFFF) == 0x082) {
51  // Termination word
52  // should normally not be decoded in case the payload size
53  // is determined correctly
54  break;
55  }
56  mDataWords[mNDataWords++] = address[iword];
57  }
58 }
