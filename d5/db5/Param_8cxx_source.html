184 void Param::IdealPosition(Int_t iCh, TGeoHMatrix* pMatrix)
185 {
186  // Construct ideal position matrix for a given chamber
187  // Arguments: iCh- chamber ID; pMatrix- pointer to precreated unity matrix where to store the results
188  // Returns: none
189  const double kAngHor = 19.5; // horizontal angle between chambers 19.5 grad
190  const double kAngVer = 20; // vertical angle between chambers 20 grad
191  const double kAngCom = 30; // common HMPID rotation with respect to x axis 30 grad
192  const double kTrans[3] = {490, 0, 0}; // center of the chamber is on window-gap surface
193  pMatrix->RotateY(90); // rotate around y since initial position is in XY plane -> now in YZ plane
194  pMatrix->SetTranslation(kTrans); // now plane in YZ is shifted along x
195  switch (iCh) {
196  case 0:
197  pMatrix->RotateY(kAngHor);
198  pMatrix->RotateZ(-kAngVer);
199  break; //right and down
200  case 1:
201  pMatrix->RotateZ(-kAngVer);
202  break; //down
203  case 2:
204  pMatrix->RotateY(kAngHor);
205  break; //right
206  case 3:
207  break; //no rotation
208  case 4:
209  pMatrix->RotateY(-kAngHor);
210  break; //left
211  case 5:
212  pMatrix->RotateZ(kAngVer);
213  break; //up
214  case 6:
215  pMatrix->RotateY(-kAngHor);
216  pMatrix->RotateZ(kAngVer);
217  break; //left and up
218  }
219  pMatrix->RotateZ(kAngCom); //apply common rotation in XY plane
220 }

273 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
274 double Param::Sigma2(double trkTheta, double trkPhi, double ckovTh, double ckovPh)
275 {
276  // Analithical calculation of total error (as a sum of localization, geometrical and chromatic errors)
277  // on Cerenkov angle for a given Cerenkov photon
278  // created by a given MIP. Fromulae according to CERN-EP-2000-058
279  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
280  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
281  // MIP beta
282  // Returns: absolute error on Cerenkov angle, [radians]
283 
284  TVector3 v(-999, -999, -999);
285  double trkBeta = 1. / (TMath::Cos(ckovTh) * GetRefIdx());
286 
287  if (trkBeta > 1)
288  trkBeta = 1; //protection against bad measured thetaCer
289  if (trkBeta < 0)
290  trkBeta = 0.0001; //
291 
292  v.SetX(SigLoc(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
293  v.SetY(SigGeom(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
294  v.SetZ(SigCrom(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
295 
296  return v.Mag2();
297 }
298 //++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
299 double Param::SigLoc(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
300 {
301  // Analitical calculation of localization error (due to finite segmentation of PC) on Cerenkov angle for a given
302  // Cerenkov photon
303  // created by a given MIP. Fromulae according to CERN-EP-2000-058
304  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
305  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
306  // MIP beta
307  // Returns: absolute error on Cerenkov angle, [radians]
308 
309  double phiDelta = phiC - trkPhi;
310 
311  double sint = TMath::Sin(trkTheta);
312  double cost = TMath::Cos(trkTheta);
313  double sinf = TMath::Sin(trkPhi);
314  double cosf = TMath::Cos(trkPhi);
315  double sinfd = TMath::Sin(phiDelta);
316  double cosfd = TMath::Cos(phiDelta);
317  double tantheta = TMath::Tan(thetaC);
318 
319  double alpha = cost - tantheta * cosfd * sint; // formula (11)
320  double k = 1. - GetRefIdx() * GetRefIdx() + alpha * alpha / (betaM * betaM); // formula (after 8 in the text)
321  if (k < 0)
322  return 1e10;
323  double mu = sint * sinf + tantheta * (cost * cosfd * sinf + sinfd * cosf); // formula (10)
324  double e = sint * cosf + tantheta * (cost * cosfd * cosf - sinfd * sinf); // formula (9)
325 
326  double kk = betaM * TMath::Sqrt(k) / (GapThick() * alpha); // formula (6) and (7)
327  // formula (6)
328  double dtdxc = kk * (k * (cosfd * cosf - cost * sinfd * sinf) - (alpha * mu / (betaM * betaM)) * sint * sinfd);
329  // formula (7) pag.4
330  double dtdyc = kk * (k * (cosfd * sinf + cost * sinfd * cosf) + (alpha * e / (betaM * betaM)) * sint * sinfd);
331 
332  double errX = 0.2, errY = 0.25; //end of page 7
333  return TMath::Sqrt(errX * errX * dtdxc * dtdxc + errY * errY * dtdyc * dtdyc);
334 }
335 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
336 double Param::SigCrom(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
337 {
338  // Analitical calculation of chromatic error (due to lack of knowledge of Cerenkov photon energy)
339  // on Cerenkov angle for a given Cerenkov photon
340  // created by a given MIP. Fromulae according to CERN-EP-2000-058
341  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
342  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
343  // MIP beta
344  // Returns: absolute error on Cerenkov angle, [radians]
345 
346  double phiDelta = phiC - trkPhi;
347 
348  double sint = TMath::Sin(trkTheta);
349  double cost = TMath::Cos(trkTheta);
350  double cosfd = TMath::Cos(phiDelta);
351  double tantheta = TMath::Tan(thetaC);
352 
353  double alpha = cost - tantheta * cosfd * sint; // formula (11)
354  double dtdn = cost * GetRefIdx() * betaM * betaM / (alpha * tantheta); // formula (12)
355 
356  // double f = 0.00928*(7.75-5.635)/TMath::Sqrt(12.);
357  double f = 0.0172 * (7.75 - 5.635) / TMath::Sqrt(24.);
358 
359  return f * dtdn;
360 } //SigCrom()
361 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
362 double Param::SigGeom(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
363 {
364  // Analitical calculation of geometric error (due to lack of knowledge of creation point in radiator)
365  // on Cerenkov angle for a given Cerenkov photon
366  // created by a given MIP. Formulae according to CERN-EP-2000-058
367  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
368  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
369  // MIP beta
370  // Returns: absolute error on Cerenkov angle, [radians]
371 
372  double phiDelta = phiC - trkPhi;
373 
374  double sint = TMath::Sin(trkTheta);
375  double cost = TMath::Cos(trkTheta);
376  double sinf = TMath::Sin(trkPhi);
377  double cosfd = TMath::Cos(phiDelta);
378  double costheta = TMath::Cos(thetaC);
379  double tantheta = TMath::Tan(thetaC);
380 
381  double alpha = cost - tantheta * cosfd * sint; // formula (11)
382 
383  double k = 1. - GetRefIdx() * GetRefIdx() + alpha * alpha / (betaM * betaM); // formula (after 8 in the text)
384  if (k < 0)
385  return 1e10;
386 
387  double eTr = 0.5 * RadThick() * betaM * TMath::Sqrt(k) / (GapThick() * alpha); // formula (14)
388  double lambda = (1. - sint * sinf) * (1. + sint * sinf); // formula (15)
389 
390  double c1 = 1. / (1. + eTr * k / (alpha * alpha * costheta * costheta)); // formula (13.a)
391  double c2 = betaM * TMath::Power(k, 1.5) * tantheta * lambda / (GapThick() * alpha * alpha); // formula (13.b)
392  double c3 = (1. + eTr * k * betaM * betaM) / ((1 + eTr) * alpha * alpha); // formula (13.c)
393  double c4 = TMath::Sqrt(k) * tantheta * (1 - lambda) / (GapThick() * betaM); // formula (13.d)
394  double dtdT = c1 * (c2 + c3 * c4);
395  double trErr = RadThick() / (TMath::Sqrt(12.) * cost);
396 
397  return trErr * dtdT;
398 } //SigGeom()

470 void Param::Lors2Pad(float x, float y, Int_t& pc, Int_t& px, Int_t& py)
471 {
472  // Check the pad of given position
473  // Arguments: x,y- position [cm] in LORS; pc,px,py- pad where to store the result
474  // Returns: none
475  pc = px = py = -1;
476  if (x > fgkMinPcX[0] && x < fgkMaxPcX[0]) {
477  pc = 0;
478  px = Int_t(x / SizePadX());
479  } //PC 0 or 2 or 4
480  else if (x > fgkMinPcX[1] && x < fgkMaxPcX[1]) {
481  pc = 1;
482  px = Int_t((x - fgkMinPcX[1]) / SizePadX());
483  } //PC 1 or 3 or 5
484  else
485  return;
486  if (y > fgkMinPcY[0] && y < fgkMaxPcY[0]) {
487  py = Int_t(y / SizePadY());
488  } //PC 0 or 1
489  else if (y > fgkMinPcY[2] && y < fgkMaxPcY[2]) {
490  pc += 2;
491  py = Int_t((y - fgkMinPcY[2]) / SizePadY());
492  } //PC 2 or 3
