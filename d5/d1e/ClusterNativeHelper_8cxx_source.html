24 void ClusterNativeHelper::convert(const char* fromFile, const char* toFile, const char* toTreeName)
25 {
26  Reader reader;
27  TreeWriter writer;
28  reader.init(fromFile);
29  writer.init(toFile, toTreeName);
30  size_t nEntries = reader.getTreeSize();
31  ClusterNativeAccess clusterIndex;
32  std::unique_ptr<ClusterNative[]> clusterBuffer;
33  MCLabelContainer mcBuffer;
34 
35  int result = 0;
36  int nClusters = 0;
37  for (size_t entry = 0; entry < nEntries; ++entry) {
38  LOG(INFO) << "converting entry " << entry;
39  reader.read(entry);
40  result = reader.fillIndex(clusterIndex, clusterBuffer, mcBuffer);
41  if (result >= 0) {
42  LOG(INFO) << "added " << result << " clusters to index";
43  } else {
44  LOG(ERROR) << "filling of clusters index failed with " << result;
45  }
46  result = writer.fillFrom(clusterIndex);
47  if (result >= 0) {
48  LOG(INFO) << "wrote " << result << " clusters to tree";
49  nClusters += result;
50  } else {
51  LOG(ERROR) << "filling of tree failed with " << result;
52  }
53  }
54  LOG(INFO) << "... done, converted " << nClusters << " clusters";
55  writer.close();
56 }

58 std::unique_ptr<ClusterNativeAccess> ClusterNativeHelper::createClusterNativeIndex(
59  std::unique_ptr<ClusterNative[]>& buffer, std::vector<ClusterNativeContainer>& clusters,
60  MCLabelContainer* bufferMC, std::vector<MCLabelContainer>* mcTruth)
61 {
62  std::unique_ptr<ClusterNativeAccess> retVal(new ClusterNativeAccess);
63  memset(retVal.get(), 0, sizeof(*retVal));
64  for (int i = 0; i < clusters.size(); i++) {
65  if (retVal->nClusters[clusters[i].sector][clusters[i].globalPadRow]) {
66  LOG(ERROR) << "Received two containers for the same sector / row";
67  return std::unique_ptr<ClusterNativeAccess>();
68  }
69  retVal->nClusters[clusters[i].sector][clusters[i].globalPadRow] = clusters[i].clusters.size();
70  retVal->nClustersTotal += clusters[i].clusters.size();
71  }
72  buffer.reset(new ClusterNative[retVal->nClustersTotal]);
73  if (bufferMC) {
74  bufferMC->clear();
75  }
76  retVal->clustersLinear = buffer.get();
77  retVal->setOffsetPtrs();
78  for (int i = 0; i < clusters.size(); i++) {
79  memcpy(&buffer[retVal->clusterOffset[clusters[i].sector][clusters[i].globalPadRow]], clusters[i].clusters.data(), sizeof(*retVal->clustersLinear) * clusters[i].clusters.size());
80  if (mcTruth) {
81  for (unsigned int j = 0; j < clusters[i].clusters.size(); j++) {
82  for (auto const& label : (*mcTruth)[i].getLabels(j)) {
83  bufferMC->addElement(retVal->clusterOffset[clusters[i].sector][clusters[i].globalPadRow] + j, label);
84  }
85  }
86  }
87  }
88  return (std::move(retVal));
89 }

102 void ClusterNativeHelper::Reader::init(const char* filename, const char* treename)
103 {
104  if (treename != nullptr && treename[0] != 0) {
105  mTreeName = treename;
106  }
107  mFile.reset(TFile::Open(filename));
108  if (!mFile) {
109  return;
110  }
111  mTree = reinterpret_cast<TTree*>(mFile->GetObjectUnchecked(mTreeName.c_str()));
112  if (!mTree) {
113  LOG(ERROR) << "can not find tree " << mTreeName << " in file " << filename;
114  return;
115  }
116  size_t nofDataBranches = 0;
117  size_t nofMCBranches = 0;
118  for (size_t sector = 0; sector < NSectors; ++sector) {
119  auto branchname = mDataBranchName + "_" + std::to_string(sector);
120  TBranch* branch = mTree->GetBranch(branchname.c_str());
121  if (branch) {
122  TBranch* sizebranch = mTree->GetBranch((branchname + "Size").c_str());
123  if (sizebranch) {
124  branch->SetAddress(&mSectorRaw[sector]);
125  sizebranch->SetAddress(&mSectorRawSize[sector]);
126  ++nofDataBranches;
127  } else {
128  LOG(ERROR) << "can not find corresponding 'Size' branch for data branch " << branchname << ", skipping it";
129  }
130  }
131  branchname = mMCBranchName + "_" + std::to_string(sector);
132  branch = mTree->GetBranch(branchname.c_str());
133  if (branch) {
134  mSectorMCPtr[sector] = &mSectorMC[sector];
135  branch->SetAddress(&mSectorMCPtr[sector]);
136  ++nofMCBranches;
137  }
138  }
139  LOG(INFO) << "reading " << nofDataBranches << " data branch(es) and " << nofMCBranches << " mc branch(es)";
140 }

179 int ClusterNativeHelper::Reader::parseSector(const char* buffer, size_t size, gsl::span<MCLabelContainer const> const& mcinput, ClusterNativeAccess& clusterIndex,
180  const MCLabelContainer* (&clustersMCTruth)[Constants::MAXSECTOR])
181 {
182  if (!buffer || size < sizeof(ClusterCountIndex)) {
183  return 0;
184  }
185 
186  auto mcIterator = mcinput.begin();
187  ClusterCountIndex const& counts = *reinterpret_cast<const ClusterCountIndex*>(buffer);
188  ClusterNative const* clusters = reinterpret_cast<ClusterNative const*>(buffer + sizeof(ClusterCountIndex));
189  size_t numberOfClusters = 0;
190  for (int i = 0; i < Constants::MAXSECTOR; i++) {
191  int nSectorClusters = 0;
192  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
193  if (counts.nClusters[i][j] == 0) {
194  continue;
195  }
196  nSectorClusters += counts.nClusters[i][j];
197  if ((numberOfClusters + counts.nClusters[i][j]) * sizeof(ClusterNative) + sizeof(ClusterCountIndex) > size) {
198  throw std::runtime_error("inconsistent buffer size");
199  }
200  clusterIndex.clusters[i][j] = clusters + numberOfClusters;
201  clusterIndex.nClusters[i][j] = counts.nClusters[i][j];
202  numberOfClusters += counts.nClusters[i][j];
203  }
204  if (nSectorClusters > 0) {
205  if (mcIterator != mcinput.end()) {
206  clustersMCTruth[i] = &(*mcIterator);
207  ++mcIterator;
208  if (mcIterator != mcinput.end()) {
209  throw std::runtime_error("can only have one MCLabel block per sector");
210  }
211  }
212  }
213  }
214 
215  return numberOfClusters;
216 }

243 int ClusterNativeHelper::TreeWriter::fillFrom(ClusterNativeAccess const& clusterIndex)
244 {
245  if (!mTree) {
246  return -1;
247  }
248  int result = 0;
249  for (size_t sector = 0; sector < Constants::MAXSECTOR; ++sector) {
250  for (size_t padrow = 0; padrow < Constants::MAXGLOBALPADROW; ++padrow) {
251  int locres = fillFrom(sector, padrow, clusterIndex.clusters[sector][padrow], clusterIndex.nClusters[sector][padrow]);
252  if (result >= 0 && locres >= 0) {
253  result += locres;
254  } else if (result >= 0) {
255  result = locres;
256  }
257  }
258  }
259  return result;
260 }
