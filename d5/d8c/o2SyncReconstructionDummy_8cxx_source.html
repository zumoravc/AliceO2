17 AlgorithmSpec simplePipe(std::string const& what)
18 {
19  return AlgorithmSpec{[what](InitContext& ic) {
20  auto delay = std::chrono::seconds(ic.options().get<int>("delay"));
21  auto messageSize = ic.options().get<int>("size");
22 
23  return [what, delay, messageSize](ProcessingContext& ctx) {
24  auto tStart = std::chrono::high_resolution_clock::now();
25  auto tEnd = std::chrono::high_resolution_clock::now();
26  auto& msg = ctx.outputs().make<char>(OutputRef{what}, messageSize);
27  while (std::chrono::duration_cast<std::chrono::seconds>(tEnd - tStart) < std::chrono::seconds(delay)) {
28  for (size_t i = 0; i < messageSize; ++i) {
29  msg[i] = 0;
30  }
31  tEnd = std::chrono::high_resolution_clock::now();
32  }
33  };
34  }};
35 }
