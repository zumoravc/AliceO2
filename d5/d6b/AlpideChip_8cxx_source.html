32 TGeoVolume* AlpideChip::createChip(const Double_t ychip,
33  const Double_t ysens,
34  const char* chipName,
35  const char* sensName,
36  const Bool_t dummy,
37  const TGeoManager* mgr)
38 {
39  //
40  // Creates the Alpide Chip
41  // Caller should then use TGeoVolume::SetName to proper set the volume name
42  //
43  // Input:
44  // ychip : the chip Y half dimensions
45  // ysens : the sensor half thickness
46  // chipName,sensName : default volume names (if not passed by caller)
47  // dummy : if true, creates a dummy air volume
48  // (for material budget studies)
49  // mgr : the GeoManager (used only to get the proper material)
50  //
51  // Output:
52  //
53  // Return:
54  //
55  // Created: 20 Oct 2017 Mario Sitta Ported from V3layer
56  //
57 
58  Double_t xchip, zchip;
59  Double_t ylen, ypos;
60 
61  xchip = 0.5 * SegmentationAlpide::SensorSizeRows;
62  zchip = 0.5 * SegmentationAlpide::SensorSizeCols;
63 
64  // First create all needed shapes
65  ylen = ysens;
66  if (ysens > ychip) {
67  LOG(WARNING) << "Sensor half thickness (" << ysens
68  << ") greater than chip half thickness (" << ychip
69  << "), setting equal";
70  ylen = ychip;
71  }
72 
73  // The chip
74  TGeoBBox* chip = new TGeoBBox(xchip, ychip, zchip);
75 
76  // The sensor
77  TGeoBBox* sensor = new TGeoBBox(xchip, ylen, zchip);
78 
79  // The metal layer
80  ylen = 0.5 * sMetalLayerThick;
81  TGeoBBox* metallay = new TGeoBBox(xchip, ylen, zchip);
82 
83  // We have all shapes: now create the real volumes
84  TGeoMedium* medSi = mgr->GetMedium("ALPIDE_SI$");
85  if (!medSi) {
86  Int_t fieldType;
87  Float_t maxField;
88  o2::base::Detector::initFieldTrackingParams(fieldType, maxField);
89  createMaterials(0, fieldType, maxField);
90  medSi = mgr->GetMedium("ALPIDE_SI$");
91  }
92  TGeoMedium* medAir = mgr->GetMedium("ALPIDE_AIR$");
93  TGeoMedium* medMetal = mgr->GetMedium("ALPIDE_METALSTACK$");
94 
95  TGeoMedium* medChip;
96 
97  if (dummy)
98  medChip = medAir;
99  else
100  medChip = medSi;
101 
102  TGeoVolume* chipVol = new TGeoVolume(chipName, chip, medChip);
103  chipVol->SetVisibility(kTRUE);
104  chipVol->SetLineColor(1);
105  chipVol->SetFillColor(chipVol->GetLineColor());
106  chipVol->SetFillStyle(4000); // 0% transparent
107 
108  TGeoVolume* sensVol = new TGeoVolume(sensName, sensor, medChip);
109  sensVol->SetVisibility(kTRUE);
110  sensVol->SetLineColor(8);
111  sensVol->SetLineWidth(1);
112  sensVol->SetFillColor(sensVol->GetLineColor());
113  sensVol->SetFillStyle(4000); // 0% transparent
114 
115  TGeoVolume* metalVol = new TGeoVolume("MetalStack", metallay, medMetal);
116  metalVol->SetVisibility(kTRUE);
117  metalVol->SetLineColor(1);
118  metalVol->SetLineWidth(1);
119  metalVol->SetFillColor(metalVol->GetLineColor());
120  metalVol->SetFillStyle(4000); // 0% transparent
121 
122  // Now build up the chip
123  ypos = chip->GetDY() - metallay->GetDY();
124  chipVol->AddNode(metalVol, 1, new TGeoTranslation(0., ypos, 0.));
125 
126  ypos -= (metallay->GetDY() + sensor->GetDY());
127  chipVol->AddNode(sensVol, 1, new TGeoTranslation(0., ypos, 0.));
128 
129  // Done, return the chip
130  return chipVol;
131 }

134 void AlpideChip::createMaterials(Int_t id, Int_t ifield, Float_t fieldm)
135 {
136  // create common materials with the MFT
137 
138  auto& mgr = o2::base::MaterialManager::Instance();
139 
140  Float_t tmaxfd = 0.1; // Degree
141  Float_t stemax = 1.0; // cm
142  Float_t deemax = 0.1; // Fraction of particle's energy 0<deemax<=1
143  Float_t epsil = 1.0E-4; // cm
144  Float_t stmin = 0.0; // cm "Default value used"
145 
146  Float_t tmaxfdSi = 0.1; // Degree
147  Float_t stemaxSi = 0.0075; // cm
148  Float_t deemaxSi = 0.1; // Fraction of particle's energy 0<deemax<=1
149  Float_t epsilSi = 1.0E-4; // cm
150  Float_t stminSi = 0.0; // cm "Default value used"
151 
152  Float_t tmaxfdAir = 0.1; // Degree
153  Float_t stemaxAir = 1.0; // cm
154  Float_t deemaxAir = 0.1; // Fraction of particle's energy 0<deemax<=1
155  Float_t epsilAir = 1.0E-4; // cm
156  Float_t stminAir = 0.0; // cm "Default value used"
157 
158  // BEOL (Metal interconnection stack in Si sensors)
159  Float_t aBEOL[3] = {26.982, 28.086, 15.999};
160  Float_t zBEOL[3] = {13, 14, 8}; // Al, Si, O
161  Float_t wBEOL[3] = {0.170, 0.388, 0.442};
162  Float_t dBEOL = 2.28;
163 
164  // AIR
165  Float_t aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
166  Float_t zAir[4] = {6., 7., 8., 18.};
167  Float_t wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
168  Float_t dAir = 1.20479E-3;
169 
170  if (mgr.getMediumID("ALPIDE", id) < 0) {
171  mgr.Mixture("ALPIDE", id, "METALSTACK$", aBEOL, zBEOL, dBEOL, 3, wBEOL);
172  mgr.Medium("ALPIDE", id, "METALSTACK$", id, 0, ifield, fieldm, tmaxfd, stemax, deemax, epsil, stmin);
173  id++;
174  }
175 
176  if (mgr.getMediumID("ALPIDE", id) < 0) {
177  mgr.Mixture("ALPIDE", id, "AIR$", aAir, zAir, dAir, 4, wAir);
178  mgr.Medium("ALPIDE", id, "AIR$", id, 0, ifield, fieldm, tmaxfdAir, stemaxAir, deemaxAir, epsilAir, stminAir);
179  id++;
180  }
181 
182  if (mgr.getMediumID("ALPIDE", id) < 0) {
183  mgr.Material("ALPIDE", id, "SI$", 0.28086E+02, 0.14000E+02, 0.23300E+01, 0.93600E+01, 0.99900E+03);
184  mgr.Medium("ALPIDE", id, "SI$", id, 0, ifield, fieldm, tmaxfdSi, stemaxSi, deemaxSi, epsilSi, stminSi);
185  }
186 }
