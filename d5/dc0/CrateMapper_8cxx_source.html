133 uint16_t CrateMapper::deLocalBoardToRO(uint8_t deId, uint8_t columnId, uint8_t lineId) const
134 {
136  auto item = mDEToROMap.find(deBoardId(detparams::getRPCLine(deId), columnId, lineId));
137  if (item == mDEToROMap.end()) {
138  std::stringstream ss;
139  ss << "Non-existant deId: " << static_cast<int>(deId) << " columnId: " << static_cast<int>(columnId) << " lineId: " << static_cast<int>(lineId);
140  throw std::runtime_error(ss.str());
141  }
142  return detparams::isRightSide(deId) ? item->second : item->second + (crateparams::sNCratesPerSide << 4);
143 }

145 uint16_t CrateMapper::roLocalBoardToDE(uint8_t crateId, uint8_t boardId) const
146 {
148  auto item = mROToDEMap.find(crateparams::makeUniqueLocID(crateId % crateparams::sNCratesPerSide, boardId));
149  if (item == mROToDEMap.end()) {
150  std::stringstream ss;
151  ss << "Non-existant crateId: " << static_cast<int>(crateId) << " boardId: " << static_cast<int>(boardId);
152  throw std::runtime_error(ss.str());
153  }
154  return item->second;
155 }
