198 int TPCFastTransformHelperO2::updateCalibration(TPCFastTransform& fastTransform, Long_t TimeStamp)
199 {
200  // Update the calibration with the new time stamp
201 
202  if (!mIsInitialized) {
203  init();
204  }
205 
206  Long_t lastTS = fastTransform.getTimeStamp();
207 
208  // deinitialize
209 
210  fastTransform.setTimeStamp(-1);
211 
212  if (TimeStamp < 0) {
213  return 0;
214  }
215 
216  // search for the calibration database ...
217 
218  auto& detParam = ParameterDetector::Instance();
219  auto& gasParam = ParameterGas::Instance();
220  auto& elParam = ParameterElectronics::Instance();
221 
222  // calibration found, set the initialized status back
223 
224  fastTransform.setTimeStamp(lastTS);
225 
226  // less than 60 seconds from the previois time stamp, don't do anything
227 
228  if (lastTS >= 0 && TMath::Abs(lastTS - TimeStamp) < 60) {
229  return 0;
230  }
231 
232  // start the initialization
233 
234  fastTransform.setTimeStamp(TimeStamp);
235 
236  // find last calibrated time bin
237 
238  const double vDrift = elParam.ZbinWidth * gasParam.DriftV; // cm/timebin
239 
240  //mLastTimeBin = detParam.getTPClength() / vDrift + 1;
241 
242  // fast transform formula:
243  // L = (t-t0)*(mVdrift + mVdriftCorrY*yLab ) + mLdriftCorr
244  // Z = Z(L) + tpcAlignmentZ
245  // spline corrections for xyz
246  // Time-of-flight correction: ldrift += dist-to-vtx*tofCorr
247 
248  const double t0 = elParam.PeakingTime / elParam.ZbinWidth;
249 
250  const double vdCorrY = 0.;
251  const double ldCorr = 0.;
252  const double tofCorr = 0.;
253  const double primVtxZ = 0.;
254 
255  fastTransform.setCalibration(TimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr, primVtxZ);
256 
257  // now calculate correction map: dx,du,dv = ( origTransform() -> x,u,v) - fastTransformNominal:x,u,v
258 
259  TPCFastSpaceChargeCorrection& correction = fastTransform.getCorrection();
260 
261  // for the future: switch TOF correction off for a while
262 
263  if (mSpaceChargeCorrection) {
264  for (int slice = 0; slice < correction.getGeometry().getNumberOfSlices(); slice++) {
265  for (int row = 0; row < correction.getGeometry().getNumberOfRows(); row++) {
266  const TPCFastSpaceChargeCorrection::SplineType& spline = correction.getSpline(slice, row);
267  float* data = correction.getSplineData(slice, row);
268  SplineHelper2D<float> helper;
269  helper.setSpline(spline, 3, 3);
270  auto F = [&](float su, float sv, float dxuv[3]) {
271  getSpaceChargeCorrection(slice, row, su, sv, dxuv[0], dxuv[1], dxuv[2]);
272  };
273  helper.approximateFunction(data, 0., 1., 0., 1., F);
274  } // row
275  } // slice
276  correction.initInverse();
277  } else {
278  correction.setNoCorrection();
279  }
280 
281  // for the future: set back the time-of-flight correction
282 
283  return 0;
284 }

286 int TPCFastTransformHelperO2::getSpaceChargeCorrection(int slice, int row, float su, float sv, float& dx, float& du, float& dv)
287 {
288  // get space charge correction in internal TPCFastTransform coordinates su,sv->dx,du,dv
289 
290  if (!mIsInitialized) {
291  init();
292  }
293 
294  dx = 0.f;
295  du = 0.f;
296  dv = 0.f;
297 
298  if (!mSpaceChargeCorrection) {
299  return 0;
300  }
301 
302  const TPCFastTransformGeo::RowInfo& rowInfo = mGeo.getRowInfo(row);
303 
304  float x = rowInfo.x;
305 
306  // x, u, v cordinates of the (su,sv) point (local cartesian coord. of slice towards central electrode )
307  float u = 0, v = 0;
308  mGeo.convScaledUVtoUV(slice, row, su, sv, u, v);
309 
310  // nominal x,y,z coordinates of the knot (without corrections and time-of-flight correction)
311  float y = 0, z = 0;
312  mGeo.convUVtoLocal(slice, u, v, y, z);
313 
314  // global coordinates of the knot
315  float gx, gy, gz;
316  mGeo.convLocalToGlobal(slice, x, y, z, gx, gy, gz);
317  float gx1 = gx, gy1 = gy, gz1 = gz;
318  {
319  double xyz[3] = {gx, gy, gz};
320  double dxyz[3] = {0., 0., 0.};
321  mSpaceChargeCorrection(slice, xyz, dxyz);
322  gx1 += dxyz[0];
323  gy1 += dxyz[1];
324  gz1 += dxyz[2];
325  }
326 
327  // corrections in the local coordinates
328  float x1, y1, z1;
329  mGeo.convGlobalToLocal(slice, gx1, gy1, gz1, x1, y1, z1);
330 
331  // correction corrections in u,v
332  float u1 = 0, v1 = 0;
333  mGeo.convLocalToUV(slice, y1, z1, u1, v1);
334 
335  dx = x1 - x;
336  du = u1 - u;
337  dv = v1 - v;
338 
339  return 0;
340 }

342 void TPCFastTransformHelperO2::testGeometry(const TPCFastTransformGeo& geo) const
343 {
344  const Mapper& mapper = Mapper::instance();
345 
346  if (geo.getNumberOfSlices() != Sector::MAXSECTOR) {
347  LOG(FATAL) << "Wrong number of sectors :" << geo.getNumberOfSlices() << " instead of " << Sector::MAXSECTOR << std::endl;
348  }
349 
350  if (geo.getNumberOfRows() != mapper.getNumberOfRows()) {
351  LOG(FATAL) << "Wrong number of rows :" << geo.getNumberOfRows() << " instead of " << mapper.getNumberOfRows() << std::endl;
352  }
353 
354  double maxDx = 0, maxDy = 0;
355 
356  for (int row = 0; row < geo.getNumberOfRows(); row++) {
357 
358  const int nPads = geo.getRowInfo(row).maxPad + 1;
359 
360  if (nPads != mapper.getNumberOfPadsInRowSector(row)) {
361  LOG(FATAL) << "Wrong number of pads :" << nPads << " instead of " << mapper.getNumberOfPadsInRowSector(row) << std::endl;
362  }
363 
364  const double x = geo.getRowInfo(row).x;
365 
366  // check if calculated pad positions are equal to the real ones
367 
368  for (int pad = 0; pad < nPads; pad++) {
369  const GlobalPadNumber p = mapper.globalPadNumber(PadPos(row, pad));
370  const PadCentre& c = mapper.padCentre(p);
371  float u = 0;
372  geo.convPadToU(row, pad, u);
373 
374  const double dx = x - c.X();
375  const double dy = u - (-c.Y()); // diferent sign convention for Y coordinate in the map
376 
377  if (fabs(dx) >= 1.e-6 || fabs(dy) >= 1.e-5) {
378  LOG(WARNING) << "wrong calculated pad position:"
379  << " row " << row << " pad " << pad << " x calc " << x << " x in map " << c.X() << " dx " << (x - c.X())
380  << " y calc " << u << " y in map " << -c.Y() << " dy " << dy << std::endl;
381  }
382  if (fabs(maxDx) < fabs(dx)) {
383  maxDx = dx;
384  }
385  if (fabs(maxDy) < fabs(dy)) {
386  maxDy = dy;
387  }
388  }
389  }
390 
391  if (fabs(maxDx) >= 1.e-4 || fabs(maxDy) >= 1.e-4) {
392  LOG(FATAL) << "wrong calculated pad position:"
393  << " max Dx " << maxDx << " max Dy " << maxDy << std::endl;
394  }
395 }
