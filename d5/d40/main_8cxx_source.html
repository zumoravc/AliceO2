43 Options* processCommandLine(int argc, char* argv[])
44 {
45  static Options options;
46  int opt;
47 
48  // put ':' in the starting of the
49  // string so that program can
50  //distinguish between '?' and ':'
51  while ((opt = getopt(argc, argv, ":if:rv")) != -1) {
52  switch (opt) {
53  case 'r':
54  options.randomTracks = true;
55  break;
56  case 'i':
57  options.itc = true;
58  break;
59  case 'v':
60  options.vsd = true;
61  break;
62  case 'f':
63  options.fileName = optarg;
64  break;
65  case ':':
66  printf("option needs a value\n");
67  return nullptr;
68  case '?':
69  printf("unknown option: %c\n", optopt);
70  return nullptr;
71  }
72  }
73 
74  // optind is for the extra arguments
75  // which are not parsed
76  for (; optind < argc; optind++) {
77  printf("extra arguments: %s\n", argv[optind]);
78  return nullptr;
79  }
80 
81  return &options;
82 }
