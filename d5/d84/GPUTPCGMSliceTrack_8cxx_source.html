97 GPUd() bool GPUTPCGMSliceTrack::FilterErrors(const GPUTPCGMMerger* merger, int iSlice, float maxSinPhi, float sinPhiMargin)
98 {
99  float lastX;
100  if (merger->Param().earlyTpcTransform && !merger->Param().rec.mergerReadFromTrackerDirectly) {
101  lastX = mOrigTrack->OutTrackCluster(mOrigTrack->NHits() - 1).GetX(); // TODO: Why is this needed, Row2X should work, but looses some tracks
102  } else {
103  //float lastX = merger->Param().tpcGeometry.Row2X(mOrigTrack->Cluster(mOrigTrack->NClusters() - 1).GetRow()); // TODO: again, why does this reduce efficiency?
104  float y, z;
105  const GPUTPCSliceOutCluster* clo;
106  int row, index;
107  if (merger->Param().rec.mergerReadFromTrackerDirectly) {
108  const GPUTPCTracker& trk = merger->GetConstantMem()->tpcTrackers[iSlice];
109  const GPUTPCHitId& ic = trk.TrackHits()[mOrigTrack->FirstHitID() + mOrigTrack->NHits() - 1];
110  index = trk.Data().ClusterDataIndex(trk.Data().Row(ic.RowIndex()), ic.HitIndex()) + merger->GetConstantMem()->ioPtrs.clustersNative->clusterOffset[iSlice][0];
111  row = ic.RowIndex();
112  } else {
113  clo = &mOrigTrack->OutTrackCluster(mOrigTrack->NHits() - 1);
114  index = clo->GetId();
115  row = clo->GetRow();
116  }
117  const ClusterNative& cl = merger->GetConstantMem()->ioPtrs.clustersNative->clustersLinear[index];
118  GPUTPCConvertImpl::convert(*merger->GetConstantMem(), iSlice, row, cl.getPad(), cl.getTime(), lastX, y, z);
119  }
120 
121  const int N = 3;
122 
123  float bz = -merger->Param().ConstBz;
124 
125  float k = mParam.mQPt * bz;
126  float dx = (1.f / N) * (lastX - mParam.mX);
127  float kdx = k * dx;
128  float dxBz = dx * bz;
129  float kdx205 = 2.f + kdx * kdx * 0.5f;
130 
131  {
132  merger->Param().GetClusterErrors2(0, mParam.mZ, mParam.mSinPhi, mParam.mDzDs, mParam.mC0, mParam.mC2);
133  float C0a, C2a;
134  merger->Param().GetClusterRMS2(0, mParam.mZ, mParam.mSinPhi, mParam.mDzDs, C0a, C2a);
135  if (C0a > mParam.mC0) {
136  mParam.mC0 = C0a;
137  }
138  if (C2a > mParam.mC2) {
139  mParam.mC2 = C2a;
140  }
141 
142  mParam.mC3 = 0;
143  mParam.mC5 = 1;
144  mParam.mC7 = 0;
145  mParam.mC9 = 1;
146  mParam.mC10 = 0;
147  mParam.mC12 = 0;
148  mParam.mC14 = 10;
149  }
150 
151  for (int iStep = 0; iStep < N; iStep++) {
152  float err2Y, err2Z;
153 
154  { // transport block
155  float ex = mParam.mCosPhi;
156  float ey = mParam.mSinPhi;
157  float ey1 = kdx + ey;
158  if (CAMath::Abs(ey1) > maxSinPhi) {
159  if (ey1 > maxSinPhi && ey1 < maxSinPhi + sinPhiMargin) {
160  ey1 = maxSinPhi - 0.01;
161  } else if (ey1 > -maxSinPhi - sinPhiMargin) {
162  ey1 = -maxSinPhi + 0.01;
163  } else {
164  return 0;
165  }
166  }
167 
168  float ss = ey + ey1;
169  float ex1 = sqrt(1.f - ey1 * ey1);
170 
171  float cc = ex + ex1;
172  float dxcci = dx / cc;
173 
174  float dy = dxcci * ss;
175  float norm2 = 1.f + ey * ey1 + ex * ex1;
176  float dl = dxcci * sqrt(norm2 + norm2);
177 
178  float dS;
179  {
180  float dSin = 0.5f * k * dl;
181  float a = dSin * dSin;
182  const float k2 = 1.f / 6.f;
183  const float k4 = 3.f / 40.f;
184  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
185  }
186 
187  float dz = dS * mParam.mDzDs;
188  float ex1i = 1.f / ex1;
189  {
190  merger->Param().GetClusterErrors2(0, mParam.mZ, mParam.mSinPhi, mParam.mDzDs, err2Y, err2Z);
191  float C0a, C2a;
192  merger->Param().GetClusterRMS2(0, mParam.mZ, mParam.mSinPhi, mParam.mDzDs, C0a, C2a);
193  if (C0a > err2Y) {
194  err2Y = C0a;
195  }
196  if (C2a > err2Z) {
197  err2Z = C2a;
198  }
199  }
200 
201  float hh = kdx205 * dxcci * ex1i;
202  float h2 = hh * mParam.mSecPhi;
203 
204  mParam.mX += dx;
205  mParam.mY += dy;
206  mParam.mZ += dz;
207  mParam.mSinPhi = ey1;
208  mParam.mCosPhi = ex1;
209  mParam.mSecPhi = ex1i;
210 
211  float h4 = bz * dxcci * hh;
212 
213  float c20 = mParam.mC3;
214  float c22 = mParam.mC5;
215  float c31 = mParam.mC7;
216  float c33 = mParam.mC9;
217  float c40 = mParam.mC10;
218  float c42 = mParam.mC12;
219  float c44 = mParam.mC14;
220 
221  float c20ph4c42 = c20 + h4 * c42;
222  float h2c22 = h2 * c22;
223  float h4c44 = h4 * c44;
224  float n7 = c31 + dS * c33;
225  float n10 = c40 + h2 * c42 + h4c44;
226  float n12 = c42 + dxBz * c44;
227 
228  mParam.mC0 += h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40);
229 
230  mParam.mC3 = c20ph4c42 + h2c22 + dxBz * n10;
231  mParam.mC10 = n10;
232 
233  mParam.mC5 = c22 + dxBz * (c42 + n12);
234  mParam.mC12 = n12;
235 
236  mParam.mC2 += dS * (c31 + n7);
237  mParam.mC7 = n7;
238  } // end transport block
239 
240  // Filter block
241 
242  float c00 = mParam.mC0, c11 = mParam.mC2, c20 = mParam.mC3, c31 = mParam.mC7, c40 = mParam.mC10;
243 
244  float mS0 = 1.f / (err2Y + c00);
245  float mS2 = 1.f / (err2Z + c11);
246 
247  // K = CHtS
248 
249  float k00, k11, k20, k31, k40;
250 
251  k00 = c00 * mS0;
252  k20 = c20 * mS0;
253  k40 = c40 * mS0;
254 
255  mParam.mC0 -= k00 * c00;
256  mParam.mC5 -= k20 * c20;
257  mParam.mC10 -= k00 * c40;
258  mParam.mC12 -= k40 * c20;
259  mParam.mC3 -= k20 * c00;
260  mParam.mC14 -= k40 * c40;
261 
262  k11 = c11 * mS2;
263  k31 = c31 * mS2;
264 
265  mParam.mC7 -= k31 * c11;
266  mParam.mC2 -= k11 * c11;
267  mParam.mC9 -= k31 * c31;
268  }
269 
270  //* Check that the track parameters and covariance matrix are reasonable
271 
272  bool ok = CAMath::Finite(mParam.mX) && CAMath::Finite(mParam.mY) && CAMath::Finite(mParam.mZ) && CAMath::Finite(mParam.mSinPhi) && CAMath::Finite(mParam.mDzDs) && CAMath::Finite(mParam.mQPt) && CAMath::Finite(mParam.mCosPhi) && CAMath::Finite(mParam.mSecPhi) && CAMath::Finite(mTZOffset) && CAMath::Finite(mParam.mC0) && CAMath::Finite(mParam.mC2) &&
273  CAMath::Finite(mParam.mC3) && CAMath::Finite(mParam.mC5) && CAMath::Finite(mParam.mC7) && CAMath::Finite(mParam.mC9) && CAMath::Finite(mParam.mC10) && CAMath::Finite(mParam.mC12) && CAMath::Finite(mParam.mC14);
274 
275  if (mParam.mC0 <= 0.f || mParam.mC2 <= 0.f || mParam.mC5 <= 0.f || mParam.mC9 <= 0.f || mParam.mC14 <= 0.f || mParam.mC0 > 5.f || mParam.mC2 > 5.f || mParam.mC5 > 2.f || mParam.mC9 > 2.f) {
276  ok = 0;
277  }
278 
279  if (ok) {
280  ok = ok && (mParam.mC3 * mParam.mC3 <= mParam.mC5 * mParam.mC0) && (mParam.mC7 * mParam.mC7 <= mParam.mC9 * mParam.mC2) && (mParam.mC10 * mParam.mC10 <= mParam.mC14 * mParam.mC0) && (mParam.mC12 * mParam.mC12 <= mParam.mC14 * mParam.mC5);
281  }
282 
283  return ok;
284 }

286 GPUd() bool GPUTPCGMSliceTrack::TransportToX(GPUTPCGMMerger* merger, float x, float Bz, GPUTPCGMBorderTrack& b, float maxSinPhi, bool doCov) const
287 {
288  Bz = -Bz;
289  float ex = mParam.mCosPhi;
290  float ey = mParam.mSinPhi;
291  float k = mParam.mQPt * Bz;
292  float dx = x - mParam.mX;
293  float ey1 = k * dx + ey;
294 
295  if (CAMath::Abs(ey1) > maxSinPhi) {
296  return 0;
297  }
298 
299  float ex1 = sqrt(1.f - ey1 * ey1);
300  float dxBz = dx * Bz;
301 
302  float ss = ey + ey1;
303  float cc = ex + ex1;
304  float dxcci = dx / cc;
305  float norm2 = 1.f + ey * ey1 + ex * ex1;
306 
307  float dy = dxcci * ss;
308 
309  float dS;
310  {
311  float dl = dxcci * sqrt(norm2 + norm2);
312  float dSin = 0.5f * k * dl;
313  float a = dSin * dSin;
314  const float k2 = 1.f / 6.f;
315  const float k4 = 3.f / 40.f;
316  // const float k6 = 5.f/112.f;
317  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
318  }
319 
320  float dz = dS * mParam.mDzDs;
321 
322  b.SetPar(0, mParam.mY + dy);
323  b.SetPar(1, mParam.mZ + dz);
324  b.SetPar(2, ey1);
325  b.SetPar(3, mParam.mDzDs);
326  b.SetPar(4, mParam.mQPt);
327  if (merger->Param().earlyTpcTransform) {
328  b.SetZOffsetLinear(mTZOffset);
329  } else {
330  b.SetZOffsetLinear(merger->GetConstantMem()->calibObjects.fastTransform->convVertexTimeToZOffset(mSlice, mTZOffset, merger->Param().continuousMaxTimeBin));
331  }
332 
333  if (!doCov) {
334  return (1);
335  }
336 
337  float ex1i = 1.f / ex1;
338  float hh = dxcci * ex1i * norm2;
339  float h2 = hh * mParam.mSecPhi;
340  float h4 = Bz * dxcci * hh;
341 
342  float c20 = mParam.mC3;
343  float c22 = mParam.mC5;
344  float c31 = mParam.mC7;
345  float c33 = mParam.mC9;
346  float c40 = mParam.mC10;
347  float c42 = mParam.mC12;
348  float c44 = mParam.mC14;
349 
350  float c20ph4c42 = c20 + h4 * c42;
351  float h2c22 = h2 * c22;
352  float h4c44 = h4 * c44;
353  float n7 = c31 + dS * c33;
354 
355  if (CAMath::Abs(mParam.mQPt) > 6.66) // Special treatment for low Pt
356  {
357  b.SetCov(0, CAMath::Max(mParam.mC0, mParam.mC0 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40))); // Do not decrease Y cov for matching!
358  float C2tmp = dS * 2.f * c31;
359  if (C2tmp < 0) {
360  C2tmp = 0;
361  }
362  b.SetCov(1, mParam.mC2 + C2tmp + dS * dS * c33); // Incorrect formula, correct would be "dS * (c31 + n7)", but we need to make sure cov(Z) increases regardless of the direction of the propagation
363  } else {
364  b.SetCov(0, mParam.mC0 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40));
365  b.SetCov(1, mParam.mC2 + dS * (c31 + n7));
366  }
367  b.SetCov(2, c22 + dxBz * (c42 + c42 + dxBz * c44));
368  b.SetCov(3, c33);
369  b.SetCov(4, c44);
370  b.SetCovD(0, c20ph4c42 + h2c22 + dxBz * (c40 + h2 * c42 + h4c44));
371  b.SetCovD(1, n7);
372 
373  b.LimitCov();
374 
375  return 1;
376 }

378 GPUd() bool GPUTPCGMSliceTrack::TransportToXAlpha(GPUTPCGMMerger* merger, float newX, float sinAlpha, float cosAlpha, float Bz, GPUTPCGMBorderTrack& b, float maxSinPhi) const
379 {
380  //*
381 
382  float c00 = mParam.mC0;
383  float c11 = mParam.mC2;
384  float c20 = mParam.mC3;
385  float c22 = mParam.mC5;
386  float c31 = mParam.mC7;
387  float c33 = mParam.mC9;
388  float c40 = mParam.mC10;
389  float c42 = mParam.mC12;
390  float c44 = mParam.mC14;
391 
392  float x, y;
393  float z = mParam.mZ;
394  float sinPhi = mParam.mSinPhi;
395  float cosPhi = mParam.mCosPhi;
396  float secPhi = mParam.mSecPhi;
397  float dzds = mParam.mDzDs;
398  float qpt = mParam.mQPt;
399 
400  // Rotate the coordinate system in XY on the angle alpha
401  {
402  float sP = sinPhi, cP = cosPhi;
403  cosPhi = cP * cosAlpha + sP * sinAlpha;
404  sinPhi = -cP * sinAlpha + sP * cosAlpha;
405 
406  if (CAMath::Abs(sinPhi) > GPUCA_MAX_SIN_PHI || CAMath::Abs(cP) < 1.e-2) {
407  return 0;
408  }
409 
410  secPhi = 1. / cosPhi;
411  float j0 = cP * secPhi;
412  float j2 = cosPhi / cP;
413  x = mParam.mX * cosAlpha + mParam.mY * sinAlpha;
414  y = -mParam.mX * sinAlpha + mParam.mY * cosAlpha;
415 
416  c00 *= j0 * j0;
417  c40 *= j0;
418 
419  c22 *= j2 * j2;
420  c42 *= j2;
421  if (cosPhi < 0.f) { // rotate to 180'
422  cosPhi = -cosPhi;
423  secPhi = -secPhi;
424  sinPhi = -sinPhi;
425  dzds = -dzds;
426  qpt = -qpt;
427  c20 = -c20;
428  c31 = -c31;
429  c40 = -c40;
430  }
431  }
432 
433  Bz = -Bz;
434  float ex = cosPhi;
435  float ey = sinPhi;
436  float k = qpt * Bz;
437  float dx = newX - x;
438  float ey1 = k * dx + ey;
439 
440  if (CAMath::Abs(ey1) > maxSinPhi) {
441  return 0;
442  }
443 
444  float ex1 = sqrt(1.f - ey1 * ey1);
445 
446  float dxBz = dx * Bz;
447 
448  float ss = ey + ey1;
449  float cc = ex + ex1;
450  float dxcci = dx / cc;
451  float norm2 = 1.f + ey * ey1 + ex * ex1;
452 
453  float dy = dxcci * ss;
454 
455  float dS;
456  {
457  float dl = dxcci * sqrt(norm2 + norm2);
458  float dSin = 0.5f * k * dl;
459  float a = dSin * dSin;
460  const float k2 = 1.f / 6.f;
461  const float k4 = 3.f / 40.f;
462  // const float k6 = 5.f/112.f;
463  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
464  }
465 
466  float ex1i = 1.f / ex1;
467  float dz = dS * dzds;
468 
469  float hh = dxcci * ex1i * norm2;
470  float h2 = hh * secPhi;
471  float h4 = Bz * dxcci * hh;
472 
473  float c20ph4c42 = c20 + h4 * c42;
474  float h2c22 = h2 * c22;
475  float h4c44 = h4 * c44;
476  float n7 = c31 + dS * c33;
477 
478  b.SetPar(0, y + dy);
479  b.SetPar(1, z + dz);
480  b.SetPar(2, ey1);
481  b.SetPar(3, dzds);
482  b.SetPar(4, qpt);
483  if (merger->Param().earlyTpcTransform) {
484  b.SetZOffsetLinear(mTZOffset);
485  } else {
486  b.SetZOffsetLinear(merger->GetConstantMem()->calibObjects.fastTransform->convVertexTimeToZOffset(mSlice, mTZOffset, merger->Param().continuousMaxTimeBin));
487  }
488 
489  b.SetCov(0, c00 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40));
490  b.SetCov(1, c11 + dS * (c31 + n7));
491  b.SetCov(2, c22 + dxBz * (c42 + c42 + dxBz * c44));
492  b.SetCov(3, c33);
493  b.SetCov(4, c44);
494  b.SetCovD(0, c20ph4c42 + h2c22 + dxBz * (c40 + h2 * c42 + h4c44));
495  b.SetCovD(1, n7);
496 
497  b.LimitCov();
498 
499  return 1;
500 }
