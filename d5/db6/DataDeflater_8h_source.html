116  template <typename ValueType, typename WriterT>
117  int writeRaw(ValueType value, uint16_t bitlength, WriterT writer)
118  {
119  auto bitsToWrite = bitlength;
120  if (bitlength > 8 * sizeof(ValueType)) {
121  // TODO: error policy
122  throw std::runtime_error("bit length exceeds width of the data type");
123  }
124  while (bitsToWrite > 0) {
125  if (mFilledBits == TargetBitWidth) {
126  mFilledBits = 0;
127  writer(mCurrent);
128  mCurrent = 0;
129  }
130  // write at max what is left to be written
131  auto writeNow = bitsToWrite;
132  // write one element of the target buffer at a time
133  if (writeNow > TargetBitWidth) {
134  writeNow = TargetBitWidth;
135  }
136  // write the remaining space in the current element
137  auto capacity = TargetBitWidth - mFilledBits;
138  if (writeNow > capacity) {
139  writeNow = capacity;
140  }
141  auto mask = (((ValueType)1 << writeNow) - 1) << (bitsToWrite - writeNow);
142  auto activebits = (value & mask) >> (bitsToWrite - writeNow);
143  mCurrent |= activebits << (capacity - writeNow);
144  mFilledBits += writeNow;
145  bitsToWrite -= writeNow;
146  assert(mFilledBits <= TargetBitWidth);
147  }
148  return bitlength - bitsToWrite;
149  }

154  using RegType = typename Codec::code_type;
155  return mCodec.write(
