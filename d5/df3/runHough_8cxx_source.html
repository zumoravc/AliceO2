66 int processData(std::string dataPath, std::string dataType, std::string dataOrigin)
67 {
68  // Open data file for reading
69  std::ifstream inputData(dataPath.c_str(), std::ifstream::binary);
70  if (!inputData) {
71  std::cerr << "Error, cluster data file " << dataPath << " could not be accessed" << endl;
72  std::exit(1);
73  }
74 
75  // Get length of file
76  inputData.seekg(0, inputData.end);
77  int dataLength = inputData.tellg();
78  inputData.seekg(0, inputData.beg);
79 
80  // Allocate memory and read file to memory
81  char* inputBuffer = new char[dataLength];
82  inputData.read(inputBuffer, dataLength);
83  inputData.close();
84 
85  // Retrieve the TPC slice and partition from the filename
86  std::string currentSliceString(dataPath, dataPath.length() - 6, 2);
87  std::string currentPartitionString(dataPath, dataPath.length() - 2, 2);
88 
89  AliHLTUInt8_t currentSlice = std::stoul(currentSliceString, nullptr, 16);
90  AliHLTUInt8_t currentPartition = std::stoul(currentPartitionString, nullptr, 16);
91 
92  // Initialize a cluster point collection
93  spacepoints = std::unique_ptr<AliHLTTPCSpacePointContainer>(new AliHLTTPCSpacePointContainer);
94  if (!spacepoints.get()) {
95  std::cerr << "Error, could not create a space point collection" << endl;
96  std::exit(1);
97  }
98 
99  // Create an AliHLTComponentBlockData object, fill it with default values and then set its pointer to the data buffer
100  AliHLTComponentBlockData bd;
101  AliHLTComponent::FillBlockData(bd);
102  bd.fPtr = inputBuffer;
103  bd.fSize = dataLength;
104  // bd.fDataType=kAliHLTVoidDataType;
105  AliHLTComponent::SetDataType(bd.fDataType, dataType.c_str(), dataOrigin.c_str());
106  bd.fSpecification = kAliHLTVoidDataSpec;
107 
108  // Set slice and partition
109  AliHLTTPCDefinitions::EncodeDataSpecification(currentSlice, currentSlice, currentPartition, currentPartition);
110 
111  // Add the AliHLTComponentBlockData object to AliHLTTPCSpacePointContainer
112  int numberOfClusters = spacepoints->AddInputBlock(&bd);
113 
114  // cout << *spacepoints << endl;
115 
116  // Retrieve the cluster information from AliHLTTPCSpacePointContainer
117  std::vector<AliHLTUInt32_t> clusterIDs;
118  spacepoints->GetClusterIDs(clusterIDs);
119 
120  // Append the cluster IDs and their X, Y and Z coordinates to the clusterCoordinates vector
121  for (vector<AliHLTUInt32_t>::const_iterator element = clusterIDs.begin(); element != clusterIDs.end(); element++) {
122  AliHLTUInt32_t clusterID = *element;
123 
124  addDataToCoordinatesVector(clusterID, spacepoints->GetX(clusterID), spacepoints->GetY(clusterID),
125  spacepoints->GetZ(clusterID));
126  }
127 
128  // De-allocate memory space
129  if (inputBuffer) {
130  delete[] inputBuffer;
131  }
132  inputBuffer = NULL;
133 
134  return numberOfClusters;
135 }

137 int main(int argc, char** argv)

148  boost::filesystem::path dataPath(dataFilename);
149  boost::filesystem::directory_iterator endIterator;
150 
151  typedef std::multimap<std::time_t, boost::filesystem::path> result_set_t;
152  result_set_t result_set;
153 
154  std::string dataType = "CLUSTERS", dataOrigin = "TPC ";
155 
156  int totalNumberOfClusters = 0, totalNumberOfDataFiles = 0;
157 
158  // Traverse the filesystem and execute processData for each cluster file found
159  if (boost::filesystem::exists(dataPath) && boost::filesystem::is_directory(dataPath)) {
160  for (boost::filesystem::directory_iterator directoryIterator(dataPath); directoryIterator != endIterator;
161  ++directoryIterator) {
162  if (boost::filesystem::is_regular_file(directoryIterator->status())) {
163  totalNumberOfClusters += processData(directoryIterator->path().string(), dataType, dataOrigin);
164  totalNumberOfDataFiles++;
165  }
166  }
167  } else {
168  std::cerr << "Path " << dataPath.string() << "/ could not be found or does not contain any valid data files"
169  << endl;
170  exit(1);
171  }
172 
173  cout << "Added " << totalNumberOfClusters << " clusters from " << totalNumberOfDataFiles << " data files" << endl;
174 
175  // printData(totalNumberOfClusters);
176 
177  drawData(totalNumberOfClusters, dataFilename);
178 
179  return 0;
180 }
