260 TGeoPcon* MakeShapeFromTemplate(const TGeoPcon* pcon, Float_t drMin, Float_t drMax)
261 {
262 
263  //
264  // Returns new shape based on a template changing
265  // the inner radii by drMin and the outer radii by drMax.
266  //
267  Int_t nz = pcon->GetNz();
268  TGeoPcon* cpcon = new TGeoPcon(0., 360., nz);
269  for (Int_t i = 0; i < nz; i++) {
270  cpcon->DefineSection(i, pcon->GetZ(i), pcon->GetRmin(i) + drMin, pcon->GetRmax(i) + drMax);
271  }
272  return cpcon;
273 }
