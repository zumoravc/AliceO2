111 template <typename T>
112 T* BasicCCDBManager::getForTimeStamp(std::string const& path, long timestamp)
113 {
114  if (!isCachingEnabled()) {
115  return mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp);
116  }
117  auto& cached = mCache[path];
118  T* ptr = mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp, &mHeaders, cached.uuid);
119  if (ptr) { // new object was shipped, old one (if any) is not valid anymore
120  cached.objPtr.reset(ptr);
121  cached.uuid = mHeaders["ETag"];
122  } else if (mHeaders.count("Error")) { // in case of errors the pointer is 0 and headers["Error"] should be set
123  clearCache(path); // in case of any error clear cache for this object
124  } else { // the old object is valid
125  ptr = reinterpret_cast<T*>(cached.objPtr.get());
126  }
127  mHeaders.clear();
128  return ptr;
129 }
