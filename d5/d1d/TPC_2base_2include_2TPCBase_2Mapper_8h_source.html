62  int getCRU(const Sector& sec, GlobalPadNumber globalPad) const
63  {
64  const auto row = mMapGlobalPadToPadPos[globalPad].getRow();
65  const auto nCRUPerSector = mMapPadRegionInfo.size();
66  int region = 0;
67  for (size_t i = 1; i < nCRUPerSector; ++i) {
68  if (row < mMapPadRegionInfo[i].getGlobalRowOffset()) {
69  break;
70  }
71  ++region;
72  }
73 
74  return int(sec * nCRUPerSector + region);
75  }

80  GlobalPadNumber getPadNumberInROC(const PadROCPos& rocPadPosition) const
81  {
82  const size_t padOffset = (rocPadPosition.getROCType() == RocType::IROC) ? 0 : mPadsInIROC;
83  const size_t rowOffset = (rocPadPosition.getROCType() == RocType::IROC) ? 0 : mNumberOfPadRowsIROC;
84 
85  return mMapPadOffsetPerRow[rocPadPosition.getRow() + rowOffset] + rocPadPosition.getPad() - padOffset;
86  }

91  GlobalPadNumber getPadNumberInPartition(int partition, int row, int pad) const
92  {
93  const auto& info = mMapPartitionInfo[partition % mMapPartitionInfo.size()];
94  const size_t rowOffset = info.getGlobalRowOffset();
95  const size_t padOffset = mMapPadOffsetPerRow[rowOffset];
96 
97  return mMapPadOffsetPerRow[row + rowOffset] + pad - padOffset;

103  GlobalPadNumber getPadNumberInRegion(CRU cru, int row, int pad) const
104  {

106  const size_t rowOffset = info.getGlobalRowOffset();

118  GlobalPadNumber getPadNumber(const PadSubset padSubset, const size_t padSubsetNumber, const int row,
119  const int pad) const
120  {
121  switch (padSubset) {
122  case PadSubset::ROC: {
123  return getPadNumberInROC(PadROCPos(padSubsetNumber, row, pad));
124  break;
125  }
126  case PadSubset::Partition: {
127  return getPadNumberInPartition(padSubsetNumber, row, pad);
128  break;
129  }
130  case PadSubset::Region: {
131  return getPadNumberInRegion(CRU(padSubsetNumber), row, pad);
132  break;
133  }
134  }
135  return 0;
136  }

233  const PadROCPos padROCPos(const CRU cru, const int fecInRegion, const int sampaOnFEC,

238  const GlobalPadNumber padNumber = globalPadNumber(fecInSector, sampaOnFEC, channelOnSAMPA);
239  const ROC roc = cru.roc();
240  PadROCPos pos(roc, padPos(padNumber));
241  if (roc.isOROC()) {

405  // ===| rotation functions |==================================================
406  static GlobalPosition3D LocalToGlobal(const LocalPosition3D& pos, const double alpha)
407  {
408  const double cs = std::cos(alpha), sn = std::sin(alpha);
409  return GlobalPosition3D(float(double(pos.X()) * cs - double(pos.Y()) * sn),
410  float(double(pos.X()) * sn + double(pos.Y() * cs)), pos.Z());
411  }
412 
413  static LocalPosition3D GlobalToLocal(const GlobalPosition3D& pos, const double alpha)
414  {
416  const double cs = std::cos(-alpha), sn = std::sin(-alpha);
417  return LocalPosition3D(float(double(pos.X()) * cs - double(pos.Y()) * sn),
418  float(double(pos.X()) * sn + double(pos.Y() * cs)), pos.Z());
419  }
420 
421  static GlobalPosition3D LocalToGlobal(const LocalPosition3D& pos, const Sector sec)
422  {
423  const double cs = CosinsPerSector[sec.getSector() % SECTORSPERSIDE],
424  sn = SinsPerSector[sec.getSector() % SECTORSPERSIDE];
425  return GlobalPosition3D(float(double(pos.X()) * cs - double(pos.Y()) * sn),
426  float(double(pos.X()) * sn + double(pos.Y() * cs)), pos.Z());
427  }

429  static LocalPosition3D GlobalToLocal(const GlobalPosition3D& pos, const Sector sec)
430  {
432  const double cs = CosinsPerSector[sec.getSector() % SECTORSPERSIDE],
433  sn = -SinsPerSector[sec.getSector() % SECTORSPERSIDE];
434  return LocalPosition3D(float(double(pos.X()) * cs - double(pos.Y()) * sn),
435  float(double(pos.X()) * sn + double(pos.Y() * cs)), pos.Z());
436  }

595 inline bool Mapper::isOutOfSector(GlobalPosition3D posEle, const Sector& sector, const float margin) const
596 {
597  int secRight = int(sector);
598  int secLeft = int(Sector::getLeft(sector));
599  const float dSectorBoundaryRight = -SinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.Y();
600  const float dSectorBoundaryLeft = -SinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.Y();
601 
602  if ((dSectorBoundaryLeft > 0 && dSectorBoundaryRight < 0) || (dSectorBoundaryLeft < 0 && dSectorBoundaryRight > 0)) {
603  return false;
604  }
605  if (std::abs(dSectorBoundaryLeft) > margin && std::abs(dSectorBoundaryRight) > margin) {
606  return true;
607  }
608  return false;
609 }
