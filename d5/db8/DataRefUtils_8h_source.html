33  template <typename T>
34  static auto as(DataRef const& ref)
35  {
36  // SFINAE makes this available only for the case we are using
37  // trivially copyable type, this is to distinguish it from the
38  // alternative below, which works for TObject (which are serialised).
39  if constexpr (is_messageable<T>::value == true) {
40  using DataHeader = o2::header::DataHeader;
41  auto header = o2::header::get<const DataHeader*>(ref.header);
42  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
43  throw std::runtime_error("Attempt to extract a POD from a wrong message kind");
44  }
45  if ((header->payloadSize % sizeof(T)) != 0) {
46  throw std::runtime_error("Cannot extract POD from message as size do not match");
47  }
48  //FIXME: provide a const collection
49  return gsl::span<T>(reinterpret_cast<T*>(const_cast<char*>(ref.payload)), header->payloadSize / sizeof(T));
50  } else if constexpr (has_root_dictionary<T>::value == true &&
51  is_messageable<T>::value == false) {
52  std::unique_ptr<T> result;
53  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
54  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
55  // Classes using the ROOT ClassDef/ClassImp macros can be detected automatically
56  // and the serialization method can be deduced. If the class is also
57  // messageable, gSerializationMethodNone is used by default. This substitution
58  // does not apply for such types, the serialization method needs to be specified
59  // explicitely using type wrapper @a ROOTSerialized.
60  using RSS = std::decay_t<decltype(*p)>;
61  using DataHeader = o2::header::DataHeader;
62  auto header = o2::header::get<const DataHeader*>(ref.header);
63  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
64  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
65  }
66 
67  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
68  auto* storedClass = ftm.GetClass();
69  auto* requestedClass = RSS::TClass::GetClass(typeid(T));
70  // should always have the class description if has_root_dictionary is true
71  assert(requestedClass != nullptr);
72 
73  auto* object = ftm.ReadObjectAny(storedClass);
74  if (object == nullptr) {
75  std::ostringstream ss;
76  ss << "Failed to read object with name "
77  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
78  << " from message using ROOT serialization.";
79  throw std::runtime_error(ss.str());
80  }
81 
82  if constexpr (std::is_base_of<typename RSS::TObject, T>::value) { // compile time switch
83  // if the type to be extracted inherits from TObject, the class descriptions
84  // do not need to match exactly, only the dynamic_cast has to work
85  // FIXME: could probably try to extend this to arbitrary types T being a base
86  // to the stored type, but this would require to cast the extracted object to
87  // the actual type. This requires this information to be available as a type
88  // in the ROOT dictionary, check if this is the case or if TClass::DynamicCast
89  // can be used for the test. Right now, this case was and is not covered and
90  // not yet checked in the unit test either.
91  auto* r = dynamic_cast<T*>(static_cast<typename RSS::TObject*>(object));
92  if (r) {
93  result.reset(r);
94  }
95  // This check includes the case: storedClass == requestedClass
96  } else if (storedClass->InheritsFrom(requestedClass)) {
97  result.reset(static_cast<T*>(object));
98  }
99  if (result == nullptr) {
100  // did not manage to cast the pointer to result
101  // delete object via the class info if available, not the case for all types,
102  // e.g. for standard containers of ROOT objects apparently this is not always
103  // the case
104  auto* delfunc = storedClass->GetDelete();
105  if (delfunc) {
106  (*delfunc)(object);
107  }
108 
109  std::ostringstream ss;
110  ss << "Attempting to extract a "
111  << (requestedClass != nullptr ? requestedClass->GetName() : "<unknown>")
112  << " but a "
113  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
114  << " is actually stored which cannot be casted to the requested one.";
115  throw std::runtime_error(ss.str());
116  }
117  // collections in ROOT can be non-owning or owning and the proper cleanup depends on
118  // this flag. Be it a bug or a feature in ROOT, but the owning flag of the extracted
119  // object only depends on the state at serialization of the original object. However,
120  // all objects created during deserialization are new and must be owned by the collection
121  // to avoid memory leak. So we call SetOwner if it is available for the type.
122  if constexpr (has_root_setowner<T>::value) {
123  result->SetOwner(true);
124  }
125  });
126 
127  return std::move(result);
128  } else if constexpr (is_specialization<T, ROOTSerialized>::value == true) {
129  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
130  // with a somewhat uniform API. ROOT serialization method is enforced by using
131  // type wrapper @a ROOTSerialized
132  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
133  using wrapped = typename T::wrapped_type;
134  using DataHeader = o2::header::DataHeader;
135  std::unique_ptr<wrapped> result;
136 
137  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
138  using RSS = std::decay_t<decltype(*p)>;
139 
140  auto header = o2::header::get<const DataHeader*>(ref.header);
141  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
142  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
143  }
144  auto* cl = RSS::TClass::GetClass(typeid(wrapped));
145  if (has_root_dictionary<wrapped>::value == false && cl == nullptr) {
146  throw std::runtime_error("ROOT serialization not supported, dictionary not found for data type");
147  }
148 
149  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
150  result.reset(static_cast<wrapped*>(ftm.ReadObjectAny(cl)));
151  if (result.get() == nullptr) {
152  std::ostringstream ss;
153  ss << "Unable to extract class ";
154  if (cl == nullptr) {
155  ss << "<name not available>";
156  } else {
157  ss << cl->GetName();
158  }
159  throw std::runtime_error(ss.str());
160  }
161  // workaround for ROOT feature, see above
162  if constexpr (has_root_setowner<T>::value) {
163  result->SetOwner(true);
164  }
165  });
166  return std::move(result);
167  }
168  }

170  static unsigned getPayloadSize(const DataRef& ref)
171  {
172  using DataHeader = o2::header::DataHeader;
173  auto* header = o2::header::get<const DataHeader*>(ref.header);
174  if (!header) {
175  return 0;
176  }
177  return header->payloadSize;
178  }

180  template <typename T>
181  static auto getHeader(const DataRef& ref)
182  {
183  using HeaderT = typename std::remove_pointer<T>::type;
184  static_assert(std::is_pointer<T>::value && std::is_base_of<o2::header::BaseHeader, HeaderT>::value,
185  "pointer to BaseHeader-derived type required");
186  return o2::header::get<T>(ref.header);
187  }

205  static bool match(DataRef const& ref, InputSpec const& spec)
206  {
207  auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
208  if (dh == nullptr) {
209  return false;
210  }
211  return DataSpecUtils::match(spec, dh->dataOrigin, dh->dataDescription, dh->subSpecification);
