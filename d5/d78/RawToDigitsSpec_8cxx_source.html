36 o2::framework::DataProcessorSpec getRawToDigitsSpec(int channel, const std::string_view inputDef, std::vector<int> const& tpcSectors)
37 {
38 
39  struct ProcessAttributes {
40  DigitDump digitDump;
41  rawreader::RawReaderCRUManager rawReader;
42  uint32_t lastOrbit{0};
43  uint32_t maxEvents{100};
44  uint64_t activeSectors{0};
45  bool quit{false};
46  std::vector<int> tpcSectors{};
47  };
48 
49  // ===| stateful initialization |=============================================
50  //
51  auto initFunction = [channel, tpcSectors](InitContext& ic) {
52  // ===| create and set up processing attributes |===
53  auto processAttributes = std::make_shared<ProcessAttributes>();
54  // set up calibration
55  {
56  auto& digitDump = processAttributes->digitDump;
57  digitDump.init();
58  digitDump.setInMemoryOnly();
59  const auto pedestalFile = ic.options().get<std::string>("pedestal-file");
60  LOGP(info, "Setting pedestal file: {}", pedestalFile);
61  digitDump.setPedestalAndNoiseFile(pedestalFile);
62 
63  processAttributes->rawReader.createReader("");
64  processAttributes->rawReader.setADCDataCallback([&digitDump](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> Int_t {
65  Int_t timeBins = digitDump.update(padROCPos, cru, data);
66  digitDump.setNumberOfProcessedTimeBins(std::max(digitDump.getNumberOfProcessedTimeBins(), size_t(timeBins)));
67  return timeBins;
68  });
69  processAttributes->maxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
70  processAttributes->tpcSectors = tpcSectors;
71  }
72 
73  // ===| data processor |====================================================
74  //
75  auto processingFct = [processAttributes, channel](ProcessingContext& pc) {
76  if (processAttributes->quit) {
77  return;
78  }
79 
80  // ===| digit snapshot |===
81  //
82  // lambda that snapshots digits to be sent out;
83  // prepares and attaches header with sector information
84  //
85  auto snapshotDigits = [&pc, processAttributes, channel](std::vector<o2::tpc::Digit> const& digits, int sector) {
86  o2::tpc::TPCSectorHeader header{sector};
87  header.activeSectors = processAttributes->activeSectors;
88  // digit for now are transported per sector, not per lane
89  // pc.outputs().snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe, header},
90  pc.outputs().snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(sector), Lifetime::Timeframe, header},
91  const_cast<std::vector<o2::tpc::Digit>&>(digits));
92  };
93 
94  // loop over all inputs
95  for (auto& input : pc.inputs()) {
96  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
97 
98  // select only RAW data
99  if (dh->dataDescription != o2::header::gDataDescriptionRawData) {
100  continue;
101  }
102 
103  // ===| extract electronics mapping information |===
104  const auto subSpecification = dh->subSpecification;
105  const auto cruID = subSpecification >> 16;
106  const auto linkID = ((subSpecification + (subSpecification >> 8)) & 0xFF) - 1;
107  const auto dataWrapperID = ((subSpecification >> 8) & 0xFF) > 0;
108  const auto globalLinkID = linkID + dataWrapperID * 12;
109  const auto sector = cruID / 10;
110 
111  // update active sectors
112  processAttributes->activeSectors |= (0x1 << sector);
113 
114  // set up mapping information for raw reader
115  auto& reader = processAttributes->rawReader.getReaders()[0];
116  reader->forceCRU(cruID);
117  reader->setLink(globalLinkID);
118 
119  LOGP(debug, "Specifier: {}/{}/{}", dh->dataOrigin.as<std::string>(), dh->dataDescription.as<std::string>(), dh->subSpecification);
120  LOGP(debug, "Payload size: {}", dh->payloadSize);
121  LOGP(debug, "CRU: {}; linkID: {}; dataWrapperID: {}; globalLinkID: {}", cruID, linkID, dataWrapperID, globalLinkID);
122 
123  try {
124  o2::framework::RawParser parser(input.payload, dh->payloadSize);
125 
126  rawreader::ADCRawData rawData;
127  rawreader::GBTFrame gFrame;
128 
129  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
130  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV4>();
131  if (!rdhPtr) {
132  break;
133  }
134  const auto& rdh = *rdhPtr;
135  //printRDH(rdh);
136 
137  // ===| event handling |===
138  //
139  // really ugly, better treatment requires extension in DPL
140  // events are are detected by close by orbit numbers
141  //
142  const auto hbOrbit = rdh.heartbeatOrbit;
143  const auto lastOrbit = processAttributes->lastOrbit;
144 
145  if ((lastOrbit > 0) && (hbOrbit > (lastOrbit + 3))) {
146  auto& digitDump = processAttributes->digitDump;
147  digitDump.incrementNEvents();
148  LOGP(info, "Number of processed events: {} ({})", digitDump.getNumberOfProcessedEvents(), processAttributes->maxEvents);
149  digitDump.sortDigits();
150 
151  // publish digits of all configured sectors
152  for (auto isector : processAttributes->tpcSectors) {
153  snapshotDigits(digitDump.getDigits(isector), isector);
154  }
155  digitDump.clearDigits();
156 
157  processAttributes->activeSectors = 0;
158  if (digitDump.getNumberOfProcessedEvents() >= processAttributes->maxEvents) {
159  LOGP(info, "Maximum number of events reached ({}), no more processing will be done", processAttributes->maxEvents);
160  processAttributes->quit = true;
161  pc.services().get<ControlService>().endOfStream();
162  //pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
163  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
164  break;
165  }
166  }
167 
168  processAttributes->lastOrbit = hbOrbit;
169  const auto size = it.size();
170  auto data = it.data();
171  LOGP(debug, "Raw data block payload size: {}", size);
172 
173  int iFrame = 0;
174  for (int i = 0; i < size; i += 16) {
175  gFrame.setFrameNumber(iFrame);
176  gFrame.setPacketNumber(iFrame / 508);
177  gFrame.readFromMemory(gsl::span<const o2::byte>(data + i, 16));
178 
179  // extract the half words from the 4 32-bit words
180  gFrame.getFrameHalfWords();
181 
182  // debug output
183  //if (CHECK_BIT(mDebugLevel, DebugLevel::GBTFrames)) {
184  //std::cout << gFrame;
185  //}
186 
187  gFrame.getAdcValues(rawData);
188  gFrame.updateSyncCheck(false);
189 
190  ++iFrame;
191  }
192  }
193 
194  reader->runADCDataCallback(rawData);
195  } catch (const std::runtime_error& e) {
196  LOG(ERROR) << "can not create raw parser form input data";
197  o2::header::hexDump("payload", input.payload, dh->payloadSize, 64);
198  LOG(ERROR) << e.what();
199  }
200  }
201  };
202 
203  return processingFct;
204  };
205 
206  std::stringstream id;
207  id << "TPCDigitizer" << channel;
208 
209  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
210  for (auto isector : tpcSectors) {
211  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
212  }
213 
214  return DataProcessorSpec{
215  id.str().c_str(),
216  select(inputDef.data()),
217  outputs,
218  AlgorithmSpec{initFunction},
219  Options{
220  {"max-events", VariantType::Int, 100, {"maximum number of events to process"}},
221  {"pedestal-file", VariantType::String, "", {"file with pedestals and noise for zero suppression"}}}};
222 }
