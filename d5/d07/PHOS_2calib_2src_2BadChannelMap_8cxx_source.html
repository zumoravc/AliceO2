47 void BadChannelMap::getHistogramRepresentation(char module, TH2* h) const
48 {
49  if (!h) {
50  LOG(ERROR) << "provide histogram to be filled";
51  }
52 
53  const char MAXX = 64,
54  MAXZ = 56;
55  if (h->GetNbinsX() != MAXX || h->GetNbinsY() != MAXZ) {
56  LOG(ERROR) << "Wrong dimentions of input histogram:" << h->GetNbinsX() << "," << h->GetNbinsY() << " instead of " << MAXX << "," << MAXZ;
57  return;
58  }
59 
60  h->Reset();
61  auto geo = Geometry::GetInstance();
62  if (!geo) {
63  LOG(ERROR) << "Geometry needs to be initialized";
64  return;
65  }
66 
67  char relid[3] = {module, 1, 1};
68  short absId;
69  for (char ix = 1; ix <= MAXX; ix++) {
70  relid[1] = ix;
71  for (char iz = 1; iz <= MAXZ; iz++) {
72  relid[2] = iz;
73  if (geo->relToAbsNumbering(relid, absId)) {
74  if (!isChannelGood(absId)) {
75  h->SetBinContent(ix, iz, 1);
76  }
77  }
78  }
79  }
80 }
