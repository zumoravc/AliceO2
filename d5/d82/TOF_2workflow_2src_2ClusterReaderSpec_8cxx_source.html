35 void ClusterReader::run(ProcessingContext& pc)
36 {
37  auto ent = mTree->GetReadEntry() + 1;
38  assert(ent < mTree->GetEntries()); // this should not happen
39  mTree->GetEntry(ent);
40  LOG(INFO) << "Pushing " << mClustersPtr->size() << " TOF clusters at entry " << ent;
41 
42  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe}, mClusters);
43  if (mUseMC) {
44  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mLabels);
45  }
46 
47  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
48  pc.services().get<ControlService>().endOfStream();
49  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
50  }
51 }

53 void ClusterReader::connectTree(const std::string& filename)
54 {
55  mTree.reset(nullptr); // in case it was already loaded
56  mFile.reset(TFile::Open(filename.c_str()));
57  assert(mFile && !mFile->IsZombie());
58  mTree.reset((TTree*)mFile->Get("o2sim"));
59  assert(mTree);
60  mTree->SetBranchAddress("TOFCluster", &mClustersPtr);
61  if (mUseMC) {
62  mTree->SetBranchAddress("TOFClusterMCTruth", &mLabelsPtr);
63  }
64  LOG(INFO) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
65 }
