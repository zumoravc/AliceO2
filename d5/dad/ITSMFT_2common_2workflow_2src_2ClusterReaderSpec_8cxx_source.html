47 void ClusterReader::run(ProcessingContext& pc)
48 {
49  auto ent = mTree->GetReadEntry() + 1;
50  assert(ent < mTree->GetEntries()); // this should not happen
51  mTree->GetEntry(ent);
52  LOG(INFO) << mDetName << "ClusterReader pushes " << mClusROFRec.size() << " ROFRecords,"
53  << mClusterArray.size() << " full clusters, " << mClusterCompArray.size()
54  << " compact clusters at entry " << ent;
55 
56  // This is a very ugly way of providing DataDescription, which anyway does not need to contain detector name.
57  // To be fixed once the names-definition class is ready
58  pc.outputs().snapshot(Output{mOrigin, mOrigin == o2::header::gDataOriginITS ? "ITSClusterROF" : "MFTClusterROF",
59  0, Lifetime::Timeframe},
60  mClusROFRec);
61  if (mUseClFull) {
62  pc.outputs().snapshot(Output{mOrigin, "CLUSTERS", 0, Lifetime::Timeframe}, mClusterArray);
63  }
64  if (mUseClComp) {
65  pc.outputs().snapshot(Output{mOrigin, "COMPCLUSTERS", 0, Lifetime::Timeframe}, mClusterCompArray);
66  }
67  if (mUsePatterns) {
68  pc.outputs().snapshot(Output{mOrigin, "PATTERNS", 0, Lifetime::Timeframe}, mPatternsArray);
69  }
70  if (mUseMC) {
71  pc.outputs().snapshot(Output{mOrigin, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mClusterMCTruth);
72  pc.outputs().snapshot(Output{mOrigin, "ClusterMC2ROF", 0, Lifetime::Timeframe}, mClusMC2ROFs);
73  }
74 
75  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
76  pc.services().get<ControlService>().endOfStream();
77  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
78  }
79 }

81 void ClusterReader::connectTree(const std::string& filename)
82 {
83  mTree.reset(nullptr); // in case it was already loaded
84  mFile.reset(TFile::Open(filename.c_str()));
85  assert(mFile && !mFile->IsZombie());
86  mTree.reset((TTree*)mFile->Get(mClusTreeName.c_str()));
87  assert(mTree);
88 
89  mTree->SetBranchAddress((mDetName + mClusROFBranchName).c_str(), &mClusROFRecPtr);
90  if (mUseClFull) {
91  mTree->SetBranchAddress((mDetName + mClusterBranchName).c_str(), &mClusterArrayPtr); // being eliminated!!!
92  }
93  if (mUseClComp) {
94  mTree->SetBranchAddress((mDetName + mClusterCompBranchName).c_str(), &mClusterCompArrayPtr);
95  }
96  if (mUsePatterns) {
97  mTree->SetBranchAddress((mDetName + mClusterPattBranchName).c_str(), &mPatternsArrayPtr);
98  }
99  if (mUseMC) {
100  if (mTree->GetBranch((mDetName + mClustMCTruthBranchName).c_str()) &&
101  mTree->GetBranch((mDetName + mClustMC2ROFBranchName).c_str())) {
102  mTree->SetBranchAddress((mDetName + mClustMCTruthBranchName).c_str(), &mClusterMCTruthPtr);
103  mTree->SetBranchAddress((mDetName + mClustMC2ROFBranchName).c_str(), &mClusMC2ROFsPtr);
104  } else {
105  LOG(INFO) << "MC-truth is missing";
106  mUseMC = false;
107  }
108  }
109  LOG(INFO) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
110 }
