47 void ClusterWriter::run(ProcessingContext& pc)
48 {
49  if (mFinished) {
50  return;
51  }
52 
53  TFile outf(mOutFileName.c_str(), "recreate");
54  if (outf.IsZombie()) {
55  LOG(FATAL) << "Failed to open output file " << mOutFileName;
56  }
57  TTree tree(mOutTreeName.c_str(), "Tree of TOF clusters");
58  auto indata = pc.inputs().get<std::vector<o2::tof::Cluster>>("tofclusters");
59  LOG(INFO) << "RECEIVED CLUSTERS SIZE " << indata.size();
60 
61  auto br = getOrMakeBranch(tree, "TOFCluster", &indata);
62  br->Fill();
63 
64  if (mUseMC) {
65  auto labeldata = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("tofclusterlabels");
66  LOG(INFO) << "TOF GOT " << labeldata->getNElements() << " LABELS ";
67  auto labeldataraw = labeldata.get();
68  // connect this to a particular branch
69 
70  auto labelbr = getOrMakeBranch(tree, "TOFClusterMCTruth", &labeldataraw);
71  labelbr->Fill();
72  }
73 
74  tree.SetEntries(1);
75  tree.Write();
76  mFinished = true;
77  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
78 }

80 DataProcessorSpec getTOFClusterWriterSpec(bool useMC)
81 {
82  std::vector<InputSpec> inputs;
83  inputs.emplace_back("tofclusters", o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe);
84  if (useMC)
85  inputs.emplace_back("tofclusterlabels", o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe);
86 
87  return DataProcessorSpec{
88  "TOFClusterWriter",
89  inputs,
90  {}, // no output
91  AlgorithmSpec{adaptFromTask<ClusterWriter>(useMC)},
92  Options{
93  {"tof-cluster-outfile", VariantType::String, "tofclusters.root", {"Name of the input file"}},
94  {"treename", VariantType::String, "o2sim", {"Name of top-level TTree"}},
95  }};
96 }
