61 void IrregularSpline2D3DCalibrator::startCalibration(std::function<void(float, float, float&, float&, float&)> F)
62 {
63  // initialize everything for the calibration
64 
65  // fill the raster data
66  mRasterData.resize(mRaster.getNumberOfKnots() * 3);
67 
68  for (int i = 0; i < mRaster.getNumberOfKnots(); ++i) {
69  float u = 0, v = 0, fx = 0, fy = 0, fz = 0;
70  mRaster.getKnotUV(i, u, v);
71  F(u, v, fx, fy, fz);
72  mRasterData[3 * i + 0] = fx;
73  mRasterData[3 * i + 1] = fy;
74  mRasterData[3 * i + 2] = fz;
75  }
76 
77  mRaster.correctEdges(mRasterData.data());
78 
79  // create current spline
80  for (int uv = 0; uv < 2; ++uv) {
81  //std::cout<<"n Raster knots: "<<mRaster.getGrid(uv).getNumberOfKnots()<<std::endl;
82  mKnots[uv].clear();
83  double du = 1. / (mMaxNKnots[uv] - 1);
84  int lastKnot = 0;
85  for (int i = 1; i < mMaxNKnots[uv] - 1; ++i) {
86  KnotData d;
87  d.uv = uv;
88  double u = i * du;
89  d.rasterKnot = nearbyint(u * (mRaster.getGrid(uv).getNumberOfKnots() - 1));
90  //std::cout<<"uv "<<uv<<" i "<<d.rasterKnot<<" u "<<u<<std::endl;
91  if (d.rasterKnot <= lastKnot)
92  continue;
93  if (d.rasterKnot >= mRaster.getGrid(uv).getNumberOfKnots() - 1)
94  continue;
95  mKnots[uv].push_back(d);
96  lastKnot = d.rasterKnot;
97  }
98  }
99 
100  createCurrentSpline();
101 }

113 void IrregularSpline2D3DCalibrator::createSpline(IrregularSpline2D3D& sp, std::vector<float>& data)
114 {
115  // recreate a spline with respect to knots in mKnots[] lists
116 
117  for (int uv = 0; uv < 2; ++uv) {
118  mTemp[uv].reserve(mMaxNKnots[uv]);
119  mTemp[uv].clear();
120  mTemp[uv].push_back(0.f);
121  for (std::list<KnotData>::iterator i = mKnots[uv].begin(); i != mKnots[uv].end(); ++i) {
122  //std::cout<<"uv "<<uv<<" i "<<i->rasterKnot<<" u "<<mRaster.getGrid(uv).getKnot(i->rasterKnot).u<<std::endl;
123  mTemp[uv].push_back(mRaster.getGrid(uv).getKnot(i->rasterKnot).u);
124  }
125  mTemp[uv].push_back(1.f);
126  }
127 
128  sp.construct(mTemp[0].size(), mTemp[0].data(), mRaster.getGrid(0).getNumberOfKnots(),
129  mTemp[1].size(), mTemp[1].data(), mRaster.getGrid(1).getNumberOfKnots());
130 
131  data.resize(sp.getNumberOfKnots() * 3);
132  for (int i = 0; i < sp.getNumberOfKnots(); ++i) {
133  float u = 0, v = 0, fx = 0, fy = 0, fz = 0;
134  sp.getKnotUV(i, u, v);
135  mRaster.getSplineVec(mRasterData.data(), u, v, fx, fy, fz);
136  data[3 * i + 0] = fx;
137  data[3 * i + 1] = fy;
138  data[3 * i + 2] = fz;
139  }
140  sp.correctEdges(data.data());
141 }

143 IrregularSpline2D3DCalibrator::Action IrregularSpline2D3DCalibrator::checkActionShift(std::list<KnotData>::iterator& knot)
144 {
145  // estimate the cost change when the knot i is shifted up or down
146 
147  Action ret;
148  ret.action = Action::Move::No;
149  ret.cost = mMaxDeviation + 1.e10;
150  ret.iter = knot;
151 
152  int uv = knot->uv;
153 
154  bool isSpaceUp = 0;
155 
156  if (knot->rasterKnot < mRaster.getGrid(uv).getNumberOfKnots() - 2) {
157  isSpaceUp = 1;
158  std::list<KnotData>::iterator next = knot;
159  ++next;
160  if (next != mKnots[uv].end()) {
161  if (next->rasterKnot <= knot->rasterKnot + 1)
162  isSpaceUp = 0;
163  }
164  }
165 
166  bool isSpaceDn = 0;
167 
168  if (knot->rasterKnot > 1) {
169  isSpaceDn = 1;
170  std::list<KnotData>::iterator prev = knot;
171  if (prev != mKnots[uv].begin()) {
172  --prev;
173  if (prev->rasterKnot >= knot->rasterKnot - 1)
174  isSpaceDn = 0;
175  }
176  }
177 
178  if (!isSpaceUp && !isSpaceDn)
179  return ret;
180 
181  // get the area of interest
182 
183  int regionKnotFirst = knot->rasterKnot;
184  int regionKnotLast = knot->rasterKnot;
185  getRegionOfInfluence(knot, regionKnotFirst, regionKnotLast);
186 
187  // get the current cost
188 
189  double costCurrent = getIntegralDeviationArea(mSpline, mSplineData, uv, regionKnotFirst, regionKnotLast);
190 
191  // get the cost when moving up
192 
193  if (isSpaceUp) {
194  knot->rasterKnot++;
195  createActionSpline();
196  knot->rasterKnot--;
197  ret.action = Action::Move::Up;
198  ret.cost = getIntegralDeviationArea(mActionSpline, mActionSplineData, uv, regionKnotFirst, regionKnotLast) - costCurrent;
199  }
200 
201  if (isSpaceDn) {
202  knot->rasterKnot--;
203  createActionSpline();
204  knot->rasterKnot++;
205  double costDn = getIntegralDeviationArea(mActionSpline, mActionSplineData, uv, regionKnotFirst, regionKnotLast) - costCurrent;
206  if (costDn < ret.cost) {
207  ret.action = Action::Move::Down;
208  ret.cost = costDn;
209  }
210  }
211  //if( ret.cost<0 ) std::cout<<"knot "<<knot->rasterKnot<<" area: "<<regionKnotFirst<<"<->"<<regionKnotLast<<" costCurrent "<<costCurrent<<std::endl;
212 
213  return ret;
214 }

216 IrregularSpline2D3DCalibrator::Action IrregularSpline2D3DCalibrator::checkActionRemove(std::list<KnotData>::iterator& knot)
217 {
218  // estimate the cost change when the knot i is shifted up or down
219 
220  Action ret;
221  ret.action = Action::Move::No;
222  ret.cost = mMaxDeviation + 1.e10;
223  ret.iter = knot;
224 
225  int uv = knot->uv;
226 
227  if (mSpline.getGrid(uv).getNumberOfKnots() <= 5)
228  return ret;
229 
230  // get the area of interest
231 
232  int regionKnotFirst = knot->rasterKnot;
233  int regionKnotLast = knot->rasterKnot;
234 
235  getRegionOfInfluence(knot, regionKnotFirst, regionKnotLast);
236 
237  // std::cout<<"uv "<<uv<<" knot "<<knot->rasterKnot<<" region: "<<regionKnotFirst<<" <-> "<<regionKnotLast<<std::endl;
238 
239  KnotData tmp = *knot;
240  std::list<KnotData>::iterator next = mKnots[uv].erase(knot);
241  createActionSpline();
242  knot = mKnots[uv].insert(next, tmp);
243 
244  // get the cost
245 
246  ret.action = Action::Move::Remove;
247  ret.cost = getMaxDeviationArea(mActionSpline, mActionSplineData, uv, regionKnotFirst, regionKnotLast);
248  ret.iter = knot;
249 
250  return ret;
251 }

253 void IrregularSpline2D3DCalibrator::getRegionOfInfluence(std::list<KnotData>::iterator knot, int& regionKnotFirst, int& regionKnotLast) const
254 {
255  int uv = knot->uv;
256  regionKnotFirst = knot->rasterKnot;
257  regionKnotLast = knot->rasterKnot;
258  std::list<KnotData>::iterator next = knot;
259  std::list<KnotData>::iterator prev = knot;
260  for (int i = 0; i < 3; ++i) {
261  if (prev != mKnots[uv].begin()) {
262  --prev;
263  regionKnotFirst = prev->rasterKnot;
264  } else {
265  regionKnotFirst = 0;
266  }
267 
268  if (next != mKnots[uv].end()) {
269  ++next;
270  if (next != mKnots[uv].end()) {
271  regionKnotLast = next->rasterKnot;
272  } else {
273  regionKnotLast = mRaster.getGrid(uv).getNumberOfKnots() - 1;
274  }
275  }

279 bool IrregularSpline2D3DCalibrator::doCalibrationStep()
280 {
281  // perform one step of the calibration
282 
283  // first, try to move a knot
284  //std::cout<<"do step.. "<<std::endl;
285  Action bestAction;
286  bestAction.action = Action::Move::No;
287  bestAction.cost = 1.e10;
288 
289  for (int uv = 0; uv < 2; ++uv) {
290  for (std::list<KnotData>::iterator i = mKnots[uv].begin(); i != mKnots[uv].end(); ++i) {
291  Action a = checkActionShift(i);
292  if (a.cost < bestAction.cost)
293  bestAction = a;
294  }
295  }
296 
297  //std::cout<<"move cost: "<<bestAction.cost<<std::endl;
298  if (bestAction.cost < 0) { // shift the best knot
299  if (bestAction.action == Action::Move::Up) {
300  //std::cout<<"move Up uv "<< bestAction.iter->uv<<" knot "<<bestAction.iter->rasterKnot<<" -> "<<bestAction.iter->rasterKnot+1<<std::endl;
301  bestAction.iter->rasterKnot++;
302  } else if (bestAction.action == Action::Move::Down) {
303  //std::cout<<"move Down uv "<< bestAction.iter->uv<<" knot "<<bestAction.iter->rasterKnot<<" -> "<<bestAction.iter->rasterKnot-1<<std::endl;
304  bestAction.iter->rasterKnot--;
305  } else {
306  std::cerr << "Internal error!!!" << std::endl;
307  return 0;
308  }
309  createCurrentSpline();
310  return 1;
311  }
312 
313  // second, try to remove a knot
314 
315  //for (int axis = 0; axis < 2; axis++) {
316  bestAction.action = Action::Move::No;
317  bestAction.cost = mMaxDeviation + 1.e10;
318 
319  for (int uv = 0; uv < 2; ++uv) {
320 
321  for (std::list<KnotData>::iterator i = mKnots[uv].begin(); i != mKnots[uv].end(); ++i) {
322  Action a = checkActionRemove(i);
323  if (a.cost < bestAction.cost)
324  bestAction = a;
325  }
326  }
327  bestAction.cost = sqrt(bestAction.cost / 3.);
328 
329  //std::cout<<"remove cost: "<<bestAction.cost<<std::endl;
330 
331  if (bestAction.cost <= mMaxDeviation) { // move the best knot
332  if (bestAction.action == Action::Move::Remove) {
333  //std::cout<<"remove uv "<< bestAction.iter->uv<<" knot "<<bestAction.iter->rasterKnot<<std::endl;
334  mKnots[bestAction.iter->uv].erase(bestAction.iter);
335  } else {
336  std::cout << "Internal error!!!" << std::endl;
337  return 0;
338  }
339  createCurrentSpline();
340  return 1;
341  }
342 
343  return 0;
344 }

363 double IrregularSpline2D3DCalibrator::getMaxDeviationLine(const IrregularSpline2D3D& spline, const std::vector<float>& data, int axis0, int knot0) const
364 {
365  int axis1 = (axis0 == 0) ? 1 : 0;
366  float u[2];
367  u[axis0] = mRaster.getGrid(axis0).getKnot(knot0).u;
368 
369  double dMax2 = 0.;
370 
371  for (int knot1 = 0; knot1 < mRaster.getGrid(axis1).getNumberOfKnots(); ++knot1) {
372  u[axis1] = mRaster.getGrid(axis1).getKnot(knot1).u;
373  float fx0, fy0, fz0, fx, fy, fz;
374  mRaster.getSplineVec(mRasterData.data(), u[0], u[1], fx0, fy0, fz0);
375  spline.getSplineVec(data.data(), u[0], u[1], fx, fy, fz);
376  double dx = fx - fx0;
377  double dy = fy - fy0;
378  double dz = fz - fz0;
379  double d2 = dx * dx + dy * dy + dz * dz;
380  if (dMax2 < d2)
381  dMax2 = d2;
382  }
383  return dMax2;
384 }
