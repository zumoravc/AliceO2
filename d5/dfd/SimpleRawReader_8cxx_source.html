34 void SimpleRawReader::init()
35 {

38  }

51 bool SimpleRawReader::loadNextTF()
52 {
53  mSTF.reset();
54 
55  if (!mReader) {
56  init();
57  }
58  int nLinks = mReader->getNLinks(), nread = 0;
59  std::vector<InputRoute> inpRoutes;
60  SimpleSTF::PartsRef partsRef;
61  SimpleSTF::Messages messages;
62 
63  auto tfID = mReader->getNextTFToRead();
64  if (tfID >= mReader->getNTimeFrames()) {
65  if (mReader->getNTimeFrames() && mLoop--) {
66  tfID = 0;
67  mReader->setNextTFToRead(tfID);
68  mLoopsDone++;
69  for (int il = 0; il < nLinks; il++) {
70  mReader->getLink(il).nextBlock2Read = 0; // think about more elaborate looping scheme, e.g. incrementing the orbits in RDHs
71  }
72  LOG(INFO) << "Starting new loop " << mLoopsDone << " from the beginning of data";
73  } else {
74  printStat();
75  mDone = true;
76  return false;
77  }
78  }
79  auto nHB = HBFUtils::Instance().getNOrbitsPerTF();
80  partsRef.reserve(nLinks);
81  inpRoutes.reserve(nLinks);
82  messages.reserve(2 * (mHBFPerMessage ? nHB * nLinks : nLinks));
83  size_t totSize = 0;
84  for (int il = 0; il < nLinks; il++) {
85  auto& link = mReader->getLink(il);
86  auto tfsz = link.getNextTFSize();
87  if (!tfsz) {
88  continue;
89  }
90  InputSpec inps(std::string("inpSpec") + std::to_string(il), link.origin, link.description, link.subspec, Lifetime::Timeframe);
91  inpRoutes.emplace_back(InputRoute{inps, size_t(il), std::string("src") + std::to_string(il)});
92 
93  o2::header::DataHeader hdrTmpl(link.description, link.origin, link.subspec); // template with 0 size
94  hdrTmpl.payloadSerializationMethod = o2::header::gSerializationMethodNone;
95  hdrTmpl.splitPayloadParts = mHBFPerMessage ? nHB : 1;
96  partsRef.emplace_back(messages.size(), hdrTmpl.splitPayloadParts); // entry and nparts the multipart in the messages
97  while (hdrTmpl.splitPayloadIndex < hdrTmpl.splitPayloadParts) {
98 
99  hdrTmpl.payloadSize = mHBFPerMessage ? link.getNextHBFSize() : tfsz;
100  o2::header::Stack headerStack{hdrTmpl, o2f::DataProcessingHeader{mTFIDaccum}};
101  auto* hdMessage = messages.emplace_back(std::make_unique<std::vector<char>>(headerStack.size()))->data(); // header message
102  memcpy(hdMessage, headerStack.data(), headerStack.size());
103 
104  auto* plMessage = messages.emplace_back(std::make_unique<std::vector<char>>(hdrTmpl.payloadSize))->data(); // payload message
105  auto bread = mHBFPerMessage ? link.readNextHBF(plMessage) : link.readNextTF(plMessage);
106  if (bread != hdrTmpl.payloadSize) {
107  LOG(ERROR) << "Link " << il << " read " << bread << " bytes instead of " << hdrTmpl.payloadSize
108  << " expected in TF=" << mTFIDaccum << " part=" << hdrTmpl.splitPayloadIndex;
109  throw std::runtime_error("error in link data reading");
110  }
111  hdrTmpl.splitPayloadIndex++; // prepare for next
112  LOG(DEBUG) << "Loaded " << tfsz << " bytes for " << link.describe();
113  }
114  totSize += tfsz;
115  nread++;
116  }
117  LOG(INFO) << "Loaded " << totSize << " bytes for " << nread << " non-empty links out of " << nLinks;
118  mSTF = std::make_unique<SimpleSTF>(std::move(inpRoutes), std::move(partsRef), std::move(messages));
119 
120  return nread > 0;
121 }
