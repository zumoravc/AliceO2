118  Int_t indexPhi;
119  Int_t startR = rIndex - rStep / 2;
120  Int_t startZ = zIndex - zStep / 2;
121 
122  if (startPhi < 0) {
123  startPhi = fNPhi + startPhi;
124  }
125  if (startR < 0) {
126  startR = 0;
127  }
128  if (startR + rStep >= fNR) {
129  startR = fNR - rStep;
130  }
131 
132  if (startZ < fMinZIndex) {
133  startZ = fMinZIndex;
134  }
135  if (startZ + zStep >= fNZ) {
136  startZ = fNZ - zStep;
137  }
138 
139  Int_t index;
140  Double_t sum_w = 0.0;
141  Double_t sum_d = 0.0;
142  Double_t shortest_d = 10000.0;
143  Int_t new_rIndex = 0;
144  Int_t new_zIndex = 0;
145  Int_t new_phiIndex = 0;
146 
147  for (Int_t iPhi = startPhi; iPhi < startPhi + phiStep; iPhi++) {
148  indexPhi = iPhi % fNPhi;

152  index = indexPhi * (fNZ * fNR) + index_r * fNZ + index_z;

156  phi0 = fPhiList[index];

164  }
165  }
166  }
167  }
168 
169  phiStep = 3;
170  rStep = 3;
171  startPhi = new_phiIndex - phiStep / 2;
172  startR = new_rIndex - rStep / 2;
173  startZ = new_zIndex - zStep / 2;
174 
175  if (startPhi < 0) {
176  startPhi = fNPhi + startPhi;
177  }
178  if (startR < 0) {
179  startR = 0;
180  }
181  if (startR + rStep >= fNR) {
182  startR = fNR - rStep;
183  }
184 
185  if (startZ < fMinZIndex) {
186  startZ = fMinZIndex;
187  }
188  if (startZ + zStep >= fNZ) {
189  startZ = fNZ - zStep;
190  }
191 
192  for (Int_t iPhi = startPhi; iPhi < startPhi + phiStep; iPhi++) {
193  indexPhi = iPhi % fNPhi;
194  for (Int_t index_r = startR; index_r < startR + rStep; index_r++) {
195  for (Int_t index_z = startZ; index_z < startZ + zStep; index_z++) {
196  // check for the closest poInt_t
197  index = indexPhi * (fNZ * fNR) + index_r * fNZ + index_z;
198 
199  r0 = fRList[indexPhi * (fNR * fNZ) + index_r * fNZ + index_z];
200  z0 = fZList[indexPhi * (fNR * fNZ) + index_r * fNZ + index_z];
201  phi0 = fPhiList[indexPhi * (fNR * fNZ) + index_r * fNZ + index_z];
202  d = Distance(r0, phi0, z0, r, phi, z);
203  if (d < MIN_DIST) {
204  return fValue[index];
205  }
206  d = 1.0 / d;
207  sum_w += (fValue[index] * d * d * d * d);
208  sum_d += d * d * d * d;
209  }
210  }
211  }
212  return (sum_w / sum_d);
213 }

224 Double_t
225  AliTPC3DCylindricalInterpolatorIrregular::Distance(Double_t r0, Double_t phi0, Double_t z0, Double_t r, Double_t phi,
226  Double_t z)
227 {
228  if (phi < 0) {
229  phi = TMath::TwoPi() + phi;
230  }
231  if (phi > TMath::TwoPi()) {
232  phi = phi - TMath::TwoPi();
233  }
234 
235  if (phi0 < 0) {
236  phi0 = TMath::TwoPi() + phi0;
237  }
238  if (phi0 > TMath::TwoPi()) {
239  phi0 = phi0 - TMath::TwoPi();
240  }
241 
242  Double_t dPhi = phi - phi0;
243  if (dPhi > TMath::Pi()) {
244  dPhi = TMath::TwoPi() - dPhi;
245  }
246  if (dPhi < -TMath::Pi()) {
247  dPhi = TMath::TwoPi() + dPhi;
248  }
249 
250  Double_t ret = r * r + r0 * r0 - 2 * r0 * r * TMath::Cos(dPhi) + (z - z0) * (z - z0);
251 
252  return TMath::Sqrt(ret);
253 }

391 void AliTPC3DCylindricalInterpolatorIrregular::Search(Int_t n, Double_t* xArray, Int_t offset, Double_t x, Int_t& low)
392 {
394 
395  Long_t middle, high;
396  Int_t ascend = 0, increment = 1;
397 
398  if (xArray[(n - 1) * offset] >= xArray[0 * offset]) {
399  ascend = 1; // Ascending ordered table if true
400  }
401  if (low < 0 || low > n - 1) {
402  low = -1;
403  high = n;
404  } else { // Ordered Search phase
405  if ((Int_t)(x >= xArray[low * offset]) == ascend) {
406  if (low == n - 1) {
407  return;
408  }
409  high = low + 1;
410  while ((Int_t)(x >= xArray[high * offset]) == ascend) {
411  low = high;
412  increment *= 2;
413  high = low + increment;
414  if (high > n - 1) {
415  high = n;
416  break;
417  }
418  }
419  } else {
420  if (low == 0) {
421  low = -1;
422  return;
423  }
424  high = low - 1;
425  while ((Int_t)(x < xArray[low * offset]) == ascend) {
426  high = low;
427  increment *= 2;
428  if (increment >= high) {
429  low = -1;
430  break;
431  } else {
432  low = high - increment;
433  }
434  }
435  }
436  }
437 
438  while ((high - low) != 1) { // Binary Search Phase
439  middle = (high + low) / 2;
440  if ((Int_t)(x >= xArray[middle * offset]) == ascend) {
441  low = middle;
442  } else {
443  high = middle;
444  }
445  }
446 
447  if (x > xArray[n - 1]) {
448  low = n;
449  }
450  if (x < xArray[0]) {
451  low = -1;
452  }
453 }
