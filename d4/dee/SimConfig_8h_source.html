22 // configuration struct (which can be passed around)
23 struct SimConfigData {
24  std::vector<std::string> mActiveDetectors; // list of active detectors

30  std::string mHepMCFileName; // file name of HepMC file
31  std::string mExtGenFileName; // file name containing the external generator configuration
32  std::string mExtGenFuncName; // function call to retrieve the external generator configuration
33  std::string mExtTrgFileName; // file name containing the external trigger configuration
34  std::string mExtTrgFuncName; // function call to retrieve the external trigger configuration
35  std::string mEmbedIntoFileName; // filename containing the reference events to be used for the embedding
36  std::string mDecayConfig; // filename containing the external decayer configuration
37  unsigned int mStartEvent; // index of first event to be taken
38  float mBMax; // maximum for impact parameter sampling
39  bool mIsMT; // chosen MT mode (Geant4 only)
40  std::string mOutputPrefix; // prefix to be used for output files
41  std::string mLogSeverity; // severity for FairLogger
42  std::string mLogVerbosity; // loglevel for FairLogger
43  std::string mKeyValueTokens; // a string holding arbitrary sequence of key-value tokens
44  // Foo.parameter1=x,Bar.parameter2=y,Baz.paramter3=hello
45  // (can be used to **loosely** change any configuration parameter from command-line)
46  std::string mConfigFile; // path to a JSON or INI config file (file extension is required to determine type).
47  // values within the config file will override values set in code by the param classes
48  // but will themselves be overridden by any values given in mKeyValueTokens.
49  int mPrimaryChunkSize; // defining max granularity for input primaries of a sim job
50  int mInternalChunkSize; //
51  int mStartSeed; // base for random number seeds
52  int mSimWorkers = 1; // number of parallel sim workers (when it applies)
53  bool mFilterNoHitEvents = false; // whether to filter out events not leaving any response
54  std::string mCCDBUrl; // the URL where to find CCDB
55  long mTimestamp; // timestamp to anchor transport simulation to
56  int mField; // L3 field setting in kGauss: +-2,+-5 and 0
57 
58  ClassDefNV(SimConfigData, 3);
