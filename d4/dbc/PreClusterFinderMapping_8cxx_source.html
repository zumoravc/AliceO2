45 auto Mapping::addPad(MpDE& de, const mapping::Segmentation& segmentation)
46 {
48  return [&de, &segmentation](int padID) {
49  MpPad& pad = de.pads[padID];
50  double padX = segmentation.padPositionX(padID);
51  double padY = segmentation.padPositionY(padID);
52  double padSizeX = segmentation.padSizeX(padID);
53  double padSizeY = segmentation.padSizeY(padID);
54  pad.area[0][0] = padX - padSizeX / 2.;
55  pad.area[0][1] = padX + padSizeX / 2.;
56  pad.area[1][0] = padY - padSizeY / 2.;
57  pad.area[1][1] = padY + padSizeY / 2.;
58  pad.nNeighbours = 0;
59  segmentation.forEachNeighbouringPad(padID, addNeighbour(pad));
60  };
61 }

64 std::vector<std::unique_ptr<Mapping::MpDE>> Mapping::createMapping()
65 {
67 
68  std::vector<std::unique_ptr<MpDE>> detectionElements{};
69 
70  // create the internal mapping for each DE
71  mapping::forEachDetectionElement([&detectionElements](int deID) {
72  auto& segmentation = mapping::segmentation(deID);
73  detectionElements.push_back(std::make_unique<MpDE>());
74  MpDE& de(*(detectionElements.back()));
75  de.uid = deID;
76  de.nPads[0] = segmentation.bending().nofPads();
77  de.nPads[1] = segmentation.nonBending().nofPads();
78  de.pads = std::make_unique<MpPad[]>(de.nPads[0] + de.nPads[1]);
79  segmentation.forEachPad(addPad(de, segmentation));
80  });
81 
82  return detectionElements;
83 }

108 bool Mapping::areOverlappingExcludeCorners(float area1[2][2], float area2[2][2])
109 {
111 
112  // precision in cm: positive = increase pad size / negative = decrease pad size
113  constexpr float precision = 1.e-4;
114 
115  if (areOverlapping(area1, area2, precision)) {
116  for (int ip1 = 0; ip1 < 2; ++ip1) {
117  for (int ip2 = 0; ip2 < 2; ++ip2) {
118  if (TMath::Abs(area1[0][ip1] - area2[0][1 - ip1]) < precision &&
119  TMath::Abs(area1[1][ip2] - area2[1][1 - ip2]) < precision) {
120  return false;
121  }
122  }
123  }
124  return true;
125  }
126 
127  return false;
128 }
