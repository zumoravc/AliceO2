78 struct ChannelStat {
79  double mean{0};
80  double rms{0};
81  double q{0};
82  int n{0};
83 
84  void incr(int v)
85  {
86  n++;
87  auto newMean = mean + (v - mean) / n;
88  q += (v - newMean) * (v - mean);
89  rms = sqrt(q / n);
90  mean = newMean;
91  }
92 };

100 std::map<std::string, ChannelStat> rawdump(std::string input, DumpOptions opt)
101 {
102  std::ifstream in(input.c_str(), std::ios::binary);
103  if (!in.good()) {
104  std::cout << "could not open file " << input << "\n";
105  return {};
106  }
107  constexpr size_t pageSize = 8192;
108 
109  std::array<std::byte, pageSize> buffer;
110  gsl::span<const std::byte> page(buffer);
111 
112  size_t nrdhs{0};
113 
114  const auto patchPage = [&](gsl::span<std::byte> rdhBuffer) {
115  auto rdhPtr = const_cast<void*>(reinterpret_cast<const void*>(rdhBuffer.data()));
116  nrdhs++;
117  auto cruId = o2::raw::RDHUtils::getCRUID(rdhPtr);
118  if (opt.cruId().has_value()) {
119  cruId = opt.cruId().value();
120  o2::raw::RDHUtils::setLinkID(rdhPtr, 0);
121  }
122  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhPtr);
123  o2::raw::RDHUtils::setFEEID(rdhPtr, cruId * 2 + endpoint);
124  if (opt.showRDHs()) {
125  std::cout << nrdhs << "--\n";
126  o2::raw::RDHUtils::printRDH(rdhPtr);
127  }
128  };

130  struct Counters {
131  std::map<std::string, int> uniqueDS;
132  std::map<std::string, int> uniqueChannel;
133  std::map<std::string, ChannelStat> statChannel;
134  int ndigits{0};
135  } counters;
136 
137  const auto channelHandler =
138  [&](DsElecId dsId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
139  auto s = asString(dsId);
140  counters.uniqueDS[s]++;
141  auto ch = fmt::format("{}-CH{}", s, channel);
142  counters.uniqueChannel[ch]++;
143  auto& chanstat = counters.statChannel[ch];
144  if (sc.isClusterSum()) {
145  chanstat.incr(sc.chargeSum);
146  } else {
147  for (auto d = 0; d < sc.nofSamples(); d++) {
148  chanstat.incr(sc.samples[d]);
149  }
150  }
151  counters.ndigits++;
152  };
153 
154  o2::mch::raw::PageDecoder decode = nullptr;
155 
156  size_t npages{0};
157  uint64_t bytesRead{0};
158 
159  char* cbuffer = reinterpret_cast<char*>(&buffer[0]);
160  char* cpayload = reinterpret_cast<char*>(&buffer[sizeof(o2::header::RDHAny)]);
161  const void* rdhPtr = reinterpret_cast<const void*>(&buffer[0]);
162 
163  while (npages < opt.maxNofRDHs()) {
164 
165  // read the next RDH, stop if no more data is available
166  in.read(cbuffer, sizeof(o2::header::RDHAny));
167  if (in.fail()) {
168  break;
169  }
170 
171  auto readSize = o2::raw::RDHUtils::getOffsetToNext(rdhPtr) - sizeof(o2::header::RDHAny);
172  in.read(cpayload, readSize);
173  if (in.fail()) {
174  break;
175  }
176 
177  npages++;
178  bytesRead += in.gcount();
179  if (!decode) {
180  decode = createPageDecoder(page, channelHandler);
181  }
182  patchPage(buffer);
183  decode(page);
184  }
185 
186  if (!opt.json()) {
187  std::cout << counters.ndigits << " digits seen - " << nrdhs << " RDHs seen - " << npages << " npages read\n";
188  std::cout << "#unique DS=" << counters.uniqueDS.size() << " #unique Channel=" << counters.uniqueChannel.size() << "\n";
189  }
190 
191  return counters.statChannel;
192 }

194 void output(const std::map<std::string, ChannelStat>& channels)
195 {
196  rapidjson::StringBuffer buffer;
197  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
198 
199  writer.StartObject();
200  writer.Key("channels");
201  writer.StartArray();
202  for (auto s : channels) {
203  writer.StartObject();
204  writer.Key("id");
205  writer.String(s.first.c_str());
206  writer.Key("ped");
207  writer.Double(s.second.mean);
208  writer.Key("noise");
209  writer.Double(s.second.rms);
210  writer.Key("nof_samples");
211  writer.Int(s.second.n);
212  writer.EndObject();
213  }
214  writer.EndArray();
215  writer.EndObject();
216  std::cout << buffer.GetString() << "\n";
217 }

248 
249  if (vm.count("help")) {

253 
254  try {

258  exit(1);

260

262 
263  if (vm.count("cru")) {
264  opt.cruId(vm["cru"].as<uint16_t>());
265  }
266 
267  auto statChannel = rawdump(inputFile, opt);
268 
269  if (jsonOutput) {
270  output(statChannel);
271  }
272  return 0;
