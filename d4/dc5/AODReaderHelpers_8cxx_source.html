42 enum AODTypeMask : uint64_t {
43  None = 0,
44  Track = 1 << 0,
45  TrackCov = 1 << 1,
46  TrackExtra = 1 << 2,
47  Calo = 1 << 3,
48  CaloTrigger = 1 << 4,
49  Muon = 1 << 5,
50  MuonCluster = 1 << 6,
51  Zdc = 1 << 7,
52  BC = 1 << 8,
53  Collision = 1 << 9,
54  FT0 = 1 << 10,
55  FV0 = 1 << 11,
56  FDD = 1 << 12,
57  UnassignedTrack = 1 << 13,
58  Run2V0 = 1 << 14,
59  McCollision = 1 << 15,
60  McTrackLabel = 1 << 16,
61  McCaloLabel = 1 << 17,
62  McParticle = 1 << 18,
63  Unknown = 1 << 19
64 };

66 uint64_t getMask(header::DataDescription description)
67 {
68 
69  if (description == header::DataDescription{"TRACKPAR"}) {
70  return AODTypeMask::Track;
71  } else if (description == header::DataDescription{"TRACKPARCOV"}) {
72  return AODTypeMask::TrackCov;
73  } else if (description == header::DataDescription{"TRACKEXTRA"}) {
74  return AODTypeMask::TrackExtra;
75  } else if (description == header::DataDescription{"CALO"}) {
76  return AODTypeMask::Calo;
77  } else if (description == header::DataDescription{"CALOTRIGGER"}) {
78  return AODTypeMask::CaloTrigger;
79  } else if (description == header::DataDescription{"MUON"}) {
80  return AODTypeMask::Muon;
81  } else if (description == header::DataDescription{"MUONCLUSTER"}) {
82  return AODTypeMask::MuonCluster;
83  } else if (description == header::DataDescription{"ZDC"}) {
84  return AODTypeMask::Zdc;
85  } else if (description == header::DataDescription{"BC"}) {
86  return AODTypeMask::BC;
87  } else if (description == header::DataDescription{"COLLISION"}) {
88  return AODTypeMask::Collision;
89  } else if (description == header::DataDescription{"FT0"}) {
90  return AODTypeMask::FT0;
91  } else if (description == header::DataDescription{"FV0"}) {
92  return AODTypeMask::FV0;
93  } else if (description == header::DataDescription{"FDD"}) {
94  return AODTypeMask::FDD;
95  } else if (description == header::DataDescription{"UNASSIGNEDTRACK"}) {
96  return AODTypeMask::UnassignedTrack;
97  } else if (description == header::DataDescription{"RUN2V0"}) {
98  return AODTypeMask::Run2V0;
99  } else if (description == header::DataDescription{"MCCOLLISION"}) {
100  return AODTypeMask::McCollision;
101  } else if (description == header::DataDescription{"MCTRACKLABEL"}) {
102  return AODTypeMask::McTrackLabel;
103  } else if (description == header::DataDescription{"MCCALOLABEL"}) {
104  return AODTypeMask::McCaloLabel;
105  } else if (description == header::DataDescription{"MCPARTICLE"}) {
106  return AODTypeMask::McParticle;
107  } else {
108  LOG(DEBUG) << "This is a tree of unknown type! " << description.str;
109  return AODTypeMask::Unknown;
110  }
111 }

113 uint64_t calculateReadMask(std::vector<OutputRoute> const& routes, header::DataOrigin const& origin)
114 {
115  uint64_t readMask = None;
116  for (auto& route : routes) {
117  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(route.matcher);
118  auto description = concrete.description;
119 
120  readMask |= getMask(description);
121  }
122  return readMask;
123 }

125 std::vector<OutputRoute> getListOfUnknown(std::vector<OutputRoute> const& routes)
126 {
127 
128  std::vector<OutputRoute> unknows;
129  for (auto& route : routes) {
130  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(route.matcher);
131 
132  if (getMask(concrete.description) == AODTypeMask::Unknown)
133  unknows.push_back(route);
134  }
135  return unknows;
136 }

138 AlgorithmSpec AODReaderHelpers::rootFileReaderCallback()
139 {
140  auto callback = AlgorithmSpec{adaptStateful([](ConfigParamRegistry const& options,
141  DeviceSpec const& spec) {
142  auto filename = options.get<std::string>("aod-file");
143 
144  // create a DataInputDirector
145  auto didir = std::make_shared<DataInputDirector>(filename);
146  if (options.isSet("json-file")) {
147  auto jsonFile = options.get<std::string>("json-file");
148  if (!didir->readJson(jsonFile)) {
149  LOGP(ERROR, "Check the JSON document! Can not be properly parsed!");
150  }
151  }
152 
153  // analyze type of requested tables
154  uint64_t readMask = calculateReadMask(spec.outputs, header::DataOrigin{"AOD"});
155  std::vector<OutputRoute> unknowns;
156  if (readMask & AODTypeMask::Unknown) {
157  unknowns = getListOfUnknown(spec.outputs);
158  }
159 
160  auto counter = std::make_shared<int>(0);
161  return adaptStateless([readMask,
162  unknowns,
163  counter,
164  didir](DataAllocator& outputs, ControlService& control, DeviceSpec const& device) {
165  // Each parallel reader reads the files whose index is associated to
166  // their inputTimesliceId
167  assert(device.inputTimesliceId < device.maxInputTimeslices);
168  size_t fi = (*counter * device.maxInputTimeslices) + device.inputTimesliceId;
169  *counter += 1;
170 
171  if (didir->atEnd(fi)) {
172  LOGP(INFO, "All input files processed");
173  didir->closeInputFiles();
174  control.endOfStream();
175  control.readyToQuit(QuitRequest::Me);
176  return;
177  }
178 
179  auto tableMaker = [&readMask, &outputs, fi, didir](auto metadata, AODTypeMask mask, char const* treeName) {
180  if (readMask & mask) {
181 
182  auto dh = header::DataHeader(decltype(metadata)::description(), decltype(metadata)::origin(), 0);
183  auto reader = didir->getTreeReader(dh, fi, treeName);
184 
185  using table_t = typename decltype(metadata)::table_t;
186  if (!reader || (reader && reader->IsInvalid())) {
187  LOGP(ERROR, "Requested \"{}\" tree not found in input file \"{}\"", treeName, didir->getInputFilename(dh, fi));
188  } else {
189  auto& builder = outputs.make<TableBuilder>(Output{decltype(metadata)::origin(), decltype(metadata)::description()});
190  RootTableBuilderHelpers::convertASoA<table_t>(builder, *reader);
191  }
192  }
193  };
194  tableMaker(o2::aod::CollisionsMetadata{}, AODTypeMask::Collision, "O2collision");
195  tableMaker(o2::aod::TracksMetadata{}, AODTypeMask::Track, "O2track");
196  tableMaker(o2::aod::TracksCovMetadata{}, AODTypeMask::TrackCov, "O2track");
197  tableMaker(o2::aod::TracksExtraMetadata{}, AODTypeMask::TrackExtra, "O2track");
198  tableMaker(o2::aod::CalosMetadata{}, AODTypeMask::Calo, "O2calo");
199  tableMaker(o2::aod::CaloTriggersMetadata{}, AODTypeMask::Calo, "O2calotrigger");
200  tableMaker(o2::aod::MuonsMetadata{}, AODTypeMask::Muon, "O2muon");
201  tableMaker(o2::aod::MuonClustersMetadata{}, AODTypeMask::Muon, "O2muoncluster");
202  tableMaker(o2::aod::ZdcsMetadata{}, AODTypeMask::Zdc, "O2zdc");
203  tableMaker(o2::aod::BCsMetadata{}, AODTypeMask::BC, "O2bc");
204  tableMaker(o2::aod::FT0sMetadata{}, AODTypeMask::FT0, "O2ft0");
205  tableMaker(o2::aod::FV0sMetadata{}, AODTypeMask::FV0, "O2fv0");
206  tableMaker(o2::aod::FDDsMetadata{}, AODTypeMask::FDD, "O2fdd");
207  tableMaker(o2::aod::UnassignedTracksMetadata{}, AODTypeMask::UnassignedTrack, "O2unassignedtrack");
208  tableMaker(o2::aod::Run2V0sMetadata{}, AODTypeMask::Run2V0, "Run2v0");
209  tableMaker(o2::aod::McCollisionsMetadata{}, AODTypeMask::McCollision, "O2mccollision");
210  tableMaker(o2::aod::McTrackLabelsMetadata{}, AODTypeMask::McTrackLabel, "O2mctracklabel");
211  tableMaker(o2::aod::McCaloLabelsMetadata{}, AODTypeMask::McCaloLabel, "O2mccalolabel");
212  tableMaker(o2::aod::McParticlesMetadata{}, AODTypeMask::McParticle, "O2mcparticle");
213 
214  // tables not included in the DataModel
215  if (readMask & AODTypeMask::Unknown) {
216 
217  // loop over unknowns
218  for (auto route : unknowns) {
219 
220  // create a TreeToTable object
221  auto concrete = DataSpecUtils::asConcreteDataMatcher(route.matcher);
222  auto dh = header::DataHeader(concrete.description, concrete.origin, concrete.subSpec);
223 
224  auto tr = didir->getDataTree(dh, fi);
225  if (!tr) {
226  char* table;
227  sprintf(table, "%s/%s/%" PRIu32, concrete.origin.str, concrete.description.str, concrete.subSpec);
228  LOGP(ERROR, "Error while retrieving the tree for \"{}\"!", table);
229  return;
230  }
231 
232  auto o = Output(dh);
233  auto& t2t = outputs.make<TreeToTable>(o, tr);
234 
235  // fill the table
236  t2t.fill();
237  }
238  }
239  });
240  })};
241 
242  return callback;
243 }
