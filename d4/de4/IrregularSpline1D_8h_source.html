273 template <typename T>
274 GPUdi() T IrregularSpline1D::getSpline(const IrregularSpline1D::Knot& knot1, T f0, T f1, T f2, T f3, float u)
275 {
278 
279  f0 -= f1;
280  f2 -= f1;
281  f3 -= f1;
282 
283  T x = T((u - knot1.u) * knot1.scale); // scaled u
284  T z1 = T(f0 * knot1.scaleL0 + f2 * knot1.scaleL2); // scaled u derivative at the knot 1
285  T z2 = T(f2 * knot1.scaleR2 + f3 * knot1.scaleR3); // scaled u derivative at the knot 2
286 
287  T x2 = x * x;
288 
289  // f(x) = ax^3 + bx^2 + cx + d
290  //
291  // f(0) = f1
292  // f(1) = f2
293  // f'(0) = z1
294  // f'(1) = z2
295  //
296 
297  // T d = f1;
298  // T c = z1;
299  T a = -f2 - f2 + z1 + z2;
300  T b = f2 - z1 - a;
301  return a * x * x2 + b * x2 + z1 * x + f1;
302 }

327 GPUdi() void IrregularSpline1D::getEdgeCorrectionCoefficients(double u0, double u1, double u2, double u3, double& c0, double& c1, double& c2, double& c3)
328 {
332 
333  double du = u2 - u1;
334  double x0 = (u0 - u1) / du;
335  // double x1 = ( u1 - u1 )/du = 0
336  // double x2 = ( u2 - u1 )/du = 1
337  double x3 = (u3 - u1) / du;
338 
339  double cL0 = -1. / (x0 * (x0 - 1.));
340  double cL2 = x0 / (x0 - 1.);
341  double cR2 = (x3 - 2.) / (x3 - 1.);
342  double cR3 = 1. / (x3 * (x3 - 1.));
343 
344  // Fi = fi - f1
345 
346  // z1 = F0corr*cL0 + F2*cL2; // scaled u derivative at the knot 1
347  // z2 = F2*cR2 + F3*cR3; // scaled u derivative at the knot 2
348 
349  // a = -2F2 + z1 + z2
350  // b = 3F2 - 2*z1 - z2 ;
351  // c = z1
352  // d = 0
353 
354  // F(x) = ax^3 + bx^2 + cx + d
355  // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*z2
356  // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*(cR2*F2 + cR3*F3)
357  // F(x) = (-2*x^3+3*x^2+(x^3-x^2)*cR2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*cR3*F3
358  // F(x) = x^2(-2*x+3+(x-1)*cR2)*F2 + x*(x-1)^2*z1 + x^2(x-1)*cR3*F3
359 
360  // z1*x*(x-1)^2 = F(x) - x^2(-2*x+3+(x-1)*cR2)*F2 - x^2(x-1)*cR3*F3
361  // z1*x0*(x0-1)^2 = F0 - x0^2(-2*x0+3+(x0-1)*cR2)*F2 - x0^2(x0-1)*cR3*F3
362 
363  double x01 = x0 - 1.;
364 
365  // coeff. for z1 at F0, F1, F2, F3:
366 
367  c0 = 1. / (x0 * x01 * x01);
368  c1 = 0;
369  c2 = -(-2 * x0 + 3. + x01 * cR2) * x0 / (x01 * x01);
370  c3 = -x0 * cR3 / x01;
371 
372  // F0corr = (z1 - F2*cL2)/cL0;
373 
374  c0 = c0 / cL0;
375  c1 = c1 / cL0;
376  c2 = (c2 - cL2) / cL0;
377  c3 = c3 / cL0;
378 
379  // F0corr = c0*F0 + c1*F1 + C2*F2 + c3*F3;
380  // f0corr-f1 = c0*(f0-f1) + c1*(f1-f1) + c2*(f2-f1) + c3*(f3-f1);
381  // f0corr-f1 = c0*(f0-f1) + c2*(f2-f1) + c3*(f3-f1);
382  // f0corr = c0*f0 + c2*f2 + c3*f3 + (1-c0-c2-c3)*f1;
383 
384  c1 = 1. - c0 - c2 - c3;
385 }

387 template <typename T>
388 GPUdi() void IrregularSpline1D::correctEdges(T* data) const
389 {
390  const IrregularSpline1D::Knot* s = getKnots();
391  double c0, c1, c2, c3;
392  getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
393  data[0] = c0 * data[0] + c1 * data[1] + c2 * data[2] + c3 * data[3];
394  int i = mNumberOfKnots - 1;
395  getEdgeCorrectionCoefficients(s[i - 0].u, s[i - 1].u, s[i - 2].u, s[i - 3].u, c0, c1, c2, c3);
396  data[i] = c0 * data[i - 0] + c1 * data[i - 1] + c2 * data[i - 2] + c3 * data[i - 3];
397 }
