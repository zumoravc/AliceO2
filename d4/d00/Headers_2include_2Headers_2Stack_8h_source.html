116  template <typename T>
117  constexpr static size_t calculateSize(T&& h) noexcept
118  {
119  //if it's a pointer (to a stack) traverse it
120  if constexpr (std::is_convertible_v<T, o2::byte*>) {
121  const BaseHeader* next = BaseHeader::get(std::forward<T>(h));
122  if (!next) {
123  return 0;
124  }
125  size_t size = next->size();
126  while ((next = next->next())) {
127  size += next->size();
128  }
129  return size;

145  template <typename T>
146  static o2::byte* inject(o2::byte* here, T&& h, bool more = false) noexcept
147  {
148  using headerType = typename std::remove_cv<typename std::remove_reference<T>::type>::type;
149  if (here == nullptr) {
150  return nullptr;
151  }
152  if constexpr (std::is_same_v<headerType, Stack>) {
153  if (h.data() == nullptr) {
154  return here;
155  }
156  std::copy(h.data(), h.data() + h.size(), here);
157  BaseHeader* last = const_cast<BaseHeader*>(lastHeader(here));
158  if (!last)
159  return here;
160  last->flagsNextHeader = more;
161  return here + h.size();
162  } else if constexpr (std::is_same_v<BaseHeader, headerType>) {
163  std::copy(h.data(), h.data() + h.size(), here);
164  reinterpret_cast<BaseHeader*>(here)->flagsNextHeader = more;
165  return here + h.size();
166  } else if constexpr (std::is_base_of_v<BaseHeader, headerType>) {
167  ::new (static_cast<void*>(here)) headerType(std::forward<T>(h));
168  reinterpret_cast<BaseHeader*>(here)->flagsNextHeader = more;
169  return here + h.size();
170  } else if constexpr (std::is_same_v<headerType, o2::byte*>) {
171  BaseHeader* from{BaseHeader::get(h)};
172  BaseHeader* last{nullptr};
173  while (from) {
174  last = reinterpret_cast<BaseHeader*>(here);
175  std::copy(from->data(), from->data() + from->size(), here);
176  here += from->size();
177  from = from->next();
178  };
179  if (last)
180  last->flagsNextHeader = more;
181  return here;
182  } else {
183  static_assert(true, "Stack can only be constructed from other stacks and BaseHeader derived classes");
184  }
185  }
