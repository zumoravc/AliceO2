42 std::vector<DataProcessorSpec> defineDataProcessing(ConfigContext const& config)
43 {
44  std::string defaultTransportConfig = config.options().get<std::string>("default-transport");
45  int nRolls = config.options().get<int>("number-of-events");
46  if (defaultTransportConfig == "zeromq") {
47  // nothing to do for the moment
48  } else if (defaultTransportConfig == "shmem") {
49  // nothing to do for the moment
50  } else {
51  throw std::runtime_error("invalid argument for option --default-transport : '" + defaultTransportConfig + "'");
52  }
53  std::vector<DataProcessorSpec> workflow;
54 
56  // a producer process steered by a timer
57  //
58  // the compute callback of the producer
59  auto producerCallback = [nRolls, counter = std::make_shared<int>()](DataAllocator& outputs, ControlService& control) {
60  outputs.make<int>(OutputRef{"data", 0}) = *counter;
61  if (++(*counter) >= nRolls) {
62  control.endOfStream();
63  control.readyToQuit(QuitRequest::Me);
64  }
65  };
66 
67  workflow.emplace_back(DataProcessorSpec{"producer",
68  {InputSpec{"timer", "TST", "TIMER", 0, Lifetime::Timer}},
69  {OutputSpec{{"data"}, "TST", "DATA", 0, Lifetime::Timeframe}},
70  AlgorithmSpec{adaptStateless(producerCallback)},
71  {ConfigParamSpec{"period-timer", VariantType::Int, 100000, {"period of timer"}}}});
72 
74  // the dpl sink proxy process
75 
76  // use the OutputChannelSpec as a tool to create the default configuration for the out-of-band channel
77  OutputChannelSpec externalChannelSpec;
78  // Note: the name has to match the binding of the input spec
79  externalChannelSpec.name = "external";
80  externalChannelSpec.type = ChannelType::Push;
81  externalChannelSpec.method = ChannelMethod::Bind;
82  externalChannelSpec.hostname = "localhost";
83  externalChannelSpec.port = 42042;
84  externalChannelSpec.listeners = 0;
85  if (!defaultTransportConfig.empty()) {
86  if (defaultTransportConfig == "zeromq") {
87  externalChannelSpec.protocol = ChannelProtocol::Network;
88  } else if (defaultTransportConfig == "shmem") {
89  externalChannelSpec.protocol = ChannelProtocol::IPC;
90  }
91  }
92  std::string channelConfig = formatExternalChannelConfiguration(externalChannelSpec);
93  // at some point the formatting tool might add the transport as well so we have to check
94  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
95  channelConfig += ",transport=" + defaultTransportConfig;
96  }
97 
98  Inputs sinkInputs = {InputSpec{"external", "TST", "DATA", 0, Lifetime::Timeframe}};
99  workflow.emplace_back(std::move(specifyFairMQDeviceOutputProxy("dpl-sink", sinkInputs, channelConfig.c_str())));
100 
102  // a simple checker process subscribing to the output of the input proxy
103  //
104  // the compute callback of the checker
105  auto checkerCallback = [nRolls](InputRecord& inputs, ControlService& control) {
106  LOG(DEBUG) << "got inputs " << inputs.size();
107  if (inputs.get<int>("datain") == nRolls - 1) {
108  LOG(INFO) << "terminating after " << nRolls << " successful event(s)";
109  control.endOfStream();
110  control.readyToQuit(QuitRequest::All);
111  }
112  };
113 
114  // the checker process connects to the proxy
115  workflow.emplace_back(DataProcessorSpec{"checker",
116  {InputSpec{"datain", "PRX", "DATA", 0, Lifetime::Timeframe}},
117  {},
118  AlgorithmSpec{adaptStateless(checkerCallback)}});
119 
121  // the input proxy process
122  // reads the messages from the output proxy via the out-of-band channel
123 
124  // converter callback for the external FairMQ device proxy ProcessorSpec generator
125  auto converter = [](FairMQDevice& device, FairMQParts& inputs, ChannelRetriever channelRetriever) {
126  ASSERT_ERROR(inputs.Size() >= 2);
127  if (inputs.Size() < 2) {
128  return;
129  }
130  int msgidx = 0;
131  auto dh = o2::header::get<o2::header::DataHeader*>(inputs.At(msgidx)->GetData());
132  if (!dh) {
133  LOG(ERROR) << "data on input " << msgidx << " does not follow the O2 data model, DataHeader missing";
134  return;
135  }
136 
137  // Note: we want to run both the output and input proxy in the same workflow and thus we need
138  // different data identifiers and change the data origin in the forwarding
139  OutputSpec query{"PRX", dh->dataDescription, dh->subSpecification};
140  auto channelName = channelRetriever(query);
141  ASSERT_ERROR(!channelName.empty());
142  LOG(DEBUG) << "using channel '" << channelName << "' for " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification});
143  if (channelName.empty()) {
144  return;
145  }
146  // make a copy of the header message, get the data header and change origin
147  auto outHeaderMessage = device.NewMessageFor(channelName, 0, inputs.At(msgidx)->GetSize());
148  memcpy(outHeaderMessage->GetData(), inputs.At(msgidx)->GetData(), inputs.At(msgidx)->GetSize());
149  // this we obviously need to fix in the get API, const'ness of the returned header pointer
150  // should depend on const'ness of the buffer
151  auto odh = const_cast<o2::header::DataHeader*>(o2::header::get<o2::header::DataHeader*>(outHeaderMessage->GetData()));
152  odh->dataOrigin = o2::header::DataOrigin("PRX");
153  FairMQParts output;
154  output.AddPart(std::move(outHeaderMessage));
155  output.AddPart(std::move(inputs.At(msgidx + 1)));
156  LOG(DEBUG) << "sending " << DataSpecUtils::describe(OutputSpec{odh->dataOrigin, odh->dataDescription, odh->subSpecification});
157  o2::framework::sendOnChannel(device, output, channelName);
158  };
159 
160  // we use the same spec to build the configuration string, ideally we would have some helpers
161  // which convert an OutputChannelSpec to an InputChannelSpec replacing 'bind' <--> 'connect'
162  // and 'push' <--> 'pull'
163  //
164  // skip the name in the configuration string as it is added in specifyExternalFairMQDeviceProxy
165  externalChannelSpec.name = "";
166  externalChannelSpec.type = ChannelType::Pull;
167  externalChannelSpec.method = ChannelMethod::Connect;
168  channelConfig = formatExternalChannelConfiguration(externalChannelSpec);
169  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
170  channelConfig += ",transport=" + defaultTransportConfig;
171  }
172 
173  // Note: in order to make the DPL output proxy and an input proxy working in the same
174  // workflow, we use different data description
175  Outputs inputProxyOutputs = {OutputSpec{"PRX", "DATA", 0, Lifetime::Timeframe}};
176  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
177  "input-proxy",
178  std::move(inputProxyOutputs),
179  channelConfig.c_str(),
180  converter));
181 
182  return workflow;
183 }
