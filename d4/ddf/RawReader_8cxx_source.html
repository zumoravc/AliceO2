60 bool RawReader::addInputFile(std::string infile)
61 {
62  typedef boost::tokenizer<boost::char_separator<char>> tokenizer;
63  boost::char_separator<char> sep{":"};
64  tokenizer tok{infile, sep};
65  if (std::distance(tok.begin(), tok.end()) < 3) {
66  LOG(ERROR) << "Not enough arguments";
67  return false;
68  }
69 
70  tokenizer::iterator it1 = tok.begin();
71  std::string path = boost::lexical_cast<std::string>(*it1);
72  int run = std::stoi(path.substr(path.find("run") + 3, 6));
73 
74  int region, link;
75  try {
76  ++it1;
77  region = boost::lexical_cast<int>(*it1);
78  } catch (boost::bad_lexical_cast) {
79  LOG(ERROR) << "Please enter a region for " << path;
80  return false;
81  }
82 
83  try {
84  ++it1;
85  link = boost::lexical_cast<int>(*it1);
86  } catch (boost::bad_lexical_cast) {
87  LOG(ERROR) << "Please enter a link number for " << path;
88  return false;
89  }
90 
91  int sampaVersion = -1;
92  if (++it1 == tok.end()) {
93  LOG(DEBUG) << "No SAMPA version was entered, assuming latest.";
94  } else {
95  try {
96  sampaVersion = boost::lexical_cast<int>(*it1);
97  } catch (boost::bad_lexical_cast) {
98  LOG(ERROR) << "Please enter a valid SAMPA version for " << path;
99  return false;
100  }
101  }
102 
103  return addInputFile(region, link, sampaVersion, path, run);
104 }

106 bool RawReader::addInputFile(int region, int link, int sampaVersion, std::string path, int run)
107 {
108  if (mRun == -1)
109  mRun = run;
110  if (mRegion == -1)
111  mRegion = region;
112  if (mLink == -1)
113  mLink = link;
114  if (mSampaVersion == -1)
115  mSampaVersion = sampaVersion;
116 
117  if (run != mRun) {
118  LOG(DEBUG) << "Run of RawReader is " << mRun << " and not " << run;
119  return false;
120  }
121 
122  if (region != mRegion) {
123  LOG(DEBUG) << "Region of RawReader is " << mRegion << " and not " << region;
124  return false;
125  }
126 
127  if (link != mLink) {
128  LOG(DEBUG) << "Link of RawReader is " << mLink << " and not " << link;
129  return false;
130  }
131 
132  if (sampaVersion != mSampaVersion) {
133  LOG(DEBUG) << "SAMPA Version of RawReader is " << mSampaVersion << " and not " << sampaVersion;
134  return false;
135  }
136 
137  std::ifstream file(path);
138  if (!file.is_open()) {
139  LOG(ERROR) << "Can't read file " << path;
140  return false;
141  }
142 
143  Header h;
144  int pos = 0;
145  file.seekg(0, file.end);
146  int length = file.tellg();
147  file.seekg(0, file.beg);
148 
149  while (pos < length) {
150  file.read((char*)&h, sizeof(h));
151  if (h.reserved_01 != 0x0F || h.reserved_2() != 0x3fec2fec1fec0fec) {
152  LOG(ERROR) << "Header does not look consistent";
153  }
154  EventInfo eD;
155  eD.path = path;
156  eD.posInFile = file.tellg();
157  eD.header = h;
158  if (h.headerVersion == 1) {
159  // auto ins = mEvents.insert(std::make_pair(h.eventCount(),std::shared_ptr<std::vector<EventInfo>>(new std::vector<EventInfo>)));
160  auto ins = mEvents.insert(std::make_pair(h.eventCount(), std::make_shared<std::vector<EventInfo>>()));
161  ins.first->second->push_back(eD);
162 
163  if (h.eventCount() == 0) {
164  LOG(DEBUG) << "Processing event 0 to find sync pattern for event synchronization";
165  loadEvent(0); // processing first event to find the SYNC pattern (mode 1) or start of data (mode 2)
166  mLastEvent = -1;
167  }
168 
169  mEventSynchronizer->addEventOffset(
170  h.eventCount(),
171  (h.timeStamp() - mTimestampOfFirstData[0]));
172 
173  file.seekg(pos + (h.nWords * 4));
174  pos = file.tellg();
175  } else {
176  LOG(ERROR) << "Header version " << (int)h.headerVersion << " not implemented.";
177  return false;
178  }
179  }
180 
181  return true;
182 }

184 bool RawReader::loadEvent(int64_t event)
185 {
186  const Mapper& mapper = Mapper::instance();
187  const PartitionInfo& partInfo = mapper.getPartitionInfo(mRegion / 2);
188  LOG(DEBUG) << "Loading event " << event << " for Link " << (mRegion * partInfo.getNumberOfFECs()) + mLink;
189 
190  if (mLastEvent == -1 && event != getFirstEvent()) {
191  LOG(DEBUG) << "First event was not loaded (maybe important for decoding RAW GBT frames) loading first event.";
192  mSyncPos.fill(-1);
193  mTimestampOfFirstData.fill(0);
194  loadEvent(getFirstEvent());
195  LOG(DEBUG) << "Continue with event " << event;
196  }
197  mData.clear();
198 
199  auto ev = mEvents.find(event);
200 
201  if (ev == mEvents.end())
202  return false;
203  mLastEvent = event;
204 
205  for (auto& eventInfo : *(ev->second)) {
206  switch (eventInfo.header.dataType) {
207  case 1: // RAW GBT frames
208  {
209  LOG(DEBUG) << "Data of readout mode 1 (RAW GBT frames)";
210  if (!decodeRawGBTFrames(eventInfo))
211  return false;
212  break;
213  }
214  case 2: // Decoded data
215  {
216  LOG(DEBUG) << "Data of readout mode 2 (decoded data)";
217  if (!decodePreprocessedData(eventInfo))
218  return false;
219  break;
220  }
221  case 3: // both, RAW GBT frames and decoded data
222  {
223  if (mUseRawInMode3) {
224  LOG(DEBUG) << "Data of readout mode 3 (decoding RAW GBT frames)";
225  if (!decodeRawGBTFrames(eventInfo))
226  return false;
227  } else {
228  LOG(DEBUG) << "Data of readout mode 3 (using decoded data)";
229  if (!decodePreprocessedData(eventInfo))
230  return false;
231  }
232  break;
233  }
234  default: {
235  LOG(DEBUG) << "Readout mode not known";
236  break;
237  }
238  }
239  }
240 
241  mDataIterator = mData.begin();
242  LOG(DEBUG);
243  LOG(DEBUG);
244  return true;
245 }

247 bool RawReader::decodePreprocessedData(EventInfo eventInfo)
248 {
249  std::ifstream file(eventInfo.path);
250  if (!file.is_open()) {
251  LOG(ERROR) << "Can't read file " << eventInfo.path;
252  return false;
253  }
254 
255  int nWords = eventInfo.header.nWords - 8;
256  uint32_t words[nWords];
257  LOG(DEBUG) << "reading " << nWords << " words from position " << eventInfo.posInFile << " in file " << eventInfo.path;
258  file.seekg(eventInfo.posInFile);
259  file.read((char*)&words, nWords * sizeof(words[0]));
260 
261  const Mapper& mapper = Mapper::instance();
262 
263  std::array<char, 5> sampas;
264  std::array<char, 5> sampaChannelStart;
265  for (char i = 0; i < 5; ++i) {
266  sampas[i] = (i == 4) ? 2 : (mRegion % 2) ? i / 2 + 3 : i / 2;
267  sampaChannelStart[i] = (i == 4) ? // 5th half SAMPA corresponds to SAMPA2
268  ((mRegion % 2) ? 16 : 0)
269  : // every even CRU receives channel 0-15 from SAMPA 2, the odd ones channel 16-31
270  ((i % 2) ? 16 : 0); // every even half SAMPA containes channel 0-15, the odd ones channel 16-31
271  }
272 
273  std::array<uint32_t, 5> ids;
274  std::array<bool, 5> writeValue;
275  writeValue.fill(false);
276  std::array<std::array<uint16_t, 16>, 5> adcValues;
277 
278  int indexStep = (eventInfo.header.dataType == 3) ? 8 : 4;
279  int offset = (eventInfo.header.dataType == 3) ? 4 : 0;
280 
281  long i_start = 0;
282  if (eventInfo.header.eventCount() > 0) {
283  LOG(DEBUG) << "Using offset [" << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) << "] "
284  << "instead of [" << eventInfo.header.timeStamp() << "] "
285  << " (diff = " << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() << ") ";
286 
287  i_start = mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() - 1;
288  i_start *= indexStep;
289  }
290 
291  LOG(DEBUG) << "Start index for Event " << eventInfo.header.eventCount() << " is " << i_start;
292 
293  for (long i = i_start; i < nWords; i = i + indexStep) {
294  ids[4] = (words[i + offset] >> 4) & 0xF;
295  ids[3] = (words[i + offset] >> 8) & 0xF;
296  ids[2] = (words[i + offset] >> 12) & 0xF;
297  ids[1] = (words[i + offset] >> 16) & 0xF;
298  ids[0] = (words[i + offset] >> 20) & 0xF;
299 
300  adcValues[4][((ids[4] & 0x7) * 2) + 1] = (((ids[4] >> 3) & 0x1) == 0) ? 0 : words[i + offset + 3] & 0x3FF;
301  adcValues[4][((ids[4] & 0x7) * 2)] = (((ids[4] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 3] >> 10) & 0x3FF;
302  adcValues[3][((ids[3] & 0x7) * 2) + 1] = (((ids[3] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 3] >> 20) & 0x3FF;
303  adcValues[3][((ids[3] & 0x7) * 2)] = (((ids[3] >> 3) & 0x1) == 0) ? 0 : ((words[i + offset + 2] & 0xFF) << 2) | ((words[i + offset + 3] >> 30) & 0x3);
304  adcValues[2][((ids[2] & 0x7) * 2) + 1] = (((ids[2] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 2] >> 8) & 0x3FF;
305  adcValues[2][((ids[2] & 0x7) * 2)] = (((ids[2] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 2] >> 18) & 0x3FF;
306  adcValues[1][((ids[1] & 0x7) * 2) + 1] = (((ids[1] >> 3) & 0x1) == 0) ? 0 : ((words[i + offset + 1] & 0x3F) << 4) | ((words[i + offset + 2] >> 28) & 0xF);
307  adcValues[1][((ids[1] & 0x7) * 2)] = (((ids[1] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 1] >> 6) & 0x3FF;
308  adcValues[0][((ids[0] & 0x7) * 2) + 1] = (((ids[0] >> 3) & 0x1) == 0) ? 0 : (words[i + offset + 1] >> 16) & 0x3FF;
309  adcValues[0][((ids[0] & 0x7) * 2)] = (((ids[0] >> 3) & 0x1) == 0) ? 0 : ((words[i + offset + 0] & 0xF) << 6) | ((words[i + offset + 1] >> 26) & 0x3F);
310 
311  for (char j = 0; j < 5; ++j) {
312  if (ids[j] == 0x8) {
313  writeValue[j] = true;
314  // header TS is one before first word -> +1
315  if (mTimestampOfFirstData[j] == 0)
316  mTimestampOfFirstData[j] = eventInfo.header.timeStamp() + 1 + i / indexStep;
317  }
318  }
319 
320  for (char j = 0; j < 5; ++j) {
321  if (writeValue[j] & (ids[j] == 0xF)) {
322  for (int k = 0; k < 16; ++k) {
323  const PadPos padPos = mapper.padPosRegion(
324  mRegion, mLink, sampas[j], k + sampaChannelStart[j]);
325 
326  if (mApplyChannelMask && // channel mask should be applied
327  (mChannelMask != nullptr) && // channel mask is available
328  !mChannelMask->getValue(CRU(mRegion), padPos.getPad(), padPos.getRow())) { // channel mask
329  continue;
330  }
331 
332  // auto ins = mData.insert(std::make_pair(padPos, std::shared_ptr<std::vector<uint16_t>>(new std::vector<uint16_t>)));
333  auto ins = mData.insert(std::make_pair(padPos, std::make_shared<std::vector<uint16_t>>()));
334  ins.first->second->push_back(adcValues[j][k]);
335  }
336  }
337  }
338  }
339  return true;
340 }

342 bool RawReader::decodeRawGBTFrames(EventInfo eventInfo)
343 {
344  std::ifstream file(eventInfo.path);
345  if (!file.is_open()) {
346  LOG(ERROR) << "Can't read file " << eventInfo.path;
347  return false;
348  }
349 
350  int nWords = eventInfo.header.nWords - 8;
351  uint32_t words[nWords];
352  LOG(DEBUG) << "reading " << nWords << " words from position " << eventInfo.posInFile << " in file " << eventInfo.path;
353  LOG(DEBUG) << "Header time stamp is " << eventInfo.header.timeStamp();
354  file.seekg(eventInfo.posInFile);
355  file.read((char*)&words, nWords * sizeof(words[0]));
356 
357  const Mapper& mapper = Mapper::instance();
358 
359  std::array<char, 5> sampas;
360  std::array<char, 5> sampaChannelStart;
361  for (char i = 0; i < 5; ++i) {
362  sampas[i] = (i == 4) ? 2 : (mRegion % 2) ? i / 2 + 3 : i / 2;
363  sampaChannelStart[i] = (i == 4) ? // 5th half SAMPA corresponds to SAMPA2
364  ((mRegion % 2) ? 16 : 0)
365  : // every even CRU receives channel 0-15 from SAMPA 2, the odd ones channel 16-31
366  ((i % 2) ? 16 : 0); // every even half SAMPA containes channel 0-15, the odd ones channel 16-31
367  }
368 
369  std::array<SyncPatternMonitor, 5> syncMon{
370  SyncPatternMonitor((mRegion % 2) ? 3 : 0, 0),
371  SyncPatternMonitor((mRegion % 2) ? 3 : 0, 1),
372  SyncPatternMonitor((mRegion % 2) ? 4 : 1, 0),
373  SyncPatternMonitor((mRegion % 2) ? 4 : 1, 1),
374  SyncPatternMonitor(2, (mRegion % 2) ? 1 : 0)};
375  std::array<AdcClockMonitor, 3> adcClockMon{
376  AdcClockMonitor((mRegion % 2) ? 3 : 0),
377  AdcClockMonitor((mRegion % 2) ? 4 : 1),
378  AdcClockMonitor(2)};
379  std::array<bool, 3> adcClockFound{false, false, false};
380 
381  std::array<short, 5> lastSyncPos;
382  std::array<std::list<uint16_t>, 5> adcValues;
383  mAdcError->clear();
384  uint64_t timebin = 0;
385 
386  int indexStep = (eventInfo.header.dataType == 3) ? 8 : 4;
387 
388  long i_start = 0;
389  if (eventInfo.header.eventCount() > 0) {
390  LOG(DEBUG) << "Using offset [" << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) << "] "
391  << "instead of [" << eventInfo.header.timeStamp() << "] "
392  << " (diff = " << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() << ") ";
393 
394  i_start = mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() - 1;
395  i_start *= indexStep;
396  }
397 
398  LOG(DEBUG) << "Start index for Event " << eventInfo.header.eventCount() << " is " << i_start;
399 
400  GBTFrame frame;
401  if (eventInfo.header.eventCount() > 0)
402  frame.setData(words[i_start - indexStep], words[i_start - indexStep + 1], words[i_start - indexStep + 2], words[i_start - indexStep + 3]);
403  GBTFrame lastFrame;
404 
405  for (long i = i_start; i < nWords; i = i + indexStep) {
406 
407  lastSyncPos = mSyncPos;
408  lastFrame = frame;
409  frame.setData(words[i], words[i + 1], words[i + 2], words[i + 3]);
410 
411  if (syncMon[0].addSequence(frame.getHalfWord(0, 0, 0), frame.getHalfWord(0, 1, 0), frame.getHalfWord(0, 2, 0), frame.getHalfWord(0, 3, 0)))
412  mSyncPos[0] = syncMon[0].getPosition();
413  if (syncMon[1].addSequence(frame.getHalfWord(0, 0, 1), frame.getHalfWord(0, 1, 1), frame.getHalfWord(0, 2, 1), frame.getHalfWord(0, 3, 1)))
414  mSyncPos[1] = syncMon[1].getPosition();
415  if (syncMon[2].addSequence(frame.getHalfWord(1, 0, 0), frame.getHalfWord(1, 1, 0), frame.getHalfWord(1, 2, 0), frame.getHalfWord(1, 3, 0)))
416  mSyncPos[2] = syncMon[2].getPosition();
417  if (syncMon[3].addSequence(frame.getHalfWord(1, 0, 1), frame.getHalfWord(1, 1, 1), frame.getHalfWord(1, 2, 1), frame.getHalfWord(1, 3, 1)))
418  mSyncPos[3] = syncMon[3].getPosition();
419  if (syncMon[4].addSequence(frame.getHalfWord(2, 0), frame.getHalfWord(2, 1), frame.getHalfWord(2, 2), frame.getHalfWord(2, 3)))
420  mSyncPos[4] = syncMon[4].getPosition();
421 
422  if (mCheckAdcClock) {
423 
424  bool adcCheckErr0 = adcClockMon[0].addSequence(frame.getAdcClock(0));
425  bool adcCheckErr1 = adcClockMon[1].addSequence(frame.getAdcClock(1));
426  bool adcCheckErr2 = adcClockMon[2].addSequence(frame.getAdcClock(2));
427 
428  if (mSyncPos[0] >= 0)
429  adcClockFound[0] = adcClockFound[0] | (!adcCheckErr0);
430  if (mSyncPos[2] >= 0)
431  adcClockFound[1] = adcClockFound[1] | (!adcCheckErr1);
432  if (mSyncPos[4] >= 0)
433  adcClockFound[2] = adcClockFound[2] | (!adcCheckErr2);
434  if (adcClockFound[0] & adcCheckErr0) {
435  adcClockFound[0] = false;
436  LOG(DEBUG) << "ADC clock error of SAMPA " << ((mRegion % 2) ? 3 : 0) << " in frame [" << i / indexStep << "]";
437  mAdcError->emplace_back(std::make_tuple((mRegion % 2) ? 3 : 0, i, mSyncPos[0]));
438  };
439  if (adcClockFound[1] & adcCheckErr1) {
440  adcClockFound[1] = false;
441  LOG(DEBUG) << "ADC clock error of SAMPA " << ((mRegion % 2) ? 4 : 1) << " in frame [" << i / indexStep << "]";
442  mAdcError->emplace_back(std::make_tuple((mRegion % 2) ? 4 : 1, i, mSyncPos[2]));
443  };
444  if (adcClockFound[2] & adcCheckErr2) {
445  adcClockFound[2] = false;
446  LOG(DEBUG) << "ADC clock error of SAMPA " << 2 << " in frame [" << i / indexStep << "]";
447  mAdcError->emplace_back(std::make_tuple(2, i, mSyncPos[4]));
448  };
449  }
450 
451  short value1;
452  short value2;
453  for (short iHalfSampa = 0; iHalfSampa < 5; ++iHalfSampa) {
454  if (mSyncPos[iHalfSampa] < 0) {
455  LOG(DEBUG1) << "Sync pattern for half SAMPA " << iHalfSampa << " not yet found";
456  continue;
457  }
458  if (lastSyncPos[iHalfSampa] < 0) {
459  LOG(DEBUG1) << "Sync pattern for half SAMPA " << iHalfSampa << " not yet found";
460  LOG(DEBUG1) << lastFrame;
461  continue;
462  }
463  if (mTimestampOfFirstData[iHalfSampa] == 0) {
464  if (iHalfSampa == 0)
465  LOG(DEBUG) << "Setting timestamp of first data to " << eventInfo.header.timeStamp() + 1 + i / indexStep << " for half SAMPA " << iHalfSampa << " (" << 1 + i / indexStep << ")";
466  mTimestampOfFirstData[iHalfSampa] = eventInfo.header.timeStamp() + 1 + i / indexStep;
467  }
468 
469  switch (mSyncPos[iHalfSampa]) {
470  case 0:
471  value1 = (frame.getHalfWord(iHalfSampa / 2, 1, iHalfSampa % 2) << 5) | frame.getHalfWord(iHalfSampa / 2, 0, iHalfSampa % 2);
472  value2 = (frame.getHalfWord(iHalfSampa / 2, 3, iHalfSampa % 2) << 5) | frame.getHalfWord(iHalfSampa / 2, 2, iHalfSampa % 2);
473  break;
474 
475  case 1:
476  value1 = (lastFrame.getHalfWord(iHalfSampa / 2, 2, iHalfSampa % 2) << 5) | lastFrame.getHalfWord(iHalfSampa / 2, 1, iHalfSampa % 2);
477  value2 = (frame.getHalfWord(iHalfSampa / 2, 0, iHalfSampa % 2) << 5) | lastFrame.getHalfWord(iHalfSampa / 2, 3, iHalfSampa % 2);
478  break;
479 
480  case 2:
481  value1 = (lastFrame.getHalfWord(iHalfSampa / 2, 3, iHalfSampa % 2) << 5) | lastFrame.getHalfWord(iHalfSampa / 2, 2, iHalfSampa % 2);
482  value2 = (frame.getHalfWord(iHalfSampa / 2, 1, iHalfSampa % 2) << 5) | frame.getHalfWord(iHalfSampa / 2, 0, iHalfSampa % 2);
483  break;
484 
485  case 3:
486  value1 = (frame.getHalfWord(iHalfSampa / 2, 0, iHalfSampa % 2) << 5) | lastFrame.getHalfWord(iHalfSampa / 2, 3, iHalfSampa % 2);
487  value2 = (frame.getHalfWord(iHalfSampa / 2, 2, iHalfSampa % 2) << 5) | frame.getHalfWord(iHalfSampa / 2, 1, iHalfSampa % 2);
488  break;
489 
490  default:
491  return false;
492  }
493 
494  if (mSampaVersion == 1 || mSampaVersion == 2) {
495  adcValues[iHalfSampa].emplace_back(value1 ^ (1 << 9)); // Invert bit 9 vor SAMPA v1 and v2
496  adcValues[iHalfSampa].emplace_back(value2 ^ (1 << 9)); // Invert bit 9 vor SAMPA v1 and v2
497  } else {
498  adcValues[iHalfSampa].emplace_back(value1);
499  adcValues[iHalfSampa].emplace_back(value2);
500  }
501  }
502 
503  for (char j = 0; j < 5; ++j) {
504  if (adcValues[j].size() == 16) {
505  for (int k = 0; k < 16; ++k) {
