51 std::string MergerInfrastructureBuilder::validateConfig()
52 {
53  std::string error;
54  const std::string preamble = "mergers::MergerInfrastructureBuilder error: ";
55  if (mInfrastructureName.empty()) {
56  error += preamble + "the infrastructure name is empty\n";
57  }
58  if (mInputs.empty()) {
59  error += preamble + "no inputs specified\n";
60  }
61  if (DataSpecUtils::validate(mOutputSpec) == false) {
62  error += preamble + "invalid output\n";
63  }
64 
65  if (mConfig.topologySize.value == TopologySize::NumberOfLayers && mConfig.topologySize.param < 1) {
66  error += preamble + "number of layers less than 1 (" + std::to_string(mConfig.topologySize.param) + ")\n";
67  }
68  if (mConfig.topologySize.value == TopologySize::ReductionFactor && mConfig.topologySize.param < 2) {
69  error += preamble + "reduction factor smaller than 2 (" + std::to_string(mConfig.topologySize.param) + ")\n";
70  }
71 
72  if (mConfig.inputObjectTimespan.value == InputObjectsTimespan::FullHistory && mConfig.mergedObjectTimespan.value == MergedObjectTimespan::LastDifference) {
73  error += preamble + "MergedObjectTimespan::LastDifference does not apply to InputObjectsTimespan::FullHistory\n";
74  }
75 
76  for (const auto& input : mInputs) {
77  if (DataSpecUtils::match(input, mOutputSpec)) {
78  error += preamble + "output '" + DataSpecUtils::label(mOutputSpec) + "' matches input '" + DataSpecUtils::label(input) + "'. That will cause a circular dependency!";
79  }
80  }
81 
82  return error;
83 }

85 framework::WorkflowSpec MergerInfrastructureBuilder::generateInfrastructure()
86 {
87  if (std::string error = validateConfig(); !error.empty()) {
88  throw std::runtime_error(error);
89  }
90 
91  framework::WorkflowSpec workflow;
92  auto layerInputs = mInputs;
93 
94  // preparing some numbers
95  auto mergersPerLayer = computeNumberOfMergersPerLayer(layerInputs.size());
96 
97  // actual topology generation
98  MergerBuilder mergerBuilder;
99  mergerBuilder.setName(mInfrastructureName);
100  for (size_t layer = 1; layer < mergersPerLayer.size(); layer++) {
101 
102  size_t numberOfMergers = mergersPerLayer[layer];
103  size_t inputsPerMerger = layerInputs.size() / numberOfMergers;
104  size_t inputsPerMergerRemainder = layerInputs.size() % numberOfMergers;
105 
106  MergerConfig layerConfig = mConfig;
107  if (layer > 1 && mConfig.inputObjectTimespan.value == InputObjectsTimespan::LastDifference) {
108  layerConfig.inputObjectTimespan = {InputObjectsTimespan::FullHistory}; // in LastDifference mode only the first layer should integrate
109  layerConfig.mergedObjectTimespan = {MergedObjectTimespan::LastDifference}; // and objects that are merged should not be used again
110  }
111  mergerBuilder.setConfig(layerConfig);
112 
113  framework::Inputs nextLayerInputs;
114  auto inputsRangeBegin = layerInputs.begin();
115 
116  for (size_t m = 0; m < numberOfMergers; m++) {
117 
118  mergerBuilder.setTopologyPosition(layer, m);
119 
120  auto inputsRangeEnd = inputsRangeBegin + inputsPerMerger + (m < inputsPerMergerRemainder);
121  mergerBuilder.setInputSpecs(framework::Inputs(inputsRangeBegin, inputsRangeEnd));
122  inputsRangeBegin = inputsRangeEnd;
123 
124  if (numberOfMergers == 1) {
125  assert(layer == mergersPerLayer.size() - 1);
126  // the last layer => use the specified external OutputSpec
127  mergerBuilder.setOutputSpec(mOutputSpec);
128  }
129 
130  auto merger = mergerBuilder.buildSpec();
131 
132  auto input = DataSpecUtils::matchingInput(merger.outputs.at(0));
133  input.binding = "in";
134  nextLayerInputs.push_back(input);
135 
136  workflow.emplace_back(std::move(merger));
137  }
138  layerInputs = nextLayerInputs; //todo: could be optimised with pointers
139  }
140 
141  return workflow;
142 }

144 std::vector<size_t> MergerInfrastructureBuilder::computeNumberOfMergersPerLayer(const size_t inputs) const
145 {
146  std::vector<size_t> mergersPerLayer{inputs};
147  if (mConfig.topologySize.value == TopologySize::NumberOfLayers) {
148  // _ _
149  // | L - i | where:
150  // | ----- | L - number of layers
151  // |V| --- | |V| L | i - layer index (0 - input layer)
152  // | |i --- | | |0 | M_i - number of mergers in i layer
153  // | |
154  //
155 
156  size_t L = mConfig.topologySize.param;
157  for (size_t i = 1; i <= L; i++) {
158  mergersPerLayer.push_back(static_cast<size_t>(ceil(pow(inputs, (L - i) / static_cast<double>(L)))));
159  }
160 
161  } else { // mConfig.topologySize.value == TopologySize::ReductionFactor
162  // _ _
163  // | |V| | where:
164  // |V| --- | | |i-1 | R - reduction factor
165  // | |i --- | -------- | i - layer index (0 - input layer)
166  // | R | M_i - number of mergers in i layer
167  //
168 
169  double R = mConfig.topologySize.param;
170  size_t Mi, prevMi = inputs;
171  do {
172  Mi = static_cast<size_t>(ceil(prevMi / R));
173  mergersPerLayer.push_back(Mi);
174  prevMi = Mi;
175  } while (Mi > 1);
176  }
177 
178  return mergersPerLayer;
179 }
