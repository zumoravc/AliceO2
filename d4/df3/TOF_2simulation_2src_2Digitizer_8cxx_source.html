29 // How data acquisition works in real data
30 /*
31  |<----------- 1 orbit ------------->|
32  ------|-----------|-----------|-----------|------
33  ^ ^ ^ ^ when triggers happen
34  |<--- latency ---|
35  |<- matching1->|
36  |<- matching2->|
37  |<- matching3->|
38  |<>| = overlap between two consecutive matching
39 
40 Norbit = number of orbits elapsed
41 Nbunch = bunch in the current orbit (0:3563)
42 Ntdc = number of tdc counts within the matching window --> for 1/3 orbit (0:3649535)
43 
44 raw time = trigger time (Norbit and Nbunch) - latency window + TDC(Ntdc)
45  */
46 
47 // What we implemented here (so far)
48 /*
49  |<----------- 1 orbit ------------->|
50  ------|-----------|-----------|-----------|------
51  |<- matching1->|
52  |<- matching2->|
53  |<- matching3->|
54  |<>| = overlap between two consecutive matching windows
55 
56 - OVERLAP between two consecutive windows: implemented, to be extensively checked
57 - NO LATENCY WINDOW (we manage it during raw encoding/decoding) then digits already corrected
58 
59 NBC = Number of bunch since timeframe beginning = Norbit*3564 + Nbunch
60 Ntdc = here within the current BC -> (0:1023)
61 
62 digit time = NBC*1024 + Ntdc
63  */

86 int Digitizer::process(const std::vector<HitType>* hits, std::vector<Digit>* digits)
87 {
88  // hits array of TOF hits for a given simulated event
89  // digits passed from external to be filled, in continuous readout mode we will push it on mDigitsPerTimeFrame vector of vectors of digits
90 
91  // printf("process event time = %f with %ld hits\n",mEventTime,hits->size());
92 
93  Int_t readoutwindow = Int_t((mEventTime - Geo::BC_TIME * (Geo::OVERLAP_IN_BC + 2)) * Geo::READOUTWINDOW_INV); // event time shifted by 2 BC as safe margin before to change current readout window to account for decalibration
94 
95  if (mContinuous && readoutwindow > mReadoutWindowCurrent) { // if we are moving in future readout windows flush previous ones (only for continuous readout mode)
96  digits->clear();
97 
98  for (; mReadoutWindowCurrent < readoutwindow;) { // mReadoutWindowCurrent incremented in fillOutputContainer!!!!
99  fillOutputContainer(*digits); // fill all windows which are before (not yet stored) of the new current one
100  checkIfReuseFutureDigits();
101  } // close loop readout window
102  } // close if continuous
103 
104  for (auto& hit : *hits) {
105  //TODO: put readout window counting/selection
106 
107  processHit(hit, mEventTime);
108  } // end loop over hits
109 
110  if (!mContinuous) { // fill output container per event
111  digits->clear();
112  fillOutputContainer(*digits);
113  }
114 
115  return 0;
116 }

120 Int_t Digitizer::processHit(const HitType& hit, Double_t event_time)
121 {
122  Float_t pos[3] = {hit.GetX(), hit.GetY(), hit.GetZ()};
123  Float_t deltapos[3];
124  Int_t detInd[5];
125  Int_t detIndOtherPad[5];
126 
127  Geo::getPadDxDyDz(pos, detInd, deltapos); // Get DetId and residuals
128 
129  detIndOtherPad[0] = detInd[0], detIndOtherPad[1] = detInd[1],
130  detIndOtherPad[2] = detInd[2]; // same sector, plate, strip
131 
132  Int_t otherraw = 0;
133  if (detInd[3] == 0)
134  otherraw = 1;
135 
136  Int_t iZshift = otherraw ? 1 : -1;
137 
138  Int_t channel = Geo::getIndex(detInd);
139 
140  Float_t charge = getCharge(hit.GetEnergyLoss());
141  // NOTE: FROM NOW ON THE TIME IS IN PS ... AND NOT IN NS
142  Double_t time = getShowerTimeSmeared((double(hit.GetTime()) + event_time) * 1E3, charge);
143 
144  Float_t xLocal = deltapos[0];
145  Float_t zLocal = deltapos[2];
146 
147  // extract trackID
148  auto trackID = hit.GetTrackID();
149 
150  // PadId - Pad Identifier
151  // E | F --> PadId = 5 | 6
152  // A | B --> PadId = 1 | 2
153  // C | D --> PadId = 3 | 4
154 
155  Int_t ndigits = 0; //Number of digits added
156 
157  UInt_t istrip = channel / Geo::NPADS;
158 
159  // check the fired PAD 1 (A)
160  if (isFired(xLocal, zLocal, charge)) {
161  ndigits++;
162  addDigit(channel, istrip, time, xLocal, zLocal, charge, 0, 0, detInd[3], trackID);
163  }
164 
165  // check PAD 2
166  detIndOtherPad[3] = otherraw;
167  detIndOtherPad[4] = detInd[4];
168  channel = Geo::getIndex(detIndOtherPad);
169  xLocal = deltapos[0]; // recompute local coordinates
170  if (otherraw)
171  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
172  else
173  zLocal = deltapos[2] + Geo::ZPAD;
174  if (isFired(xLocal, zLocal, charge)) {
175  ndigits++;
176  addDigit(channel, istrip, time, xLocal, zLocal, charge, 0, iZshift, detInd[3], trackID);
177  }
178 
179  // check PAD 3
180  detIndOtherPad[3] = detInd[3];
181  detIndOtherPad[4] = detInd[4] - 1;
182  if (detIndOtherPad[4] >= 0) {
183  channel = Geo::getIndex(detIndOtherPad);
184  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
185  zLocal = deltapos[2]; // recompute local coordinates
186  if (isFired(xLocal, zLocal, charge)) {
187  ndigits++;
188  addDigit(channel, istrip, time, xLocal, zLocal, charge, -1, 0, detInd[3], trackID);
189  }
190  }
191 
192  // check PAD 5
193  detIndOtherPad[3] = detInd[3];
194  detIndOtherPad[4] = detInd[4] + 1;
195  if (detIndOtherPad[4] < Geo::NPADX) {
196  channel = Geo::getIndex(detIndOtherPad);
197  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
198  zLocal = deltapos[2]; // recompute local coordinates
199  if (isFired(xLocal, zLocal, charge)) {
200  ndigits++;
201  addDigit(channel, istrip, time, xLocal, zLocal, charge, 1, 0, detInd[3], trackID);
202  }
203  }
204 
205  // check PAD 4
206  detIndOtherPad[3] = otherraw;
207  detIndOtherPad[4] = detInd[4] - 1;
208  if (detIndOtherPad[4] >= 0) {
209  channel = Geo::getIndex(detIndOtherPad);
210  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
211  if (otherraw)
212  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
213  else
214  zLocal = deltapos[2] + Geo::ZPAD;
215  if (isFired(xLocal, zLocal, charge)) {
216  ndigits++;
217  addDigit(channel, istrip, time, xLocal, zLocal, charge, -1, iZshift, detInd[3], trackID);
218  }
219  }
220 
221  // check PAD 6
222  detIndOtherPad[3] = otherraw;
223  detIndOtherPad[4] = detInd[4] + 1;
224  if (detIndOtherPad[4] < Geo::NPADX) {
225  channel = Geo::getIndex(detIndOtherPad);
226  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
227  if (otherraw)
228  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
229  else
230  zLocal = deltapos[2] + Geo::ZPAD;
231  if (isFired(xLocal, zLocal, charge)) {
232  ndigits++;
233  addDigit(channel, istrip, time, xLocal, zLocal, charge, 1, iZshift, detInd[3], trackID);
234  }
235  }
236  return ndigits;
237 }

240 void Digitizer::addDigit(Int_t channel, UInt_t istrip, Double_t time, Float_t x, Float_t z, Float_t charge, Int_t iX, Int_t iZ,
241  Int_t padZfired, Int_t trackID)
242 {
243  // TOF digit requires: channel, time and time-over-threshold
244 
245  time = getDigitTimeSmeared(time, x, z, charge); // add time smearing
246 
247  charge *= getFractionOfCharge(x, z);
248  Float_t tot = 12; // time-over-threshold
249 
250  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
251  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
252  Float_t border = TMath::Min(xborder, zborder);
253 
254  Float_t timewalkX = x * mTimeWalkeSlope;
255  Float_t timewalkZ = (z - (padZfired - 0.5) * Geo::ZPAD) * mTimeWalkeSlope;
256 
257  if (border < 0) { // keep the effect onlu if hit out of pad
258  border *= -1;
259  Float_t extraTimeSmear = border * mTimeSlope;
260  time += gRandom->Gaus(mTimeDelay, extraTimeSmear);
261  } else {
262  border = 1 - border;
263  // if(border > 0) printf("deltat =%f\n",mTimeDelay*border*border*border);
264  // else printf("deltat=0\n");
265  // getchar();
266  if (border > 0)
267  time += mTimeDelay * border * border * border;
268  }
269  time += TMath::Sqrt(timewalkX * timewalkX + timewalkZ * timewalkZ) - mTimeDelayCorr - mTimeWalkeSlope * 2;
270 
271  // Decalibrate
272  time -= mCalibApi->getTimeDecalibration(channel, tot); //TODO: to be checked that "-" is correct, and we did not need "+" instead :-)
273 
274  Int_t nbc = Int_t(time * Geo::BC_TIME_INPS_INV); // time elapsed in number of bunch crossing
275  //Digit newdigit(time, channel, (time - Geo::BC_TIME_INPS * nbc) * Geo::NTDCBIN_PER_PS, tot * Geo::NTOTBIN_PER_NS, nbc);
276 
277  int tdc = int((time - Geo::BC_TIME_INPS * nbc) * Geo::NTDCBIN_PER_PS);
278 
279  // additional check to avoid very rare truncation
280  if (tdc < 0) {
281  nbc--;
282  tdc += 1024;
283  } else if (tdc >= 1024) {
284  nbc++;
285  tdc -= 1024;
286  }
287 
288  // printf("tdc = %d\n",tdc);
289 
290  int lblCurrent = 0;
291 
292  bool iscurrent = true; // if we are in the current readout window
293  Int_t isnext = -1;
294  Int_t isIfOverlap = -1;
295 
296  if (mContinuous) {
297  isnext = nbc / Geo::BC_IN_WINDOW - mReadoutWindowCurrent;
298  // isnext = Int_t(time * 1E-3 * Geo::READOUTWINDOW_INV) - mReadoutWindowCurrent; // to be replaced with uncalibrated time
299  isIfOverlap = (nbc - Geo::OVERLAP_IN_BC) / Geo::BC_IN_WINDOW - mReadoutWindowCurrent;
300  // isIfOverlap = Int_t((time - Geo::BC_TIME_INPS * Geo::OVERLAP_IN_BC) * 1E-3 * Geo::READOUTWINDOW_INV) - mReadoutWindowCurrent; // to be replaced with uncalibrated time;
301 
302  if (isnext == isIfOverlap)
303  isIfOverlap = -1;
304  else if (isnext < 0 && isIfOverlap >= 0) {
305  isnext = isIfOverlap;
306  isIfOverlap = -1;
307  } else if (isnext >= MAXWINDOWS && isIfOverlap < MAXWINDOWS) {
308  isnext = isIfOverlap;
309  isIfOverlap = MAXWINDOWS;
310  }
311 
312  if (isnext < 0) {
313  LOG(ERROR) << "error: isnext =" << isnext << "(current window = " << mReadoutWindowCurrent << ")"
314  << " nbc = " << nbc << " -- event time = " << mEventTime << "\n";
315 
316  return;
317  }
318 
319  if (isnext < 0 || isnext >= MAXWINDOWS) {
320 
321  lblCurrent = mFutureIevent.size(); // this is the size of mHeaderArray;
322  mFutureIevent.push_back(mEventID);
323  mFutureIsource.push_back(mSrcID);
324  mFutureItrackID.push_back(trackID);
325 
326  // fill temporary digits array
327  insertDigitInFuture(channel, tdc, tot * Geo::NTOTBIN_PER_NS, nbc, lblCurrent);
328  return; // don't fill if doesn't match any available readout window
329  } else if (isIfOverlap == MAXWINDOWS) { // add in future digits but also in one of the current readout windows (beacuse of windows overlap)
330  lblCurrent = mFutureIevent.size();
331  mFutureIevent.push_back(mEventID);
332  mFutureIsource.push_back(mSrcID);
333  mFutureItrackID.push_back(trackID);
334 
335  // fill temporary digits array
336  insertDigitInFuture(channel, tdc, tot * Geo::NTOTBIN_PER_NS, nbc, lblCurrent);
337  }
338 
339  if (isnext)
340  iscurrent = false;
341  }
342 
343  //printf("add TOF digit c=%i n=%i\n",iscurrent,isnext);
344 
345  std::vector<Strip>* strips;
346  o2::dataformats::MCTruthContainer<o2::tof::MCLabel>* mcTruthContainer;
347 
348  if (iscurrent) {
349  strips = mStripsCurrent;
350  mcTruthContainer = mMCTruthContainerCurrent;
351  } else {
352  strips = mStripsNext[isnext - 1];
353  mcTruthContainer = mMCTruthContainerNext[isnext - 1];
354  }
355 
356  int eventcounter = mReadoutWindowCurrent + isnext;
357  int hittimeTDC = (nbc - eventcounter * Geo::BC_IN_WINDOW) * 1024 + tdc; // time in TDC bin within the TOF WINDOW
358  if (hittimeTDC < 0)
359  LOG(ERROR) << "1) Negative hit " << hittimeTDC << ", something went wrong in filling readout window: isnext=" << isnext << ", isIfOverlap=" << isIfOverlap;
360  else
361  fillDigitsInStrip(strips, mcTruthContainer, channel, tdc, tot, nbc, istrip, trackID, mEventID, mSrcID);
362 
363  if (isIfOverlap > -1 && isIfOverlap < MAXWINDOWS) { // fill also a second readout window because of the overlap
364  if (!isIfOverlap) {
365  strips = mStripsCurrent;
366  mcTruthContainer = mMCTruthContainerCurrent;
367  } else {
368  strips = mStripsNext[isIfOverlap - 1];
369  mcTruthContainer = mMCTruthContainerNext[isIfOverlap - 1];
370  }
371 
372  int eventcounter = mReadoutWindowCurrent + isIfOverlap;
373  int hittimeTDC = (nbc - eventcounter * Geo::BC_IN_WINDOW) * 1024 + tdc; // time in TDC bin within the TOF WINDOW
374  if (hittimeTDC < 0)
375  LOG(ERROR) << "2) Negative hit " << hittimeTDC << ", something went wrong in filling readout window: isnext=" << isnext << ", isIfOverlap=" << isIfOverlap;
376  else
377  fillDigitsInStrip(strips, mcTruthContainer, channel, tdc, tot, nbc, istrip, trackID, mEventID, mSrcID);
378  }
379 }

381 void Digitizer::fillDigitsInStrip(std::vector<Strip>* strips, o2::dataformats::MCTruthContainer<o2::tof::MCLabel>* mcTruthContainer, int channel, int tdc, int tot, int nbc, UInt_t istrip, Int_t trackID, Int_t eventID, Int_t sourceID)
382 {
383  int lblCurrent;
384  if (mcTruthContainer) {
385  lblCurrent = mcTruthContainer->getIndexedSize(); // this is the size of mHeaderArray;
386  }
387 
388  Int_t lbl = (*strips)[istrip].addDigit(channel, tdc, tot * Geo::NTOTBIN_PER_NS, nbc, lblCurrent);
389 
390  if (mcTruthContainer) {
391  if (lbl == lblCurrent) { // it means that the digit was a new one --> we have to add the info in the MC container
392  o2::tof::MCLabel label(trackID, eventID, sourceID, tdc);
393  mcTruthContainer->addElement(lbl, label);
394  } else {
395  o2::tof::MCLabel label(trackID, eventID, sourceID, tdc);
396  mcTruthContainer->addElementRandomAccess(lbl, label);
397 
398  // sort the labels according to increasing tdc value
399  auto labels = mcTruthContainer->getLabels(lbl);
400  std::sort(labels.begin(), labels.end(),
401  [](o2::tof::MCLabel a, o2::tof::MCLabel b) { return a.getTDC() < b.getTDC(); });
402  }
403  }

431 Bool_t Digitizer::isFired(Float_t x, Float_t z, Float_t charge)
432 {
433  if (TMath::Abs(x) > Geo::XPAD * 0.5 + 0.3)
434  return kFALSE;
435  if (TMath::Abs(z) > Geo::ZPAD * 0.5 + 0.3)
436  return kFALSE;
437 
438  Float_t effX = getEffX(x);
439  Float_t effZ = getEffZ(z);
440 
441  Float_t efficiency = TMath::Min(effX, effZ);
442 
443  if (gRandom->Rndm() > efficiency)
444  return kFALSE;
445 
446  return kTRUE;
447 }

449 //______________________________________________________________________
450 Float_t Digitizer::getEffX(Float_t x)
451 {
452  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
453 
454  if (xborder > 0) {
455  if (xborder > mBound1)
456  return mEffCenter;
457  else if (xborder > mBound2)
458  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (xborder - mBound2) / (mBound1 - mBound2);
459  else
460  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * xborder / mBound2;
461  } else {
462  xborder *= -1;
463  if (xborder > mBound4)
464  return 0;
465  else if (xborder > mBound3)
466  return mEffBoundary3 - mEffBoundary3 * (xborder - mBound3) / (mBound4 - mBound3);
467  else
468  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * xborder / mBound3;
469  }
470 
471  return 0;
472 }

474 //______________________________________________________________________
475 Float_t Digitizer::getEffZ(Float_t z)
476 {
477  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
478 
479  if (zborder > 0) {
480  if (zborder > mBound1)
481  return mEffCenter;
482  else if (zborder > mBound2)
483  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (zborder - mBound2) / (mBound1 - mBound2);
484  else
485  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * zborder / mBound2;
486  } else {
487  zborder *= -1;
488  if (zborder > mBound4)
489  return 0;
490  else if (zborder > mBound3)
491  return mEffBoundary3 - mEffBoundary3 * (zborder - mBound3) / (mBound4 - mBound3);
492  else
493  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * zborder / mBound3;
494  }
495 
496  return 0;
497 }

502 void Digitizer::initParameters()
503 {
504  // boundary references for interpolation of efficiency and resolution
505  mBound1 = 0.4; // distance from border when efficiency starts to decrese
506  mBound2 = 0.15; // second step in the fired pad
507  mBound3 = 0.55; // distance from border (not fired pad)
508  mBound4 = 0.9; // distance from border (not fired pad) when efficiency vanishes
509 
510  // resolution parameters
511  mTOFresolution = 60; // TOF global resolution in ps
512  mShowerResolution = 50; // smearing correlated for all digits of the same hit
513  if (mTOFresolution > mShowerResolution)
514  mDigitResolution = TMath::Sqrt(mTOFresolution * mTOFresolution -
515  mShowerResolution * mShowerResolution); // independent smearing for each digit
516  else {
517  mShowerResolution = mTOFresolution;
518  mDigitResolution = 0;
519  }
520 
521  mTimeSlope = 100; // ps/cm extra smearing if hit out of pad propto the distance from the border
522  mTimeDelay = 70; // time delay if hit out of pad
523  mTimeWalkeSlope = 40; // ps/cm
524 
525  if (mMode == 0) {
526  mTimeSlope = 0;
527  mTimeDelay = 0;
528  mTimeWalkeSlope = 0;
529  }
530 
531  mTimeDelayCorr = mTimeDelay / 3.5;
532  if (mShowerResolution > mTimeDelayCorr)
533  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeDelayCorr * mTimeDelayCorr);
534  else
535  mShowerResolution = 0;
536 
537  if (mShowerResolution > mTimeWalkeSlope * 0.8)
538  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeWalkeSlope * mTimeWalkeSlope * 0.64);
539  else
540  mShowerResolution = 0;
541 
542  // efficiency parameters
543  mEffCenter = 0.995; // efficiency in the center of the fired pad
544  mEffBoundary1 = 0.94; // efficiency in mBound2
545  mEffBoundary2 = 0.833; // efficiency in the pad border
546  mEffBoundary3 = 0.1; // efficiency in mBound3
547 }

864 void Digitizer::checkIfReuseFutureDigits()
865 {
866  // check if digits stored very far in future match the new readout windows currently available
867  int idigit = mFutureDigits.size() - 1;
868 
869  int bclimit = 999999; // if bc is larger than this value stop the search in the next loop since bc are ordered in descending order
870 
871  for (std::vector<Digit>::reverse_iterator digit = mFutureDigits.rbegin(); digit != mFutureDigits.rend(); ++digit) {
872 
873  if (digit->getBC() > bclimit)
874  break;
875 
876  double timestamp = digit->getBC() * Geo::BC_TIME + digit->getTDC() * Geo::TDCBIN * 1E-3; // in ns
877  int isnext = Int_t(timestamp * Geo::READOUTWINDOW_INV) - (mReadoutWindowCurrent + 1); // to be replaced with uncalibrated time
878  int isIfOverlap = Int_t((timestamp - Geo::BC_TIME_INPS * Geo::OVERLAP_IN_BC * 1E-3) * Geo::READOUTWINDOW_INV) - (mReadoutWindowCurrent + 1); // to be replaced with uncalibrated time;
879 
880  if (isnext == isIfOverlap)
881  isIfOverlap = -1;
882  else if (isnext < 0 && isIfOverlap >= 0) {
883  isnext = isIfOverlap;
884  isIfOverlap = -1;
885  } else if (isnext >= MAXWINDOWS && isIfOverlap < MAXWINDOWS) {
886  isnext = isIfOverlap;
887  isIfOverlap = MAXWINDOWS;
888  }
889 
890  if (isnext < 0) { // we jump too ahead in future, digit will be not stored
891  LOG(INFO) << "Digit lost because we jump too ahead in future. Current RO window=" << isnext << "\n";
892

922  mcTruthContainer = mMCTruthContainerNext[isnext - 1];
923  }
924

931  // remove digit from array in the future
932  int labelremoved = digit->getLabel();
933  mFutureDigits.erase(mFutureDigits.begin() + idigit);
934 
935  /* NOT TO REMOVE LABELS TO SAVE CPU TIME (clear of vector when flushing)
