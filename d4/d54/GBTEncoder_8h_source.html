93 inline int phase(int i, bool forceNoPhase)
94 {
95  // generate the phase for the i-th ElinkEncoder
96  // the default value of -1 means it will be random and decided
97  // by the ElinkEncoder ctor
98  //
99  // if > 0 it will set a fixed phase at the beginning of the life
100  // of the ElinkEncoder
101  //
102  // returning zero will simply disable the phase
103 
104  if (forceNoPhase) {
105  return 0;
106  }
107  return -1;
108 }
