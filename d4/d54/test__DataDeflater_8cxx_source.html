46 template <typename DataContainerT, typename DeflatedDataT>
47 bool compare(const DataContainerT& container, std::size_t bitwidth, const DeflatedDataT& targetBuffer)
48 {
49  unsigned wordcount = 0;
50  auto bufferWord = targetBuffer[wordcount];
51  using target_type = typename DeflatedDataT::value_type;
52  auto targetWidth = 8 * sizeof(target_type);
53  int position = targetWidth;
54  for (const auto c : container) {
55  int length = bitwidth;
56  while (length > 0) {
57  if (position == 0) {
58  ++wordcount;
59  BOOST_REQUIRE(wordcount < targetBuffer.size());
60  position = targetWidth;
61  bufferWord = targetBuffer[wordcount];
62  }
63  auto comparing = length;
64  if (comparing > position) {
65  comparing = position;
66  }
67  position -= comparing;
68  target_type mask = ((target_type)1 << comparing) - 1;
69  if (((bufferWord >> position) & mask) != ((c >> (length - comparing)) & mask)) {
70  std::cout << "Decoding error at wordcount: " << wordcount << std::endl
71  << " length: " << length << std::endl
72  << " comparing: " << comparing << std::endl
73  << " position: " << position << std::endl
74  << " mask: " << std::hex << mask << std::endl
75  << std::endl
76  << " bufferWord: " << std::hex << bufferWord << std::endl
77  << " c: " << std::hex << c << std::endl;
78  }
79 
80  BOOST_REQUIRE(((bufferWord >> position) & mask) == ((c >> (length - comparing)) & mask));
81 
82  length -= comparing;
83  }
84  }
85 
86  return true;
87 }

134 // define a simple parameter model to mask a data value
135 template <int NBits>
136 class ParameterModelBitMask
137 {
138  public:
139  ParameterModelBitMask() = default;
140  ~ParameterModelBitMask() = default;
141 
142  static const int sBitlength = NBits;
143  using converted_type = uint64_t;
144 
145  template <typename T>
146  int convert(T value, converted_type& content, uint8_t& bitlength)
147  {
148  bitlength = sBitlength; // number of valid bits in the value
149  uint32_t mask = 0x1 << bitlength;
150  mask -= 1;
151  content = value & mask;
152  return 0;
153  }
154 
155  void reset() {}
156 
157  private:
158 };
