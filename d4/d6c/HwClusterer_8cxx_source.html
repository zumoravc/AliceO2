29 HwClusterer::HwClusterer(
30  std::vector<ClusterHardwareContainer8kb>* clusterOutputContainer,
31  int sectorid, MCLabelContainer* labelOutput)
32  : Clusterer(),
33  mNumRows(0),
34  mNumRowSets(0),
35  mCurrentMcContainerInBuffer(0),
36  mSplittingMode(0),
37  mClusterSector(sectorid),
38  mLastTimebin(-1),
39  mLastHB(0),
40  mPeakChargeThreshold(2),
41  mContributionChargeThreshold(0),
42  mClusterCounter(0),
43  mIsContinuousReadout(true),
44  mRejectSinglePadClusters(false),
45  mRejectSingleTimeClusters(false),
46  mRejectLaterTimebin(false),
47  mPadsPerRowSet(),
48  mGlobalRowToRegion(),
49  mGlobalRowToLocalRow(),
50  mGlobalRowToVcIndex(),
51  mGlobalRowToRowSet(),
52  mDataBuffer(),
53  mIndexBuffer(),
54  mMCtruth(),
55  mTmpClusterArray(),
56  mTmpLabelArray(),
57  mClusterMcLabelArray(labelOutput),
58  mClusterArray(clusterOutputContainer)
59 {
60  LOG(DEBUG) << "Enter Initializer of HwClusterer";
61 
62  // Given sector ID must be within 0 and 35 for a proper CRU ID calculation
63  assert(sectorid >= 0 && sectorid < 36);
64 
65  /*
66  * Prepare temporary storage for digits
67  */
68  Mapper& mapper = Mapper::instance();
69 
70  mNumRows = mapper.getNumberOfRows();
71  mNumRowSets = std::ceil(mapper.getNumberOfRows() / Vc::uint_v::Size);
72  mDataBuffer.resize(mNumRowSets);
73  mIndexBuffer.resize(mNumRowSets);
74  mPadsPerRowSet.resize(mNumRowSets);
75 
76  mGlobalRowToVcIndex.resize(mNumRows);
77  mGlobalRowToRowSet.resize(mNumRows);
78  for (unsigned short row = 0; row < mNumRowSets; ++row) {
79  // add two empty pads on the left and on the right
80  int max = 0;
81  for (int subrow = 0; subrow < Vc::uint_v::Size; ++subrow) {
82  max = std::max(max, mapper.getNumberOfPadsInRowSector(row * Vc::uint_v::Size + subrow) + 2 + 2);
83  mGlobalRowToRowSet[row * Vc::uint_v::Size + subrow] = row;
84  mGlobalRowToVcIndex[row * Vc::uint_v::Size + subrow] = subrow;
85  }
86  mPadsPerRowSet[row] = max;
87 
88  // prepare for number of timebins
89  mDataBuffer[row].resize(mPadsPerRowSet[row] * mTimebinsInBuffer, 0);
90  mIndexBuffer[row].resize(mPadsPerRowSet[row] * mTimebinsInBuffer, -1);
91  }
92 
93  mTmpClusterArray.resize(10);
94  mTmpLabelArray.resize(10);
95  for (unsigned short region = 0; region < 10; ++region) {
96  mTmpClusterArray[region] = std::make_unique<std::vector<ClusterHardware>>();
97  mTmpLabelArray[region] = std::make_unique<std::vector<std::vector<std::pair<MCCompLabel, unsigned>>>>();
98  }
99 
100  mGlobalRowToRegion.resize(mNumRows);
101  mGlobalRowToLocalRow.resize(mNumRows);
102  unsigned short row = 0;
103  for (unsigned short region = 0; region < 10; ++region) {
104  for (unsigned short localRow = 0; localRow < mapper.getNumberOfRowsRegion(region); ++localRow) {
105  mGlobalRowToRegion[row] = region;
106  mGlobalRowToLocalRow[row] = localRow;
107  ++row;
108  }
109  }
110  mMCtruth.resize(mTimebinsInBuffer, nullptr);
111 }

128 void HwClusterer::process(gsl::span<o2::tpc::Digit const> const& digits, MCLabelContainer const* mcDigitTruth, bool clearContainerFirst)
129 {
130  if (clearContainerFirst) {
131  if (mClusterArray)
132  mClusterArray->clear();
133 
134  if (mClusterMcLabelArray)
135  mClusterMcLabelArray->clear();
136  mClusterCounter = 0;
137  }
138 
139  int digitIndex = 0;
140  int index;
141  unsigned HB;
142  mCurrentMcContainerInBuffer = 0;
143 
144  /*
145  * Loop over all (time ordered) digits
146  */
147  for (const auto& digit : digits) {
148  /*
149  * This loop does the following:
150  * - add digits to the tmp storage
151  * - look for clusters
152  * - fill cluster output
153  *
154  * Before adding current digit to the storage, the new timebin has to be
155  * prepared first, by setting it completely 0, or to pedestal or ...
156  *
157  * This needs to be done only of the timestamps changes, otherwise it was
158  * already done.
159  */
160 
161  if (digit.getTimeStamp() != mLastTimebin) {
162  /*
163  * If the timebin changes, it could change by more then just 1 (not every
164  * timebin has digits). Since the tmp storage covers mTimebinsInBuffer,
165  * at most mTimebinsInBuffer new timebins need to be prepared and checked
166  * for clusters.
167  */
168  for (int i = mLastTimebin; (i < digit.getTimeStamp()) && (i - mLastTimebin < mTimebinsInBuffer); ++i) {
169 
170  /*
171  * If the HB of the cluster which will be found in a few lines, NOT the
172  * current timebin is a new one, we have to fill the output container
173  * with the so far found clusters. Because cluster center and timebin
174  * have an offset of two with respect to each other (see next comment),
175  * the HB is calculated with (i-2). By the way, it is not possible, that
176  * a cluster is found with a negative HB, because at least 2 timebins
177  * have to be filled to be able to find a cluster.
178  */
179  HB = i < 3 ? 0 : (i - 3) / 447; // integer division on purpose
180  if (HB != mLastHB) {
181  writeOutputWithTimeOffset(mLastHB * 447);
182  }
183 
184  /*
185  * For each row(set), we first compute all the pad relations for the
186  * latest timebin (i), afterwards the clusters for timebin i-2 are
187  * collected and computed. We need the -2 because a cluster spreads
188  * over 5 timbins, and the relations are always computed with respect
189  * to the older timebin. Also a (i-1) and (i-2) would be possible, but
190  * doens't matter.
191  *
192  * If mTimebinsInBuffer would be 5 and i 5 is the new digit timebin (4
193  * would be mLastTimebin), then 0 is the oldest one timebin and will to
194  * be replaced by the new arriving one. The cluster which could be
195  * found, would then range from timebin 0 to 4 and has its center at
196  * timebin 2. Threrefore we are looking in (i - 2) for clusters and
197  * clearing (i - 4), or (i + 1) afterwards.
198  * ---------
199  * -> 0 |
200  * 1 |
201  * 2 | XXXXXX
202  * 3 |
203  * 4 |
204  * ---------
205  */
206  findPeaksForTime(i);
207  computeClusterForTime(i - 3);
208 
209  clearBuffer(i + 1);
210 
211  mLastHB = HB;
212  }
213 
214  // we have to copy the MC truth container because we need the information
215  // maybe only in the next events (we store permanently 5 timebins), where
216  // the original pointer could already point to the next container.
217  if (mcDigitTruth) {
218  if (mCurrentMcContainerInBuffer == 0)
219  mMCtruth[mapTimeInRange(digit.getTimeStamp())] = std::make_shared<MCLabelContainer const>(*mcDigitTruth);
220  else
221  mMCtruth[mapTimeInRange(digit.getTimeStamp())] = std::shared_ptr<MCLabelContainer const>(mMCtruth[getFirstSetBitOfField()]);
222 
223  mCurrentMcContainerInBuffer |= (0x1 << (mapTimeInRange(digit.getTimeStamp())));
224  }
225  }
226 
227  /*
228  * add current digit to storage
229  */
230  index = mapTimeInRange(digit.getTimeStamp()) * mPadsPerRowSet[mGlobalRowToRowSet[digit.getRow()]] + (digit.getPad() + 2);
231  // offset of digit pad because of 2 empty pads on both sides
232 
233  float charge = digit.getChargeFloat();
234 
235  // TODO: fill noise here as well if necessary
236  if (mPedestalObject) {
237  charge -= mPedestalObject->getValue(CRU(digit.getCRU()), digit.getRow(), digit.getPad());
238  }
239  /*
240  * charge could be smaller than 0 due to pedestal subtraction, if so set it to zero
241  * noise thresholds for zero suppression could also be done here ...
242  */
243  mDataBuffer[mGlobalRowToRowSet[digit.getRow()]][index][mGlobalRowToVcIndex[digit.getRow()]] =
244  charge < 0 ? 0 : ((charge < float(0x3FFF) / (1 << 4)) ? (charge * (1 << 4)) : 0x3FFF);
245 
246  mIndexBuffer[mGlobalRowToRowSet[digit.getRow()]][index][mGlobalRowToVcIndex[digit.getRow()]] = digitIndex++;
247 
248  mLastTimebin = digit.getTimeStamp();
249  }
250 
251  if (!mIsContinuousReadout)
252  finishFrame(true);
253 
254  if (digits.size() != 0)
255  LOG(DEBUG) << "Event ranged from time bin " << digits[0].getTimeStamp() << " to " << digits[digits.size() - 1].getTimeStamp() << ".";
256 }

259 void HwClusterer::finishProcess(gsl::span<o2::tpc::Digit const> const& digits, MCLabelContainer const* mcDigitTruth, bool clearContainerFirst)
260 {
261  // Process the last digits (if there are any)
262  process(digits, mcDigitTruth, clearContainerFirst);
263 
264  // Search in last remaining timebins
265  finishFrame(false);
266 }

269 void HwClusterer::hwClusterProcessor(const Vc::uint_m peakMask, unsigned qMaxIndex, short centerPad, int centerTime, unsigned short row)
270 {
271  Vc::uint_v qTot = 0;
272  Vc::int_v pad = 0;
273  Vc::int_v time = 0;
274  Vc::int_v sigmaPad2 = 0;
275  Vc::int_v sigmaTime2 = 0;
276  Vc::int_v flags = 0;
277 
278  using labelPair = std::pair<MCCompLabel, unsigned>;
279  std::vector<std::unique_ptr<std::vector<labelPair>>> mcLabels(Vc::uint_v::Size);
280  for (int i = 0; i < Vc::uint_v::Size; ++i) {
281  mcLabels[i] = std::make_unique<std::vector<labelPair>>();
282  }
283 
284  const unsigned llttIndex = mapTimeInRange(centerTime - 2) * mPadsPerRowSet[row] + centerPad - 2;
285  const unsigned lltIndex = mapTimeInRange(centerTime - 1) * mPadsPerRowSet[row] + centerPad - 2;
286  const unsigned llIndex = mapTimeInRange(centerTime + 0) * mPadsPerRowSet[row] + centerPad - 2;
287  const unsigned llbIndex = mapTimeInRange(centerTime + 1) * mPadsPerRowSet[row] + centerPad - 2;
288  const unsigned llbbIndex = mapTimeInRange(centerTime + 2) * mPadsPerRowSet[row] + centerPad - 2;
289 
290  const unsigned lttIndex = llttIndex + 1;
291  const unsigned ltIndex = lltIndex + 1;
292  const unsigned lIndex = llIndex + 1;
293  const unsigned lbIndex = llbIndex + 1;
294  const unsigned lbbIndex = llbbIndex + 1;
295 
296  const unsigned ttIndex = lttIndex + 1;
297  const unsigned tIndex = ltIndex + 1;
298  const unsigned bIndex = lbIndex + 1;
299  const unsigned bbIndex = lbbIndex + 1;
300 
301  const unsigned rttIndex = ttIndex + 1;
302  const unsigned rtIndex = tIndex + 1;
303  const unsigned rIndex = lIndex + 2;
304  const unsigned rbIndex = bIndex + 1;
305  const unsigned rbbIndex = bbIndex + 1;
306 
307  const unsigned rrttIndex = rttIndex + 1;
308  const unsigned rrtIndex = rtIndex + 1;
309  const unsigned rrIndex = rIndex + 1;
310  const unsigned rrbIndex = rbIndex + 1;
311  const unsigned rrbbIndex = rbbIndex + 1;
312 
313  Vc::uint_m selectionMask;
314  if (mSplittingMode == 0) {
315  // Charge is not splitted between nearby peaks, every peak uses all charges
316 
317  // Cluster:
318  //
319  // o o o o o
320  // o i i i o
321  // o i C i o
322  // o i i i o
323  // o o o o o
324 
325  // Use always inner 3x3 matrix
326  // i i i
327  // i C i
328  // i i i
329  for (short dt = -1; dt <= 1; ++dt) {
330  for (short dp = -1; dp <= 1; ++dp) {
331  updateCluster(peakMask, row, centerPad, centerTime, dp, dt, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
332  }
333  }
334 
335  // Use outer pads (o) only, if corresponding inner pad (i) is above contribution threshold
336  // o
337  // i
338  // o i C i o
339  // i
340  // o
341  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][lIndex]) > (mContributionChargeThreshold << 4));
342  updateCluster(selectionMask, row, centerPad, centerTime, -2, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
343 
344  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][rIndex]) > (mContributionChargeThreshold << 4));
345  updateCluster(selectionMask, row, centerPad, centerTime, +2, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
346 
347  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][tIndex]) > (mContributionChargeThreshold << 4));
348  updateCluster(selectionMask, row, centerPad, centerTime, 0, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
349 
350  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][bIndex]) > (mContributionChargeThreshold << 4));
351  updateCluster(selectionMask, row, centerPad, centerTime, 0, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
352 
353  // o o o o
354  // o i i o
355  // C
356  // o i i o
357  // o o o o
358  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][ltIndex]) > (mContributionChargeThreshold << 4));
359  updateCluster(selectionMask, row, centerPad, centerTime, -2, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
360  updateCluster(selectionMask, row, centerPad, centerTime, -2, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
361  updateCluster(selectionMask, row, centerPad, centerTime, -1, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
362 
363  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][lbIndex]) > (mContributionChargeThreshold << 4));
364  updateCluster(selectionMask, row, centerPad, centerTime, -2, +1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
365  updateCluster(selectionMask, row, centerPad, centerTime, -2, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
366  updateCluster(selectionMask, row, centerPad, centerTime, -1, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
367 
368  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][rtIndex]) > (mContributionChargeThreshold << 4));
369  updateCluster(selectionMask, row, centerPad, centerTime, +2, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
370  updateCluster(selectionMask, row, centerPad, centerTime, +2, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
371  updateCluster(selectionMask, row, centerPad, centerTime, +1, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
372 
373  selectionMask = peakMask & (getFpOfADC(mDataBuffer[row][rbIndex]) > (mContributionChargeThreshold << 4));
374  updateCluster(selectionMask, row, centerPad, centerTime, +2, +1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
375  updateCluster(selectionMask, row, centerPad, centerTime, +2, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
376  updateCluster(selectionMask, row, centerPad, centerTime, +1, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
377 
378  } else if (mSplittingMode == 1) {
379  // Charge is split at minimum, the minimum itself contributes half too all
380  // adjacent peaks (even if there are 3).
381 
382  // Use always inner 3x3 matrix
383  // i i i
384  // i C i
385  // i i i
386  // but only half the charge if (i) is minimum
387 
388  // center
389  updateCluster(peakMask, row, centerPad, centerTime, 0, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels);
390 
391  // horizontal, look for minimum in pad direction
392  auto splitMask = (((mDataBuffer[row][lIndex] >> 26) & 0x1) == 0x1);
393  updateCluster(peakMask, row, centerPad, centerTime, -1, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
394 
395  splitMask = (((mDataBuffer[row][rIndex] >> 26) & 0x1) == 0x1);
396  updateCluster(peakMask, row, centerPad, centerTime, +1, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
397 
398  // vertical look for minimum in time direction
399  splitMask = (((mDataBuffer[row][tIndex] >> 25) & 0x1) == 0x1);
400  updateCluster(peakMask, row, centerPad, centerTime, 0, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
401 
402  splitMask = (((mDataBuffer[row][bIndex] >> 25) & 0x1) == 0x1);
403  updateCluster(peakMask, row, centerPad, centerTime, 0, +1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
404 
405  // diagonal tl/br, look for minimum in 1. diagonal + vertical + horizontal direction
406  splitMask = (((mDataBuffer[row][ltIndex] >> 24) & 0x1) == 0x1) |
407  (((mDataBuffer[row][ltIndex] >> 25) & 0x1) == 0x1) |
408  (((mDataBuffer[row][ltIndex] >> 26) & 0x1) == 0x1);
409  updateCluster(peakMask, row, centerPad, centerTime, -1, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
410 
411  splitMask = (((mDataBuffer[row][rbIndex] >> 24) & 0x1) == 0x1) |
412  (((mDataBuffer[row][rbIndex] >> 25) & 0x1) == 0x1) |
413  (((mDataBuffer[row][rbIndex] >> 26) & 0x1) == 0x1);
414  updateCluster(peakMask, row, centerPad, centerTime, +1, +1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
415 
416  // diagonal tr/bl, look for minimum in 2. diagonal + vertical + horizontal direction
417  splitMask = (((mDataBuffer[row][rtIndex] >> 23) & 0x1) == 0x1) |
418  (((mDataBuffer[row][rtIndex] >> 25) & 0x1) == 0x1) |
419  (((mDataBuffer[row][rtIndex] >> 26) & 0x1) == 0x1);
420  updateCluster(peakMask, row, centerPad, centerTime, +1, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
421 
422  splitMask = (((mDataBuffer[row][lbIndex] >> 23) & 0x1) == 0x1) |
423  (((mDataBuffer[row][lbIndex] >> 25) & 0x1) == 0x1) |
424  (((mDataBuffer[row][lbIndex] >> 26) & 0x1) == 0x1);
425  updateCluster(peakMask, row, centerPad, centerTime, -1, +1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
426 
427  // Use outer pads (o) only, if corresponding inner pad (i) is above contribution threshold
428  // o
429  // i
430  // o i C i o
431  // i
432  // o
433  // and only if inner pad (i) is not a minimum (in corresponding direction).
434  // Split charge, if (o) is a minimum.
435  selectionMask = peakMask & // using (o) (if we have a peak in the center)
436  (getFpOfADC(mDataBuffer[row][lIndex]) > (mContributionChargeThreshold << 4)) & // AND (i) is above threshold
437  (((mDataBuffer[row][lIndex] >> 26) & 0x1) != 0x1); // AND (i) is not minimum in corresponding direction
438  splitMask = (((mDataBuffer[row][llIndex] >> 26) & 0x1) == 0x1); // split (o) if it is miminum in corresponding direction
439  updateCluster(selectionMask, row, centerPad, centerTime, -2, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
440 
441  selectionMask = peakMask &
442  (getFpOfADC(mDataBuffer[row][rIndex]) > (mContributionChargeThreshold << 4)) &
443  (((mDataBuffer[row][rIndex] >> 26) & 0x1) != 0x1);
444  splitMask = (((mDataBuffer[row][rrIndex] >> 26) & 0x1) == 0x1);
445  updateCluster(selectionMask, row, centerPad, centerTime, +2, 0, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
446 
447  selectionMask = peakMask &
448  (getFpOfADC(mDataBuffer[row][tIndex]) > (mContributionChargeThreshold << 4)) &
449  (((mDataBuffer[row][tIndex] >> 25) & 0x1) != 0x1);
450  splitMask = (((mDataBuffer[row][ttIndex] >> 26) & 0x1) == 0x1);
451  updateCluster(selectionMask, row, centerPad, centerTime, 0, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
452 
453  selectionMask = peakMask &
454  (getFpOfADC(mDataBuffer[row][bIndex]) > (mContributionChargeThreshold << 4)) &
455  (((mDataBuffer[row][bIndex] >> 25) & 0x1) != 0x1);
456  splitMask = (((mDataBuffer[row][bbIndex] >> 26) & 0x1) == 0x1);
457  updateCluster(selectionMask, row, centerPad, centerTime, 0, +2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
458 
459  // o o o o
460  // o i i o
461  // C
462  // o i i o
463  // o o o o
464  selectionMask = peakMask & // using (o) (if we have a peak in the center)
465  (getFpOfADC(mDataBuffer[row][ltIndex]) > (mContributionChargeThreshold << 4)) & // AND (i) is above threshold
466  (((mDataBuffer[row][ltIndex] >> 26) & 0x1) != 0x1); // AND (i) is not minimum in corresponding direction
467  splitMask = (((mDataBuffer[row][lltIndex] >> 26) & 0x1) == 0x1) | // split (o) if it is miminum in corresponding directions
468  (((mDataBuffer[row][lltIndex] >> 24) & 0x1) == 0x1);
469  updateCluster(selectionMask, row, centerPad, centerTime, -2, -1, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);
470 
471  selectionMask = peakMask & // using (o) (if we have a peak in the center)
472  (getFpOfADC(mDataBuffer[row][ltIndex]) > (mContributionChargeThreshold << 4)) & // AND (i) is above threshold
473  (((mDataBuffer[row][ltIndex] >> 24) & 0x1) != 0x1) & // AND (i) is not minimum in corresponding directions
474  (((mDataBuffer[row][ltIndex] >> 25) & 0x1) != 0x1) &
475  (((mDataBuffer[row][ltIndex] >> 26) & 0x1) != 0x1) &
476  (((mDataBuffer[row][lltIndex] >> 25) & 0x1) != 0x1) & // AND other (o) is not minimum in corresponding direction
477  (((mDataBuffer[row][lttIndex] >> 26) & 0x1) != 0x1); // AND other (o) is not minimum in corresponding direction
478  splitMask = (((mDataBuffer[row][llttIndex] >> 24) & 0x1) == 0x1) | // split (o) if it is miminum in corresponding direction
479  (((mDataBuffer[row][llttIndex] >> 25) & 0x1) == 0x1) |
480  (((mDataBuffer[row][llttIndex] >> 26) & 0x1) == 0x1);
481  updateCluster(selectionMask, row, centerPad, centerTime, -2, -2, qTot, pad, time, sigmaPad2, sigmaTime2, mcLabels, splitMask);

597 void HwClusterer::hwPeakFinder(unsigned qMaxIndex, short centerPad, int mappedCenterTime, unsigned short row)
598 {
599 
600  // Always the center pad is compared with a fixed other pad. The first
601  // comparison is in pad direction. That bin (p,t) is a peak in pad direction,
602  // it has to be >= then (p-1,t) and in the next iteration, where the next pad
603  // is the center, the comparison of (the new center with the new left one)
604  // must be false, so that (p+1,t) > (p,t). Afterwards, this is also done in
605  // time direction and both diagonal directions. With those 4 comparisons, all
606  // 8 neighboring pads are checked iteratively.
607  //
608  // Example in pad direction:
609  // p-2 p-1 p p+1 p+2 p-1 p p+1 p+2 p+3
610  // t-2 o o o o o o o o o o
611  // t-1 o i i i o next iteration o i i i o
612  // t o I<->C i o ---------------> O I<->c i o
613  // t+1 o i i i o o i i i o
614  // t+2 o o o o o o o o o o
615  //
616  //
617  // Meaning of the set bit:
618  //
619  // bit | meaning if set
620  // 31 | peak in pad direction
621  // 30 | peak in time direction
622  // 29 | peak in 1. diagonal direction, top left to bottom right
623  // 28 | peak in 2. diagonal direction, top right to bottom left
624  // 27 | ADC above peak threshold
625  // 26 | minimum in pad direction
626  // 25 | minimum in time direction
627  // 24 | minimum in 1. diagonal direction
628  // 23 | minimum in 2. diagonal direction
629 
630  const int compareIndex0 = mappedCenterTime * mPadsPerRowSet[row] + centerPad - 1;
631  const int compareIndex1 = mapTimeInRange(mappedCenterTime - 1) * mPadsPerRowSet[row] + centerPad;
632  const int compareIndex2 = compareIndex1 - 1;
633  const int compareIndex3 = compareIndex1 + 1;
634  const auto adcMask = (getFpOfADC(mDataBuffer[row][qMaxIndex]) == 0) &
635  (getFpOfADC(mDataBuffer[row][compareIndex0]) == 0) &
636  (getFpOfADC(mDataBuffer[row][compareIndex1]) == 0) &
637  (getFpOfADC(mDataBuffer[row][compareIndex2]) == 0) &
638  (getFpOfADC(mDataBuffer[row][compareIndex3]) == 0);
639  if (adcMask.isFull()) {
640  // if all 0, we can skip the following part and directly set the bits where
641  // the comparisons were true, TODO: still possible if difference is applied?
642  mDataBuffer[row][qMaxIndex] |= (0x1 << 31);
643  mDataBuffer[row][compareIndex0] &= ~(0x1 << 31);
644  mDataBuffer[row][qMaxIndex] |= (0x1 << 30);
645  mDataBuffer[row][compareIndex1] &= ~(0x1 << 30);
646  mDataBuffer[row][qMaxIndex] |= (0x1 << 29);
647  mDataBuffer[row][compareIndex2] &= ~(0x1 << 29);
648  mDataBuffer[row][qMaxIndex] |= (0x1 << 28);
649  mDataBuffer[row][compareIndex3] &= ~(0x1 << 28);
650  return;
651  }
652 
654  // Comparison in pad direction
655  // TODO: define needed difference
656  auto tmpMask = getFpOfADC(mDataBuffer[row][qMaxIndex]) >= getFpOfADC(mDataBuffer[row][compareIndex0]);
657  // if true:
658  // - current center could be peak
659  // - cleare possible maxBit of other
660  // - other is minimum if minBit was already set before
661  Vc::where(tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 31);
662  Vc::where(tmpMask) | mDataBuffer[row][compareIndex0] &= ~(0x1 << 31);
663 
664  // if false:
665  // - current center could be minimum
666  // - cleare possible minBit of other
667  // - other is peak if maxBit was already set before
668  Vc::where(!tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 26);
669  Vc::where(!tmpMask) | mDataBuffer[row][compareIndex0] &= ~(0x1 << 26);
670 
672  // Comparison in time direction
673  tmpMask = getFpOfADC(mDataBuffer[row][qMaxIndex]) >= getFpOfADC(mDataBuffer[row][compareIndex1]);
674  Vc::where(tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 30);
675  Vc::where(tmpMask) | mDataBuffer[row][compareIndex1] &= ~(0x1 << 30);
676  Vc::where(!tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 25);
677  Vc::where(!tmpMask) | mDataBuffer[row][compareIndex1] &= ~(0x1 << 25);
678 
680  // Comparison in 1. diagonal direction
681  tmpMask = getFpOfADC(mDataBuffer[row][qMaxIndex]) >= getFpOfADC(mDataBuffer[row][compareIndex2]);
682  Vc::where(tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 29);
683  Vc::where(tmpMask) | mDataBuffer[row][compareIndex2] &= ~(0x1 << 29);
684  Vc::where(!tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 24);
685  Vc::where(!tmpMask) | mDataBuffer[row][compareIndex2] &= ~(0x1 << 24);
686 
688  // Comparison in 2. diagonal direction
689  tmpMask = getFpOfADC(mDataBuffer[row][qMaxIndex]) >= getFpOfADC(mDataBuffer[row][compareIndex3]);
690  Vc::where(tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 28);
691  Vc::where(tmpMask) | mDataBuffer[row][compareIndex3] &= ~(0x1 << 28);
692  Vc::where(!tmpMask) | mDataBuffer[row][qMaxIndex] |= (0x1 << 23);
693  Vc::where(!tmpMask) | mDataBuffer[row][compareIndex3] &= ~(0x1 << 23);
694 
696  // Comparison peak threshold
697  Vc::where(getFpOfADC(mDataBuffer[row][qMaxIndex]) > (mPeakChargeThreshold << 4)) | mDataBuffer[row][qMaxIndex] |= (0x1 << 27);
698 }

763 void HwClusterer::computeClusterForTime(int timebin)
764 {
765  if (timebin < 0)
766  return;
767 
768  const unsigned timeBinWrapped = mapTimeInRange(timebin);
769  if (mRejectLaterTimebin) {
770  const unsigned previousTimeBinWrapped = mapTimeInRange(timebin - 2);
771  for (unsigned short row = 0; row < mNumRowSets; ++row) {
772  const unsigned padOffset = timeBinWrapped * mPadsPerRowSet[row];
773  const unsigned previousPadOffset = previousTimeBinWrapped * mPadsPerRowSet[row];
774  // two empty pads on the left and right without a cluster peak
775  for (short pad = 2; pad < mPadsPerRowSet[row] - 2; ++pad) {
776  const unsigned qMaxIndex = padOffset + pad;
777  const unsigned qMaxPreviousIndex = previousPadOffset + pad;
778 
779  // TODO: define needed difference
780  const auto peakMask = ((mDataBuffer[row][qMaxIndex] >> 27) == 0x1F) & // True if current pad is peak AND
781  (getFpOfADC(mDataBuffer[row][qMaxIndex]) > getFpOfADC(mDataBuffer[row][qMaxPreviousIndex]) | // previous has smaller charge
782  !((mDataBuffer[row][qMaxPreviousIndex] >> 27) == 0x1F)); // or previous one was not a peak
783  if (peakMask.isEmpty())
784  continue;
785 
786  hwClusterProcessor(peakMask, qMaxIndex, pad, timebin, row);
787  }
788  }
789  } else {
790  for (unsigned short row = 0; row < mNumRowSets; ++row) {
791  const unsigned padOffset = timeBinWrapped * mPadsPerRowSet[row];
792  // two empty pads on the left and right without a cluster peak
793  for (short pad = 2; pad < mPadsPerRowSet[row] - 2; ++pad) {
794  const unsigned qMaxIndex = padOffset + pad;
795 
796  const auto peakMask = ((mDataBuffer[row][qMaxIndex] >> 27) == 0x1F);

846 void HwClusterer::updateCluster(
847  const Vc::uint_m selectionMask, int row, short centerPad, int centerTime, short dp, short dt,
848  Vc::uint_v& qTot, Vc::int_v& pad, Vc::int_v& time, Vc::int_v& sigmaPad2, Vc::int_v& sigmaTime2,

850 {

852  return;
853 
854  const int mappedTime = mapTimeInRange(centerTime + dt);
855  const int index = mappedTime * mPadsPerRowSet[row] + centerPad + dp;
856 
857  // If the charge should be split, only half of the charge is used
858  Vc::where(selectionMask & splitMask) | qTot += (getFpOfADC(mDataBuffer[row][index]) >> 1);
859  Vc::where(selectionMask & splitMask) | pad += (getFpOfADC(mDataBuffer[row][index]) >> 1) * dp;
860  Vc::where(selectionMask & splitMask) | time += (getFpOfADC(mDataBuffer[row][index]) >> 1) * dt;

862  Vc::where(selectionMask & splitMask) | sigmaTime2 += (getFpOfADC(mDataBuffer[row][index]) >> 1) * dt * dt;
863 
864  // Otherwise the full charge
865  Vc::where(selectionMask & (!splitMask)) | qTot += getFpOfADC(mDataBuffer[row][index]);
866  Vc::where(selectionMask & (!splitMask)) | pad += getFpOfADC(mDataBuffer[row][index]) * dp;
867  Vc::where(selectionMask & (!splitMask)) | time += getFpOfADC(mDataBuffer[row][index]) * dt;
868  Vc::where(selectionMask & (!splitMask)) | sigmaPad2 += getFpOfADC(mDataBuffer[row][index]) * dp * dp;
869  Vc::where(selectionMask & (!splitMask)) | sigmaTime2 += getFpOfADC(mDataBuffer[row][index]) * dt * dt;
870 
871  for (int i = 0; i < Vc::uint_v::Size; ++i) {
872  if (selectionMask[i] && mMCtruth[mappedTime] != nullptr) {
873  for (auto& label : mMCtruth[mappedTime]->getLabels(mIndexBuffer[row][index][i])) {
874  bool isKnown = false;
875  for (auto& vecLabel : *mcLabels[i]) {
876  if (label == vecLabel.first) {
877  ++vecLabel.second;
878  isKnown = true;
879  }
880  }
881  if (!isKnown) {
882  mcLabels[i]->emplace_back(label, 1);
883  }
884  }
885  }
