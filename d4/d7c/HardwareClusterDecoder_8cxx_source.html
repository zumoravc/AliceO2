35 int HardwareClusterDecoder::decodeClusters(std::vector<std::pair<const ClusterHardwareContainer*, std::size_t>>& inputClusters,
36  HardwareClusterDecoder::OutputAllocator outputAllocator,
37  const std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>* inMCLabels,
38  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* outMCLabels)
39 {
40  if (mIntegrator == nullptr)
41  mIntegrator.reset(new DigitalCurrentClusterIntegrator);
42  // MCLabelContainer does only allow appending new labels, so we need to write to separate
43  // containers per {sector,padrow} and merge at the end;
44  std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> outMCLabelContainers;
45  if (!inMCLabels) {
46  outMCLabels = nullptr;
47  }
48  ClusterNative* outputClusterBuffer = nullptr;
49  // the number of clusters in a {sector,row}
50  int nRowClusters[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
51  // offset of first cluster of {sector,row} in the output buffer
52  size_t clusterOffsets[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
53  int containerRowCluster[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
54  Mapper& mapper = Mapper::instance();
55  int numberOfOutputContainers = 0;
56  for (int loop = 0; loop < 2; loop++) {
57  int nTotalClusters = 0;
58  for (int i = 0; i < inputClusters.size(); i++) {
59  if (outMCLabels && inputClusters[i].second > 1) {
60  LOG(ERROR) << "Decoding of ClusterHardware to ClusterNative with MC labels is yet only support for single 8kb pages of ClusterHardwareContainer\n";
61  return (1);
62  }
63  for (int j = 0; j < inputClusters[i].second; j++) {
64  const char* tmpPtr = reinterpret_cast<const char*>(inputClusters[i].first);
65  tmpPtr += j * 8192; //TODO: FIXME: Compute correct offset based on the size of the actual packet in the RDH
66  const ClusterHardwareContainer& cont = *(reinterpret_cast<const ClusterHardwareContainer*>(tmpPtr));
67  const CRU cru(cont.CRU);
68  const Sector sector = cru.sector();
69  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
70  const int rowOffset = region.getGlobalRowOffset();
71 
72  // TODO: make sure that input clusters are sorted in ascending row, so we
73  // can write the MCLabels directly in consecutive order following the cluster sequence
74  // Note: also the sorting below would need to be adjusted.
75  for (int k = 0; k < cont.numberOfClusters; k++) {
76  const int padRowGlobal = rowOffset + cont.clusters[k].getRow();
77  int& nCls = nRowClusters[sector][padRowGlobal];
78  if (loop == 1) {
79  //Fill cluster in the respective output buffer
80  const ClusterHardware& cIn = cont.clusters[k];
81  ClusterNative& cOut = outputClusterBuffer[clusterOffsets[sector][padRowGlobal] + nCls];
82  float pad = cIn.getPad();
83  cOut.setPad(pad);
84  cOut.setTimeFlags(cIn.getTimeLocal() + cont.timeBinOffset, cIn.getFlags());
85  cOut.setSigmaPad(std::sqrt(cIn.getSigmaPad2()));
86  cOut.setSigmaTime(std::sqrt(cIn.getSigmaTime2()));
87  cOut.qMax = cIn.getQMax();
88  cOut.qTot = cIn.getQTot();
89  mIntegrator->integrateCluster(sector, padRowGlobal, pad, cIn.getQTot());
90  if (outMCLabels) {
91  auto& mcOut = outMCLabelContainers[containerRowCluster[sector][padRowGlobal]];
92  for (const auto& element : (*inMCLabels)[i].getLabels(k)) {
93  mcOut.addElement(nCls, element);
94  }
95  }
96  } else {
97  //Count how many output buffers we need (and how large they are below)
98  if (nCls == 0)
99  numberOfOutputContainers++;
100  }
101  nCls++;
102  nTotalClusters++;
103  }
104  }
105  }
106  if (loop == 1) {
107  //We are done with filling the buffers, sort all output buffers
108  for (int i = 0; i < Constants::MAXSECTOR; i++) {
109  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
110  if (nRowClusters[i][j] == 0) {
111  continue;
112  }
113  if (outMCLabels) {
114  sortClustersAndMC(outputClusterBuffer + clusterOffsets[i][j], nRowClusters[i][j], outMCLabelContainers[containerRowCluster[i][j]]);
115  } else {
116  auto* cl = outputClusterBuffer + clusterOffsets[i][j];
117  std::sort(cl, cl + nRowClusters[i][j]);
118  }
119  }
120  }
121  } else {
122  //Now we know the size of all output buffers, allocate them
123  if (outMCLabels) {
124  outMCLabelContainers.resize(numberOfOutputContainers);
125  }
126  size_t rawOutputBufferSize = sizeof(ClusterCountIndex) + nTotalClusters * sizeof(ClusterNative);
127  char* rawOutputBuffer = outputAllocator(rawOutputBufferSize);
128  auto& clusterCounts = *(reinterpret_cast<ClusterCountIndex*>(rawOutputBuffer));
129  outputClusterBuffer = reinterpret_cast<ClusterNative*>(rawOutputBuffer + sizeof(ClusterCountIndex));
130  nTotalClusters = 0;
131  numberOfOutputContainers = 0;
132  for (int i = 0; i < Constants::MAXSECTOR; i++) {
133  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
134  clusterCounts.nClusters[i][j] = nRowClusters[i][j];
135  if (nRowClusters[i][j] == 0) {
136  continue;
137  }
138  containerRowCluster[i][j] = numberOfOutputContainers++;
139  clusterOffsets[i][j] = nTotalClusters;
140  nTotalClusters += nRowClusters[i][j];
141  mIntegrator->initRow(i, j);
142  }
143  }
144  memset(nRowClusters, 0, sizeof(nRowClusters));
145  }
146  }
147  // Finally merge MC label containers into one container following the cluster sequence in the
148  // output buffer
149  if (outMCLabels) {
150  auto& labels = *outMCLabels;
151  int nCls = 0;
152  for (int i = 0; i < Constants::MAXSECTOR; i++) {
153  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
154  if (nRowClusters[i][j] == 0) {
155  continue;
156  }
157  for (int k = 0, end = outMCLabelContainers[containerRowCluster[i][j]].getIndexedSize(); k < end; k++, nCls++) {
158  assert(end == nRowClusters[i][j]);
159  assert(clusterOffsets[i][j] + k == nCls);
160  for (const auto& element : outMCLabelContainers[containerRowCluster[i][j]].getLabels(k)) {
161  labels.addElement(nCls, element);
162  }
163  }
164  }
165  }
166  }
167  return (0);
168 }

170 void HardwareClusterDecoder::sortClustersAndMC(ClusterNative* clusters, size_t nClusters,
171  o2::dataformats::MCTruthContainer<o2::MCCompLabel>& mcTruth)
172 {
173  std::vector<unsigned int> indizes(nClusters);
174  std::iota(indizes.begin(), indizes.end(), 0);
175  std::sort(indizes.begin(), indizes.end(), [&clusters](const auto a, const auto b) {
176  return clusters[a] < clusters[b];
177  });
178  std::vector<ClusterNative> buffer(clusters, clusters + nClusters);
179  MCLabelContainer tmpMC = std::move(mcTruth);
180  assert(mcTruth.getIndexedSize() == 0);
181  for (int i = 0; i < nClusters; i++) {
182  clusters[i] = buffer[indizes[i]];
183  for (auto const& label : tmpMC.getLabels(indizes[i])) {
184  mcTruth.addElement(i, label);
185  }
186  }
187 }
