30 template <typename T = o2::Transform3D>
31 class MatrixCache
32 {
33  // matrices (per sensor) for specific transformation type
34 
35  public:
36  MatrixCache() = default;
37  ~MatrixCache() = default;
38  MatrixCache(const MatrixCache& src) = delete;
39  MatrixCache& operator=(const MatrixCache& src) = delete;
40 
42  void setSize(int s)
43  {
44  if (!mCache.size())
45  mCache.resize(s);
46  }
47 
49  int getSize() const { return mCache.size(); }
51  void setMatrix(const T& mat, int sensID)
52  {
53  // assign matrix for given sensor. The cache must be booked in advance
54  if ((unsigned int)sensID >= mCache.size()) {
55  LOG(FATAL) << "SensID " << sensID << " exceeds cache size of " << mCache.size();
56  }
57  mCache[sensID] = mat;
58  }
59 
60  const T& getMatrix(int sensID) const { return mCache[sensID]; }
61  bool isFilled() const { return !mCache.empty(); }
62 
63  private:
64  std::vector<T> mCache;
65  ClassDefNV(MatrixCache, 1);
66 };

162  void useT2GRotCache() { mT2GRot.setSize(mSize); }
163 
164  private:
165  int getCacheHelper(int sensID)
166  {
167  if (sensID >= mIndirectSize) {
168  LOG(FATAL) << "SendID " << sensID << " exceeds indirect cache size of " << mIndirectSize;
169  }
170  if (mIndirection[sensID] >= 0) {
171  return mIndirection[sensID];
172  }
173  return (mIndirection[sensID] = mNextEntry++);
174  }
175 
176  int mIndirectSize = 0;
177  int mNextEntry = 0;
178  std::vector<short> mIndirection;
179
