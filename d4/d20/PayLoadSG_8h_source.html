27 class PayLoadSG
28 {
29  // scatter-gather buffer for the payload: base pointer + vector of references for pieces to collect
30  public:
31  using DataType = unsigned char;

37  void add(const DataType* ptr, size_t n)
38  {
39  if (n) {
40  mBuffer.emplace_back(ptr, n);
41  }
42  }

45  bool current(char& v)
46  {
47  if (mCurrentPieceID < mBuffer.size()) {
48  const auto& piece = mBuffer[mCurrentPieceID];
49  if (mCurrentEntryInPiece < piece.size) {
50  v = piece.data[mCurrentEntryInPiece];
51  return true;
52  } else {
53  nextPiece();
54  return current(v);
55  }
56  }
57  return false;
58  }
59 
61  bool next(char& v)
62  {
63  if (mCurrentPieceID < mBuffer.size()) {
64  const auto& piece = mBuffer[mCurrentPieceID];
65  if (mCurrentEntryInPiece < piece.size) {
66  v = piece.data[mCurrentEntryInPiece++];
67  return true;
68  } else {
69  nextPiece();
70  return next(v);
71  }
72  }
73  return false;
74  }
75 
77  bool next(uint16_t& v)
78  {
79  char b0, b1;
80  if (next(b0) && next(b1)) {
81  v = (b0 << 8) | b1;
82  return true;
83  }
84  return false;
85  }

88  void rewind()

117  const SGPiece* nextPiece()
