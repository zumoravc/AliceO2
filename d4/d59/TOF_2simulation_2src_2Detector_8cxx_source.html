61 Bool_t Detector::ProcessHits(FairVolume* v)
62 {
63  // This method is called from the MC stepping for the sensitive volume only
64  if (static_cast<int>(fMC->TrackCharge()) == 0) {
65  // set a very large step size for neutral particles
66  return kFALSE; // take only charged particles
67  }
68 
69  float pos2x, pos2y, pos2z;
70  fMC->TrackPosition(pos2x, pos2y, pos2z);
71  Float_t radius = std::sqrt(pos2x * pos2x + pos2y * pos2y);
72  LOG(DEBUG) << "Process hit in TOF volume ar R=" << radius << " - Z=" << pos2z;
73 
74  Float_t enDep = fMC->Edep();
75  if (enDep < 1E-8)
76  return kFALSE; // wo se need a threshold?
77 
78  // ADD HIT
79  float posx, posy, posz;
80  fMC->TrackPosition(posx, posy, posz);
81  float time = fMC->TrackTime() * 1.0e09;
82  auto stack = static_cast<o2::data::Stack*>(fMC->GetStack());
83  int trackID = stack->GetCurrentTrackNumber();
84  int sensID = v->getMCid();
85  Int_t det[5];
86  Float_t pos[3] = {posx, posy, posz};
87  Float_t delta[3];
88  Geo::getPadDxDyDz(pos, det, delta);
89  auto channel = Geo::getIndex(det);
90  HitType newhit(posx, posy, posz, time, enDep, trackID, sensID);
91  if (channel != mLastChannelID || !isMergable(newhit, mHits->back())) {
92  mHits->push_back(newhit);
93  stack->addHit(GetDetId());
94  } else {
95  mHits->back().SetEnergyLoss(mHits->back().GetEnergyLoss() + newhit.GetEnergyLoss());
96  // LOG(INFO)<<"Merging hit "<<"\n";
97  // <<mHits->back().GetId()<<"with new hit "<<newhit.GetId()<<"\n";
98  }
99  mLastChannelID = channel;
100 
101  return kTRUE;
102 }

277 void Detector::MaterialMixer(Float_t* p, const Float_t* const a, const Float_t* const m, Int_t n) const
278 {
279  // a[] atomic weights vector (in)
280  // (atoms present in more compound appear separately)
281  // m[] number of corresponding atoms in the compound (in)
282  Float_t t = 0.;
283  for (Int_t i = 0; i < n; ++i) {
284  p[i] = a[i] * m[i];
285  t += p[i];
286  }
287  for (Int_t i = 0; i < n; ++i) {
288  p[i] = p[i] / t;
289  }
290 }
