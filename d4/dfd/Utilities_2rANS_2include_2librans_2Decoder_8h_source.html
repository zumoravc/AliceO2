87 void Decoder<coder_T, stream_T, source_T>::process(const source_IT outputBegin, const stream_IT inputBegin, size_t numSymbols) const
88 {
89  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
90  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
91 
92  State<coder_T> rans0, rans1;
93  stream_T* ptr = &(*inputBegin);
94  source_IT it = outputBegin;
95  ransDecoder::decInit(&rans0, &ptr);
96  ransDecoder::decInit(&rans1, &ptr);
97 
98  for (size_t i = 0; i < (numSymbols & ~1); i += 2) {
99  const stream_T s0 =
100  (*mReverseLUT)[ransDecoder::decGet(&rans0, mProbabilityBits)];
101  const stream_T s1 =
102  (*mReverseLUT)[ransDecoder::decGet(&rans1, mProbabilityBits)];
103  *it++ = s0;
104  *it++ = s1;
105  ransDecoder::decAdvanceSymbolStep(&rans0, &(*mSymbolTable)[s0],
106  mProbabilityBits);
107  ransDecoder::decAdvanceSymbolStep(&rans1, &(*mSymbolTable)[s1],
108  mProbabilityBits);
109  ransDecoder::decRenorm(&rans0, &ptr);
110  ransDecoder::decRenorm(&rans1, &ptr);
111  }
112 
113  // last byte, if number of bytes was odd
114  if (numSymbols & 1) {
115  const stream_T s0 =
116  (*mReverseLUT)[ransDecoder::decGet(&rans0, mProbabilityBits)];
117  *it = s0;
118  ransDecoder::decAdvanceSymbol(&rans0, &ptr, &(*mSymbolTable)[s0],
119  mProbabilityBits);
120  }
121 }
