62 int GPUCATracking::runTracking(GPUO2InterfaceIOPtrs* data, GPUInterfaceOutputs* outputs)
63 {
64  if ((int)(data->tpcZS != nullptr) + (int)(data->o2Digits != nullptr) + (int)(data->clusters != nullptr) != 1) {
65  return 0;
66  }
67 
68  std::vector<TrackTPC>* outputTracks = data->outputTracks;
69  std::vector<uint32_t>* outClusRefs = data->outputClusRefs;
70  MCLabelContainer* outputTracksMCTruth = data->outputTracksMCTruth;
71 
72  if (!outputTracks || !outClusRefs) {
73  LOG(ERROR) << "Output tracks or clusRefs vectors are not initialized";
74  return 0;
75  }
76  auto& detParam = ParameterDetector::Instance();
77  auto& gasParam = ParameterGas::Instance();
78  auto& elParam = ParameterElectronics::Instance();
79  float vzbin = (elParam.ZbinWidth * gasParam.DriftV);
80  float vzbinInv = 1.f / vzbin;
81  Mapper& mapper = Mapper::instance();
82 
83  const ClusterNativeAccess* clusters;
84  std::vector<o2::tpc::Digit> gpuDigits[Sector::MAXSECTOR];
85  GPUTrackingInOutDigits gpuDigitsMap;
86  GPUTPCDigitsMCInput gpuDigitsMC;
87  GPUTrackingInOutPointers ptrs;
88 
89  if (data->tpcZS) {
90  ptrs.tpcZS = data->tpcZS;
91  } else if (data->o2Digits) {
92  ptrs.clustersNative = nullptr;
93  const float zsThreshold = mTrackingCAO2Interface->getConfig().configReconstruction.tpcZSthreshold;
94  const int maxContTimeBin = (o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * o2::constants::lhc::LHCMaxBunches + Constants::LHCBCPERTIMEBIN - 1) / Constants::LHCBCPERTIMEBIN;
95  for (int i = 0; i < Sector::MAXSECTOR; i++) {
96  const auto& d = (*(data->o2Digits))[i];
97  gpuDigits[i].reserve(d.size());
98  gpuDigitsMap.tpcDigits[i] = gpuDigits[i].data();
99  for (int j = 0; j < d.size(); j++) {
100  if (d[j].getTimeStamp() >= maxContTimeBin) {
101  throw std::runtime_error("Digit time bin exceeds time frame length");
102  }
103  if (d[j].getChargeFloat() >= zsThreshold) {
104  gpuDigits[i].emplace_back(d[j]);
105  }
106  }
107  gpuDigitsMap.nTPCDigits[i] = gpuDigits[i].size();
108  }
109  if (data->o2DigitsMC) {
110  for (int i = 0; i < Sector::MAXSECTOR; i++) {
111  gpuDigitsMC.v[i] = (*data->o2DigitsMC)[i].get();
112  }
113  gpuDigitsMap.tpcDigitsMC = &gpuDigitsMC;
114  }
115  ptrs.tpcPackedDigits = &gpuDigitsMap;
116  } else {
117  clusters = data->clusters;
118  ptrs.clustersNative = clusters;
119  ptrs.tpcPackedDigits = nullptr;
120  }
121  int retVal = mTrackingCAO2Interface->RunTracking(&ptrs, outputs);
122  if (data->o2Digits || data->tpcZS) {
123  clusters = ptrs.clustersNative;
124  }
125  const GPUTPCGMMergedTrack* tracks = ptrs.mergedTracks;
126  int nTracks = ptrs.nMergedTracks;
127  const GPUTPCGMMergedTrackHit* trackClusters = ptrs.mergedTrackHits;
128 
129  if (retVal) {
130  return retVal;
131  }
132 
133  std::vector<std::pair<int, float>> trackSort(nTracks);
134  int tmp = 0, tmp2 = 0;
135  uint32_t clBuff = 0;
136  for (char cside = 0; cside < 2; cside++) {
137  for (int i = 0; i < nTracks; i++) {
138  if (tracks[i].OK() && tracks[i].CSide() == cside) {
139  trackSort[tmp++] = {i, tracks[i].GetParam().GetTZOffset()};
140  auto ncl = tracks[i].NClusters();
141  clBuff += ncl + (ncl + 1) / 2; // actual N clusters to store will be less
142  }
143  }
144  std::sort(trackSort.data() + tmp2, trackSort.data() + tmp,
145  [](const auto& a, const auto& b) { return (a.second > b.second); });
146  tmp2 = tmp;
147  if (cside == 0)
148  mNTracksASide = tmp;
149  }
150  nTracks = tmp;
151 
152  outputTracks->resize(nTracks);
153  outClusRefs->resize(clBuff);
154  clBuff = 0;
155 
156  for (int iTmp = 0; iTmp < nTracks; iTmp++) {
157  auto& oTrack = (*outputTracks)[iTmp];
158  const int i = trackSort[iTmp].first;
159  float time0 = 0.f, tFwd = 0.f, tBwd = 0.f;
160 
161  if (mTrackingCAO2Interface->GetParamContinuous()) {
162  time0 = tracks[i].GetParam().GetTZOffset();
163 
164  if (tracks[i].CCE()) {
165  bool lastSide = trackClusters[tracks[i].FirstClusterRef()].slice < Sector::MAXSECTOR / 2;
166  float delta = 0.f;
167  for (int iCl = 1; iCl < tracks[i].NClusters(); iCl++) {
168  if (lastSide ^ (trackClusters[tracks[i].FirstClusterRef() + iCl].slice < Sector::MAXSECTOR / 2)) {
169  auto& cacl1 = trackClusters[tracks[i].FirstClusterRef() + iCl];
170  auto& cacl2 = trackClusters[tracks[i].FirstClusterRef() + iCl - 1];
171  auto& cl1 = clusters->clustersLinear[cacl1.num];
172  auto& cl2 = clusters->clustersLinear[cacl2.num];
173  delta = fabs(cl1.getTime() - cl2.getTime()) * 0.5f;
174  break;
175  }
176  }
177  tFwd = tBwd = delta;
178  } else {
179  // estimate max/min time increments which still keep track in the physical limits of the TPC
180  auto& c1 = trackClusters[tracks[i].FirstClusterRef()];
181  auto& c2 = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1];
182  float t1 = clusters->clustersLinear[c1.num].getTime();
183  float t2 = clusters->clustersLinear[c2.num].getTime();
184  auto times = std::minmax(t1, t2);
185  tFwd = times.first - time0;
186  tBwd = time0 - (times.second - detParam.TPClength * vzbinInv);
187  }
188  }
189 
190  oTrack =
191  TrackTPC(tracks[i].GetParam().GetX(), tracks[i].GetAlpha(),
192  {tracks[i].GetParam().GetY(), tracks[i].GetParam().GetZ(), tracks[i].GetParam().GetSinPhi(),
193  tracks[i].GetParam().GetDzDs(), tracks[i].GetParam().GetQPt()},
194  {tracks[i].GetParam().GetCov(0), tracks[i].GetParam().GetCov(1), tracks[i].GetParam().GetCov(2),
195  tracks[i].GetParam().GetCov(3), tracks[i].GetParam().GetCov(4), tracks[i].GetParam().GetCov(5),
196  tracks[i].GetParam().GetCov(6), tracks[i].GetParam().GetCov(7), tracks[i].GetParam().GetCov(8),
197  tracks[i].GetParam().GetCov(9), tracks[i].GetParam().GetCov(10), tracks[i].GetParam().GetCov(11),
198  tracks[i].GetParam().GetCov(12), tracks[i].GetParam().GetCov(13), tracks[i].GetParam().GetCov(14)});
199  oTrack.setTime0(time0);
200  oTrack.setDeltaTBwd(tBwd);
201  oTrack.setDeltaTFwd(tFwd);
202  if (tracks[i].CCE()) {
203  oTrack.setHasCSideClusters();
204  oTrack.setHasASideClusters();
205  } else if (tracks[i].CSide()) {
206  oTrack.setHasCSideClusters();
207  } else {
208  oTrack.setHasASideClusters();
209  }
210 
211  oTrack.setChi2(tracks[i].GetParam().GetChi2());
212  auto& outerPar = tracks[i].OuterParam();
213  oTrack.setdEdx(tracks[i].dEdxInfo());
214  oTrack.setOuterParam(o2::track::TrackParCov(
215  outerPar.X, outerPar.alpha,
216  {outerPar.P[0], outerPar.P[1], outerPar.P[2], outerPar.P[3], outerPar.P[4]},
217  {outerPar.C[0], outerPar.C[1], outerPar.C[2], outerPar.C[3], outerPar.C[4], outerPar.C[5],
218  outerPar.C[6], outerPar.C[7], outerPar.C[8], outerPar.C[9], outerPar.C[10], outerPar.C[11],
219  outerPar.C[12], outerPar.C[13], outerPar.C[14]}));
220  int nOutCl = 0;
221  for (int j = 0; j < tracks[i].NClusters(); j++) {
222  if (!(trackClusters[tracks[i].FirstClusterRef() + j].state & GPUTPCGMMergedTrackHit::flagReject)) {
223  nOutCl++;
224  }
225  }
226  oTrack.setClusterRef(clBuff, nOutCl); // register the references
227  uint32_t* clIndArr = &(*outClusRefs)[clBuff]; // cluster indices start here
228  uint8_t* sectorIndexArr = reinterpret_cast<uint8_t*>(clIndArr + nOutCl);
229  uint8_t* rowIndexArr = sectorIndexArr + nOutCl;
230 
231  clBuff += nOutCl + (nOutCl + 1) / 2;
232  std::vector<std::pair<MCCompLabel, unsigned int>> labels;
233  nOutCl = 0;
234  for (int j = 0; j < tracks[i].NClusters(); j++) {
235  if (trackClusters[tracks[i].FirstClusterRef() + j].state & GPUTPCGMMergedTrackHit::flagReject) {
236  continue;
237  }
238  int clusterIdGlobal = trackClusters[tracks[i].FirstClusterRef() + j].num;
239  Sector sector = trackClusters[tracks[i].FirstClusterRef() + j].slice;
240  int globalRow = trackClusters[tracks[i].FirstClusterRef() + j].row;
241  int clusterIdInRow = clusterIdGlobal - clusters->clusterOffset[sector][globalRow];
242  int regionNumber = 0;
243  while (globalRow > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber)) {
244  regionNumber++;
245  }
246  clIndArr[nOutCl] = clusterIdInRow;
247  sectorIndexArr[nOutCl] = sector;
248  rowIndexArr[nOutCl] = globalRow;
249  nOutCl++;
250  if (outputTracksMCTruth && clusters->clustersMCTruth) {
251  for (const auto& element : clusters->clustersMCTruth->getLabels(clusterIdGlobal)) {
252  bool found = false;
253  for (int l = 0; l < labels.size(); l++) {
254  if (labels[l].first == element) {
255  labels[l].second++;
256  found = true;
257  break;
258  }
259  }
260  if (!found)
261  labels.emplace_back(element, 1);
262  }
263  }
264  }
265  if (outputTracksMCTruth) {
266  if (labels.size() == 0) {
267  outputTracksMCTruth->addElement(iTmp, MCCompLabel()); //default constructor creates NotSet label
268  } else {
269  int bestLabelNum = 0, bestLabelCount = 0;
270  for (int j = 0; j < labels.size(); j++) {
271  if (labels[j].second > bestLabelCount) {
272  bestLabelNum = j;
273  bestLabelCount = labels[j].second;
274  }
275  }
276  MCCompLabel& bestLabel = labels[bestLabelNum].first;
277  if (bestLabelCount < (1.f - sTrackMCMaxFake) * nOutCl) {
278  bestLabel.setFakeFlag();
279  }
280  outputTracksMCTruth->addElement(iTmp, bestLabel);
281  }
282  }
283  int lastSector = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].slice;
284  }
285  outClusRefs->resize(clBuff); // remove overhead
286  data->compressedClusters = ptrs.tpcCompressedClusters;
287  if (data->o2Digits || data->tpcZS) {
288  data->clusters = ptrs.clustersNative;
289  }
290  mTrackingCAO2Interface->Clear(false);
291 
292  return (retVal);
293 }
