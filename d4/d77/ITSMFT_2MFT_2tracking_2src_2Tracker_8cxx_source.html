43 void Tracker::computeCells(ROframe& event)
44 {
45  MCCompLabel mcCompLabel;
46  Int_t layer1, layer2;
47  Int_t nClsInLayer1, nClsInLayer2;
48  Int_t disk1, disk2;
49  Int_t binR_proj, binPhi_proj, bin;
50  Int_t binIndex, clsMinIndex, clsMaxIndex;
51  std::array<Int_t, 3> binsR, binsPhi;
52  Int_t cellId = 0;
53  for (layer1 = 0; layer1 < (constants::mft::LayersNumber - 1); ++layer1) {
54  nClsInLayer1 = event.getClustersInLayer(layer1).size();
55  disk1 = layer1 / 2;
56  layer2 = layer1 + 1; // only (L+1) tracklets !
57  disk2 = layer2 / 2;
58  nClsInLayer2 = event.getClustersInLayer(layer2).size();
59  for (Int_t clsLayer1 = 0; clsLayer1 < nClsInLayer1; ++clsLayer1) {
60  const Cluster& cluster1 = event.getClustersInLayer(layer1).at(clsLayer1);
61  // project to next layer and get the bin index in R and Phi
62  getRPhiProjectionBin(cluster1, layer1, layer2, binR_proj, binPhi_proj);
63  // define the search window in bins x bins (3x3, 5x5, etc.)
64  for (Int_t i = 0; i < constants::index_table::LTFinterBinWin; ++i) {
65  binsR[i] = binR_proj + (i - constants::index_table::LTFinterBinWin / 2);
66  binsPhi[i] = binPhi_proj + (i - constants::index_table::LTFinterBinWin / 2);
67  }
68  // loop over the bins in the search window
69  for (auto binR : binsR) {
70  for (auto binPhi : binsPhi) {
71  // the global bin index
72  bin = constants::index_table::getBinIndex(binR, binPhi);
73  if (!getBinClusterRange(event, layer2, bin, clsMinIndex, clsMaxIndex)) {
74  continue;
75  }
76  for (Int_t clsLayer2 = clsMinIndex; clsLayer2 <= clsMaxIndex; ++clsLayer2) {
77  event.addCellToLayer(layer1, layer2, clsLayer1, clsLayer2, cellId++);
78  } // end clusters bin layer2
79  } // end binPhi
80  } // end binR
81  } // end clusters layer1
82  } // end layers
83 }

85 void Tracker::findTracksLTF(ROframe& event)
86 {
87  // find (high momentum) tracks by the Linear Track Finder (LTF) method
88 
89  MCCompLabel mcCompLabel;
90  Int_t layer1, layer2, nPointDisks;
91  Int_t nClsInLayer1, nClsInLayer2, nClsInLayer;
92  Int_t binR_proj, binPhi_proj, bin;
93  Int_t binIndex, clsMinIndex, clsMaxIndex, clsMinIndexS, clsMaxIndexS;
94  Float_t dR, dRmin, dRcut = constants::mft::LTFclsRCut;
95  std::vector<Int_t> binsR, binsPhi, binsRS, binsPhiS;
96  Bool_t hasDisk[constants::mft::DisksNumber], newPoint, seed = kTRUE;
97 
98  binsRS.resize(constants::index_table::LTFseed2BinWin);
99  binsPhiS.resize(constants::index_table::LTFseed2BinWin);
100 
101  binsR.resize(constants::index_table::LTFinterBinWin);
102  binsPhi.resize(constants::index_table::LTFinterBinWin);
103 
104  Int_t step = 0;
105 
106  layer1 = 0;
107 
108  while (seed) {
109 
110  if (step == 0) {
111  layer2 = constants::mft::LayersNumber - 1;
112  } else {
113  layer2--;
114  }
115 
116  step++;
117 
118  if (layer2 < layer1 + (constants::mft::MinTrackPoints - 1)) {
119  ++layer1;
120  if (layer1 > (constants::mft::LayersNumber - (constants::mft::MinTrackPoints - 1))) {
121  break;
122  }
123  step = 0;
124  continue;
125  }
126 
127  nClsInLayer1 = event.getClustersInLayer(layer1).size();
128  nClsInLayer2 = event.getClustersInLayer(layer2).size();
129 
130  for (Int_t clsLayer1 = 0; clsLayer1 < nClsInLayer1; ++clsLayer1) {
131  if (event.isClusterUsed(layer1, clsLayer1)) {
132  continue;
133  }
134  const Cluster& cluster1 = event.getClustersInLayer(layer1)[clsLayer1];
135 
136  // project to the second seed layer and get the bin index in R and Phi
137  getRPhiProjectionBin(cluster1, layer1, layer2, binR_proj, binPhi_proj);
138  // define the search window in bins x bins (3x3, 5x5, etc.)
139  for (Int_t i = 0; i < constants::index_table::LTFseed2BinWin; ++i) {
140  binsRS[i] = binR_proj + (i - constants::index_table::LTFseed2BinWin / 2);
141  binsPhiS[i] = binPhi_proj + (i - constants::index_table::LTFseed2BinWin / 2);
142  }
143  // loop over the bins in the search window
144  for (auto binRS : binsRS) {
145  for (auto binPhiS : binsPhiS) {
146  // the global bin index
147  bin = constants::index_table::getBinIndex(binRS, binPhiS);
148  if (!getBinClusterRange(event, layer2, bin, clsMinIndexS, clsMaxIndexS)) {
149  continue;
150  }
151  for (Int_t clsLayer2 = clsMinIndexS; clsLayer2 <= clsMaxIndexS; ++clsLayer2) {
152  if (event.isClusterUsed(layer2, clsLayer2)) {
153  continue;
154  }
155  const Cluster& cluster2 = event.getClustersInLayer(layer2)[clsLayer2];
156 
157  for (Int_t i = 0; i < (constants::mft::DisksNumber); i++) {
158  hasDisk[i] = kFALSE;
159  }
160 
161  hasDisk[layer1 / 2] = kTRUE;
162  hasDisk[layer2 / 2] = kTRUE;
163 
164  // start a track LTF
165  event.addTrackLTF();
166 
167  // add the first seed-point
168  mcCompLabel = event.getClusterLabels(layer1, cluster1.clusterId);
169  newPoint = kTRUE;
170  event.getCurrentTrackLTF().setPoint(cluster1.xCoordinate, cluster1.yCoordinate, cluster1.zCoordinate, layer1, clsLayer1, mcCompLabel, newPoint);
171 
172  for (Int_t layer = (layer1 + 1); layer <= (layer2 - 1); ++layer) {
173 
174  nClsInLayer = event.getClustersInLayer(layer).size();
175 
176  newPoint = kTRUE;
177 
178  // project to the intermediate layer and get the bin index in R and Phi
179  getRPhiProjectionBin(cluster1, layer1, layer, binR_proj, binPhi_proj);
180 
181  // define the search window in bins x bins (3x3, 5x5, etc.)
182  for (Int_t i = 0; i < constants::index_table::LTFinterBinWin; ++i) {
183  binsR[i] = binR_proj + (i - constants::index_table::LTFinterBinWin / 2);
184  binsPhi[i] = binPhi_proj + (i - constants::index_table::LTFinterBinWin / 2);
185  }
186  // loop over the bins in the search window
187  dRmin = dRcut;
188  for (auto binR : binsR) {
189  for (auto binPhi : binsPhi) {
190  // the global bin index
191  bin = constants::index_table::getBinIndex(binR, binPhi);
192  if (!getBinClusterRange(event, layer, bin, clsMinIndex, clsMaxIndex)) {
193  continue;
194  }
195  for (Int_t clsLayer = clsMinIndex; clsLayer <= clsMaxIndex; ++clsLayer) {
196  if (event.isClusterUsed(layer, clsLayer)) {
197  continue;
198  }
199  const Cluster& cluster = event.getClustersInLayer(layer)[clsLayer];
200 
201  dR = getDistanceToSeed(cluster1, cluster2, cluster);
202  // retain the closest point within a radius dRcut
203  if (dR >= dRmin) {
204  continue;
205  }
206  dRmin = dR;
207 
208  hasDisk[layer / 2] = kTRUE;
209  mcCompLabel = event.getClusterLabels(layer, cluster.clusterId);
210  event.getCurrentTrackLTF().setPoint(cluster.xCoordinate, cluster.yCoordinate, cluster.zCoordinate, layer, clsLayer, mcCompLabel, newPoint);
211  } // end clusters bin intermediate layer
212  } // end intermediate layers
213  } // end binPhi
214  } // end binR
215 
216  // add the second seed-point
217  mcCompLabel = event.getClusterLabels(layer2, cluster2.clusterId);
218  newPoint = kTRUE;
219  event.getCurrentTrackLTF().setPoint(cluster2.xCoordinate, cluster2.yCoordinate, cluster2.zCoordinate, layer2, clsLayer2, mcCompLabel, newPoint);
220 
221  // keep only tracks fulfilling the minimum length condition
222  if (event.getCurrentTrackLTF().getNPoints() < constants::mft::MinTrackPoints) {
223  event.removeCurrentTrackLTF();
224  continue;
225  }
226  nPointDisks = 0;
227  for (Int_t disk = 0; disk < (constants::mft::DisksNumber); ++disk) {
228  if (hasDisk[disk])
229  ++nPointDisks;
230  }
231  if (nPointDisks < constants::mft::MinTrackPoints) {
232  event.removeCurrentTrackLTF();
233  continue;
234  }
235  // mark the used clusters
236  //Int_t lay, layMin = 10, layMax = -1;
237  for (Int_t point = 0; point < event.getCurrentTrackLTF().getNPoints(); ++point) {
238  event.markUsedCluster(event.getCurrentTrackLTF().getLayers()[point], event.getCurrentTrackLTF().getClustersId()[point]);
239  //lay = event.getCurrentTrackLTF().getLayers()[point];
240  //layMin = (lay < layMin) ? lay : layMin;
241  //layMax = (lay > layMax) ? lay : layMax;
242  }
243 
244  } // end seed clusters bin layer2
245  } // end binPhi
246  } // end binR
247  } // end clusters layer1
248 
249  } // end seeding
250 }

252 void Tracker::findTracksCA(ROframe& event)
253 {
254  // layers: 0, 1, 2, ..., 9
255  // rules for combining first/last plane in a road:
256  // 0 with 6, 7, 8, 9
257  // 1 with 6, 7, 8, 9
258  // 2 with 8, 9
259  // 3 with 8, 9
260  Int_t layer1Min = 0, layer1Max = 3;
261  Int_t layer2Min[4] = {6, 6, 8, 8};
262  Int_t layer2Max[4] = {9, 9, 9, 9};
263 
264  MCCompLabel mcCompLabel;
265  Int_t roadId, nPointDisks;
266  Int_t nClsInLayer1, nClsInLayer2, nClsInLayer;
267  Int_t binR_proj, binPhi_proj, bin;
268  Int_t binIndex, clsMinIndex, clsMaxIndex, clsMinIndexS, clsMaxIndexS;
269  Float_t dR, dRcut = constants::mft::ROADclsRCut;
270  std::vector<Int_t> binsR, binsPhi, binsRS, binsPhiS;
271  Bool_t hasDisk[constants::mft::DisksNumber], newPoint;
272 
273  binsRS.resize(constants::index_table::LTFseed2BinWin);
274  binsPhiS.resize(constants::index_table::LTFseed2BinWin);
275 
276  binsR.resize(constants::index_table::LTFinterBinWin);
277  binsPhi.resize(constants::index_table::LTFinterBinWin);
278 
279  roadId = 0;
280 
281  for (Int_t layer1 = layer1Min; layer1 <= layer1Max; ++layer1) {
282 
283  nClsInLayer1 = event.getClustersInLayer(layer1).size();
284 
285  for (Int_t layer2 = layer2Max[layer1]; layer2 >= layer2Min[layer1]; --layer2) {
286 
287  nClsInLayer2 = event.getClustersInLayer(layer2).size();
288 
289  for (Int_t clsLayer1 = 0; clsLayer1 < nClsInLayer1; ++clsLayer1) {
290 
291  if (event.isClusterUsed(layer1, clsLayer1)) {
292  continue;
293  }
294  const Cluster& cluster1 = event.getClustersInLayer(layer1)[clsLayer1];
295 
296  // project to the second seed layer and get the bin index in R and Phi
297  getRPhiProjectionBin(cluster1, layer1, layer2, binR_proj, binPhi_proj);
298  // define the search window in bins x bins (3x3, 5x5, etc.)
299  for (Int_t i = 0; i < constants::index_table::LTFseed2BinWin; ++i) {
300  binsRS[i] = binR_proj + (i - constants::index_table::LTFseed2BinWin / 2);
301  binsPhiS[i] = binPhi_proj + (i - constants::index_table::LTFseed2BinWin / 2);
302  }
303 
304  // loop over the bins in the search window
305  for (auto binRS : binsRS) {
306  for (auto binPhiS : binsPhiS) {
307  // the global bin index
308  bin = constants::index_table::getBinIndex(binRS, binPhiS);
309  if (!getBinClusterRange(event, layer2, bin, clsMinIndexS, clsMaxIndexS)) {
310  continue;
311  }
312  for (Int_t clsLayer2 = clsMinIndexS; clsLayer2 <= clsMaxIndexS; ++clsLayer2) {
313  if (event.isClusterUsed(layer2, clsLayer2)) {
314  continue;
315  }
316  const Cluster& cluster2 = event.getClustersInLayer(layer2)[clsLayer2];
317 
318  for (Int_t i = 0; i < (constants::mft::DisksNumber); i++) {
319  hasDisk[i] = kFALSE;
320  }
321 
322  hasDisk[layer1 / 2] = kTRUE;
323  hasDisk[layer2 / 2] = kTRUE;
324 
325  // start a road
326  event.addRoad();
327 
328  // add the 1st/2nd road points
329  mcCompLabel = event.getClusterLabels(layer1, cluster1.clusterId);
330  newPoint = kTRUE;
331  event.getCurrentRoad().setPoint(cluster1.xCoordinate, cluster1.yCoordinate, cluster1.zCoordinate, layer1, clsLayer1, mcCompLabel, newPoint);
332 
333  for (Int_t layer = (layer1 + 1); layer <= (layer2 - 1); ++layer) {
334 
335  nClsInLayer = event.getClustersInLayer(layer).size();
336 
337  // project to the intermediate layer and get the bin index in R and Phi
338  getRPhiProjectionBin(cluster1, layer1, layer, binR_proj, binPhi_proj);
339  // define the search window in bins x bins (3x3, 5x5, etc.)
340  for (Int_t i = 0; i < constants::index_table::LTFinterBinWin; ++i) {
341  binsR[i] = binR_proj + (i - constants::index_table::LTFinterBinWin / 2);
342  binsPhi[i] = binPhi_proj + (i - constants::index_table::LTFinterBinWin / 2);
343  }
344 
345  // loop over the bins in the search window
346  for (auto binR : binsR) {
347  for (auto binPhi : binsPhi) {
348  // the global bin index
349  bin = constants::index_table::getBinIndex(binR, binPhi);
350  if (!getBinClusterRange(event, layer, bin, clsMinIndex, clsMaxIndex)) {
351  continue;
352  }
353  for (Int_t clsLayer = clsMinIndex; clsLayer <= clsMaxIndex; ++clsLayer) {
354  if (event.isClusterUsed(layer, clsLayer)) {
355  continue;
356  }
357  const Cluster& cluster = event.getClustersInLayer(layer)[clsLayer];
358 
359  dR = getDistanceToSeed(cluster1, cluster2, cluster);
360  // add all points within a radius dRcut
361  if (dR >= dRcut) {
362  continue;
363  }
364 
365  hasDisk[layer / 2] = kTRUE;
366  mcCompLabel = event.getClusterLabels(layer, cluster.clusterId);
367  newPoint = kTRUE;
368  event.getCurrentRoad().setPoint(cluster.xCoordinate, cluster.yCoordinate, cluster.zCoordinate, layer, clsLayer, mcCompLabel, newPoint);
369 
370  } // end clusters bin intermediate layer
371  } // end intermediate layers
372  } // end binPhi
373  } // end binR
374 
375  // add the second seed-point
376  mcCompLabel = event.getClusterLabels(layer2, cluster2.clusterId);
377  newPoint = kTRUE;
378  event.getCurrentRoad().setPoint(cluster2.xCoordinate, cluster2.yCoordinate, cluster2.zCoordinate, layer2, clsLayer2, mcCompLabel, newPoint);
379 
380  // keep only roads fulfilling the minimum length condition
381  if (event.getCurrentRoad().getNPoints() < constants::mft::MinTrackPoints) {
382  event.removeCurrentRoad();
383  continue;
384  }
385  nPointDisks = 0;
386  for (Int_t disk = 0; disk < (constants::mft::DisksNumber); ++disk) {
387  if (hasDisk[disk])
388  ++nPointDisks;
389  }
390  if (nPointDisks < constants::mft::MinTrackPoints) {
391  event.removeCurrentRoad();
392  continue;
393  }
394  event.getCurrentRoad().setNDisks(nPointDisks);
395  event.getCurrentRoad().setRoadId(roadId);
396  ++roadId;
397 
398  computeCellsInRoad(event.getCurrentRoad());
399  runForwardInRoad(event);
400  runBackwardInRoad(event);
401 
402  } // end clusters bin layer2
403  } // end binPhiS
404  } // end binRS
405  } // end clusters in layer1
406  } // end layer2
407  } // end layer1
408 }

410 void Tracker::computeCellsInRoad(Road& road)
411 {
412  Int_t layer1, layer1min, layer1max, layer2, layer2min, layer2max;
413  Int_t nPtsInLayer1, nPtsInLayer2;
414  Int_t clsLayer1, clsLayer2;
415  Bool_t noCell;
416 
417  road.getLength(layer1min, layer1max);
418  --layer1max;
419 
420  Int_t cellId = 0;
421  for (layer1 = layer1min; layer1 <= layer1max; ++layer1) {
422  layer2min = layer1 + 1;
423  layer2max = std::min(layer1 + (constants::mft::DisksNumber - isDiskFace(layer1)), constants::mft::LayersNumber - 1);
424  nPtsInLayer1 = road.getNPointsInLayer(layer1);
425  for (Int_t point1 = 0; point1 < nPtsInLayer1; ++point1) {
426  clsLayer1 = road.getClustersIdInLayer(layer1)[point1];
427  layer2 = layer2min;
428  noCell = kTRUE;
429  while (noCell && (layer2 <= layer2max)) {
430  nPtsInLayer2 = road.getNPointsInLayer(layer2);
431  /*
432  if (nPtsInLayer2 > 1) {
433  LOG(INFO) << "BV===== more than one point in road " << road.getRoadId() << " in layer " << layer2 << " : " << nPtsInLayer2 << "\n";
434  }
435  */
436  for (Int_t point2 = 0; point2 < nPtsInLayer2; ++point2) {
437  clsLayer2 = road.getClustersIdInLayer(layer2)[point2];
438  noCell = kFALSE;
439  // create a cell
440  road.addCellToLayer(layer1, layer2, clsLayer1, clsLayer2, cellId++);
441  } // end points in layer2
442  ++layer2;
443  } // end while(noCell && (layer2 <= layer2max))
444  } // end points in layer1
445  } // end layer1
446 }

448 void Tracker::runForwardInRoad(ROframe& event)
449 {
450  Int_t layerR, layerL, icellR, icellL;
451  Int_t iter = 0;
452  Bool_t levelChange = kTRUE;
453 
454  Road& road = event.getCurrentRoad();
455 
456  while (levelChange) {
457 
458  levelChange = kFALSE;
459  ++iter;
460 
461  // R = right, L = left
462  for (layerL = 0; layerL < (constants::mft::LayersNumber - 2); ++layerL) {
463 
464  for (icellL = 0; icellL < road.getCellsInLayer(layerL).size(); ++icellL) {
465  const Cell& cellL = road.getCellsInLayer(layerL)[icellL];
466 
467  if (cellL.getLevel() == 0) {
468  continue;
469  }
470 
471  layerR = cellL.getSecondLayerId();
472  if (layerR >= (constants::mft::LayersNumber - 1)) {
473  continue;
474  }
475 
476  for (icellR = 0; icellR < road.getCellsInLayer(layerR).size(); ++icellR) {
477  const Cell& cellR = road.getCellsInLayer(layerR)[icellR];
478 
479  if (cellR.getLevel() == 0) {
480  continue;
481  }
482  if ((cellL.getLevel() == cellR.getLevel()) && getCellsConnect(event, cellL, cellR)) {
483  if (iter == 1) {
484  road.addRightNeighbourToCell(layerL, icellL, layerR, icellR);
485  road.addLeftNeighbourToCell(layerR, icellR, layerL, icellL);
486  }
487  road.incrementCellLevel(layerR, icellR);
488  levelChange = kTRUE;
489  } // end matching cells
490  } // end loop cellR
491  } // end loop cellL
492  } // end loop layer
493 
494  updateCellStatusInRoad(road);
495 
496  } // end while (step)
497 }

499 void Tracker::runBackwardInRoad(ROframe& event)
500 {
501  if (mMaxCellLevel == 1)
502  return; // we have only isolated cells
503 
504  Bool_t addCellToNewTrack, hasDisk[constants::mft::DisksNumber];
505 
506  Int_t iSelectChisquare, iSelectDeviation, lastCellLayer, lastCellId;
507  Int_t icell, layerC, cellIdC, nPointDisks;
508  Float_t chisquarePrev, deviationPrev, deviation, chisquare;
509 
510  // start layer
511  Int_t minLayer = 6;
512  Int_t maxLayer = 8;
513 
514  Road& road = event.getCurrentRoad();
515 
516  for (Int_t layer = maxLayer; layer >= minLayer; --layer) {
517 
518  for (icell = 0; icell < road.getCellsInLayer(layer).size(); ++icell) {
519  if (road.getCellLevel(layer, icell) == 0) {
520  continue;
521  }
522  if (road.isCellUsed(layer, icell)) {
523  continue;
524  }
525  if (road.getCellLevel(layer, icell) < (constants::mft::MinTrackPoints - 1)) {
526  continue;
527  }
528 
529  // start a track CA
530  event.addTrackCA();
531  event.getCurrentTrackCA().setRoadId(road.getRoadId());
532  if (addCellToCurrentTrackCA(layer, icell, event)) {
533  road.setCellUsed(layer, icell, kTRUE);
534  }
535 
536  // add cells to new track
537  addCellToNewTrack = kTRUE;
538  while (addCellToNewTrack) {
539  Int_t layerRC = event.getCurrentTrackCA().getCellsLayer()[event.getCurrentTrackCA().getNCells() - 1];
540  Int_t cellIdRC = event.getCurrentTrackCA().getCellsId()[event.getCurrentTrackCA().getNCells() - 1];
541  const Cell& cellRC = road.getCellsInLayer(layerRC)[cellIdRC];
542  addCellToNewTrack = kFALSE;
543 
544  // find the left neighbor giving the smalles chisquare
545  iSelectChisquare = 0;
546  chisquarePrev = 0.;
547 
548  // ... or
549 
550  // find the left neighbor giving the smallest deviation
551  iSelectDeviation = 0;
552  deviationPrev = -1.;
553 
554  // loop over left neighbours
555  deviationPrev = o2::constants::math::TwoPI;
556  chisquarePrev = 1.E5;
557 
558  for (Int_t iLN = 0; iLN < cellRC.getNLeftNeighbours(); ++iLN) {
559  auto leftNeighbour = cellRC.getLeftNeighbours()[iLN];
560  Int_t layerL = leftNeighbour.first;
561  Int_t cellIdL = leftNeighbour.second;
562 
563  const Cell& cellL = road.getCellsInLayer(layerL)[cellIdL];
564 
565  if (road.getCellLevel(layerL, cellIdL) == 0) {
566  continue;
567  }
568  if (road.isCellUsed(layerL, cellIdL)) {
569  continue;
570  }
571  if (road.getCellLevel(layerL, cellIdL) != (road.getCellLevel(layerRC, cellIdRC) - 1)) {
572  continue;
573  }
574  /*
575  // ... smallest deviation

662 {

775 const Bool_t Tracker::LinearRegression(Int_t npoints, Float_t* x, Float_t* y, Float_t* yerr, Float_t& apar, Float_t& aparerr, Float_t& bpar, Float_t& bparerr, Float_t& chisq, Int_t skippoint) const
776 {
777  // y = apar * x + bpar
778 
779  // work with only part of the points
780  Float_t xCl[constants::mft::MaxTrackPoints], yCl[constants::mft::MaxTrackPoints], yClErr[constants::mft::MaxTrackPoints];
781  Int_t ipoints = 0;
782  for (Int_t i = 0; i < npoints; ++i) {
783  if (i == skippoint) {
784  continue;
785  }
786  xCl[ipoints] = x[i];
787  yCl[ipoints] = y[i];
788  yClErr[ipoints] = yerr[i];
789  ipoints++;
790  }
791 
792  // calculate the regression parameters
793  Float_t S1, SXY, SX, SY, SXX, SsXY, SsXX, SsYY, Xm, Ym, s, delta, difx;
794  S1 = SXY = SX = SY = SXX = 0.0;
795  SsXX = SsYY = SsXY = Xm = Ym = 0.0;
796  difx = 0.;
797  for (Int_t i = 0; i < ipoints; ++i) {
798  S1 += 1.0 / (yClErr[i] * yClErr[i]);
799  SXY += xCl[i] * yCl[i] / (yClErr[i] * yClErr[i]);
800  SX += xCl[i] / (yClErr[i] * yClErr[i]);
801  SY += yCl[i] / (yClErr[i] * yClErr[i]);
802  SXX += xCl[i] * xCl[i] / (yClErr[i] * yClErr[i]);
803  if (i > 0)
804  difx += TMath::Abs(xCl[i] - xCl[i - 1]);
805  Xm += xCl[i];
806  Ym += yCl[i];
807  SsXX += xCl[i] * xCl[i];
808  SsYY += yCl[i] * yCl[i];
809  SsXY += xCl[i] * yCl[i];
810  }
811  delta = SXX * S1 - SX * SX;
812  if (delta == 0.) {
813  return kFALSE;
814  }
815  apar = (SXY * S1 - SX * SY) / delta;
816  bpar = (SY * SXX - SX * SXY) / delta;
817 
818  // calculate the chisquare
819  chisq = 0.0;
820  for (Int_t i = 0; i < ipoints; ++i) {
821  chisq += (yCl[i] - (apar * xCl[i] + bpar)) * (yCl[i] - (apar * xCl[i] + bpar)) / (yerr[i] * yerr[i]);
822  }
823 
824  // calculate the errors of the regression parameters
825  Ym /= (Float_t)ipoints;
826  Xm /= (Float_t)ipoints;
827  SsYY -= (Float_t)ipoints * (Ym * Ym);
828  SsXX -= (Float_t)ipoints * (Xm * Xm);
829  SsXY -= (Float_t)ipoints * (Ym * Xm);
830  Float_t eps = 1.E-24;
831  if ((ipoints > 2) && (TMath::Abs(difx) > eps) && ((SsYY - (SsXY * SsXY) / SsXX) > 0.0)) {
832  s = std::sqrt((SsYY - (SsXY * SsXY) / SsXX) / (ipoints - 2));
833  bparerr = s * std::sqrt(1. / (Float_t)ipoints + (Xm * Xm) / SsXX);
834  aparerr = s / std::sqrt(SsXX);
835  } else {
836  bparerr = 0.;
837  aparerr = 0.;
838  }
839 
840  return kTRUE;
841 }
