34  auto initFunction = [config, propagateMC, creator](o2::framework::InitContext& ic) {
35  // get the option from the init context
36  auto filename = ic.options().get<std::string>("infile");
37  auto treename = ic.options().get<std::string>("treename");
38  auto dtbrName = ic.options().get<std::string>(config.databranch.option.c_str()); // databranch name
39  auto trgbrName = ic.options().get<std::string>(config.triggerrecordbranch.option.c_str()); // triggerbranch name
40  auto mcbrName = ic.options().get<std::string>(config.mcbranch.option.c_str()); // mcbranch name
41  auto nofEvents = ic.options().get<int>("nevents");
42  auto publishingMode = nofEvents == -1 ? o2::framework::RootTreeReader::PublishingMode::Single : o2::framework::RootTreeReader::PublishingMode::Loop;
43 
44  auto processAttributes = std::make_shared<ProcessAttributes>();
45  {
46  using Reader = o2::framework::RootTreeReader;
47  using TriggerInputType = std::vector<o2::emcal::TriggerRecord>;
48  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
49  processAttributes->finished = false;
50  processAttributes->datatype = config.databranch.defval;
51  o2::header::DataHeader::SubSpecificationType subSpec = 0;
52  processAttributes->reader = creator(treename.c_str(), // tree name
53  filename.c_str(), // input file name
54  nofEvents, // number of entries to publish
55  publishingMode,
56  dtbrName.c_str(), // databranch name
57  trgbrName.c_str(), // triggerbranch name
58  mcbrName.c_str()); // mcbranch name
59  }
60 
61  auto processFunction = [processAttributes, propagateMC](o2::framework::ProcessingContext& pc) {
62  if (processAttributes->finished) {
63  return;
64  }
65 
66  auto publish = [&processAttributes, &pc, propagateMC]() {
67  o2::emcal::EMCALBlockHeader emcheader(true);
68  if (processAttributes->reader->next()) {
69  (*processAttributes->reader)(pc, emcheader);
70  } else {
71  processAttributes->reader.reset();
72  return false;
73  }
74  return true;
75  };
76 
77  if (!publish()) {
78  pc.services().get<o2::framework::ControlService>().endOfStream();
79  pc.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
80  }
81  };
82 
83  return processFunction;
84  };

86  auto createOutputSpecs = [&config, propagateMC]() {
87  std::vector<o2::framework::OutputSpec> outputSpecs;
88  auto dto = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.dataoutput);
89  auto tro = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.triggerrecordoutput);
90  auto mco = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.mcoutput);
91  outputSpecs.emplace_back(o2::framework::OutputSpec{{"output"}, dto.origin, dto.description, 0, o2::framework::Lifetime::Timeframe});
92  outputSpecs.emplace_back(o2::framework::OutputSpec{{"outputTRG"}, tro.origin, tro.description, 0, o2::framework::Lifetime::Timeframe});
93  if (propagateMC) {
94  outputSpecs.emplace_back(o2::framework::OutputSpec{{"outputMC"}, mco.origin, mco.description, 0, o2::framework::Lifetime::Timeframe});
95  }
96  return std::move(outputSpecs);
97  };
98 
99  auto& dtb = config.databranch;
100  auto& mcb = config.mcbranch;
101  auto& trb = config.triggerrecordbranch;
102  return o2::framework::DataProcessorSpec{
103  config.processName.c_str(),
104  o2::framework::Inputs{}, // no inputs
105  {createOutputSpecs()},
106  o2::framework::AlgorithmSpec(initFunction),
107  o2::framework::Options{
108  {"infile", o2::framework::VariantType::String, "", {"Name of the input file"}},
109  {"treename", o2::framework::VariantType::String, config.defaultTreeName.c_str(), {"Name of input tree"}},
110  {dtb.option.c_str(), o2::framework::VariantType::String, dtb.defval.c_str(), {dtb.help.c_str()}},
111  {trb.option.c_str(), o2::framework::VariantType::String, trb.defval.c_str(), {trb.help.c_str()}},
112  {mcb.option.c_str(), o2::framework::VariantType::String, mcb.defval.c_str(), {mcb.help.c_str()}},
113  {"nevents", o2::framework::VariantType::Int, -1, {"number of events to run"}},
114  {"terminate-on-eod", o2::framework::VariantType::Bool, true, {"terminate on end-of-data"}},
115  }};
116 }
