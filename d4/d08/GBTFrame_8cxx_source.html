184 void GBTFrame::setAdcClock(int sampa, int clock)
185 {
186  switch (sampa) {
187  case 0:
188  mWords[1] = (mWords[1] & 0xFFFFF0FF) | ((clock & 0xF) << 8);
189  break;
190  case 1:
191  mWords[2] = (mWords[2] & 0xFF0FFFFF) | ((clock & 0xF) << 20);
192  break;
193  case 2:
194  mWords[3] = (mWords[3] & 0xFFFF0FFF) | ((clock & 0xF) << 12);
195  break;
196  case -1:
197  mWords[1] = (mWords[1] & 0xFFFFF0FF) | ((clock & 0xF) << 8);
198  mWords[2] = (mWords[2] & 0xFF0FFFFF) | ((clock & 0xF) << 20);
199  mWords[3] = (mWords[3] & 0xFFFF0FFF) | ((clock & 0xF) << 12);
200  break;
201  default:
202  std::cout << "don't know SAMPA " << sampa << std::endl;
203  break;
204  }
205  calculateAdcClock();
206 }
