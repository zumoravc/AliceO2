94 template <typename stream_IT, typename source_IT>
95 const stream_IT Encoder<coder_T, stream_T, source_T>::Encoder::process(
96  const stream_IT outputBegin, const stream_IT outputEnd, const source_IT inputBegin, const source_IT inputEnd) const
97 {
98  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
99  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
100 
101  State<coder_T> rans0, rans1;
102  ransCoder::encInit(&rans0);
103  ransCoder::encInit(&rans1);
104 
105  stream_T* ptr = &(*outputEnd);
106  source_IT inputIT = inputEnd;
107 
108  const auto inputBufferSize = inputEnd - inputBegin;
109 
110  // odd number of bytes?
111  if (inputBufferSize & 1) {
112  const coder_T s = *(--inputIT);
113  ransCoder::encPutSymbol(&rans0, &ptr, &(*mSymbolTable)[s],
114  mProbabilityBits);
115  }
116 
117  while (inputIT > inputBegin) { // NB: working in reverse!
118  const coder_T s1 = *(--inputIT);
119  const coder_T s0 = *(--inputIT);
120  ransCoder::encPutSymbol(&rans1, &ptr, &(*mSymbolTable)[s1],
121  mProbabilityBits);
122  ransCoder::encPutSymbol(&rans0, &ptr, &(*mSymbolTable)[s0],
123  mProbabilityBits);
124  }
125  ransCoder::encFlush(&rans1, &ptr);
126  ransCoder::encFlush(&rans0, &ptr);
127 
128  assert(&(*outputBegin) < ptr);
129 
130  return outputBegin + std::distance(&(*outputBegin), ptr);
131 };
