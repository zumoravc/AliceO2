264 GPUdi() void TPCFastTransform::convPadTimeToUV(int slice, int row, float pad, float time, float& u, float& v, float vertexTime) const
265 {
266  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
267 
268  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
269  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
270 
271  float x = rowInfo.x;
272  u = (pad - 0.5 * rowInfo.maxPad) * rowInfo.padWidth;
273 
274  float y = sideC ? -u : u; // pads are mirrorred on C-side
275  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
276 
277  v = (time - mT0 - vertexTime) * (mVdrift + mVdriftCorrY * yLab) + mLdriftCorr; // drift length cm
278 }

310 GPUdi() void TPCFastTransform::convUVtoPadTime(int slice, int row, float u, float v, float& pad, float& time, float vertexTime) const
311 {
312  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
313 
314  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
315  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
316 
317  pad = u / rowInfo.padWidth + 0.5 * rowInfo.maxPad;
318 
319  float x = rowInfo.x;
320  float y = sideC ? -u : u; // pads are mirrorred on C-side
321  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
322  time = mT0 + vertexTime + (v - mLdriftCorr) / (mVdrift + mVdriftCorrY * yLab);
323 }

343 GPUdi() void TPCFastTransform::getTOFcorrection(int slice, int /*row*/, float x, float y, float z, float& dz) const
344 {
345  // calculate time of flight correction for z coordinate
346 
347  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
348  float distZ = z - mPrimVtxZ;
349  float dv = -GPUCommonMath::Sqrt(x * x + y * y + distZ * distZ) * mTOFcorr;
350  dz = sideC ? dv : -dv;
351 }

353 GPUdi() void TPCFastTransform::Transform(int slice, int row, float pad, float time, float& x, float& y, float& z, float vertexTime) const
354 {
360 
361  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
362 
363  // const SliceInfo &sliceInfo = getSliceInfo( slice );
364  // bool sideC = ( slice >= NumberOfSlices / 2 );
365 
366  x = rowInfo.x;

372  mCorrection.getCorrection(slice, row, u, v, dx, du, dv);
373  x += dx;

376  }

379 
380  float dzTOF = 0;
381  getTOFcorrection(slice, row, x, y, z, dzTOF);
382  z += dzTOF;
