41 // Generic class to keep position, time and hit value
42 // T is basic type for position,
43 // E is basic type for time,
44 // V is basic type for hit value.
45 template <typename T, typename E, typename V = float>
46 class BasicXYZVHit : public BaseHit
47 {
48  Point3D<T> mPos; // cartesian position of Hit
49  E mTime; // time of flight
50  V mHitValue; // hit value
51  short mDetectorID; // the detector/sensor id
52 
53  public:
54  BasicXYZVHit() = default; // for ROOT IO
55 
56  // constructor
57  BasicXYZVHit(T x, T y, T z, E time, V val, int trackid, short did)
58  : mPos(x, y, z), mTime(time), mHitValue(val), BaseHit(trackid), mDetectorID(did)
59  {
60  }
61 
62  // getting the cartesian coordinates
63  T GetX() const { return mPos.X(); }
64  T GetY() const { return mPos.Y(); }
65  T GetZ() const { return mPos.Z(); }
66  Point3D<T> GetPos() const { return mPos; }
67  // getting hit value
68  V GetHitValue() const { return mHitValue; }
69  // getting the time
70  E GetTime() const { return mTime; }
71  // get detector + track information
72  short GetDetectorID() const { return mDetectorID; }
73 
74  // modifiers
75  void SetTime(E time) { mTime = time; }
76  void SetHitValue(V val) { mHitValue = val; }
77  void SetDetectorID(short detID) { mDetectorID = detID; }
78  void SetX(T x) { mPos.SetX(x); }
79  void SetY(T y) { mPos.SetY(y); }
80  void SetZ(T z) { mPos.SetZ(z); }
81  void SetXYZ(T x, T y, T z)
82  {
83  SetX(x);
84  SetY(y);
85  SetZ(z);
86  }
87  void SetPos(Point3D<T> const& p) { mPos = p; }
88 
89  ClassDefNV(BasicXYZVHit, 1);
90 };
