34 WorkflowSpec defineDataProcessing(ConfigContext const& config)
35 {
36  WorkflowSpec workflow;
37  workflow.emplace_back(DataProcessorSpec{
38  "raw-parser",
39  select(config.options().get<std::string>("input-spec").c_str()),
40  Outputs{},
41  AlgorithmSpec{[](InitContext& setup) {
42  auto loglevel = setup.options().get<int>("log-level");
43  return adaptStateless([loglevel](InputRecord& inputs, DataAllocator& outputs) {
44  DPLRawParser parser(inputs);
45  o2::header::DataHeader const* lastDataHeader = nullptr;
46  std::stringstream rdhprintout;
47  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
48  // retrieving RDH v4
49  auto const* rdh = it.get_if<o2::header::RAWDataHeaderV4>();
50  // retrieving the raw pointer of the page
51  auto const* raw = it.raw();
52  // retrieving payload pointer of the page
53  auto const* payload = it.data();
54  // size of payload
55  size_t payloadSize = it.size();
56  // offset of payload in the raw page
57  size_t offset = it.offset();
58 
59  // Note: the following code is only for printing out raw page information
60  const auto* dh = it.o2DataHeader();
61  if (loglevel > 0) {
62  if (dh != lastDataHeader) {
63  if (!rdhprintout.str().empty()) {
64  LOG(INFO) << rdhprintout.str();
65  rdhprintout.str(std::string());
66  }
67  // print the DataHeader information only for the first part or if we have high verbosity
68  if (loglevel > 1 || dh->splitPayloadIndex == 0) {
69  rdhprintout << dh->dataOrigin.as<std::string>() << "/"
70  << dh->dataDescription.as<std::string>() << "/"
71  << dh->subSpecification << " ";
72  // at high verbosity print part number, otherwise only the total number of parts
73  if (loglevel > 1) {
74  rdhprintout << "part " + std::to_string(dh->splitPayloadIndex) + " of " + std::to_string(dh->splitPayloadParts);
75  } else {
76  rdhprintout << " " + std::to_string(dh->splitPayloadParts) + " part(s)";
77  }
78  rdhprintout << " payload size " << dh->payloadSize
79  << std::endl;
80  }
81  rdhprintout << DPLRawParser::RDHInfo(it) << std::endl;
82  }
83  rdhprintout << it << " payload size " << it.size() << std::endl;
84  }
85  lastDataHeader = dh;
86  }
87  if (loglevel > 0) {
88  LOG(INFO) << rdhprintout.str();
89  }
90  }); }},
91  Options{
92  {"log-level", VariantType::Int, 1, {"Logging level [0-2]"}}}});
93  return workflow;
94 }
