83 void DataSpecUtils::describe(char* buffer, size_t size, InputSpec const& spec)
84 {
85  if (auto concrete = std::get_if<ConcreteDataMatcher>(&spec.matcher)) {
86  char origin[5];
87  origin[4] = 0;
88  char description[17];
89  description[16] = 0;
90  snprintf(buffer, size, "%s/%s/%" PRIu32, (strncpy(origin, concrete->origin.str, 4), origin),
91  (strncpy(description, concrete->description.str, 16), description), concrete->subSpec);
92  } else if (auto matcher = std::get_if<DataDescriptorMatcher>(&spec.matcher)) {
93  std::ostringstream ss;
94  ss << "<matcher query: " << *matcher << ">";
95  strncpy(buffer, ss.str().c_str(), size - 1);
96  } else {
97  throw std::runtime_error("Unsupported InputSpec");
98  }

164  using namespace header;
165  if (spec.binding.empty()) {
166  return false;
167  }
168  if (auto concrete = std::get_if<ConcreteDataMatcher>(&spec.matcher)) {
169  return (concrete->description != DataDescription("")) &&
170  (concrete->description != header::gDataDescriptionInvalid) &&

172  (concrete->origin != header::gDataOriginInvalid);
173  }
174  return true;

188 {
189  return std::visit(overloaded{
190  [](ConcreteDataMatcher const& matcher) {
191  // We return false because the matcher is more
192  // qualified (has subSpec) than the target.
193  return false;
194  },
195  [&target](DataDescriptorMatcher const& matcher) {
196  // FIXME: to do it properly we should actually make sure that the
197  // matcher is invariant for changes of SubSpecification. Maybe it's
198  // enough to check that none of the nodes actually match on SubSpec.
199  ConcreteDataMatcher concreteExample{
200  target.origin,
201  target.description,
202  static_cast<header::DataHeader::SubSpecificationType>(0xffffffff)};
203  data_matcher::VariableContext context;
204  return matcher.match(concreteExample, context);
205  }},
206  spec.matcher);
207 }

234 bool DataSpecUtils::match(OutputSpec const& left, OutputSpec const& right)
235 {
236  if (auto leftConcrete = std::get_if<ConcreteDataMatcher>(&left.matcher)) {
237  return match(right, *leftConcrete);
238  } else if (auto rightConcrete = std::get_if<ConcreteDataMatcher>(&right.matcher)) {
239  return match(left, *rightConcrete);
240  } else {
241  // both sides are ConcreteDataTypeMatcher without subspecification, we simply specify 0
242  // this is ignored in the mathing since also left hand object is ConcreteDataTypeMatcher
243  ConcreteDataTypeMatcher dataType = DataSpecUtils::asConcreteDataTypeMatcher(right);
244  return match(left, dataType.origin, dataType.description, 0);
245  }
246 }

285 struct MatcherInfo {
286  header::DataOrigin origin = header::gDataOriginInvalid; // Whether or not we found an origins (should be a bad query!)
287  header::DataDescription description = header::gDataDescriptionInvalid; // Whether or not we found a description
288  header::DataHeader::SubSpecificationType subSpec = 0; // Whether or not we found a description
289  bool hasOrigin = false;
290  bool hasDescription = false;
291  bool hasSubSpec = false;
292  bool hasUniqueOrigin = false; // Whether the matcher involves a unique origin
293  bool hasUniqueDescription = false; // Whether the matcher involves a unique origin
294  bool hasUniqueSubSpec = false;
295  bool hasError = false;
296 };

298 MatcherInfo extractMatcherInfo(DataDescriptorMatcher const& top)
299 {
300  using namespace data_matcher;
301  using ops = DataDescriptorMatcher::Op;
302 
303  MatcherInfo state;
304  auto nodeWalker = overloaded{
305  [&state](EdgeActions::EnterNode action) {
306  if (state.hasError) {
307  return VisitNone;
308  }
309  // For now we do not support extracting a type from things
310  // which have an OR, so we reset all the uniqueness attributes.
311  if (action.node->getOp() == ops::Or || action.node->getOp() == ops::Xor) {
312  state.hasError = true;
313  return VisitNone;
314  }
315  if (action.node->getOp() == ops::Just) {
316  return VisitLeft;
317  }
318  return VisitBoth;
319  },
320  [](auto) { return VisitNone; }};
321 
322  auto leafWalker = overloaded{
323  [&state](OriginValueMatcher const& valueMatcher) {
324  // FIXME: If we found already more than one data origin, it means
325  // we are ANDing two incompatible origins. Until we support OR,
326  // this is an error.
327  // In case we find a ContextRef, it means we have
328  // a wildcard, so there is not a unique origin.
329  if (state.hasOrigin) {
330  state.hasError = false;
331  return;
332  }
333  state.hasOrigin = true;
334 
335  valueMatcher.visit(overloaded{
336  [&state](std::string const& s) {
337  strncpy(state.origin.str, s.data(), 4);
338  state.hasUniqueOrigin = true;
339  },
340  [&state](auto) { state.hasUniqueOrigin = false; }});
341  },
342  [&state](DescriptionValueMatcher const& valueMatcher) {
343  if (state.hasDescription) {
344  state.hasError = true;
345  return;
346  }
347  state.hasDescription = true;
348  valueMatcher.visit(overloaded{
349  [&state](std::string const& s) {
350  strncpy(state.description.str, s.data(), 16);
351  state.hasUniqueDescription = true;
352  },
353  [&state](auto) { state.hasUniqueDescription = false; }});
354  },
355  [&state](SubSpecificationTypeValueMatcher const& valueMatcher) {
356  if (state.hasSubSpec) {
357  state.hasError = true;
358  return;
359  }
360  state.hasSubSpec = true;
361  valueMatcher.visit(overloaded{
362  [&state](uint32_t const& data) {
363  state.subSpec = data;
364  state.hasUniqueSubSpec = true;
365  },
366  [&state](auto) { state.hasUniqueSubSpec = false; }});
367  },
368  [](auto t) {}};
369  DataMatcherWalker::walk(top, nodeWalker, leafWalker);
370  return state;
371 }

405 OutputSpec DataSpecUtils::asOutputSpec(InputSpec const& spec)
406 {
407  return std::visit(overloaded{
408  [&spec](ConcreteDataMatcher const& concrete) {
409  return OutputSpec{{spec.binding}, concrete, spec.lifetime};
410  },
411  [&spec](DataDescriptorMatcher const& matcher) {
412  auto state = extractMatcherInfo(matcher);
413  if (state.hasUniqueOrigin && state.hasUniqueDescription && state.hasUniqueSubSpec) {
414  return OutputSpec{{spec.binding}, ConcreteDataMatcher{state.origin, state.description, state.subSpec}, spec.lifetime};
415  } else if (state.hasUniqueOrigin && state.hasUniqueDescription) {
416  return OutputSpec{{spec.binding}, ConcreteDataTypeMatcher{state.origin, state.description}, spec.lifetime};
417  }
418 
419  throw std::runtime_error("Could not extract neither ConcreteDataMatcher nor ConcreteDataTypeMatcher from query" + describe(spec));
420  }},
421  spec.matcher);
422 }
