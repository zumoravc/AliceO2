99 void optionsTable(const char* label, std::vector<ConfigParamSpec> const& options, const DeviceControl& control)
100 {
101  if (options.empty()) {
102  return;
103  }
104  if (ImGui::CollapsingHeader(label, ImGuiTreeNodeFlags_DefaultOpen)) {
105  ImGui::Columns(2);
106  auto labels = {"Name", "Value"};
107  for (auto& label : labels) {
108  ImGui::TextUnformatted(label);
109  ImGui::NextColumn();
110  }
111  for (auto& option : options) {
112  ImGui::TextUnformatted(option.name.c_str());
113  ImGui::NextColumn();
114  auto currentValueIt = control.options.find(option.name);
115 
116  // Did not find the option
117  if (currentValueIt == control.options.end()) {
118  switch (option.type) {
119  case VariantType::String:
120  ImGui::Text(R"("%s" (default))", option.defaultValue.get<const char*>());
121  break;
122  case VariantType::Int:
123  ImGui::Text("%d (default)", option.defaultValue.get<int>());
124  break;
125  case VariantType::Float:
126  ImGui::Text("%f (default)", option.defaultValue.get<float>());
127  break;
128  case VariantType::Double:
129  ImGui::Text("%f (default)", option.defaultValue.get<double>());
130  break;
131  case VariantType::Empty:
132  ImGui::TextUnformatted(""); // no default value
133  default:
134  ImGui::TextUnformatted("unknown");
135  }
136  } else {
137  ImGui::TextUnformatted(currentValueIt->second.c_str());
138  }
139  ImGui::NextColumn();
140  }
141  }
142 
143  ImGui::Columns(1);
144 }

146 void displayDeviceInspector(DeviceSpec const& spec,
147  DeviceInfo const& info,
148  DeviceMetricsInfo const& metrics,
149  DataProcessorInfo const& metadata,
150  DeviceControl& control)
151 {
152  ImGui::Text("Name: %s", spec.name.c_str());
153  ImGui::Text("Executable: %s", metadata.executable.c_str());
154  if (info.active) {
155  ImGui::Text("Pid: %d", info.pid);
156  } else {
157  ImGui::Text("Pid: %d (exit status: %d)", info.pid, info.exitStatus);
158  }
159  ImGui::Text("Rank: %zu/%zu%%%zu/%zu", spec.rank, spec.nSlots, spec.inputTimesliceId, spec.maxInputTimeslices);
160 
161  if (ImGui::Button("Attach debugger")) {
162  std::string pid = std::to_string(info.pid);
163  setenv("O2DEBUGGEDPID", pid.c_str(), 1);
164 #ifdef __APPLE__
165  std::string defaultAppleDebugCommand =
166  "osascript -e 'tell application \"Terminal\" to activate'"
167  " -e 'tell application \"Terminal\" to do script \"lldb -p " +
168  pid + "\"'";
169  setenv("O2DPLDEBUG", defaultAppleDebugCommand.c_str(), 0);
170 #else
171  setenv("O2DPLDEBUG", "xterm -hold -e gdb attach $O2DEBUGGEDPID &", 0);
172 #endif
173  int retVal = system(getenv("O2DPLDEBUG"));
174  (void)retVal;
175  }
176 
177  ImGui::SameLine();
178  if (ImGui::Button("Profile 30s")) {
179  std::string pid = std::to_string(info.pid);
180  setenv("O2PROFILEDPID", pid.c_str(), 1);
181 #ifdef __APPLE__
182  std::string defaultAppleProfileCommand =
183  "osascript -e 'tell application \"Terminal\" to activate'"
184  " -e 'tell application \"Terminal\" to do script \"instruments -D dpl-profile-" +
185  pid +
186  ".trace -l 30000 -t Time\\\\ Profiler -p " +
187  pid + " && open dpl-profile-" + pid + ".trace && exit\"'";
188  setenv("O2DPLPROFILE", defaultAppleProfileCommand.c_str(), 0);
189 #else
190  setenv("O2DPLPROFILE", "xterm -hold -e perf record -a -g -p $O2PROFILEDPID > perf-$O2PROFILEDPID.data &", 0);
191 #endif
192  LOG(ERROR) << getenv("O2DPLPROFILE");
193  int retVal = system(getenv("O2DPLPROFILE"));
194  (void)retVal;
195  }
196 
197  deviceInfoTable(info, metrics);
198  for (auto& option : info.currentConfig) {
199  ImGui::Text("%s: %s", option.first.c_str(), option.second.data().c_str());
200  }
201  configurationTable(info.currentConfig, info.currentProvenance);
202  optionsTable("Workflow Options", metadata.workflowOptions, control);
203  if (ImGui::CollapsingHeader("Command line arguments", ImGuiTreeNodeFlags_DefaultOpen)) {
204  for (auto& arg : metadata.cmdLineArgs) {
205  ImGui::Text("%s", arg.c_str());
206  }
207  }
208 
209  if (ImGui::CollapsingHeader("Channels", ImGuiTreeNodeFlags_DefaultOpen)) {
210  ImGui::Text("# channels: %lu", spec.inputChannels.size() + spec.outputChannels.size());
211  ChannelsTableHelper::channelsTable("Inputs:", spec.inputChannels);
212  ChannelsTableHelper::channelsTable("Outputs:", spec.outputChannels);
213  }
214  if (ImGui::CollapsingHeader("Data relayer")) {
215  ImGui::Text("Completion policy: %s", spec.completionPolicy.name.c_str());
216  }
217  if (ImGui::CollapsingHeader("Signals", ImGuiTreeNodeFlags_DefaultOpen)) {
218  if (ImGui::Button("SIGSTOP")) {
219  kill(info.pid, SIGSTOP);
220  }
221  ImGui::SameLine();
222  if (ImGui::Button("SIGTERM")) {
223  kill(info.pid, SIGTERM);
224  }
225  ImGui::SameLine();
226  if (ImGui::Button("SIGKILL")) {
227  kill(info.pid, SIGKILL);
228  }
229  if (ImGui::Button("SIGCONT")) {
230  kill(info.pid, SIGCONT);
231  }
232  ImGui::SameLine();

234  kill(info.pid, SIGUSR1);
235  }
236  ImGui::SameLine();

238  kill(info.pid, SIGUSR2);
239  }
240  }
