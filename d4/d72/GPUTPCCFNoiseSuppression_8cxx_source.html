38 GPUdii() void GPUTPCCFNoiseSuppression::noiseSuppressionImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUSharedMemory& smem,
39  const Array2D<PackedCharge>& chargeMap,
40  const Array2D<uchar>& peakMap,
41  const ChargePos* peakPositions,
42  const uint peaknum,
43  uchar* isPeakPredicate)
44 {
45  SizeT idx = get_global_id(0);
46 
47  ChargePos pos = peakPositions[CAMath::Min(idx, (SizeT)(peaknum - 1))];
48  Charge charge = chargeMap[pos].unpack();
49 
50  ulong minimas, bigger, peaksAround;
51  findMinimaAndPeaksScratchpad(
52  chargeMap,
53  peakMap,
54  charge,
55  pos,
56  smem.posBcast,
57  smem.buf,
58  &minimas,
59  &bigger,
60  &peaksAround);
61 
62  peaksAround &= bigger;
63 
64  bool keepMe = keepPeak(minimas, peaksAround);
65 
66  bool iamDummy = (idx >= peaknum);
67  if (iamDummy) {
68  return;
69  }
70 
71  isPeakPredicate[idx] = keepMe;
72 }

74 GPUd() void GPUTPCCFNoiseSuppression::updatePeaksImpl(int nBlocks, int nThreads, int iBlock, int iThread,
75  const ChargePos* peakPositions,
76  const uchar* isPeak,
77  const uint peakNum,
78  Array2D<uchar>& peakMap)
79 {
80  SizeT idx = get_global_id(0);
81 
82  if (idx >= peakNum) {
83  return;
84  }
85 
86  ChargePos pos = peakPositions[idx];
87 
88  uchar peak = isPeak[idx];
89 
90  peakMap[pos] = 0b10 | peak; // if this positions was marked as peak at some point, then its charge must exceed the charge threshold.
91  // So we can just set the bit and avoid rereading the charge
92 }

94 GPUd() void GPUTPCCFNoiseSuppression::checkForMinima(
95  float q,
96  float epsilon,
97  PackedCharge other,
98  int pos,
99  ulong* minimas,
100  ulong* bigger)
101 {
102  float r = other.unpack();
103 
104  ulong isMinima = (q - r > epsilon);
105  *minimas |= (isMinima << pos);
106 
107  ulong lq = (r > q);
108  *bigger |= (lq << pos);
109 }

142 GPUd() bool GPUTPCCFNoiseSuppression::keepPeak(
143  ulong minima,
144  ulong peaks)
145 {
146  bool keepMe = true;
147 
148  for (int i = 0; i < NOISE_SUPPRESSION_NEIGHBOR_NUM; i++) {
149  bool otherPeak = (peaks & (ulong(1) << i));
150  bool minimaBetween = (minima & CfConsts::NoiseSuppressionMinima[i]);
151 
152  keepMe &= (!otherPeak || minimaBetween);
153  }
154 
155  return keepMe;
156 }
