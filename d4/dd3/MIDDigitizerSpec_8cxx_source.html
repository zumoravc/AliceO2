53  void run(framework::ProcessingContext& pc)
54  {
55  static bool finished = false;
56  if (finished) {
57  return;
58  }
59  LOG(DEBUG) << "Doing MID digitization";
60 
61  // read collision context from input
62  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
63  context->initSimChains(o2::detectors::DetID::MID, mSimChains);
64  auto& irecords = context->getEventRecords();
65 
66  auto& eventParts = context->getEventParts();
67  std::vector<o2::mid::ColumnDataMC> digits, digitsAccum;
68  std::vector<o2::mid::ROFRecord> rofRecords;
69  o2::dataformats::MCTruthContainer<o2::mid::MCLabel> labels, labelsAccum;
70 
71  // loop over all composite collisions given from context
72  // (aka loop over all the interaction records)
73  for (int collID = 0; collID < irecords.size(); ++collID) {
74  // for each collision, loop over the constituents event and source IDs
75  // (background signal merging is basically taking place here)
76  auto firstEntry = digitsAccum.size();
77  for (auto& part : eventParts[collID]) {
78  mDigitizer->setEventID(part.entryID);
79  mDigitizer->setSrcID(part.sourceID);
80 
81  // get the hits for this event and this source
82  std::vector<o2::mid::Hit> hits;
83  context->retrieveHits(mSimChains, "MIDHit", part.sourceID, part.entryID, &hits);
84  LOG(DEBUG) << "For collision " << collID << " eventID " << part.entryID << " found MID " << hits.size() << " hits ";
85 
86  mDigitizer->process(hits, digits, labels);
87  if (digits.empty()) {
88  continue;
89  }
90  digitsAccum.insert(digitsAccum.end(), digits.begin(), digits.end());
91  labelsAccum.mergeAtBack(labels);
92  }
93  auto nEntries = digitsAccum.size() - firstEntry;
94  if (nEntries > 0) {
95  rofRecords.emplace_back(irecords[collID], EventType::Standard, firstEntry, nEntries);
96  }
97  }
98 
99  mDigitsMerger.process(digitsAccum, labelsAccum, rofRecords);
100 
101  LOG(DEBUG) << "MID: Sending " << digitsAccum.size() << " digits.";
102  pc.outputs().snapshot(Output{"MID", "DIGITS", 0, Lifetime::Timeframe}, mDigitsMerger.getColumnData());
103  pc.outputs().snapshot(Output{"MID", "DIGITSROF", 0, Lifetime::Timeframe}, mDigitsMerger.getROFRecords());
104  if (pc.outputs().isAllowed({"MID", "DIGITLABELS", 0})) {
105  pc.outputs().snapshot(Output{"MID", "DIGITLABELS", 0, Lifetime::Timeframe}, mDigitsMerger.getMCContainer());
106  }
107  LOG(DEBUG) << "MID: Sending ROMode= " << mROMode << " to GRPUpdater";
108  pc.outputs().snapshot(Output{"MID", "ROMode", 0, Lifetime::Timeframe}, mROMode);
109 
110  // we should be only called once; tell DPL that this process is ready to exit
111  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
112  finished = true;
113  }
114 
115  private:
116  std::unique_ptr<Digitizer> mDigitizer;
117  DigitsMerger mDigitsMerger;
118  std::vector<TChain*> mSimChains;
119  // RS: at the moment using hardcoded flag for continuos readout
120  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
121 };
