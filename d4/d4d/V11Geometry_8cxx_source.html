61 void V11Geometry::intersectLines(Double_t m, Double_t x0, Double_t y0, Double_t n, Double_t x1,
62  Double_t y1, Double_t& xi, Double_t& yi) const
63 {
64  if (TMath::Abs(m - n) < 0.000001) {
65  LOG(ERROR) << "Lines are parallel: m = " << m << " n = " << n;
66  return;
67  }
68 
69  xi = (y1 - n * x1 - y0 + m * x0) / (m - n);
70  yi = y0 + m * (xi - x0);
71 
72  return;
73 }

75 Bool_t V11Geometry::intersectCircle(Double_t m, Double_t x0, Double_t y0, Double_t rr, Double_t xc,
76  Double_t yc, Double_t& xi1, Double_t& yi1, Double_t& xi2,
77  Double_t& yi2)
78 {
79  Double_t p = m * x0 - y0;
80  Double_t q = m * m + 1;
81 
82  p = p - m * xc + yc;
83 
84  Double_t delta = m * m * p * p - q * (p * p - rr * rr);
85 
86  if (delta < 0) {
87  return kFALSE;
88  } else {
89  Double_t root = TMath::Sqrt(delta);
90  xi1 = (m * p + root) / q + xc;
91  xi2 = (m * p - root) / q + xc;
92  yi1 = m * (xi1 - x0) + y0;
93  yi2 = m * (xi2 - x0) + y0;
94  return kTRUE;
95  }
96 }

284 void V11Geometry::insidePoint(Double_t x0, Double_t y0, Double_t x1, Double_t y1, Double_t x2,
285  Double_t y2, Double_t c, Double_t& x, Double_t& y) const
286 {
287  Double_t dx01, dx12, dy01, dy12, r01, r12, m;
288 
289  // printf("InsidePoint: x0=% #12.7g y0=% #12.7g x1=% #12.7g y1=% #12.7g "
290  // "x2=% #12.7g y2=% #12.7g c=% #12.7g ",x0,y0,x1,y2,x2,y2,c);
291  dx01 = x0 - x1; // cout <<"L410 dx01="<<dx01<<endl;
292  dx12 = x1 - x2; // cout <<"L411 dx12="<<dx12<<endl;
293  dy01 = y0 - y1; // cout <<"L412 dy01="<<dy01<<endl;
294  dy12 = y1 - y2; // cout <<"L413 dy12="<<dy12<<endl;
295  r01 = TMath::Sqrt(dy01 * dy01 + dx01 * dx01); // cout <<"L414 r01="<<r01<<endl;
296  r12 = TMath::Sqrt(dy12 * dy12 + dx12 * dx12); // cout <<"L415 r12="<<r12<<endl;
297  m = dx12 * dy01 - dy12 * dx01;
298  if (m * m < DBL_EPSILON) { // m == n
299  if (dy01 == 0.0) { // line are =
300  x = x1 + c; // cout <<"L419 x="<<x<<endl;
301  y = y1; // cout <<"L420 y="<<y<<endl;
302  // printf("dy01==0.0 x=% #12.7g y=% #12.7g\n",x,y);
303  return;
304  } else if (dx01 == 0.0) {
305  x = x1;
306  y = y1 + c;
307  // printf("dx01==0.0 x=% #12.7g y=% #12.7g\n",x,y);
308  return;
309  } else { // dx01!=0 and dy01 !=0.
310  x = x1 - 0.5 * c * r01 / dy01; // cout <<"L434 x="<<x<<endl;
311  y = y1 + 0.5 * c * r01 / dx01; // cout <<"L435 y="<<y<<endl;
312  // printf("m*m<DBL_E x=% #12.7g y=% #12.7g\n",x,y);
313  } // end if
314  return;
315  }
316  x = x1 + c * (dx12 * r01 - dx01 * r12) / m; // cout <<"L442 x="<<x<<endl;
317  y = y1 + c * (dy12 * r01 - dy01 * r12) / m; // cout <<"L443 y="<<y<<endl;
318  // printf(" x=% #12.7g y=% #12.7g\n",x,y);
319  // cout <<"=============================================="<<endl;
320  return;
321 }
