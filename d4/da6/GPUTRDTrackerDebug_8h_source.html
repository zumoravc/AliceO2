200  // track parameters
201  void SetTrackParameter(const T& trk, int ly)
202  {
203  fTrackX(ly) = trk.getX();
204  fTrackY(ly) = trk.getY();
205  fTrackZ(ly) = trk.getZ();
206  fTrackPhi(ly) = trk.getSnp();
207  fTrackLambda(ly) = trk.getTgl();
208  fTrackPt(ly) = trk.getPt();
209  fTrackQPt(ly) = trk.getQ2Pt();
210  fTrackSector(ly) = GetSector(trk.getAlpha());
211  fTrackYerr(ly) = trk.getSigmaY2();
212  fTrackZerr(ly) = trk.getSigmaZ2();
213  }
214  void SetTrackParameterNoUp(const T& trk, int ly)
215  {
216  fTrackNoUpX(ly) = trk.getX();
217  fTrackNoUpY(ly) = trk.getY();
218  fTrackNoUpZ(ly) = trk.getZ();
219  fTrackNoUpPhi(ly) = trk.getSnp();
220  fTrackNoUpLambda(ly) = trk.getTgl();
221  fTrackNoUpPt(ly) = trk.getPt();
222  fTrackNoUpSector(ly) = GetSector(trk.getAlpha());
223  fTrackNoUpYerr(ly) = trk.getSigmaY2();
224  fTrackNoUpZerr(ly) = trk.getSigmaZ2();
225  }
226  void SetTrackParameterReal(const T& trk, int ly)
227  {
228  fTrackXReal(ly) = trk.getX();
229  fTrackYReal(ly) = trk.getY();
230  fTrackZReal(ly) = trk.getZ();
231  fTrackSecReal(ly) = GetSector(trk.getAlpha());
232  }
233  void SetTrack(const T& trk)
234  {
235  fChi2 = trk.GetChi2();
236  fNlayers = trk.GetNlayers();
237  fNtrklts = trk.GetNtracklets();
238  fNtrkltsRef = trk.GetNtrackletsOffline(0);
239  fNtrkltsRefMatch = trk.GetNtrackletsOffline(1);
240  fNtrkltsRefRelated = trk.GetNtrackletsOffline(2);
241  fNtrkltsRefFake = trk.GetNtrackletsOffline(3);
242  fTrackIDref = trk.GetLabelOffline();
243  for (int iLy = 0; iLy < 6; iLy++) {
244  if (trk.GetIsFindable(iLy)) {
245  fFindable(iLy) = 1;
246  }
247  }
248  }

323  void Output()
324  {
325  (*fStreamer) << "tracksFinal"
326  << "event=" << fEv << // event number
327  "nTPCtracks=" << fNTPCtracks << // total number of TPC tracks for this event
328  "iTrack=" << fTrk << // track index in event
329  "trackID=" << mTrackId << // TPC MC track label
330  "trackPtTPC=" << fPtTPC << // track pT before any propagation
331  "trackX.=" << &fTrackX << // x-pos of track (layerwise)
332  "trackY.=" << &fTrackY << // y-pos of track (layerwise)
333  "trackZ.=" << &fTrackZ << // z-pos of track (layerwise)
334  "trackPhi.=" << &fTrackPhi << // phi angle of track (track.fP[2])
335  "trackLambda.=" << &fTrackLambda << // lambda angle of track (track.fP[3])
336  "trackQPt.=" << &fTrackQPt << // track q/pT (track.fP[4])
337  "trackPt.=" << &fTrackPt << // track pT (layerwise)
338  "trackYerr.=" << &fTrackYerr << // sigma_y^2 for track
339  "trackZerr.=" << &fTrackZerr << // sigma_z^2 for track
340  "trackSec.=" << &fTrackSector << // TRD sector of track
341  "trackNoUpX.=" << &fTrackNoUpX << // x-pos of track w/o updates (layerwise)
342  "trackNoUpY.=" << &fTrackNoUpY << // y-pos of track w/o updates (layerwise)
343  "trackNoUpZ.=" << &fTrackNoUpZ << // z-pos of track w/o updates (layerwise)
344  "trackNoUpPhi.=" << &fTrackNoUpPhi << // phi angle of track w/o updates (track.fP[2])
345  "trackNoUpLambda.=" << &fTrackNoUpLambda << // lambda angle of track w/o updates (track.fP[3])
346  "trackNoUpPt.=" << &fTrackNoUpPt << // track pT w/o updates (layerwise)
347  "trackNoUpYerr.=" << &fTrackNoUpYerr << // sigma_y^2 for track w/o updates
348  "trackNoUpZerr.=" << &fTrackNoUpZerr << // sigma_z^2 for track w/o updates
349  "trackNoUpSec.=" << &fTrackNoUpSector << // TRD sector of track w/o updates
350  "trackletX.=" << &fTrackletX << // x position of tracklet used for update (sector coords)
351  "trackletY.=" << &fTrackletYcorr << // y position of tracklet used for update (sector coords, tilt corrected position)
352  "trackletZ.=" << &fTrackletZcorr << // z position of tracklet used for update (sector coords, tilt corrected position)
353  "trackletYRaw.=" << &fTrackletY << // y position of tracklet used for update (sector coords)
354  "trackletZRaw.=" << &fTrackletZ << // z position of tracklet used for update (sector coords)
355  "trackletYerr.=" << &fTrackletY2err << // sigma_y^2 for tracklet
356  "trackletYZerr.=" << &fTrackletYZerr << // sigma_yz for tracklet
357  "trackletZerr.=" << &fTrackletZ2err << // sigma_z^2 for tracklet
358  "trackletDy.=" << &fTrackletDy << // deflection for tracklet
359  "trackletDet.=" << &fTrackletDet << // TRD chamber of tracklet
360  "trackXReal.=" << &fTrackXReal << // x-pos for track at first found tracklet radius w/ matching MC label
361  "trackYReal.=" << &fTrackYReal << // y-pos for track at first found tracklet radius w/ matching MC label
362  "trackZReal.=" << &fTrackZReal << // z-pos for track at first found tracklet radius w/ matching MC label
363  "trackSecReal.=" << &fTrackSecReal << // TRD sector for track at first found tracklet w/ matching MC label
364  "trackletXReal.=" << &fTrackletXReal << // x position (sector coords) for matching or related tracklet if available, otherwise 0
365  "trackletYReal.=" << &fTrackletYcorrReal << // y position (sector coords, tilt correctet position) for matching or related tracklet if available, otherwise 0
366  "trackletZReal.=" << &fTrackletZcorrReal << // z position (sector coords, tilt correctet position) for matching or related tracklet if available, otherwise 0
367  "trackletYRawReal.=" << &fTrackletYReal << // y position (sector coords) for matching or related tracklet if available, otherwise 0
368  "trackletZRawReal.=" << &fTrackletZReal << // z position (sector coords) for matching or related tracklet if available, otherwise 0
369  "trackletSecReal.=" << &fTrackletSecReal << // sector number for matching or related tracklet if available, otherwise -1
370  "trackletDetReal.=" << &fTrackletDetReal << // detector number for matching or related tracklet if available, otherwise -1
371  "chi2Update.=" << &fChi2Update << // chi2 for update
372  "chi2Real.=" << &fChi2Real << // chi2 for first tracklet w/ matching MC label
373  "chi2YZPhiUpdate.=" << &fChi2YZPhiUpdate << // chi2 for update taking into account full tracklet information (y, z, dY aka sin(phi))
374  "chi2YZPhiReal.=" << &fChi2YZPhiReal << // chi2 for first tracklet w/ matching MC label taking into account full tracklet information (y, z, dY aka sin(phi))
375  "chi2Total=" << fChi2 << // total chi2 for track
376  "nLayers=" << fNlayers << // number of layers in which track was findable
377  "nTracklets=" << fNtrklts << // number of attached tracklets
378  "nTrackletsOffline=" << fNtrkltsRef << // number of attached offline tracklets
379  "nTrackletsOfflineMatch=" << fNtrkltsRefMatch << // number of attached offline tracklets
380  "nTrackletsOfflineRelated=" << fNtrkltsRefRelated << // number of attached offline tracklets
381  "nTrackletsOfflineFake=" << fNtrkltsRefFake << // number of attached offline tracklets
382  "labelRef=" << fTrackIDref << // TRD MC track label from offline, if provided
383  "roadY.=" << &fRoadY << // search road width in Y
384  "roadZ.=" << &fRoadZ << // search road width in Z
385  "findable.=" << &fFindable << // whether or not track was in active TRD volume (layerwise)
386  "findableMC.=" << &fFindableMC << // whether or not a MC hit existed inside the TRD for the track (layerwise)
387  "update.=" << &fUpdates << // layerwise tracklet attachment (0 - no tracklet, [1-3] matching tracklet, [4-6] related tracklet, 9 fake tracklet)
388  "nRelated=" << fNrelated << // number of attached related tracklets
389  "nMatching=" << fNmatch << // number of attached matching tracklets
390  "nFake=" << fNfake << // number of attached fake tracklets
391  "nMatchingTracklets.=" << &fNmatchesAvail << // number of matching + related tracklets for this track in each layer
392  "XvMC=" << fXvMC << // MC production vertex x
393  "YvMC=" << fYvMC << // MC production vertex y
394  "ZvMC=" << fZvMC << // MC production vertex z
395  "pdgCode=" << fPdgCode << // MC PID
396  "\n";
397  }
