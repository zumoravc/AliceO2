111 inline RecoGeomHelper::Relation RecoGeomHelper::RecoLadder::isPhiOutside(float phi, float toler) const
112 {
113  // check if phi+-toler is out of the limits of the ladder phi.
114  // return -1 or +1 if phi is above or below the region. If inside, return 0
115  float dif = phi - phiMean, difa = fabs(dif);
116  if (difa < dphiH + toler) {
117  return RecoGeomHelper::Inside;
118  }
119  if (difa > o2::constants::math::PI) { // wraps?
120  difa = o2::constants::math::TwoPI - difa;
121  if (difa < dphiH + toler) {
122  return RecoGeomHelper::Inside;
123  }
124  return dif < 0 ? RecoGeomHelper::Above : RecoGeomHelper::Below;
125  }
126  return dif < 0 ? RecoGeomHelper::Below : RecoGeomHelper::Above;
127 }

130 inline int RecoGeomHelper::RecoLayer::getChipID(float z) const
131 {
132  // Get chip ID within the ladder corresponding to this phi
133  // Note: this is an approximate method, one should check also the neighbouring ladders +/-1
134  int ic = (z - zRange.min()) * z2chipID;
135  return ic < 0 ? 0 : (ic < lastChipInLadder ? ic : lastChipInLadder);
136 }

139 inline int RecoGeomHelper::RecoLayer::getLadderID(float phi) const
140 {
141  // Get ladder ID corresponding to phi.
142  // Note: this is an approximate method, precise within 1/3 of average ladder width,
143  // one should check also the neighbouring ladders +/-1
144  o2::utils::BringTo02Pi(phi);
145  constexpr float PI2Inv = 1.f / o2::constants::math::TwoPI;
146  return phi2ladder[int(phi * PI2Inv * phi2ladder.size())];
147 }
