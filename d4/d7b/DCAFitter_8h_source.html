70  struct TrackCovI {
71  ftype_t sxx, syy, syz, szz;
72 
73  TrackCovI(const Track& trc) { set(trc); }
74  TrackCovI() CONSTRDEF;
75  void set(const Track& trc)
76  {
77  // we assign Y error to X for DCA calculation of 2 points
78  // (otherwise for quazi-collinear tracks the X will not be constrained)
79  ftype_t cyy = trc.getSigmaY2(), czz = trc.getSigmaZ2(), cyz = trc.getSigmaZY(), cxx = cyy;
80  ftype_t detYZ = cyy * czz - cyz * cyz;
81  if (detYZ > 0.) {
82  sxx = 1. / cxx;
83  syy = czz / detYZ;
84  syz = -cyz / detYZ;
85  szz = cyy / detYZ;
86  } else {
87  syy = 0.0; // failure
88  }
89  }
90  };

94  struct TrackDeriv2 {
95  ftype_t dydx, dzdx, d2ydx2, d2zdx2;
96 
97  TrackDeriv2() CONSTRDEF;
98  TrackDeriv2(const Track& trc, ftype_t bz) { set(trc, bz); }
99  void set(const Track& trc, ftype_t bz)
100  {
101  ftype_t snp = trc.getSnp(), csp = TMath::Sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz) * cspI;
102  dydx = snp * cspI; // = snp/csp
103  dzdx = trc.getTgl() * cspI; // = tgl/csp
104  d2ydx2 = crv2c * cspI * cspI; // = crv/csp^3
105  d2zdx2 = crv2c * dzdx * dydx; // = crv*tgl*snp/csp^3
106  }
107  };

195  void notTouchingXY(ftype_t dist, ftype_t xDist, ftype_t yDist, const TrcAuxPar& trcA, ftype_t rBSign)
196  {
197  // fast method to calculate DCA between 2 circles, assuming that they don't touch each outer:
198  // the parametric equation of lines connecting the centers is x = xA + t/dist * xDist, y = yA + t/dist * yDist
199  // with xA,yY being the center of the circle A ( = trcA.xCen, trcA.yCen ), xDist = trcB.xCen = trcA.xCen ...
200  // There are 2 special cases:
201  // (a) small circle is inside the large one: provide rBSign as -trcB.r
202  // (b) circle are side by side: provide rBSign as trcB.r
203  nDCA = 1;
204  auto t2d = (dist + trcA.r - rBSign) / dist;
205  xDCA[0] = trcA.xCen + 0.5 * (xDist * t2d);
206  yDCA[0] = trcA.yCen + 0.5 * (yDist * t2d);
207  }
208  };
