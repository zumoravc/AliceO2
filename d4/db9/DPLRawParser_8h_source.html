215  private:
216  // the iterator over the parts in one channel
217  using part_iterator = typename input_iterator::const_iterator;
218  // the iterator over the over the parser pages

221  bool next()
222  {
223  while (mInputIterator != mEnd) {
224  bool isInitial = mParser == nullptr;
225  while (mPartIterator != mInputIterator.end()) {
226  // first increment on the parser level
227  if (mParser && mCurrent != mParser->end() && ++mCurrent != mParser->end()) {
228  // we have an active parser and there is still data at the incremented iterator
229  return true;
230  }
231  // now increment on the level of one input
232  mParser.reset();
233  if (!isInitial && (mPartIterator == mInputIterator.end() || ++mPartIterator == mInputIterator.end())) {
234  // no more parts, go to next input
235  break;
236  }
237  isInitial = false;
238  // check filter rules
239  if (mFilterSpecs.size() > 0) {
240  bool isSelected = false;
241  for (auto const& spec : mFilterSpecs) {
242  if ((isSelected = DataRefUtils::match(*mPartIterator, spec)) == true) {
243  break;
244  }
245  }
246  if (!isSelected) {
247  continue;
248  }
249  }
250  gsl::span<const char> raw;
251  try {
252  raw = mParent.get<gsl::span<char>>(*mPartIterator);
253  } catch (const std::runtime_error& e) {
254  // TODO: need some better handling to avoid to be spammed by error messages
255  LOG(ERROR) << "failed to read data from " << (*mInputIterator).spec->binding;
256  LOG(ERROR) << e.what();
257  }
258  if (raw.size() == 0) {
259  continue;
260  }
261 
262  try {
263  mParser = std::make_unique<parser_type>(raw.data(), raw.size());
264  } catch (const std::runtime_error& e) {
265  LOG(ERROR) << "can not create raw parser form input data";
266  LOG(ERROR) << e.what();
267  }
268 
269  if (mParser != nullptr) {
270  mCurrent = mParser->begin();
271  return true;
272  }
273  } // end loop over parts on one input
274  ++mInputIterator;
275  mPartIterator = mInputIterator.begin();
276  } // end loop over inputs
277  return false;
278  }
