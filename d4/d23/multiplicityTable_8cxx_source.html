40  void process(aod::Collision const& collision, aod::BCs const& bcs, aod::Zdcs const& zdcs, aod::Run2V0s const& vzeros)
41  {
42  auto zdc = getZdc(collision.bc(), zdcs);
43  auto vzero = getVZero(collision.bc(), vzeros);
44  float multV0A = vzero.multA();
45  float multV0C = vzero.multC();
46  float multZNA = zdc.energyCommonZNA();
47  float multZNC = zdc.energyCommonZNC();
48 
49  LOGF(info, "multV0A=%5.0f multV0C=%5.0f multZNA=%6.0f multZNC=%6.0f", multV0A, multV0C, multZNA, multZNC);
50  // fill multiplicity columns
51  mult(multV0A, multV0C, multZNA, multZNC);
52  }
