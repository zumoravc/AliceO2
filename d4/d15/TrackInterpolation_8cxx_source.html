45 void TrackInterpolation::process()
46 {
47  // main processing function
48  if (!mInitDone) {
49  LOG(error) << "Initialization not yet done. Aborting...";
50  return;
51  }
52 
53 #ifdef TPC_RUN2
54  // processing will not work if the run 2 geometry is defined in the parameter class SpacePointsCalibParam.h
55  LOG(FATAL) << "Run 2 parameters compiled for the TPC geometry. Creating residual trees from Run 3 data will not work. Aborting...";
56  return;
57 #endif
58 
59  std::set<unsigned int> tracksDone; // to store indices of ITS-TPC matched tracks that have been processed
60 
61  LOG(INFO) << "Processing " << mTOFMatchesArray.size() << " ITS-TPC-TOF matched tracks out of "
62  << mITSTPCTracksArray.size() << " ITS-TPC matched tracks";
63 
64  // TODO reserve only a fraction of the needed space for all tracks? How many tracks pass on average the quality cuts with how many TPC clusters?
65  mTrackData.reserve(mTOFMatchesArray.size());
66  mClRes.reserve(mTOFMatchesArray.size() * param::NPadRows);
67 
68  int nTracksTPC = mTPCTracksArray.size();
69 
70  for (const auto& trkTOF : mTOFMatchesArray) {
71  // process ITS-TPC-TOF matched tracks
72  if (!trackPassesQualityCuts(mITSTPCTracksArray[trkTOF.getTrackIndex()])) {
73  LOG(DEBUG) << "Abandoning track due to bad quality";
74  continue;
75  }
76  mTrackData.emplace_back();
77  if (!interpolateTrackITSTOF(trkTOF)) {
78  LOG(DEBUG) << "Failed to interpolate ITS-TOF track";
79  mTrackData.pop_back();
80  continue;
81  }
82  mTrackData.back().nTracksInEvent = nTracksTPC;
83  tracksDone.insert(trkTOF.getTrackIndex());
84  }
85 
86  LOG(INFO) << "Could process " << tracksDone.size() << " ITS-TPC-TOF matched tracks successfully";
87 
88  if (mDoITSOnlyTracks) {
89  size_t nTracksDoneITS = 0;
90  size_t nTracksSkipped = 0;
91  for (std::size_t iTrk = 0; iTrk < mITSTPCTracksArray.size(); ++iTrk) {
92  // process ITS-TPC matched tracks that were not matched to TOF
93  if (tracksDone.find(iTrk) != tracksDone.end()) {
94  // track also has a matching cluster in TOF and has already been processed
95  ++nTracksSkipped;
96  continue;
97  }
98  const auto& trk = mITSTPCTracksArray[iTrk];
99  if (!trackPassesQualityCuts(trk, false)) {
100  continue;
101  }
102  mTrackData.emplace_back();
103  const auto& trkTPC = mTPCTracksArray[trk.getRefTPC()];
104  const auto& trkITS = mITSTracksArray[trk.getRefITS()];
105  if (!extrapolateTrackITS(trkITS, trkTPC, trk.getTimeMUS().getTimeStamp(), trk.getRefTPC())) {
106  mTrackData.pop_back();
107  continue;
108  }
109  mTrackData.back().nTracksInEvent = nTracksTPC;
110  ++nTracksDoneITS;
111  }
112  LOG(INFO) << "Could process " << nTracksDoneITS << " ITS-TPC matched tracks successfully";
113  LOG(INFO) << "Skipped " << nTracksSkipped << " tracks, as they were successfully propagated to TOF";
114  }
115 }

117 bool TrackInterpolation::trackPassesQualityCuts(const o2::dataformats::TrackTPCITS& matchITSTPC, bool hasOuterPoint) const
118 {
119  // apply track quality cuts (assume different settings for track with and without points in TRD or TOF)
120  const auto& trkTPC = mTPCTracksArray[matchITSTPC.getRefTPC()];
121  const auto& trkITS = mITSTracksArray[matchITSTPC.getRefITS()];
122  if (hasOuterPoint) {
123  // track has a match in TRD or TOF
124  if (trkTPC.getNClusterReferences() < param::MinTPCNCls ||
125  trkITS.getNumberOfClusters() < param::MinITSNCls) {
126  printf("TPC clusters (%i), ITS clusters(%i)\n", trkTPC.getNClusterReferences(), trkITS.getNumberOfClusters());
127  return false;
128  }
129  if (trkTPC.getChi2() / trkTPC.getNClusterReferences() > param::MaxTPCChi2 ||
130  trkITS.getChi2() / trkITS.getNumberOfClusters() > param::MaxITSChi2) {
131  printf("TPC reduced chi2 (%.2f), ITS reduced chi2 (%.2f)\n", trkTPC.getChi2() / trkTPC.getNClusterReferences(), trkITS.getChi2() / trkITS.getNumberOfClusters());
132  return false;
133  }
134  } else {
135  // ITS-TPC only track
136  if (trkTPC.getNClusterReferences() < param::MinTPCNClsNoOuterPoint ||
137  trkITS.getNumberOfClusters() < param::MinITSNClsNoOuterPoint) {
138  return false;
139  }
140  if (trkTPC.getChi2() / trkTPC.getNClusterReferences() > param::MaxTPCChi2 ||
141  trkITS.getChi2() / trkITS.getNumberOfClusters() > param::MaxITSChi2) {
142  return false;
143  }
144  }
145  return true;
146 }

148 bool TrackInterpolation::interpolateTrackITSTOF(const o2::dataformats::MatchInfoTOF& matchTOF)
149 {
150  // get TPC cluster residuals to ITS-TOF only tracks
151  size_t trkIdx = mTrackData.size() - 1;
152  auto propagator = o2::base::Propagator::Instance();
153  const auto& matchITSTPC = mITSTPCTracksArray[matchTOF.getTrackIndex()];
154  const auto& clTOF = mTOFClustersArray[matchTOF.getTOFClIndex()];
155  //const int clTOFSec = (TMath::ATan2(-clTOF.getY(), -clTOF.getX()) + o2::constants::math::PI) * o2::constants::math::Rad2Deg * 0.05; // taken from TOF cluster class as there is no const getter for the sector
156  const int clTOFSec = clTOF.getCount();
157  const float clTOFAlpha = o2::utils::Sector2Angle(clTOFSec);
158  const auto& trkTPC = mTPCTracksArray[matchITSTPC.getRefTPC()];
159  const auto& trkITS = mITSTracksArray[matchITSTPC.getRefITS()];
160  auto trkWork = trkITS.getParamOut();
161  // reset the cache array (sufficient to set )
162  for (auto& elem : mCache) {
163  elem.clAvailable = 0;
164  }
165  mTrackData[trkIdx].clIdx.setFirstEntry(mClRes.size()); // reference the first cluster residual belonging to this track
166  //printf("=== New Track with pt = %.3f, nClsTPC = %i ===\n", trkWork.getQ2Pt(), trkTPC.getNClusterReferences());
167 
168  // store the TPC cluster positions in the cache
169  float clusterTimeBinOffset = matchITSTPC.getTimeMUS().getTimeStamp() / mTPCTimeBinMUS;
170  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
171  uint8_t sector, row;
172  uint32_t clusterIndexInRow;
173  const auto& clTPC = trkTPC.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
174  float clTPCX;
175  std::array<float, 2> clTPCYZ;
176  mFastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), clTPCX, clTPCYZ[0], clTPCYZ[1], clusterTimeBinOffset);
177  sector %= SECTORSPERSIDE;
178  mCache[row].clAvailable = 1;
179  mCache[row].clY = clTPCYZ[0];
180  mCache[row].clZ = clTPCYZ[1];
181  mCache[row].clAngle = o2::utils::Sector2Angle(sector);
182  }
183 
184  // first extrapolate through TPC and store track position at each pad row
185  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
186  if (!mCache[iRow].clAvailable) {
187  continue;
188  }
189  if (!trkWork.rotate(mCache[iRow].clAngle)) {
190  LOG(DEBUG) << "Failed to rotate track during first extrapolation";
191  return false;
192  }
193  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
194  LOG(DEBUG) << "Failed on first extrapolation";
195  return false;
196  }
197  mCache[iRow].y[ExtOut] = trkWork.getY();
198  mCache[iRow].z[ExtOut] = trkWork.getZ();
199  mCache[iRow].sy2[ExtOut] = trkWork.getSigmaY2();
200  mCache[iRow].szy[ExtOut] = trkWork.getSigmaZY();
201  mCache[iRow].sz2[ExtOut] = trkWork.getSigmaZ2();
202  mCache[iRow].phi[ExtOut] = trkWork.getSnp();
203  mCache[iRow].tgl[ExtOut] = trkWork.getTgl();
204  //printf("Track alpha at row %i: %.2f, Y(%.2f), Z(%.2f)\n", iRow, trkWork.getAlpha(), trkWork.getY(), trkWork.getZ());
205  }
206 
207  // now continue to TOF and update track with TOF cluster
208  if (!trkWork.rotate(clTOFAlpha)) {
209  LOG(DEBUG) << "Failed to rotate into TOF cluster sector frame";
210  return false;
211  }
212  //float ca, sa;
213  //o2::utils::sincosf(clTOFAlpha, sa, ca);
214  //float clTOFX = clTOF.getX() * ca + clTOF.getY() * sa; // cluster x in sector coordinate frame
215  //std::array<float, 2> clTOFYZ{ -clTOF.getX() * sa + clTOF.getY() * ca, clTOF.getZ() }; // cluster y and z in sector coordinate frame
216  float clTOFX = clTOF.getX();
217  std::array<float, 2> clTOFYZ{clTOF.getY(), clTOF.getZ()};
218  std::array<float, 3> clTOFCov{mSigYZ2TOF, 0.f, mSigYZ2TOF}; // assume no correlation between y and z and equal cluster error sigma^2 = (3cm)^2 / 12
219  if (!propagator->PropagateToXBxByBz(trkWork, clTOFX, o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
220  LOG(DEBUG) << "Failed final propagation to TOF radius";
221  return false;
222  }
223  if (!trkWork.update(clTOFYZ, clTOFCov)) {
224  LOG(DEBUG) << "Failed to update extrapolated ITS track with TOF cluster";
225  return false;
226  }
227 
228  // go back through the TPC and store updated track positions
229  for (int iRow = param::NPadRows; iRow--;) {
230  if (!mCache[iRow].clAvailable) {
231  continue;
232  }
233  if (!trkWork.rotate(mCache[iRow].clAngle)) {
234  LOG(DEBUG) << "Failed to rotate track during back propagation";
235  return false;
236  }
237  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
238  LOG(DEBUG) << "Failed on back propagation";
239  //printf("trkX(%.2f), clX(%.2f), clY(%.2f), clZ(%.2f), alphaTOF(%.2f)\n", trkWork.getX(), param::RowX[iRow], clTOFYZ[0], clTOFYZ[1], clTOFAlpha);
240  return false;
241  }
242  mCache[iRow].y[ExtIn] = trkWork.getY();
243  mCache[iRow].z[ExtIn] = trkWork.getZ();
244  mCache[iRow].sy2[ExtIn] = trkWork.getSigmaY2();
245  mCache[iRow].szy[ExtIn] = trkWork.getSigmaZY();
246  mCache[iRow].sz2[ExtIn] = trkWork.getSigmaZ2();
247  mCache[iRow].phi[ExtIn] = trkWork.getSnp();
248  mCache[iRow].tgl[ExtIn] = trkWork.getTgl();
249  }
250 
251  // calculate weighted mean at each pad row (assume for now y and z are uncorrelated) and store residuals to TPC clusters
252  unsigned short deltaRow = 0;
253  unsigned short nMeasurements = 0;
254  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
255  if (!mCache[iRow].clAvailable) {
256  ++deltaRow;
257  continue;
258  }
259  float wTotY = 1.f / mCache[iRow].sy2[ExtOut] + 1.f / mCache[iRow].sy2[ExtIn];
260  float wTotZ = 1.f / mCache[iRow].sz2[ExtOut] + 1.f / mCache[iRow].sz2[ExtIn];
261  mCache[iRow].y[Int] = (mCache[iRow].y[ExtOut] / mCache[iRow].sy2[ExtOut] + mCache[iRow].y[ExtIn] / mCache[iRow].sy2[ExtIn]) / wTotY;
262  mCache[iRow].z[Int] = (mCache[iRow].z[ExtOut] / mCache[iRow].sz2[ExtOut] + mCache[iRow].z[ExtIn] / mCache[iRow].sz2[ExtIn]) / wTotZ;
263 
264  // simple average w/o weighting for angles
265  mCache[iRow].phi[Int] = (mCache[iRow].phi[ExtOut] + mCache[iRow].phi[ExtIn]) / 2.f;
266  mCache[iRow].tgl[Int] = (mCache[iRow].tgl[ExtOut] + mCache[iRow].tgl[ExtIn]) / 2.f;
267 
268  TPCClusterResiduals res;
269  res.setDY(mCache[iRow].clY - mCache[iRow].y[Int]);
270  res.setDZ(mCache[iRow].clZ - mCache[iRow].z[Int]);
271  res.setY(mCache[iRow].y[Int]);
272  res.setZ(mCache[iRow].z[Int]);
273  res.setPhi(mCache[iRow].phi[Int]);
274  res.setTgl(mCache[iRow].tgl[Int]);
275  res.sec = o2::utils::Angle2Sector(mCache[iRow].clAngle);
276  res.dRow = deltaRow;
277  res.row = iRow;
278  mClRes.push_back(std::move(res));
279  ++nMeasurements;
280  deltaRow = 1;
281  }
282 
283  mTrackData[trkIdx].trkId = matchITSTPC.getRefTPC();
284  mTrackData[trkIdx].eta = trkTPC.getEta();
285  mTrackData[trkIdx].phi = trkTPC.getSnp();
286  mTrackData[trkIdx].qPt = trkTPC.getQ2Pt();
287  mTrackData[trkIdx].chi2TPC = trkTPC.getChi2();
288  mTrackData[trkIdx].chi2ITS = trkITS.getChi2();
289  mTrackData[trkIdx].nClsTPC = trkTPC.getNClusterReferences();
290  mTrackData[trkIdx].nClsITS = trkITS.getNumberOfClusters();
291  mTrackData[trkIdx].clIdx.setEntries(nMeasurements);
292 
293  LOG(DEBUG) << "Track interpolation successfull";
294  return true;
295 }

297 bool TrackInterpolation::extrapolateTrackITS(const o2::its::TrackITS& trkITS, const TrackTPC& trkTPC, float trkTime, int trkIdTPC)
298 {
299  // extrapolate ITS-only track through TPC and store residuals to TPC clusters in the output vectors
300  size_t trkIdx = mTrackData.size() - 1;
301  mTrackData[trkIdx].clIdx.setFirstEntry(mClRes.size());
302  auto trk = trkITS.getParamOut();
303  float clusterTimeBinOffset = trkTime / mTPCTimeBinMUS;
304  auto propagator = o2::base::Propagator::Instance();
305  unsigned short rowPrev = 0;
306  unsigned short nMeasurements = 0;
307  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
308  uint8_t sector, row;
309  uint32_t clusterIndexInRow;
310  const auto& cl = trkTPC.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
311  float x = 0, y = 0, z = 0;
312  mFastTransform->TransformIdeal(sector, row, cl.getPad(), cl.getTime(), x, y, z, clusterTimeBinOffset);
313  if (!trk.rotate(o2::utils::Sector2Angle(sector))) {
314  return false;
315  }
316  if (!propagator->PropagateToXBxByBz(trk, x, o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
317  return false;
318  }
319  TPCClusterResiduals res;
320  res.setDY(y - trk.getY());
321  res.setDY(z - trk.getZ());
322  res.setY(trk.getY());
323  res.setZ(trk.getZ());
324  res.setPhi(trk.getSnp());
325  res.setTgl(trk.getTgl());
326  res.sec = o2::utils::Angle2Sector(trk.getAlpha());
327  res.dRow = row - rowPrev;
328  res.row = row;
329  rowPrev = row;
330  mClRes.push_back(std::move(res));
331  ++nMeasurements;
332  }
333  mTrackData[trkIdx].trkId = trkIdTPC;
334  mTrackData[trkIdx].eta = trkTPC.getEta();
335  mTrackData[trkIdx].phi = trkTPC.getSnp();
336  mTrackData[trkIdx].qPt = trkTPC.getQ2Pt();
337  mTrackData[trkIdx].chi2TPC = trkTPC.getChi2();
338  mTrackData[trkIdx].chi2ITS = trkITS.getChi2();
339  mTrackData[trkIdx].nClsTPC = trkTPC.getNClusterReferences();
340  mTrackData[trkIdx].nClsITS = trkITS.getNumberOfClusters();
341  mTrackData[trkIdx].clIdx.setEntries(nMeasurements);
342 
343  return true;
344 }
