36 Point3D<double> HitFinder::getIntersectInDefaultPlane(const Track& track, int chamber) const
37 {
39  double defaultZ = geoparams::DefaultChamberZ[chamber];
40  double linePar = ((track.getPositionZ() - defaultZ) * mTanTheta - track.getPositionY()) /
41  (track.getDirectionY() - track.getDirectionZ() * mTanTheta);
42  Point3D<double> point;
43  point.SetX(track.getPositionX() + linePar * track.getDirectionX());
44  point.SetY(track.getPositionY() + linePar * track.getDirectionY());
45  point.SetZ(track.getPositionZ() + linePar * track.getDirectionZ());
46  return point;
47 }

84 std::vector<int> HitFinder::getFiredDE(const Track& track, int chamber) const
85 {
90  std::vector<int> deIdList;
91  Point3D<double> defPos = getIntersectInDefaultPlane(track, chamber);
92  double xPos = defPos.x();
93  double xErr = std::sqrt(track.getCovarianceParameter(Track::CovarianceParamIndex::VarX));
94  double yPos = defPos.y() / mCosTheta;
95  double yErr = std::sqrt(track.getCovarianceParameter(Track::CovarianceParamIndex::VarY));
96  double positions[3] = {yPos, yPos - yErr, yPos + yErr};
97  int centerRpc = -1;
98  for (int ipos = 0; ipos < 3; ++ipos) {
99  int rpc = guessRPC(positions[ipos], chamber);
100  if (rpc < 0 || rpc > 8) {
101  continue;
102  }
103  if (ipos == 0) {
104  centerRpc = rpc;
105  } else if (rpc == centerRpc) {
106  continue;
107  }
108  if (xPos > 0) {
109  deIdList.push_back(detparams::getDEId(true, chamber, rpc));
110  if (xPos - xErr < 0) {
111  deIdList.push_back(detparams::getDEId(false, chamber, rpc));
112  }
113  } else {
114  deIdList.push_back(detparams::getDEId(false, chamber, rpc));
115  if (xPos + xErr > 0) {
116  deIdList.push_back(detparams::getDEId(true, chamber, rpc));
117  }
118  }
119  }
120 
121  return deIdList;
122 }

125 std::vector<Cluster2D> HitFinder::getLocalPositions(const Track& track, int chamber, bool withUncertainties) const
126 {
131  std::vector<int> deIdList = getFiredDE(track, chamber);
132 
133  std::vector<Cluster2D> points;
134  for (auto& deId : deIdList) {
135  Cluster2D cl = getIntersect(track, deId);
136  int rpc = deId % 9;
137  double hl = geoparams::getRPCHalfLength(chamber, rpc);
138  double hh = geoparams::getRPCHalfHeight(chamber);
139  if (cl.xCoor < -hl || cl.xCoor > hl) {
140  continue;
141  }
142  if (cl.yCoor < -hh || cl.yCoor > hh) {
143  continue;
144  }
145  if (withUncertainties) {
146  addUncertainty(cl, track);
147  }
148  points.emplace_back(cl);
149  }
150 
151  return points;
152 }

156 {

159  track.propagateToZ(globalPos.z());
