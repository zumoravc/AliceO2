100 TGeoVolume* createSegment(int i)
101 {
102 
106 
107  const char* segmentName = Form("Segment %d", i);
108  auto segment = new TGeoVolumeAssembly(segmentName);
109 
110  const int kNLayers = 6;
111 
112  const std::string kLayerName[kNLayers] = {"gas", "cathode", "insulator", "rohacell", "MEB", "EERB"};
113  const std::array<TGeoMedium*, kNLayers> kLayerMedium = {assertMedium(Medium::Gas), assertMedium(Medium::Copper), assertMedium(Medium::FR4), assertMedium(Medium::Rohacell), assertMedium(Medium::Epoxy), assertMedium(Medium::Copper)};
114  const double kLayerHalfThickness[kNLayers] = {kGasHalfThickness, kCathodeHalfThickness, kInsuHalfThickness, kRohaHalfThickness, kMEBHalfThickness, kEERBHalfThickness};
115 
116  // volume dimensions
117  double halfLength = kBoxSegHalfLength[i], halfHeight = kBoxSegHalfHeight[i], halfThickness = kLayerHalfThickness[0];
118 
119  double z = 0.; // increment this variable when adding a new layer
120 
121  switch (i) {
122  case 1: // polygon
123 
124  // parameters
125  double par[10];
126  par[0] = kStartAngle; // initial angle
127  par[1] = kStopAngle; // increment in angle starting from initial angle
128  par[2] = kNEdges; // number of edges
129  par[3] = kNPlanes; // number of planes
130  par[4] = -halfThickness; // z-position of the first plane
131  par[5] = kSegmentRadius[0]; // inner radius first plane
132  par[6] = kSegmentRadius[1]; // outer radius first plane
133  par[7] = halfThickness; // z-position of the second plane
134  par[8] = par[5]; // inner radius of second plane
135  par[9] = par[6]; // outer radius of second plane
136 
137  // create and place the layers in the segment
138 
139  segment->AddNode(new TGeoVolume(Form("%s %s", segmentName, kLayerName[0].data()), new TGeoPgon(par), kLayerMedium[0]), 1);
140 
141  z = halfThickness;
142 
143  for (int j = 1; j < kNLayers; j++) {
144 
145  halfThickness = kLayerHalfThickness[j];
146 
147  par[4] = -halfThickness;
148  par[7] = halfThickness;
149  auto layer = new TGeoVolume(Form("Segment %d %s", i, kLayerName[j].data()), new TGeoPgon(par), kLayerMedium[j]);
150 
151  z += halfThickness;
152  segment->AddNode(layer, 1, new TGeoTranslation(0., 0., z));
153  segment->AddNode(layer, 2, new TGeoTranslation(0., 0., -z));
154  z += halfThickness;
155 
156  } // end of the layer loop
157  break;
158 
159  default: //box
160  // create and place the layers in the segment
161 
162  // start with gas
163  halfThickness = kLayerHalfThickness[0];
164 
165  segment->AddNode(gGeoManager->MakeBox(Form("%s %s", segmentName, kLayerName[0].data()), kLayerMedium[0], halfLength, halfHeight, halfThickness), 1);
166 
167  z = halfThickness;
168 
169  for (int j = 1; j < kNLayers; j++) {
170 
171  halfThickness = kLayerHalfThickness[j];
172 
173  auto layer = gGeoManager->MakeBox(Form("%s %s", segmentName, kLayerName[j].data()), kLayerMedium[j], halfLength, halfHeight, halfThickness);
174 
175  z += halfThickness;
176  segment->AddNode(layer, 1, new TGeoTranslation(0., 0., z));
177  segment->AddNode(layer, 2, new TGeoTranslation(0., 0., -z));
178  z += halfThickness;
179 
180  } // end of the layer loop
181 
182  break;
183  } // end of the switch
184 
185  return segment;
186 }

200  0.,
201  0.,

205  kFrameHalfLength[7] - kRibHalfLength[7] // right edge

209  kFrameHalfHeight[0] - kRibHalfHeight[0], // upper edge

211  0.,

216  0.};

318 TGeoVolume* createQuadrant()
319 {
320 
322  auto quadrant = new TGeoVolumeAssembly("Station 2 quadrant");
323 
324  // create and place the segments in the quadrant
325  const double kSegXPos[kNSegments] = {kSegmentRadius[0] + kBoxSegHalfLength[0], 0., -kBoxSegHalfLength[2]};
326  const double kSegYPos[kNSegments] = {-kBoxSegHalfHeight[0], 0., kSegmentRadius[0] + kBoxSegHalfHeight[2]};
327 
328  for (int i = 0; i < kNSegments; i++)
329  quadrant->AddNode(createSegment(i), 0, new TGeoTranslation(kSegXPos[i], kSegYPos[i], 0.));
330 
331  // create and place the frames in the quadrant
332  createFrames();
333 
334  // positions
335  const double kFrameXPos[kNFrames] = {
336  kSegXPos[0], // frame n1 aligned with the segment 0
337  kSegXPos[0] + kBoxSegHalfLength[0] + kFrameHalfLength[2 - 1], // frame n2 at the right of the segment 0
338  0.,
339  kSegXPos[2], // frame n4 aligned with the segment 2
340  kSegXPos[2] - kBoxSegHalfLength[2] - kFrameHalfLength[5 - 1], // frame n5 at the left of the segment 2
341  kSegXPos[2], // frame n6 aligned with the segment 2
342  0.,
343  kSegXPos[0] - kBoxSegHalfLength[0] - kFrameHalfLength[8 - 1], // frame n8 at the left of the segment 0
344  };
345 
346  const double kFrameYPos[kNFrames] = {
347  kSegYPos[0] - kBoxSegHalfHeight[0] - kFrameHalfHeight[1 - 1], // frame n1 below the segment 0
348  kSegYPos[0], // frame n2 aligned with the segment 0
349  0.,
350  kSegYPos[2] + kBoxSegHalfHeight[2] + kFrameHalfHeight[4 - 1], // frame n4 above the segment 2
351  kSegYPos[2], // frame n5 aligned with the segment 2
352  kSegYPos[2] - kBoxSegHalfHeight[2] - kFrameHalfHeight[6 - 1], // frame n6 below the segment 2
353  0.,
354  kSegYPos[0], // frame n8 aligned with the segment 0
355  };
356 
357  for (int i = 1; i <= kNFrames; i++)
358  quadrant->AddNode(gGeoManager->GetVolume(Form("Frame %d", i)), 1,
359  new TGeoTranslation(kFrameXPos[i - 1], kFrameYPos[i - 1], 0.));
360 
361  return quadrant;
362 }

365 void createStation2Geometry(TGeoVolume& topVolume)
366 {
368 
369  // create a quadrant
370  auto quadrant = createQuadrant();
371 
372  const int kNQuad = 4;
373 
374  // rotation matrices to place the quadrants in the half-chambers
375  auto rot0 = new TGeoRotation();
376  auto rot1 = new TGeoRotation("reflXZ", 90., 180., 90., 90., 180., 0.);
377  auto rot2 = new TGeoRotation("reflXY", 90., 180., 90., 270., 0., 0.);
378  auto rot3 = new TGeoRotation("reflYZ", 90., 0., 90., -90., 180., 0.);
379  std::array<TGeoRotation*, kNQuad> rot = {rot0, rot1, rot2, rot3};
380 
381  // build the two chambers
382  double z = kQuadrantZPos;
383  int detElemID = 0;
384  const int kFirstChamberNumber = 3;
385 
386  for (int ich = kFirstChamberNumber; ich < kFirstChamberNumber + 2; ich++) {
387 
388  // create two half-chambers
389  auto in = new TGeoVolumeAssembly(Form("SC0%dI", ich));
390  auto out = new TGeoVolumeAssembly(Form("SC0%dO", ich));
391 
392  // place the quadrants in the half-chambers
393  for (int i = 0; i < kNQuad; i++) {
394  // alternate the z position
395  z *= -1.;
396 
397  // compute the detection element ID
398  detElemID = 100 * ich + i;
399 
400  if (i == 0 || i == 3) {
401  in->AddNode(quadrant, detElemID, new TGeoCombiTrans(0., 0., z, rot[i]));
402  } else
403  out->AddNode(quadrant, detElemID, new TGeoCombiTrans(0., 0., z, rot[i]));
404  }
405 
406  // place the half-chambers in the top volume
407  topVolume.AddNode(in, 2 * (ich - 1), new TGeoTranslation(0., 0., kChamberZPos[ich - kFirstChamberNumber]));
408  topVolume.AddNode(out, 2 * ich - 1, new TGeoTranslation(0., 0., kChamberZPos[ich - kFirstChamberNumber]));
409 
410  } // end of the chamber loop
411 }
