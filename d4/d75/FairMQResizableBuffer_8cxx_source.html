36 arrow::Status FairMQResizableBuffer::Resize(const int64_t newSize, bool shrink_to_fit)
37 {
38  if (newSize < this->capacity_ && shrink_to_fit == true) {
39  auto newMessage = mCreator(newSize);
40  memcpy(newMessage->GetData(), mMessage->GetData(), newSize);
41  mMessage = std::move(newMessage);
42  this->mutable_data_ = reinterpret_cast<uint8_t*>(mMessage->GetData());
43  this->data_ = this->mutable_data_;
44  assert(this->data_);
45  this->capacity_ = static_cast<int64_t>(mMessage->GetSize());
46  assert(newSize == this->capacity_);
47  } else if (newSize > this->capacity_) {
48  auto status = this->Reserve(newSize);
49  if (status.ok() == false) {
50  return status;
51  }
52  }
53  assert(newSize <= this->capacity_);
54 
55  this->size_ = newSize;
56  assert(this->size_ <= mMessage->GetSize());

60 arrow::Status FairMQResizableBuffer::Reserve(const int64_t capacity)
61 {
62  assert(!mMessage || this->size_ <= mMessage->GetSize());
63  assert(!mMessage || this->capacity_ == mMessage->GetSize());
64  if (capacity <= this->capacity_) {
65  return arrow::Status::OK();
66  }
67  auto newMessage = mCreator(capacity);
68  assert(!mMessage || capacity > mMessage->GetSize());
69  if (mMessage) {
70  memcpy(newMessage->GetData(), mMessage->GetData(), mMessage->GetSize());
71  }
72  mMessage = std::move(newMessage);
73  assert(mMessage);
74  this->mutable_data_ = reinterpret_cast<uint8_t*>(mMessage->GetData());
75  this->data_ = this->mutable_data_;
76  this->capacity_ = static_cast<int64_t>(mMessage->GetSize());
