53 ExpirationHandler::Creator LifetimeHelpers::enumDrivenCreation(size_t start, size_t end, size_t step, size_t inputTimeslice, size_t maxInputTimeslices)
54 {
55  auto last = std::make_shared<size_t>(start + inputTimeslice * step);
56  return [start, end, step, last, inputTimeslice, maxInputTimeslices](TimesliceIndex& index) -> TimesliceSlot {
57  for (size_t si = 0; si < index.size(); si++) {
58  if (*last > end) {
59  return TimesliceSlot{TimesliceSlot::INVALID};
60  }
61  auto slot = TimesliceSlot{si};
62  if (index.isValid(slot) == false) {
63  TimesliceId timestamp{*last};
64  *last += step * maxInputTimeslices;
65  index.associate(timestamp, slot);
66  return slot;
67  }
68  }
69  return TimesliceSlot{TimesliceSlot::INVALID};
70  };
71 }

73 ExpirationHandler::Creator LifetimeHelpers::timeDrivenCreation(std::chrono::microseconds period)
74 {
75  auto start = getCurrentTime();
76  auto last = std::make_shared<decltype(start)>(start);
77  // FIXME: should create timeslices when period expires....
78  return [last, period](TimesliceIndex& index) -> TimesliceSlot {
79  // Nothing to do if the time has not expired yet.
80  auto current = getCurrentTime();
81  auto delta = current - *last;
82  if (delta < period.count()) {
83  return TimesliceSlot{TimesliceSlot::INVALID};
84  }
85  // We first check if the current time is not already present
86  // FIXME: this should really be done by query matching? Ok
87  // for now to avoid duplicate entries.
88  for (size_t i = 0; i < index.size(); ++i) {
89  TimesliceSlot slot{i};
90  if (index.isValid(slot) == false) {
91  continue;
92  }
93  if (index.getTimesliceForSlot(slot).value == current) {
94  return TimesliceSlot{TimesliceSlot::INVALID};
95  }
96  }
97  // If we are here the timer has expired and a new slice needs
98  // to be created.
99  *last = current;
100  data_matcher::VariableContext newContext;
101  newContext.put({0, static_cast<uint64_t>(current)});
102  newContext.commit();
103  auto [action, slot] = index.replaceLRUWith(newContext);
104  switch (action) {
105  case TimesliceIndex::ActionTaken::ReplaceObsolete:
106  case TimesliceIndex::ActionTaken::ReplaceUnused:
107  index.associate(TimesliceId{current}, slot);
108  break;
109  case TimesliceIndex::ActionTaken::DropInvalid:
110  case TimesliceIndex::ActionTaken::DropObsolete:
111  break;
112  }
113  return slot;
114  };
115 }

127 ExpirationHandler::Checker LifetimeHelpers::expireTimed(std::chrono::microseconds period)
128 {
129  auto start = getCurrentTime();
130  auto last = std::make_shared<decltype(start)>(start);
131  return [last, period](int64_t) -> bool {
132  auto current = getCurrentTime();
133  auto delta = current - *last;
134  if (delta > period.count()) {
135  *last = current;
136  return true;
137  }
138  return false;
139  };

150 // We simply put everything in a stringstream and read it afterwards.
151 size_t readToMessage(void* p, size_t size, size_t nmemb, void* userdata)
152 {
153  if (nmemb == 0) {
154  return 0;
155  }
156  if (size == 0) {
157  return 0;
158  }
159  o2::vector<char>* buffer = (o2::vector<char>*)userdata;
160  size_t oldSize = buffer->size();
161  buffer->resize(oldSize + nmemb * size);
162  memcpy(buffer->data() + oldSize, userdata, nmemb * size);
163  return size * nmemb;
164 }

174 ExpirationHandler::Handler
175  LifetimeHelpers::fetchFromCCDBCache(ConcreteDataMatcher const& matcher,
176  std::string const& prefix,
177  std::string const& overrideTimestamp,
178  std::string const& sourceChannel)
179 {
180  char* err;
181  uint64_t overrideTimestampMilliseconds = strtoll(overrideTimestamp.c_str(), &err, 10);
182  if (*err != 0) {
183  throw std::runtime_error("fetchFromCCDBCache: Unable to parse forced timestamp for conditions");
184  }
185  if (overrideTimestampMilliseconds) {
186  LOGP(info, "fetchFromCCDBCache: forcing timestamp for conditions to {} milliseconds from epoch UTC", overrideTimestampMilliseconds);
187  }
188  return [matcher, sourceChannel, serverUrl = prefix, overrideTimestampMilliseconds](ServiceRegistry& services, PartRef& ref, uint64_t timestamp) -> void {
189  // We should invoke the handler only once.
190  assert(!ref.header);
191  assert(!ref.payload);
192 
193  auto& rawDeviceService = services.get<RawDeviceService>();
194  auto&& transport = rawDeviceService.device()->GetChannel(sourceChannel, 0).Transport();
195  auto channelAlloc = o2::pmr::getTransportAllocator(transport);
196  o2::vector<char> payloadBuffer;
197  payloadBuffer.reserve(10000); // we begin with messages of 10KB
198  auto payload = o2::pmr::getMessage(std::forward<o2::vector<char>>(payloadBuffer), transport->GetMemoryResource());
199 
200  CURL* curl = curl_easy_init();
201  if (curl == nullptr) {
202  throw std::runtime_error("fetchFromCCDBCache: Unable to initialise CURL");
203  }
204  CURLcode res;
205  if (overrideTimestampMilliseconds) {
206  timestamp = overrideTimestampMilliseconds;
207  }
208  auto path = std::string("/") + matcher.origin.as<std::string>() + "/" + matcher.description.as<std::string>() + "/" + std::to_string(timestamp / 1000);
209  auto url = serverUrl + path;
210  LOG(INFO) << "fetchFromCCDBCache: Fetching " << url;
211 
212  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
213  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &payloadBuffer);
214  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, readToMessage);
215  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
216 
217  res = curl_easy_perform(curl);
218  if (res != CURLE_OK) {
219  throw std::runtime_error(std::string("fetchFromCCDBCache: Unable to fetch ") + url + " from CCDB");
220  }
221  long responseCode;
222  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &responseCode);
223 
224  if (responseCode != 200) {
225  throw std::runtime_error(std::string("fetchFromCCDBCache: HTTP error ") + std::to_string(responseCode) + " while fetching " + url + " from CCDB");
226  }
227 
228  curl_easy_cleanup(curl);
229 
230  DataHeader dh;
231  dh.dataOrigin = matcher.origin;
232  dh.dataDescription = matcher.description;
233  dh.subSpecification = matcher.subSpec;
234  // FIXME: should use curl_off_t and CURLINFO_SIZE_DOWNLOAD_T, but
235  // apparently not there on some platforms.
236  double dl;
237  res = curl_easy_getinfo(curl, CURLINFO_SIZE_DOWNLOAD, &dl);
238  dh.payloadSize = payloadBuffer.size();
239  dh.payloadSerializationMethod = gSerializationMethodNone;
240 
241  DataProcessingHeader dph{timestamp, 1};
242  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, dph});
243 
244  ref.header = std::move(header);
245  ref.payload = std::move(payload);
246 
247  return;
248  };
249 }

274 ExpirationHandler::Handler LifetimeHelpers::enumerate(ConcreteDataMatcher const& matcher, std::string const& sourceChannel)
275 {
276  using counter_t = int64_t;
277  auto counter = std::make_shared<counter_t>(0);
278  auto f = [matcher, counter, sourceChannel](ServiceRegistry& services, PartRef& ref, uint64_t timestamp) -> void {
279  // We should invoke the handler only once.
280  assert(!ref.header);
281  assert(!ref.payload);
282  auto& rawDeviceService = services.get<RawDeviceService>();
283 
284  DataHeader dh;
285  dh.dataOrigin = matcher.origin;
286  dh.dataDescription = matcher.description;
287  dh.subSpecification = matcher.subSpec;
288  dh.payloadSize = sizeof(counter_t);
289  dh.payloadSerializationMethod = gSerializationMethodNone;
290 
291  DataProcessingHeader dph{timestamp, 1};
292 
293  auto&& transport = rawDeviceService.device()->GetChannel(sourceChannel, 0).Transport();
294  auto channelAlloc = o2::pmr::getTransportAllocator(transport);
295  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, dph});
296  ref.header = std::move(header);
297 
298  auto payload = rawDeviceService.device()->NewMessage(sizeof(counter_t));
299  *(counter_t*)payload->GetData() = *counter;
300  ref.payload = std::move(payload);
301  (*counter)++;
302  };
303  return f;
304 }
