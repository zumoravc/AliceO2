35 void Cell::setAbsId(short absId)
36 {
37  //14 bits available
38  if (absId > kNmaxCell || absId < 0)
39  absId = kNmaxCell;
40  ULong_t t = (ULong_t)absId;
41 
42  ULong_t b = getLong() & 0xffffffc000; // 1111 1111 1111 1111 1111 1111 1100 0000 0000 0000
43  mBits = b + t;
44 }

77 void Cell::setTime(float time)
78 {
79  //10 bits available for time
80  ULong_t t = 0;
81  //Convert time to long
82  t = ULong_t((time + kTime0) / kTimeAccuracy);
83  if (t > 0x3ff) {
84  t = 0x3ff;
85  } else {
86  if (t < 0) {
87  t = 0;
88  }
89  }
90 
91  t <<= 14;
92  ULong_t b = getLong() & 0xffff003fff; // 1111 1111 1111 1111 0000 0000 0011 1111 1111 1111
93  mBits = b + t;
94 }

96 float Cell::getTime() const
97 {
98  ULong_t t = getLong();
99  t >>= 14;
100  t &= 0x3ff;
101  //Convert back long to float
102

106 void Cell::setEnergy(float energy)
107 {
108  //15 bits
109  ULong_t a = static_cast<ULong_t>(energy / kEnergyConv);
110  a = a & 0x7FFF; //15 bits
111 
112  a <<= 24;
113  ULong_t b = getLong() & 0x8000ffffff; // 1000 0000 0000 0000 1111 1111 1111 1111 1111 1111
114  mBits = b + a;
115 }
