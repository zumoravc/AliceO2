103 void TrackExtrap::linearExtrapToZCov(TrackParam* trackParam, double zEnd, bool updatePropagator)
104 {
107 
108  if (trackParam->getZ() == zEnd) {
109  return; // nothing to be done if same z
110  }
111 
112  // No need to propagate the covariance matrix if it does not exist
113  if (!trackParam->hasCovariances()) {
114  LOG(WARNING) << "Covariance matrix does not exist";
115  // Extrapolate linearly track parameters to "zEnd"
116  linearExtrapToZ(trackParam, zEnd);
117  return;
118  }
119 
120  // Compute track parameters
121  double dZ = zEnd - trackParam->getZ();
122  trackParam->setNonBendingCoor(trackParam->getNonBendingCoor() + trackParam->getNonBendingSlope() * dZ);
123  trackParam->setBendingCoor(trackParam->getBendingCoor() + trackParam->getBendingSlope() * dZ);
124  trackParam->setZ(zEnd);
125 
126  // Calculate the jacobian related to the track parameters linear extrapolation to "zEnd"
127  TMatrixD jacob(5, 5);
128  jacob.UnitMatrix();
129  jacob(0, 1) = dZ;
130  jacob(2, 3) = dZ;
131 
132  // Extrapolate track parameter covariances to "zEnd"
133  TMatrixD tmp(trackParam->getCovariances(), TMatrixD::kMultTranspose, jacob);
134  TMatrixD tmp2(jacob, TMatrixD::kMult, tmp);
135  trackParam->setCovariances(tmp2);
136 
137  // Update the propagator if required
138  if (updatePropagator) {
139  trackParam->updatePropagator(jacob);
140  }
141 }

158 //__________________________________________________________________________
159 bool TrackExtrap::extrapToZCov(TrackParam* trackParam, double zEnd, bool updatePropagator)
160 {
163 
164  ++sNCallExtrapToZCov;
165 
166  if (trackParam->getZ() == zEnd) {
167  return true; // nothing to be done if same z
168  }
169 
170  if (!sFieldON) { // linear extrapolation if no magnetic field
171  linearExtrapToZCov(trackParam, zEnd, updatePropagator);
172  return true;
173  }
174 
175  // No need to propagate the covariance matrix if it does not exist
176  if (!trackParam->hasCovariances()) {
177  LOG(WARNING) << "Covariance matrix does not exist";
178  // Extrapolate track parameters to "zEnd"
179  return extrapToZ(trackParam, zEnd);
180  }
181 
182  // Save the actual track parameters
183  TrackParam trackParamSave(*trackParam);
184  TMatrixD paramSave(trackParamSave.getParameters());
185  double zBegin = trackParamSave.getZ();
186 
187  // Get reference to the parameter covariance matrix
188  const TMatrixD& kParamCov = trackParam->getCovariances();
189 
190  // Extrapolate track parameters to "zEnd"
191  // Do not update the covariance matrix if the extrapolation failed
192  if (!extrapToZ(trackParam, zEnd)) {
193  return false;
194  }
195 
196  // Get reference to the extrapolated parameters
197  const TMatrixD& extrapParam = trackParam->getParameters();
198 
199  // Calculate the jacobian related to the track parameters extrapolation to "zEnd"
200  TMatrixD jacob(5, 5);
201  jacob.Zero();
202  TMatrixD dParam(5, 1);
203  double direction[5] = {-1., -1., 1., 1., -1.};
204  for (int i = 0; i < 5; i++) {
205  // Skip jacobian calculation for parameters with no associated error
206  if (kParamCov(i, i) <= 0.) {
207  continue;
208  }
209 
210  // Small variation of parameter i only
211  for (int j = 0; j < 5; j++) {
212  if (j == i) {
213  dParam(j, 0) = TMath::Sqrt(kParamCov(i, i));
214  dParam(j, 0) *= TMath::Sign(1., direction[j] * paramSave(j, 0)); // variation always in the same direction
215  } else {
216  dParam(j, 0) = 0.;
217  }
218  }
219 
220  // Set new parameters
221  trackParamSave.setParameters(paramSave);
222  trackParamSave.addParameters(dParam);
223  trackParamSave.setZ(zBegin);
224 
225  // Extrapolate new track parameters to "zEnd"
226  if (!extrapToZ(&trackParamSave, zEnd)) {
227  LOG(WARNING) << "Bad covariance matrix";
228  return false;
229  }
230 
231  // Calculate the jacobian
232  TMatrixD jacobji(trackParamSave.getParameters(), TMatrixD::kMinus, extrapParam);
233  jacobji *= 1. / dParam(i, 0);
234  jacob.SetSub(0, i, jacobji);
235  }
236 
237  // Extrapolate track parameter covariances to "zEnd"
238  TMatrixD tmp(kParamCov, TMatrixD::kMultTranspose, jacob);
239  TMatrixD tmp2(jacob, TMatrixD::kMult, tmp);
240  trackParam->setCovariances(tmp2);
241 
242  // Update the propagator if required
243  if (updatePropagator) {
244  trackParam->updatePropagator(jacob);
245  }
246 
247  return true;
248 }

251 bool TrackExtrap::extrapToVertex(TrackParam* trackParam, double xVtx, double yVtx, double zVtx,
252  double errXVtx, double errYVtx, bool correctForMCS, bool correctForEnergyLoss)
253 {
261 
262  if (trackParam->getZ() == zVtx) {
263  return true; // nothing to be done if already at vertex
264  }
265 
266  // Check the vertex position with respect to the absorber (spectro z<0)
267  if (zVtx < SAbsZBeg) {
268  if (zVtx < SAbsZEnd) {
269  LOG(WARNING) << "Ending Z (" << zVtx << ") downstream the front absorber (zAbsorberEnd = " << SAbsZEnd << ")";
270  return false;
271  } else {
272  LOG(WARNING) << "Ending Z (" << zVtx << ") inside the front absorber (" << SAbsZBeg << ", " << SAbsZEnd << ")";
273  return false;
274  }
275  }
276 
277  // Check the track position with respect to the vertex and the absorber (spectro z<0)
278  if (trackParam->getZ() > SAbsZEnd) {
279  if (trackParam->getZ() > zVtx) {
280  LOG(WARNING) << "Starting Z (" << trackParam->getZ() << ") upstream the vertex (zVtx = " << zVtx << ")";
281  return false;
282  } else if (trackParam->getZ() > SAbsZBeg) {
283  LOG(WARNING) << "Starting Z (" << trackParam->getZ() << ") upstream the front absorber (zAbsorberBegin = " << SAbsZBeg << ")";
284  return false;
285  } else {
286  LOG(WARNING) << "Starting Z (" << trackParam->getZ() << ") inside the front absorber (" << SAbsZBeg << ", " << SAbsZEnd << ")";
287  return false;
288  }
289  }
290 
291  // Extrapolate track parameters (and covariances if any) to the end of the absorber
292  if ((trackParam->hasCovariances() && !extrapToZCov(trackParam, SAbsZEnd)) ||
293  (!trackParam->hasCovariances() && !extrapToZ(trackParam, SAbsZEnd))) {
294  return false;
295  }
296 
297  // Get absorber correction parameters assuming linear propagation in absorber
298  double trackXYZOut[3] = {trackParam->getNonBendingCoor(), trackParam->getBendingCoor(), trackParam->getZ()};
299  double trackXYZIn[3] = {0., 0., 0.};
300  if (correctForMCS) { // assume linear propagation to the vertex
301  trackXYZIn[2] = SAbsZBeg;
302  trackXYZIn[0] = trackXYZOut[0] + (xVtx - trackXYZOut[0]) / (zVtx - trackXYZOut[2]) * (trackXYZIn[2] - trackXYZOut[2]);
303  trackXYZIn[1] = trackXYZOut[1] + (yVtx - trackXYZOut[1]) / (zVtx - trackXYZOut[2]) * (trackXYZIn[2] - trackXYZOut[2]);
304  } else { // or standard propagation without vertex constraint
305  TrackParam trackParamIn(*trackParam);
306  if (!extrapToZ(&trackParamIn, SAbsZBeg)) {
307  return false;
308  }
309  trackXYZIn[0] = trackParamIn.getNonBendingCoor();
310  trackXYZIn[1] = trackParamIn.getBendingCoor();
311  trackXYZIn[2] = trackParamIn.getZ();
312  }
313  double pTot = trackParam->p();
314  double pathLength(0.), f0(0.), f1(0.), f2(0.), meanRho(0.), totalELoss(0.), sigmaELoss2(0.);
315  if (!getAbsorberCorrectionParam(trackXYZIn, trackXYZOut, pTot, pathLength, f0, f1, f2, meanRho, totalELoss, sigmaELoss2)) {
316  return false;
317  }
318 
319  // Compute track parameters and covariances at vertex according to correctForMCS and correctForEnergyLoss flags
320  if (correctForMCS) {
321  if (correctForEnergyLoss) {
322  // Correct for multiple scattering and energy loss
323  correctELossEffectInAbsorber(trackParam, 0.5 * totalELoss, 0.5 * sigmaELoss2);
324  if (!correctMCSEffectInAbsorber(trackParam, xVtx, yVtx, zVtx, errXVtx, errYVtx, trackXYZIn[2], pathLength, f0, f1, f2)) {
325  return false;
326  }
327  correctELossEffectInAbsorber(trackParam, 0.5 * totalELoss, 0.5 * sigmaELoss2);
328  } else {
329  // Correct for multiple scattering
330  if (!correctMCSEffectInAbsorber(trackParam, xVtx, yVtx, zVtx, errXVtx, errYVtx, trackXYZIn[2], pathLength, f0, f1, f2)) {
331  return false;
332  }
333  }
334  } else {
335  if (correctForEnergyLoss) {
336  // Correct for energy loss add multiple scattering dispersion in covariance matrix
337  correctELossEffectInAbsorber(trackParam, 0.5 * totalELoss, 0.5 * sigmaELoss2);
338  addMCSEffectInAbsorber(trackParam, -pathLength, f0, f1, f2); // (spectro. (z<0))
339  if (!extrapToZCov(trackParam, trackXYZIn[2])) {
340  return false;
341  }
342  correctELossEffectInAbsorber(trackParam, 0.5 * totalELoss, 0.5 * sigmaELoss2);
343  if (!extrapToZCov(trackParam, zVtx)) {
344  return false;
345  }
346  } else {
347  // add multiple scattering dispersion in covariance matrix
348  addMCSEffectInAbsorber(trackParam, -pathLength, f0, f1, f2); // (spectro. (z<0))
349  if (!extrapToZCov(trackParam, zVtx)) {
350  return false;
351  }
352  }
353  }
354 
355  return true;
356 }

359 bool TrackExtrap::getAbsorberCorrectionParam(double trackXYZIn[3], double trackXYZOut[3], double pTotal,
360  double& pathLength, double& f0, double& f1, double& f2,
361  double& meanRho, double& totalELoss, double& sigmaELoss2)
362 {
372 
373  // Check whether the geometry is available
374  if (!gGeoManager) {
375  LOG(WARNING) << "geometry is missing";
376  return false;
377  }
378 
379  // Initialize starting point and direction
380  pathLength = TMath::Sqrt((trackXYZOut[0] - trackXYZIn[0]) * (trackXYZOut[0] - trackXYZIn[0]) +
381  (trackXYZOut[1] - trackXYZIn[1]) * (trackXYZOut[1] - trackXYZIn[1]) +
382  (trackXYZOut[2] - trackXYZIn[2]) * (trackXYZOut[2] - trackXYZIn[2]));
383  if (pathLength < TGeoShape::Tolerance()) {
384  LOG(WARNING) << "path length is too small";
385  return false;
386  }
387  double b[3] = {(trackXYZOut[0] - trackXYZIn[0]) / pathLength, (trackXYZOut[1] - trackXYZIn[1]) / pathLength, (trackXYZOut[2] - trackXYZIn[2]) / pathLength};
388  TGeoNode* currentnode = gGeoManager->InitTrack(trackXYZIn, b);
389  if (!currentnode) {
390  LOG(WARNING) << "starting point out of geometry";
391  return false;
392  }
393 
394  // loop over absorber slices and calculate absorber's parameters
395  f0 = f1 = f2 = meanRho = totalELoss = 0.;
396  double sigmaELoss(0.);
397  double zB = trackXYZIn[2];
398  double remainingPathLength = pathLength;
399  do {
400 
401  // Get material properties
402  TGeoMaterial* material = currentnode->GetVolume()->GetMedium()->GetMaterial();
403  double rho = material->GetDensity(); // material density (g/cm3)
404  double x0 = material->GetRadLen(); // radiation-length (cm-1)
405  double atomicZ = material->GetZ(); // Z of material
406  double atomicZoverA(0.); // Z/A of material
407  if (material->IsMixture()) {
408  TGeoMixture* mixture = static_cast<TGeoMixture*>(material);
409  double sum(0.);
410  for (int iel = 0; iel < mixture->GetNelements(); ++iel) {
411  sum += mixture->GetWmixt()[iel];
412  atomicZoverA += mixture->GetWmixt()[iel] * mixture->GetZmixt()[iel] / mixture->GetAmixt()[iel];
413  }
414  atomicZoverA /= sum;
415  } else {
416  atomicZoverA = atomicZ / material->GetA();
417  }
418 
419  // Get path length within this material
420  gGeoManager->FindNextBoundary(remainingPathLength);
421  double localPathLength = gGeoManager->GetStep() + 1.e-6;
422  // Check if boundary within remaining path length. If so, make sure to cross the boundary to prepare the next step
423  if (localPathLength >= remainingPathLength) {
424  localPathLength = remainingPathLength;
425  } else {
426  currentnode = gGeoManager->Step();
427  if (!currentnode) {
428  LOG(WARNING) << "navigation failed";
429  return false;
430  }
431  if (!gGeoManager->IsEntering()) {
432  // make another small step to try to enter in new absorber slice
433  gGeoManager->SetStep(0.001);
434  currentnode = gGeoManager->Step();
435  if (!gGeoManager->IsEntering() || !currentnode) {
436  LOG(WARNING) << "navigation failed";
437  return false;
438  }
439  localPathLength += 0.001;
440  }
441  }
442 
443  // calculate absorber's parameters
444  double zE = b[2] * localPathLength + zB;
445  double dzB = zB - trackXYZIn[2];
446  double dzE = zE - trackXYZIn[2];
447  f0 += localPathLength / x0;
448  f1 += (dzE * dzE - dzB * dzB) / b[2] / b[2] / x0 / 2.;
449  f2 += (dzE * dzE * dzE - dzB * dzB * dzB) / b[2] / b[2] / b[2] / x0 / 3.;
450  meanRho += localPathLength * rho;
451  totalELoss += betheBloch(pTotal, localPathLength, rho, atomicZ, atomicZoverA);
452  sigmaELoss += energyLossFluctuation(pTotal, localPathLength, rho, atomicZoverA);
453 
454  // prepare next step
455  zB = zE;
456  remainingPathLength -= localPathLength;
457  } while (remainingPathLength > TGeoShape::Tolerance());
458 
459  meanRho /= pathLength;
460  sigmaELoss2 = sigmaELoss * sigmaELoss;
461 
462  return true;
463 }

466 double TrackExtrap::getMCSAngle2(const TrackParam& param, double dZ, double x0)
467 {
471  double bendingSlope = param.getBendingSlope();
472  double nonBendingSlope = param.getNonBendingSlope();
473  double inverseTotalMomentum2 = param.getInverseBendingMomentum() * param.getInverseBendingMomentum() *
474  (1.0 + bendingSlope * bendingSlope) /
475  (1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
476  // Path length in the material
477  double pathLength = TMath::Abs(dZ) * TMath::Sqrt(1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
478  // relativistic velocity
479  double velo = 1.;
480  // Angular dispersion square of the track (variance) in a plane perpendicular to the trajectory
481  double theta02 = 0.0136 / velo * (1 + 0.038 * TMath::Log(pathLength / x0));
482  return theta02 * theta02 * inverseTotalMomentum2 * pathLength / x0;
483 }

485 //__________________________________________________________________________
486 void TrackExtrap::addMCSEffect(TrackParam* trackParam, double dZ, double x0)
487 {
493 
494  double bendingSlope = trackParam->getBendingSlope();
495  double nonBendingSlope = trackParam->getNonBendingSlope();
496  double inverseBendingMomentum = trackParam->getInverseBendingMomentum();
497  double inverseTotalMomentum2 = inverseBendingMomentum * inverseBendingMomentum * (1.0 + bendingSlope * bendingSlope) /
498  (1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
499  // Path length in the material
500  double signedPathLength = dZ * TMath::Sqrt(1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
501  double pathLengthOverX0 = (x0 > 0.) ? TMath::Abs(signedPathLength) / x0 : TMath::Abs(signedPathLength);
502  // relativistic velocity
503  double velo = 1.;
504  // Angular dispersion square of the track (variance) in a plane perpendicular to the trajectory
505  double theta02 = 0.0136 / velo * (1 + 0.038 * TMath::Log(pathLengthOverX0));
506  theta02 *= theta02 * inverseTotalMomentum2 * pathLengthOverX0;
507 
508  double varCoor = (x0 > 0.) ? signedPathLength * signedPathLength * theta02 / 3. : 0.;
509  double varSlop = theta02;
510  double covCorrSlope = (x0 > 0.) ? signedPathLength * theta02 / 2. : 0.;
511 
512  // Set MCS covariance matrix
513  TMatrixD newParamCov(trackParam->getCovariances());
514  // Non bending plane
515  newParamCov(0, 0) += varCoor;
516  newParamCov(0, 1) += covCorrSlope;
517  newParamCov(1, 0) += covCorrSlope;
518  newParamCov(1, 1) += varSlop;
519  // Bending plane
520  newParamCov(2, 2) += varCoor;
521  newParamCov(2, 3) += covCorrSlope;
522  newParamCov(3, 2) += covCorrSlope;
523  newParamCov(3, 3) += varSlop;
524 
525  // Set momentum related covariances if B!=0
526  if (sFieldON) {
527  // compute derivative d(q/Pxy) / dSlopeX and d(q/Pxy) / dSlopeY
528  double dqPxydSlopeX =
529  inverseBendingMomentum * nonBendingSlope / (1. + nonBendingSlope * nonBendingSlope + bendingSlope * bendingSlope);
530  double dqPxydSlopeY = -inverseBendingMomentum * nonBendingSlope * nonBendingSlope * bendingSlope /
531  (1. + bendingSlope * bendingSlope) /
532  (1. + nonBendingSlope * nonBendingSlope + bendingSlope * bendingSlope);
533  // Inverse bending momentum (due to dependences with bending and non bending slopes)
534  newParamCov(4, 0) += dqPxydSlopeX * covCorrSlope;
535  newParamCov(0, 4) += dqPxydSlopeX * covCorrSlope;
536  newParamCov(4, 1) += dqPxydSlopeX * varSlop;
537  newParamCov(1, 4) += dqPxydSlopeX * varSlop;
538  newParamCov(4, 2) += dqPxydSlopeY * covCorrSlope;
539  newParamCov(2, 4) += dqPxydSlopeY * covCorrSlope;
540  newParamCov(4, 3) += dqPxydSlopeY * varSlop;
541  newParamCov(3, 4) += dqPxydSlopeY * varSlop;
542  newParamCov(4, 4) += (dqPxydSlopeX * dqPxydSlopeX + dqPxydSlopeY * dqPxydSlopeY) * varSlop;
543  }
544 
545  // Set new covariances
546  trackParam->setCovariances(newParamCov);
547 }

549 //__________________________________________________________________________
550 void TrackExtrap::addMCSEffectInAbsorber(TrackParam* param, double signedPathLength, double f0, double f1, double f2)
551 {
554 
555  // absorber related covariance parameters
556  double bendingSlope = param->getBendingSlope();
557  double nonBendingSlope = param->getNonBendingSlope();
558  double inverseBendingMomentum = param->getInverseBendingMomentum();
559  double alpha2 = 0.0136 * 0.0136 * inverseBendingMomentum * inverseBendingMomentum * (1.0 + bendingSlope * bendingSlope) /
560  (1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope); // velocity = 1
561  double pathLength = TMath::Abs(signedPathLength);
562  double varCoor = alpha2 * (pathLength * pathLength * f0 - 2. * pathLength * f1 + f2);
563  double covCorrSlope = TMath::Sign(1., signedPathLength) * alpha2 * (pathLength * f0 - f1);
564  double varSlop = alpha2 * f0;
565 
566  // Set MCS covariance matrix
567  TMatrixD newParamCov(param->getCovariances());
568  // Non bending plane
569  newParamCov(0, 0) += varCoor;
570  newParamCov(0, 1) += covCorrSlope;
571  newParamCov(1, 0) += covCorrSlope;
572  newParamCov(1, 1) += varSlop;
573  // Bending plane
574  newParamCov(2, 2) += varCoor;
575  newParamCov(2, 3) += covCorrSlope;
576  newParamCov(3, 2) += covCorrSlope;
577  newParamCov(3, 3) += varSlop;
578 
579  // Set momentum related covariances if B!=0
580  if (sFieldON) {
581  // compute derivative d(q/Pxy) / dSlopeX and d(q/Pxy) / dSlopeY

602 double TrackExtrap::betheBloch(double pTotal, double pathLength, double rho, double atomicZ, double atomicZoverA)
603 {
607 
608  static constexpr double mK = 0.307075e-3; // [GeV*cm^2/g]
609  static constexpr double me = 0.511e-3; // [GeV/c^2]
610  static constexpr double x0 = 0.2 * 2.303; // density effect first junction point * 2.303
611  static constexpr double x1 = 3. * 2.303; // density effect second junction point * 2.303
612 
613  double bg = pTotal / SMuMass; // beta*gamma
614  double bg2 = bg * bg;
615  double maxT = 2 * me * bg2; // neglecting the electron mass
616 
617  // mean exitation energy (GeV)
618  double mI = (atomicZ < 13) ? (12. * atomicZ + 7.) * 1.e-9 : (9.76 * atomicZ + 58.8 * TMath::Power(atomicZ, -0.19)) * 1.e-9;
619 
620  // density effect
621  double x = TMath::Log(bg);
622  double lhwI = TMath::Log(28.816 * 1e-9 * TMath::Sqrt(rho * atomicZoverA) / mI);
623  double d2(0.);
624  if (x > x1) {
625  d2 = lhwI + x - 0.5;
626  } else if (x > x0) {
627  double r = (x1 - x) / (x1 - x0);
628  d2 = lhwI + x - 0.5 + (0.5 - lhwI - x0) * r * r * r;
629  }
630 
631  return pathLength * rho * (mK * atomicZoverA * (1 + bg2) / bg2 * (0.5 * TMath::Log(2 * me * bg2 * maxT / (mI * mI)) - bg2 / (1 + bg2) - d2));

754 void TrackExtrap::cov2CovP(const TMatrixD& param, TMatrixD& cov)
755 {
758 
759  // charge * total momentum
760  double qPTot = TMath::Sqrt(1. + param(1, 0) * param(1, 0) + param(3, 0) * param(3, 0)) /

764  TMatrixD jacob(5, 5);
765  jacob.UnitMatrix();

768  (1. + param(3, 0) * param(3, 0)) / (1. + param(1, 0) * param(1, 0) + param(3, 0) * param(3, 0));
769  jacob(4, 4) = -qPTot / param(4, 0);

832 bool TrackExtrap::extrapToZRungekutta(TrackParam* trackParam, double zEnd)
833 {
837  if (trackParam->getZ() == zEnd) {
838  return true; // nothing to be done if same Z
839  }
840  double forwardBackward; // +1 if forward, -1 if backward
841  if (zEnd < trackParam->getZ()) {
842  forwardBackward = 1.0; // spectro. z<0
843  } else {
844  forwardBackward = -1.0;
845  }
846  // sign of charge (sign of fInverseBendingMomentum if forward motion)
847  // must be changed if backward extrapolation
848  double chargeExtrap = forwardBackward * TMath::Sign(double(1.0), trackParam->getInverseBendingMomentum());
849  double v3[7] = {0.}, v3New[7] = {0.};
850  double dZ(0.), step(0.);
851  int stepNumber = 0;
852 
853  // Extrapolation loop (until within tolerance or the track turn around)
854  double residue = zEnd - trackParam->getZ();
855  while (TMath::Abs(residue) > SRungeKuttaMaxResidue) {
856 
857  dZ = zEnd - trackParam->getZ();
858  // step length assuming linear trajectory
859  step = dZ * TMath::Sqrt(1.0 + trackParam->getBendingSlope() * trackParam->getBendingSlope() +
860  trackParam->getNonBendingSlope() * trackParam->getNonBendingSlope());
861  convertTrackParamForExtrap(trackParam, forwardBackward, v3);
862 
863  do { // reduce step length while zEnd overstepped
864  if (++stepNumber > SMaxStepNumber) {
865  LOG(WARNING) << "Too many trials";
866  return false;
867  }
868  step = TMath::Abs(step);
869  if (!extrapOneStepRungekutta(chargeExtrap, step, v3, v3New)) {
870  return false;
871  }
872  residue = zEnd - v3New[2];
873  step *= dZ / (v3New[2] - trackParam->getZ());
874  } while (residue * dZ < 0 && TMath::Abs(residue) > SRungeKuttaMaxResidue);
875 
876  if (v3New[5] * v3[5] < 0) { // the track turned around
877  LOG(WARNING) << "The track turned around";
878  return false;
879  }
880 
881  recoverTrackParam(v3New, chargeExtrap * forwardBackward, trackParam);
882  }
883 
884  // terminate the extropolation with a straight line up to the exact "zEnd" value
885  trackParam->setNonBendingCoor(trackParam->getNonBendingCoor() + residue * trackParam->getNonBendingSlope());
886  trackParam->setBendingCoor(trackParam->getBendingCoor() + residue * trackParam->getBendingSlope());
887  trackParam->setZ(zEnd);
888 
889  return true;
890 }
