110 inline Ray::Ray(const Point3D<float> point0, const Point3D<float> point1)
111  : mP{point0.X(), point0.Y(), point0.Z()}, mD{point1.X() - point0.X(), point1.Y() - point0.Y(), point1.Z() - point0.Z()}
112 {
113  mDistXY2 = mD[0] * mD[0] + mD[1] * mD[1];
114  mDistXY2i = mDistXY2 > 0 ? 1.f / mDistXY2 : 0.f;
115  mDistXYZ = o2::gpu::CAMath::Sqrt(mDistXY2 + mD[2] * mD[2]);
116  mXDxPlusYDy = point0.X() * mD[0] + point0.Y() * mD[1];
117  mXDxPlusYDyRed = -mXDxPlusYDy * mDistXY2i;
118  mXDxPlusYDy2 = mXDxPlusYDy * mXDxPlusYDy;
119  mR02 = point0.Perp2();
120  mR12 = point1.Perp2();
121 }

125 GPUdi() Ray::Ray(float x0, float y0, float z0, float x1, float y1, float z1)
126  : mP{x0, y0, z0}, mD{x1 - x0, y1 - y0, z1 - z0}
127 {
128  mDistXY2 = mD[0] * mD[0] + mD[1] * mD[1];
129  mDistXY2i = mDistXY2 > 0 ? 1.f / mDistXY2 : 0.f;
130  mDistXYZ = o2::gpu::CAMath::Sqrt(mDistXY2 + mD[2] * mD[2]);
131  mXDxPlusYDy = x0 * mD[0] + y0 * mD[1];
132  mXDxPlusYDyRed = -mXDxPlusYDy * mDistXY2i;
133  mXDxPlusYDy2 = mXDxPlusYDy * mXDxPlusYDy;
134  mR02 = x0 * x0 + y0 * y0;
135  mR12 = x1 * x1 + y1 * y1;
136 }

139 GPUdi() float Ray::crossRadial(float cs, float sn) const
140 {
141  // calculate t of crossing with radial line with inclination cosine and sine
142  float den = mD[0] * sn - mD[1] * cs;
143  if (o2::gpu::CAMath::Abs(den) < Tiny) {
144  return InvalidT;
145  }
146  return (mP[1] * cs - mP[0] * sn) / den;
147 }

149 //______________________________________________________
150 GPUdi() bool Ray::crossCircleR(float r2, float& cross1, float& cross2) const
151 {
152  // calculate parameters t of intersection with circle of radius r^2
153  // calculated as solution of equation
154  // t^2*mDistXY2 +- sqrt( mXDxPlusYDy^2 - mDistXY2*(mR02 - r^2) )
155  //
156  float det = mXDxPlusYDy2 - mDistXY2 * (mR02 - r2);
157  if (det < 0)
158  return false; // no intersection
159  float detRed = o2::gpu::CAMath::Sqrt(det) * mDistXY2i;
160  cross1 = mXDxPlusYDyRed + detRed; // (-mXDxPlusYDy + det)*mDistXY2i;
161  cross2 = mXDxPlusYDyRed - detRed; // (-mXDxPlusYDy - det)*mDistXY2i;
162  return true;
163 }

179 //______________________________________________________
180 GPUdi() bool Ray::validateZRange(float& cpar1, float& cpar2, const MatLayerCyl& lr) const
181 {
182  // make sure that estimated crossing parameters are compatible
183  // with Z coverage of the layer
184  MatLayerCyl::RangeStatus zout0 = lr.isZOutside(getZ(cpar1)), zout1 = lr.isZOutside(getZ(cpar2));
185  if (zout0 == zout1) { // either both points outside w/o crossing or boht inside
186  return zout0 == MatLayerCyl::Within ? true : false;
187  }
188  // at least 1 point is outside, but there is a crossing
189  if (zout0 != MatLayerCyl::Within) {
190  cpar1 = crossZ(zout0 == MatLayerCyl::Below ? lr.getZMin() : lr.getZMax());
191  }
192  if (zout1 != MatLayerCyl::Within) {
193  cpar2 = crossZ(zout1 == MatLayerCyl::Below ? lr.getZMin() : lr.getZMax());
194  }
195  return true;
196 }

199 GPUdi() void Ray::getMinMaxR2(float& rmin2, float& rmax2) const
200 {
201  // calculate min and max R2
202  if (mR02 > mR12) {
203  rmin2 = mR12;
204  rmax2 = mR02;
205  } else {
206  rmin2 = mR02;
207  rmax2 = mR12;
208  }
209  if (mXDxPlusYDyRed > 0.f && mXDxPlusYDyRed < 1.f) {
210  // estimate point of closest approach to origin as the crossing of normal from the origin to input vector
211  // use r^2(t) = mR02 + t^2 (mD[0]^2+mD[1]^2) + 2t*mXDxPlusYDy
212  float xMin = mP[0] + mXDxPlusYDyRed * mD[0], yMin = mP[1] + mXDxPlusYDyRed * mD[1];
213  rmin2 = xMin * xMin + yMin * yMin;
214  }
