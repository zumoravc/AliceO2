93 Int_t CalibPulser::updateROC(const Int_t roc, const Int_t row, const Int_t pad,
94  const Int_t timeBin, Float_t signal)
95 {
96  // ===| range checks |========================================================
97  if (timeBin < mFirstTimeBin || timeBin > mLastTimeBin) {
98  return 0;
99  }
100 
101  // ---| pedestal subtraction |---
102  if (mPedestal) {
103  signal -= mPedestal->getValue(ROC(roc), row, pad);
104  }
105 
106  if (signal < mADCMin || signal > mADCMax)
107  return 0;
108 
109  // ===| temporary calibration data |==========================================
110  const PadROCPos padROCPos(roc, row, pad);
111  VectorType& adcData = mPulserData[padROCPos];
112 
113  if (!adcData.size()) {
114  // accept first and last time bin, so difference +1
115  adcData.resize(mLastTimeBin - mFirstTimeBin + 1);
116  //printf("new pad pos\n");
117  }
118 
119  adcData[timeBin - mFirstTimeBin] = signal;
120  //printf("%2d, %3d, %3d, %3d: %.2f\n", roc, row, pad, timeBin, signal);
121  return 1;
122 }

133  //std::cout << (int)padROCPos.getROC().getRoc() << ", " << padROCPos.getRow() << ", " << padROCPos.getPad() << std::endl;
134  //for (auto& val: adcData) std::cout << val << ", ";
135  //std::cout << std::endl;
136 
137  const auto data = processPadData(padROCPos, adcData);
138  //std::cout << data.mT0+mFirstTimeBin << " : " << data.mQtot << " : " << data.mWidth << "\n";
139 
140  // fill histograms
141  getHistoT0(padROCPos.getROC(), true)->Fill(data.mT0 + mFirstTimeBin, currentChannel);
142  getHistoQtot(padROCPos.getROC(), true)->Fill(data.mQtot, currentChannel);
143  getHistoSigma(padROCPos.getROC(), true)->Fill(data.mWidth, currentChannel);
144  }
145 
146  // reset the adc data to free space
147  mPulserData.clear();
148 }

169 CalibPulser::PulserData CalibPulser::processPadData(const PadROCPos& padROCPos, const CalibPulser::VectorType& adcData)
170 {
171  // data to return
172  PulserData data;
173 
174  const auto vectorSize = adcData.size();
175  const auto maxElement = std::max_element(std::begin(adcData), std::end(adcData));
176  const auto maxPosition = std::distance(std::begin(adcData), maxElement);
177 
178  double weightedSum = 0.;
179  double weightedSum2 = 0.;
180  double chargeSum = 0.;
181  for (int t = maxPosition - mPeakIntMinus; t <= maxPosition + mPeakIntPlus; ++t) {
182  const auto signal = adcData[t];
183  // check time bounds
184  if (t < 0 || t >= vectorSize)
185  continue;
186  weightedSum += signal * (t + 0.5); // +0.5 to get the center of the time bin
187  weightedSum2 += signal * (t + 0.5) * (t + 0.5);
188  chargeSum += signal;
189  }
190 
191  if (chargeSum > mMinimumQtot) {
192  weightedSum /= chargeSum;
193  weightedSum2 = std::sqrt(std::abs(weightedSum2 / chargeSum - weightedSum * weightedSum));
194  // L1 phase correction?
195  data.mT0 = weightedSum;
196  data.mWidth = weightedSum2;
197  data.mQtot = chargeSum;
198  }
199 
200  return data;
201 }

218 //______________________________________________________________________________
219 void CalibPulser::analyse()
220 {
221  for (ROC roc; !roc.looped(); ++roc) {
222  auto histT0 = mT0Histograms.at(roc).get();
223  auto histWidth = mWidthHistograms.at(roc).get();
224  auto histQtot = mQtotHistograms.at(roc).get();
225 
226  if (!histT0 || !histWidth || !histQtot)
227  continue;
228 
229  // array pointer
230  const auto arrT0 = histT0->GetArray();
231  const auto arrWidth = histWidth->GetArray();
232  const auto arrQtot = histQtot->GetArray();
233 
234  const auto numberOfPads = mMapper.getNumberOfPads(roc);
235  for (auto iChannel = 0; iChannel < numberOfPads; ++iChannel) {
236  const int offsetT0 = (mNbinsT0 + 2) * (iChannel + 1) + 1;
237  const int offsetQtot = (mNbinsQtot + 2) * (iChannel + 1) + 1;
238  const int offsetWidth = (mNbinsWidth + 2) * (iChannel + 1) + 1;
239 
240  StatisticsData dataT0 = getStatisticsData(arrT0 + offsetT0, mNbinsT0, mXminT0, mXmaxT0);
241  StatisticsData dataWidth = getStatisticsData(arrWidth + offsetWidth, mNbinsWidth, mXminWidth, mXmaxWidth);
242  StatisticsData dataQtot = getStatisticsData(arrQtot + offsetQtot, mNbinsQtot, mXminQtot, mXmaxQtot);
243 
244  mT0.getCalArray(roc).setValue(iChannel, dataT0.mCOG);
245  mWidth.getCalArray(roc).setValue(iChannel, dataWidth.mCOG);
246  mQtot.getCalArray(roc).setValue(iChannel, dataQtot.mCOG);
247  }
248  }
249 }
