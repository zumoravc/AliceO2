300 Bool_t TrackParamMFT::isCompatibleTrackParamMFT(const TrackParamMFT& TrackParamMFT, Double_t sigma2Cut, Double_t& chi2) const
301 {
306 
307  // reset chi2 value
308  chi2 = 0.;
309 
310  // ckeck covariance matrices
311  if (!mCovariances && !TrackParamMFT.mCovariances) {
312  LOG(ERROR) << "Covariance matrix must exist for at least one set of parameters";
313  return kFALSE;
314  }
315 
316  Double_t maxChi2 = 5. * sigma2Cut * sigma2Cut; // 5 degrees of freedom
317 
318  // check Z parameters
319  if (mZ != TrackParamMFT.mZ)
320  LOG(WARN) << "Parameters are given at different Z position (" << mZ << " : " << TrackParamMFT.mZ
321  << "): results are meaningless";
322 
323  // compute the parameter residuals
324  TMatrixD deltaParam(mParameters, TMatrixD::kMinus, TrackParamMFT.mParameters);
325 
326  // build the error matrix
327  TMatrixD weight(5, 5);
328  if (mCovariances)
329  weight += *mCovariances;
330  if (TrackParamMFT.mCovariances)
331  weight += *(TrackParamMFT.mCovariances);
332 
333  // invert the error matrix to get the parameter weights if possible
334  if (weight.Determinant() == 0) {
335  LOG(ERROR) << "Cannot compute the compatibility chi2";
336  return kFALSE;
337  }
338  weight.Invert();
339 
340  // compute the compatibility chi2
341  TMatrixD tmp(deltaParam, TMatrixD::kTransposeMult, weight);
342  TMatrixD mChi2(tmp, TMatrixD::kMult, deltaParam);
343 
344  // set chi2 value
345  chi2 = mChi2(0, 0);
346 
347  // check compatibility
348  if (chi2 > maxChi2)
349  return kFALSE;
350 
351  return kTRUE;
352 }
