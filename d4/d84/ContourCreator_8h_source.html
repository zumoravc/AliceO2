32 template <typename T>
33 Contour<T> createContour(const std::vector<Polygon<T>>& polygons)
34 {
35  if (polygons.empty()) {
36  return {};
37  }
38 
39  if (!isCounterClockwiseOriented(polygons)) {
40  throw std::invalid_argument("polygons should be oriented counterclockwise");
41  }
42 
43  // trivial case : only one input polygon
44  if (polygons.size() == 1) {
45  Contour<T> trivialContour;
46  trivialContour.addPolygon(polygons.front());
47  return trivialContour;
48  }
49 
50  std::vector<impl::VerticalEdge<T>> polygonVerticalEdges{impl::getVerticalEdges(polygons)};
51 
52  sortVerticalEdges(polygonVerticalEdges);
53 
54  // Initialize the segment tree that is used by the sweep() function
55  std::unique_ptr<impl::Node<T>> segmentTree{impl::createSegmentTree(impl::getYPositions(polygons))};
56 
57  // Find the vertical edges of the merged contour. This is the meat of the algorithm...
58  std::vector<impl::VerticalEdge<T>> contourVerticalEdges{impl::sweep(segmentTree.get(), polygonVerticalEdges)};
59 
60  // Deduce the horizontal edges from the vertical ones
61  std::vector<impl::HorizontalEdge<T>> contourHorizontalEdges{impl::verticalsToHorizontals(contourVerticalEdges)};
62 
63  return impl::finalizeContour(contourVerticalEdges, contourHorizontalEdges);
64 }

66 template <typename T>
67 Contour<T> getEnvelop(const std::vector<Contour<T>>& list)
68 {
70  std::vector<o2::mch::contour::Polygon<T>> polygons;
71  for (const auto& c : list) {
72  for (auto j = 0; j < c.size(); ++j) {
73  polygons.push_back(c[j]);
74  }
75  }
76  return createContour(polygons);
77 }
