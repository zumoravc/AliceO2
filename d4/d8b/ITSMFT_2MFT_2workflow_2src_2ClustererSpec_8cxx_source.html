41 void ClustererDPL::init(InitContext& ic)
42 {
43  o2::base::GeometryManager::loadGeometry(); // for generating full clusters
44  o2::mft::GeometryTGeo* geom = o2::mft::GeometryTGeo::Instance();
45  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L));
46 
47  mClusterer = std::make_unique<o2::itsmft::Clusterer>();
48  mClusterer->setGeometry(geom);
49  mClusterer->setNChips(o2::itsmft::ChipMappingMFT::getNChips());
50  LOG(INFO) << "MFT ClustererDPL::init total number of sensors " << o2::itsmft::ChipMappingMFT::getNChips() << "\n";
51 
52  //mClusterer->setMaskOverflowPixels(false);
53 
54  auto filenameGRP = ic.options().get<std::string>("grp-file");
55  const auto grp = o2::parameters::GRPObject::loadFrom(filenameGRP.c_str());
56 
57  if (grp) {
58  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut("MFT"));
59  } else {
60  LOG(ERROR) << "Cannot retrieve GRP from the " << filenameGRP.c_str() << " file !";
61  mState = 0;
62  return;
63  }
64 
65  mFullClusters = ic.options().get<bool>("full-clusters");
66  mPatterns = !ic.options().get<bool>("no-patterns");
67  mNThreads = ic.options().get<int>("nthreads");
68 
69  // settings for the fired pixel overflow masking
70  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::MFT>::Instance();
71  const auto& clParams = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance();
72  mClusterer->setMaxBCSeparationToMask(alpParams.roFrameLength / o2::constants::lhc::LHCBunchSpacingNS + clParams.maxBCDiffToMaskBias);
73  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
74 
75  std::string dictPath = ic.options().get<std::string>("mft-dictionary-path");
76  std::string dictFile = o2::base::NameConf::getDictionaryFileName(o2::detectors::DetID::MFT, dictPath, ".bin");
77  if (o2::base::NameConf::pathExists(dictFile)) {
78  mClusterer->loadDictionary(dictFile);
79  LOG(INFO) << "ITSClusterer running with a provided dictionary: " << dictFile;
80  } else {
81  LOG(INFO) << "Dictionary " << dictFile << " is absent, ITSClusterer expects cluster patterns";
82  }
83  mState = 1;
84  mClusterer->print();
85 }

87 void ClustererDPL::run(ProcessingContext& pc)
88 {
89  auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
90  auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
91 
92  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> labels;
93  gsl::span<const o2::itsmft::MC2ROFRecord> mc2rofs;
94  if (mUseMC) {
95  labels = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
96  mc2rofs = pc.inputs().get<gsl::span<o2::itsmft::MC2ROFRecord>>("MC2ROframes");
97  }
98 
99  LOG(INFO) << "MFTClusterer pulled " << digits.size() << " digits, in "
100  << rofs.size() << " RO frames";
101 
102  o2::itsmft::DigitPixelReader reader;
103  reader.setDigits(digits);
104  reader.setROFRecords(rofs);
105  if (mUseMC) {
106  reader.setMC2ROFRecords(mc2rofs);
107  reader.setDigitsMCTruth(labels.get());
108  }
109  reader.init();
110  auto orig = o2::header::gDataOriginMFT;
111  std::vector<o2::itsmft::Cluster> clusVec;
112  std::vector<o2::itsmft::CompClusterExt> clusCompVec;
113  std::vector<o2::itsmft::ROFRecord> clusROFVec;
114  std::vector<unsigned char> clusPattVec;
115 
116  std::unique_ptr<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> clusterLabels;
117  if (mUseMC) {
118  clusterLabels = std::make_unique<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>();
119  }
120  mClusterer->process(mNThreads, reader, mFullClusters ? &clusVec : nullptr, &clusCompVec, mPatterns ? &clusPattVec : nullptr, &clusROFVec, clusterLabels.get());
121  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
122  pc.outputs().snapshot(Output{orig, "MFTClusterROF", 0, Lifetime::Timeframe}, clusROFVec);
123  pc.outputs().snapshot(Output{orig, "CLUSTERS", 0, Lifetime::Timeframe}, clusVec);
124  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
125 
126  if (mUseMC) {
127  pc.outputs().snapshot(Output{orig, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, *clusterLabels.get()); // at the moment requires snapshot
128  std::vector<o2::itsmft::MC2ROFRecord> clusterMC2ROframes(mc2rofs.size());
129  for (int i = mc2rofs.size(); i--;) {
130  clusterMC2ROframes[i] = mc2rofs[i]; // Simply, replicate it from digits ?
131  }
132  pc.outputs().snapshot(Output{orig, "MFTClusterMC2ROF", 0, Lifetime::Timeframe}, clusterMC2ROframes);
133  }
134 
135  // TODO: in principle, after masking "overflow" pixels the MC2ROFRecord maxROF supposed to change, nominally to minROF
136  // -> consider recalculationg maxROF
137  LOG(INFO) << "MFTClusterer pushed " << clusCompVec.size() << " clusters, in " << clusROFVec.size() << " RO frames";
138 }

140 DataProcessorSpec getClustererSpec(bool useMC)
141 {
142  std::vector<InputSpec> inputs;
143  inputs.emplace_back("digits", "MFT", "DIGITS", 0, Lifetime::Timeframe);
144  inputs.emplace_back("ROframes", "MFT", "MFTDigitROF", 0, Lifetime::Timeframe);
145 
146  std::vector<OutputSpec> outputs;
147  outputs.emplace_back("MFT", "COMPCLUSTERS", 0, Lifetime::Timeframe);
148  outputs.emplace_back("MFT", "PATTERNS", 0, Lifetime::Timeframe);
149  outputs.emplace_back("MFT", "CLUSTERS", 0, Lifetime::Timeframe);
150  outputs.emplace_back("MFT", "MFTClusterROF", 0, Lifetime::Timeframe);
151 
152  if (useMC) {
153  inputs.emplace_back("labels", "MFT", "DIGITSMCTR", 0, Lifetime::Timeframe);
154  inputs.emplace_back("MC2ROframes", "MFT", "MFTDigitMC2ROF", 0, Lifetime::Timeframe);
155  outputs.emplace_back("MFT", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
156  outputs.emplace_back("MFT", "MFTClusterMC2ROF", 0, Lifetime::Timeframe);
157  }
158 
159  return DataProcessorSpec{
160  "mft-clusterer",
161  inputs,
162  outputs,
163  AlgorithmSpec{adaptFromTask<ClustererDPL>(useMC)},
164  Options{
165  {"mft-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
166  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
167  {"full-clusters", o2::framework::VariantType::Bool, true, {"Produce full clusters"}}, // RSTODO temporary set to true
168  {"no-patterns", o2::framework::VariantType::Bool, false, {"Do not save rare cluster patterns"}},
169  {"nthreads", VariantType::Int, 0, {"Number of clustering threads (<1: rely on openMP default)"}}}};
170 }
