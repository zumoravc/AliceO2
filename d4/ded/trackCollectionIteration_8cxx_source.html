19 // This is a very simple example showing how to iterate over tracks
20 // and operate on them.
21 // FIXME: this should really inherit from AnalysisTask but
22 // we need GCC 7.4+ for that
23 struct ATask {
24  void process(aod::Tracks const& tracks)
25  {
26  // FIXME: to see some output, we create the histogram
27  // for every timeframe. In general this is not the way it
28  // should be done.
29  auto hPhi = new TH1F("phi", "Phi", 100, 0, 2 * M_PI);
30  auto hEta = new TH1F("eta", "Eta", 100, 0, 2 * M_PI);
31  for (auto& track : tracks) {
32  auto phi = asin(track.snp()) + track.alpha() + M_PI;
33  auto eta = log(tan(0.25 * M_PI - 0.5 * atan(track.tgl())));
34  hPhi->Fill(phi);
35  hEta->Fill(eta);
36  }
37  TFile f("result1.root", "RECREATE");
38  hPhi->SetName("Phi");
39  hPhi->Write();
40  hEta->SetName("Eta");
41  hEta->Write();
42  }
43 };
