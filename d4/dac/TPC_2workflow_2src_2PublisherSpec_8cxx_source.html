43 DataProcessorSpec createPublisherSpec(PublisherConf const& config, bool propagateMC, workflow_reader::Creator creator)
44 {
45  if (config.tpcSectors.size() == 0 || config.outputIds.size() == 0) {
46  throw std::invalid_argument("need TPC sector and output id configuration");
47  }
48  constexpr static size_t NSectors = o2::tpc::Sector::MAXSECTOR;
49  enum struct SectorMode {
50  Sector, // stored in sector branches
51  Full, // full TPC stored in one branch
52  };
53  struct ProcessAttributes {
54  std::vector<int> sectors;
55  std::vector<int> outputIds;
56  std::vector<o2::header::DataHeader::SubSpecificationType> zeroLengthOutputs;
57  uint64_t activeSectors = 0;
58  std::array<std::shared_ptr<RootTreeReader>, NSectors> readers;
59  bool terminateOnEod = false;
60  bool finished = false;
61  SectorMode sectorMode = SectorMode::Sector;
62  };
63 
64  auto initFunction = [config, propagateMC, creator](InitContext& ic) {
65  // get the option from the init context
66  auto filename = ic.options().get<std::string>("infile");
67  auto treename = ic.options().get<std::string>("treename");
68  auto clbrName = ic.options().get<std::string>(config.databranch.option.c_str());
69  auto mcbrName = ic.options().get<std::string>(config.mcbranch.option.c_str());
70  auto nofEvents = ic.options().get<int>("nevents");
71  auto publishingMode = nofEvents == -1 ? RootTreeReader::PublishingMode::Single : RootTreeReader::PublishingMode::Loop;
72 
73  // do a runtime check if the branch name without sector number suffix is found in the file
74  // if found the publisher will publish the single data set at one output route and empty
75  // messages at all the others
76  auto checkSectorMode = [&filename, &treename, &clbrName]() -> SectorMode {
77  std::unique_ptr<TFile> file(TFile::Open(filename.c_str()));
78  if (file) {
79  TTree* tree = reinterpret_cast<TTree*>(file->GetObjectChecked(treename.c_str(), "TTree"));
80  if (tree) {
81  const auto brlist = tree->GetListOfBranches();
82  for (TObject const* entry : *brlist) {
83  if (clbrName == entry->GetName()) {
84  return SectorMode::Full;
85  }
86  }
87  }
88  file->Close();
89  }
90  return SectorMode::Sector;
91  };
92 
93  auto processAttributes = std::make_shared<ProcessAttributes>();
94  {
95  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
96  processAttributes->sectorMode = checkSectorMode();
97  auto& sectors = processAttributes->sectors;
98  auto& activeSectors = processAttributes->activeSectors;
99  auto& readers = processAttributes->readers;
100  auto& outputIds = processAttributes->outputIds;
101  auto& sectorMode = processAttributes->sectorMode;
102 
103  sectors = config.tpcSectors;
104  outputIds = config.outputIds;
105  for (auto const& s : sectors) {
106  // set the mask of active sectors
107  if (s >= NSectors) {
108  std::string message = std::string("invalid sector range specified, allowed 0-") + std::to_string(NSectors - 1);
109  // FIXME should probably be FATAL, but this doesn't seem to be handled in the DPL control flow
110  // at least the process is not marked dead in the DebugGUI
111  LOG(ERROR) << message;
112  throw std::invalid_argument(message);
113  }
114  activeSectors |= (uint64_t)0x1 << s;
115  }
116 
117  // set up the tree interface
118  // TODO: parallelism on sectors needs to be implemented as selector in the reader
119  // the data is now in parallel branches, as first attempt use an array of readers
120  auto outputId = outputIds.begin();
121  for (auto const& sector : sectors) {
122  o2::header::DataHeader::SubSpecificationType subSpec = *outputId;
123  std::string sectorfile = filename;
124  if (filename.find('%') != std::string::npos) {
125  vector<char> formattedname(filename.length() + 10, 0);
126  snprintf(formattedname.data(), formattedname.size() - 1, filename.c_str(), sector);
127  sectorfile = formattedname.data();
128  }
129  std::string clusterbranchname = clbrName;
130  std::string mcbranchname = mcbrName;
131  if (sectorMode == SectorMode::Sector) {
132  clusterbranchname += "_" + std::to_string(sector);
133  mcbranchname += "_" + std::to_string(sector);
134  }
135  readers[sector] = creator(treename.c_str(), // tree name
136  sectorfile.c_str(), // input file name
137  nofEvents, // number of entries to publish
138  publishingMode,
139  subSpec,
140  clusterbranchname.c_str(), // name of data branch
141  mcbranchname.c_str() // name of mc label branch
142  );
143  if (sectorMode == SectorMode::Full) {
144  break;
145  }
146  if (++outputId == outputIds.end()) {
147  outputId = outputIds.begin();
148  }
149  }
150  if (sectorMode == SectorMode::Full) {
151  // the slot of the first configured sector is used to publish the full set, all others removed
152  sectors.resize(1);
153  // the data will be published at first configured output id, zero-length data on all other output ids
154  processAttributes->zeroLengthOutputs.assign(++outputId, outputIds.end());
155  }
156  }
157 
158  // set up the processing function
159  // using by-copy capture of the worker instance shared pointer
160  // the shared pointer makes sure to clean up the instance when the processing
161  // function gets out of scope
162  // FIXME: wanted to use it = sectors.begin() in the variable capture but the iterator
163  // is const and can not be incremented
164  auto processingFct = [processAttributes, config](ProcessingContext& pc) {
165  if (processAttributes->finished) {
166  return;
167  }
168 
169  bool eos = false;
170  auto const& sectors = processAttributes->sectors;
171  for (auto const& sector : sectors) {
172  auto& activeSectors = processAttributes->activeSectors;
173  auto& readers = processAttributes->readers;
174  o2::tpc::TPCSectorHeader header{sector};
175  if (processAttributes->sectorMode == SectorMode::Full) {
176  header.sectorBits = activeSectors;
177  }
178  header.activeSectors = activeSectors;
179  auto& r = *(readers[sector].get());
180 
181  // increment the reader and invoke it for the processing context
182  if (r.next()) {
183  // there is data, run the reader
184  r(pc, header);
185  } else {
186  // no more data, delete the reader
187  readers[sector].reset();
188  eos = true;
189  }
190  }
191 
192  if (eos) {
193  processAttributes->finished = true;
194  pc.services().get<ControlService>().endOfStream();
195  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
196  } else {
197  // publish empty events
198  auto dto = DataSpecUtils::asConcreteDataTypeMatcher(config.dataoutput);
199  auto mco = DataSpecUtils::asConcreteDataTypeMatcher(config.mcoutput);
200  o2::tpc::TPCSectorHeader header{0};
201  header.sectorBits = 0;
202  header.activeSectors = processAttributes->activeSectors;
203  for (auto const& subSpec : processAttributes->zeroLengthOutputs) {
204  pc.outputs().make<char>({dto.origin, dto.description, subSpec, Lifetime::Timeframe, {header}});
205  if (pc.outputs().isAllowed({mco.origin, mco.description, subSpec})) {
206  pc.outputs().make<char>({mco.origin, mco.description, subSpec, Lifetime::Timeframe, {header}});
207  }
208  }
209  }
210  };
211 
212  // return the actual processing function as a lambda function using variables
213  // of the init function
214  return processingFct;
215  };
216 
217  auto createOutputSpecs = [&config, propagateMC]() {
218  std::vector<OutputSpec> outputSpecs;
219  for (size_t n = 0; n < config.outputIds.size(); ++n) {
220  o2::header::DataHeader::SubSpecificationType subSpec = config.outputIds[n];
221  auto dto = DataSpecUtils::asConcreteDataTypeMatcher(config.dataoutput);
222  auto mco = DataSpecUtils::asConcreteDataTypeMatcher(config.mcoutput);
223  outputSpecs.emplace_back(OutputSpec{{"output"}, dto.origin, dto.description, subSpec, Lifetime::Timeframe});
224  if (propagateMC) {
225  outputSpecs.emplace_back(OutputSpec{{"outputMC"}, mco.origin, mco.description, subSpec, Lifetime::Timeframe});
226  }
227  }
228  return std::move(outputSpecs);
229  };
230 
231  auto& dtb = config.databranch;
232  auto& mcb = config.mcbranch;
233  return DataProcessorSpec{config.processName.c_str(),
234  Inputs{}, // no inputs
235  {createOutputSpecs()},
236  AlgorithmSpec(initFunction),
237  Options{
238  {"infile", VariantType::String, "", {"Name of the input file"}},
239  {"treename", VariantType::String, config.defaultTreeName.c_str(), {"Name of input tree"}},
240  {dtb.option.c_str(), VariantType::String, dtb.defval.c_str(), {dtb.help.c_str()}},
241  {mcb.option.c_str(), VariantType::String, mcb.defval.c_str(), {mcb.help.c_str()}},
242  {"nevents", VariantType::Int, -1, {"number of events to run"}},
243  {"terminate-on-eod", VariantType::Bool, true, {"terminate on end-of-data"}},
244  }};
245 }
