27 // because comparing the whole thing is a pain.
28 void lineByLineComparision(const std::string& as, const std::string& bs)
29 {
30  std::istringstream a(as);
31  std::istringstream b(bs);
32 
33  char bufferA[1024];
34  char bufferB[1024];
35  while (a.good() && b.good()) {
36  a.getline(bufferA, 1024);
37  b.getline(bufferB, 1024);
38  BOOST_CHECK_EQUAL(std::string(bufferA), std::string(bufferB));
39  }
40  BOOST_CHECK(a.eof());
41  BOOST_CHECK(b.eof());
42 }
