40 BOOST_AUTO_TEST_CASE(test_tableview_reverse)
41 {
42  using FrameT = o2::algorithm::Composite<HeartbeatHeader, HeartbeatTrailer>;
43  using TestFrame = o2::algorithm::StaticSequenceAllocator;
44  // the length of the data is set in the trailer word
45  // the header is used as column description, using slightly different
46  // orbit numbers in the to data sets which will result in two complete
47  // columns at beginning and end, while the two in the middle only have
48  // one row entry
49  TestFrame tf1(FrameT({0x1100000000000000}, "heartbeatdata", {0x510000000000000e}),
50  FrameT({0x1100000000000001}, "test", {0x5100000000000005}),
51  FrameT({0x1100000000000003}, "dummydata", {0x510000000000000a}));
52  TestFrame tf2(FrameT({0x1100000000000000}, "frame2a", {0x5100000000000008}),
53  FrameT({0x1100000000000002}, "frame2b", {0x5100000000000008}),
54  FrameT({0x1100000000000003}, "frame2c", {0x5100000000000008}));
55  hexDump("Test frame 1", tf1.buffer.get(), tf1.size());
56  hexDump("Test frame 2", tf2.buffer.get(), tf2.size());
57 
58  // the payload length is set in the trailer, so we need a reverse parser
59  using ParserT = o2::algorithm::ReverseParser<typename FrameT::HeaderType,
60  typename FrameT::TrailerType>;
61 
62  // define the view type for DataHeader as row descriptor,
63  // HeartbeatHeader as column descriptor and the reverse parser
64  using ViewType = o2::algorithm::TableView<o2::header::DataHeader,
65  o2::header::HeartbeatHeader,
66  ParserT>;
67  ViewType heartbeatview;
68 
69  o2::header::DataHeader dh1;
70  dh1.dataDescription = o2::header::DataDescription("FIRSTROW");
71  dh1.dataOrigin = o2::header::DataOrigin("TST");
72  dh1.subSpecification = 0;
73  dh1.payloadSize = 0;
74 
75  o2::header::DataHeader dh2;
76  dh2.dataDescription = o2::header::DataDescription("SECONDROW");
77  dh2.dataOrigin = o2::header::DataOrigin("TST");
78  dh2.subSpecification = 0xdeadbeef;
79  dh2.payloadSize = 0;
80 
81  heartbeatview.addRow(dh1, tf1.buffer.get(), tf1.size());
82  heartbeatview.addRow(dh2, tf2.buffer.get(), tf2.size());
83 
84  std::cout << "slots: " << heartbeatview.getNRows()
85  << " columns: " << heartbeatview.getNColumns()
86  << std::endl;
87 
88  // definitions for the data check
89  const char* dataset1[] = {
90  "heartbeatdata",
91  "test",
92  "dummydata"};
93  const char* dataset2[] = {
94  "frame2a",
95  "frame2b",
96  "frame2c"};
97 
98  // four orbits are populated, 0 and 3 with 2 rows, 1 and 2 with one row
99  BOOST_REQUIRE(heartbeatview.getNColumns() == 4);
100  BOOST_REQUIRE(heartbeatview.getNRows() == 2);
101  unsigned requiredNofRowsInColumn[] = {2, 1, 1, 2};
102 
103  unsigned colidx = 0;
104  unsigned dataset1idx = 0;
105  unsigned dataset2idx = 0;
106  for (auto columnIt = heartbeatview.begin(), end = heartbeatview.end();
107  columnIt != end; ++columnIt, ++colidx) {
108  unsigned rowidx = 0;
109  std::cout << "---------------------------------------" << std::endl;
110  for (auto row : columnIt) {
111  auto dataset = (rowidx == 1 || colidx == 2) ? dataset2 : dataset1;
112  auto& datasetidx = (rowidx == 1 || colidx == 2) ? dataset2idx : dataset1idx;
113  hexDump("Entry", row.buffer, row.size);
114  BOOST_CHECK(memcmp(row.buffer, dataset[datasetidx++], row.size) == 0);
115  ++rowidx;
116  }
117  BOOST_CHECK(rowidx == requiredNofRowsInColumn[colidx]);
118  }
119 }
