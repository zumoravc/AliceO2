35 int GPUTPCO2Interface::Initialize(const GPUO2InterfaceConfiguration& config)
36 {
37  if (mInitialized) {
38  return (1);
39  }
40  mConfig.reset(new GPUO2InterfaceConfiguration(config));
41  mContinuous = mConfig->configEvent.continuousMaxTimeBin != 0;
42  mRec.reset(GPUReconstruction::CreateInstance(mConfig->configProcessing));
43  if (mRec == nullptr) {
44  GPUError("Error obtaining instance of GPUReconstruction");
45  return 1;
46  }
47  mChain = mRec->AddChain<GPUChainTracking>(mConfig->configInterface.maxTPCHits, mConfig->configInterface.maxTRDTracklets);
48  mChain->mConfigDisplay = &mConfig->configDisplay;
49  mChain->mConfigQA = &mConfig->configQA;
50  if (mConfig->configWorkflow.inputs.isSet(GPUDataTypes::InOutType::TPCRaw)) {
51  mConfig->configEvent.needsClusterer = 1;
52  }
53  mRec->SetSettings(&mConfig->configEvent, &mConfig->configReconstruction, &mConfig->configDeviceProcessing, &mConfig->configWorkflow);
54  mChain->SetTPCFastTransform(mConfig->configCalib.fastTransform);
55  mChain->SetdEdxSplines(mConfig->configCalib.dEdxSplines);
56  mChain->SetMatLUT(mConfig->configCalib.matLUT);
57  mChain->SetTRDGeometry(mConfig->configCalib.trdGeometry);
58  if (mConfig->configInterface.outputToExternalBuffers) {
59  mOutputCompressedClusters.reset(new GPUOutputControl);
60  mChain->SetOutputControlCompressedClusters(mOutputCompressedClusters.get());
61  mOutputClustersNative.reset(new GPUOutputControl);
62  mChain->SetOutputControlClustersNative(mOutputClustersNative.get());
63  }
64 
65  if (mRec->Init()) {
66  return (1);
67  }
68  if (!mRec->IsGPU() && mConfig->configDeviceProcessing.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
69  mRec->MemoryScalers()->factor *= 2;
70  }
71  mInitialized = true;
72  return (0);
73 }

84 int GPUTPCO2Interface::RunTracking(GPUTrackingInOutPointers* data, GPUInterfaceOutputs* outputs)
85 {
86  if (!mInitialized) {
87  return (1);
88  }
89  static int nEvent = 0;
90  if (mConfig->configInterface.dumpEvents) {
91  mChain->ClearIOPointers();
92  mChain->mIOPtrs.clustersNative = data->clustersNative;
93  mChain->mIOPtrs.tpcPackedDigits = data->tpcPackedDigits;
94  mChain->mIOPtrs.tpcZS = data->tpcZS;
95 
96  char fname[1024];
97  sprintf(fname, "event.%d.dump", nEvent);
98  mChain->DumpData(fname);
99  if (nEvent == 0) {
100  mRec->DumpSettings();
101  }
102  if (mConfig->configInterface.dumpEvents >= 2) {
103  return 0;
104  }
105  }
106 
107  mChain->mIOPtrs = *data;
108  if (mConfig->configInterface.outputToExternalBuffers) {
109  if (outputs->compressedClusters.allocator) {
110  mOutputCompressedClusters->set(outputs->compressedClusters.allocator);
111  } else if (outputs->compressedClusters.ptr) {
112  mOutputCompressedClusters->set(outputs->compressedClusters.ptr, outputs->compressedClusters.size);
113  } else {
114  mOutputCompressedClusters->reset();
115  }
116  if (outputs->clustersNative.allocator) {
117  mOutputClustersNative->set(outputs->clustersNative.allocator);
118  } else if (outputs->clustersNative.ptr) {
119  mOutputClustersNative->set(outputs->clustersNative.ptr, outputs->clustersNative.size);
120  } else {
121  mOutputClustersNative->reset();
122  }
123  }
124  int retVal = mRec->RunChains();
125  if (retVal == 2) {
126  retVal = 0; // 2 signals end of event display, ignore
127  }
128  if (retVal) {
129  mRec->ClearAllocatedMemory();
130  return retVal;
131  }
132  if (mConfig->configInterface.outputToExternalBuffers) {
133  outputs->compressedClusters.size = mOutputCompressedClusters->EndOfSpace ? 0 : mChain->mIOPtrs.tpcCompressedClusters->totalDataSize;
134  outputs->clustersNative.size = mOutputClustersNative->EndOfSpace ? 0 : (mChain->mIOPtrs.clustersNative->nClustersTotal * sizeof(*mChain->mIOPtrs.clustersNative->clustersLinear));
135  }
136  *data = mChain->mIOPtrs;
137 
138  nEvent++;
139  return 0;
140 }
