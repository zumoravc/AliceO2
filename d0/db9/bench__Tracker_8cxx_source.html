26 std::vector<o2::mid::Cluster2D> generateTestData(int nTracks, o2::mid::TrackGenerator& trackGen,
27  const o2::mid::HitFinder& hitFinder, const o2::mid::Mapping& mapping)
28 {
29  o2::mid::Mapping::MpStripIndex stripIndex;
30  o2::mid::MpArea area;
31  std::vector<o2::mid::Cluster2D> clusters;
32  o2::mid::Cluster2D cl;
33  std::vector<o2::mid::Track> tracks = trackGen.generate(nTracks);
34  for (auto& track : tracks) {
35  for (int ich = 0; ich < 4; ++ich) {
36  auto hits = hitFinder.getLocalPositions(track, ich);
37  bool isFired = false;
38  for (auto& hit : hits) {
39  int deId = hit.deId;
40  float xPos = hit.xCoor;
41  float yPos = hit.yCoor;
42  stripIndex = mapping.stripByPosition(xPos, yPos, 0, deId, false);
43  if (!stripIndex.isValid()) {
44  continue;
45  }
46  cl.deId = deId;
47  area = mapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
48  cl.yCoor = area.getCenterY();
49  float halfSize = area.getHalfSizeY();
50  cl.sigmaY2 = halfSize * halfSize / 3.;
51  stripIndex = mapping.stripByPosition(xPos, yPos, 1, deId, false);
52  area = mapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
53  cl.xCoor = area.getCenterX();
54  halfSize = area.getHalfSizeX();
55  cl.sigmaX2 = halfSize * halfSize / 3.;
56  clusters.push_back(cl);
57  } // loop on fired pos
58  } // loop on chambers
59  } // loop on tracks
60  return clusters;
61 }

63 static void BM_TRACKER(benchmark::State& state)
64 {
65  o2::mid::GeometryTransformer geoTrans = o2::mid::createDefaultTransformer();
66  o2::mid::TrackGenerator trackGen;
67  o2::mid::HitFinder hitFinder(geoTrans);
68  o2::mid::Mapping mapping;
69  o2::mid::Tracker tracker(geoTrans);
70 
71  int nTracksPerEvent = state.range(0);
72  tracker.init((state.range(1) == 1));
73  double num{0};
74 
75  std::vector<o2::mid::Cluster2D> inputData;
76 
77  for (auto _ : state) {
78  state.PauseTiming();
79  inputData = generateTestData(nTracksPerEvent, trackGen, hitFinder, mapping);
80  state.ResumeTiming();
81  tracker.process(inputData);
82  ++num;
83  }
84 
85  state.counters["num"] = benchmark::Counter(num, benchmark::Counter::kIsRate);
86 }
