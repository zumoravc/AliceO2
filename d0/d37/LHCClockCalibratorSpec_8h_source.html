35 class LHCClockCalibDevice : public o2::framework::Task
36 {
37  public:
38  void init(o2::framework::InitContext& ic) final
39  {
40  int minEnt = std::max(300, ic.options().get<int>("min-entries"));
41  int nb = std::max(500, ic.options().get<int>("nbins"));
42  int slotL = ic.options().get<int>("tf-per-slot");
43  int delay = ic.options().get<int>("max-delay");
44  mCalibrator = std::make_unique<o2::tof::LHCClockCalibrator>(minEnt, nb);
45  mCalibrator->setSlotLength(slotL);
46  mCalibrator->setMaxSlotsDelay(delay);
47  }
48 
49  void run(o2::framework::ProcessingContext& pc) final
50  {
51  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
52  auto data = pc.inputs().get<gsl::span<o2::dataformats::CalibInfoTOF>>("input");
53  LOG(INFO) << "Processing TF " << tfcounter << " with " << data.size() << " tracks";
54  mCalibrator->process(tfcounter, data);
55  sendOutput(pc.outputs());
56  const auto& infoVec = mCalibrator->getLHCphaseInfoVector();
57  LOG(INFO) << "Created " << infoVec.size() << " objects for TF " << tfcounter;
58  }
59 
60  void endOfStream(o2::framework::EndOfStreamContext& ec) final
61  {
62  LOG(INFO) << "Finalizing calibration";
63  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
64  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
65  sendOutput(ec.outputs());
66  }
67 
68  private:
69  std::unique_ptr<o2::tof::LHCClockCalibrator> mCalibrator;
70 
71  //________________________________________________________________
72  void sendOutput(DataAllocator& output)
73  {
74  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
75  // TODO in principle, this routine is generic, can be moved to Utils.h
76  using clbUtils = o2::calibration::Utils;
77  const auto& payloadVec = mCalibrator->getLHCphaseVector();
78  auto& infoVec = mCalibrator->getLHCphaseInfoVector(); // use non-const version as we update it
79  assert(payloadVec.size() == infoVec.size());
80 
81  for (uint32_t i = 0; i < payloadVec.size(); i++) {
82  auto& w = infoVec[i];
83  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
84  LOG(INFO) << "Senging object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
85  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
86  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, i}, *image.get()); // vector<char>
87  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, i}, w); // root-serialized
88  }
89  if (payloadVec.size()) {
90  mCalibrator->initOutput(); // reset the outputs once they are already sent
91  }
92  }
93 };
