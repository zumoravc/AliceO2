75 // ensure all clusters are either in sample mode or in
76 // chargesum mode, no mixing allowed
77 // and that all clusters share a single bunch crossing counter value
78 template <typename CHARGESUM>
79 void assertNotMixingClusters(const std::vector<SampaCluster>& data)
80 {
81  CHARGESUM a;
82  auto refValue = a();
83  uint32_t bunchCrossingCounter{0};
84  for (auto i = 0; i < data.size(); i++) {
85  if (data[i].isClusterSum() != refValue) {
86  throw std::invalid_argument(fmt::format("all cluster of this encoder should be of the same type ({}) but {}-th does not match ", (refValue ? "clusterSum" : "samples"), i));
87  }
88  auto bc = data[i].bunchCrossing;
89  if (i == 0) {
90  bunchCrossingCounter = bc;
91  }
92  if (bc != bunchCrossingCounter) {
93  throw std::invalid_argument(fmt::format("all sampa clusters should have the same bunch crossing number : first is {} and got {}\n", bunchCrossingCounter, bc));
94  }
95  }
96 }
