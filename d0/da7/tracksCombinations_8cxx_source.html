42 struct HashTask {
43  std::vector<float> xBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
44  std::vector<float> yBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
45  Produces<aod::Hashes> hashes;
46 
47  // Calculate hash for an element based on 2 properties and their bins.
48  int getHash(const std::vector<float>& xBins, const std::vector<float>& yBins, float colX, float colY)
49  {
50  for (int i = 0; i < xBins.size(); i++) {
51  if (colX < xBins[i]) {
52  for (int j = 0; j < yBins.size(); j++) {
53  if (colY < yBins[j]) {
54  return i + j * (xBins.size() + 1);
55  }
56  }
57  // overflow for yBins only
58  return i + yBins.size() * (xBins.size() + 1);
59  }
60  }
61 
62  // overflow for xBins only
63  for (int j = 0; j < yBins.size(); j++) {
64  if (colY < yBins[j]) {
65  return xBins.size() + j * (xBins.size() + 1);
66  }
67  }
68 
69  // overflow for both bins
70  return (yBins.size() + 1) * (xBins.size() + 1) - 1;
71  }
72 
73  void process(aod::Tracks const& tracks)
74  {
75  for (auto& track : tracks) {
76  int hash = getHash(xBins, yBins, track.x(), track.y());
77  LOGF(info, "Track: %d (%f, %f, %f) hash: %d", track.index(), track.x(), track.y(), track.z(), hash);
78  hashes(hash);
79  }
80  }
81 };
