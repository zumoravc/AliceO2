70  uint16_t composeFEEId(uint16_t layer, uint16_t ruOnLayer, uint16_t link) const
71  {
72  // only one link is used
73  // ruOnLayer is 0, 1, 2, 3 for half = 0
74  // 4, 5, 6, 7 1
75  auto dhalf = std::div(ruOnLayer, 4);
76  uint16_t half = dhalf.quot;
77  uint16_t zone = dhalf.rem;
78  auto ddisk = std::div(layer, 2);
79  uint16_t disk = ddisk.quot;
80  uint16_t plane = layer % 2;
81  return (half << 6) + (disk << 3) + (plane << 2) + zone;
82  }
83 
85  void expandFEEId(uint16_t feeID, uint16_t& layer, uint16_t& ruOnLayer, uint16_t& link) const
86  {
87  link = 0;
88  uint16_t half = feeID >> 6;
89  uint16_t disk = (feeID >> 3) & 0x7;
90  uint16_t plane = (feeID >> 2) & 0x1;
91  uint16_t zone = feeID & 0x3;
92  layer = 2 * disk + plane;
93  ruOnLayer = 2 * half + zone;
94  }
