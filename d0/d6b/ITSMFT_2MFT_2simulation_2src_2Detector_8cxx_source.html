101 Bool_t Detector::ProcessHits(FairVolume* vol)
102 {
103  // This method is called from the MC stepping
104 
105  // do not track neutral particles
106  if (!(fMC->TrackCharge())) {
107  return kFALSE;
108  }
109 
110  Geometry* mftGeo = Geometry::instance();
111 
112  Int_t copy;
113  // Check if hit is into a MFT sensor volume
114  if (fMC->CurrentVolID(copy) != mftGeo->getSensorVolumeID())
115  return kFALSE;
116 
117  // Get The Sensor Unique ID
118  Int_t sensorID = -1, ladderID = -1, diskID = -1, halfID = -1, level = 0;
119  fMC->CurrentVolOffID(++level, sensorID);
120  fMC->CurrentVolOffID(++level, ladderID);
121  fMC->CurrentVolOffID(++level, diskID);
122  fMC->CurrentVolOffID(++level, halfID);
123 
124  Int_t sensorIndex = mGeometryTGeo->getSensorIndex(halfID, diskID, ladderID, sensorID);
125 
126  // LOG(INFO) << "Found hit into half = " << halfID << "; disk = " << diskID << "; ladder = " << ladderID << "; sensor
127  // = " << sensorID ;
128 
129  bool startHit = false, stopHit = false;
130  unsigned char status = 0;
131  if (fMC->IsTrackEntering()) {
132  status |= Hit::kTrackEntering;
133  }
134  if (fMC->IsTrackInside()) {
135  status |= Hit::kTrackInside;
136  }
137  if (fMC->IsTrackExiting()) {
138  status |= Hit::kTrackExiting;
139  }
140  if (fMC->IsTrackOut()) {
141  status |= Hit::kTrackOut;
142  }
143  if (fMC->IsTrackStop()) {
144  status |= Hit::kTrackStopped;
145  }
146  if (fMC->IsTrackAlive()) {
147  status |= Hit::kTrackAlive;
148  }
149 
150  // track is entering or created in the volume
151  if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
152  startHit = true;
153  } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
154  stopHit = true;
155  }
156 
157  // increment energy loss at all steps except entrance
158  if (!startHit)
159  mTrackData.mEnergyLoss += fMC->Edep();
160  if (!(startHit | stopHit))
161  return kFALSE; // do noting
162 
163  if (startHit) {
164 
165  mTrackData.mEnergyLoss = 0.;
166  fMC->TrackMomentum(mTrackData.mMomentumStart);
167  fMC->TrackPosition(mTrackData.mPositionStart);
168  mTrackData.mTrkStatusStart = status;
169  mTrackData.mHitStarted = true;
170  }
171 
172  if (stopHit) {
173 
174  TLorentzVector positionStop;
175  fMC->TrackPosition(positionStop);
176 
177  Int_t trackID = fMC->GetStack()->GetCurrentTrackNumber();
178  // Int_t detID = vol->getMCid();
179 
180  Hit* p = addHit(trackID, sensorIndex, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
181  mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
182  mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
183 
184  o2::data::Stack* stack = (o2::data::Stack*)fMC->GetStack();
185  stack->addHit(GetDetId());
186  }
187 
188  return kTRUE;
189 }
