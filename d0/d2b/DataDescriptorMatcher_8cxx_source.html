25 ContextElement::Value const& VariableContext::get(size_t pos) const
26 {
27  // First we check if there is any pending update
28  for (size_t i = 0; i < mPerformedUpdates; ++i) {
29  if (mUpdates[i].position == pos) {
30  return mUpdates[i].newValue;
31  }
32  }
33  // Otherwise we return the element.
34  return mElements.at(pos).value;

53 bool OriginValueMatcher::match(header::DataHeader const& header, VariableContext& context) const
54 {
55  if (auto ref = std::get_if<ContextRef>(&mValue)) {
56  auto& variable = context.get(ref->index);
57  if (auto value = std::get_if<std::string>(&variable)) {
58  return strncmp(header.dataOrigin.str, value->c_str(), 4) == 0;
59  }
60  auto maxSize = strnlen(header.dataOrigin.str, 4);
61  context.put({ref->index, std::string(header.dataOrigin.str, maxSize)});
62  return true;
63  } else if (auto s = std::get_if<std::string>(&mValue)) {
64  return strncmp(header.dataOrigin.str, s->c_str(), 4) == 0;
65  }
66  throw std::runtime_error("Mismatching type for variable");
67 }

121  mRight{ConstantValueMatcher{false}}
122 {

126  mLeft = *pval1;

130  mLeft = std::move(std::make_unique<DataDescriptorMatcher>(*pval3->get()));
131  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&other.mLeft)) {
132  mLeft = *pval4;

136  std::cerr << (other.mLeft.index() == std::variant_npos) << std::endl;

138  }
139 
140  if (auto pval0 = std::get_if<OriginValueMatcher>(&other.mRight)) {
141  mRight = *pval0;
142  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&other.mRight)) {
143  mRight = *pval1;
144  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&other.mRight)) {
145  mRight = *pval2;

147  mRight = std::move(std::make_unique<DataDescriptorMatcher>(*pval3->get()));

149  mRight = *pval4;

151  mRight = *pval5;

210 // actual polymorphic matcher which is able to cast the pointer to the correct
211 // kind of header.
212 bool DataDescriptorMatcher::match(char const* d, VariableContext& context) const
213 {
214  bool leftValue = false, rightValue = false;
215 
216  // FIXME: Using std::visit is not API compatible due to a new
217  // exception being thrown. This is the ABI compatible version.
218  // Replace with:
219  //
220  // auto eval = [&d](auto&& arg) -> bool {
221  // using T = std::decay_t<decltype(arg)>;
222  // if constexpr (std::is_same_v<T, std::unique_ptr<DataDescriptorMatcher>>) {
223  // return arg->match(d, context);
224  // if constexpr (std::is_same_v<T, ConstantValueMatcher>) {
225  // return arg->match(d);
226  // } else {
227  // return arg.match(d, context);
228  // }
229  // };
230  // switch (mOp) {
231  // case Op::Or:
232  // return std::visit(eval, mLeft) || std::visit(eval, mRight);
233  // case Op::And:
234  // return std::visit(eval, mLeft) && std::visit(eval, mRight);
235  // case Op::Xor:
236  // return std::visit(eval, mLeft) ^ std::visit(eval, mRight);
237  // case Op::Just:
238  // return std::visit(eval, mLeft);
239  // }
240  // When we drop support for macOS 10.13
241  if (auto pval0 = std::get_if<OriginValueMatcher>(&mLeft)) {
242  auto dh = o2::header::get<header::DataHeader*>(d);
243  if (dh == nullptr) {
244  throw std::runtime_error("Cannot find DataHeader");
245  }
246  leftValue = pval0->match(*dh, context);
247  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mLeft)) {
248  auto dh = o2::header::get<header::DataHeader*>(d);
249  if (dh == nullptr) {
250  throw std::runtime_error("Cannot find DataHeader");
251  }
252  leftValue = pval1->match(*dh, context);
253  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mLeft)) {
254  auto dh = o2::header::get<header::DataHeader*>(d);
255  if (dh == nullptr) {
256  throw std::runtime_error("Cannot find DataHeader");
257  }
258  leftValue = pval2->match(*dh, context);
259  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mLeft)) {
260  leftValue = (*pval3)->match(d, context);
261  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mLeft)) {
262  leftValue = pval4->match();
263  } else if (auto pval5 = std::get_if<StartTimeValueMatcher>(&mLeft)) {
264  auto dph = o2::header::get<DataProcessingHeader*>(d);
265  if (dph == nullptr) {
266  throw std::runtime_error("Cannot find DataProcessingHeader");
267  }
268  leftValue = pval5->match(*dph, context);
269  } else {
270  throw std::runtime_error("Bad parsing tree");
271  }
272  // Common speedup.
273  if (mOp == Op::And && leftValue == false) {
274  return false;
275  }
276  if (mOp == Op::Or && leftValue == true) {
277  return true;
278  }
279  if (mOp == Op::Just) {
280  return leftValue;
281  }
282 
283  if (auto pval0 = std::get_if<OriginValueMatcher>(&mRight)) {
284  auto dh = o2::header::get<header::DataHeader*>(d);
285  rightValue = pval0->match(*dh, context);
286  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mRight)) {
287  auto dh = o2::header::get<header::DataHeader*>(d);
288  rightValue = pval1->match(*dh, context);
289  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mRight)) {
290  auto dh = o2::header::get<header::DataHeader*>(d);
291  rightValue = pval2->match(*dh, context);
292  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mRight)) {
293  rightValue = (*pval3)->match(d, context);
294  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mRight)) {
295  rightValue = pval4->match();
296  } else if (auto pval5 = std::get_if<StartTimeValueMatcher>(&mRight)) {
297  auto dph = o2::header::get<DataProcessingHeader*>(d);
298  rightValue = pval5->match(*dph, context);
299  }
300  // There are cases in which not having a rightValue might be legitimate,
301  // so we do not throw an exception.
302  switch (mOp) {
303  case Op::Or:
304  return leftValue || rightValue;
305  case Op::And:
306  return leftValue && rightValue;
307  case Op::Xor:
308  return leftValue ^ rightValue;
309  case Op::Just:
310  return leftValue;
311  }
312  throw std::runtime_error("Bad parsing tree");
313 };

315 bool DataDescriptorMatcher::operator==(DataDescriptorMatcher const& other) const
316 {
317  if (mOp != this->mOp) {
318  return false;
319  }
320 
321  bool leftValue = false;
322 
323  {
324  auto v1 = std::get_if<OriginValueMatcher>(&this->mLeft);
325  auto v2 = std::get_if<OriginValueMatcher>(&other.mLeft);
326  if (v1 && v2 && *v1 == *v2) {
327  leftValue = true;
328  }
329  }
330 
331  {
332  auto v1 = std::get_if<DescriptionValueMatcher>(&this->mLeft);
333  auto v2 = std::get_if<DescriptionValueMatcher>(&other.mLeft);
334  if (v1 && v2 && *v1 == *v2) {
335  leftValue = true;
336  }
337  }
338 
339  {
340  auto v1 = std::get_if<SubSpecificationTypeValueMatcher>(&this->mLeft);
341  auto v2 = std::get_if<SubSpecificationTypeValueMatcher>(&other.mLeft);
342  if (v1 && v2 && *v1 == *v2) {
343  leftValue = true;
344  }
345  }
346 
347  {
348  auto v1 = std::get_if<ConstantValueMatcher>(&this->mLeft);
349  auto v2 = std::get_if<ConstantValueMatcher>(&other.mLeft);
350  if (v1 && v2 && *v1 == *v2) {
351  leftValue = true;
352  }
353  }
354 
355  {
356  auto v1 = std::get_if<StartTimeValueMatcher>(&this->mLeft);
357  auto v2 = std::get_if<StartTimeValueMatcher>(&other.mLeft);
358  if (v1 && v2 && *v1 == *v2) {
359  leftValue = true;
360  }
361  }
362 
363  {
364  auto v1 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&this->mLeft);
365  auto v2 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&other.mLeft);
366  if (v1 && v2 && v1->get() && v2->get() && (**v1 == **v2)) {
367  leftValue = true;
368  }
369  }
370 
371  // Shortcut the fact that the left side is different.
372  if (leftValue == false) {
373  return false;
374  }
375 
376  if (mOp == Op::Just) {
377  return true;
378  }
379 
380  {
381  auto v1 = std::get_if<OriginValueMatcher>(&this->mRight);
382  auto v2 = std::get_if<OriginValueMatcher>(&other.mRight);
383  if (v1 && v2 && *v1 == *v2) {
384  return true;
385  }
386  }
387 
388  {
389  auto v1 = std::get_if<DescriptionValueMatcher>(&this->mRight);
390  auto v2 = std::get_if<DescriptionValueMatcher>(&other.mRight);
391  if (v1 && v2 && *v1 == *v2) {
392  return true;
393  }
394  }
395 
396  {
397  auto v1 = std::get_if<SubSpecificationTypeValueMatcher>(&this->mRight);
398  auto v2 = std::get_if<SubSpecificationTypeValueMatcher>(&other.mRight);
399  if (v1 && v2 && *v1 == *v2) {
400  return true;
401  }
402  }
403 
404  {
405  auto v1 = std::get_if<ConstantValueMatcher>(&this->mRight);
406  auto v2 = std::get_if<ConstantValueMatcher>(&other.mRight);
407  if (v1 && v2 && *v1 == *v2) {
408  return true;
409  }
410  }
411 
412  {
413  auto v1 = std::get_if<StartTimeValueMatcher>(&this->mRight);
414  auto v2 = std::get_if<StartTimeValueMatcher>(&other.mRight);
415  if (v1 && v2 && *v1 == *v2) {
416  return true;
417  }
418  }
419 
420  {
421  auto v1 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&this->mRight);
422  auto v2 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&other.mRight);
423  if (v1 && v2 && v1->get() && v2->get() && (**v1 == **v2)) {
424  return true;
425  }
426  }
427  // We alredy know the left side is true.
428  return false;
429 }

431 std::ostream& operator<<(std::ostream& os, DataDescriptorMatcher const& matcher)
432 {
433  auto edgeWalker = overloaded{
434  [&os](EdgeActions::EnterNode action) {
435  os << "(" << action.node->mOp;
436  if (action.node->mOp == DataDescriptorMatcher::Op::Just) {
437  return ChildAction::VisitLeft;
438  }
439  return ChildAction::VisitBoth;
440  },
441  [&os](EdgeActions::EnterLeft) { os << " "; },
442  [&os](EdgeActions::ExitLeft) { os << " "; },
443  [&os](EdgeActions::EnterRight) { os << " "; },
444  [&os](EdgeActions::ExitRight) { os << " "; },
445  [&os](EdgeActions::ExitNode) { os << ")"; },
446  [&os](auto) {}};
447  auto leafWalker = overloaded{
448  [&os](OriginValueMatcher const& origin) { os << "origin:" << origin; },
449  [&os](DescriptionValueMatcher const& description) { os << "description:" << description; },
450  [&os](SubSpecificationTypeValueMatcher const& subSpec) { os << "subSpec:" << subSpec; },
451  [&os](StartTimeValueMatcher const& startTime) { os << "startTime:" << startTime; },
452  [&os](ConstantValueMatcher const& constant) {},
453  [&os](auto t) { os << "not implemented " << typeid(decltype(t)).name(); }};
454  DataMatcherWalker::walk(matcher,
455  edgeWalker,
456  leafWalker);
457 
458  return os;
459 }
