111 std::vector<Digit> mergeDigits_map(const std::vector<Digit>& inputDigits, const std::vector<o2::MCCompLabel>& inputLabels)
112 {
113  int iter = 0;
114  int index = 0;
115  std::map<int, int> padidmap;
116  std::set<int> forRemoval;
117  std::vector<Digit> digits{inputDigits};
118  std::vector<o2::MCCompLabel> labels{inputLabels};
119 
120  for (auto& digit : digits) {
121  int count = 0;
122  int padid = digit.getPadID();
123  count = padidmap.count(padid);
124  if (count) {
125  std::pair<std::map<int, int>::iterator, std::map<int, int>::iterator> ret;
126 
127  ret = padidmap.equal_range(padid);
128  index = ret.first->second;
129  (digits.at(index)).setADC(digits.at(index).getADC() + digit.getADC());
130  forRemoval.emplace(iter);
131  } else {
132  padidmap.emplace(padid, iter);
133  }
134  ++iter;
135  }
136 
137  int rmcounts = 0;
138  for (auto& rmindex : forRemoval) {
139  digits.erase(digits.begin() + rmindex - rmcounts);
140  labels.erase(labels.begin() + rmindex - rmcounts);
141  ++rmcounts;
142  }
143  return digits;
144 }
