216  void
217  decodeTF(framework::ProcessingContext& pc, std::vector<o2::mch::Digit>& digits)
218  {
219  // get the input buffer
220  auto& inputs = pc.inputs();
221  DPLRawParser parser(inputs, o2::framework::select("TF:MCH/RAWDATA"));
222 
223  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
224  // retrieving RDH v4
225  auto const* rdh = it.get_if<o2::header::RAWDataHeaderV4>();
226  // retrieving the raw pointer of the page
227  auto const* raw = it.raw();
228  // size of payload
229  size_t payloadSize = it.size();
230 
231  if (payloadSize == 0) {
232  continue;
233  }
234 
235  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), sizeof(o2::header::RAWDataHeaderV4) + payloadSize);
236  decodeBuffer(buffer, digits);
237  }
238  }

240  //_________________________________________________________________________________________________
241  void decodeReadout(const o2::framework::DataRef& input, std::vector<o2::mch::Digit>& digits)
242  {
243  static int nFrame = 1;
244  // get the input buffer
245  if (input.spec->binding != "readout") {
246  return;
247  }
248 
249  const auto* header = o2::header::get<header::DataHeader*>(input.header);
250  if (!header) {
251  return;
252  }
253 
254  auto const* raw = input.payload;
255  // size of payload
256  size_t payloadSize = header->payloadSize;
257 
258  if (mPrint) {
259  std::cout << nFrame << " payloadSize=" << payloadSize << std::endl;
260  }
261  nFrame += 1;
262  if (payloadSize == 0) {
263  return;
264  }
265 
266  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), payloadSize);
267  decodeBuffer(buffer, digits);
268  }

270  //_________________________________________________________________________________________________
271  void run(framework::ProcessingContext& pc)
272  {
273  std::vector<o2::mch::Digit> digits;
274 
275  decodeTF(pc, digits);
276  for (auto&& input : pc.inputs()) {
277  if (input.spec->binding == "readout")
278  decodeReadout(input, digits);
279  }
280 
281  if (mPrint) {
282  for (auto d : digits) {
283  std::cout << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC() << " time " << d.getTime().sampaTime << std::endl;
284  }
285  }
286 
287  const size_t OUT_SIZE = sizeof(o2::mch::Digit) * digits.size();
288 
289  // send the output buffer via DPL
290  char* outbuffer = nullptr;
291  outbuffer = (char*)realloc(outbuffer, OUT_SIZE);
292  memcpy(outbuffer, digits.data(), OUT_SIZE);
293 
294  // create the output message
295  auto freefct = [](void* data, void*) { free(data); };
296  pc.outputs().adoptChunk(Output{"MCH", "DIGITS", 0}, outbuffer, OUT_SIZE, freefct, nullptr);
297  }
