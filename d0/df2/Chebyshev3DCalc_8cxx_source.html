205 Float_t Chebyshev3DCalc::evaluateDerivative(int dim, const Float_t* par) const
206 {
207  int ncfRC;
208  for (int id0 = mNumberOfRows; id0--;) {
209  int nCLoc = mNumberOfColumnsAtRow[id0]; // number of significant coefs on this row
210  if (!nCLoc) {
211  mTemporaryCoefficients1D[id0] = 0;
212  continue;
213  }
214  //
215  int col0 = mColumnAtRowBeginning[id0]; // beginning of local column in the 2D boundary matrix
216  for (int id1 = nCLoc; id1--;) {
217  int id = id1 + col0;
218  if (!(ncfRC = mCoefficientBound2D0[id])) {
219  mTemporaryCoefficients2D[id1] = 0;
220  continue;
221  }
222  if (dim == 2) {
223  mTemporaryCoefficients2D[id1] =
224  chebyshevEvaluation1Derivative(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC);
225  } else {
226  mTemporaryCoefficients2D[id1] = chebyshevEvaluation1D(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC);
227  }
228  }
229  if (dim == 1) {
230  mTemporaryCoefficients1D[id0] = chebyshevEvaluation1Derivative(par[1], mTemporaryCoefficients2D, nCLoc);
231  } else {
232  mTemporaryCoefficients1D[id0] = chebyshevEvaluation1D(par[1], mTemporaryCoefficients2D, nCLoc);
233  }
234  }
235  return (dim == 0) ? chebyshevEvaluation1Derivative(par[0], mTemporaryCoefficients1D, mNumberOfRows)
236  : chebyshevEvaluation1D(par[0], mTemporaryCoefficients1D, mNumberOfRows);
237 }

239 Float_t Chebyshev3DCalc::evaluateDerivative2(int dim1, int dim2, const Float_t* par) const
240 {
241  Bool_t same = dim1 == dim2;
242  int ncfRC;
243  for (int id0 = mNumberOfRows; id0--;) {
244  int nCLoc = mNumberOfColumnsAtRow[id0]; // number of significant coefs on this row
245  if (!nCLoc) {
246  mTemporaryCoefficients1D[id0] = 0;
247  continue;
248  }
249  int col0 = mColumnAtRowBeginning[id0]; // beginning of local column in the 2D boundary matrix
250  for (int id1 = nCLoc; id1--;) {
251  int id = id1 + col0;
252  if (!(ncfRC = mCoefficientBound2D0[id])) {
253  mTemporaryCoefficients2D[id1] = 0;
254  continue;
255  }
256  if (dim1 == 2 || dim2 == 2) {
257  mTemporaryCoefficients2D[id1] =
258  same ? chebyshevEvaluation1Derivative2(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC)
259  : chebyshevEvaluation1Derivative(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC);
260  } else {
261  mTemporaryCoefficients2D[id1] = chebyshevEvaluation1D(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC);
262  }
263  }
264  if (dim1 == 1 || dim2 == 1) {
265  mTemporaryCoefficients1D[id0] = same ? chebyshevEvaluation1Derivative2(par[1], mTemporaryCoefficients2D, nCLoc)
266  : chebyshevEvaluation1Derivative(par[1], mTemporaryCoefficients2D, nCLoc);
267  } else {
268  mTemporaryCoefficients1D[id0] = chebyshevEvaluation1D(par[1], mTemporaryCoefficients2D, nCLoc);
269  }
270  }
271  return (dim1 == 0 || dim2 == 0)
272  ? (same ? chebyshevEvaluation1Derivative2(par[0], mTemporaryCoefficients1D, mNumberOfRows)
273  : chebyshevEvaluation1Derivative(par[0], mTemporaryCoefficients1D, mNumberOfRows))
274  : chebyshevEvaluation1D(par[0], mTemporaryCoefficients1D, mNumberOfRows);
275 }

315 void Chebyshev3DCalc::loadData(FILE* stream)
316 {
317  if (!stream) {
318  Error("LoadData", "No stream provided.\nStop");
319  exit(1);
320  }
321  TString buffs;
322  Clear();
323  readLine(buffs, stream);
324 
325  if (!buffs.BeginsWith("START")) {
326  Error("LoadData", "Expected: \"START <fit_name>\", found \"%s\"\nStop\n", buffs.Data());
327  exit(1);
328  }
329 
330  if (buffs.First(' ') > 0) {
331  SetName(buffs.Data() + buffs.First(' ') + 1);
332  }
333 
334  readLine(buffs, stream); // NRows
335  mNumberOfRows = buffs.Atoi();
336 
337  if (mNumberOfRows < 0 || !buffs.IsDigit()) {
338  Error("LoadData", "Expected: '<number_of_rows>', found \"%s\"\nStop\n", buffs.Data());
339  exit(1);
340  }
341 
342  mNumberOfColumns = 0;
343  mNumberOfElementsBound2D = 0;
344  initializeRows(mNumberOfRows);
345 
346  for (int id0 = 0; id0 < mNumberOfRows; id0++) {
347  readLine(buffs, stream); // n.cols at this row
348  mNumberOfColumnsAtRow[id0] = buffs.Atoi();
349  mColumnAtRowBeginning[id0] = mNumberOfElementsBound2D; // begining of this row in 2D boundary surface
350  mNumberOfElementsBound2D += mNumberOfColumnsAtRow[id0];
351  if (mNumberOfColumns < mNumberOfColumnsAtRow[id0]) {
352  mNumberOfColumns = mNumberOfColumnsAtRow[id0];
353  }
354  }
355  initializeColumns(mNumberOfColumns);
356 
357  mNumberOfCoefficients = 0;
358  initializeElementBound2D(mNumberOfElementsBound2D);
359 
360  for (int i = 0; i < mNumberOfElementsBound2D; i++) {
361  readLine(buffs, stream); // n.coeffs at 3-d dimension for the given column/row
362  mCoefficientBound2D0[i] = buffs.Atoi();
363  mCoefficientBound2D1[i] = mNumberOfCoefficients;
364  mNumberOfCoefficients += mCoefficientBound2D0[i];
365  }
366 
367  initializeCoefficients(mNumberOfCoefficients);
368  for (int i = 0; i < mNumberOfCoefficients; i++) {
369  readLine(buffs, stream);
370  mCoefficients[i] = buffs.Atof();
371  }
372  // read precision
373  readLine(buffs, stream);
374  mPrecision = buffs.Atof();
375 
376  // check end_of_data record
377  readLine(buffs, stream);
378  if (!buffs.BeginsWith("END") || !buffs.Contains(GetName())) {
379  Error("LoadData", "Expected \"END %s\", found \"%s\".\nStop\n", GetName(), buffs.Data());
380  exit(1);
381  }
382 }

469 Float_t Chebyshev3DCalc::chebyshevEvaluation1Derivative(Float_t x, const Float_t* array, int ncf)
470 {
471  if (--ncf < 1) {
472  return 0;
473  }
474  Float_t b0, b1, b2;
475  Float_t x2 = x + x;
476  b1 = b2 = 0;
477  float dcf0 = 0, dcf1, dcf2 = 0;
478  b0 = dcf1 = 2 * ncf * array[ncf];
479  if (!(--ncf)) {
480  return b0 / 2;
481  }
482 
483  for (int i = ncf; i--;) {
484  b2 = b1;
485  b1 = b0;
486  dcf0 = dcf2 + 2 * (i + 1) * array[i + 1];
487  b0 = dcf0 + x2 * b1 - b2;
488  dcf2 = dcf1;
489  dcf1 = dcf0;
490  }
491 
492  return b0 - x * b1 - dcf0 / 2;
493 }

495 Float_t Chebyshev3DCalc::chebyshevEvaluation1Derivative2(Float_t x, const Float_t* array, int ncf)
496 {
497  if (--ncf < 2) {
498  return 0;
499  }
500  Float_t b0, b1, b2;
501  Float_t x2 = x + x;
502  b1 = b2 = 0;
503  float dcf0 = 0, dcf1 = 0, dcf2 = 0;
504  float ddcf0 = 0, ddcf1, ddcf2 = 0;
505 
506  dcf2 = 2 * ncf * array[ncf];
507  --ncf;
508 
509  dcf1 = 2 * ncf * array[ncf];
510  b0 = ddcf1 = 2 * ncf * dcf2;
511 
512  if (!(--ncf)) {
513  return b0 / 2;
514  }
515 
516  for (int i = ncf; i--;) {
517  b2 = b1;
518  b1 = b0;
519  dcf0 = dcf2 + 2 * (i + 1) * array[i + 1];
520  ddcf0 = ddcf2 + 2 * (i + 1) * dcf1;
521  b0 = ddcf0 + x2 * b1 - b2;
522 
523  ddcf2 = ddcf1;
524  ddcf1 = ddcf0;
525 
526  dcf2 = dcf1;
527  dcf1 = dcf0;
528  }
529  return b0 - x * b1 - ddcf0 / 2;
530 }
