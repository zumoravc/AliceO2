34  DataRef ref;
35  TMessage* tm = new TMessage(kMESS_OBJECT);
36  tm->WriteObject(obj);
37 
38  ref.payload = tm->Buffer();
39 
40  auto dh = new o2::header::DataHeader{};
41  dh->payloadSerializationMethod = o2::header::gSerializationMethodROOT;
42  dh->payloadSize = tm->BufferSize();

44 
45  return ref;
46  };
