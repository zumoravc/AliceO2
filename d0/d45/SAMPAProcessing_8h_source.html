137 template <typename T>
138 inline T SAMPAProcessing::getADCvalue(T nElectrons) const
139 {
140  T conversion = mEleParam->ElectronCharge * 1.e15 * mEleParam->ChipGain * mEleParam->ADCsaturation /
141  mEleParam->ADCdynamicRange; // 1E-15 is to convert Coulomb in fC
142  return nElectrons * conversion;
143 }

145 template <DigitzationMode MODE>
146 inline float SAMPAProcessing::makeSignal(float ADCcounts, const int sector, const int globalPadInSector, const float commonMode,
147  float& pedestal, float& noise)
148 {
149  float signal = ADCcounts;
150  pedestal = getPedestal(sector, globalPadInSector);
151  noise = getNoise(sector, globalPadInSector);
152  switch (MODE) {
153  case DigitzationMode::FullMode: {
154  signal -= commonMode;
155  signal += noise;
156  signal += pedestal;
157  return getADCSaturation(signal);
158  break;
159  }
160  case DigitzationMode::SubtractPedestal: {
161  signal -= commonMode;
162  signal += noise;
163  signal += pedestal;
164  float signalSubtractPedestal = getADCSaturation(signal) - pedestal;
165  return signalSubtractPedestal;
166  break;
167  }
168  case DigitzationMode::NoSaturation: {
169  signal -= commonMode;
170  signal += noise;
171  signal += pedestal;
172  return signal;
173  break;
174  }
175  case DigitzationMode::PropagateADC: {
176  return signal;
177  break;
178  }
179  }
180  return signal;
181 }

183 inline float SAMPAProcessing::getADCSaturation(const float signal) const
184 {
185  const float adcSaturation = mEleParam->ADCsaturation;
186  if (signal > adcSaturation - 1)
187  return adcSaturation - 1;
188  return signal;

191 template <typename T>
192 inline T SAMPAProcessing::getGamma4(T time, T startTime, T ADC) const
193 {
194  const auto tmp0 = (time - startTime) / mEleParam->PeakingTime;
195  const auto cond = (tmp0 > 0);
196  T tmp{};
197  if constexpr (std::is_floating_point_v<T>) {
198  if (!cond) {
199  return T{};
200  }
201  tmp = tmp0;
202  } else {
203  tmp(cond) = tmp0;
204  }
205  const auto tmp2 = tmp * tmp;
206  return 55.f * ADC * std::exp(-4.f * tmp) * tmp2 * tmp2;
207 }
