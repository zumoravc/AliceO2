139 template <typename T>
140 GPUd() int GPUTRDTrack_t<T>::GetNmissingConsecLayers(int iLayer) const
141 {
142  //------------------------------------------------------------------
143  // returns number of consecutive layers in which the track was
144  // inside the deadzone up to (and including) the given layer
145  //------------------------------------------------------------------
146  int res = 0;
147  while (!mIsFindable[iLayer]) {
148  ++res;
149  --iLayer;
150  if (iLayer < 0) {
151  break;
152  }
153  }
154  return res;
155 }

179 template <typename T>
180 GPUd() void GPUTRDTrack_t<T>::ConvertFrom(const GPUTRDTrackDataRecord& t)
181 {
182  //------------------------------------------------------------------
183  // convert from GPU structure
184  //------------------------------------------------------------------
185  T::set(t.fX, t.mAlpha, &(t.fY), t.fC);
186  SetTPCtrackId(t.fTPCTrackID);
187  mChi2 = 0.f;
188  mMass = 0.13957f;
189  mLabel = -1;
190  mNTracklets = 0;
191  mNMissingConsecLayers = 0;
192  mLabelOffline = -1;
193  mIsStopped = false;
194  for (int iLayer = 0; iLayer < kNLayers; iLayer++) {
195  mAttachedTracklets[iLayer] = t.fAttachedTracklets[iLayer];
196  mIsFindable[iLayer] = 0;
197  if (mAttachedTracklets[iLayer] >= 0) {
198  mNTracklets++;
199  }
200  }
201  for (int j = 0; j < 4; ++j) {
202  mNTrackletsOffline[j] = 0;
203  }
204 }
