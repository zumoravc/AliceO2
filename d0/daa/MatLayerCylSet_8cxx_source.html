33 void MatLayerCylSet::addLayer(float rmin, float rmax, float zmax, float dz, float drphi)
34 {
35  // add new layer checking for overlaps
36  assert(mConstructionMask != Constructed);
37  assert(rmin < rmax && zmax > 0 && dz > 0 && drphi > 0);
38  mConstructionMask = InProgress;
39  int nlr = getNLayers();
40  if (!nlr) {
41  // book local storage
42  auto sz = sizeof(MatLayerCylSetLayout);
43  o2::gpu::resizeArray(mFlatBufferContainer, 0, sz);
44  mFlatBufferPtr = mFlatBufferContainer;
45  mFlatBufferSize = sz;
46  //--------------????
47  get()->mRMin = 1.e99;
48  get()->mRMax = 0.;
49  }
50 
51  for (int il = 0; il < nlr; il++) {
52  const auto& lr = getLayer(il);
53  if (lr.getRMax() > rmin && rmax > lr.getRMin()) {
54  LOG(FATAL) << "new layer overlaps with layer " << il;
55  }
56  }
57  auto* oldLayers = o2::gpu::resizeArray(get()->mLayers, nlr, nlr + 1);
58  // dynamyc buffers of old layers were used in new ones, detach them
59  for (int i = nlr; i--;) {
60  oldLayers[i].clearInternalBufferPtr();
61  }
62  delete[] oldLayers;
63  get()->mLayers[nlr].initSegmentation(rmin, rmax, zmax, dz, drphi);
64  get()->mNLayers++;
65  get()->mRMin = get()->mRMin > rmin ? rmin : get()->mRMin;
66  get()->mRMax = get()->mRMax < rmax ? rmax : get()->mRMax;
67  get()->mZMax = get()->mZMax < zmax ? zmax : get()->mZMax;
68  get()->mRMin2 = get()->mRMin * get()->mRMin;
69  get()->mRMax2 = get()->mRMax * get()->mRMax;
70 }

73 void MatLayerCylSet::populateFromTGeo(int ntrPerCell)
74 {
76  assert(mConstructionMask == InProgress);
77 
78  int nlr = getNLayers();
79  if (!nlr) {
80  LOG(ERROR) << "The LUT is not yet initialized";
81  return;
82  }
83  if (get()->mR2Intervals) {
84  LOG(ERROR) << "The LUT is already populated";
85  return;
86  }
87  for (int i = 0; i < nlr; i++) {
88  printf("Populating with %d trials Lr %3d ", ntrPerCell, i);
89  get()->mLayers[i].print();
90  get()->mLayers[i].populateFromTGeo(ntrPerCell);
91  }
92  // build layer search structures
93  int nR2Int = 2 * (nlr + 1);
94  o2::gpu::resizeArray(get()->mR2Intervals, 0, nR2Int);
95  o2::gpu::resizeArray(get()->mInterval2LrID, 0, nR2Int);
96  get()->mR2Intervals[0] = get()->mRMin2;
97  get()->mR2Intervals[1] = get()->mRMax2;
98  get()->mInterval2LrID[0] = 0;
99  auto& nRIntervals = get()->mNRIntervals;
100  nRIntervals = 1;
101 
102  for (int i = 1; i < nlr; i++) {
103  const auto& lr = getLayer(i);
104  if (std::sqrt(lr.getRMin2()) > std::sqrt(get()->mR2Intervals[nRIntervals] + Ray::Tiny)) {
105  // register gap
106  get()->mInterval2LrID[nRIntervals] = -1;
107  get()->mR2Intervals[++nRIntervals] = lr.getRMin2();
108  }
109  get()->mInterval2LrID[nRIntervals] = i;
110  get()->mR2Intervals[++nRIntervals] = lr.getRMax2();
111  }
112  delete[] o2::gpu::resizeArray(get()->mInterval2LrID, nR2Int, nRIntervals); // rebook with precise size
113  delete[] o2::gpu::resizeArray(get()->mR2Intervals, nR2Int, ++nRIntervals); // rebook with precise size
114  //
115 }

118 void MatLayerCylSet::dumpToTree(const std::string outName) const
119 {
121 
122  o2::utils::TreeStreamRedirector dump(outName.data(), "recreate");
123  for (int i = 0; i < getNLayers(); i++) {
124  const auto& lr = getLayer(i);
125  float r = 0.5 * (lr.getRMin() + lr.getRMax());
126  // per cell dump
127  int nphib = lr.getNPhiBins();
128  for (int ip = 0; ip < nphib; ip++) {
129  float phi = 0.5 * (lr.getPhiBinMin(ip) + lr.getPhiBinMax(ip));
130  float sn, cs;
131  int ips = lr.phiBin2Slice(ip);
132  char merge = 0; // not mergeable
133  if (ip + 1 < nphib) {
134  int ips1 = lr.phiBin2Slice(ip + 1);
135  merge = ips == ips1 ? -1 : lr.canMergePhiSlices(ips, ips1); // -1 for already merged
136  } else
137  merge = -2; // last one
138  o2::utils::sincosf(phi, sn, cs);
139  float x = r * cs, y = r * sn;
140  for (int iz = 0; iz < lr.getNZBins(); iz++) {
141  float z = 0.5 * (lr.getZBinMin(iz) + lr.getZBinMax(iz));
142  auto cell = lr.getCellPhiBin(ip, iz);
143  dump << "cell"
144  << "ilr=" << i << "r=" << r << "phi=" << phi << "x=" << x << "y=" << y << "z=" << z << "ip=" << ip << "ips=" << ips << "iz=" << iz
145  << "mrgnxt=" << merge << "val=" << cell << "\n";
146  }
147  }
148  //
149  // statistics per layer
150  MatCell mean, rms;
151  lr.getMeanRMS(mean, rms);
152  dump << "lay"
153  << "ilr=" << i << "r=" << r << "mean=" << mean << "rms=" << rms << "\n";
154  }
155 }

242 GPUd() MatBudget MatLayerCylSet::getMatBudget(float x0, float y0, float z0, float x1, float y1, float z1) const
243 {
244  // get material budget traversed on the line between point0 and point1
245  MatBudget rval;
246  Ray ray(x0, y0, z0, x1, y1, z1);
247  short lmin, lmax; // get innermost and outermost relevant layer
248  if (!getLayersRange(ray, lmin, lmax)) {
249  return rval;
250  }
251  short lrID = lmax;
252  while (lrID >= lmin) { // go from outside to inside
253  const auto& lr = getLayer(lrID);
254  int nc = ray.crossLayer(lr);
255  for (int ic = nc; ic--;) {
256  float cross1, cross2;
257  ray.getCrossParams(ic, cross1, cross2); // tmax,tmin of crossing the layer
258  auto phi0 = ray.getPhi(cross1), phi1 = ray.getPhi(cross2), dPhi = phi0 - phi1;
259  auto phiID = lr.getPhiSliceID(phi0), phiIDLast = lr.getPhiSliceID(phi1);
260  // account for eventual wrapping around 0
261  if (dPhi > 0.f) {
262  if (dPhi > o2::constants::math::PI) { // wraps around phi=0
263  phiIDLast += lr.getNPhiSlices();
264  }
265  } else {
266  if (dPhi < -o2::constants::math::PI) { // wraps around phi=0
267  phiID += lr.getNPhiSlices();
268  }
269  }
270  int stepPhiID = phiID > phiIDLast ? -1 : 1;
271  bool checkMorePhi = true;
272  auto tStartPhi = cross1, tEndPhi = 0.f;
273  do {
274  // get the path in the current phi slice
275  if (phiID == phiIDLast) {
276  tEndPhi = cross2;
277  checkMorePhi = false;
278  } else { // last phi slice still not reached
279  tEndPhi = ray.crossRadial(lr, (stepPhiID > 0 ? phiID + 1 : phiID) % lr.getNPhiSlices());
280  }
281  auto zID = lr.getZBinID(ray.getZ(tStartPhi));
282  auto zIDLast = lr.getZBinID(ray.getZ(tEndPhi));
283  // check if Zbins are crossed
284 
285 #ifdef _DBG_LOC_
286  printf("-- Zdiff (%3d : %3d) mode: t: %+e %+e\n", zID, zIDLast, tStartPhi, tEndPhi);
287 #endif
288 
289  if (zID != zIDLast) {
290  auto stepZID = zID < zIDLast ? 1 : -1;
291  bool checkMoreZ = true;
292  auto tStartZ = tStartPhi, tEndZ = 0.f;
293  do {
294  if (zID == zIDLast) {
295  tEndZ = tEndPhi;
296  checkMoreZ = false;
297  } else {
298  tEndZ = ray.crossZ(lr.getZBinMin(stepZID > 0 ? zID + 1 : zID));
299  }
300  // account materials of this step
301  float step = tEndZ - tStartZ; // the real step is ray.getDist(tEnd-tStart), will rescale all later
302  const auto& cell = lr.getCell(phiID, zID);
303  rval.meanRho += cell.meanRho * step;
304  rval.meanX2X0 += cell.meanX2X0 * step;
305  rval.length += step;
306 
307 #ifdef _DBG_LOC_
308  float pos0[3] = {ray.getPos(tStartZ, 0), ray.getPos(tStartZ, 1), ray.getPos(tStartZ, 2)};
309  float pos1[3] = {ray.getPos(tEndZ, 0), ray.getPos(tEndZ, 1), ray.getPos(tEndZ, 2)};
310  printf(
311  "Lr#%3d / cross#%d : account %f<t<%f at phiSlice %d | Zbin: %3d (%3d) |[%+e %+e +%e]:[%+e %+e %+e] "
312  "Step: %.3e StrpCor: %.3e\n",
313  lrID, ic, tEndZ, tStartZ, phiID % lr.getNPhiSlices(), zID, zIDLast,
314  pos0[0], pos0[1], pos0[2], pos1[0], pos1[1], pos1[2], step, ray.getDist(step));
315 #endif
316 
317  tStartZ = tEndZ;
318  zID += stepZID;
319  } while (checkMoreZ);
320  } else {
321  float step = tEndPhi - tStartPhi; // the real step is |ray.getDist(tEnd-tStart)|, will rescale all later
322  const auto& cell = lr.getCell(phiID, zID);
323  rval.meanRho += cell.meanRho * step;
324  rval.meanX2X0 += cell.meanX2X0 * step;
325  rval.length += step;
326 
327 #ifdef _DBG_LOC_
328  float pos0[3] = {ray.getPos(tStartPhi, 0), ray.getPos(tStartPhi, 1), ray.getPos(tStartPhi, 2)};
329  float pos1[3] = {ray.getPos(tEndPhi, 0), ray.getPos(tEndPhi, 1), ray.getPos(tEndPhi, 2)};
330  printf(
331  "Lr#%3d / cross#%d : account %f<t<%f at phiSlice %d | Zbin: %3d ----- |[%+e %+e +%e]:[%+e %+e %+e]"
332  "Step: %.3e StrpCor: %.3e\n",
333  lrID, ic, tEndPhi, tStartPhi, phiID % lr.getNPhiSlices(), zID,
334  pos0[0], pos0[1], pos0[2], pos1[0], pos1[1], pos1[2], step, ray.getDist(step));
335 #endif
336  }
337  //
338  tStartPhi = tEndPhi;
339  phiID += stepPhiID;
340 
341  } while (checkMorePhi);
342  }
343  lrID--;
344  } // loop over layers
345 
346  if (rval.length != 0.f) {
347  rval.meanRho /= rval.length; // average
348  float norm = (rval.length < 0.f) ? -ray.getDist() : ray.getDist(); // normalize
349  rval.meanX2X0 *= norm;
350  rval.length *= norm;
351  }
352 #ifdef _DBG_LOC_
353  printf("<rho> = %e, x2X0 = %e | step = %e\n", rval.meanRho, rval.meanX2X0, rval.length);
354 #endif
355  return rval;
356 }

359 GPUd() bool MatLayerCylSet::getLayersRange(const Ray& ray, short& lmin, short& lmax) const
360 {
361  // get range of layers corresponding to rmin/rmax
362  //
363  lmin = lmax = -1;
364  float rmin2, rmax2;
365  ray.getMinMaxR2(rmin2, rmax2);
366 
367  if (rmin2 >= getRMax2() || rmax2 <= getRMin2()) {
368  return false;
369  }
370  int lmxInt, lmnInt;
371  lmxInt = rmax2 < getRMax2() ? searchSegment(rmax2, 0) : get()->mNRIntervals - 2;
372  lmnInt = rmin2 >= getRMin2() ? searchSegment(rmin2, 0, lmxInt + 1) : 0;
373  const auto* interval2LrID = get()->mInterval2LrID;
374  lmax = interval2LrID[lmxInt];
375  lmin = interval2LrID[lmnInt];
376  // make sure lmnInt and/or lmxInt are not in the gap
377  if (lmax < 0) {
378  lmax = interval2LrID[--lmxInt]; // rmax2 is in the gap, take highest layer below rmax2
379  }
380  if (lmin < 0) {
381  lmin = interval2LrID[++lmnInt]; // rmin2 is in the gap, take lowest layer above rmin2
382  }
383  return lmin <= lmax; // valid if both are not in the same gap
384 }

386 GPUd() int MatLayerCylSet::searchSegment(float val, int low, int high) const
387 {
389  if (low < 0) {
390  low = 0;
391  }
392  if (high < 0) {
393  high = get()->mNRIntervals;
394  }
395  int mid = (low + high) >> 1;
396  const auto* r2Intervals = get()->mR2Intervals;
397  while (mid != low) {
398  if (val < r2Intervals[mid]) {
399  high = mid;
400  } else {
401  low = mid;
402  }
403  mid = (low + high) >> 1;
404  }
405  return mid;
406 }

410 void MatLayerCylSet::flatten()
411 {
412  // make object flat: move all content to single internally allocated buffer
413  assert(mConstructionMask == InProgress);
414 
415  int sz = estimateFlatBufferSize();
416  // create new internal buffer with total size and copy data
417  delete[] o2::gpu::resizeArray(mFlatBufferContainer, mFlatBufferSize, sz);
418  mFlatBufferPtr = mFlatBufferContainer;
419  mFlatBufferSize = sz;
420  int nLr = getNLayers();
421 
422  auto offs = alignSize(sizeof(MatLayerCylSetLayout), getBufferAlignmentBytes()); // account for the alignment
423  // move array of layer pointers to the flat array
424  auto* oldLayers = o2::gpu::resizeArray(get()->mLayers, nLr, nLr, (MatLayerCyl*)(mFlatBufferPtr + offs));
425  // dynamyc buffers of old layers were used in new ones, detach them
426  for (int i = nLr; i--;) {
427  oldLayers[i].clearInternalBufferPtr();
428  }
429  delete[] oldLayers;
430  offs = alignSize(offs + nLr * sizeof(MatLayerCyl), MatLayerCyl::getClassAlignmentBytes()); // account for the alignment
431 
432  // move array of R2 boundaries to the flat array
433  delete[] o2::gpu::resizeArray(get()->mR2Intervals, nLr + 1, nLr + 1, (float*)(mFlatBufferPtr + offs));
434  offs = alignSize(offs + (nLr + 1) * sizeof(float), getBufferAlignmentBytes()); // account for the alignment
435 
436  // move array of R2 boundaries to the flat array
437  delete[] o2::gpu::resizeArray(get()->mInterval2LrID, nLr, nLr, (int*)(mFlatBufferPtr + offs));
438  offs = alignSize(offs + nLr * sizeof(int), getBufferAlignmentBytes()); // account for the alignment
439 
440  for (int il = 0; il < nLr; il++) {
441  MatLayerCyl& lr = get()->mLayers[il];
442  lr.flatten(mFlatBufferPtr + offs);
443  offs = alignSize(offs + lr.getFlatBufferSize(), getBufferAlignmentBytes()); // account for the alignment
444  }
445  mConstructionMask = Constructed;
446 }
