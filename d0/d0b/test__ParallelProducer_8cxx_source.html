52 // This is a simple consumer / producer workflow where both are
53 // stateful, i.e. they have context which comes from their initialization.
54 WorkflowSpec defineDataProcessing(ConfigContext const& context)
55 {
56  // This is an example of how we can parallelize by subSpec.
57  // templatedProducer will be instanciated 32 times and the lambda function
58  // passed to the parallel statement will be applied to each one of the
59  // instances in order to modify it. Parallel will also make sure the name of
60  // the instance is amended from "some-producer" to "some-producer-<index>".
61  auto jobs = context.options().get<int>("jobs");
62  WorkflowSpec workflow = parallel(templateProducer(), jobs, [](DataProcessorSpec& spec, size_t index) {
63  DataSpecUtils::updateMatchingSubspec(spec.outputs[0], index);
64  });
65  workflow.push_back(DataProcessorSpec{
66  "merger",
67  mergeInputs(InputSpec{"x", "TST", "A", 0, Lifetime::Timeframe},
68  jobs,
69  [](InputSpec& input, size_t index) {
70  DataSpecUtils::updateMatchingSubspec(input, index);
71  }),
72  {},
73  AlgorithmSpec{[](InitContext& setup) {
74  return [](ProcessingContext& ctx) {
75  // Create a single output.
76  LOG(DEBUG) << "Invoked" << std::endl;
77  };
78  }}});
79 
80  return workflow;
81 }
