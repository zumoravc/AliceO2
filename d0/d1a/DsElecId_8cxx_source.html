29 uint32_t encode(const DsElecId& id)
30 {

36 std::optional<DsElecId> decodeDsElecId(uint32_t code)
37 {
38  uint16_t solarId = (code & 0xFFFF0000) >> 16; // 11 (LSB) bits used
39 
40  uint8_t groupId = (code & 0xFF00) >> 8; // 3 (LSB) bits used
41 
42  uint8_t index = (code & 0xFF) >> 0; // 3 (LSB) bits used
43 
44  if (groupId > 7) {
45  return std::nullopt;
46  }
47  if (index > 4) {
48  return std::nullopt;
49  }
50  return DsElecId(solarId, groupId, index);
51 }

53 std::optional<DsElecId> decodeDsElecId(std::string rep)
54 {
55  std::istringstream is(rep);
56  std::string line;
57  std::vector<std::string> tokens;
58  while (getline(is, line, '-')) {
59  tokens.emplace_back(line);
60  }
61  if (tokens.size() < 3) {
62  // not a valid representation of a DsElecId
63  return std::nullopt;
64  }
65  if (tokens[0].empty() || tokens[0][0] != 'S') {
66  // token[0] is not a valid representation of a solarId
67  return std::nullopt;
68  }
69  if (tokens[1].empty() || tokens[1][0] != 'J') {
70  // token[1] is not a valid representation of a groupId
71  return std::nullopt;
72  }
73  if (tokens[2].size() < 3 || tokens[2][0] != 'D' || tokens[2][1] != 'S') {
74  // token is not a valid representation of a DS
75  return std::nullopt;
76  }
77  uint16_t solarId = std::atoi(tokens[0].substr(1).c_str());
78  uint8_t groupId = std::atoi(tokens[1].substr(1).c_str());
79  uint8_t index = std::atoi(tokens[2].substr(2).c_str());
80  return DsElecId(solarId, groupId, index);
81 }

83 std::optional<uint8_t> decodeChannelId(std::string rep)
84 {
85  auto dsElecId = decodeDsElecId(rep);
86  if (!dsElecId.has_value()) {
87  // must be a valid dsElecId
88  return std::nullopt;
89  }
90  std::istringstream is(rep);
91  std::string line;
92  std::vector<std::string> tokens;
93  while (getline(is, line, '-')) {
94  tokens.emplace_back(line);
95  }
96  if (tokens.size() < 4) {
97  // not a valid representation of a {DsElecId,ChannelId}
98  return std::nullopt;
99  }
100  if (tokens[3].size() < 3 || tokens[3][0] != 'C' || tokens[3][1] != 'H') {
101  // token[3] is not a valid representation of a CH
102  return std::nullopt;
103  }
104  auto chId = std::atoi(tokens[3].substr(2).c_str());
105  if (chId >= 0 && chId <= 63) {
106  return chId;
107  }
108  return std::nullopt;
109 }
