51  auto makeBoxGen = [](int pdgid, int mult, double etamin, double etamax, double pmin, double pmax, double phimin, double phimax, bool debug = false) {
52  auto gen = new FairBoxGenerator(pdgid, mult);
53  gen->SetEtaRange(etamin, etamax);
54  gen->SetPRange(pmin, pmax);
55  gen->SetPhiRange(phimin, phimax);
56  gen->SetDebug(debug);
57  return gen;
58  };

60 #ifdef GENERATORS_WITH_PYTHIA8
61  auto makePythia8Gen = [](std::string& config) {
62  auto gen = new o2::eventgen::GeneratorPythia8();
63  LOG(INFO) << "Reading \'Pythia8\' base configuration: " << config << std::endl;
64  gen->readFile(config);
65  auto& param = GeneratorPythia8Param::Instance();
66  LOG(INFO) << "Init \'Pythia8\' generator with following parameters";
67  LOG(INFO) << param;
68  gen->setConfig(param.config);
69  gen->setHooksFileName(param.hooksFileName);
70  gen->setHooksFuncName(param.hooksFuncName);
71  return gen;

77  auto genconfig = conf.getGenerator();

100  } else if (genconfig.compare("fwrootino") == 0) {
101  // a simple "box" generator for forward rootinos
102  LOG(INFO) << "Init box forward rootinos generator";
103  auto boxGen = makeBoxGen(0, 1, -4, -2.5, 1, 5, 0, 360);
104  primGen->AddGenerator(boxGen);
105  } else if (genconfig.compare("zdcgen") == 0) {
106  // a simple "box" generator for forward neutrons
107  LOG(INFO) << "Init box forward/backward zdc generator";
108  auto boxGenC = makeBoxGen(2112 /*neutrons*/, 1, -8, -9999, 500, 1000, 0., 360.);
109  auto boxGenA = makeBoxGen(2112 /*neutrons*/, 1, 8, 9999, 500, 1000, 0., 360.);
110  primGen->AddGenerator(boxGenC);
111  primGen->AddGenerator(boxGenA);
112  } else if (genconfig.compare("emcgenele") == 0) {
113  // box generator with one electron per event
114  LOG(INFO) << "Init box generator for electrons in EMCAL";

116  auto elecgen = makeBoxGen(11, 1, -0.67, 0.67, 15, 15, 80, 187);
117  primGen->AddGenerator(elecgen);
118  } else if (genconfig.compare("emcgenphoton") == 0) {
119  LOG(INFO) << "Init box generator for photons in EMCAL";

121  primGen->AddGenerator(photongen);
122  } else if (genconfig.compare("fddgen") == 0) {
123  LOG(INFO) << "Init box FDD generator";
124  auto boxGenFDC = makeBoxGen(13, 1000, -7, -4.8, 10, 500, 0, 360.);
125  auto boxGenFDA = makeBoxGen(13, 1000, 4.9, 6.3, 10, 500, 0., 360);
126  primGen->AddGenerator(boxGenFDA);
127  primGen->AddGenerator(boxGenFDC);
128  } else if (genconfig.compare("extkin") == 0) {
129  // external kinematics

215  Trigger trigger = nullptr;
216  DeepTrigger deeptrigger = nullptr;
217 
218  auto trgconfig = conf.getTrigger();
219  if (trgconfig.empty()) {
220  return;
221  } else if (trgconfig.compare("particle") == 0) {
222  trigger = TriggerParticle(TriggerParticleParam::Instance());
223  } else if (trgconfig.compare("external") == 0) {
224  // external trigger via configuration macro
225  auto external_trigger_filename = conf.getExtTriggerFileName();
226  auto external_trigger_func = conf.getExtTriggerFuncName();
227  trigger = GetFromMacro<o2::eventgen::Trigger>(external_trigger_filename, external_trigger_func, "o2::eventgen::Trigger", "trigger");
228  if (!trigger) {
229  LOG(INFO) << "Trying to retrieve a \'o2::eventgen::DeepTrigger\' type" << std::endl;
230  deeptrigger = GetFromMacro<o2::eventgen::DeepTrigger>(external_trigger_filename, external_trigger_func, "o2::eventgen::DeepTrigger", "deeptrigger");
231  }
232  if (!trigger && !deeptrigger)
233  LOG(FATAL) << "Failed to retrieve \'external trigger\': problem with configuration ";
234  } else {
235  LOG(FATAL) << "Invalid trigger";
236  }
237 
239  auto generators = primGen->GetListOfGenerators();
240  for (int igen = 0; igen < generators->GetEntries(); ++igen) {
241  auto generator = dynamic_cast<o2::eventgen::Generator*>(generators->At(igen));
242  if (!generator) {
243  LOG(FATAL) << "request to add a trigger to an unsupported generator";
244  return;
245  }
246  generator->setTriggerMode(o2::eventgen::Generator::kTriggerOR);
247  if (trigger)
248  generator->addTrigger(trigger);
249  if (deeptrigger)
250  generator->addDeepTrigger(deeptrigger);
251  }
