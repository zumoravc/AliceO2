51 void GPUTPCCompression::SetPointersCompressedClusters(void*& mem, T& c, unsigned int nClA, unsigned int nTr, unsigned int nClU, bool reducedClA)
52 {
53  computePointerWithAlignment(mem, c.qTotU, nClU);
54  computePointerWithAlignment(mem, c.qMaxU, nClU);
55  computePointerWithAlignment(mem, c.flagsU, nClU);
56  computePointerWithAlignment(mem, c.padDiffU, nClU);
57  computePointerWithAlignment(mem, c.timeDiffU, nClU);
58  computePointerWithAlignment(mem, c.sigmaPadU, nClU);
59  computePointerWithAlignment(mem, c.sigmaTimeU, nClU);
60  computePointerWithAlignment(mem, c.nSliceRowClusters, GPUCA_ROW_COUNT * NSLICES);
61 
62  unsigned int nClAreduced = reducedClA ? nClA - nTr : nClA;
63 
64  if (!(mRec->GetParam().rec.tpcCompressionModes & GPUSettings::CompressionTrackModel)) {
65  return; // Track model disabled, do not allocate memory
66  }
67  computePointerWithAlignment(mem, c.qTotA, nClA);
68  computePointerWithAlignment(mem, c.qMaxA, nClA);
69  computePointerWithAlignment(mem, c.flagsA, nClA);
70  computePointerWithAlignment(mem, c.rowDiffA, nClAreduced);
71  computePointerWithAlignment(mem, c.sliceLegDiffA, nClAreduced);
72  computePointerWithAlignment(mem, c.padResA, nClAreduced);
73  computePointerWithAlignment(mem, c.timeResA, nClAreduced);
74  computePointerWithAlignment(mem, c.sigmaPadA, nClA);
75  computePointerWithAlignment(mem, c.sigmaTimeA, nClA);
76 
77  computePointerWithAlignment(mem, c.qPtA, nTr);
78  computePointerWithAlignment(mem, c.rowA, nTr);
79  computePointerWithAlignment(mem, c.sliceA, nTr);
80  computePointerWithAlignment(mem, c.timeA, nTr);
81  computePointerWithAlignment(mem, c.padA, nTr);
82 
83  computePointerWithAlignment(mem, c.nTrackClusters, nTr);
84 }
