58 std::tuple<int, int> CaloRawFitter::selectSubarray(const gsl::span<double> data, short maxindex, int cut) const
59 {
60 
61  int first(0), last(0);
62  int tmpfirst = maxindex;
63  int tmplast = maxindex;
64  double prevFirst = data[maxindex];
65  double prevLast = data[maxindex];
66  bool firstJump = false;
67  bool lastJump = false;
68 
69  while ((tmpfirst >= 0) && (data[tmpfirst] >= cut) && (!firstJump)) {
70  // jump check:
71  if (tmpfirst != maxindex) { // neighbor to maxindex can share peak with maxindex
72  if (data[tmpfirst] >= prevFirst) {
73  firstJump = true;
74  }
75  }
76  prevFirst = data[tmpfirst];
77  tmpfirst--;
78  }
79 
80  while ((tmplast < data.size()) && (data[tmplast] >= cut) && (!lastJump)) {
81  // jump check:
82  if (tmplast != maxindex) { // neighbor to maxindex can share peak with maxindex
83  if (data[tmplast] >= prevLast) {
84  lastJump = true;
85  }
86  }
87  prevLast = data[tmplast];
88  tmplast++;
89  }
90 
91  // we keep one pre- or post- sample if we can (as in online)
92  // check though if we ended up on a 'jump', or out of bounds: if so, back up
93  if (firstJump || tmpfirst < 0)
94  tmpfirst++;
95  if (lastJump || tmplast >= data.size())
96  tmplast--;
97 
98  first = tmpfirst;
99  last = tmplast;
100 
101  return std::make_tuple(first, last);
102 }

174 std::tuple<short, short, short> CaloRawFitter::selectBunch(const std::vector<Bunch>& bunchvector)
175 {
176  short bunchindex = -1;
177  short maxall = -1;
178  int indx = -1;
179  short maxampbin(0), maxamplitude(0);
180 
181  for (unsigned int i = 0; i < bunchvector.size(); i++) {
182  short max = maxAmp(bunchvector.at(i), indx); // CRAP PTH, bug fix, trouble if more than one bunches
183  if (isInTimeRange(indx, mMaxTimeIndex, mMinTimeIndex)) {
184  if (max > maxall) {
185  maxall = max;
186  bunchindex = i;
187  maxampbin = indx;
188  maxamplitude = max;
189  }
190  }
191  }
192 
193  if (bunchindex >= 0) {
194  bool bunchOK = checkBunchEdgesForMax(bunchvector.at(bunchindex));
195  if (!bunchOK) {
196  bunchindex = -1;
197  }
198  }
199 
200  return std::make_tuple(bunchindex, maxampbin, maxamplitude);
201 }

212 double CaloRawFitter::calculateChi2(double amp, double time,
213  int first, int last,
214  double adcErr, double tau) const
215 {
216  if (first == last || first < 0) { // signal consists of single sample, chi2 estimate (0) not too well defined..
217  // or, first is negative, the indices are not valid
218  return -1;
219  }
220 
221  int nsamples = last - first + 1;
222 
223  double chi2 = 0;
224 
225  for (int i = 0; i < nsamples; i++) {
226  int x = first + i; // timebin
227  double xx = (x - time + tau) / tau; // help variable
228  double f = 0.0;
229 
230  if (xx > 0) {
231  f = amp * xx * xx * TMath::Exp(2 * (1 - xx));
232  }
233 
234  double dy = mReversed[x] - f;
235  chi2 += dy * dy;
236  }
237 
238  if (adcErr > 0.0) { // weight chi2
239  chi2 /= (adcErr * adcErr);
240  }
241 
242  return chi2;
243 }
244 std::tuple<int, int, float, short, short, float, int, int> CaloRawFitter::preFitEvaluateSamples(const std::vector<Bunch>& bunchvector,
245  std::optional<unsigned int> altrocfg1, std::optional<unsigned int> altrocfg2, int acut)
246 {
247 
248  float maxf(0), ped(0);
249  int nsamples(0), first(0), last(0);
250  short maxrev(0);
251 
252  // select the bunch with the highest amplitude unless any time constraints is set
253  auto [index, maxampindex, maxamp] = selectBunch(bunchvector);
254 
255  // something valid was found, and non-zero amplitude
256  if (index >= 0 && maxamp >= acut) {
257  // use more convenient numbering and possibly subtract pedestal
258 
259  //std::tie(ped, mReversed) = reverseAndSubtractPed((bunchvector.at(index)), altrocfg1, altrocfg2);
260  //maxf = (float)*std::max_element(mReversed.begin(), mReversed.end());
261 
262  int length = bunchvector.at(index).getBunchLength();
263  const std::vector<uint16_t>& sig = bunchvector.at(index).getADC();
264 
265  ped = evaluatePedestal(sig, length);
266 
267  for (int i = 0; i < length; i++) {
268  mReversed[i] = sig[length - i - 1] - ped;
269  if (maxf < mReversed[i])
270  maxf = mReversed[i];
271  }
272 
273  if (maxf >= acut) // possibly significant signal
274  {
275  // select array around max to possibly be used in fit
276  maxrev = maxampindex - bunchvector.at(index).getStartTime();
277  std::tie(first, last) = selectSubarray(gsl::span<double>(&mReversed[0], bunchvector.at(index).getBunchLength()), maxrev, acut);
278 
279  // sanity check: maximum should not be in first or last bin
280  // if we should do a fit
281  if (first != maxrev && last != maxrev) {
282  // calculate how many samples we have
283  nsamples = last - first + 1;
284  }
285  }
286  }
287 
288  return std::make_tuple(nsamples, index, maxf, maxamp, maxrev, ped, first, last);
289 }
