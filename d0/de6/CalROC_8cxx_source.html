215 double CalROC::getLTM(double* sigma, double fraction, CalROC* const outlierROC)
216 {
217  //
218  // Calculate LTM mean and sigma
219  //
220 
221  std::vector<double> ddata(mNchannels);
222  double mean = 0, lsigma = 0;
223  unsigned int nPoints = 0;
224  for (int i = 0; i < mNchannels; i++) {
225  if (!outlierROC || !(outlierROC->getValue(i))) {
226  if (mData[i] > 0.000000000000001) {
227  ddata[nPoints] = (double)mData[i] / 10000;
228  nPoints++;
229  }
230  }
231  }
232  int hh = TMath::Min(TMath::Nint(fraction * nPoints), int(nPoints));
233  TRobustEstimator tre;
234  tre.EvaluateUni(nPoints, ddata.data(), mean, lsigma, hh);
235  if (sigma) {
236  *sigma = lsigma;
237  }
238  return mean;
239 }

344 bool CalROC::unfold()
345 {
346  //
347  // Compute the mean value per pad col
348  // Divide with this value each pad col
349  // This is for the noise study
350  // Return false if one or more of the pad col was not normalised
351  //
352 
353  bool result = true;
354  float eps = 0.00000000000000001;
355  double mmeannotnull = getMeanNotNull();
356  double rmsnotnull = getRMSNotNull();
357  //printf("mmeannotnull %f and rmsnotnull %f\n",mmeannotnull,rmsnotnull);
358  if ((mmeannotnull < 0.) || (rmsnotnull < 0.)) {
359  return false;
360  }
361  // calcul the mean value per col
362  for (int icol = 0; icol < mNcols; icol++) {
363  float mean = 0.0;
364  float nb = 0.0;
365  for (int irow = 0; irow < mNrows; irow++) {
366  if (TMath::Abs(getValue(icol, irow) - mmeannotnull) < 5 * rmsnotnull) {
367  mean += getValue(icol, irow);
368  nb += 1.0;
369  }
370  }
371  if (nb > eps) {
372  mean = mean / nb;
373  if (mean > eps) {
374  for (int irow = 0; irow < mNrows; irow++) {
375  float value = getValue(icol, irow);
376  setValue(icol, irow, (float)(value / mean));
377  }
378  } else {
379  result = false;
380  }
381  } else {
382  result = false;
383  }
384  }
385  return result;
386 }

388 TH2F* CalROC::makeHisto2D(float min, float max, int type, float mu)
389 {
390  //
391  // make 2D histo
392  // type -1 = user defined range
393  // 0 = nsigma cut nsigma=min
394  // 1 = delta cut around median delta=min
395 
396  if (type >= 0) {
397  float epsr = 0.005;
398  if (type == 0) {
399  // nsigma range
400  float mean = getMean();
401  float sigma = getRMS();
402  float nsigma = TMath::Abs(min);
403  sigma *= nsigma;
404  if (sigma < epsr)
405  sigma = epsr;
406  min = mean - sigma;
407  max = mean + sigma;
408  }
409  if (type == 1) {
410  // fixed range
411  float mean = getMedian();
412  float delta = min;
413  min = mean - delta;
414  max = mean + delta;
415  }
416  if (type == 2) {
417  double sigma;
418  float mean = getLTM(&sigma, max);
419  sigma *= min;
420  if (sigma < epsr)
421  sigma = epsr;
422  min = mean - sigma;
423  max = mean + sigma;
424  }
425  }
426  std::stringstream title;
427  title << mTitle.c_str() << " 2D Plane " << mPla << " Chamber " << mCha;
428  std::string titlestr = title.str();
429  TH2F* his = new TH2F(mName.c_str(), titlestr.c_str(), mNrows, 0, mNrows, mNcols, 0, mNcols);
430  for (int irow = 0; irow < mNrows; irow++) {
431  for (int icol = 0; icol < mNcols; icol++) {
432  his->Fill(irow + 0.5, icol + 0.5, getValue(icol, irow) * mu);
433  }
434  }
435  his->SetStats(0);
436  his->SetMaximum(max);
437  his->SetMinimum(min);
438  return his;
439 }
