32 void Digitizer::process(const std::vector<o2::fdd::Hit>& hits,
33  std::vector<o2::fdd::Digit>& digitsBC,
34  std::vector<o2::fdd::ChannelData>& digitsCh,
35  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel>& labels)
36 {
37  // loop over all hits and produce digits
38  //LOG(INFO) << "Processing IR = " << mIntRecord << " | NHits = " << hits.size();
39 
40  flush(digitsBC, digitsCh, labels); // flush cached signal which cannot be affect by new event
41 
42  auto sorted_hits{hits};
43  std::sort(sorted_hits.begin(), sorted_hits.end(), [](o2::fdd::Hit const& a, o2::fdd::Hit const& b) {
44  return a.GetTrackID() < b.GetTrackID();
45  });
46  LOG(INFO) << "Pulse";
47  //Conversion of hits to the analogue pulse shape
48  for (auto& hit : sorted_hits) {
49  if (hit.GetTime() > 20e3) {
50  const int maxWarn = 10;
51  static int warnNo = 0;
52  if (warnNo < maxWarn) {
53  LOG(WARNING) << "Ignoring hit with time_in_event = " << hit.GetTime() << " ns"
54  << ((++warnNo < maxWarn) ? "" : " (suppressing further warnings)");
55  }
56  continue;
57  }
58 
59  std::array<o2::InteractionRecord, NBC2Cache> cachedIR;
60  int iChannel = hit.GetDetectorID();
61  int nPhotoElectrons = simulateLightYield(iChannel, hit.GetNphot());
62 
63  double delayScintillator = mRndScintDelay.getNextValue();
64  double timeHit = delayScintillator + hit.GetTime();
65 
66  timeHit -= getTOFCorrection(int(iChannel / 4)); // account for TOF to detector
67  timeHit += mIntRecord.timeNS;
68  o2::InteractionRecord irHit(timeHit); // BC in which the hit appears (might be different from interaction BC for slow particles)
69 
70  int nCachedIR = 0;
71  for (int i = BCCacheMin; i < BCCacheMax + 1; i++) {
72  double tNS = timeHit + o2::constants::lhc::LHCBunchSpacingNS * i;
73  cachedIR[nCachedIR].setFromNS(tNS);
74  if (tNS < 0 && cachedIR[nCachedIR] > irHit) {
75  continue; // don't go to negative BC/orbit (it will wrap)
76  }
77  setBCCache(cachedIR[nCachedIR++]); // ensure existence of cached container
78  }
79  // if digit for this sector does not exist, create one otherwise add to it
80  createPulse(nPhotoElectrons, hit.GetTrackID(), timeHit, cachedIR, nCachedIR, iChannel);
81  } //hit loop
82 }

85 void Digitizer::createPulse(int nPhE, int parID, double timeHit, std::array<o2::InteractionRecord, NBC2Cache> const& cachedIR, int nCachedIR, int channel)
86 {
87  auto const roundVc = [&](int i) -> int {
88  return (i / Vc::float_v::Size) * Vc::float_v::Size;
89  };
90 
91  double time0 = cachedIR[0].bc2ns(); // start time of the 1st cashed BC
92  float timeDiff = time0 - timeHit;
93  if (channel < 9)
94  timeDiff += parameters.TimeDelayFDC;
95  else
96  timeDiff += parameters.TimeDelayFDA;
97 
98  //LOG(INFO) <<"Ch = "<<channel<<" NphE = " << nPhE <<" timeDiff "<<timeDiff;
99  float charge = TMath::Qe() * parameters.PmGain * mBinSize / (mPmtTimeIntegral * ChargePerADC);
100 
101  Bool_t added[nCachedIR];
102  for (int ir = 0; ir < nCachedIR; ir++)
103  added[ir] = kFALSE;
104 
105  constexpr float BinSizeInv = 1.0 / mBinSize;
106  for (int iPhE = 0; iPhE < nPhE; ++iPhE) {
107  float tPhE = timeDiff + mRndSignalShape.getNextValue();
108  int const firstBin = roundVc(TMath::Max((int)0, (int)((tPhE - PMTransitTime) * BinSizeInv)));
109  int const lastBin = TMath::Min((int)NBC2Cache * NTimeBinsPerBC - 1, (int)((tPhE + 2.0 * PMTransitTime) * BinSizeInv));
110  //LOG(INFO) << "firstBin = "<<firstBin<<" lastbin "<<lastBin;
111 
112  float const tempT = mBinSize * (0.5f + firstBin) - tPhE;
113  long iStart = std::lround((tempT + 2.0f * PMTransitTime) * BinSizeInv);
114  float const offset = tempT + 2.0f * PMTransitTime - float(iStart) * mBinSize;
115  long const iOffset = std::lround(offset * BinSizeInv * float(parameters.NResponseTables - 1));
116  if (iStart < 0) { // this should not happen
117  LOG(ERROR) << "FDDDigitizer: table lookup failure";
118  }
119  iStart = roundVc(std::max(long(0), iStart));
120 
121  Vc::float_v workVc;
122  Vc::float_v pmtVc;
123  float const* q = mPMResponseTables[parameters.NResponseTables / 2 + iOffset].data() + iStart;
124  float const* qEnd = &mPMResponseTables[parameters.NResponseTables / 2 + iOffset].back();
125 
126  for (int ir = firstBin / NTimeBinsPerBC; ir <= lastBin / NTimeBinsPerBC; ir++) {
127  int localFirst = (ir == firstBin / NTimeBinsPerBC) ? firstBin : 0;
128  int localLast = (ir < lastBin / NTimeBinsPerBC) ? NTimeBinsPerBC : (lastBin - ir * NTimeBinsPerBC);
129  auto bcCache = getBCCache(cachedIR[ir]);
130  auto& analogSignal = (*bcCache).pulse[channel];
131  float* p = analogSignal.data() + localFirst;
132 
133  for (int localBin = localFirst, iEnd = roundVc(localLast); q < qEnd && localBin < iEnd; localBin += Vc::float_v::Size) {
134  pmtVc.load(q);
135  q += Vc::float_v::Size;
136  Vc::prefetchForOneRead(q);
137  workVc.load(p);
138  workVc += mRndGainVar.getNextValueVc() * charge * pmtVc;
139  workVc.store(p);
140  p += Vc::float_v::Size;
141  Vc::prefetchForOneRead(p);
142  }
143  added[ir] = kTRUE;
144  }
145  }
146  for (int ir = 0; ir < nCachedIR; ir++) {
147  if (added[ir]) {
148  auto bcCache = getBCCache(cachedIR[ir]);
149  (*bcCache).labels.emplace_back(parID, mEventID, mSrcID, channel);
150  }
151  }
152 }

154 void Digitizer::flush(std::vector<o2::fdd::Digit>& digitsBC,
155  std::vector<o2::fdd::ChannelData>& digitsCh,
156  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel>& labels)
157 {

160  int nCached = mCache.size();
161  if (nCached < 1) {
162  return;
163  }
164  if (mIntRecord.differenceInBC(mCache.back()) > -BCCacheMin) {
165  LOG(DEBUG) << "Generating new pedestal BL fluct. for BC range " << mCache.front() << " : " << mCache.back();
166  //generatePedestal();
167  } else {
168  return;
169  }
170  //o2::InteractionRecord ir0(mCache.front());
171  //int cacheSpan = 1 + mCache.back().differenceInBC(ir0);
172  //LOG(INFO) << "Cache spans " << cacheSpan << " with " << nCached << " BCs cached";
173 
174  for (int ibc = 0; ibc < nCached; ibc++) { // digitize BCs which might not be affected by future events
175  auto& bc = mCache[ibc];
176  storeBC(bc, digitsBC, digitsCh, labels);
177  }
178  // clean cache for BCs which are not needed anymore
179  //LOG(INFO) << "Cleaning cache";
180  mCache.erase(mCache.begin(), mCache.end());
181 }

216 float Digitizer::simulateTimeCFD(const ChannelBCDataF& pulse)
217 {
218 
219  std::fill(mTimeCFD.begin(), mTimeCFD.end(), 0);
220  float timeCFD = -1024;
221  int binShift = TMath::Nint(parameters.TimeShiftCFD / mBinSize);
222  for (int iBin = 0; iBin < NTimeBinsPerBC; ++iBin) {
223  //if (mTime[channel][iBin] != 0) std::cout << mTime[channel][iBin] / parameters.mChargePerADC << ", ";
224  if (iBin >= binShift)
225  mTimeCFD[iBin] = 5.0 * pulse[iBin - binShift] - pulse[iBin];
226  else
227  mTimeCFD[iBin] = -1.0 * pulse[iBin];
228  }
229  for (int iBin = 1; iBin < NTimeBinsPerBC; ++iBin) {
230  if (mTimeCFD[iBin - 1] < 0 && mTimeCFD[iBin] >= 0) {
231  timeCFD = mBinSize * float(iBin);
232  break;
233  }
234  }
235  //LOG(INFO) <<" Time " << timeCFD;
236  return timeCFD;
237 }

239 o2::fdd::Digitizer::BCCache& Digitizer::setBCCache(const o2::InteractionRecord& ir)
240 {
241  if (mCache.empty() || mCache.back() < ir) {
242  mCache.emplace_back();
243  auto& cb = mCache.back();
244  cb = ir;
245  return cb;
246  }
247  if (mCache.front() > ir) {
248  mCache.emplace_front();
249  auto& cb = mCache.front();
250  cb = ir;
251  return cb;
252  }
253 
254  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
255  if ((*cb) == ir) {
256  return *cb;
257  }
258  if (ir < (*cb)) {
259  auto cbnew = mCache.emplace(cb); // insert new element before cb
260  (*cbnew) = ir;
261  return (*cbnew);
262  }
263  }
264  return mCache.front();
265 }

283 void Digitizer::init()
284 {
285  auto const roundVc = [&](int i) -> int {
286  return (i / Vc::float_v::Size) * Vc::float_v::Size;
287  };
288  // set up PMT response tables
289  float offset = -0.5f * mBinSize; // offset \in [-0.5..0.5] * mBinSize
290  int const nBins = roundVc(std::lround(4.0f * PMTransitTime / mBinSize));
291  for (auto& table : mPMResponseTables) {
292  table.resize(nBins);
293  float t = -2.0f * PMTransitTime + offset; // t \in offset + [-2 2] * DP::mPmtTransitTime
294  for (int j = 0; j < nBins; ++j) {
295  table[j] = Digitizer::PMResponse(t);
296  t += mBinSize;
297  }
298  offset += mBinSize / float(parameters.NResponseTables - 1);
299  }
300 
301  TF1 scintDelayFn("fScintDelay", "gaus", -6.0f * IntTimeRes, +6.0f * IntTimeRes);
302  scintDelayFn.SetParameters(1, 0, IntTimeRes);
303  mRndScintDelay.initialize(scintDelayFn);
304 
305  // Initialize function describing the PMT time response
306  TF1 pmtResponseFn("mPmtResponseFn", &Digitizer::PMResponse, -1.0f * PMTransitTime, +2.0f * PMTransitTime, 0);
307  pmtResponseFn.SetNpx(100);
308  mPmtTimeIntegral = pmtResponseFn.Integral(-1.0f * PMTransitTime, +2.0f * PMTransitTime);
309 
310  // Initialize function describing PMT response to the single photoelectron
311  TF1 singlePhESpectrumFn("mSinglePhESpectrum",
312  &Digitizer::SinglePhESpectrum, 0, 30, 0);
313  float const meansPhE = singlePhESpectrumFn.Mean(0, 30);
314  mRndGainVar.initialize([&]() -> float {
315  return singlePhESpectrumFn.GetRandom(0, 30) / meansPhE;
316  });
317 
318  TF1 signalShapeFn("signalShape", "crystalball", 0, 300);
319  signalShapeFn.SetParameters(1, parameters.ShapeSigma, parameters.ShapeSigma, parameters.ShapeAlpha, parameters.ShapeN);
320  mRndSignalShape.initialize([&]() -> float {
321  return signalShapeFn.GetRandom(0, 200);
322  });
323 }

328 int Digitizer::simulateLightYield(int pmt, int nPhot)
329 {
330  const float p = parameters.LightYield * PhotoCathodeEfficiency;
331  if (p == 1.0f || nPhot == 0)
332  return nPhot;
333  const int n = int(nPhot < 100 ? gRandom->Binomial(nPhot, p) : gRandom->Gaus(p * nPhot + 0.5, TMath::Sqrt(p * (1 - p) * nPhot)));
334  return n;
335 }

342 Double_t Digitizer::PMResponse(Double_t x)
343 {
344  // this function describes the PM time response to a single photoelectron
345  Double_t y = x + PMTransitTime;
346  return y * y * TMath::Exp(-y * y / (PMTransitTime * PMTransitTime));
347 }

349 Double_t Digitizer::SinglePhESpectrum(Double_t* x, Double_t*)
350 {
351  // this function describes the PM amplitude response to a single photoelectron
352  Double_t y = x[0];
353  if (y < 0)
354  return 0;
355  return (TMath::Poisson(y, PMNbOfSecElec) + PMTransparency * TMath::Poisson(y, 1.0));
356 }
