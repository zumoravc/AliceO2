21 GPUd() int Ray::crossLayer(const MatLayerCyl& lr)
22 {
23  // Calculate parameters t of intersection with cyl.layer
24  // Calculated as solution of equation for ray crossing with circles of r (rmin and rmax)
25  // t^2*mDistXY2 +- sqrt( mXDxPlusYDy^2 - mDistXY2*(mR02 - r^2) )
26  // Region of valid t is 0:1.
27  // Straigh line may have 2 crossings with cyl. layer
28  float detMax = mXDxPlusYDy2 - mDistXY2 * (mR02 - lr.getRMax2());
29  if (detMax < 0)
30  return 0; // does not reach outer R, hence inner also
31  float detMaxRed = CAMath::Sqrt(detMax) * mDistXY2i;
32  float tCross0Max = mXDxPlusYDyRed + detMaxRed; // largest possible t
33 
34  if (tCross0Max < 0) { // max t is outside of the limiting point -> other t's also
35  return 0;
36  }
37 
38  float tCross0Min = mXDxPlusYDyRed - detMaxRed; // smallest possible t
39  if (tCross0Min > 1.f) { // min t is outside of the limiting point -> other t's also
40  return 0;
41  }
42  float detMin = mXDxPlusYDy2 - mDistXY2 * (mR02 - lr.getRMin2());
43  if (detMin < 0) { // does not reach inner R -> just 1 tangential crossing
44  mCrossParams1[0] = tCross0Min > 0.f ? tCross0Min : 0.f;
45  mCrossParams2[0] = tCross0Max < 1.f ? tCross0Max : 1.f;
46  return validateZRange(mCrossParams1[0], mCrossParams2[0], lr);
47  }
48  int nCross = 0;
49  float detMinRed = CAMath::Sqrt(detMin) * mDistXY2i;
50  float tCross1Max = mXDxPlusYDyRed + detMinRed;
51  float tCross1Min = mXDxPlusYDyRed - detMinRed;
52 
53  if (tCross1Max < 1.f) {
54  mCrossParams1[0] = tCross0Max < 1.f ? tCross0Max : 1.f;
55  mCrossParams2[0] = tCross1Max > 0.f ? tCross1Max : 0.f;
56  if (validateZRange(mCrossParams1[nCross], mCrossParams2[nCross], lr)) {
57  nCross++;
58  }
59  }
60 
61  if (tCross1Min > -0.f) {
62  mCrossParams1[nCross] = tCross1Min < 1.f ? tCross1Min : 1.f;
63  mCrossParams1[nCross] = tCross0Min > 0.f ? tCross0Min : 0.f;
64  if (validateZRange(mCrossParams1[nCross], mCrossParams2[nCross], lr)) {
65  nCross++;
66  }
67  }
68  return nCross;
69 }
