224 template <class Mapping>
225 GBTLink::CollectedDataStatus GBTLink::collectROFCableData(const Mapping& chmap)
226 {
227  int nw = 0;
228  status = None;
229  auto* currRawPiece = rawData.currentPiece();
230  while (currRawPiece) { // we may loop over multiple CRU page
231  if (dataOffset >= currRawPiece->size) {
232  dataOffset = 0; // start of the RDH
233  if (!(currRawPiece = rawData.nextPiece())) { // fetch next CRU page
234  break; // Data chunk (TF?) is done
235  }
236  }
237  if (!dataOffset) { // here we always start with the RDH
238  const auto* rdh = reinterpret_cast<const RDH*>(&currRawPiece->data[dataOffset]);
239  if (verbosity >= VerboseHeaders) {
240  printRDH(*rdh);
241  }
242  GBTLINK_DECODE_ERRORCHECK(checkErrorsRDH(*rdh)); // make sure we are dealing with RDH
243  GBTLINK_DECODE_ERRORCHECK(checkErrorsRDHStop(*rdh)); // if new HB starts, the lastRDH must have stop
244  GBTLINK_DECODE_ERRORCHECK(checkErrorsRDHStopPageEmpty(*rdh)); // end of HBF should be an empty page with stop
245  lastRDH = rdh;
246  statistics.nPackets++;
247  dataOffset += sizeof(RDH);
248  continue;
249  }
250 
251  ruPtr->nCables = ruPtr->ruInfo->nCables; // RSTODO is this needed? TOREMOVE
252 
253  // data must start with GBT trigger word (unless we work with old format)
254  const GBTTrigger* gbtTrg = nullptr;
255  if (format == NewFormat) {
256  gbtTrg = reinterpret_cast<const GBTTrigger*>(&currRawPiece->data[dataOffset]); // process GBT trigger
257  dataOffset += GBTPaddedWordLength;
258  if (verbosity >= VerboseHeaders) {
259  printTrigger(gbtTrg);
260  }
261  GBTLINK_DECODE_ERRORCHECK(checkErrorsTriggerWord(gbtTrg));
262  }
263 
264  // next the GBTHeader must come
265  auto gbtH = reinterpret_cast<const GBTDataHeader*>(&currRawPiece->data[dataOffset]); // process GBT header
266  dataOffset += GBTPaddedWordLength;
267  if (verbosity >= VerboseHeaders) {
268  printHeader(gbtH);
269  }
270  GBTLINK_DECODE_ERRORCHECK(checkErrorsHeaderWord(gbtH));
271  lanesActive = gbtH->activeLanes; // TODO do we need to update this for every page?
272  GBTLINK_DECODE_ERRORCHECK(checkErrorsActiveLanes(chmap.getCablesOnRUType(ruPtr->ruInfo->ruType)));
273  if (format == OldFormat && RDHUtils::getPageCounter(*lastRDH)) { // RSTODO reset flags in case of 1st page of new ROF (old format: judge by RDH)
274  lanesStop = 0;
275  lanesWithData = 0;
276  } else if (gbtH->packetIdx == 0) { // reset flags in case of 1st page of new ROF (new format: judge by header)
277  lanesStop = 0;
278  lanesWithData = 0;
279  }
280  auto gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
281  while (!gbtD->isDataTrailer()) { // start reading real payload
282  nw++;
283  int cableHW = gbtD->getCableID(), cableSW = chmap.cableHW2SW(ruPtr->ruInfo->ruType, cableHW);
284  if (verbosity >= VerboseData) {
285  gbtD->printX();
286  }
287  GBTLINK_DECODE_ERRORCHECK(checkErrorsGBTData(cableHW, cableSW));
288  ruPtr->cableData[cableSW].add(gbtD->getW8(), 9);
289  ruPtr->cableHWID[cableSW] = cableHW;
290  ruPtr->cableLinkID[cableSW] = idInRU;
291  ruPtr->cableLinkPtr[cableSW] = this;
292  dataOffset += GBTPaddedWordLength;
293  gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
294  } // we are at the trailer, packet is over, check if there are more data on the next page
295 
296  auto gbtT = reinterpret_cast<const o2::itsmft::GBTDataTrailer*>(&currRawPiece->data[dataOffset]); // process GBT trailer
297  dataOffset += GBTPaddedWordLength;
298  if (verbosity >= VerboseHeaders) {
299  printTrailer(gbtT);
300  }
301  GBTLINK_DECODE_ERRORCHECK(checkErrorsTrailerWord(gbtT));
302  // we finished the GBT page, but there might be continuation on the next CRU page
303  if (!gbtT->packetDone) {
304  GBTLINK_DECODE_ERRORCHECK(checkErrorsPacketDoneMissing(gbtT, dataOffset < currRawPiece->size));
305  continue; // keep reading next CRU page
306  }
307  GBTLINK_DECODE_ERRORCHECK(checkErrorsLanesStops());
308  // accumulate packet states
309  statistics.packetStates[gbtT->getPacketState()]++;
310  // before quitting, store the trigger and IR
311  if (format == NewFormat) {
312  ir.bc = gbtTrg->bc;
313  ir.orbit = gbtTrg->orbit;
314  trigger = gbtTrg->triggerType;
315  } else {
316  ir = RDHUtils::getTriggerIR(lastRDH);
317  trigger = RDHUtils::getTriggerType(lastRDH);
318  }
319  return (status = DataSeen);
320  }
321 
322  return (status = StoppedOnEndOfData);
323 }
