36 short Geometry::relToAbsId(char moduleNumber, int strip, int cell)
37 {
38  // calculates absolute cell Id from moduleNumber, strip (number) and cell (number)
39  // PHOS layout parameters:
40  const short nStrpZ = 28; // Number of strips along z-axis
41  const short nCrystalsInModule = 56 * 64; // Total number of crystals in module
42  const short nCellsXInStrip = 8; // Number of crystals in strip unit along x-axis
43  const short nZ = 56; // nStripZ * nCellsZInStrip
44 
45  short row = nStrpZ - (strip - 1) % nStrpZ;
46  short col = (int)std::ceil((float)strip / (nStrpZ)) - 1;
47 
48  return (moduleNumber - 1) * nCrystalsInModule + row * 2 + (col * nCellsXInStrip + (cell - 1) / 2) * nZ -
49  (cell & 1 ? 1 : 0);
50 }

52 bool Geometry::absToRelNumbering(short absId, char* relid)
53 {
54  // Converts the absolute numbering into the following array
55  // relid[0] = PHOS Module number 1:fNModules
56  // relid[1] = Row number inside a PHOS module (Z coordinate)
57  // relid[2] = Column number inside a PHOS module (Phi coordinate)
58  const short nZ = 56; // nStripZ * nCellsZInStrip
59  const short nPhi = 64; // nStripZ * nCellsZInStrip
60 
61  short phosmodulenumber = (absId - 1) / (nZ * nPhi);
62 
63  relid[0] = phosmodulenumber + 1;
64  absId -= phosmodulenumber * nPhi * nZ;
65  relid[1] = 1 + (absId - 1) / nZ;
66  relid[2] = absId - (relid[1] - 1) * nZ;
67 
68  return true;
69 }

78 int Geometry::areNeighbours(short absId1, short absId2)
79 {
80 
81  // Gives the neighbourness of two digits = 0 are not neighbour but continue searching
82  // = 1 are neighbour
83  // = 2 are not neighbour but do not continue searching
84  // =-1 are not neighbour, continue searching, but do not look before d2 next
85  // time
86  // neighbours are defined as digits having at least a common vertex
87  // The order of d1 and d2 is important: first (d1) should be a digit already in a cluster
88  // which is compared to a digit (d2) not yet in a cluster
89 
90  char relid1[3];
91  absToRelNumbering(absId1, relid1);
92 
93  char relid2[3];
94  absToRelNumbering(absId2, relid2);
95 
96  if (relid1[0] == relid2[0]) { // inside the same PHOS module
97  char rowdiff = TMath::Abs(relid1[1] - relid2[1]);
98  char coldiff = TMath::Abs(relid1[2] - relid2[2]);
99 
100  if ((coldiff <= 1) && (rowdiff <= 1)) { // At least common vertex
101  return 1;
102  } else {
103  if ((relid2[1] > relid1[1]) && (relid2[2] > relid1[2] + 1))
104  return 2; // Difference in row numbers is too large to look further
105  }
106  return 0;
107 
108  } else {
109  if (relid1[0] > relid2[0]) // we switched to the next module
110  return -1;
111  return 2;
112  }
113  return 0;
114 }
