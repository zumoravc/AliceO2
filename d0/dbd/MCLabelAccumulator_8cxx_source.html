31 void MCLabelAccumulator::collect(const ChargePos& pos, Charge q)
32 {
33  if (q == 0 || !engaged()) {
34  return;
35  }
36 
37  uint index = mIndexMap[pos];
38 
39  auto labels = mLabels->getLabels(index);
40 
41  for (const auto& label : labels) {
42  int h = label.getRawValue() % mMaybeHasLabel.size();
43 
44  if (mMaybeHasLabel[h]) {
45  auto lookup = std::find(mClusterLabels.begin(), mClusterLabels.end(), label);
46  if (lookup != mClusterLabels.end()) {
47  continue;
48  }
49  }
50 
51  mMaybeHasLabel[h] = true;
52  mClusterLabels.emplace_back(label);
53  }
54 }

56 void MCLabelAccumulator::commit(Row row, uint indexInRow, uint maxElemsPerBucket)
57 {
58  if (indexInRow >= maxElemsPerBucket || !engaged()) {
59  return;
60  }
61 
62  auto& out = mOutput[row * maxElemsPerBucket + indexInRow];
63  out.labels = std::move(mClusterLabels);
64 }
