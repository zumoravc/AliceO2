21 static int _makefiles_opencl_obtain_program_helper(cl_context context, cl_uint num_devices, cl_device_id* devices, cl_program* program, char* binaries)
22 {
23  const char* magic_bytes = "QOCLPB";
24  if (strncmp(magic_bytes, binaries, strlen(magic_bytes)) != 0) {
25  printf("Internal error accessing opencl program\n");
26  return (1);
27  }
28  char* current_ptr = binaries + strlen(magic_bytes) + 1;
29  _makefiles_opencl_platform_info* pinfo = (_makefiles_opencl_platform_info*)current_ptr;
30  current_ptr += sizeof(_makefiles_opencl_platform_info);
31 
32  if (num_devices != pinfo->count) {
33  printf("Number of devices differs from number of devices in opencl program\n");
34  return (1);
35  }
36  // printf("Obtaining program for OpenCL Platform: (%s %s) %s %s\n", pinfo->platform_profile, pinfo->platform_version, pinfo->platform_vendor, pinfo->platform_name);
37 
38  std::vector<size_t> program_sizes(pinfo->count);
39  std::vector<char*> program_binaries(pinfo->count);
40 
41  for (unsigned int i = 0; i < pinfo->count; i++) {
42  char device_name[64], device_vendor[64];
43  cl_uint nbits;
44  clGetDeviceInfo(devices[i], CL_DEVICE_NAME, 64, device_name, nullptr);
45  clGetDeviceInfo(devices[i], CL_DEVICE_VENDOR, 64, device_vendor, nullptr);
46  clGetDeviceInfo(devices[i], CL_DEVICE_ADDRESS_BITS, sizeof(nbits), &nbits, nullptr);
47  _makefiles_opencl_device_info* dinfo = (_makefiles_opencl_device_info*)current_ptr;
48  if (strcmp(device_name, dinfo->device_name) != 0 || strcmp(device_vendor, dinfo->device_vendor) != 0) {
49  printf("Device list is different to device list from opencl program (Device %d: '%s - %s' != '%s - %s')\n", i, device_vendor, device_name, dinfo->device_vendor, dinfo->device_name);
50  return (1);
51  }
52  if (nbits != dinfo->nbits) {
53  printf("Pointer size of device and stored device binary differs\n");
54  return (1);
55  }
56  current_ptr += sizeof(_makefiles_opencl_device_info);
57  // printf("Device %d: %s %s (size %lld)\n", i, dinfo->device_vendor, dinfo->device_name, (long long int) dinfo->binary_size);
58  program_sizes[i] = dinfo->binary_size;
59  program_binaries[i] = current_ptr;
60  current_ptr += dinfo->binary_size;
61  }
62 
63  cl_int return_status[pinfo->count];
64  cl_int ocl_error;
65  *program = clCreateProgramWithBinary(context, num_devices, devices, program_sizes.data(), (const unsigned char**)program_binaries.data(), return_status, &ocl_error);
66 
67  if (ocl_error != CL_SUCCESS) {
68  printf("Error loading program\n");
69  return (1);
70  }
71 
72  for (unsigned int i = 0; i < pinfo->count; i++) {
73  if (return_status[i] != CL_SUCCESS) {
74  printf("Error loading program for device %d\n", i);
75  clReleaseProgram(*program);
76  return (1);
77  }
78  }
79 
80  ocl_error = clBuildProgram(*program, num_devices, devices, "", nullptr, nullptr);
81  if (ocl_error != CL_SUCCESS) {
82  printf("Error building program\n");
83  clReleaseProgram(*program);
84  return (1);
85  }
86 
87  return (0);
88 }
