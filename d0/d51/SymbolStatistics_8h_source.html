120 template <typename IT>
121 void SymbolStatistics::buildFrequencyTable(const IT begin, const IT end,
122  size_t range)
123 {
124  // find min_ and max_
125  const auto minmax = std::minmax_element(begin, end);
126 
127  if (range > 0) {
128  mMin = 0;
129  mMax = (1 << range) - 1;
130 
131  // do checks
132  if (static_cast<unsigned int>(mMin) > *minmax.first) {
133  throw std::runtime_error("min of data too small for given minimum");
134  }
135 
136  if (static_cast<unsigned int>(mMax) < *minmax.second) {
137  throw std::runtime_error("max of data too big for given maximum");
138  }
139  } else {
140  mMin = *minmax.first;
141  mMax = *minmax.second;
142  }
143 
144  mFrequencyTable.resize(std::abs(mMax - mMin) + 1, 0);
145 
146  for (IT it = begin; it != end; it++) {
147  mFrequencyTable[*it - mMin]++;
148  }
149 }
