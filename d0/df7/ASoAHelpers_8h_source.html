66 template <typename T2, typename ARRAY>
67 std::vector<std::pair<uint64_t, uint64_t>> doGroupTable(const std::shared_ptr<arrow::Table>& table, const std::string& categoryColumnName, int minCatSize, const T2& outsider)
68 {
69  auto columnIndex = table->schema()->GetFieldIndex(categoryColumnName);
70  auto chunkedArray = table->column(columnIndex);
71 
72  uint64_t ind = 0;
73  std::vector<std::pair<uint64_t, uint64_t>> groupedIndices;
74  for (uint64_t ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
75  auto chunk = chunkedArray->chunk(ci);
76  // Note: assuming that the table is not empty
77  T2 const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
78  for (uint64_t ai = 0; ai < chunk->length(); ++ai) {
79  if (data[ai] != outsider) {
80  groupedIndices.emplace_back(data[ai], ind);
81  }
82  ind++;
83  }
84  }
85 
86  // Do a stable sort so that same categories entries are
87  // grouped together.
88  std::stable_sort(groupedIndices.begin(), groupedIndices.end());
89 
90  // Remove categories of too small size
91  if (minCatSize > 1) {
92  auto catBegin = groupedIndices.begin();
93  while (catBegin != groupedIndices.end()) {
94  auto catEnd = std::upper_bound(catBegin, groupedIndices.end(), *catBegin, sameCategory);
95  if (std::distance(catBegin, catEnd) < minCatSize) {
96  catEnd = groupedIndices.erase(catBegin, catEnd);
97  }
98  catBegin = catEnd;
99  }
100  }
101 
102  return groupedIndices;
103 }

105 template <typename T, typename T2>
106 auto groupTable(const T& table, const std::string& categoryColumnName, int minCatSize, const T2& outsider)
107 {
108  auto arrowTable = table.asArrowTable();
109  auto columnIndex = arrowTable->schema()->GetFieldIndex(categoryColumnName);
110  auto dataType = arrowTable->column(columnIndex)->type();
111  if (dataType->id() == arrow::Type::UINT64) {
112  return doGroupTable<uint64_t, arrow::UInt64Array>(arrowTable, categoryColumnName, minCatSize, outsider);
113  }
114  if (dataType->id() == arrow::Type::INT64) {
115  return doGroupTable<int64_t, arrow::Int64Array>(arrowTable, categoryColumnName, minCatSize, outsider);
116  }
117  if (dataType->id() == arrow::Type::UINT32) {
118  return doGroupTable<uint32_t, arrow::UInt32Array>(arrowTable, categoryColumnName, minCatSize, outsider);
119  }
120  if (dataType->id() == arrow::Type::INT32) {
121  return doGroupTable<int32_t, arrow::Int32Array>(arrowTable, categoryColumnName, minCatSize, outsider);
122  }
123  if (dataType->id() == arrow::Type::FLOAT) {
124  return doGroupTable<float, arrow::FloatArray>(arrowTable, categoryColumnName, minCatSize, outsider);
125  }
126  // FIXME: Should we support other types as well?
127  throw std::runtime_error("Combinations: category column must be of integral type");
128 }

130 // Synchronize categories so as groupedIndices contain elements only of categories common to all tables
131 template <std::size_t K>
132 void syncCategories(std::array<std::vector<std::pair<uint64_t, uint64_t>>, K>& groupedIndices)
133 {
134  std::vector<std::pair<uint64_t, uint64_t>> firstCategories;
135  std::vector<std::pair<uint64_t, uint64_t>> commonCategories;
136  std::unique_copy(groupedIndices[0].begin(), groupedIndices[0].end(), std::back_inserter(firstCategories), diffCategory);
137 
138  for (auto& cat : firstCategories) {
139  bool isCommon = true;
140  for (int i = 1; i < K; i++) {
141  if (!std::binary_search(groupedIndices[i].begin(), groupedIndices[i].end(), cat, sameCategory)) {
142  isCommon = false;
143  break;
144  }
145  }
146  if (isCommon) {
147  commonCategories.push_back(cat);
148  }
149  }
150 
151  for (int i = 0; i < K; i++) {
152  auto nonCatBegin = groupedIndices[i].begin();
153  for (auto& cat : commonCategories) {
154  auto nonCatEnd = std::lower_bound(nonCatBegin, groupedIndices[i].end(), cat, sameCategory);
155  nonCatEnd = groupedIndices[i].erase(nonCatBegin, nonCatEnd);
156  nonCatBegin = std::upper_bound(nonCatEnd, groupedIndices[i].end(), cat, sameCategory);
157  }
158  }
159 }

161 template <typename... Ts>
162 struct CombinationsIndexPolicyBase {
163  using CombinationType = std::tuple<typename Ts::iterator...>;
164  using IndicesType = typename NTupleType<uint64_t, sizeof...(Ts)>::type;
165 
166  CombinationsIndexPolicyBase(const Ts&... tables) : mIsEnd(false),
167  mMaxOffset(tables.end().index...),
168  mCurrent(tables.begin()...)
169  {
170  if (((tables.size() == 0) || ...)) {
171  this->mIsEnd = true;
172  }
173  }
174 
175  void moveToEnd()
176  {
177  constexpr auto k = sizeof...(Ts);
178  for_<k>([&, this](auto i) {
179  std::get<i.value>(this->mCurrent).moveToEnd();
180  });
181  this->mIsEnd = true;
182  }
183 
184  void addOne() {}
185 
186  CombinationType mCurrent;
187  IndicesType mMaxOffset; // one position past maximum acceptable position for each element of combination
188  bool mIsEnd; // whether there are any more tuples available
189 };

191 template <typename... Ts>
192 struct CombinationsUpperIndexPolicy : public CombinationsIndexPolicyBase<Ts...> {
193  using CombinationType = typename CombinationsIndexPolicyBase<Ts...>::CombinationType;
194 
195  CombinationsUpperIndexPolicy(const Ts&... tables) : CombinationsIndexPolicyBase<Ts...>(tables...) {}
196 
197  void addOne()
198  {
199  constexpr auto k = sizeof...(Ts);
200  bool modify = true;
201  for_<k>([&, this](auto i) {
202  if (modify) {
203  constexpr auto curInd = k - i.value - 1;
204  std::get<curInd>(this->mCurrent)++;
205  if (*std::get<1>(std::get<curInd>(this->mCurrent).getIndices()) != std::get<curInd>(this->mMaxOffset)) {
206  modify = false;
207  for_<i.value>([&, this](auto j) {
208  constexpr auto curJ = k - i.value + j.value;
209  int64_t nextInd = *std::get<1>(std::get<curJ - 1>(this->mCurrent).getIndices());
210  if (nextInd < std::get<curJ>(this->mMaxOffset)) {
211  std::get<curJ>(this->mCurrent).setCursor(nextInd);
212  } else {
213  modify = true;
214  }
215  });
216  }
217  }
218  });
219  this->mIsEnd = modify;
220  }
221 };

295 struct CombinationsBlockIndexPolicyBase : public CombinationsIndexPolicyBase<Ts...> {
296  using CombinationType = typename CombinationsIndexPolicyBase<Ts...>::CombinationType;
297  using IndicesType = typename NTupleType<uint64_t, sizeof...(Ts)>::type;
298 
299  CombinationsBlockIndexPolicyBase(const std::string& categoryColumnName, int categoryNeighbours, const T& outsider, const Ts&... tables) : CombinationsIndexPolicyBase<Ts...>(tables...), mSlidingWindowSize(categoryNeighbours + 1)
300  {
301  constexpr auto k = sizeof...(Ts);
302  if (this->mIsEnd) {
303  return;
304  }
305  if (mSlidingWindowSize < 1) {
306  this->mIsEnd = true;
307  return;
308  }
309 
310  int tableIndex = 0;
311  ((this->mGroupedIndices[tableIndex++] = groupTable(tables, categoryColumnName, 1, outsider)), ...);
312 
313  // Synchronize categories across tables
314  syncCategories(this->mGroupedIndices);
315 
316  for (int i = 0; i < k; i++) {
317  if (this->mGroupedIndices[i].size() == 0) {
318  this->mIsEnd = true;
319  return;
320  }
321  }
322 
323  for_<k>([this](auto i) {
324  std::get<i.value>(this->mCurrentIndices) = 0;
325  });
326  }
327 
328  std::array<std::vector<std::pair<uint64_t, uint64_t>>, sizeof...(Ts)> mGroupedIndices;
329  IndicesType mCurrentIndices;
330  IndicesType mBeginIndices;
331  uint64_t mSlidingWindowSize;
332 };

334 template <typename T, typename... Ts>
335 struct CombinationsBlockUpperIndexPolicy : public CombinationsBlockIndexPolicyBase<T, Ts...> {
336  using CombinationType = typename CombinationsBlockIndexPolicyBase<T, Ts...>::CombinationType;
337 
338  CombinationsBlockUpperIndexPolicy(const std::string& categoryColumnName, int categoryNeighbours, const T& outsider, const Ts&... tables) : CombinationsBlockIndexPolicyBase<T, Ts...>(categoryColumnName, categoryNeighbours, outsider, tables...)
339  {
340  if (!this->mIsEnd) {
341  setRanges();
342  }
343  }
344 
345  void setRanges()
346  {
347  constexpr auto k = sizeof...(Ts);
348  for_<k>([&, this](auto i) {
349  auto catBegin = this->mGroupedIndices[i.value].begin() + std::get<i.value>(this->mCurrentIndices);
350  auto range = std::equal_range(catBegin, this->mGroupedIndices[i.value].end(), *catBegin, sameCategory);
351  std::get<i.value>(this->mBeginIndices) = std::distance(this->mGroupedIndices[i.value].begin(), range.first);
352  std::get<i.value>(this->mCurrent).setCursor(range.first->second);
353  std::get<i.value>(this->mMaxOffset) = std::distance(this->mGroupedIndices[i.value].begin(), range.second);
354  });
355  }
356 
357  void addOne()
358  {
359  constexpr auto k = sizeof...(Ts);
360  bool modify = true;
361  bool nextCatAvailable = true;
362  for_<k - 1>([&, this](auto i) {
363  if (modify) {
364  constexpr auto curInd = k - i.value - 1;
365  std::get<curInd>(this->mCurrentIndices)++;
366  uint64_t curGroupedInd = std::get<curInd>(this->mCurrentIndices);
367  std::get<curInd>(this->mCurrent).setCursor(this->mGroupedIndices[curInd][curGroupedInd].second);
368  uint64_t maxForWindow = std::get<curInd>(this->mBeginIndices) + this->mSlidingWindowSize;
369 
370  // If we remain within the same sliding window
371  if (curGroupedInd < maxForWindow && curGroupedInd < std::get<curInd>(this->mMaxOffset)) {
372  modify = false;
373  for_<i.value>([&, this](auto j) {
374  constexpr auto curJ = k - i.value + j.value;
375  if (std::get<curJ - 1>(this->mCurrentIndices) < std::get<curJ>(this->mMaxOffset)) {
376  std::get<curJ>(this->mCurrentIndices) = std::get<curJ - 1>(this->mCurrentIndices);
377  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
378  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
379  } else {
380  modify = true;
381  }
382  });
383  }
384  }
385  });
386 
387  // First iterator processed separately
388  if (modify) {
389  std::get<0>(this->mCurrentIndices)++;
390  std::get<0>(this->mBeginIndices)++;
391  uint64_t curGroupedInd = std::get<0>(this->mCurrentIndices);
392  std::get<0>(this->mCurrent).setCursor(this->mGroupedIndices[0][curGroupedInd].second);
393 
394  // If we remain within the same category - slide window
395  if (curGroupedInd < std::get<0>(this->mMaxOffset)) {
396  modify = false;
397  for_<k - 1>([&, this](auto j) {
398  constexpr auto curJ = j.value + 1;
399  std::get<curJ>(this->mBeginIndices)++;
400  if (std::get<curJ>(this->mBeginIndices) < std::get<curJ>(this->mMaxOffset)) {
401  std::get<curJ>(this->mCurrentIndices) = std::get<curJ>(this->mBeginIndices);
402  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
403  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
404  } else {
405  modify = true;
406  }
407  });
408  }
409  }
410 
411  // No more combinations within this category - move to the next category, if possible
412  if (modify) {
413  for_<k>([&, this](auto m) {
414  std::get<m.value>(this->mCurrentIndices) = std::get<m.value>(this->mMaxOffset);
415  if (std::get<m.value>(this->mCurrentIndices) == this->mGroupedIndices[m.value].size()) {
416  nextCatAvailable = false;
417  }
418  });
419  if (nextCatAvailable) {
420  setRanges();
421  }
422  }
423 
424  this->mIsEnd = modify && !nextCatAvailable;
425  }
426 };

428 template <typename T, typename... Ts>
429 struct CombinationsBlockFullIndexPolicy : public CombinationsBlockIndexPolicyBase<T, Ts...> {
430  using CombinationType = typename CombinationsBlockIndexPolicyBase<T, Ts...>::CombinationType;
431  using IndicesType = typename NTupleType<uint64_t, sizeof...(Ts)>::type;
432 
433  CombinationsBlockFullIndexPolicy(const std::string& categoryColumnName, int categoryNeighbours, const T& outsider, const Ts&... tables) : CombinationsBlockIndexPolicyBase<T, Ts...>(categoryColumnName, categoryNeighbours, outsider, tables...), mCurrentlyFixed(0)
434  {
435  if (!this->mIsEnd) {
436  setRanges();
437  }
438  }

440  void setRanges()
441  {
442  constexpr auto k = sizeof...(Ts);
443  for_<k>([&, this](auto i) {
444  auto catBegin = this->mGroupedIndices[i.value].begin() + std::get<i.value>(this->mCurrentIndices);
445  auto range = std::equal_range(catBegin, this->mGroupedIndices[i.value].end(), *catBegin, sameCategory);
446  std::get<i.value>(this->mBeginIndices) = std::distance(this->mGroupedIndices[i.value].begin(), range.first);
447  std::get<i.value>(this->mMaxOffset) = std::distance(this->mGroupedIndices[i.value].begin(), range.second);
448  std::get<i.value>(this->mCurrent).setCursor(range.first->second);
449  });
450  }
451 
452  void addOne()
453  {
454  constexpr auto k = sizeof...(Ts);
455  bool modify = true;
456  bool nextCatAvailable = true;
457  for_<k>([&, this](auto i) {
458  if (modify) {
459  constexpr auto curInd = k - i.value - 1;
460  std::get<curInd>(this->mCurrentIndices)++;
461  uint64_t curGroupedInd = std::get<curInd>(this->mCurrentIndices);
462  std::get<curInd>(this->mCurrent).setCursor(this->mGroupedIndices[curInd][curGroupedInd].second);
463  uint64_t windowOffset = curInd == this->mCurrentlyFixed ? 1 : this->mSlidingWindowSize;
464  uint64_t maxForWindow = std::get<curInd>(this->mBeginIndices) + windowOffset;
465 
466  // If we remain within the same sliding window and fixed index
467  if (curGroupedInd < maxForWindow && curGroupedInd < std::get<curInd>(this->mMaxOffset)) {
468  for_<i.value>([&, this](auto j) {
469  constexpr auto curJ = k - i.value + j.value;
470  if (curJ < this->mCurrentlyFixed) { // To assure no repetitions
471  std::get<curJ>(this->mCurrentIndices) = std::get<curJ>(this->mBeginIndices) + 1;
472  } else {
473  std::get<curJ>(this->mCurrentIndices) = std::get<curJ>(this->mBeginIndices);
474  }
475  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
476  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
477  });
478  modify = false;
479  }
480  }
481  });
482 
483  // Currently fixed iterator processed separately
484  if (modify) {
485  // If we haven't finished with window starting element
486  if (this->mCurrentlyFixed < k - 1 && std::get<0>(this->mBeginIndices) < std::get<0>(this->mMaxOffset) - 1) {
487  this->mCurrentlyFixed++;
488  for_<k>([&, this](auto s) {
489  if (s.value < this->mCurrentlyFixed) { // To assure no repetitions
490  std::get<s.value>(this->mCurrentIndices) = std::get<s.value>(this->mBeginIndices) + 1;
491  } else {
492  std::get<s.value>(this->mCurrentIndices) = std::get<s.value>(this->mBeginIndices);
493  }
494  uint64_t curGroupedI = std::get<s.value>(this->mCurrentIndices);
495  std::get<s.value>(this->mCurrent).setCursor(this->mGroupedIndices[s.value][curGroupedI].second);
496  });
497  modify = false;
498  } else {
499  this->mCurrentlyFixed = 0;
500  std::get<0>(this->mBeginIndices)++;
501  std::get<0>(this->mCurrentIndices) = std::get<0>(this->mBeginIndices);
502  uint64_t curGroupedInd = std::get<0>(this->mCurrentIndices);
503  std::get<0>(this->mCurrent).setCursor(this->mGroupedIndices[0][curGroupedInd].second);
504 
505  // If we remain within the same category - slide window
506  if (std::get<0>(this->mBeginIndices) < std::get<0>(this->mMaxOffset)) {
507  modify = false;
508  for_<k - 1>([&, this](auto j) {
509  constexpr auto curJ = j.value + 1;
510  std::get<curJ>(this->mBeginIndices)++;
511  if (std::get<curJ>(this->mBeginIndices) < std::get<curJ>(this->mMaxOffset)) {
512  std::get<curJ>(this->mCurrentIndices) = std::get<curJ>(this->mBeginIndices);
513  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
514  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
515  } else {
516  modify = true;
517  }
518  });
519  }
520  }
521  }
522 
523  // No more combinations within this category - move to the next category, if possible
524  if (modify) {
525  for_<k>([&, this](auto m) {
526  std::get<m.value>(this->mCurrentIndices) = std::get<m.value>(this->mMaxOffset);
527  if (std::get<m.value>(this->mCurrentIndices) == this->mGroupedIndices[m.value].size()) {
528  nextCatAvailable = false;
529  }
530  });
531  if (nextCatAvailable) {
532  setRanges();
533  }
534  }
535 
536  this->mIsEnd = modify && !nextCatAvailable;
537  }

542 template <typename T1, typename T, typename... Ts>
543 struct CombinationsBlockSameIndexPolicyBase : public CombinationsIndexPolicyBase<T, Ts...> {
544  using CombinationType = typename CombinationsIndexPolicyBase<T, Ts...>::CombinationType;
545  using IndicesType = typename NTupleType<uint64_t, sizeof...(Ts) + 1>::type;
546 
547  CombinationsBlockSameIndexPolicyBase(const std::string& categoryColumnName, int categoryNeighbours, const T1& outsider, int minWindowSize, const T& table, const Ts&... tables) : CombinationsIndexPolicyBase<T, Ts...>(table, tables...), mSlidingWindowSize(categoryNeighbours + 1)
548  {
549  constexpr auto k = sizeof...(Ts) + 1;
550  // minWindowSize == 1 for upper and full, and k for strictly upper k-combination
551  if (mSlidingWindowSize < minWindowSize) {
552  this->mIsEnd = true;
553  return;
554  }
555 
556  this->mGroupedIndices = groupTable(table, categoryColumnName, minWindowSize, outsider);
