194  void moveUnusedToHead()
195  {
196  auto left = getUnusedSize();
197  if (left < getOffset()) {
198  std::memcpy(mBuffer.data(), mPtr, left); // there is no overlap
199  } else {
200  std::memmove(mBuffer.data(), mPtr, left); // there is an overlap
201  }
202  mPtr = mBuffer.data();
203  mEnd = mPtr + left;
204  }

207  // (attemtint to use all free capacity) using the method provided via getNext

209  {
