111 template <typename DataT>
112 template <bool isConsistentT>
113 void SplineHelper2D<DataT>::approximateFunction(
114  Spline2DBase<DataT, isConsistentT>& spline,
115  DataT x1Min, DataT x1Max, DataT x2Min, DataT x2Max,
116  std::function<void(DataT x1, DataT x2, DataT f[/*spline.getFdimensions()*/])> F,
117  int nAxiliaryDataPointsU1, int nAxiliaryDataPointsU2)

127 template <typename DataT>
128 template <bool isConsistentT>
129 int SplineHelper2D<DataT>::setSpline(
130  const Spline2DBase<DataT, isConsistentT>& spline, int nAxiliaryPointsU, int nAxiliaryPointsV)
131 {
132  // Prepare creation of 2D irregular spline
133  // The should be at least one (better, two) axiliary measurements on each segnment between two knots and at least 2*nKnots measurements in total
134  // Returns 0 when the spline can not be constructed with the given nAxiliaryPoints
135 
136  int ret = 0;
137  mFdimensions = spline.getFdimensions();
138  if (mHelperU1.setSpline(spline.getGridU1(), mFdimensions, nAxiliaryPointsU) != 0) {
139  ret = storeError(-2, "SplineHelper2D::setSpline2D: error by setting U axis");
140  }
141  if (mHelperU2.setSpline(spline.getGridU2(), mFdimensions, nAxiliaryPointsV) != 0) {
142  ret = storeError(-3, "SplineHelper2D::setSpline2D: error by setting V axis");
143  }
144  return ret;
145 }
