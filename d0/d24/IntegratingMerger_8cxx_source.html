45 void IntegratingMerger::run(framework::ProcessingContext& ctx)
46 {
47  // we have to avoid mistaking the timer input with data inputs.
48  auto* timerHeader = ctx.inputs().get("timer-publish").header;
49 
50  for (const DataRef& ref : InputRecordWalker(ctx.inputs())) {
51  if (ref.header != timerHeader) {
52  if (std::holds_alternative<std::monostate>(mMergedObject)) {
53  mMergedObject = object_store_helpers::extractObjectFrom(ref);
54 
55  } else if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
56  // We expect that if the first object was TObject, then all should.
57  auto other = TObjectPtr(framework::DataRefUtils::as<TObject>(ref).release(), algorithm::deleteTCollections);
58  auto target = std::get<TObjectPtr>(mMergedObject);
59  algorithm::merge(target.get(), other.get());
60 
61  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
62  // We expect that if the first object inherited MergeInterface, then all should.
63  auto other = framework::DataRefUtils::as<MergeInterface>(ref);
64  std::get<MergeInterfacePtr>(mMergedObject)->merge(other.get());
65  } else {
66  throw std::runtime_error("mMergedObject' variant has no value.");
67  }
68  mObjectsMerged++;
69  }
70  }
71 
72  if (ctx.inputs().isValid("timer-publish")) {
73 
74  publish(ctx.outputs());
75 
76  if (mConfig.mergedObjectTimespan.value == MergedObjectTimespan::LastDifference) {
77  mMergedObject = std::monostate{};
78  }
79  }
80 }

82 void IntegratingMerger::publish(framework::DataAllocator& allocator)
83 {
84  if (std::holds_alternative<std::monostate>(mMergedObject)) {
85  LOG(INFO) << "Nothing to publish yet";
86  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
87  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
88  *std::get<MergeInterfacePtr>(mMergedObject));
89  } else if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
90  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
91  *std::get<TObjectPtr>(mMergedObject));
92  } else {
93  throw std::runtime_error("mMergedObject' variant has no value.");
94  }
95 
96  mTotalObjectsMerged += mObjectsMerged;
97  mCollector->send({mTotalObjectsMerged, "total_objects_merged"}, monitoring::DerivedMetricMode::RATE);
98  mCollector->send({mObjectsMerged, "objects_merged_since_last_publication"});
99  mObjectsMerged = 0;
100 }
