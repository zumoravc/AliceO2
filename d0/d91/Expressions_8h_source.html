54 template <typename T>
55 constexpr auto selectArrowType()
56 {
57  if constexpr (std::is_same_v<T, int>) {
58  return atype::INT32;
59  } else if constexpr (std::is_same_v<T, bool>) {
60  return atype::BOOL;
61  } else if constexpr (std::is_same_v<T, float>) {
62  return atype::FLOAT;
63  } else if constexpr (std::is_same_v<T, double>) {
64  return atype::DOUBLE;
65  } else {
66  return atype::NA;
67  }
68  O2_BUILTIN_UNREACHABLE();
69 }
