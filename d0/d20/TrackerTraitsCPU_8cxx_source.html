34 void TrackerTraitsCPU::computeLayerTracklets()
35 {
36  PrimaryVertexContext* primaryVertexContext = mPrimaryVertexContext;
37  for (int iLayer{0}; iLayer < constants::its::TrackletsPerRoad; ++iLayer) {
38  if (primaryVertexContext->getClusters()[iLayer].empty() || primaryVertexContext->getClusters()[iLayer + 1].empty()) {
39  return;
40  }
41 
42  const float3& primaryVertex = primaryVertexContext->getPrimaryVertex();
43  const int currentLayerClustersNum{static_cast<int>(primaryVertexContext->getClusters()[iLayer].size())};
44 
45  for (int iCluster{0}; iCluster < currentLayerClustersNum; ++iCluster) {
46  const Cluster& currentCluster{primaryVertexContext->getClusters()[iLayer][iCluster]};
47 
48  if (primaryVertexContext->isClusterUsed(iLayer, currentCluster.clusterId)) {
49  continue;
50  }
51 
52  const float tanLambda{(currentCluster.zCoordinate - primaryVertex.z) / currentCluster.rCoordinate};
53  const float directionZIntersection{tanLambda * (constants::its::LayersRCoordinate()[iLayer + 1] -
54  currentCluster.rCoordinate) +
55  currentCluster.zCoordinate};
56 
57  const int4 selectedBinsRect{getBinsRect(currentCluster, iLayer, directionZIntersection,
58  mTrkParams.TrackletMaxDeltaZ[iLayer], mTrkParams.TrackletMaxDeltaPhi)};
59 
60  if (selectedBinsRect.x == 0 && selectedBinsRect.y == 0 && selectedBinsRect.z == 0 && selectedBinsRect.w == 0) {
61  continue;
62  }
63 
64  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
65 
66  if (phiBinsNum < 0) {
67  phiBinsNum += constants::index_table::PhiBins;
68  }
69 
70  for (int iPhiBin{selectedBinsRect.y}, iPhiCount{0}; iPhiCount < phiBinsNum;
71  iPhiBin = ++iPhiBin == constants::index_table::PhiBins ? 0 : iPhiBin, iPhiCount++) {
72  const int firstBinIndex{index_table_utils::getBinIndex(selectedBinsRect.x, iPhiBin)};
73  const int maxBinIndex{firstBinIndex + selectedBinsRect.z - selectedBinsRect.x + 1};
74  const int firstRowClusterIndex = primaryVertexContext->getIndexTables()[iLayer][firstBinIndex];
75  const int maxRowClusterIndex = primaryVertexContext->getIndexTables()[iLayer][maxBinIndex];
76 
77  for (int iNextLayerCluster{firstRowClusterIndex}; iNextLayerCluster < maxRowClusterIndex;
78  ++iNextLayerCluster) {
79 
80  const Cluster& nextCluster{primaryVertexContext->getClusters()[iLayer + 1][iNextLayerCluster]};
81 
82  if (primaryVertexContext->isClusterUsed(iLayer + 1, nextCluster.clusterId)) {
83  continue;
84  }
85 
86  const float deltaZ{gpu::GPUCommonMath::Abs(tanLambda * (nextCluster.rCoordinate - currentCluster.rCoordinate) +
87  currentCluster.zCoordinate - nextCluster.zCoordinate)};
88  const float deltaPhi{gpu::GPUCommonMath::Abs(currentCluster.phiCoordinate - nextCluster.phiCoordinate)};
89 
90  if (deltaZ < mTrkParams.TrackletMaxDeltaZ[iLayer] &&
91  (deltaPhi < mTrkParams.TrackletMaxDeltaPhi ||
92  gpu::GPUCommonMath::Abs(deltaPhi - constants::math::TwoPi) < mTrkParams.TrackletMaxDeltaPhi)) {
93 
94  if (iLayer > 0 &&
95  primaryVertexContext->getTrackletsLookupTable()[iLayer - 1][iCluster] == constants::its::UnusedIndex) {
96 
97  primaryVertexContext->getTrackletsLookupTable()[iLayer - 1][iCluster] =
98  primaryVertexContext->getTracklets()[iLayer].size();
99  }
100 
101  primaryVertexContext->getTracklets()[iLayer].emplace_back(iCluster, iNextLayerCluster, currentCluster,
102  nextCluster);
103  }
104  }
105  }
106  }
107  }

110 void TrackerTraitsCPU::computeLayerCells()
111 {
112  PrimaryVertexContext* primaryVertexContext = mPrimaryVertexContext;
113  for (int iLayer{0}; iLayer < constants::its::CellsPerRoad; ++iLayer) {
114 
115  if (primaryVertexContext->getTracklets()[iLayer + 1].empty() ||
116  primaryVertexContext->getTracklets()[iLayer].empty()) {
117 
118  return;
119  }
120 
121  const float3& primaryVertex = primaryVertexContext->getPrimaryVertex();
122  const int currentLayerTrackletsNum{static_cast<int>(primaryVertexContext->getTracklets()[iLayer].size())};
123 
124  for (int iTracklet{0}; iTracklet < currentLayerTrackletsNum; ++iTracklet) {
125 
126  const Tracklet& currentTracklet{primaryVertexContext->getTracklets()[iLayer][iTracklet]};
127  const int nextLayerClusterIndex{currentTracklet.secondClusterIndex};
128  const int nextLayerFirstTrackletIndex{
129  primaryVertexContext->getTrackletsLookupTable()[iLayer][nextLayerClusterIndex]};
130 
131  if (nextLayerFirstTrackletIndex == constants::its::UnusedIndex) {
132 
133  continue;
134  }
135 
136  const Cluster& firstCellCluster{primaryVertexContext->getClusters()[iLayer][currentTracklet.firstClusterIndex]};
137  const Cluster& secondCellCluster{
138  primaryVertexContext->getClusters()[iLayer + 1][currentTracklet.secondClusterIndex]};
139  const float firstCellClusterQuadraticRCoordinate{firstCellCluster.rCoordinate * firstCellCluster.rCoordinate};
140  const float secondCellClusterQuadraticRCoordinate{secondCellCluster.rCoordinate *
141  secondCellCluster.rCoordinate};
142  const float3 firstDeltaVector{secondCellCluster.xCoordinate - firstCellCluster.xCoordinate,
143  secondCellCluster.yCoordinate - firstCellCluster.yCoordinate,
144  secondCellClusterQuadraticRCoordinate - firstCellClusterQuadraticRCoordinate};
145  const int nextLayerTrackletsNum{static_cast<int>(primaryVertexContext->getTracklets()[iLayer + 1].size())};
146 
147  for (int iNextLayerTracklet{nextLayerFirstTrackletIndex};
148  iNextLayerTracklet < nextLayerTrackletsNum &&
149  primaryVertexContext->getTracklets()[iLayer + 1][iNextLayerTracklet].firstClusterIndex ==
150  nextLayerClusterIndex;
151  ++iNextLayerTracklet) {
152 
153  const Tracklet& nextTracklet{primaryVertexContext->getTracklets()[iLayer + 1][iNextLayerTracklet]};
154  const float deltaTanLambda{std::abs(currentTracklet.tanLambda - nextTracklet.tanLambda)};
155  const float deltaPhi{std::abs(currentTracklet.phiCoordinate - nextTracklet.phiCoordinate)};
156 
157  if (deltaTanLambda < mTrkParams.CellMaxDeltaTanLambda &&
158  (deltaPhi < mTrkParams.CellMaxDeltaPhi ||
159  std::abs(deltaPhi - constants::math::TwoPi) < mTrkParams.CellMaxDeltaPhi)) {
160 
161  const float averageTanLambda{0.5f * (currentTracklet.tanLambda + nextTracklet.tanLambda)};
162  const float directionZIntersection{-averageTanLambda * firstCellCluster.rCoordinate +
163  firstCellCluster.zCoordinate};
164  const float deltaZ{std::abs(directionZIntersection - primaryVertex.z)};
165 
166  if (deltaZ < mTrkParams.CellMaxDeltaZ[iLayer]) {
167 
168  const Cluster& thirdCellCluster{
169  primaryVertexContext->getClusters()[iLayer + 2][nextTracklet.secondClusterIndex]};
170 
171  const float thirdCellClusterQuadraticRCoordinate{thirdCellCluster.rCoordinate *
172  thirdCellCluster.rCoordinate};
173 
174  const float3 secondDeltaVector{thirdCellCluster.xCoordinate - firstCellCluster.xCoordinate,
175  thirdCellCluster.yCoordinate - firstCellCluster.yCoordinate,
176  thirdCellClusterQuadraticRCoordinate -
177  firstCellClusterQuadraticRCoordinate};
178 
179  float3 cellPlaneNormalVector{math_utils::crossProduct(firstDeltaVector, secondDeltaVector)};
180 
181  const float vectorNorm{std::sqrt(cellPlaneNormalVector.x * cellPlaneNormalVector.x +
182  cellPlaneNormalVector.y * cellPlaneNormalVector.y +
183  cellPlaneNormalVector.z * cellPlaneNormalVector.z)};
184 
185  if (vectorNorm < constants::math::FloatMinThreshold ||
186  std::abs(cellPlaneNormalVector.z) < constants::math::FloatMinThreshold) {
187 
188  continue;
189  }
190 
191  const float inverseVectorNorm{1.0f / vectorNorm};
192  const float3 normalizedPlaneVector{cellPlaneNormalVector.x * inverseVectorNorm,
193  cellPlaneNormalVector.y * inverseVectorNorm,
194  cellPlaneNormalVector.z * inverseVectorNorm};
195  const float planeDistance{-normalizedPlaneVector.x * (secondCellCluster.xCoordinate - primaryVertex.x) -
196  (normalizedPlaneVector.y * secondCellCluster.yCoordinate - primaryVertex.y) -
197  normalizedPlaneVector.z * secondCellClusterQuadraticRCoordinate};
198  const float normalizedPlaneVectorQuadraticZCoordinate{normalizedPlaneVector.z * normalizedPlaneVector.z};
199  const float cellTrajectoryRadius{std::sqrt(
200  (1.0f - normalizedPlaneVectorQuadraticZCoordinate - 4.0f * planeDistance * normalizedPlaneVector.z) /
201  (4.0f * normalizedPlaneVectorQuadraticZCoordinate))};
202  const float2 circleCenter{-0.5f * normalizedPlaneVector.x / normalizedPlaneVector.z,
203  -0.5f * normalizedPlaneVector.y / normalizedPlaneVector.z};
204  const float distanceOfClosestApproach{std::abs(
205  cellTrajectoryRadius - std::sqrt(circleCenter.x * circleCenter.x + circleCenter.y * circleCenter.y))};
206 
207  if (distanceOfClosestApproach >
208  mTrkParams.CellMaxDCA[iLayer]) {
209 
210  continue;
211  }
212 
213  const float cellTrajectoryCurvature{1.0f / cellTrajectoryRadius};
214  if (iLayer > 0 &&
215  primaryVertexContext->getCellsLookupTable()[iLayer - 1][iTracklet] == constants::its::UnusedIndex) {
216 
217  primaryVertexContext->getCellsLookupTable()[iLayer - 1][iTracklet] =
218  primaryVertexContext->getCells()[iLayer].size();
219  }
220 
221  primaryVertexContext->getCells()[iLayer].emplace_back(
222  currentTracklet.firstClusterIndex, nextTracklet.firstClusterIndex, nextTracklet.secondClusterIndex,
223  iTracklet, iNextLayerTracklet, normalizedPlaneVector, cellTrajectoryCurvature);
224  }
225  }
226  }
227  }
228  }

231 void TrackerTraitsCPU::refitTracks(const std::array<std::vector<TrackingFrameInfo>, 7>& tf, std::vector<TrackITSExt>& tracks)
232 {
233  std::array<const Cell*, 5> cells;
234  for (int iLayer = 0; iLayer < 5; iLayer++) {
235  cells[iLayer] = mPrimaryVertexContext->getCells()[iLayer].data();
236  }
237  std::array<const Cluster*, 7> clusters;
238  for (int iLayer = 0; iLayer < 7; iLayer++) {
239  clusters[iLayer] = mPrimaryVertexContext->getClusters()[iLayer].data();
240  }
241  mChainRunITSTrackFit(*mChain, mPrimaryVertexContext->getRoads(), clusters, cells, tf, tracks);
242 }
