85 framework::WorkflowSpec getWorkflow(std::vector<int> const& tpcSectors, std::vector<int> const& laneConfiguration,
86  bool propagateMC, unsigned nLanes, std::string const& cfgInput, std::string const& cfgOutput,
87  int caClusterer, int zsOnTheFly, int zs10bit, float zsThreshold)
88 {
89  InputType inputType;
90 
91  try {
92  inputType = InputMap.at(cfgInput);
93  } catch (std::out_of_range&) {
94  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
95  }
96  std::vector<OutputType> outputTypes;
97  try {
98  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
99  } catch (std::out_of_range&) {
100  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
101  }
102  auto isEnabled = [&outputTypes](OutputType type) {
103  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
104  };
105 
106  if (inputType == InputType::ClustersHardware && isEnabled(OutputType::Digits)) {
107  throw std::invalid_argument("input/output type mismatch, can not produce 'digits' from 'clustershardware'");
108  }
109  if (inputType == InputType::Clusters && (isEnabled(OutputType::Digits) || isEnabled(OutputType::ClustersHardware))) {
110  throw std::invalid_argument("input/output type mismatch, can not produce 'digits', nor 'clustershardware' from 'clusters'");
111  }
112  if (inputType == InputType::ZSRaw && isEnabled(OutputType::ClustersHardware)) {
113  throw std::invalid_argument("input/output type mismatch, can not produce 'clustershardware' from 'zsraw'");
114  }
115 
116  if (inputType == InputType::ZSRaw && !caClusterer) {
117  throw std::invalid_argument("zsraw input needs caclusterer");
118  }
119  if (caClusterer && (inputType == InputType::Clusters || inputType == InputType::ClustersHardware)) {
120  throw std::invalid_argument("ca-clusterer requires digits as input");
121  }
122  if (caClusterer && (isEnabled(OutputType::ClustersHardware))) {
123  throw std::invalid_argument("ca-clusterer cannot produce clustershardware output");
124  }
125 
126  WorkflowSpec specs;
127 
128  // The OutputSpec of the PublisherSpec is configured depending on the input
129  // type. Note that the configuration of the dispatch trigger in the main file
130  // needs to be done in accordance. This means, if a new input option is added
131  // also the dispatch trigger needs to be updated.
132  if (inputType == InputType::Digits) {
133  using Type = std::vector<o2::tpc::Digit>;
134  specs.emplace_back(o2::tpc::getPublisherSpec<Type>(PublisherConf{
135  "tpc-digit-reader",
136  "o2sim",
137  {"digitbranch", "TPCDigit", "Digit branch"},
138  {"mcbranch", "TPCDigitMCTruth", "MC label branch"},
139  OutputSpec{"TPC", "DIGITS"},
140  OutputSpec{"TPC", "DIGITSMCTR"},
141  tpcSectors,
142  laneConfiguration,
143  },
144  propagateMC));
145  } else if (inputType == InputType::ClustersHardware) {
146  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
147  "tpc-clusterhardware-reader",
148  "tpcclustershardware",
149  {"databranch", "TPCClusterHw", "Branch with TPC ClustersHardware"},
150  {"mcbranch", "TPCClusterHwMCTruth", "MC label branch"},
151  OutputSpec{"TPC", "CLUSTERHW"},
152  OutputSpec{"TPC", "CLUSTERHWMCLBL"},
153  tpcSectors,
154  laneConfiguration,
155  },
156  propagateMC));
157  } else if (inputType == InputType::Clusters) {
158  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
159  "tpc-native-cluster-reader",
160  "tpcrec",
161  {"clusterbranch", "TPCClusterNative", "Branch with TPC native clusters"},
162  {"clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch"},
163  OutputSpec{"TPC", "CLUSTERNATIVE"},
164  OutputSpec{"TPC", "CLNATIVEMCLBL"},
165  tpcSectors,
166  laneConfiguration,
167  },
168  propagateMC));
169  } else if (inputType == InputType::CompClusters) {
170  // TODO: need to check if we want to store the MC labels alongside with compressed clusters
171  // for the moment reading of labels is disabled (last parameter is false)
172  // TODO: make a different publisher spec for only one output spec, for now using the
173  // PublisherSpec with only sector 0, '_0' is thus appended to the branch name
174  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
175  "tpc-compressed-cluster-reader",
176  "tpcrec",
177  {"clusterbranch", "TPCCompClusters", "Branch with TPC compressed clusters"},
178  {"clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch"},
179  OutputSpec{"TPC", "COMPCLUSTERS"},
180  OutputSpec{"TPC", "CLNATIVEMCLBL"},
181  std::vector<int>(1, 0),
182  std::vector<int>(1, 0),
183  },
184  false));
185  } else if (inputType == InputType::EncodedClusters) {
186  // TODO: need to check if we want to store the MC labels alongside with encoded clusters
187  // for the moment reading of labels is disabled (last parameter is false)
188  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
189  "tpc-encoded-cluster-reader",
190  "tpcrec",
191  {"clusterbranch", "TPCEncodedClusters", "Branch with TPC encoded clusters"},
192  {"clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch"},
193  OutputSpec{"TPC", "ENCCLUSTERS"},
194  OutputSpec{"TPC", "CLNATIVEMCLBL"},
195  std::vector<int>(1, 0),
196  std::vector<int>(1, 0),
197  },
198  false));
199  }
200 
201  // output matrix
202  // Note: the ClusterHardware format is probably a deprecated legacy format and also the
203  // ClusterDecoderRawSpec
204  bool produceCompClusters = isEnabled(OutputType::CompClusters);
205  bool produceTracks = isEnabled(OutputType::Tracks);
206  bool runTracker = produceTracks || produceCompClusters;
207  bool runHWDecoder = !caClusterer && (runTracker || isEnabled(OutputType::Clusters));
208  bool runClusterer = !caClusterer && (runHWDecoder || isEnabled(OutputType::ClustersHardware));
209  bool zsDecoder = inputType == InputType::ZSRaw;
210  bool runClusterEncoder = isEnabled(OutputType::EncodedClusters);
211  // input matrix
212  runClusterer &= inputType == InputType::Digitizer || inputType == InputType::Digits;
213  runHWDecoder &= runClusterer || inputType == InputType::ClustersHardware;
214  runTracker &= caClusterer || (runHWDecoder || inputType == InputType::Clusters);
215 
216  bool outRaw = inputType == InputType::Digits && isEnabled(OutputType::ZSRaw);
217  //bool runZSDecode = inputType == InputType::ZSRaw;
218  bool zsToDigit = inputType == InputType::ZSRaw && isEnabled(OutputType::Digits);
219 
220  WorkflowSpec parallelProcessors;
222  //
223  // clusterer process(es)
224  //
225  //
226  if (runClusterer) {
227  parallelProcessors.push_back(o2::tpc::getClustererSpec(propagateMC));
228  }
229 
231  //
232  // cluster decoder process(es)
233  //
234  //
235  if (runHWDecoder) {
236  parallelProcessors.push_back(o2::tpc::getClusterDecoderRawSpec(propagateMC));
237  }
238 
240  //
241  // set up parallel TPC lanes
242  //
243  // the parallelPipeline helper distributes the subspec ids from the lane configuration
244  // among the pipelines. All inputs and outputs of processors of one pipeline will be
245  // cloned by the number of subspecs served by this pipeline and amended with the subspecs
246  parallelProcessors = parallelPipeline(
247  parallelProcessors, nLanes,
248  [&laneConfiguration]() { return laneConfiguration.size(); },
249  [&laneConfiguration](size_t index) { return laneConfiguration[index]; });
250  specs.insert(specs.end(), parallelProcessors.begin(), parallelProcessors.end());
251 
253  //
254  // generation of processor specs for various types of outputs
255  // based on generic RootTreeWriter and MakeRootTreeWriterSpec generator
256  //
257  // -------------------------------------------------------------------------------------------
258  // the callbacks for the RootTreeWriter
259  //
260  // The generic writer needs a way to associate incoming data with the individual branches for
261  // the TPC sectors. The sector number is transmitted as part of the sector header, the callback
262  // finds the corresponding index in the vector of configured sectors
263  auto getIndex = [tpcSectors](o2::framework::DataRef const& ref) {
264  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
265  if (!tpcSectorHeader) {
266  throw std::runtime_error("TPC sector header missing in header stack");
267  }
268  if (tpcSectorHeader->sector() < 0) {
269  // special data sets, don't write
270  return ~(size_t)0;
271  }
272  size_t index = 0;
273  for (auto const& sector : tpcSectors) {
274  if (sector == tpcSectorHeader->sector()) {
275  return index;
276  }
277  ++index;
278  }
279  throw std::runtime_error("sector " + std::to_string(tpcSectorHeader->sector()) + " not configured for writing");
280  };
281  auto getName = [tpcSectors](std::string base, size_t index) {
282  return base + "_" + std::to_string(tpcSectors.at(index));
283  };
284 
285  // -------------------------------------------------------------------------------------------
286  // helper to create writer specs for different types of output
287  auto makeWriterSpec = [tpcSectors, laneConfiguration, propagateMC, getIndex, getName](const char* processName,
288  const char* defaultFileName,
289  const char* defaultTreeName,
290  auto&& databranch,
291  auto&& mcbranch,
292  bool singleBranch = false) {
293  if (tpcSectors.size() == 0) {
294  throw std::invalid_argument(std::string("writer process configuration needs list of TPC sectors"));
295  }
296 
297  auto amendInput = [tpcSectors, laneConfiguration](InputSpec& input, size_t index) {
298  input.binding += std::to_string(laneConfiguration[index]);
299  DataSpecUtils::updateMatchingSubspec(input, laneConfiguration[index]);
300  };
301  auto amendBranchDef = [laneConfiguration, amendInput, tpcSectors, getIndex, getName, singleBranch](auto&& def, bool enableMC = true) {
302  if (!singleBranch) {
303  def.keys = mergeInputs(def.keys, laneConfiguration.size(), amendInput);
304  // the branch is disabled if set to 0
305  def.nofBranches = enableMC ? tpcSectors.size() : 0;
306  def.getIndex = getIndex;
307  def.getName = getName;
308  } else {
309  // instead of the separate sector branches only one is going to be written
310  def.nofBranches = enableMC ? 1 : 0;
311  }
312  return std::move(def);
313  };
314 
315  return std::move(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
316  std::move(amendBranchDef(databranch)),
317  std::move(amendBranchDef(mcbranch, propagateMC)))());
318  };
319 
321  //
322  // a writer process for digits
323  //
324  // selected by output type 'difits'
325  if (isEnabled(OutputType::Digits) && !isEnabled(OutputType::DisableWriter)) {
326  using DigitOutputType = std::vector<o2::tpc::Digit>;
327  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
328  specs.push_back(makeWriterSpec("tpc-digits-writer",
329  inputType == InputType::ZSRaw ? "tpc-zs-digits.root" : inputType == InputType::Digits ? "tpc-filtered-digits.root" : "tpcdigits.root",
330  "o2sim",
331  BranchDefinition<DigitOutputType>{InputSpec{"data", "TPC", "DIGITS", 0},
332  "TPCDigit",
333  "digit-branch-name"},
334  BranchDefinition<MCLabelContainer>{InputSpec{"mc", "TPC", "DIGITSMCTR", 0},
335  "TPCDigitMCTruth",
336  "digitmc-branch-name"}));
337  }
338 
340  //
341  // a writer process for hardware clusters
342  //
343  // selected by output type 'clustershardware'
344  if (isEnabled(OutputType::ClustersHardware) && !isEnabled(OutputType::DisableWriter)) {
345  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
346  specs.push_back(makeWriterSpec("tpc-clusterhardware-writer",
347  inputType == InputType::ClustersHardware ? "tpc-filtered-clustershardware.root" : "tpc-clustershardware.root",
348  "tpcclustershardware",
349  BranchDefinition<const char*>{InputSpec{"data", "TPC", "CLUSTERHW", 0},
350  "TPCClusterHw",
351  "databranch"},
352  BranchDefinition<MCLabelContainer>{InputSpec{"mc", "TPC", "CLUSTERHWMCLBL", 0},
353  "TPCClusterHwMCTruth",
354  "mcbranch"}));
355  }
356 
358  //
359  // a writer process for TPC native clusters
360  //
361  // selected by output type 'clusters'
362  if (isEnabled(OutputType::Clusters) && !isEnabled(OutputType::DisableWriter)) {
363  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
364  // if the caClusterer is enabled, only one data set with the full TPC is produced, and the writer
365  // is configured to write one single branch
366  specs.push_back(makeWriterSpec("tpc-native-cluster-writer",
367  inputType == InputType::Clusters ? "tpc-filtered-native-clusters.root" : "tpc-native-clusters.root",
368  "tpcrec",
369  BranchDefinition<const char*>{InputSpec{"data", ConcreteDataTypeMatcher{"TPC", "CLUSTERNATIVE"}},
370  "TPCClusterNative",
371  "databranch"},
372  BranchDefinition<MCLabelContainer>{InputSpec{"mc", ConcreteDataTypeMatcher{"TPC", "CLNATIVEMCLBL"}},
373  "TPCClusterNativeMCTruth",
374  "mcbranch"},
375  caClusterer));
376  }
377 
378  if (zsOnTheFly) {
379  specs.emplace_back(o2::tpc::getZSEncoderSpec(laneConfiguration, zs10bit, zsThreshold, outRaw));
380  }
381 
382  if (zsToDigit) {
383  specs.emplace_back(o2::tpc::getZStoDigitsSpec(laneConfiguration));
384  }
385 
387  //
388  // tracker process
389  //
390  // selected by output type 'tracks'
391  if (runTracker) {
392  specs.emplace_back(o2::tpc::getCATrackerSpec(ca::Config{
393  propagateMC ? ca::Operation::ProcessMC : ca::Operation::Noop,
394  caClusterer ? ca::Operation::CAClusterer : ca::Operation::Noop,
395  zsDecoder ? ca::Operation::ZSDecoder : ca::Operation::Noop,
396  zsOnTheFly ? ca::Operation::ZSOnTheFly : ca::Operation::Noop,
397  produceTracks ? ca::Operation::OutputTracks : ca::Operation::Noop,
398  produceCompClusters ? ca::Operation::OutputCompClusters : ca::Operation::Noop,
399  runClusterEncoder ? ca::Operation::OutputCompClustersFlat : ca::Operation::Noop,
400  isEnabled(OutputType::Clusters) && caClusterer ? ca::Operation::OutputCAClusters : ca::Operation::Noop,
401  },
402  laneConfiguration));
403  }
404 
406  //
407  // tracker process
408  //
409  // selected by output type 'encoded-clusters'
410  if (runClusterEncoder) {
411  specs.emplace_back(o2::tpc::getEntropyEncoderSpec(!runTracker));
412  }
413 
415  //
416  // a writer process for tracks
417  //
418  // selected by output type 'tracks'
419  if (produceTracks && !isEnabled(OutputType::DisableWriter)) {
420  // defining the track writer process using the generic RootTreeWriter and generator tool
421  //
422  // defaults
423  const char* processName = "tpc-track-writer";
424  const char* defaultFileName = "tpctracks.root";
425  const char* defaultTreeName = "tpcrec";
426 
427  //branch definitions for RootTreeWriter spec
428  using TrackOutputType = std::vector<o2::tpc::TrackTPC>;
429 
430  using ClusRefsOutputType = std::vector<o2::tpc::TPCClRefElem>;
431 
432  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
433  // a spectator callback which will be invoked by the tree writer with the extracted object
434  // we are using it for printing a log message
435  auto logger = BranchDefinition<TrackOutputType>::Spectator([](TrackOutputType const& tracks) {
436  LOG(INFO) << "writing " << tracks.size() << " track(s)";
437  });
438  auto tracksdef = BranchDefinition<TrackOutputType>{InputSpec{"inputTracks", "TPC", "TRACKS", 0}, //
439  "TPCTracks", "track-branch-name", //
440  1, //
441  logger}; //
442  auto clrefdef = BranchDefinition<ClusRefsOutputType>{InputSpec{"inputClusRef", "TPC", "CLUSREFS", 0}, //
443  "ClusRefs", "trackclusref-branch-name"}; //
444  auto mcdef = BranchDefinition<MCLabelContainer>{InputSpec{"mcinput", "TPC", "TRACKSMCLBL", 0}, //
445  "TPCTracksMCTruth", //
446  (propagateMC ? 1 : 0), //
447  "trackmc-branch-name"}; //
448 
449  // depending on the MC propagation flag, branch definition for MC labels is disabled
450  specs.push_back(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
451  std::move(tracksdef), std::move(clrefdef),
452  std::move(mcdef))());
453  }
454 
456  //
457  // a writer process for compressed clusters container
458  //
459  // selected by output type 'compressed-clusters'
460  if (produceCompClusters && !isEnabled(OutputType::DisableWriter)) {
461  // defining the track writer process using the generic RootTreeWriter and generator tool
462  //
463  // defaults
464  const char* processName = "tpc-compcluster-writer";
465  const char* defaultFileName = "tpc-compclusters.root";
466  const char* defaultTreeName = "tpcrec";
467 
468  //branch definitions for RootTreeWriter spec
469  using CCluSerializedType = ROOTSerialized<CompressedClustersROOT>;
470  auto ccldef = BranchDefinition<CCluSerializedType>{InputSpec{"inputCompCl", "TPC", "COMPCLUSTERS"}, //
471  "TPCCompClusters_0", "compcluster-branch-name"}; //
472 
473  specs.push_back(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName, //
474  std::move(ccldef))()); //
475  }
476 
477  return std::move(specs);
478 }
