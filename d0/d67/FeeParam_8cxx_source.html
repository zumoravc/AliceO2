202 //_____________________________________________________________________________
203 int FeeParam::getPadColFromADC(int irob, int imcm, int iadc) const
204 {
205  //
206  // Return which pad is connected to this adc channel.
207  //
208  // Return virtual pad number even if ADC is outside chamber
209  // to keep compatibility of data processing at the edge MCM.
210  // User has to check that this is in the chamber if it is essential.
211  // Return -100 if iadc is invalid.
212  //
213  // Caution: ADC ordering in the online data is opposite to the pad column ordering.
214  // And it is not one-by-one correspondence. Precise drawing can be found in:
215  // http://wiki.kip.uni-heidelberg.de/ti/TRD/index.php/Image:ROB_MCM_numbering.pdf
216  //
217 
218  if (iadc < 0 || iadc > mgkNadcMcm)
219  return -100;
220  int mcmcol = imcm % mgkNmcmRobInCol + getRobSide(irob) * mgkNmcmRobInCol; // MCM column number on ROC [0..7]
221  int padcol = mcmcol * mgkNcolMcm + mgkNcolMcm + 1 - iadc;
222  if (padcol < 0 || padcol >= mgkNcol) {
223  return -1; // this is commented because of reason above OK
224  }
225  return padcol;
226 }

260 int FeeParam::getMCMfromSharedPad(int irow, int icol) const
261 {
262  //
263  // Return on which MCM this pad is directry connected.
264  // Return -1 for error.
265  //
266 
267  if (irow < 0 || icol < 0 || irow > mgkNrowC1 || icol > mgkNcol + 8 * 3)

270  int adc = 20 - (icol % 18) - 1;
271  switch (adc) {
272  case 2:
273  icol += 5;
274  break;
275  case 18:

282  icol += 0;
283  break;

285 
286  return (icol % (mgkNcol / 2)) / mgkNcolMcm + mgkNmcmRobInCol * (irow % mgkNmcmRobInRow);

347 int FeeParam::extAliToAli(unsigned int dest, unsigned short linkpair, unsigned short rocType, int* mcmList, int listSize)
348 {
349  // Converts an extended ALICE ID which identifies a single MCM or a group of MCMs to
350  // the corresponding list of MCMs. Only broadcasts (127) are encoded as 127
351  // The return value is the number of MCMs in the list
352 
353  mcmList[0] = -1;
354 
355  short nmcm = 0;
356  unsigned int mcm, rob, robAB;
357  unsigned int cmA = 0, cmB = 0; // Chipmask for each A and B side
358 
359  // Default chipmask for 4 linkpairs (each bit correponds each alice-mcm)
360  static const unsigned int gkChipmaskDefLp[4] = {0x1FFFF, 0x1FFFF, 0x3FFFF, 0x1FFFF};
361 
362  rob = dest >> 7; // Extract ROB pattern from dest.
363  mcm = dest & 0x07F; // Extract MCM pattern from dest.
364  robAB = getRobAB(rob, linkpair); // Get which ROB sides are selected.
365 
366  // Abort if no ROB is selected
367  if (robAB == 0) {
368  return 0;
369  }
370 
371  // Special case
372  if (mcm == 127) {
373  if (robAB == 3) { // This is very special 127 can stay only if two ROBs are selected
374  mcmList[0] = 127; // broadcase to ALL
375  mcmList[1] = -1;
376  return 1;
377  }
378  cmA = cmB = 0x3FFFF;
379  } else if ((mcm & 0x40) != 0) { // If top bit is 1 but not 127, this is chip group.
380  if ((mcm & 0x01) != 0) {
381  cmA |= 0x04444;
382  cmB |= 0x04444;
383  } // chip_cmrg
384  if ((mcm & 0x02) != 0) {
385  cmA |= 0x10000;
386  cmB |= 0x10000;
387  } // chip_bmrg
388  if ((mcm & 0x04) != 0 && rocType == 0) {
389  cmA |= 0x20000;
390  cmB |= 0x20000;
391  } // chip_hm3
392  if ((mcm & 0x08) != 0 && rocType == 1) {
393  cmA |= 0x20000;
394  cmB |= 0x20000;
395  } // chip_hm4
396  if ((mcm & 0x10) != 0) {
397  cmA |= 0x01111;
398  cmB |= 0x08888;
399  } // chip_edge
400  if ((mcm & 0x20) != 0) {
401  cmA |= 0x0aaaa;
402  cmB |= 0x03333;
403  } // chip_norm
404  } else { // Otherwise, this is normal chip ID, turn on only one chip.
405  cmA = 1 << mcm;
406  cmB = 1 << mcm;
407  }
408 
409  // Mask non-existing MCMs
410  cmA &= gkChipmaskDefLp[linkpair];
411  cmB &= gkChipmaskDefLp[linkpair];
412  // Remove if only one side is selected
413  if (robAB == 1)
414  cmB = 0;
415  if (robAB == 2)
416  cmA = 0;
417  if (robAB == 4 && linkpair != 2)
418  cmA = cmB = 0; // Restrict to only T3A and T3B
419 
420  // Finally convert chipmask to list of slaves
421  nmcm = chipmaskToMCMlist(cmA, cmB, linkpair, mcmList, listSize);
422 
423  return nmcm;
424 }

426 short FeeParam::getRobAB(unsigned short robsel, unsigned short linkpair)
427 {
428  // Converts the ROB part of the extended ALICE ID to robs
429 
430  if ((robsel & 0x8) != 0) { // 1000 .. direct ROB selection. Only one of the 8 ROBs are used.
431  robsel = robsel & 7;
432  if ((robsel % 2) == 0 && (robsel / 2) == linkpair)
433  return 1; // Even means A side (position 0,2,4,6)
434  if ((robsel % 2) == 1 && (robsel / 2) == linkpair)
435  return 2; // Odd means B side (position 1,3,5,7)
436  return 0;
437  }
438 
439  // ROB group
440  if (robsel == 0) {
441  return 3;
442  } // Both ROB
443  if (robsel == 1) {
444  return 1;
445  } // A-side ROB
446  if (robsel == 2) {
447  return 2;
448  } // B-side ROB
449  if (robsel == 3) {
450  return 3;
451  } // Both ROB
452  if (robsel == 4) {
453  return 4;
454  } // Only T3A and T3B
455  // Other number 5 to 7 are ignored (not defined)
456 
457  return 0;
458 }

460 short FeeParam::chipmaskToMCMlist(unsigned int cmA, unsigned int cmB, unsigned short linkpair, int* mcmList, int listSize)
461 {
462  // Converts the chipmask to a list of MCMs
463 
464  short nmcm = 0;
465  short i;
466  for (i = 0; i < 18; i++) { // 18: number of MCMs on a ROB
467  if ((cmA & (1 << i)) != 0 && nmcm < listSize) {
468  mcmList[nmcm] = ((linkpair * 2) << 7) | i;
469  ++nmcm;
470  }
471  if ((cmB & (1 << i)) != 0 && nmcm < listSize) {
472  mcmList[nmcm] = ((linkpair * 2 + 1) << 7) | i;
473  ++nmcm;
474  }
475  }
476 
477  mcmList[nmcm] = -1;
478  return nmcm;
479 }

500 void FeeParam::createPad2MCMLookUpTable()
501 {
502 
503  //
504  // Initializes the Look Up Table to relate
505  // pad numbering and mcm channel numbering
506  //
507  LOG(info) << "Entering : " << __FILE__ << ":" << __func__ << ":" << __LINE__;
508  if (!mgLUTPadNumberingFilled) {

512  memset(&mgLUTPadNumbering[0], 0, sizeof(mgLUTPadNumbering[0]) * getNcol());
513  for (int mcm = 0; mcm < 8; mcm++) {
514  int lowerlimit = 0 + mcm * 18;
515  int upperlimit = 18 + mcm * 18;
516  int shiftposition = 1 + 3 * mcm;
517  for (int index = lowerlimit; index < upperlimit; index++) {
518  mgLUTPadNumbering[index] = index + shiftposition;
519  }
520  }
521  mgLUTPadNumberingFilled = true;
522  }
523  LOG(info) << "Leaving : " << __FILE__ << ":" << __func__ << ":" << __LINE__;
524 }

526 int FeeParam::getDyCorrection(int det, int rob, int mcm) const
527 {
528  // calculate the correction of the deflection
529  // i.e. Lorentz angle and tilt correction (if active)
530 
531  int layer = det % 6;
532 
533  float dyTilt = (mgDriftLength * std::tan(mgTiltingAngle[layer] * M_PI / 180.) *
534  getLocalZ(det, rob, mcm) * mgInvX[layer]);
535 
536  // calculate Lorentz correction
537  float dyCorr = -mOmegaTau * mgDriftLength;
538 
539  if (mTiltCorr)
540  dyCorr += dyTilt; // add tilt correction
541 
542  return (int)TMath::Nint(dyCorr * mgScalePad * mgInvWidthPad[layer]);
543 }

545 void FeeParam::getDyRange(int det, int rob, int mcm, int ch,
546  int& dyMinInt, int& dyMaxInt) const
547 {
548  // calculate the deflection range in which tracklets are accepted
549 
550  dyMinInt = mgDyMin;
551  dyMaxInt = mgDyMax;
552 
553  // deflection cut is considered for |B| > 0.1 T only
554  if (std::abs(mMagField) < 0.1)
555  return;
556 
557  float e = 0.30;
558 
559  float maxDeflTemp = getPerp(det, rob, mcm, ch) / 2. * // Sekante/2 (cm)
560  (e * 1e-2 * std::abs(mMagField) * mInvPtMin); // 1/R (1/cm)
561 
562  float phi = getPhi(det, rob, mcm, ch);
563  if (maxDeflTemp < std::cos(phi)) {
564  float maxDeflAngle = std::asin(maxDeflTemp);
565 
566  float dyMin = (mgDriftLength *
567  std::tan(phi - maxDeflAngle));
568 
569  dyMinInt = int(dyMin / mgBinDy);
570  // clipping to allowed range
571  if (dyMinInt < mgDyMin)
572  dyMinInt = mgDyMin;
573  else if (dyMinInt > mgDyMax)
574  dyMinInt = mgDyMax;
575 
576  float dyMax = (mgDriftLength *
577  std::tan(phi + maxDeflAngle));
578 
579  dyMaxInt = int(dyMax / mgBinDy);
580  // clipping to allowed range
581  if (dyMaxInt > mgDyMax)
582  dyMaxInt = mgDyMax;
583  else if (dyMaxInt < mgDyMin)
584  dyMaxInt = mgDyMin;
585  } else if (maxDeflTemp < 0.) {
586  // this must not happen
587  printf("Inconsistent calculation of sin(alpha): %f\n", maxDeflTemp);
588  } else {
589  // TRD is not reached at the given pt threshold
590  // max range
591  }
592 
593  if ((dyMaxInt - dyMinInt) <= 0) {
594  LOG(info) << "strange dy range: [" << dyMinInt << "," << dyMaxInt << "], using max range now";
595  dyMaxInt = mgDyMax;
596  dyMinInt = mgDyMin;
597  }
598 }

600 float FeeParam::getElongation(int det, int rob, int mcm, int ch) const
601 {
602  // calculate the ratio of the distance to the primary vertex and the
603  // distance in x-direction for the given ADC channel
604 
605  int layer = det % 6;
606 
607  float elongation = std::abs(getDist(det, rob, mcm, ch) * mgInvX[layer]);
608

611  elongation = 1.;
612  }
613  return elongation;

616 void FeeParam::getCorrectionFactors(int det, int rob, int mcm, int ch,
617  unsigned int& cor0, unsigned int& cor1, float gain) const
618 {
619  // calculate the gain correction factors for the given ADC channel

630  cor1 = (unsigned int)(mScaleQ1 * Invgain);
