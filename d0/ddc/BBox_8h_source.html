28 template <typename T>
29 class BBox
30 {
31  public:
32  BBox(T xmin, T ymin, T xmax, T ymax) : mXmin{xmin}, mXmax{xmax}, mYmin{ymin}, mYmax{ymax}
33  {
34  if (xmin >= xmax || ymin >= ymax) {
35  throw std::invalid_argument("BBox should be created valid (xmin<xmax and ymin<ymax)");
36  }
37  }
38 
39  T xmin() const { return mXmin; }
40 
41  T xmax() const { return mXmax; }
42 
43  T ymin() const { return mYmin; }
44 
45  T ymax() const { return mYmax; }
46 
47  T xcenter() const { return (xmin() + xmax()) / T{2}; }
48 
49  T ycenter() const { return (ymin() + ymax()) / T{2}; }
50 
51  T width() const { return xmax() - xmin(); }
52 
53  T height() const { return ymax() - ymin(); }
54 
55  friend bool operator==(const BBox& lhs, const BBox& rhs)
56  {
57  return impl::areEqual(lhs.xmin(), rhs.xmin()) && impl::areEqual(lhs.ymin(), rhs.ymin()) &&
58  impl::areEqual(lhs.xmax(), rhs.xmax()) && impl::areEqual(lhs.ymax(), rhs.ymax());
59  }
60 
61  friend bool operator!=(const BBox& lhs, const BBox& rhs) { return !(rhs == lhs); }
62 
63  friend std::ostream& operator<<(std::ostream& os, const BBox& box)
64  {
65  os << "mTopLeft: " << box.xmin() << "," << box.ymax() << " mBottomRight: " << box.xmax() << "," << box.ymin();
66  return os;
67  }
68 
69  bool contains(const BBox<T>& a) const
70  {
71  return !(a.xmin() < xmin() || a.xmax() > xmax() || a.ymin() < ymin() || a.ymax() > ymax());
72  }
73 
74  private:
75  T mXmin;
76  T mXmax;
77  T mYmin;
78  T mYmax;
79 };

93 }
