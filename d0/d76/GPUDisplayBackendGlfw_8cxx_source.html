138 void GPUDisplayBackendGlfw::GetKey(int key, int scancode, int mods, int& keyOut, int& keyPressOut)
139 {
140  int specialKey = GetKey(key);
141  const char* str = glfwGetKeyName(key, scancode);
142  char localeKey = str ? str[0] : 0;
143  if ((mods & GLFW_MOD_SHIFT) && localeKey >= 'a' && localeKey <= 'z') {
144  localeKey += 'A' - 'a';
145  }
146  // GPUInfo("Key: key %d (%c) -> %d (%c) special %d (%c)", key, (char) key, (int) localeKey, localeKey, specialKey, (char) specialKey);
147 
148  if (specialKey) {
149  keyOut = keyPressOut = specialKey;
150  } else {
151  keyOut = keyPressOut = localeKey;
152  if (keyPressOut >= 'a' && keyPressOut <= 'z') {
153  keyPressOut += 'A' - 'a';
154  }
155  }
156 }

160 void GPUDisplayBackendGlfw::key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
161 {
162  int handleKey = 0, keyPress = 0;
163  GetKey(key, scancode, mods, handleKey, keyPress);
164  if (action == GLFW_PRESS) {
165  me->mKeys[keyPress] = true;
166  me->mKeysShift[keyPress] = mods & GLFW_MOD_SHIFT;
167  } else if (action == GLFW_RELEASE) {
168  if (me->mKeys[keyPress]) {
169  me->HandleKeyRelease(handleKey);
170  }
171  me->mKeys[keyPress] = false;
172  me->mKeysShift[keyPress] = false;
173  }
174 }

177 {

184  me->mMouseDnX = me->mouseMvX;
185  me->mMouseDnY = me->mouseMvY;
186  } else if (action == GLFW_RELEASE) {
187  if (button == 0) {
188  me->mMouseDn = false;
189  } else if (button == 1) {
190  me->mMouseDnR = false;

219 int GPUDisplayBackendGlfw::OpenGLMain()
220 {
221  me = this;
222 
223  if (!glfwInit()) {
224  fprintf(stderr, "Error initializing glfw\n");
225  return (-1);
226  }
227  glfwSetErrorCallback(error_callback);
228 
229  glfwWindowHint(GLFW_MAXIMIZED, 1);
230  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, GL_MIN_VERSION_MAJOR);
231  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, GL_MIN_VERSION_MINOR);
232  glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, 0);
233  glfwWindowHint(GLFW_OPENGL_PROFILE, GPUCA_DISPLAY_OPENGL_CORE_FLAGS ? GLFW_OPENGL_CORE_PROFILE : GLFW_OPENGL_COMPAT_PROFILE);
234  mWindow = glfwCreateWindow(INIT_WIDTH, INIT_HEIGHT, GL_WINDOW_NAME, nullptr, nullptr);
235  if (!mWindow) {
236  fprintf(stderr, "Error creating glfw window\n");
237  glfwTerminate();
238  return (-1);
239  }
240  glfwMakeContextCurrent(mWindow);
241 
242  glfwSetKeyCallback(mWindow, key_callback);
243  glfwSetMouseButtonCallback(mWindow, mouseButton_callback);
244  glfwSetScrollCallback(mWindow, scroll_callback);
245  glfwSetCursorPosCallback(mWindow, cursorPos_callback);
246  glfwSetWindowSizeCallback(mWindow, resize_callback);
247 
248  pthread_mutex_lock(&mSemLockExit);
249  mGlfwRunning = true;
250  pthread_mutex_unlock(&mSemLockExit);
251 
252  if (GPUDisplayExtInit()) {
253  fprintf(stderr, "Error initializing GL extension wrapper\n");
254  return (-1);
255  }
256 
257 #if defined(GPUCA_O2_LIB) && !defined(GPUCA_DISPLAY_GL3W)
258  if (gl3wInit()) {
259  fprintf(stderr, "Error initializing gl3w (2)\n");
260  return (-1); // Hack: We have to initialize gl3w as well, as the DebugGUI uses it.
261  }
262 #endif
263 
264  if (InitGL()) {
265  fprintf(stderr, "Error in OpenGL initialization\n");
266  return (1);
267  }
268 
269 #ifdef GPUCA_O2_LIB
270  ImGui_ImplGlfwGL3_Init(mWindow, false);
271 #endif
272 
273 #ifdef GPUCA_O2_LIB
274  while (o2::framework::pollGUI(mWindow, DisplayLoop)) {
275  }
276 #else
277  while (!glfwWindowShouldClose(mWindow)) {
278  HandleSendKey();
279  if (DrawGLScene()) {
280  fprintf(stderr, "Error drawing GL scene\n");
281  return (1);
282  }
283  glfwSwapBuffers(mWindow);
284  glfwPollEvents();
285  }
286 #endif
287 
288  mDisplayControl = 2;
289  pthread_mutex_lock(&mSemLockExit);
290 #ifdef GPUCA_O2_LIB
291  ImGui_ImplGlfwGL3_Shutdown();
292 #endif
293  glfwDestroyWindow(mWindow);
294  glfwTerminate();
295  mGlfwRunning = false;
296  pthread_mutex_unlock(&mSemLockExit);
297 
298  return 0;
299 }
