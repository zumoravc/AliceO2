33 class TFDispatcher : public o2::framework::Task
34 {
35  public:
36  void init(o2::framework::InitContext& ic) final
37  {
38  mMaxTF = ic.options().get<int64_t>("max-timeframes");
39  }
40 
41  void run(o2::framework::ProcessingContext& pc) final
42  {
43  for (auto& input : pc.inputs()) {
44  auto tfid = header::get<o2::framework::DataProcessingHeader*>(input.header)->startTime;
45  if (tfid >= mMaxTF) {
46  LOG(INFO) << "Data generator reached TF " << tfid << ", stopping";
47  pc.services().get<o2::framework::ControlService>().endOfStream();
48  pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
49  break;
50  }
51  }
52  int size = 100 + gRandom->Integer(100); // push dummy output
53  pc.outputs().snapshot(o2::framework::OutputRef{"output", 0}, size);
54  }
55 
56  private:
57  uint64_t mMaxTF = 1;
58 };

60 class TFProcessor : public o2::framework::Task
61 {
62  public:
63  void init(o2::framework::InitContext& ic) final
64  {
65  mDevCopy = ic.services().get<const o2::framework::DeviceSpec>().inputTimesliceId;
66  gRandom->SetSeed(mDevCopy);
67  mMeanLatency = std::max(1, ic.options().get<int>("mean-latency"));
68  mLatencyRMS = std::max(1, ic.options().get<int>("latency-spread"));
69  LOG(INFO) << "TFProcessorCopy: " << mDevCopy << " MeanLatency: " << mMeanLatency << " LatencyRMS: " << mLatencyRMS;
70  }
71 
72  void run(o2::framework::ProcessingContext& pc) final
73  {
74  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
75  // introduceDelay
76  uint32_t delay = std::abs(gRandom->Gaus(mMeanLatency, mLatencyRMS));
77  LOG(INFO) << "TFProcessorCopy: " << mDevCopy << " Simulate latency of " << delay << " mcs for TF " << tfcounter;
78  usleep(delay);
79 
80  // push dummy output
81  auto size = pc.inputs().get<int>("input");
82  auto& output = pc.outputs().make<std::vector<o2::dataformats::CalibInfoTOF>>(o2::framework::OutputRef{"output", 0});
83  output.reserve(size);
84 
85  double clockShift = 1e3 * std::sin(tfcounter / 100. * o2::constants::math::PI);
86 
87  for (int i = size; i--;) {
88  output.emplace_back(gRandom->Integer(o2::tof::Geo::NCHANNELS), 0, gRandom->Gaus(clockShift, 100.), 0, 0);
89  }
90  }
91 
92  private:
93  int mDevCopy = 0;
94  uint32_t mMeanLatency = 0;
95  uint32_t mLatencyRMS = 1;
96 };
