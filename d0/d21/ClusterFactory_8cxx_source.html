46 template <class InputType>
47 o2::emcal::AnalysisCluster ClusterFactory<InputType>::buildCluster(int clusterIndex) const
48 {
49  if (clusterIndex >= mClustersContainer.size())
50  throw ClusterRangeException(clusterIndex, mClustersContainer.size());
51 
52  o2::emcal::AnalysisCluster clusterAnalysis;
53  clusterAnalysis.setID(clusterIndex);
54 
55  int firstCellIndex = mClustersContainer[clusterIndex].getCellIndexFirst();
56  int nCells = mClustersContainer[clusterIndex].getNCells();
57 
58  gsl::span<int> inputsIndices = gsl::span<int>(&mCellsIndices[firstCellIndex], nCells);
59 
60  // First calculate the index of input with maximum amplitude and get
61  // the supermodule number where it sits.
62 
63  auto [inputIndMax, inputEnergyMax, cellAmp] = getMaximalEnergyIndex(inputsIndices);
64 
65  clusterAnalysis.setIndMaxInput(inputIndMax);
66 
67  clusterAnalysis.setE(cellAmp);
68 
69  mSuperModuleNumber = mGeomPtr->GetSuperModuleNumber(mInputsContainer[inputIndMax].getTower());
70 
71  clusterAnalysis.setNCells(inputsIndices.size());
72 
73  std::vector<unsigned short> cellsIdices;
74 
75  for (auto cellIndex : inputsIndices) {
76  cellsIdices.push_back(cellIndex);
77  }
78 
79  clusterAnalysis.setCellsIndices(cellsIdices);
80 
81  // evaluate global and local position
82  evalGlobalPosition(inputsIndices, clusterAnalysis);
83  evalLocalPosition(inputsIndices, clusterAnalysis);
84 
85  // evaluate shower parameters
86  evalElipsAxis(inputsIndices, clusterAnalysis);
87  evalDispersion(inputsIndices, clusterAnalysis);
88 
89  evalCoreEnergy(inputsIndices, clusterAnalysis);
90  evalTime(inputsIndices, clusterAnalysis);
91 
92  // TODO to be added at a later stage
93  //evalPrimaries(inputsIndices, clusterAnalysis);
94  //evalParents(inputsIndices, clusterAnalysis);
95 
96  // TODO to be added at a later stage
97  // Called last because it sets the global position of the cluster?
98  // Do not call it when recalculating clusters out of standard reconstruction
99  //if (!mJustCluster)
100  // evalLocal2TrackingCSTransform();
101 
102  return clusterAnalysis;
103 }

109 template <class InputType>
110 void ClusterFactory<InputType>::evalDispersion(gsl::span<int> inputsIndices, AnalysisCluster& clusterAnalysis) const
111 {
112  double d = 0., wtot = 0.;
113  int nstat = 0;
114 
115  // Calculates the dispersion in cell units
116  double etaMean = 0.0, phiMean = 0.0;
117 
118  // Calculate mean values
119  for (auto iInput : inputsIndices) {
120 
121  if (clusterAnalysis.E() > 0 && mInputsContainer[iInput].getEnergy() > 0) {
122  auto [nSupMod, nModule, nIphi, nIeta] = mGeomPtr->GetCellIndex(mInputsContainer[iInput].getTower());
123  auto [iphi, ieta] = mGeomPtr->GetCellPhiEtaIndexInSModule(nSupMod, nModule, nIphi, nIeta);
124 
125  // In case of a shared cluster, index of SM in C side, columns start at 48 and ends at 48*2
126  // C Side impair SM, nSupMod%2=1; A side pair SM nSupMod%2=0
127  if (mSharedCluster && nSupMod % 2)
128  ieta += EMCAL_COLS;
129 
130  double etai = (double)ieta;
131  double phii = (double)iphi;
132  double w = TMath::Max(0., mLogWeight + TMath::Log(mInputsContainer[iInput].getEnergy() / clusterAnalysis.E()));
133 
134  if (w > 0.0) {
135  phiMean += phii * w;
136  etaMean += etai * w;
137  wtot += w;
138  }
139  }
140  }
141 
142  if (wtot > 0) {
143  phiMean /= wtot;
144  etaMean /= wtot;
145  } else
146  LOG(ERROR) << Form("Wrong weight %f\n", wtot);
147 
148  // Calculate dispersion
149  for (auto iInput : inputsIndices) {
150 
151  if (clusterAnalysis.E() > 0 && mInputsContainer[iInput].getEnergy() > 0) {
152  auto [nSupMod, nModule, nIphi, nIeta] = mGeomPtr->GetCellIndex(mInputsContainer[iInput].getTower());
153  auto [iphi, ieta] = mGeomPtr->GetCellPhiEtaIndexInSModule(nSupMod, nModule, nIphi, nIeta);
154 
155  // In case of a shared cluster, index of SM in C side, columns start at 48 and ends at 48*2
156  // C Side impair SM, nSupMod%2=1; A side pair SM, nSupMod%2=0
157  if (mSharedCluster && nSupMod % 2)
158  ieta += EMCAL_COLS;
159 
160  double etai = (double)ieta;
161  double phii = (double)iphi;
162  double w = TMath::Max(0., mLogWeight + TMath::Log(mInputsContainer[iInput].getEnergy() / clusterAnalysis.E()));
163 
164  if (w > 0.0) {
165  nstat++;
166  d += w * ((etai - etaMean) * (etai - etaMean) + (phii - phiMean) * (phii - phiMean));
167  }
168  }
169  }
170 
171  if (wtot > 0 && nstat > 1)
172  d /= wtot;
173  else
174  d = 0.;
175 
176  clusterAnalysis.setDispersion(TMath::Sqrt(d));
177 }

182 template <class InputType>
183 void ClusterFactory<InputType>::evalLocalPosition(gsl::span<int> inputsIndices, AnalysisCluster& clusterAnalysis) const
184 {
185 
186  int nstat = 0;
187 
188  double dist = tMaxInCm(double(clusterAnalysis.E()));
189 
190  double clXYZ[3] = {0., 0., 0.}, clRmsXYZ[3] = {0., 0., 0.}, xyzi[3], wtot = 0., w = 0.;
191 
192  for (auto iInput : inputsIndices) {
193 
194  try {
195  mGeomPtr->RelPosCellInSModule(mInputsContainer[iInput].getTower(), dist).GetCoordinates(xyzi[0], xyzi[1], xyzi[2]);
196  } catch (InvalidCellIDException& e) {
197  LOG(ERROR) << e.what();
198  continue;
199  }
200 
201  //Temporal patch, due to mapping problem, need to swap "y" in one of the 2 SM, although no effect in position calculation. GCB 05/2010
202  if (mSharedCluster && mSuperModuleNumber != mGeomPtr->GetSuperModuleNumber(mInputsContainer[iInput].getTower()))
203  xyzi[1] *= -1;
204 
205  if (mLogWeight > 0.0)
206  w = TMath::Max(0., mLogWeight + TMath::Log(mInputsContainer[iInput].getEnergy() / clusterAnalysis.E()));
207  else
208  w = mInputsContainer[iInput].getEnergy(); // just energy
209 
210  if (w > 0.0) {
211  wtot += w;
212  nstat++;
213 
214  for (int i = 0; i < 3; i++) {
215  clXYZ[i] += (w * xyzi[i]);
216  clRmsXYZ[i] += (w * xyzi[i] * xyzi[i]);
217  }
218  } // w > 0
219  } // dig loop
220 
221  // cout << " wtot " << wtot << endl;
222 
223  if (wtot > 0) {
224  // xRMS = TMath::Sqrt(x2m - xMean*xMean);
225  for (int i = 0; i < 3; i++) {
226  clXYZ[i] /= wtot;
227 
228  if (nstat > 1) {
229  clRmsXYZ[i] /= (wtot * wtot);
230  clRmsXYZ[i] = clRmsXYZ[i] - clXYZ[i] * clXYZ[i];
231 
232  if (clRmsXYZ[i] > 0.0)
233  clRmsXYZ[i] = TMath::Sqrt(clRmsXYZ[i]);
234  else
235  clRmsXYZ[i] = 0;
236  } else
237  clRmsXYZ[i] = 0;
238  }
239  } else {
240  for (int i = 0; i < 3; i++) {
241  clXYZ[i] = clRmsXYZ[i] = -1.;
242  }
243  }
244 
245  clusterAnalysis.setLocalPosition(Point3D<float>(clXYZ[0], clXYZ[1], clXYZ[2]));
246 }

395 template <class InputType>
396 void ClusterFactory<InputType>::getDeffW0(const double esum, double& deff, double& w0)
397 {
398  double e = 0.0;
399  const double kdp0 = 9.25147, kdp1 = 1.16700; // Hard coded now
400  const double kwp0 = 4.83713, kwp1 = -2.77970e-01, kwp2 = 4.41116;
401 
402  // No extrapolation here
403  e = esum < 0.5 ? 0.5 : esum;
404  e = e > 100. ? 100. : e;
405 
406  deff = kdp0 + kdp1 * TMath::Log(e);
407  w0 = kwp0 / (1. + TMath::Exp(kwp1 * (e + kwp2)));
408 }

418 template <class InputType>
419 void ClusterFactory<InputType>::evalCoreEnergy(gsl::span<int> inputsIndices, AnalysisCluster& clusterAnalysis) const
420 {
421 
422  float coreEnergy = 0.;
423 
424  if (!clusterAnalysis.getLocalPosition().Mag2())
425  evalLocalPosition(inputsIndices, clusterAnalysis);
426 
427  double phiPoint = clusterAnalysis.getLocalPosition().Phi();
428  double etaPoint = clusterAnalysis.getLocalPosition().Eta();
429  for (auto iInput : inputsIndices) {
430 
431  auto [eta, phi] = mGeomPtr->EtaPhiFromIndex(mInputsContainer[iInput].getTower());
432  phi = phi * TMath::DegToRad();
433 
434  double distance = TMath::Sqrt((eta - etaPoint) * (eta - etaPoint) + (phi - phiPoint) * (phi - phiPoint));
435 
436  if (distance < mCoreRadius)
437  coreEnergy += mInputsContainer[iInput].getEnergy();
438  }
439  clusterAnalysis.setCoreEnergy(coreEnergy);
440 }

446 template <class InputType>
447 void ClusterFactory<InputType>::evalElipsAxis(gsl::span<int> inputsIndices, AnalysisCluster& clusterAnalysis) const
448 {
449  TString gn(mGeomPtr->GetName());
450 
451  double wtot = 0.;
452  double x = 0.;
453  double z = 0.;
454  double dxx = 0.;
455  double dzz = 0.;
456  double dxz = 0.;
457 
458  std::array<float, 2> lambda;
459 
460  for (auto iInput : inputsIndices) {
461 
462  auto [nSupMod, nModule, nIphi, nIeta] = mGeomPtr->GetCellIndex(mInputsContainer[iInput].getTower());
463  auto [iphi, ieta] = mGeomPtr->GetCellPhiEtaIndexInSModule(nSupMod, nModule, nIphi, nIeta);
464 
465  // In case of a shared cluster, index of SM in C side, columns start at 48 and ends at 48*2
466  // C Side impair SM, nSupMod%2=1; A side pair SM, nSupMod%2=0
467  if (mSharedCluster && nSupMod % 2)
468  ieta += EMCAL_COLS;
469 
470  double etai = (double)ieta;
471  double phii = (double)iphi;
472 
473  double w = TMath::Max(0., mLogWeight + TMath::Log(mInputsContainer[iInput].getEnergy() / clusterAnalysis.E()));
474  // clusterAnalysis.E() summed amplitude of inputs, i.e. energy of cluster
475  // Gives smaller value of lambda than log weight
476  // w = mEnergyList[iInput] / clusterAnalysis.E(); // Nov 16, 2006 - try just energy
477 
478  dxx += w * etai * etai;
479  x += w * etai;
480  dzz += w * phii * phii;
481  z += w * phii;
482 
483  dxz += w * etai * phii;
484 
485  wtot += w;
486  }
487 
488  if (wtot > 0) {
489  dxx /= wtot;
490  x /= wtot;
491  dxx -= x * x;
492  dzz /= wtot;
493  z /= wtot;
494  dzz -= z * z;
495  dxz /= wtot;
496  dxz -= x * z;
497 
498  lambda[0] = 0.5 * (dxx + dzz) + TMath::Sqrt(0.25 * (dxx - dzz) * (dxx - dzz) + dxz * dxz);
499 
500  if (lambda[0] > 0)
501  lambda[0] = TMath::Sqrt(lambda[0]);
502  else
503  lambda[0] = 0;
504 
505  lambda[1] = 0.5 * (dxx + dzz) - TMath::Sqrt(0.25 * (dxx - dzz) * (dxx - dzz) + dxz * dxz);
506 
507  if (lambda[1] > 0) //To avoid exception if numerical errors lead to negative lambda.
508  lambda[1] = TMath::Sqrt(lambda[1]);
509  else
510  lambda[1] = 0.;
511  } else {
512  lambda[0] = 0.;
513  lambda[1] = 0.;
514  }
515 
516  clusterAnalysis.setM02(lambda[0] * lambda[0]);
517  clusterAnalysis.setM20(lambda[1] * lambda[1]);
518 }

523 template <class InputType>
524 std::tuple<int, float, float> ClusterFactory<InputType>::getMaximalEnergyIndex(gsl::span<int> inputsIndices) const
525 {
526 
527  float energy = 0.;
528  int mid = 0;
529  float cellAmp = 0;
530  for (auto iInput : inputsIndices) {
531  if (iInput >= mInputsContainer.size())
532  throw CellIndexRangeException(iInput, mInputsContainer.size());
533  cellAmp += mInputsContainer[iInput].getEnergy();
534  if (mInputsContainer[iInput].getEnergy() > energy) {
535  energy = mInputsContainer[iInput].getEnergy();
536  mid = iInput;
537  }
538  } // loop on cluster inputs
539 
540  return std::make_tuple(mid, energy, cellAmp);
541 }

580 template <class InputType>
581 double ClusterFactory<InputType>::tMaxInCm(const double e, const int key) const
582 {
583  const double ca = 4.82; // shower max parameter - first guess; ca=TMath::Log(1000./8.07)
584  double tmax = 0.; // position of electromagnetic shower max in cm
585 
586  const double x0 = 1.31; // radiation lenght (cm)
587 
588  if (e > 0.1) {
589  tmax = TMath::Log(e) + ca;
590  if (key == 0)
591  tmax += 0.5;
592  else
593  tmax -= 0.5;
594  tmax *= x0; // convert to cm
595  }
596 
597  return tmax;
598 }
