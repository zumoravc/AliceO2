93 inline void DigitTime::addDigit(const MCCompLabel& label, const CRU& cru, GlobalPadNumber globalPad, float signal)
94 {
95  auto& paddigit = mGlobalPads[globalPad];
96  if (paddigit.getID() == -1) {
97  // this means we have a new digit
98  paddigit.setID(mDigitCounter++);
99  // could also register this pad in a vector of digits
100  }
101  paddigit.addDigit(label, signal, mLabels);
102  mCommonMode[cru.gemStack()] += signal;
103 }

122 inline void DigitTime::fillOutputContainer(std::vector<Digit>& output, dataformats::MCTruthContainer<MCCompLabel>& mcTruth,
123  std::vector<CommonMode>& commonModeOutput, const Sector& sector, TimeBin timeBin,
124  float commonMode)
125 {
126  static Mapper& mapper = Mapper::instance();
127  GlobalPadNumber globalPad = 0;
128  for (size_t i = 0; i < mCommonMode.size(); ++i) {
129  const float cm = getCommonMode(GEMstack(i));
130  if (cm > 0.) {
131  commonModeOutput.push_back({cm, timeBin, static_cast<unsigned char>(i)});
132  }
133  }
134  for (auto& pad : mGlobalPads) {
135  if (pad.getChargePad() > 0.) {
136  const CRU cru = mapper.getCRU(sector, globalPad);
137  pad.fillOutputContainer<MODE>(output, mcTruth, cru, timeBin, globalPad, mLabels, getCommonMode(cru));
138  }
139  ++globalPad;
140  }
