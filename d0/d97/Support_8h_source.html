124 //Template function to perform serial TGeo boolean operations
125 template <class L, class SHAPE, class EL, class OP>
126 auto serialBoolOperation(L&& base, SHAPE&& shape, EL&& elements, OP&& op)
127 {
128  TGeoCompositeShape* localCS = nullptr;
129  SHAPE* localshape;
130  TGeoTranslation* localTranslation;
131 
132  for (auto par : elements) {
133  //Info("SerialBoolOperation", Form("params: %f %f %f %f %f %f,", par[0], par[1], par[2], par[3], par[4], par[5]), 0, 0);
134 
135  localshape = new SHAPE(par[0], par[1], par[2], nullptr);
136  localTranslation = new TGeoTranslation(par[3], par[4], par[5]);
137 
138  //The first subtraction needs a shape, the base shape
139  if (!localCS)
140  localCS = new TGeoCompositeShape(nullptr, compositeOperation(base, localshape, localTranslation, std::forward<OP>(op)));
141  else
142  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localshape, localTranslation, std::forward<OP>(op)));
143  }
144  return localCS;
145 }
