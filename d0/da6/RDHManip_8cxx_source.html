87 
88  os << fmt::format("version {:03d} headerSize {:03d} triggerType {:08x} {:s}\n",
89  o2::raw::RDHUtils::getVersion(rdh),
90  headerSize,
91  triggerType,
92  triggerTypeAsString(triggerType));
93 
94  os << fmt::format("cruId {:03d} dpwId {:02d} linkId {:03d}\n",
95  o2::raw::RDHUtils::getCRUID(rdh),
96  o2::raw::RDHUtils::getEndPointID(rdh),
97  o2::raw::RDHUtils::getLinkID(rdh));
98 
99  auto memorySize = o2::raw::RDHUtils::getMemorySize(rdh);
100 
101  os << fmt::format("offsetToNext {:05d} memorySize {:05d} {:s}\n",
102  o2::raw::RDHUtils::getOffsetToNext(rdh),
103  memorySize,
104  memorySize == headerSize ? "EMPTY" : "");
105 
106  os << fmt::format("heartbeatOrbit{:010d} heartbeatBC {:04d} feeId {:6d}\n",
107  o2::raw::RDHUtils::getHeartBeatOrbit(rdh),
108  o2::raw::RDHUtils::getHeartBeatBC(rdh),
109  o2::raw::RDHUtils::getFEEID(rdh));
110 
111  auto stop = o2::raw::RDHUtils::getStop(rdh);
112  os << fmt::format("stopBit {:1d} pagesCounter {:03d} packetCounter {:03d} {:s}\n",
113  stop,
114  o2::raw::RDHUtils::getPageCounter(rdh),
115  o2::raw::RDHUtils::getPacketCounter(rdh),
116  stop ? "STOP" : "");
117 
118  return os;

128 void append(std::vector<std::byte>& buffer, uint64_t w)
129 {
130  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x00000000000000FF)) >> 0)});
131  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x000000000000FF00)) >> 8)});
132  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x0000000000FF0000)) >> 16)});
133  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x00000000FF000000)) >> 24)});
134  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x000000FF00000000)) >> 32)});
135  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x0000FF0000000000)) >> 40)});

158 uint64_t eightBytes(gsl::span<const std::byte> buffer)
159 {

170 o2::header::RDHAny createRDH(gsl::span<const std::byte> buffer, int version)
171 {
172  auto minSize = sizeof(o2::header::RDHAny);
173  if (buffer.size() < minSize) {
174  throw std::invalid_argument(fmt::format("buffer should be at least {} bytes", minSize));
175  }
176  o2::header::RDHAny rdh;
177 
178  o2::raw::RDHUtils::setVersion(rdh, version);
179 
180  rdh.word0 = eightBytes(buffer.subspan(0));
181  rdh.word1 = eightBytes(buffer.subspan(8));
182  rdh.word2 = eightBytes(buffer.subspan(16));
183  rdh.word3 = eightBytes(buffer.subspan(24));
184  rdh.word4 = eightBytes(buffer.subspan(32));
185  rdh.word5 = eightBytes(buffer.subspan(40));
186  rdh.word6 = eightBytes(buffer.subspan(48));
187  rdh.word7 = eightBytes(buffer.subspan(56));
188 
189  return rdh;
190 }

192 int forEachRDH(gsl::span<const std::byte> buffer, std::function<void(const void*)> f = nullptr)
193 {
194  int index{0};
195  int nrdh{0};
196  while (index < buffer.size()) {
197  const void* rdhPtr = reinterpret_cast<const void*>(buffer.data() + index);
198  if (!o2::raw::RDHUtils::checkRDH(rdhPtr, true)) {
199  break;
200  }
201  nrdh++;
202  if (f) {
203  f(rdhPtr);
204  }
205  auto offsetToNext = o2::raw::RDHUtils::getOffsetToNext(rdhPtr);
206  if (offsetToNext == 0) {
207  return -1;
208  }
209  index += offsetToNext;
210  }
211  return nrdh;
212 }

214 int showRDHs(gsl::span<const std::byte> buffer)
215 {
216  assert(buffer.size() % 4 == 0);
217  return forEachRDH(buffer, [](const void* rdhPtr) {
218  auto version = o2::raw::RDHUtils::getVersion(rdhPtr);
219  switch (version) {
220  case 3:
221  case 4:
222  std::cout << (*reinterpret_cast<const RAWDataHeaderV4*>(rdhPtr)) << "\n";
223  break;
224  case 5:
225  std::cout << (*reinterpret_cast<const RAWDataHeaderV5*>(rdhPtr)) << "\n";
226  break;
227  case 6:
228  std::cout << (*reinterpret_cast<const RAWDataHeaderV6*>(rdhPtr)) << "\n";
229  break;
230  default:
231  throw std::invalid_argument(fmt::format("RDH version {} not yet supported by showRDHs function",
232  version));
233  }
234  });
235 }
