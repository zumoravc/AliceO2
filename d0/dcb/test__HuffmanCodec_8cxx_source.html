40 template <class RandvalStreamT, class EncodedStreamT, class CodecT>
41 void decoderProcess(RandvalStreamT& fifoRandvals, EncodedStreamT& fifoEncoded, CodecT& codec)
42 {
43  uint16_t decodedLen = 0;
44  typename CodecT::model_type::value_type decodedValue;
45  do {
46  } while (fifoEncoded.pull([&](typename EncodedStreamT::value_type c) {
47  codec.Decode(decodedValue, c, decodedLen);
48  return fifoRandvals.pull([&](typename RandvalStreamT::value_type v) {
49  if (decodedValue != v) {
50  throw std::runtime_error("decoding mismatch");
51  return false;
52  } else {
53  // std::cout << "decoded: "
54  // << std::setw(4) << decodedValue
55  // << " code: " << c
56  // << std::endl;
57  }
58  return true;
59  });
60  }));
61 }

105 auto setupCodec(int verbosity = 0)
106 {
107  // defining a contiguous alphabet of integral 16 bit unsigned numbers
108  // in the range [-7, 10] including the upper bound
109  // the first definition is a data type, then an object of this type is
110  // defined
111  using TestDistribution_t = o2::test::normal_distribution<double>;
112  using DataGenerator_t = o2::test::DataGenerator<int16_t, TestDistribution_t>;
113  DataGenerator_t dg(-7, 10, 1, 0., 1.);
114  using SimpleRangeAlphabet_t = ContiguousAlphabet<DataGenerator_t::value_type, -7, 10>;
115  SimpleRangeAlphabet_t alphabet;
116 
118  // Using the Huffman propability model for the alphabet
119  //
120  // HuffmanModel_t is a data type, huffmanmodel an object of this type
121  // the node type is defined to be HuffmanNode specialized to bitset<16>
122  // third template parameter determines whether code has to be decoded
123  // MSB to LSB (true) or LSB to MSB (false)
124  using HuffmanModel_t =
125  HuffmanModel<ProbabilityModel<SimpleRangeAlphabet_t>, std::bitset<32>, true>;
126  HuffmanModel_t huffmanmodel;
127 
128  huffmanmodel.init(0.);
129  if (verbosity > 0) {
130  std::cout << std::endl
131  << "Huffman probability model after initialization: " << std::endl;
132  for (auto s : alphabet) {
133  std::cout << "val = " << std::setw(2) << s << " --> weight = " << huffmanmodel[s] << std::endl;
134  }
135  }
136 
137  // add probabilities from data generator as weights for every symbol
138  for (auto s : alphabet) {
139  huffmanmodel.addWeight(s, dg.getProbability(s));
140  }
141 
142  // normalizing the weight to the total weight thus having the probability
143  // for every symbol
144  huffmanmodel.normalize();
145  if (verbosity > 0) {
146  std::cout << std::endl
147  << "Probabilities from DataGenerator:" << std::endl;
148  for (auto i : huffmanmodel) {
149  std::cout << "val = " << std::setw(2) << i.first << " --> weight = " << i.second << std::endl;
150  }
151  }
152 
154  // generate the Huffman tree in the Huffman model and create a Huffman
155  // codec operating on the probability table
156  huffmanmodel.GenerateHuffmanTree();
157  if (verbosity > 0) {
158  std::cout << std::endl
159  << "Generating binary tree and Huffman codes" << std::endl;
160  huffmanmodel.print();
161  }
162  return std::pair<HuffmanCodec<HuffmanModel_t>, DataGenerator_t>(huffmanmodel, dg);
163 }
