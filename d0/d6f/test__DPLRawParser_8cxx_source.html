31 // simple helper struct to keep the InputRecord and ownership of messages
32 struct DataSet {
33  // not nice with the double vector but for quick unit test ok
34  using Messages = std::vector<std::vector<std::unique_ptr<std::vector<char>>>>;
35  DataSet(std::vector<InputRoute>&& s, Messages&& m, std::vector<int>&& v)
36  : schema{std::move(s)}, messages{std::move(m)}, record{schema, {[this](size_t i, size_t part) {
37  BOOST_REQUIRE(i < this->messages.size());
38  BOOST_REQUIRE(part < this->messages[i].size() / 2);
39  auto header = static_cast<char const*>(this->messages[i].at(2 * part)->data());
40  auto payload = static_cast<char const*>(this->messages[i].at(2 * part + 1)->data());
41  return DataRef{nullptr, header, payload};
42  },
43  [this](size_t i) { return i < this->messages.size() ? messages[i].size() / 2 : 0; }, this->messages.size()}},
44  values{std::move(v)}
45  {
46  BOOST_REQUIRE(messages.size() == schema.size());
47  }
48 
49  std::vector<InputRoute> schema;
50  Messages messages;
51  InputRecord record;
52  std::vector<int> values;
53 };

55 DataSet createData()
56 {
57  // Create the routes we want for the InputRecord
58  std::vector<InputSpec> inputspecs = {
59  InputSpec{"tpc0", "TPC", "RAWDATA", 0, Lifetime::Timeframe},
60  InputSpec{"its1", "ITS", "RAWDATA", 0, Lifetime::Timeframe},
61  InputSpec{"its1", "ITS", "RAWDATA", 1, Lifetime::Timeframe}};
62 
63  size_t i = 0;
64  auto createRoute = [&i](const char* source, InputSpec& spec) {
65  return InputRoute{
66  spec,
67  i++,
68  source};
69  };
70 
71  std::vector<InputRoute> schema = {
72  createRoute("tpc_source", inputspecs[0]),
73  createRoute("its_source", inputspecs[1]),
74  createRoute("tof_source", inputspecs[2])};
75 
76  std::vector<int> checkValues;
77  DataSet::Messages messages;
78 
79  auto initRawPage = [&checkValues](char* buffer, size_t size, int value) {
80  char* wrtptr = buffer;
81  while (wrtptr < buffer + size) {
82  auto* header = reinterpret_cast<RAWDataHeaderV4*>(wrtptr);
83  *header = RAWDataHeaderV4();
84  header->offsetToNext = PAGESIZE;
85  *reinterpret_cast<decltype(value)*>(wrtptr + header->headerSize) = value;
86  wrtptr += PAGESIZE;
87  checkValues.emplace_back(value);
88  ++value;
89  }
90  };
91 
92  auto createMessage = [&messages, &initRawPage](DataHeader dh, int value) {
93  DataProcessingHeader dph{0, 1};
94  Stack stack{dh, dph};
95  if (dh.splitPayloadParts == 0 || dh.splitPayloadIndex == 0) {
96  // add new message collection
97  messages.emplace_back();
98  }
99  messages.back().emplace_back(std::make_unique<std::vector<char>>(stack.size()));
100  memcpy(messages.back().back()->data(), stack.data(), messages.back().back()->size());
101  messages.back().emplace_back(std::make_unique<std::vector<char>>(dh.payloadSize));
102  initRawPage(messages.back().back()->data(), messages.back().back()->size(), value);
103  };
104 
105  // we create message for the 3 input routes, the messages have different page size
106  // and the second messages has 3 parts, each with the same page size
107  // the test value is written as payload after the RDH and all values are cached for
108  // later checking when parsing the data set
109  DataHeader dh1;
110  dh1.dataDescription = "RAWDATA";
111  dh1.dataOrigin = "TPC";
112  dh1.subSpecification = 0;
113  dh1.payloadSerializationMethod = o2::header::gSerializationMethodNone;
114  dh1.payloadSize = 5 * PAGESIZE;
115  DataHeader dh2;
116  dh2.dataDescription = "RAWDATA";
117  dh2.dataOrigin = "ITS";
118  dh2.subSpecification = 0;
119  dh2.payloadSerializationMethod = o2::header::gSerializationMethodNone;
120  dh2.payloadSize = 3 * PAGESIZE;
121  dh2.splitPayloadParts = 3;
122  dh2.splitPayloadIndex = 0;
123  DataHeader dh3;
124  dh3.dataDescription = "RAWDATA";
125  dh3.dataOrigin = "ITS";
126  dh3.subSpecification = 1;
127  dh3.payloadSerializationMethod = o2::header::gSerializationMethodNone;
128  dh3.payloadSize = 4 * PAGESIZE;
129  createMessage(dh1, 10);
130  createMessage(dh2, 20);
131  dh2.splitPayloadIndex++;
132  createMessage(dh2, 23);
133  dh2.splitPayloadIndex++;
134  createMessage(dh2, 26);
135  createMessage(dh3, 30);
136 
137  return {std::move(schema), std::move(messages), std::move(checkValues)};
138 }

140 BOOST_AUTO_TEST_CASE(test_DPLRawParser)
141 {
142  auto dataset = createData();
143  InputRecord& inputs = dataset.record;
144  BOOST_REQUIRE(dataset.messages.size() > 0);
145  BOOST_REQUIRE(dataset.messages[0].at(0) != nullptr);
146  BOOST_REQUIRE(inputs.size() > 0);
147  BOOST_CHECK((*inputs.begin()).header == dataset.messages[0].at(0)->data());
148  DPLRawParser parser(inputs);
149  int count = 0;
150  o2::header::DataHeader const* last = nullptr;
151  for (auto it = parser.begin(), end = parser.end(); it != end; ++it, ++count) {
152  LOG(INFO) << "data " << count << " " << *((int*)it.data());
153  // now check the iterator API
154  // retrieving RDH v4
155  auto const* rdh = it.get_if<o2::header::RAWDataHeaderV4>();
156  // retrieving the raw pointer of the page
157  auto const* raw = it.raw();
158  // retrieving payload pointer of the page
159  auto const* payload = it.data();
160  // size of payload
161  size_t payloadSize = it.size();
162  // offset of payload in the raw page
163  size_t offset = it.offset();
164  BOOST_REQUIRE(rdh != nullptr);
165  BOOST_REQUIRE(offset == sizeof(o2::header::RAWDataHeaderV4));
166  BOOST_REQUIRE(payload == raw + offset);
167  BOOST_REQUIRE(*reinterpret_cast<int const*>(payload) == dataset.values[count]);
168  BOOST_REQUIRE(payloadSize == PAGESIZE - sizeof(o2::header::RAWDataHeaderV4));
169  auto const* dh = it.o2DataHeader();
170  if (last != dh) {
171  // this is a special wrapper to print the RDU info and table header, this will
172  // be extended
173  std::cout << DPLRawParser::RDHInfo(it) << std::endl;
174  last = dh;
175  }
176  std::cout << it << " payload size " << it.size() << std::endl;
177  }
178 
179  // test the parser with filter on data specs, this will filter out the first input
180  // route with 5 raw pages in the payload, so we start checking at count 5
181  DPLRawParser filteredparser(inputs, o2::framework::select("its:ITS/RAWDATA"));
182  count = 5;
183  for (auto it = filteredparser.begin(), end = filteredparser.end(); it != end; ++it, ++count) {
184  LOG(INFO) << "data " << count << " " << *((int*)it.data());
185  BOOST_REQUIRE(*reinterpret_cast<int const*>(it.data()) == dataset.values[count]);
186  }
187 
188  // test with filter not matching any input route
189  DPLRawParser nomatchingparser(inputs, o2::framework::select("nmatch:NO/MATCH"));
190  count = 0;
191  for (auto it = nomatchingparser.begin(), end = nomatchingparser.end(); it != end; ++it, ++count) {
192  LOG(INFO) << "data " << count << " " << *((int*)it.data());
193  }
194  BOOST_CHECK(count == 0);
195 }
