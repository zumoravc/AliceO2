243 template <typename DataT, bool isConsistentT>
244 int Spline2DBase<DataT, isConsistentT>::test(const bool draw, const bool drawDataPoints)
245 {
246  using namespace std;
247 
248  const int Ndim = 3;
249 
250  const int Fdegree = 4;
251 
252  double Fcoeff[Ndim][4 * (Fdegree + 1) * (Fdegree + 1)];
253 
254  constexpr int nKnots = 4;
255  constexpr int nAxiliaryPoints = 1;
256  constexpr int uMax = nKnots * 3;
257 
258  auto F = [&](DataT u, DataT v, DataT Fuv[]) {
259  const double scale = TMath::Pi() / uMax;
260  double uu = u * scale;
261  double vv = v * scale;
262  double cosu[Fdegree + 1], sinu[Fdegree + 1], cosv[Fdegree + 1], sinv[Fdegree + 1];
263  double ui = 0, vi = 0;
264  for (int i = 0; i <= Fdegree; i++, ui += uu, vi += vv) {
265  GPUCommonMath::SinCos(ui, sinu[i], cosu[i]);
266  GPUCommonMath::SinCos(vi, sinv[i], cosv[i]);
267  }
268  for (int dim = 0; dim < Ndim; dim++) {
269  double f = 0; // Fcoeff[dim][0]/2;
270  for (int i = 1; i <= Fdegree; i++) {
271  for (int j = 1; j <= Fdegree; j++) {
272  double* c = &(Fcoeff[dim][4 * (i * Fdegree + j)]);
273  f += c[0] * cosu[i] * cosv[j];
274  f += c[1] * cosu[i] * sinv[j];
275  f += c[2] * sinu[i] * cosv[j];
276  f += c[3] * sinu[i] * sinv[j];
277  }
278  }
279  Fuv[dim] = f;
280  }
281  };
282 
283  TCanvas* canv = nullptr;
284  TNtuple* nt = nullptr;
285  TNtuple* knots = nullptr;
286 
287  auto ask = [&]() -> bool {
288  if (!canv) {
289  return 0;
290  }
291  canv->Update();
292  cout << "type 'q ' to exit" << endl;
293  std::string str;
294  std::getline(std::cin, str);
295  return (str != "q" && str != ".q");
296  };
297 
298  std::cout << "Test 2D interpolation with the compact spline" << std::endl;
299 
300  int nTries = 10;
301 
302  if (draw) {
303  canv = new TCanvas("cQA", "Spline2D QA", 1500, 800);
304  nTries = 10000;
305  }
306 
307  long double statDf = 0;
308  long double statDf1D = 0;
309  long double statN = 0;
310 
311  for (int seed = 1; seed < nTries + 1; seed++) {
312  //cout << "next try.." << endl;
313 
314  gRandom->SetSeed(seed);
315 
316  for (int dim = 0; dim < Ndim; dim++) {
317  for (int i = 0; i < 4 * (Fdegree + 1) * (Fdegree + 1); i++) {
318  Fcoeff[dim][i] = gRandom->Uniform(-1, 1);
319  }
320  }
321 
322  Spline2D<DataT, Ndim> spline;
323 
324  int knotsU[nKnots], knotsV[nKnots];
325  do {
326  knotsU[0] = 0;
327  knotsV[0] = 0;
328  double du = 1. * uMax / (nKnots - 1);
329  for (int i = 1; i < nKnots; i++) {
330  knotsU[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
331  knotsV[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
332  }
333  knotsU[nKnots - 1] = uMax;
334  knotsV[nKnots - 1] = uMax;
335  spline.recreate(nKnots, knotsU, nKnots, knotsV);
336 
337  if (nKnots != spline.getGridU1().getNumberOfKnots() ||
338  nKnots != spline.getGridU2().getNumberOfKnots()) {
339  cout << "warning: n knots changed during the initialisation " << nKnots
340  << " -> " << spline.getNumberOfKnots() << std::endl;
341  continue;
342  }
343  } while (0);
344 
345  std::string err = FlatObject::stressTest(spline);
346  if (!err.empty()) {
347  cout << "error at FlatObject functionality: " << err << endl;
348  return -1;
349  } else {
350  // cout << "flat object functionality is ok" << endl;
351  }
352 
353  // Ndim-D spline
354  spline.approximateFunction(0., uMax, 0., uMax, F, 4, 4);
355 
356  //if (itry == 0)
357  if (1) {
358  TFile outf("testSpline2D.root", "recreate");
359  if (outf.IsZombie()) {
360  cout << "Failed to open output file testSpline2D.root " << std::endl;
361  } else {
362  const char* name = "spline2Dtest";
363  spline.writeToFile(outf, name);
364  Spline2D<DataT, Ndim>* p = Spline2D<DataT, Ndim>::readFromFile(outf, name);
365  if (p == nullptr) {
366  cout << "Failed to read Spline1D from file testSpline1D.root " << std::endl;
367  } else {
368  spline = *p;
369  }
370  outf.Close();
371  }
372  }
373 
374  // 1-D splines for each of Ndim dimensions
375 
376  Spline2D<DataT, 1> splines1D[Ndim];
377 
378  for (int dim = 0; dim < Ndim; dim++) {
379  auto F1 = [&](DataT x1, DataT x2, DataT f[]) {
380  DataT ff[Ndim];
381  F(x1, x2, ff);
382  f[0] = ff[dim];
383  };
384  splines1D[dim].recreate(nKnots, knotsU, nKnots, knotsV);
385  splines1D[dim].approximateFunction(0., uMax, 0., uMax, F1, 4, 4);
386  }
387 
388  double stepU = .1;
389  for (double u = 0; u < uMax; u += stepU) {
390  for (double v = 0; v < uMax; v += stepU) {
391  DataT f[Ndim];
392  F(u, v, f);
393  DataT s[Ndim];
394  DataT s1;
395  spline.interpolate(u, v, s);
396  for (int dim = 0; dim < Ndim; dim++) {
397  statDf += (s[dim] - f[dim]) * (s[dim] - f[dim]);
398  splines1D[dim].interpolate(u, v, &s1);
399  statDf1D += (s[dim] - s1) * (s[dim] - s1);
400  }
401  statN += Ndim;
402  // cout << u << " " << v << ": f " << f << " s " << s << " df "
403  // << s - f << " " << sqrt(statDf / statN) << std::endl;
404  }
405  }
406  // cout << "Spline2D standard deviation : " << sqrt(statDf / statN)
407  // << std::endl;
408 
409  if (draw) {
410  delete nt;
411  delete knots;
412  nt = new TNtuple("nt", "nt", "u:v:f:s");
413  knots = new TNtuple("knots", "knots", "type:u:v:s");
414  double stepU = .3;
415  for (double u = 0; u < uMax; u += stepU) {
416  for (double v = 0; v < uMax; v += stepU) {
417  DataT f[Ndim];
418  F(u, v, f);
419  DataT s[Ndim];
420  spline.interpolate(u, v, s);
421  nt->Fill(u, v, f[0], s[0]);
422  }
423  }
424  nt->SetMarkerStyle(8);
425 
426  nt->SetMarkerSize(.5);
427  nt->SetMarkerColor(kBlue);
428  nt->Draw("s:u:v", "", "");
429 
430  nt->SetMarkerColor(kGray);
431  nt->SetMarkerSize(2.);
432  nt->Draw("f:u:v", "", "same");
433 
434  nt->SetMarkerSize(.5);
435  nt->SetMarkerColor(kBlue);
436  nt->Draw("s:u:v", "", "same");
437 
438  for (int i = 0; i < nKnots; i++) {
439  for (int j = 0; j < nKnots; j++) {
440  double u = spline.getGridU1().getKnot(i).u;
441  double v = spline.getGridU2().getKnot(j).u;
442  DataT s[Ndim];
443  spline.interpolate(u, v, s);
444  knots->Fill(1, u, v, s[0]);
445  }
446  }
447 
448  knots->SetMarkerStyle(8);
449  knots->SetMarkerSize(1.5);
450  knots->SetMarkerColor(kRed);
451  knots->SetMarkerSize(1.5);
452  knots->Draw("s:u:v", "type==1", "same"); // knots
453 
454  if (drawDataPoints) {
455  SplineHelper2D<DataT> helper;
456  helper.setSpline(spline, 4, 4);
457  for (int ipu = 0; ipu < helper.getHelperU1().getNumberOfDataPoints(); ipu++) {
458  const typename SplineHelper1D<DataT>::DataPoint& pu = helper.getHelperU1().getDataPoint(ipu);
459  for (int ipv = 0; ipv < helper.getHelperU2().getNumberOfDataPoints(); ipv++) {
460  const typename SplineHelper1D<DataT>::DataPoint& pv = helper.getHelperU2().getDataPoint(ipv);
461  if (pu.isKnot && pv.isKnot) {
462  continue;
463  }
464  DataT s[Ndim];
465  spline.interpolate(pu.u, pv.u, s);
466  knots->Fill(2, pu.u, pv.u, s[0]);
467  }
468  }
469  knots->SetMarkerColor(kBlack);
470  knots->SetMarkerSize(1.);
471  knots->Draw("s:u:v", "type==2", "same"); // data points
472  }
473 
474  if (!ask()) {
475  break;
476  }
477  }
478  }
479  // delete canv;
480  // delete nt;
481  // delete knots;
482 
483  statDf = sqrt(statDf / statN);
484  statDf1D = sqrt(statDf1D / statN);
485 
486  cout << "\n std dev for Spline2D : " << statDf << std::endl;
487  cout << " mean difference between 1-D and " << Ndim
488  << "-D splines : " << statDf1D << std::endl;
489 
490  if (statDf < 0.15 && statDf1D < 1.e-20) {
491  cout << "Everything is fine" << endl;
492  } else {
493  cout << "Something is wrong!!" << endl;
494  return -2;
495  }
496 
497  return 0;
498 }
