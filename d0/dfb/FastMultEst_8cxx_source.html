22 void FastMultEst::fillNClPerLayer(const gsl::span<const o2::itsmft::CompClusterExt>& clusters)
23 {
24  int lr = FastMultEst::NLayers - 1, nchAcc = o2::itsmft::ChipMappingITS::getNChips() - o2::itsmft::ChipMappingITS::getNChipsPerLr(lr);
25  std::memset(&nClPerLayer[0], 0, sizeof(int) * FastMultEst::NLayers);
26  for (int i = clusters.size(); i--;) { // profit from clusters being ordered in chip increasing order
27  while (clusters[i].getSensorID() < nchAcc) {
28  assert(lr >= 0);
29  nchAcc -= o2::itsmft::ChipMappingITS::getNChipsPerLr(--lr);
30  }
31  nClPerLayer[lr]++;
32  }
33 }

37 float FastMultEst::processNoiseFree(const std::array<int, NLayers> ncl)
38 {
39  // we assume that on the used layers the observed number of clusters is defined by the
40  // the noise ~ nu * Nchips and contribution from the signal tracks Ntr*mAccCorr
41  const auto& conf = FastMultEstConfig::Instance();
42 
43  float mat[3] = {0}, b[2] = {0};
44  nLayersUsed = 0;
45  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
46  if (ncl[il] > 0) {
47  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
48  float err2i = 1. / ncl[il];
49  float m2n = nch * err2i;
50  mat[0] += err2i * conf.accCorr[il] * conf.accCorr[il];
51  mat[2] += nch * m2n;
52  mat[1] += conf.accCorr[il] * m2n; // non-diagonal element
53  b[0] += conf.accCorr[il];
54  b[1] += nch;
55  nLayersUsed++;
56  }
57  }
58  mult = noisePerChip = chi2 = -1;
59  float det = mat[0] * mat[2] - mat[1] * mat[1];
60  if (nLayersUsed < 2 || std::abs(det) < 1e-15) {
61  return -1;
62  }
63  float detI = 1. / det;
64  mult = detI * (b[0] * mat[2] - b[1] * mat[1]);
65  noisePerChip = detI * (b[1] * mat[0] - b[0] * mat[1]);
66  cov[0] = mat[2] * detI;
67  cov[2] = mat[0] * detI;
68  cov[1] = -mat[1] * detI;
69  chi2 = 0.;
70  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
71  if (ncl[il] > 0) {
72  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
73  float err2i = 1. / ncl[il];
74  float diff = mult * conf.accCorr[il] + nch * noisePerChip - ncl[il];
75  chi2 += diff * diff / ncl[il];
76  }
77  }
78  chi2 = nLayersUsed > 2 ? chi2 / (nLayersUsed - 2) : 0.;
79  return mult;
80 }

84 float FastMultEst::processNoiseImposed(const std::array<int, NLayers> ncl)
85 {
86  // we assume that on the used layers the observed number of clusters is defined by the
87  // the noise ~ nu * Nchips and contribution from the signal tracks Ntr*conf.accCorr
88  //
89  // minimize the form sum_lr (noise_i - mu nchips_i)^2 / (mu nchips_i) + lambda_i * (noise_i + mult*acc_i - ncl_i)
90  // whith noise_i being estimate of the noise clusters in nchips_i of layer i, mu is the mean noise per chip,
91  // mult is the number of signal clusters on the ref. (1st) layer and the acc_i is the acceptance of layer i wrt 1st.
92  // The lambda_i is hust a Lagrange multiplier.
93 
94  const auto& conf = FastMultEstConfig::Instance();
95  float w2sum = 0., wnsum = 0., wsum = 0.;
96  nLayersUsed = 0;
97  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
98  if (ncl[il] > 0) {
99  float nchInv = 1. / o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
100  w2sum += conf.accCorr[il] * conf.accCorr[il] * nchInv;
101  wnsum += ncl[il] * nchInv * conf.accCorr[il];
102  wsum += conf.accCorr[il];
103  nLayersUsed++;
104  }
105  }
106  mult = 0;
107  chi2 = -1;
108  noisePerChip = conf.imposeNoisePerChip;
109  if (nLayersUsed < 1) {
110  return -1;
111  }
112  auto w2sumI = 1. / w2sum;
113  mult = (wnsum - noisePerChip * wsum) * w2sumI;
114  cov[0] = wnsum * w2sumI;
115  cov[2] = 0.;
116  cov[1] = 0.;
117 
118  chi2 = 0.;
119  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
120  if (ncl[il] > 0) {
121  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
122  float noise = ncl[il] - mult * conf.accCorr[il], estNoise = o2::itsmft::ChipMappingITS::getNChipsPerLr(il) * noisePerChip;
123  float diff = noise - estNoise;
124  chi2 += diff * diff / estNoise;
125  }
126  }
127  chi2 = nLayersUsed > 2 ? chi2 / (nLayersUsed - 2) : 0.;
128  return mult;
129 }
