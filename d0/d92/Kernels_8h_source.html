78 template <typename T, typename ARRAY>
79 class ARROW_EXPORT SortedGroupByKernel : public arrow::compute::UnaryKernel
80 {
81  public:
82  explicit SortedGroupByKernel(GroupByOptions<T> options = {}) : mOptions(options){};
83  arrow::Status Call(arrow::compute::FunctionContext* ctx,
84  arrow::compute::Datum const& table,
85  arrow::compute::Datum* outputRanges) override
86  {
87  using namespace arrow;
88  if (table.kind() == arrow::compute::Datum::TABLE) {
89  auto atable = util::get<std::shared_ptr<arrow::Table>>(table.value);
90  auto columnIndex = atable->schema()->GetFieldIndex(mOptions.columnName);
91  auto chunkedArray = atable->column(columnIndex);
92  return doGrouping(chunkedArray, outputRanges);
93  };
94  return arrow::Status::OK();
95  };
96 #pragma GCC diagnostic push
97 #ifdef __clang__
98 #pragma GCC diagnostic ignored "-Winconsistent-missing-override"
99 #endif // __clang__
100  std::shared_ptr<arrow::DataType> out_type() const final
101  {
102  return mType;
103  }
104 #pragma GCC diagnostic pop
105 
106  private:
107  arrow::Status doGrouping(std::shared_ptr<arrow::ChunkedArray> chunkedArray, arrow::compute::Datum* outputRanges)
108  {
109  o2::framework::TableBuilder builder;
110  auto writer = builder.persist<T, T, T>({"start", "count", "index"});
111  auto zeroChunk = std::static_pointer_cast<ARRAY>(chunkedArray->chunk(0));
112  if (zeroChunk->length() == 0) {
113  *outputRanges = std::move(builder.finalize());
114  return arrow::Status::OK();
115  }
116  T currentIndex = 0;
117  T currentCount = 0;
118  T currentOffset = 0;
119  for (auto ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
120  auto chunk = chunkedArray->chunk(ci);
121  T const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
122  for (auto ai = 0; ai < chunk->length(); ++ai) {
123  if (currentIndex == data[ai]) {
124  currentCount++;
125  } else {
126  writer(0, currentOffset, currentCount, currentIndex);
127  currentOffset += currentCount;
128  while (data[ai] - currentIndex > 1) {
129  writer(0, currentOffset, 0, ++currentIndex);
130  }
131  currentIndex++;
132  currentCount = 1;
133  }
134  }
135  }
136  writer(0, currentOffset, currentCount, currentIndex);
137  while (currentIndex < mOptions.size - 1) {
138  writer(0, currentOffset, 0, ++currentIndex);
139  }
140  *outputRanges = std::move(builder.finalize());
141  return arrow::Status::OK();
142  }
143  std::shared_ptr<arrow::DataType> mType;
144  GroupByOptions<T> mOptions;
145 };

153 template <typename T>
154 arrow::Status sliceByColumn(arrow::compute::FunctionContext* context,
155  std::string const& key,
156  T size,
157  arrow::compute::Datum const& inputTable,
158  std::vector<arrow::compute::Datum>* outputSlices,
159  std::vector<uint64_t>* offsets = nullptr)
160 {
161  if (inputTable.kind() != arrow::compute::Datum::TABLE) {
162  return arrow::Status::Invalid("Input Datum was not a table");
163  }
164  // build all the ranges on the fly.
165  arrow::compute::Datum outRanges;
166  auto table = arrow::util::get<std::shared_ptr<arrow::Table>>(inputTable.value);
167  o2::framework::SortedGroupByKernel<T, soa::arrow_array_for_t<T>> kernel{GroupByOptions<T>{key, size}};
168 
169  ARROW_RETURN_NOT_OK(kernel.Call(context, inputTable, &outRanges));
170  auto ranges = arrow::util::get<std::shared_ptr<arrow::Table>>(outRanges.value);
171  outputSlices->reserve(ranges->num_rows());
172  if (offsets) {
173  offsets->reserve(ranges->num_rows());
174  }
175 
176  auto startChunks = ranges->column(0);
177  assert(startChunks->num_chunks() == 1);
178  auto countChunks = ranges->column(1);
179  assert(countChunks->num_chunks() == 1);
180  auto startData = std::static_pointer_cast<soa::arrow_array_for_t<T>>(startChunks->chunk(0))->raw_values();
181  auto countData = std::static_pointer_cast<soa::arrow_array_for_t<T>>(countChunks->chunk(0))->raw_values();
182 
183  for (auto ri = 0; ri < ranges->num_rows(); ++ri) {
184  auto start = startData[ri];
185  auto count = countData[ri];
186  auto schema = table->schema();
187  std::vector<std::shared_ptr<arrow::ChunkedArray>> slicedColumns;
188  slicedColumns.reserve(schema->num_fields());
189  // if (count != 0) {
190  for (auto ci = 0; ci < schema->num_fields(); ++ci) {
191  slicedColumns.emplace_back(table->column(ci)->Slice(start, count));
192  }
193  // }
194  outputSlices->emplace_back(arrow::compute::Datum(arrow::Table::Make(table->schema(), slicedColumns)));
195  if (offsets) {
196  offsets->emplace_back(start);
197  }
198  }
199  return arrow::Status::OK();
200 }
