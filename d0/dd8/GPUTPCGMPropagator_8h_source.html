102  GPUd() int RejectCluster(float chiY, float chiZ, unsigned char clusterState)
103  {
104  if (chiY > 9.f || chiZ > 9.f) {
105  return 2;
106  }
107  if ((chiY > 6.25f || chiZ > 6.25f) && (clusterState & (GPUTPCGMMergedTrackHit::flagSplit | GPUTPCGMMergedTrackHit::flagShared))) {
108  return 2;
109  }
110  if ((chiY > 1.f || chiZ > 6.25f) && (clusterState & (GPUTPCGMMergedTrackHit::flagEdge | GPUTPCGMMergedTrackHit::flagSingle))) {
111  return 2;
112  }
113  return 0;
114  }
