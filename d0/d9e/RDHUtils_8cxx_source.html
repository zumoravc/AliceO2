61 void RDHUtils::printRDH(const void* rdhP)
62 {
63  int version = getVersion(rdhP);
64  switch (version) {
65  case 3:
66  case 4:
67  printRDH(*reinterpret_cast<const RAWDataHeaderV4*>(rdhP));
68  break;
69  case 5:
70  printRDH(*reinterpret_cast<const RAWDataHeaderV5*>(rdhP));
71  break;
72  case 6:
73  printRDH(*reinterpret_cast<const RAWDataHeaderV6*>(rdhP));
74  break;
75  default:
76  LOG(ERROR) << "Unexpected RDH version " << version << " from";
77  dumpRDH(rdhP);
78  throw std::runtime_error("invalid RDH provided");
79  break;
80  };

94 bool RDHUtils::checkRDH(const void* rdhP, bool verbose)
95 {
96  int version = getVersion(rdhP);
97  bool ok = true;
98  switch (version) {
99  case 3:
100  case 4:
101  ok = checkRDH(*reinterpret_cast<const RAWDataHeaderV4*>(rdhP), verbose);
102  break;
103  case 5:
104  ok = checkRDH(*reinterpret_cast<const RAWDataHeaderV5*>(rdhP), verbose);
105  break;
106  case 6:
107  ok = checkRDH(*reinterpret_cast<const RAWDataHeaderV6*>(rdhP), verbose);
108  break;
109  default:
110  ok = false;
111  if (verbose) {
112  LOG(ERROR) << "Unexpected RDH version " << version << " from";
113  }
114  break;
115  };
116  if (!ok && verbose) {
117  dumpRDH(rdhP);
118  }
119  return ok;
120 }

123 bool RDHUtils::checkRDH(const RAWDataHeaderV4& rdh, bool verbose)
124 {
125  // check if rdh conforms with RDH4 fields
126  bool ok = true;
127  if (rdh.version != 4 && rdh.version != 3) {
128  if (verbose) {
129  LOG(ERROR) << "RDH version 4 is expected instead of " << int(rdh.version);
130  }
131  ok = false;
132  }
133  if (rdh.headerSize != 64) {
134  if (verbose) {
135  LOG(ERROR) << "RDH with header size of 64 B is expected instead of " << int(rdh.headerSize);
136  }
137  ok = false;
138  }
139  if (rdh.memorySize < 64 || rdh.offsetToNext < 64 || rdh.memorySize > MAXCRUPage || rdh.offsetToNext > MAXCRUPage) {
140  if (verbose) {
141  LOG(ERROR) << "RDH expected to have memorySize/offsetToNext in 64 : 8192 bytes range instead of "
142  << int(rdh.memorySize) << '/' << int(rdh.offsetToNext);
143  }
144  ok = false;
145  }
146  if (rdh.zero0 || rdh.word3 || rdh.zero41 || rdh.zero42 || rdh.word5 || rdh.zero6 || rdh.word7) {
147  if (verbose) {
148  LOG(ERROR) << "Some reserved fields of RDH v4 are not empty";
149  }
150  ok = false;
151  }
152  if (!ok && verbose) {
153  dumpRDH(rdh);
154  }
155  return ok;
156 }

233 uint32_t RDHUtils::fletcher32(const uint16_t* data, int len)
234 {
235  uint32_t c0, c1;
236  // We similarly solve for n > 0 and n * (n+1) / 2 * (2^16-1) < (2^32-1) here.
237  // On modern computers, using a 64-bit c0/c1 could allow a group size of 23726746.
238  for (c0 = c1 = 0; len > 0; len -= 360) {
239  int blocklen = len < 360 ? len : 360;
240  for (int i = 0; i < blocklen; ++i) {
241  c0 = c0 + *data++;
242  c1 = c1 + c0;
243  }
244  c0 = c0 % 65535;
245  c1 = c1 % 65535;
246  }
247  return (c1 << 16 | c0);
248 }
