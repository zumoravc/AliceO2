99  }

102  }
103  int leg = 0;
104  GPUTPCGMSliceTrack *trbase = &track, *tr = &track;
105  tr->SetPrevSegmentNeighbour(1000000000);
106  while (true) {
107  int iTrk = tr - mSliceTrackInfos;
108  if (trkUsed[iTrk]) {
109  GPUError("FAILURE: double use");
110  }
111  trkUsed[iTrk] = true;
112 
113  int jtr = tr->NextSegmentNeighbour();
114  if (jtr >= 0) {
115  tr = &(mSliceTrackInfos[jtr]);
116  tr->SetPrevSegmentNeighbour(1000000002);
117  continue;
118  }
119  jtr = trbase->NextNeighbour();
120  if (jtr >= 0) {
121  trbase = &(mSliceTrackInfos[jtr]);
122  tr = trbase;
123  if (tr->PrevSegmentNeighbour() >= 0) {
124  break;
125  }
126  tr->SetPrevSegmentNeighbour(1000000001);

128  continue;
129  }
130  break;
131  }
132  }
133  for (int i = 0; i < SliceTrackInfoLocalTotal(); i++) {

140 int GPUTPCGMMerger::GetTrackLabel(const GPUTPCGMBorderTrack& trk)
141 {
142  GPUTPCGMSliceTrack* track = &mSliceTrackInfos[trk.TrackID()];
143  int nClusters = track->OrigTrack()->NHits();
144  std::vector<int> labels;
145  for (int i = 0; i < nClusters; i++) {
146  int id;
147  if (Param().rec.mergerReadFromTrackerDirectly) {
148  const GPUTPCTracker& tracker = GetConstantMem()->tpcTrackers[track->Slice()];
149  const GPUTPCHitId& ic = tracker.TrackHits()[track->OrigTrack()->FirstHitID() + i];
150  id = tracker.Data().ClusterDataIndex(tracker.Data().Row(ic.RowIndex()), ic.HitIndex()) + GetConstantMem()->ioPtrs.clustersNative->clusterOffset[track->Slice()][0];
151  } else {
152  id = track->OrigTrack()->OutTrackClusters()[i].GetId();
153  }
154  for (int j = 0; j < 3; j++) {
155  int label = GetConstantMem()->ioPtrs.mcLabelsTPC[id].fClusterID[j].fMCID;
156  if (label >= 0) {
157  labels.push_back(label);
158  }
159  }
160  }
161  if (labels.size() == 0) {
162  return (-1);
163  }
164  labels.push_back(-1);
165  std::sort(labels.begin(), labels.end());
166  int bestLabel = -1, bestLabelCount = 0;
167  int curLabel = labels[0], curCount = 1;
168  for (unsigned int i = 1; i < labels.size(); i++) {
169  if (labels[i] == curLabel) {
170  curCount++;
171  } else {
172  if (curCount > bestLabelCount) {
173  bestLabelCount = curCount;
174  bestLabel = curLabel;
175  }
176  curLabel = labels[i];
177  curCount = 1;
178  }
179  }
180  return bestLabel;
181 }

186 void GPUTPCGMMerger::PrintMergeGraph(const GPUTPCGMSliceTrack* trk, std::ostream& out)
187 {
188  const GPUTPCGMSliceTrack* orgTrack = trk;
189  while (trk->PrevSegmentNeighbour() >= 0) {
190  trk = &mSliceTrackInfos[trk->PrevSegmentNeighbour()];
191  }
192  const GPUTPCGMSliceTrack* orgTower = trk;
193  while (trk->PrevNeighbour() >= 0) {
194  trk = &mSliceTrackInfos[trk->PrevNeighbour()];
195  }
196 
197  int nextId = trk - mSliceTrackInfos;
198  out << "Graph of track %d" << (orgTrack - mSliceTrackInfos) << "\n";
199  while (nextId >= 0) {
200  trk = &mSliceTrackInfos[nextId];
201  if (trk->PrevSegmentNeighbour() >= 0) {
202  out << "TRACK TREE INVALID!!! " << trk->PrevSegmentNeighbour() << " --> " << nextId << "\n";
203  }
204  out << (trk == orgTower ? "--" : " ");
205  while (nextId >= 0) {
206  GPUTPCGMSliceTrack* trk2 = &mSliceTrackInfos[nextId];
207  if (trk != trk2 && (trk2->PrevNeighbour() >= 0 || trk2->NextNeighbour() >= 0)) {
208  out << " (TRACK TREE INVALID!!! " << trk2->PrevNeighbour() << " <-- " << nextId << " --> " << trk2->NextNeighbour() << ") ";
209  }
210  char tmp[128];
211  snprintf(tmp, 128, " %s%5d(%5.2f)", trk2 == orgTrack ? "!" : " ", nextId, trk2->QPt());
212  out << tmp;
213  nextId = trk2->NextSegmentNeighbour();
214  }
215  out << "\n";
216  nextId = trk->NextNeighbour();
217  }

343 GPUd() int GPUTPCGMMerger::RefitSliceTrack(GPUTPCGMSliceTrack& sliceTrack, const GPUTPCTrack* inTrack, float alpha, int slice)
344 {
345  GPUTPCGMPropagator prop;
346  prop.SetMaterialTPC();
347  prop.SetMaxSinPhi(GPUCA_MAX_SIN_PHI);
348  prop.SetToyMCEventsFlag(false);
349  prop.SetSeedingErrors(true); // Larger errors for seeds, better since we don't start with good hypothesis
350  prop.SetFitInProjections(false);
351  prop.SetPolynomialField(&Param().polynomialField);
352  GPUTPCGMTrackParam trk;
353  trk.X() = inTrack->Param().GetX();
354  trk.Y() = inTrack->Param().GetY();
355  trk.Z() = inTrack->Param().GetZ();
356  trk.SinPhi() = inTrack->Param().GetSinPhi();
357  trk.DzDs() = inTrack->Param().GetDzDs();
358  trk.QPt() = inTrack->Param().GetQPt();
359  trk.TZOffset() = Param().earlyTpcTransform ? inTrack->Param().GetZOffset() : GetConstantMem()->calibObjects.fastTransform->convZOffsetToVertexTime(slice, inTrack->Param().GetZOffset(), Param().continuousMaxTimeBin);
360  trk.ShiftZ(this, slice, sliceTrack.ClusterZT0(), sliceTrack.ClusterZTN());
361  for (int way = 0; way < 2; way++) {
362  if (way) {
363  prop.SetFitInProjections(true);
364  prop.SetPropagateBzOnly(true);
365  }
366  trk.ResetCovariance();
367  prop.SetTrack(&trk, alpha);
368  int start = way ? inTrack->NHits() - 1 : 0;
369  int end = way ? 0 : (inTrack->NHits() - 1);
370  int incr = way ? -1 : 1;
371  for (int i = start; i != end; i += incr) {
372  float x, y, z;
373  int row, flags;
374  if (Param().rec.mergerReadFromTrackerDirectly) {
375  const GPUTPCTracker& tracker = GetConstantMem()->tpcTrackers[slice];
376  const GPUTPCHitId& ic = tracker.TrackHits()[inTrack->FirstHitID() + i];
377  int clusterIndex = tracker.Data().ClusterDataIndex(tracker.Data().Row(ic.RowIndex()), ic.HitIndex());
378  row = ic.RowIndex();
379  const ClusterNative& cl = GetConstantMem()->ioPtrs.clustersNative->clustersLinear[GetConstantMem()->ioPtrs.clustersNative->clusterOffset[slice][0] + clusterIndex];
380  flags = cl.getFlags();
381  if (Param().earlyTpcTransform) {
382  x = tracker.Data().ClusterData()[clusterIndex].x;
383  y = tracker.Data().ClusterData()[clusterIndex].y;
384  z = tracker.Data().ClusterData()[clusterIndex].z - trk.TZOffset();
385  } else {
386  GetConstantMem()->calibObjects.fastTransform->Transform(slice, row, cl.getPad(), cl.getTime(), x, y, z, trk.TZOffset());
387  }
388  } else {
389  const GPUTPCSliceOutCluster& clo = inTrack->OutTrackCluster(i);
390  row = clo.GetRow();
391  flags = clo.GetFlags();
392  if (Param().earlyTpcTransform) {
393  x = clo.GetX();
394  y = clo.GetY();
395  z = clo.GetZ() - trk.TZOffset();
396  } else {
397  const ClusterNative& cl = GetConstantMem()->ioPtrs.clustersNative->clustersLinear[clo.GetId()];
398  GetConstantMem()->calibObjects.fastTransform->Transform(slice, row, cl.getPad(), cl.getTime(), x, y, z, trk.TZOffset());
399  }
400  }
401  if (prop.PropagateToXAlpha(x, alpha, true)) {
402  return way == 0;
403  }
404  trk.ConstrainSinPhi();
405  if (prop.Update(y, z, row, Param(), flags & GPUTPCGMMergedTrackHit::clustererAndSharedFlags, 0, nullptr, false)) {
406  return way == 0;
407  }
408  trk.ConstrainSinPhi();
409  }
410  if (way) {
411  sliceTrack.SetParam2(trk);
412  } else {
413  sliceTrack.Set(trk, inTrack, alpha, slice);
414  sliceTrack.SetX2(0.f);
415  }
416  }
417  return 0;
418 }
