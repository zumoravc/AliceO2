44  void configure(const boost::property_tree::ptree& config) override
45  {
46  std::string libraryName = "lib" + config.get<std::string>("moduleName");
47  std::string className = config.get<std::string>("className");
48 
49  int libLoaded = gSystem->Load(libraryName.c_str(), "", true);
50  if (libLoaded < 0) {
51  throw std::runtime_error("Failed to load the library: " + libraryName);
52  }
53 
54  // it does not seem to be documented anywhere, but this pointer to
55  // a dictionary should not be deleted - it results in segfaults.
56  TClass* dictionary = TClass::GetClass(className.c_str());
57  if (!dictionary) {
58  throw std::runtime_error("Failed to load the dictionary of the class: " + className + " from the library: " + libraryName);
59  }
60 
61  mCondition.reset(static_cast<DataSamplingCondition*>(dictionary->New()));
62  if (mCondition == nullptr) {
63  throw std::runtime_error("Failed to instantiate the class: " + className + " from the library: " + libraryName);
64  }
65 
66  mCondition->configure(config);
67  }
