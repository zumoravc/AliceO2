86 /******************************************************************************
87  * @brief Meta program to create a vector of wrapped types from a sequencs
88  * of types. A new mpl vector containing Wrapper<Type> for every
89  * element of the original sequence is created
90  * mpl::fold recursivly applies the elements of the list to the previous result
91  * placeholder _2 refers to the element, placeholder _1 to the previous result
92  * or initial condition for the first element
93  */

103 /******************************************************************************
104  * @brief apply functor to element at mpl sequence position
105  * This meta function recurses through the list while incrementing the index
106  * and calls the functor at the required position
107  *
108  * @note this is a helper macro to the apply_at function
109  */
110 template <typename _IndexT, // data type of position index
111  typename _Iterator, // current iterator position
112  typename _End, // end iterator position
113  typename _ReturnT, // type of the return value
114  _IndexT _Index, // current index
115  typename F // functor
116  >
117 struct apply_at_sequence_position {
118  static _ReturnT apply(_IndexT position, F f)
119  {
120  if (position == _Index) {
121  // this is the queried position, execute function and
122  // terminate loop by forwarding _End as _Iterator and thus
123  // calling the specialization
124  // TODO: check the performance penalty of the element instantiation
125  typename boost::mpl::deref<_Iterator>::type element;
126  return f(element);
127  } else {
128  // go to next element
129  return apply_at_sequence_position< //
130  _IndexT, //
131  typename boost::mpl::next<_Iterator>::type, //
132  _End, //
133  _ReturnT, //
134  _Index + 1, //
135  F //
136  >::apply(position, f);
137  }
138  }
139 };
140 // specialization: end of recursive loop, kicks in if _Iterator matches
141 // _End.
142 // here we end up if the position parameter is out of bounds
143 template <typename _IndexT, typename _End, typename _ReturnT, _IndexT _Index, typename F>
144 struct apply_at_sequence_position<_IndexT, _End, _End, _ReturnT, _Index, F> {
145  static _ReturnT apply(_IndexT position, F f)
146  {
147  // TODO: this is probably the place to through an exeption because
148  // we are out of bound
149  return _ReturnT(0);
150  }
151 };

164 /******************************************************************************
165  * @brief Apply a functor to an element of a compile time sequence
166  * This meta function is a bridge to the runtime environment to apply a functor
167  * to an element of a compile time list of types.
168  * Required template parameter: sequence - typename defining the list of types
169  *
170  * @arg position position of element in the list
171  * @arg f functor
172  */
173 template <typename _Sequence, typename _IndexT = int, typename F = defaultFunctor>
174 typename F::return_type apply_at(_IndexT position, F f)
175 {
176  return apply_at_sequence_position<_IndexT, typename boost::mpl::begin<_Sequence>::type,
177  typename boost::mpl::end<_Sequence>::type, typename F::return_type, 0,
178  F>::apply(position, f);
179 }
