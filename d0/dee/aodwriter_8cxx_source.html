119 struct ATask {
120  Produces<aod::Uno> table_uno;
121  Produces<aod::Due> table_due;

124  void init(InitContext&)

126  cnt = 0;

129  void process(aod::Tracks const& tracks)
130  {
131  for (auto& track : tracks) {
132  bool ok = (cnt % 2) == 0;
133  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
134  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
135  double mom = track.tgl();
136  double pt = track.signed1Pt();
137  int id = (int)cnt;
138 
139  table_uno(phi, eta, mom);
140  table_due(ok, phi, eta, mom, pt);
141  table_tre(phi, eta, mom, id);
142  LOGF(INFO, "Values (%i): (%i %f, %f, %f, %f, %i)", cnt, ok, eta, phi, mom, pt, id);
143  cnt++;
144  }
145 
146  LOGF(INFO, "ATask Processed %i data points from Tracks", cnt);
147  }
148 
149  size_t cnt = 0;
150 };
