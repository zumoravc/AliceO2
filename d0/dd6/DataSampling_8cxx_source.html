37 void DataSampling::GenerateInfrastructure(WorkflowSpec& workflow, const std::string& policiesSource, size_t threads)
38 {
39  LOG(DEBUG) << "Generating Data Sampling infrastructure...";
40  Dispatcher dispatcher(createDispatcherName(), policiesSource);
41  Options options;
42 
43  std::unique_ptr<ConfigurationInterface> cfg = ConfigurationFactory::getConfiguration(policiesSource);
44  auto policiesTree = cfg->getRecursive("dataSamplingPolicies");
45 
46  for (auto&& policyConfig : policiesTree) {
47 
48  std::unique_ptr<DataSamplingPolicy> policy;
49 
50  // We don't want the Dispatcher to exit due to one faulty Policy
51  try {
52  policy = std::make_unique<DataSamplingPolicy>(policyConfig.second);
53  } catch (const std::exception& ex) {
54  LOG(WARN) << "Could not load the Data Sampling Policy '"
55  << policyConfig.second.get_optional<std::string>("id").value_or("") << "', because: " << ex.what();
56  continue;
57  } catch (...) {
58  LOG(WARN) << "Could not load the Data Sampling Policy '"
59  << policyConfig.second.get_optional<std::string>("id").value_or("") << "'";
60  continue;
61  }
62 
63  for (const auto& path : policy->getPathMap()) {
64  dispatcher.registerPath({path.first, path.second});
65  }
66 
67  if (!policy->getFairMQOutputChannel().empty()) {
68  options.push_back({"channel-config", VariantType::String, policy->getFairMQOutputChannel().c_str(), {"Out-of-band channel config"}});
69  LOG(DEBUG) << " - registering output FairMQ channel '" << policy->getFairMQOutputChannel() << "'";
70  }
71  }
72  options.push_back({"period-timer-stats", framework::VariantType::Int, 10 * 1000000, {"Dispatcher's stats timer period"}});
73 
74  if (dispatcher.getInputSpecs().size() > 0) {
75  DataProcessorSpec spec;
76  spec.name = dispatcher.getName();
77  spec.inputs = dispatcher.getInputSpecs();
78  spec.outputs = dispatcher.getOutputSpecs();
79  spec.algorithm = adaptFromTask<Dispatcher>(std::move(dispatcher));
80  spec.maxInputTimeslices = threads;
81  spec.labels = {{"DataSampling"}, {"Dispatcher"}};
82  spec.options = options;
83 
84  workflow.emplace_back(std::move(spec));
85  } else {
86  LOG(DEBUG) << "No input to this dispatcher, it won't be added to the workflow.";
87  }
88 }

108 std::vector<InputSpec> DataSampling::InputSpecsForPolicy(ConfigurationInterface* const config, const std::string& policyName)
109 {
110  std::vector<InputSpec> inputs;
111  auto policiesTree = config->getRecursive("dataSamplingPolicies");
112 
113  for (auto&& policyConfig : policiesTree) {
114  if (policyConfig.second.get<std::string>("id") == policyName) {
115  DataSamplingPolicy policy(policyConfig.second);
116  for (const auto& path : policy.getPathMap()) {
117  InputSpec input = DataSpecUtils::matchingInput(path.second);
118  inputs.push_back(input);
119  }
120  break;
121  }
122  }
123  return inputs;
124 }

166 std::vector<std::string> DataSampling::MachinesForPolicy(configuration::ConfigurationInterface* const config, const std::string& policyName)
167 {
168  std::vector<std::string> machines;
169  auto policiesTree = config->getRecursive("dataSamplingPolicies");
170  for (auto&& policyConfig : policiesTree) {
171  if (policyConfig.second.get<std::string>("id") == policyName) {
172  for (const auto& machine : policyConfig.second.get_child("machines")) {
173  machines.emplace_back(machine.second.get<std::string>(""));
174  }
175  return machines;
176  }
177  }
178  throw std::runtime_error("Could not find the policy '" + policyName + "'");
179 }
