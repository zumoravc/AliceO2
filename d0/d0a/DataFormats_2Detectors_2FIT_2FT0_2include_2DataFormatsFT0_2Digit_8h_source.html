33 struct Triggers {
34  enum { bitA,
35  bitC,
36  bitVertex,
37  bitCen,
38  bitSCen };
39  uint8_t triggersignals; // T0 trigger signals
40  int8_t nChanA; // number of faired channels A side
41  int8_t nChanC; // number of faired channels A side
42  int32_t amplA; // sum amplitude A side
43  int32_t amplC; // sum amplitude C side
44  int16_t timeA; // average time A side
45  int16_t timeC; // average time C side
46  Triggers() = default;
47  Triggers(uint8_t signals, int8_t chanA, int8_t chanC, int32_t aamplA, int32_t aamplC, int16_t atimeA, int16_t atimeC)
48  {
49  triggersignals = signals;
50  nChanA = chanA;
51  nChanC = chanC;
52  amplA = aamplA;
53  amplC = aamplC;
54  timeA = atimeA;
55  timeC = atimeC;
56  }
57  bool getOrA() { return (triggersignals & (1 << bitA)) != 0; }
58  bool getOrC() { return (triggersignals & (1 << bitC)) != 0; }
59  bool getVertex() { return (triggersignals & (1 << bitVertex)) != 0; }
60  bool getCen() { return (triggersignals & (1 << bitCen)) != 0; }
61  bool getSCen() { return (triggersignals & (1 << bitSCen)) != 0; }

63  void setTriggers(Bool_t isA, Bool_t isC, Bool_t isVrtx, Bool_t isCnt, Bool_t isSCnt, int8_t chanA, int8_t chanC, int32_t aamplA,
64  int32_t aamplC, int16_t atimeA, int16_t atimeC)
65  {
66  triggersignals = (isA << bitA) | (isC << bitC) | (isVrtx << bitVertex) | (isCnt << bitCen) | (isSCnt << bitSCen);
67  nChanA = chanA;
68  nChanC = chanC;
69  amplA = aamplA;
70  amplC = aamplC;
71  timeA = atimeA;
72  timeC = atimeC;
73  }
74  void cleanTriggers()
75  {

77  nChanA = nChanC = -1;
78  amplA = amplC = -1000;
