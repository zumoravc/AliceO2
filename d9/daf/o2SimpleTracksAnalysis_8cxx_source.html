38 // A dummy workflow which creates a few of the tables proposed by Ruben,

40 WorkflowSpec defineDataProcessing(ConfigContext const& specs)
41 {
42  // Workflow definition. A workflow can be one or more DataProcessors
43  // each implementing (part of) an analysis. Each DataProcessor has
44  // (at least) a name, some Inputs, some Outputs and they get arranged
45  // together accordingly.
46  WorkflowSpec workflow{
47  // Multiple DataProcessor specs
48  // can be provided per workflow
49  DataProcessorSpec{
50  // The name of my analysis
51  "tracks-analysis",
52  Inputs{
53  // Dangling inputs of type AOD will be automatically picked up by DPL
54  // and an extra reader device will be instanciated to read them from
55  // file. In this particular case the signature AOD/TRACKPAR is
56  // associated to the basic trajectory parameters for a track described
57  // in Ruben's table. The first string is just a label so that the
58  // algorithm can be in principle be reused for different kind of
59  // tracks.
60  InputSpec{"tracks", "AOD", "TRACKPAR"},
61  },
62  // No outputs for the time being.
63  Outputs{
64  OutputSpec{{"derived"}, "AOD", "TRACKDERIVED"}},
65  AlgorithmSpec{
66  // This is the actual per "message" loop, where a message could
67  // be the contents of a file or part of it.
68  // FIXME: Too much boilerplate.
69  adaptStateless([](InputRecord& inputs, DataAllocator& outputs) {
71  auto input = inputs.get<TableConsumer>("tracks");
73  auto& etaPhiBuilder = outputs.make<TableBuilder>(Output{"AOD", "TRACKDERIVED"});
74  auto etaPhiWriter = etaPhiBuilder.cursor<o2::aod::TracksDerived>();
75 
79  auto tracks = aod::Tracks(input->asArrowTable());
80 
81  for (auto& track : tracks) {
82  auto phi = asin(track.snp()) + track.alpha() + M_PI;
83  auto eta = log(tan(0.25 * M_PI - 0.5 * atan(track.tgl())));
84  etaPhiWriter(0, eta, phi);
85  }
86  })}},
87  DataProcessorSpec{
88  "phi-consumer",
89  Inputs{
90  InputSpec{"etaphi", "AOD", "TRACKDERIVED"}},
91  Outputs{},
92  AlgorithmSpec{
93  adaptStateless([](InputRecord& inputs) {
94  auto input = inputs.get<TableConsumer>("etaphi");
95  auto tracks = o2::analysis::doSingleLoopOn(input);
96 
97  auto h = tracks.Histo1D("fPhi");
98 
99  TFile f("result1.root", "RECREATE");
100  h->SetName("Phi");
101  h->Write();
102  })}},
103  DataProcessorSpec{
104  "eta-consumer",
105  Inputs{
106  InputSpec{"etaphi", "AOD", "TRACKDERIVED"}},
107  Outputs{},
108  AlgorithmSpec{
109  adaptStateless([](InputRecord& inputs) {
110  auto input = inputs.get<TableConsumer>("etaphi");
111  auto tracks = o2::analysis::doSingleLoopOn(input);
112 
113  auto h2 = tracks.Histo1D("fEta");
114 
115  TFile f("result2.root", "RECREATE");
116  h2->SetName("Eta");
117  h2->Write();
118  })}}};
119  return workflow;
120 }
