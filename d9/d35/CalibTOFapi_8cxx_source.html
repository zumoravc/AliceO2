90 float CalibTOFapi::getTimeCalibration(int ich, float tot)
91 {
92 
93  // time calibration to correct measured TOF times
94 
95  float corr = 0;
96  if (!mLHCphase || !mSlewParam) {
97  LOG(WARNING) << "Either LHC phase or slewing object null: mLHCphase = " << mLHCphase << ", mSlewParam = " << mSlewParam;
98  return corr;
99  }
100  // LHCphase
101  corr += mLHCphase->getLHCphase(int(mTimeStamp / 1000)); // timestamp that we use in LHCPhase is in seconds, but for CCDB we need it in ms
102 
103  // time slewing + channel offset
104  //printf("eval time sleweing calibration: ch=%d tot=%f (lhc phase = %f)\n",ich,tot,corr);
105  corr += mSlewParam->evalTimeSlewing(ich, tot);
106  //printf("corr = %f\n",corr);
107  return corr;
108 }
