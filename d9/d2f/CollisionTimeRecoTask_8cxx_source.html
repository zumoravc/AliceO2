29 o2::ft0::RecPoints CollisionTimeRecoTask::process(o2::ft0::Digit const& bcd,
30  gsl::span<const o2::ft0::ChannelData> inChData,
31  gsl::span<o2::ft0::ChannelDataFloat> outChData)
32 {
33  LOG(INFO) << "Running reconstruction on new event";
34 
35  Int_t ndigitsC = 0, ndigitsA = 0;
36 
37  constexpr Int_t nMCPsA = 4 * Geometry::NCellsA;
38  // constexpr Int_t nMCPsC = 4 * Geometry::NCellsC;
39  // constexpr Int_t nMCPs = nMCPsA + nMCPsC;
40 
41  Float_t sideAtime = 0, sideCtime = 0;
42 
43  auto timeStamp = o2::InteractionRecord::bc2ns(bcd.mIntRecord.bc, bcd.mIntRecord.orbit);
44 
45  LOG(INFO) << " event time " << timeStamp << " orbit " << bcd.mIntRecord.orbit << " bc " << bcd.mIntRecord.bc;
46 
47  int nch = inChData.size();
48  for (int ich = 0; ich < nch; ich++) {
49  outChData[ich] = o2::ft0::ChannelDataFloat{inChData[ich].ChId,
50  inChData[ich].CFDTime * Geometry::ChannelWidth,
51  (double)inChData[ich].QTCAmpl * Geometry::MV_2_NchannelsInverse,
52  inChData[ich].ChainQTC};
53 
54  if (std::fabs(outChData[ich].CFDTime) < 2000) {
55  if (outChData[ich].ChId < nMCPsA) {
56  sideAtime += outChData[ich].CFDTime;
57  ndigitsA++;
58  } else {
59  sideCtime += outChData[ich].CFDTime;
60  ndigitsC++;
61  }
62  }
63  }
64  std::array<Float_t, 4> mCollisionTime = {2 * o2::InteractionRecord::DummyTime,
65  2 * o2::InteractionRecord::DummyTime,
66  2 * o2::InteractionRecord::DummyTime,
67  2 * o2::InteractionRecord::DummyTime};
68  mCollisionTime[TimeA] = (ndigitsA > 0) ? sideAtime / Float_t(ndigitsA) : 2 * o2::InteractionRecord::DummyTime;
69  mCollisionTime[TimeC] = (ndigitsC > 0) ? sideCtime / Float_t(ndigitsC) : 2 * o2::InteractionRecord::DummyTime;
70 
71  if (ndigitsA > 0 && ndigitsC > 0) {
72  mCollisionTime[Vertex] = (mCollisionTime[TimeA] - mCollisionTime[TimeC]) / 2.;
73  mCollisionTime[TimeMean] = (mCollisionTime[TimeA] + mCollisionTime[TimeC]) / 2.;
74  } else {
75  mCollisionTime[TimeMean] = std::min(mCollisionTime[TimeA], mCollisionTime[TimeC]);
76  }
77  LOG(INFO) << " Collision time " << mCollisionTime[TimeA] << " " << mCollisionTime[TimeC] << " " << mCollisionTime[TimeMean] << " " << mCollisionTime[Vertex];
78  return RecPoints{
79  mCollisionTime, bcd.ref.getFirstEntry(), bcd.ref.getEntries(), bcd.mIntRecord, bcd.mTriggers};
80 }
