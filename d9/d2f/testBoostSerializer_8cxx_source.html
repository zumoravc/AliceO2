105

108  inputV.emplace_back(TestCluster{(uint8_t)i, 0.3f * iFloat, 0.5f * iFloat, 0.7f / iFloat, 0.9f / iFloat});

110 
111  auto msgStr = BoostSerialize(inputV).str();
112  auto inputV2 = BoostDeserialize<contType>(msgStr);
113 
114  BOOST_TEST(inputV.size() == inputV2.size());
