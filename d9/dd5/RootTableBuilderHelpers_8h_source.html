96 // When reading from a ROOT file special care must happen
97 // because uint64_t is platform specific while ULong64_t is
98 // always long long unsigned int (same for the signed version).
99 // By using this traits we make sure that any 64 bit quantity
100 // read from a root file uses the ROOT datatype, not the platform one.
101 template <typename T>
102 struct Remap64Bit {
103  using type = T;
104 };

156 struct RootTableBuilderHelpers {
157  template <typename... TTREEREADERVALUE>
158  static void convertTTree(TableBuilder& builder,
159  TTreeReader& reader,
160  ReaderHolder<TTREEREADERVALUE>... holders)
161  {
162  std::vector<std::string> branchNames = {holders.reader->GetBranchName()...};
163  LOG(INFO) << "Branch names";
164  for (auto bn : branchNames)
165  LOG(INFO) << " " << bn;
166 
167  auto filler = builder.preallocatedPersist<typename std::decay_t<decltype(holders)>::Type...>(branchNames, reader.GetEntries(true));
168  reader.Restart();
169  while (reader.Next()) {
170  filler(0, ValueExtractor::deref(holders)...);
171  }
172  }
173 
174  template <typename... C>
175  static void convertASoAColumns(TableBuilder& builder, TTreeReader& reader, pack<C...>)
176  {
177  return convertTTree(builder, reader, ColumnReaderTrait<C>::createReader(reader)...);
178  }
179 
180  template <typename T>
181  static void convertASoA(TableBuilder& builder, TTreeReader& reader)
182  {
183  return convertASoAColumns(builder, reader, typename T::persistent_columns_t{});
184  }
185 };
