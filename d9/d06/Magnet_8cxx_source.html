58 void Magnet::createMaterials()
59 {
60  auto& matmgr = o2::base::MaterialManager::Instance();
61  //
62  // Create materials for L3 magnet
63  //
64  Int_t isxfld = 2.;
65  Float_t sxmgmx = 10.;
66  o2::base::Detector::initFieldTrackingParams(isxfld, sxmgmx);
67  Float_t epsil, stmin, deemax, tmaxfd, stemax;
68 
69  // --- Define the various materials for GEANT ---
70  // Steel
71  Float_t asteel[4] = {55.847, 51.9961, 58.6934, 28.0855};
72  Float_t zsteel[4] = {26., 24., 28., 14.};
73  Float_t wsteel[4] = {.715, .18, .1, .005};
74  Float_t aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
75  Float_t zAir[4] = {6., 7., 8., 18.};
76  Float_t wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
77  Float_t dAir = 1.20479E-3;
78  Float_t aWater[2] = {1.00794, 15.9994};
79  Float_t zWater[2] = {1., 8.};
80  Float_t wWater[2] = {0.111894, 0.888106};
81 
82  // Aluminum
83  matmgr.Material("MAG", 9, "Al0$", 26.98, 13., 2.7, 8.9, 37.2);
84  matmgr.Material("MAG", 29, "Al1$", 26.98, 13., 2.7, 8.9, 37.2);
85 
86  // Stainless Steel
87  matmgr.Mixture("MAG", 19, "STAINLESS STEEL1", asteel, zsteel, 7.88, 4, wsteel);
88  matmgr.Mixture("MAG", 39, "STAINLESS STEEL2", asteel, zsteel, 7.88, 4, wsteel);
89  matmgr.Mixture("MAG", 59, "STAINLESS STEEL3", asteel, zsteel, 7.88, 4, wsteel);
90 
91  // Iron
92  matmgr.Material("MAG", 10, "Fe0$", 55.85, 26., 7.87, 1.76, 17.1);
93  matmgr.Material("MAG", 30, "Fe1$", 55.85, 26., 7.87, 1.76, 17.1);
94 
95  // Air
96  matmgr.Mixture("MAG", 15, "AIR0$", aAir, zAir, dAir, 4, wAir);
97  matmgr.Mixture("MAG", 35, "AIR1$", aAir, zAir, dAir, 4, wAir);
98 
99  // Water
100  matmgr.Mixture("MAG", 16, "WATER", aWater, zWater, 1., 2, wWater);
101 
102  // ****************
103  // Defines tracking media parameters.
104  // Les valeurs sont commentees pour laisser le defaut
105  // a GEANT (version 3-21, page CONS200), f.m.
106  epsil = .001; // Tracking precision,
107  stemax = -1.; // Maximum displacement for multiple scat
108  tmaxfd = -20.; // Maximum angle due to field deflection
109  deemax = -.3; // Maximum fractional energy loss, DLS
110  stmin = -.8;
111  // ***************
112 
113  // IRON
114  matmgr.Medium("MAG", 10, "FE_C0", 10, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
115  matmgr.Medium("MAG", 30, "FE_C1", 30, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
116 
117  // ALUMINUM
118  matmgr.Medium("MAG", 9, "ALU_C0", 9, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
119  matmgr.Medium("MAG", 29, "ALU_C1", 29, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
120 
121  // AIR
122  matmgr.Medium("MAG", 15, "AIR_C0", 15, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
123  matmgr.Medium("MAG", 35, "AIR_C1", 35, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
124 
125  // Steel
126  matmgr.Medium("MAG", 19, "ST_C0", 19, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
127  matmgr.Medium("MAG", 39, "ST_C1", 39, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
128  matmgr.Medium("MAG", 59, "ST_C3", 59, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
129  // WATER
130  matmgr.Medium("MAG", 16, "WATER", 16, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
131 }

133 void Magnet::ConstructGeometry()
134 {
135  createMaterials();
136 
137  // Octagon
138  const Int_t kNSides = 8;
139  const Float_t kStartAngle = 22.5; // deg
140  const Float_t kFullAngle = 360.0; // deg
141  // Mother volume
142  const Float_t kRBMotherInner = 600.00; // cm
143  const Float_t kRBMotherOuter = 790.50; // cm
144  const Float_t kLBMother = 706.00; // cm
145  // Yoke
146  const Float_t kRYokeInner = 703.50; // cm
147  const Float_t kRYokeOuter = 790.50; // cm
148  const Float_t kLYoke = 620.00; // cm
149  // Coil
150  const Float_t kRCoilInner = 593.00; // cm
151  const Float_t kRCoilOuter = 682.00; // cm
152  const Float_t kLCoil = 588.00; // cm
153  // Cooling
154  const Float_t kRCoolingOuter = 1.70; // cm
155  const Float_t kRCoolingInner = 1.00; // cm
156  // Thermal Shield
157  const Float_t kRThermalShieldInner = 566.00; // cm
158  const Float_t kRThermalShieldOuter = 571.00; // cm
159  // Crown
160  const Float_t kRCrownInner = 600.00; // cm
161  const Float_t kRCrownOuter = 785.50; // cm
162  const Float_t kLCrown1 = 605.00; // cm
163  const Float_t kLCrown2 = 620.00; // cm
164  const Float_t kLCrown3 = 706.00; // cm
165  // Door
166  const Float_t kRDoorOuter = 600.00; // cm
167  const Float_t kRPlugInner = 183.50; // cm
168  const Float_t kLDoor1 = 615.50; // cm
169  const Float_t kLDoor2 = 714.60; // cm
170  //
171  const Float_t kDegRad = TMath::Pi() / 180.;
172 
173  //
174  // Top volume
175  TGeoVolume* top = gGeoManager->GetVolume("cave");
176  assert(top);
177 
178  // Media
179  auto& matmgr = o2::base::MaterialManager::Instance();
180  auto medAir = matmgr.getTGeoMedium("MAG_AIR_C1");
181  auto medAlu = matmgr.getTGeoMedium("MAG_ALU_C1");
182  auto medAluI = matmgr.getTGeoMedium("MAG_ALU_C0");
183  auto medSteel = matmgr.getTGeoMedium("MAG_ST_C1");
184  auto medWater = matmgr.getTGeoMedium("MAG_WATER");
185  //
186  // Offset between LHC and LEP axis
187  Float_t os = -30.;
188 
189  //
190  // Define Barrel Mother
191  //
192  TGeoPgon* shBMother = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 2);
193  shBMother->DefineSection(0, -kLBMother, kRBMotherInner, kRBMotherOuter);
194  shBMother->DefineSection(1, kLBMother, kRBMotherInner, kRBMotherOuter);
195  //
196  TGeoVolumeAssembly* voBMother = new TGeoVolumeAssembly("L3BM");
197  //
198  // Define Thermal Shield
199  //
200  // Only one layer
201  // This can be improved: replace by (protection - shield - insulation) !
202  //
203  TGeoPgon* shThermSh = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 2);
204  shThermSh->DefineSection(0, -kLCoil, kRThermalShieldInner, kRThermalShieldOuter);
205  shThermSh->DefineSection(1, kLCoil, kRThermalShieldInner, kRThermalShieldOuter);
206  //
207  TGeoVolume* voThermSh = new TGeoVolume("L3TS", shThermSh, medAluI);
208  voBMother->AddNode(voThermSh, 1, new TGeoTranslation(0., 0., 0.));
209  //
210  // Define Coils and cooling circuits
211  //
212  TGeoPgon* shCoilMother = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 2);
213  shCoilMother->DefineSection(0, -kLCoil, kRCoilInner - 2. * kRCoolingOuter, kRCoilOuter + 2. * kRCoolingOuter);
214  shCoilMother->DefineSection(1, kLCoil, kRCoilInner - 2. * kRCoolingOuter, kRCoilOuter + 2. * kRCoolingOuter);
215  //
216  // Coils
217  TGeoVolume* voCoilMother = new TGeoVolume("L3CM", shCoilMother, medAir);
218  voBMother->AddNode(voCoilMother, 1, new TGeoTranslation(0., 0., 0.));
219  // Divide into the 168 turns
220  TGeoVolume* voCoilTurn = voCoilMother->Divide("L3CD", 3, 168, 0., 0.);
221  TGeoPgon* shCoils = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 2);
222  shCoils->DefineSection(0, -3., kRCoilInner, kRCoilOuter);
223  shCoils->DefineSection(1, 3., kRCoilInner, kRCoilOuter);
224  //
225  TGeoVolume* voCoils = new TGeoVolume("L3C0", shCoils, medAlu);
226  voCoilTurn->AddNode(voCoils, 1, new TGeoTranslation(0., 0., 0.));
227  //
228  // Hexagonal Cooling circuits
229  //
230  const Float_t kRCC = kRCoolingOuter;
231  const Float_t kRCW = kRCoolingInner;
232  const Float_t kRCL = kRCC * TMath::Tan(30. / 180. * TMath::Pi());
233  const Float_t kRWL = kRCW * TMath::Tan(30. / 180. * TMath::Pi());
234  // Outer Circuits
235  //
236  // Pipe
237  TGeoPgon* shCoolingPipeO = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 4);
238  shCoolingPipeO->DefineSection(0, -kRCC, kRCoilOuter + kRCC, kRCoilOuter + kRCC + 0.01);
239  shCoolingPipeO->DefineSection(1, -kRCL, kRCoilOuter, kRCoilOuter + 2. * kRCC);
240  shCoolingPipeO->DefineSection(2, kRCL, kRCoilOuter, kRCoilOuter + 2. * kRCC);
241  shCoolingPipeO->DefineSection(3, kRCC, kRCoilOuter + kRCC, kRCoilOuter + kRCC + 0.01);
242  //
243  TGeoVolume* voCoolingPipeO = new TGeoVolume("L3CCO", shCoolingPipeO, medAlu);
244  voCoilTurn->AddNode(voCoolingPipeO, 1, new TGeoTranslation(0., 0., 0.));
245  //
246  TGeoPgon* shCoolingWaterO = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 4);
247  shCoolingWaterO->DefineSection(0, -kRCW, kRCoilOuter + kRCC, kRCoilOuter + kRCC + 0.01);
248  shCoolingWaterO->DefineSection(1, -kRWL, kRCoilOuter + (kRCC - kRCW), kRCoilOuter + kRCC + kRCW);
249  shCoolingWaterO->DefineSection(2, kRWL, kRCoilOuter + (kRCC - kRCW), kRCoilOuter + kRCC + kRCW);
250  shCoolingWaterO->DefineSection(3, kRCW, kRCoilOuter + kRCC, kRCoilOuter + kRCC + 0.01);
251  //
252  TGeoVolume* voCoolingWaterO = new TGeoVolume("L3CWO", shCoolingWaterO, medWater);
253  voCoolingPipeO->AddNode(voCoolingWaterO, 1, new TGeoTranslation(0., 0., 0.));
254 
255  // Inner Circuits
256  //
257  // Pipe
258  TGeoPgon* shCoolingPipeI = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 4);
259  shCoolingPipeI->DefineSection(0, -kRCC, kRCoilInner - kRCC, kRCoilInner - kRCC + 0.01);
260  shCoolingPipeI->DefineSection(1, -kRCL, kRCoilInner - 2. * kRCC, kRCoilInner);
261  shCoolingPipeI->DefineSection(2, kRCL, kRCoilInner - 2. * kRCC, kRCoilInner);
262  shCoolingPipeI->DefineSection(3, kRCC, kRCoilInner - kRCC, kRCoilInner - kRCC + 0.01);
263  //
264  TGeoVolume* voCoolingPipeI = new TGeoVolume("L3CCI", shCoolingPipeI, medAlu);
265  voCoilTurn->AddNode(voCoolingPipeI, 1, new TGeoTranslation(0., 0., 0.));
266  //
267  TGeoPgon* shCoolingWaterI = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 4);
268  shCoolingWaterI->DefineSection(0, -kRCW, kRCoilInner - kRCC, kRCoilInner - kRCC + 0.01);
269  shCoolingWaterI->DefineSection(1, -kRWL, kRCoilInner - kRCC - kRCW, kRCoilInner - (kRCC - kRCW));
270  shCoolingWaterI->DefineSection(2, kRWL, kRCoilInner - kRCC - kRCW, kRCoilInner - (kRCC - kRCW));
271  shCoolingWaterI->DefineSection(3, kRCW, kRCoilInner - kRCC, kRCoilInner - kRCC + 0.01);
272  //
273  TGeoVolume* voCoolingWaterI = new TGeoVolume("L3CWI", shCoolingWaterI, medWater);
274  voCoolingPipeI->AddNode(voCoolingWaterI, 1, new TGeoTranslation(0., 0., 0.));
275 
276  //
277  // Define Yoke
278  //
279  TGeoPgon* shYoke = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 2);
280  shYoke->DefineSection(0, -kLYoke, kRYokeInner, kRYokeOuter);
281  shYoke->DefineSection(1, +kLYoke, kRYokeInner, kRYokeOuter);
282  //
283  TGeoVolume* voYoke = new TGeoVolume("L3YO", shYoke, medSteel);
284  voBMother->AddNode(voYoke, 1, new TGeoTranslation(0., 0., 0.));
285 
286  //
287  // Define Crown
288  //
289  TGeoPgon* shCrown = new TGeoPgon(kStartAngle, kFullAngle, kNSides, 4);
290  shCrown->DefineSection(0, kLCrown1, kRCrownInner, kRYokeInner);
291  shCrown->DefineSection(1, kLCrown2, kRCrownInner, kRYokeInner);
292  shCrown->DefineSection(2, kLCrown2, kRCrownInner, kRCrownOuter);
293  shCrown->DefineSection(3, kLCrown3, kRCrownInner, kRCrownOuter);
294  //
295  TGeoVolume* voCrown = new TGeoVolume("L3CR", shCrown, medSteel);
296 
297  //
298  // Door including "Plug"
299  //
300  Float_t slo = 2. * kRDoorOuter * TMath::Tan(22.5 * kDegRad);
301  Float_t sli = 2. * kRPlugInner * TMath::Tan(22.5 * kDegRad);
302  Double_t xpol1[12], xpol2[12], ypol1[12], ypol2[12];
303 
304  xpol1[0] = 2.;
305  ypol1[0] = kRDoorOuter;
306  xpol1[1] = slo / 2.;
307  ypol1[1] = kRDoorOuter;
308  xpol1[2] = kRDoorOuter;
309  ypol1[2] = slo / 2.;
310  xpol1[3] = kRDoorOuter;
311  ypol1[3] = -slo / 2.;
312  xpol1[4] = slo / 2.;
313  ypol1[4] = -kRDoorOuter;
314  xpol1[5] = 2.;
315  ypol1[5] = -kRDoorOuter;
316  xpol1[6] = 2.;
317  ypol1[6] = -kRPlugInner - os;
318  xpol1[7] = sli / 2.;
319  ypol1[7] = -kRPlugInner - os;
320  xpol1[8] = kRPlugInner;
321  ypol1[8] = -sli / 2. - os;
322  xpol1[9] = kRPlugInner;
323  ypol1[9] = sli / 2. - os;
324  xpol1[10] = sli / 2.;
325  ypol1[10] = kRPlugInner - os;
326  xpol1[11] = 2.;
327  ypol1[11] = kRPlugInner - os;
328 
329  TGeoXtru* shL3DoorR = new TGeoXtru(2);
330  shL3DoorR->DefinePolygon(12, xpol1, ypol1);
331  shL3DoorR->DefineSection(0, kLDoor1);
332  shL3DoorR->DefineSection(1, kLDoor2);
333  TGeoVolume* voL3DoorR = new TGeoVolume("L3DoorR", shL3DoorR, medSteel);
334 
335  for (Int_t i = 0; i < 12; i++) {
336  xpol2[i] = -xpol1[11 - i];
337  ypol2[i] = ypol1[11 - i];
338  }
339 
340  TGeoXtru* shL3DoorL = new TGeoXtru(2);
341  shL3DoorL->DefinePolygon(12, xpol2, ypol2);
342  shL3DoorL->DefineSection(0, kLDoor1);
343  shL3DoorL->DefineSection(1, kLDoor2);
344  TGeoVolume* voL3DoorL = new TGeoVolume("L3DoorL", shL3DoorL, medSteel);
345  //
346  // Plug support plate
347  //
348  Float_t ro = kRPlugInner + 50.;
349  slo = 2. * ro * TMath::Tan(22.5 * kDegRad);
350 
351  xpol1[0] = 2.;
352  ypol1[0] = ro - os;
353  xpol1[1] = slo / 2.;
354  ypol1[1] = ro - os;
355  xpol1[2] = ro;
356  ypol1[2] = slo / 2. - os;
357  xpol1[3] = ro;
358  ypol1[3] = -slo / 2. - os;
359  xpol1[4] = slo / 2.;
360  ypol1[4] = -ro - os;
361  xpol1[5] = 2.;
362  ypol1[5] = -ro - os;
363 
364  for (Int_t i = 0; i < 12; i++) {
365  xpol2[i] = -xpol1[11 - i];
366  ypol2[i] = ypol1[11 - i];
367  }
368 
369  TGeoXtru* shL3PlugSPR = new TGeoXtru(2);
370  shL3PlugSPR->DefinePolygon(12, xpol1, ypol1);
371  shL3PlugSPR->DefineSection(0, kLDoor1 - 10.);
372  shL3PlugSPR->DefineSection(1, kLDoor1);
373  TGeoVolume* voL3PlugSPR = new TGeoVolume("L3PlugSPR", shL3PlugSPR, medSteel);
374 
375  TGeoXtru* shL3PlugSPL = new TGeoXtru(2);
376  shL3PlugSPL->DefinePolygon(12, xpol2, ypol2);
377  shL3PlugSPL->DefineSection(0, kLDoor1 - 10.);
378  shL3PlugSPL->DefineSection(1, kLDoor1);
379  TGeoVolume* voL3PlugSPL = new TGeoVolume("L3PlugSPL", shL3PlugSPL, medSteel);
380 
381  // Position crown and door
382  TGeoRotation* rotxz = new TGeoRotation("rotxz", 90., 0., 90., 90., 180., 0.);
383 
384  TGeoVolumeAssembly* l3 = new TGeoVolumeAssembly("L3MO");
385  voBMother->AddNode(voCrown, 1, new TGeoTranslation(0., 0., 0.));
386  voBMother->AddNode(voCrown, 2, new TGeoCombiTrans(0., 0., 0., rotxz));
387  l3->AddNode(voBMother, 1, new TGeoTranslation(0., 0., 0.));
388  l3->AddNode(voL3DoorR, 1, new TGeoTranslation(0., 0., 0.));
389  l3->AddNode(voL3DoorR, 2, new TGeoCombiTrans(0., 0., 0., rotxz));
390  l3->AddNode(voL3DoorL, 1, new TGeoTranslation(0., 0., 0.));
391  l3->AddNode(voL3DoorL, 2, new TGeoCombiTrans(0., 0., 0., rotxz));
392  l3->AddNode(voL3PlugSPR, 1, new TGeoTranslation(0., 0., 0.));
393  l3->AddNode(voL3PlugSPR, 2, new TGeoCombiTrans(0., 0., 0., rotxz));
394  l3->AddNode(voL3PlugSPL, 1, new TGeoTranslation(0., 0., 0.));
395  l3->AddNode(voL3PlugSPL, 2, new TGeoCombiTrans(0., 0., 0., rotxz));
396  top->AddNode(l3, 1, new TGeoTranslation(0., os, 0.));
397 }
