42 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
43 {
44 
45  auto config = cfgc.options().get<std::string>("tof-compressor-config");
46  // auto outputDesc = cfgc.options().get<std::string>("output-desc");
47  auto rdhVersion = cfgc.options().get<int>("tof-compressor-rdh-version");
48  auto verbose = cfgc.options().get<bool>("tof-compressor-verbose");
49  std::vector<OutputSpec> outputs;
50  outputs.emplace_back(OutputSpec(ConcreteDataTypeMatcher{"TOF", "CRAWDATA"}));
51 
52  AlgorithmSpec algoSpec;
53  if (rdhVersion == 4) {
54  if (verbose) {
55  algoSpec = AlgorithmSpec{adaptFromTask<o2::tof::CompressorTask<o2::header::RAWDataHeaderV4, true>>()};
56  } else {
57  algoSpec = AlgorithmSpec{adaptFromTask<o2::tof::CompressorTask<o2::header::RAWDataHeaderV4, false>>()};
58  }
59  } else if (rdhVersion == 6) {
60  if (verbose) {
61  algoSpec = AlgorithmSpec{adaptFromTask<o2::tof::CompressorTask<o2::header::RAWDataHeaderV6, true>>()};
62  } else {
63  algoSpec = AlgorithmSpec{adaptFromTask<o2::tof::CompressorTask<o2::header::RAWDataHeaderV6, false>>()};
64  }
65  }
66 
67  WorkflowSpec workflow;
68 
84  std::stringstream ssconfig(config);
85  std::string iconfig;
86  int idevice = 0;
87 
88  while (getline(ssconfig, iconfig, ',')) {
89  workflow.emplace_back(DataProcessorSpec{
90  std::string("tof-compressor-") + std::to_string(idevice),
91  select(iconfig.c_str()),
92  outputs,
93  algoSpec,
94  Options{
95  {"tof-compressor-conet-mode", VariantType::Bool, false, {"Decoder CONET flag"}},
96  {"tof-compressor-decoder-verbose", VariantType::Bool, false, {"Decoder verbose flag"}},
97  {"tof-compressor-encoder-verbose", VariantType::Bool, false, {"Encoder verbose flag"}},
98  {"tof-compressor-checker-verbose", VariantType::Bool, false, {"Checker verbose flag"}}}});
99  idevice++;
100  }
101 
102  return workflow;
103 }
