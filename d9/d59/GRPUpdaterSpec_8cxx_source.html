36 class GRPDPLUpdatedTask
37 {
38  using GRP = o2::parameters::GRPObject;
39 
40  public:
41  GRPDPLUpdatedTask(const std::string& grpfilename) : mGRPFileName{grpfilename} {}

43  void init(framework::InitContext& ic)
44  {
45  // nothing special to be set up
46  }

48  void run(framework::ProcessingContext& pc)
49  {
50  const std::string grpName = "GRP";
51  if (mFinished) {
52  return;
53  }
54 
55  TFile flGRP(mGRPFileName.c_str(), "update");
56  if (flGRP.IsZombie()) {
57  LOG(ERROR) << "Failed to open in update mode " << mGRPFileName;
58  return;
59  }
60  std::unique_ptr<GRP> grp(static_cast<GRP*>(flGRP.GetObjectChecked(grpName.c_str(), GRP::Class())));
61  for (auto det : sDetList) { // get readout mode data from different detectors
62  auto roMode = pc.inputs().get<o2::parameters::GRPObject::ROMode>(det.getName());
63  if (!(roMode & o2::parameters::GRPObject::PRESENT)) {
64  LOG(ERROR) << "Detector " << det.getName() << " is read out while processor set ABSENT";
65  continue;
66  }
67  grp->setDetROMode(det, roMode);
68  }
69  LOG(INFO) << "Updated GRP in " << mGRPFileName << " for detectors RO mode";
70  grp->print();
71  flGRP.WriteObjectAny(grp.get(), grp->Class(), grpName.c_str());
72  flGRP.Close();
73  mFinished = true;
74 
75  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
76  }
77 
78  private:
79  bool mFinished = false;
80  std::string mGRPFileName = "o2sim_grp.root";
81 };

84 o2::framework::DataProcessorSpec getGRPUpdaterSpec(const std::string& grpfilename, const std::vector<o2::detectors::DetID>& detList)
85 {
86  sDetList = detList;
87  static constexpr std::array<o2::header::DataOrigin, o2::detectors::DetID::nDetectors> sOrigins = {
88  o2::header::gDataOriginITS, o2::header::gDataOriginTPC, o2::header::gDataOriginTRD,
89  o2::header::gDataOriginTOF, o2::header::gDataOriginPHS, o2::header::gDataOriginCPV,
90  o2::header::gDataOriginEMC, o2::header::gDataOriginHMP, o2::header::gDataOriginMFT,
91  o2::header::gDataOriginMCH, o2::header::gDataOriginMID, o2::header::gDataOriginZDC,
92  o2::header::gDataOriginFT0, o2::header::gDataOriginFV0, o2::header::gDataOriginFDD,
93  o2::header::gDataOriginACO};
94 
95  // prepare specs
96  std::vector<InputSpec> inputs;
97  for (const auto det : detList) {
98  inputs.emplace_back(InputSpec{det.getName(), sOrigins[det], "ROMode",
99  static_cast<SubSpecificationType>(0 /*det.second*/), Lifetime::Timeframe});
100  }
101 
102  return DataProcessorSpec{
103  "GRPUpdater",
104  inputs, // input status from each detector
105  {}, // no output
106  AlgorithmSpec{adaptFromTask<GRPDPLUpdatedTask>(grpfilename)},
107  Options{/* for the moment no options */}};
108 }
