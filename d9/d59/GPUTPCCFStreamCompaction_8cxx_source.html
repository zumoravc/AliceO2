135 GPUdii() void GPUTPCCFStreamCompaction::compactImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUSharedMemory& smem,
136  const ChargePos* in,
137  ChargePos* out,
138  const uchar* predicate,
139  int* newIdx,
140  const int* incr,
141  int nElems,
142  SizeT bufferSize)
143 {
144  int gid = get_group_id(0);
145  int idx = get_global_id(0);
146 
147  int lastItem = get_global_size(0) - 1;
148 
149  bool iAmDummy = (idx >= nElems);
150 
151  int pred = (iAmDummy) ? 0 : predicate[idx];
152  int scanRes = work_group_scan_inclusive_add(pred);
153 
154  SizeT compIdx = scanRes;
155  if (gid) {
156  compIdx += incr[gid - 1];
157  }
158 
159  SizeT tgtIdx = compIdx - 1;
160  if (pred && tgtIdx < bufferSize) {
161  out[tgtIdx] = in[idx];
162  }
163 
164  if (idx == lastItem) {
165  newIdx[idx] = CAMath::Min(compIdx, bufferSize); // TODO: Eventually, we can just return the last value, no need to store to memory
166  }
167 }
