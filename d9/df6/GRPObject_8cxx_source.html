27 float GRPObject::getSqrtS() const
28 {
29  // get center of mass energy
30  double e0 = getBeamEnergyPerNucleon(BeamClockWise);
31  double e1 = getBeamEnergyPerNucleon(BeamAntiClockWise);
32  if (e0 <= MassProton || e1 <= MassProton) {
33  return 0.f;
34  }
35  double beta0 = 1. - MassProton * MassProton / (e0 * e0);
36  double beta1 = 1. - MassProton * MassProton / (e1 * e1);
37  beta0 = beta0 > 0 ? sqrt(beta0) : 0.;
38  beta1 = beta1 > 0 ? sqrt(beta1) : 0.;
39  double ss = 2. * (MassProton * MassProton + e0 * e1 * (1. + beta0 * beta1 * cos(mCrossingAngle)));
40  return ss > 0. ? sqrt(ss) : 0.;
41 }

73 void GRPObject::setDetROMode(o2::detectors::DetID id, ROMode status)
74 {
76  if (!(status & PRESENT)) {
77  remDetReadOut(id);
78  return;
79  }
80  addDetReadOut(id);
81  if ((status & CONTINUOUS) == CONTINUOUS) {
82  addDetContinuousReadOut(id);
83  } else {
84  remDetContinuousReadOut(id);
85  }
86  if ((status & TRIGGERING) == TRIGGERING) {
87  addDetTrigger(id);
88  } else {
89  remDetTrigger(id);
90  }

94 GRPObject::ROMode GRPObject::getDetROMode(o2::detectors::DetID id) const
95 {
96  GRPObject::ROMode status = ABSENT;
97  if (isDetReadOut(id)) {
98  status = PRESENT;
99  } else {
100  if (isDetContinuousReadOut(id)) {
101  status = GRPObject::ROMode(status | CONTINUOUS);
102  }
103  if (isDetTriggers(id)) {
104  status = GRPObject::ROMode(status | TRIGGERING);
105  }
106  }
107  return status;
108 }
