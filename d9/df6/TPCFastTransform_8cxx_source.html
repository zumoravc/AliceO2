156 int TPCFastTransform::writeToFile(std::string outFName, std::string name)
157 {
159  assert(isConstructed());
160 
161  if (outFName.empty())
162  outFName = "tpcFastTransform.root";
163 
164  if (name.empty())
165  name = "TPCFastTransform";
166 
167  TFile outf(outFName.data(), "recreate");
168  if (outf.IsZombie()) {
169  LOG(ERROR) << "Failed to open output file " << outFName;
170  return -1;
171  }
172 
173  bool isBufferExternal = !isBufferInternal();
174  if (isBufferExternal)
175  adoptInternalBuffer(mFlatBufferPtr);
176  outf.WriteObjectAny(this, Class(), name.data());
177  outf.Close();
178  if (isBufferExternal)
179  clearInternalBufferPtr();
180  return 0;
181 }

183 TPCFastTransform* TPCFastTransform::loadFromFile(std::string inpFName, std::string name)
184 {
186 
187  if (inpFName.empty())
188  inpFName = "tpcFastTransform.root";
189 
190  if (name.empty())
191  name = "TPCFastTransform";
192 
193  TFile inpf(inpFName.data());
194  if (inpf.IsZombie()) {
195  LOG(ERROR) << "Failed to open input file " << inpFName;
196  return nullptr;
197  }
198  TPCFastTransform* transform = reinterpret_cast<TPCFastTransform*>(inpf.GetObjectChecked(name.data(), TPCFastTransform::Class()));
199  if (!transform) {
200  LOG(ERROR) << "Failed to load " << name << " from " << inpFName;
201  return nullptr;
202  }
203  if (transform->mFlatBufferSize > 0 && transform->mFlatBufferContainer == nullptr) {
204  LOG(ERROR) << "Failed to load " << name << " from " << inpFName << ": empty flat buffer container";
205  return nullptr;
206  }
207  transform->setActualBufferAddress(transform->mFlatBufferContainer);
208  return transform;
209 }
