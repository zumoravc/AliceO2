167 GPUdi() void GPUTPCGMPolynomialField::GetField(float x, float y, float z, float B[3]) const
168 {
169  const float* fBxS = &mTpcBx[1];
170  const float* fByS = &mTpcBy[1];
171  const float* fBzS = &mTpcBz[1];
172 
173  const float f[NTPCM - 1] = {x, y, z, x * x, x * y, x * z, y * y, y * z, z * z};
174  float bx = mTpcBx[0], by = mTpcBy[0], bz = mTpcBz[0];
175  for (int i = NTPCM - 1; i--;) {
176  // for (int i=0;i<NTPCM-1; i++){
177  bx += fBxS[i] * f[i];
178  by += fByS[i] * f[i];
179  bz += fBzS[i] * f[i];
180  }
181  B[0] = bx;
182  B[1] = by;
183  B[2] = bz;
184 }

223 GPUdi() void GPUTPCGMPolynomialField::GetFieldTrd(float x, float y, float z, float B[3]) const
224 {
225  float f[NTRDM];
226  GetPolynomsTrd(x, y, z, f);
227  float bx = 0.f, by = 0.f, bz = 0.f;
228  for (int i = 0; i < NTRDM; i++) {
229  bx += mTrdBx[i] * f[i];
230  by += mTrdBy[i] * f[i];
231  bz += mTrdBz[i] * f[i];
232  }
233  B[0] = bx;
234  B[1] = by;
235  B[2] = bz;
236 }
