53 {
54  constexpr static size_t NSectors = o2::tpc::Sector::MAXSECTOR;
55  using DataDescription = o2::header::DataDescription;

64  auto initFunction = [sendMC](InitContext& ic) {
65  // there is nothing to init at the moment
66  auto processAttributes = std::make_shared<ProcessAttributes>();
67  processAttributes->decoder = std::make_unique<HardwareClusterDecoder>();

70  auto processSectorFunction = [processAttributes](ProcessingContext& pc, DataRef const& ref, DataRef const& mclabelref) {
71  auto& decoder = processAttributes->decoder;
72  auto& verbosity = processAttributes->verbosity;
73  auto& activeInputs = processAttributes->activeInputs;
74  // this will return a span of TPC clusters
75  auto size = o2::framework::DataRefUtils::getPayloadSize(ref);
76  auto const* dataHeader = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
77  o2::header::DataHeader::SubSpecificationType fanSpec = dataHeader->subSpecification;
78 
79  // init the stacks for forwarding the sector header
80  // FIXME check if there is functionality in the DPL to forward the stack
81  // FIXME make one function
82  o2::header::Stack rawHeaderStack;
83  o2::header::Stack mcHeaderStack;
84  o2::tpc::TPCSectorHeader const* sectorHeaderMC = nullptr;
85  if (DataRefUtils::isValid(mclabelref)) {
86  sectorHeaderMC = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(mclabelref);
87  if (sectorHeaderMC) {
88  o2::header::Stack actual{*sectorHeaderMC};
89  std::swap(mcHeaderStack, actual);
90  if (sectorHeaderMC->sector() < 0) {
91  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLNATIVEMCLBL"), fanSpec, Lifetime::Timeframe, std::move(mcHeaderStack)}, fanSpec);
92  }
93  }
94  }
95  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
96  if (sectorHeader) {
97  o2::header::Stack actual{*sectorHeader};
98  std::swap(rawHeaderStack, actual);
99  if (sectorHeader->sector() < 0) {
100  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLUSTERNATIVE"), fanSpec, Lifetime::Timeframe, std::move(rawHeaderStack)}, fanSpec);
101  return;
102  }
103  }
104  assert(sectorHeaderMC == nullptr || sectorHeader->sector() == sectorHeaderMC->sector());
105 
106  // input to the decoder is a vector of raw pages description ClusterHardwareContainer,
107  // each specified as a pair of pointer to ClusterHardwareContainer and the number
108  // of pages in that buffer
109  // FIXME: better description of the raw page
110  size_t nPages = size / 8192;
111  std::vector<std::pair<const ClusterHardwareContainer*, std::size_t>> inputList;
112  if (verbosity > 0 && !DataRefUtils::isValid(mclabelref)) {
113  LOG(INFO) << "Decoder input: " << size << ", " << nPages << " pages for sector " << sectorHeader->sector();
114  }
115 
116  // MC labels are received as one container of labels in the sequence matching clusters
117  // in the raw pages
118  std::vector<MCLabelContainer> mcinCopies;
119  std::unique_ptr<const MCLabelContainer> mcin;
120  if (DataRefUtils::isValid(mclabelref)) {
121  mcin = std::move(pc.inputs().get<MCLabelContainer*>(mclabelref));
122  mcinCopies.resize(nPages);
123  if (verbosity > 0) {
124  LOG(INFO) << "Decoder input: " << size << ", " << nPages << " pages, " << mcin->getIndexedSize() << " MC label sets for sector " << sectorHeader->sector();
125  }
126  }
127 
128  // FIXME: the decoder takes vector of MCLabelContainers as input and the retreived
129  // input can not be moved because the input is const, so we have to copy
130  // Furthermore, the current decoder implementation supports handling of MC labels
131  // only for single 8kb pages. So we have to add the raw pages individually and create
132  // MC label containers for the corresponding clusters.
133  size_t mcinPos = 0;
134  size_t totalNumberOfClusters = 0;
135  for (size_t page = 0; page < nPages; page++) {
136  inputList.emplace_back(reinterpret_cast<const ClusterHardwareContainer*>(ref.payload + page * 8192), 1);
137  const ClusterHardwareContainer& container = *(inputList.back().first);
138  if (verbosity > 1) {
139  LOG(INFO) << "Decoder input in page " << std::setw(2) << page << ": " //
140  << "CRU " << std::setw(3) << container.CRU << " " //
141  << std::setw(3) << container.numberOfClusters << " cluster(s)"; //
142  }
143  totalNumberOfClusters += container.numberOfClusters;
144  if (mcin) {
145  for (size_t mccopyPos = 0;
146  mccopyPos < container.numberOfClusters && mcinPos < mcin->getIndexedSize();
147  mccopyPos++, mcinPos++) {
148  for (auto const& label : mcin->getLabels(mcinPos)) {
149  mcinCopies[page].addElement(mccopyPos, label);
150  }
151  }
152  }
153  }
154  // FIXME: introduce error handling policy: throw, ignore, warn
155  //assert(!mcin || mcinPos == mcin->getIndexedSize());
156  if (mcin && mcinPos != totalNumberOfClusters) {
157  LOG(ERROR) << "inconsistent number of MC label objects processed"
158  << ", expecting MC label objects for " << totalNumberOfClusters << " cluster(s)"
159  << ", got " << mcin->getIndexedSize();
160  }
161  // output of the decoder is sorted in (sector,globalPadRow) coordinates, individual
162  // containers are created for clusters and MC labels per (sector,globalPadRow) address
163  char* outputBuffer = nullptr;
164  auto outputAllocator = [&pc, &fanSpec, &outputBuffer, &rawHeaderStack](size_t size) -> char* {
165  outputBuffer = pc.outputs().newChunk(Output{gDataOriginTPC, DataDescription("CLUSTERNATIVE"), fanSpec, Lifetime::Timeframe, std::move(rawHeaderStack)}, size).data();
166  return outputBuffer;
167  };
168  MCLabelContainer mcout;
169  decoder->decodeClusters(inputList, outputAllocator, (mcin ? &mcinCopies : nullptr), &mcout);
170 
171  // TODO: reestablish the logging messages on the raw buffer
172  // if (verbosity > 1) {
173  // LOG(INFO) << "decoder " << std::setw(2) << sectorHeader->sector() //
174  // << ": decoded " << std::setw(4) << coll.clusters.size() << " clusters on sector " //
175  // << std::setw(2) << (int)coll.sector << "[" << (int)coll.globalPadRow << "]"; //
176  // }
177 
178  if (DataRefUtils::isValid(mclabelref)) {
179  if (verbosity > 0) {
180  LOG(INFO) << "sending " << mcout.getIndexedSize()
181  << " label object(s)" << std::endl;
182  }
183  // serialize the complete list of MC label containers
184  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLNATIVEMCLBL"), fanSpec, Lifetime::Timeframe, std::move(mcHeaderStack)}, mcout);
185  }
186  };

188  auto processingFct = [processAttributes, processSectorFunction](ProcessingContext& pc) {
189  struct SectorInputDesc {
190  DataRef dataref;
191  DataRef mclabelref;
192  };
193  // loop over all inputs and their parts and associate data with corresponding mc truth data
194  // by the subspecification
195  std::map<int, SectorInputDesc> inputs;
196  std::vector<InputSpec> filter = {
197  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHW"}, Lifetime::Timeframe},
198  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHWMCLBL"}, Lifetime::Timeframe},
199  };
200  for (auto const& inputRef : InputRecordWalker(pc.inputs(), filter)) {
201  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(inputRef);
202  if (sectorHeader == nullptr) {
203  LOG(ERROR) << "sector header missing on header stack for input on " << inputRef.spec->binding;
204  continue;
205  }
206  const int sector = sectorHeader->sector();
207  if (DataRefUtils::match(inputRef, {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHW"}})) {
208  inputs[sector].dataref = inputRef;
209  }
210  if (DataRefUtils::match(inputRef, {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHWMCLBL"}})) {
211  inputs[sector].mclabelref = inputRef;
212  }
213  }
214  for (auto const& input : inputs) {
215  if (processAttributes->sendMC && !DataRefUtils::isValid(input.second.mclabelref)) {
216  throw std::runtime_error("missing the required MC label data for sector " + std::to_string(input.first));
217  }
218  processSectorFunction(pc, input.second.dataref, input.second.mclabelref);
219  }
220  };
221 
222  // return the actual processing function as a lambda function using variables
223  // of the init function
224  return processingFct;
225  };

227  auto createInputSpecs = [](bool makeMcInput) {
228  std::vector<InputSpec> inputSpecs{
229  InputSpec{{"rawin"}, gDataOriginTPC, "CLUSTERHW", 0, Lifetime::Timeframe},
230  };
231  if (makeMcInput) {
232  // FIXME: define common data type specifiers
233  constexpr o2::header::DataDescription datadesc("CLUSTERHWMCLBL");
234  inputSpecs.emplace_back(InputSpec{"mclblin", gDataOriginTPC, datadesc, 0, Lifetime::Timeframe});
235  }
236  return std::move(inputSpecs);
237  };
238 
239  auto createOutputSpecs = [](bool makeMcOutput) {
240  std::vector<OutputSpec> outputSpecs{
241  OutputSpec{{"clusterout"}, gDataOriginTPC, "CLUSTERNATIVE", 0, Lifetime::Timeframe},
242  };
243  if (makeMcOutput) {
244  OutputLabel label{"mclblout"};
245  // have to use a new data description, routing is only based on origin and decsription
246  constexpr o2::header::DataDescription datadesc("CLNATIVEMCLBL");
247  outputSpecs.emplace_back(label, gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
248  }
249  return std::move(outputSpecs);
250  };
251 
252  return DataProcessorSpec{processorName,
253  {createInputSpecs(sendMC)},
254  {createOutputSpecs(sendMC)},
255  AlgorithmSpec(initFunction)};
