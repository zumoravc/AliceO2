137 inline double TRDSimParam::TimeResponse(double time) const
138 {
139  //
140  // Applies the preamp shaper time response
141  // (We assume a signal rise time of 0.2us = fTRFlo/2.
142  //
143 
144  double rt = (time - .5 * mTRFlo) * mInvTRFwid;
145  int iBin = (int)rt;
146  double dt = rt - iBin;
147  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
148  return mTRFsmp[iBin] + (mTRFsmp[iBin + 1] - mTRFsmp[iBin]) * dt;
149  } else {
150  return 0.0;
151  }
152 }
153 
154 inline double TRDSimParam::CrossTalk(double time) const
155 {
156  //
157  // Applies the pad-pad capacitive cross talk
158  //
159 
160  double rt = (time - mTRFlo) * mInvTRFwid;
161  int iBin = (int)rt;
162  double dt = rt - iBin;
163  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
164  return mCTsmp[iBin] + (mCTsmp[iBin + 1] - mCTsmp[iBin]) * dt;
165  } else {
166  return 0.0;
167  }
168 }
