43 template <class RawHeader>
44 void RawReaderMemory<RawHeader>::next()
45 {
46  mRawPayload.reset();
47  bool isDataTerminated = false;
48  do {
49  nextPage(false);
50  // check if we find a valid RCU trailer
51  // the payload must be at the end of the buffer
52  // if not present and error will be thrown
53  try {
54  RCUTrailer::constructFromPayloadWords(mRawBuffer.getDataWords());
55  isDataTerminated = true;
56  } catch (...) {
57  }
58  } while (!isDataTerminated);
59 }

61 template <class RawHeader>
62 void RawReaderMemory<RawHeader>::nextPage(bool doResetPayload)
63 {
64  if (!hasNext())
65  throw RawDecodingError(RawDecodingError::ErrorType_t::PAGE_NOTFOUND);
66  if (doResetPayload)
67  mRawPayload.reset();
68  mRawHeaderInitialized = false;
69  mPayloadInitialized = false;
70  // Use std::string stream as byte stream
71  std::string headerwords(mRawMemoryBuffer.data() + mCurrentPosition, sizeof(RawHeader) / sizeof(char));
72  std::stringstream headerstream(headerwords);
73  // Read header
74  try {
75  headerstream >> mRawHeader;
76  mRawHeaderInitialized = true;
77  } catch (...) {
78  throw RawDecodingError(RawDecodingError::ErrorType_t::HEADER_DECODING);
79  }
80  if (mCurrentPosition + sizeof(RawHeader) + mRawHeader.memorySize >= mRawMemoryBuffer.size()) {
81  // Payload incomplete
82  throw RawDecodingError(RawDecodingError::ErrorType_t::PAYLOAD_DECODING);
83  } else {
84  mRawBuffer.readFromMemoryBuffer(gsl::span<const char>(mRawMemoryBuffer.data() + mCurrentPosition + sizeof(RawHeader), mRawHeader.memorySize));
85  mRawPayload.appendPayloadWords(mRawBuffer.getDataWords());
86  mRawPayload.increasePageCount();
87  }
88  mCurrentPosition += mRawHeader.offsetToNext;
89 }
