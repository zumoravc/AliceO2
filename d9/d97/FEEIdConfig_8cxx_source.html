78 bool FEEIdConfig::load(const char* filename)
79 {
84 
85  mGBTIdToFeeId.clear();
86  std::ifstream inFile(filename);
87  if (!inFile.is_open()) {
88  return false;
89  }
90  std::string line, token;
91  while (std::getline(inFile, line)) {
92  if (std::count(line.begin(), line.end(), ' ') < 3) {
93  continue;
94  }
95  if (line.find('#') < line.find(' ')) {
96  continue;
97  }
98  std::stringstream ss;
99  ss << line;
100  std::getline(ss, token, ' ');
101  uint16_t feeId = std::atoi(token.c_str());
102  std::getline(ss, token, ' ');
103  uint8_t linkId = std::atoi(token.c_str());
104  std::getline(ss, token, ' ');
105  uint8_t endPointId = std::atoi(token.c_str());
106  std::getline(ss, token, ' ');
107  uint16_t cruId = std::atoi(token.c_str());
108  mGBTIdToFeeId[getGBTId(linkId, endPointId, cruId)] = feeId;
109  }
110  inFile.close();
111  return true;
112 }
