31 bool CalibParams::setGain(TH2* h, char module)
32 {
33  const char MAXX = 64,
34  MAXZ = 56;
35  if (!h) {
36  LOG(ERROR) << "no input histogam";
37  return false;
38  }
39 
40  if (h->GetNbinsX() != MAXX || h->GetNbinsY() != MAXZ) {
41  LOG(ERROR) << "Wrong dimentions of input histogram:" << h->GetNbinsX() << "," << h->GetNbinsY() << " instead of " << MAXX << "," << MAXZ;
42  return false;
43  }
44 
45  auto geo = Geometry::GetInstance();
46  if (!geo) {
47  LOG(ERROR) << "Geometry needs to be initialized";
48  return false;
49  }
50 
51  char relid[3] = {module, 1, 1};
52  short absId;
53  for (char ix = 1; ix <= MAXX; ix++) {
54  relid[1] = ix;
55  for (char iz = 1; iz <= MAXZ; iz++) {
56  relid[2] = iz;
57 
58  if (geo->relToAbsNumbering(relid, absId)) {
59  mGainCalib[absId] = h->GetBinContent(ix, iz);
60  }
61  }
62  }
63  return true;
64 }
