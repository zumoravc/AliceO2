19 // createDigits generates N digits with random id
20 // (hence some might get duplicated).
21 std::vector<Digit> createDigits(int N)
22 {
23  std::vector<Digit> digits;
24  float dummyadc{42.0};
25  std::srand(std::time(nullptr)); // use current time as seed for random generator
26  Digit::Time dummytime;
27  int dummydetID = 100; //to be improved, timing depending on that
28 
29  for (auto i = 0; i < N; i++) {
30  int randomPadID = std::rand() * N;
31  digits.emplace_back(dummydetID, randomPadID, dummyadc, dummytime);
32  }
33 
34  return digits;
35 }

52 // benchDigitMerging create fake digits and merges them
53 // using one of the merging functions.
54 static void benchDigitMerging(benchmark::State& state)
55 {
56  auto digits = createDigits(100);
57  auto labels = createLabels(100);
58 
59  auto mergingFunction = mergingFunctions()[state.range(0)];
60 
61  for (auto _ : state) {
62  mergingFunction(digits, labels);
63  }
64 }
