11 // Compatibility layer for a few common builtins found in different compilers.
12 // These are trivial and the goal is to make sure we can compile on non GCC /
13 // clang compilers if requested.
14 #ifndef O2_FRAMEWORK_COMPILER_BUILTINS_H_
15 #define O2_FRAMEWORK_COMPILER_BUILTINS_H_
16 #if __GNUC__
17 #define O2_BUILTIN_UNREACHABLE __builtin_unreachable
18 #elif __clang__
19 #define O2_BUILTIN_UNREACHABLE __builtin_unreachable
20 #else
21 #define O2_BUILTIN_UNREACHABLE
22 #endif
23 
24 #if __GNUC__
25 #define O2_BUILTIN_LIKELY(x) __builtin_expect((x), 1)
26 #define O2_BUILTIN_UNLIKELY(x) __builtin_expect((x), 0)
27 #elif __clang__
28 #define O2_BUILTIN_LIKELY(x) __builtin_expect((x), 1)
29 #define O2_BUILTIN_UNLIKELY(x) __builtin_expect((x), 0)
30 #else
31 #define O2_BUILTIN_LIKELY(x)
32 #define O2_BUILTIN_UNLIKELY(x)
33 #endif
34 
35 #if __GNUC__
36 #define O2_BUILTIN_PREFETCH(x, ...) __builtin_prefetch((x), __VA_ARGS__)
37 #elif __clang__
38 #define O2_BUILTIN_PREFETCH(x, ...) __builtin_prefetch((x), __VA_ARGS__)
39 #else
40 #define O2_BUILTIN_PREFETCH(x, ...)
41 #endif
42 
43 #if __GNUC__
44 #define O2_VISIBILITY_HIDDEN __attribute__((visibility("hidden")))
45 #elif __clang__
46 #define O2_VISIBILITY_HIDDEN __attribute__((visibility("hidden")))
47 #else
48 #define O2_VISIBILITY_HIDDEN
49 #endif
50 
51 #endif // O2_FRAMEWORK_COMPILER_BUILTINS_H_
