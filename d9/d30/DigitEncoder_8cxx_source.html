46 // create a function that return the (DsElecId,chid) of a digit
47 Digit2ElecMapper createDigit2ElecMapper(Det2ElecMapper det2elec)
48 {
49  return [det2elec](const o2::mch::Digit& digit) -> std::optional<std::pair<DsElecId, int>> {
50  int deid = digit.getDetID();
51  int dsid = mapping::segmentation(deid).padDualSampaId(digit.getPadID());
52  DsDetId detId{deid, dsid};
53  auto dselocopt = det2elec(DsDetId(deid, dsid));
54  if (!dselocopt.has_value()) {
55  //std::cout << fmt::format("WARNING : got no location for (de,ds)=({},{})\n", deid, dsid);
56  return std::nullopt;
57  }
58  DsElecId elecId = dselocopt.value();
59  int dschid = mapping::segmentation(deid).padDualSampaChannel(digit.getPadID());
60  return std::make_pair(dselocopt.value(), dschid);
61  };

64 template <typename FORMAT, typename CHARGESUM>
65 DigitEncoder createDigitEncoder(Det2ElecMapper det2elec)
66 {
67  auto encoder = createPayloadEncoder<FORMAT, CHARGESUM>().release();
68  auto digit2elec = createDigit2ElecMapper(det2elec);
69 
70  return [encoder, digit2elec](gsl::span<o2::mch::Digit> digits,
71  std::vector<std::byte>& buffer,
72  uint32_t orbit,
73  uint16_t bc) {
74  static int nadd{0};
75  encoder->startHeartbeatFrame(orbit, bc);
76 
77  for (auto d : digits) {
78  auto optElecId = digit2elec(d);
79  if (!optElecId.has_value()) {
80  continue;
81  }
82  auto elecId = optElecId.value().first;
83  int dschid = optElecId.value().second;
84  // FIXME: add a warning if timestamp is not in range
85  uint16_t ts(static_cast<int>(d.getTime().sampaTime) & 0x3FF);
86  int sampachid = dschid % 32;
87  // FIXME: add a warning if ADC is not in range
88  auto clusters = createSampaClusters<CHARGESUM>(ts, d.getADC() & 0xFFFFF);
89  encoder->addChannelData(elecId, sampachid, clusters);
90  }
91  encoder->moveToBuffer(buffer);
92  };
93 }
