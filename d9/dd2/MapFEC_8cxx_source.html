21 MapFEC::MapFEC(std::string_view content)
22 {
23  mDsMap.fill({-1, -1, -1});
24  int link_id, group_id, de, ds_id[5];
25  std::istringstream in(std::string{content});
26  while (in >> link_id >> group_id >> de >> ds_id[0] >> ds_id[1] >> ds_id[2] >> ds_id[3] >> ds_id[4]) {
27  for (int i = 0; i < 5; i++) {
28  if (ds_id[i] <= 0) {
29  continue;
30  }
31  int ds_addr = group_id * 5 + i;
32  int ix = index(link_id, ds_addr);
33  if (ix < 0) {
34  continue;
35  }
36  mDsMap.at(ix) = {de, ds_id[i], 0};
37  }
38  }
39 }

59 std::optional<o2::mch::raw::DsDetId> MapFEC::operator()(const o2::mch::raw::DsElecId& dsElecId) const
60 {
61  auto link_id = dsElecId.solarId();
62  auto ds_addr = dsElecId.elinkId();
63 
64  if (!size()) {
65  return std::nullopt;
66  }
67 
68  int ix = index(link_id, ds_addr);
69  if (ix < 0) {
70  return std::nullopt;
71  }
72  if (mDsMap.at(ix).bad == 1) {
73  return std::nullopt;
74  }
75  auto de = mDsMap.at(ix).deId;
76  auto dsid = mDsMap.at(ix).dsId;
77  if (de >= 0 && dsid >= 0) {
78  return o2::mch::raw::DsDetId(de, dsid);
79  }
80  return std::nullopt;
