60 float ModelGEM::getElectronCollectionEfficiency(float elecFieldAbove, float gemPotential, int geom)
61 {
62  // Fit model: CollTop 6.0 (fitted in a range of 0.0 <= ElecFieldRatioAbove <= 0.16)
63  float ElecFieldGEM = (0.001 * gemPotential) / (0.0001 * mFitElecEffThickness); //in kV/cm
64  float ElecFieldRatioAbove = elecFieldAbove / ElecFieldGEM;
65 
66  float eta1 = mFitElecEffTuneEta1[geom] * ElecFieldRatioAbove;
67  float eta2 = mFitElecEffFieldBelow / mFitElecEffFieldGEM;
68 
69  float paramC7Bar = getParameterC7Bar(eta1, eta2, geom);
70  float paramC8Bar = getParameterC8Bar(eta1, eta2, geom);
71  float paramC9Bar = getParameterC9Bar(eta1, eta2, geom);
72 
73  return 2.0 * Pi * (paramC7Bar + paramC9Bar * eta1 + paramC8Bar * eta2) /
74  (mParamC1[geom] + mParamC3[geom] * eta1 + mParamC2[geom] * eta2);
75 }

77 float ModelGEM::getElectronExtractionEfficiency(float elecFieldBelow, float gemPotential, int geom)
78 {
79  // Fit model: ExtrBot 6.1 (fitted in a range of 0.0 <= ElecFieldRatioBelow <= 0.16)
80  float ElecFieldGEM = (0.001 * gemPotential) / (0.0001 * mFitElecEffThickness); //in kV/cm
81  float ElecFieldRatioBelow = elecFieldBelow / ElecFieldGEM;
82 
83  float eta1 = mFitElecEffFieldAbove / mFitElecEffFieldGEM;
84  float eta2 = mFitElecEffTuneEta2[geom] * ElecFieldRatioBelow;
85 
86  float paramC7 = getParameterC7(eta1, eta2, geom);
87  float paramC8 = getParameterC8(eta1, eta2, geom);
88  float paramC9 = getParameterC9(eta1, eta2, geom);
89 
90  return 2.0 * Pi * (paramC7 + paramC8 * eta1 + paramC9 * eta2) /
91  (std::pow(mFitElecEffTuneDiffusion[geom], 3.5) * mParamC4[geom] + mParamC5[geom] * eta1 + 1.0 / mFitElecEffTuneDiffusion[geom] * mParamC6[geom] * eta2);
92 }

113 float ModelGEM::getStackEnergyResolution()
114 {
115  float PrimaryCharges = PhotonEnergy / Wi;
116 
117  // Number of electrons which are collected at each GEM stage
118  std::array<float, 4> NumElectrons;
119 
120  const int NumOfGEMs = NumElectrons.size();
121 
122  for (int n = 0; n < NumElectrons.size(); ++n) {
123  float Attachment = 1.0 - mAttachment * mDistance[n]; //1.0 - 1/cm * cm => unitless
124 
125  if (n == 0) {
126  NumElectrons[n] = PrimaryCharges * getElectronCollectionEfficiency(mElectricField[n], mPotential[n], mGeometry[n]) * Attachment;
127 
128  } else {
129  NumElectrons[n] = NumElectrons[n - 1] * getAbsoluteGain(mPotential[n - 1], mGeometry[n - 1]) * getElectronExtractionEfficiency(mElectricField[n], mPotential[n - 1], mGeometry[n - 1]) * getElectronCollectionEfficiency(mElectricField[n], mPotential[n], mGeometry[n]) * Attachment;
130  }
131  }
132 
133  // Number of charges after extraction from last amplification stage
134  float Attachment = 1.0 - mAttachment * mDistance[NumOfGEMs];
135  NumElectrons[NumOfGEMs] = NumElectrons[NumOfGEMs - 1] * getAbsoluteGain(mPotential[NumOfGEMs - 1], mGeometry[NumOfGEMs - 1]) * getElectronExtractionEfficiency(mElectricField[NumOfGEMs], mPotential[NumOfGEMs - 1], mGeometry[NumOfGEMs - 1]) * Attachment;
136 
137  float SigmaOverMuSquare = 0.0;
138 
139  for (int n = 0; n <= NumOfGEMs - 1; ++n) {
140  float SingleGainFluctuation = getSingleGainFluctuation(mPotential[n], mGeometry[n]);
141 
142  if (n == 0) {
143  SigmaOverMuSquare += (Fano + SingleGainFluctuation) / NumElectrons[n];
144  } else {
145  SigmaOverMuSquare += SingleGainFluctuation / NumElectrons[n];
146  }
147  }
148 
149  mStackEnergyCalculated = 1;
150  mStackEffectiveGain = NumElectrons[NumOfGEMs] / PrimaryCharges;
151 
152  return std::sqrt(SigmaOverMuSquare);
153 }

278 float ModelGEM::getIntXEndBot(float eta1, float eta2, int geom)
279 {
280  float result;
281 
282  if (eta2 <= getEta2Kink1(eta1, geom)) {
283  result = -(mFitElecEffHoleDiameter + mFitElecEffWidth[geom]) / 4.0;
284  } else if (getEta2Kink1(eta1, geom) < eta2 && eta2 < getEta2Kink2(eta1, geom)) {
285  result = -mFitElecEffPitch[geom] / 2.0 + sqrt(getHtop2(geom) * (eta1 - 1.0) * (2.0 * Pi * eta2 + getHtop0(geom) * (1.0 - eta1))) /
286  (getHtop2(geom) * (eta1 - 1.0));
287  } else {
288  result = -mFitElecEffHoleDiameter / 2.0;
289  }
290 
291  return result;
