311 void V3Layer::createLayer(TGeoVolume* motherVolume)
312 {
313  const Int_t nameLen = 30;
314  char volumeName[nameLen];
315  Double_t xpos, ypos, zpos;
316  Double_t alpha;
317 
318  // Check if the user set the proper parameters
319  if (mLayerRadius <= 0) {
320  LOG(FATAL) << "Wrong layer radius " << mLayerRadius;
321  }
322 
323  if (mNumberOfStaves <= 0) {
324  LOG(FATAL) << "Wrong number of staves " << mNumberOfStaves;
325  }
326 
327  if (mNumberOfChips <= 0) {
328  LOG(FATAL) << "Wrong number of chips " << mNumberOfChips;
329  }
330 
331  if (mLayerNumber >= sNumberOfInnerLayers && mNumberOfModules <= 0) {
332  LOG(FATAL) << "Wrong number of modules " << mNumberOfModules;
333  }
334 
335  if (mChipThickness <= 0) {
336  LOG(FATAL) << "Chip thickness wrong or not set " << mChipThickness;
337  }
338 
339  if (mSensorThickness <= 0) {
340  LOG(FATAL) << "Sensor thickness wrong or not set " << mSensorThickness;
341  }
342 
343  if (mSensorThickness > mChipThickness) {
344  LOG(FATAL) << "Sensor thickness " << mSensorThickness << " is greater than chip thickness " << mChipThickness;
345  }
346 
347  // If a Turbo layer is requested, do it and exit
348  if (mIsTurbo) {
349  createLayerTurbo(motherVolume);
350  return;
351  }
352 
353  // First create the stave container
354  alpha = (360. / (2 * mNumberOfStaves)) * DegToRad();
355 
356  // mStaveWidth = mLayerRadius*Tan(alpha);
357 
358  snprintf(volumeName, nameLen, "%s%d", GeometryTGeo::getITSLayerPattern(), mLayerNumber);
359  TGeoVolume* layerVolume = new TGeoVolumeAssembly(volumeName);
360  layerVolume->SetUniqueID(mChipTypeID);
361 
362  // layerVolume->SetVisibility(kFALSE);
363  layerVolume->SetVisibility(kTRUE);
364  layerVolume->SetLineColor(1);
365 
366  TGeoVolume* stavVol = createStave();
367 
368  // Now build up the layer
369  alpha = 360. / mNumberOfStaves;
370  Double_t r = mLayerRadius + (static_cast<TGeoBBox*>(stavVol->GetShape()))->GetDY();
371  for (Int_t j = 0; j < mNumberOfStaves; j++) {
372  Double_t phi = j * alpha + mPhi0;
373  xpos = r * cosD(phi); // r*sinD(-phi);
374  ypos = r * sinD(phi); // r*cosD(-phi);
375  zpos = 0.;
376  phi += 90;
377  layerVolume->AddNode(stavVol, j, new TGeoCombiTrans(xpos, ypos, zpos, new TGeoRotation("", phi, 0, 0)));
378  }
379 
380  // Finally put everything in the mother volume
381  motherVolume->AddNode(layerVolume, 1, nullptr);
382 
383  // geometry is served
384  return;
385 }

387 void V3Layer::createLayerTurbo(TGeoVolume* motherVolume)
388 {
389  const Int_t nameLen = 30;
390  char volumeName[nameLen];
391  Double_t xpos, ypos, zpos;
392  Double_t alpha;
393 
394  // Check if the user set the proper (remaining) parameters
395  if (mStaveWidth <= 0) {
396  LOG(FATAL) << "Wrong stave width " << mStaveWidth;
397  }
398 
399  if (Abs(mStaveTilt) > 45) {
400  LOG(WARNING) << "Stave tilt angle (" << mStaveTilt << ") greater than 45deg";
401  }
402 
403  snprintf(volumeName, nameLen, "%s%d", GeometryTGeo::getITSLayerPattern(), mLayerNumber);
404  TGeoVolume* layerVolume = new TGeoVolumeAssembly(volumeName);
405  layerVolume->SetUniqueID(mChipTypeID);
406  layerVolume->SetVisibility(kTRUE);
407  layerVolume->SetLineColor(1);
408  TGeoVolume* stavVol = createStave();
409 
410  // Now build up the layer
411  alpha = 360. / mNumberOfStaves;
412  Double_t r = mLayerRadius /* +chip thick ?! */;
413  for (Int_t j = 0; j < mNumberOfStaves; j++) {
414  Double_t phi = j * alpha + mPhi0;
415  xpos = r * cosD(phi); // r*sinD(-phi);
