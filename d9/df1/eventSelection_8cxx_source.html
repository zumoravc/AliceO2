88  void process(aod::Collision const& collision, aod::BCs const& bcs, aod::Zdcs const& zdcs, aod::Run2V0s const& vzeros)
89  {
90  LOGF(debug, "Starting new event");
91  // CTP info
92  uint64_t triggerMask = collision.bc().triggerMask();
93  LOGF(debug, "triggerMask=%llu", triggerMask);
94  // fill fired aliases
95  int32_t alias[nAliases] = {0};
96  for (auto& al : mAliasToClassIds) {
97  for (auto& classIndex : al.second) {
98  alias[al.first] |= triggerMask & (1ul << classIndex);
99  }
100  }
101  // ZDC info
102  auto zdc = getZdc(collision.bc(), zdcs);
103  bool bbZNA = zdc.timeZNA() > -2. && zdc.timeZNA() < 2.;
104  bool bbZNC = zdc.timeZNC() > -2. && zdc.timeZNC() < 2.;
105  // VZERO info
106  auto vzero = getVZero(collision.bc(), vzeros);
107  float timeV0A = vzero.timeA();
108  float timeV0C = vzero.timeC();
109  // TODO replace it with configurable cuts from CCDB
110  bool bbV0A = timeV0A > 0. && timeV0A < 25.; // ns
111  bool bbV0C = timeV0C > 0. && timeV0C < 25.; // ns
112  bool bgV0A = timeV0A > -25. && timeV0A < 0.; // ns
113  bool bgV0C = timeV0C > -25. && timeV0C < 0.; // ns
114 
115  // Fill event selection columns
116  evsel(alias, bbV0A, bbV0C, bgV0A, bgV0C, bbZNA, bbZNC);
117  }
118 };
