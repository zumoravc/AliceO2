67 // a generic looper of data members of a TClass; calling a callback
68 // reused in various functions below
69 void loopOverMembers(TClass* cl, void* obj,
70  std::function<void(const TDataMember*, int, int)>&& callback)
71 {
72  auto memberlist = cl->GetListOfDataMembers();
73  for (int i = 0; i < memberlist->GetEntries(); ++i) {
74  auto dm = (TDataMember*)memberlist->At(i);
75 
76  auto isValidComplex = [dm]() {
77  return isString(*dm) || dm->IsEnum();
78  };
79 
80  // filter out static members for now
81  if (dm->Property() & kIsStatic) {
82  continue;
83  }
84 
85  if (dm->IsaPointer()) {
86  LOG(WARNING) << "Pointer types not supported in ConfigurableParams";
87  continue;
88  }
89  if (!dm->IsBasic() && !isValidComplex()) {
90  LOG(WARNING) << "Generic complex types not supported in ConfigurableParams";
91  continue;
92  }
93 
94  const auto dim = dm->GetArrayDim();
95  // we support very simple vectored data in 1D for now
96  if (dim > 1) {
97  LOG(WARNING) << "We support at most 1 dimensional arrays in ConfigurableParams";
98  continue;
99  }
100 
101  const auto size = (dim == 1) ? dm->GetMaxIndex(dim - 1) : 1; // size of array (1 if scalar)
102  for (int index = 0; index < size; ++index) {
103  callback(dm, index, size);
104  }
105  }
106 }

123 const char* asString(TDataMember const& dm, char* pointer)
124 {
125  // first check if this is a basic data type, in which case
126  // we let ROOT do the work
127  if (auto dt = dm.GetDataType()) {
128  auto val = dt->AsString(pointer);
129 
130  // For enums we grab the string value of the member
131  // and use that instead if its int value
132  if (dm.IsEnum()) {
133  const auto enumtype = TEnum::GetEnum(dm.GetTypeName());
134  assert(enumtype != nullptr);
135  const auto constantlist = enumtype->GetConstants();
136  assert(constantlist != nullptr);
137  if (enumtype) {
138  for (int i = 0; i < constantlist->GetEntries(); ++i) {
139  const auto e = (TEnumConstant*)(constantlist->At(i));
140  if (val == std::to_string((int)e->GetValue())) {
141  return e->GetName();
142  }
143  }
144  }
145  }
146 
147  return val;
148  }
149 
150  // if data member is a std::string just return
151  else if (isString(dm)) {
152  return ((std::string*)pointer)->c_str();
153  }
154  // potentially other cases to be added here
155 
156  LOG(ERROR) << "COULD NOT REPRESENT AS STRING";
157  return nullptr;
158 }

162 std::vector<ParamDataMember>* _ParamHelper::getDataMembersImpl(std::string const& mainkey, TClass* cl, void* obj,
163  std::map<std::string, ConfigurableParam::EParamProvenance> const* provmap)
164 {
165  std::vector<ParamDataMember>* members = new std::vector<ParamDataMember>;
166 
167  auto toDataMember = [&members, obj, mainkey, provmap](const TDataMember* dm, int index, int size) {
168  auto dt = dm->GetDataType();
169  auto TS = dt ? dt->Size() : 0;
170  char* pointer = ((char*)obj) + dm->GetOffset() + index * TS;
171  const std::string name = getName(dm, index, size);
172  const char* value = asString(*dm, pointer);
173 
174  std::string prov = "";
175  auto iter = provmap->find(mainkey + "." + name);
176  if (iter != provmap->end()) {
177  prov = ConfigurableParam::toString(iter->second);
178  }
179  ParamDataMember member{name, value, prov};
180  members->push_back(member);
181  };
182 
183  loopOverMembers(cl, obj, toDataMember);
184  return members;
185 }

189 // a function converting a string representing a type to the type_info
190 // because unfortunately typeid(double) != typeid("double")
191 // but we can take the TDataType (if it exists) as a hint in order to
192 // minimize string comparisons
193 std::type_info const& nameToTypeInfo(const char* tname, TDataType const* dt)
194 {
195  if (dt) {
196  switch (dt->GetType()) {
197  case kChar_t: {
198  return typeid(char);
199  }
200  case kUChar_t: {
201  return typeid(unsigned char);
202  }
203  case kShort_t: {
204  return typeid(short);
205  }
206  case kUShort_t: {
207  return typeid(unsigned short);
208  }
209  case kInt_t: {
210  return typeid(int);
211  }
212  case kUInt_t: {
213  return typeid(unsigned int);
214  }
215  case kLong_t: {
216  return typeid(long);
217  }
218  case kULong_t: {
219  return typeid(unsigned long);
220  }
221  case kFloat_t: {
222  return typeid(float);
223  }
224  case kDouble_t: {
225  return typeid(double);
226  }
227  case kDouble32_t: {
228  return typeid(double);
229  }
230  case kBool_t: {
231  return typeid(bool);
232  }
233  case kLong64_t: {
234  return typeid(long long);
235  }
236  case kULong64_t: {
237  return typeid(unsigned long long);
238  }
239  default: {
240  break;
241  }
242  }
243  }
244  // if we get here none of the above worked
245  if (strcmp(tname, "string") == 0 || strcmp(tname, "std::string")) {
246  return typeid(std::string);
247  }
248  LOG(ERROR) << "ENCOUNTERED AN UNSUPPORTED TYPE " << tname << "IN A CONFIGURABLE PARAMETER";
249  return typeid("ERROR");
250 }

254 void _ParamHelper::fillKeyValuesImpl(std::string const& mainkey, TClass* cl, void* obj, boost::property_tree::ptree* tree,
255  std::map<std::string, std::pair<std::type_info const&, void*>>* keytostoragemap,
256  EnumRegistry* enumRegistry)
257 {
258  boost::property_tree::ptree localtree;
259  auto fillMap = [obj, &mainkey, &localtree, &keytostoragemap, &enumRegistry](const TDataMember* dm, int index, int size) {
260  const auto name = getName(dm, index, size);
261  auto dt = dm->GetDataType();
262  auto TS = dt ? dt->Size() : 0;
263  char* pointer = ((char*)obj) + dm->GetOffset() + index * TS;
264  localtree.put(name, asString(*dm, pointer));
265 
266  auto key = mainkey + "." + name;
267 
268  // If it's an enum, we need to store separately all the legal
269  // values so that we can map to them from the command line
270  if (dm->IsEnum()) {
271  enumRegistry->add(key, dm);
272  }
273 
274  using mapped_t = std::pair<std::type_info const&, void*>;
275  auto& ti = nameToTypeInfo(dm->GetTrueTypeName(), dt);
276  keytostoragemap->insert(std::pair<std::string, mapped_t>(key, mapped_t(ti, pointer)));
277  };
278  loopOverMembers(cl, obj, fillMap);
279  tree->add_child(mainkey, localtree);
280 }

315 void _ParamHelper::assignmentImpl(std::string const& mainkey, TClass* cl, void* to, void* from,
316  std::map<std::string, ConfigurableParam::EParamProvenance>* provmap)
317 {
318  auto assignifchanged = [to, from, &mainkey, provmap](const TDataMember* dm, int index, int size) {
319  const auto name = getName(dm, index, size);
320  auto dt = dm->GetDataType();
321  auto TS = dt ? dt->Size() : 0;
322  char* pointerto = ((char*)to) + dm->GetOffset() + index * TS;
323  char* pointerfrom = ((char*)from) + dm->GetOffset() + index * TS;
324 
325  // lambda to update the provenance
326  auto updateProv = [&mainkey, name, provmap]() {
327  auto key = mainkey + "." + name;
328  auto iter = provmap->find(key);
329  if (iter != provmap->end()) {
330  iter->second = ConfigurableParam::EParamProvenance::kCCDB; // TODO: change to "current STATE"??
331  } else {
332  LOG(WARN) << "KEY " << key << " NOT FOUND WHILE UPDATING PARAMETER PROVENANCE";
333  }
334  };
335 
336  // TODO: this could dispatch to the same method used in ConfigurableParam::setValue
337  // but will be slower
338 
339  // test if a complicated case
340  if (isString(*dm)) {
341  std::string& target = *(std::string*)pointerto;
342  std::string const& origin = *(std::string*)pointerfrom;
343  if (target.compare(origin) != 0) {
344  updateProv();
345  target = origin;
346  }
347  return;
348  }
349 
350  //
351  if (!isMemblockDifferent(pointerto, pointerfrom, TS)) {
352  updateProv();
353  // actually copy
354  std::memcpy(pointerto, pointerfrom, dt->Size());
355  }
356  };
357  loopOverMembers(cl, to, assignifchanged);
358 }
