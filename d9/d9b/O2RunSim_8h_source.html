52  void Init() final
53  {
54  LOG(INFO) << "O2RUNSIM SPECIFIC INIT CALLED";
55 
56  fRootManager->InitSink();
57 
58  // original FairRunSim follows
59  FairGeoLoader* loader = new FairGeoLoader(fLoaderName->Data(), "Geo Loader");
60  FairGeoInterface* GeoInterFace = loader->getGeoInterface();
61  GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
62  GeoInterFace->setMediaFile(MatFname.Data());
63  GeoInterFace->readMedia();
64 
65  if (mDeviceMode) {
66  fApp = new O2MCApplication("Fair", "The Fair VMC App", ListOfModules, MatFname);
67  } else {
68  fApp = new O2MCApplicationBase("Fair", "The Fair VMC App", ListOfModules, MatFname);
69  }
70 
71  fApp->SetGenerator(fGen);
72 
73  // Add a Generated run ID to the FairRunTimeDb
74  FairRunIdGenerator genid;
75  // FairRuntimeDb *rtdb= GetRuntimeDb();
76  fRunId = genid.generateId();
77  fRtdb->addRun(fRunId);
78 
79  fFileHeader->SetRunId(fRunId);
81  FairBaseParSet* par = dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
82  if (par) {
83  par->SetDetList(GetListOfModules());
84  par->SetGen(GetPrimaryGenerator());
85  par->SetBeamMom(fBeamMom);
86  }
87 
89  FairGeoParSet* geopar = dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
90  if (geopar) {
91  geopar->SetGeometry(gGeoManager);
92  }
93 
94  // Set global Parameter Info
95  if (fPythiaDecayer) {
96  fApp->SetPythiaDecayer(fPythiaDecayer);
97  if (fPythiaDecayerConfig) {
98  fApp->SetPythiaDecayerConfig(fPythiaDecayerConfig);
99  }
100  }
101  if (fUserDecay) {
102  fApp->SetUserDecay(fUserDecay);
103  if (fUserDecayConfig) {
104  fApp->SetUserDecayConfig(fUserDecayConfig);
105  }
106  }
107 
108  if (fField) {
109  fField->Init();
110  }
111  fApp->SetField(fField);
112  SetFieldContainer();
113 
114  TList* containerList = fRtdb->getListOfContainers();
115  TIter next(containerList);
116  FairParSet* cont;
117  TObjArray* ContList = new TObjArray();
118  while ((cont = dynamic_cast<FairParSet*>(next()))) {
119  ContList->Add(new TObjString(cont->GetName()));
120  }
121  if (par) {
122  par->SetContListStr(ContList);
123  par->SetRndSeed(gRandom->GetSeed());
124  par->setChanged();
125  par->setInputVersion(fRunId, 1);
126  }
127  if (geopar) {
128  geopar->setChanged();
129  geopar->setInputVersion(fRunId, 1);
130  }
131 
132  fSimSetup();
133  fApp->InitMC("foo", "bar");
134  fRootManager->WriteFileHeader(fFileHeader);
135  }
136 
137  void Run(int n = 0, int b = 0) final
138  {
139  FairRunSim::Run(n, b);
140  }
141 
142  private:
143  bool mDeviceMode = false;
144 
145  ClassDefOverride(O2RunSim, 0);
146 };
