78 Int_t SimpleEventDisplay::updateROC(const Int_t roc,
79  const Int_t row,
80  const Int_t pad,
81  const Int_t timeBin,
82  const Float_t signal)
83 {
84  //
85  // Signal filling methode on the fly pedestal and time offset correction if necessary.
86  // no extra analysis necessary. Assumes knowledge of the signal shape!
87  // assumes that it is looped over consecutive time bins of one pad
88  //
89  //printf("update called: %d, %d, %d, %d, %.3f\n", roc, row, pad, timeBin, signal);
90  if (row < 0)
91  return 0;
92  if (pad < 0)
93  return 0;
94  if (timeBin < 0)
95  return 0;
96  if ((timeBin > mLastTimeBin) || (timeBin < mFirstTimeBin))
97  return 0;
98  if (mSectorLoop && roc % 36 != mSelectedSector % 36)
99  return 0;
100 
101  if (row < 0 || pad < 0) {
102  printf("Wrong Pad or Row number, skipping!");
103  return 0;
104  }
105 
106  float corrSignal = signal;
107 
108  // ===| get pedestal |========================================================
109  if (mPedestals) {
110  corrSignal -= mPedestals->getValue(ROC(roc), row, pad);
111  }
112 
113  const int iChannel = mTPCmapper.getPadNumberInROC(PadROCPos(roc, row, pad));
114  //Int_t iChannel = mTPCmapper.globalPadNumber(PadPos(row, pad)) - (roc >= mTPCmapper.getNumberOfIROCs()) * mTPCmapper.getPadsInIROC();
115 
116  //init first pad and roc in this event
117  if (mCurrentChannel == -1) {
118  mCurrentChannel = iChannel;
119  mCurrentROC = roc;
120  mCurrentRow = row;
121  mCurrentPad = pad;
122  }
123 
124  //process last pad if we change to a new one
125  if (iChannel != mCurrentChannel) {
126  //ProcessPad();
127  mLastSector = mCurrentROC;
128  mCurrentChannel = iChannel;
129  mCurrentROC = roc;
130  mCurrentRow = row;
131  mCurrentPad = pad;
132  mMaxPadSignal = 0;
133  }
134  // if (signal>0) printf("%02d:%03d:%03d:%05d: %.3f\n",mCurrentROC,mCurrentRow,mCurrentPad,mCurrentChannel,signal);
135  //fill signals for current pad
136  if (mCurrentROC % 36 == mSelectedSector % 36) {
137  const Int_t nbins = mLastTimeBin - mFirstTimeBin;
138  const Int_t offset = (nbins + 2) * (iChannel + 1) + timeBin + 1;
139 
140  if ((UInt_t)roc < mTPCmapper.getNumberOfIROCs()) {
141  mHSigIROC->GetArray()[offset] = corrSignal;
142  } else {
143  mHSigOROC->GetArray()[offset] = corrSignal;
144  }
145  }
146 
147  CalROC& calROC = mPadMax.getCalArray(mCurrentROC);
148  //auto val = calROC.getValue(mCurrentRow, mCurrentPad);
149  auto val = calROC.getValue(row, pad);
150 
151  if (corrSignal > val) {
152  //printf("sec: %2d, row: %2d, pad: %3d, val: %.2f, sig: (%.2f) %.2f\n", mCurrentROC, mCurrentRow, mCurrentPad, val, corrSignal, signal);
153  //calROC.setValue(mCurrentRow,mCurrentPad,signal);
154  calROC.setValue(row, pad, corrSignal);
155  mMaxPadSignal = corrSignal;
156  mMaxTimeBin = timeBin;
157  }
158  //printf("update done\n");
159  return 0;
160 }

163 TH1D* SimpleEventDisplay::makePadSignals(Int_t roc, Int_t row, Int_t pad)
164 {
165  // TODO: check
166  //if (roc<0||roc>=(Int_t)mROC->GetNSectors()) return nullptr;
167  //if (row<0||row>=(Int_t)mROC->GetNRows(roc)) return nullptr;
168  //if (pad<0||pad>=(Int_t)mROC->GetNPads(roc,row)) return nullptr;
169  // TODO: possible bug for OROC
170  //const Int_t channel =
171  //mTPCmapper.globalPadNumber(PadPos(row, pad)) - (roc >= mTPCmapper.getNumberOfIROCs()) * mTPCmapper.getPadsInIROC();
172 
173  const int padOffset = (roc > 35) * Mapper::getPadsInIROC();
174  const int channel = mTPCmapper.getPadNumberInROC(PadROCPos(roc, row, pad));
175 
176  const FECInfo& fecInfo = mTPCmapper.getFECInfo(PadROCPos(roc, row, pad));
177  const int cruNumber = mTPCmapper.getCRU(ROC(roc).getSector(), channel + padOffset);
178  const CRU cru(cruNumber);
179  const PartitionInfo& partInfo = mTPCmapper.getMapPartitionInfo()[cru.partition()];
180  const int nFECs = partInfo.getNumberOfFECs();
181  const int fecOffset = (nFECs + 1) / 2;
182  const int fecInPartition = fecInfo.getIndex() - partInfo.getSectorFECOffset();
183  const int dataWrapperID = fecInPartition >= fecOffset;
184  const int globalLinkID = (fecInPartition % fecOffset) + dataWrapperID * 12;
185 
186  mSelectedSector = roc;
187  // mLastSelSector=roc;
188  //attention change for if event has changed
189  if (mSelectedSector % 36 != mLastSelSector % 36) {
190  mSectorLoop = kTRUE;
191  processEvent(getPresentEventNumber());
192  mLastSelSector = mSelectedSector;
193  mSectorLoop = kFALSE;
194  }
195  TH1D* h = nullptr;
196  const Int_t nbins = mLastTimeBin - mFirstTimeBin;
197  if (nbins <= 0)
198  return nullptr;
199  const Int_t offset = (nbins + 2) * (channel + 1);
200  Double_t* arrP = nullptr;
201 
202  TString title("#splitline{#lower[.1]{#scale[.5]{");
203 
204  if (roc < (Int_t)mTPCmapper.getNumberOfIROCs()) {
205  h = (TH1D*)gROOT->FindObject("PadSignals_IROC");
206  if (!h) {
207  h = new TH1D("PadSignals_IROC", "PadSignals IROC;time bins (200ns);amplitude (ADC counts)", nbins, mFirstTimeBin, mLastTimeBin);
208  }
209  h->SetFillColor(kBlue - 10);
210  arrP = mHSigIROC->GetArray() + offset;
211  // title+="IROC ";
212  } else {
213  h = (TH1D*)gROOT->FindObject("PadSignals_OROC");
214  if (!h) {
215  h = new TH1D("PadSignals_OROC", "PadSignals OROC;time bins (200ns);amplitude (ADC counts)", nbins, mFirstTimeBin, mLastTimeBin);
216  }
217  h->SetFillColor(kBlue - 10);
218  arrP = mHSigOROC->GetArray() + offset;
219  title += "OROC ";
220  }
221  title += (roc / 18 % 2 == 0) ? "A" : "C";
222  title += Form("%02d (%02d) row: %02d, pad: %03d, globalpad: %05d (in roc)}}}{#scale[.5]{FEC: %02d (%02d), Chip: %02d, Chn: %02d, CRU: %d, Link: %02d (%s%d)}}",
223  roc % 18, roc, row, pad, channel, fecInfo.getIndex(), fecInPartition, fecInfo.getSampaChip(), fecInfo.getSampaChannel(), cruNumber % CRU::CRUperSector, globalLinkID, dataWrapperID ? "B" : "A", globalLinkID % 12);
224  //title+=Form("row: %02d, pad: %03d, cpad: %03d, globalpad: %05d}}}{#scale[.5]{br: %d, FEC: %02d, Chip: %02d, Chn: %02d = HW: %d}}",
225  //row,pad,pad-mROC->GetNPads(roc,row)/2,channel,
226  //mTPCmapper.GetBranch(roc,row,pad),
227  //mTPCmapper.GetFEChw(roc,row,pad),
228  //mTPCmapper.GetChip(roc,row,pad),
229  //mTPCmapper.GetChannel(roc,row,pad),
230  //mTPCmapper.GetHWAddress(roc,row,pad)
231  //);
232  h->SetTitle(title.Data());
233  Int_t entries = 0;
234  for (Int_t i = 0; i < nbins; i++) {
235  entries += (Int_t)arrP[i + 1];
236  }
237  h->Set(nbins + 2, arrP);
238  h->SetEntries(entries);
239  return h;
240 }
