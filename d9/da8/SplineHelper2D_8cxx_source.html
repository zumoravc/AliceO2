61 template <typename DataT>
62 void SplineHelper2D<DataT>::approximateFunctionBatch(
63  DataT* Fparameters, DataT x1Min, DataT x1Max, DataT x2Min, DataT x2Max,
64  std::function<void(const std::vector<DataT>& x1, const std::vector<DataT>& x2, std::vector<DataT> f[/*mFdimensions*/])> F,

77  x1.reserve(batchsize);

87  dataPointFTmp[iDim].reserve(batchsize);
88  }
89 
90  unsigned int counter = 0;
91  for (int iv = 0; iv < getNumberOfDataPointsU2(); iv++) {
92  DataT x2Tmp = x2Min + mHelperU2.getDataPoint(iv).u * scaleX2;
93  for (int iu = 0; iu < getNumberOfDataPointsU1(); iu++) {
94  DataT x1Tmp = x1Min + mHelperU1.getDataPoint(iu).u * scaleX1;
95  x1.emplace_back(x1Tmp);
96  x2.emplace_back(x2Tmp);
97  index.emplace_back((iv * getNumberOfDataPointsU1() + iu) * mFdimensions);
98  ++counter;
99 
100  if (counter == batchsize || (iu == (getNumberOfDataPointsU1() - 1) && (iv == (getNumberOfDataPointsU2() - 1)))) {
101  counter = 0;
102  F(x1, x2, dataPointFTmp);
103  unsigned int entries = index.size();
104 
105  for (unsigned int i = 0; i < entries; ++i) {
106  const unsigned int indexTmp = index[i];
107  for (unsigned int iDim = 0; iDim < mFdimensions; ++iDim) {
108  dataPointF[indexTmp + iDim] = dataPointFTmp[iDim][i];
109  }
110  }
111 
112  x1.clear();
113  x2.clear();
114  index.clear();
115  for (unsigned int iDim = 0; iDim < mFdimensions; ++iDim) {
116  dataPointFTmp[iDim].clear();
117  }
118  }
119  }
120  }
121  approximateFunction(Fparameters, dataPointF.data());
122 }

124 template <typename DataT>
125 void SplineHelper2D<DataT>::approximateFunction(
126  DataT* Fparameters, const DataT DataPointF[/*getNumberOfDataPoints() x nFdim*/]) const
127 {
129 
130  const int Ndim = mFdimensions;
131  const int Ndim2 = 2 * Ndim;
132  const int Ndim3 = 3 * Ndim;
133  const int Ndim4 = 4 * Ndim;
134 
135  int nDataPointsU = getNumberOfDataPointsU1();
136  int nDataPointsV = getNumberOfDataPointsU2();
137 
138  int nKnotsU = mHelperU1.getSpline().getNumberOfKnots();
139  int nKnotsV = mHelperU2.getSpline().getNumberOfKnots();
140 
141  std::unique_ptr<DataT[]> rotDataPointF(new DataT[nDataPointsU * nDataPointsV * Ndim]); // U DataPoints x V DataPoints : rotated DataPointF for one output dimension
142  std::unique_ptr<DataT[]> Dv(new DataT[nKnotsV * nDataPointsU * Ndim]); // V knots x U DataPoints
143 
144  std::unique_ptr<DataT[]> parU(new DataT[mHelperU1.getSpline().getNumberOfParameters(Ndim)]);
145  std::unique_ptr<DataT[]> parV(new DataT[mHelperU2.getSpline().getNumberOfParameters(Ndim)]);
146 
147  // rotated data points (u,v)->(v,u)
148 
149  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
150  for (int ipv = 0; ipv < nDataPointsV; ipv++) {
151  for (int dim = 0; dim < Ndim; dim++) {
152  rotDataPointF[Ndim * (ipu * nDataPointsV + ipv) + dim] = DataPointF[Ndim * (ipv * nDataPointsU + ipu) + dim];
153  }
154  }
155  }
156 
157  // get S and S'u at all the knots by interpolating along the U axis
158 
159  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
160  int ipv = mHelperU2.getKnotDataPoint(iKnotV);
161  const DataT* DataPointFrow = &(DataPointF[Ndim * ipv * nDataPointsU]);
162  mHelperU1.approximateFunctionGradually(parU.get(), DataPointFrow);
163 
164  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
165  DataT* knotPar = &Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU)];
166  for (int dim = 0; dim < Ndim; ++dim) {
167  knotPar[dim] = parU[Ndim * (2 * iKnotU) + dim]; // store S for all the knots
168  knotPar[Ndim2 + dim] = parU[Ndim * (2 * iKnotU) + Ndim + dim]; // store S'u for all the knots //SG!!!
169  }
170  }
171 
172  // recalculate F values for all ipu DataPoints at V = ipv
173  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
174  DataT splineF[Ndim];
175  DataT u = mHelperU1.getDataPoint(ipu).u;
176  mHelperU1.getSpline().interpolateU(Ndim, parU.get(), u, splineF);
177  for (int dim = 0; dim < Ndim; dim++) {
178  rotDataPointF[(ipu * nDataPointsV + ipv) * Ndim + dim] = splineF[dim];
179  }
180  }
181  }
182 
183  // calculate S'v at all data points with V == V of a knot
184 
185  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
186  const DataT* DataPointFcol = &(rotDataPointF[ipu * nDataPointsV * Ndim]);
187  mHelperU2.approximateFunctionGradually(parV.get(), DataPointFcol);
188  for (int iKnotV = 0; iKnotV < nKnotsV; iKnotV++) {
189  for (int dim = 0; dim < Ndim; dim++) {
190  DataT dv = parV[(iKnotV * 2 + 1) * Ndim + dim];
191  Dv[(iKnotV * nDataPointsU + ipu) * Ndim + dim] = dv;
192  }
193  }
194  }
195 
196  // fit S'v and S''_vu at all the knots
197 
198  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
199  const DataT* Dvrow = &(Dv[iKnotV * nDataPointsU * Ndim]);
200  mHelperU1.approximateFunction(parU.get(), Dvrow);
201  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
202  for (int dim = 0; dim < Ndim; ++dim) {
203  Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim + dim] = parU[Ndim * 2 * iKnotU + dim]; // store S'v for all the knots
204  Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim3 + dim] = parU[Ndim * 2 * iKnotU + Ndim + dim]; // store S''vu for all the knots
205  }
206  }
207  }
208 }
