72  if (nel == 0)

98 double Response::chargefrac1d(float min, float max, double k2, double sqrtk3, double k4)
99 {
100  // The Mathieson function integral (1D)
101  double u1 = sqrtk3 * TMath::TanH(k2 * min);
102  double u2 = sqrtk3 * TMath::TanH(k2 * max);
103  return 2. * k4 * (TMath::ATan(u2) - TMath::ATan(u1));
104 }

106 unsigned long Response::response(unsigned long adc)
107 {
108  //DecalibrateTrackerDigit functionality from
109  //AliMuonDigitizerV3 in aliroot
110  int fgNSigma = 5.0; //aliroot no
111  //no channel-by-channel noise map as in aliroot

113  float adc_out = adc;
114  float pedestalMean = 0;
115  float adcNoise = 0.0;
116  //TODO: parameter choices for match with aliroot
117 
118  adc = TMath::Nint(adc_out + pedestalMean + adcNoise + 0.5);
119 
120  if (adc_out < TMath::Nint(pedestalMean + fgNSigma * pedestalSigma + 0.5))
121  adc = 0;
122  if (adc > mMaxADC)
123  adc = mMaxADC;
124  return adc;
125 }
