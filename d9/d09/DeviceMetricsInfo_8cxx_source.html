27 // Parses a metric in the form
28 //
29 // [METRIC] <name>,<type> <value> <timestamp> [<tag>,<tag>]
30 bool DeviceMetricsHelper::parseMetric(std::string_view const s, ParsedMetricMatch& match)
31 {
34  constexpr size_t PREFIXSIZE = 9;
35  if (s.size() < PREFIXSIZE) {
36  return false;
37  }
38  if (memcmp(s.data(), "[METRIC] ", 9) != 0) {
39  return false;
40  }
41  char const* comma; // first comma
42  char const* spaces[256]; // list of spaces
43  char const** space = spaces; // first element to fill
44 
45  comma = (char const*)memchr((void*)s.data(), ',', s.size());
46  if (comma == nullptr) {
47  return false;
48  }
49  // Find all spaces
50  char const* nextSpace = s.data();
51  while (space - spaces < 256) {
52  *space = strchr(nextSpace, ' ');
53  if (*space == nullptr) {
54  break;
55  }
56  nextSpace = *space + 1;
57  space++;
58  }
59  // First space should always come before comma
60  if (spaces[0] > comma) {
61  return false;
62  }
63  match.beginKey = spaces[0] + 1;
64  match.endKey = comma;
65  // type is alway 1 char after the comma, followed by space
66  if ((spaces[1] - comma) != 2) {
67  return false;
68  }
69  char* ep = nullptr;
70  match.type = static_cast<MetricType>(strtol(comma + 1, &ep, 10));
71  if (ep != spaces[1]) {
72  return false;
73  }
74  // We need at least 4 spaces
75  if (space - spaces < 4) {
76  return false;
77  }
78  // Value is between the second and the last but one space
79  switch (match.type) {
80  case MetricType::Int:
81  match.intValue = strtol(spaces[1] + 1, &ep, 10);
82  if (ep != *(space - 2)) {
83  return false;
84  }
85  break;
86  case MetricType::Float:
87  match.floatValue = strtof(spaces[1] + 1, &ep);
88  if (ep != *(space - 2)) {
89  return false;
90  }
91  break;
92  case MetricType::String:
93  match.beginStringValue = spaces[1] + 1;
94  match.endStringValue = *(space - 2);
95  break;
96  default:
97  return false;
98  }
99  // Timestamp is between the last but one and the last space
100  match.timestamp = strtol(*(space - 2) + 1, &ep, 10);
101  if (ep != *(space - 1)) {
102  return false;
103  }
104  return true;
105 }

107 bool DeviceMetricsHelper::processMetric(ParsedMetricMatch& match,
108  DeviceMetricsInfo& info,
109  DeviceMetricsHelper::NewMetricCallback newMetricsCallback)
110 {
111  // get the type
112  size_t metricIndex = -1;
113 
114  StringMetric stringValue;
115  switch (match.type) {
116  case MetricType::Float:
117  case MetricType::Int:
118  break;
119  case MetricType::String: {
120  auto lastChar = std::min(match.endStringValue - match.beginStringValue, StringMetric::MAX_SIZE - 1);
121  memcpy(stringValue.data, match.beginStringValue, lastChar);
122  stringValue.data[lastChar] = '\0';
123  } break;
124  default:
125  return false;
126  break;
127  };
128 
129  // Find the metric based on the label. Create it if not found.
130  auto cmpFn = [namePtr = match.beginKey,
131  nameSize = match.endKey - match.beginKey](MetricLabelIndex const& a, MetricLabelIndex const& b)
132  -> bool {
133  return strncmp(a.label, namePtr, nameSize) < 0;
134  };
135  auto mi = std::lower_bound(info.metricLabelsIdx.begin(),
136  info.metricLabelsIdx.end(),
137  MetricLabelIndex{},
138  cmpFn);
139 
140  // We could not find the metric, lets insert a new one.
141  if (mi == info.metricLabelsIdx.end() || (strncmp(mi->label, match.beginKey, std::min(match.endKey - match.beginKey, (long)MetricLabelIndex::MAX_METRIC_LABEL_SIZE - 1)) != 0)) {
142  // Create a new metric
143  MetricInfo metricInfo;
144  metricInfo.pos = 0;
145  metricInfo.type = match.type;
146  // Add a new empty buffer for it of the correct kind
147  switch (match.type) {
148  case MetricType::Int:
149  metricInfo.storeIdx = info.intMetrics.size();
150  info.intMetrics.emplace_back(std::array<int, 1024>{});
151  break;
152  case MetricType::String:
153  metricInfo.storeIdx = info.stringMetrics.size();
154  info.stringMetrics.emplace_back(std::array<StringMetric, 32>{});
155  break;
156  case MetricType::Float:
157  metricInfo.storeIdx = info.floatMetrics.size();
158  info.floatMetrics.emplace_back(std::array<float, 1024>{});
159  break;
160  default:
161  return false;
162  };
163  // Add the timestamp buffer for it
164  info.timestamps.emplace_back(std::array<size_t, 1024>{});
165  info.max.push_back(std::numeric_limits<float>::lowest());
166  info.min.push_back(std::numeric_limits<float>::max());
167  info.maxDomain.push_back(std::numeric_limits<size_t>::lowest());
168  info.minDomain.push_back(std::numeric_limits<size_t>::max());
169 
170  // Add the index by name in the correct position
171  // this will require moving the tail of the index,
172  // but inserting should happen only once for each metric,
173  // so who cares.
174  MetricLabelIndex metricLabelIdx;
175  auto lastChar = std::min(match.endKey - match.beginKey, (ptrdiff_t)MetricLabelIndex::MAX_METRIC_LABEL_SIZE - 1);
176  memcpy(metricLabelIdx.label, match.beginKey, lastChar);
177  metricLabelIdx.label[lastChar] = '\0';
178  metricLabelIdx.index = info.metrics.size();
179  info.metricLabelsIdx.insert(mi, metricLabelIdx);
180  // Add the the actual Metric info to the store
181  metricIndex = info.metrics.size();
182  assert(metricInfo.storeIdx != -1);
183  assert(metricLabelIdx.label[0] != '\0');
184  if (newMetricsCallback != nullptr) {
185  newMetricsCallback(metricLabelIdx.label, metricInfo, match.intValue, metricIndex);
186  }
187  info.metrics.push_back(metricInfo);
188  } else {
189  metricIndex = mi->index;
190  }
191  assert(metricIndex != -1);
192  // We are now guaranteed our metric is present at metricIndex.
193  MetricInfo& metricInfo = info.metrics[metricIndex];
194 
195  // auto mod = info.timestamps[metricIndex].size();
196  info.minDomain[metricIndex] = std::min(info.minDomain[metricIndex], (size_t)match.timestamp);
197  info.maxDomain[metricIndex] = std::max(info.maxDomain[metricIndex], (size_t)match.timestamp);
198 
199  switch (metricInfo.type) {
200  case MetricType::Int: {
201  info.intMetrics[metricInfo.storeIdx][metricInfo.pos] = match.intValue;
202  info.max[metricIndex] = std::max(info.max[metricIndex], (float)match.intValue);
203  info.min[metricIndex] = std::min(info.min[metricIndex], (float)match.intValue);
204  // Save the timestamp for the current metric we do it here
205  // so that we do not update timestamps for broken metrics
206  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
207  // Update the position where to write the next metric
208  metricInfo.pos = (metricInfo.pos + 1) % info.intMetrics[metricInfo.storeIdx].size();
209  } break;
210  case MetricType::String: {
211  info.stringMetrics[metricInfo.storeIdx][metricInfo.pos] = stringValue;
212  // Save the timestamp for the current metric we do it here
213  // so that we do not update timestamps for broken metrics
214  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
215  metricInfo.pos = (metricInfo.pos + 1) % info.stringMetrics[metricInfo.storeIdx].size();
216  } break;
217  case MetricType::Float: {
218  info.floatMetrics[metricInfo.storeIdx][metricInfo.pos] = match.floatValue;
219  info.max[metricIndex] = std::max(info.max[metricIndex], match.floatValue);
220  info.min[metricIndex] = std::min(info.min[metricIndex], match.floatValue);
221  // Save the timestamp for the current metric we do it here
222  // so that we do not update timestamps for broken metrics
223  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
224  metricInfo.pos = (metricInfo.pos + 1) % info.floatMetrics[metricInfo.storeIdx].size();
225  } break;
226  default:
227  return false;
228  break;
229  };
230 
231  return true;
232 }

235 size_t
236  DeviceMetricsHelper::metricIdxByName(const std::string& name, const DeviceMetricsInfo& info)
237 {
238  size_t i = 0;
239  while (i < info.metricLabelsIdx.size()) {
240  auto& metricName = info.metricLabelsIdx[i];
241  if (metricName.label == name) {
242  return metricName.index;
243  }
244  ++i;
245  }
246  return i;
247 }
