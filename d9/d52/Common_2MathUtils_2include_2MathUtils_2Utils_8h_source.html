165 GPUhdi() float FastATan2(float y, float x)
166 {
167  // Fast atan2(y,x) for any angle [-Pi,Pi]
168  // Average inaccuracy: 0.00048
169  // Max inaccuracy: 0.00084
170  // Speed: 6.2 times faster than atan2f()
171 
172  constexpr float kPi = 3.1415926535897f;
173 
174  auto atan = [](float a) -> float {
175  // returns the arctan for the angular range [-Pi/4, Pi/4]
176  // the polynomial coefficients are taken from:
177  // https://stackoverflow.com/questions/42537957/fast-accurate-atan-arctan-approximation-algorithm
178  constexpr float kA = 0.0776509570923569f;
179  constexpr float kB = -0.287434475393028f;
180  constexpr float kC = (kPi / 4 - kA - kB);
181  float a2 = a * a;
182  return ((kA * a2 + kB) * a2 + kC) * a;
183  };
184 
185  auto atan2P = [atan](float yy, float xx) -> float {
186  // fast atan2(yy,xx) for the angular range [0,+Pi]
187  constexpr float kPi025 = 1 * kPi / 4;
188  constexpr float kPi075 = 3 * kPi / 4;
189  float x1 = xx + yy; // point p1 (x1,y1) = (xx,yy) - Pi/4
190  float y1 = yy - xx;
191  float phi0, tan;
192  if (xx < 0) { // p1 is in the range [Pi/4, 3*Pi/4]
193  phi0 = kPi075;
194  tan = -x1 / y1;
195  } else { // p1 is in the range [-Pi/4, Pi/4]
196  phi0 = kPi025;
197  tan = y1 / x1;
198  }
199  return phi0 + atan(tan);
200  };
201 
202  // fast atan2(y,x) for any angle [-Pi,Pi]
203  return o2::gpu::GPUCommonMath::Copysign(atan2P(o2::gpu::CAMath::Abs(y), x), y);
204 }
