22 // structure describing an entity of work

24 struct SubEventInfo {

26  uint32_t eventID = 0; // which event ID

33  int32_t npersistenttracks = -1; // the number of persistent tracks for this SubEvent (might be set to cache it)
34  int32_t nprimarytracks = -1; // the number of primary tracks for this SubEvent
35  // might add more fields (such as which process treated this chunk etc)
36

39  ClassDefNV(SubEventInfo, 1);

42 inline bool operator<(SubEventInfo const& a, SubEventInfo const& b)

44  return a.eventID <= b.eventID && (a.part < b.part);
