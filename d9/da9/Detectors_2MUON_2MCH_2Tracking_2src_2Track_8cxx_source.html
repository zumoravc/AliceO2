93 {
96 
97  int chMin = 2 * stMin;
98  int chMax = 2 * stMax + 1;
99  int nClustersInCommon(0);
100 
101  for (const auto& param1 : *this) {
102 
103  int ch1 = param1.getClusterPtr()->getChamberId();
104  if (ch1 < chMin || ch1 > chMax) {
105  continue;
106  }
107 
108  for (const auto& param2 : track) {
109 
110  int ch2 = param2.getClusterPtr()->getChamberId();
111  if (ch2 < chMin || ch2 > chMax) {
112  continue;
113  }
114 
115  if (param1.getClusterPtr()->getUniqueId() == param2.getClusterPtr()->getUniqueId()) {
116  ++nClustersInCommon;
117  break;
118  }
119  }
120  }

122  return nClustersInCommon;

136 void Track::tagRemovableClusters(uint8_t requestedStationMask)
137 {
141 
142  int previousCh(-1), previousSt(-1), nChHitInSt45(0);
143  TrackParam* previousParam(nullptr);
144 
145  for (auto& param : *this) {
146 
147  int currentCh = param.getClusterPtr()->getChamberId();
148  int currentSt = currentCh / 2;
149 
150  // set the cluster as removable if the station is not requested or if it is not alone in the station
151  if (((1 << currentSt) & requestedStationMask) == 0) {
152  param.setRemovable(true);
153  } else if (currentSt == previousSt) {
154  previousParam->setRemovable(true);
155  param.setRemovable(true);
156  } else {
157  param.setRemovable(false);
158  previousSt = currentSt;
159  previousParam = &param;
160  }
161 
162  // count the number of chambers in station 4 & 5 that contain cluster(s)
163  if (currentCh > 5 && currentCh != previousCh) {
164  ++nChHitInSt45;
165  previousCh = currentCh;
166  }
167  }
168 
169  // if there are less than 3 chambers containing cluster(s) in station 4 & 5
170  if (nChHitInSt45 < 3) {
171 
172  previousCh = -1;
173  previousParam = nullptr;
174 
175  for (auto itParam = this->rbegin(); itParam != this->rend(); ++itParam) {
176 
177  int currentCh = itParam->getClusterPtr()->getChamberId();
178 
179  if (currentCh < 6) {
180  break;
181  }
182 
183  // set the cluster as not removable unless it is not alone in the chamber
184  if (currentCh == previousCh) {
185  previousParam->setRemovable(true);
186  itParam->setRemovable(true);
187  } else {
188  itParam->setRemovable(false);
189  previousCh = currentCh;
190  previousParam = &*itParam;
191  }
192  }
193  }
194 }
