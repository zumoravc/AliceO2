132 static int GPUFailedMsgAI(int error, const char* file, int line)
133 {
134  // Check for OPENCL Error and in the case of an error display the corresponding error string
135  if (error == CL_SUCCESS) {
136  return (0);
137  }
138  GPUError("OCL Error: %d / %s (%s:%d)", error, opencl_error_string(error), file, line);
139  return 1;
140 }

152 static inline int OCLsetKernelParameters_helper(cl_kernel& kernel, int i, const T& firstParameter, const Args&... restOfParameters)
153 {
154  GPUFailedMsg(clSetKernelArg(kernel, i, sizeof(T), &firstParameter));

185 template <typename K, typename... Args>
186 int GPUReconstructionOCL::runKernelBackendCommon(krnlSetup& _xyz, K& k, const Args&... args)
187 {
188  auto& x = _xyz.x;
189  auto& y = _xyz.y;
190  auto& z = _xyz.z;
191  if (y.num <= 1) {
192  if (OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, args...)) {
193  return 1;
194  }
195  } else {
196  if (OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, y.num, args...)) {
197  return 1;
198  }
199  }
200 
201  cl_event ev;
202  cl_event* evr;
203  bool tmpEvent = false;
204  if (z.ev == nullptr && mDeviceProcessingSettings.deviceTimers) {
205  evr = &ev;
206  tmpEvent = true;
207  } else {
208  evr = (cl_event*)z.ev;
209  }
210  int retVal = clExecuteKernelA(mInternals->command_queue[x.stream], k, x.nThreads, x.nThreads * x.nBlocks, evr, (cl_event*)z.evList, z.nEvents);
211  if (mDeviceProcessingSettings.deviceTimers) {
212  cl_ulong time_start, time_end;
213  GPUFailedMsg(clWaitForEvents(1, evr));
214  GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_START, sizeof(time_start), &time_start, nullptr));
215  GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_END, sizeof(time_end), &time_end, nullptr));
216  _xyz.t = (time_end - time_start) * 1.e-9;
217  if (tmpEvent) {
218  GPUFailedMsg(clReleaseEvent(ev));
219  }
220  }
221  return retVal;
222 }

224 template <class T, int I>
225 int GPUReconstructionOCL::AddKernel(bool multi)
226 {
227  std::string name(GetKernelName<T, I>());
228  if (multi) {
229  name += "_multi";
230  }
231  std::string kname("krnl_" + name);
232 
233  cl_int ocl_error;
234  cl_kernel krnl = clCreateKernel(mInternals->program, kname.c_str(), &ocl_error);
235  if (GPUFailedMsgI(ocl_error)) {
236  GPUError("Error creating OPENCL Kernel: %s", name.c_str());
237  return 1;
238  }
239  mInternals->kernels.emplace_back(krnl, name);
240  return 0;
241 }

243 template <class T, int I>
244 inline unsigned int GPUReconstructionOCL::FindKernel(int num)
245 {
246  std::string name(GetKernelName<T, I>());
247  if (num > 1) {
248  name += "_multi";
249  }
250 
251  for (unsigned int k = 0; k < mInternals->kernels.size(); k++) {
252  if (mInternals->kernels[k].second == name) {
253  return (k);
254  }
255  }
256  GPUError("Could not find OpenCL kernel %s", name.c_str());
257  throw ::std::runtime_error("Requested unsupported OpenCL kernel");
258 }
