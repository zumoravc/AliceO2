45  void process(aod::Collision const& collision, aod::Tracks const& tracks)
46  {
47  LOGF(info, "Tracks for collision: %d (z: %f V0: %f)", tracks.size(), -1.0, collision.posZ());
48 
49  for (auto it1 = tracks.begin(); it1 != tracks.end(); ++it1) {
50  auto& track1 = *it1;
51  if (track1.pt() < 0.5)
52  continue;
53 
54  double eventValues[3];
55  eventValues[0] = track1.pt();
56  eventValues[1] = 0; //collision.v0mult();
57  eventValues[2] = collision.posZ();
58 
59  same->getEventHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
60  //mixed->getEventHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
61 
62  for (auto it2 = it1 + 1; it2 != tracks.end(); ++it2) {
63  auto& track2 = *it2;
64  //if (track1 == track2)
65  // continue;
66  if (track2.pt() < 0.5)
67  continue;
68 
69  double values[6];
70 
71  values[0] = track1.eta() - track2.eta();
72  values[1] = track1.pt();
73  values[2] = track2.pt();
74  values[3] = 0; //collision.v0mult();
75 
76  values[4] = track1.phi() - track2.phi();
77  if (values[4] > 1.5 * TMath::Pi())
78  values[4] -= TMath::TwoPi();
79  if (values[4] < -0.5 * TMath::Pi())
80  values[4] += TMath::TwoPi();
81 
82  values[5] = collision.posZ();
83 
84  same->getTrackHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
85  //mixed->getTrackHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
86  }
87  }
88  }
