100 Bool_t GeneratorHepMC::importParticles()
101 {
105  auto particles = mEvent->particles();
106  for (int i = 0; i < particles.size(); ++i) {
107 
109  auto particle = particles.at(i);
110  auto pdg = particle->pid();
111  auto st = particle->status();
112  auto momentum = particle->momentum();
113  auto vertex = particle->production_vertex()->position();
114  auto parents = particle->parents(); // less efficient than via vertex
115  auto children = particle->children(); // less efficient than via vertex
116 
118  auto px = momentum.x();
119  auto py = momentum.y();
120  auto pz = momentum.z();
121  auto et = momentum.t();
122 
124  auto vx = vertex.x();
125  auto vy = vertex.y();
126  auto vz = vertex.z();
127  auto vt = vertex.t();
128 
130  auto m1 = parents.empty() ? -1 : parents.front()->id() - 1;
131  auto m2 = parents.empty() ? -1 : parents.back()->id() - 1;
132 
134  auto d1 = children.empty() ? -1 : children.front()->id() - 1;
135  auto d2 = children.empty() ? -1 : children.back()->id() - 1;
136 
138  mParticles.push_back(TParticle(pdg, st, m1, m2, d1, d2, px, py, pz, et, vx, vy, vz, vt));
139 
140  }
143  return kTRUE;
144 }

148 Bool_t GeneratorHepMC::Init()
149 {
153  Generator::Init();
154 
156  mStream.open(mFileName);
157  if (!mStream.is_open()) {
158  LOG(FATAL) << "Cannot open input file: " << mFileName << std::endl;
159  return kFALSE;
160  }
161 
163  switch (mVersion) {
164  case 2:
165  mStream.close();
166  mReader = new HepMC::ReaderAsciiHepMC2(mFileName);
167  break;
168  case 3:
169  mReader = new HepMC::ReaderAscii(mStream);
170  break;
171  default:
172  LOG(FATAL) << "Unsupported HepMC version: " << mVersion << std::endl;
173  return kFALSE;
174  }
175 
177  return !mReader->failed();
178 }
