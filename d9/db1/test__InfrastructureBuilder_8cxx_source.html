55 BOOST_AUTO_TEST_CASE(InfrastructureBuilderLayers)
56 {
57  MergerInfrastructureBuilder builder;
58  builder.setInfrastructureName("name");
59  builder.setInputSpecs({{"one", "TST", "test", 1},
60  {"two", "TST", "test", 2},
61  {"thr", "TST", "test", 3},
62  {"fou", "TST", "test", 4},
63  {"fiv", "TST", "test", 5},
64  {"six", "TST", "test", 6},
65  {"sev", "TST", "test", 7}});
66  builder.setOutputSpec({{"main"}, "TST", "test", 0});
67  MergerConfig config;
68 
69  {
70  config.topologySize = {TopologySize::NumberOfLayers, 0};
71  builder.setConfig(config);
72  BOOST_CHECK_THROW(builder.generateInfrastructure(), std::runtime_error);
73  }
74 
75  {
76  config.topologySize = {TopologySize::NumberOfLayers, 1};
77  builder.setConfig(config);
78  auto mergersTopology = builder.generateInfrastructure();
79 
80  BOOST_REQUIRE_EQUAL(mergersTopology.size(), 1);
81  BOOST_CHECK_EQUAL(mergersTopology[0].inputs.size(), 8);
82  BOOST_REQUIRE_EQUAL(mergersTopology[0].outputs.size(), 1);
83 
84  auto concrete = DataSpecUtils::asConcreteDataMatcher(mergersTopology[0].outputs[0]);
85  BOOST_CHECK_EQUAL(concrete.origin.str, "TST");
86  BOOST_CHECK_EQUAL(concrete.description.str, "test");
87  BOOST_CHECK_EQUAL(concrete.subSpec, 0);
88  }
89 
90  {
91  config.topologySize = {TopologySize::NumberOfLayers, 2};
92  builder.setConfig(config);
93  auto mergersTopology = builder.generateInfrastructure();
94 
95  BOOST_REQUIRE_EQUAL(mergersTopology.size(), 4);
96 
97  // the first layer
98  BOOST_CHECK_EQUAL(mergersTopology[0].inputs.size(), 4);
99  BOOST_CHECK_EQUAL(mergersTopology[0].outputs.size(), 1);
100  BOOST_CHECK_EQUAL(mergersTopology[1].inputs.size(), 3);
101  BOOST_CHECK_EQUAL(mergersTopology[1].outputs.size(), 1);
102  BOOST_CHECK_EQUAL(mergersTopology[2].inputs.size(), 3);
103  BOOST_CHECK_EQUAL(mergersTopology[2].outputs.size(), 1);
104 
105  // the second layer
106  BOOST_CHECK_EQUAL(mergersTopology[3].inputs.size(), 4);
107  BOOST_CHECK_EQUAL(mergersTopology[3].outputs.size(), 1);
108  auto concrete = DataSpecUtils::asConcreteDataMatcher(mergersTopology[3].outputs[0]);
109  BOOST_REQUIRE_EQUAL(concrete.origin.str, "TST");
110  BOOST_CHECK_EQUAL(concrete.description.str, "test");
111  BOOST_CHECK_EQUAL(concrete.subSpec, 0);
112  }
