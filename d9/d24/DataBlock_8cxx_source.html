22 int forEachDataBlockRef(gsl::span<const std::byte> buffer,
23  std::function<void(DataBlockRef ref)> f)
24 {
25  int index{0};
26  int nheaders{0};
27  DataBlockHeader header;
28  while (index < buffer.size()) {
29  memcpy(&header, &buffer[index], sizeof(header));
30  nheaders++;
31  if (f) {
32  DataBlock block{header, buffer.subspan(index + sizeof(header), header.payloadSize)};
33  DataBlockRef ref{block, index};
34  f(ref);
35  }
36  index += header.payloadSize + sizeof(header);
37  }
38  return nheaders;
39 }
