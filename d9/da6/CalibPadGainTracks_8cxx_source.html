30 void CalibPadGainTracks::processTrack(o2::tpc::TrackTPC track, float momMin, float momMax)
31 {
32  // make momentum cut
33  const float mom = track.getP();
34  if (mom < momMin || mom > momMax) {
35  return;
36  }
37 
38  std::array<std::vector<float>, 2> vTempdEdx{}; // this vector is filled with the cluster charge and used for dedx calculation
39  vTempdEdx[0].reserve(NROWSIROC);
40  vTempdEdx[1].reserve(NROWSOROC);
41  std::vector<o2::tpc::ClusterNative> clNat{};
42  clNat.reserve(NROWS);
43  std::vector<std::tuple<unsigned char, unsigned char, unsigned char, float>> tClTrk{}; //temp tuple which will be filled
44  tClTrk.reserve(NROWS);
45  const int nClusters = track.getNClusterReferences();
46  float effectiveLengthLastCluster = 0; // set the effective length of the track over pad for the last cluster
47 
48  for (int iCl = 0; iCl < nClusters; iCl++) { //loop over cluster
49  const float effectiveLength = getTrackTopologyCorrection(track, iCl);
50  clNat.emplace_back(track.getCluster(*mTPCTrackClIdxVecInput, iCl, *mClusterIndex));
51  unsigned char sectorIndex = 0;
52  unsigned char rowIndex = 0;
53  unsigned int clusterIndexNumb = 0;
54 
55  // this function sets sectorIndex, rowIndex, clusterIndexNumb
56  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
57  const float charge = clNat[static_cast<unsigned int>(iCl)].qMax;
58  const unsigned char pad = static_cast<unsigned char>(clNat[static_cast<unsigned int>(iCl)].getPad() + 0.5f); // the left side of the pad ist defined at e.g. 3.5 and the right side at 4.5
59 
60  // propagateTo delivers sometimes wrong values! break if the effLength doesnt change
61  // TODO check this
62  if (effectiveLength - effectiveLengthLastCluster == 0) {
63  break;
64  }
65 
66  // fill IROC dedx
67  if (rowIndex < mapper.getNumberOfRowsROC(0)) {
68  vTempdEdx[0].emplace_back(charge / effectiveLength);
69  } else {
70  vTempdEdx[1].emplace_back(charge / effectiveLength);
71  }
72  tClTrk.emplace_back(std::make_tuple(sectorIndex, rowIndex, pad, charge / effectiveLength)); // fill with dummy dedx value
73  effectiveLengthLastCluster = effectiveLength;
74  }
75 
76  // use dedx from track as reference
77  if (mMode == DedxTrack) {
78  const float dedxIROC = getTruncMean(vTempdEdx[0]);
79  const float dedxOROC = getTruncMean(vTempdEdx[1]);
80 
81  // set the dEdx
82  for (auto& x : tClTrk) {
83  const unsigned char globRow = std::get<1>(x);
84  const unsigned char pad = std::get<2>(x);
85 
86  // get globalPadNumber (index)
87  const auto rowsIROC = mapper.getNumberOfRowsROC(0);
88  const auto roctype = (globRow < rowsIROC) ? RocType::IROC : RocType::OROC;
89  const float fillVal = (roctype == RocType::IROC) ? (std::get<3>(x) / dedxIROC) : (std::get<3>(x) / dedxOROC);
90  ROC roc(std::get<0>(x), roctype);
91  const o2::tpc::PadSubset sub = mPadHistosDet->getPadSubset();
92  const int num = mPadHistosDet->getCalArray(roc.getRoc()).getPadSubsetNumber();
93  const int rowinROC = roctype == RocType::IROC ? std::get<1>(x) : std::get<1>(x) - rowsIROC;
94  auto index = static_cast<unsigned int>(getIndex(sub, num, rowinROC, pad)); // index=globalpadnumber
95  //fill the normalizes charge in pad histogram
96  mPadHistosDet->getCalArray(roc.getRoc()).getData()[index].fill(fillVal);
97  }
98  } else {
99  }
100 }

102 void CalibPadGainTracks::writeTree() const
103 {
104  // loop over the tuple and fill the entries in a tree
105  TFile file("debug.root", "RECREATE");
106  TTree tCluster("debug", "debug");
107  int sector{};
108  int padRow{};
109  unsigned int pad{};
110  o2::tpc::FastHisto<float> fastHisto{};
111 
112  tCluster.Branch("sector", &sector);
113  tCluster.Branch("padRow", &padRow);
114  tCluster.Branch("pad", &pad);
115  tCluster.Branch("hist", &fastHisto);
116 
117  unsigned char iROC = 0; // roc: 0...71
118  for (auto& calArray : mPadHistosDet->getData()) {
119  o2::tpc::ROC roc(iROC);
120  GlobalPadNumber globPadNumber = roc.rocType() == RocType::OROC ? mapper.getPadsInIROC() : 0;
121 
122  for (auto& val : calArray.getData()) {
123  fastHisto = val;
124  const PadPos pos = mapper.padPos(globPadNumber);
125  padRow = pos.getRow();
126  pad = pos.getPad();
127  sector = roc.getSector();
128  tCluster.Fill();
129  ++globPadNumber;
130  }
131  ++iROC;
132  }
133  file.cd();
134  tCluster.Write();
135  file.Close();
136 }

138 float CalibPadGainTracks::getTruncMean(std::vector<float> vCharge, float low, float high) const
139 {
140  // returns the truncated mean for input vector
141  std::sort(vCharge.begin(), vCharge.end()); //sort the vector for performing truncated mean
142  float dEdx = 0.f; // count total dedx
143  int nClustersTrunc = 0; // count number of clusters
144  const int nClustersUsed = static_cast<int>(vCharge.size());
145 
146  for (int icl = static_cast<int>(low * nClustersUsed); icl < static_cast<int>(high * nClustersUsed); ++icl) {
147  dEdx += vCharge[static_cast<unsigned int>(icl)];
148  ++nClustersTrunc;
149  }
150 
151  if (nClustersTrunc > 0) {
152  dEdx /= nClustersTrunc;
153  }
154  return dEdx;
155 }

178 float CalibPadGainTracks::getTrackTopologyCorrection(o2::tpc::TrackTPC& track, int iCl)
179 {
180  unsigned char sectorIndex = 0;
181  unsigned char rowIndex = 0;
182  unsigned int clusterIndexNumb = 0;
183  // this function sets sectorIndex, rowIndex, clusterIndexNumb
184  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
185 
186  int nPadRows = 0; // used for getting the current region (e.g. set the padLength and padHeight)
187  float padLength = 0;
188 
189  // TODO optimize for loop
190  for (unsigned char iRegion = 0; iRegion < 10; iRegion++) {
191  const PadRegionInfo& region = mapper.getPadRegionInfo(iRegion);
192  padLength = region.getPadHeight();
193  nPadRows += static_cast<int>(region.getNumberOfPadRows());
194  if (static_cast<int>(rowIndex) < nPadRows) {
195  break;
196  }
197  }
198 
199  // to correct the cluster charge for the track topology, the track parameters have to be propagated to the x position if the cluster
200  const float xPosition = mapper.getPadCentre(PadPos(rowIndex, 0)).X();
201  const float bField = 5.00668f; // magnetic field in "kilo Gaus" // TODO Get b field from o2.
202  track.propagateTo(xPosition, bField); // propagate this track to the plane X=xk (cm) in the field "b" (kG)
203  const float sinPhi = track.getSnp();
204  const float tgl = track.getTgl();
205  const float snp2 = sinPhi * sinPhi;
206  const float effectiveLength = padLength * std::sqrt((1 + tgl * tgl) / (1 - snp2)); // calculate the trace length of the track over the pad
207  return effectiveLength;
208 }
