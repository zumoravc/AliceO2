26 template <>
27 GPUdii() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step0attached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& smem, processorType& processors)
28 {
29  const GPUTPCGMMerger& GPUrestrict() merger = processors.tpcMerger;
30  const o2::tpc::ClusterNativeAccess* GPUrestrict() clusters = processors.ioPtrs.clustersNative;
31  GPUTPCCompression& GPUrestrict() compressor = processors.tpcCompressor;
32  const GPUParam& GPUrestrict() param = processors.param;
33 
34  char lastLeg = 0;
35  int myTrack = 0;
36  for (unsigned int i = get_global_id(0); i < (unsigned int)merger.NOutputTracks(); i += get_global_size(0)) {
37  const GPUTPCGMMergedTrack& GPUrestrict() trk = merger.OutputTracks()[i];
38  if (!trk.OK()) {
39  continue;
40  }
41  bool rejectTrk = CAMath::Abs(trk.GetParam().GetQPt()) > processors.param.rec.tpcRejectQPt;
42  unsigned int nClustersStored = 0;
43  CompressedClustersPtrs& GPUrestrict() c = compressor.mPtrs;
44  unsigned int lastRow = 0, lastSlice = 0; // BUG: These should be unsigned char, but then CUDA breaks
45  GPUTPCCompressionTrackModel track;
46  for (int k = trk.NClusters() - 1; k >= 0; k--) {
47  const GPUTPCGMMergedTrackHit& GPUrestrict() hit = merger.Clusters()[trk.FirstClusterRef() + k];
48  if (hit.state & GPUTPCGMMergedTrackHit::flagReject) {
49  continue;
50  }
51 
52  int hitId = hit.num;
53  int attach = merger.ClusterAttachment()[hitId];
54  if ((attach & GPUTPCGMMergerTypes::attachTrackMask) != i) {
55  continue; // Main attachment to different track
56  }
57  bool rejectCluster = processors.param.rec.tpcRejectionMode && (rejectTrk || ((attach & GPUTPCGMMergerTypes::attachGoodLeg) == 0) || (attach & GPUTPCGMMergerTypes::attachHighIncl));
58  if (rejectCluster) {
59  compressor.mClusterStatus[hitId] = 1; // Cluster rejected, do not store
60  continue;
61  }
62 
63  if (!(param.rec.tpcCompressionModes & GPUSettings::CompressionTrackModel)) {
64  continue; // No track model compression
65  }
66  const ClusterNative& GPUrestrict() orgCl = clusters->clusters[hit.slice][hit.row][hit.num - clusters->clusterOffset[hit.slice][hit.row]];
67  float x = param.tpcGeometry.Row2X(hit.row);
68  float y = param.tpcGeometry.LinearPad2Y(hit.slice, hit.row, orgCl.getPad());
69  float z = param.tpcGeometry.LinearTime2Z(hit.slice, orgCl.getTime());
70  if (nClustersStored) {
71  if ((hit.slice < GPUCA_NSLICES) ^ (lastSlice < GPUCA_NSLICES)) {
72  break;
73  }
74  if (lastLeg != hit.leg && track.Mirror()) {
75  break;
76  }
77  if (track.Propagate(param.tpcGeometry.Row2X(hit.row), param.SliceParam[hit.slice].Alpha)) {
78  break;
79  }
80  }
81 
82  compressor.mClusterStatus[hitId] = 1; // Cluster compressed in track model, do not store as difference
83 
84  int cidx = trk.FirstClusterRef() + nClustersStored++;
85  if (nClustersStored == 1) {
86  unsigned char qpt = fabs(trk.GetParam().GetQPt()) < 20.f ? (trk.GetParam().GetQPt() * (127.f / 20.f) + 127.5f) : (trk.GetParam().GetQPt() > 0 ? 254 : 0);
87  track.Init(x, y, z, param.SliceParam[hit.slice].Alpha, qpt, param); // TODO: Compression track model must respect Z offset!
88 
89  myTrack = CAMath::AtomicAdd(&compressor.mMemory->nStoredTracks, 1u);
90  compressor.mAttachedClusterFirstIndex[myTrack] = trk.FirstClusterRef();
91  lastLeg = hit.leg;
92  c.qPtA[myTrack] = qpt;
93  c.rowA[myTrack] = hit.row;
94  c.sliceA[myTrack] = hit.slice;
95  c.timeA[myTrack] = orgCl.getTimePacked();
96  c.padA[myTrack] = orgCl.padPacked;
97  } else {
98  unsigned int row = hit.row;
99  unsigned int slice = hit.slice;
100 
101  if (param.rec.tpcCompressionModes & GPUSettings::CompressionDifferences) {
102  if (lastRow > row) {
103  row += GPUCA_ROW_COUNT;
104  }
105  row -= lastRow;
106  if (lastSlice > slice) {
107  slice += compressor.NSLICES;
108  }
109  slice -= lastSlice;
110  }
111  c.rowDiffA[cidx] = row;
112  c.sliceLegDiffA[cidx] = (hit.leg == lastLeg ? 0 : compressor.NSLICES) + slice;
113  float pad = CAMath::Max(0.f, CAMath::Min((float)param.tpcGeometry.NPads(GPUCA_ROW_COUNT - 1), param.tpcGeometry.LinearY2Pad(hit.slice, hit.row, track.Y())));
114  c.padResA[cidx] = orgCl.padPacked - orgCl.packPad(pad);
115  float time = CAMath::Max(0.f, param.tpcGeometry.LinearZ2Time(hit.slice, track.Z()));
116  c.timeResA[cidx] = (orgCl.getTimePacked() - orgCl.packTime(time)) & 0xFFFFFF;
117  lastLeg = hit.leg;
118  }
119  lastRow = hit.row;
120  lastSlice = hit.slice;
121  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
122  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
123  if (param.rec.tpcCompressionModes & GPUSettings::CompressionTruncate) {
124  compressor.truncateSignificantBitsChargeMax(qmax, param);
125  compressor.truncateSignificantBitsCharge(qtot, param);
126  compressor.truncateSignificantBitsWidth(sigmapad, param);
127  compressor.truncateSignificantBitsWidth(sigmatime, param);
128  }
129  c.qTotA[cidx] = qtot;
130  c.qMaxA[cidx] = qmax;
131  c.sigmaPadA[cidx] = sigmapad;
132  c.sigmaTimeA[cidx] = sigmatime;
133  c.flagsA[cidx] = orgCl.getFlags();
134  if (k && track.Filter(y, z, hit.row)) {
135  break;
136  }
137  }
138  if (nClustersStored) {
139  CAMath::AtomicAdd(&compressor.mMemory->nStoredAttachedClusters, nClustersStored);
140  c.nTrackClusters[myTrack] = nClustersStored;
141  }
142  }
143 }

175 template <>
176 GPUdii() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step1unattached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& GPUrestrict() smem, processorType& GPUrestrict() processors)
177 {
178  const GPUTPCGMMerger& GPUrestrict() merger = processors.tpcMerger;
179  const o2::tpc::ClusterNativeAccess* GPUrestrict() clusters = processors.ioPtrs.clustersNative;
180  GPUTPCCompression& GPUrestrict() compressor = processors.tpcCompressor;
181  GPUParam& GPUrestrict() param = processors.param;
182  unsigned int* sortBuffer = smem.step1.sortBuffer;
183  for (int iSliceRow = iBlock; iSliceRow < GPUCA_NSLICES * GPUCA_ROW_COUNT; iSliceRow += nBlocks) {
184  const int iSlice = iSliceRow / GPUCA_ROW_COUNT;
185  const int iRow = iSliceRow % GPUCA_ROW_COUNT;
186  const int idOffset = clusters->clusterOffset[iSlice][iRow];
187  if (iThread == nThreads - 1) {
188  smem.step1.nCount = 0;
189  }
190  unsigned int totalCount = 0;
191  GPUbarrier();
192 
193  CompressedClustersPtrs& GPUrestrict() c = compressor.mPtrs;
194 
195  const unsigned int nn = GPUCommonMath::nextMultipleOf<GPUCA_GET_THREAD_COUNT(GPUCA_LB_GPUTPCCompressionKernels_step1unattached)>(clusters->nClusters[iSlice][iRow]);
196  for (unsigned int i = iThread; i < nn + nThreads; i += nThreads) {
197  const int idx = idOffset + i;
198  int cidx = 0;
199  do {
200  if (i >= clusters->nClusters[iSlice][iRow]) {
201  break;
202  }
203  if (compressor.mClusterStatus[idx]) {
204  break;
205  }
206  int attach = merger.ClusterAttachment()[idx];
207  bool unattached = attach == 0;
208 
209  if (unattached) {
210  if (processors.param.rec.tpcRejectionMode >= GPUSettings::RejectionStrategyB) {
211  break;
212  }
213  } else if (processors.param.rec.tpcRejectionMode >= GPUSettings::RejectionStrategyA) {
214  if ((attach & GPUTPCGMMergerTypes::attachGoodLeg) == 0) {
215  break;
216  }
217  if (attach & GPUTPCGMMergerTypes::attachHighIncl) {
218  break;
219  }
220  int id = attach & GPUTPCGMMergerTypes::attachTrackMask;
221  if (CAMath::Abs(merger.OutputTracks()[id].GetParam().GetQPt()) > processors.param.rec.tpcRejectQPt) {
222  break;
223  }
224  }
225  cidx = 1;
226  } while (false);
227 
228  GPUbarrier();
229  int myIndex = work_group_scan_inclusive_add(cidx);
230  int storeLater = -1;
231  if (cidx) {
232  if (smem.step1.nCount + myIndex <= GPUCA_TPC_COMP_CHUNK_SIZE) {
233  sortBuffer[smem.step1.nCount + myIndex - 1] = i;
234  } else {
235  storeLater = smem.step1.nCount + myIndex - 1 - GPUCA_TPC_COMP_CHUNK_SIZE;
236  }
237  }
238  GPUbarrier();
239  if (iThread == nThreads - 1) {
240  smem.step1.nCount += myIndex;
241  }
242  GPUbarrier();
243 
244  if (smem.step1.nCount < GPUCA_TPC_COMP_CHUNK_SIZE && i < nn) {
245  continue;
246  }
247 
248  const unsigned int count = CAMath::Min(smem.step1.nCount, (unsigned int)GPUCA_TPC_COMP_CHUNK_SIZE);
249  if (param.rec.tpcCompressionModes & GPUSettings::CompressionDifferences) {
250  if (param.rec.tpcCompressionSortOrder == GPUSettings::SortZPadTime) {
251  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortZPadTime>(clusters->clusters[iSlice][iRow]));
252  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortZTimePad) {
253  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortZTimePad>(clusters->clusters[iSlice][iRow]));
254  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortPad) {
255  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortPad>(clusters->clusters[iSlice][iRow]));
256  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortTime) {
257  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortTime>(clusters->clusters[iSlice][iRow]));
258  }
259  GPUbarrier();
260  }
261 
262  for (unsigned int j = get_local_id(0); j < count; j += get_local_size(0)) {
263  int outidx = idOffset + totalCount + j;
264  const ClusterNative& GPUrestrict() orgCl = clusters->clusters[iSlice][iRow][sortBuffer[j]];
265  unsigned int lastTime = 0;
266  unsigned int lastPad = 0;
267  if (j != 0) {
268  const ClusterNative& GPUrestrict() orgClPre = clusters->clusters[iSlice][iRow][sortBuffer[j - 1]];
269  lastPad = orgClPre.padPacked;
270  lastTime = orgClPre.getTimePacked();
271  } else if (totalCount != 0) {
272  const ClusterNative& GPUrestrict() orgClPre = clusters->clusters[iSlice][iRow][smem.step1.lastIndex];
273  lastPad = orgClPre.padPacked;
274  lastTime = orgClPre.getTimePacked();
275  }
276 
277  c.padDiffU[outidx] = orgCl.padPacked - lastPad;
278  c.timeDiffU[outidx] = (orgCl.getTimePacked() - lastTime) & 0xFFFFFF;
279 
280  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
281  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
282  if (param.rec.tpcCompressionModes & GPUSettings::CompressionTruncate) {
283  compressor.truncateSignificantBitsChargeMax(qmax, param);
284  compressor.truncateSignificantBitsCharge(qtot, param);
285  compressor.truncateSignificantBitsWidth(sigmapad, param);
286  compressor.truncateSignificantBitsWidth(sigmatime, param);
287  }
288  c.qTotU[outidx] = qtot;
289  c.qMaxU[outidx] = qmax;
290  c.sigmaPadU[outidx] = sigmapad;
291  c.sigmaTimeU[outidx] = sigmatime;
292  c.flagsU[outidx] = orgCl.getFlags();
293  }
294 
295  GPUbarrier();
296  if (storeLater > 0) {
297  sortBuffer[storeLater] = i;
298  }
299  totalCount += count;
300  if (iThread == nThreads - 1 && count) {
301  smem.step1.lastIndex = sortBuffer[count - 1];
302  smem.step1.nCount -= count;
303  }
304  }
305 
306  if (iThread == nThreads - 1) {
307  c.nSliceRowClusters[iSlice * GPUCA_ROW_COUNT + iRow] = totalCount;
308  CAMath::AtomicAdd(&compressor.mMemory->nStoredUnattachedClusters, totalCount);
309  }
310  GPUbarrier();
311  }
312 }

384 template <typename Scalar, typename BaseVector>
385 GPUdi() void GPUTPCCompressionKernels::compressorMemcpyVectorised(Scalar* dst, const Scalar* src, unsigned int size, int nThreads, int iThread)
386 {
387  if (not isAlignedTo<BaseVector>(dst)) {
388  size_t dsti = reinterpret_cast<size_t>(dst);
389  int offset = (alignof(BaseVector) - dsti % alignof(BaseVector)) / sizeof(Scalar);
390  if (iThread < offset) {
391  dst[iThread] = src[iThread];
392  }
393  src += offset;
394  dst += offset;
395  size -= offset;
396  }
397 
398  BaseVector* GPUrestrict() dstAligned = reinterpret_cast<BaseVector*>(dst);
399 
400  using CpyVec = CpyVector<Scalar, BaseVector>;
401  unsigned int sizeAligned = size / CpyVec::Size;
402 
403  if (isAlignedTo<BaseVector>(src)) {
404  const BaseVector* GPUrestrict() srcAligned = reinterpret_cast<const BaseVector*>(src);
405  compressorMemcpyBasic(dstAligned, srcAligned, sizeAligned, nThreads, iThread);
406  } else {
407  for (unsigned int i = iThread; i < sizeAligned; i += nThreads) {
408  CpyVec buf;
409  for (unsigned int j = 0; j < CpyVec::Size; j++) {
410  buf.elems[j] = src[i * CpyVec::Size + j];
411  }
412  dstAligned[i] = buf.all;
413  }
414  }
415 
416  int leftovers = size % CpyVec::Size;
417  if (iThread < leftovers) {
418  dst[size - leftovers + iThread] = src[size - leftovers + iThread];
419  }
420 }

422 template <typename T>
423 GPUdi() unsigned int GPUTPCCompressionKernels::calculateWarpOffsets(GPUSharedMemory& smem, T* nums, unsigned int start, unsigned int end, int iWarp, int nLanes, int iLane)
424 {
425  unsigned int offset = 0;
426  if (iWarp > -1) {
427  for (unsigned int i = start + iLane; i < end; i += nLanes) {
428  offset += nums[i];
429  }
430  }
431  offset = work_group_scan_inclusive_add(int(offset)); // FIXME: use scan with unsigned int
432  if (iWarp > -1 && iLane == nLanes - 1) {
433  smem.step2.warpOffset[iWarp] = offset;
434  }
435  GPUbarrier();
436  offset = (iWarp <= 0) ? 0 : smem.step2.warpOffset[iWarp - 1];
437 
438  return offset;
439 }
440 template <>
441 GPUdii() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step2gather>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& GPUrestrict() smem, processorType& GPUrestrict() processors)
442 {
443  GPUTPCCompression& GPUrestrict() compressor = processors.tpcCompressor;
444  const o2::tpc::ClusterNativeAccess* GPUrestrict() clusters = processors.ioPtrs.clustersNative;
445 
446  int nWarps = nThreads / GPUCA_WARP_SIZE;
447  int iWarp = iThread / GPUCA_WARP_SIZE;
448 
449  int nLanes = GPUCA_WARP_SIZE;
450  int iLane = iThread % GPUCA_WARP_SIZE;
451 
452  if (iBlock == 0) {
453 
454  unsigned int nRows = compressor.NSLICES * GPUCA_ROW_COUNT;
455  unsigned int rowsPerWarp = (nRows + nWarps - 1) / nWarps;
456  unsigned int rowStart = rowsPerWarp * iWarp;
457  unsigned int rowEnd = CAMath::Min(nRows, rowStart + rowsPerWarp);
458 
459  unsigned int rowsOffset = calculateWarpOffsets(smem, compressor.mPtrs.nSliceRowClusters, rowStart, rowEnd, iWarp, nLanes, iLane);
460 
461  compressorMemcpy(compressor.mOutput->nSliceRowClusters, compressor.mPtrs.nSliceRowClusters, compressor.NSLICES * GPUCA_ROW_COUNT, nThreads, iThread);
462  compressorMemcpy(compressor.mOutput->nTrackClusters, compressor.mPtrs.nTrackClusters, compressor.mMemory->nStoredTracks, nThreads, iThread);
463  compressorMemcpy(compressor.mOutput->qPtA, compressor.mPtrs.qPtA, compressor.mMemory->nStoredTracks, nThreads, iThread);
464  compressorMemcpy(compressor.mOutput->rowA, compressor.mPtrs.rowA, compressor.mMemory->nStoredTracks, nThreads, iThread);
465  compressorMemcpy(compressor.mOutput->sliceA, compressor.mPtrs.sliceA, compressor.mMemory->nStoredTracks, nThreads, iThread);
466  compressorMemcpy(compressor.mOutput->timeA, compressor.mPtrs.timeA, compressor.mMemory->nStoredTracks, nThreads, iThread);
467  compressorMemcpy(compressor.mOutput->padA, compressor.mPtrs.padA, compressor.mMemory->nStoredTracks, nThreads, iThread);
468 
469  unsigned int sliceStart = rowStart / GPUCA_ROW_COUNT;
470  unsigned int sliceEnd = rowEnd / GPUCA_ROW_COUNT;
471 
472  unsigned int sliceRowStart = rowStart % GPUCA_ROW_COUNT;
473  unsigned int sliceRowEnd = rowEnd % GPUCA_ROW_COUNT;
