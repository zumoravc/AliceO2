42 int AlpideCoder::encodeChip(PayLoadCont& buffer, const o2::itsmft::ChipPixelData& chipData,
43  uint16_t chipInModule, uint16_t bc, uint16_t roflags)
44 {
45  // Encode chip data into provided buffer. Data must be provided sorted in row/col, no check is done
46  int nfound = 0;
47  auto pixels = chipData.getData();
48  if (pixels.size()) { // non-empty chip
49  for (const auto& pix : pixels) {
50  addPixel(pix.getRow(), pix.getCol());
51  }
52  buffer.addFast(makeChipHeader(chipInModule, bc)); // chip header
53  for (int ir = 0; ir < NRegions; ir++) {
54  nfound += procRegion(buffer, ir);
55  }
56  buffer.addFast(makeChipTrailer(roflags));
57  resetMap();
58  } else {
59  buffer.addFast(makeChipEmpty(chipInModule, bc));
60  }
61  //
62  return nfound;
63 }

66 int AlpideCoder::procDoubleCol(PayLoadCont& buffer, short reg, short dcol)
67 {
68  // process double column: encoding
69  short hits[2 * NRows];
70  int nHits = 0, nData = 0;
71  //
72  int nr = mFirstInRow.size();
73  short col0 = ((reg * NDColInReg + dcol) << 1), col1 = col0 + 1; // 1st,2nd column of double column
74  int prevRow = -1;
75  for (int ir = 0; ir < nr; ir++) {
76  int linkID = mFirstInRow[ir];
77  if (linkID == -1 || mPix2Encode[linkID].col > col1) { // no pixels left on this row or higher column IDs
78  continue;
79  }
80  short rowID = mPix2Encode[linkID].row;
81  // process fired pixels
82  bool left = 0, right = 0;
83  if (mPix2Encode[linkID].col == col0) { // pixel in left column
84  left = 1;
85  linkID = mPix2Encode[linkID].nextInRow; // unlink processed pixel
86  }
87  if (linkID != -1 && mPix2Encode[linkID].col == col1) { // pixel in right column
88  right = 1;
89  linkID = mPix2Encode[linkID].nextInRow; // unlink processed pixel
90  }
91  short addr0 = rowID << 1;
92  if (rowID & 0x1) { // odd rows: right to left numbering
93  if (right) {
94  hits[nHits++] = addr0;
95  }
96  if (left) {
97  hits[nHits++] = addr0 + 1;
98  }
99  } else { // even rows: left to right numbering
100  if (left) {
101  hits[nHits++] = addr0;
102  }
103  if (right) {
104  hits[nHits++] = addr0 + 1;
105  }
106  }
107  if (linkID == -1) {
108  mFirstInRow[ir] = -1;
109  continue;
110  } // this row is finished
111  mFirstInRow[ir] = linkID; // link remaining hit pixels to row
112  }
113  //
114  int ih = 0;
115  if (nHits) {
116  buffer.addFast(makeRegion(reg)); // flag region start
117  }
118 
119  while ((ih < nHits)) {
120  short addrE, addrW = hits[ih++]; // address of the reference hit
121  uint8_t mask = 0;
122  short addrLim = addrW + HitMapSize + 1; // 1+address of furthest hit can be put in the map
123  while ((ih < nHits && (addrE = hits[ih]) < addrLim)) {
124  mask |= 0x1 << (addrE - addrW - 1);
125  ih++;
126  }
127  if (mask) { // flag DATALONG
128  buffer.addFast(makeDataLong(dcol, addrW));
129  buffer.addFast(mask);
130  } else {
131  buffer.addFast(makeDataShort(dcol, addrW));
132  }
133  nData++;
134  }
135  //
136  return nData;
137 }
