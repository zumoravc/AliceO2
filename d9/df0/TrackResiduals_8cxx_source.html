76 void TrackResiduals::setY2XBinning(const std::vector<float>& binning)
77 {
78  if (mIsInitialized) {
79  LOG(error) << "Binning already initialized, not changing y/x binning";
80  return;
81  }
82  int nBins = binning.size() - 1;
83  if (fabsf(binning[0] + 1.f) > param::sEps || fabsf(binning[nBins] - 1.f) > param::sEps) {
84  LOG(error) << "Provided binning for y/x not in range -1 to 1: " << binning[0] << " - " << binning[nBins] << ". Not changing y/x binning";
85  return;
86  }
87  setNY2XBins(nBins);
88  mUniformBins[VoxF] = false;
89  mY2XBinsDH.clear();
90  mY2XBinsDI.clear();
91  mY2XBinsCenter.clear();
92  for (int iBin = 0; iBin < nBins; ++iBin) {
93  mY2XBinsDH.push_back(.5f * (binning[iBin + 1] - binning[iBin]));
94  mY2XBinsDI.push_back(.5f / mY2XBinsDH[iBin]);
95  mY2XBinsCenter.push_back(binning[iBin] + mY2XBinsDH[iBin]);
96  }
97 }

123 //______________________________________________________________________________
124 void TrackResiduals::initBinning()
125 {
126  // initialize binning structures
127  //
128  // X binning
129  if (mNXBins > 0 && mNXBins < param::NPadRows) {
130  // uniform binning in X
131  O2INFO("X-binning is uniform with %i bins from %.2f to %.2f", mNXBins, param::MinX, param::MaxX);
132  mDXI = mNXBins / (param::MaxX - param::MinX);
133  mDX = 1.0f / mDXI;
134  mUniformBins[VoxX] = true;
135  } else {
136  // binning per pad row
137  O2INFO("X-binning is per pad-row");
138  mNXBins = param::NPadRows;
139  mUniformBins[VoxX] = false;
140  mDX = param::RowDX[0];
141  mDXI = 1.f / mDX; // should not be used
142  }
143  //
144  // Y/X binning
145  mMaxY2X.resize(mNXBins);
146  mDY2XI.resize(mNXBins);
147  mDY2X.resize(mNXBins);
148  //
149  for (int ix = 0; ix < mNXBins; ++ix) {
150  float x = getX(ix);
151  mMaxY2X[ix] = tan(.5f * SECPHIWIDTH) - sDeadZone / x;
152  mDY2XI[ix] = mNY2XBins / (2.f * mMaxY2X[ix]);
153  mDY2X[ix] = 1.f / mDY2XI[ix];
154  }
155  if (mUniformBins[VoxF]) {
156  O2INFO("Y/X-binning is uniform with %i bins from -MaxY2X to +MaxY2X", mNY2XBins);
157  for (int ip = 0; ip < mNY2XBins; ++ip) {
158  mY2XBinsDH.push_back(1.f / mNY2XBins);
159  mY2XBinsDI.push_back(.5f / mY2XBinsDH[ip]);
160  mY2XBinsCenter.push_back(-1.f + (ip + 0.5f) * 2.f * mY2XBinsDH[ip]);
161  O2DEBUG("Bin %i: center (%.3f), half bin width (%.3f)", ip, mY2XBinsCenter.back(), mY2XBinsDH.back());
162  }
163  }
164  //
165  // Z/X binning
166  mDZ2XI = mNZ2XBins / sMaxZ2X;
167  mDZ2X = 1.0f / mDZ2XI; // for uniform case only
168  if (mUniformBins[VoxZ]) {
169  O2INFO("Z/X-binning is uniform with %i bins from 0 to sMaxZ2X", mNY2XBins);
170  for (int iz = 0; iz < mNZ2XBins; ++iz) {
171  mZ2XBinsDH.push_back(.5f * mDZ2X);
172  mZ2XBinsDI.push_back(mDZ2XI);
173  mZ2XBinsCenter.push_back((iz + 0.5f) * mDZ2X);
174  O2DEBUG("Bin %i: center (%.3f), half bin width (%.3f)", iz, mZ2XBinsCenter.back(), mZ2XBinsDH.back());
175  }
176  }
177  //
178  mNVoxPerSector = mNY2XBins * mNZ2XBins * mNXBins;
179 }

213 int TrackResiduals::getRowID(float x) const
214 {
215  int ix;
216 
217 #ifdef TPC_RUN2
218 
219  if (x < param::RowX[param::NRowsAccumulated[0] - 1] + param::RowDX[0]) {
220  // we are in the IROC
221  ix = (x - (param::RowX[0] - .5f * param::RowDX[0])) / param::RowDX[0];
222  if (ix < 0) {
223  // x is smaller than the inner radius of the first pad row
224  ix = -1;
225  }
226  } else if (x >= param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1]) {
227  // we are in the OROC2
228  ix = (x - (param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1])) / param::RowDX[param::NROCTypes - 1] + param::NRowsAccumulated[param::NROCTypes - 2];
229  if (ix >= param::NPadRows) {
230  // x is larger than the outer radius of the last OROC pad row
231  ix = -1;
232  }
233  } else {
234  // we are in the OROC1
235  ix = (x - (param::RowX[param::NRowsAccumulated[0]] - .5f * param::RowDX[1])) / param::RowDX[1] + param::NRowsAccumulated[0];
236  if (ix < param::NRowsAccumulated[0]) {
237  // The given x is between IROC and OROC1, where there is a gap of 2.5 cm. Between OROC1 and OROC2 there is no gap in the geometrical description of the TPC
238  ix = -1;
239  }
240  }
241 
242 #else // use TPC geometry for Run 3 and beyond
243 
244  // calculations are slightly more complex, since differently to Run 1 + 2 there are gaps between all ROCs
245  if (x < param::RowX[param::NRowsAccumulated[0] - 1] + param::RowDX[0]) {
246  // we are in the IROC
247  ix = (x - (param::RowX[0] - .5f * param::RowDX[0])) / param::RowDX[0];
248  if (ix < 0) {
249  // x is smaller than the inner radius of the first pad row
250  ix = -1;
251  }
252  } else if (x >= param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1]) {
253  // we are in the OROC3
254  ix = (x - (param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1])) / param::RowDX[param::NROCTypes - 1] + param::NRowsAccumulated[param::NROCTypes - 2];
255  if (ix >= param::nPadRows) {
256  // x is larger than the outer radius of the last OROC pad row
257  ix = -1;
258  }
259  } else if (x > param::RowX[param::NRowsAccumulated[0]] - .5f * param::RowDX[1] && x < param::RowX[param::NRowsAccumulated[1] - 1] + .5f * param::RowDX[1]) {
260  // we are in the OROC1
261  ix = (x - (param::RowX[param::NRowsAccumulated[0]] - .5f * param::RowDX[1])) / param::RowDX[1] + param::NRowsAccumulated[0];
262  } else if (x > param::RowX[param::NRowsAccumulated[1]] - .5f * param::RowDX[2] && x < param::RowX[param::NRowsAccumulated[2] - 1] + .5f * param::RowDX[2]) {
263  // we are in the OROC2
264  ix = (x - (param::RowX[param::NRowsAccumulated[1]] - .5f * param::RowDX[2])) / param::RowDX[2] + param::NRowsAccumulated[1];
265  } else {
266  // x is in one of the gaps between the ROCs
267  ix = -1;
268  }
269 
270 #endif
271 
272  return ix;
273 }

275 bool TrackResiduals::findVoxelBin(int secID, float x, float y, float z, std::array<unsigned char, VoxDim>& bvox) const
276 {
277  // Z/X bin
278  if (fabs(z / x) > sMaxZ2X) {
279  return false;
280  }
281  int bz = getZ2XBinExact(secID < SECTORSPERSIDE ? z / x : -z / x);
282  if (bz < 0) {
283  return false;
284  }
285  // X bin
286  int bx = getXBinExact(x);
287  if (bx < 0 || bx >= mNXBins) {
288  return false;
289  }
290  // Y/X bin
291  int bp = getY2XBinExact(y / x, bx);
292  if (bp < 0 || bp >= mNY2XBins) {
293  return false;
294  }
295 
296  bvox[VoxZ] = bz;
297  bvox[VoxX] = bx;
298  bvox[VoxF] = bp;
299  return true;
300 }

302 void TrackResiduals::setKernelType(KernelType kernel, float bwX, float bwP, float bwZ, float scX, float scP, float scZ)
303 {
304  // set kernel type and widths in terms of binning in x, y/x, z/x and define aux variables
305  mKernelType = kernel;
306 
307  mKernelScaleEdge[VoxX] = scX;
308  mKernelScaleEdge[VoxF] = scP;
309  mKernelScaleEdge[VoxZ] = scZ;
310 
311  mKernelWInv[VoxX] = (bwX > 0) ? 1. / bwX : 1.;
312  mKernelWInv[VoxF] = (bwP > 0) ? 1. / bwP : 1.;
313  mKernelWInv[VoxZ] = (bwZ > 0) ? 1. / bwZ : 1.;
314 
315  if (mKernelType == KernelType::Epanechnikov) {
316  // bandwidth 1
317  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX + 0.5));
318  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP + 0.5));
319  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ + 0.5));
320  } else if (mKernelType == KernelType::Gaussian) {
321  // look in ~5 sigma
322  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX * 5. + 0.5));
323  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP * 5. + 0.5));
324  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ * 5. + 0.5));
325  } else {
326  LOG(error) << "given kernel type is not defined";
327  }
328  for (int i = VoxDim; i--;) {
329  if (mStepKern[i] < 1) {
330  mStepKern[i] = 1;
331  }
332  }
333 }

341 void TrackResiduals::buildLocalResidualTreesFromRun2Data()
342 {
343  // prepare trees for storage of local residuals
344  prepareLocalResidualTrees();
345  // access delta trees created by AliTPCcalibAlignInterpolation::Process();
346  mRun2DeltaTree = std::make_unique<TChain>(mResidualDataTreeName.data());
347  mRun2DeltaTree->AddFile((mPathToResidualFiles + mResidualDataFileName).data());
348  prepareDeltaTreeBranches();
349  std::array<float, param::NPadRows> residHelixY;
350  std::array<float, param::NPadRows> residHelixZ;
351  auto* brTRDOK = mRun2DeltaTree->GetBranch("trdOK");
352  auto* brITSOK = mRun2DeltaTree->GetBranch("itsOK");
353  auto nTracks = mRun2DeltaTree->GetEntries();
354  int nTracksSelected = 0;
355  int nTracksSelectedWithOutliers = 0;
356  std::array<int, 3> counterTrkValidation{0};
357  int nRejCl = 0, nRejHelix = 0, nRejQpt = 0, nRejValidation = 0;
358  LOG(info) << "Building local residual trees from " << nTracks << " tracks.";
359  std::vector<DebugOutliers> debugOutliers;
360  for (int iTrk = 0; iTrk < nTracks; ++iTrk) {
361  /*
362  printf("Checking track %i\n", iTrk);
363  if (iTrk > 2) {
364  break;
365  }
366  */
367  DebugOutliers debug;
368  brTRDOK->GetEntry(iTrk);
369  brITSOK->GetEntry(iTrk);
370  if (!mDeltaStruct.trdOK || !mDeltaStruct.itsOK) {
371  //printf("Track %i trd or its not OK\n", iTrk);
372  continue;
373  }
374  mRun2DeltaTree->GetEntry(iTrk);
375  const Float_t* vSec = mDeltaStruct.vecSec->GetMatrixArray();
376  const Float_t* vPhi = mDeltaStruct.vecPhi->GetMatrixArray();
377  const Float_t* vR = mDeltaStruct.vecR->GetMatrixArray();
378  const Float_t* vZ = mDeltaStruct.vecZ->GetMatrixArray();
379  const Float_t* vDY = mDeltaStruct.vecDYtrd->GetMatrixArray();
380  const Float_t* vDZ = mDeltaStruct.vecDZtrd->GetMatrixArray();
381  const Float_t* vDYits = mDeltaStruct.vecDYits->GetMatrixArray();
382  const Float_t* vDZits = mDeltaStruct.vecDZits->GetMatrixArray();
383  mQpt = mDeltaStruct.param[4];
384  mTgl = mDeltaStruct.param[3];
385  mNCl = 0;
386 
387  //printf("qpt(%.2f), tgl(%.2f), nPoints(%03i), current mNCl(%03i)\n", mQpt, mTgl, mDeltaStruct.npValid, mNCl);
388 
389  // load all points into buffer
390  for (int iCl = 0; iCl < mDeltaStruct.npValid; ++iCl) {
391  //printf("checking point %03i, mNCl(%03i), nPadRows(%i)\n", iCl, mNCl, param::NPadRows);
392  // check if point is OK
393  if (vR[iCl] < param::InvalidR || vDY[iCl] < param::InvalidRes || vDYits[iCl] < param::InvalidRes) {
394  //printf("point %03i not OK: vR(%.2f), vDY(%.2f), vDTits(%.2f)\n", iCl, vR[iCl], vDY[iCl], vDYits[iCl]);
395  continue;
396  }
397  // then fill buffer arrays
398  mArrX[mNCl] = -1;
399  mArrR[mNCl] = vR[iCl];
400  mArrZTr[mNCl] = vZ[iCl] + vDZ[iCl] - vDZits[iCl];
401  mArrDY[mNCl] = vDY[iCl];
402  mArrDZ[mNCl] = vDZ[iCl];
403  mArrPhi[mNCl] = vPhi[iCl];
404  if (mArrPhi[mNCl] < 0) {
405  mArrPhi[mNCl] += o2::constants::math::TwoPI;
406  }
407  mArrSecId[mNCl] = static_cast<int>(nearbyint(vSec[iCl])) % (SECTORSPERSIDE * SIDES); // 0..35 for sectors from A0 to C17
408  ++mNCl;
409  }
410  if (mNCl < param::MinNCl) {
411  //printf("Not enough points: %i \n", mNCl);
412  ++nRejCl;
413  continue;
414  }
415  ++nTracksSelectedWithOutliers;
416  //printf("Checking track %i\n", iTrk);
417  bool resHelix = compareToHelix(residHelixY, residHelixZ);
418  debug.idx = iTrk;
419  debug.x = mArrR;
420  debug.dY = mArrDY;
421  debug.dZ = mArrDZ;
422  debug.residHelixY = residHelixY;
423  debug.residHelixZ = residHelixZ;
424 
425  /*
426  printf("Printing helix residuals for track %i\n", iTrk);
427  for (int i = 0; i < param::NPadRows; ++i) {
428  printf("residHelixY[%03i]=% .4f \t \t residHelixZ[%03i]=% .4f\n", i, residHelixY[i], i, residHelixZ[i]);
429  }
430  */
431 
432  if (mFilterOutliers && !resHelix) {
433  // too strong deviation to helix -> discard track
434  //printf("Track %i deviates strongly from helix\n", iTrk);
435  ++nRejHelix;
436  debug.flags = 1 << 0; // rejected by helix fit
437  debugOutliers.push_back(debug);
438  continue;
439  }
440  if (fabsf(mQpt) > param::MaxQ2Pt) {
441  // discard low pt tracks now that a more precise q/pt estimate is available
442  ++nRejQpt;
443  debug.flags = 1 << 1; // rejected by qpt cut
444  debugOutliers.push_back(debug);
445  continue;
446  }
447 
448  // now everything needs to be converted to the sector frame
449  int nClTmp = mNCl;
450  mNCl = 0;
451  for (int iCl = 0; iCl < nClTmp; ++iCl) {
452  int side = mArrSecId[iCl] / SECTORSPERSIDE; // TODO check if side is always correct!
453  float cs = cos(mArrPhi[iCl] - (.5f + mArrSecId[iCl] % SECTORSPERSIDE) * o2::constants::math::SectorSpanRad);
454  float sn = std::sqrt((1.f + cs) * (1.f - cs)); // sin^2 + cos^2 = 1 => sin = sqrt(1-cos^2)
455 
456  // by using propagation in the cluster frame in AliTPCcalibAlignInterpolation::Process(),
457  // the x of the track is evaluated not at the pad-row x = r * cs, but at x = r * cs - dy * sn
458  float xRow = mArrR[iCl] * cs;
459  float dx = mArrDY[iCl] * sn;
460  float yCl = mArrR[iCl] * sn; // cluster y in sector frame
461  float yTrk = yCl + mArrDY[iCl] * cs; // track Y in sector frame at x = xTrk = xRow - dx
462  float zTrk = mArrZTr[iCl]; // track Z at x = xTrk = xRow - dx
463  float zCl = zTrk - mArrDZ[iCl]; // cluster z is zTrk - deltaZ
464  // use linear approximation to take the track to the real pad-row x
465  float tgSlp = mArrTgSlp[iCl];
466  if (fabsf(tgSlp) > param::MaxTgSlp) {
467  continue;
468  }
469  yTrk += dx * tgSlp;
470  float csXtrkInv = std::sqrt(1.f + tgSlp * tgSlp); // invers cosine of track angle
471  zTrk += dx * mTgl * csXtrkInv;
472 
473  // assign recalculated residuals to arrays
474  mArrX[mNCl] = xRow;
475  mArrYTr[mNCl] = yTrk;
476  mArrZTr[mNCl] = zTrk;
477  mArrYCl[mNCl] = yCl;
478  mArrZCl[mNCl] = zCl;
479  mArrDY[mNCl] = yTrk - yCl;
480  mArrDZ[mNCl] = zTrk - zCl;
481  // prevent under-/overflows
482  if (fabsf(mArrDY[mNCl]) > param::MaxResid - param::sEps) {
483  continue;
484  }
485  if (fabsf(mArrDZ[mNCl]) > param::MaxResid - param::sEps) {
486  continue;
487  }
488  if (mArrX[mNCl] < param::MinX || mArrX[mNCl] > param::MaxX) {
489  continue;
490  }
491  if (fabsf(mArrZCl[mNCl]) > param::ZLimit[side]) {
492  continue;
493  }
494  // done converting everything to sector frame
495  ++mNCl;
496  }
497  debug.x = mArrX;
498  debug.dY = mArrDY;
499  debug.dZ = mArrDZ;
500  if (mFilterOutliers && !validateTrack(counterTrkValidation)) {
501  ++nRejValidation;
502  debug.flags = 1 << 2;
503  debugOutliers.push_back(debug);
504  continue;

567 int TrackResiduals::checkResiduals(std::bitset<param::NPadRows>& rejCl, float& rmsLong)
568 {
569  int secStart = mArrSecId[0];
570  int iClFirst = 0;
571  int iClLast = mNCl - 1;
572 
573  // arrays with differences / abs(differences) of points to their neighbourhood, initialized to zero
574  std::array<float, param::NPadRows> yDiffLL{};
575  std::array<float, param::NPadRows> zDiffLL{};
576  std::array<float, param::NPadRows> absDevY{};
577  std::array<float, param::NPadRows> absDevZ{};
578 
579  for (int iCl = 0; iCl < mNCl; ++iCl) {
580  if (iCl < iClLast && mArrSecId[iCl] == secStart) {
581  continue;
582  }
583  // sector changed or last cluster reached
584  // now run estimators for all points in the same sector
585  int nClSec = iCl - iClFirst;
586  if (iCl == iClLast) {
587  ++nClSec;
588  }
589  diffToLocLine(nClSec, iClFirst, mArrX, mArrDY, yDiffLL);
590  diffToLocLine(nClSec, iClFirst, mArrX, mArrDZ, zDiffLL);
591  iClFirst = iCl;
592  secStart = mArrSecId[iCl];
593  }
594  // store abs deviations
595  int nAccY = 0;
596  int nAccZ = 0;
597  for (int iCl = mNCl; iCl--;) {
598  if (fabsf(yDiffLL[iCl]) > param::sEps) {
599  absDevY[nAccY++] = fabsf(yDiffLL[iCl]);
600  }
601  if (fabsf(zDiffLL[iCl]) > param::sEps) {
602  absDevZ[nAccZ++] = fabsf(zDiffLL[iCl]);
603  }
604  }
605  if (nAccY < param::MinNumberOfAcceptedResiduals || nAccZ < param::MinNumberOfAcceptedResiduals) {
606  // mask all clusters
607  rejCl.set();
608  return mNCl;
609  }
610  // estimate rms on 90% of the smallest deviations
611  int nKeepY = static_cast<int>(.9 * nAccY);
612  int nKeepZ = static_cast<int>(.9 * nAccZ);
613  std::nth_element(absDevY.begin(), absDevY.begin() + nKeepY, absDevY.begin() + nAccY);
614  std::nth_element(absDevZ.begin(), absDevZ.begin() + nKeepZ, absDevZ.begin() + nAccZ);
615  float rmsYkeep = 0.f;
616  float rmsZkeep = 0.f;
617  for (int i = nKeepY; i--;) {
618  rmsYkeep += absDevY[i] * absDevY[i];
619  }
620  for (int i = nKeepZ; i--;) {
621  rmsZkeep += absDevZ[i] * absDevZ[i];
622  }
623  rmsYkeep = std::sqrt(rmsYkeep / nKeepY);
624  rmsZkeep = std::sqrt(rmsZkeep / nKeepZ);
625  if (rmsYkeep < param::sEps || rmsZkeep < param::sEps) {
626  LOG(warning) << "Too small RMS: " << rmsYkeep << "(y), " << rmsZkeep << "(z).";
627  rejCl.set();
628  return mNCl;
629  }
630  float rmsYkeepI = 1.f / rmsYkeep;
631  float rmsZkeepI = 1.f / rmsZkeep;
632  int nAcc = 0;
633  std::array<float, param::NPadRows> yAcc;
634  std::array<float, param::NPadRows> yDiffLong;
635  for (int iCl = 0; iCl < mNCl; ++iCl) {
636  yDiffLL[iCl] *= rmsYkeepI;
637  zDiffLL[iCl] *= rmsZkeepI;
638  if (yDiffLL[iCl] * yDiffLL[iCl] + zDiffLL[iCl] * zDiffLL[iCl] > param::mMaxStdDevMA) {
639  rejCl.set(iCl);
640  } else {
641  yAcc[nAcc++] = mArrDY[iCl];
642  }
643  }
644  if (nAcc > mNMALong) {
645  diffToMA(nAcc, yAcc, yDiffLong);
646  float average = 0.f;
647  float rms = 0.f;
648  for (int i = 0; i < nAcc; ++i) {
649  average += yDiffLong[i];
650  rms += yDiffLong[i] * yDiffLong[i];
651  }
652  average /= nAcc;
653  rmsLong = rms / nAcc - average * average;
654  rmsLong = (rmsLong > 0) ? std::sqrt(rmsLong) : 0.f;
655  }
656  return rejCl.count();
657 }

999 void TrackResiduals::processVoxelResiduals(std::vector<float>& dy, std::vector<float>& dz, std::vector<float>& tg, VoxRes& resVox)
1000 {
1001  size_t nPoints = dy.size();
1002  //LOG(debug) << "processing voxel residuals for vox " << getGlbVoxBin(resVox.bvox) << " with " << nPoints << " points";
1003  if (nPoints < mMinEntriesPerVoxel) {
1004  LOG(info) << "voxel " << getGlbVoxBin(resVox.bvox) << " is skipped due to too few entries (" << nPoints << " < " << mMinEntriesPerVoxel << ")";
1005  return;
1006  }
1007  std::array<float, 7> zResults;
1008  resVox.flags = 0;
1009  std::vector<size_t> indices(dz.size());
1010  if (!o2::math_utils::math_base::LTMUnbinned(dz, indices, zResults, mLTMCut)) {
1011  LOG(debug) << "failed trimming input array for voxel " << getGlbVoxBin(resVox.bvox);
1012  return;
1013  }
1014  std::array<float, 2> res{0.f};
1015  std::array<float, 3> err{0.f};
1016  float sigMAD = fitPoly1Robust(tg, dy, res, err, mLTMCut);
1017  if (sigMAD < 0) {
1018  LOG(debug) << "failed robust linear fit, sigMAD = " << sigMAD;
1019  return;
1020  }
1021  float corrErr = err[0] * err[2];
1022  corrErr = corrErr > 0 ? err[1] / std::sqrt(corrErr) : -999;
1023  //
1024  resVox.D[ResX] = -res[1];
1025  resVox.D[ResY] = res[0];
1026  resVox.D[ResZ] = zResults[1];
1027  resVox.E[ResX] = std::sqrt(err[2]);
1028  resVox.E[ResY] = std::sqrt(err[0]);
1029  resVox.E[ResZ] = zResults[4];

1257  std::array<std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>, ResDim> cmat;

1756 void TrackResiduals::medFit(int nPoints, int offset, const std::vector<float>& x, const std::vector<float>& y, float& a, float& b, std::array<float, 3>& err) const
1757 {
1758  // fitting a straight line y(x|a, b) = a + b * x
1759  // to given x and y data minimizing the absolute deviation
1760  float aa, bb, chi2 = 0.f;
1761  if (nPoints < 2) {
1762  a = b = 0.f;
1763  err[0] = err[1] = err[2] = 999.f;
1764  return;
1765  }
1766  // do least squares minimization as first guess
1767  float sx = 0.f, sxx = 0.f, sy = 0.f, sxy = 0.f;
1768  for (int j = nPoints + offset; j-- > offset;) { // same order as in AliRoot version such that resulting sums are identical
1769  sx += x[j];
1770  sxx += x[j] * x[j];
1771  sy += y[j];
1772  sxy += x[j] * y[j];
1773  }
1774  float del = nPoints * sxx - sx * sx;
1775  float delI = 1. / del;
1776  aa = (sxx * sy - sx * sxy) * delI;
1777  bb = (nPoints * sxy - sx * sy) * delI;
1778  err[0] = sxx * delI;
1779  err[1] = sx * delI;
1780  err[2] = nPoints * delI;
1781 
1782  for (int j = nPoints + offset; j-- > offset;) {
1783  float tmp = y[j] - (aa + bb * x[j]);
1784  chi2 += tmp * tmp;
1785  }
1786  float sigb = std::sqrt(chi2 * delI); // expected sigma for b
1787  float b1 = bb;
1788  float f1 = roFunc(nPoints, offset, x, y, b1, aa);
1789  if (sigb > 0) {
1790  float b2 = bb + std::copysign(3.f * sigb, f1);
1791  float f2 = roFunc(nPoints, offset, x, y, b2, aa);
1792  if (fabs(f1 - f2) < sFloatEps) {
1793  a = aa;
1794  b = bb;
1795  return;
1796  }
1797  while (f1 * f2 > 0.f) {
1798  bb = b2 + 1.6f * (b2 - b1);
1799  b1 = b2;
1800  f1 = f2;
1801  b2 = bb;
1802  f2 = roFunc(nPoints, offset, x, y, b2, aa);
1803  }
1804  sigb = .01f * sigb;
1805  while (fabs(b2 - b1) > sigb) {
1806  bb = b1 + .5f * (b2 - b1);
1807  if (bb == b1 || bb == b2) {
1808  break;
1809  }
1810  float f = roFunc(nPoints, offset, x, y, bb, aa);
1811  if (f * f1 >= .0f) {
1812  f1 = f;
1813  b1 = bb;
1814  } else {
1815  f2 = f;
1816  b2 = bb;
1817  }
1818  }
1819  }
1820  a = aa;
1821  b = bb;
1822 }
