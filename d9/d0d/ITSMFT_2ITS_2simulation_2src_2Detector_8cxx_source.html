74 static double radii2Turbo(double rMin, double rMid, double rMax, double sensW)
75 {
76  // compute turbo angle from radii and sensor width
77  return TMath::ASin((rMax * rMax - rMin * rMin) / (2 * rMid * sensW)) * TMath::RadToDeg();
78 }

80 static void configITS(Detector* its)
81 {
82  // build ITS upgrade detector
83  const int kNLr = 7;
84  const int kNLrInner = 3;
85  const int kBuildLevel = 0;
86  const int kSensTypeID = 0; // dummy id for Alpide sensor
87 
88  const float ChipThicknessIB = 50.e-4;
89  const float ChipThicknessOB = 100.e-4;
90 
91  enum { kRmn,
92  kRmd,
93  kRmx,
94  kNModPerStave,
95  kPhi0,
96  kNStave,
97  kNPar };
98  // Radii are from last TDR (ALICE-TDR-017.pdf Tab. 1.1, rMid is mean value)
99  const double tdr5dat[kNLr][kNPar] = {
100  {2.24, 2.34, 2.67, 9., 16.42, 12}, // for each inner layer: rMin,rMid,rMax,NChip/Stave, phi0, nStaves
101  {3.01, 3.15, 3.46, 9., 12.18, 16},
102  {3.78, 3.93, 4.21, 9., 9.55, 20},
103  {-1, 19.6, -1, 4., 0., 24}, // for others: -, rMid, -, NMod/HStave, phi0, nStaves // 24 was 49
104  {-1, 24.55, -1, 4., 0., 30}, // 30 was 61
105  {-1, 34.39, -1, 7., 0., 42}, // 42 was 88
106  {-1, 39.34, -1, 7., 0., 48} // 48 was 100
107  };
108  const int nChipsPerModule = 7; // For OB: how many chips in a row
109  const double zChipGap = 0.01; // For OB: gap in Z between chips
110  const double zModuleGap = 0.01; // For OB: gap in Z between modules
111 
112  double dzLr, rLr, phi0, turbo;
113  int nStaveLr, nModPerStaveLr;
114 
115  its->setStaveModelIB(o2::its::Detector::kIBModel4);
116  its->setStaveModelOB(o2::its::Detector::kOBModel2);
117 
118  const int kNWrapVol = 3;
119  const double wrpRMin[kNWrapVol] = {2.1, 19.3, 33.32};
120  const double wrpRMax[kNWrapVol] = {15.4, 29.14, 46.0};
121  const double wrpZSpan[kNWrapVol] = {70., 93., 163.6};
122 
123  for (int iw = 0; iw < kNWrapVol; iw++) {
124  its->defineWrapperVolume(iw, wrpRMin[iw], wrpRMax[iw], wrpZSpan[iw]);
125  }
126 
127  for (int idLr = 0; idLr < kNLr; idLr++) {
128  rLr = tdr5dat[idLr][kRmd];
129  phi0 = tdr5dat[idLr][kPhi0];
130 
131  nStaveLr = TMath::Nint(tdr5dat[idLr][kNStave]);
132  nModPerStaveLr = TMath::Nint(tdr5dat[idLr][kNModPerStave]);
133  int nChipsPerStaveLr = nModPerStaveLr;
134  if (idLr >= kNLrInner) {
135  its->defineLayer(idLr, phi0, rLr, nStaveLr, nModPerStaveLr, ChipThicknessOB, Segmentation::SensorLayerThickness,
136  kSensTypeID, kBuildLevel);
137  } else {
138  turbo = radii2Turbo(tdr5dat[idLr][kRmn], rLr, tdr5dat[idLr][kRmx], Segmentation::SensorSizeRows);
139  its->defineLayerTurbo(idLr, phi0, rLr, nStaveLr, nChipsPerStaveLr, Segmentation::SensorSizeRows, turbo,
140  ChipThicknessIB, Segmentation::SensorLayerThickness, kSensTypeID, kBuildLevel);
141  }
142  }
143 }

271 Bool_t Detector::ProcessHits(FairVolume* vol)
272 {
273  // This method is called from the MC stepping
274  if (!(fMC->TrackCharge())) {
275  return kFALSE;
276  }
277 
278  Int_t lay = 0, volID = vol->getMCid();
279 
280  // FIXME: Determine the layer number. Is this information available directly from the FairVolume?
281  bool notSens = false;
282  while ((lay < sNumberLayers) && (notSens = (volID != mLayerID[lay]))) {
283  ++lay;
284  }
285  if (notSens)
286  return kFALSE; // RS: can this happen? This method must be called for sensors only?
287 
288  // Is it needed to keep a track reference when the outer ITS volume is encountered?
289  auto stack = (o2::data::Stack*)fMC->GetStack();
290  if (fMC->IsTrackExiting() && (lay == 0 || lay == 6)) {
291  // Keep the track refs for the innermost and outermost layers only
292  o2::TrackReference tr(*fMC, GetDetId());
293  tr.setTrackID(stack->GetCurrentTrackNumber());
294  tr.setUserId(lay);
295  stack->addTrackReference(tr);
296  }
297  bool startHit = false, stopHit = false;
298  unsigned char status = 0;
299  if (fMC->IsTrackEntering()) {
300  status |= Hit::kTrackEntering;
301  }
302  if (fMC->IsTrackInside()) {
303  status |= Hit::kTrackInside;
304  }
305  if (fMC->IsTrackExiting()) {
306  status |= Hit::kTrackExiting;
307  }
308  if (fMC->IsTrackOut()) {
309  status |= Hit::kTrackOut;
310  }
311  if (fMC->IsTrackStop()) {
312  status |= Hit::kTrackStopped;
313  }
314  if (fMC->IsTrackAlive()) {
315  status |= Hit::kTrackAlive;
316  }
317 
318  // track is entering or created in the volume
319  if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
320  startHit = true;
321  } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
322  stopHit = true;
323  }
324 
325  // increment energy loss at all steps except entrance
326  if (!startHit)
327  mTrackData.mEnergyLoss += fMC->Edep();
328  if (!(startHit | stopHit))
329  return kFALSE; // do noting
330 
331  if (startHit) {
332  mTrackData.mEnergyLoss = 0.;
333  fMC->TrackMomentum(mTrackData.mMomentumStart);
334  fMC->TrackPosition(mTrackData.mPositionStart);
335  mTrackData.mTrkStatusStart = status;
336  mTrackData.mHitStarted = true;
337  }
338  if (stopHit) {
339  TLorentzVector positionStop;
340  fMC->TrackPosition(positionStop);
341  // Retrieve the indices with the volume path
342  int stave(0), halfstave(0), chipinmodule(0), module;
343  fMC->CurrentVolOffID(1, chipinmodule);
344  fMC->CurrentVolOffID(2, module);
345  fMC->CurrentVolOffID(3, halfstave);
346  fMC->CurrentVolOffID(4, stave);
347  int chipindex = mGeometryTGeo->getChipIndex(lay, stave, halfstave, module, chipinmodule);
348 
349  Hit* p = addHit(stack->GetCurrentTrackNumber(), chipindex, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
350  mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
351  mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
352  // p->SetTotalEnergy(vmc->Etot());
353 
354  // RS: not sure this is needed
355  // Increment number of Detector det points in TParticle
356  stack->addHit(GetDetId());
357  }
358 
359  return kTRUE;
360 }

696 TGeoVolume* Detector::createWrapperVolume(Int_t id)
697 {
698  // Creates an air-filled wrapper cylindrical volume
699  // For OB a Pcon is needed to host the support rings
700  // while avoiding overlaps with MFT structures and OB cones
701 
702  const Double_t suppRingAZlen = 4.;
703  const Double_t coneRingARmax = 33.96;
704  const Double_t coneRingAZlen = 5.6;
705  const Double_t suppRingCZlen[3] = {4.8, 4.0, 2.4};
706  const Double_t suppRingsRmin[3] = {23.35, 20.05, 35.4};
707 
708  if (mWrapperMinRadius[id] < 0 || mWrapperMaxRadius[id] < 0 || mWrapperZSpan[id] < 0) {
709  LOG(FATAL) << "Wrapper volume " << id << " was requested but not defined";
710  }
711 
712  // Now create the actual shape and volume
713  TGeoShape* tube;
714  Double_t zlen;
715  switch (id) {
716  case 0: // IB Layer 0,1,2: simple cylinder
717  {
718  TGeoTube* wrap = new TGeoTube(mWrapperMinRadius[id], mWrapperMaxRadius[id], mWrapperZSpan[id] / 2.);
719  tube = (TGeoShape*)wrap;
720  } break;
721  case 1: // MB Layer 3,4: complex Pcon to avoid MFT overlaps
722  {
723  TGeoPcon* wrap = new TGeoPcon(0, 360, 6);
724  zlen = mWrapperZSpan[id] / 2 + suppRingCZlen[0];
725  wrap->DefineSection(0, -zlen, suppRingsRmin[0], mWrapperMaxRadius[id]);
726  zlen = mWrapperZSpan[id] / 2 + suppRingCZlen[1];
727  wrap->DefineSection(1, -zlen, suppRingsRmin[0], mWrapperMaxRadius[id]);
728  wrap->DefineSection(2, -zlen, suppRingsRmin[1], mWrapperMaxRadius[id]);
729  wrap->DefineSection(3, -mWrapperZSpan[id] / 2., suppRingsRmin[1], mWrapperMaxRadius[id]);
730  wrap->DefineSection(4, -mWrapperZSpan[id] / 2., mWrapperMinRadius[id], mWrapperMaxRadius[id]);
731  zlen = mWrapperZSpan[id] / 2 + suppRingAZlen;
732  wrap->DefineSection(5, zlen, mWrapperMinRadius[id], mWrapperMaxRadius[id]);
733  tube = (TGeoShape*)wrap;
734  } break;
735  case 2: // OB Layer 5,6: simpler Pcon to avoid OB cones overlaps
736  {
737  TGeoPcon* wrap = new TGeoPcon(0, 360, 6);
738  zlen = mWrapperZSpan[id] / 2;
739  wrap->DefineSection(0, -zlen, suppRingsRmin[2], mWrapperMaxRadius[id]);
740  zlen -= suppRingCZlen[2];
741  wrap->DefineSection(1, -zlen, suppRingsRmin[2], mWrapperMaxRadius[id]);
742  wrap->DefineSection(2, -zlen, mWrapperMinRadius[id], mWrapperMaxRadius[id]);
743  zlen = mWrapperZSpan[id] / 2 - coneRingAZlen;
744  wrap->DefineSection(3, zlen, mWrapperMinRadius[id], mWrapperMaxRadius[id]);
745  wrap->DefineSection(4, zlen, coneRingARmax, mWrapperMaxRadius[id]);
746  wrap->DefineSection(5, mWrapperZSpan[id] / 2, coneRingARmax, mWrapperMaxRadius[id]);
747  tube = (TGeoShape*)wrap;
748  } break;
749  default: // Can never happen, keeps gcc quiet
750  break;
751  }
752 
753  TGeoMedium* medAir = gGeoManager->GetMedium("ITS_AIR$");
754 
755  char volnam[30];
756  snprintf(volnam, 29, "%s%d", GeometryTGeo::getITSWrapVolPattern(), id);
757 
758  auto* wrapper = new TGeoVolume(volnam, tube, medAir);
759 
760  return wrapper;
761 }

786

790 
791  // Check that we have all needed parameters
792  for (Int_t j = 0; j < sNumberLayers; j++) {
793  if (mLayerRadii[j] <= 0) {
794  LOG(FATAL) << "Wrong layer radius for layer " << j << "(" << mLayerRadii[j] << ")";
795  }
796  if (mStavePerLayer[j] <= 0) {
797  LOG(FATAL) << "Wrong number of staves for layer " << j << "(" << mStavePerLayer[j] << ")";
798  }
799  if (mUnitPerStave[j] <= 0) {
800  LOG(FATAL) << "Wrong number of chips for layer " << j << "(" << mUnitPerStave[j] << ")";
801  }
802  if (mChipThickness[j] < 0) {
803  LOG(FATAL) << "Wrong chip thickness for layer " << j << "(" << mChipThickness[j] << ")";
804  }
805  if (mTurboLayer[j] && mStaveWidth[j] <= 0) {
806  LOG(FATAL) << "Wrong stave width for layer " << j << "(" << mStaveWidth[j] << ")";
807  }
808  if (mDetectorThickness[j] < 0) {
809  LOG(FATAL) << "Wrong Sensor thickness for layer " << j << "(" << mDetectorThickness[j] << ")";
810  }
811 
812  if (j > 0) {
813  if (mLayerRadii[j] <= mLayerRadii[j - 1]) {
814  LOG(FATAL) << "Layer " << j << " radius (" << mLayerRadii[j] << ") is smaller than layer " << j - 1
815  << " radius (" << mLayerRadii[j - 1] << ")";
816  }
817  }
818 
819  if (mChipThickness[j] == 0) {
820  LOG(INFO) << "Chip thickness for layer " << j << " not set, using default";
821  }
822  }
823 
824  // Create the wrapper volumes
825  TGeoVolume** wrapVols = nullptr;
826 
827  if (sNumberOfWrapperVolumes) {
828  wrapVols = new TGeoVolume*[sNumberOfWrapperVolumes];
829  for (int id = 0; id < sNumberOfWrapperVolumes; id++) {
830  wrapVols[id] = createWrapperVolume(id);
831  vITSV->AddNode(wrapVols[id], 1, nullptr);
832  }
833  }

836  for (Int_t j = 0; j < sNumberLayers; j++) {
837  TGeoVolume* dest = vITSV;
838  mWrapperLayerId[j] = -1;
839 
840  if (mTurboLayer[j]) {
841  mGeometry[j] = new V3Layer(j, kTRUE, kFALSE);
842  mGeometry[j]->setStaveWidth(mStaveWidth[j]);
843  mGeometry[j]->setStaveTilt(mStaveTilt[j]);

845  mGeometry[j] = new V3Layer(j, kFALSE);
846  }
847 
848  mGeometry[j]->setPhi0(mLayerPhi0[j]);
849  mGeometry[j]->setRadius(mLayerRadii[j]);
850  mGeometry[j]->setNumberOfStaves(mStavePerLayer[j]);
851  mGeometry[j]->setNumberOfUnits(mUnitPerStave[j]);
852  mGeometry[j]->setChipType(mChipTypeID[j]);
853  mGeometry[j]->setBuildLevel(mBuildLevel[j]);
854 
855  if (j < sNumberInnerLayers) {
856  mGeometry[j]->setStaveModel(mStaveModelInnerBarrel);
857  } else {
858  mGeometry[j]->setStaveModel(mStaveModelOuterBarrel);
859  }
860 
861  LOG(DEBUG1) << "mBuildLevel: " << mBuildLevel[j];
862 
863  if (mChipThickness[j] != 0) {
864  mGeometry[j]->setChipThick(mChipThickness[j]);
865  }
866  if (mDetectorThickness[j] != 0) {
867  mGeometry[j]->setSensorThick(mDetectorThickness[j]);
868  }
869 
870  for (int iw = 0; iw < sNumberOfWrapperVolumes; iw++) {
871  if (mLayerRadii[j] > mWrapperMinRadius[iw] && mLayerRadii[j] < mWrapperMaxRadius[iw]) {
872  LOG(DEBUG) << "Will embed layer " << j << " in wrapper volume " << iw;

874  dest = wrapVols[iw];

877  }

880  }
881

884

886  createMiddlBarrelServices(wrapVols[1]);
887  createOuterBarrelServices(wrapVols[2]);
888  createOuterBarrelSupports(vITSV);
889 
890  // TEMPORARY - These routines will be obsoleted once the new services are completed - TEMPORARY
891  // createServiceBarrel(kTRUE, wrapVols[0]);
892  // createServiceBarrel(kFALSE, wrapVols[2]);
893 
894  delete[] wrapVols; // delete pointer only, not the volumes
