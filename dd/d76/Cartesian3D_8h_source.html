46 class Rotation2D
47 {
48  //
49  // class to perform rotation of 3D (around Z) and 2D points
50 
51  public:
52  Rotation2D() = default;
53  Rotation2D(float cs, float sn) : mCos(cs), mSin(sn) {}
54  Rotation2D(float phiZ) : mCos(cos(phiZ)), mSin(sin(phiZ)) {}
55  ~Rotation2D() = default;
56  Rotation2D(const Rotation2D& src) = default;
57  Rotation2D& operator=(const Rotation2D& src) = default;
58 
59  void set(float phiZ)
60  {
61  mCos = cos(phiZ);
62  mSin = sin(phiZ);
63  }
64 
65  void set(float cs, float sn)
66  {
67  mCos = cs;
68  mSin = sn;
69  }
70 
71  void getComponents(float& cs, float& sn) const
72  {
73  cs = mCos;
74  sn = mSin;
75  }
76 
77  template <typename T>
78  Point3D<T> operator()(const Point3D<T>& v) const
79  { // local->master
80  return Point3D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos, v.Z());
81  }
82 
83  template <typename T>
84  Point3D<T> operator^(const Point3D<T>& v) const
85  { // master->local
86  return Point3D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos, v.Z());
87  }
88 
89  template <typename T>
90  Vector3D<T> operator()(const Vector3D<T>& v) const
91  { // local->master
92  return Vector3D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos, v.Z());
93  }
94 
95  template <typename T>
96  Vector3D<T> operator^(const Vector3D<T>& v) const
97  { // master->local
98  return Vector3D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos, v.Z());
99  }
100 
101  template <typename T>
102  Point2D<T> operator()(const Point2D<T>& v) const
103  { // local->master
104  return Point2D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos);
105  }
106 
107  template <typename T>
108  Point2D<T> operator^(const Point2D<T>& v) const
109  { // master->local
110  return Point2D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos);
111  }
112 
113  template <typename T>
114  Vector2D<T> operator()(const Vector2D<T>& v) const
115  { // local->master
116  return Vector2D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos);
117  }
118 
119  template <typename T>
120  Vector2D<T> operator^(const Vector2D<T>& v) const
121  { // master->local
122  return Vector2D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos);
123  }
124 
125  private:
126  float mCos = 1.f;
127  float mSin = 0.f;
128

132 class Transform3D : public ROOT::Math::Transform3D
133 {
134  //
135  // Class to perform geom.transformations (rotation and displacements only) in
136  // double precision over the cartesian points and vectors (float or double).
137  // Adds to the base ROOT::Math::Transform3D<double> class a convertor from
138  // TGeoMatrix.
139  // To be used instead of TGeoHMatrix for all transformations of hits,
140  // clusters etc.
141  //
142 
143  public:
144  Transform3D() = default;
145  Transform3D(const TGeoMatrix& m);
146  ~Transform3D() = default;
147 
148  // inherit assignment operators of the base class
149  using ROOT::Math::Transform3D::operator=;
150 
151  // to avoid conflict between the base Transform3D(const ForeignMatrix & m) and
152  // Transform3D(const TGeoMatrix &m) constructors we cannot inherit base c-tors,
153  // therefore we redefine them here
154  Transform3D(const ROOT::Math::Rotation3D& r, const Vector& v) : ROOT::Math::Transform3D(r, v) {}
155  Transform3D(const ROOT::Math::Rotation3D& r, const ROOT::Math::Translation3D& t) : ROOT::Math::Transform3D(r, t) {}
156  template <class IT>
157  Transform3D(IT begin, IT end) : ROOT::Math::Transform3D(begin, end)
158  {
159  }
160 
161  // conversion operator to TGeoHMatrix
162  operator TGeoHMatrix&() const
163  {
164  static TGeoHMatrix tmp;
165  double rot[9], tra[3];
166  GetComponents(rot[0], rot[1], rot[2], tra[0], rot[3], rot[4], rot[5], tra[1], rot[6], rot[7], rot[8], tra[2]);
167  tmp.SetRotation(rot);
168  tmp.SetTranslation(tra);
169  return tmp;
170  }
171 
172  void set(const TGeoMatrix& m); // set parameters from TGeoMatrix
173 
174  using ROOT::Math::Transform3D::operator();
175  // the local->master transformation for points and vectors can be
176  // done in operator form (inherited from base Transform3D) as
177  // Point3D pnt;
178  // Transform3D trans;
179  // auto pntTr0 = trans(pnt); // 1st version
180  // auto pntTr1 = trans*pnt; // 2nd version
181  //
182  // For the inverse transform we define our own operator^
183 
184  template <typename T>
185  Point3D<T> operator^(const Point3D<T>& p) const
186  { // master->local
187  return ApplyInverse(p);
188  }
189 
190  template <typename T>
191  Vector3D<T> operator^(const Vector3D<T>& v) const
192  { // local->master
193  return ApplyInverse(v);
194  }
195 
196  // TGeoHMatrix-like aliases
197  template <typename T>
198  void LocalToMaster(const Point3D<T>& loc, Point3D<T>& mst) const
199  {
200  mst = operator()(loc);
201  }
202 
203  template <typename T>
204  void MasterToLocal(const Point3D<T>& mst, Point3D<T>& loc) const
205  {
206  loc = operator^(mst);
207  }
208 
209  template <typename T>
210  void LocalToMasterVect(const Point3D<T>& loc, Point3D<T>& mst) const
211  {

213  }
214 
215  template <typename T>
216  void MasterToLocalVect(const Point3D<T>& mst, Point3D<T>& loc) const
217  {
218  loc = operator^(mst);
219  }
