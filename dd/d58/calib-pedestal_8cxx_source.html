57 void printRDH(const RDH& rdh)
58 {
59  const int globalLinkID = int(rdh.linkID) + (((rdh.word1 >> 32) >> 28) * 12);
60 
61  LOGP(debug, "{:>5} {:>4} {:>4} {:>4} {:>3} {:>4} {:>10} {:>5} {:>1} {:#010X}", (uint64_t)rdh.packetCounter, (uint64_t)rdh.pageCnt, (uint64_t)rdh.feeId, (uint64_t)(rdh.memorySize), (uint64_t)rdh.cruID, (uint64_t)globalLinkID, (uint64_t)rdh.heartbeatOrbit, (uint64_t)rdh.heartbeatBC, (uint64_t)rdh.stop, (uint64_t)rdh.triggerType);
62 }

64 WorkflowSpec defineDataProcessing(ConfigContext const& config)
65 {
66  using namespace o2::tpc;
67 
68  // set up configuration
69  o2::conf::ConfigurableParam::updateFromFile(config.options().get<std::string>("configFile"));
70  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
71  o2::conf::ConfigurableParam::writeINI("o2tpccalibration_configuration.ini");
72 
73  struct ProcessAttributes {
74  CalibPedestal calibPedestal;
75  rawreader::RawReaderCRUManager rawReader;
76  uint32_t lastOrbit{0};
77  uint64_t lastTFID{0};
78  uint32_t maxEvents{100};
79  bool quit{false};
80  bool dumped{false};
81  };
82 
83  auto initFunction = [](InitContext& ic) {
84  auto processAttributes = std::make_shared<ProcessAttributes>();
85  // set up calibration
86  // TODO:
87  // it is a bit ugly to use the RawReaderCRUManager for this is.
88  // At some point the raw reader code should be cleaned up and modularized
89  {
90  auto& pedestal = processAttributes->calibPedestal;
91  pedestal.init(); // initialize configuration via configKeyValues
92  processAttributes->rawReader.createReader("");
93  processAttributes->rawReader.setADCDataCallback([&pedestal](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> Int_t {
94  Int_t timeBins = pedestal.update(padROCPos, cru, data);
95  pedestal.setNumberOfProcessedTimeBins(std::max(pedestal.getNumberOfProcessedTimeBins(), size_t(timeBins)));
96  return timeBins;
97  });
98  processAttributes->maxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
99  }
100 
101  auto processingFct = [processAttributes](ProcessingContext& pc) {
102  // in case the maximum number of events was reached don't do further processing
103  if (processAttributes->quit) {
104  return;
105  }
106 
107  if (pc.inputs().isValid("TFID")) {
108  auto tfid = pc.inputs().get<uint64_t>("TFID");
109  LOGP(info, "TFid: {}", tfid);
110  processAttributes->lastTFID = tfid;
111  }
112 
113  for (auto& input : pc.inputs()) {
114  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
115 
116  // ---| only process RAWDATA, there might be a nicer way to do this |---
117  if (dh == nullptr || dh->dataDescription != o2::header::gDataDescriptionRawData) {
118  continue;
119  }
120 
121  // ---| extract hardware information to do the processing |---
122  const auto subSpecification = dh->subSpecification;
123  const auto cruID = subSpecification >> 16;
124  const auto linkID = ((subSpecification + (subSpecification >> 8)) & 0xFF) - 1;
125  const auto dataWrapperID = ((subSpecification >> 8) & 0xFF) > 0;
126  const auto globalLinkID = linkID + dataWrapperID * 12;
127 
128  // ---| update hardware information in the reader |---
129  auto& reader = processAttributes->rawReader.getReaders()[0];
130  reader->forceCRU(cruID);
131  reader->setLink(globalLinkID);
132 
133  LOGP(debug, "Specifier: {}/{}/{}", dh->dataOrigin.as<std::string>(), dh->dataDescription.as<std::string>(), dh->subSpecification);
134  LOGP(debug, "Payload size: {}", dh->payloadSize);
135  LOGP(debug, "CRU: {}; linkID: {}; dataWrapperID: {}; globalLinkID: {}", cruID, linkID, dataWrapperID, globalLinkID);
136 
137  printHeader();
138 
139  // TODO: exception handling needed?
140  try {
141  o2::framework::RawParser parser(input.payload, dh->payloadSize);
142 
143  // TODO: it would be better to have external event handling and then moving the event processing functionality to CalibRawBase and RawReader to not repeat it in other places
144  rawreader::ADCRawData rawData;
145  rawreader::GBTFrame gFrame;
146 
147  auto& calibPedestal = processAttributes->calibPedestal;
148 
149  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
150  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV4>();
151  if (!rdhPtr) {
152  break;
153  }
154  const auto& rdh = *rdhPtr;
155  printRDH(rdh);
156  // ===| event handling |===
157  //
158  // really ugly, better treatment required extension in DPL
159  // events are are detected by close by orbit numbers
160  // might be possible to change this using the TFID information
161  //
162  const auto hbOrbit = rdh.heartbeatOrbit;
163  const auto lastOrbit = processAttributes->lastOrbit;
164 
165  if ((lastOrbit > 0) && (hbOrbit > (lastOrbit + 3))) {
166  calibPedestal.incrementNEvents();
167  LOGP(info, "Number of processed events: {} ({})", calibPedestal.getNumberOfProcessedEvents(), processAttributes->maxEvents);
168  if (calibPedestal.getNumberOfProcessedEvents() >= processAttributes->maxEvents) {
169  LOGP(info, "Maximm number of events reached ({}), no more processing will be done", processAttributes->maxEvents);
170  processAttributes->quit = true;
171  break;
172  }
173  }
174 
175  processAttributes->lastOrbit = hbOrbit;
176  const auto size = it.size();
177  auto data = it.data();
178  //LOGP(info, "Data size: {}", size);
179 
180  int iFrame = 0;
181  for (int i = 0; i < size; i += 16) {
182  gFrame.setFrameNumber(iFrame);
183  gFrame.setPacketNumber(iFrame / 508);
184  gFrame.readFromMemory(gsl::span<const o2::byte>(data + i, 16));
185 
186  // extract the half words from the 4 32-bit words
187  gFrame.getFrameHalfWords();
188 
189  // debug output
190  //if (CHECK_BIT(mDebugLevel, DebugLevel::GBTFrames)) {
191  //std::cout << gFrame;
192  //}
193 
194  gFrame.getAdcValues(rawData);
195  gFrame.updateSyncCheck(false);
196 
197  ++iFrame;
198  }
199  }
200 
201  reader->runADCDataCallback(rawData);
202  } catch (const std::runtime_error& e) {
203  LOGP(error, "can not create raw parser form input data");
204  o2::header::hexDump("payload", input.payload, dh->payloadSize, 64);
205  LOG(ERROR) << e.what();
206  }
207  }
208 
209  // TODO: For the moment simply dump calibration output to file, to check if everything is working as expected
210  if (processAttributes->quit && !processAttributes->dumped) {
211  LOGP(info, "Dumping output");
212  processAttributes->calibPedestal.analyse();
213  processAttributes->calibPedestal.dumpToFile("pedestals.root");
214  processAttributes->dumped = true;
215  //pc.services().get<ControlService>().endOfStream();
216  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
217  }
218  };
219 
220  return processingFct;
221  };
222 
223  WorkflowSpec workflow;
224  workflow.emplace_back(DataProcessorSpec{
225  "calib-pedestal",
226  select(config.options().get<std::string>("input-spec").c_str()),
227  Outputs{},
228  AlgorithmSpec{initFunction},
229  Options{{"max-events", VariantType::Int, 100, {"maximum number of events to process"}}}});
230 
231  return workflow;
232 }
