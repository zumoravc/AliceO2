115 template <typename T>
116 auto squaredDistanceOfPointToSegment(const Vertex<T>& p, const Vertex<T>& p0, const Vertex<T>& p1)
117  -> decltype(p0.x * p1.x)
118 {
120  auto v = p1 - p0;
121  auto w = p - p0;
122 
123  auto c1 = dot(w, v);
124 
125  if (c1 <= 0) {
126  return squaredDistance(p, p0);
127  }
128 
129  auto c2 = dot(v, v);
130  if (c2 <= c1) {
131  return squaredDistance(p, p1);
132  }
133 
134  auto b = c1 / c2;
135  Vertex<T> pbase{p0.x + b * v.x, p0.y + b * v.y};
136  return squaredDistance(p, pbase);
137 }
