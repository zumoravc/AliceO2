512 BOOST_AUTO_TEST_CASE(DataQuery)
513 {
514  auto empty_bindings = [](std::runtime_error const& ex) -> bool {
515  BOOST_CHECK_EQUAL(ex.what(), "Parse error: empty binding string");
516  return true;
517  };
518  auto missing_origin = [](std::runtime_error const& ex) -> bool {
519  BOOST_CHECK_EQUAL(ex.what(), "Parse error: origin needs to be between 1 and 4 char long");
520  return true;
521  };
522  auto missing_description = [](std::runtime_error const& ex) -> bool {
523  BOOST_CHECK_EQUAL(ex.what(), "Parse error: description needs to be between 1 and 16 char long");
524  return true;
525  };
526  auto missing_subspec = [](std::runtime_error const& ex) -> bool {
527  BOOST_CHECK_EQUAL(ex.what(), "Parse error: Expected a number");
528  return true;
529  };
530  auto missing_timemodulo = [](std::runtime_error const& ex) -> bool {
531  BOOST_CHECK_EQUAL(ex.what(), "Parse error: Expected a number");
532  return true;
533  };
534  auto trailing_semicolon = [](std::runtime_error const& ex) -> bool {
535  BOOST_CHECK_EQUAL(ex.what(), "Parse error: Remove trailing ;");
536  return true;
537  };
538  // Empty query.
539  BOOST_CHECK(DataDescriptorQueryBuilder::parse().empty() == true);
540  // Empty bindings.
541  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse(":"), std::runtime_error, empty_bindings);
542  // Missing origin
543  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:"), std::runtime_error, missing_origin);
544  // Origin too long
545  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:bacjasbjkca"), std::runtime_error, missing_origin);
546  // This is a valid expression, short for x:TST/*/* or x:TST/$1/$2
547  BOOST_CHECK_NO_THROW(DataDescriptorQueryBuilder::parse("x:TST"));
548  // This one is not, as we expect a description after a /
549  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:TST/"), std::runtime_error, missing_description);
550  // This one is not, as the description is too long
551  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:TST/cdjancajncjancjkancjkadncancnacaklmcak"), std::runtime_error, missing_description);
552  // This one is ok, short for "x:TST/A1/*"
553  BOOST_CHECK_NO_THROW(DataDescriptorQueryBuilder::parse("x:TST/A1"));
554  // This one is not, as subspec needs to be a value or a range.
555  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:TST/A1/"), std::runtime_error, missing_subspec);
556  // Not valid as subspec should be a number.
557  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:TST/A1/a0"), std::runtime_error, missing_subspec);
558 
559  // Let's verify that the contents are correct.
560  auto result0 = DataDescriptorQueryBuilder::parse("x:TST/A1/77");
561  BOOST_CHECK_EQUAL(result0.size(), 1);
562  DataDescriptorMatcher expectedMatcher00{
563  DataDescriptorMatcher::Op::And,
564  OriginValueMatcher{"TST"},
565  std::make_unique<DataDescriptorMatcher>(
566  DataDescriptorMatcher::Op::And,
567  DescriptionValueMatcher{"A1"},
568  std::make_unique<DataDescriptorMatcher>(
569  DataDescriptorMatcher::Op::And,
570  SubSpecificationTypeValueMatcher{77},
571  std::make_unique<DataDescriptorMatcher>(DataDescriptorMatcher::Op::Just,
572  StartTimeValueMatcher{ContextRef{0}})))};
573  auto matcher = std::get_if<DataDescriptorMatcher>(&result0[0].matcher);
574  BOOST_REQUIRE(matcher != nullptr);
575  BOOST_CHECK(expectedMatcher00 == *matcher);
576  std::ostringstream ss0;
577  ss0 << *matcher;
578  std::ostringstream expectedSS00;
579  expectedSS00 << expectedMatcher00;
580  BOOST_CHECK_EQUAL(ss0.str(), "(and origin:TST (and description:A1 (and subSpec:77 (just startTime:$0 ))))");
581  BOOST_CHECK_EQUAL(expectedSS00.str(), "(and origin:TST (and description:A1 (and subSpec:77 (just startTime:$0 ))))");
582  BOOST_CHECK_EQUAL(ss0.str(), expectedSS00.str());
583 
584  // This is valid. TimeModulo is 1.
585  BOOST_CHECK_NO_THROW(DataDescriptorQueryBuilder::parse("x:TST/A1/0"));
586  // Not valid as timemodulo should be a number.
587  BOOST_CHECK_EXCEPTION(DataDescriptorQueryBuilder::parse("x:TST/A1/0%"), std::runtime_error, missing_timemodulo);
