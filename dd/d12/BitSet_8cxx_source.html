34 template <typename T>
35 void setRangeFromIntegerFast(BitSet& bs, int a, int b, T v)
36 {
37  assertRange<T>(a, b, v);
38  if (b > bs.size()) {
39  bs.grow(b);
40  }
41  for (int i = 0; i <= b - a; i++) {
42  T check = static_cast<T>(1) << i;
43  bs.setFast(i + a, (v & check) != 0);
44  }

47 template <typename T>
48 T uint(const BitSet& bs, int a, int b)
49 {
50  if (b < 0) {
51  b = bs.len() - 1;
52  }
53  if (a < 0 || b < 0 || (b - a) > (sizeof(T) * 8) || a >= bs.size() || b >= bs.size()) {
54  throw std::out_of_range(fmt::format("Range [{0},{1}] out of range. bs.size()={2}", a, b, bs.size()));
55  }
56  T value{0};
57  for (T i = a; i <= b; i++) {
58  if (bs.get(i)) {
59  value |= (static_cast<T>(1) << (i - a));
60  }
61  }
62  return value;
63 }

65 template <typename T>
66 int appendT(BitSet& bs, T val, int n)
67 {
68  if (n <= 0) {
69  if (val > 0) {
70  n = impl::nofBits(val);
71  } else {
72  n = 1;
73  }
74  }
75 
76  if (n > sizeof(T) * 8) {
77  throw std::invalid_argument(fmt::format("n={0} is invalid : should be between 0 and {1}",
78  n, sizeof(T)));
79  }
80 
81  for (T i = 0; i < static_cast<T>(n); i++) {
82  T p = static_cast<T>(1) << i;
83  if ((val & p) == p) {
84  bs.append(true);
85  } else {
86  bs.append(false);
87  }
88  }
89  return n;

233 bool BitSet::grow(int n)
234 {
235  if (n <= 0) {
236  throw std::invalid_argument("n should be >= 0");
237  }
238  if (n < size()) {
239  return false;
240  }
241  if (n > BitSet::maxSize()) {
242  throw std::length_error(fmt::format("trying to allocate a bitset of more than {0} bytes", BitSet::maxSize()));
243  }
244  auto nbytes = mBytes.size();
245  while (nbytes * 8 < n) {
246  nbytes *= 2;
247  }
248  mBytes.resize(nbytes, 0);
249  mSize = mBytes.size() * 8;
250  return true;
251 }

253 BitSet BitSet::last(int n) const
254 {
255  if (len() < n) {
256  throw std::length_error(fmt::format("cannot get {0} bits out of a bitset of size {1}", n, len()));
257  }
258  auto subbs = subset(len() - n, len() - 1);
259  if (subbs.len() != n) {
260  throw std::logic_error("subset not of the expected len");
261  }
262  return subbs;
263 }

287 void BitSet::setFast(int pos, bool val)
288 {
289  uint8_t ix = pos % 8;
290  uint8_t& b = mBytes[pos / 8];
291 
292  uint8_t mask = 1 << ix;
293 
294  if (val) {
295  b |= mask;
296  } else {
297  b &= ~mask;
298  }
299 
300  if ((pos + 1) > mLen) {
301  mLen = pos + 1;
302  mMaxLen = std::max(mMaxLen, mLen);
303  }
304 }

383 {

416 // append n bits to BitSet bs.
417 // those n bits are taken from ringBuffer, starting at ringBuffer[startBit]
418 // return the value of startBit to be used for a future call to this method
419 // so the sequence of ringBuffer is not lost.
420 int circularAppend(BitSet& bs, const BitSet& ringBuffer, int startBit, int n)
421 {
422  int ix = startBit;
423 
424  while (n > 0) {
425  bs.append(ringBuffer.get(ix));
426  --n;
427  ix++;
428  ix %= ringBuffer.len();
429  }

439 std::string compactString(const BitSet& bs)
440 {
441  // replaces multiple sync patterns by nxSYNC
442  static BitSet syncWord(sampaSync().uint64(), 50);
443 
444  if (bs.size() < 49) {
445  return bs.stringLSBLeft();
446  }
447  std::string s;
448 
449  int i = 0;
450  int nsync = 0;
451  while (i + 49 < bs.len()) {
452  bool sync{false};
453  while (i + 49 < bs.len() && bs.subset(i, i + 49) == syncWord) {
454  i += 50;
455  nsync++;
456  sync = true;
457  }
458  if (sync) {
459  s += fmt::format("[{}SYNC]", nsync);
460  } else {
461  nsync = 0;
462  s += bs.get(i) ? "1" : "0";
463  i++;
464  }
465  }
466  for (int j = i; j < bs.len(); j++) {
467  s += bs.get(j) ? "1" : "0";
468  }
469  return s;
470 }
