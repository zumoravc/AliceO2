33 float checkResults(o2::utils::TreeStreamRedirector& outs, std::string& treeName, const FITTER& fitter,
34  Vec3D& vgen, TLorentzVector& genPar, const std::vector<double>& dtMass)
35 {
36  int nCand = fitter.getNCandidates();
37  std::array<float, 3> p;
38  float distMin = 1e9;
39  for (int ic = 0; ic < nCand; ic++) {
40  const auto& vtx = fitter.getPCACandidate(ic);
41  auto df = vgen;
42  df -= vtx;
43 
44  TLorentzVector moth, prong;
45  for (int i = 0; i < fitter.getNProngs(); i++) {
46  const auto& trc = fitter.getTrack(i, ic);
47  trc.getPxPyPzGlo(p);
48  prong.SetVectM({p[0], p[1], p[2]}, dtMass[i]);
49  moth += prong;
50  }
51  auto nIter = fitter.getNIterations(ic);
52  auto chi2 = fitter.getChi2AtPCACandidate(ic);
53  double dst = TMath::Sqrt(df[0] * df[0] + df[1] * df[1] + df[2] * df[2]);
54  distMin = dst < distMin ? dst : distMin;
55  // float genX
56  outs << treeName.c_str() << "cand=" << ic << "ncand=" << nCand << "nIter=" << nIter << "chi2=" << chi2
57  << "genPart=" << genPar << "recPart=" << moth
58  << "genX=" << vgen[0] << "genY=" << vgen[1] << "genZ=" << vgen[2]
59  << "dx=" << df[0] << "dy=" << df[1] << "dz=" << df[2] << "dst=" << dst << "\n";
60  }
61  return distMin;
62 }

64 TLorentzVector generate(Vec3D& vtx, std::vector<o2::track::TrackParCov>& vctr, float bz,
65  TGenPhaseSpace& genPHS, double parMass, const std::vector<double>& dtMass, std::vector<int> forceQ)
66 {
67  const float errYZ = 1e-2, errSlp = 1e-3, errQPT = 2e-2;
68  std::array<float, 15> covm = {
69  errYZ * errYZ,
70  0., errYZ * errYZ,
71  0, 0., errSlp * errSlp,
72  0., 0., 0., errSlp * errSlp,
73  0., 0., 0., 0., errQPT * errQPT};
74  bool accept = true;
75  TLorentzVector parent, d0, d1, d2;
76  do {
77  accept = true;
78  double y = gRandom->Rndm() - 0.5;
79  double pt = 0.1 + gRandom->Rndm() * 3;
80  double mt = TMath::Sqrt(parMass * parMass + pt * pt);
81  double pz = mt * TMath::SinH(y);
82  double phi = gRandom->Rndm() * TMath::Pi() * 2;
83  double en = mt * TMath::CosH(y);
84  double rdec = 10.; // radius of the decay
85  vtx[0] = rdec * TMath::Cos(phi);
86  vtx[1] = rdec * TMath::Sin(phi);
87  vtx[2] = rdec * pz / pt;
88  parent.SetPxPyPzE(pt * TMath::Cos(phi), pt * TMath::Sin(phi), pz, en);
89  int nd = dtMass.size();
90  genPHS.SetDecay(parent, nd, dtMass.data());
91  genPHS.Generate();
92  vctr.clear();
93  float p[4];
94  for (int i = 0; i < nd; i++) {
95  auto* dt = genPHS.GetDecay(i);
96  if (dt->Pt() < 0.05) {
97  accept = false;
98  break;
99  }
100  dt->GetXYZT(p);
101  float s, c, x;
102  std::array<float, 5> params;
103  o2::utils::sincosf(dt->Phi(), s, c);
104  o2::utils::rotateZInv(vtx[0], vtx[1], x, params[0], s, c);
105 
106  params[1] = vtx[2];
107  params[2] = 0.; // since alpha = phi
108  params[3] = 1. / TMath::Tan(dt->Theta());
109  params[4] = (i % 2 ? -1. : 1.) / dt->Pt();
110  covm[14] = errQPT * errQPT * params[4] * params[4];
111  //
112  // randomize
113  float r1, r2;
114  gRandom->Rannor(r1, r2);
115  params[0] += r1 * errYZ;
116  params[1] += r2 * errYZ;
117  gRandom->Rannor(r1, r2);
118  params[2] += r1 * errSlp;
119  params[3] += r2 * errSlp;
120  params[4] *= gRandom->Gaus(1., errQPT);
121  if (forceQ[i] == 0) {
122  params[4] = 0.; // impose straight track
123  }
124  auto& trc = vctr.emplace_back(x, dt->Phi(), params, covm);
125  float rad = forceQ[i] == 0 ? 600. : TMath::Abs(1. / trc.getCurvature(bz));
126  if (!trc.propagateTo(trc.getX() + (gRandom->Rndm() - 0.5) * rad * 0.05, bz) ||
127  !trc.rotate(trc.getAlpha() + (gRandom->Rndm() - 0.5) * 0.2)) {
128  printf("Failed to randomize ");
129  trc.print();
130  }
131  }
132  } while (!accept);
133 
134  return parent;
135 }

157 
158  o2::vertexing::DCAFitterN<2> ft; // 2 prong fitter
159  ft.setBz(bz);
160  ft.setPropagateToPCA(true); // After finding the vertex, propagate tracks to the DCA. This is default anyway
161  ft.setMaxR(200); // do not consider V0 seeds with 2D circles crossing above this R. This is default anyway
162  ft.setMaxDZIni(4); // do not consider V0 seeds with tracks Z-distance exceeding this. This is default anyway
163  ft.setMinParamChange(1e-3); // stop iterations if max correction is below this value. This is default anyway

165 
166  std::string treeName2A = "pr2a", treeName2W = "pr2w";
167  TStopwatch swA, swW;
168  int nfoundA = 0, nfoundW = 0;
169  double meanDA = 0, meanDW = 0;
170  swA.Stop();
171  swW.Stop();
172  for (int iev = 0; iev < NTest; iev++) {
173  auto genParent = generate(vtxGen, vctracks, bz, genPHS, k0, k0dec, forceQ);
174 
175  ft.setUseAbsDCA(true);
176  swA.Start(false);
177  int ncA = ft.process(vctracks[0], vctracks[1]); // HERE WE FIT THE VERTICES
178  swA.Stop();
179  LOG(DEBUG) << "fit abs.dist " << iev << " NC: " << ncA << " Chi2: " << (ncA ? ft.getChi2AtPCACandidate(0) : -1);
180  if (ncA) {
181  auto minD = checkResults(outStream, treeName2A, ft, vtxGen, genParent, k0dec);
182  meanDA += minD;
183  nfoundA++;
184  }
185 
186  ft.setUseAbsDCA(false);
187  swW.Start(false);
188  int ncW = ft.process(vctracks[0], vctracks[1]); // HERE WE FIT THE VERTICES
189  swW.Stop();
190  LOG(DEBUG) << "fit wgh.dist " << iev << " NC: " << ncW << " Chi2: " << (ncW ? ft.getChi2AtPCACandidate(0) : -1);
191  if (ncW) {
192  auto minD = checkResults(outStream, treeName2W, ft, vtxGen, genParent, k0dec);
193  meanDW += minD;
194  nfoundW++;
195  }
196  }
197  ft.print();
198  meanDA /= nfoundA ? nfoundA : 1;
199  meanDW /= nfoundW ? nfoundW : 1;
200  LOG(INFO) << "Processed " << NTest << " 2-prong vertices Helix : Helix";
201  LOG(INFO) << "2-prongs with abs.dist minization: eff= " << float(nfoundA) / NTest
202  << " mean.dist to truth: " << meanDA << " CPU time: " << swA.CpuTime();
203  LOG(INFO) << "2-prongs with wgh.dist minization: eff= " << float(nfoundW) / NTest
204  << " mean.dist to truth: " << meanDW << " CPU time: " << swW.CpuTime();
205  BOOST_CHECK(nfoundA > 0.99 * NTest);
206  BOOST_CHECK(nfoundW > 0.99 * NTest);
207  BOOST_CHECK(meanDA < 0.1);
208  BOOST_CHECK(meanDW < 0.1);
