32 GPUdii() bool GPUITSFitterKernel::fitTrack(GPUITSFitter& GPUrestrict() Fitter, GPUTPCGMPropagator& GPUrestrict() prop, GPUITSTrack& GPUrestrict() track, int start, int end, int step)
33 {
34  for (int iLayer{start}; iLayer != end; iLayer += step) {
35  if (track.mClusters[iLayer] == o2::its::constants::its::UnusedIndex) {
36  continue;
37  }
38  const TrackingFrameInfo& GPUrestrict() trackingHit = Fitter.trackingFrame()[iLayer][track.mClusters[iLayer]];
39 
40  if (prop.PropagateToXAlpha(trackingHit.xTrackingFrame, trackingHit.alphaTrackingFrame, step > 0)) {
41  return false;
42  }
43 
44  if (prop.Update(trackingHit.positionTrackingFrame[0], trackingHit.positionTrackingFrame[1], 0, false, trackingHit.covarianceTrackingFrame[0], trackingHit.covarianceTrackingFrame[2])) {
45  return false;
46  }
47 
48  /*const float xx0 = (iLayer > 2) ? 0.008f : 0.003f; // Rough layer thickness //FIXME
49  constexpr float radiationLength = 9.36f; // Radiation length of Si [cm]
50  constexpr float density = 2.33f; // Density of Si [g/cm^3]
51  if (!track.correctForMaterial(xx0, xx0 * radiationLength * density, true))
52  return false;*/
53  }
54  return true;
55 }

58 GPUdii() void GPUITSFitterKernel::Thread<0>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& GPUrestrict() smem, processorType& GPUrestrict() processors)
59 {
60  GPUITSFitter& Fitter = processors.itsFitter;
61 
62  GPUTPCGMPropagator prop;
63  prop.SetPolynomialField(&processors.param.polynomialField);
64  prop.SetMaxSinPhi(GPUCA_MAX_SIN_PHI);
65  prop.SetToyMCEventsFlag(0);
66  prop.SetFitInProjections(1);
67  float bz = -5.f; // FIXME
68 
69 #ifdef CA_DEBUG
70  int roadCounters[4]{0, 0, 0, 0};
71  int fitCounters[4]{0, 0, 0, 0};
72  int backpropagatedCounters[4]{0, 0, 0, 0};
73  int refitCounters[4]{0, 0, 0, 0};
74 #endif
75  for (int iRoad = get_global_id(0); iRoad < Fitter.NumberOfRoads(); iRoad += get_global_size(0)) {
76  Road& road = Fitter.roads()[iRoad];
77  int clusters[7] = {o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex, o2::its::constants::its::UnusedIndex};
78  int lastCellLevel = o2::its::constants::its::UnusedIndex;
79  CA_DEBUGGER(int nClusters = 2);
80 
81  for (int iCell{0}; iCell < o2::its::constants::its::CellsPerRoad; ++iCell) {
82  const int cellIndex = road[iCell];
83  if (cellIndex == o2::its::constants::its::UnusedIndex) {
84  continue;
85  } else {
86  clusters[iCell] = Fitter.cells()[iCell][cellIndex].getFirstClusterIndex();
87  clusters[iCell + 1] = Fitter.cells()[iCell][cellIndex].getSecondClusterIndex();
88  clusters[iCell + 2] = Fitter.cells()[iCell][cellIndex].getThirdClusterIndex();
89  lastCellLevel = iCell;
90  CA_DEBUGGER(nClusters++);
91  }
92  }
93 
94  CA_DEBUGGER(roadCounters[nClusters - 4]++);
95 
96  if (lastCellLevel == o2::its::constants::its::UnusedIndex) {
97  continue;
98  }
99 
101  for (int iC{0}; iC < 7; iC++) {
102  if (clusters[iC] != o2::its::constants::its::UnusedIndex) {
103  clusters[iC] = Fitter.clusters()[iC][clusters[iC]].clusterId;
104  }
105  }
107  const auto& cluster1 = Fitter.trackingFrame()[lastCellLevel + 2][clusters[lastCellLevel + 2]];
108  const auto& cluster2 = Fitter.trackingFrame()[lastCellLevel + 1][clusters[lastCellLevel + 1]];
109  const auto& cluster3 = Fitter.trackingFrame()[lastCellLevel][clusters[lastCellLevel]];
110 
111  GPUITSTrack temporaryTrack;
112  {
113  const float ca = CAMath::Cos(cluster3.alphaTrackingFrame), sa = CAMath::Sin(cluster3.alphaTrackingFrame);
114  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
115  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
116  const float z1 = cluster1.zCoordinate;
117  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
118  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
119  const float z2 = cluster2.zCoordinate;
120  const float x3 = cluster3.xTrackingFrame;
121  const float y3 = cluster3.positionTrackingFrame[0];
122  const float z3 = cluster3.positionTrackingFrame[1];
123 
124  const float crv = math_utils::computeCurvature(x1, y1, x2, y2, x3, y3);
125  const float x0 = math_utils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
126  const float tgl12 = math_utils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
127  const float tgl23 = math_utils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
128 
129  const float r2 = CAMath::Sqrt(cluster2.xCoordinate * cluster2.xCoordinate + cluster2.yCoordinate * cluster2.yCoordinate);
130  const float r3 = CAMath::Sqrt(cluster3.xCoordinate * cluster3.xCoordinate + cluster3.yCoordinate * cluster3.yCoordinate);
131  const float fy = 1. / (r2 - r3);
132  const float& tz = fy;
133  const float cy = (math_utils::computeCurvature(x1, y1, x2, y2 + o2::its::constants::its::Resolution, x3, y3) - crv) / (o2::its::constants::its::Resolution * bz * constants::math::B2C) * 20.f; // FIXME: MS contribution to the cov[14] (*20 added)
134  constexpr float s2 = o2::its::constants::its::Resolution * o2::its::constants::its::Resolution;
135 
136  temporaryTrack.X() = cluster3.xTrackingFrame;
137  temporaryTrack.Y() = y3;
138  temporaryTrack.Z() = z3;
139  temporaryTrack.SinPhi() = crv * (x3 - x0);
140  temporaryTrack.DzDs() = 0.5f * (tgl12 + tgl23);
141  temporaryTrack.QPt() = CAMath::Abs(bz) < constants::math::Almost0 ? constants::math::Almost0 : crv / (bz * constants::math::B2C);
142  temporaryTrack.TZOffset() = 0;
143  temporaryTrack.Cov()[0] = s2;
144  temporaryTrack.Cov()[1] = 0.f;
145  temporaryTrack.Cov()[2] = s2;
146  temporaryTrack.Cov()[3] = s2 * fy;
147  temporaryTrack.Cov()[4] = 0.f;
148  temporaryTrack.Cov()[5] = s2 * fy * fy;
149  temporaryTrack.Cov()[6] = 0.f;
150  temporaryTrack.Cov()[7] = s2 * tz;
151  temporaryTrack.Cov()[8] = 0.f;
152  temporaryTrack.Cov()[9] = s2 * tz * tz;
153  temporaryTrack.Cov()[10] = s2 * cy;
154  temporaryTrack.Cov()[11] = 0.f;
155  temporaryTrack.Cov()[12] = s2 * fy * cy;
156  temporaryTrack.Cov()[13] = 0.f;
157  temporaryTrack.Cov()[14] = s2 * cy * cy;
158  temporaryTrack.SetChi2(0);
159  temporaryTrack.SetNDF(-5);
160 
161  prop.SetTrack(&temporaryTrack, cluster3.alphaTrackingFrame);
162  }
163 
164  for (size_t iC = 0; iC < 7; ++iC) {
165  temporaryTrack.mClusters[iC] = clusters[iC];
166  }
167  bool fitSuccess = fitTrack(Fitter, prop, temporaryTrack, o2::its::constants::its::LayersNumber - 4, -1, -1);
168  if (!fitSuccess) {
169  continue;
170  }
171  CA_DEBUGGER(fitCounters[nClusters - 4]++);
172  temporaryTrack.ResetCovariance();
173  fitSuccess = fitTrack(Fitter, prop, temporaryTrack, 0, o2::its::constants::its::LayersNumber, 1);
174  if (!fitSuccess) {
175  continue;
176  }
177  CA_DEBUGGER(backpropagatedCounters[nClusters - 4]++);
178  for (int k = 0; k < 5; k++) {
179  temporaryTrack.mOuterParam.P[k] = temporaryTrack.Par()[k];
180  }
181  for (int k = 0; k < 15; k++) {
182  temporaryTrack.mOuterParam.C[k] = temporaryTrack.Cov()[k];
183  }
184  temporaryTrack.mOuterParam.X = temporaryTrack.X();
185  temporaryTrack.mOuterParam.alpha = prop.GetAlpha();
186  temporaryTrack.ResetCovariance();
187  fitSuccess = fitTrack(Fitter, prop, temporaryTrack, o2::its::constants::its::LayersNumber - 1, -1, -1);
188  if (!fitSuccess) {
189  continue;
190  }
191  CA_DEBUGGER(refitCounters[nClusters - 4]++);
192  int trackId = CAMath::AtomicAdd(&Fitter.NumberOfTracks(), 1u);
193  Fitter.tracks()[trackId] = temporaryTrack;
194  }
195 #ifdef CA_DEBUG
196  GPUInfo("Roads: %i %i %i %i", roadCounters[0], roadCounters[1], roadCounters[2], roadCounters[3]);
197  GPUInfo("Fitted tracks: %i %i %i %i", fitCounters[0], fitCounters[1], fitCounters[2], fitCounters[3]);
198  GPUInfo("Backpropagated tracks: %i %i %i %i", backpropagatedCounters[0], backpropagatedCounters[1], backpropagatedCounters[2], backpropagatedCounters[3]);
199  GPUInfo("Refitted tracks: %i %i %i %i", refitCounters[0], refitCounters[1], refitCounters[2], refitCounters[3]);
200 #endif
201 }
