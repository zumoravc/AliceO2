38 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
39 {
40  WorkflowSpec wf;
41  auto doClusters = !cfgc.options().get<bool>("no-clusters");
42  auto doPatterns = doClusters && !cfgc.options().get<bool>("no-cluster-patterns");
43  auto doDigits = cfgc.options().get<bool>("digits");
44  auto dict = cfgc.options().get<std::string>("dict-file");
45 
46  // Update the (declared) parameters if changed from the command line
47  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
48 
49  if (cfgc.options().get<bool>("mft")) {
50  wf.emplace_back(o2::itsmft::getSTFDecoderMFTSpec(doClusters, doPatterns, doDigits, dict));
51  } else {
52  wf.emplace_back(o2::itsmft::getSTFDecoderITSSpec(doClusters, doPatterns, doDigits, dict));
53  }
54  return wf;
55 }
