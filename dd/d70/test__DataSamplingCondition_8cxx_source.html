39  {
40  std::vector<bool> correctDecision{
41  true, false, true, false, true, false, false, true, false, false, true, true, false, false, true, false, false,
42  true, false, false, true, true, true, false, false, false, true, false, true, true, true, false, false, true,

44  for (DataProcessingHeader::StartTime id = 1; id < 50; id++) {
45  DataProcessingHeader dph{id, 0};
46  o2::header::Stack headerStack{dph};
47  DataRef dr{nullptr, reinterpret_cast<const char*>(headerStack.data()), nullptr};

50  }
