31 void AlpideSimResponse::initData()
32 {
33  /*
34  * read grid parameters and load data
35  */
36  if (mData.size()) {
37  cout << "Object already initialized" << endl;
38  print();
39  return;
40  }
41 
42  const float kTiny = 1e-6; // to check 0 values
43 
44  // if needed, append path with slash
45  if (mDataPath.length() && mDataPath.back() != '/') {
46  mDataPath.push_back('/');
47  }
48  mDataPath = gSystem->ExpandPathName(mDataPath.data());
49  string inpfname = mDataPath + mGridColName;
50  std::ifstream inpGrid;
51 
52  // read X grid
53  inpGrid.open(inpfname, std::ifstream::in);
54  if (inpGrid.fail()) {
55  LOG(FATAL) << "Failed to open file " << inpfname;
56  }
57 
58  while (inpGrid >> mStepInvCol && inpGrid.good()) {
59  mNBinCol++;
60  }
61 
62  if (!mNBinCol || mStepInvCol < kTiny) {
63  LOG(FATAL) << "Failed to read X(col) binning from " << inpfname;
64  }
65  mMaxBinCol = mNBinCol - 1;
66  mStepInvCol = mMaxBinCol / mStepInvCol; // inverse of the X bin width
67  inpGrid.close();
68 
69  // read Y grid
70  inpfname = mDataPath + mGridRowName;
71  inpGrid.open(inpfname, std::ifstream::in);
72  if (inpGrid.fail()) {
73  LOG(FATAL) << "Failed to open file " << inpfname;
74  }
75 
76  while (inpGrid >> mStepInvRow && inpGrid.good())
77  mNBinRow++;
78  if (!mNBinRow || mStepInvRow < kTiny) {
79  LOG(FATAL) << "Failed to read Y(row) binning from " << inpfname;
80  }
81  mMaxBinRow = mNBinRow - 1;
82  mStepInvRow = mMaxBinRow / mStepInvRow; // inverse of the Row bin width
83  inpGrid.close();
84 
85  // load response data
86  int nz = 0;
87  size_t cnt = 0;
88  float val, gx, gy, gz;
89  int lost, untrck, dead, nele;
90  size_t dataSize = 0;
91  mDptMax = -2.e9;
92  mDptMin = 2.e9;
93  const int npix = AlpideRespSimMat::getNPix();
94 
95  for (int ix = 0; ix < mNBinCol; ix++) {
96  for (int iy = 0; iy < mNBinRow; iy++) {
97  inpfname = composeDataName(ix, iy);
98  inpGrid.open(inpfname, std::ifstream::in);
99  if (inpGrid.fail()) {
100  LOG(FATAL) << "Failed to open file " << inpfname;
101  }
102  inpGrid >> nz;
103  if (cnt == 0) {
104  mNBinDpt = nz;
105  dataSize = mNBinCol * mNBinRow * mNBinDpt;
106  mData.reserve(dataSize); // reserve space for data
107  } else if (nz != mNBinDpt) {
108  LOG(FATAL) << "Mismatch in Nz slices of bin X(col): " << ix << " Y(row): " << iy
109  << " wrt bin 0,0. File " << inpfname;
110  }
111 
112  // load data
113  for (int iz = 0; iz < nz; iz++) {
114  AlpideRespSimMat mat;
115 
116  std::array<float, AlpideRespSimMat::MatSize>* arr = mat.getArray();
117  for (int ip = 0; ip < npix * npix; ip++) {
118  inpGrid >> val;
119  (*arr)[ip] = val;
120  cnt++;
121  }
122  inpGrid >> lost >> dead >> untrck >> nele >> gx >> gy >> gz;
123 
124  if (inpGrid.bad()) {
125  LOG(FATAL) << "Failed reading data for depth(Z) slice " << iz << " from "
126  << inpfname;
127  }
128  if (!nele) {
129  LOG(FATAL) << "Wrong normalization Nele=" << nele << "for depth(Z) slice "
130  << iz << " from " << inpfname;
131  }
132 
133  if (mDptMax < -1e9)
134  mDptMax = gz;
135  if (mDptMin > gz)
136  mDptMin = gz;
137 
138  // normalize
139  float norm = 1.f / nele;
140  for (int ip = npix * npix; ip--;)
141  (*arr)[ip] *= norm;
142  mData.push_back(mat); // store in the final container
143  } // loop over z
144 
145  inpGrid.close();
146 
147  } // loop over y
148  } // loop over x
149 
150  // final check
151  if (dataSize != mData.size()) {
152  LOG(FATAL) << "Mismatch between expected " << dataSize << " and loaded " << mData.size()
153  << " number of bins";
154  }
155 
156  // normalize Dpt boundaries
157 
158  mStepInvCol /= micron2cm;
159  mStepInvRow /= micron2cm;
160 
161  mDptMin *= micron2cm;
162  mDptMax *= micron2cm;
163  mStepInvDpt = (mNBinDpt - 1) / (mDptMax - mDptMin);
164  mDptMin -= 0.5 / mStepInvDpt;
165  mDptMax += 0.5 / mStepInvDpt;
166  mDptShift = 0.5 * (mDptMax + mDptMin);
167  print();
168 }

184 string AlpideSimResponse::composeDataName(int colBin, int rowBin)
185 {
186  /*
187  * compose the file-name to read data for bin colBin,rowBin
188  */
189 
190  // ugly but safe way to compose the file name
191  float vcol = colBin / mStepInvCol, vrow = rowBin / mStepInvRow;
192  size_t size = snprintf(nullptr, 0, mColRowDataFmt.data(), vcol, vrow) + 1;
193  unique_ptr<char[]> tmp(new char[size]);
194  snprintf(tmp.get(), size, mColRowDataFmt.data(), vcol, vrow);
195  return mDataPath + string(tmp.get(), tmp.get() + size - 1);
196 }

199 bool AlpideSimResponse::getResponse(float vRow, float vCol, float vDepth, AlpideRespSimMat& dest) const
200 {
201  /*
202  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
203  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)
204  */
205  if (!mNBinDpt) {
206  LOG(FATAL) << "response object is not initialized";
207  }
208  bool flipCol = false, flipRow = true;
209  if (vDepth < mDptMin || vDepth > mDptMax)
210  return false;
211  if (vCol < 0) {
212  vCol = -vCol;
213  flipCol = true;
214  }
215  if (vCol > mColMax)
216  return false;
217  if (vRow < 0) {
218  vRow = -vRow;
219  flipRow = false;
220  }
221  if (vRow > mRowMax)
222  return false;
223 
224  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
225  if (bin >= mData.size()) {
226  // this should not happen
227  LOG(FATAL) << "requested bin " << bin << "row/col/depth: " << getRowBin(vRow) << ":" << getColBin(vCol)
228  << ":" << getDepthBin(vDepth) << ")"
229  << ">= maxBin " << mData.size()
230  << " for X(row)=" << vRow << " Z(col)=" << vCol << " Y(depth)=" << vDepth;
231  }
232  // printf("bin %d %d %d\n",getColBin(vCol),getRowBin(vRow),getDepthBin(vDepth));
233  // return &mData[bin];
234  dest.adopt(mData[bin], flipRow, flipCol);
235  return true;
236 }

239 const AlpideRespSimMat* AlpideSimResponse::getResponse(float vRow, float vCol, float vDepth, bool& flipRow, bool& flipCol) const
240 {
241  /*
242  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
243  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)
244  */
245  if (!mNBinDpt) {
246  LOG(FATAL) << "response object is not initialized";
247  }
248  if (vDepth < mDptMin || vDepth > mDptMax)
249  return nullptr;
250  if (vCol < 0) {
251  vCol = -vCol;
252  flipCol = true;
253  } else {
254  flipCol = false;
255  }
256  if (vCol > mColMax)
257  return nullptr;
258  if (vRow < 0) {
259  vRow = -vRow;
260  flipRow = false;
261  } else {
262  flipRow = true;
263  }
264  if (vRow > mRowMax)
265  return nullptr;
266 
267  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
268  if (bin >= mData.size()) {
269  // this should not happen
270  LOG(FATAL) << "requested bin " << bin << "row/col/depth: " << getRowBin(vRow) << ":" << getColBin(vCol)
271  << ":" << getDepthBin(vDepth) << ")"
272  << ">= maxBin " << mData.size()
273  << " for X(row)=" << vRow << " Z(col)=" << vCol << " Y(depth)=" << vDepth;
274  }
275  return &mData[bin];
