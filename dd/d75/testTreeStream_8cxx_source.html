112 bool UnitTestSparse(Double_t scale, Int_t testEntries)
113 {
114  // Unit test for the TreeStreamRedirector
115  // 1.) Test TTreeRedirector
116  // a.) Fill tree with random vectors
117  // b.) Fill downscaled version of vectors
118  // c.) The same skipping first entry
119  // 2.) Check results wtitten to terminale
120  // a.) Disk consumption
121  // skip data should be scale time smaller than full
122  // zerro replaced ata should be compresed time smaller than full
123  // b.) Test invariants
124  // Input parameter scale => downscaling of sprse element
125 
126  std::string outFName("testTreeStreamSparse.root");
127  if (scale <= 0)
128  scale = 1;
129  if (scale > 1)
130  scale = 1;
131  TreeStreamRedirector* pcstream = new TreeStreamRedirector(outFName.data(), "recreate");
132  for (Int_t ientry = 0; ientry < testEntries; ientry++) {
133  TVectorD vecRandom(200);
134  TVectorD vecZerro(200); // zerro vector
135  for (Int_t j = 0; j < 200; j++)
136  vecRandom[j] = j + ientry + 0.1 * gRandom->Rndm();
137  Bool_t isSelected = (gRandom->Rndm() < scale);
138  TVectorD* pvecFull = &vecRandom;
139  TVectorD* pvecSparse = isSelected ? &vecRandom : nullptr;
140  TVectorD* pvecSparse0 = isSelected ? &vecRandom : nullptr;
141  TVectorD* pvecSparse1 = isSelected ? &vecRandom : &vecZerro;
142 
143  if (ientry == 0) {
144  pvecSparse0 = nullptr;
145  pvecSparse = &vecRandom;
146  }
147  (*pcstream) << "Full" << // stored all vectors
148  "ientry=" << ientry << "vec.=" << pvecFull << "\n";
149  (*pcstream) << "SparseSkip" << // fraction of vectors stored
150  "ientry=" << ientry << "vec.=" << pvecSparse << "\n";
151  (*pcstream) << "SparseSkip0" << // fraction with -pointer
152  "ientry=" << ientry << "vec.=" << pvecSparse0 << "\n";
153  (*pcstream) << "SparseZerro" << // all vectors filled, franction filled with 0
154  "ientry=" << ientry << "vec.=" << pvecSparse1 << "\n";
155  }
156  delete pcstream;
157  //
158  // 2.) check results
159  //
160 
161  TFile* f = TFile::Open(outFName.data());
162  if (!f) {
163  printf("Failed to open file: %s\n", outFName.data());
164  return false;
165  }
166  TTree* treeFull = (TTree*)f->Get("Full");
167  TTree* treeSparseSkip = (TTree*)f->Get("SparseSkip");
168  TTree* treeSparseSkip0 = (TTree*)f->Get("SparseSkip0");
169  TTree* treeSparseZerro = (TTree*)f->Get("SparseZerro");
170  // a.) data volume
171  //
172  Double_t ratio = (1. / scale) * treeSparseSkip->GetZipBytes() / Double_t(treeFull->GetZipBytes());
173  Double_t ratio0 = (1. / scale) * treeSparseSkip0->GetZipBytes() / Double_t(treeFull->GetZipBytes());
174  Double_t ratio1 = (1. / scale) * treeSparseZerro->GetZipBytes() / Double_t(treeFull->GetZipBytes());
175  printf("#UnitTest:\tTestSparse(%f)\tRatioSkip\t%f\n", scale, ratio);
176  printf("#UnitTest:\tTestSparse(%f)\tRatioSkip0\t%f\n", scale, ratio0);
177  printf("#UnitTest:\tTestSparse(%f)\tRatioZerro\t%f\n", scale, ratio1);
178  // b.) Integrity
179  Int_t outlyersSparseSkip = treeSparseSkip->Draw("1", "(vec.fElements-ientry-Iteration$-0.5)>0.5", "goff");
180  Int_t outlyersSparseSkip0 = treeSparseSkip0->Draw("1", "(vec.fElements-ientry-Iteration$-0.5)>0.5", "goff");
181  printf("#UnitTest:\tTestSparse(%f)\tOutlyersSkip\t%d\n", scale, outlyersSparseSkip != 0);
182  printf("#UnitTest:\tTestSparse(%f)\tOutlyersSkip0\t%d\n", scale, outlyersSparseSkip0 != 0);
183  // c.) Number of entries
184  //
185  Int_t entries = treeFull->GetEntries();
186  Int_t entries0 = treeSparseSkip0->GetEntries();
187  Bool_t isOKStat = (entries == entries0);
188  printf("#UnitTest:\tTestSparse(%f)\tEntries\t%d\n", scale, isOKStat);
189  //
190  // d.)Reading test
191  TVectorD* pvecRead = nullptr;
192  treeSparseSkip0->SetBranchAddress("vec.", &pvecRead);
193  Bool_t readOK = kTRUE;
194  for (Int_t ientry = 0; ientry < testEntries; ientry++) {
195  if (!pvecRead)
196  continue;
197  if (pvecRead->GetNrows() == 0)
198  continue;
199  if (TMath::Abs((*pvecRead)[0] - ientry) > 0.5)
200  readOK = kFALSE;
201  }
202  printf("#UnitTest:\tTestSparse(%f)\tReadOK\t%d\n", scale, readOK);
203  //
204  // e.)Global test
205  Bool_t isOK = (outlyersSparseSkip0 == 0) && isOKStat && readOK;
206  printf("#UnitTest:\tTestSparse(%f)\tisOk\t%d\n", scale, isOK);
207 
208  return isOK;
209 }
