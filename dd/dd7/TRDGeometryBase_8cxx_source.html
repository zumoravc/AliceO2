16 GPUd() int TRDGeometryBase::getStack(float z, int layer) const
17 {
18  //
19  // Reconstruct the chamber number from the z position and layer number
20  //
21  // The return function has to be protected for positiveness !!
22  //
23 
24  if ((layer < 0) || (layer >= kNlayer))
25  return -1;
26 
27  int istck = kNstack;
28  float zmin = 0.0;
29  float zmax = 0.0;
30 
31  do {
32  istck--;
33  if (istck < 0)
34  break;
35  const TRDPadPlane* pp = getPadPlane(layer, istck);
36  zmax = pp->getRow0();
37  int nrows = pp->getNrows();
38  zmin = zmax - 2 * pp->getLengthOPad() - (nrows - 2) * pp->getLengthIPad() - (nrows - 1) * pp->getRowSpacing();
39  } while ((z < zmin) || (z > zmax));
40 
41  return istck;
42 }

45 GPUd() bool TRDGeometryBase::isOnBoundary(int det, float y, float z, float eps) const
46 {
47  //
48  // Checks whether position is at the boundary of the sensitive volume
49  //
50 
51  int ly = getLayer(det);
52  if ((ly < 0) || (ly >= kNlayer))
53  return true;
54 
55  int stk = getStack(det);
56  if ((stk < 0) || (stk >= kNstack))
57  return true;
58 
59  const TRDPadPlane* pp = &mPadPlanes[getDetectorSec(ly, stk)];
60  if (!pp)
61  return true;
62 
63  float max = pp->getRow0();
64  int n = pp->getNrows();
65  float min = max - 2 * pp->getLengthOPad() - (n - 2) * pp->getLengthIPad() - (n - 1) * pp->getRowSpacing();
66  if (z < min + eps || z > max - eps) {
67  // printf("z : min[%7.2f (%7.2f)] %7.2f max[(%7.2f) %7.2f]\n", min, min+eps, z, max-eps, max);
68  return true;
69  }
70  min = pp->getCol0();
71  n = pp->getNcols();
72  max = min + 2 * pp->getWidthOPad() + (n - 2) * pp->getWidthIPad() + (n - 1) * pp->getColSpacing();
73  if (y < min + eps || y > max - eps) {
74  // printf("y : min[%7.2f (%7.2f)] %7.2f max[(%7.2f) %7.2f]\n", min, min+eps, y, max-eps, max);
75  return true;
76  }
77 
78  return false;
79 }
