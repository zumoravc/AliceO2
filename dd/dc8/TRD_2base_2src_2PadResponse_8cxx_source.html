120  // More sampling precision with linear interpolation
121  std::array<float, kPRFbin> pad{};
122  int sPRFbin = kPRFbin;
123  float sPRFwid = (mPRFhi - mPRFlo) / ((float)sPRFbin);
124  for (int iPad = 0; iPad < sPRFbin; iPad++) {
125  pad[iPad] = ((float)iPad + 0.5) * sPRFwid + mPRFlo;
126  }
127 
128  mPRFwid = (mPRFhi - mPRFlo) / ((float)mPRFbin);
129  mPRFpad = ((int)(1.0 / mPRFwid));
130 
131  int ipos1;
132  int ipos2;
133  float diff;
134 
135  for (int iLayer = 0; iLayer < kNlayer; ++iLayer) {
136  for (int iBin = 0; iBin < mPRFbin; ++iBin) {
137  float bin = (((float)iBin) + 0.5) * mPRFwid + mPRFlo;
138  ipos1 = ipos2 = 0;
139  diff = 0;
140  do {
141  diff = bin - pad[ipos2++];
142  } while ((diff > 0) && (ipos2 < kPRFbin));
143  if (ipos2 == kPRFbin) {
144  mPRFsmp[iLayer * mPRFbin + iBin] = prf[iLayer][ipos2 - 1];
145  } else if (ipos2 == 1) {
146  mPRFsmp[iLayer * mPRFbin + iBin] = prf[iLayer][ipos2 - 1];
147  } else {
148  --ipos2;
149  if (ipos2 >= kPRFbin)
150  ipos2 = kPRFbin - 1;
151  ipos1 = ipos2 - 1;
152  mPRFsmp[iLayer * mPRFbin + iBin] = prf[iLayer][ipos2] + diff * (prf[iLayer][ipos2] - prf[iLayer][ipos1]) / sPRFwid;
153  }
154  }

158 int PadResponse::getPRF(double signal, double dist, int layer, double* pad) const
159 {
160  //
161  // Applies the pad response
162  //
163  int iBin = ((int)((-dist - mPRFlo) / mPRFwid));
164  int iOff = layer * mPRFbin;
165 
166  int iBin0 = iBin - mPRFpad + iOff;
167  int iBin1 = iBin + iOff;
168  int iBin2 = iBin + mPRFpad + iOff;
169 
170  pad[0] = 0;
171  pad[1] = 0;
172  pad[2] = 0;
173 
174  if ((iBin1 >= 0) && (iBin1 < (mPRFbin * kNlayer))) {
175  if (iBin0 >= 0) {
176  pad[0] = signal * mPRFsmp[iBin0];
177  }
178  pad[1] = signal * mPRFsmp[iBin1];
179  if (iBin2 < (mPRFbin * kNlayer)) {
180  pad[2] = signal * mPRFsmp[iBin2];
181  }
182  return 1;
183  } else {
184  return 0;
185  }
