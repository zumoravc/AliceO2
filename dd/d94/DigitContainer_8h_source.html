90 inline DigitContainer::DigitContainer()
91 {
92  auto& detParam = ParameterDetector::Instance();
93  auto& gasParam = ParameterGas::Instance();
94  auto& eleParam = ParameterElectronics::Instance();
95  mTmaxTriggered = detParam.TmaxTriggered;
96 
97  // always have 50 % contingency for the size of the container depending on the input
98  mOffset = static_cast<TimeBin>(1.5 * detParam.TPClength / gasParam.DriftV / eleParam.ZbinWidth);
99  mTimeBins.resize(mOffset);
100 }
