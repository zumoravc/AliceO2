60 // Remove leading whitespace
61 std::string ltrimSpace(std::string src)
62 {
63  return src.erase(0, src.find_first_not_of(' '));
64 }
65 
66 // Remove trailing whitespace
67 std::string rtrimSpace(std::string src)
68 {
69  return src.erase(src.find_last_not_of(' ') + 1);
70 }

72 // Remove leading/trailing whitespace
73 std::string trimSpace(std::string const& src)
74 {
75  return ltrimSpace(rtrimSpace(src));
76 }

78 // Split a given string on a delim character, return vector of tokens
79 // If trim is true, then also remove leading/trailing whitespace of each token.
80 std::vector<std::string> splitString(const std::string& src, char delim, bool trim = false)
81 {
82  std::stringstream ss(src);
83  std::string token;
84  std::vector<std::string> tokens;
85 
86  while (std::getline(ss, token, delim)) {
87  token = (trim ? trimSpace(token) : token);
88  if (!token.empty()) {
89  tokens.push_back(std::move(token));
90  }
91  }
92 
93  return tokens;
94 }

107  }

175 // getIntValue takes a string value which is supposed to be
176 // a legal enum value and tries to cast it to an int.
177 // If it succeeds, and if the int value is legal, it is returned.
178 // If it fails, and if it is a legal string enum value, we look up
179 // and return the equivalent int value. In any case, if it is not
180 // a legal value we return -1 to indicate this fact.
181 int EnumLegalValues::getIntValue(const std::string& value) const
182 {
183  try {
184  int val = boost::lexical_cast<int>(value);
185  if (isLegal(val)) {
186  return val;
187  }
188  } catch (const boost::bad_lexical_cast& e) {
189  if (isLegal(value)) {
190  for (auto& pair : vvalues) {
191  if (pair.first == value) {
192  return pair.second;
193  }
194  }
195  }
196  }
197 
198  return -1;
199 }

393 // Update the storage map of params from the given configuration file.
394 // It can be in JSON or INI format.
395 void ConfigurableParam::updateFromFile(std::string const& configFile)
396 {
397  if (!sIsFullyInitialized) {
398  initialize();
399  }
400 
401  auto cfgfile = trimSpace(configFile);
402 
403  if (cfgfile.length() == 0) {
404  return;
405  }
406 
407  boost::property_tree::ptree pt = readConfigFile(cfgfile);
408 
409  std::vector<std::pair<std::string, std::string>> keyValPairs;
410 
411  try {
412  for (auto& section : pt) {
413  std::string mainKey = section.first;
414  for (auto& subKey : section.second) {
415  auto name = subKey.first;
416  auto value = subKey.second.get_value<std::string>();
417  std::string key = mainKey + "." + name;
418  std::pair<std::string, std::string> pair = std::make_pair(key, trimSpace(value));
419  keyValPairs.push_back(pair);
420  }
421  }
422  } catch (std::exception const& error) {
423  LOG(ERROR) << "Error while updating params " << error.what();
424  } catch (...) {
425  LOG(ERROR) << "Unknown while updating params ";
426  }
427 
428  try {
429  setValues(keyValPairs);
430  } catch (std::exception const& error) {
431  LOG(ERROR) << "Error while setting values " << error.what();
432  }
433 }

438 void ConfigurableParam::updateFromString(std::string const& configString)
439 {
440  if (!sIsFullyInitialized) {
441  initialize();
442  }
443 
444  auto cfgStr = trimSpace(configString);
445  if (cfgStr.length() == 0) {
446  return;
447  }
448 
449  // Take a vector of strings with elements of form a=b, and
450  // return a vector of pairs with each pair of form <a, b>
451  auto toKeyValPairs = [](std::vector<std::string>& tokens) {
452  std::vector<std::pair<std::string, std::string>> pairs;
453 
454  for (auto& token : tokens) {
455  auto keyval = splitString(token, '=');
456  if (keyval.size() != 2) {
457  LOG(FATAL) << "Illegal command-line key/value string: " << token;
458  continue;
459  }
460 
461  std::pair<std::string, std::string> pair = std::make_pair(keyval[0], trimSpace(keyval[1]));
462  pairs.push_back(pair);
463  }
464 
465  return pairs;
466  };
467 
468  // Simple check that the string starts/ends with an open square bracket
469  // Find the maximum index of a given key with array value.
470  // We store string keys for arrays as a[0]...a[size_of_array]
471  /*
472  auto maxIndex = [](std::string baseName) {
473  bool isFound = true;
474  int index = -1;
475  do {
476  index++;
477  std::string key = baseName + "[" + std::to_string(index) + "]";
478  isFound = keyInTree(sPtree, key);
479  } while (isFound);
480 
481  return index;
482  };
483 */
484 
485  // ---- end of helper functions --------------------
486 
487  // Command-line string is a ;-separated list of key=value params
488  auto params = splitString(configString, ';', true);
489 
490  // Now split each key=value string into its std::pair<key, value> parts
491  auto keyValues = toKeyValPairs(params);
492 
493  setValues(keyValues);
494 }

496 // setValues takes a vector of pairs where each pair is a key and value
497 // to be set in the storage map
498 void ConfigurableParam::setValues(std::vector<std::pair<std::string, std::string>> const& keyValues)
499 {
500  auto isArray = [](std::string& el) {
501  return el.size() > 0 && (el.at(0) == '[') && (el.at(el.size() - 1) == ']');
502  };
503 
504  // Take a vector of param key/value pairs
505  // and update the storage map for each of them by calling setValue.
506  // 1. For string/scalar types this is simple.
507  // 2. For array values we need to iterate over each array element
508  // and call setValue on the element, using an appropriately constructed key.
509  // 3. For enum types we check for the existence of the key in the enum registry
510  // and also confirm that the value is in the list of legal values
511  for (auto& keyValue : keyValues) {
512  std::string key = keyValue.first;
513  std::string value = trimSpace(keyValue.second);
514 
515  if (!keyInTree(sPtree, key)) {
516  LOG(FATAL) << "Inexistant ConfigurableParam key: " << key;
517  }
518 
519  if (sEnumRegistry->contains(key)) {
520  setEnumValue(key, value);
521  } else if (isArray(value)) {
522  setArrayValue(key, value);
523  } else {
524  assert(sKeyToStorageMap->find(key) != sKeyToStorageMap->end());
525 
526  // If the value is given as a boolean true|false, change to 1|0 int equivalent
527  if (value == "true") {
528  value = "1";
529  } else if (value == "false") {
530  value = "0";
531  }
532 
533  // TODO: this will trap complex types like maps and structs.
534  // These need to be broken into their own cases, so that we only
535  // get strings and scalars here.
536  setValue(key, value);
537  }
538  }
539 }

541 void ConfigurableParam::setArrayValue(const std::string& key, const std::string& value)
542 {
543  // We remove the lead/trailing square bracket
544  // value.erase(0, 1).pop_back();
545  auto elems = splitString(value.substr(1, value.length() - 2), ',', true);
546 
547  // TODO:
548  // 1. Should not assume each array element is a scalar/string. We may need to recurse.
549  // 2. Should not assume each array element - even if not complex - is correctly written. Validate.
550  // 3. Validation should include finding same types as in provided defaults.
551  for (int i = 0; i < elems.size(); ++i) {
552  std::string indexKey = key + "[" + std::to_string(i) + "]";
553  setValue(indexKey, elems[i]);
554  }

596 bool ConfigurableParam::updateThroughStorageMap(std::string mainkey, std::string subkey, std::type_info const& tinfo,

614  }

727 template <typename T>
728 bool ConvertAndCopy(std::string const& valuestring, void* targetaddr)
729 {

731  if (isMemblockDifferent<T>(targetaddr, (void*)&addr)) {

737 // special version for std::string
738 template <>
739 bool ConvertAndCopy<std::string>(std::string const& valuestring, void* targetaddr)
