97 template <typename T>
98 VectorHIP<T>::VectorHIP(const T* const source, const int size, const int initialSize) : mCapacity{size}, mIsWeak{false}
99 {
100  if (size > 0) {
101  try {
102 
103  Utils::HostHIP::gpuMalloc(reinterpret_cast<void**>(&mArrayPointer), size * sizeof(T));
104  Utils::HostHIP::gpuMalloc(reinterpret_cast<void**>(&mDeviceSize), sizeof(int));
105 
106  if (source != nullptr) {
107 
108  Utils::HostHIP::gpuMemcpyHostToDevice(mArrayPointer, source, size * sizeof(T));
109  Utils::HostHIP::gpuMemcpyHostToDevice(mDeviceSize, &size, sizeof(int));
110 
111  } else {
112 
113  Utils::HostHIP::gpuMemcpyHostToDevice(mDeviceSize, &initialSize, sizeof(int));
114  }
115 
116  } catch (...) {
117 
118  destroy();
119 
120  throw;
121  }
122  }
123 }

199 template <typename T>
200 void VectorHIP<T>::reset(const T* const source, const int size, const int initialSize)
201 {
202  if (size > mCapacity) {
203  if (mArrayPointer != nullptr) {
204  Utils::HostHIP::gpuFree(mArrayPointer);
205  }
206 
207  Utils::HostHIP::gpuMalloc(reinterpret_cast<void**>(&mArrayPointer), size * sizeof(T));
208  mCapacity = size;
209  }
210 
211  if (source != nullptr) {
212 
213  Utils::HostHIP::gpuMemcpyHostToDevice(mArrayPointer, source, size * sizeof(T));
214  Utils::HostHIP::gpuMemcpyHostToDevice(mDeviceSize, &size, sizeof(int));
215 
216  } else {
217  Utils::HostHIP::gpuMemcpyHostToDevice(mDeviceSize, &initialSize, sizeof(int));
218  }
219 }
