42 // small helper function to append to vector at arbitrary position
43 template <typename T, typename I>
44 void insertInVector(std::vector<T>& v, I index, T e)
45 {
46  auto currentsize = v.size();
47  if (index >= currentsize) {
48  const auto newsize = std::max(index + 1, (I)(1 + currentsize * 1.2));
49  v.resize(newsize, T(-1));
50  }
51  // new size must at least be as large as index
52  // assert(index < v.size());
53  v[index] = e;
54 }

149 void Stack::PushTrack(Int_t toBeDone, Int_t parentId, Int_t pdgCode, Double_t px, Double_t py, Double_t pz, Double_t e,
150  Double_t vx, Double_t vy, Double_t vz, Double_t time, Double_t polx, Double_t poly, Double_t polz,
151  TMCProcess proc, Int_t& ntr, Double_t weight, Int_t is, Int_t secondparentID)
152 {
153  // printf("Pushing %s toBeDone %5d parentId %5d pdgCode %5d is %5d entries %5d \n",
154  // proc == kPPrimary ? "Primary: " : "Secondary: ",
155  // toBeDone, parentId, pdgCode, is, mNumberOfEntriesInParticles);
156 
157  //
158  // This method is called
159  //
160  // - during serial simulation directly from the event generator
161  // - during parallel simulation to fill the stack of the primary generator device
162  // - in all cases to push a secondary particle
163  //
164  //
165  // Create new TParticle and add it to the TParticle array
166 
167  Int_t trackId = mNumberOfEntriesInParticles;
168  // Set track variable
169  ntr = trackId;
170  Int_t nPoints = 0;
171  Int_t daughter1Id = -1;
172  Int_t daughter2Id = -1;
173  TParticle p(pdgCode, trackId, parentId, nPoints, daughter1Id, daughter2Id, px, py, pz, e, vx, vy, vz, time);
174  p.SetPolarisation(polx, poly, polz);
175  p.SetWeight(weight);
176  p.SetUniqueID(proc); // using the unique ID to transfer process ID
177  mNumberOfEntriesInParticles++;
178 
179  insertInVector(mTrackIDtoParticlesEntry, trackId, (int)(mParticles.size()));
180 
181  // Push particle on the stack if toBeDone is set
182  if (proc == kPPrimary) {
183  // This is a particle from the primary particle generator
184  //
185  // SetBit is used to pass information about the primary particle to the stack during transport.
186  // Sime particles have already decayed or are partons from a shower. They are needed for the
187  // event history in the stack, but not for transport.
188  //
189  mIndexMap[trackId] = trackId;
190  p.SetBit(ParticleStatus::kKeep);
191  p.SetBit(ParticleStatus::kPrimary);
192  if (toBeDone == 1) {
193  p.SetBit(ParticleStatus::kToBeDone, 1);
194  mNumberOfPrimariesforTracking++;
195  } else {
196  p.SetBit(ParticleStatus::kToBeDone, 0);
197  }
198  mNumberOfPrimaryParticles++;
199  mPrimaryParticles.push_back(p);
200  mTracks->emplace_back(p);
201  } else {
202  p.SetBit(ParticleStatus::kPrimary, 0);
203  if (toBeDone == 1) {
204  p.SetBit(ParticleStatus::kToBeDone, 1);
205  } else {
206  p.SetBit(ParticleStatus::kToBeDone, 0);
207  }
208  mParticles.emplace_back(p);
209  mCurrentParticle = p;
210  }
211  mStack.push(p);
212 }

247 // calculates a hash based on particle properties
248 // hash may serve as seed for this track
249 ULong_t getHash(TParticle const& p)
250 {
251  auto asLong = [](double x) {
252  return (ULong_t) * (reinterpret_cast<ULong_t*>(&x));
253  };
254 
255  ULong_t hash;
256  o2::MCTrackT<double> track(p);
257 
258  hash = asLong(track.GetStartVertexCoordinatesX());
259  hash ^= asLong(track.GetStartVertexCoordinatesY());
260  hash ^= asLong(track.GetStartVertexCoordinatesZ());
261  hash ^= asLong(track.GetStartVertexCoordinatesT());
262  hash ^= asLong(track.GetStartVertexMomentumX());
263  hash ^= asLong(track.GetStartVertexMomentumY());
264  hash ^= asLong(track.GetStartVertexMomentumZ());
265  hash += (ULong_t)track.GetPdgCode();
266  return hash;
267 }

269 TParticle* Stack::PopNextTrack(Int_t& iTrack)
270 {
271 
272  // This functions is mainly used by Geant3?
273  Int_t nprod = (int)(mParticles.size());
274 
275  // If end of stack: Return empty pointer
276  if (mStack.empty()) {
277  iTrack = -1;
278  return nullptr;
279  }
280  Bool_t found = kFALSE;
281 
282  TParticle* nextParticle = nullptr;
283  while (!found && !mStack.empty()) {
284  // get next particle from stack
285  mCurrentParticle = mStack.top();
286  // remove particle from the top
287  mStack.pop();
288  // test if primary to be transported
289  if (mCurrentParticle.TestBit(ParticleStatus::kToBeDone)) {
290  if (mCurrentParticle.TestBit(ParticleStatus::kPrimary)) {
291  // particle is primary and needs to be tracked -> indicates that previous particle finished
292  mNumberOfPrimariesPopped++;
293  mIndexOfCurrentPrimary = mCurrentParticle.GetStatusCode();
294  } else {
295  //printf("........ it's a secondary \n");
296  }
297  mIndexOfCurrentTrack = mCurrentParticle.GetStatusCode();
298  iTrack = mIndexOfCurrentTrack;
299 
300  if (o2::conf::SimCutParams::Instance().trackSeed) {
301  auto hash = getHash(mCurrentParticle);
302  // LOG(INFO) << "SEEDING NEW TRACK USING HASH" << hash;
303  // init seed per track
304  gRandom->SetSeed(hash);
305 
306  // NOTE: THE BETTER PLACE WOULD BE IN PRETRACK HOOK BUT THIS DOES NOT SEEM TO WORK
307  // WORKS ONLY WITH G3 SINCE G4 DOES NOT CALL THIS FUNCTION
308  } // .trackSeed ?
309  nextParticle = &mCurrentParticle;
310  found = kTRUE;
311  } else {
312  iTrack = -1;
313  nextParticle = nullptr;
314  }
315  } // while
316  return nextParticle;
317 }

357 void Stack::FinishPrimary()
358 {
359  // Here transport of a primary and all its secondaries is finished
360  // we can do some cleanup of the memory structures
361  mPrimariesDone++;
362  LOG(DEBUG) << "Finish primary hook " << mPrimariesDone;
363  // preserve particles and theire ancestors that produced hits
364 
365  auto selected = selectTracks();
366 
367  // loop over current particle buffer
368  // - build index map indicesKept
369  // - update mother index information
370  int indexOld = 0;
371  int indexNew = 0;
372  int indexoffset = mTracks->size();
373  int neglected = 0;
374  std::vector<int> indicesKept((int)(mParticles.size()));
375  std::vector<MCTrack> tmpTracks;
376  Int_t ic = 0;
377 
378  // mTrackIDtoParticlesEntry
379  // trackID to mTrack -> index in mParticles
380  //
381  // indicesKept
382  // index in mParticles -> index in mTrack - highWaterMark
383  //
384  // mReorderIndices
385  // old (mTrack-highWaterMark) -> new (mTrack-highWaterMark)
386  //
387 
388  for (auto& particle : mParticles) {
389  if (particle.getStore() || !mPruneKinematics) {
390  // map the global track index to the new persistent index
391  auto imother = particle.getMotherTrackId();
392  // here mother is relative to the mParticles array
393  if (imother >= 0) {
394  // daughter of a secondary: obtain index from lookup table
395  imother = indicesKept[imother];
396  particle.SetMotherTrackId(imother);
397  }
398  // at this point we have the correct mother index in mParticles or
399  // a negative one which is a pointer to a primary
400  tmpTracks.emplace_back(particle);
401  indicesKept[indexOld] = indexNew;
402  indexNew++;
403  } else {
404  indicesKept[indexOld] = -1;
405  neglected++;
406  }
407  indexOld++;
408  mTracksDone++;
409  }
410  Int_t ntr = (int)(tmpTracks.size());
411  std::vector<int> reOrderedIndices(ntr);
412  std::vector<int> invreOrderedIndices(ntr);
413  for (Int_t i = 0; i < ntr; i++) {
414  invreOrderedIndices[i] = i;
415  reOrderedIndices[i] = i;
416  }
417 
418  if (mIsG4Like) {
419  ReorderKine(tmpTracks, reOrderedIndices);
420  for (Int_t i = 0; i < ntr; i++) {
421  Int_t index = reOrderedIndices[i];
422  invreOrderedIndices[index] = i;
423  }
424  }
425  for (Int_t i = 0; i < ntr; i++) {
426  Int_t index = reOrderedIndices[i];
427  auto& particle = tmpTracks[index];
428  Int_t imo = particle.getMotherTrackId();
429  Int_t imo0 = imo;
430  if (imo >= 0) {
431  imo = invreOrderedIndices[imo];
432  }
433  imo += indexoffset;
434  particle.SetMotherTrackId(imo);
435  mTracks->emplace_back(particle);
436  auto& mother = mTracks->at(imo);
437  if (mother.getFirstDaughterTrackId() == -1) {
438  mother.SetFirstDaughterTrackId((int)(mTracks->size()) - 1);
439  }
440  mother.SetLastDaughterTrackId((int)(mTracks->size()) - 1);
441  }
442 
443  //
444  // Update index map
445  //
446  Int_t imax = mNumberOfEntriesInParticles;
447  Int_t imin = imax - mParticles.size();
448  for (Int_t idTrack = imin; idTrack < imax; idTrack++) {
449  Int_t index1 = mTrackIDtoParticlesEntry[idTrack];
450  Int_t index2 = indicesKept[index1];
451  if (index2 == -1)
452  continue;
453  Int_t index3 = (mIsG4Like) ? invreOrderedIndices[index2] : index2;
454  mIndexMap[idTrack] = index3 + indexoffset;
455  }
456 
457  // we can now clear the particles buffer!
458  reOrderedIndices.clear();
459  invreOrderedIndices.clear();
460  mParticles.clear();
461  tmpTracks.clear();
462  mTransportedIDs.clear();
463  mTrackIDtoParticlesEntry.clear();
464  mIndexOfPrimaries.clear();
465 }

467 void Stack::UpdateTrackIndex(TRefArray* detList)
468 {
469  // we can avoid any updating in case no tracks have been filtered out
470  // check this like this
471  if (mIndexMap.size() == 0) {
472  LOG(INFO) << "No TrackIndex update necessary\n";
473  return;
474  }
475 
476  // we are getting the detectorlist from FairRoot as TRefArray
477  // at each call, but this list never changes so we cache it here
478  // as the right type to avoid repeated dynamic casts
479  if (mActiveDetectors.size() == 0) {
480  if (detList == nullptr) {
481  LOG(FATAL) << "No detList passed to Stack";
482  }
483  auto iter = detList->MakeIterator();
484  while (auto det = iter->Next()) {
485  auto o2det = dynamic_cast<o2::base::Detector*>(det);
486  if (o2det) {
487  mActiveDetectors.emplace_back(o2det);
488  } else {
489  LOG(INFO) << "Found nonconforming detector " << det->GetName();
490  }
491  }
492  }
493 
494  LOG(DEBUG) << "Stack::UpdateTrackIndex: Stack: Updating track indices...";
495  Int_t nColl = 0;
496 
497  // update track references
498  // use some caching since repeated trackIDs
499  for (auto& ref : *mTrackRefs) {
500  const auto id = ref.getTrackID();
501  auto iter = mIndexMap.find(id);
502  if (iter == mIndexMap.end()) {
503  LOG(INFO) << "Invalid trackref ... needs to be rmoved \n";
504  ref.setTrackID(-1);
505  } else {
506  ref.setTrackID(iter->second);
507  }
508  }
509 
510  // sort trackrefs according to new track index
511  // then according to track length
512  std::sort(mTrackRefs->begin(), mTrackRefs->end(), [](const o2::TrackReference& a, const o2::TrackReference& b) {
513  if (a.getTrackID() == b.getTrackID()) {
514  return a.getLength() < b.getLength();
515  }
516  return a.getTrackID() < b.getTrackID();
517  });
518 
519  // make final indexed container for track references
520  // fill empty
521  for (auto& ref : *mTrackRefs) {
522  if (ref.getTrackID() >= 0) {
523  mIndexedTrackRefs->addElement(ref.getTrackID(), ref);
524  }
525  }
526 
527  for (auto det : mActiveDetectors) {
528  // update the track indices by delegating to specialized detector functions
529  det->updateHitTrackIndices(mIndexMap);
530  } // List of active detectors
531 
532  LOG(DEBUG) << "Stack::UpdateTrackIndex: ...stack and " << nColl << " collections updated.";
533 }

616 bool Stack::selectTracks()
617 {
618  bool tracksdiscarded = false;
619  // Check particles in the fParticle array
620  int prim = -1; // counter how many primaries seen (mainly to constrain search in motherindex remapping)
621  LOG(DEBUG) << "Stack: Entering track selection on " << mParticles.size() << " tracks";
622  for (auto& thisPart : mParticles) {
623  Bool_t store = kTRUE;
624  // Get track parameters
625  Bool_t isPrimary = (thisPart.getProcess() == 0);
626  Int_t iMother = thisPart.getMotherTrackId();
627  auto& mother = mParticles[iMother];
628  Bool_t motherIsPrimary = (iMother < mNumberOfPrimaryParticles);
629 
630  if (isPrimary) {
631  prim++;
632  // for primaries we are done quickly
633  // in fact the primary has already been stored
634  // so this should not happen
635  store = kTRUE;
636  } else {
637  // for other particles we potentially need to correct the mother indices
638  if (!motherIsPrimary) {
639  // the mapping is from the index in the stack to the index in mParticles
640  thisPart.SetMotherTrackId(mTrackIDtoParticlesEntry[iMother]);
641  } else {
642  // for a secondary which is a direct decendant of a primary use a negative index which will be restored later
643  thisPart.SetMotherTrackId(iMother - (int)(mTracks->size()));
644  }
645  // no secondaries; also done
646  if (!mStoreSecondaries) {
647  store = kFALSE;
648  tracksdiscarded = true;
649  } else {
650  // Calculate number of hits created by this track
651  // Note: we only distinguish between no hit and more than 0 hits
652  int nHits = thisPart.hasHits();
653  // Check for cuts (store primaries in any case)
654  if (nHits < mMinHits) {
655  store = kFALSE;
656  tracksdiscarded = true;
657  }
658  // only if we have non-trival energy cut
659  if (mEnergyCut > 0.) {
660  Double_t energy = thisPart.GetEnergy();
661  Double_t mass = thisPart.GetMass();
662  Double_t eKin = energy - mass;
663  if (eKin < mEnergyCut) {
664  store = kFALSE;
665  tracksdiscarded = true;
666  }
667  }
668  if (keepPhysics(thisPart)) {
669  store = kTRUE;
670  tracksdiscarded = false;
671  }
672  }
673  }
674  thisPart.setStore(store);
675  }
676 
677  // If flag is set, flag recursively mothers of selected tracks
678  //
679  if (mStoreMothers) {
680  for (auto& particle : mParticles) {
681  if (particle.getStore()) {
682  Int_t iMother = particle.getMotherTrackId();
683  while (iMother >= 0) {
684  auto& mother = mParticles[iMother];
685  mother.setStore(true);
686  iMother = mother.getMotherTrackId();
687  } // while mother
688  } // store ?
689  } // particle loop
690  }
691 
692  return !tracksdiscarded;
693 }

795 void Stack::ReorderKine(std::vector<MCTrack>& particles, std::vector<int>& reOrderedIndices)
796 {
797  //
798  // Particles are ordered in a way that descendants of a particle appear next to each other.
799  // This has the advantage that their position in the stack can be identified by two number,
800  // for example the index of the first and last descentant.
801  // The result of the ordering is returned via the look-up table reOrderedIndices
802  //
803 
804  Int_t ntr = (int)(particles.size());
805  std::vector<bool> done(ntr, false);
806 
807  int indexoffset = mTracks->size();
808  Int_t index = 0;
809  Int_t imoOld = 0;
810  for (Int_t i = 0; i < ntr; i++)
811  reOrderedIndices[i] = i;
812 
813  for (Int_t i = -1; i < ntr; i++) {
814  if (i != -1) {
815  // secondaries
816  if (!done[i]) {
817  reOrderedIndices[index] = i;
818  index++;
819  done[i] = true;
820  }
821  imoOld = i;
822  } else {
823  // current primary
824  imoOld = mIndexOfCurrentPrimary - indexoffset;
825  }
826  for (Int_t j = i + 1; j < ntr; j++) {
827  if (!done[j]) {
828  if ((particles[j]).getMotherTrackId() == imoOld) {
829  reOrderedIndices[index] = j;
830  index++;
831  done[j] = true;
832  } // child found
833  } // done
834  } // j
835  } // i
836 }
