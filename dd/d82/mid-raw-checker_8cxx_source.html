63  if (vm.count("help")) {
64  std::cout << "Usage: " << argv[0] << " <input_raw_filename> [input_raw_filename_1 ...]\n";
65  std::cout << generic << std::endl;
66  return 2;
67  }
68  if (vm.count("input") == 0) {
69  std::cout << "no input file specified" << std::endl;
70  return 1;
71  }

73  std::vector<std::string> inputfiles{vm["input"].as<std::vector<std::string>>()};
74 
75  o2::mid::CRUBareDecoder decoder;
76  decoder.init(true);
77 
78  o2::mid::CRUBareDataChecker checker;
79 
80  unsigned long int iHB = 0;
81 
82  for (auto& filename : inputfiles) {
83  o2::mid::RawFileReader<uint8_t> rawFileReader;
84  std::string outFilename = filename;
85  auto pos = outFilename.find_last_of("/");
86  if (pos == std::string::npos) {
87  pos = 0;
88  } else {
89  ++pos;
90  }
91  outFilename.insert(pos, "check_");
92  outFilename += ".txt";
93  std::ofstream outFile(outFilename.c_str());
94  if (!rawFileReader.init(filename.c_str())) {
95  return 2;
96  }
97  if (ignoreRDH) {
98  rawFileReader.setCustomRDH(buildCustomRDH());
99  }
100  std::vector<o2::mid::LocalBoardRO> data;
101  std::vector<o2::mid::ROFRecord> rofRecords;
102  std::stringstream ss;
103  bool isFirst = true;
104  while (rawFileReader.getState() == 0) {
105  rawFileReader.readHB(onlyClosedHBs);
106  decoder.process(rawFileReader.getData());
107  rawFileReader.clear();
108  size_t offset = data.size();
109  std::copy(decoder.getData().begin(), decoder.getData().end(), std::back_inserter(data));
110  for (auto& rof : decoder.getROFRecords()) {
111  rofRecords.emplace_back(rof.interactionRecord, rof.eventType, rof.firstEntry + offset, rof.nEntries);
112  }
113  ss << " HB: " << iHB << " (line: " << 512 * iHB + 1 << ")";
114  if (decoder.isComplete()) {
115  // The check assumes that we have all data corresponding to one event.
116  // However this might not be true since we read one HB at the time.
117  // So we must test that the event was fully read before running the check.
118  if (!checker.process(data, rofRecords, isFirst)) {
119  outFile << ss.str() << "\n";
120  outFile << checker.getDebugMessage() << "\n";
121  }
122  isFirst = false;
123  std::stringstream clean;
124  ss.swap(clean);
125  data.clear();
126  rofRecords.clear();
127  }
128  ++iHB;
129  if (nHBs > 0 && iHB >= nHBs) {
130  break;
131  }
132  }
133  outFile << "Fraction of faulty events: " << checker.getNBCsFaulty() << " / " << checker.getNBCsProcessed() << " = " << static_cast<double>(checker.getNBCsFaulty()) / static_cast<double>(checker.getNBCsProcessed());
134 
135  outFile.close();
136  }
137 
138  return 0;
139 }
