35 int GPUChainTracking::DoProfile()
36 {
37 #ifdef GPUCA_TRACKLET_CONSTRUCTOR_DO_PROFILE
38  std::unique_ptr<char[]> stageAtSync{new char[PROFILE_MAX_SIZE]};
39  mRec->GPUMemCpy(stageAtSync.get(), processorsShadow()->tpcTrackers[0].mStageAtSync, PROFILE_MAX_SIZE, -1, false);
40 
41  FILE* fp = fopen("profile.txt", "w+");
42  FILE* fp2 = fopen("profile.bmp", "w+b");
43 
44  const int bmpheight = 8192;
45  BITMAPFILEHEADER bmpFH;
46  BITMAPINFOHEADER bmpIH;
47  memset(&bmpFH, 0, sizeof(bmpFH));
48  memset(&bmpIH, 0, sizeof(bmpIH));
49 
50  bmpFH.bfType = 19778; //"BM"
51  bmpFH.bfSize = sizeof(bmpFH) + sizeof(bmpIH) + (ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1) * bmpheight;
52  bmpFH.bfOffBits = sizeof(bmpFH) + sizeof(bmpIH);
53 
54  bmpIH.biSize = sizeof(bmpIH);
55  bmpIH.biWidth = ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1;
56  bmpIH.biHeight = bmpheight;
57  bmpIH.biPlanes = 1;
58  bmpIH.biBitCount = 32;
59 
60  fwrite(&bmpFH, 1, sizeof(bmpFH), fp2);
61  fwrite(&bmpIH, 1, sizeof(bmpIH), fp2);
62 
63  int nEmptySync = 0;
64  for (unsigned int i = 0; i < bmpheight * ConstructorBlockCount() * ConstructorThreadCount(); i += ConstructorBlockCount() * ConstructorThreadCount()) {
65  int fEmpty = 1;
66  for (unsigned int j = 0; j < ConstructorBlockCount() * ConstructorThreadCount(); j++) {
67  fprintf(fp, "%d\t", stageAtSync[i + j]);
68  int color = 0;
69  if (stageAtSync[i + j] == 1) {
70  color = RGB(255, 0, 0);
71  }
72  if (stageAtSync[i + j] == 2) {
73  color = RGB(0, 255, 0);
74  }
75  if (stageAtSync[i + j] == 3) {
76  color = RGB(0, 0, 255);
77  }
78  if (stageAtSync[i + j] == 4) {
79  color = RGB(255, 255, 0);
80  }
81  fwrite(&color, 1, sizeof(int), fp2);
82  if (j > 0 && j % 32 == 0) {
83  color = RGB(255, 255, 255);
84  fwrite(&color, 1, 4, fp2);
85  }
86  if (stageAtSync[i + j]) {
87  fEmpty = 0;
88  }
89  }
90  fprintf(fp, "\n");
91  if (fEmpty) {
92  nEmptySync++;
93  } else {
94  nEmptySync = 0;
95  }
96  (void)nEmptySync;
97  // if (nEmptySync == GPUCA_SCHED_ROW_STEP + 2) break;
98  }
99 
100  fclose(fp);
101  fclose(fp2);
102 #endif
103  return 0;
104 }

106 void GPUChainTracking::PrintMemoryStatistics()
107 {
108  unsigned int nStartHits = 0, nMaxStartHits = 0;
109  unsigned int nTracklets = 0, nMaxTracklets = 0;
110  unsigned int nTrackletHits = 0, nMaxTrackletHits = 0;
111  unsigned int nSectorTracks = 0, nMaxSectorTracks = 0;
112  unsigned int nSectorTrackHits = 0, nMaxSectorTrackHits = 0;
113  float maxUsageHits = 0.f, maxUsageTracklets = 0.f, maxUsageTrackletHits = 0.f, maxUsageTracks = 0.f, maxUsageTrackHits = 0.f;
114  for (int i = 0; i < NSLICES; i++) {
115  nMaxStartHits += processors()->tpcTrackers[i].NMaxStartHits();
116  nStartHits += *processors()->tpcTrackers[i].NStartHits();
117  maxUsageHits = CAMath::Max(maxUsageHits, 100.f * *processors()->tpcTrackers[i].NStartHits() / std::max(1u, processors()->tpcTrackers[i].NMaxStartHits()));
118  nMaxTracklets += processors()->tpcTrackers[i].NMaxTracklets();
119  nTracklets += *processors()->tpcTrackers[i].NTracklets();
120  maxUsageTracklets = CAMath::Max(maxUsageTracklets, 100.f * *processors()->tpcTrackers[i].NTracklets() / std::max(1u, processors()->tpcTrackers[i].NMaxTracklets()));
121  nMaxTrackletHits += processors()->tpcTrackers[i].NMaxRowHits();
122  nTrackletHits += *processors()->tpcTrackers[i].NRowHits();
123  maxUsageTrackletHits = CAMath::Max(maxUsageTrackletHits, 100.f * *processors()->tpcTrackers[i].NRowHits() / std::max(1u, processors()->tpcTrackers[i].NMaxRowHits()));
124  nMaxSectorTracks += processors()->tpcTrackers[i].NMaxTracks();
125  nSectorTracks += *processors()->tpcTrackers[i].NTracks();
126  maxUsageTracks = CAMath::Max(maxUsageTracks, 100.f * *processors()->tpcTrackers[i].NTracks() / std::max(1u, processors()->tpcTrackers[i].NMaxTracks()));
127  nMaxSectorTrackHits += processors()->tpcTrackers[i].NMaxTrackHits();
128  nSectorTrackHits += *processors()->tpcTrackers[i].NTrackHits();
129  maxUsageTrackHits = CAMath::Max(maxUsageTrackHits, 100.f * *processors()->tpcTrackers[i].NTrackHits() / std::max(1u, processors()->tpcTrackers[i].NMaxTrackHits()));
130  }
131  unsigned int nTracks = processors()->tpcMerger.NOutputTracks();
132  unsigned int nMaxTracks = processors()->tpcMerger.NMaxTracks();
133  unsigned int nTrackHits = processors()->tpcMerger.NOutputTrackClusters();
134  unsigned int nMaxTrackHits = processors()->tpcMerger.NMaxOutputTrackClusters();
135 
136  GPUInfo("Mem Usage Start Hits : %8u / %8u (%3.0f%% / %3.0f%%)", nStartHits, nMaxStartHits, 100.f * nStartHits / std::max(1u, nMaxStartHits), maxUsageHits);
137  GPUInfo("Mem Usage Tracklets : %8u / %8u (%3.0f%% / %3.0f%%)", nTracklets, nMaxTracklets, 100.f * nTracklets / std::max(1u, nMaxTracklets), maxUsageTracklets);
138  GPUInfo("Mem Usage TrackletHits : %8u / %8u (%3.0f%% / %3.0f%%)", nTrackletHits, nMaxTrackletHits, 100.f * nTrackletHits / std::max(1u, nMaxTrackletHits), maxUsageTrackletHits);
139  GPUInfo("Mem Usage SectorTracks : %8u / %8u (%3.0f%% / %3.0f%%)", nSectorTracks, nMaxSectorTracks, 100.f * nSectorTracks / std::max(1u, nMaxSectorTracks), maxUsageTracks);
140  GPUInfo("Mem Usage SectorTrackHits: %8u / %8u (%3.0f%% / %3.0f%%)", nSectorTrackHits, nMaxSectorTrackHits, 100.f * nSectorTrackHits / std::max(1u, nMaxSectorTrackHits), maxUsageTrackHits);
141  GPUInfo("Mem Usage Tracks : %8u / %8u (%3.0f%%)", nTracks, nMaxTracks, 100.f * nTracks / nMaxTracks);
142  GPUInfo("Mem Usage TrackHits : %8u / %8u (%3.0f%%)", nTrackHits, nMaxTrackHits, 100.f * nTrackHits / nMaxTrackHits);
143 }
