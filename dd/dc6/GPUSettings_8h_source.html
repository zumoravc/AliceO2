43 // Settings concerning the reconstruction
44 struct GPUSettingsRec {
45 #ifndef GPUCA_GPUCODE
46  GPUSettingsRec()
47  {
48  SetDefaults();
49  }
50  void SetDefaults();
51  void SetMinTrackPt(float v) { MaxTrackQPt = CAMath::Abs(v) > 0.001 ? 1. / CAMath::Abs(v) : 1. / 0.001; }
52 #endif
53 
54  // There must be no bool in here, use char, as sizeof(bool) is compiler dependent and fails on GPUs!!!!!!
55  float HitPickUpFactor; // multiplier for the chi2 window for hit pick up procedure
56  float NeighboursSearchArea; // area in cm for the search of neighbours
57  float ClusterError2CorrectionY; // correction for the squared cluster error during tracking
58  float ClusterError2CorrectionZ; // correction for the squared cluster error during tracking
59  int MinNTrackClusters; //* required min number of clusters on the track
60  float MaxTrackQPt; //* required max Q/Pt (==min Pt) of tracks
61  char NWays; // Do N fit passes in final fit of merger
62  char NWaysOuter; // Store outer param
63  char RejectMode; // 0: no limit on rejection or missed hits, >0: break after n rejected hits, <0: reject at max -n hits
64  char GlobalTracking; // Enable Global Tracking (prolong tracks to adjacent sectors to find short segments)
65  float SearchWindowDZDR; // Use DZDR window for seeding instead of vertex window
66  float TrackReferenceX; // Transport all tracks to this X after tracking (disabled if > 500)
67  char NonConsecutiveIDs; // Non-consecutive cluster IDs as in HLT, disables features that need access to slice data in TPC merger
68  char ForceEarlyTPCTransform; // Force early TPC transformation also for continuous data (-1 = auto)
69  unsigned char DisableRefitAttachment; // Bitmask to disable cluster attachment steps in refit: 1: attachment, 2: propagation, 4: loop following, 8: mirroring
70  unsigned char dEdxTruncLow; // Low truncation threshold, fraction of 128
71  unsigned char dEdxTruncHigh; // High truncation threshold, fraction of 128
72  unsigned char tpcRejectionMode; // 0: do not reject clusters, 1: do reject identified junk, 2: reject everything but good tracks
73  float tpcRejectQPt; // Reject tracks below this Pt
74  unsigned char tpcCompressionModes; // Enabled steps of TPC compression as flags: 1=truncate charge/width LSB, 2=differences, 4=track-model
75  unsigned char tpcCompressionSortOrder; // Sort order for clusters storred as differences (0 = time, 1 = pad, 2 = Z-curve-time-pad, 3 = Z-curve-pad-time)
76  unsigned char tpcSigBitsCharge; // Number of significant bits for TPC cluster charge in compression mode 1
77  unsigned char tpcSigBitsWidth; // Number of significant bits for TPC cluster width in compression mode 1
78  unsigned char tpcZSthreshold; // TPC Zero Suppression Threshold (for loading digits / forwarging digits as clusters)
79  unsigned char fwdTPCDigitsAsClusters; // Simply forward TPC digits as clusters
80  unsigned char bz0Pt; // Nominal Pt to set when bz = 0 (in 10 MeV)
81  unsigned char dropLoopers; // Drop all clusters after starting from the second loop from tracks
82  unsigned char mergerCovSource; // 0 = simpleFilterErrors, 1 = use from track following
83  unsigned char mergerInterpolateErrors; // Use interpolation for cluster rejection based on chi-2 instead of extrapolation
84  char fitInProjections; // -1 for automatic
85  char fitPropagateBzOnly; // Use only Bz for the propagation during the fit in the first n passes, -1 = NWays -1
86  char retryRefit; // Retry refit with larger cluster errors when fit fails
87  char loopInterpolationInExtraPass; // Perform the loop interpolation in an extra pass
88  char mergerReadFromTrackerDirectly; // Make the TPC merger read the output directly from the tracker class
89  char useMatLUT; // Use material lookup table for TPC refit
90 };

135  int nThreads; // Numnber of threads on CPU, 0 = auto-detect
136  bool ompKernels; // OMP Parallelization inside kernels
137  int deviceNum; // Device number to use, in case the backend provides multiple devices (-1 = auto-select)
138  int platformNum; // Platform to use, in case the backend provides multiple platforms (-1 = auto-select)
139  bool globalInitMutex; // Global mutex to synchronize initialization over multiple instances
140  bool gpuDeviceOnly; // Use only GPU as device (i.e. no CPU for OpenCL)
141  int nDeviceHelperThreads; // Additional CPU helper-threads for CPU parts of processing with accelerator
142  int debugLevel; // Level of debug output (-1 = silent)
143  int debugMask; // Mask for debug output dumps to file
144  bool comparableDebutOutput; // Make CPU and GPU debug output comparable (sort / skip concurrent parts)
145  int resetTimers; // Reset timers every event
146  GPUDisplayBackend* eventDisplay; // Run event display after processing, ptr to backend
147  bool runQA; // Run QA after processing
148  bool runCompressionStatistics; // Run statistics and verification for cluster compression
149  int stuckProtection; // Timeout in us, When AMD GPU is stuck, just continue processing and skip tracking, do not crash or stall the chain
150  int memoryAllocationStrategy; // 0 = auto, 1 = new/delete per resource (default for CPU), 2 = big chunk single allocation (default for device)
151  bool keepAllMemory; // Allocate all memory on both device and host, and do not reuse
152  bool keepDisplayMemory; // Like keepAllMemory, but only for memory required for event display
153  int nStreams; // Number of parallel GPU streams
154  char trackletConstructorInPipeline; // Run tracklet constructor in pileline like the preceeding tasks instead of as one big block
155  char trackletSelectorInPipeline; // Run tracklet selector in pipeline, requres also tracklet constructor in pipeline
156  char trackletSelectorSlices; // Number of slices to processes in parallel at max
157  size_t forceMemoryPoolSize; // Override size of memory pool to be allocated on GPU / Host (set =1 to force allocating all device memory, if supported)
158  int nTPCClustererLanes; // Number of TPC clusterers that can run in parallel
159  bool deviceTimers; // Use device timers instead of host-based timers
160  bool registerStandaloneInputMemory; // Automatically register memory for the GPU which is used as input for the standalone benchmark
161  int tpcCompressionGatherMode; // Modes: 0 = gather by DMA, 1 = DMA + gather on host, 2 = gather by kernel
162  char mergerSortTracks; // Sort track indices for GPU track fit
163  bool runMC; // Process MC labels
164  float memoryScalingFactor; // Factor to apply to all memory scalers
165  bool fitSlowTracksInOtherPass; // Do a second pass on tracks that are supposed to take long, an attempt to reduce divergence on the GPU
166  bool fullMergerOnGPU; // Perform full TPC track merging on GPU instead of only refit
167  char alternateBorderSort; // Alternative scheduling for sorting of border tracks
168  bool delayedOutput; // Delay output to be parallel to track fit
169  bool tpccfGatherKernel; // Use a kernel instead of the DMA engine to gather the clusters
170  bool prefetchTPCpageScan; // Prefetch headers during TPC page scan
171 };
