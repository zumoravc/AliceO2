63 template <class Mapping>
64 int RawPixelDecoder<Mapping>::decodeNextTrigger()
65 {
66  mTimerDecode.Start(false);
67  mNChipsFiredROF = 0;
68  mNPixelsFiredROF = 0;
69  mInteractionRecord.clear();
70  int nLinksWithData = 0, nru = mRUDecodeVec.size();
71 #ifdef WITH_OPENMP
72  if (mNThreads > 0) { // otherwhise, rely on default
73  omp_set_num_threads(mNThreads);
74  } else {
75  mNThreads = omp_get_num_threads();
76  LOG(INFO) << mSelfName << " will use " << mNThreads << " threads";
77  }
78 #pragma omp parallel for schedule(dynamic) reduction(+ \
79  : nLinksWithData, mNChipsFiredROF, mNPixelsFiredROF)
80 #endif
81  for (int iru = 0; iru < nru; iru++) {
82  nLinksWithData += decodeNextTrigger(iru);
83  mNChipsFiredROF += mRUDecodeVec[iru].nChipsFired;
84  int npix = 0;
85  for (int ic = mRUDecodeVec[iru].nChipsFired; ic--;) {
86  npix += mRUDecodeVec[iru].chipsData[ic].getData().size();
87  }
88  mNPixelsFiredROF += npix;
89  }
90 
91  if (nLinksWithData) { // fill some statistics
92  mROFCounter++;
93  mNChipsFired += mNChipsFiredROF;
94  mNPixelsFired += mNPixelsFiredROF;
95  mCurRUDecodeID = 0; // getNextChipData will start from here
96  // set IR and trigger from the 1st non empty link
97  for (const auto& link : mGBTLinks) {
98  if (link.status == GBTLink::DataSeen) {
99  mInteractionRecord = link.ir;
100  mInteractionRecordHB = o2::raw::RDHUtils::getHeartBeatIR(*link.lastRDH);
101  mTrigger = link.trigger;
102  break;
103  }
104  }
105  }
106  mTimerDecode.Stop();
107  // LOG(INFO) << "Chips Fired: " << mNChipsFiredROF << " NPixels: " << mNPixelsFiredROF << " at IR " << mInteractionRecord << " of HBF " << mInteractionRecordHB;
108  return nLinksWithData;
109 }

130 template <class Mapping>
131 int RawPixelDecoder<Mapping>::decodeNextTrigger(int iru)
132 {
133  auto& ru = mRUDecodeVec[iru];
134  ru.clear();
135  int ndec = 0; // number of yet non-empty links
136  for (int il = 0; il < RUDecodeData::MaxLinksPerRU; il++) {
137  auto* link = getGBTLink(ru.links[il]);
138  if (link) {
139  auto res = link->collectROFCableData(mMAP);
140  if (res == GBTLink::DataSeen) { // at the moment process only DataSeen
141  ndec++;
142  }
143  }
144  }
145  if (ndec) {
146  ru.decodeROF(mMAP);
147  }
148  return ndec;
149 }

153 template <class Mapping>
154 void RawPixelDecoder<Mapping>::setupLinks(InputRecord& inputs)
155 {
156  mCurRUDecodeID = NORUDECODED;
157  auto nLinks = mGBTLinks.size();
158  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{mMAP.getOrigin(), "RAWDATA"}, Lifetime::Timeframe}};
159  DPLRawParser parser(inputs, filter);
160  uint32_t currSSpec = 0xffffffff; // dummy starting subspec
161  int linksAdded = 0;
162  for (auto it = parser.begin(); it != parser.end(); ++it) {
163  auto const* dh = it.o2DataHeader();
164  auto& lnkref = mSubsSpec2LinkID[dh->subSpecification];
165  const auto& rdh = *reinterpret_cast<const header::RDHAny*>(it.raw()); // RSTODO this is a hack in absence of generic header getter
166 
167  if (lnkref.entry == -1) { // new link needs to be added
168  lnkref.entry = int(mGBTLinks.size());
169  auto& lnk = mGBTLinks.emplace_back(RDHUtils::getCRUID(rdh), RDHUtils::getFEEID(rdh), RDHUtils::getEndPointID(rdh), RDHUtils::getLinkID(rdh), lnkref.entry);
170  getCreateRUDecode(mMAP.FEEId2RUSW(RDHUtils::getFEEID(rdh))); // make sure there is a RU for this link
171  lnk.verbosity = GBTLink::Verbosity(mVerbosity);
172  lnk.format = mFormat;
173  LOG(INFO) << mSelfName << " registered new link " << lnk.describe() << " RUSW=" << int(mMAP.FEEId2RUSW(lnk.feeID));
174  linksAdded++;
175  }
176  auto& link = mGBTLinks[lnkref.entry];
177  if (currSSpec != dh->subSpecification) { // this is the 1st part for this link in this TF, next parts must follow contiguously!!!
178  link.clear(false, true); // clear link data except statistics
179  currSSpec = dh->subSpecification;
180  }
181  link.cacheData(it.raw(), RDHUtils::getMemorySize(rdh));
182  }
183 
184  if (linksAdded) { // new links were added, update link<->RU mapping, usually is done for 1st TF only
185  if (nLinks) {
186  LOG(WARNING) << mSelfName << " New links appeared although the initialization was already done";
187  for (auto& ru : mRUDecodeVec) { // reset RU->link references since they may have been changed
188  memset(&ru.links[0], -1, RUDecodeData::MaxLinksPerRU * sizeof(int));
189  }
190  }
191  // sort RUs in stave increasing order
192  std::sort(mRUDecodeVec.begin(), mRUDecodeVec.end(), [](const RUDecodeData& ruA, const RUDecodeData& ruB) -> bool { return ruA.ruSWID < ruB.ruSWID; });
193  for (auto i = 0; i < mRUDecodeVec.size(); i++) {
194  mRUEntry[mRUDecodeVec[i].ruSWID] = i;
195  }
196  nLinks = mGBTLinks.size();
197  // attach link to corresponding RU: this can be done once all RUs are created, to make sure their pointers don't change
198  for (int il = 0; il < nLinks; il++) {
199  auto& link = mGBTLinks[il];
200  bool newLinkAdded = (link.ruPtr == nullptr);
201  link.ruPtr = getRUDecode(mMAP.FEEId2RUSW(link.feeID)); // link to RU reference, reattach even it was already set before
202  uint16_t lr, ruOnLr, linkInRU;
203  mMAP.expandFEEId(link.feeID, lr, ruOnLr, linkInRU);
204  if (newLinkAdded) {
205  LOG(INFO) << mSelfName << " Attaching " << link.describe() << " to RU#" << int(mMAP.FEEId2RUSW(link.feeID))
206  << " (stave " << ruOnLr << " of layer " << lr << ')';
207  }
208  link.idInRU = linkInRU;
209  link.ruPtr->links[linkInRU] = il; // RU to link reference
210  }
211  }
212 }

216 template <class Mapping>
217 RUDecodeData& RawPixelDecoder<Mapping>::getCreateRUDecode(int ruSW)
218 {
219  assert(ruSW < mMAP.getNRUs());
220  if (mRUEntry[ruSW] < 0) {
221  mRUEntry[ruSW] = mRUDecodeVec.size();
222  auto& ru = mRUDecodeVec.emplace_back();
223  ru.ruSWID = ruSW;
224  ru.ruInfo = mMAP.getRUInfoSW(ruSW); // info on the stave/RU
225  ru.chipsData.resize(mMAP.getNChipsOnRUType(ru.ruInfo->ruType));
226  LOG(INFO) << mSelfName << " Defining container for RU " << ruSW << " at slot " << mRUEntry[ruSW];
227  }
228  return mRUDecodeVec[mRUEntry[ruSW]];
229 }

232 template <class Mapping>
233 ChipPixelData* RawPixelDecoder<Mapping>::getNextChipData(std::vector<ChipPixelData>& chipDataVec)
234 {
235  // decode new RU if no cached non-empty chips
236 
237  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
238  auto& ru = mRUDecodeVec[mCurRUDecodeID];
239  if (ru.lastChipChecked < ru.nChipsFired) {
240  auto& chipData = ru.chipsData[ru.lastChipChecked++];
241  int id = chipData.getChipID();
242  chipDataVec[id].swap(chipData);
243  return &chipDataVec[id];
244  }
245  }
246  // will need to decode new trigger
247  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no more data to decode
248  return nullptr;
249  }
250  return getNextChipData(chipDataVec);
251 }

255 bool RawPixelDecoder<Mapping>::getNextChipData(ChipPixelData& chipData)
256 {
258  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
259  auto& ru = mRUDecodeVec[mCurRUDecodeID];
260  if (ru.lastChipChecked < ru.nChipsFired) {
261  chipData.swap(ru.chipsData[ru.lastChipChecked++]);
262  return true;
263  }
264  }
265  // will need to decode new trigger
266  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no more data to decode
267  return false;
268  }
269  return getNextChipData(chipData); // is it ok to use recursion here?
270 }
