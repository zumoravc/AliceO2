39 void DataProcessor::doSend(FairMQDevice& device, MessageContext& context)
40 {
41  std::unordered_map<std::string const*, FairMQParts> outputs;
42  auto contextMessages = context.getMessagesForSending();
43  for (auto& message : contextMessages) {
44  // monitoringService.send({ message->parts.Size(), "outputs/total" });
45  FairMQParts parts = std::move(message->finalize());
46  assert(message->empty());
47  assert(parts.Size() == 2);
48  for (auto& part : parts) {
49  outputs[&(message->channel())].AddPart(std::move(part));
50  }
51  }
52  for (auto& [channel, parts] : outputs) {
53  device.Send(parts, *channel, 0);
54  }
55 }

57 void DataProcessor::doSend(FairMQDevice& device, StringContext& context)
58 {
59  for (auto& messageRef : context) {
60  FairMQParts parts;
61  FairMQMessagePtr payload(device.NewMessage());
62  auto a = messageRef.payload.get();
63  // Rebuild the message using the string as input. For now it involves a copy.
64  payload->Rebuild(reinterpret_cast<void*>(const_cast<char*>(strdup(a->data()))), a->size(), nullptr, nullptr);
65  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
66  // sigh... See if we can avoid having it const by not
67  // exposing it to the user in the first place.
68  DataHeader* dh = const_cast<DataHeader*>(cdh);
69  dh->payloadSize = payload->GetSize();
70  parts.AddPart(std::move(messageRef.header));
71  parts.AddPart(std::move(payload));
72  device.Send(parts, messageRef.channel, 0);
73  }
74 }

98 void DataProcessor::doSend(FairMQDevice& device, RawBufferContext& context)
99 {
100  for (auto& messageRef : context) {
101  FairMQParts parts;
102  FairMQMessagePtr payload(device.NewMessage());
103  auto buffer = messageRef.serializeMsg().str();
104  // Rebuild the message using the serialized ostringstream as input. For now it involves a copy.
105  size_t size = buffer.length();
106  payload->Rebuild(size);
107  std::memcpy(payload->GetData(), buffer.c_str(), size);
108  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
109  // sigh... See if we can avoid having it const by not
110  // exposing it to the user in the first place.
111  DataHeader* dh = const_cast<DataHeader*>(cdh);
112  dh->payloadSize = size;
113  parts.AddPart(std::move(messageRef.header));
114  parts.AddPart(std::move(payload));
