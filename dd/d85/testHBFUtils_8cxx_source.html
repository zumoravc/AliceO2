28 BOOST_AUTO_TEST_CASE(HBFUtils)
29 {
30  using RDH = o2::header::RAWDataHeaderV5;
31  using IR = o2::InteractionRecord;
32 
33  const bool useContinuous = true;
34 
35  o2::steer::InteractionSampler irSampler;
36  irSampler.setInteractionRate(12000); // ~1.5 interactions per orbit
37  irSampler.init();
38 
39  int nIRs = 500;
40  std::vector<o2::InteractionTimeRecord> irs(nIRs);
41  irSampler.generateCollisionTimes(irs);
42 
43  LOG(INFO) << "Emulate RDHs for raw data between IRs " << irs.front() << " and " << irs.back();
44 
45  // default sampler with BC filling like in TPC TDR, 50kHz
46  const auto& sampler = o2::raw::HBFUtils::Instance();
47 
48  uint8_t packetCounter = 0;
49  std::vector<o2::InteractionRecord> HBIRVec;
50  auto irFrom = sampler.getFirstIR(); // TFs are counted from this IR
51  int nHBF = 0, nHBFEmpty = 0, nTF = 0;
52  int nHBFOpen = 0, nHBFClose = 0;
53  RDH rdh;
54  IR rdhIR;
55  auto flushRDH = [&]() {
56  bool empty = rdh.offsetToNext == sizeof(RDH);
57  std::bitset<32> trig(rdh.triggerType);
58  int hbfID = sampler.getHBF(rdhIR);
59  auto tfhb = sampler.getTFandHBinTF(rdhIR);
60  static bool firstCall = true;
61 
62  printf("%s HBF%4d (TF%3d/HB%3d) Sz:%4d| HB Orbit/BC :%4d/%4d Trigger:(0x%08x) %s Packet: %3d Page: %3d Stop: %d\n",
63  rdh.stop ? "Close" : "Open ", hbfID, tfhb.first, tfhb.second, rdh.memorySize, rdhIR.orbit, rdhIR.bc,
64  int(rdh.triggerType), trig.to_string().c_str(), rdh.packetCounter, int(rdh.pageCnt), int(rdh.stop));
65  bool sox = (rdh.triggerType & o2::trigger::SOC || rdh.triggerType & o2::trigger::SOT);
66 
67  if (rdh.stop) {
68  nHBFClose++;
69  } else {
70  nHBFOpen++;
71  if (rdh.triggerType & o2::trigger::TF) {
72  nTF++;
73  }
74  if (rdh.triggerType & (o2::trigger::ORBIT | o2::trigger::HB)) {
75  nHBF++;
76  }
77  if (empty) {
78  nHBFEmpty++;
79  }
80  BOOST_CHECK(firstCall == sox);
81  firstCall = false;
82  }
83  };
84 
85  bool flagSOX = true; // the 1st RDH must provide readout mode: SOT or SOC
86 
87  for (int i = 0; i < nIRs; i++) {
88  int nHBF = sampler.fillHBIRvector(HBIRVec, irFrom, irs[i]);
89  irFrom = irs[i] + 1;
90 
91  // nHBF-1 HBframes don't have data, we need to create empty HBFs for them
92  if (nHBF) {
93  if (rdh.stop) { // do we need to close previous HBF?
94  flushRDH();
95  }
96  for (int j = 0; j < nHBF - 1; j++) {
97  rdhIR = HBIRVec[j];
98  rdh = sampler.createRDH<RDH>(rdhIR);
99  // dress rdh with cruID/FEE/Link ID ...
100  rdh.packetCounter = packetCounter++;
101  rdh.memorySize = sizeof(rdh);
102  rdh.offsetToNext = sizeof(rdh);
103 
104  if (flagSOX) {
105  rdh.triggerType |= useContinuous ? o2::trigger::SOC : o2::trigger::SOT;
106  flagSOX = false;
107  }
108  flushRDH(); // open empty HBH
109  rdh.packetCounter = packetCounter++;
110  rdh.stop = 0x1;
111  rdh.pageCnt++;
112  flushRDH(); // close empty HBF
113  }
114 
115  rdhIR = HBIRVec.back();
116  rdh = sampler.createRDH<RDH>(rdhIR);
117  rdh.packetCounter = packetCounter++;
118  rdh.memorySize = sizeof(rdh) + 16 + gRandom->Integer(8192 - sizeof(rdh) - 16); // random payload
119  rdh.offsetToNext = rdh.memorySize;
120  flushRDH(); // open non-empty HBH
121  rdh.packetCounter = packetCounter++;
122  rdh.stop = 0x1; // flag that it should be closed
123  rdh.pageCnt++;
124  }
125  // flush payload
126  printf("Flush payload for Orbit/BC %4d/%d\n", irs[i].orbit, irs[i].bc);
127  }
128  // close last packet
129  if (rdh.stop) { // do we need to close previous HBF?
130  flushRDH();
131  } else {
132  BOOST_CHECK(false); // lost closing RDH?
133  }
134 
135  // the TF must be completed, generate HBF till the end of the current TF
136  int tf = sampler.getTF(irs.back());
137  auto lastHBIR = sampler.getIRTF(tf + 1) - 1; // last IR of the current TF
138  sampler.fillHBIRvector(HBIRVec, irs.back(), lastHBIR);
139  for (const auto& ir : HBIRVec) {
140  rdh = sampler.createRDH<RDH>(rdhIR);
141  // dress rdh with cruID/FEE/Link ID ...
142  rdh.packetCounter = packetCounter++;
143  rdh.memorySize = sizeof(rdh);
144  rdh.offsetToNext = sizeof(rdh);
145 
146  flushRDH(); // open empty HBH
147  rdh.stop = 0x1;
148  rdh.pageCnt++;
149  flushRDH(); // close empty HBF
150  }
151 
152  printf("\nN_TF=%d, N_HBF=%d (%d empty), Opened %d / Closed %d\n", nTF, nHBF, nHBFEmpty, nHBFOpen, nHBFClose);
153  BOOST_CHECK(nHBF > nHBFEmpty);
154  BOOST_CHECK(nTF > 0);
155  BOOST_CHECK(nHBFOpen == nHBFClose);
156  BOOST_CHECK(nHBF == nTF * sampler.getNOrbitsPerTF()); // make sure all TFs are complete
157 }
