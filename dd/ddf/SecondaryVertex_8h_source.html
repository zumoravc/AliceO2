85 float invmass2prongs(float px0, float py0, float pz0, float mass0,
86  float px1, float py1, float pz1, float mass1)
87 {
88 
89  float energy0_ = energy(px0, py0, pz0, mass0);
90  float energy1_ = energy(px1, py1, pz1, mass1);
91  float energytot = energy0_ + energy1_;
92 
93  float psum2 = (px0 + px1) * (px0 + px1) +
94  (py0 + py1) * (py0 + py1) +
95  (pz0 + pz1) * (pz0 + pz1);
96  float mass = sqrtf(energytot * energytot - psum2);
97  return mass;
98 };
