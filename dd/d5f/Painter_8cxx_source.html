30 template <class T>
31 TCanvas* painter::draw(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D)
32 {
33  using DetType = CalDet<T>;
34  using CalType = CalArray<T>;
35 
36  static const Mapper& mapper = Mapper::instance();
37 
38  // ===| name and title |======================================================
39  const auto title = calDet.getName().c_str();
40  std::string name = calDet.getName();
41  std::replace(name.begin(), name.end(), ' ', '_');
42 
43  // ===| define histograms |===================================================
44  // TODO: auto scaling of ranges based on mean and variance?
45  // for the moment use roots auto scaling
46 
47  // set buffer size such that autoscaling uses the full range. This is about 2MB per histogram!
48  const int bufferSize = TH1::GetDefaultBufferSize();
49  TH1::SetDefaultBufferSize(Sector::MAXSECTOR * mapper.getPadsInSector());
50 
51  auto hAside1D = new TH1F(Form("h_Aside_1D_%s", name.c_str()), Form("%s (A-Side)", title),
52  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
53 
54  auto hCside1D = new TH1F(Form("h_Cside_1D_%s", name.c_str()), Form("%s (C-Side)", title),
55  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
56 
57  auto hAside2D = new TH2F(Form("h_Aside_2D_%s", name.c_str()), Form("%s (A-Side);x (cm);y (cm)", title),
58  300, -300, 300, 300, -300, 300);
59 
60  auto hCside2D = new TH2F(Form("h_Cside_2D_%s", name.c_str()), Form("%s (C-Side);x (cm);y (cm)", title),
61  300, -300, 300, 300, -300, 300);
62 
63  for (ROC roc; !roc.looped(); ++roc) {
64 
65  auto hist2D = hAside2D;
66  auto hist1D = hAside1D;
67  if (roc.side() == Side::C) {
68  hist2D = hCside2D;
69  hist1D = hCside1D;
70  }
71 
72  const int nrows = mapper.getNumberOfRowsROC(roc);
73  for (int irow = 0; irow < nrows; ++irow) {
74  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
75  for (int ipad = 0; ipad < npads; ++ipad) {
76  const auto val = calDet.getValue(roc, irow, ipad);
77  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
78  const int bin = hist2D->FindBin(pos.X(), pos.Y());
79  if (!hist2D->GetBinContent(bin))
80  hist2D->SetBinContent(bin, val);
81  hist1D->Fill(val);
82  }
83  }
84  }
85 
86  if (xMax1D > xMin1D) {
87  hAside2D->SetMinimum(xMin1D);
88  hAside2D->SetMaximum(xMax1D);
89  hCside2D->SetMinimum(xMin1D);
90  hCside2D->SetMaximum(xMax1D);
91  }
92 
93  // ===| Draw histograms |=====================================================
94  auto c = new TCanvas(Form("c_%s", name.c_str()), title, 1000, 1000);
95  c->Divide(2, 2);
96 
97  c->cd(1);
98  hAside2D->Draw("colz");
99 
100  c->cd(2);
101  hCside2D->Draw("colz");
102 
103  c->cd(3);
104  hAside1D->Draw();
105 
106  c->cd(4);
107  hCside1D->Draw();
108 
109  // reset the buffer size
110  TH1::SetDefaultBufferSize(bufferSize);
111 
112  return c;
113 }

131 template <class T>
132 TH2* painter::getHistogram2D(const CalDet<T>& calDet, Side side)
133 {
134  static const Mapper& mapper = Mapper::instance();
135 
136  const auto title = calDet.getName().c_str();
137  std::string name = calDet.getName();
138  std::replace(name.begin(), name.end(), ' ', '_');
139  const char side_name = (side == Side::A) ? 'A' : 'C';
140 
141  auto h2D = new TH2F(Form("h_%cside_2D_%s", side_name, name.c_str()),
142  Form("%s (%c-Side);x (cm);y (cm)", title, side_name),
143  300, -300, 300, 300, -300, 300);
144 
145  for (ROC roc; !roc.looped(); ++roc) {
146  if (roc.side() != side)
147  continue;
148  const int nrows = mapper.getNumberOfRowsROC(roc);
149  for (int irow = 0; irow < nrows; ++irow) {
150  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
151  for (int ipad = 0; ipad < npads; ++ipad) {
152  const auto val = calDet.getValue(roc, irow, ipad);
153  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
154  const int bin = h2D->FindBin(pos.X(), pos.Y());
155  if (!h2D->GetBinContent(bin))
156  h2D->SetBinContent(bin, val);
157  }
158  }
159  }
160 
161  return h2D;
162 }

165 template <class T>
166 TH2* painter::getHistogram2D(const CalArray<T>& calArray)
167 {
168  static const Mapper& mapper = Mapper::instance();
169 
170  const size_t position = calArray.getPadSubsetNumber();
171 
172  const PadSubset padSubset = calArray.getPadSubset();
173  // ===| maximum number of rows and pads |=====================================
174  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
175  const int npads = mapper.getNumberOfPadsInRow(padSubset, position, nrows - 1);
176 
177  // ===| create histogram |====================================================
178  const auto title = calArray.getName().c_str();
179  std::string name = calArray.getName();
180  std::replace(name.begin(), name.end(), ' ', '_');
181  auto hist = new TH2F(Form("h_%s", name.c_str()),
182  Form("%s;pad row;pad", title),
183  nrows, 0., nrows,
184  npads, -npads / 2, npads / 2);
185 
186  // ===| fill hist |===========================================================
187  for (int irow = 0; irow < nrows; ++irow) {
188  const int padsInRow = mapper.getNumberOfPadsInRow(padSubset, position, irow);
189  for (int ipad = 0; ipad < padsInRow; ++ipad) {
190  const GlobalPadNumber pad = mapper.getPadNumber(padSubset, position, irow, ipad);
191  const auto val = calArray.getValue(pad);
192  const int cpad = ipad - padsInRow / 2;
193  hist->Fill(irow, cpad, val);
194  //printf("%d %d: %f\n", irow, cpad, (double)val);
195  }
196  }
197  return hist;
198 }

212 template <class T>
213 std::vector<TCanvas*> painter::makeSummaryCanvases(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, bool onlyFilled)
214 {
215 
216  std::vector<TCanvas*> vecCanvases;
217 
218  auto nROCs = calDet.getData().size();
219 
220  if (onlyFilled) {
221  nROCs = 0;
222  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
223  const auto& roc = calDet.getCalArray(iroc);
224 
225  if (hasData(roc)) {
226  ++nROCs;
227  }
228  }
229 
230  if (!nROCs) {
231  return vecCanvases;
232  }
233  }
234 
235  // ===| set up canvases |===
236  const std::string_view calName = calDet.getName();
237  auto cSides = draw(calDet, nbins1D, xMin1D, xMax1D);
238  auto cROCs1D = new TCanvas(fmt::format("c_ROCs_{}_1D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
239  auto cROCs2D = new TCanvas(fmt::format("c_ROCs_{}_2D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
240  vecCanvases.emplace_back(cSides);
241  vecCanvases.emplace_back(cROCs1D);
242  vecCanvases.emplace_back(cROCs2D);
243 
244  cROCs1D->DivideSquare(nROCs);
245  cROCs2D->DivideSquare(nROCs);
246 
247  // ===| produce plots for each ROC |===
248  size_t pad = 1;
249  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
250  const auto& roc = calDet.getCalArray(iroc);
251 
252  if (onlyFilled && !hasData(roc)) {
253  continue;
254  }
255 
256  // ===| 1D histogram |===
257  auto h1D = new TH1F(fmt::format("h_{}_{:02d}", calName, iroc).data(), fmt::format("{} distribution ROC {:02d};ADC value", calName, iroc).data(), nbins1D, xMin1D, xMax1D);
258  for (const auto& val : roc.getData()) {
259  h1D->Fill(val);
260  }
261 
262  // ===| 2D histogram |===
263  auto h2D = painter::getHistogram2D(roc);
264  h2D->SetStats(0);
265  if (xMax1D > xMin1D) {
266  h2D->SetMinimum(xMin1D);
267  h2D->SetMaximum(xMax1D);
268  }
269  h2D->SetUniqueID(iroc);
270 
271  cROCs1D->cd(pad);
272  h1D->Draw();
273 
274  cROCs2D->cd(pad);
275  h2D->Draw("colz");
276 
277  ++pad;
278  }
279 
280  return vecCanvases;
281 }
