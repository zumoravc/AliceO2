104 inline const Float_t Tracker::getDistanceToSeed(const Cluster& cluster1, const Cluster& cluster2, const Cluster& cluster) const
105 {
106  // the seed is between "cluster1" and "cluster2" and cuts the plane
107  // of the "cluster" at a distance dR from it
108  Float_t dxSeed, dySeed, dzSeed, dz, dR, xSeed, ySeed;
109  dxSeed = cluster2.xCoordinate - cluster1.xCoordinate;
110  dySeed = cluster2.yCoordinate - cluster1.yCoordinate;
111  dzSeed = cluster2.zCoordinate - cluster1.zCoordinate;
112  dz = cluster.zCoordinate - cluster1.zCoordinate;
113  xSeed = cluster1.xCoordinate + dxSeed * dz / dzSeed;
114  ySeed = cluster1.yCoordinate + dySeed * dz / dzSeed;
115  dR = std::sqrt((cluster.xCoordinate - xSeed) * (cluster.xCoordinate - xSeed) + (cluster.yCoordinate - ySeed) * (cluster.yCoordinate - ySeed));
116  return dR;
117 }

119 inline void Tracker::getRPhiProjectionBin(const Cluster& cluster1, const Int_t layer1, const Int_t layer, Int_t& binR_proj, Int_t& binPhi_proj) const
120 {
121  Float_t dz, x_proj, y_proj, r_proj, phi_proj;
122  dz = constants::mft::LayerZCoordinate()[layer] - constants::mft::LayerZCoordinate()[layer1];
123  x_proj = cluster1.xCoordinate + dz * cluster1.xCoordinate * constants::mft::InverseLayerZCoordinate()[layer1];
124  y_proj = cluster1.yCoordinate + dz * cluster1.yCoordinate * constants::mft::InverseLayerZCoordinate()[layer1];
125  auto clsPoint2D = Point2D<Float_t>(x_proj, y_proj);
126  r_proj = clsPoint2D.R();
127  phi_proj = clsPoint2D.Phi();
128  o2::utils::BringTo02PiGen(phi_proj);
129  binR_proj = constants::index_table::getRBinIndex(r_proj);
130  binPhi_proj = constants::index_table::getPhiBinIndex(phi_proj);
131  return;
132 }

152 inline const Float_t Tracker::getCellDeviation(const ROframe& event, const Cell& cell1, const Cell& cell2) const
153 {
154  Int_t cell1layer1 = cell1.getFirstLayerId();
155  Int_t cell1layer2 = cell1.getSecondLayerId();
156 
157  Int_t cell2layer1 = cell2.getFirstLayerId();
158  Int_t cell2layer2 = cell2.getSecondLayerId();
159 
160  Int_t cell1cls1 = cell1.getFirstClusterIndex();
161  Int_t cell1cls2 = cell1.getSecondClusterIndex();
162 
163  Int_t cell2cls1 = cell2.getFirstClusterIndex();
164  Int_t cell2cls2 = cell2.getSecondClusterIndex();
165 
166  auto cluster11 = event.getClustersInLayer(cell1layer1)[cell1cls1];
167  auto cluster12 = event.getClustersInLayer(cell1layer2)[cell1cls2];
168  auto cluster21 = event.getClustersInLayer(cell2layer1)[cell2cls1];
169  auto cluster22 = event.getClustersInLayer(cell2layer2)[cell2cls2];
170 
171  Float_t cell1x1 = cluster11.xCoordinate;
172  Float_t cell1y1 = cluster11.yCoordinate;
173  Float_t cell1z1 = cluster11.zCoordinate;
174 
175  Float_t cell1x2 = cluster12.xCoordinate;
176  Float_t cell1y2 = cluster12.yCoordinate;
177  Float_t cell1z2 = cluster12.zCoordinate;
178 
179  Float_t cell2x1 = cluster21.xCoordinate;
180  Float_t cell2y1 = cluster21.yCoordinate;
181  Float_t cell2z1 = cluster21.zCoordinate;
182 
183  Float_t cell2x2 = cluster22.xCoordinate;
184  Float_t cell2y2 = cluster22.yCoordinate;
185  Float_t cell2z2 = cluster22.zCoordinate;
186 
187  Float_t cell1dx = cell1x2 - cell1x1;
188  Float_t cell1dy = cell1y2 - cell1y1;
189  Float_t cell1dz = cell1z2 - cell1z1;
190 
191  Float_t cell2dx = cell2x2 - cell2x1;
192  Float_t cell2dy = cell2y2 - cell2y1;
193  Float_t cell2dz = cell2z2 - cell2z1;
194 
195  Float_t cell1mod = std::sqrt(cell1dx * cell1dx + cell1dy * cell1dy + cell1dz * cell1dz);
196  Float_t cell2mod = std::sqrt(cell2dx * cell2dx + cell2dy * cell2dy + cell2dz * cell2dz);
197 
198  Float_t cosAngle = (cell1dx * cell2dx + cell1dy * cell2dy + cell1dz * cell2dz) / (cell1mod * cell2mod);
199 
200  return std::acos(cosAngle);
201 }
