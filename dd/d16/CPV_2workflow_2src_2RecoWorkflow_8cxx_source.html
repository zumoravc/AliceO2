55 o2::framework::WorkflowSpec getWorkflow(bool propagateMC,
56  bool enableDigitsPrinter,
57  std::string const& cfgInput,
58  std::string const& cfgOutput)
59 {
60  InputType inputType;
61 
62  try {
63  inputType = InputMap.at(cfgInput);
64  } catch (std::out_of_range&) {
65  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
66  }
67  std::vector<OutputType> outputTypes;
68  try {
69  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
70  } catch (std::out_of_range&) {
71  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
72  }
73  auto isEnabled = [&outputTypes](OutputType type) {
74  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
75  };

110  // check if the process is ready to quit
111  // this is decided upon the meta information in the CPV block header, the operation is set
112  // value kNoPayload in case of no data or no operation
113  // see also PublisherSpec.cxx
114  // in this workflow, the EOD is sent after the last real data, and all inputs will receive EOD,
115  // so it is enough to check on the first occurence
116  // FIXME: this will be changed once DPL can propagate control events like EOD
117  auto checkReady = [](o2::framework::DataRef const& ref) {
118  auto const* cpvheader = o2::framework::DataRefUtils::getHeader<o2::cpv::CPVBlockHeader*>(ref);
119  // sector number -1 indicates end-of-data
120  if (cpvheader != nullptr) {
121  // indicate normal processing if not ready and skip if ready
122  if (!cpvheader->mHasPayload) {
123  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::SkipProcessing, true);
124  }
125  }
126  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::DoProcessing, false);
127  };
