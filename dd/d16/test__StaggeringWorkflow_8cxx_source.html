33 void customize(std::vector<o2::framework::DispatchPolicy>& policies)
34 {
35  // we customize all devices to dispatch data immediately
36  auto producerMatcher = [](auto const& spec) {
37  return std::regex_match(spec.name.begin(), spec.name.end(), std::regex("producer.*"));
38  };
39  auto processorMatcher = [](auto const& spec) {
40  return std::regex_match(spec.name.begin(), spec.name.end(), std::regex("processor.*"));
41  };
42  auto triggerMatcher = [](auto const& query) {
43  o2::framework::Output reference{"PROD", "TRIGGER"};
44  return reference.origin == query.origin && reference.description == query.description;
45  };
46  policies.push_back({"producer-policy", producerMatcher, o2::framework::DispatchPolicy::DispatchOp::WhenReady, triggerMatcher});
47  policies.push_back({"processor-policy", processorMatcher, o2::framework::DispatchPolicy::DispatchOp::WhenReady});
48 }

73 std::vector<DataProcessorSpec> defineDataProcessing(ConfigContext const&)
74 {
75  // first fill the subspecifications
76  using MyDataType = o2::header::DataHeader::SubSpecificationType;
77  std::vector<o2::header::DataHeader::SubSpecificationType> subspecs(nChannels);
78  std::generate(subspecs.begin(), subspecs.end(), [counter = std::make_shared<int>(0)]() { return (*counter)++; });
79  std::vector<OutputSpec> producerOutputs;
80  for (auto const& subspec : subspecs) {
81  producerOutputs.emplace_back(OutputSpec{"PROD", "CHANNEL", subspec, Lifetime::Timeframe});
82  producerOutputs.emplace_back(OutputSpec{"PROD", "TRIGGER", subspec, Lifetime::Timeframe});
83  }
84 
85  auto producerFct = adaptStateless([subspecs](DataAllocator& outputs, RawDeviceService& device, ControlService& control) {
86 
87  for (auto const& subspec : subspecs) {
88  // since the snapshot copy is ready for sending it is scheduled but held back
89  // because of the CompletionPolicy trigger matcher. This message will be
90  // sent together with the second message.
91  outputs.snapshot(Output{"PROD", "CHANNEL", subspec, Lifetime::Timeframe}, subspec);
92  device.device()->WaitFor(std::chrono::milliseconds(100));
93  outputs.snapshot(Output{"PROD", "TRIGGER", subspec, Lifetime::Timeframe}, subspec);
94  device.device()->WaitFor(std::chrono::milliseconds(100));
95  }
96  control.endOfStream();
97  control.readyToQuit(QuitRequest::Me);
98  });
99 
100  auto processorFct = [](ProcessingContext& pc) {
101  int nActiveInputs = 0;
102  LOG(INFO) << "processing ...";
103  for (auto const& input : pc.inputs()) {
104  if (pc.inputs().isValid(input.spec->binding) == false) {
105  // this input slot is empty
106  continue;
107  }
108  auto& data = pc.inputs().get<MyDataType>(input.spec->binding.c_str());
109  LOG(INFO) << "processing " << input.spec->binding << " " << data;
110  // check if the channel binding starts with 'trigger'
111  if (input.spec->binding.find("trigger") == 0) {
112  pc.outputs().make<MyDataType>(Output{"PROC", "CHANNEL", data, Lifetime::Timeframe}) = data;
113  }
114  nActiveInputs++;
115  }
116  // since we publish with delays, and two channels are always sent together
117  ASSERT_ERROR(nActiveInputs == 2);
118  };
119  auto amendSinkInput = [subspecs](InputSpec& input, size_t index) {
120  input.binding += std::to_string(subspecs[index]);
121  DataSpecUtils::updateMatchingSubspec(input, subspecs[index]);
122  };
123 
124  auto sinkFct = adaptStateful([](CallbackService& callbacks) {
125  callbacks.set(CallbackService::Id::EndOfStream, [](EndOfStreamContext& context) {
126  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
127  });
128  return adaptStateless([](InputRecord& inputs) {
129  for (auto const& input : inputs) {
130  auto& data = inputs.get<MyDataType>(input.spec->binding.c_str());
131  LOG(INFO) << "received channel " << data;
132  } });
133  });
134 
135  std::vector<DataProcessorSpec> workflow = parallelPipeline(
136  std::vector<DataProcessorSpec>{DataProcessorSpec{
137  "processor",
138  {InputSpec{"input", "PROD", "CHANNEL", 0, Lifetime::Timeframe},
139  InputSpec{"trigger", "PROD", "TRIGGER", 0, Lifetime::Timeframe}},
140  {OutputSpec{"PROC", "CHANNEL", 0, Lifetime::Timeframe}},
141  AlgorithmSpec{processorFct}}},
142  nPipelines,
143  [&subspecs]() { return subspecs.size(); },
144  [&subspecs](size_t index) { return subspecs[index]; });
145 
146  workflow.emplace_back(DataProcessorSpec{
147  "producer",
148  Inputs{},
149  producerOutputs,
150  AlgorithmSpec{producerFct}});
151 
152  workflow.emplace_back(DataProcessorSpec{
153  "sink",
154  mergeInputs(InputSpec{"input", "PROC", "CHANNEL", 0, Lifetime::Timeframe}, nChannels, amendSinkInput),
155  {},
156  AlgorithmSpec{sinkFct}});
157  return workflow;
158 }
