236 template <class S, int I, int J, typename... Args>
237 inline int GPUReconstructionCPU::runKernel(const krnlExec& x, const krnlRunRange& y, const krnlEvent& z, const Args&... args)
238 {
239  HighResTimer* t = nullptr;
240  GPUCA_RECO_STEP myStep = S::GetRecoStep() == GPUCA_RECO_STEP::NoRecoStep ? x.step : S::GetRecoStep();
241  if (myStep == GPUCA_RECO_STEP::NoRecoStep) {
242  throw std::runtime_error("Failure running general kernel without defining RecoStep");
243  }
244  int cpuFallback = IsGPU() ? (x.device == krnlDeviceType::CPU ? 2 : (mRecoStepsGPU & myStep) != myStep) : 0;
245  unsigned int nThreads = x.nThreads;
246  unsigned int nBlocks = x.nBlocks;
247  auto prop = getKernelProperties<S, I>();
248  const int autoThreads = cpuFallback ? 1 : prop.nThreads;
249  const int autoBlocks = cpuFallback ? 1 : (prop.minBlocks * mBlockCount);
250  if (nBlocks == (unsigned int)-1) {
251  nBlocks = (nThreads + autoThreads - 1) / autoThreads;
252  nThreads = autoThreads;
253  } else if (nBlocks == (unsigned int)-2) {
254  nBlocks = nThreads;
255  nThreads = autoThreads;
256  } else if (nBlocks == (unsigned int)-3) {
257  nBlocks = autoBlocks;
258  nThreads = autoThreads;
259  } else if ((int)nThreads < 0) {
260  nThreads = cpuFallback ? 1 : -nThreads;
261  }
262  if (nThreads > GPUCA_MAX_THREADS) {
263  throw std::runtime_error("GPUCA_MAX_THREADS exceeded");
264  }
265  if (mDeviceProcessingSettings.debugLevel >= 3) {
266  GPUInfo("Running kernel %s (Stream %d, Range %d/%d, Grid %d/%d) on %s", GetKernelName<S, I>(), x.stream, y.start, y.num, nBlocks, nThreads, cpuFallback == 2 ? "CPU (forced)" : cpuFallback ? "CPU (fallback)" : mDeviceName.c_str());
267  }
268  if (nThreads == 0 || nBlocks == 0) {
269  return 0;
270  }
271  if (mDeviceProcessingSettings.debugLevel >= 0) {
272  t = &getKernelTimer<S, I, J>(myStep, !IsGPU() || cpuFallback ? getOMPThreadNum() : x.stream);
273  if (!mDeviceProcessingSettings.deviceTimers || !IsGPU() || cpuFallback) {
274  t->Start();
275  }
276  }
277  krnlSetup setup{{nBlocks, nThreads, x.stream, x.device, x.step}, y, z, 0.};
278  if (cpuFallback) {
279  if (GPUReconstructionCPU::runKernelImpl(classArgument<S, I>(), setup, args...)) {
280  return 1;
281  }
282  } else {
283  if (runKernelImpl(classArgument<S, I>(), setup, args...)) {
284  return 1;
285  }
286  }
287  if (mDeviceProcessingSettings.debugLevel >= 0) {
288  if (GPUDebug(GetKernelName<S, I>(), x.stream)) {
289  throw std::runtime_error("kernel failure");
290  }
291  if (t) {
292  if (!mDeviceProcessingSettings.deviceTimers || !IsGPU() || cpuFallback) {
293  t->Stop();
294  } else {
295  t->AddTime(setup.t);
296  }
297  }
298  }
299  return 0;
300 }

318 template <class T, int I, int J>
319 HighResTimer& GPUReconstructionCPU::getKernelTimer(RecoStep step, int num, size_t addMemorySize)
320 {
321  static int id = getNextTimerId();
322  timerMeta* timer = getTimerById(id);
323  if (timer == nullptr) {
324  timer = insertTimer(id, GetKernelName<T, I>(), J, NSLICES, 0, step);
325  }
326  if (addMemorySize) {
327  timer->memSize += addMemorySize;
328  }
329  if (num < 0 || num >= timer->num) {
330  throw std::runtime_error("Invalid timer requested");
331  }
332  return timer->timer[num];
333 }

335 template <class T, int J>
336 HighResTimer& GPUReconstructionCPU::getTimer(const char* name, int num)
337 {
338  static int id = getNextTimerId();
339  timerMeta* timer = getTimerById(id);
340  if (timer == nullptr) {
341  int max = std::max({getOMPMaxThreads(), mDeviceProcessingSettings.nDeviceHelperThreads + 1, mDeviceProcessingSettings.nStreams});
342  timer = insertTimer(id, name, J, max, 1, RecoStep::NoRecoStep);
343  }
344  if (num == -1) {
345  num = getOMPThreadNum();
346  }
347  if (num < 0 || num >= timer->num) {
348  throw std::runtime_error("Invalid timer requested");
349  }
350  return timer->timer[num];
