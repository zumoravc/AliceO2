44 Bool_t TrivialVertexer::openInputFile(const Char_t* fname)
45 {
46  mFile = TFile::Open(fname, "old");
47  if (!mFile) {
48  LOG(ERROR) << "TrivialVertexer::openInputFile() : "
49  << "Cannot open the input file !";
50  return kFALSE;
51  }
52  mTree = (TTree*)mFile->Get("o2sim");
53  if (!mTree) {
54  LOG(ERROR) << "TrivialVertexer::openInputFile() : "
55  << "Cannot get the input tree !";
56  return kFALSE;
57  }
58  Int_t rc = mTree->SetBranchAddress("MCEventHeader.", &mHeader);
59  if (rc != 0) {
60  LOG(ERROR) << "TrivialVertexer::openInputFile() : "
61  << "Cannot get the input branch ! rc=" << rc;
62  return kFALSE;
63  }
64  return kTRUE;
65 }

67 void TrivialVertexer::process(const std::vector<Cluster>& clusters, std::vector<std::array<Double_t, 3>>& vertices)
68 {
69  if (mClsLabels == nullptr) {
70  LOG(INFO) << "TrivialVertexer::process() : "
71  << "No cluster labels available ! Running with a default MC vertex...";
72  vertices.emplace_back(std::array<Double_t, 3>{0., 0., 0.});
73  return;
74  }
75 
76  if (mTree == nullptr) {
77  LOG(INFO) << "TrivialVertexer::process() : "
78  << "No MC information available ! Running with a default MC vertex...";
79  vertices.emplace_back(std::array<Double_t, 3>{0., 0., 0.});
80  return;
81  }
82 
83  Int_t lastEventID = 0;
84  Int_t firstEventID = std::numeric_limits<Int_t>::max();
85 
86  // Find the first and last MC event within this TF
87  for (Int_t i = 0; i < clusters.size(); ++i) {
88  auto mclab = (mClsLabels->getLabels(i))[0];
89  if (mclab.getTrackID() == -1)
90  continue; // noise
91  auto id = mclab.getEventID();
92  if (id < firstEventID)
93  firstEventID = id;
94  if (id > lastEventID)
95  lastEventID = id;
96  }
97 
98  for (Int_t mcEv = firstEventID; mcEv <= lastEventID; ++mcEv) {
99  mTree->GetEvent(mcEv);
100  Double_t vx = mHeader->GetX();
101  Double_t vy = mHeader->GetY();
102  Double_t vz = mHeader->GetZ();
103  vertices.emplace_back(std::array<Double_t, 3>{vx, vy, vz});
104  LOG(INFO) << "TrivialVertexer::process() : "
105  << "MC event #" << mcEv << " with vertex (" << vx << ',' << vy << ',' << vz << ')';
106  }
107 }
