27 template <typename RDH, typename Container>
28 void fillPages(Container& buffer)
29 {
30  RDH defaultHeader;
31  size_t const NofPages = buffer.size() / PageSize;
32  memset(buffer.data(), 0, buffer.size());
33  for (int pageNo = 0; pageNo < NofPages; pageNo++) {
34  auto* rdh = reinterpret_cast<RDH*>(buffer.data() + pageNo * PageSize);
35  rdh->version = defaultHeader.version;
36  rdh->headerSize = sizeof(RDH);
37  rdh->offsetToNext = PageSize;
38  rdh->memorySize = PageSize;
39  rdh->pageCnt = NofPages;
40  rdh->packetCounter = pageNo;
41  rdh->stop = pageNo + 1 == NofPages;
42  auto* data = reinterpret_cast<size_t*>(buffer.data() + pageNo * PageSize + rdh->headerSize);
43  *data = pageNo;
44  }
45 }
