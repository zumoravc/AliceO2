91 template <typename I, typename Cmp>
92 GPUdi() void GPUCommonAlgorithm::InsertionSort(I f, I l, Cmp cmp) noexcept
93 {
94  auto it0{f};
95  while (it0 != l) {
96  auto tmp{*it0};
97 
98  auto it1{it0};
99  while (it1 != f && cmp(tmp, it1[-1])) {
100  it1[0] = it1[-1];
101  --it1;
102  }
103  it1[0] = tmp;
104 
105  ++it0;
106  }
107 }

109 template <typename I, typename Cmp>
110 GPUdi() I GPUCommonAlgorithm::MedianOf3Select(I f, I l, Cmp cmp) noexcept
111 {
112  auto m = f + (l - f) / 2;
113 
114  --l;
115 
116  if (cmp(*f, *m)) {
117  if (cmp(*m, *l))
118  return m;
119  else if (cmp(*f, *l))
120  return l;
121  else
122  return f;
123  } else if (cmp(*f, *l))
124  return f;
125  else if (cmp(*m, *l))
126  return l;
127  else
128  return m;
129 }

131 template <typename I, typename T, typename Cmp>
132 GPUdi() I GPUCommonAlgorithm::UnguardedPartition(I f, I l, T piv, Cmp cmp) noexcept
133 {
134  do {
135  while (cmp(*f, piv))
136  ++f;
137  --l;
138  while (cmp(piv, *l))
139  --l;
140 
141  if (l <= f)
142  return f;
143  IterSwap(f, l);
144  ++f;
145  } while (true);
146 }

148 template <typename I, typename Cmp>
149 GPUdi() void GPUCommonAlgorithm::QuickSort(I f, I l, Cmp cmp) noexcept
150 {
151  if (f == l)
152 
153  return;
154 
155  using IndexType = unsigned short;
156 
157  struct pair {
158  IndexType first;
159  IndexType second;
160  };
161 
162  struct Stack {
163  pair data[11];
164  unsigned char n{0};
165 
166  GPUd() void emplace(IndexType x, IndexType y)
167  {
168  data[n++] = {x, y};
169  }
170  GPUd() bool empty() const { return n == 0; }
171  GPUd() pair& top() { return data[n - 1]; }
172  GPUd() void pop() { --n; }
173  };
174 
175  Stack s;
176  s.emplace(0, l - f);
177  while (!s.empty()) {
178  const auto it0 = f + s.top().first;
179  const auto it1 = f + s.top().second;
180  s.pop();
181 
182  const auto piv = *MedianOf3Select(it0, it1, cmp);
183  const auto pp = UnguardedPartition(it0, it1, piv, cmp);
184 
185  constexpr auto cutoff = 50u;
186  const auto lsz = pp - it0;
187  const auto rsz = it1 - pp;
188  if (lsz < rsz) {
189  if (rsz > cutoff)
190  s.emplace(pp - f, it1 - f);
191  if (lsz > cutoff)
192  s.emplace(it0 - f, pp - f);
193  } else {
194  if (lsz > cutoff)
195  s.emplace(it0 - f, pp - f);
196  if (rsz > cutoff)
197  s.emplace(pp - f, it1 - f);
198  }
199  }
200  InsertionSort(f, l, cmp);
201 }

287 template <class T, class S>
288 GPUdi() void GPUCommonAlgorithm::sortInBlock(T* begin, T* end, const S& comp)
289 {
290 #ifndef GPUCA_GPUCODE_DEVICE
291  GPUCommonAlgorithm::sort(begin, end, comp);
292 #else
293  int n = end - begin;
294  for (int i = 0; i < n; i++) {
295  for (int tIdx = get_local_id(0); tIdx < n; tIdx += get_local_size(0)) {
296  int offset = i % 2;
297  int curPos = 2 * tIdx + offset;
298  int nextPos = curPos + 1;
299 
300  if (nextPos < n) {
301  if (!comp(begin[curPos], begin[nextPos])) {
302  IterSwap(&begin[curPos], &begin[nextPos]);
303  }
304  }
305  }
306  GPUbarrier();
307  }
308 #endif
309 }
