429 inline void MagneticWrapperChebyshev::cylindricalToCartesianCylB(const Double_t* rphiz, const Double_t* brphiz,
430  Double_t* bxyz)
431 {
432  Double_t btr = TMath::Sqrt(brphiz[0] * brphiz[0] + brphiz[1] * brphiz[1]);
433  Double_t psiPLUSphi = TMath::ATan2(brphiz[1], brphiz[0]) + rphiz[1];
434  bxyz[0] = btr * TMath::Cos(psiPLUSphi);
435  bxyz[1] = btr * TMath::Sin(psiPLUSphi);
436  bxyz[2] = brphiz[2];
437 }

440 inline void MagneticWrapperChebyshev::cylindricalToCartesianCartB(const Double_t* xyz, const Double_t* brphiz,
441  Double_t* bxyz)
442 {
443  Double_t btr = TMath::Sqrt(brphiz[0] * brphiz[0] + brphiz[1] * brphiz[1]);
444  Double_t phiPLUSpsi = TMath::ATan2(xyz[1], xyz[0]) + TMath::ATan2(brphiz[1], brphiz[0]);
445  bxyz[0] = btr * TMath::Cos(phiPLUSpsi);
446  bxyz[1] = btr * TMath::Sin(phiPLUSpsi);
447  bxyz[2] = brphiz[2];

451 inline void MagneticWrapperChebyshev::cartesianToCylindricalCartB(const Double_t* xyz, const Double_t* bxyz,
452  Double_t* brphiz)
453 {
454  Double_t btr = TMath::Sqrt(bxyz[0] * bxyz[0] + bxyz[1] * bxyz[1]);
455  Double_t psiMINphi = TMath::ATan2(bxyz[1], bxyz[0]) - TMath::ATan2(xyz[1], xyz[0]);
456 
457  brphiz[0] = btr * TMath::Cos(psiMINphi);
458  brphiz[1] = btr * TMath::Sin(psiMINphi);
459  brphiz[2] = bxyz[2];
460 }
