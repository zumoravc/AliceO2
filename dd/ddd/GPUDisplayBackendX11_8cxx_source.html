164 int GPUDisplayBackendX11::OpenGLMain()
165 {
166  XSetWindowAttributes windowAttributes;
167  XVisualInfo* visualInfo = nullptr;
168  XEvent event;
169  Colormap colorMap;
170  GLXContext glxContext;
171  int errorBase;
172  int eventBase;
173 
174  // Open a connection to the X server
175  mDisplay = XOpenDisplay(nullptr);
176 
177  if (mDisplay == nullptr) {
178  GPUError("could not open display");
179  return (-1);
180  }
181 
182  // Make sure OpenGL's GLX extension supported
183  if (!glXQueryExtension(mDisplay, &errorBase, &eventBase)) {
184  GPUError("X server has no OpenGL GLX extension");
185  return (-1);
186  }
187 
188  const char* glxExt = glXQueryExtensionsString(mDisplay, DefaultScreen(mDisplay));
189  if (strstr(glxExt, "GLX_EXT_swap_control") == nullptr) {
190  GPUError("No vsync support!");
191  vsync_supported = false;
192  } else {
193  vsync_supported = true;
194  }
195 
196  // Require MSAA, double buffering, and Depth buffer
197  int attribs[] = {GLX_X_RENDERABLE, True, GLX_DRAWABLE_TYPE, GLX_WINDOW_BIT, GLX_RENDER_TYPE, GLX_RGBA_BIT, GLX_X_VISUAL_TYPE, GLX_TRUE_COLOR, GLX_RED_SIZE, 8, GLX_GREEN_SIZE, 8, GLX_BLUE_SIZE, 8, GLX_ALPHA_SIZE, 8, GLX_DEPTH_SIZE, 24, GLX_STENCIL_SIZE, 8, GLX_DOUBLEBUFFER, True,
198  // GLX_SAMPLE_BUFFERS , 1, //Disable MSAA here, we do it by rendering to offscreenbuffer
199  // GLX_SAMPLES , MSAA_SAMPLES,
200  None};
201 
202  GLXFBConfig fbconfig = 0;
203  int fbcount;
204  GLXFBConfig* fbc = glXChooseFBConfig(mDisplay, DefaultScreen(mDisplay), attribs, &fbcount);
205  if (fbc == nullptr || fbcount == 0) {
206  GPUError("Failed to get MSAA GLXFBConfig");
207  return (-1);
208  }
209  fbconfig = fbc[0];
210  XFree(fbc);
211  visualInfo = glXGetVisualFromFBConfig(mDisplay, fbconfig);
212 
213  if (visualInfo == nullptr) {
214  GPUError("no RGB visual with depth buffer");
215  return (-1);
216  }
217 
218  // Create an OpenGL rendering context
219  glXCreateContextAttribsARBProc glXCreateContextAttribsARB = (glXCreateContextAttribsARBProc)glXGetProcAddressARB((const GLubyte*)"glXCreateContextAttribsARB");
220  if (glXCreateContextAttribsARB) {
221  int context_attribs[] = {
222  GLX_CONTEXT_MAJOR_VERSION_ARB, GL_MIN_VERSION_MAJOR,
223  GLX_CONTEXT_MINOR_VERSION_ARB, GL_MIN_VERSION_MINOR,
224  GLX_CONTEXT_PROFILE_MASK_ARB, GPUCA_DISPLAY_OPENGL_CORE_FLAGS ? GLX_CONTEXT_CORE_PROFILE_BIT_ARB : GLX_CONTEXT_COMPATIBILITY_PROFILE_BIT_ARB,
225  None};
226  glxContext = glXCreateContextAttribsARB(mDisplay, fbconfig, nullptr, GL_TRUE, context_attribs);
227  } else {
228  glxContext = glXCreateContext(mDisplay, visualInfo, nullptr, GL_TRUE);
229  }
230  if (glxContext == nullptr) {
231  GPUError("could not create rendering context");
232  return (-1);
233  }
234 
235  Window win = RootWindow(mDisplay, visualInfo->screen);
236  colorMap = XCreateColormap(mDisplay, win, visualInfo->visual, AllocNone);
237  windowAttributes.colormap = colorMap;
238  windowAttributes.border_pixel = 0;
239  windowAttributes.event_mask = ExposureMask | VisibilityChangeMask | KeyPressMask | KeyReleaseMask | ButtonPressMask | ButtonReleaseMask | PointerMotionMask | StructureNotifyMask | SubstructureNotifyMask | FocusChangeMask;
240 
241  // Create an X window with the selected visual
242  mWindow = XCreateWindow(mDisplay, win, 50, 50, INIT_WIDTH, INIT_HEIGHT, // Position / Width and height of window
243  0, visualInfo->depth, InputOutput, visualInfo->visual, CWBorderPixel | CWColormap | CWEventMask, &windowAttributes);
244  XSetStandardProperties(mDisplay, mWindow, GL_WINDOW_NAME, GL_WINDOW_NAME, None, nullptr, 0, nullptr);
245  glXMakeCurrent(mDisplay, mWindow, glxContext);
246  XMapWindow(mDisplay, mWindow);
247 
248  // Maximize window
249  ToggleMaximized(true);
250 
251  // Receive signal when window closed
252  Atom WM_DELETE_WINDOW = XInternAtom(mDisplay, "WM_DELETE_WINDOW", False);
253  XSetWMProtocols(mDisplay, mWindow, &WM_DELETE_WINDOW, 1);
254 #ifndef GPUCA_DISPLAY_OPENGL_CORE
255  // Prepare fonts
256  mFontBase = glGenLists(256);
257  if (!glIsList(mFontBase)) {
258  GPUError("Out of display lists.");
259  return (-1);
260  }
261  const char* f = "fixed";
262  XFontStruct* font_info = XLoadQueryFont(mDisplay, f);
263  if (!font_info) {
264  GPUError("XLoadQueryFont failed.");
265  return (-1);
266  } else {
267  int first = font_info->min_char_or_byte2;
268  int last = font_info->max_char_or_byte2;
269  glXUseXFont(font_info->fid, first, last - first + 1, mFontBase + first);
270  }
271 #endif
272 
273  // Init OpenGL...
274  if (GPUDisplayExtInit()) {
275  fprintf(stderr, "Error initializing GL extension wrapper\n");
276  return (-1);
277  }
278 
279  XMapWindow(mDisplay, mWindow);
280  XFlush(mDisplay);
281  int x11_fd = ConnectionNumber(mDisplay);
282 
283  // Enable vsync
284  if (vsync_supported) {
285  mGlXSwapIntervalEXT = (PFNGLXSWAPINTERVALEXTPROC)glXGetProcAddressARB((const GLubyte*)"glXSwapIntervalEXT");
286  if (mGlXSwapIntervalEXT == nullptr) {
287  GPUError("Cannot enable vsync");
288  return (-1);
289  }
290  mGlXSwapIntervalEXT(mDisplay, glXGetCurrentDrawable(), 0);
291  }
292 
293  if (InitGL()) {
294  return (1);
295  }
296 
297  pthread_mutex_lock(&mSemLockExit);
298  mDisplayRunning = true;
299  pthread_mutex_unlock(&mSemLockExit);
300 
301  while (1) {
302  int num_ready_fds;
303  struct timeval tv;
304  fd_set in_fds;
305  int waitCount = 0;
306  do {
307  FD_ZERO(&in_fds);
308  FD_SET(x11_fd, &in_fds);
309  tv.tv_usec = 10000;
310  tv.tv_sec = 0;
311  num_ready_fds = mMaxFPSRate || XPending(mDisplay) || select(x11_fd + 1, &in_fds, nullptr, nullptr, &tv);
312  if (num_ready_fds < 0) {
313  GPUError("Error (num_ready_fds)");
314  }
315  if (mDisplayControl == 2) {
316  break;
317  }
318  if (mSendKey) {
319  mNeedUpdate = 1;
320  }
321  if (waitCount++ != 100) {
322  mNeedUpdate = 1;
323  }
324  } while (!(num_ready_fds || mNeedUpdate));
325  mNeedUpdate = 0;
326 
327  do {
328  if (mDisplayControl == 2) {
329  break;
330  }
331  HandleSendKey();
332  if (!XPending(mDisplay)) {
333  event.type = Expose;
334  } else {
335  XNextEvent(mDisplay, &event);
336  }
337  switch (event.type) {
338  case ButtonPress: {
339  if (event.xbutton.button == 4) {
340  mMouseWheel += 100;
341  } else if (event.xbutton.button == 5) {
342  mMouseWheel -= 100;
343  } else {
344  if (event.xbutton.button == 1) {
345  mMouseDn = true;
346  } else {
347  mMouseDnR = true;
348  }
349  mMouseDnX = event.xmotion.x;
350  mMouseDnY = event.xmotion.y;
351  }
352  break;
353  }
354 
355  case ButtonRelease: {
356  if (event.xbutton.button != 4 && event.xbutton.button != 5) {
357  if (event.xbutton.button == 1) {
358  mMouseDn = false;
359  } else {
360  mMouseDnR = false;
361  }
362  }
363  break;
364  }
365 
366  case KeyPress: {
367  int handleKey = 0, keyPress = 0;
368  GetKey(event, handleKey, keyPress);
369  mKeysShift[keyPress] = mKeys[KEY_SHIFT];
370  mKeys[keyPress] = true;
371  break;
372  }
373 
374  case KeyRelease: {
375  int handleKey = 0, keyPress = 0;
376  GetKey(event, handleKey, keyPress);
377  if (mKeys[keyPress]) {
378  HandleKeyRelease(handleKey);
379  }
380  mKeys[keyPress] = false;
381  mKeysShift[keyPress] = false;
382  break;
383  }
384 
385  case MotionNotify: {
386  mouseMvX = event.xmotion.x;
387  mouseMvY = event.xmotion.y;
388  break;
389  }
390 
391  case Expose: {
392  break;
393  }
394 
395  case ConfigureNotify: {
396  ReSizeGLScene(event.xconfigure.width, event.xconfigure.height);
397  break;
398  }
399 
400  case ClientMessage: {
401  if (event.xclient.message_type == XInternAtom(mDisplay, "_NET_WM_STATE", False)) {
402  XFlush(mDisplay);
403  } else {
404  mDisplayControl = 2;
405  }
406  break;
407  }
408  }
409  } while (XPending(mDisplay)); // Loop to compress events
410  if (mDisplayControl == 2) {
411  break;
412  }
413 
414  DrawGLScene();
415  glXSwapBuffers(mDisplay, mWindow); // Buffer swap does implicit glFlush
416  }
417 
418 #ifndef GPUCA_DISPLAY_OPENGL_CORE
419  glDeleteLists(mFontBase, 256);
420  XUnloadFont(mDisplay, font_info->fid);
421 #endif
422  ExitGL();
423  glXDestroyContext(mDisplay, glxContext);
424  XFree(visualInfo);
425  XDestroyWindow(mDisplay, mWindow);
426  XCloseDisplay(mDisplay);
427 
428  pthread_mutex_lock(&mSemLockExit);
429  mDisplayRunning = false;
430  pthread_mutex_unlock(&mSemLockExit);
431 
432  return (0);
433 }

447 void GPUDisplayBackendX11::SwitchFullscreen(bool set)
448 {
449  XEvent xev;
450  memset(&xev, 0, sizeof(xev));
451  xev.type = ClientMessage;
452  xev.xclient.window = mWindow;
453  xev.xclient.message_type = XInternAtom(mDisplay, "_NET_WM_STATE", False);
454  xev.xclient.format = 32;
455  xev.xclient.data.l[0] = 2; // _NET_WM_STATE_TOGGLE

457  xev.xclient.data.l[2] = 0;
458  XSendEvent(mDisplay, DefaultRootWindow(mDisplay), False, SubstructureNotifyMask, &xev);
