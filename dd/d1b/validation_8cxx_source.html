46  void process(aod::Collision const& collision, aod::BCs const& bcs, soa::Join<aod::Tracks, aod::TracksCov> const& tracks)
47  {
48  hrun_number->Fill(collision.bc().runNumber());
49  LOGF(info, "Tracks for collision: %d", tracks.size());
50  for (auto& track : tracks) {
51  hpt_nocuts->Fill(track.pt());
52  hfCYY->Fill(track.cYY());
53  hfCZY->Fill(track.cZY());
54  hfCZZ->Fill(track.cZZ());
55  hfCSnpY->Fill(track.cSnpY());
56  hfCSnpZ->Fill(track.cSnpZ());
57  hfCSnpSnp->Fill(track.cSnpSnp());
58  hfCTglY->Fill(track.cTglY());
59  hfCTglZ->Fill(track.cTglZ());
60  hfCTglSnp->Fill(track.cTglSnp());
61  hfCTglTgl->Fill(track.cTglTgl());
62  hfC1PtY->Fill(track.c1PtY());
63  hfC1PtZ->Fill(track.c1PtZ());
64  hfC1PtSnp->Fill(track.c1PtSnp());
65  hfC1PtTgl->Fill(track.c1PtTgl());
66  hfC1Pt21Pt2->Fill(track.c1Pt21Pt2());
67  LOGF(info, "track tgl ss s%f", track.tgl());
68  }
69  }
70 };
