81 void IdPath::InitPath()
82 {
83  // sets mLevel0, mLevel1, mLevel2, validity flagss from mPath
84 
85  TSubString strippedString = mPath.Strip(TString::kBoth);
86  TString aString(strippedString);
87  strippedString = aString.Strip(TString::kBoth, '/');
88 
89  TObjArray* anArray = TString(strippedString).Tokenize("/");
90  Int_t paramCount = anArray->GetEntriesFast();
91 
92  if (paramCount == 1) {
93  if (mPath == "*") {
94  mLevel0 = "*";
95  mLevel1 = "*";
96  mLevel2 = "*";
97 
98  mValid = kTRUE;
99  } else {
100  mValid = kFALSE;
101  }
102 
103  } else if (paramCount == 2) {
104  mLevel0 = ((TObjString*)anArray->At(0))->GetString();
105  TString bString = ((TObjString*)anArray->At(1))->GetString();
106 
107  if (isWord(mLevel0) && bString == "*") {
108  mLevel1 = "*";
109  mLevel2 = "*";
110 
111  mValid = kTRUE;
112 
113  } else {
114  mValid = kFALSE;
115  }
116 
117  } else if (paramCount == 3) {
118  mLevel0 = ((TObjString*)anArray->At(0))->GetString();
119  mLevel1 = ((TObjString*)anArray->At(1))->GetString();
120  mLevel2 = ((TObjString*)anArray->At(2))->GetString();
121 
122  if ((isWord(mLevel0) || mLevel0 == "*") && (isWord(mLevel1) || mLevel1 == "*") &&
123  (isWord(mLevel2) || mLevel2 == "*")) {
124 
125  mValid = kTRUE;
126  } else {
127  mValid = kFALSE;
128  }
129 
130  } else {
131  mValid = kFALSE;
132  }
133 
134  if (!mValid) {
135  LOG(INFO) << R"(Invalid Path ")" << mPath.Data() << R"("!)";
136  } else {
137  mPath = Form("%s/%s/%s", mLevel0.Data(), mLevel1.Data(), mLevel2.Data());
138  }
139 
140  delete anArray;
141 
142  init();
143 }
