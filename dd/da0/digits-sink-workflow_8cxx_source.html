55 class DigitsSinkTask
56 {
57  public:
58  //_________________________________________________________________________________________________
59  void init(framework::InitContext& ic)
60  {
62  LOG(INFO) << "initializing digits sink";
63 
64  mText = ic.options().get<bool>("txt");
65 
66  auto outputFileName = ic.options().get<std::string>("outfile");
67  mOutputFile.open(outputFileName, (mText ? std::ios::out : (std::ios::out | std::ios::binary)));
68  if (!mOutputFile.is_open()) {
69  throw std::invalid_argument("Cannot open output file" + outputFileName);
70  }
71 
72  auto stop = [this]() {
74  LOG(INFO) << "stop file reader";
75  this->mOutputFile.close();
76  };
77  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
78  }
79 
80  //_________________________________________________________________________________________________
81  void run(framework::ProcessingContext& pc)
82  {
83  // get the input digits
84  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
85 
86  if (mText) {
87  for (auto d : digits) {
88  mOutputFile << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC() << " time " << d.getTime().sampaTime << std::endl;
89  }
90  } else {
91  int nDigits = digits.size();
92  mOutputFile.write(reinterpret_cast<char*>(&nDigits), sizeof(int));
93  mOutputFile.write(reinterpret_cast<const char*>(digits.data()), digits.size_bytes());
94  }
95  }
96 
97  private:
98  std::ofstream mOutputFile{};
99  bool mText = false;
100 };
