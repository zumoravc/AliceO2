58 void digit2raw(const std::string& input,
59  DigitEncoder& encoder,
60  PayloadPaginator& paginate)
61 {
62  o2::InteractionRecord ir;
63  std::vector<o2::mch::Digit> digits;
64  DigitReader dr(input.c_str());
65 
66  while (dr.nextDigits(ir, digits)) {
67 
68  std::vector<std::byte> buffer;
69 
70  encoder(digits, buffer, ir.orbit, ir.bc);
71 
72  paginate(buffer);
73  }
74 }

90  ("output-dir,o",po::value<std::string>()->default_value("./"),"output directory for file(s)")
91  ("input-file,i",po::value<std::string>(&input)->required(),"input file name")
92  ("configKeyValues", po::value<std::string>()->default_value(""), "comma-separated configKeyValues")
93  ("no-empty-hbf,e", po::value<bool>()->default_value(true), "do not create empty HBF pages (except for HBF starting TF)")

95  // clang-format on
96 
97  po::options_description cmdline;
98  cmdline.add(generic);
99 
100  po::store(po::command_line_parser(argc, argv).options(cmdline).run(), vm);
101 
102  if (vm.count("help")) {
103  std::cout << generic << "\n";
104  return 2;
105  }
106 
107  try {
108  po::notify(vm);
109  } catch (boost::program_options::error& e) {
110  std::cout << "Error: " << e.what() << "\n";
111  exit(1);
112  }
113 
114  po::notify(vm);
115 
116  o2::conf::ConfigurableParam::updateFromString(vm["configKeyValues"].as<std::string>());
117 
118  if (vm.count("verbosity")) {
119  fair::Logger::SetVerbosity(vm["verbosity"].as<std::string>());
120  }
121 
122  if (dummyElecMap) {
123  std::cout << "WARNING: using dummy electronic mapping\n";
124  }
125 
126  auto det2elec = (dummyElecMap ? createDet2ElecMapper<ElectronicMapperDummy>() : createDet2ElecMapper<ElectronicMapperGenerated>());
127 
128  DigitEncoder encoder = createDigitEncoder(userLogic, det2elec);
129 
130  auto solar2feelink = (dummyElecMap ? createSolar2FeeLinkMapper<ElectronicMapperDummy>() : createSolar2FeeLinkMapper<ElectronicMapperGenerated>());
131 
132  o2::raw::RawFileWriter fw(o2::header::DAQID(o2::header::DAQID::MCH).getO2Origin());
133 
134  if (vm["no-empty-hbf"].as<bool>()) {
135  fw.setDontFillEmptyHBF(true);
136  }
137 
138  auto outDirName = vm["output-dir"].as<std::string>();
139 
140  if (gSystem->AccessPathName(outDirName.c_str())) {
141  if (gSystem->mkdir(outDirName.c_str(), kTRUE)) {
142  LOG(FATAL) << "could not create output directory " << outDirName;
143  } else {
144  LOG(INFO) << "created output directory " << outDirName;
145  }
146  }
147 
148  fw.writeConfFile("MCH", "RAWDATA", fmt::format("{}/MCHraw.cfg", outDirName));
149 
150  std::string output = fmt::format("{}/mch.raw", outDirName);
151  PayloadPaginator paginator(fw, output, solar2feelink, userLogic, chargeSumMode);
152 
153  digit2raw(input, encoder, paginator);
154 
155  return 0;
156 }
