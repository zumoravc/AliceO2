353  interpolate(DataT x1, DataT x2, GPUgeneric() DataT S[]) const
354 {
356  assert(isConsistentT);
357  if (isConsistentT) {

363  }

396 template <typename DataT, int nFdimT, bool isConsistentT>
397 GPUhdi() void Spline2D<DataT, nFdimT, isConsistentT>::interpolateU(
398  GPUgeneric() const DataT Fparameters[],
399  DataT u, DataT v, GPUgeneric() DataT S[]) const
400 {
403 
404  int nu = mGridU1.getNumberOfKnots();
405  int iu = mGridU1.getKnotIndexU(u);
406  int iv = mGridU2.getKnotIndexU(v);
407 
408  const typename Spline1D<DataT>::Knot& knotU = mGridU1.getKnot(iu);
409  const typename Spline1D<DataT>::Knot& knotV = mGridU2.getKnot(iv);
410 
411 #if defined(GPUCA_GPUCODE) // constexpr array size for the GPU compiler
412  const int nFdim = nFdimT;
413 #else
414  const int nFdim = getFdimensions();
415 #endif
416 
417  const int nFdim2 = nFdimT * 2;
418  const int nFdim4 = nFdimT * 4;
419 
420  // X:=Sx, Y:=Sy, Z:=Sz
421 
422  const DataT* par00 = Fparameters + (nu * iv + iu) * nFdim4; // values { {X,Y,Z}, {X,Y,Z}'v, {X,Y,Z}'u, {X,Y,Z}''vu } at {u0, v0}
423  const DataT* par10 = par00 + nFdim4; // values { ... } at {u1, v0}
424  const DataT* par01 = par00 + nFdim4 * nu; // values { ... } at {u0, v1}
425  const DataT* par11 = par01 + nFdim4; // values { ... } at {u1, v1}
426 
427  DataT Su0[nFdim4]; // values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) }, at u0
428  DataT Du0[nFdim4]; // derivatives {}'_u at u0
429  DataT Su1[nFdim4]; // values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) }, at u1
430  DataT Du1[nFdim4]; // derivatives {}'_u at u1
431 
432  for (int i = 0; i < nFdim2; i++) {
433  Su0[i] = par00[i];
434  Su0[nFdim2 + i] = par01[i];
435 
436  Du0[i] = par00[nFdim2 + i];
437  Du0[nFdim2 + i] = par01[nFdim2 + i];
438 
439  Su1[i] = par10[i];
440  Su1[nFdim2 + i] = par11[i];
441 
442  Du1[i] = par10[nFdim2 + i];
443  Du1[nFdim2 + i] = par11[nFdim2 + i];
444  }
445 
446  DataT parU[nFdim4]; // interpolated values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) } at u
447  mGridU1.interpolateU(nFdim4, knotU, Su0, Du0, Su1, Du1, u, parU);
448 
449  const DataT* Sv0 = parU + 0;
450  const DataT* Dv0 = parU + nFdim;
451  const DataT* Sv1 = parU + nFdim2;
452  const DataT* Dv1 = parU + nFdim2 + nFdim;
453 
454  mGridU2.interpolateU(nFdim, knotV, Sv0, Dv0, Sv1, Dv1, v, S);
455 }
