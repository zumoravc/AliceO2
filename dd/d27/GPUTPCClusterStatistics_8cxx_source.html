67 INode* BuildTree(const double* frequencies, unsigned int UniqueSymbols)
68 {
69  std::priority_queue<INode*, std::vector<INode*>, NodeCmp> trees;
70 
71  for (unsigned int i = 0; i < UniqueSymbols; ++i) {
72  if (frequencies[i] != 0) {
73  trees.push(new LeafNode(frequencies[i], i));
74  }
75  }
76  while (trees.size() > 1) {
77  INode* childR = trees.top();
78  trees.pop();
79 
80  INode* childL = trees.top();
81  trees.pop();
82 
83  INode* parent = new InternalNode(childR, childL);
84  trees.push(parent);
85  }
86  return trees.top();
87 }

89 void GenerateCodes(const INode* node, const HuffCode& prefix, HuffCodeMap& outCodes)
90 {
91  if (const LeafNode* lf = dynamic_cast<const LeafNode*>(node)) {
92  outCodes[lf->c] = prefix;
93  } else if (const InternalNode* in = dynamic_cast<const InternalNode*>(node)) {
94  HuffCode leftPrefix = prefix;
95  leftPrefix.push_back(false);
96  GenerateCodes(in->left, leftPrefix, outCodes);
97 
98  HuffCode rightPrefix = prefix;
99  rightPrefix.push_back(true);
100  GenerateCodes(in->right, rightPrefix, outCodes);
101  }
102 }

105 void GPUTPCClusterStatistics::RunStatistics(const o2::tpc::ClusterNativeAccess* clustersNative, const o2::tpc::CompressedClusters* clustersCompressed, const GPUParam& param)
106 {
107  bool decodingError = false;
108  o2::tpc::ClusterNativeAccess clustersNativeDecoded;
109  std::vector<o2::tpc::ClusterNative> clusterBuffer;
110  GPUInfo("Compression statistics, decoding: %d attached (%d tracks), %d unattached", clustersCompressed->nAttachedClusters, clustersCompressed->nTracks, clustersCompressed->nUnattachedClusters);
111  mDecoder.decompress(clustersCompressed, clustersNativeDecoded, clusterBuffer, param);
112  std::vector<o2::tpc::ClusterNative> tmpClusters;
113  if (param.rec.tpcRejectionMode == GPUSettings::RejectionNone) { // verification does not make sense if we reject clusters during compression
114  for (unsigned int i = 0; i < NSLICES; i++) {
115  for (unsigned int j = 0; j < GPUCA_ROW_COUNT; j++) {
116  if (clustersNative->nClusters[i][j] != clustersNativeDecoded.nClusters[i][j]) {
117  GPUError("Number of clusters mismatch slice %u row %u: expected %d v.s. decoded %d", i, j, clustersNative->nClusters[i][j], clustersNativeDecoded.nClusters[i][j]);
118  decodingError = true;
119  continue;
120  }
121  tmpClusters.resize(clustersNative->nClusters[i][j]);
122  for (unsigned int k = 0; k < clustersNative->nClusters[i][j]; k++) {
123  tmpClusters[k] = clustersNative->clusters[i][j][k];
124  if (param.rec.tpcCompressionModes & GPUSettings::CompressionTruncate) {
125  GPUTPCCompression::truncateSignificantBitsChargeMax(tmpClusters[k].qMax, param);
126  GPUTPCCompression::truncateSignificantBitsCharge(tmpClusters[k].qTot, param);
127  GPUTPCCompression::truncateSignificantBitsWidth(tmpClusters[k].sigmaPadPacked, param);
128  GPUTPCCompression::truncateSignificantBitsWidth(tmpClusters[k].sigmaTimePacked, param);
129  }
130  }
131  std::sort(tmpClusters.begin(), tmpClusters.end());
132  for (unsigned int k = 0; k < clustersNative->nClusters[i][j]; k++) {
133  const o2::tpc::ClusterNative& c1 = tmpClusters[k];
134  const o2::tpc::ClusterNative& c2 = clustersNativeDecoded.clusters[i][j][k];
135  if (c1.timeFlagsPacked != c2.timeFlagsPacked || c1.padPacked != c2.padPacked || c1.sigmaTimePacked != c2.sigmaTimePacked || c1.sigmaPadPacked != c2.sigmaPadPacked || c1.qMax != c2.qMax || c1.qTot != c2.qTot) {
136  GPUWarning("Cluster mismatch: slice %2u row %3u hit %5u: %6d %3d %4d %3d %3d %4d %4d", i, j, k, (int)c1.getTimePacked(), (int)c1.getFlags(), (int)c1.padPacked, (int)c1.sigmaTimePacked, (int)c1.sigmaPadPacked, (int)c1.qMax, (int)c1.qTot);
137  GPUWarning("%45s %6d %3d %4d %3d %3d %4d %4d", "", (int)c2.getTimePacked(), (int)c2.getFlags(), (int)c2.padPacked, (int)c2.sigmaTimePacked, (int)c2.sigmaPadPacked, (int)c2.qMax, (int)c2.qTot);
138  decodingError = true;
139  }
140  }
141  }
142  }
143  if (decodingError) {
144  mDecodingError = true;
145  } else {
146  GPUInfo("Cluster decoding verification: PASSED");
147  }
148  }
149 
150  FillStatistic(mPqTotA, clustersCompressed->qTotA, clustersCompressed->nAttachedClusters);
151  FillStatistic(mPqMaxA, clustersCompressed->qMaxA, clustersCompressed->nAttachedClusters);

156  FillStatistic(mPtimeResA, clustersCompressed->timeResA, clustersCompressed->nAttachedClustersReduced);
157  FillStatistic(mPsigmaPadA, clustersCompressed->sigmaPadA, clustersCompressed->nAttachedClusters);
158  FillStatistic(mPsigmaTimeA, clustersCompressed->sigmaTimeA, clustersCompressed->nAttachedClusters);
159  FillStatistic(mPqPtA, clustersCompressed->qPtA, clustersCompressed->nTracks);
160  FillStatistic(mProwA, clustersCompressed->rowA, clustersCompressed->nTracks);
161  FillStatistic(mPsliceA, clustersCompressed->sliceA, clustersCompressed->nTracks);
162  FillStatistic(mPtimeA, clustersCompressed->timeA, clustersCompressed->nTracks);
163  FillStatistic(mPpadA, clustersCompressed->padA, clustersCompressed->nTracks);
164  FillStatistic(mPqTotU, clustersCompressed->qTotU, clustersCompressed->nUnattachedClusters);
165  FillStatistic(mPqMaxU, clustersCompressed->qMaxU, clustersCompressed->nUnattachedClusters);
166  FillStatistic(mPflagsU, clustersCompressed->flagsU, clustersCompressed->nUnattachedClusters);
167  FillStatistic(mPpadDiffU, clustersCompressed->padDiffU, clustersCompressed->nUnattachedClusters);
168  FillStatistic(mPtimeDiffU, clustersCompressed->timeDiffU, clustersCompressed->nUnattachedClusters);
169  FillStatistic(mPsigmaPadU, clustersCompressed->sigmaPadU, clustersCompressed->nUnattachedClusters);
170  FillStatistic(mPsigmaTimeU, clustersCompressed->sigmaTimeU, clustersCompressed->nUnattachedClusters);
171  FillStatistic<short unsigned int, 1>(mPnTrackClusters, clustersCompressed->nTrackClusters, clustersCompressed->nTracks);
172  FillStatistic<unsigned int, 1>(mPnSliceRowClusters, clustersCompressed->nSliceRowClusters, clustersCompressed->nSliceRows);
173  FillStatisticCombined(mPsigmaA, clustersCompressed->sigmaPadA, clustersCompressed->sigmaTimeA, clustersCompressed->nAttachedClusters, P_MAX_SIGMA);

176  FillStatisticCombined(mPQU, clustersCompressed->qMaxU, clustersCompressed->qTotU, clustersCompressed->nUnattachedClusters, P_MAX_QMAX);
177  FillStatisticCombined(mProwSliceA, clustersCompressed->rowDiffA, clustersCompressed->sliceLegDiffA, clustersCompressed->nAttachedClustersReduced, GPUCA_ROW_COUNT);
178  mNTotalClusters += clustersCompressed->nAttachedClusters + clustersCompressed->nUnattachedClusters;

221  GPUInfo("Combined Sigma: %6.4f --> %6.4f (%6.4f%%)", eSigma, eSigmaCombined, eSigma > 1e-3 ? (100. * (eSigma - eSigmaCombined) / eSigma) : 0.f);
222  GPUInfo("Combined Q: %6.4f --> %6.4f (%6.4f%%)", eQ, eQCombined, eQ > 1e-3 ? (100. * (eQ - eQCombined) / eQ) : 0.f);
223

227 float GPUTPCClusterStatistics::Analyze(std::vector<int>& p, const char* name, bool count)
228 {
229  double entropy = 0.;
230  double huffmanSize = 0;
231 
232  std::vector<double> prob(p.size());
233  double log2 = log(2.);
234  size_t total = 0;
235  for (unsigned int i = 0; i < p.size(); i++) {
236  total += p[i];
237  }
238  if (total) {
239  for (unsigned int i = 0; i < prob.size(); i++) {
240  if (p[i]) {
241  prob[i] = (double)p[i] / total;
242  double I = -log(prob[i]) / log2;
243  double H = I * prob[i];
244 
245  entropy += H;
246  }
247  }
248 
249  INode* root = BuildTree(prob.data(), prob.size());
250 
251  HuffCodeMap codes;
252  GenerateCodes(root, HuffCode(), codes);
253  delete root;
254 
255  for (HuffCodeMap::const_iterator it = codes.begin(); it != codes.end(); ++it) {
256  huffmanSize += it->second.size() * prob[it->first];
257  }
258 
259  if (count) {
260  mEntropy += entropy * total;
261  mHuffman += huffmanSize * total;
262  }
263  }
264  GPUInfo("Size: %30s: Entropy %7.4f Huffman %7.4f (Count) %9lld", name, entropy, huffmanSize, (long long int)total);
265  return entropy;
266 }

270 {
