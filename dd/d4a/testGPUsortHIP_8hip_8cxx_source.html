72 template <typename T>
73 void testAlmostEqualArray(T* correct, T* testing, size_t size)
74 {
75  for (size_t i = 0; i < size; i++) {
76  if (std::fabs(correct[i]) < TOLERANCE) {
77  BOOST_CHECK_SMALL(testing[i], TOLERANCE);
78  } else {
79  BOOST_CHECK_CLOSE(correct[i], testing[i], TOLERANCE);
80  }
81  }
