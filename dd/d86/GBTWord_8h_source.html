44 struct GBTWord {
46  union {
47  struct {
48  uint64_t packetIdx : 16;
49  uint64_t activeLanes : 28;
50  uint64_t na0h : 20;
51  uint64_t na1h : 8;
52  uint64_t id : 8;
53  }; // HEADER
54  struct {
55  uint64_t lanesStops : 28;
56  uint64_t na0t : 4;
57  uint64_t lanesTimeout : 28;
58  uint64_t na1t : 4;
59  uint64_t packetDone : 1;
60  uint64_t transmissionTimeout : 1;
61  uint64_t packetOverflow : 1;
62  uint64_t laneStartsViolation : 1;
63  uint64_t laneTimeouts : 1;
64  uint64_t na2t : 3;
65  // uint8_t id : 8; /// = 0xf0; Trailer Status Word (TSW) identifier
66  }; // TRAILER
67  struct {
68  uint64_t triggerType : 12;
69  uint64_t internal : 1;
70  uint64_t noData : 1;
71  uint64_t na0tr : 1;
72  uint64_t na1tr : 1;
73  uint64_t bc : 12;
74  uint64_t na2tr : 4;
75  uint64_t orbit : 32;
76  uint64_t na3tr : 6;
77  // uint8_t id : 8; /// = 0xc0; Trigger Status Word (TSW) identifier
78  }; // TRIGGER
79 
80  uint8_t data8[16]; // 80 bits GBT word + optional padding to 128 bits
81  uint64_t data64[2] = {0};
82  };
83 
84  GBTWord() = default;
85 
87  bool isDataHeader() const { return id == GBTFlagDataHeader; }
88 
90  bool isDataTrailer() const { return id == GBTFlagDataTrailer; }
91 
93  bool isTriggerWord() const { return id == GBTFlagTrigger; }
94 
96  bool isDataIB() const { return (id & 0xe0) == GBTFlagDataIB; }
97 
99  bool isDataOB() const { return (id & 0xe0) == GBTFlagDataOB; }

103 
104  uint8_t getHeader() const { return id; }

106  void printX(bool padded = true) const;
107  void printB(bool padded = true) const;

109  ClassDefNV(GBTWord, 1);
110 };

131 struct GBTDataTrailer : public GBTWord {
145  enum PacketStates : int {
146  PacketDone = 0, // Packet finished
147  TransmissionTimeout = 1, // if timeout of transmission (lanes)
148  PacketOverflow = 2, // if max number of packets reached
149  LaneStartsViolation = 3, // if at least 1 lane had a start violation
150  LaneTimeouts = 4, // if at least 1 lane timed out
151  NStatesDefined, // total number of states defined
152  MaxStateCombinations = 0x1 << NStatesDefined // number of possible combinations
153  };
154 
155  GBTDataTrailer() { id = GBTFlagDataTrailer; }
156  GBTDataTrailer(int lanesStop, int lanesTimeOut, bool done, bool transmTO, bool overflow, bool laneStViol, bool laneTO)
157  {
158  lanesStops = lanesStop;
159  lanesTimeout = lanesTimeOut;
160  packetDone = done;
161  transmissionTimeout = transmTO;
162  packetOverflow = overflow;
163  laneStartsViolation = laneStViol;
164  laneTimeouts = laneTO;
165  id = GBTFlagDataTrailer;
166  }
167 
168  uint8_t getPacketState() const { return data8[8]; }

196 struct GBTData : public GBTWord {
199  public:
201  static int getLaneIB(uint8_t v) { return v & 0x1f; }
203  static int getConnectorOB(uint8_t v) { return (v & 0x18) >> 3; } // lane only
204  static int getLaneOB(uint8_t v) { return v & 0x7; } // connector only
205  static int getLaneOB(uint8_t v, int& connector)
206  { // separately lane and connector
207  connector = getConnectorOB(v);
208  return getLaneOB(v);
209  }
210  static int getCableID(uint8_t v) { return v & 0x1f; } // combined connector and lane
211 
213  int getLaneIB() const { return getLaneIB(id); }
215  int getConnectorOB() const { return getConnectorOB(id); } // lane only
216  int getLaneOB() const { return getLaneOB(id); } // connector only
217  int getLaneOB(int& connector) const { return getLaneOB(id, connector); } // separately lane and connector
218  int getCableID() const { return getCableID(id); } // combined connector and lane
219 
220  ClassDefNV(GBTData, 1);
221 };
