52 void DataReaderVSD::open()
53 {
54  TString ESDFileName = "events_0.root";
55  Warning("GotoEvent", "OPEN");
56  mMaxEv = -1;
57  mCurEv = -1;
58  mFile = TFile::Open(ESDFileName);
59  if (!mFile) {
60  Error("VSD_Reader", "Can not open file '%s' ... terminating.",
61  ESDFileName.Data());
62  gSystem->Exit(1);
63  }
64 
65  assert(mEvDirKeys.size() == 0);
66 
67  TPMERegexp name_re("Event\\d+");
68  TObjLink* lnk = mFile->GetListOfKeys()->FirstLink();
69  while (lnk) {
70  if (name_re.Match(lnk->GetObject()->GetName())) {
71  mEvDirKeys.push_back((TKey*)lnk->GetObject());
72  }
73  lnk = lnk->Next();
74  }
75 
76  mMaxEv = mEvDirKeys.size();
77  if (mMaxEv == 0) {
78  Error("VSD_Reader", "No events to show ... terminating.");
79  gSystem->Exit(1);
80  }
81 }
