26 int TreeStream::CheckIn(Char_t type, void* pointer)
27 {
28  // Insert object
29 
30  if (mCurrentIndex >= static_cast<int>(mElements.size())) {
31  mElements.emplace_back();
32  auto& element = mElements.back();
33  element.type = type;
34  TString name = mNextName;
35  if (name.Length()) {
36  if (mNextNameCounter > 0) {
37  name += mNextNameCounter;
38  }
39  } else {
40  name = TString::Format("B%d.", static_cast<int>(mElements.size()));
41  }
42  element.name = name.Data();
43  element.ptr = pointer;
44  } else {
45  auto& element = mElements[mCurrentIndex];
46  if (element.type != type) {
47  mStatus++;
48  return 1; // mismatched data element
49  }
50  element.ptr = pointer;
51  }
52  mCurrentIndex++;
53  return 0;
54 }

57 void TreeStream::BuildTree()
58 {
59  // Build the Tree
60 
61  int entriesFilled = mTree.GetEntries();
62  if (mBranches.size() < mElements.size())
63  mBranches.resize(mElements.size());
64 
65  TString name;
66  TBranch* br = nullptr;
67  for (int i = 0; i < static_cast<int>(mElements.size()); i++) {
68  //
69  auto& element = mElements[i];
70  if (mBranches[i])
71  continue;
72  name = element.name.data();
73  if (name.IsNull()) {
74  name = TString::Format("B%d", i);
75  }
76  if (element.cls) {
77  br = mTree.Branch(name.Data(), element.cls->GetName(), &(element.ptr));
78  mBranches[i] = br;
79  if (entriesFilled) {
80  br->SetAddress(nullptr);
81  for (int ientry = 0; ientry < entriesFilled; ientry++) {
82  br->Fill();
83  }
84  br->SetAddress(&(element.ptr));
85  }
86  }
87 
88  if (element.type > 0) {
89  TString nameC = TString::Format("%s/%c", name.Data(), element.type);
90  br = mTree.Branch(name.Data(), element.ptr, nameC.Data());
91  if (entriesFilled) {
92  br->SetAddress(nullptr);
93  for (int ientry = 0; ientry < entriesFilled; ientry++) {
94  br->Fill();
95  }
96  br->SetAddress(element.ptr);
97  }
98  mBranches[i] = br;
99  }
100  }
101 }

104 void TreeStream::Fill()
105 {
106  // Fill the tree
107 
108  int entries = mElements.size();
109  if (entries > mTree.GetNbranches()) {
110  BuildTree();
111  }
112  for (int i = 0; i < entries; i++) {
113  auto& element = mElements[i];
114  if (!element.type)
115  continue;
116  auto br = mBranches[i];
117  if (br) {
118  if (element.type)
119  br->SetAddress(element.ptr);
120  }
121  }
122  if (!mStatus)
123  mTree.Fill(); // fill only in case of non conflicts
124  mStatus = 0;
125 }
