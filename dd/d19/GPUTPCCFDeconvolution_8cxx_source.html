31 GPUdii() void GPUTPCCFDeconvolution::countPeaksImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUSharedMemory& smem,
32  const Array2D<uchar>& peakMap,
33  Array2D<PackedCharge>& chargeMap,
34  const ChargePos* positions,
35  const uint digitnum)
36 {
37  SizeT idx = get_global_id(0);
38 
39  bool iamDummy = (idx >= digitnum);
40  idx = iamDummy ? digitnum - 1 : idx;
41 
42  ChargePos pos = positions[idx];
43 
44  bool iamPeak = CfUtils::isPeak(peakMap[pos]);
45 
46  char peakCount = (iamPeak) ? 1 : 0;
47 
48  ushort ll = get_local_id(0);
49  ushort partId = ll;
50 
51  ushort in3x3 = 0;
52  partId = CfUtils::partition(smem, ll, iamPeak, SCRATCH_PAD_WORK_GROUP_SIZE, &in3x3);
53 
54  if (partId < in3x3) {
55  smem.posBcast1[partId] = pos;
56  }
57  GPUbarrier();
58 
59  CfUtils::blockLoad(
60  peakMap,
61  in3x3,
62  SCRATCH_PAD_WORK_GROUP_SIZE,
63  ll,
64  0,
65  8,
66  CfConsts::InnerNeighbors,
67  smem.posBcast1,
68  smem.buf);
69 
70  uchar aboveThreshold = 0;
71  if (partId < in3x3) {
72  peakCount = countPeaksScratchpadInner(partId, smem.buf, &aboveThreshold);
73  }
74 
75  ushort in5x5 = 0;
76  partId = CfUtils::partition(smem, partId, peakCount > 0 && !iamPeak, in3x3, &in5x5);
77 
78  if (partId < in5x5) {
79  smem.posBcast1[partId] = pos;
80  smem.aboveThresholdBcast[partId] = aboveThreshold;
81  }
82  GPUbarrier();
83 
84  CfUtils::condBlockLoad<uchar, true>(
85  peakMap,
86  in5x5,
87  SCRATCH_PAD_WORK_GROUP_SIZE,
88  ll,
89  0,
90  16,
91  CfConsts::OuterNeighbors,
92  smem.posBcast1,
93  smem.aboveThresholdBcast,
94  smem.buf);
95 
96  if (partId < in5x5) {
97  peakCount = countPeaksScratchpadOuter(partId, aboveThreshold, smem.buf);
98  peakCount *= -1;
99  }
100 
101  if (iamDummy) {
102  return;
103  }
104 
105  bool has3x3 = (peakCount > 0);
106  peakCount = CAMath::Abs(int(peakCount));
107  bool split = (peakCount > 1);
108 
109  peakCount = (peakCount == 0) ? 1 : peakCount;
110 
111  PackedCharge charge = chargeMap[pos];
112  PackedCharge p(charge.unpack() / peakCount, has3x3, split);
113 
114  chargeMap[pos] = p;
115 }

118  ushort ll,
119  const uchar* isPeak,
120  uchar* aboveThreshold)
121 {

123  GPUCA_UNROLL(U(), U())

128  }
129
