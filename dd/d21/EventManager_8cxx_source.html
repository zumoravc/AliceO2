82 void EventManager::GotoEvent(Int_t no)
83 {
84  if (no == -1) {
85  no = getDataSource()->GetEventCount() - 1;
86  }
87 
88  this->currentEvent = no;
89 
90  MultiView::getInstance()->destroyAllEvents();
91 
92  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
93  dataTypeLists[i] = new TEveElementList(gDataTypeNames[i].c_str());
94  }
95 
96  for (int i = 0; i < EVisualisationGroup::NvisualisationGroups; ++i) {
97  for (int dataType = 0; dataType < EVisualisationDataType::NdataTypes; ++dataType) {
98  DataInterpreter* interpreter = dataInterpreters[i];
99  if (interpreter) {
100  TObject* data = getDataSource()->getEventData(no, (EVisualisationGroup)i);
101  std::unique_ptr<VisualisationEvent> event = interpreter->interpretDataForType(data, (EVisualisationDataType)dataType);
102  displayVisualisationEvent(*event, gVisualisationGroupName[i]);
103  }
104  }
105  }
106 
107  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
108  MultiView::getInstance()->registerElement(dataTypeLists[i]);
109  }
110 
111  MultiView::getInstance()->redraw3D();
112 }

161 void EventManager::displayVisualisationEvent(VisualisationEvent& event, const std::string& detectorName)
162 {
163  size_t trackCount = event.getTrackCount();
164 
165  auto* list = new TEveTrackList(detectorName.c_str());
166  list->IncDenyDestroy();
167 
168  for (size_t i = 0; i < trackCount; ++i) {
169  VisualisationTrack track = event.getTrack(i);
170  TEveRecTrackD t;
171  double* p = track.getMomentum();
172  t.fP = {p[0], p[1], p[2]};
173  t.fSign = track.getCharge() > 0 ? 1 : -1;
174  auto* vistrack = new TEveTrack(&t, &TEveTrackPropagator::fgDefault);
175  vistrack->SetLineColor(kMagenta);
176  size_t pointCount = track.getPointCount();
177  vistrack->Reset(pointCount);
178 
179  for (size_t j = 0; j < pointCount; ++j) {
180  auto point = track.getPoint(j);
181  vistrack->SetNextPoint(point[0], point[1], point[2]);
182  }
183  list->AddElement(vistrack);
184  }
185 
186  if (trackCount != 0) {
187  dataTypeLists[EVisualisationDataType::ESD]->AddElement(list);
188  }
189 
190  size_t clusterCount = event.getClusterCount();
191  auto* point_list = new TEvePointSet(detectorName.c_str());
192  point_list->IncDenyDestroy();
193  point_list->SetMarkerColor(kBlue);
194 
195  for (size_t i = 0; i < clusterCount; ++i) {
196  VisualisationCluster cluster = event.getCluster(i);
197  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
198  }
199 
200  if (clusterCount != 0) {
201  dataTypeLists[EVisualisationDataType::Clusters]->AddElement(point_list);
202  }
203 }
