63  //printf("binx, biny: %d %d\n",binx,biny);
64 
65  return h;

68 void utils::addFECInfo()
69 {
70  using namespace o2::tpc;
71  const int event = gPad->GetEvent();
72  if (event != 51) {
73  return;
74  }
75 
76  int binx, biny;
77  float bincx, bincy;
78  TH1* h = utils::getBinInfoXY(binx, biny, bincx, bincy);
79  if (!h) {
80  return;
81  }
82 
83  const float binValue = h->GetBinContent(binx, biny);
84  const int row = int(std::floor(bincx));
85  const int cpad = int(std::floor(bincy));
86 
87  const auto& mapper = Mapper::instance();
88 
89  const int roc = h->GetUniqueID();
90  if (roc < 0 || roc >= (int)ROC::MaxROC)
91  return;
92  if (row < 0 || row >= (int)mapper.getNumberOfRowsROC(roc))
93  return;
94  const int nPads = mapper.getNumberOfPadsInRowROC(roc, row);
95  const int pad = cpad + nPads / 2;
96  //printf("row %d, cpad %d, pad %d, nPads %d\n", row, cpad, pad, nPads);
97  if (pad < 0 || pad >= (int)nPads) {
98  return;
99  }
100  const int channel = mapper.getPadNumberInROC(PadROCPos(roc, row, pad));
101 
102  const auto& fecInfo = mapper.getFECInfo(PadROCPos(roc, row, pad));
103 
104  std::string title("#splitline{#lower[.1]{#scale[.5]{");
105  title += (roc / 18 % 2 == 0) ? "A" : "C";
106  title += fmt::format("{:02d} ({:02d}) row: {:02d}, pad: {:03d}, globalpad: {:05d} (in roc)}}}{#scale[.5]{FEC: {:02d}, Chip: {:02d}, Chn: {:02d}, Value: {:.3f}}}",
107  roc % 18, roc, row, pad, channel, fecInfo.getIndex(), fecInfo.getSampaChip(), fecInfo.getSampaChannel(), binValue);
108 
109  h->SetTitle(title.data());
110 }
