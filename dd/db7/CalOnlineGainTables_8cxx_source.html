47 float CalOnlineGainTables::getGainCorrectionFactor(int det, int row, int col) const
48 {
49  //notes for my sanity ...
50  //det gives us the [0-540]
51  //row and col give us the [0-128] 16*rob+mcm
52  int arrayoffset = getArrayOffset(det, row, col);
53  int channel = getChannel(col);
54  float GainCorrectionFactor = 0.0;
55  if (mGainTable[arrayoffset].mAdcdac == 0) {
56  if (mGainTable[arrayoffset].mFGFN[channel] < 0) {
57  GainCorrectionFactor = -1.0;
58  } else if (mGainTable[arrayoffset].mFGFN[channel] > 511) {
59  GainCorrectionFactor = CalOnlineGainTables::UnDef;
60  } else
61  GainCorrectionFactor = (mGainTable[arrayoffset].mFGFN[channel] / 2048.) + 0.875;
62  } else {
63  float ADCCorrection = (1. / (1. + ((float)mGainTable[arrayoffset].mAdcdac / 31.) * 0.4 / 1.05));
64  GainCorrectionFactor = ADCCorrection * (((mGainTable[arrayoffset].mFGFN[channel]) / 2048.) + 0.875);
65  }
66  return GainCorrectionFactor;
67 }
