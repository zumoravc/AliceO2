33 double radii2Turbo(double rMin, double rMid, double rMax, double sensW)
34 {
35  // compute turbo angle from radii and sensor width
36  return TMath::ASin((rMax * rMax - rMin * rMin) / (2 * rMid * sensW)) * TMath::RadToDeg();
37 }
