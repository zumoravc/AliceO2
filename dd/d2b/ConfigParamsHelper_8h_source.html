42  template <typename ContainerType>
43  static boost::program_options::options_description
44  prepareOptionDescriptions(ContainerType const& workflow,
45  std::vector<ConfigParamSpec> const& currentWorkflowOptions,
46  options_description vetos = options_description(),
47  std::string mode = "full")
48  {
49  boost::program_options::options_description toplevel;
50  boost::program_options::options_description wo("Global workflow options");
51  if (dpl2BoostOptions(currentWorkflowOptions, wo, vetos)) {
52  toplevel.add(wo);
53  }
54  std::string specOptionsDescription = "Available data processors";
55  if (mode == "short") {
56  specOptionsDescription += " (full info with '--help full')";
57  }
58  options_description specOptions(specOptionsDescription);
59  for (const auto& spec : workflow) {
60  std::string name = "Data processor options: " + spec.name;
61  boost::program_options::options_description processorOptions(name);
62  if (dpl2BoostOptions(spec.options, processorOptions, vetos)) {
63  // if vetos have been provided to the function we also need to make
64  // sure that there are no duplicate option definitions for the individual
65  // processor specs, so we add in order to be vetos for all subsequent specs.
66  // Note: this only concerns the main parser, all individual options are
67  // handled when starting individual processors.
68  if (vetos.options().size() > 0) {
69  vetos.add(processorOptions);
70  }
71  if (mode == "full") {
72  specOptions.add(processorOptions);
73  } else if (mode == spec.name) {
74  toplevel.add(processorOptions);
75  break;
76  }
77  }
78  if (mode == "full" || mode == "short") {
79  std::string help;
80  if (mode == "full") {
81  help = "Option groups by process name: --" + spec.name + R"( "<processor options>")";
82  } else if (mode == "short" && processorOptions.options().size() > 0) {
83  help = "Use '--help " + spec.name + "' to display processor options";
84  } else if (mode == "short" && processorOptions.options().size() == 0) {
85  help = "No processor options";
86  }
87  specOptions.add_options()(spec.name.c_str(),
88  boost::program_options::value<std::string>(),
89  help.c_str());
90  }
91  }
92  if (workflow.size() > 0 && (mode == "full" || mode == "short")) {
93  toplevel.add(specOptions);
94  }
95  return toplevel;
96  }

98  template <VariantType V>
99  static void addConfigSpecOption(const ConfigParamSpec& spec,
100  boost::program_options::options_description& options)
101  {
102  const char* name = spec.name.c_str();
103  const char* help = spec.help.c_str();
104  using Type = typename variant_type<V>::type;
105  using BoostType = typename std::conditional<V == VariantType::String, std::string, Type>::type;
106  auto value = boost::program_options::value<BoostType>();
107  if (spec.defaultValue.type() != VariantType::Empty) {
108  // set the default value if provided in the config spec
109  value = value->default_value(spec.defaultValue.get<Type>());
110  }
111  if (V == VariantType::Bool) {
112  // for bool values we also support the zero_token option to make
113  // the option usable as a single switch
114  value = value->zero_tokens();
115  }
116  options.add_options()(name, value, help);
117  }
118 };
