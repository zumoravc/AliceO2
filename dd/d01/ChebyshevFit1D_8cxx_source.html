66 void ChebyshevFit1D::fit()
67 {
68  for (int i = 0; i < mN; i++) {
69  for (int j = 0; j < i; j++) {
70  mA[i * mN + j] = mA[j * mN + i];
71  }
72  }
73  //print();
74  {
75  double* Ai = mA.data();
76  for (int i = 0; i < mN; i++, Ai += mN) {
77  double a = Ai[i];
78  if (fabs(a) < 1.e-6) {
79  Ai[i] = 0;
80  continue;
81  }
82  double* Aj = Ai + mN;
83  for (int j = i + 1; j < mN; j++, Aj += mN) {
84  double c = Aj[i] / a;
85  for (int k = i + 1; k < mN; k++) {
86  Aj[k] -= c * Ai[k];
87  }
88  mB[j] -= c * mB[i];
89  }

91  }

95  for (int i = mN - 1; i >= 0; i--, Ai -= mN) {
96  double s = mB[i];
