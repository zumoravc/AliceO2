24 BOOST_AUTO_TEST_CASE(CocreteData)
25 {
26  OutputSpec spec{
27  "TEST",
28  "FOOO",
29  1,
30  Lifetime::Timeframe};
31 
32  InputSpec inputSpec{
33  "binding",
34  "TEST",
35  "FOOO",
36  1,
37  Lifetime::Timeframe};
38 
39  BOOST_REQUIRE(DataSpecUtils::validate(inputSpec));
40 
41  {
42  ConcreteDataMatcher concrete = DataSpecUtils::asConcreteDataMatcher(spec);
43  BOOST_CHECK_EQUAL(std::string(concrete.origin.as<std::string>()), "TEST");
44  BOOST_CHECK_EQUAL(std::string(concrete.description.as<std::string>()), "FOOO");
45  BOOST_CHECK_EQUAL(concrete.subSpec, 1);
46  BOOST_CHECK_EQUAL(DataSpecUtils::describe(spec), "TEST/FOOO/1");
47  BOOST_CHECK_EQUAL(*DataSpecUtils::getOptionalSubSpec(spec), 1);
48 
49  ConcreteDataTypeMatcher dataType = DataSpecUtils::asConcreteDataTypeMatcher(spec);
50  BOOST_CHECK_EQUAL(std::string(dataType.origin.as<std::string>()), "TEST");
51  BOOST_CHECK_EQUAL(std::string(dataType.description.as<std::string>()), "FOOO");
52 
53  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 1}));
54  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == false);
55  DataSpecUtils::updateMatchingSubspec(spec, 0);
56  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == true);
57  BOOST_CHECK(DataSpecUtils::match(inputSpec, ConcreteDataMatcher{"TEST", "FOOO", 1}));
58  BOOST_CHECK(DataSpecUtils::match(inputSpec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == false);
59  DataSpecUtils::updateMatchingSubspec(inputSpec, 0);
60  BOOST_CHECK(DataSpecUtils::match(inputSpec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == true);
61  }
62 }

131 BOOST_AUTO_TEST_CASE(MatchingOutputs)
132 {
133  OutputSpec output1{
134  "TST", "A1", 0, Lifetime::Timeframe};
135 
136  OutputSpec output2{
137  "TST", "B1", 0, Lifetime::Timeframe};
138 
139  OutputSpec output3{
140  {"TST", "A1"}, Lifetime::Timeframe};
141 
142  InputSpec input1{
143  "binding", "TST", "A1", 0, Lifetime::Timeframe};
144 
145  InputSpec input2{
146  "binding", "TST", "A1", 1, Lifetime::Timeframe};
147 
148  InputSpec input3{
149  "binding", "TST", "B1", 0, Lifetime::Timeframe};
150 
151  InputSpec input4{
152  "binding", {"TST", "A1"}, Lifetime::Timeframe};
153 
154  // matching inputs to outputs
155  BOOST_CHECK(DataSpecUtils::match(input1, output1) == true);
156  BOOST_CHECK(DataSpecUtils::match(input1, output2) == false);
157  BOOST_CHECK(DataSpecUtils::match(input1, output3) == false); // Wildcard on output!
158  BOOST_CHECK(DataSpecUtils::match(input2, output1) == false);
159  BOOST_CHECK(DataSpecUtils::match(input2, output2) == false);
160  BOOST_CHECK(DataSpecUtils::match(input2, output3) == false); // Wildcard on output!
161  BOOST_CHECK(DataSpecUtils::match(input3, output1) == false);
162  BOOST_CHECK(DataSpecUtils::match(input3, output2) == true);
163  BOOST_CHECK(DataSpecUtils::match(input3, output3) == false); // Wildcard on output!
164  BOOST_CHECK(DataSpecUtils::match(input4, output1) == true); // Wildcard in input!
165  BOOST_CHECK(DataSpecUtils::match(input4, output2) == false);
166  BOOST_CHECK(DataSpecUtils::match(input4, output3) == true); // Wildcard on both!
167 
168  // matching outputs to output definitions
169  // ConcreteDataMatcher on both sides
170  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1", 0}) == true);
171  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1", 1}) == false);
172 
173  // ConcreteDataMatcher left, ConcreteDataTypeMatcher right (subspec ignored)
174  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1"}) == true);
175 
176  // ConcreteDataTypeMatcher left (subspec ignored), ConcreteDataMatcher right
177  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1", 0}) == true);
178  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1", 1}) == true);
179 
180  // ConcreteDataTypeMatcher on both sides
181  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1"}) == true);
182 }
