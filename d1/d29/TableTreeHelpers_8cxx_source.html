54 bool branchIterator::initBranch(TTree* tree)
55 {
56  // try to find branch in tree
57  mbranchPtr = tree->GetBranch(mbranchName.c_str());
58  if (mbranchPtr) {
59  return true;
60  }
61 
62  // create new branch of given data type
63  switch (marrowType) {
64  case arrow::Type::type::BOOL:
65  mleaflistString = mbranchName + "/O";
66  break;
67  case arrow::Type::type::UINT8:
68  mleaflistString = mbranchName + "/b";
69  break;
70  case arrow::Type::type::FLOAT:
71  mleaflistString = mbranchName + "/F";
72  break;
73  case arrow::Type::type::DOUBLE:
74  mleaflistString = mbranchName + "/D";
75  break;
76  case arrow::Type::type::UINT16:
77  mleaflistString = mbranchName + "/s";
78  break;
79  case arrow::Type::type::UINT32:
80  mleaflistString = mbranchName + "/i";
81  break;
82  case arrow::Type::type::UINT64:
83  mleaflistString = mbranchName + "/l";
84  break;
85  case arrow::Type::type::INT16:
86  mleaflistString = mbranchName + "/S";
87  break;
88  case arrow::Type::type::INT32:
89  mleaflistString = mbranchName + "/I";
90  break;
91  case arrow::Type::type::INT64:
92  mleaflistString = mbranchName + "/L";
93  break;
94  default:
95  LOGP(FATAL, "Type {} not handled!", marrowType);
96  break;
97  }
98 
99  mbranchPtr = tree->Branch(mbranchName.c_str(), dbuf, mleaflistString.c_str());
100  if (mbranchPtr) {
101  return true;
102  } else {
103  return false;
104  }
105 }

107 bool branchIterator::initDataBuffer(Int_t ib)
108 {
109  // reset actual row number
110  mcounterRow = 0;
111 
112  // get next chunk of given data type marrowType
113  switch (marrowType) {
114  case arrow::Type::type::BOOL:
115  var_o = std::dynamic_pointer_cast<arrow::BooleanArray>(mchunks.at(ib));
116  boolValueHolder = (Bool_t)var_o->Value(mcounterRow);
117  v = (void*)&boolValueHolder;
118  break;
119  case arrow::Type::type::UINT8:
120  var_b = (UChar_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt8Type>>(mchunks.at(ib))->raw_values();
121  v = (void*)var_b;
122  break;
123  case arrow::Type::type::FLOAT:
124  var_f = (Float_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::FloatType>>(mchunks.at(ib))->raw_values();
125  v = (void*)var_f;
126  break;
127  case arrow::Type::type::DOUBLE:
128  var_d = (Double_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::DoubleType>>(mchunks.at(ib))->raw_values();
129  v = (void*)var_d;
130  break;
131  case arrow::Type::type::UINT16:
132  vs = (UShort_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt16Type>>(mchunks.at(ib))->raw_values();
133  v = (void*)vs;
134  break;
135  case arrow::Type::type::UINT32:
136  vi = (UInt_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt32Type>>(mchunks.at(ib))->raw_values();
137  v = (void*)vi;
138  break;
139  case arrow::Type::type::UINT64:
140  vl = (ULong64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt64Type>>(mchunks.at(ib))->raw_values();
141  v = (void*)vl;
142  break;
143  case arrow::Type::type::INT16:
144  var_s = (Short_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int16Type>>(mchunks.at(ib))->raw_values();
145  v = (void*)var_s;
146  break;
147  case arrow::Type::type::INT32:
148  var_i = (Int_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int32Type>>(mchunks.at(ib))->raw_values();
149  v = (void*)var_i;
150  break;
151  case arrow::Type::type::INT64:
152  var_l = (Long64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int64Type>>(mchunks.at(ib))->raw_values();
153  v = (void*)var_l;
154  break;
155  default:
156  LOGP(FATAL, "Type {} not handled!", marrowType);
157  break;
158  }
159  mbranchPtr->SetAddress(v);
160 
161  // reset number of rows mnumberRows and row counter mcounterRow
162  mnumberRows = mchunks.at(ib)->length();
163 
164  return true;

167 bool branchIterator::push()
168 {
169  // increment row counter
170  mcounterRow++;
171 
172  // mcounterChunk and mcounterRow contain the current chunk and row
173  // return the next element if available
174  if (mcounterRow >= mnumberRows) {
175  mcounterChunk++;
176  if (mcounterChunk < mnumberChuncs) {
177  initDataBuffer(mcounterChunk);
178  } else {
179  // end of data buffer reached
180  return false;
181  }
182  } else {
183  switch (marrowType) {
184  case arrow::Type::type::BOOL:
185  boolValueHolder = (Bool_t)var_o->Value(mcounterRow);
186  v = (void*)&boolValueHolder;
187  break;
188  case arrow::Type::type::UINT8:
189  v = (void*)++var_b;
190  break;
191  case arrow::Type::type::FLOAT:
192  v = (void*)++var_f;
193  break;
194  case arrow::Type::type::DOUBLE:
195  v = (void*)++var_d;
196  break;
197  case arrow::Type::type::UINT16:
198  v = (void*)++vs;
199  break;
200  case arrow::Type::type::UINT32:
201  v = (void*)++vi;
202  break;
203  case arrow::Type::type::UINT64:
204  v = (void*)++vl;
205  break;
206  case arrow::Type::type::INT16:
207  v = (void*)++var_s;
208  break;
209  case arrow::Type::type::INT32:
210  v = (void*)++var_i;
211  break;
212  case arrow::Type::type::INT64:
213  v = (void*)++var_l;
214  break;
215  default:
216  LOGP(FATAL, "Type {} not handled!", marrowType);
217  break;
218  }
219  }
220  mbranchPtr->SetAddress(v);
221 
222  return true;
223 }

273 TTree* TableToTree::process()
274 {
275 
276  bool togo = true;
277  LOGP(DEBUG, "Number of colums: {}", mbranchIterators.size());
278  while (togo) {
279  // fill the tree
280  mtreePtr->Fill();
281 
282  // update the branches
283  for (auto brit : mbranchIterators) {
284  togo &= brit->push();
285  }
286  }
287  mtreePtr->Write();
288 
289  return mtreePtr;
290 }
