48 void Compensator::createMaterials()

52  //

163 TGeoVolume* Compensator::createMagnetYoke()
164 {
165  TGeoVolumeAssembly* voMagnet = new TGeoVolumeAssembly("DCM0");
166  voMagnet->SetName("DCM0");
167  TGeoRotation* Ry180 = new TGeoRotation("Ry180", 180., 180., 0.);
168  auto& matmgr = o2::base::MaterialManager::Instance();
169  auto kMedAlu = matmgr.getTGeoMedium("COMP_ALU_C0");
170  auto kMedCooper = matmgr.getTGeoMedium("COMP_Cu_C0");
171  auto kMedIron = matmgr.getTGeoMedium("COMP_FE_C0");
172 
173  // we use a special optimized tracking medium for the inner part
174  // of the YOKE (FE_C1 instead of FE_C2)
175  auto kMedIronInner = matmgr.getTGeoMedium("COMP_FE_C1");
176 
177  const double innerUpLx = 8.;
178  const double innerUpLy = 32.2;
179  const double innerDwLx = 46.;
180  const double innerDwLy = 23.;
181  const double outerLx = 116.4;
182  const double outerLy = 90.2;
183  const double Lz = 250.;
184 
185  new TGeoBBox("shMagnetYokeOuter", outerLx / 2.0, outerLy / 2.0, Lz / 2.0);
186  new TGeoBBox("shMagnetYokeInnerUp", innerUpLx / 2.0, innerUpLy / 2.0, Lz / 1.0);
187  new TGeoBBox("shMagnetYokeInnerDw", innerDwLx / 2.0, innerDwLy / 2.0, Lz / 1.0);
188  (new TGeoTranslation("trMagnetYokeOuter", 0.0, -29.1, 0.0))->RegisterYourself();
189  (new TGeoTranslation("trMagnetYokeInnerUp", 0.0, 0.0, 0.0))->RegisterYourself();
190  (new TGeoTranslation("trMagnetYokeInnerDw", 0.0, -27.5, 0.0))->RegisterYourself();
191 
192  TGeoCompositeShape* shMagnetYoke =
193  new TGeoCompositeShape("shMagnetBulk",
194  "shMagnetYokeOuter:trMagnetYokeOuter-(shMagnetYokeInnerUp:trMagnetYokeInnerUp+"
195  "shMagnetYokeInnerDw:trMagnetYokeInnerDw)");
196  TGeoVolume* voMagnetYoke = new TGeoVolume("voMagnetYoke", shMagnetYoke, kMedIron);
197 
198  // make a second version of volume which is smaller than the first and which can be embedded
199  // into the first for the purpose of defining a "fast physics" region
200  // introduce a thin layer dimension "delta" in which we treat physics correctly
201  // and which we can tune
202  auto& param = o2::passive::HallSimParam::Instance();
203  const double delta = param.yokeDelta;
204 
205  new TGeoBBox("shMagnetYokeOuterFast", (outerLx - delta) / 2.0, (outerLy - delta) / 2.0, (Lz - delta) / 2.0);
206  new TGeoBBox("shMagnetYokeInnerUpFast", (innerUpLx + delta) / 2.0, (innerUpLy + delta) / 2.0, Lz / 1.0);
207  new TGeoBBox("shMagnetYokeInnerDwFast", (innerDwLx + delta) / 2.0, (innerDwLy + delta) / 2.0, Lz / 1.0);
208 
209  TGeoCompositeShape* shMagnetYokeFast =
210  new TGeoCompositeShape("shMagnetInner",
211  "shMagnetYokeOuterFast:trMagnetYokeOuter-(shMagnetYokeInnerUpFast:trMagnetYokeInnerUp+"
212  "shMagnetYokeInnerDwFast:trMagnetYokeInnerDw)");
213 
214  TGeoVolume* voMagnetYokeInner = new TGeoVolume("voMagnetYokeInner", shMagnetYokeFast, kMedIronInner);
215  if (delta >= 0.) {
216  voMagnetYoke->AddNode(voMagnetYokeInner, 1, new TGeoTranslation(0., 0., 0.0));
217  }
218 
219  // Make the coils:
220  TGeoVolume* voCoilH = gGeoManager->MakeBox("voCoilH", kMedCooper, 12.64 / 2.0, 21.46 / 2.0, 310.5 / 2.0);
221  TGeoVolume* voCoilV = gGeoManager->MakeBox("voCoilV", kMedCooper, 12.64 / 2.0, 35.80 / 2.0, 26.9 / 2.0);
222 
223  // Make the top coil supports:
224  // Polygone Coordinates (x,y)
225  Double_t x, y;
226  const Double_t kDegToRad = TMath::Pi() / 180.;
227  const Double_t AngleInner = 4.5 * kDegToRad;
228  const Double_t AngleOuter = 56.0 * kDegToRad;
229  const Double_t ArcStart = 90. - AngleOuter / kDegToRad;
230  const Double_t ArcEnd = 90. + AngleInner / kDegToRad;
231  const Double_t b = 13.6;
232  const Double_t Lx = 37.2;
233  const Double_t Ly = 25.7;
234  const Double_t LxV = 14.9;
235  const Double_t R = 9.50;
236  const Double_t dz = 2.00 / 2.0;
237  const Int_t npoints = 8;
238  Double_t CenterX;
239  Double_t CenterY;
240  Double_t PointsX[npoints] = {0.};
241  Double_t PointsY[npoints] = {0.};
242  Int_t ip = 0;
243  // Start point:
244  x = 0.0;
245  y = 0.0;
246  PointsX[ip] = x;
247  PointsY[ip] = y;
248  ip++;
249  // 1st step:
250  x = 0.00;
251  y = 1.95;
252  PointsX[ip] = x;
253  PointsY[ip] = y;
254  ip++;
255  // 2nd step:
256  x += b;
257  y += b * TMath::Tan(AngleInner);
258  PointsX[ip] = x;
259  PointsY[ip] = y;
260  ip++;
261  // Center of Arc:
262  x += R * TMath::Sin(AngleInner);
263  y -= R * TMath::Cos(AngleInner);
264  CenterX = x;
265  CenterY = y;
266  TGeoTubeSeg* shPolygonArc = new TGeoTubeSeg("shPolygonArc", R - 2.0, R, dz, ArcStart, ArcEnd);
267  (new TGeoTranslation("trPolygonArc", x, y, 0.))->RegisterYourself();
268  // 3rd Step:
269  x += R * TMath::Sin(AngleOuter);
270  y += R * TMath::Cos(AngleOuter);
271  PointsX[ip] = x;
272  PointsY[ip] = y;
273  ip++;
274  // 4th Step:
275  Double_t a = Lx - b - R * TMath::Sin(AngleInner) - R * TMath::Sin(AngleOuter);
276  x = Lx;
277  y -= a * TMath::Tan(AngleOuter);
278  PointsX[ip] = x;
279  PointsY[ip] = y;
280  ip++;
281  // 5th Step:
282  x = Lx;
283  y = -Ly;
284  PointsX[ip] = x;
285  PointsY[ip] = y;
286  ip++;
287  // 6th Step:
288  x = LxV;
289  y = -Ly;
290  PointsX[ip] = x;
291  PointsY[ip] = y;
292  ip++;
293  // 7th Step:
294  x = LxV;
295  y = 0.0;
296  PointsX[ip] = x;
297  PointsY[ip] = y;
298  ip++;
299  //
300  //
301  //
302  TGeoXtru* shPolygon = new TGeoXtru(2);
303  shPolygon->SetNameTitle("shPolygon", "shPolygon");
304  shPolygon->DefinePolygon(npoints, PointsX, PointsY);
305  shPolygon->DefineSection(0, -dz, 0., 0., 1.0); // index, Z position, offset (x,y) and scale for first section
306  shPolygon->DefineSection(1, +dz, 0., 0., 1.0); // idem, second section
307 
308  TGeoCompositeShape* shCoilSupportV = new TGeoCompositeShape("shCoilSupportV", "shPolygon+shPolygonArc:trPolygonArc");
309  TGeoVolume* voCoilSupportV = new TGeoVolume("voCoilSupportV", shCoilSupportV, kMedAlu);
310 
311  const Double_t MagCoilDx = 12.64 / 2.;
312  const Double_t MagCoilDy = 21.46 / 2.;
313  const Double_t SqOuterDx = MagCoilDx + 2.8;
314  const Double_t SqInnerDx = MagCoilDx + 0.6;
315  const Double_t SqOuterDy = 29.2 / 2.;
316  const Double_t SqInnerDy = 24.8 / 2.;
317  const Double_t SqOuterDz = 15.5 / 2.;
318  const Double_t SqInnerDz = SqOuterDz * 2.;
319  TGeoBBox* shCoilSupportSqOuter = new TGeoBBox("shCoilSupportSqOuter", SqOuterDx, SqOuterDy, SqOuterDz);
320  TGeoBBox* shCoilSupportSqInner = new TGeoBBox("shCoilSupportSqInner", SqInnerDx, SqInnerDy, SqInnerDz);
321  TGeoCompositeShape* shCoilSupportSq =
322  new TGeoCompositeShape("shCoilSupportSq", "shCoilSupportSqOuter - shCoilSupportSqInner");
323  TGeoVolume* voCoilSupportSq = new TGeoVolume("voCoilSupportSq", shCoilSupportSq, kMedAlu);
324 
325  const Double_t HSuppDx = (Lx - LxV + 0.6) / 2.0;
326  const Double_t HSuppDy = 2.2 / 2.0;
327  const Double_t HSuppDz = SqOuterDz;
328 
329  TGeoVolume* voCoilSupportH = gGeoManager->MakeBox("voCoilSupportH", kMedAlu, HSuppDx, HSuppDy, HSuppDz);
330 
331  TGeoVolumeAssembly* voCoilSupport = new TGeoVolumeAssembly("voCoilSupport");
332  voCoilSupportV->SetLineColor(kViolet + 9);
333  voCoilSupportSq->SetLineColor(kBlue - 5);
334  voCoilSupportH->SetLineColor(kPink);
335  // voCoilSupportH -> SetTransparency(16);
336  voCoilSupport->AddNode(voCoilSupportV, 1, new TGeoTranslation(SqOuterDx - LxV, SqOuterDy, 0.));
337  voCoilSupport->AddNode(voCoilSupportSq, 1, new TGeoTranslation(0., 0., 0.));
338  voCoilSupport->AddNode(voCoilSupportH, 1, new TGeoTranslation(SqOuterDx + HSuppDx, SqOuterDy - Ly - HSuppDy, 0.));
339 
340  // Make the Top Support for Geodesic reference points:
341  TGeoVolume* voSupportHTop = gGeoManager->MakeBox("voSupportHTop", kMedAlu, 66.0 / 2.0, 2.0 / 2.0, 17.0 / 2.0);
342  TGeoVolume* voSupportHBot = gGeoManager->MakeBox("voSupportHBot", kMedAlu, 14.0 / 2.0, 2.0 / 2.0, 17.0 / 2.0);
343  TGeoVolume* voSupportVert = gGeoManager->MakeBox("voSupportVert", kMedAlu, 3.0 / 2.0, 25.0 / 2.0, 17.0 / 2.0);
344 
345  TGeoVolumeAssembly* voSupportGeoRefPoint = new TGeoVolumeAssembly("voSupportGeoRefPoint");
346  voSupportHTop->SetLineColor(kGreen);
347  voSupportHBot->SetLineColor(kGreen);
348  voSupportVert->SetLineColor(kGreen);
349  voSupportGeoRefPoint->AddNode(voSupportHTop, 1, new TGeoTranslation(0.0, 28.0, 0.));
350  voSupportGeoRefPoint->AddNode(voSupportHBot, 1, new TGeoTranslation(+33.0, 1.0, 0.));
351  voSupportGeoRefPoint->AddNode(voSupportHBot, 2, new TGeoTranslation(-33.0, 1.0, 0.));
352  voSupportGeoRefPoint->AddNode(voSupportVert, 1, new TGeoTranslation(+31.5, 14.5, 0.));
353  voSupportGeoRefPoint->AddNode(voSupportVert, 2, new TGeoTranslation(-31.5, 14.5, 0.));
354 
355  // Add some color:
356  voMagnetYoke->SetLineColor(kAzure - 7);
357  voCoilH->SetLineColor(kOrange - 3);
358  voCoilV->SetLineColor(kOrange - 3);
359  // Assembling:
360 
361  voMagnet->AddNode(voMagnetYoke, 1, new TGeoTranslation(0., 0., 0.0));
362  voMagnet->AddNode(voCoilH, 1, new TGeoTranslation(+16.14, +29.83, 0.0));
363  voMagnet->AddNode(voCoilH, 2, new TGeoTranslation(-16.14, +29.83, 0.0));
364  voMagnet->AddNode(voCoilH, 3, new TGeoTranslation(+16.14, -27.43, 0.0));
365  voMagnet->AddNode(voCoilH, 4, new TGeoTranslation(-16.14, -27.43, 0.0));
366  voMagnet->AddNode(voCoilV, 1, new TGeoTranslation(+16.14, 1.20, +141.8));
367  voMagnet->AddNode(voCoilV, 2, new TGeoTranslation(-16.14, 1.20, +141.8));
368  voMagnet->AddNode(voCoilV, 3, new TGeoTranslation(+16.14, 1.20, -141.8));
369  voMagnet->AddNode(voCoilV, 4, new TGeoTranslation(-16.14, 1.20, -141.8));
370  Double_t zGeoRef = 74.0 / 2. + SqOuterDz + 9.0 + 17.0 / 2.0;
371  voMagnet->AddNode(voSupportGeoRefPoint, 1, new TGeoTranslation(0., 16.0, +zGeoRef));
372  voMagnet->AddNode(voSupportGeoRefPoint, 2, new TGeoTranslation(0., 16.0, -zGeoRef));
373  Double_t zCoilSupp = 29.83 - MagCoilDy - 0.6 + SqInnerDy;
374  voMagnet->AddNode(voCoilSupport, 1, new TGeoTranslation(+16.14, zCoilSupp, 74.0 * 0.5));
375  voMagnet->AddNode(voCoilSupport, 2, new TGeoTranslation(+16.14, zCoilSupp, -74.0 * 0.5));
376  voMagnet->AddNode(voCoilSupport, 3, new TGeoTranslation(+16.14, zCoilSupp, 74.0 * 1.5));
377  voMagnet->AddNode(voCoilSupport, 4, new TGeoTranslation(+16.14, zCoilSupp, -74.0 * 1.5));
378  //
379  voMagnet->AddNode(voCoilSupport, 5, new TGeoCombiTrans(-16.14, zCoilSupp, 74.0 * 0.5, Ry180));
380  voMagnet->AddNode(voCoilSupport, 6, new TGeoCombiTrans(-16.14, zCoilSupp, -74.0 * 0.5, Ry180));
381  voMagnet->AddNode(voCoilSupport, 7, new TGeoCombiTrans(-16.14, zCoilSupp, 74.0 * 1.5, Ry180));
382  voMagnet->AddNode(voCoilSupport, 8, new TGeoCombiTrans(-16.14, zCoilSupp, -74.0 * 1.5, Ry180));
383 
384  return (TGeoVolume*)voMagnet;
385 }
