27 // check function for different versions of the cluster type, versions
28 // can differ by internal members and scaling factors
29 template <typename ClusterType>
30 bool checkClusterType()
31 {
32  constexpr float step = 0.01;
33  constexpr auto padSeparation = 1. / ClusterType::scalePadPacked;
34  constexpr auto timeSeparation = 1. / ClusterType::scaleTimePacked;
35  constexpr auto sigmaPadSeparation = 1. / ClusterType::scaleSigmaPadPacked;
36  constexpr auto sigmaTimeSeparation = 1. / ClusterType::scaleSigmaTimePacked;
37 
38  // the step size must be small then the achievable separation of values
39  static_assert(step < padSeparation, "inconsistent step size");
40  static_assert(step < timeSeparation, "inconsistent step size");
41 
42  ClusterType somecluster;
43  for (float v = 0.; v < 5.; v += step) {
44  somecluster.setPad(v);
45  auto readback = somecluster.getPad();
46  auto delta = padSeparation;
47  BOOST_REQUIRE(readback > v - delta && readback < v + delta);
48 
49  v += step;
50  somecluster.setTime(v);
51  readback = somecluster.getTime();
52  delta = timeSeparation;
53  BOOST_REQUIRE(readback > v - delta && readback < v + delta);
54 
55  v += step;
56  somecluster.setSigmaPad(v);
57  readback = somecluster.getSigmaPad();
58  delta = sigmaPadSeparation;
59  BOOST_REQUIRE(readback > v - delta && readback < v + delta);
60 
61  v += step;
62  somecluster.setSigmaTime(v);
63  readback = somecluster.getSigmaTime();
64  delta = sigmaTimeSeparation;
65  BOOST_REQUIRE(readback > v - delta && readback < v + delta);
66  }
67 
68  // currently a time frame is supposed to be 256 orbits at most, which is less
69  // than 256 TPC drift lengths; 9 bit time sampling per drift length
70  float maxTime = 256 * 512;
71  uint8_t flags = 0x66;
72  somecluster.setTimeFlags(maxTime, flags);
73  auto read = somecluster.getTime();
74  BOOST_REQUIRE(read > maxTime - timeSeparation && read < maxTime + timeSeparation);
75  BOOST_REQUIRE(somecluster.getFlags() == 0x66);
76 
77  // check that there is no crosstalk to the flags field if time is too large
78  somecluster.setFlags(0);
79  somecluster.setTime(4 * maxTime);
80  BOOST_REQUIRE(somecluster.getFlags() == 0);
81 
82  return true;
83 }
