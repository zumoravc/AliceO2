44 void TrackerDPL::init(InitContext& ic)
45 {
46  auto filename = ic.options().get<std::string>("grp-file");
47  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
48  if (grp) {
49  mGRP.reset(grp);
50  o2::base::Propagator::initFieldFromGRP(grp);
51  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
52 
53  o2::base::GeometryManager::loadGeometry();
54  o2::mft::GeometryTGeo* geom = o2::mft::GeometryTGeo::Instance();
55  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L, o2::TransformType::T2GRot,
56  o2::TransformType::T2G));
57 
58  mTracker = std::make_unique<o2::mft::Tracker>();
59  double origD[3] = {0., 0., 0.};
60  mTracker->setBz(field->getBz(origD));
61  } else {
62  LOG(ERROR) << "Cannot retrieve GRP from the " << filename.c_str() << " file !";
63  mState = 0;
64  }
65  mState = 1;
66 }

68 void TrackerDPL::run(ProcessingContext& pc)
69 {
70  if (mState != 1)
71  return;
72 
73  auto compClusters = pc.inputs().get<const std::vector<o2::itsmft::CompClusterExt>>("compClusters");
74  auto clusters = pc.inputs().get<const std::vector<o2::itsmft::Cluster>>("clusters");
75 
76  // code further down does assignment to the rofs and the altered object is used for output
77  // we therefore need a copy of the vector rather than an object created directly on the input data,
78  // the output vector however is created directly inside the message memory thus avoiding copy by
79  // snapshot
80  auto rofsinput = pc.inputs().get<const std::vector<o2::itsmft::ROFRecord>>("ROframes");
81  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"MFT", "MFTTrackROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
82 
83  LOG(INFO) << "MFTTracker pulled " << clusters.size() << " clusters in "
84  << rofs.size() << " RO frames";
85 
86  const dataformats::MCTruthContainer<MCCompLabel>* labels = mUseMC ? pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("labels").release() : nullptr;
87  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
88  if (mUseMC) {
89  // get the array as read-only span, a snapshot of the object is sent forward
90  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
91  LOG(INFO) << labels->getIndexedSize() << " MC label objects , in "
92  << mc2rofs.size() << " MC events";
93  }
94 
95  //std::vector<o2::mft::TrackMFTExt> tracks;
96  std::vector<int> allClusIdx;
97  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackLabels;
98  std::vector<o2::mft::TrackMFT> allTracks;
99  o2::dataformats::MCTruthContainer<o2::MCCompLabel> allTrackLabels;
100  std::vector<o2::mft::TrackLTF> tracksLTF;
101  auto& allTracksLTF = pc.outputs().make<std::vector<o2::mft::TrackLTF>>(Output{"MFT", "TRACKSLTF", 0, Lifetime::Timeframe});
102  std::vector<o2::mft::TrackCA> tracksCA;
103  auto& allTracksCA = pc.outputs().make<std::vector<o2::mft::TrackCA>>(Output{"MFT", "TRACKSCA", 0, Lifetime::Timeframe});
104 
105  std::uint32_t roFrame = 0;
106  o2::mft::ROframe event(0);
107 
108  Bool_t continuous = mGRP->isDetContinuousReadOut("MFT");
109  LOG(INFO) << "MFTTracker RO: continuous=" << continuous;
110 
111  // snippet to convert found tracks to final output tracks with separate cluster indices
112  auto copyTracks = [](auto& tracks, auto& allTracks, auto& allClusIdx, int offset = 0) {
113  for (auto& trc : tracks) {
114  trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
115  int ncl = trc.getNumberOfClusters();
116  for (int ic = 0; ic < ncl; ic++) {
117  allClusIdx.push_back(trc.getClusterIndex(ic) + offset);
118  }
119  allTracks.emplace_back(trc);
120  }
121  };
122 
123  if (continuous) {
124  for (const auto& rof : rofs) {
125  Int_t nclUsed = o2::mft::ioutils::loadROFrameData(rof, event, gsl::span(clusters.data(), clusters.size()), labels);
126  if (nclUsed) {
127  event.setROFrameId(roFrame);
128  event.initialise();
129  LOG(INFO) << "ROframe: " << roFrame << ", clusters loaded : " << nclUsed;
130  mTracker->setROFrame(roFrame);
131  mTracker->clustersToTracks(event);
132  tracksLTF.swap(event.getTracksLTF());
133  tracksCA.swap(event.getTracksCA());
134  LOG(INFO) << "Found tracks LTF: " << tracksLTF.size();
135  LOG(INFO) << "Found tracks CA: " << tracksCA.size();
136  trackLabels = mTracker->getTrackLabels();
137  int first = allTracks.size();
138  int shiftIdx = -rof.getFirstEntry();
139  rofs[roFrame].setFirstEntry(first);
140  std::copy(tracksLTF.begin(), tracksLTF.end(), std::back_inserter(allTracksLTF));
141  std::copy(tracksCA.begin(), tracksCA.end(), std::back_inserter(allTracksCA));
142  allTrackLabels.mergeAtBack(trackLabels);
143  }
144  roFrame++;
145  }
146  }
147 
148  //LOG(INFO) << "MFTTracker pushed " << allTracks.size() << " tracks";
149  LOG(INFO) << "MFTTracker pushed " << allTracksLTF.size() << " tracks LTF";
150  LOG(INFO) << "MFTTracker pushed " << allTracksCA.size() << " tracks CA";
151  if (mUseMC) {
152  pc.outputs().snapshot(Output{"MFT", "TRACKSMCTR", 0, Lifetime::Timeframe}, allTrackLabels);
153  pc.outputs().snapshot(Output{"MFT", "MFTTrackMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
154  }
155 
156  mState = 2;
157  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
158 }

160 DataProcessorSpec getTrackerSpec(bool useMC)
161 {
162  std::vector<InputSpec> inputs;
163  inputs.emplace_back("compClusters", "MFT", "COMPCLUSTERS", 0, Lifetime::Timeframe);
164  inputs.emplace_back("clusters", "MFT", "CLUSTERS", 0, Lifetime::Timeframe);
165  inputs.emplace_back("ROframes", "MFT", "MFTClusterROF", 0, Lifetime::Timeframe);
166 
167  std::vector<OutputSpec> outputs;
168  outputs.emplace_back("MFT", "TRACKSLTF", 0, Lifetime::Timeframe);
169  outputs.emplace_back("MFT", "TRACKSCA", 0, Lifetime::Timeframe);
170  outputs.emplace_back("MFT", "MFTTrackROF", 0, Lifetime::Timeframe);
171 
172  if (useMC) {
173  inputs.emplace_back("labels", "MFT", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
174  inputs.emplace_back("MC2ROframes", "MFT", "MFTClusterMC2ROF", 0, Lifetime::Timeframe);
175  outputs.emplace_back("MFT", "TRACKSMCTR", 0, Lifetime::Timeframe);
176  outputs.emplace_back("MFT", "MFTTrackMC2ROF", 0, Lifetime::Timeframe);
177  }
178 
179  return DataProcessorSpec{
180  "mft-tracker",
181  inputs,
182  outputs,
183  AlgorithmSpec{adaptFromTask<TrackerDPL>(useMC)},
184  Options{
185  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the output file"}},
186  }};
187 }
