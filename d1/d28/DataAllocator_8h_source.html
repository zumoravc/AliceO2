98  template <typename T, typename... Args>
99  decltype(auto) make(const Output& spec, Args... args)
100  {
101  if constexpr (is_specialization<T, std::vector>::value && has_messageable_value_type<T>::value) {
102  // this catches all std::vector objects with messageable value type before checking if is also
103  // has a root dictionary, so non-serialized transmission is preferred
104  using ValueType = typename T::value_type;
105  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
106  auto context = mContextRegistry->get<MessageContext>();
107 
108  // Note: initial payload size is 0 and will be set by the context before sending
109  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, 0);
110  return context->add<MessageContext::VectorObject<ValueType>>(std::move(headerMessage), channel, 0, std::forward<Args>(args)...).get();
111  } else if constexpr (has_root_dictionary<T>::value == true && is_messageable<T>::value == false) {
112  // Extended support for types implementing the Root ClassDef interface, both TObject
113  // derived types and others
114  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
115  auto context = mContextRegistry->get<MessageContext>();
116 
117  // Note: initial payload size is 0 and will be set by the context before sending
118  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodROOT, 0);
119  return context->add<MessageContext::RootSerializedObject<T>>(std::move(headerMessage), channel, std::forward<Args>(args)...).get();
120  } else if constexpr (std::is_base_of_v<std::string, T>) {
121  std::string* s = new std::string(args...);
122  adopt(spec, s);
123  return *s;
124  } else if constexpr (std::is_base_of_v<struct TableBuilder, T>) {
125  TableBuilder* tb = nullptr;
126  call_if_defined<struct TableBuilder>([&](auto* p) {
127  tb = new std::decay_t<decltype(*p)>(args...);
128  adopt(spec, tb);
129  });
130  return *tb;
131  } else if constexpr (std::is_base_of_v<struct TreeToTable, T>) {
132  void* t2tr = nullptr;
133  call_if_defined<struct TreeToTable>([&](auto* p) {
134  auto t2t = new std::decay_t<decltype(*p)>(args...);
135  t2t->addAllColumns();
136  adopt(spec, t2t);
137  t2tr = t2t;
138  });
139  return *reinterpret_cast<TreeToTable*>(t2tr);
140  } else if constexpr (sizeof...(Args) == 0) {
141  if constexpr (is_messageable<T>::value == true) {
142  return *reinterpret_cast<T*>(newChunk(spec, sizeof(T)).data());
143  } else if constexpr (is_specialization<T, BoostSerialized>::value == true) {
144  return make_boost<typename T::wrapped_type>(std::move(spec));
145  } else if constexpr (is_specialization<T, BoostSerialized>::value == false && framework::is_boost_serializable<T>::value == true && std::is_base_of<std::string, T>::value == false) {
146  return make_boost<T>(std::move(spec));
147  } else {
148  static_assert(always_static_assert_v<T>, ERROR_STRING);
149  }
150  } else if constexpr (sizeof...(Args) == 1) {
151  using FirstArg = typename std::tuple_element<0, std::tuple<Args...>>::type;
152  if constexpr (std::is_integral_v<FirstArg>) {
153  if constexpr (is_messageable<T>::value == true) {
154  auto [nElements] = std::make_tuple(args...);
155  auto size = nElements * sizeof(T);
156  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
157  auto context = mContextRegistry->get<MessageContext>();
158 
159  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, size);
160  return context->add<MessageContext::SpanObject<T>>(std::move(headerMessage), channel, 0, nElements).get();
161  }
162  } else if constexpr (std::is_same_v<FirstArg, std::shared_ptr<arrow::Schema>>) {
163  if constexpr (std::is_base_of_v<arrow::ipc::RecordBatchWriter, T>) {
164  auto [schema] = std::make_tuple(args...);
165  std::shared_ptr<arrow::ipc::RecordBatchWriter> writer;
166  create(spec, &writer, schema);
167  return writer;
168  }
169  } else if constexpr (is_specialization<T, BoostSerialized>::value) {
170  return make_boost<FirstArg>(std::move(spec));
171  } else {
172  static_assert(always_static_assert_v<T>, ERROR_STRING);
173  }
174  } else {
175  static_assert(always_static_assert_v<T>, ERROR_STRING);
176  }

254  template <typename T>
255  void snapshot(const Output& spec, T const& object)
256  {
257  auto proxy = mContextRegistry->get<MessageContext>()->proxy();
258  FairMQMessagePtr payloadMessage;
259  auto serializationType = o2::header::gSerializationMethodNone;
260  if constexpr (is_messageable<T>::value == true) {
261  // Serialize a snapshot of a trivially copyable, non-polymorphic object,
262  payloadMessage = proxy.createMessage(sizeof(T));
263  memcpy(payloadMessage->GetData(), &object, sizeof(T));
264 
265  serializationType = o2::header::gSerializationMethodNone;
266  } else if constexpr (is_specialization<T, std::vector>::value == true ||
267  (gsl::details::is_span<T>::value && has_messageable_value_type<T>::value)) {
268  using ElementType = typename std::remove_pointer<typename T::value_type>::type;
269  if constexpr (is_messageable<ElementType>::value) {
270  // Serialize a snapshot of a std::vector of trivially copyable, non-polymorphic elements
271  // Note: in most cases it is better to use the `make` function und work with the provided
272  // reference object
273  constexpr auto elementSizeInBytes = sizeof(ElementType);
274  auto sizeInBytes = elementSizeInBytes * object.size();
275  payloadMessage = proxy.createMessage(sizeInBytes);
276 
277  if constexpr (std::is_pointer<typename T::value_type>::value == false) {
278  // vector of elements
279  memcpy(payloadMessage->GetData(), object.data(), sizeInBytes);
280  } else {
281  // serialize vector of pointers to elements
282  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
283  for (auto const& pointer : object) {
284  memcpy(target, pointer, elementSizeInBytes);
285  target += elementSizeInBytes;
286  }
287  }
288 
289  serializationType = o2::header::gSerializationMethodNone;
290  } else if constexpr (has_root_dictionary<ElementType>::value) {
291  return snapshot(spec, ROOTSerialized<T const>(object));
292  } else {
293  static_assert(always_static_assert_v<T>,
294  "value type of std::vector not supported by API, supported types:"
295  "\n - messageable tyeps (trivially copyable, non-polymorphic structures)"
296  "\n - pointers to those"
297  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
298  }
299  } else if constexpr (has_root_dictionary<T>::value == true || is_specialization<T, ROOTSerialized>::value == true) {
300  // Serialize a snapshot of an object with root dictionary
301  payloadMessage = proxy.createMessage();
302  if constexpr (is_specialization<T, ROOTSerialized>::value == true) {
303  // Explicitely ROOT serialize a snapshot of object.
304  // An object wrapped into type `ROOTSerialized` is explicitely marked to be ROOT serialized
305  // and is expected to have a ROOT dictionary. Availability can not be checked at compile time
306  // for all cases.
307  using WrappedType = typename T::wrapped_type;
308  static_assert(std::is_same<typename T::hint_type, const char>::value ||
309  std::is_same<typename T::hint_type, TClass>::value ||
310  std::is_void<typename T::hint_type>::value,
311  "class hint must be of type TClass or const char");
312 
313  const TClass* cl = nullptr;
314  if (object.getHint() == nullptr) {
315  // get TClass info by wrapped type
316  cl = TClass::GetClass(typeid(WrappedType));
317  } else if (std::is_same<typename T::hint_type, TClass>::value) {
318  // the class info has been passed directly
319  cl = reinterpret_cast<const TClass*>(object.getHint());
320  } else if (std::is_same<typename T::hint_type, const char>::value) {
321  // get TClass info by optional name
322  cl = TClass::GetClass(reinterpret_cast<const char*>(object.getHint()));
323  }
324  if (has_root_dictionary<WrappedType>::value == false && cl == nullptr) {
325  std::string msg("ROOT serialization not supported, dictionary not found for type ");
326  if (std::is_same<typename T::hint_type, const char>::value) {
327  msg += reinterpret_cast<const char*>(object.getHint());
328  } else {
329  msg += typeid(WrappedType).name();
330  }
331  throw std::runtime_error(msg);
332  }
333  TMessageSerializer().Serialize(*payloadMessage, &object(), cl);
334  } else {
335  TMessageSerializer().Serialize(*payloadMessage, &object, TClass::GetClass(typeid(T)));
336  }
337  serializationType = o2::header::gSerializationMethodROOT;
338  } else {
339  static_assert(always_static_assert_v<T>,
340  "data type T not supported by API, \n specializations available for"
341  "\n - trivially copyable, non-polymorphic structures"
342  "\n - std::vector of messageable structures or pointers to those"
343  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
344  }
345  addPartToContext(std::move(payloadMessage), spec, serializationType);
346  }
