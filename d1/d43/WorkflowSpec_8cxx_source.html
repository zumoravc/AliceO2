54 WorkflowSpec parallelPipeline(const WorkflowSpec& specs,
55  size_t nPipelines,
56  std::function<size_t()> getNumberOfSubspecs,
57  std::function<size_t(size_t)> getSubSpec)
58 {
59  WorkflowSpec result;
60  size_t numberOfSubspecs = getNumberOfSubspecs();
61  if (numberOfSubspecs < nPipelines) {
62  // no need to create more pipelines than the number of parallel Ids, in that case
63  // each pipeline serves one id
64  nPipelines = numberOfSubspecs;
65  }
66  for (auto process : specs) {
67  size_t channels = numberOfSubspecs;
68  size_t inputMultiplicity = numberOfSubspecs / nPipelines;
69  if (numberOfSubspecs % nPipelines) {
70  // some processes will get one more channel to handle all channels
71  inputMultiplicity += 1;
72  }
73  auto amendProcess = [numberOfSubspecs, nPipelines, &channels, &inputMultiplicity, getSubSpec](DataProcessorSpec& spec, size_t pipeline) {
74  auto inputs = std::move(spec.inputs);
75  auto outputs = std::move(spec.outputs);
76  spec.inputs.reserve(inputMultiplicity);
77  spec.outputs.reserve(inputMultiplicity);
78  for (size_t inputNo = 0; inputNo < inputMultiplicity; ++inputNo) {
79  for (auto& input : inputs) {
80  spec.inputs.push_back(input);
81  spec.inputs.back().binding += std::to_string(inputNo);
82  DataSpecUtils::updateMatchingSubspec(spec.inputs.back(), getSubSpec(pipeline + inputNo * nPipelines));
83  }
84  for (auto& output : outputs) {
85  spec.outputs.push_back(output);
86  spec.outputs.back().binding.value += std::to_string(inputNo);
87  // FIXME: this will be unneeded once we have a subSpec-less variant...
88  DataSpecUtils::updateMatchingSubspec(spec.outputs.back(), getSubSpec(pipeline + inputNo * nPipelines));
89  }
90  }
91  channels -= inputMultiplicity;
92  if (inputMultiplicity > numberOfSubspecs / nPipelines &&
93  (channels % (nPipelines - (pipeline + 1))) == 0) {
94  // if the remaining ids can be distributed equally among the remaining pipelines
95  // we can decrease multiplicity
96  inputMultiplicity = numberOfSubspecs / nPipelines;
97  }
98  };
99 
100  if (nPipelines > 1) {
101  // add multiple processes and distribute inputs among them
102  auto amendedProcessors = parallel(process, nPipelines, amendProcess);
103  result.insert(result.end(), amendedProcessors.begin(), amendedProcessors.end());
104  } else if (nPipelines == 1) {
105  // add one single process with all the inputs
106  amendProcess(process, 0);
107  result.push_back(process);
108  }
109  }
110  return result;
111 }
