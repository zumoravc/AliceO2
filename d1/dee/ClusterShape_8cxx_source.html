56 Bool_t ClusterShape::IsValidShape()
57 {
58  // Check the size
59  if (mShape.size() > mNrows * mNcols)
60  return false;
61 
62  // Check for duplicates and the validity of the position
63  std::sort(mShape.begin(), mShape.end());
64  for (size_t i = 0; i < mShape.size() - 1; i++) {
65  if (mShape[i] >= mNrows * mNcols || mShape[i + 1] >= mNrows * mNcols)
66  return false;
67  if (mShape[i] == mShape[i + 1])
68  return false;
69  }
70 
71  return true;
72 }

75 Long64_t ClusterShape::GetShapeID() const
76 {
77  // DJBX33X
78  Long64_t id = 5381;
79  id = ((id << 5) + id) ^ mNrows;
80  id = ((id << 5) + id) ^ mNcols;
81  for (UInt_t i = 0; i < mShape.size(); ++i) {
82  id = ((id << 5) + id) ^ mShape[i];
83  }
84  return id;

100 UInt_t ClusterShape::ComputeCenter(UInt_t n)
101 {
102  UInt_t c = 0;
103  if (n % 2 == 0) {
104  UInt_t r = gRandom->Integer(2); // 0 or 1
105  c = (UInt_t)r + n / 2;
106  } else {
107  c = (UInt_t)(n + 1) / 2;
108  }
109  return c - 1; // 0-based
110 }
