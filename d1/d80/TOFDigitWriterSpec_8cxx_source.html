49 DataProcessorSpec getTOFDigitWriterSpec(bool useMC)
50 {
51  auto initFunction = [useMC](InitContext& ic) {
52  // get the option from the init context
53  auto filename = ic.options().get<std::string>("tof-digit-outfile");
54  auto treename = ic.options().get<std::string>("treename");
55 
56  auto outputfile = std::make_shared<TFile>(filename.c_str(), "RECREATE");
57  auto outputtree = std::make_shared<TTree>(treename.c_str(), treename.c_str());
58 
59  int* nCalls = new int;
60  *nCalls = 0;
61 
62  // the callback to be set as hook at stop of processing for the framework
63  auto finishWriting = [outputfile, outputtree, nCalls]() {
64  printf("finish writing with %d entries in the tree\n", *nCalls);
65  outputtree->SetEntries(*nCalls);
66  outputtree->Write();
67  outputfile->Close();
68  };
69  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
70 
71  // setup the processing function
72  // using by-copy capture of the worker instance shared pointer
73  // the shared pointer makes sure to clean up the instance when the processing
74  // function gets out of scope
75  auto processingFct = [outputfile, outputtree, useMC, nCalls](ProcessingContext& pc) {
76  static bool finished = false;
77  if (finished) {
78  // avoid being executed again when marked as finished;
79  return;
80  }
81  (*nCalls)++;
82  // retrieve the digits from the input
83  auto indata = pc.inputs().get<std::vector<o2::tof::Digit>*>("tofdigits");
84  LOG(INFO) << "Call " << *nCalls;
85  LOG(INFO) << "RECEIVED DIGITS SIZE " << indata->size();
86  auto row = pc.inputs().get<std::vector<o2::tof::ReadoutWindowData>*>("readoutwin");
87  LOG(INFO) << "RECEIVED READOUT WINDOWS " << row->size();
88 
89  auto digVect = indata.get();
90  auto rowVect = row.get();
91 
92  // connect this to a particular branch
93  auto brDig = getOrMakeBranch(*outputtree.get(), "TOFDigit", &digVect);
94  brDig->Fill();
95  auto brRow = getOrMakeBranch(*outputtree.get(), "TOFReadoutWindow", &rowVect);
96  brRow->Fill();
97 
98  // retrieve labels from the input
99  if (useMC) {
100  auto labeldata = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
101  // for (int i = 0; i < labeldata->size(); i++) {
102  // LOG(INFO) << "TOF GOT " << labeldata->at(i).getNElements() << " LABELS ";
103  // }
104  auto labeldataraw = labeldata.get();
105  // connect this to a particular branch
106  auto labelbr = getOrMakeBranch(*outputtree.get(), "TOFDigitMCTruth", &labeldataraw);
107  labelbr->Fill();
108  }
109 
110  // finished = true;
111  // pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
112  };
113 
114  // return the actual processing function as a lambda function using variables
115  // of the init function
116  return processingFct;
117  };
118 
119  std::vector<InputSpec> inputs;
120  inputs.emplace_back("tofdigits", o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
121  inputs.emplace_back("readoutwin", o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
122  if (useMC)
123  inputs.emplace_back("tofdigitlabels", o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
124 
125  return DataProcessorSpec{
126  "TOFDigitWriter",
127  inputs,
128  {}, // no output
129  AlgorithmSpec(initFunction),
130  Options{
131  {"tof-digit-outfile", VariantType::String, "tofdigits.root", {"Name of the input file"}},
132  {"treename", VariantType::String, "o2sim", {"Name of top-level TTree"}},
133  }};
134 }
