58  void run(framework::ProcessingContext& pc)
59  {
60  if (mFinished) {
61  return;
62  }
63  LOG(INFO) << "Doing ZDC digitization";
64 
65  // TODO: this should eventually come from the framework and depend on the TF timestamp
66  mDigitizer.refreshCCDB();
67 
68  // read collision context from input
69  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
70  context->initSimChains(o2::detectors::DetID::ZDC, mSimChains);
71 
72  const auto& grp = context->getGRP();
73  mDigitizer.setTimeStamp(grp.getTimeStart());
74 
75  auto& irecords = context->getEventRecords();
76  auto& eventParts = context->getEventParts();
77 
78  // loop over all composite collisions given from context
79  // (aka loop over all the interaction records)
80  std::vector<o2::zdc::Hit> hits;
81 
82  for (int collID = 0; collID < irecords.size(); ++collID) {
83 
84  const auto& irec = irecords[collID];
85  mDigitizer.setInteractionRecord(irec);
86 
87  for (auto& part : eventParts[collID]) {
88 
89  context->retrieveHits(mSimChains, "ZDCHit", part.sourceID, part.entryID, &hits);
90  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found ZDC " << hits.size() << " hits ";
91 
92  mDigitizer.setEventID(part.entryID);
93  mDigitizer.setSrcID(part.sourceID);
94 
95  mDigitizer.process(hits, mDigitsBC, mDigitsCh, mLabels);
96  }
97  }
98 
99  o2::InteractionTimeRecord terminateIR;
100  terminateIR.orbit = 0xffffffff; // supply IR in the infinite future to flush all cached BC
101  mDigitizer.setInteractionRecord(terminateIR);
102  mDigitizer.flush(mDigitsBC, mDigitsCh, mLabels);
103 
104  // send out to next stage
105  pc.outputs().snapshot(Output{"ZDC", "DIGITSBC", 0, Lifetime::Timeframe}, mDigitsBC);
106  pc.outputs().snapshot(Output{"ZDC", "DIGITSCH", 0, Lifetime::Timeframe}, mDigitsCh);
107  if (pc.outputs().isAllowed({"ZDC", "DIGITLBL", 0})) {
108  pc.outputs().snapshot(Output{"ZDC", "DIGITLBL", 0, Lifetime::Timeframe}, mLabels);
109  }
110 
111  LOG(INFO) << "ZDC: Sending ROMode= " << mROMode << " to GRPUpdater";
112  pc.outputs().snapshot(Output{"ZDC", "ROMode", 0, Lifetime::Timeframe}, mROMode);
113 
114  // we should be only called once; tell DPL that this process is ready to exit
115  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
116  mFinished = true;
117  }
118 
119  private:
120  bool mFinished = false;
121  Digitizer mDigitizer;
122  std::vector<TChain*> mSimChains;
123  std::vector<o2::zdc::ChannelData> mDigitsCh;
124  std::vector<o2::zdc::BCData> mDigitsBC;
125  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel> mLabels; // labels which get filled
126 
127  // RS: at the moment using hardcoded flag for continuous readout
128  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
129 };
