46 template <typename RAWDataHeader>
47 bool DecoderBaseT<RAWDataHeader>::processHBF()
48 {
49 
50 #ifdef DECODER_VERBOSE
51  if (mDecoderVerbose) {
52  std::cout << colorBlue
53  << "--- PROCESS HBF"
54  << colorReset
55  << std::endl;
56  }
57 #endif
58 
59  mDecoderRDH = reinterpret_cast<const RAWDataHeader*>(mDecoderPointer);
60  auto rdh = mDecoderRDH;
61 
63  while (!rdh->stop) {
64 
65 #ifdef DECODER_VERBOSE
66  if (mDecoderVerbose) {
67  std::cout << colorBlue
68  << "--- RDH open/continue detected"
69  << colorReset
70  << std::endl;
71  o2::raw::RDHUtils::printRDH(*rdh);
72  }
73 #endif
74 
76  rdhHandler(rdh);
77 
78  auto headerSize = rdh->headerSize;
79  auto memorySize = rdh->memorySize;
80  auto offsetToNext = rdh->offsetToNext;
81  auto drmPayload = memorySize - headerSize;
82 
84  std::memcpy(mDecoderSaveBuffer + mDecoderSaveBufferDataSize, reinterpret_cast<const char*>(rdh) + headerSize, drmPayload);
85  mDecoderSaveBufferDataSize += drmPayload;
86 
88  rdh = reinterpret_cast<const RAWDataHeader*>(reinterpret_cast<const char*>(rdh) + offsetToNext);
89 
91  if (reinterpret_cast<const char*>(rdh) < mDecoderBuffer + mDecoderBufferSize)
92  continue;
93 
95  return true;
96  }
97 
98 #ifdef DECODER_VERBOSE
99  if (mDecoderVerbose) {
100  std::cout << colorBlue
101  << "--- RDH close detected"
102  << colorReset
103  << std::endl;
104  o2::raw::RDHUtils::printRDH(*rdh);
105  }
106 #endif
107 
109  mDecoderPointer = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer);
110  mDecoderPointerMax = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer + mDecoderSaveBufferDataSize);
111  while (mDecoderPointer < mDecoderPointerMax) {
112  if (processDRM())
113  break;
114  }
115  mDecoderSaveBufferDataSize = 0;
116 
118  rdhHandler(rdh);
119 
120 #ifdef DECODER_VERBOSE
121  if (mDecoderVerbose) {
122  std::cout << colorBlue
123  << "--- END PROCESS HBF"
124  << colorReset
125  << std::endl;
126  }
127 #endif
128 
130  mDecoderPointer = reinterpret_cast<const uint32_t*>(reinterpret_cast<const char*>(rdh) + rdh->offsetToNext);
131 
133  if (reinterpret_cast<const char*>(mDecoderPointer) < mDecoderBuffer + mDecoderBufferSize)
134  return false;
135 
137  return true;
138 }

140 template <typename RAWDataHeader>
141 bool DecoderBaseT<RAWDataHeader>::processDRM()
142 {
143 
144 #ifdef DECODER_VERBOSE
145  if (mDecoderVerbose) {
146  std::cout << colorBlue << "--- PROCESS DRM"
147  << colorReset
148  << std::endl;
149  }
150 #endif
151 
152  if ((*mDecoderPointer & 0x80000000) != 0x80000000) {
153 #ifdef DECODER_VERBOSE
154  if (mDecoderVerbose) {
155  printf(" %08x [ERROR] \n ", *mDecoderPointer);
156  }
157 #endif
158  return true;
159  }
160 
162  auto crateHeader = reinterpret_cast<const CrateHeader_t*>(mDecoderPointer);
163 #ifdef DECODER_VERBOSE
164  if (mDecoderVerbose) {
165  printf(" %08x CrateHeader (drmID=%d) \n ", *mDecoderPointer, crateHeader->drmID);
166  }
167 #endif
168  mDecoderPointer++;
169 
171  auto crateOrbit = reinterpret_cast<const CrateOrbit_t*>(mDecoderPointer);
172 #ifdef DECODER_VERBOSE
173  if (mDecoderVerbose) {
174  printf(" %08x CrateOrbit (orbit=0x%08x) \n ", *mDecoderPointer, crateOrbit->orbitID);
175  }
176 #endif
177  mDecoderPointer++;
178 
180  headerHandler(crateHeader, crateOrbit);
181 
182  while (true) {
183 
185  if (*mDecoderPointer & 0x80000000) {
186  auto crateTrailer = reinterpret_cast<const CrateTrailer_t*>(mDecoderPointer);
187 #ifdef DECODER_VERBOSE
188  if (mDecoderVerbose) {
189  printf(" %08x CrateTrailer (numberOfDiagnostics=%d, numberOfErrors=%d) \n ", *mDecoderPointer, crateTrailer->numberOfDiagnostics, numberOfErrors);
190  }
191 #endif
192  mDecoderPointer++;
193  auto diagnostics = reinterpret_cast<const Diagnostic_t*>(mDecoderPointer);
194 #ifdef DECODER_VERBOSE
195  if (mDecoderVerbose) {
196  for (int i = 0; i < crateTrailer->numberOfDiagnostics; ++i) {
197  auto diagnostic = reinterpret_cast<const Diagnostic_t*>(mDecoderPointer + i);
198  printf(" %08x Diagnostic (slotId=%d) \n ", *(mDecoderPointer + i), diagnostic->slotID);
199  }
200  }
201 #endif
202  mDecoderPointer += crateTrailer->numberOfDiagnostics;
203  auto errors = reinterpret_cast<const Error_t*>(mDecoderPointer);
204 #ifdef DECODER_VERBOSE
205  if (mDecoderVerbose) {
206  for (int i = 0; i < crateTrailer->numberOfErrors; ++i) {
207  auto error = reinterpret_cast<const Error_t*>(mDecoderPointer + i);
208  printf(" %08x Error (slotId=%d) \n ", *(mDecoderPointer + i), error->slotID);
209  }
210  }
211 #endif
212  mDecoderPointer += crateTrailer->numberOfErrors;
213 
215  trailerHandler(crateHeader, crateOrbit, crateTrailer, diagnostics, errors);
216 
217  return false;
218  }
219 
221  auto frameHeader = reinterpret_cast<const FrameHeader_t*>(mDecoderPointer);
222 #ifdef DECODER_VERBOSE
223  if (mDecoderVerbose) {
224  printf(" %08x FrameHeader (numberOfHits=%d) \n ", *mDecoderPointer, frameHeader->numberOfHits);
225  }
226 #endif
227  mDecoderPointer++;
228  auto packedHits = reinterpret_cast<const PackedHit_t*>(mDecoderPointer);
229 #ifdef DECODER_VERBOSE
230  if (mDecoderVerbose) {
231  for (int i = 0; i < frameHeader->numberOfHits; ++i) {
232  auto packedHit = reinterpret_cast<const PackedHit_t*>(mDecoderPointer + 1);
233  printf(" %08x PackedHit (tdcID=%d) \n ", *(mDecoderPointer + 1), packedHit->tdcID);
234  packedHits++;
235  }
236  }
237 #endif
238  mDecoderPointer += frameHeader->numberOfHits;
239 
241  frameHandler(crateHeader, crateOrbit, frameHeader, packedHits);
242  }
243 
245  return false;
246 }
