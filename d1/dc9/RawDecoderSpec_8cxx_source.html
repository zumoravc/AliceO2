43  void init(of::InitContext& ic)
44  {
45  auto stop = [this]() {
46  LOG(INFO) << "Capacities: ROFRecords: " << mDecoder.getROFRecords().capacity() << " LocalBoards: " << mDecoder.getData().capacity();
47  double scaleFactor = 1.e6 / mNROFs;
48  LOG(INFO) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us decoding: " << mTimerAlgo.count() * scaleFactor << " us";
49  };
50  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
51 
52  auto feeIdConfigFilename = ic.options().get<std::string>("feeId-config-file");
53  if (!feeIdConfigFilename.empty()) {
54  o2::mid::FEEIdConfig feeIdConfig(feeIdConfigFilename.c_str());
55  mDecoder.setFeeIdConfig(feeIdConfig);
56  }
57  auto crateMasksFilename = ic.options().get<std::string>("crate-masks-file");
58  if (!crateMasksFilename.empty()) {
59  o2::mid::CrateMasks crateMasks(crateMasksFilename.c_str());
60  mDecoder.setCrateMasks(crateMasks);
61  }

63  auto isDebugMode = ic.options().get<bool>("debug-mode");
64  mDecoder.init(isDebugMode);
65  }

67  void run(of::ProcessingContext& pc)
68  {
69  auto tStart = std::chrono::high_resolution_clock::now();
70 
71  auto msg = pc.inputs().get("mid_raw");
72  auto buffer = of::DataRefUtils::as<const uint8_t>(msg);
73 
74  auto tAlgoStart = std::chrono::high_resolution_clock::now();
75  of::DPLRawParser parser(pc.inputs());
76 
77  mDecoder.clear();
78  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
79  // retrieving RDH v4
80  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
81  gsl::span<const uint8_t> payload(it.data(), it.size());
82  mDecoder.process(payload, *rdhPtr);
83  }
84 
85  mDecoder.flush();
86  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
87 
88  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODED", 0, of::Lifetime::Timeframe}, mDecoder.getData());
89  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODEDROF", 0, of::Lifetime::Timeframe}, mDecoder.getROFRecords());
90 
91  mTimer += std::chrono::high_resolution_clock::now() - tStart;
92  mNROFs += mDecoder.getROFRecords().size();
93  }
94 
95  private:
96  Decoder<GBTDECODER> mDecoder{};
97  std::chrono::duration<double> mTimer{0};
98  std::chrono::duration<double> mTimerAlgo{0};
99  unsigned int mNROFs{0};
100 };
