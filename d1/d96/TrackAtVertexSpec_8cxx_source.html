82  //_________________________________________________________________________________________________
83  void run(framework::ProcessingContext& pc)
84  {
87 
88  // get the vertex
89  Point3D<double> vertex(0., 0., 0.);
90  int eventVtx = readVertex(pc.inputs().get<gsl::span<char>>("vertex"), vertex);
91 
92  // get the tracks
93  std::list<TrackStruct> tracks{};
94  int eventTracks = readTracks(pc.inputs().get<gsl::span<char>>("tracks"), tracks);
95 
96  if (eventVtx > -1 && eventVtx != eventTracks) {
97  throw runtime_error("vertex and tracks are from different events");
98  }
99 
100  // propagate the tracks to the vertex
101  auto tStart = std::chrono::high_resolution_clock::now();
102  for (auto itTrack = tracks.begin(); itTrack != tracks.end();) {
103  if (extrapTrackToVertex(*itTrack, vertex)) {
104  ++itTrack;
105  } else {
106  itTrack = tracks.erase(itTrack);
107  }
108  }
109  auto tEnd = std::chrono::high_resolution_clock::now();
110  mElapsedTime += tEnd - tStart;
111 
112  // calculate the size of the payload for the output message, excluding the event header
113  int trackSize = getSize(tracks);
114 
115  // create the output message
116  auto msgOut = pc.outputs().make<char>(Output{"MCH", "TRACKSATVERTEX", 0, Lifetime::Timeframe}, 2 * SSizeOfInt + trackSize);
117  auto bufferPtrOut = msgOut.data();
118 
119  // write the event header
120  writeHeader(eventTracks, trackSize, bufferPtrOut);
121 
122  // write the tracks
123  if (trackSize > 0) {
124  writeTracks(tracks, bufferPtrOut);
125  }
126  }

139  int readVertex(const gsl::span<const char>& msgIn, Point3D<double>& vertex) const
140  {
143 
144  auto bufferPtr = msgIn.data();
145  int size = msgIn.size();
146 
147  if (size != SSizeOfInt + SSizeOfPoint3D) {
148  throw length_error("incorrect payload size");
149  }
150 
151  const int& event = *reinterpret_cast<const int*>(bufferPtr);
152  bufferPtr += SSizeOfInt;
153 
154  vertex = *reinterpret_cast<const Point3D<double>*>(bufferPtr);
155 
156  return event;
157  }

159  //_________________________________________________________________________________________________
160  int readTracks(const gsl::span<const char>& msgIn, std::list<TrackStruct>& tracks) const
161  {
164 
165  auto bufferPtr = msgIn.data();
166  int size = msgIn.size();
167 
168  if (size < 2 * SSizeOfInt) {
169  throw out_of_range("missing event header");
170  }
171  const int& event = *reinterpret_cast<const int*>(bufferPtr);
172  bufferPtr += SSizeOfInt;
173  int sizeLeft = *reinterpret_cast<const int*>(bufferPtr);
174  bufferPtr += SSizeOfInt;
175 
176  if (sizeLeft != size - 2 * SSizeOfInt) {
177  throw length_error("incorrect payload size");
178  }
179 
180  if (sizeLeft > 0) {
181 
182  if (sizeLeft < SSizeOfInt) {
183  throw out_of_range("missing number of tracks");
184  }
185  const int& nTracks = *reinterpret_cast<const int*>(bufferPtr);
186  bufferPtr += SSizeOfInt;
187  sizeLeft -= SSizeOfInt;
188 
189  for (int iTrack = 0; iTrack < nTracks; ++iTrack) {
190 
191  tracks.emplace_back();
192  auto& track = tracks.back();
193 
194  if (sizeLeft < SSizeOfTrackParamStruct) {
195  throw out_of_range("missing track parameters");
196  }
197  track.paramAt1stCluster = *reinterpret_cast<const TrackParamStruct*>(bufferPtr);
198  bufferPtr += SSizeOfTrackParamStruct;
199  sizeLeft -= SSizeOfTrackParamStruct;
200 
201  if (sizeLeft < SSizeOfDouble) {
202  throw out_of_range("missing chi2");
203  }
204  track.chi2 = *reinterpret_cast<const double*>(bufferPtr);
205  bufferPtr += SSizeOfDouble;
206  sizeLeft -= SSizeOfDouble;
207 
208  if (sizeLeft < SSizeOfInt) {
209  throw out_of_range("missing number of clusters");
210  }
211  const int& nClusters = *reinterpret_cast<const int*>(bufferPtr);
212  bufferPtr += SSizeOfInt;
213  sizeLeft -= SSizeOfInt;
214 
215  for (int iCl = 0; iCl < nClusters; ++iCl) {
216 
217  if (sizeLeft < SSizeOfClusterStruct) {
218  throw out_of_range("missing cluster");
219  }
220  track.clusters.emplace_back(*reinterpret_cast<const ClusterStruct*>(bufferPtr));
221  bufferPtr += SSizeOfClusterStruct;
222  sizeLeft -= SSizeOfClusterStruct;
223  }
224  }
225 
226  if (sizeLeft != 0) {
227  throw length_error("incorrect payload size");
228  }
229  }
230 
231  return event;
232  }

234  //_________________________________________________________________________________________________
235  bool extrapTrackToVertex(TrackStruct& track, Point3D<double>& vertex)
236  {
238 
239  // convert parameters at first cluster in internal format
240  TrackParam trackParam;
241  trackParam.setNonBendingCoor(track.paramAt1stCluster.x);
242  trackParam.setBendingCoor(track.paramAt1stCluster.y);
243  trackParam.setZ(track.paramAt1stCluster.z);
244  trackParam.setNonBendingSlope(track.paramAt1stCluster.px / track.paramAt1stCluster.pz);
245  trackParam.setBendingSlope(track.paramAt1stCluster.py / track.paramAt1stCluster.pz);
246  trackParam.setInverseBendingMomentum(track.paramAt1stCluster.sign / TMath::Sqrt(track.paramAt1stCluster.py * track.paramAt1stCluster.py + track.paramAt1stCluster.pz * track.paramAt1stCluster.pz));
247 
248  // extrapolate to vertex
249  TrackParam trackParamAtVertex(trackParam);
250  if (!TrackExtrap::extrapToVertex(&trackParamAtVertex, vertex.x(), vertex.y(), vertex.z(), 0., 0.)) {
251  return false;
252  }
253  track.paramAtVertex.x = trackParamAtVertex.getNonBendingCoor();
254  track.paramAtVertex.y = trackParamAtVertex.getBendingCoor();
255  track.paramAtVertex.z = trackParamAtVertex.getZ();
256  track.paramAtVertex.px = trackParamAtVertex.px();
257  track.paramAtVertex.py = trackParamAtVertex.py();
258  track.paramAtVertex.pz = trackParamAtVertex.pz();
259  track.paramAtVertex.sign = trackParamAtVertex.getCharge();
260 
261  // extrapolate to DCA
262  TrackParam trackParamAtDCA(trackParam);
263  if (!TrackExtrap::extrapToVertexWithoutBranson(&trackParamAtDCA, vertex.z())) {
264  return false;
265  }
266  double dcaX = trackParamAtDCA.getNonBendingCoor() - vertex.x();
267  double dcaY = trackParamAtDCA.getBendingCoor() - vertex.y();
268  track.dca = TMath::Sqrt(dcaX * dcaX + dcaY * dcaY);
269 
270  // extrapolate to the end of the absorber
271  if (!TrackExtrap::extrapToZ(&trackParam, -505.)) {
272  return false;
273  }
274  double xAbs = trackParam.getNonBendingCoor();
275  double yAbs = trackParam.getBendingCoor();
276  track.rAbs = TMath::Sqrt(xAbs * xAbs + yAbs * yAbs);
277 
278  return true;
279  }

282  int getSize(const std::list<TrackStruct>& tracks)
283  {
285 
286  int size(0);
287  for (const auto& track : tracks) {
288  size += 2 * SSizeOfTrackParamStruct + 3 * SSizeOfDouble + SSizeOfInt + track.clusters.size() * SSizeOfClusterStruct;

311  //_________________________________________________________________________________________________
312  void writeTracks(const std::list<TrackStruct>& tracks, char*& bufferPtr) const
313  {
315 
316  // write the number of tracks
317  int nTracks = tracks.size();
318  memcpy(bufferPtr, &nTracks, SSizeOfInt);
319  bufferPtr += SSizeOfInt;
320 
321  for (const auto& track : tracks) {
322 
323  // write track parameters at vertex
324  memcpy(bufferPtr, &(track.paramAtVertex), SSizeOfTrackParamStruct);
325  bufferPtr += SSizeOfTrackParamStruct;
326 
327  // write dca
328  memcpy(bufferPtr, &(track.dca), SSizeOfDouble);
329  bufferPtr += SSizeOfDouble;
330 
331  // write rAbs
332  memcpy(bufferPtr, &(track.rAbs), SSizeOfDouble);
333  bufferPtr += SSizeOfDouble;
334 
335  // write track parameters at first cluster
336  memcpy(bufferPtr, &(track.paramAt1stCluster), SSizeOfTrackParamStruct);
337  bufferPtr += SSizeOfTrackParamStruct;
338 
339  // write track chi2
340  memcpy(bufferPtr, &(track.chi2), SSizeOfDouble);
341  bufferPtr += SSizeOfDouble;
342 
343  // write the number of clusters
344  int nClusters = track.clusters.size();
345  memcpy(bufferPtr, &nClusters, SSizeOfInt);
346  bufferPtr += SSizeOfInt;
347 
348  // write clusters
349  for (const auto& cluster : track.clusters) {
350  memcpy(bufferPtr, &cluster, SSizeOfClusterStruct);
351  bufferPtr += SSizeOfClusterStruct;
352  }
353  }
354  }
355 
356  static constexpr int SSizeOfInt = sizeof(int);

358  static constexpr int SSizeOfTrackParamStruct = sizeof(TrackParamStruct);
359  static constexpr int SSizeOfClusterStruct = sizeof(ClusterStruct);
360  static constexpr int SSizeOfPoint3D = sizeof(Point3D<double>);
361 
362  std::chrono::duration<double> mElapsedTime{};
363 };
