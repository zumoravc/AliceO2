25 class PackedCharge
26 {
27  public:
28  using BasicType = unsigned short;
29  static_assert(sizeof(BasicType) == 2);
30 
31  enum Constants : BasicType {
32  Null = 0,
33  ADCBits = 10,
34  DecimalBits = 4,
35  ChargeBits = ADCBits + DecimalBits,
36  ChargeMask = (1 << ChargeBits) - 1,
37  MaxVal = ChargeMask,
38  Has3x3PeakMask = 1 << ChargeBits,
39  IsSplitMask = 1 << (ChargeBits + 1),
40  };
41 
42  GPUdDefault() PackedCharge() CON_DEFAULT;
43  GPUdi() explicit PackedCharge(tpccf::Charge q) : PackedCharge(q, false, false) {}
44  GPUdi() PackedCharge(tpccf::Charge q, bool peak3x3, bool wasSplit)
45  {
46  mVal = q * tpccf::Charge(1 << DecimalBits);
47  mVal = CAMath::Min<BasicType>(MaxVal, mVal); // ensure only lower 14 bits are set
48  mVal |= (peak3x3) ? Has3x3PeakMask : Null;
49  mVal |= (wasSplit) ? IsSplitMask : Null;
50  }
51 
52  GPUdi() tpccf::Charge unpack() const { return tpccf::Charge(mVal & ChargeMask) / tpccf::Charge(1 << DecimalBits); }
53  GPUdi() bool has3x3Peak() const { return mVal & Has3x3PeakMask; }
54  GPUdi() bool isSplit() const { return mVal & IsSplitMask; }
55 
56  private:
57  BasicType mVal;
58 };
