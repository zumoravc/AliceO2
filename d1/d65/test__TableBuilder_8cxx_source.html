246 {
247  using namespace o2::framework;
248  TableBuilder builder1;
249  auto rowWriter1 = builder1.persist<int, int, int>({"x", "y", "event"});

258  rowWriter2(0, i, i);
259  }
260  auto table2 = builder2.finalize();
261  BOOST_REQUIRE_EQUAL(table2->num_columns(), 2);
262  BOOST_REQUIRE_EQUAL(table2->num_rows(), 8);
263  for (size_t i = 0; i < 8; ++i) {
264  auto p2 = std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int32Type>>(table2->column(0)->chunk(0));
265  BOOST_CHECK_EQUAL(p2->Value(i), i);
266  }
267 
268  auto source1 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
269  auto source2 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
270  auto cross = ROOT::RDF::MakeCrossProductDataFrame(std::move(source1), std::move(source2));
271  auto source3 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
272  auto source4 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
273  auto indexed = ROOT::RDF::MakeColumnIndexedDataFrame(std::move(source3), std::move(source4), "x");
274  auto source5 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
275  auto source6 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
276  auto unionDS = ROOT::RDF::MakeFriendDataFrame(std::move(source5), std::move(source6));
277  auto source7 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
278  auto source8 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{}); // Notice the table needs to be the same
279  auto blockDS = ROOT::RDF::MakeBlockAntiDataFrame(std::move(source7), std::move(source8), "event");
280 
281  ROOT::RDataFrame finalDF{std::move(cross)};
282  ROOT::RDataFrame indexedDF{std::move(indexed)};
283  ROOT::RDataFrame unionDF{std::move(unionDS)};
284  ROOT::RDataFrame blockDF{std::move(blockDS)};
285 
286  BOOST_CHECK_EQUAL(*finalDF.Count(), 64); // Full cross product of 8x8 rows, 64 entries
287  BOOST_CHECK_EQUAL(*indexedDF.Count(), 8); // Indexing the left table using a column of the right table

289  BOOST_CHECK_EQUAL(*unionDF.Count(), 8); // Pairing one by one the rows of the two tables, still 8
