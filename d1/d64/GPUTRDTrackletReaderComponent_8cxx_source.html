132 int GPUTRDTrackletReaderComponent::DoInit(int argc, const char** argv)
133 {
134 
135  int iResult = 0;
136 
137  do {
138 
139  fRawReaderMem = new AliRawReaderMemory;
140  if (!fRawReaderMem) {
141  iResult = -ENOMEM;
142  break;
143  }
144 
145  fTrackletArray = new TClonesArray("AliTRDtrackletWord", 1000);
146  if (!fTrackletArray) {
147  iResult = -ENOMEM;
148  break;
149  }
150 
151  fRawReaderTrd = new AliTRDrawStream(fRawReaderMem);
152  if (!fRawReaderTrd) {
153  iResult = -ENOMEM;
154  break;
155  }
156 
157  fRawReaderTrd->SetTrackletArray(fTrackletArray);
158 
159  // Disable raw reader error messages that could flood HLT logbook
160  AliLog::SetClassDebugLevel("AliTRDrawStream", 0);
161  fRawReaderTrd->SetErrorDebugLevel(AliTRDrawStream::kLinkMonitor, 1);
162 
163  } while (0);
164 
165  if (iResult < 0) {
166 
167  if (fRawReaderTrd) {
168  delete fRawReaderTrd;
169  }
170  fRawReaderTrd = nullptr;
171 
172  if (fRawReaderMem) {
173  delete fRawReaderMem;
174  }
175  fRawReaderMem = nullptr;
176 
177  if (fTrackletArray) {
178  delete fTrackletArray;
179  }
180  fTrackletArray = nullptr;
181  }
182 
183  vector<const char*> remainingArgs;
184  for (int i = 0; i < argc; ++i) {
185  remainingArgs.push_back(argv[i]);
186  }
187 
188  if (argc > 0) {
189  ConfigureFromArgumentString(remainingArgs.size(), &(remainingArgs[0]));
190  }
191 
192  return iResult;
193 }

240 int GPUTRDTrackletReaderComponent::DoEvent(const AliHLTComponentEventData& hltEventData, AliHLTComponentTriggerData& /*trigData*/)
241 {
242 
243  fEventId = hltEventData.fEventID;
244 
245  HLTInfo("### START DoEvent [event id: %llu, %d blocks, size: %d]", hltEventData.fEventID, hltEventData.fBlockCnt, hltEventData.fStructSize);
246 
247  // event processing function
248  int iResult = 0;
249 
250  fTrackletArray->Clear();
251  fRawReaderMem->ClearBuffers();
252 
253  if (!IsDataEvent()) { // process data events only
254  HLTInfo("### END DoEvent [event id: %llu, %d blocks, size: %d] (skipped: no data event)", hltEventData.fEventID, hltEventData.fBlockCnt, hltEventData.fStructSize);
255  return iResult;
256  }
257 
258  std::vector<GPUTRDTrackletWord> outputTrkls;
259  std::vector<GPUTRDTrackletLabels> outputTrklsMC;
260 
261  { // read raw data
262 
263  TString infoStr("");
264  unsigned int sourceSectors = 0;
265 
266  // loop over all incoming TRD raw data blocks
267  for (const AliHLTComponentBlockData* pBlock = GetFirstInputBlock(kAliHLTDataTypeDDLRaw | kAliHLTDataOriginTRD); pBlock != nullptr && iResult >= 0; pBlock = GetNextInputBlock()) {
268 
269  int trdSector = -1;
270 
271  // determine sector from block specification
272  for (unsigned pos = 0; pos < 8 * sizeof(AliHLTUInt32_t); pos++) {
273  if (pBlock->fSpecification & (0x1 << pos)) {
274  if (trdSector >= 0) {
275  HLTWarning("Cannot uniquely identify DDL number from specification, skipping data block %s 0x%08x", DataType2Text(pBlock->fDataType).c_str(), pBlock->fSpecification);
276  trdSector = -1;
277  break;
278  }
279  trdSector = pos;
280  }
281  }
282  if (trdSector < 0) {
283  continue;
284  }
285 
286  // add data block to rawreader
287  infoStr += Form("%02d, ", trdSector);
288  sourceSectors |= pBlock->fSpecification;
289  if (!fRawReaderMem->AddBuffer((unsigned char*)pBlock->fPtr, pBlock->fSize, trdSector + 1024)) {
290  LogError("Could not add buffer of data block %s, 0x%08x to rawreader", DataType2Text(pBlock->fDataType).c_str(), pBlock->fSpecification);
291  continue;
292  }
293  } // loop over all incoming TRD raw data blocks
294 
295  if (sourceSectors) {
296  infoStr.Remove(infoStr.Length() - 2, 2);
297  LogDebug("preprocessing raw data from sectors: %s...", infoStr.Data());
298 
299  // extract header info and TRD tracklets from raw data
300  fRawReaderTrd->ReadEvent();
301 
302  // read and process TRD tracklets
303  int nTracklets = fTrackletArray->GetEntriesFast();
304 
305  HLTInfo("There are %i tracklets in this event\n", nTracklets);
306  for (int iTracklet = 0; iTracklet < nTracklets; ++iTracklet) {
307  GPUTRDTrackletWord trkl = *((AliTRDtrackletWord*)fTrackletArray->At(iTracklet));
308  trkl.SetId(iTracklet);
309  outputTrkls.push_back(trkl);
310  }
311  LogDebug("pushing data for sectors: 0x%05x", sourceSectors);
312  }
313  fRawReaderMem->ClearBuffers();
314  }
315 
316  { // loop over all incoming TRD MC tracklets data blocks
317 
318  for (const TObject* iter = GetFirstInputObject(kAliHLTDataTypeAliTreeD | kAliHLTDataOriginTRD); iter != nullptr; iter = GetNextInputObject()) {
319  TTree* trackletTree = dynamic_cast<TTree*>(const_cast<TObject*>(iter));
320  if (!trackletTree) {
321  HLTFatal("No Tracklet Tree found");
322  return -EINVAL;
323  }
324 
325  TBranch* trklbranch = trackletTree->GetBranch("mcmtrklbranch");
326  if (!trklbranch) {
327  HLTFatal("No tracklet branch found in tracklet tree");
328  return -EINVAL;
329  }
330  int nTracklets = trklbranch->GetEntries();
331  HLTInfo("Input tree with %d TRD MCM tracklets", nTracklets);
332 
333  //-----------------------------------
334  // Deploy same hack as in ITS Clusterizer
335  AliRunLoader* pRunLoader = AliRunLoader::Instance();
336  if (!pRunLoader) {
337  HLTError("failed to get global runloader instance");
338  return -ENOSYS;
339  }
340  pRunLoader->GetEvent(GetEventCount());
341  const char* loaderType = "TRDLoader";
342  AliLoader* pLoader = pRunLoader->GetLoader(loaderType);
343  if (!pLoader) {
344  HLTError("can not get loader \"%s\" from runloader", loaderType);
345  return -ENOSYS;
346  }
347  pLoader->LoadDigits("read");
348  AliDataLoader* dataLoader = pLoader->GetDataLoader("tracklets");
349  if (dataLoader) {
350  trackletTree = dataLoader->Tree();
351  dataLoader->Load("read");
352  } else {
353  HLTWarning("TRD tracklet loader not found");
354  }
355  trklbranch = trackletTree->GetBranch("mcmtrklbranch");
356  if (!trklbranch) {
357  HLTFatal("No tracklet branch found in tracklet tree");
358  return -EINVAL;
359  }
360  if (trklbranch->GetEntries() != nTracklets) {
361  HLTFatal("Incorrect number of tracklets in tree");
362  return -EINVAL;
363  }
364  //-----------------------------------
365 
366  AliTRDtrackletMCM* trkl = 0x0;
367  trklbranch->SetAddress(&trkl);
368 
369  for (int iTracklet = 0; iTracklet < nTracklets; iTracklet++) {
370  int nbytes = trklbranch->GetEntry(iTracklet, 1);
371  if (!trkl || nbytes <= 0) {
372  HLTWarning("Can not read entry from tracklet branch");
373  continue;
374  }
375  GPUTRDTrackletWord hltTrkl = *trkl;
376  hltTrkl.SetId(iTracklet);
377  outputTrkls.push_back(hltTrkl);
378  GPUTRDTrackletLabels trklMC;
379  trklMC.mLabel[0] = trkl->GetLabel(0);
380  trklMC.mLabel[1] = trkl->GetLabel(1);
381  trklMC.mLabel[2] = trkl->GetLabel(2);
382  outputTrklsMC.push_back(trklMC);
383  }
384  }
385  }
386 
387  if (outputTrkls.size() > 0) {
388  iResult = PushBack(&outputTrkls[0], outputTrkls.size() * sizeof(outputTrkls[0]), AliHLTTRDDefinitions::fgkTRDTrackletDataType, 0);
389  }
390  if (outputTrklsMC.size() > 0) {
391  iResult = PushBack(&outputTrklsMC[0], outputTrklsMC.size() * sizeof(outputTrklsMC[0]), AliHLTTRDDefinitions::fgkTRDMCTrackletDataType, 0);
392  }
393 
394  HLTInfo("### END DoEvent [event id: %llu, %d blocks, size: %d, output tracklets: %d]", hltEventData.fEventID, hltEventData.fBlockCnt, hltEventData.fStructSize, outputTrkls.size());
395 
396  return iResult;
397 }
