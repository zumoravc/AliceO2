47 void Dispatcher::init(InitContext& ctx)
48 {
49  LOG(DEBUG) << "Reading Data Sampling Policies...";
50 
51  std::unique_ptr<ConfigurationInterface> cfg = ConfigurationFactory::getConfiguration(mReconfigurationSource);
52  auto policiesTree = cfg->getRecursive("dataSamplingPolicies");
53  mPolicies.clear();
54 
55  for (auto&& policyConfig : policiesTree) {
56  // we don't want the Dispatcher to exit due to one faulty Policy
57  try {
58  mPolicies.emplace_back(std::make_shared<DataSamplingPolicy>(policyConfig.second));
59  } catch (std::exception& ex) {
60  LOG(WARN) << "Could not load the Data Sampling Policy '"
61  << policyConfig.second.get_optional<std::string>("id").value_or("") << "', because: " << ex.what();
62  } catch (...) {
63  LOG(WARN) << "Could not load the Data Sampling Policy '"
64  << policyConfig.second.get_optional<std::string>("id").value_or("") << "'";
65  }
66  }
67 }

69 void Dispatcher::run(ProcessingContext& ctx)
70 {
71  for (const auto& input : ctx.inputs()) {
72  if (input.header != nullptr && input.spec != nullptr) {
73  const auto* inputHeader = header::get<header::DataHeader*>(input.header);
74  ConcreteDataMatcher inputMatcher{inputHeader->dataOrigin, inputHeader->dataDescription, inputHeader->subSpecification};
75 
76  for (auto& policy : mPolicies) {
77  // todo: consider getting the outputSpec in match to improve performance
78  // todo: consider matching (and deciding) in completion policy to save some time
79 
80  if (policy->match(inputMatcher) && policy->decide(input)) {
81  // We copy every header which is not DataHeader or DataProcessingHeader,
82  // so that custom data-dependent headers are passed forward,
83  // and we add a DataSamplingHeader.
84  header::Stack headerStack{
85  std::move(extractAdditionalHeaders(input.header)),
86  std::move(prepareDataSamplingHeader(*policy.get(), ctx.services().get<const DeviceSpec>()))};
87 
88  if (!policy->getFairMQOutputChannel().empty()) {
89  sendFairMQ(ctx.services().get<RawDeviceService>().device(), input, policy->getFairMQOutputChannelName(),
90  std::move(headerStack));
91  } else {
92  Output output = policy->prepareOutput(inputMatcher, input.spec->lifetime);
93  output.metaHeader = std::move(header::Stack{std::move(output.metaHeader), std::move(headerStack)});
94  send(ctx.outputs(), input, std::move(output));
95  }
96  }
97  }
98  }
99  }
100 
101  if (ctx.inputs().isValid("timer-stats")) {
102  reportStats(ctx.services().get<Monitoring>());
103  }
104 }

106 void Dispatcher::reportStats(Monitoring& monitoring) const
107 {
108  uint64_t dispatcherTotalEvaluatedMessages = 0;
109  uint64_t dispatcherTotalAcceptedMessages = 0;
110 
111  for (const auto& policy : mPolicies) {
112  dispatcherTotalEvaluatedMessages += policy->getTotalEvaluatedMessages();
113  dispatcherTotalAcceptedMessages += policy->getTotalAcceptedMessages();
114  }
115 
116  monitoring.send({dispatcherTotalEvaluatedMessages, "Dispatcher_messages_evaluated"});
117  monitoring.send({dispatcherTotalAcceptedMessages, "Dispatcher_messages_passed"});
118 }

120 DataSamplingHeader Dispatcher::prepareDataSamplingHeader(const DataSamplingPolicy& policy, const DeviceSpec& spec)
121 {
122  uint64_t sampleTime = static_cast<uint64_t>(std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
123 
124  DataSamplingHeader::DeviceIDType id;
125  id.runtimeInit(spec.id.substr(0, DataSamplingHeader::deviceIDTypeSize).c_str());
126 
127  return {
128  sampleTime,
129  policy.getTotalAcceptedMessages(),
130  policy.getTotalEvaluatedMessages(),
131  id};

134 header::Stack Dispatcher::extractAdditionalHeaders(const char* inputHeaderStack) const
135 {
136  header::Stack headerStack;
137 
138  const auto* first = header::BaseHeader::get(reinterpret_cast<const byte*>(inputHeaderStack));
139  for (const auto* current = first; current != nullptr; current = current->next()) {
140  if (current->description != header::DataHeader::sHeaderType &&
141  current->description != DataProcessingHeader::sHeaderType) {
142  headerStack = std::move(header::Stack{std::move(headerStack), *current});
143  }
144  }
145 
146  return headerStack;
147 }

155 // ideally this should be in a separate proxy device or use Lifetime::External
156 void Dispatcher::sendFairMQ(FairMQDevice* device, const DataRef& inputData, const std::string& fairMQChannel,
157  header::Stack&& stack) const
158 {
159  const auto* dh = header::get<header::DataHeader*>(inputData.header);
160  assert(dh);
161  const auto* dph = header::get<DataProcessingHeader*>(inputData.header);
162  assert(dph);
163 
164  header::DataHeader dhout{dh->dataDescription, dh->dataOrigin, dh->subSpecification, dh->payloadSize};
165  dhout.payloadSerializationMethod = dh->payloadSerializationMethod;
166  DataProcessingHeader dphout{dph->startTime, dph->duration};
167  o2::header::Stack headerStack{dhout, dphout, stack};
168 
169  auto channelAlloc = o2::pmr::getTransportAllocator(device->Transport());
170  FairMQMessagePtr msgHeaderStack = o2::pmr::getMessage(std::move(headerStack), channelAlloc);
171 
172  char* payloadCopy = new char[dh->payloadSize];
173  memcpy(payloadCopy, inputData.payload, dh->payloadSize);
174  auto cleanupFcn = [](void* data, void*) { delete[] reinterpret_cast<char*>(data); };
175  FairMQMessagePtr msgPayload(device->NewMessage(payloadCopy, dh->payloadSize, cleanupFcn, payloadCopy));
176 
177  FairMQParts message;
178  message.AddPart(move(msgHeaderStack));
179  message.AddPart(move(msgPayload));
180 
181  int64_t bytesSent = device->Send(message, fairMQChannel);
182 }

184 void Dispatcher::registerPath(const std::pair<InputSpec, OutputSpec>& path)
185 {
186  //todo: take care of inputs inclusive in others, when subSpec matchers are supported
187  auto cmp = [a = path.first](const InputSpec b) {
188  return a.matcher == b.matcher && a.lifetime == b.lifetime;
189  };
190 
191  if (std::find_if(inputs.begin(), inputs.end(), cmp) == inputs.end()) {
192  inputs.push_back(path.first);
193  LOG(DEBUG) << "Registering input " << DataSpecUtils::describe(path.first);
194  } else {
195  LOG(DEBUG) << "Input " << DataSpecUtils::describe(path.first)
196  << " already registered";
197  }
198 
199  outputs.push_back(path.second);
200 }
