26 LogLevel LogParsingHelpers::parseTokenLevel(std::string_view const s)
27 {
28 
29  // Example format: [99:99:99][ERROR] (string begins with that, longest is 17 chars)
30  constexpr size_t MAXPREFLEN = 17;
31  constexpr size_t LABELPOS = 10;
32  if (s.size() < MAXPREFLEN) {
33  return LogLevel::Unknown;
34  }
35 
36  // Check if first chars match [NN:NN:NN]
37  // 0123456789
38  if ((unsigned char)s[0] != '[' || (unsigned char)s[9] != ']' ||
39  (unsigned char)s[3] != ':' || (unsigned char)s[6] != ':' ||
40  (unsigned char)s[1] - '0' > 9 || (unsigned char)s[2] - '0' > 9 ||
41  (unsigned char)s[4] - '0' > 9 || (unsigned char)s[5] - '0' > 9 ||
42  (unsigned char)s[7] - '0' > 9 || (unsigned char)s[8] - '0' > 9) {
43  return LogLevel::Unknown;
44  }
45 
46  if (s.compare(LABELPOS, 8, "[DEBUG] ") == 0) {
47  return LogLevel::Debug;
48  } else if (s.compare(LABELPOS, 7, "[INFO] ") == 0 ||
49  s.compare(LABELPOS, 8, "[STATE] ") == 0) {
50  return LogLevel::Info;
51  } else if (s.compare(LABELPOS, 7, "[WARN] ") == 0) {
52  return LogLevel::Warning;
53  } else if (s.compare(LABELPOS, 8, "[ERROR] ") == 0) {
54  return LogLevel::Error;
55  }
56  return LogLevel::Unknown;
57 }
