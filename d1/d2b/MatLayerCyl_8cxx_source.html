54 void MatLayerCyl::initSegmentation(float rMin, float rMax, float zHalfSpan, int nz, int nphi)
55 {
56  // Init and precalculate aux parameters. The initialization is done in the own memory.
57  assert(mConstructionMask == NotConstructed);
58  assert(rMin < rMax);
59  assert(nz > 0);
60  assert(nphi > 0);
61 
62  // book local storage
63  auto sz = estimateFlatBufferSize(nphi, nphi, nz);
64 
65  //--------------????
66  mFlatBufferPtr = mFlatBufferContainer = new char[sz];
67  mFlatBufferSize = sz;
68  //--------------????
69 
70  mRMin2 = rMin * rMin;
71  mRMax2 = rMax * rMax;
72  mZHalf = zHalfSpan;
73  mNZBins = nz;
74 
75  mDZ = 2. * zHalfSpan / nz;
76  mDZInv = 1.f / mDZ;
77 
78  mDPhi = o2::constants::math::TwoPI / nphi;
79  mDPhiInv = 1.f / mDPhi;
80  //
81  int offs = 0;
82 
83  o2::gpu::resizeArray(mPhiBin2Slice, 0, nphi, reinterpret_cast<short*>(mFlatBufferPtr + offs));
84  mNPhiSlices = mNPhiBins = nphi;
85 
86  for (int i = nphi; i--;) {
87  mPhiBin2Slice[i] = i; // fill with trivial mapping
88  }
89 
90  offs = alignSize(offs + nphi * sizeof(short), getBufferAlignmentBytes()); // account for alignment
91 
92  o2::gpu::resizeArray(mSliceCos, 0, nphi, reinterpret_cast<float*>(mFlatBufferPtr + offs)); // in the beginning nslice = nphi
93  offs = alignSize(offs + nphi * sizeof(float), getBufferAlignmentBytes()); // account for alignment
94 
95  o2::gpu::resizeArray(mSliceSin, 0, nphi, reinterpret_cast<float*>(mFlatBufferPtr + offs)); // in the beginning nslice = nphi
96  offs = alignSize(offs + nphi * sizeof(float), getBufferAlignmentBytes()); // account for alignment
97 
98  for (int i = nphi; i--;) {
99  mSliceCos[i] = std::cos(getPhiBinMin(i));
100  mSliceSin[i] = std::sin(getPhiBinMin(i));
101  }
102

122 //________________________________________________________________________________
123 void MatLayerCyl::populateFromTGeo(int ip, int iz, int ntrPerCell)
124 {
126 
127  float zmn = getZBinMin(iz), phmn = getPhiBinMin(ip), sn, cs, rMin = getRMin(), rMax = getRMax();
128  double meanRho = 0., meanX2X0 = 0., lgt = 0.;
129  ;
130  float dz = getDZ() / ntrPerCell;
131  for (int isz = ntrPerCell; isz--;) {
132  float zs = zmn + (isz + 0.5) * dz;
133  float dzt = zs > 0.f ? 0.25 * dz : -0.25 * dz; // to avoid 90 degree polar angle
134  for (int isp = ntrPerCell; isp--;) {
135  o2::utils::sincosf(phmn + (isp + 0.5) * getDPhi() / ntrPerCell, sn, cs);
136  auto bud = o2::base::GeometryManager::meanMaterialBudget(rMin * cs, rMin * sn, zs - dzt, rMax * cs, rMax * sn, zs + dzt);
137  if (bud.length > 0.) {
138  meanRho += bud.length * bud.meanRho;
139  meanX2X0 += bud.meanX2X0; // we store actually not X2X0 but 1./X0
140  lgt += bud.length;
141  }
142  }
143  }
144  if (lgt > 0.) {
145  auto& cell = mCells[getCellIDPhiBin(ip, iz)];
146  cell.meanRho = meanRho / lgt; // mean rho
147  cell.meanX2X0 = meanX2X0 / lgt; // mean 1./X0 seen in this cell
148  }
149 }

152 bool MatLayerCyl::canMergePhiSlices(int i, int j, float maxRelDiff, int maxDifferent) const
153 {
154  if (std::abs(i - j) > 1 || i == j || std::max(i, j) >= getNPhiSlices()) {
155  LOG(ERROR) << "Only existing " << getNPhiSlices() << " slices with diff. of 1 can be merged, input is " << i << " and " << j;
156  return false;
157  }
158  int ndiff = 0; // number of different cells
159  for (int iz = getNZBins(); iz--;) {
160  const auto& cellI = getCellPhiBin(i, iz);
161  const auto& cellJ = getCellPhiBin(j, iz);
162  if (cellsDiffer(cellI, cellJ, maxRelDiff)) {
163  if (++ndiff > maxDifferent) {
164  return false;
165  }
166  }
167  }
168  return true;
169 }

171 //________________________________________________________________________________
172 bool MatLayerCyl::cellsDiffer(const MatCell& cellA, const MatCell& cellB, float maxRelDiff) const
173 {
175  float rav = 0.5 * (cellA.meanRho + cellB.meanRho), xav = 0.5 * (cellA.meanX2X0 + cellB.meanX2X0);
176  float rdf = 0.5 * (cellA.meanRho - cellB.meanRho), xdf = 0.5 * (cellA.meanX2X0 - cellB.meanX2X0);
177  if (rav > 0 && std::abs(rdf / rav) > maxRelDiff)
178  return true;
179  if (xav > 0 && std::abs(xdf / xav) > maxRelDiff)
180  return true;
181  return false;
182 }

185 void MatLayerCyl::optimizePhiSlices(float maxRelDiff)
186 {
187  // merge compatible phi slices
188  if (getNPhiSlices() < getNPhiBins()) {
189  LOG(ERROR) << getNPhiBins() << " phi bins were already merged to " << getNPhiSlices() << " slices";
190  return;
191  }
192  int newSl = 0;
193  for (int is = 1; is < getNPhiSlices(); is++) {
194  if (!canMergePhiSlices(is - 1, is, maxRelDiff)) {
195  newSl++;
196  }
197  mPhiBin2Slice[is] = newSl;
198  }
199  LOG(INFO) << newSl + 1 << " slices out of " << getNPhiBins();
200  if (newSl + 1 == getNPhiSlices()) {
201  return;
202  }
203  newSl = 0;
204  int slMin = 0, slMax = 0, is = 0;
205  while (is++ < getNPhiSlices()) {
206  while (is < getNPhiSlices() && mPhiBin2Slice[is] == newSl) { // select similar slices
207  slMax++;
208  is++;
209  }
210  if (slMax > slMin || newSl != slMin) { // merge or shift slices
211  mSliceCos[newSl] = mSliceCos[slMin];
212  mSliceSin[newSl] = mSliceSin[slMin];
213  float norm = 1.f / (1.f + slMax - slMin);
214  for (int iz = getNZBins(); iz--;) {
215  int iDest = newSl * getNZBins() + iz, iSrc = slMin * getNZBins() + iz;
216  mCells[iDest] = mCells[iSrc];
217  for (int ism = slMin + 1; ism <= slMax; ism++) {
218  iSrc = ism * getNZBins() + iz;
219  mCells[iDest].meanX2X0 += mCells[iSrc].meanX2X0;
220  mCells[iDest].meanRho += mCells[iSrc].meanRho;
221  }
222  mCells[iDest].scale(norm);
223  }
224  LOG(INFO) << "mapping " << slMin << ":" << slMax << " to new slice " << newSl;
225  }
226  newSl++;
227  slMin = slMax = is;
228  }
229  mNPhiSlices = newSl;
230 
231  // relocate arrays to avoid spaces after optimization
232  // mSliceCos pointer does not change, but sliceSin needs to be relocated
233  auto offs = alignSize(newSl * sizeof(float), getBufferAlignmentBytes());
234  char* dst = ((char*)mSliceCos) + offs; // account for alignment
235  o2::gpu::resizeArray(mSliceSin, getNPhiBins(), newSl, reinterpret_cast<float*>(dst));
236  // adjust mCells array
237  dst = ((char*)mSliceSin) + offs; // account for alignment
238  o2::gpu::resizeArray(mCells, getNPhiBins() * getNZBins(), newSl * getNZBins(), reinterpret_cast<MatCell*>(dst));
239  mFlatBufferSize = estimateFlatBufferSize();
240  LOG(INFO) << "Updated Nslices = " << getNPhiSlices();
241 }

244 void MatLayerCyl::getMeanRMS(MatCell& mean, MatCell& rms) const
245 {
246  // mean and RMS over layer
247  mean.meanRho = rms.meanRho = 0.f;
248  mean.meanX2X0 = rms.meanX2X0 = 0.f;
249  for (int ip = getNPhiBins(); ip--;) {
250  for (int iz = getNZBins(); iz--;) {
251  const auto& cell = getCellPhiBin(ip, iz);
252  mean.meanRho += cell.meanRho;
253  mean.meanX2X0 += cell.meanX2X0;
254  rms.meanRho += cell.meanRho * cell.meanRho;
255  rms.meanX2X0 += cell.meanX2X0 * cell.meanX2X0;
256  }
257  }
258  int nc = getNPhiBins() * getNZBins();
259  mean.meanRho /= nc;
260  mean.meanX2X0 /= nc;
261  rms.meanRho /= nc;
262  rms.meanX2X0 /= nc;
263  rms.meanRho -= mean.meanRho * mean.meanRho;
264  rms.meanX2X0 -= mean.meanX2X0 * mean.meanX2X0;
265  rms.meanRho = rms.meanRho > 0.f ? std::sqrt(rms.meanRho) : 0.f;
266  rms.meanX2X0 = rms.meanX2X0 > 0.f ? std::sqrt(rms.meanX2X0) : 0.f;
267 }

324 GPUd() int MatLayerCyl::getNPhiBinsInSlice(int iSlice, int& binMin, int& binMax) const
325 {
326  // slow method to get number of phi bins for given phi slice
327  int nb = 0;
328  binMin = binMax = -1;
329  for (int ib = getNPhiBins(); ib--;) {
330  if (phiBin2Slice(ib) == iSlice) {
331  binMax < 0 ? binMin = binMax = ib : binMin = ib;
332  nb++;
333  continue;
334  }
335  if (binMax >= 0) {
336  break; // no more bins since they are consecutive
337  }
338  }
339  return nb;
340 }
