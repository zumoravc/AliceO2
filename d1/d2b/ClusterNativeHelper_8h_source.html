157  {
158  if (buffer == nullptr || size < sizeof(AttributeT) || (size - sizeof(AttributeT)) % sizeof(ClusterNative) != 0) {

161  }

164  auto ptrClusters = reinterpret_cast<ClusterNative*>(buffer + sizeof(groupAttribute));
165  clusterIndex.clusters[groupAttribute.sector][groupAttribute.globalPadRow] = ptrClusters;
166  clusterIndex.nClusters[groupAttribute.sector][groupAttribute.globalPadRow] = nofClusters;

347 template <typename DataArrayType, typename MCArrayType, typename CheckFct>
348 int ClusterNativeHelper::Reader::fillIndex(ClusterNativeAccess& clusterIndex,
349  std::unique_ptr<ClusterNative[]>& clusterBuffer, MCLabelContainer& mcBuffer,
350  DataArrayType& inputs, MCArrayType const& mcinputs, CheckFct checkFct)
351 {
352  if (mcinputs.size() > 0 && mcinputs.size() != inputs.size()) {
353  std::runtime_error("inconsistent size of MC label array " + std::to_string(mcinputs.size()) + ", expected " + std::to_string(inputs.size()));
354  }
355  memset(&clusterIndex, 0, sizeof(clusterIndex));
356  if (inputs.size() == 1) {
357  if (inputs[0].size() >= sizeof(ClusterCountIndex)) {
358  // there is only one data block and we can set the index directly from it
359  const ClusterCountIndex* hdr = reinterpret_cast<ClusterCountIndex const*>(inputs[0].data());
360  memcpy((void*)&clusterIndex.nClusters[0][0], hdr, sizeof(*hdr));
361  clusterIndex.clustersLinear = reinterpret_cast<const ClusterNative*>(inputs[0].data() + sizeof(*hdr));
362  clusterIndex.setOffsetPtrs();
363  if (mcinputs.size() > 0) {
364  clusterIndex.clustersMCTruth = mcinputs[0].get();
365  }
366  }
367  if (sizeof(ClusterCountIndex) + clusterIndex.nClustersTotal * sizeof(ClusterNative) > inputs[0].size()) {
368  throw std::runtime_error("inconsistent input buffer, expecting size " + std::to_string(sizeof(ClusterCountIndex) + clusterIndex.nClustersTotal * sizeof(ClusterNative)) + " got " + std::to_string(inputs[0].size()));
369  }
370  return clusterIndex.nClustersTotal;
371  }
372 
373  // multiple data blocks need to be merged into the single block
374  const MCLabelContainer* clustersMCTruth[NSectors] = {nullptr};
375  int result = 0;
376  for (size_t index = 0, end = inputs.size(); index < end; index++) {
377  if (!checkFct(index)) {
378  continue;
379  }
380  MCLabelContainer const* labelsptr = nullptr;
381  int extent = 0;
382  if (index < mcinputs.size()) {
383  labelsptr = mcinputs[index].get();
384  extent = 1;
385  }
386  int locres = parseSector(inputs[index], {labelsptr, extent}, clusterIndex, clustersMCTruth);
387  if (locres < 0) {
388  return locres;
389  }
390  result += locres;
391  }
392 
393  // Now move all data to a new consecutive buffer
394  ClusterNativeAccess old = clusterIndex;
395  clusterBuffer.reset(new ClusterNative[result]);
396  mcBuffer.clear();
397  bool mcPresent = false;
398  clusterIndex.clustersLinear = clusterBuffer.get();
399  clusterIndex.setOffsetPtrs();
400  for (unsigned int i = 0; i < NSectors; i++) {
401  int sectorLabelId = 0;
402  for (unsigned int j = 0; j < NPadRows; j++) {
403  memcpy(&clusterBuffer[clusterIndex.clusterOffset[i][j]], old.clusters[i][j], sizeof(*old.clusters[i][j]) * old.nClusters[i][j]);
404  if (clustersMCTruth[i]) {
405  mcPresent = true;
406  for (unsigned int k = 0; k < old.nClusters[i][j]; k++, sectorLabelId++) {
407  for (auto const& label : clustersMCTruth[i]->getLabels(sectorLabelId)) {
408  mcBuffer.addElement(clusterIndex.clusterOffset[i][j] + k, label);
409  }
410  }
411  }
412  }
413  }
414  if (mcPresent) {
415  clusterIndex.clustersMCTruth = &mcBuffer;
416  }
417 
418  return result;
419 }

421 template <typename BufferType, typename MCArrayType>
422 void ClusterNativeHelper::copySectorData(ClusterNativeAccess const& index, int sector, BufferType& target, MCArrayType& mcTarget)
423 {
424  static_assert(sizeof(typename BufferType::value_type) == 1, "Target container must be byte-type");
425  if (index.clustersLinear == nullptr) {
426  return;
427  }
428  size_t nRows = 0;
429  size_t nClusters = 0;
430  for (unsigned int row = 0; row < NPadRows; row++) {
431  // count rows with clusters
432  nRows += index.nClusters[sector][row] > 0 ? 1 : 0;
433  nClusters += index.nClusters[sector][row];
434  }
435  size_t rawSize = nRows * sizeof(ClusterNativeBuffer) + nClusters * sizeof(ClusterNative);
436  target.resize(rawSize);
437  ClusterNativeBuffer* current = reinterpret_cast<ClusterNativeBuffer*>(target.data());
438  for (unsigned int row = 0; row < NPadRows; row++) {
439  if (index.nClusters[sector][row] == 0) {
440  continue;
441  }
442  current->sector = sector;
443  current->globalPadRow = row;
444  current->nClusters = index.nClusters[sector][row];
445  memcpy(current->clusters, index.clusters[sector][row], sizeof(*(current->clusters)) * current->nClusters);
446  current = reinterpret_cast<ClusterNativeBuffer*>(current->clusters + current->nClusters);
447  if (index.clustersMCTruth) {
448  mcTarget.emplace_back();
449  for (unsigned int k = 0; k < index.nClusters[sector][row]; k++) {
450  for (auto const& label : index.clustersMCTruth->getLabels(k + index.clusterOffset[sector][row])) {
451  mcTarget.back().addElement(k, label);
452  }
453  }
454  }
455  }
456 }
