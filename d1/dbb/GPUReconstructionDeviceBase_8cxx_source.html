56 void* GPUReconstructionDeviceBase::helperWrapper(GPUReconstructionHelpers::helperParam* par)
57 {
58  if (mDeviceProcessingSettings.debugLevel >= 3) {
59  GPUInfo("\tHelper thread %d starting", par->num);
60  }
61 
62  // cpu_set_t mask; //TODO add option
63  // CPU_ZERO(&mask);
64  // CPU_SET(par->num * 2 + 2, &mask);
65  // sched_setaffinity(0, sizeof(mask), &mask);
66 
67  par->mutex[0].lock();
68  while (par->terminate == false) {
69  for (int i = par->num + 1; i < par->count; i += mDeviceProcessingSettings.nDeviceHelperThreads + 1) {
70  // if (mDeviceProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Running, Slice %d+%d, Phase %d", par->num, i, par->phase);
71  if ((par->functionCls->*par->function)(i, par->num + 1, par)) {
72  par->error = 1;
73  }
74  if (par->reset) {
75  break;
76  }
77  par->done = i + 1;
78  // if (mDeviceProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Finished, Slice %d+%d, Phase %d", par->num, i, par->phase);
79  }
80  ResetThisHelperThread(par);
81  par->mutex[0].lock();
82  }
83  if (mDeviceProcessingSettings.debugLevel >= 3) {
84  GPUInfo("\tHelper thread %d terminating", par->num);
85  }
86  par->mutex[1].unlock();
87  pthread_exit(nullptr);
88  return (nullptr);
89 }

100 int GPUReconstructionDeviceBase::GetGlobalLock(void*& pLock)
101 {
102 #ifdef _WIN32
103  HANDLE* semLock = new HANDLE;
104  *semLock = CreateSemaphore(nullptr, 1, 1, SemLockName);
105  if (*semLock == nullptr) {
106  GPUError("Error creating GPUInit Semaphore");
107  return (1);
108  }
109  WaitForSingleObject(*semLock, INFINITE);
110 #elif !defined(__APPLE__) // GPU not supported on MacOS anyway
111  sem_t* semLock = sem_open(SemLockName, O_CREAT, 0x01B6, 1);
112  if (semLock == SEM_FAILED) {
113  GPUError("Error creating GPUInit Semaphore");
114  return (1);
115  }
116  timespec semtime;
117  clock_gettime(CLOCK_REALTIME, &semtime);
118  semtime.tv_sec += 10;
119  while (sem_timedwait(semLock, &semtime) != 0) {
120  GPUError("Global Lock for GPU initialisation was not released for 10 seconds, assuming another thread died");
121  GPUWarning("Resetting the global lock");
122  sem_post(semLock);
123  }
124 #else
125  void* semLock = nullptr;
126 #endif
127  pLock = semLock;
128  return 0;
129 }

131 void GPUReconstructionDeviceBase::ReleaseGlobalLock(void* sem)
132 {
133 // Release the global named semaphore that locks GPU Initialization
134 #ifdef _WIN32
135  HANDLE* h = (HANDLE*)sem;
136  ReleaseSemaphore(*h, 1, nullptr);
137  CloseHandle(*h);
138  delete h;
139 #elif !defined(__APPLE__) // GPU not supported on MacOS anyway
140  sem_t* pSem = (sem_t*)sem;
141  sem_post(pSem);

147 {
148  GPUImportant("Error occurred, GPU tracker helper threads will be reset (Number of threads %d (%d))", mDeviceProcessingSettings.nDeviceHelperThreads, mNSlaveThreads);
149  SynchronizeGPU();
150  for (int i = 0; i < mDeviceProcessingSettings.nDeviceHelperThreads; i++) {
151  mHelperParams[i].reset = true;
152  if (helpers || i >= mDeviceProcessingSettings.nDeviceHelperThreads) {
153  pthread_mutex_lock(&((pthread_mutex_t*)mHelperParams[i].mutex)[1]);
154  }
155  }
156  GPUImportant("GPU Tracker helper threads have ben reset");
157 }

159 int GPUReconstructionDeviceBase::StartHelperThreads()
160 {
161  int nThreads = mDeviceProcessingSettings.nDeviceHelperThreads;
162  if (nThreads) {
163  mHelperParams = new GPUReconstructionHelpers::helperParam[nThreads];
164  if (mHelperParams == nullptr) {
165  GPUError("Memory allocation error");
166  ExitDevice();
167  return (1);
168  }
169  for (int i = 0; i < nThreads; i++) {
170  mHelperParams[i].cls = this;
171  mHelperParams[i].terminate = false;
172  mHelperParams[i].reset = false;
173  mHelperParams[i].num = i;
174  for (int j = 0; j < 2; j++) {
175  mHelperParams[i].mutex[j].lock();
176  }
177 
178  if (pthread_create(&mHelperParams[i].threadId, nullptr, helperWrapper_static, &mHelperParams[i])) {
179  GPUError("Error starting slave thread");
180  ExitDevice();
181  return (1);
182  }
183  }
184  }
185  mNSlaveThreads = nThreads;
186  return (0);
187 }

189 int GPUReconstructionDeviceBase::StopHelperThreads()
190 {
191  if (mNSlaveThreads) {
192  for (int i = 0; i < mNSlaveThreads; i++) {
193  mHelperParams[i].terminate = true;
194  mHelperParams[i].mutex[0].unlock();
195  mHelperParams[i].mutex[1].lock();
196  if (pthread_join(mHelperParams[i].threadId, nullptr)) {
197  GPUError("Error waiting for thread to terminate");
198  return (1);
199  }
200  }
201  delete[] mHelperParams;
202  }
203  mNSlaveThreads = 0;
204  return (0);
205 }

226 int GPUReconstructionDeviceBase::InitDevice()
227 {
228  // cpu_set_t mask;
229  // CPU_ZERO(&mask);
230  // CPU_SET(0, &mask);
231  // sched_setaffinity(0, sizeof(mask), &mask);
232 
233  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
234  GPUError("Individual memory allocation strategy unsupported for device\n");
235  return (1);
236  }
237  if (mDeviceProcessingSettings.nStreams > GPUCA_MAX_STREAMS) {
238  GPUError("Too many straems requested %d > %d\n", mDeviceProcessingSettings.nStreams, GPUCA_MAX_STREAMS);
239  return (1);
240  }
241  mThreadId = GetThread();
242 
243  void* semLock = nullptr;
244  if (mDeviceProcessingSettings.globalInitMutex && GetGlobalLock(semLock)) {
245  return (1);
246  }
247 
248  if (mDeviceProcessingSettings.deviceTimers) {
249  AddGPUEvents(mDebugEvents);
250  }
251 
252  int retVal = InitDevice_Runtime();
253  if (retVal) {
254  GPUImportant("GPU Tracker initialization failed");
255  return (1);
256  }
257 
258  if (mDeviceProcessingSettings.globalInitMutex) {
259  ReleaseGlobalLock(semLock);
260  }
261 
262  mDeviceMemoryPermanent = mDeviceMemoryBase;
263  mHostMemoryPermanent = mHostMemoryBase;
264  ClearAllocatedMemory();
265 
266  mProcShadow.InitGPUProcessor(this, GPUProcessor::PROCESSOR_TYPE_SLAVE);
267  mProcShadow.mMemoryResProcessors = RegisterMemoryAllocation(&mProcShadow, &GPUProcessorProcessors::SetPointersDeviceProcessor, GPUMemoryResource::MEMORY_PERMANENT | GPUMemoryResource::MEMORY_HOST, "Processors");
268  AllocateRegisteredMemory(mProcShadow.mMemoryResProcessors);
269 
270  if (StartHelperThreads()) {
271  return (1);
272  }
273 
274  if (mMaster == nullptr || mDeviceProcessingSettings.debugLevel >= 2) {
275  GPUInfo("GPU Tracker initialization successfull"); // Verbosity reduced because GPU backend will print GPUImportant message!
276  }
277 
278  return (retVal);
279 }
