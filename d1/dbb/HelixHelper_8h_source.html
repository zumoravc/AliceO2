49 struct CrossInfo {
50  float xDCA[2];
51  float yDCA[2];
52  int nDCA;
53 
54  int circlesCrossInfo(const TrackAuxPar& trax0, const TrackAuxPar& trax1)
55  {
56  const auto& trcA = trax0.rC > trax1.rC ? trax0 : trax1; // designate the largest circle as A
57  const auto& trcB = trax0.rC > trax1.rC ? trax1 : trax0;
58  float xDist = trcB.xC - trcA.xC, yDist = trcB.yC - trcA.yC;
59  float dist2 = xDist * xDist + yDist * yDist, dist = std::sqrt(dist2), rsum = trcA.rC + trcB.rC;
60  if (std::abs(dist) < 1e-12) {
61  return nDCA; // circles are concentric?
62  }
63  if (dist > rsum) { // circles don't touch, chose a point in between
64  // the parametric equation of lines connecting the centers is
65  // x = x0 + t/dist * (x1-x0), y = y0 + t/dist * (y1-y0)
66  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
67  } else if (dist + trcB.rC < trcA.rC) { // the small circle is nestled into large one w/o touching
68  // select the point of closest approach of 2 circles
69  notTouchingXY(dist, xDist, yDist, trcA, -trcB.rC);
70  } else { // 2 intersection points
71  // to simplify calculations, we move to new frame x->x+Xc0, y->y+Yc0, so that
72  // the 1st one is centered in origin
73  if (std::abs(xDist) < std::abs(yDist)) {
74  float a = (trcA.rC * trcA.rC - trcB.rC * trcB.rC + dist2) / (2. * yDist), b = -xDist / yDist, ab = a * b, bb = b * b;
75  float det = ab * ab - (1. + bb) * (a * a - trcA.rC * trcA.rC);
76  if (det > 0.) {
77  det = std::sqrt(det);
78  xDCA[0] = (-ab + det) / (1. + b * b);
79  yDCA[0] = a + b * xDCA[0] + trcA.yC;
80  xDCA[0] += trcA.xC;
81  xDCA[1] = (-ab - det) / (1. + b * b);
82  yDCA[1] = a + b * xDCA[1] + trcA.yC;
83  xDCA[1] += trcA.xC;
84  nDCA = 2;
85  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
86  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
87  }
88  } else {
89  float a = (trcA.rC * trcA.rC - trcB.rC * trcB.rC + dist2) / (2. * xDist), b = -yDist / xDist, ab = a * b, bb = b * b;
90  float det = ab * ab - (1. + bb) * (a * a - trcA.rC * trcA.rC);
91  if (det > 0.) {
92  det = std::sqrt(det);
93  yDCA[0] = (-ab + det) / (1. + bb);
94  xDCA[0] = a + b * yDCA[0] + trcA.xC;
95  yDCA[0] += trcA.yC;
96  yDCA[1] = (-ab - det) / (1. + bb);
97  xDCA[1] = a + b * yDCA[1] + trcA.xC;
98  yDCA[1] += trcA.yC;
99  nDCA = 2;
100  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
101  notTouchingXY(dist, xDist, yDist, trcA, trcB.rC);
102  }
103  }
104  }
105  return nDCA;
106  }

108  void notTouchingXY(float dist, float xDist, float yDist, const TrackAuxPar& trcA, float rBSign)
109  {
110  // fast method to calculate DCA between 2 circles, assuming that they don't touch each outer:
111  // the parametric equation of lines connecting the centers is x = xA + t/dist * xDist, y = yA + t/dist * yDist
112  // with xA,yY being the center of the circle A ( = trcA.xC, trcA.yC ), xDist = trcB.xC = trcA.xC ...
113  // There are 2 special cases:
114  // (a) small circle is inside the large one: provide rBSign as -trcB.rC
115  // (b) circle are side by side: provide rBSign as trcB.rC
116  nDCA = 1;
117  auto t2d = (dist + trcA.rC - rBSign) / dist;
118  xDCA[0] = trcA.xC + 0.5 * (xDist * t2d);
119  yDCA[0] = trcA.yC + 0.5 * (yDist * t2d);
120  }

122  int linesCrossInfo(const TrackAuxPar& trax0, const TrackPar& tr0,
123  const TrackAuxPar& trax1, const TrackPar& tr1)
124  {
138 
139  float dx = trax1.xC - trax0.xC; // for straight line TrackAuxPar stores lab coordinates at referene point!!!
140  float dy = trax1.yC - trax0.yC; //
141  float dz = tr1.getZ() - tr0.getZ();
142  auto csp0i2 = 1. / tr0.getCsp2(); // 1 / csp^2
143  auto csp0i = std::sqrt(csp0i2);
144  auto tgp0 = tr0.getSnp() * csp0i;
145  float kx0 = trax0.c - trax0.s * tgp0;
146  float ky0 = trax0.s + trax0.c * tgp0;
147  float kz0 = tr0.getTgl() * csp0i;
148  auto csp1i2 = 1. / tr1.getCsp2(); // 1 / csp^2
149  auto csp1i = std::sqrt(csp1i2);
150  auto tgp1 = tr1.getSnp() * std::sqrt(csp1i2);
151  float kx1 = trax1.c - trax1.s * tgp1;
152  float ky1 = trax1.s + trax1.c * tgp1;
153  float kz1 = tr1.getTgl() * csp1i;
162  float a00 = (1.f + tr0.getTgl() * tr0.getTgl()) * csp0i2, a11 = (1.f + tr1.getTgl() * tr1.getTgl()) * csp1i2, a01 = -(kx0 * kx1 + ky0 * ky1 + kz0 * kz1);
163  float b0 = dx * kx0 + dy * ky0 + dz * kz0, b1 = -(dx * kx1 + dy * ky1 + dz * kz1);
164  float det = a00 * a11 - a01 * a01, det0 = b0 * a11 - b1 * a01, det1 = a00 * b1 - a01 * b0;
165  if (std::abs(det) > o2::constants::math::Almost0) {
166  auto detI = 1. / det;
167  auto t0 = det0 * detI;
168  auto t1 = det1 * detI;
169  xDCA[0] = (trax0.xC + kx0 * t0 + trax1.xC + kx1 * t1) * 0.5;
170  yDCA[0] = (trax0.yC + ky0 * t0 + trax1.yC + ky1 * t1) * 0.5;
171  nDCA = 1;
172  }
173  return nDCA;
174  }

176  int circleLineCrossInfo(const TrackAuxPar& trax0, const TrackPar& tr0,
177  const TrackAuxPar& trax1, const TrackPar& tr1)
178  {
192 
193  const auto& traxH = trax0.rC > trax1.rC ? trax0 : trax1; // circle (for the line rC is set to 0)
194  const auto& traxL = trax0.rC > trax1.rC ? trax1 : trax0; // line
195  const auto& trcL = trax0.rC > trax1.rC ? tr1 : tr0; // track of the line
196 
197  // solve quadratic equation of line crossing the circle
198  float dx = traxL.xC - traxH.xC; // X distance between the line lab reference and circle center
199  float dy = traxL.yC - traxH.yC; // Y...
200  // t^2(kx^2+ky^2) + 2t(dx*kx+dy*ky) + dx^2 + dy^2 - r^2 = 0
201  auto cspi2 = 1. / trcL.getCsp2(); // 1 / csp^2 == kx^2 + ky^2
202  auto cspi = std::sqrt(cspi2);
203  auto tgp = trcL.getSnp() * cspi;
204  float kx = traxL.c - traxL.s * tgp;
205  float ky = traxL.s + traxL.c * tgp;
206  double dk = dx * kx + dy * ky;
207  double det = dk * dk - cspi2 * (dx * dx + dy * dy - traxH.rC * traxH.rC);
208  if (det > 0) { // 2 crossings
209  det = std::sqrt(det);
210  float t0 = (-dk + det) * cspi2;
211  float t1 = (-dk - det) * cspi2;
212  xDCA[0] = traxL.xC + kx * t0;
213  yDCA[0] = traxL.yC + ky * t0;
214  xDCA[1] = traxL.xC + kx * t1;
215  yDCA[1] = traxL.yC + ky * t1;
216  nDCA = 2;
217  } else {
218  // there is no crossing, find the point of the closest approach on the line as the sames which is closest to the circle center
219  float t = -dk * cspi2;
220  float xL = traxL.xC + kx * t, yL = traxL.yC + ky * t; // point on the line, need to average with point on the circle
221  float dxc = xL - traxH.xC, dyc = yL - traxH.yC, drcf = traxH.rC / std::sqrt(dxc * dxc + dyc * dyc); // radius / distance to circle center
222  float xH = traxH.xC + dxc * drcf, yH = traxH.yC + dyc * drcf;
223  xDCA[0] = (xL + xH) * 0.5;
224  yDCA[0] = (yL + yH) * 0.5;
225  nDCA = 1;
226  }
227  return nDCA;
228  }

230  int set(const TrackAuxPar& trax0, const TrackPar& tr0, const TrackAuxPar& trax1, const TrackPar& tr1)
231  {
232  // calculate up to 2 crossings between 2 circles
233  nDCA = 0;
234  if (trax0.rC > o2::constants::math::Almost0 && trax1.rC > o2::constants::math::Almost0) { // both are not straight lines
235  nDCA = circlesCrossInfo(trax0, trax1);
236  } else if (trax0.rC < o2::constants::math::Almost0 && trax1.rC < o2::constants::math::Almost0) { // both are straigt lines
237  nDCA = linesCrossInfo(trax0, tr0, trax1, tr1);
238  } else {
239  nDCA = circleLineCrossInfo(trax0, tr0, trax1, tr1);
240  }
241  //
242  return nDCA;
243  }
