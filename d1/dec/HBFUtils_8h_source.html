28 /*
29  In the MC->Raw conversion we have to make sure that
30  1) The HB and TF starts are in sync for all detectors regardless on time (bc/orbir)
31  distribution of its signal.
32  2) All HBF and TF (RAWDataHeaders with corresponding HB and TF trigger flags) are present
33  in the emulated raw data, even if some of them had no data in particular detector.
34 
35  The HBFUtils class provides tools for interaction record -> HBF conversion and sampling
36  of IRs for which the HBF RDH should be added to the raw data from the CRU.
37 
38  See testHBFUtils.cxx for the outline of generating HBF frames for simulated data.
39  */

84  /*//-------------------------------------------------------------------------------------
85  Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
86  BCs between interaction records "fromIR" and "toIR" (inclusive).
87  This method provides the IRs for RDHs to add obligatory for the MC->raw conversion,
88  in order to avoid missing HBFs (or even TFs)
89  Typical use case: assume we are converting to RAW data the digits corresponding to triggers
90  for Int.records ir[0], ir[1], ... ir[N]
91  The pseudo-code:
92 
93  HBFUtils sampler;
94  uint8_t packetCounter = 0;
95  std::vector<o2::InteractionRecord> HBIRVec;
96  auto irFrom = sampler.getFirstIR();
97  for (int i=0;i<N;i++) {
98  int nHBF = sampler.fillHBIRvector(HBIRVec, irFrom, ir[i]);
99  irFrom = ir[i]+1;
100  // nHBF-1 HBframes don't have data, we need to create empty HBFs for them
101  for (int j=0;j<nHBF-1;j++) {
102  auto rdh = sampler.createRDH<RAWDataHeader>( HBIRVec[j] );
103  // dress rdh with cruID/FEE/Link ID
104  RDHUtils::setPacketCounter(rdh, packetCounter++);
105  RDHUtils::setMemorySize(rdh, sizeof(rdh));
106  RDHUtils::setOffsetToNext(rdh, sizeof(rdh));
107  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // open empty HBF
108  RDHUtils::setStop(rdh, 0x1);
109  RDHUtils::setPageCounter(rdh, RDHUtils::getPageCounter()+1 );
110  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // close empty HBF
111  }
112  // write RDH for the HBF with data
113  auto rdh = HBIRVec.back();
114  // dress rdh with cruID/FEE/Link ID, estimate size, offset etc. and flush
115  // flush raw data payload
116  }
117  // see tstHBFUtils for more details
118  //-------------------------------------------------------------------------------------*/
119  int fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const;
120 
121  void print() const { printKeyValues(true); }
122 
123  int nHBFPerTF = 1 + 0xff; // number of orbits per BC
124  uint16_t bcFirst = 0;
125  uint32_t orbitFirst = 0;
126 
127  O2ParamDef(HBFUtils, "HBFUtils");
128 };

131 template <typename H>
132 void HBFUtils::updateRDH(H& rdh, const IR& rec) const
133 {
134  auto tfhb = getTFandHBinTF(rec);
135  RDHUtils::setHeartBeatBC(rdh, bcFirst);
136  RDHUtils::setHeartBeatOrbit(rdh, rec.orbit);
137  if (RDHUtils::getVersion(rdh) < 5) { // v3,4 have separate fields for trigger IR
138  RDHUtils::setTriggerBC(rdh, bcFirst);
139  RDHUtils::setTriggerOrbit(rdh, rec.orbit);
140  }
141 
142  if (rec.bc == bcFirst) { // if we are starting new HB, set the HB trigger flag
143  auto trg = RDHUtils::getTriggerType(rdh) | (o2::trigger::ORBIT | o2::trigger::HB);
144  if (tfhb.second == 0) { // if we are starting new TF, set the TF trigger flag
145  trg |= o2::trigger::TF;
146  }
147  RDHUtils::setTriggerType(rdh, trg);
148  }
149 }
