27 // a structure describing EventPart
28 // (an elementary constituent of a collision)
29 struct EventPart {
30  EventPart() = default;
31  EventPart(int s, int e) : sourceID(s), entryID(e) {}
32  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)
33  // the sourceID should correspond to the chain ID
34  int entryID = 0; // the event/entry ID inside the chain corresponding to sourceID
35 
36  static bool isSignal(EventPart e) { return e.sourceID > 1; }
37  static bool isBackGround(EventPart e) { return !isSignal(e); }
38  ClassDefNV(EventPart, 1);
39 };
