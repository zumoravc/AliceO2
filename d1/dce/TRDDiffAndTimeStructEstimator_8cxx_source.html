439 float TRDDiffusionAndTimeStructEstimator::TimeStruct(float vdrift, float dist, float z)
440 {
441  //
442  // Applies the time structure of the drift cells (by C.Lippmann).
443  // The drift time of electrons to the anode wires depends on the
444  // distance to the wire (z) and on the position in the drift region.
445  //
446  // input :
447  // dist = radial distance from (cathode) pad plane [cm]
448  // z = distance from anode wire (parallel to cathode planes) [cm]
449  //
450  // output :
451  // tdrift = the drift time of an electron at the given position
452  //
453  // We interpolate between the drift time values at the two drift
454  // velocities fVDlo and fVDhi, being smaller and larger than
455  // fDriftVelocity. We use the two stored drift time maps fTimeStruct1
456  // and fTimeStruct2, calculated for the two mentioned drift velocities.
457  //
458 
459  SampleTimeStruct(vdrift);
460 
461  // Indices:
462  int r1 = (int)(10 * dist);
463  if (r1 < 0)
464  r1 = 0;
465  if (r1 > 37)
466  r1 = 37;
467  int r2 = r1 + 1;
468  if (r2 > 37)
469  r2 = 37;
470  const int kz1 = ((int)(100 * z / 2.5));
471  const int kz2 = kz1 + 1;
472 
473  if ((r1 < 0) || (r1 > 37) || (kz1 < 0) || (kz1 > 10)) {
474  LOG(WARN) << Form("TRD: Time struct indices out of range: dist=%.2f, z=%.2f, r1=%d, kz1=%d", dist, z, r1, kz1);
475  }
476 
477  const float ky111 = mTimeStruct1[r1 + 38 * kz1];
478  const float ky221 = ((r2 <= 37) && (kz2 <= 10)) ? mTimeStruct1[r2 + 38 * kz2] : mTimeStruct1[37 + 38 * 10];
479  const float ky121 = (kz2 <= 10) ? mTimeStruct1[r1 + 38 * kz2] : mTimeStruct1[r1 + 38 * 10];
480  const float ky211 = mTimeStruct1[r2 + 38 * kz1];
481 
482  const float ky112 = mTimeStruct2[r1 + 38 * kz1];
483  const float ky222 = ((r2 <= 37) && (kz2 <= 10)) ? mTimeStruct2[r2 + 38 * kz2] : mTimeStruct2[37 + 38 * 10];
484  const float ky122 = (kz2 <= 10) ? mTimeStruct2[r1 + 38 * kz2] : mTimeStruct2[r1 + 38 * 10];
485  const float ky212 = mTimeStruct2[r2 + 38 * kz1];
486 
487  // Interpolation in dist-directions, lower drift time map
488  const float ky11 = (ky211 - ky111) * 10 * dist + ky111 - (ky211 - ky111) * r1;
489  const float ky21 = (ky221 - ky121) * 10 * dist + ky121 - (ky221 - ky121) * r1;
490 
491  // Interpolation in dist-direction, larger drift time map
492  const float ky12 = (ky212 - ky112) * 10 * dist + ky112 - (ky212 - ky112) * r1;
493  const float ky22 = (ky222 - ky122) * 10 * dist + ky122 - (ky222 - ky122) * r1;
494 
495  // Dist now is the drift distance to the anode wires (negative if electrons are
496  // between anode wire plane and cathode pad plane)
497  dist -= TRDGeometry::amThick() * 0.5;
498 
499  // Interpolation in z-directions, lower drift time map
500  const bool condition = ((std::abs(dist) > 0.005) || (z > 0.005));
501  const float ktdrift1 =
502  condition ? (ky21 - ky11) * 100 * z / 2.5 + ky11 - (ky21 - ky11) * kz1 : 0.0;
503  // Interpolation in z-directions, larger drift time map
504  const float ktdrift2 =
505  condition ? (ky22 - ky12) * 100 * z / 2.5 + ky12 - (ky22 - ky12) * kz1 : 0.0;
506 
507  // Interpolation between the values at fVDlo and fVDhi
508  const float a = (ktdrift2 - ktdrift1) * mInvBinWidth; // 1./(mVDhi - mVDlo);
509  const float b = ktdrift2 - a * mVDhi;
510  const float t = a * vdrift + b;
511 
512  return t;
513 }
