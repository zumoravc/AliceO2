141 template <class S, class T>
142 void* qThreadCls<S, T>::qThreadWrapperCls(T* arg)
143 {
144  qThreadParamCls<S>* const arg_A = (qThreadParamCls<S>*)arg;
145  if (arg_A->pinCPU != -1) {
146  cpu_set_t tmp_mask;
147  CPU_ZERO(&tmp_mask);
148  CPU_SET(arg_A->pinCPU, &tmp_mask);
149  sched_setaffinity(0, sizeof(tmp_mask), &tmp_mask);
150  }
151 
152  void (S::*pFunc)(T*) = (void (S::*)(T*))arg_A->pFunc;
153  (arg_A->pCls->*pFunc)(arg);
154 
155  arg_A->threadMutex[1].Unlock();
156  pthread_exit(nullptr);
157  return (nullptr);
158 }
