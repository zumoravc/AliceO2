28 template <typename T, std::enable_if_t<std::is_integral<T>::value, int> = 1>
29 void dumpByteBuffer(gsl::span<T> buffer)
30 {
31 
32  int i{0};
33  while (i < buffer.size()) {
34  if (i % (16 / sizeof(T)) == 0) {
35  std::cout << fmt::format("\n{:8d} : ", i * sizeof(T));
36  }
37  std::cout << fmt::format("{:0{}X} ", buffer[i], sizeof(T) * 2);
38  i++;
39  }
40  std::cout << "\n";
41 }

43 inline void append(std::vector<std::byte>& buffer, uint64_t w)
44 {
45  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x00000000000000FF)))});
46  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x000000000000FF00)) >> 8)});
47  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x0000000000FF0000)) >> 16)});
48  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x00000000FF000000)) >> 24)});
49  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x000000FF00000000)) >> 32)});
50  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x0000FF0000000000)) >> 40)});
51  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0x00FF000000000000)) >> 48)});
52  buffer.emplace_back(std::byte{static_cast<uint8_t>((w & UINT64_C(0xFF00000000000000)) >> 56)});
53 }

64 template <>
65 void dumpWord<o2::mch::raw::BareFormat>(std::ostream& out, uint64_t w)
66 {
67  // show word itself and it 10-bits parts
68  out << fmt::format("{:016X} {:4d} {:4d} {:4d} {:4d} {:4d} ",
69  w,
70  (w & 0x3FF0000000000) >> 40,
71  (w & 0xFFC0000000) >> 30,
72  (w & 0x3FF00000) >> 20,
73  (w & 0xFFC00) >> 10,
74  (w & 0x3FF));

83 template <>
84 void dumpWordInfo<o2::mch::raw::BareFormat>(std::ostream& out, uint64_t w, const char* /*spacer*/)
85 {
86  static constexpr uint64_t FIFTYBITSATONE = (static_cast<uint64_t>(1) << 50) - 1;
87  SampaHeader h(w & FIFTYBITSATONE);
88  if (h == sampaSync()) {
89  out << "SYNC !!";
90  } else if (h.packetType() == SampaPacketType::Sync) {
91  out << "SYNC " << std::boolalpha << (h == sampaSync());
92  } else if (h.packetType() == SampaPacketType::Data) {
93  out << fmt::format(" n10 {:4d} chip {:2d} ch {:2d}",
94  h.nof10BitWords(), h.chipAddress(), h.channelAddress());
95  }
96 }

98 template <>
99 void dumpWordInfo<o2::mch::raw::UserLogicFormat>(std::ostream& out, uint64_t w,
100  const char* spacer)
101 {
102  if (o2::mch::raw::isSampaSync(w)) {
103  out << "SYNC ";
104  }
105  if (w != 0xFEEDDEEDFEEDDEED && w != 0) {
106  if (!o2::mch::raw::isSampaSync(w)) {
107  out << spacer;
108  }
109  out << fmt::format("GBT(0.11) {:2d} ELINKID(0..39) {:2d} ERR {:1d}",
110  (w >> 59) & 0x1F,
111  (w >> 53) & 0x3F,
112  (w >> 50) & 0x7);
113  if (!o2::mch::raw::isSampaSync(w)) {
114  out << fmt::format("{:4d} {:4d} {:4d} {:4d} {:4d} ",
115  (w & 0x3FF0000000000) >> 40,
116  (w & 0xFFC0000000) >> 30,
117  (w & 0x3FF00000) >> 20,
118  (w & 0xFFC00) >> 10,
119  (w & 0x3FF));
120  }
121  }

124 template <typename FORMAT>
125 void dumpBuffer(gsl::span<const std::byte> buffer, std::ostream& out, size_t maxbytes)
126 {
127  int i{0};
128  int inRDH{-1};
129  const void* rdhP{nullptr};
130  const char* spacer = " ";
131 
132  if (buffer.size() < 8) {
133  out << "Should at least get 8 bytes to be able to dump\n";
134  return;
135  }
136  while ((i < buffer.size()) && i < maxbytes) {
137  if (i % 8 == 0) {
138  out << fmt::format("\n{:8d} : ", i);
139  }
140  uint64_t w = b8to64(buffer, i);
141  dumpWord<FORMAT>(out, w);
142 
143  if (inRDH >= 0) {
144  --inRDH;
145  }
146 
147  if (buffer.size() >= i + 64) {
148  const void* testRDH = reinterpret_cast<const void*>(buffer.data() + i);
149  if (o2::raw::RDHUtils::checkRDH(testRDH, false)) {
150  rdhP = testRDH;
151  inRDH = 8;
152  out << "RDH ";
153  }
154  }
155 
156  if (inRDH == 8) {
157  auto version = o2::raw::RDHUtils::getVersion(rdhP);
158  if (version >= 6) {
159  out << "SOURCE " << o2::raw::RDHUtils::getSourceID(rdhP) << " ";
160  }
161  out << fmt::format("VER {:2d} SIZ {:3d} FEEID {:5d}",
162  o2::raw::RDHUtils::getVersion(rdhP),
163  o2::raw::RDHUtils::getHeaderSize(rdhP),
164  o2::raw::RDHUtils::getFEEID(rdhP));
165  }
166 
167  if (inRDH == 7) {
168  std::cout << spacer;
169  out << fmt::format("LINK {:3d} CRU {:3d} EP {:1d}",
170  o2::raw::RDHUtils::getLinkID(rdhP),
171  o2::raw::RDHUtils::getCRUID(rdhP),
172  o2::raw::RDHUtils::getEndPointID(rdhP));
173  }
174 
175  if (inRDH == 6) {
176  std::cout << spacer;
177  out << fmt::format("ORBIT {:10d} BX {:4d}",
178  o2::raw::RDHUtils::getHeartBeatOrbit(rdhP),
179  o2::raw::RDHUtils::getHeartBeatBC(rdhP));
180  }
181 
182  if (inRDH == 4) {
183  out << spacer;
184  out << fmt::format("TRIG 0x{:08X} PAGECOUNT {:5d}",
185  o2::raw::RDHUtils::getTriggerType(rdhP),
186  o2::raw::RDHUtils::getPageCounter(rdhP));
187  if (o2::raw::RDHUtils::getStop(rdhP)) {
188  out << " *STOP*";
189  }
190  }
191 
192  if (inRDH == 2) {
193  out << spacer;
194  out << fmt::format("DET PAR {:5d} DET FIELD {:10d}",
195  o2::raw::RDHUtils::getDetectorPAR(rdhP),
196  o2::raw::RDHUtils::getDetectorField(rdhP));
197  }
198 
199  if (inRDH <= 0) {
200  dumpWordInfo<FORMAT>(out, w, spacer);
201  }
202  i += 8;
203  }
204  out << "\n";
205 } // namespace o2::mch::raw::impl

209 {
210  std::vector<std::byte> b8;
