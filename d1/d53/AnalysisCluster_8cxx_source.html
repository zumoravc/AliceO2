31 TLorentzVector AnalysisCluster::getMomentum(std::array<const float, 3> vertex) const
32 {
33 
34  TLorentzVector p;
35 
36  float pos[3] = {mGlobalPos.X(), mGlobalPos.Y(), mGlobalPos.Z()};
37  pos[0] -= vertex[0];
38  pos[1] -= vertex[1];
39  pos[2] -= vertex[2];
40 
41  float r = TMath::Sqrt(pos[0] * pos[0] + pos[1] * pos[1] + pos[2] * pos[2]);
42 
43  if (r > 0)
44  p.SetPxPyPzE(mEnergy * pos[0] / r, mEnergy * pos[1] / r, mEnergy * pos[2] / r, mEnergy);
45  else
46  LOG(INFO) << "Null cluster radius, momentum calculation not possible";
47 
48  return p;
49 }
