39 // a helper class to do the checking at the end of the program when
40 // the destructor of the class is called.
41 class StaticChecker
42 {
43  public:
44  StaticChecker() = default;
45  // have to throw an exception in the destructor if checking fails
46  // this is ok in this cae because no other instances which would require proper
47  // cleanup are expected
48  ~StaticChecker() noexcept(false)
49  {
50  // the check in the desctructor makes sure that the workflow has been run at all
51  if (mChecks.size() > 0) {
52  throw std::runtime_error("Workflow error: Checks have not been executed");
53  }
54  }
55 
56  struct Attributes {
57  std::string fileName;
58  int nEntries = 0;
59  int nBranches = 0;
60  };
61 
62  void runChecks()
63  {
64  for (auto const& check : mChecks) {
65  TFile* file = TFile::Open(check.fileName.c_str());
66  if (file == nullptr) {
67  setError(std::string("missing file ") + check.fileName.c_str());
68  continue;
69  }
70  TTree* tree = reinterpret_cast<TTree*>(file->GetObjectChecked("testtree", "TTree"));
71  if (tree == nullptr) {
72  setError(std::string("can not find tree 'testtree' in file ") + check.fileName.c_str());
73  } else if (tree->GetEntries() != check.nEntries) {
74  setError(std::string("inconsistent number of entries in 'testtree' of file ") + check.fileName + " expecting " + std::to_string(check.nEntries) + " got " + std::to_string(tree->GetEntries()));
75  } else if (tree->GetNbranches() != check.nBranches) {
76  setError(std::string("inconsistent number of branches in 'testtree' of file ") + check.fileName + " expecting " + std::to_string(check.nBranches) + " got " + std::to_string(tree->GetNbranches()));
77  }
78  file->Close();
79  boost::filesystem::remove(check.fileName.c_str());
80  }
81  mChecks.clear();
82  if (mErrorMessage.empty() == false) {
83  throw std::runtime_error(mErrorMessage);
84  }
85  }
86 
87  void addCheck(std::string filename, int entries, int branches = 0)
88  {
89  mChecks.emplace_back(Attributes{filename, entries, branches});
90  }
91 
92  template <typename T>
93  void setError(T const& message)
94  {
95  if (mErrorMessage.empty()) {
96  mErrorMessage = message;
97  }
98  }
99 
100  void clear()
101  {
102  mChecks.clear();
103  mErrorMessage.clear();
104  }
105 
106  private:
107  std::vector<Attributes> mChecks;
108  std::string mErrorMessage;
109 };

128 static constexpr int sTreeSize = 10; // elements to send and write
129 DataProcessorSpec getSourceSpec()
130 {
131  auto initFct = [](InitContext& ic) {
132  auto counter = std::make_shared<int>();
133  *counter = 0;
134 
135  auto processingFct = [counter](ProcessingContext& pc) {
136  if (*counter >= sTreeSize) {
137  // don't publish more
138  return;
139  }
140  o2::test::Polymorphic a(*counter);
141  pc.outputs().snapshot(OutputRef{"output", 0}, a);
142  pc.outputs().snapshot(OutputRef{"output", 1}, a);
143  int& metadata = pc.outputs().make<int>(Output{"TST", "METADATA", 0, Lifetime::Timeframe});
144  metadata = *counter;
145  *counter = *counter + 1;
146  if (*counter >= sTreeSize) {
147  pc.services().get<ControlService>().endOfStream();
148  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
149  }
150  };
151 
152  return processingFct;
153  };
154  return DataProcessorSpec{"source", // name of the processor
155  {},
156  {OutputSpec{{"output"}, "TST", "SOMEOBJECT", 0, Lifetime::Timeframe},
157  OutputSpec{{"output"}, "TST", "SOMEOBJECT", 1, Lifetime::Timeframe},
158  OutputSpec{{"meta"}, "TST", "METADATA", 0, Lifetime::Timeframe}},
159  AlgorithmSpec(initFct)};
160 }
