236 /******************************************************************************
237  * @brief apply functor to the wrapped member object in the runtime container
238  * This meta function recurses through the list while incrementing the index
239  * and calls the functor at the required position
240  *
241  * @note internal meta function for the RuntimeContainers' apply function
242  */
243 template <
244  typename _ContainerT // container type
245  , typename _IndexT // data type of position index
246  , typename _Iterator // current iterator position
247  , typename _End // end iterator position
248  , _IndexT _Index // current index
249  , typename F // functor
250  >
251 struct rc_apply_at
252 {
253  static typename F::return_type apply( _ContainerT& c, _IndexT position, F& f )
254  {
255  if ( position == _Index ) {
256  // this is the queried position, make the type cast to the current
257  // stage of the runtime container and execute function for it.
258  // Terminate loop by forwarding _End as _Iterator and thus
259  // calling the specialization
260  typedef typename boost::mpl::deref< _Iterator >::type stagetype;
261  stagetype& stage = static_cast<stagetype&>(c);
262  return f(stage);
263  } else {
264  // go to next element
265  return rc_apply_at<
266  _ContainerT
267  , _IndexT
268  , typename boost::mpl::next< _Iterator >::type
269  , _End
270  , _Index + 1
271  , F
272  >::apply( c, position, f );
273  }
274  }
275 };
276 // specialization: end of recursive loop, kicks in if _Iterator matches
277 // _End.
278 // here we end up if the position parameter is out of bounds

280  typename _ContainerT // container type
281  , typename _IndexT // data type of position index
282  , typename _End // end iterator position
283  , _IndexT _Index // current index

285  >
286 struct rc_apply_at<_ContainerT
287  , _IndexT
288  , _End
289  , _End
290  , _Index
291  , F
292  >
293 {
294  static typename F::return_type apply( _ContainerT& c, _IndexT position, F& f )

468  /*
469  * Apply a functor to the runtime container at index
470  *
471  * For performance tests there is a template option to do an explicite loop
472  * unrolling for the first n (=10) elements. This is however only effective
473  * if the compiler optimization is switched of. This is in the end a nice
474  * demonstrator for the potential of compiler optimization. Unrolling is
475  * switched on with the compile time switch RC_UNROLL.
476  */
477  template<typename F
478 #ifdef RC_UNROLL
479  , bool unroll = true
480 #else
481  , bool unroll = false
482 #endif
483  >
484  typename F::return_type apply(int index, F f) {
485  if (unroll) {// this is a compile time switch
486  // do unrolling for the first n elements and forward to generic
487  // recursive function for the rest.
488  switch (index) {
489  case 0: return rc_dispatcher<mixin_type, F, boost::mpl::int_<0>, int>::apply(*this, 0, f);
490  case 1: return rc_dispatcher<mixin_type, F, boost::mpl::int_<1>, int>::apply(*this, 1, f);
491  case 2: return rc_dispatcher<mixin_type, F, boost::mpl::int_<2>, int>::apply(*this, 2, f);
492  case 3: return rc_dispatcher<mixin_type, F, boost::mpl::int_<3>, int>::apply(*this, 3, f);
493  case 4: return rc_dispatcher<mixin_type, F, boost::mpl::int_<4>, int>::apply(*this, 4, f);
494  case 5: return rc_dispatcher<mixin_type, F, boost::mpl::int_<5>, int>::apply(*this, 5, f);
495  case 6: return rc_dispatcher<mixin_type, F, boost::mpl::int_<6>, int>::apply(*this, 6, f);
496  case 7: return rc_dispatcher<mixin_type, F, boost::mpl::int_<7>, int>::apply(*this, 7, f);
497  case 8: return rc_dispatcher<mixin_type, F, boost::mpl::int_<8>, int>::apply(*this, 8, f);
498  case 9: return rc_dispatcher<mixin_type, F, boost::mpl::int_<9>, int>::apply(*this, 9, f);
499  }
500  }
501  return rc_dispatcher<mixin_type, F>::apply(*this, index, f);
502  }
503 
504 private:
505  T mMember;
506 };

533 template<typename Types, typename Base, typename N = boost::mpl::size<Types>>
534 struct create_rtc
535 {
536  typedef typename boost::mpl::lambda<
537  // mpl fold loops over all elements in the list of the first template
538  // parameter and provides this as placeholder _2; for every element the
539  // operation of the third template parameter is applied to the result of
540  // the previous stage which is provided as placeholder _1 to the operation
541  // and initialized to the second template argument for the very first
542  // operation
543  typename boost::mpl::fold<
544  // list of types, each element provided as placeholder _1
545  Types
546  // initializer for the _1 placeholder
547  , Base
548  // recursively applied operation, depending on the outcome of rtc_less
549  // either the next mixin level is applied or the current state is used
550  , boost::mpl::if_<
551  rtc_less<_1, N >
552  // apply mixin level
553  , boost::mpl::apply2< boost::mpl::protect<apply_rc_mixin>::type, _1, _2 >
554  // keep current state by identity
555  , boost::mpl::identity<_1>
556  >
