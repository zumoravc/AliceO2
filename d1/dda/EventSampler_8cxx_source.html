113 void EventSampler::Run()
114 {
116  int iResult=0;
117 
118  boost::thread samplerThread(boost::bind(&EventSampler::samplerLoop, this));
119 
120  unique_ptr<FairMQPoller> poller(fTransportFactory->CreatePoller(fChannels["data-in"]));
121 
122  bool received = false;
123 
124  // inherited variables of FairMQDevice:
125  // fChannels
126  // fTransportFactory
127  int numInputs = fChannels["data-in"].size();
128  int NoOfMsgParts=numInputs-1;
129  int errorCount=0;
130  const int maxError=10;
131 
132  vector<unique_ptr<FairMQMessage>> inputMessages;
133  vector<int> inputMessageCntPerSocket(numInputs, 0);
134  int nReadCycles=0;
135 
136  std::ofstream latencyLog(mLatencyLogFileName);
137 
138  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this)) {
139 
140  // read input messages
141  poller->Poll(mPollingTimeout);
142  for(int i = 0; i < numInputs; i++) {
143  if (poller->CheckInput(i)) {
144  if (fChannels.at("data-in").at(i).Receive(inputMessages)) {
145  inputMessageCntPerSocket[i] = inputMessages.size();
146  if (mVerbosity > 3) {
147  LOG(INFO) << " |---- receive Msgs from socket " << i;
148  }
149  }
150  if (mVerbosity > 2) {
151  LOG(INFO) << "------ received " << inputMessageCntPerSocket[i] << " message(s) from socket " << i;
152  }
153  }
154  }
155 
156  system_clock::time_point timestamp = system_clock::now();
157  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(timestamp - dayref);
158  auto useconds = std::chrono::duration_cast<std::chrono::microseconds>(timestamp - dayref - seconds);
159 
160  for (vector<unique_ptr<FairMQMessage>>::iterator mit=inputMessages.begin();
161  mit!=inputMessages.end(); mit++) {
162  AliHLTComponentEventData* evtData=reinterpret_cast<AliHLTComponentEventData*>((*mit)->GetData());
163  if ((*mit)->GetSize() >= sizeof(AliHLTComponentEventData) &&
164  evtData && evtData->fStructSize == sizeof(AliHLTComponentEventData) &&
165  (evtData->fEventCreation_s>0 || evtData->fEventCreation_us>0)) {
166  unsigned latencySeconds=seconds.count() - evtData->fEventCreation_s;
167  unsigned latencyUSeconds=0;
168  if (useconds.count() < evtData->fEventCreation_us) {
169  latencySeconds--;
170  latencyUSeconds=(1000000 + useconds.count()) - evtData->fEventCreation_us;
171  } else {
172  latencyUSeconds=useconds.count() - evtData->fEventCreation_us;
173  }
174  if (mVerbosity>0) {
175  const char* unit="";
176  unsigned value=0;
177  if (latencySeconds>=10) {
178  unit=" s";
179  value=latencySeconds;
180  } else if (latencySeconds>0 || latencyUSeconds>10000) {
181  value=latencySeconds*1000 + latencyUSeconds/1000;
182  unit=" ms";
183  } else {
184  value=latencyUSeconds;
185  unit=" us";
186  }
187  LOG(DEBUG) << "received event " << evtData->fEventID << " at " << seconds.count() << "s " << useconds.count() << "us - latency " << value << unit;
188  }
189  latencyUSeconds+=latencySeconds*1000000; // max 4294s, should be enough for latency
190  if (latencyLog.is_open()) {
191  latencyLog << evtData->fEventID << " " << latencyUSeconds << endl;
192  }
193  }
194  }
195  inputMessages.clear();
196  for (vector<int>::iterator mcit=inputMessageCntPerSocket.begin();
197  mcit!=inputMessageCntPerSocket.end(); mcit++) {
198  *mcit=0;
199  }
200  }
201 
202  if (latencyLog.is_open()) {
203  latencyLog.close();
204  }
205 
206  samplerThread.interrupt();
207  samplerThread.join();
208 }

210 void EventSampler::samplerLoop()
211 {
213  LOG(INFO) << "initializing sampler loop, then waiting for " << mInitialDelay << " ms";
214  // wait until the first event is sent
215  // usleep expects arguments in the range [0,1000000]
216  unsigned initialDelayInSeconds=mInitialDelay/1000;
217  unsigned initialDelayInUSeconds=mInitialDelay%1000;
218  unsigned eventPeriodInSeconds=mEventPeriod/1000000;
219  if (initialDelayInSeconds>0) sleep(initialDelayInSeconds);
220  usleep(initialDelayInUSeconds);
221 
222  unique_ptr<FairMQMessage> msg(fTransportFactory->CreateMessage());
223  msg->Rebuild(sizeof(AliHLTComponentEventData));
224  if (msg->GetSize() < sizeof(AliHLTComponentEventData)) {
225  // fatal error
226  LOG(ERROR) << "failed to allocate message of size " << sizeof(AliHLTComponentEventData) << ", aborting event generation";
227  return;
228  }
229  AliHLTComponentEventData* evtData = reinterpret_cast<AliHLTComponentEventData*>(msg->GetData());
230  memset(evtData, 0, sizeof(AliHLTComponentEventData));
231  evtData->fStructSize = sizeof(AliHLTComponentEventData);
232 
233  int numOutputs = (fChannels.find("data-out") == fChannels.end() ? 0 : fChannels["data-out"].size());
234 
235  LOG(INFO) << "starting sampler loop, period " << mEventPeriod << " us";
236  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this)) {
237  msg->Rebuild(sizeof(AliHLTComponentEventData));
238  evtData = reinterpret_cast<AliHLTComponentEventData*>(msg->GetData());
239  memset(evtData, 0, sizeof(AliHLTComponentEventData));
240  evtData->fStructSize = sizeof(AliHLTComponentEventData);
241  evtData->fEventID=mNEvents;
242  system_clock::time_point timestamp = system_clock::now();
243  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(timestamp - dayref);
244  evtData->fEventCreation_s=seconds.count();
245  auto useconds = std::chrono::duration_cast<std::chrono::microseconds>(timestamp - dayref - seconds);
246  evtData->fEventCreation_us=useconds.count();
247  if (mVerbosity>0) {
248  LOG(DEBUG) << "send event " << evtData->fEventID << " at " << evtData->fEventCreation_s << "s " << evtData->fEventCreation_us << "us";
249  }
250 
251  for (int iOutput=0; iOutput<numOutputs; iOutput++) {
252  fChannels["data-out"].at(iOutput).Send(msg);
253  }
254 
255  mNEvents++;
256  if (eventPeriodInSeconds>0) sleep(eventPeriodInSeconds);
257  else usleep(mEventPeriod);
258  }
259 }
