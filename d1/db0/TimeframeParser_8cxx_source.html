63 void streamTimeframe(std::istream& stream,
64  std::function<void(FairMQParts& parts, char* buffer, size_t size)> onAddPart,
65  std::function<void(FairMQParts& parts)> onSend)
66 {
67  FairMQParts parts;
68  StreamingState state;
69  assert(state.state == PARSE_BEGIN_STREAM);
70  while (true) {
71  switch (state.state) {
72  case PARSE_BEGIN_STREAM:
73  LOG(INFO) << "In PARSE_BEGIN_STREAM\n";
74  state.state = PARSE_BEGIN_TIMEFRAME;
75  break;
76  case PARSE_BEGIN_TIMEFRAME:
77  LOG(INFO) << "In PARSE_BEGIN_TIMEFRAME\n";
78  state.state = PARSE_BEGIN_PAIR;
79  break;
80  case PARSE_BEGIN_PAIR:
81  LOG(INFO) << "In PARSE_BEGIN_PAIR\n";
82  state.state = PARSE_DATA_HEADER;
83  state.hasDataHeader = false;
84  state.payloadBuffer = nullptr;
85  state.headerBuffer = nullptr;
86  break;
87  case PARSE_DATA_HEADER:
88  LOG(INFO) << "In PARSE_DATA_HEADER\n";
89  if (state.hasDataHeader) {
90  throw std::runtime_error("DataHeader already present.");
91  } else if (state.payloadBuffer) {
92  throw std::runtime_error("Unexpected payload.");
93  }
94  LOG(INFO) << "Reading dataheader of " << sizeof(state.dh) << " bytes\n";
95  stream.read(reinterpret_cast<char*>(&state.dh), sizeof(state.dh));
96  // If we have a TIMEFRAMEINDEX part and we find the eof, we are done.
97  if (stream.eof()) {
98  throw std::runtime_error("Premature end of stream");
99  }
100 
101  // Otherwise we move to the state which is responsible for parsing the
102  // kind of header.
103  state.state = PARSE_CONCRETE_HEADER;
104  break;
105  case PARSE_CONCRETE_HEADER:
106  LOG(INFO) << "In PARSE_CONCRETE_HEADER\n";
107  if (state.headerBuffer) {
108  throw std::runtime_error("File has two consecutive headers");
109  }
110  if (state.dh.headerSize < sizeof(DataHeader)) {
111  std::ostringstream str;
112  str << "Bad header size. Should be greater then "
113  << sizeof(DataHeader)
114  << ". Found " << state.dh.headerSize << "\n";
115  throw std::runtime_error(str.str());
116  }
117  // We get the full header size and read the rest of the header
118  state.headerBuffer = malloc(state.dh.headerSize);
119  memcpy(state.headerBuffer, &state.dh, sizeof(state.dh));
120  LOG(INFO) << "Reading rest of the header of " << state.dh.headerSize - sizeof(state.dh) << " bytes\n";
121  stream.read(reinterpret_cast<char*>(state.headerBuffer) + sizeof(state.dh),
122  state.dh.headerSize - sizeof(state.dh));
123  // Handle the case the file was truncated.
124  if (stream.eof()) {
125  throw std::runtime_error("Unexpected end of file");
126  }
127  onAddPart(parts, reinterpret_cast<char*>(state.headerBuffer), state.dh.headerSize);
128  // Move to parse the payload
129  state.state = PARSE_PAYLOAD;
130  break;
131  case PARSE_PAYLOAD:
132  LOG(INFO) << "In PARSE_PAYLOAD\n";
133  if (state.payloadBuffer) {
134  throw std::runtime_error("File has two consecutive payloads");
135  }
136  state.payloadBuffer = new char[state.dh.payloadSize];
137  LOG(INFO) << "Reading payload of " << state.dh.payloadSize << " bytes\n";
138  stream.read(reinterpret_cast<char*>(state.payloadBuffer), state.dh.payloadSize);
139  if (stream.eof()) {
140  throw std::runtime_error("Unexpected end of file");
141  }
142  onAddPart(parts, reinterpret_cast<char*>(state.payloadBuffer), state.dh.payloadSize);
143  state.state = PARSE_END_PAIR;
144  break;
145  case PARSE_END_PAIR:
146  LOG(INFO) << "In PARSE_END_PAIR\n";
147  state.state = state.dh == DataDescription("TIMEFRAMEINDEX") ? PARSE_END_TIMEFRAME : PARSE_BEGIN_PAIR;
148  break;
149  case PARSE_END_TIMEFRAME:
150  LOG(INFO) << "In PARSE_END_TIMEFRAME\n";
151  onSend(parts);
152  // Check if we have more. If not, we can declare success.
153  stream.peek();
154  if (stream.eof()) {
155  state.state = PARSE_END_STREAM;
156  } else {
157  state.state = PARSE_BEGIN_TIMEFRAME;
158  }
159  break;
160  case PARSE_END_STREAM:
161  return;
162  break;
163  default:
164  break;
165  }
166  }
167 }

169 void streamTimeframe(std::ostream& stream, FairMQParts& parts)
170 {
171  if (parts.Size() < 2) {
172  throw std::runtime_error("Expecting at least 2 parts\n");
173  }
174 
175  auto indexHeader = o2::header::get<DataHeader*>(parts.At(parts.Size() - 2)->GetData());
176  // FIXME: Provide iterator pair API for the index
177  // Index should really be something which provides an
178  // iterator pair API so that we can sort / find / lower_bound
179  // easily. Right now we simply use it a C-style array.
180  auto index = reinterpret_cast<IndexElement*>(parts.At(parts.Size() - 1)->GetData());
181 
182  LOG(INFO) << "This time frame has " << parts.Size() << " parts.\n";
183  auto indexEntries = indexHeader->payloadSize / sizeof(IndexElement);
184  if (indexHeader->dataDescription != DataDescription("TIMEFRAMEINDEX")) {
185  throw std::runtime_error("Could not find a valid index header\n");
186  }
187  LOG(INFO) << indexHeader->dataDescription.str << "\n";
188  LOG(INFO) << "This time frame has " << indexEntries << "entries in the index.\n";
189  if ((indexEntries * 2 + 2) != (parts.Size())) {
190  std::stringstream err;
191  err << "Mismatched index and received parts. Expected "
192  << (parts.Size() - 2 * 2) << " found " << indexEntries;
193  throw std::runtime_error(err.str());
194  }
195 
196  LOG(INFO) << "Everything is fine with received timeframe\n";
197  for (size_t i = 0; i < parts.Size(); ++i) {
198  stream.write(reinterpret_cast<const char*>(parts.At(i)->GetData()),
199  parts.At(i)->GetSize());
200  }
201 }
