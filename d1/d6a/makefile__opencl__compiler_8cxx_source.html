82  clGetPlatformInfo(platforms[i_platform], CL_PLATFORM_VERSION, 64, pinfo.platform_version, nullptr);

88  printf("AMD OpenCL Platform found (%u)\n", i_platform);

97  if (clGetDeviceIDs(platform, CL_DEVICE_TYPE_ALL, 0, nullptr, &pinfo.count) != CL_SUCCESS) {

104  quit("Memory allocation error");
105  }

110  _makefiles_opencl_device_info dinfo;

118  clGetDeviceInfo(devices[i], CL_DEVICE_NAME, 64, dinfo.device_name, nullptr);

122  clGetDeviceInfo(devices[i], CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(shaders), &shaders, nullptr);
123  clGetDeviceInfo(devices[i], CL_DEVICE_ADDRESS_BITS, sizeof(dinfo.nbits), &dinfo.nbits, nullptr);
124  printf("Found Device %u : %s %s (Frequency %d, Shaders %d, %d bit)\n", i, dinfo.device_vendor, dinfo.device_name, (int)freq, (int)shaders, (int)dinfo.nbits);
125  }
126 
127  if (files.size() == 0) {

219  memset(binary_buffers[i], 0, binary_sizes[i]);

226  printf("Programs obtained successfully, cleaning up opencl\n");

241  clGetDeviceInfo(devices[i], CL_DEVICE_VENDOR, 64, dinfo.device_vendor, nullptr);
242  dinfo.binary_size = binary_sizes[i];

244  fwrite(binary_buffers[i], 1, binary_sizes[i], fp);

249  for (unsigned int i = 0; i < pinfo.count; i++) {
