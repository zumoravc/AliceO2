50 HalfSegmentation::HalfSegmentation(const Char_t* nameGeomFile, const Short_t id) : VSegmentation(), mHalfDisks(nullptr)
51 {
52 
53  Geometry* mftGeom = Geometry::instance();
54 
55  UInt_t halfUniqueID = mftGeom->getObjectID(Geometry::HalfType, id);
56  SetUniqueID(halfUniqueID);
57  SetName(Form("%s_%d", GeometryTGeo::getMFTHalfPattern(), id));
58 
59  mHalfDisks = new TClonesArray("o2::mft::HalfDiskSegmentation", constants::DisksNumber);
60  mHalfDisks->SetOwner(kTRUE);
61 
62  // Create XML engine
63  auto* geomFile = new TXMLEngine;
64 
65  // take access to main node
66  XMLDocPointer_t xmldoc = geomFile->ParseFile(nameGeomFile);
67  if (xmldoc == nullptr) {
68  delete geomFile;
69  LOG(FATAL) << "Could not parse Geometry XML File named " << nameGeomFile;
70  }
71  XMLNodePointer_t mainnode = geomFile->DocGetRootElement(xmldoc);
72 
73  // Find Half-MFT node in the XML file
74  XMLNodePointer_t halfnode;
75  findHalf(geomFile, mainnode, halfnode);
76 
77  // Create Half Disks belonging to that Half-MFT
78  createHalfDisks(geomFile, halfnode);
79 
80  // Release memory
81  geomFile->FreeDoc(xmldoc);
82  delete geomFile;

109 void HalfSegmentation::createHalfDisks(TXMLEngine* xml, XMLNodePointer_t node)
110 {
111  // this function display all accessible information about xml node and its children
112  Int_t idisk;
113  Int_t nladder;
114  Double_t pos[3] = {0., 0., 0.};
115  Double_t ang[3] = {0., 0., 0.};
116 
117  Geometry* mftGeom = Geometry::instance();
118 
119  TString nodeName = xml->GetNodeName(node);
120  if (!nodeName.CompareTo("disk")) {
121  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
122  while (attr != nullptr) {
123  TString attrName = xml->GetAttrName(attr);
124  TString attrVal = xml->GetAttrValue(attr);
125  if (!attrName.CompareTo("idisk")) {
126  idisk = attrVal.Atoi();
127  if (idisk >= constants::DisksNumber || idisk < 0) {
128  LOG(FATAL) << "Wrong disk number : " << idisk;
129  }
130  } else if (!attrName.CompareTo("nladder")) {
131  nladder = attrVal.Atoi();
132  } else if (!attrName.CompareTo("xpos")) {
133  pos[0] = attrVal.Atof();
134  } else if (!attrName.CompareTo("ypos")) {
135  pos[1] = attrVal.Atof();
136  } else if (!attrName.CompareTo("zpos")) {
137  pos[2] = attrVal.Atof();
138  } else if (!attrName.CompareTo("phi")) {
139  ang[0] = attrVal.Atof();
140  } else if (!attrName.CompareTo("theta")) {
141  ang[1] = attrVal.Atof();
142  } else if (!attrName.CompareTo("psi")) {
143  ang[2] = attrVal.Atof();
144  } else {
145  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr);
146  }
147  attr = xml->GetNextAttr(attr);
148  }
149 
150  UInt_t diskUniqueID = mftGeom->getObjectID(Geometry::HalfDiskType, mftGeom->getHalfID(GetUniqueID()), idisk);
151 
152  auto* halfDisk = new HalfDiskSegmentation(diskUniqueID);
153  halfDisk->setPosition(pos);
154  halfDisk->setRotationAngles(ang);
155  halfDisk->setNLadders(nladder);
156  halfDisk->createLadders(xml, node);
157  if (halfDisk->getNLaddersBuild() != halfDisk->getNLadders()) {
158  LOG(FATAL) << "Number of ladder build " << halfDisk->getNLaddersBuild()
159  << " does not correspond to the number declared " << halfDisk->getNLadders() << " Check XML file";
160  }
161  new ((*mHalfDisks)[idisk]) HalfDiskSegmentation(*halfDisk);
162  delete halfDisk;
163  }
164 
165  // display all child nodes
166  XMLNodePointer_t child = xml->GetChild(node);
167  while (child != nullptr) {
168  createHalfDisks(xml, child);
169  child = xml->GetNext(child);
170  }
171 }

176 void HalfSegmentation::findHalf(TXMLEngine* xml, XMLNodePointer_t node, XMLNodePointer_t& retnode)
177 {
178  // Find in the XML Geometry File the node corresponding to the Half-MFT being build
179  // Set Position and Orientation of the Half-MFT
180  Int_t isTop;
181  Int_t ndisk;
182  Double_t pos[3] = {0., 0., 0.};
183  Double_t ang[3] = {0., 0., 0.};
184 
185  TString nodeName = xml->GetNodeName(node);
186  if (!nodeName.CompareTo("half")) {
187  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
188  while (attr != nullptr) {
189  TString attrName = xml->GetAttrName(attr);
190  TString attrVal = xml->GetAttrValue(attr);
191  if (!attrName.CompareTo("top")) {
192  isTop = attrVal.Atoi();
193  if (isTop > 1 || isTop < 0) {
194  LOG(FATAL) << "Wrong Half MFT number : " << isTop;
195  }
196  } else if (!attrName.CompareTo("ndisk")) {
197  ndisk = attrVal.Atoi();
198  if (ndisk > 5 || ndisk < 0) {
199  LOG(ERROR) << "Wrong number of disk : " << ndisk;
200  }
201 
202  } else if (!attrName.CompareTo("xpos")) {
203  pos[0] = attrVal.Atof();
204  } else if (!attrName.CompareTo("ypos")) {
205  pos[1] = attrVal.Atof();
206  } else if (!attrName.CompareTo("zpos")) {
207  pos[2] = attrVal.Atof();
208  } else if (!attrName.CompareTo("phi")) {
209  ang[0] = attrVal.Atof();
210  } else if (!attrName.CompareTo("theta")) {
211  ang[1] = attrVal.Atof();
212  } else if (!attrName.CompareTo("psi")) {
213  ang[2] = attrVal.Atof();
214  } else {
215  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr);
216  }
217 
218  attr = xml->GetNextAttr(attr);
219  }
220 
221  Geometry* mftGeom = Geometry::instance();
222  if (isTop == mftGeom->getHalfID(GetUniqueID())) {
223  setPosition(pos);
224  setRotationAngles(ang);
225  retnode = node;
226  return;
227  }
228  }
229 
230  // display all child nodes
231  XMLNodePointer_t child = xml->GetChild(node);
232  while (child != nullptr) {
233  findHalf(xml, child, retnode);
234  child = xml->GetNext(child);
235  }
236 }
