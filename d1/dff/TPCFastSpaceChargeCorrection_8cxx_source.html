299 void TPCFastSpaceChargeCorrection::finishConstruction()
300 {
302 
303  assert(mConstructionMask & ConstructionState::InProgress);
304 
305  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
306  assert(mConstructionRowInfos[i].splineScenarioID >= 0);
307  }
308  for (int i = 0; i < mNumberOfScenarios; i++) {
309  assert(mConstructionScenarios[i].isConstructed());
310  }
311 
312  // organize memory for the flat buffer and caculate its size
313 
314  size_t rowsOffset = 0;
315  size_t rowsSize = sizeof(RowInfo) * mGeo.getNumberOfRows();
316 
317  size_t sliceRowsOffset = rowsSize;
318  size_t sliceRowsSize = sizeof(SliceRowInfo) * mGeo.getNumberOfRows() * mGeo.getNumberOfSlices();
319 
320  size_t scOffset = alignSize(sliceRowsOffset + sliceRowsSize, SplineType::getClassAlignmentBytes());
321  size_t scSize = sizeof(SplineType) * mNumberOfScenarios;
322 
323  size_t scBufferOffsets[mNumberOfScenarios];
324 
325  scBufferOffsets[0] = alignSize(scOffset + scSize, SplineType::getBufferAlignmentBytes());
326  size_t scBufferSize = 0;
327  for (int i = 0; i < mNumberOfScenarios; i++) {
328  SplineType& sp = mConstructionScenarios[i];
329  scBufferOffsets[i] = scBufferOffsets[0] + scBufferSize;
330  scBufferSize = alignSize(scBufferSize + sp.getFlatBufferSize(), sp.getBufferAlignmentBytes());
331  }
332  size_t bufferSize = scBufferOffsets[0] + scBufferSize;
333  size_t sliceDataOffset[3];
334  for (int is = 0; is < 3; is++) {
335  sliceDataOffset[is] = alignSize(bufferSize, SplineType::getParameterAlignmentBytes());
336  mSliceDataSizeBytes[is] = 0;
337  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
338  RowInfo& row = mConstructionRowInfos[i];
339  SplineType& spline = mConstructionScenarios[row.splineScenarioID];
340  row.dataOffsetBytes[is] = alignSize(mSliceDataSizeBytes[is], SplineType::getParameterAlignmentBytes());
341  mSliceDataSizeBytes[is] = row.dataOffsetBytes[is] + spline.getSizeOfParameters();
342  }
343  mSliceDataSizeBytes[is] = alignSize(mSliceDataSizeBytes[is], SplineType::getParameterAlignmentBytes());
344  bufferSize = sliceDataOffset[is] + mSliceDataSizeBytes[is] * mGeo.getNumberOfSlices();
345  }
346 
347  FlatObject::finishConstruction(bufferSize);
348 
349  mRowInfoPtr = reinterpret_cast<RowInfo*>(mFlatBufferPtr + rowsOffset);
350  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
351  mRowInfoPtr[i] = mConstructionRowInfos[i];
352  }
353 
354  mSliceRowInfoPtr = reinterpret_cast<SliceRowInfo*>(mFlatBufferPtr + sliceRowsOffset);
355  for (int s = 0; s < mGeo.getNumberOfSlices(); s++) {
356  for (int r = 0; r < mGeo.getNumberOfRows(); r++) {
357  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].CorrU0 = 0;
358  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].scaleCorrUtoGrid = 0;
359  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].scaleCorrVtoGrid = 0;
360  }
361  }
362 
363  mScenarioPtr = reinterpret_cast<SplineType*>(mFlatBufferPtr + scOffset);
364 
365  for (int i = 0; i < mNumberOfScenarios; i++) {
366  SplineType& sp0 = mConstructionScenarios[i];
367  SplineType& sp1 = mScenarioPtr[i];
368  new (&sp1) SplineType(); // first, call a constructor
369  sp1.cloneFromObject(sp0, mFlatBufferPtr + scBufferOffsets[i]);
370  }
371 
372  for (int is = 0; is < 3; is++) {
373  mSplineData[is] = reinterpret_cast<char*>(mFlatBufferPtr + sliceDataOffset[is]);
374  }
375  releaseConstructionMemory();
376 
377  mTimeStamp = -1;
378 
379  setNoCorrection();
380 }

382 GPUd() void TPCFastSpaceChargeCorrection::setNoCorrection()
383 {
384  // initialise all corrections to 0.
385  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
386  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
387  SliceInfo& sliceInfo = getSliceInfo(slice);
388  sliceInfo.vMax = vLength;
389  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
390  const SplineType& spline = getSpline(slice, row);
391 
392  for (int is = 0; is < 3; is++) {
393  float* data = getSplineData(slice, row, is);
394  int nPar = spline.getNumberOfParameters();
395  if (is == 1) {
396  nPar = nPar / 3;
397  }
398  if (is == 2) {
399  nPar = nPar * 2 / 3;
400  }
401  for (int i = 0; i < nPar; i++) {
402  data[i] = 0.f;
403  }
404  }
405 
406  SliceRowInfo& info = getSliceRowInfo(slice, row);
407  RowActiveArea& area = info.activeArea;
408  for (int i = 1; i < 5; i++) {
409  area.maxDriftLengthCheb[i] = 0;
410  }
411  area.maxDriftLengthCheb[0] = vLength;
412  mGeo.convPadToU(row, 0., area.cuMin);
413  area.cuMax = -area.cuMin;
414  area.vMax = vLength;
415  area.cvMax = vLength;
416  info.CorrU0 = area.cuMin;
417  info.scaleCorrUtoGrid = (spline.getGridU1().getNumberOfKnots() - 1) / (area.cuMax - area.cuMin);
418  info.scaleCorrVtoGrid = (spline.getGridU2().getNumberOfKnots() - 1) / area.cvMax;
419  } // row
420  } // slice
421 }

423 void TPCFastSpaceChargeCorrection::initMaxDriftLength(bool prn)
424 {
425  double tpcR2min = mGeo.getRowInfo(0).x - 1.;
426  tpcR2min = tpcR2min * tpcR2min;
427  double tpcR2max = mGeo.getRowInfo(mGeo.getNumberOfRows() - 1).x;
428  tpcR2max = tpcR2max / cos(2 * M_PI / mGeo.getNumberOfSlicesA() / 2) + 1.;
429  tpcR2max = tpcR2max * tpcR2max;
430 
431  ChebyshevFit1D chebFitter;
432 
433  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
434  if (prn) {
435  std::cout << "init MaxDriftLength for slice " << slice << std::endl;
436  }
437  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
438  SliceInfo& sliceInfo = getSliceInfo(slice);
439  sliceInfo.vMax = 0.f;
440 
441  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
442  RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
443  area.cvMax = 0;
444  area.vMax = 0;
445  mGeo.convPadToU(row, 0., area.cuMin);
446  area.cuMax = -area.cuMin;
447  chebFitter.reset(4, 0., mGeo.getRowInfo(row).maxPad);
448  double x = mGeo.getRowInfo(row).x;
449  for (int pad = 0; pad < mGeo.getRowInfo(row).maxPad; pad++) {
450  float u = 0;
451  mGeo.convPadToU(row, pad, u);
452  float v0 = 0;
453  float v1 = 1.1 * vLength;
454  float vLastValid = -1;
455  float cvLastValid = -1;
456  while (v1 - v0 > 0.1) {
457  float v = 0.5 * (v0 + v1);
458  float dx, du, dv;
459  getCorrection(slice, row, u, v, dx, du, dv);
460  double cx = x + dx;
461  double cu = u + du;
462  double cv = v + dv;
463  double r2 = cx * cx + cu * cu;
464  if (cv < 0) {
465  v0 = v;
466  } else if (cv <= vLength && r2 >= tpcR2min && r2 <= tpcR2max) {
467  v0 = v;
468  vLastValid = v;
469  cvLastValid = cv;
470  } else {
471  v1 = v;
472  }
473  }
474  if (vLastValid > 0.) {
475  chebFitter.addMeasurement(pad, vLastValid);
476  }
477  if (area.vMax < vLastValid) {
478  area.vMax = vLastValid;
479  }
480  if (area.cvMax < cvLastValid) {
481  area.cvMax = cvLastValid;
482  }
483  }
484  chebFitter.fit();
485  for (int i = 0; i < 5; i++) {
486  area.maxDriftLengthCheb[i] = chebFitter.getCoefficients()[i];
487  }
488  if (sliceInfo.vMax < area.vMax) {
489  sliceInfo.vMax = area.vMax;
490  }
491  } // row
492  } // slice
493 }

495 void TPCFastSpaceChargeCorrection::initInverse(bool prn)
496 {
497 
498  initMaxDriftLength(prn);
499 
500  SplineHelper2D<float> helper;
501  std::vector<float> dataPointF;
502  std::vector<float> splineParameters;
503 
504  ChebyshevFit1D chebFitterX, chebFitterU, chebFitterV;
505 
506  double tpcR2min = mGeo.getRowInfo(0).x - 1.;
507  tpcR2min = tpcR2min * tpcR2min;
508  double tpcR2max = mGeo.getRowInfo(mGeo.getNumberOfRows() - 1).x;
509  tpcR2max = tpcR2max / cos(2 * M_PI / mGeo.getNumberOfSlicesA() / 2) + 1.;
510  tpcR2max = tpcR2max * tpcR2max;
511 
512  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
513  //std::cout << "inverse transform for slice " << slice << std::endl;
514  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
515  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
516  const SplineType& spline = getSpline(slice, row);
517  helper.setSpline(spline, 3, 3);
518 
519  float u0, u1, v0, v1;
520  mGeo.convScaledUVtoUV(slice, row, 0., 0., u0, v0);
521  mGeo.convScaledUVtoUV(slice, row, 1., 1., u1, v1);
522 
523  double x = mGeo.getRowInfo(row).x;
524  double stepU = (u1 - u0) / (1. * (helper.getNumberOfDataPointsU1() - 1));
525  double stepV = (v1 - v0) / (1. * (helper.getNumberOfDataPointsU2() - 1));
526 
527  if (prn) {
528  std::cout << "u0 " << u0 << " u1 " << u1 << " v0 " << v0 << " v1 " << v1 << std::endl;
529  }
530  RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
531  area.cuMin = 1.e10;
532  area.cuMax = -1.e10;
533 
534  v1 = area.vMax;
535  stepV = (v1 - v0) / (1. * (helper.getNumberOfDataPointsU2() - 1));
536  if (stepV < 1.f) {
537  stepV = 1.f;
538  }
539  int nCheb = helper.getNumberOfDataPointsU2();
540  nCheb = 20;
541  chebFitterV.reset(nCheb - 1, 0, vLength);
542 
543  struct Entry {
544  double cu, cv, dx, du, dv;
545  };
546  std::vector<Entry> dataRowsV[helper.getNumberOfDataPointsU2()];
547 
548  for (double u = u0; u < u1 + stepU; u += stepU) {
549  chebFitterV.reset();
550  //double vvMax = 0;
551  for (double v = v0; v < v1 + stepV; v += stepV) {
552  float dx, du, dv;
553  getCorrection(slice, row, u, v, dx, du, dv);
554  double cx = x + dx;
555  double cu = u + du;
556  double cv = v + dv;
557  double r2 = cx * cx + cu * cu;
558  if (cv < 0 || cv > vLength || r2 < tpcR2min || r2 > tpcR2max) {
559  continue;
560  }
561  if (cu < area.cuMin) {
562  area.cuMin = cu;
563  }
564  if (cu > area.cuMax) {
565  area.cuMax = cu;
566  }
567  //if (v > vvMax) {
568  //vvMax = v;
569  //}
570  if (prn) {

692 double TPCFastSpaceChargeCorrection::testInverse(bool prn)
693 {
694  if (prn) {
695  std::cout << "Test inverse transform " << std::endl;
696  }
697 
698  double tpcR2min = mGeo.getRowInfo(0).x - 1.;
699  tpcR2min = tpcR2min * tpcR2min;
700  double tpcR2max = mGeo.getRowInfo(mGeo.getNumberOfRows() - 1).x;
701  tpcR2max = tpcR2max / cos(2 * M_PI / mGeo.getNumberOfSlicesA() / 2) + 1.;
702  tpcR2max = tpcR2max * tpcR2max;
703 
704  double maxDtpc[3] = {0, 0, 0};
705  double maxD = 0;
706 
707  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
708  if (prn) {
709  std::cout << "check inverse transform for slice " << slice << std::endl;
710  }
711  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
712  double maxDslice[3] = {0, 0, 0};
713  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
714  const SplineType& spline = getSpline(slice, row);
715  float u0, u1, v0, v1;
716  mGeo.convScaledUVtoUV(slice, row, 0., 0, u0, v0);
717  mGeo.convScaledUVtoUV(slice, row, 1., 1, u1, v1);
718  double x = mGeo.getRowInfo(row).x;
719  double stepU = (u1 - u0) / 100.;
720  double stepV = (v1 - v0) / 100.;
721  double maxDrow[3] = {0, 0, 0};
722  for (double u = u0; u < u1; u += stepU) {
723  for (double v = v0; v < v1; v += stepV) {
724  float dx, du, dv;
725  getCorrection(slice, row, u, v, dx, du, dv);
726  double cx = x + dx;
727  double cu = u + du;
728  double cv = v + dv;
729  double r2 = cx * cx + cu * cu;
730  if (cv < 0 || cv > vLength || r2 < tpcR2min || r2 > tpcR2max) {
731  continue;
732  }
733  float nx, nu, nv;
734  getCorrectionInvCorrectedX(slice, row, cu, cv, nx);
735  getCorrectionInvUV(slice, row, cu, cv, nu, nv);
736  double d[3] = {nx - cx, nu - u, nv - v};
737  for (int i = 0; i < 3; i++) {
738  if (fabs(d[i]) > fabs(maxDrow[i]))
739  maxDrow[i] = d[i];
740  }
741 
742  if (prn && fabs(d[0]) + fabs(d[1]) + fabs(d[2]) > 0.1) {

756  maxDslice[i] = maxDrow[i];
757  }
