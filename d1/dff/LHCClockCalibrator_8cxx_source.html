35 void LHCClockDataHisto::fill(const gsl::span<const o2::dataformats::CalibInfoTOF> data)
36 {
37  // fill container
38  for (int i = data.size(); i--;) {
39  auto dt = data[i].getDeltaTimePi();
40  dt += range;
41  if (dt > 0 && dt < 2 * range) {
42  histo[int(dt * v2Bin)]++;
43  entries++;
44  }
45  }
46 }

75 //_____________________________________________
76 void LHCClockCalibrator::finalizeSlot(Slot& slot)
77 {
78  // Extract results for the single slot
79  o2::tof::LHCClockDataHisto* c = slot.getContainer();
80  LOG(INFO) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " with "
81  << c->getEntries() << " entries";
82  std::vector<float> fitValues;
83  float* array = &c->histo[0];
84  int fitres = fitGaus(c->nbins, array, -(c->range), c->range, fitValues);
85  if (fitres >= 0) {
86  LOG(INFO) << "Fit result " << fitres << " Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
87  } else {
88  LOG(ERROR) << "Fit failed with result = " << fitres;
89  }
90 
91  // TODO: the timestamp is now given with the TF index, but it will have
92  // to become an absolute time. This is true both for the lhc phase object itself
93  // and the CCDB entry
94  std::map<std::string, std::string> md;
95  LHCphase l;
96  l.addLHCphase(slot.getTFStart(), fitValues[1]);
97  auto clName = o2::utils::MemFileHelper::getClassName(l);
98  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
99  mInfoVector.emplace_back("TOF/LHCphase", clName, flName, md, slot.getTFStart(), 99999999999999);
100  mLHCphaseVector.emplace_back(l);
101 
102  slot.print();
103 }
