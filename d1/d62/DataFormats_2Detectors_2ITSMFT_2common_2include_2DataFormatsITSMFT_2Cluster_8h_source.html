174 inline void Cluster::setClusterUsage(Int_t n)
175 {
176  // set cluster usage counter
177  mNxNzN &= ~(kMaskClUse << kOffsClUse);
178  mNxNzN |= (n & kMaskClUse) << kOffsClUse;

182  resetBit(kUsed);
