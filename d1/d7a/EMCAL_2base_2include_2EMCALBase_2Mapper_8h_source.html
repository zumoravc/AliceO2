59  struct ChannelIDHasher {
60 
64  size_t operator()(const ChannelID& s) const
65  {
66  size_t h1 = std::hash<int>()(s.mRow);
67  size_t h2 = std::hash<int>()(s.mColumn);
68  size_t h3 = std::hash<int>()(o2::emcal::channelTypeToInt(s.mChannelType));
69  return ((h1 ^ (h2 << 1)) >> 1) ^ (h3 << 1);
70  return h1 ^ (h2 << 1);
71  }
