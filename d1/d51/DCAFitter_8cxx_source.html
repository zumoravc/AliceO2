26 void DCAFitter::CrossInfo::set(const TrcAuxPar& trc0, const TrcAuxPar& trc1)
27 {
28  // calculate up to 2 crossings between 2 circles
29  nDCA = 0;
30  const auto& trcA = trc0.r > trc1.r ? trc0 : trc1; // designate the largest circle as A
31  const auto& trcB = trc0.r > trc1.r ? trc1 : trc0;
32  ftype_t xDist = trcB.xCen - trcA.xCen, yDist = trcB.yCen - trcA.yCen;
33  ftype_t dist2 = xDist * xDist + yDist * yDist, dist = TMath::Sqrt(dist2), rsum = trcA.r + trcB.r;
34  if (TMath::Abs(dist) < 1e-12) {
35  return; // circles are concentric?
36  }
37  if (dist > rsum) { // circles don't touch, chose a point in between
38  // the parametric equation of lines connecting the centers is
39  // x = x0 + t/dist * (x1-x0), y = y0 + t/dist * (y1-y0)
40  notTouchingXY(dist, xDist, yDist, trcA, trcB.r);
41  } else if (dist + trcB.r < trcA.r) { // the small circle is nestled into large one w/o touching
42  // select the point of closest approach of 2 circles
43  notTouchingXY(dist, xDist, yDist, trcA, -trcB.r);
44  } else { // 2 intersection points
45  // to simplify calculations, we move to new frame x->x+Xc0, y->y+Yc0, so that
46  // the 1st one is centered in origin
47  if (TMath::Abs(xDist) < TMath::Abs(yDist)) {
48  ftype_t a = (trcA.r * trcA.r - trcB.r * trcB.r + dist2) / (2. * yDist), b = -xDist / yDist, ab = a * b, bb = b * b;
49  ftype_t det = ab * ab - (1. + bb) * (a * a - trcA.r * trcA.r);
50  if (det > 0.) {
51  det = TMath::Sqrt(det);
52  xDCA[0] = (-ab + det) / (1. + b * b);
53  yDCA[0] = a + b * xDCA[0] + trcA.yCen;
54  xDCA[0] += trcA.xCen;
55  xDCA[1] = (-ab - det) / (1. + b * b);
56  yDCA[1] = a + b * xDCA[1] + trcA.yCen;
57  xDCA[1] += trcA.xCen;
58  nDCA = 2;
59  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
60  notTouchingXY(dist, xDist, yDist, trcA, trcB.r);
61  }
62  } else {
63  ftype_t a = (trcA.r * trcA.r - trcB.r * trcB.r + dist2) / (2. * xDist), b = -yDist / xDist, ab = a * b, bb = b * b;
64  ftype_t det = ab * ab - (1. + bb) * (a * a - trcA.r * trcA.r);
65  if (det > 0.) {
66  det = TMath::Sqrt(det);
67  yDCA[0] = (-ab + det) / (1. + bb);
68  xDCA[0] = a + b * yDCA[0] + trcA.xCen;
69  yDCA[0] += trcA.yCen;
70  yDCA[1] = (-ab - det) / (1. + bb);
71  xDCA[1] = a + b * yDCA[1] + trcA.xCen;
72  yDCA[1] += trcA.yCen;
73  nDCA = 2;
74  } else { // due to the finite precision the det<=0, i.e. the circles are barely touching, fall back to this special case
75  notTouchingXY(dist, xDist, yDist, trcA, trcB.r);
76  }
77  }
78  }
79 }

82 void DCAFitter::TrcAuxPar::setRCen(const Track& tr, ftype_t bz)
83 {
84  // set track radius and circle coordinates in global frame, no check for bz==0!
85  r = 1. / tr.getCurvature(bz);
86  ftype_t sn = tr.getSnp(), cs = TMath::Sqrt((1. - sn) * (1. + sn));
87  loc2glo(tr.getX() - sn * r, tr.getY() + cs * r, xCen, yCen);
88  r = TMath::Abs(r);
89 }

92 int DCAFitter::process(const Track& trc0, const TrcAuxPar& trc0Aux,
93  const Track& trc1, const TrcAuxPar& trc1Aux)
94 {
95  // find dca of 2 tracks with aux info preclaculated
96  mNCandidates = 0;
97  mCrossings.set(trc0Aux, trc1Aux); // find at most 2 candidates of 2 circles crossing
98  if (!mCrossings.nDCA) {
99  return 0; // no crossing
100  }
101  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
102  // both XY crossings may eventually converge to the same point. To stop asap this redundant step
103  // we abandon fit if it appears to be closer to alternative crossing
104  mCrossIDCur = ic;
105  mCrossIDAlt = mCrossings.nDCA == 2 ? 1 - ic : -1;
106  // check if radius is acceptable
107  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
108  continue;
109  }
110  // find dca starting from proximity of transverse point xv,yv
111  ftype_t xl, yl;
112  mCandTr0[mNCandidates] = trc0;
113  mCandTr1[mNCandidates] = trc1;
114  trc0Aux.glo2loc(mCrossings.xDCA[ic], mCrossings.yDCA[ic], xl, yl);
115  if (!mCandTr0[mNCandidates].propagateTo(xl, mBz)) {
116  continue;
117  }
118  trc1Aux.glo2loc(mCrossings.xDCA[ic], mCrossings.yDCA[ic], xl, yl);
119  if (!mCandTr1[mNCandidates].propagateTo(xl, mBz)) {
120  continue;
121  }
122  if (mUseAbsDCA ? processCandidateDCA(trc0Aux, trc1Aux) : processCandidateChi2(trc0Aux, trc1Aux)) {
123  mNCandidates++; // candidate validated
124  }
125  }
126  return mNCandidates;
127 }

130 bool DCAFitter::processCandidateChi2(const TrcAuxPar& trc0Aux, const TrcAuxPar& trc1Aux)
131 {
132  // find best chi2 (weighted DCA) of 2 tracks already propagated to their approximate vicinity
133  mChi2[mNCandidates] = 1e9;
134  Track &trc0 = mCandTr0[mNCandidates], &trc1 = mCandTr1[mNCandidates];
135  TrackCovI &trcEI0 = mTrcEI0[mNCandidates], &trcEI1 = mTrcEI1[mNCandidates];
136  // get error matrices at initial point
137  trcEI0.set(trc0);
138  trcEI1.set(trc1);
139  TrackCoefVtx &trCFVT0 = mTrCFVT0[mNCandidates], &trCFVT1 = mTrCFVT1[mNCandidates]; // get coefficients of PCA vs track points
140  calcPCACoefs(trc0Aux, trcEI0, trc1Aux, trcEI1, trCFVT0, trCFVT1);
141  TrackDeriv2 tDer0, tDer1; // their derivatives over track param X
142  Derivatives deriv; // chi2 1st and 2nd derivatives over tracks params
143 
144  ftype_t chi2 = 0, chi2Prev = 0;
145  int iter = 0;
146  do {
147  tDer0.set(trc0, mBz); // tracks derivatives
148  tDer1.set(trc1, mBz); // over their X params
149  chi2Deriv(trc0, tDer0, trc0Aux, trcEI0, trCFVT0, trc1, tDer1, trc1Aux, trcEI1, trCFVT1, deriv);
150 
151  // do Newton-Rapson iteration with corrections = - dchi2/d{x0,x1} * [ d^2chi2/d{x0,x1}^2 ]^-1
152  dtype_t detDer2 = deriv.dChidx0dx0 * deriv.dChidx1dx1 - deriv.dChidx0dx1 * deriv.dChidx0dx1;
153  dtype_t detDer2I = 1. / detDer2;
154  dtype_t dX0 = -(deriv.dChidx0 * deriv.dChidx1dx1 - deriv.dChidx1 * deriv.dChidx0dx1) * detDer2I;
155  dtype_t dX1 = -(deriv.dChidx1 * deriv.dChidx0dx0 - deriv.dChidx0 * deriv.dChidx0dx1) * detDer2I;
156  if (!trc0.propagateParamTo(trc0.getX() + dX0, mBz) || !trc1.propagateParamTo(trc1.getX() + dX1, mBz)) {
157  return false;
158  }
159  Triplet pca;
160  calcPCA(trc0, trCFVT0, trc1, trCFVT1, pca);
161  // if there are 2 XY crossings, fit with both of them used as a starting point may converge to
162  // the same point. To stop asap this redundant step we abandon fit if it appears to be closer
163  // to alternative crossing
164  if (mCrossIDAlt >= 0 && closerToAlternative(pca.x, pca.y)) {
165  return false;
166  }
167 
168  chi2 = calcChi2(pca, trc0, trc0Aux, trcEI0, trc1, trc1Aux, trcEI1);
169 
170  if ((TMath::Abs(dX0) < mMinParamChange && TMath::Abs(dX1) < mMinParamChange) ||
171  (iter && chi2 / chi2Prev > mMinRelChi2Change)) {
172  break; // converged
173  }
174  chi2Prev = chi2;
175  } while (++iter < mMaxIter);
176  //
177  if (chi2 < mMaxChi2) {
178  auto& pca = mPCA[mNCandidates];
179  calcPCA(trc0, trCFVT0, trc1, trCFVT1, pca);
180  mChi2[mNCandidates] = calcChi2(pca, trc0, trc0Aux, trcEI0, trc1, trc1Aux, trcEI1);
181  return true;
182  }
183  return false;
184 }

187 int DCAFitter::processAsIs(const Track& trc0, const Track& trc1)
188 {
189  // find dca of 2 tracks w/o preliminary propagation to XY crossing points
190  clear();
191  TrcAuxPar trc0Aux(trc0, mBz), trc1Aux(trc1, mBz);
192  mCandTr0[mNCandidates] = trc0;
193  mCandTr1[mNCandidates] = trc1;
194  if (mUseAbsDCA ? processCandidateDCA(trc0Aux, trc1Aux) : processCandidateChi2(trc0Aux, trc1Aux)) {
195  mNCandidates++; // candidate validated
196  }
197  return mNCandidates;
198 }

200 //___________________________________________________________________
201 bool DCAFitter::processCandidateDCA(const TrcAuxPar& trc0Aux, const TrcAuxPar& trc1Aux)
202 {
203  // find DCA of 2 tracks already propagated to its approximate vicinity W/O APPLYING ANY ERRORS
204  // i.e. the absolute distance is minimized
205  Track &trc0 = mCandTr0[mNCandidates], &trc1 = mCandTr1[mNCandidates];
206 
207  TrackDeriv2 tDer0, tDer1; // their derivatives over track param X
208  Derivatives deriv; // chi2 1st and 2nd derivatives over tracks params
209 
210  ftype_t chi2 = 0, chi2Prev = 0;
211  int iter = 0;
212  do {
213  tDer0.set(trc0, mBz); // tracks derivatives
214  tDer1.set(trc1, mBz); // over their X params
215  DCADeriv(trc0, tDer0, trc0Aux, trc1, tDer1, trc1Aux, deriv);
216 
217  // do Newton-Rapson iteration with corrections = - dchi2/d{x0,x1} * [ d^2chi2/d{x0,x1}^2 ]^-1
218  dtype_t detDer2 = deriv.dChidx0dx0 * deriv.dChidx1dx1 - deriv.dChidx0dx1 * deriv.dChidx0dx1;
219  dtype_t detDer2I = 1. / detDer2;
220  dtype_t dX0 = -(deriv.dChidx0 * deriv.dChidx1dx1 - deriv.dChidx1 * deriv.dChidx0dx1) * detDer2I;
221  dtype_t dX1 = -(deriv.dChidx1 * deriv.dChidx0dx0 - deriv.dChidx0 * deriv.dChidx0dx1) * detDer2I;
222  if (!trc0.propagateParamTo(trc0.getX() + dX0, mBz) || !trc1.propagateParamTo(trc1.getX() + dX1, mBz)) {
223  return false;
224  }
225 
226  if (mCrossIDAlt >= 0) {
227  Triplet pca;
228  calcPCA(trc0, trc0Aux, trc1, trc1Aux, pca);
229  // if there are 2 XY crossings, fit with both of them used as a starting point may converge to
230  // the same point. To stop asap this redundant step we abandon fit if it appears to be closer
231  // to alternative crossing
232  if (closerToAlternative(pca.x, pca.y)) {
233  return false;
234  }
235  }
236  chi2 = calcDCA(trc0, trc0Aux, trc1, trc1Aux);
237  if ((TMath::Abs(dX0) < mMinParamChange && TMath::Abs(dX1) < mMinParamChange) ||
238  (iter && chi2 / chi2Prev > mMinRelChi2Change)) {
239  break; // converged
240  }
241  chi2Prev = chi2;
242  } while (++iter < mMaxIter);
243  //
244  if (chi2 < mMaxChi2) {
245  auto& pca = mPCA[mNCandidates];
246  calcPCA(trc0, trc0Aux, trc1, trc1Aux, pca);
247  mChi2[mNCandidates] = calcDCA(trc0, trc0Aux, trc1, trc1Aux);
248  return true;
249  }
250  return false;
251 }

253 //___________________________________________________________________
254 void DCAFitter::calcPCA(const Track& trc0, const TrackCoefVtx& trCFVT0,
255  const Track& trc1, const TrackCoefVtx& trCFVT1,
256  Triplet& pca) const
257 {
258  // calculate the PCA (Vx,Vy,Vz) to 2 points in lab frame represented via local points coordinates
259  pca.x = trCFVT0.mXX * trc0.getX() + trCFVT0.mXY * trc0.getY() + trCFVT0.mXZ * trc0.getZ() + trCFVT1.mXX * trc1.getX() + trCFVT1.mXY * trc1.getY() + trCFVT1.mXZ * trc1.getZ();
260  pca.y = trCFVT0.mYX * trc0.getX() + trCFVT0.mYY * trc0.getY() + trCFVT0.mYZ * trc0.getZ() + trCFVT1.mYX * trc1.getX() + trCFVT1.mYY * trc1.getY() + trCFVT1.mYZ * trc1.getZ();
261  pca.z = trCFVT0.mZX * trc0.getX() + trCFVT0.mZY * trc0.getY() + trCFVT0.mZZ * trc0.getZ() + trCFVT1.mZX * trc1.getX() + trCFVT1.mZY * trc1.getY() + trCFVT1.mZZ * trc1.getZ();
262 }

264 //___________________________________________________________________
265 void DCAFitter::calcPCA(const Track& trc0, const TrcAuxPar& trc0Aux,
266  const Track& trc1, const TrcAuxPar& trc1Aux,
267  Triplet& pca) const
268 {
269  // calculate the PCA (Vx,Vy,Vz) to 2 points in lab frame represented via local points coordinates
270  // w/o accounting for the errors of the points
271  ftype_t xg, yg;
272  trc0Aux.loc2glo(trc0.getX(), trc0.getY(), xg, yg);
273  trc1Aux.loc2glo(trc1.getX(), trc1.getY(), pca.x, pca.y);
274  pca.x += xg;
275  pca.y += yg;
276  pca.z = 0.5 * (trc0.getZ() + trc1.getZ());
277  pca.x *= 0.5;
278  pca.y *= 0.5;
279 }

281 //___________________________________________________________________
282 void DCAFitter::chi2Deriv(const Track& trc0, const TrackDeriv2& tDer0, const TrcAuxPar& trc0Aux, const TrackCovI& trcEI0, const TrackCoefVtx& trCFVT0,
283  const Track& trc1, const TrackDeriv2& tDer1, const TrcAuxPar& trc1Aux, const TrackCovI& trcEI1, const TrackCoefVtx& trCFVT1,
284  Derivatives& deriv) const
285 {
286  // calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters X
287  Triplet vtx;
288  calcPCA(trc0, trCFVT0, trc1, trCFVT1, vtx); // calculate PCA for current track-points positions
289  // calculate residuals
290  auto res0 = calcResid(trc0, trc0Aux, vtx);
291  auto res1 = calcResid(trc1, trc1Aux, vtx);
292 
293  // res0.x,dy0,dz0 = x0 - Xl0, y0 - Yl0, ... , whith x0,y0,z0: track coords in its frame, and Xl0, Yl0, Zl0 : vertex rotated to same frame
294 
295  // aux params to minimize multiplications
296  ftype_t dx0s = res0.x * trcEI0.sxx;
297  ftype_t dy0sz0t = res0.y * trcEI0.syy + res0.z * trcEI0.syz;
298  ftype_t dy0tz0s = res0.y * trcEI0.syz + res0.z * trcEI0.szz;
299 
300  ftype_t dx1s = res1.x * trcEI1.sxx;
301  ftype_t dy1sz1t = res1.y * trcEI1.syy + res1.z * trcEI1.syz;
302  ftype_t dy1tz1s = res1.y * trcEI1.syz + res1.z * trcEI1.szz;
303 
304  //-------------------------------
305  // At the moment keep this in double. TODO: check if ftype_t is ok
306  dtype_t xx0DtXYXZx0 = trCFVT0.mXX + trCFVT0.mXY * tDer0.dydx + trCFVT0.mXZ * tDer0.dzdx;
307  dtype_t xx1DtXYXZx1 = trCFVT1.mXX + trCFVT1.mXY * tDer1.dydx + trCFVT1.mXZ * tDer1.dzdx;
308  dtype_t yz0DtYYYZx0 = trCFVT0.mYX + trCFVT0.mYY * tDer0.dydx + trCFVT0.mYZ * tDer0.dzdx;
309  dtype_t yz1DtYYYZx1 = trCFVT1.mYX + trCFVT1.mYY * tDer1.dydx + trCFVT1.mYZ * tDer1.dzdx;
310 
311  dtype_t DtXYXZx02 = trCFVT0.mXY * tDer0.d2ydx2 + trCFVT0.mXZ * tDer0.d2zdx2;
312  dtype_t DtYYYZx02 = trCFVT0.mYY * tDer0.d2ydx2 + trCFVT0.mYZ * tDer0.d2zdx2;
313  dtype_t DtXYXZx12 = trCFVT1.mXY * tDer1.d2ydx2 + trCFVT1.mXZ * tDer1.d2zdx2;
314  dtype_t DtYYYZx12 = trCFVT1.mYY * tDer1.d2ydx2 + trCFVT1.mYZ * tDer1.d2zdx2;
315 
316  dtype_t FDdx0Dx0 = 1. - (trc0Aux.c * xx0DtXYXZx0 + trc0Aux.s * yz0DtYYYZx0);
317  dtype_t FDdy0Dx0 = trc0Aux.s * xx0DtXYXZx0 - trc0Aux.c * yz0DtYYYZx0 + tDer0.dydx;
318  dtype_t FDdz0Dx0 = -trCFVT0.mZX - trCFVT0.mZY * tDer0.dydx + tDer0.dzdx * (1. - trCFVT0.mZZ);
319 
320  dtype_t FDdx1Dx1 = 1. - (trc1Aux.c * xx1DtXYXZx1 + trc1Aux.s * yz1DtYYYZx1);
321  dtype_t FDdy1Dx1 = trc1Aux.s * xx1DtXYXZx1 - trc1Aux.c * yz1DtYYYZx1 + tDer1.dydx;
322  dtype_t FDdz1Dx1 = -trCFVT1.mZX - trCFVT1.mZY * tDer1.dydx + tDer1.dzdx * (1. - trCFVT1.mZZ);
323 
324  dtype_t FDdx0Dx1 = -(trc0Aux.c * xx1DtXYXZx1 + trc0Aux.s * yz1DtYYYZx1);
325  dtype_t FDdy0Dx1 = trc0Aux.s * xx1DtXYXZx1 - trc0Aux.c * yz1DtYYYZx1;
326  dtype_t FDdz0Dx1 = FDdz1Dx1 - tDer1.dzdx; // -trCFVT1.mZX - trCFVT1.mZY*tDer1.dydx - trCFVT1.mZZ*tDer1.dzdx;
327 
328  dtype_t FDdx1Dx0 = -(trc1Aux.c * xx0DtXYXZx0 + trc1Aux.s * yz0DtYYYZx0);
329  dtype_t FDdy1Dx0 = trc1Aux.s * xx0DtXYXZx0 - trc1Aux.c * yz0DtYYYZx0;
330  dtype_t FDdz1Dx0 = FDdz0Dx0 - tDer0.dzdx; // -trCFVT0.mZX - trCFVT0.mZY*tDer0.dydx - trCFVT0.mZZ*tDer0.dzdx;
331 
332  dtype_t FDdx0Dx0x0 = -(trc0Aux.c * DtXYXZx02 + trc0Aux.s * DtYYYZx02);
333  dtype_t FDdy0Dx0x0 = tDer0.d2ydx2 + trc0Aux.s * DtXYXZx02 - trc0Aux.c * DtYYYZx02;
334  dtype_t FDdz0Dx0x0 = -trCFVT0.mZY * tDer0.d2ydx2 + tDer0.d2zdx2 * (1. - trCFVT0.mZZ);
335 
336  dtype_t FDdx1Dx1x1 = -(trc1Aux.c * DtXYXZx12 + trc1Aux.s * DtYYYZx12);
337  dtype_t FDdy1Dx1x1 = tDer1.d2ydx2 + trc1Aux.s * DtXYXZx12 - trc1Aux.c * DtYYYZx12;
338  dtype_t FDdz1Dx1x1 = -trCFVT1.mZY * tDer1.d2ydx2 + tDer1.d2zdx2 * (1. - trCFVT1.mZZ);
339 
340  dtype_t FDdx0Dx1x1 = -(trc0Aux.c * DtXYXZx12 + trc0Aux.s * DtYYYZx12);
341  dtype_t FDdx1Dx0x0 = -(trc1Aux.c * DtXYXZx02 + trc1Aux.s * DtYYYZx02);
342 
343  dtype_t FDdy1Dx0x0 = trc1Aux.s * DtXYXZx02 - trc1Aux.c * DtYYYZx02;
344  dtype_t FDdy0Dx1x1 = trc0Aux.s * DtXYXZx12 - trc0Aux.c * DtYYYZx12;
345 
346  dtype_t FDdz0Dx1x1 = FDdz1Dx1x1 - tDer1.d2zdx2; // -(trCFVT1.mZY*tDer1.d2ydx2 + trCFVT1.mZZ*tDer1.d2zdx2);
347  dtype_t FDdz1Dx0x0 = FDdz0Dx0x0 - tDer0.d2zdx2; // -(trCFVT0.mZY*tDer0.d2ydx2 + trCFVT0.mZZ*tDer0.d2zdx2);
348 
349  dtype_t FD00YYYZ = FDdy0Dx0 * trcEI0.syy + FDdz0Dx0 * trcEI0.syz;
350  dtype_t FD11YYYZ = FDdy1Dx1 * trcEI1.syy + FDdz1Dx1 * trcEI1.syz;
351  dtype_t FD10YYYZ = FDdy1Dx0 * trcEI1.syy + FDdz1Dx0 * trcEI1.syz;
352  dtype_t FD01YYYZ = FDdy0Dx1 * trcEI0.syy + FDdz0Dx1 * trcEI0.syz;
353  dtype_t FD00YZZZ = FDdy0Dx0 * trcEI0.syz + FDdz0Dx0 * trcEI0.szz;

390 void DCAFitter::DCADeriv(const Track& trc0, const TrackDeriv2& tDer0, const TrcAuxPar& trc0Aux,
391  const Track& trc1, const TrackDeriv2& tDer1, const TrcAuxPar& trc1Aux,
392  Derivatives& deriv) const
393 {
394  // DCA derivative calculation with Chi2 defined as the absolule distance (no errors applied)
395  ftype_t cosDA = trc0Aux.c * trc1Aux.c + trc0Aux.s * trc1Aux.s; // cos(A0-A1)
396  ftype_t sinDA = trc0Aux.s * trc1Aux.c - trc0Aux.c * trc1Aux.s; // sin(A0-A1)
397  ftype_t dx01 = trc0.getX() - (trc1.getX() * cosDA + trc1.getY() * sinDA);
398  ftype_t dx10 = trc1.getX() - (trc0.getX() * cosDA - trc0.getY() * sinDA);
399  ftype_t dy01 = trc0.getY() - (trc1.getY() * cosDA - trc1.getX() * sinDA);
400  ftype_t dy10 = trc1.getY() - (trc0.getY() * cosDA + trc0.getX() * sinDA);
401  ftype_t dz = trc0.getZ() - trc1.getZ();
402 
403  // 1st derivatives over track params x
404  deriv.dChidx0 = 0.5 * (tDer0.dydx * dy01 + dx01 + dz * tDer0.dzdx);
405  deriv.dChidx1 = 0.5 * (tDer1.dydx * dy10 + dx10 - dz * tDer1.dzdx);
406 
407  // 2nd derivative over track params x
408  deriv.dChidx0dx0 = 0.5 * (1. + tDer0.dydx * tDer0.dydx + tDer0.dzdx * tDer0.dzdx + dz * tDer0.d2zdx2 + tDer0.d2ydx2 * dy01);
409  deriv.dChidx1dx1 = 0.5 * (1. + tDer1.dydx * tDer1.dydx + tDer1.dzdx * tDer1.dzdx - dz * tDer1.d2zdx2 + tDer1.d2ydx2 * dy10);
410  deriv.dChidx0dx1 = 0.5 * (sinDA * (tDer0.dydx - tDer1.dydx) - cosDA * (1. + tDer0.dydx * tDer1.dydx) - tDer0.dzdx * tDer1.dzdx);
411 }

414 void DCAFitter::calcPCACoefs(const TrcAuxPar& trc0Aux, const TrackCovI& trcEI0,
415  const TrcAuxPar& trc1Aux, const TrackCovI& trcEI1,
416  TrackCoefVtx& trCFVT0, TrackCoefVtx& trCFVT1) const
417 {
418  // calculate coefficients of the PCA (Vx,Vy,Vz) to 2 points in lab frame represented via local points coordinates as
419  // Vx = mXX0*x0+mXY0*y0+mXZ0*z0 + mXX1*x1+mXY1*y1+mXZ1*z1
420  // Vy = mYX0*x0+mYY0*y0+mYZ0*z0 + mYX1*x1+mYY1*y1+mYZ1*z1
421  // Vz = mZX0*x0+mZY0*y0+mZZ0*z0 + mZX1*x1+mZY1*y1+mZZ1*z1
422  // where {x0,y0,z0} and {x1,y1,z1} are track positions in their local frames
423  //
424  // we find the PCA of 2 tracks poins weighted by their errors, i.e. minimizing
425  // chi2 = ....
426  // these are the coefficients of dChi2/d{Vx,Vy,vZ} = 0
427 
428  // At the moment keep this in dtype_t. TODO: check if ftype_t is ok
429  dtype_t axx = trc0Aux.cc * trcEI0.sxx + trc1Aux.cc * trcEI1.sxx + trc0Aux.ss * trcEI0.syy + trc1Aux.ss * trcEI1.syy;
430  dtype_t axy = trc0Aux.cs * (trcEI0.sxx - trcEI0.syy) + trc1Aux.cs * (trcEI1.sxx - trcEI1.syy);
431  dtype_t axz = -(trc0Aux.s * trcEI0.syz + trc1Aux.s * trcEI1.syz);
432  dtype_t ayy = trc0Aux.ss * trcEI0.sxx + trc1Aux.ss * trcEI1.sxx + trc0Aux.cc * trcEI0.syy + trc1Aux.cc * trcEI1.syy; // = (trcEI0.sxx + trcEI1.sxx + trcEI0.syy + trcEI1.syy) - axx
433  dtype_t ayz = trc0Aux.c * trcEI0.syz + trc1Aux.c * trcEI1.syz;
434  dtype_t azz = trcEI0.szz + trcEI1.szz;
435  //
436  // define some aux variables
437  dtype_t axxyy = axx * ayy, axxzz = axx * azz, axxyz = axx * ayz,
438  axyxy = axy * axy, axyxz = axy * axz, axyyz = axy * ayz, axyzz = axy * azz,
439  axzxz = axz * axz, axzyy = axz * ayy, axzyz = axz * ayz,
440  ayzyz = ayz * ayz, ayyzz = ayy * azz;
441  dtype_t dAxxyyAxyxy = axxyy - axyxy, dAxyyzAxzyy = axyyz - axzyy, dAxyxzAxxyz = axyxz - axxyz;
442  dtype_t dAxzyzAxyzz = axzyz - axyzz, dAyyzzAyzyz = ayyzz - ayzyz, dAxxzzAxzxz = axxzz - axzxz;
443  dtype_t det = -dAxyyzAxzyy * axz + dAxyxzAxxyz * ayz + dAxxyyAxyxy * azz, detI = 1. / det;
444 
445  dtype_t dfxPCS0 = dAyyzzAyzyz * trc0Aux.c + dAxzyzAxyzz * trc0Aux.s, dfxQCS0 = dAxzyzAxyzz * trc0Aux.c - dAyyzzAyzyz * trc0Aux.s;
446  dtype_t dfyPCS0 = dAxzyzAxyzz * trc0Aux.c + dAxxzzAxzxz * trc0Aux.s, dfyQCS0 = dAxxzzAxzxz * trc0Aux.c - dAxzyzAxyzz * trc0Aux.s;
447  dtype_t dfzPCS0 = dAxyyzAxzyy * trc0Aux.c + dAxyxzAxxyz * trc0Aux.s, dfzQCS0 = dAxyxzAxxyz * trc0Aux.c - dAxyyzAxzyy * trc0Aux.s;
448 
449  dtype_t dfxPCS1 = dAyyzzAyzyz * trc1Aux.c + dAxzyzAxyzz * trc1Aux.s, dfxQCS1 = dAxzyzAxyzz * trc1Aux.c - dAyyzzAyzyz * trc1Aux.s;
450  dtype_t dfyPCS1 = dAxzyzAxyzz * trc1Aux.c + dAxxzzAxzxz * trc1Aux.s, dfyQCS1 = dAxxzzAxzxz * trc1Aux.c - dAxzyzAxyzz * trc1Aux.s;
451  dtype_t dfzPCS1 = dAxyyzAxzyy * trc1Aux.c + dAxyxzAxxyz * trc1Aux.s, dfzQCS1 = dAxyxzAxxyz * trc1Aux.c - dAxyyzAxzyy * trc1Aux.s;
452  //
453  trCFVT0.mXX = detI * (dfxPCS0 * trcEI0.sxx);
454  trCFVT0.mXY = detI * (dfxQCS0 * trcEI0.syy + dAxyyzAxzyy * trcEI0.syz);
455  trCFVT0.mXZ = detI * (dfxQCS0 * trcEI0.syz + dAxyyzAxzyy * trcEI0.szz);
456 
457  trCFVT0.mYX = detI * (dfyPCS0 * trcEI0.sxx);
458  trCFVT0.mYY = detI * (dfyQCS0 * trcEI0.syy + dAxyxzAxxyz * trcEI0.syz);
459  trCFVT0.mYZ = detI * (dfyQCS0 * trcEI0.syz + dAxyxzAxxyz * trcEI0.szz);
460 
461  trCFVT0.mZX = detI * (dfzPCS0 * trcEI0.sxx);
462  trCFVT0.mZY = detI * (dfzQCS0 * trcEI0.syy + dAxxyyAxyxy * trcEI0.syz);
463  trCFVT0.mZZ = detI * (dfzQCS0 * trcEI0.syz + dAxxyyAxyxy * trcEI0.szz);
464 
465  trCFVT1.mXX = detI * (dfxPCS1 * trcEI1.sxx);
466  trCFVT1.mXY = detI * (dfxQCS1 * trcEI1.syy + dAxyyzAxzyy * trcEI1.syz);
467  trCFVT1.mXZ = detI * (dfxQCS1 * trcEI1.syz + dAxyyzAxzyy * trcEI1.szz);
468 
469  trCFVT1.mYX = detI * (dfyPCS1 * trcEI1.sxx);
470  trCFVT1.mYY = detI * (dfyQCS1 * trcEI1.syy + dAxyxzAxxyz * trcEI1.syz);
471  trCFVT1.mYZ = detI * (dfyQCS1 * trcEI1.syz + dAxyxzAxxyz * trcEI1.szz);
472 
473  trCFVT1.mZX = detI * (dfzPCS1 * trcEI1.sxx);
474  trCFVT1.mZY = detI * (dfzQCS1 * trcEI1.syy + dAxxyyAxyxy * trcEI1.syz);
475  trCFVT1.mZZ = detI * (dfzQCS1 * trcEI1.syz + dAxxyyAxyxy * trcEI1.szz);
476 }

479 ftype_t DCAFitter::calcChi2(const Triplet& pca,
480  const Track& trc0, const TrcAuxPar& trc0Aux, const TrackCovI& trcEI0,
481  const Track& trc1, const TrcAuxPar& trc1Aux, const TrackCovI& trcEI1) const
482 {
483  ftype_t chi2 = 0;
484  ftype_t xl, yl, dx, dy, dz;
485  trc0Aux.glo2loc(pca.x, pca.y, xl, yl);
486  dx = trc0.getX() - xl;
487  dy = trc0.getY() - yl;
488  dz = trc0.getZ() - pca.z;
489  chi2 += dx * dx * trcEI0.sxx + dy * dy * trcEI0.syy + dz * dz * trcEI0.szz + 2. * dy * dz * trcEI0.syz;
490  trc1Aux.glo2loc(pca.x, pca.y, xl, yl);
491  dx = trc1.getX() - xl;
492  dy = trc1.getY() - yl;
493  dz = trc1.getZ() - pca.z;
494  chi2 += dx * dx * trcEI1.sxx + dy * dy * trcEI1.syy + dz * dz * trcEI1.szz + 2. * dy * dz * trcEI1.syz;
495  return chi2;

499 ftype_t DCAFitter::calcDCA(const Triplet& pca,
500  const Track& trc0, const TrcAuxPar& trc0Aux, const Track& trc1, const TrcAuxPar& trc1Aux) const
501 {
502  // calculate distance (non-weighted) of closest approach of 2 points in their local frame
503  // (long way, see alternative getDCA w/o explicit vertex calculation)
504  ftype_t chi2 = 0;
505  ftype_t xl, yl, dx, dy, dz;
506  trc0Aux.glo2loc(pca.x, pca.y, xl, yl);
507  dx = trc0.getX() - xl;
508  dy = trc0.getY() - yl;
509  dz = trc0.getZ() - pca.z;
510  chi2 += dx * dx + dy * dy + dz * dz;
511  trc1Aux.glo2loc(pca.x, pca.y, xl, yl);
512  dx = trc1.getX() - xl;
513  dy = trc1.getY() - yl;
514  dz = trc1.getZ() - pca.z;
515  chi2 += dz * dz + dy * dy + dz * dz;
516  return chi2;
517 }

520 ftype_t DCAFitter::calcDCA(const Track& trc0, const TrcAuxPar& trc0Aux, const Track& trc1, const TrcAuxPar& trc1Aux) const
521 {
522  // calculate distance (non-weighted) of closest approach of 2 points in their local frame
523  ftype_t chi2 = 0;
524  dtype_t cosDA = trc0Aux.c * trc1Aux.c + trc0Aux.s * trc1Aux.s; // cos(A0-A1)
525  dtype_t sinDA = trc0Aux.s * trc1Aux.c - trc0Aux.c * trc1Aux.s; // sin(A0-A1)
526  dtype_t dx = trc0.getX() - trc1.getX(), dy = trc0.getY() - trc1.getY(), dz = trc0.getZ() - trc1.getZ();
527  chi2 = 0.5 * (dx * dx + dy * dy + dz * dz) + (1. - cosDA) * (trc0.getX() * trc1.getX() + trc0.getY() * trc1.getY()) +
528  sinDA * (trc0.getY() * trc1.getX() - trc1.getY() * trc0.getX());
529 
530  return chi2;
