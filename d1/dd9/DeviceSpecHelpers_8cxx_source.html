197 void DeviceSpecHelpers::processOutEdgeActions(std::vector<DeviceSpec>& devices,
198  std::vector<DeviceId>& deviceIndex,
199  std::vector<DeviceConnectionId>& connections,
200  ResourceManager& resourceManager,
201  const std::vector<size_t>& outEdgeIndex,
202  const std::vector<DeviceConnectionEdge>& logicalEdges,
203  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
204  const std::vector<OutputSpec>& outputsMatchers,
205  const std::vector<ChannelConfigurationPolicy>& channelPolicies,
206  ComputingOffer const& defaultOffer)
207 {
208  // The topology cannot be empty or not connected. If that is the case, than
209  // something before this went wrong.
210  // FIXME: is that really true???
211  assert(!workflow.empty());
212 
213  // Edges are navigated in order for each device, so the device associaited to
214  // an edge is always the last one created.
215  auto deviceForEdge = [&actions, &workflow, &devices, &logicalEdges, &resourceManager, &defaultOffer](size_t ei, ComputingOffer& acceptedOffer) {
216  auto& edge = logicalEdges[ei];
217  auto& action = actions[ei];
218 
219  if (action.requiresNewDevice == false) {
220  assert(devices.empty() == false);
221  return devices.size() - 1;
222  }
223  if (acceptedOffer.hostname != "") {
224  resourceManager.notifyAcceptedOffer(acceptedOffer);
225  }
226 
227  auto processor = workflow[edge.producer];
228 
229  acceptedOffer.cpu = defaultOffer.cpu;
230  acceptedOffer.memory = defaultOffer.memory;
231  for (auto offer : resourceManager.getAvailableOffers()) {
232  if (offer.cpu < acceptedOffer.cpu) {
233  continue;
234  }
235  if (offer.memory < acceptedOffer.memory) {
236  continue;
237  }
238  acceptedOffer.hostname = offer.hostname;
239  acceptedOffer.startPort = offer.startPort;
240  acceptedOffer.rangeSize = 0;
241  break;
242  }
243 
244  DeviceSpec device;
245  device.name = processor.name;
246  device.id = processor.name;
247  if (processor.maxInputTimeslices != 1) {
248  device.id = processor.name + "_t" + std::to_string(edge.producerTimeIndex);
249  }
250  device.algorithm = processor.algorithm;
251  device.options = processor.options;
252  device.rank = processor.rank;
253  device.nSlots = processor.nSlots;
254  device.inputTimesliceId = edge.producerTimeIndex;
255  device.maxInputTimeslices = processor.maxInputTimeslices;
256  device.resource = {acceptedOffer};
257  devices.push_back(device);
258  return devices.size() - 1;
259  };
260 
261  auto channelFromDeviceEdgeAndPort = [&connections, &workflow, &channelPolicies](const DeviceSpec& device,
262  ComputingResource& deviceResource,
263  ComputingOffer& acceptedOffer,
264  const DeviceConnectionEdge& edge) {
265  OutputChannelSpec channel;
266  auto& consumer = workflow[edge.consumer];
267  std::string consumerDeviceId = consumer.name;
268  if (consumer.maxInputTimeslices != 1) {
269  consumerDeviceId += "_t" + std::to_string(edge.timeIndex);
270  }
271  channel.name = "from_" + device.id + "_to_" + consumerDeviceId;
272  channel.port = acceptedOffer.startPort + acceptedOffer.rangeSize;
273  channel.hostname = acceptedOffer.hostname;
274  deviceResource.usedPorts += 1;
275  acceptedOffer.rangeSize += 1;
276 
277  for (auto& policy : channelPolicies) {
278  if (policy.match(device.id, consumerDeviceId)) {
279  policy.modifyOutput(channel);
280  break;
281  }
282  }
283  DeviceConnectionId id{edge.producer, edge.consumer, edge.timeIndex, edge.producerTimeIndex, channel.port};
284  connections.push_back(id);
285  return std::move(channel);
286  };
287 
288  auto isDifferentDestinationDeviceReferredBy = [&actions](size_t ei) { return actions[ei].requiresNewChannel; };
289 
290  // This creates a new channel for a given edge, if needed. Notice that we
291  // navigate edges in a per device fashion (creating those if they are not
292  // alredy there) and create a new channel only if it connects two new
293  // devices. Whether or not this is the case was previously computed
294  // in the action.requiresNewChannel field.
295  auto createChannelForDeviceEdge = [&devices, &logicalEdges, &channelFromDeviceEdgeAndPort,
296  &outputsMatchers, &deviceIndex,
297  &workflow](size_t di, size_t ei, ComputingOffer& offer) {
298  auto& device = devices[di];
299  auto& edge = logicalEdges[ei];
300 
301  deviceIndex.emplace_back(DeviceId{edge.producer, edge.producerTimeIndex, di});
302 
303  OutputChannelSpec channel = channelFromDeviceEdgeAndPort(device, device.resource, offer, edge);
304 
305  device.outputChannels.push_back(channel);
306  return device.outputChannels.size() - 1;
307  };
308 
309  // Notice how we need to behave in two different ways depending
310  // whether this is a real OutputRoute or if it's a forward from
311  // a previous consumer device.
312  // FIXME: where do I find the InputSpec for the forward?
313  auto appendOutputRouteToSourceDeviceChannel = [&outputsMatchers, &workflow, &devices, &logicalEdges](
314  size_t ei, size_t di, size_t ci) {
315  assert(ei < logicalEdges.size());
316  assert(di < devices.size());
317  assert(ci < devices[di].outputChannels.size());
318  auto& edge = logicalEdges[ei];
319  auto& device = devices[di];
320  assert(edge.consumer < workflow.size());
321  auto& consumer = workflow[edge.consumer];
322  auto& channel = devices[di].outputChannels[ci];
323  assert(edge.outputGlobalIndex < outputsMatchers.size());
324 
325  if (edge.isForward == false) {
326  OutputRoute route{
327  edge.timeIndex,
328  consumer.maxInputTimeslices,
329  outputsMatchers[edge.outputGlobalIndex],
330  channel.name};
331  device.outputs.emplace_back(route);
332  } else {
333  ForwardRoute route{
334  edge.timeIndex,
335  consumer.maxInputTimeslices,
336  workflow[edge.consumer].inputs[edge.consumerInputIndex],
337  channel.name};
338  device.forwards.emplace_back(route);
339  }
340  };
341 
342  auto sortDeviceIndex = [&deviceIndex]() { std::sort(deviceIndex.begin(), deviceIndex.end()); };
343 
344  auto lastChannelFor = [&devices](size_t di) {
345  assert(di < devices.size());
346  assert(devices[di].outputChannels.empty() == false);
347  return devices[di].outputChannels.size() - 1;
348  };
349 
350  //
351  // OUTER LOOP
352  //
353  // We need to create all the channels going out of a device, and associate
354  // routes to them for this reason
355  // we iterate over all the edges (which are per-datatype exchanged) and
356  // whenever we need to connect to a new device we create the channel. `device`
357  // here refers to the source device. This loop will therefore not create the
358  // devices which acts as sink, which are done in the preocessInEdgeActions
359  // function.
360  ComputingOffer acceptedOffer;
361  for (auto edge : outEdgeIndex) {
362  auto device = deviceForEdge(edge, acceptedOffer);
363  size_t channel = -1;
364  if (isDifferentDestinationDeviceReferredBy(edge)) {
365  channel = createChannelForDeviceEdge(device, edge, acceptedOffer);
366  } else {
367  channel = lastChannelFor(device);
368  }
369  appendOutputRouteToSourceDeviceChannel(edge, device, channel);
370  }
371  resourceManager.notifyAcceptedOffer(acceptedOffer);
372  sortDeviceIndex();
373 }

375 void DeviceSpecHelpers::processInEdgeActions(std::vector<DeviceSpec>& devices,
376  std::vector<DeviceId>& deviceIndex,
377  const std::vector<DeviceConnectionId>& connections,
378  ResourceManager& resourceManager,
379  const std::vector<size_t>& inEdgeIndex,
380  const std::vector<DeviceConnectionEdge>& logicalEdges,
381  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
382  std::vector<LogicalForwardInfo> const& availableForwardsInfo,
383  std::vector<ChannelConfigurationPolicy> const& channelPolicies,
384  ComputingOffer const& defaultOffer)
385 {
386  auto const& constDeviceIndex = deviceIndex;
387 
388  auto findProducerForEdge = [&logicalEdges, &constDeviceIndex](size_t ei) {
389  auto& edge = logicalEdges[ei];
390 
391  DeviceId pid{edge.producer, edge.producerTimeIndex, 0};
392  auto deviceIt = std::lower_bound(constDeviceIndex.cbegin(), constDeviceIndex.cend(), pid);
393  // By construction producer should always be there
394  assert(deviceIt != constDeviceIndex.end());
395  assert(deviceIt->processorIndex == pid.processorIndex && deviceIt->timeslice == pid.timeslice);
396  return deviceIt->deviceIndex;
397  };
398 
399  // Notice that to start with, consumer exists only if they also are
400  // producers, so we need to create one if it does not exist. Given this is
401  // stateful, we keep an eye on what edge was last searched to make sure we
402  // are not screwing up.
403  //
404  // Notice this is not thread safe.
405  decltype(deviceIndex.begin()) lastConsumerSearch;
406  size_t lastConsumerSearchEdge;
407  auto hasConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge, &deviceIndex,
408  &logicalEdges](size_t ei) -> int {
409  auto& edge = logicalEdges[ei];
410  DeviceId cid{edge.consumer, edge.timeIndex, 0};
411  lastConsumerSearchEdge = ei; // This will invalidate the cache
412  lastConsumerSearch = std::lower_bound(deviceIndex.begin(), deviceIndex.end(), cid);
413  return lastConsumerSearch != deviceIndex.end() && cid.processorIndex == lastConsumerSearch->processorIndex &&
414  cid.timeslice == lastConsumerSearch->timeslice;
415  };
416 
417  // The passed argument is there just to check. We do know that the last searched
418  // is the one we want.
419  auto getConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge](size_t ei) {
420  assert(ei == lastConsumerSearchEdge);
421  return lastConsumerSearch->deviceIndex;
422  };
423 
424  auto createNewDeviceForEdge = [&workflow, &logicalEdges, &devices, &deviceIndex, &resourceManager, &defaultOffer](size_t ei, ComputingOffer& acceptedOffer) {
425  auto& edge = logicalEdges[ei];
426 
427  if (acceptedOffer.hostname != "") {
428  resourceManager.notifyAcceptedOffer(acceptedOffer);
429  }
430 
431  auto& processor = workflow[edge.consumer];
432 
433  acceptedOffer.cpu = defaultOffer.cpu;
434  acceptedOffer.memory = defaultOffer.memory;
435  for (auto offer : resourceManager.getAvailableOffers()) {
436  if (offer.cpu < acceptedOffer.cpu) {
437  continue;
438  }
439  if (offer.memory < acceptedOffer.memory) {
440  continue;
441  }
442  acceptedOffer.hostname = offer.hostname;
443  acceptedOffer.startPort = offer.startPort;
444  acceptedOffer.rangeSize = 0;
445  break;
446  }
447 
448  DeviceSpec device;
449  device.name = processor.name;
450  device.id = processor.name;
451  if (processor.maxInputTimeslices != 1) {
452  device.id += "_t" + std::to_string(edge.timeIndex);
453  }
454  device.algorithm = processor.algorithm;
455  device.options = processor.options;
456  device.rank = processor.rank;
457  device.nSlots = processor.nSlots;
458  device.inputTimesliceId = edge.timeIndex;
459  device.maxInputTimeslices = processor.maxInputTimeslices;
460  device.resource = {acceptedOffer};
461 
462  // FIXME: maybe I should use an std::map in the end
463  // but this is really not performance critical
464  auto id = DeviceId{edge.consumer, edge.timeIndex, devices.size()};
465  devices.push_back(device);
466  deviceIndex.push_back(id);
467  std::sort(deviceIndex.begin(), deviceIndex.end());
468  return devices.size() - 1;
469  };
470 
471  // We search for a preexisting outgoing connection associated to this edge.
472  // This is to retrieve the port of the source.
473  // This has to exists, because we already created all the outgoing connections
474  // so it's just a matter of looking it up.
475  auto findMatchingOutgoingPortForEdge = [&logicalEdges, &connections](size_t ei) {
476  auto const& edge = logicalEdges[ei];
477  DeviceConnectionId connectionId{edge.producer, edge.consumer, edge.timeIndex, edge.producerTimeIndex, 0};
478 
479  auto it = std::lower_bound(connections.begin(), connections.end(), connectionId);
480 
481  assert(it != connections.end());
482  assert(it->producer == connectionId.producer);
483  assert(it->consumer == connectionId.consumer);
484  assert(it->timeIndex == connectionId.timeIndex);
485  assert(it->producerTimeIndex == connectionId.producerTimeIndex);
486  return it->port;
487  };
488 
489  auto checkNoDuplicatesFor = [](std::vector<InputChannelSpec> const& channels, const std::string& name) {
490  for (auto const& channel : channels) {
491  if (channel.name == name) {
492  return false;
493  }
494  }
495  return true;
496  };
497  auto appendInputChannelForConsumerDevice = [&devices, &connections, &checkNoDuplicatesFor, &channelPolicies](
498  size_t pi, size_t ci, unsigned short port) {
499  auto const& producerDevice = devices[pi];
500  auto& consumerDevice = devices[ci];
501  InputChannelSpec channel;
502  channel.name = "from_" + producerDevice.id + "_to_" + consumerDevice.id;
503  channel.hostname = producerDevice.resource.hostname;
504  channel.port = port;
505  for (auto& policy : channelPolicies) {
506  if (policy.match(producerDevice.id, consumerDevice.id)) {
507  policy.modifyInput(channel);
508  break;
509  }
510  }
511  assert(checkNoDuplicatesFor(consumerDevice.inputChannels, channel.name));
512  consumerDevice.inputChannels.push_back(channel);
513  return consumerDevice.inputChannels.size() - 1;
514  };
515 
516  // I think this is trivial, since I think it should always be the last one,
517  // in case it's not actually the case, I should probably do an actual lookup
518  // here.
519  auto getChannelForEdge = [&devices](size_t pi, size_t ci) {
520  auto& consumerDevice = devices[ci];
521  return consumerDevice.inputChannels.size() - 1;
522  };
523 
524  // This is always called when adding a new channel, so we can simply refer
525  // to back. Notice also that this is the place where it makes sense to
526  // assign the forwarding, given that the forwarded stuff comes from some
527  // input.
528  auto appendInputRouteToDestDeviceChannel = [&devices, &logicalEdges, &workflow](size_t ei, size_t di, size_t ci) {
529  auto const& edge = logicalEdges[ei];
530  auto const& consumer = workflow[edge.consumer];
531  auto& consumerDevice = devices[di];
532 
533  auto const& inputSpec = consumer.inputs[edge.consumerInputIndex];
534  auto const& sourceChannel = consumerDevice.inputChannels[ci].name;
535 
536  InputRoute route{
537  inputSpec,
538  edge.consumerInputIndex,
539  sourceChannel,
540  edge.producerTimeIndex,
541  std::nullopt};
542 
543  switch (consumer.inputs[edge.consumerInputIndex].lifetime) {
544  case Lifetime::Condition:
545  route.configurator = {
546  ExpirationHandlerHelpers::dataDrivenConfigurator(),
547  ExpirationHandlerHelpers::danglingConditionConfigurator(),
548  ExpirationHandlerHelpers::expiringConditionConfigurator(inputSpec, sourceChannel)};
549  break;
550  case Lifetime::QA:
551  route.configurator = {
552  ExpirationHandlerHelpers::dataDrivenConfigurator(),
553  ExpirationHandlerHelpers::danglingQAConfigurator(),
554  ExpirationHandlerHelpers::expiringQAConfigurator()};
555  break;
556  case Lifetime::Timer:
557  route.configurator = {
558  ExpirationHandlerHelpers::timeDrivenConfigurator(inputSpec),
559  ExpirationHandlerHelpers::danglingTimerConfigurator(inputSpec),
560  ExpirationHandlerHelpers::expiringTimerConfigurator(inputSpec, sourceChannel)};
561  break;
562  case Lifetime::Enumeration:
563  route.configurator = {
564  ExpirationHandlerHelpers::enumDrivenConfigurator(inputSpec, consumerDevice.inputTimesliceId, consumerDevice.maxInputTimeslices),
565  ExpirationHandlerHelpers::danglingEnumerationConfigurator(inputSpec),
566  ExpirationHandlerHelpers::expiringEnumerationConfigurator(inputSpec, sourceChannel)};
567  break;
568  case Lifetime::Transient:
569  route.configurator = {
570  ExpirationHandlerHelpers::dataDrivenConfigurator(),
571  ExpirationHandlerHelpers::danglingTransientConfigurator(),
572  ExpirationHandlerHelpers::expiringTransientConfigurator(inputSpec)};
573  break;
574  default:
575  break;
576  }
577 
578  // In case we have wildcards, we must make sure that some other edge
579  // produced the same route, i.e. has the same matcher. Without this,
580  // otherwise, we would end up with as many input routes as the outputs that
581  // can be matched by the wildcard.
582  for (size_t iri = 0; iri < consumerDevice.inputs.size(); ++iri) {
583  auto& existingRoute = consumerDevice.inputs[iri];
584  if (existingRoute.timeslice != edge.producerTimeIndex) {
585  continue;
586  }
587  if (existingRoute.inputSpecIndex == edge.consumerInputIndex) {
588  return;
589  }
590  }
591 
592  consumerDevice.inputs.push_back(route);
593  };
594 
595  // Outer loop. A new device is needed for each
596  // of the sink data processors.
597  // New InputChannels need to refer to preexisting OutputChannels we create
598  // previously.
599  ComputingOffer acceptedOffer;
600  for (size_t edge : inEdgeIndex) {
601  auto& action = actions[edge];
602 
603  size_t consumerDevice;
604

629 void DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(WorkflowSpec const& workflow,

631  std::vector<CompletionPolicy> const& completionPolicies,
632  std::vector<DispatchPolicy> const& dispatchPolicies,
633  std::vector<DeviceSpec>& devices,
634  ResourceManager& resourceManager,
635  std::string const& uniqueWorkflowId)
636 {

639  std::vector<DeviceConnectionEdge> logicalEdges;
640  std::vector<DeviceConnectionId> connections;
641  std::vector<DeviceId> deviceIndex;
642 
643  // This is a temporary store for inputs and outputs,
644  // including forwarded channels, so that we can construct
645  // them before assigning to a device.
646  std::vector<OutputSpec> outputs;
647 
648  WorkflowHelpers::verifyWorkflow(workflow);
649  WorkflowHelpers::constructGraph(workflow, logicalEdges, outputs, availableForwardsInfo);
650 
651  // We need to instanciate one device per (me, timeIndex) in the
652  // DeviceConnectionEdge. For each device we need one new binding
653  // server per (me, other) -> port Moreover for each (me, other,
654  // outputGlobalIndex) we need to insert either an output or a
655  // forward.
656  //
657  // We then sort by other. For each (other, me) we need to connect to
658  // port (me, other) and add an input.
659 
660  // Fill an index to do the sorting
661  std::vector<size_t> inEdgeIndex;
662  std::vector<size_t> outEdgeIndex;
663  WorkflowHelpers::sortEdges(inEdgeIndex, outEdgeIndex, logicalEdges);
664 
665  std::vector<EdgeAction> outActions = WorkflowHelpers::computeOutEdgeActions(logicalEdges, outEdgeIndex);
666  // Crete the connections on the inverse map for all of them
667  // lookup for port and add as input of the current device.
668  std::vector<EdgeAction> inActions = WorkflowHelpers::computeInEdgeActions(logicalEdges, inEdgeIndex);
669  size_t deviceCount = 0;
670  for (auto& action : outActions) {
671  deviceCount += action.requiresNewDevice ? 1 : 0;
672  }
673  for (auto& action : inActions) {
674  deviceCount += action.requiresNewDevice ? 1 : 0;
675  }
676 
677  ComputingOffer defaultOffer;
678  for (auto& offer : resourceManager.getAvailableOffers()) {
679  defaultOffer.cpu += offer.cpu;
680  defaultOffer.memory += offer.memory;
681  }
682 
684  defaultOffer.cpu /= deviceCount + 1;
685  defaultOffer.memory /= deviceCount + 1;
686 
687  processOutEdgeActions(devices, deviceIndex, connections, resourceManager, outEdgeIndex, logicalEdges,
688  outActions, workflow, outputs, channelPolicies, defaultOffer);
689 
690  // FIXME: is this not the case???
691  std::sort(connections.begin(), connections.end());
692 
693  processInEdgeActions(devices, deviceIndex, connections, resourceManager, inEdgeIndex, logicalEdges,
694  inActions, workflow, availableForwardsInfo, channelPolicies, defaultOffer);
695  // We apply the completion policies here since this is where we have all the
696  // devices resolved.
697  for (auto& device : devices) {
698  for (auto& policy : completionPolicies) {
699  if (policy.matcher(device) == true) {
700  device.completionPolicy = policy;
701  break;
702  }
703  }
704  for (auto& policy : dispatchPolicies) {
705  if (policy.deviceMatcher(device) == true) {
706  device.dispatchPolicy = policy;
707  break;
708  }
709  }
710  }
711 
712  auto findDeviceIndex = [&deviceIndex](size_t processorIndex, size_t timeslice) {
713  for (auto& deviceEdge : deviceIndex) {
714  if (deviceEdge.processorIndex != processorIndex) {
715  continue;
716  }
717  if (deviceEdge.timeslice != timeslice) {
718  continue;
719  }
720  return deviceEdge.deviceIndex;
721  }
722  throw std::runtime_error("Unable to find device.");
723  };
724 
725  // Optimize the topology when two devices are
726  // running on the same node.
727  for (auto& connection : connections) {
728  auto& device1 = devices[findDeviceIndex(connection.consumer, connection.timeIndex)];
729  auto& device2 = devices[findDeviceIndex(connection.producer, connection.producerTimeIndex)];
730  // No need to do anything if they are not on the same host
731  if (device1.resource.hostname != device2.resource.hostname) {
732  continue;
733  }
734  for (auto& input : device1.inputChannels) {
735  for (auto& output : device2.outputChannels) {
736  if (input.hostname == output.hostname && input.port == output.port) {
737  input.protocol = ChannelProtocol::IPC;
738  output.protocol = ChannelProtocol::IPC;
739  input.hostname += uniqueWorkflowId;
740  output.hostname += uniqueWorkflowId;
741  }
742  }
743  }
744  }

775 void DeviceSpecHelpers::prepareArguments(bool defaultQuiet, bool defaultStopped,
776  std::vector<DataProcessorInfo> const& processorInfos,
777  std::vector<DeviceSpec> const& deviceSpecs,
778  std::vector<DeviceExecution>& deviceExecutions,
779  std::vector<DeviceControl>& deviceControls,
780  std::string const& uniqueWorkflowId)
781 {
782  assert(deviceSpecs.size() == deviceExecutions.size());
783  assert(deviceControls.size() == deviceExecutions.size());
784  for (size_t si = 0; si < deviceSpecs.size(); ++si) {
785  auto& spec = deviceSpecs[si];
786  auto& control = deviceControls[si];
787  auto& execution = deviceExecutions[si];
788 
789  control.quiet = defaultQuiet;
790  control.stopped = defaultStopped;
791 
792  int argc;
793  char** argv;
794  std::vector<ConfigParamSpec> workflowOptions;
798  for (auto& processorInfo : processorInfos) {
799  if (processorInfo.name == spec.id) {
800  argc = processorInfo.cmdLineArgs.size() + 1;
801  argv = (char**)malloc(sizeof(char**) * (argc + 1));
802  argv[0] = strdup(processorInfo.executable.data());
803  for (size_t ai = 0; ai < processorInfo.cmdLineArgs.size(); ++ai) {
804  auto& arg = processorInfo.cmdLineArgs[ai];
805  argv[ai + 1] = strdup(arg.data());
806  }
807  argv[argc] = nullptr;
808  workflowOptions = processorInfo.workflowOptions;
809  break;
810  }
811  }
812 
813  // We duplicate the list of options, filtering only those
814  // which are actually relevant for the given device. The additional
815  // four are to add
816  // * name of the executable
817  // * --framework-id <id> so that we can use the workflow
818  // executable also in other context where we do not fork, e.g. DDS.
819  // * final NULL required by execvp
820  //
821  // We do it here because we are still in the parent and we can therefore
822  // capture them to be displayed in the GUI or to populate the DDS configuration
823  // to dump
824 
825  // Set up options for the device running underneath
826  // FIXME: add some checksum in framework id. We could use this
827  // to avoid redeploys when only a portion of the workflow is changed.
828  // FIXME: this should probably be done in one go with char *, but I am lazy.
829  std::vector<std::string> tmpArgs = {argv[0],
830  "--id", spec.id.c_str(),
831  "--control", "static",
832  "--shm-monitor", "false",
833  "--log-color", "false",
834  "--color", "false"};
835  if (defaultStopped) {
836  tmpArgs.push_back("-s");
837  }
838 
839  // do the filtering of options:
840  // 1) forward options belonging to this specific DeviceSpec
841  // 2) global options defined in getForwardedDeviceOptions and workflow option are
842  // always forwarded and need to be handled separately
843  const char* name = spec.name.c_str();
844  bpo::options_description od; // option descriptions per process
845  bpo::options_description foDesc; // forwarded options for all processes
846  ConfigParamsHelper::dpl2BoostOptions(spec.options, od);
847  od.add_options()(name, bpo::value<std::string>());
848  ConfigParamsHelper::dpl2BoostOptions(workflowOptions, foDesc);
849  foDesc.add(getForwardedDeviceOptions());
850 
851  // has option --session been specified on the command line?
852  bool haveSessionArg = false;
853  using FilterFunctionT = std::function<void(decltype(argc), decltype(argv), decltype(od))>;
854 
855  // the filter function will forward command line arguments based on the option
856  // definition passed to it. All options of the program option definition will be forwarded
857  // if found in the argument list. If not found they will be added with the default value
858  FilterFunctionT filterArgsFct = [&](int largc, char** largv, const bpo::options_description& odesc) {
859  // spec contains options
860  bpo::command_line_parser parser{largc, largv};
861  parser.options(odesc).allow_unregistered();
862  bpo::parsed_options parsed_options = parser.run();
863 
864  bpo::variables_map varmap;
865  bpo::store(parsed_options, varmap);
866 
867  // options can be grouped per processor spec, the group is entered by
868  // the option created from the actual processor spec name
869  // if specified, the following string is interpreted as a sequence
870  // of arguments
871  if (varmap.count(name) > 0) {
872  // strangely enough, the first argument of the group argument string
873  // is marked as defaulted by the parser and is thus ignored. not fully
874  // understood but adding a dummy argument in front cures this
875  auto arguments = "--unused " + varmap[name].as<std::string>();
876  wordexp_t expansions;
877  wordexp(arguments.c_str(), &expansions, 0);
878  bpo::options_description realOdesc = odesc;
879  realOdesc.add_options()("severity", bpo::value<std::string>());
880  realOdesc.add_options()("child-driver", bpo::value<std::string>());
881  realOdesc.add_options()("rate", bpo::value<std::string>());
882  realOdesc.add_options()("shm-segment-size", bpo::value<std::string>());
883  realOdesc.add_options()("shm-monitor", bpo::value<std::string>());
884  realOdesc.add_options()("session", bpo::value<std::string>());
885  filterArgsFct(expansions.we_wordc, expansions.we_wordv, realOdesc);
886  wordfree(&expansions);
887  return;
888  }
889 
890  const char* child_driver_key = "child-driver";
891  if (varmap.count(child_driver_key) > 0) {
892  auto arguments = varmap[child_driver_key].as<std::string>();
893  wordexp_t expansions;
894  wordexp(arguments.c_str(), &expansions, 0);
895  tmpArgs.insert(tmpArgs.begin(), expansions.we_wordv, expansions.we_wordv + expansions.we_wordc);
896  }
897 
898  haveSessionArg = haveSessionArg || varmap.count("session") != 0;
899 
900  for (const auto varit : varmap) {
901  // find the option belonging to key, add if the option has been parsed
902  // and is not defaulted
903  const auto* description = odesc.find_nothrow(varit.first, false);
904  if (description && varmap.count(varit.first)) {
905  // check the semantics of the value
906  auto semantic = description->semantic();
907  const char* optarg = "";
908  if (semantic) {
909  // the value semantics allows different properties like
910  // multitoken, zero_token and composing
911  // currently only the simple case is supported
912  assert(semantic->min_tokens() <= 1);
913  //assert(semantic->max_tokens() && semantic->min_tokens());
914  if (semantic->min_tokens() > 0) {
915  tmpArgs.emplace_back("--");
916  tmpArgs.back() += varit.first;
917  // add the token
918  tmpArgs.emplace_back(varit.second.as<std::string>());
919  optarg = tmpArgs.back().c_str();
920  } else if (semantic->min_tokens() == 0 && varit.second.as<bool>()) {
921  tmpArgs.emplace_back("--");
922  tmpArgs.back() += varit.first;
923  }
924  }
925  control.options.insert(std::make_pair(varit.first, optarg));
926  }
927  }
928  };
929 
930  // filter global options and workflow options independent of option groups
931  filterArgsFct(argc, argv, foDesc);
932  // filter device options, and handle option groups
933  filterArgsFct(argc, argv, od);
934 
935  // Add the channel configuration
936  for (auto& channel : spec.outputChannels) {
937  tmpArgs.emplace_back(std::string("--channel-config"));
938  tmpArgs.emplace_back(outputChannel2String(channel));
939  }
940  for (auto& channel : spec.inputChannels) {
941  tmpArgs.emplace_back(std::string("--channel-config"));
942  tmpArgs.emplace_back(inputChannel2String(channel));
943  }
944 
945  // add the session id if not already specified on command line
946  if (!haveSessionArg) {
947  tmpArgs.emplace_back(std::string("--session"));
948  tmpArgs.emplace_back("dpl_" + uniqueWorkflowId);
949  }
950 
951  // We create the final option list, depending on the channels
952  // which are present in a device.
953  for (auto& arg : tmpArgs) {
954  execution.args.emplace_back(strdup(arg.c_str()));
955  }
956  // execvp wants a NULL terminated list.
957  execution.args.push_back(nullptr);
958 
959  // FIXME: this should probably be reflected in the GUI
960  std::ostringstream str;
961  for (size_t ai = 0; ai < execution.args.size() - 1; ai++) {
962  if (execution.args[ai] == nullptr) {
963  LOG(ERROR) << "Bad argument for " << execution.args[ai - 1];
964  }
965  assert(execution.args[ai]);
966  str << " " << execution.args[ai];
967  }
968  LOG(DEBUG) << "The following options are being forwarded to " << spec.id << ":" << str.str();
969  }
970 }
