40 void DecodedDataAggregator::addData(const LocalBoardRO& loc, size_t firstEntry)
41 {
43  uint8_t uniqueLocId = loc.boardId;
44  uint8_t crateId = crateparams::getCrateId(uniqueLocId);
45  bool isRightSide = crateparams::isRightSide(crateId);
46  uint16_t deBoardId = mCrateMapper.roLocalBoardToDE(crateId, crateparams::getLocId(loc.boardId));
47  auto rpcLineId = mCrateMapper.getRPCLine(deBoardId);
48  auto columnId = mCrateMapper.getColumnId(deBoardId);
49  auto lineId = mCrateMapper.getLineId(deBoardId);
50  for (int ich = 0; ich < 4; ++ich) {
51  if (((loc.firedChambers >> ich) & 0x1) == 0) {
52  continue;
53  }
54  uint8_t deId = detparams::getDEId(isRightSide, ich, rpcLineId);
55  auto& col = FindColumnData(deId, columnId, firstEntry);
56  col.setBendPattern(loc.patternsBP[ich], lineId);
57  col.setNonBendPattern(loc.patternsNBP[ich]);
58  }
59 }

61 void DecodedDataAggregator::process(gsl::span<const LocalBoardRO> localBoards, gsl::span<const ROFRecord> rofRecords)
62 {
64 
65  // First clear the output
66  mData.clear();
67  mROFRecords.clear();
68 
69  // Fill the map with ordered events
70  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
71  mOrderIndexes[rofIt->interactionRecord.toLong()].emplace_back(rofIt - rofRecords.begin());
72  }
73 
74  const ROFRecord* rof = nullptr;
75  for (auto& item : mOrderIndexes) {
76  size_t firstEntry = mData.size();
77  for (auto& idx : item.second) {
78  // In principle all of these ROF records have the same timestamp
79  rof = &rofRecords[idx];
80  for (size_t iloc = rof->firstEntry; iloc < rof->firstEntry + rof->nEntries; ++iloc) {
81  addData(localBoards[iloc], firstEntry);
82  }
83  }
84  mROFRecords.emplace_back(rof->interactionRecord, rof->eventType, firstEntry, mData.size() - firstEntry);
85  }
86 
87  // Clear the inner objects when the computation is done
88  mOrderIndexes.clear();
89 }
