73  float getR() // Cluster Radius (it is the same in sector and global frame)
74  {
75  if (mR == RadiusOutOfRange) {
76  mR = TMath::Sqrt(getX() * getX() + getY() * getY() + getZ() * getZ());
77  }
78  return mR;
79  }
80 
81  float getPhi() // Cluster Phi in sector frame
82  {

84  mPhi = TMath::ATan2(getY(), getX());
85  }

87  }
