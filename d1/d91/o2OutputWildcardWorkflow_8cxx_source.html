25 // This allows defining a workflow where the subSpecification
26 // for the inputs and the outputs are left unspecified.
27 // The source will produce one message at the time with random
28 // subspecification and the receiver will match the any message
29 // with origin "TST" and description "A1" regardless of the subspec.
30 WorkflowSpec defineDataProcessing(ConfigContext const& specs)
31 {
32  return WorkflowSpec{
33  {"A",
34  Inputs{},
35  {OutputSpec{{"a1"}, {"TST", "A1"}}},
36  AlgorithmSpec{adaptStateless(
37  [](DataAllocator& outputs) {
38  auto rn = rand() % 5;
39  std::this_thread::sleep_for(std::chrono::seconds(rn));
40  auto& aData = outputs.make<int>(OutputRef{"a1", static_cast<DataAllocator::SubSpecificationType>(rn)});
41  LOGP(info, "A random subspec:{}", rn);
42  })}},
43  {"B",
44  {InputSpec{"x", {"TST", "A1"}}},
45  {},
46  AlgorithmSpec{adaptStateless(
47  [](InputRecord& inputs) {
48  DataRef ref = inputs.getByPos(0);
49  auto const* header = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
50  LOGP(info, "A random subspec:{}", header->subSpecification);
51  })}},
52  };
53 }
