103 class RawClusterArray
104 {
105  public:
106  RawClusterArray() : mBuffer(nullptr), mBufferSize(0), mNClusters(0), mClusters(NULL), mClustersEnd(NULL) {}
107  RawClusterArray(const char* filename) : mBuffer(nullptr), mBufferSize(0), mNClusters(0), mClusters(NULL), mClustersEnd(NULL)
108  {
109  init(filename);
110  }
111  RawClusterArray(unsigned char* buffer, int size) : mBuffer(nullptr), mBufferSize(0), mNClusters(0), mClusters(NULL), mClustersEnd(NULL)
112  {
113  init(buffer, size);
114  }
115  ~RawClusterArray() {}
116 
117  typedef uint8_t Buffer_t;
118 
119  int init(const char* filename)
120  {
121  std::ifstream input(filename, std::ifstream::binary);
122  clear(0);
123  if (input) {
124  // get length of file:
125  input.seekg(0, input.end);
126  int length = input.tellg();
127  input.seekg(0, input.beg);
128 
129  // allocate memory:
130  mBuffer = new Buffer_t[length];
131  mBufferSize = length;
132 
133  // read data as a block:
134  input.read(reinterpret_cast<char*>(mBuffer), length);
135  if (!input.good()) {
136  clear(-1);
137  std::cerr << "failed to read " << length << " byte(s) from file " << filename << std::endl;
138  }
139 
140  input.close();
141  return init();
142  }
143  std::cerr << "failed to open file " << filename << std::endl;
144  return -1;
145  }
146 
147  int init(unsigned char* buffer, int size)
148  {
149  if (!buffer || size <= 0)
150  return -1;
151  clear(0);
152  mBuffer = new Buffer_t[size];
153  mBufferSize = size;
154  memcpy(mBuffer, buffer, size);
155  return init();
156  }
157 
158  int GetNClusters() const { return mNClusters; }
159 
160  RawCluster* begin() { return mClusters; }
161 
162  RawCluster* end() { return mClustersEnd; }
163 
164  RawCluster& operator[](int i)
165  {
166  if (i + 1 > mNClusters) {
167  // runtime exeption?
168  static RawCluster dummy;
169  return dummy;
170  }
171  return *(mClusters + i);
172  }
173 
174  void print() { print(std::cout); }
175 
176  template <typename StreamT>
177  StreamT& print(StreamT& stream)
178  {
179  std::cout << "RawClusterArray: " << mNClusters << " cluster(s)" << std::endl;
180  for (RawCluster* cluster = mClusters; cluster != mClustersEnd; cluster++) {
181  std::cout << " " << *cluster << std::endl;
182  }
183  return stream;
184  }
185 
186  private:
187  int init()
188  {
189  if (mBuffer == nullptr || mBufferSize == 0)
190  return 0;
191  if (mBufferSize < sizeof(RawClusterData))
192  return -1;
193  RawClusterData& clusterData = *reinterpret_cast<RawClusterData*>(mBuffer);
194 
195  if (clusterData.fCount * sizeof(RawCluster) + sizeof(RawClusterData) > mBufferSize) {
196  std::cerr << "Format error, " << clusterData.fCount << " cluster(s) "
197  << "would require "
198  << (clusterData.fCount * sizeof(RawCluster) + sizeof(RawClusterData))
199  << " byte(s), but only " << mBufferSize << " available" << std::endl;
200  return clear(-1);
201  }
202 
203  mNClusters = clusterData.fCount;
204  mClusters = clusterData.fClusters;
205  mClustersEnd = mClusters + mNClusters;
206 
207  return mNClusters;
208  }
209 
210  int clear(int returnValue)
211  {
212  mNClusters = 0;
213  mClusters = NULL;
214  mClustersEnd = NULL;
215  delete[] mBuffer;
216  mBuffer = nullptr;
217  mBufferSize = 0;
218 
219  return returnValue;
220  }
221 
222  Buffer_t* mBuffer;
223  int mBufferSize;
224  int mNClusters;
225  RawCluster* mClusters;
226  RawCluster* mClustersEnd;
227 };
