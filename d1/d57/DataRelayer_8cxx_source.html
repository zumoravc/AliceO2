73 bool DataRelayer::processDanglingInputs(std::vector<ExpirationHandler> const& expirationHandlers,
74  ServiceRegistry& services)
75 {
77  if (expirationHandlers.empty()) {
78  return false;
79  }
80  // Create any slot for the time based fields
81  std::vector<TimesliceSlot> slotsCreatedByHandlers;
82  for (auto& handler : expirationHandlers) {
83  slotsCreatedByHandlers.push_back(handler.creator(mTimesliceIndex));
84  }
85  bool didWork = slotsCreatedByHandlers.empty() == false;
86  // Outer loop, we process all the records because the fact that the record
87  // expires is independent from having received data for it.
88  for (size_t ti = 0; ti < mTimesliceIndex.size(); ++ti) {
89  TimesliceSlot slot{ti};
90  if (mTimesliceIndex.isValid(slot) == false) {
91  continue;
92  }
93  assert(mDistinctRoutesIndex.empty() == false);
94  auto timestamp = mTimesliceIndex.getTimesliceForSlot(slot);
95  // We iterate on all the hanlders checking if they need to be expired.
96  for (size_t ei = 0; ei < expirationHandlers.size(); ++ei) {
97  auto& expirator = expirationHandlers[ei];
98  // We check that no data is already there for the given cell
99  // it is enough to check the first element
100  auto& part = mCache[ti * mDistinctRoutesIndex.size() + expirator.routeIndex.value];
101  if (part.size() > 0 && part[0].header != nullptr) {
102  continue;
103  }
104  if (part.size() > 0 && part[0].payload != nullptr) {
105  continue;
106  }
107  // We check that the cell can actually be expired.
108  if (!expirator.checker) {
109  continue;
110  }
111  if (slotsCreatedByHandlers[ei] != slot) {
112  continue;
113  }
114  if (expirator.checker(timestamp.value) == false) {
115  continue;
116  }
117 
118  assert(ti * mDistinctRoutesIndex.size() + expirator.routeIndex.value < mCache.size());
119  assert(expirator.handler);
120  // expired, so we create one entry
121  if (part.size() == 0) {
122  part.parts.resize(1);
123  }
124  expirator.handler(services, part[0], timestamp.value);
125  didWork = true;
126  mTimesliceIndex.markAsDirty(slot, true);
127  assert(part[0].header != nullptr);
128  assert(part[0].payload != nullptr);
129  }
130  }
131  return didWork;
132 }

137 size_t matchToContext(void* data,
138  std::vector<DataDescriptorMatcher> const& matchers,
139  VariableContext& context)
140 {
141  for (size_t ri = 0, re = matchers.size(); ri < re; ++ri) {
142  auto& matcher = matchers[ri];
143 
144  if (matcher.match(reinterpret_cast<char const*>(data), context)) {
145  context.commit();
146  return ri;
147  }
148  context.discard();
149  }
150  return INVALID_INPUT;
151 }

172 DataRelayer::RelayChoice
173  DataRelayer::relay(std::unique_ptr<FairMQMessage>&& header,
174  std::unique_ptr<FairMQMessage>&& payload)
175 {
176  // STATE HOLDING VARIABLES
177  // This is the class level state of the relaying. If we start supporting
178  // multithreading this will have to be made thread safe before we can invoke
179  // relay concurrently.
180  auto& index = mTimesliceIndex;
181 
182  auto& cache = mCache;
183  auto const& readonlyCache = mCache;
184  auto& metrics = mMetrics;
185  auto numInputTypes = mDistinctRoutesIndex.size();
186 
187  // IMPLEMENTATION DETAILS
188  //
189  // This returns the identifier for the given input. We use a separate
190  // function because while it's trivial now, the actual matchmaking will
191  // become more complicated when we will start supporting ranges.
192  auto getInputTimeslice = [& matchers = mInputMatchers,
193  &header,
194  &index](VariableContext& context)
195  -> std::tuple<int, TimesliceId> {
198  auto input = matchToContext(header->GetData(), matchers, context);
199 
200  if (input == INVALID_INPUT) {
201  return {
202  INVALID_INPUT,
203  TimesliceId{TimesliceId::INVALID},
204  };
205  }
208  if (auto pval = std::get_if<uint64_t>(&context.get(0))) {
209  TimesliceId timeslice{*pval};
210  return {input, timeslice};
211  }
212  // If we get here it means we need to push something out of the cache.
213  return {
214  INVALID_INPUT,
215  TimesliceId{TimesliceId::INVALID},
216  };
217  };
218 
219  // We need to prune the cache from the old stuff, if any. Otherwise we
220  // simply store the payload in the cache and we mark relevant bit in the
221  // hence the first if.
222  auto pruneCache = [&cache,
223  &cachedStateMetrics = mCachedStateMetrics,
224  &numInputTypes,
225  &index,
226  &metrics](TimesliceSlot slot) {
227  assert(cache.empty() == false);
228  assert(index.size() * numInputTypes == cache.size());
229  // Prune old stuff from the cache, hopefully deleting it...
230  // We set the current slot to the timeslice value, so that old stuff
231  // will be ignored.
232  assert(numInputTypes * slot.index < cache.size());
233  for (size_t ai = slot.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
234  cache[ai].clear();
235  cachedStateMetrics[ai] = 0;
236  }
237  };
238 
239  // Actually save the header / payload in the slot
240  auto saveInSlot = [&header,
241  &cachedStateMetrics = mCachedStateMetrics,
242  &payload,
243  &cache,
244  &numInputTypes,
245  &metrics](TimesliceId timeslice, int input, TimesliceSlot slot) {
246  auto cacheIdx = numInputTypes * slot.index + input;
247  std::vector<PartRef>& parts = cache[cacheIdx].parts;
248  cachedStateMetrics[cacheIdx] = 1;
249  // TODO: make sure that multiple parts can only be added within the same call of
250  // DataRelayer::relay
251  PartRef entry{std::move(header), std::move(payload)};
252  parts.emplace_back(std::move(entry));
253  assert(header.get() == nullptr && payload.get() == nullptr);
254  };
255 
256  auto updateStatistics = [& stats = mStats](TimesliceIndex::ActionTaken action) {
257  // Update statistics for what happened
258  switch (action) {
259  case TimesliceIndex::ActionTaken::DropObsolete:
260  stats.droppedIncomingMessages++;
261  break;
262  case TimesliceIndex::ActionTaken::DropInvalid:
263  stats.malformedInputs++;
264  stats.droppedIncomingMessages++;
265  break;
266  case TimesliceIndex::ActionTaken::ReplaceUnused:
267  stats.relayedMessages++;
268  break;
269  case TimesliceIndex::ActionTaken::ReplaceObsolete:
270  stats.droppedComputations++;
271  stats.relayedMessages++;
272  break;
273  }
274  };
275 
276  // OUTER LOOP
277  //
278  // This is the actual outer loop processing input as part of a given
279  // timeslice. All the other implementation details are hidden by the lambdas
280  auto input = INVALID_INPUT;
281  auto timeslice = TimesliceId{TimesliceId::INVALID};
282  auto slot = TimesliceSlot{TimesliceSlot::INVALID};
283 
284  // First look for matching slots which already have some
285  // partial match.
286  for (size_t ci = 0; ci < index.size(); ++ci) {
287  slot = TimesliceSlot{ci};
288  if (index.isValid(slot) == false) {
289  continue;
290  }
291  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
292  if (input != INVALID_INPUT) {
293  break;
294  }
295  }
296 
297  // If we did not find anything, look for slots which
298  // are invalid.
299  if (input == INVALID_INPUT) {
300  for (size_t ci = 0; ci < index.size(); ++ci) {
301  slot = TimesliceSlot{ci};
302  if (index.isValid(slot) == true) {
303  continue;
304  }
305  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
306  if (input != INVALID_INPUT) {
307  break;
308  }
309  }
310  }
311 
313  if (input != INVALID_INPUT && TimesliceId::isValid(timeslice) && TimesliceSlot::isValid(slot)) {
314  O2_SIGNPOST(O2_PROBE_DATARELAYER, timeslice.value, 0, 0, 0);
315  saveInSlot(timeslice, input, slot);
316  index.publishSlot(slot);
317  index.markAsDirty(slot, true);
318  mStats.relayedMessages++;
319  return WillRelay;
320  }
321 
324  VariableContext pristineContext;
325  std::tie(input, timeslice) = getInputTimeslice(pristineContext);
326 
327  auto DataHeaderInfo = [&header]() {
328  std::string error;
329  const auto* dh = o2::header::get<o2::header::DataHeader*>(header->GetData());
330  if (dh) {
331  error += dh->dataOrigin.as<std::string>() + "/" + dh->dataDescription.as<std::string>() + "/" + dh->subSpecification;
332  } else {
333  error += "invalid header";
334  }
335  return error;
336  };
337 
338  if (input == INVALID_INPUT) {
339  LOG(ERROR) << "Could not match incoming data to any input route: " << DataHeaderInfo();
340  mStats.malformedInputs++;
341  mStats.droppedIncomingMessages++;
342  return WillNotRelay;
343  }
344 
345  if (TimesliceId::isValid(timeslice) == false) {
346  LOG(ERROR) << "Could not determine the timeslice for input: " << DataHeaderInfo();
347  mStats.malformedInputs++;
348  mStats.droppedIncomingMessages++;
349  return WillNotRelay;
350  }
351 
352  TimesliceIndex::ActionTaken action;
353  std::tie(action, slot) = index.replaceLRUWith(pristineContext);
354 
355  updateStatistics(action);
356 
357  if (action == TimesliceIndex::ActionTaken::DropObsolete) {
358  LOG(WARNING) << "Incoming data is already obsolete, not relaying.";
359  return WillNotRelay;
360  }
361 
362  if (action == TimesliceIndex::ActionTaken::DropInvalid) {
363  LOG(WARNING) << "Incoming data is invalid, not relaying.";
364  return WillNotRelay;
365  }
366 
367  // At this point the variables match the new input but the
368  // cache still holds the old data, so we prune it.
369  pruneCache(slot);
370  saveInSlot(timeslice, input, slot);
371  index.publishSlot(slot);
372  index.markAsDirty(slot, true);
373 
374  return WillRelay;
375 }

377 std::vector<DataRelayer::RecordAction> DataRelayer::getReadyToProcess()
378 {
379  // THE STATE
380  std::vector<RecordAction> completed;
381  completed.reserve(16);
382  const auto& cache = mCache;
383  const auto numInputTypes = mDistinctRoutesIndex.size();
384  //
385  // THE IMPLEMENTATION DETAILS
386  //
387  // We use this to bail out early from the check as soon as we find something
388  // which we know is not complete.
389  auto getPartialRecord = [&cache, &numInputTypes](int li) -> gsl::span<MessageSet const> {
390  auto offset = li * numInputTypes;
391  assert(cache.size() >= offset + numInputTypes);
392  auto const start = cache.data() + offset;
393  auto const end = cache.data() + offset + numInputTypes;
394  return gsl::span<MessageSet const>(start, end);
395  };
396 
397  // These two are trivial, but in principle the whole loop could be parallelised
398  // or vectorised so "completed" could be a thread local variable which needs
399  // merging at the end.
400  auto updateCompletionResults = [&completed](TimesliceSlot li, CompletionPolicy::CompletionOp op) {
401  completed.emplace_back(RecordAction{li, op});
402  };
403 
404  auto completionResults = [&completed]() -> std::vector<RecordAction> {
405  return completed;
406  };
407 
408  // THE OUTER LOOP
409  //
410  // To determine if a line is complete, we iterate on all the arguments
411  // and check if they are ready. We do it this way, because in the end
412  // the number of inputs is going to be small and having a more complex
413  // structure will probably result in a larger footprint in any case.
414  // Also notice that ai == inputsNumber only when we reach the end of the
415  // iteration, that means we have found all the required bits.
416  //
417  // Notice that the only time numInputTypes is 0 is when we are a dummy
418  // device created as a source for timers / conditions.
419  if (numInputTypes == 0) {
420  return {};
421  }
422  size_t cacheLines = cache.size() / numInputTypes;
423  assert(cacheLines * numInputTypes == cache.size());
424 
425  for (size_t li = 0; li < cacheLines; ++li) {
426  TimesliceSlot slot{li};
427  // We only check the cachelines which have been updated by an incoming
428  // message.
429  if (mTimesliceIndex.isDirty(slot) == false) {
430  continue;
431  }
432  auto partial = getPartialRecord(li);
433  auto getter = [&partial](size_t idx, size_t part) {
434  if (partial[idx].size() > 0 && partial[idx].at(part).header && partial[idx].at(part).payload) {
435  return DataRef{nullptr,
436  reinterpret_cast<const char*>(partial[idx].at(part).header->GetData()),
437  reinterpret_cast<const char*>(partial[idx].at(part).payload->GetData())};
438  }
439  return DataRef{};
440  };
441  auto nPartsGetter = [&partial](size_t idx) {
442  return partial[idx].size();
443  };
444  auto action = mCompletionPolicy.callback({getter, nPartsGetter, static_cast<size_t>(partial.size())});
445  switch (action) {
446  case CompletionPolicy::CompletionOp::Consume:
447  case CompletionPolicy::CompletionOp::Process:
448  case CompletionPolicy::CompletionOp::Discard:
449  updateCompletionResults(slot, action);
450  break;
451  case CompletionPolicy::CompletionOp::Wait:
452  break;
453  }
454  // Given we have created an action for this cacheline, we need to wait for
455  // a new message before we look again into the given cacheline.
456  mTimesliceIndex.markAsDirty(slot, false);
457  }
458  return completionResults();
459 }

461 std::vector<o2::framework::MessageSet> DataRelayer::getInputsForTimeslice(TimesliceSlot slot)
462 {
463  const auto numInputTypes = mDistinctRoutesIndex.size();
464  // State of the computation
465  std::vector<MessageSet> messages(numInputTypes);
466  auto& cache = mCache;
467  auto& index = mTimesliceIndex;
468  auto& metrics = mMetrics;
469 
470  // Nothing to see here, this is just to make the outer loop more understandable.
471  auto jumpToCacheEntryAssociatedWith = [](TimesliceSlot) {
472  return;
473  };
474 
475  // We move ownership so that the cache can be reused once the computation is
476  // finished. We mark the given cache slot invalid, so that it can be reused
477  // This means we can still handle old messages if there is still space in the
478  // cache where to put them.
479  auto moveHeaderPayloadToOutput = [&messages,
480  &cachedStateMetrics = mCachedStateMetrics,
481  &cache, &index, &numInputTypes, &metrics](TimesliceSlot s, size_t arg) {
482  auto cacheId = s.index * numInputTypes + arg;
483  cachedStateMetrics[cacheId] = 2;
484  // TODO: in the original implementation of the cache, there have been only two messages per entry,
485  // check if the 2 above corresponds to the number of messages.
486  if (cache[cacheId].size() > 0) {
487  messages[arg] = std::move(cache[cacheId]);
488  }
489  index.markAsInvalid(s);
490  };
491 
492  // An invalid set of arguments is a set of arguments associated to an invalid
493  // timeslice, so I can simply do that. I keep the assertion there because in principle
494  // we should have dispatched the timeslice already!
495  // FIXME: what happens when we have enough timeslices to hit the invalid one?
496  auto invalidateCacheFor = [&numInputTypes, &index, &cache](TimesliceSlot s) {
497  for (size_t ai = s.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
498  assert(std::accumulate(cache[ai].begin(), cache[ai].end(), true, [](bool result, auto const& element) { return result && element.header.get() == nullptr && element.payload.get() == nullptr; }));
499  cache[ai].clear();
500  }
501  index.markAsInvalid(s);
502  };
503 
504  // Outer loop here.
505  jumpToCacheEntryAssociatedWith(slot);
506  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
507  moveHeaderPayloadToOutput(slot, ai);
508  }
509  invalidateCacheFor(slot);
510 
511  return std::move(messages);
512 }
