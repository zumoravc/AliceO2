322  void attachHits(FairMQChannel& channel, FairMQParts& parts) override
323  {
324  int probe = 0;
325  // check if there is anything to be attached
326  // at least the first hit index should return non nullptr
327  if (static_cast<Det*>(this)->Det::getHits(0) == nullptr) {
328  return;
329  }

331  attachDetIDHeaderMessage(GetDetId(), channel, parts); // the DetId s are universal as they come from o2::detector::DetID
332 
333  while (auto hits = static_cast<Det*>(this)->Det::getHits(probe++)) {
334  if (!UseShm<Det>::value || !o2::utils::ShmManager::Instance().isOperational()) {
335  attachTMessage(*hits, channel, parts);
336  } else {
337  // this is the shared mem variant
338  // we will just send the sharedmem ID and the offset inside
339  *mShmBusy[mCurrentBuffer] = true;
340  attachShmMessage((void*)hits, channel, parts, mShmBusy[mCurrentBuffer]);

345  // this merges several entries from the TBranch brname from the origin TTree
346  // into a single entry in a target TTree / same branch
347  // (assuming T is typically a vector; merging is simply done by appending)
348  // make this function a (static helper)
349  template <typename T>
350  void mergeAndAdjustHits(std::string const& brname, TTree& origin, TTree& target,
351  std::vector<int> const& trackoffsets, std::vector<int> const& nprimaries, std::vector<int> const& subevtsOrdered)
352  {
353  auto originbr = origin.GetBranch(brname.c_str());
354  if (originbr) {
355  auto targetdata = new T;
356  T* incomingdata = nullptr;
357  originbr->SetAddress(&incomingdata);
358 
359  T* filladdress = nullptr;
360  if (origin.GetEntries() == 1) {
361  originbr->GetEntry(0);
362  filladdress = incomingdata;
363  } else {
364  Int_t entries = origin.GetEntries();
365  Int_t nprimTot = 0;
366  for (auto entry = 0; entry < entries; entry++)
367  nprimTot += nprimaries[entry];
368  // offset for pimary track index
369  Int_t idelta0 = 0;
370  // offset for secondary track index
371  Int_t idelta1 = nprimTot;
372  for (int entry = entries - 1; entry >= 0; --entry) {
373  // proceed in the order of subevent Ids
374  Int_t index = subevtsOrdered[entry];
375  // numbe of primaries for this event
376  Int_t nprim = nprimaries[index];
377  idelta1 -= nprim;
378  filladdress = targetdata;
379  originbr->GetEntry(index);
380  if (incomingdata) {
381  // fix the trackIDs for this data
382  for (auto& hit : *incomingdata) {
383  const auto oldID = hit.GetTrackID();
384  // offset depends on whether the trackis a primary or secondary
385  Int_t offset = (oldID < nprim) ? idelta0 : idelta1;
386  hit.SetTrackID(oldID + offset);
387  }
388  // this could be further generalized by using a policy for T
389  std::copy(incomingdata->begin(), incomingdata->end(), std::back_inserter(*targetdata));
390  // adjust offset
391  delete incomingdata;
392  incomingdata = nullptr;
393  }
394  // adjust offsets for next subevent
395  idelta0 += nprim;
396  idelta1 += trackoffsets[index];
397  } // subevent loop
398  }
399  // fill target for this event
400  auto targetbr = o2::base::getOrMakeBranch(target, brname.c_str(), &filladdress);
401  targetbr->SetAddress(&filladdress);
402  targetbr->Fill();
403  targetbr->ResetAddress();
404  targetdata->clear();
405  if (incomingdata) {
406  delete incomingdata;
407  incomingdata = nullptr;
408  }
409  delete targetdata;
410  }
411  }

413  void mergeHitEntries(TTree& origin, TTree& target, std::vector<int> const& trackoffsets, std::vector<int> const& nprimaries, std::vector<int> const& subevtsOrdered) final
414  {
415  // loop over hit containers / different branches
416  // adjust trackID in hits on the go
417  int probe = 0;
418  using Hit_t = decltype(static_cast<Det*>(this)->Det::getHits(probe));
419  std::string name = static_cast<Det*>(this)->getHitBranchNames(probe++);
420  while (name.size() > 0) {
421  mergeAndAdjustHits<typename std::remove_pointer<Hit_t>::type>(name, origin, target, trackoffsets, nprimaries, subevtsOrdered);
422  // next name
423  name = static_cast<Det*>(this)->getHitBranchNames(probe++);
424  }
425  }

427  public:
428  void fillHitBranch(TTree& tr, FairMQParts& parts, int& index) override
429  {
430  int probe = 0;
431  bool* busy = nullptr;
432  using Hit_t = decltype(static_cast<Det*>(this)->Det::getHits(probe));
433  std::string name = static_cast<Det*>(this)->getHitBranchNames(probe++);
434  while (name.size() > 0) {
435  if (!UseShm<Det>::value || !o2::utils::ShmManager::Instance().isOperational()) {
436 
437  // for each branch name we extract/decode hits from the message parts ...
438  auto hitsptr = decodeTMessage<Hit_t>(parts, index++);
439  if (hitsptr) {
440  // ... and fill the tree branch
441  auto br = getOrMakeBranch(tr, name.c_str(), hitsptr);
442  br->SetAddress(static_cast<void*>(&hitsptr));
443  br->Fill();
444  br->ResetAddress();
445  delete hitsptr;
446  }
447  } else {
448  // for each branch name we extract/decode hits from the message parts ...
449  auto hitsptr = decodeShmMessage<Hit_t>(parts, index++, busy);
450  // ... and fill the tree branch
451  auto br = getOrMakeBranch(tr, name.c_str(), hitsptr);
452  br->SetAddress(static_cast<void*>(&hitsptr));
453  br->Fill();
454  br->ResetAddress();
455  }
456  // next name
457  name = static_cast<Det*>(this)->getHitBranchNames(probe++);
458  }
459  // there is only one busy flag per detector so we need to clear it only
460  // at the end (after all branches have been treated)
461  if (busy) {
462  *busy = false;
463  }
464  }

497  // creating a number of hit buffers (in shared mem) -- to which
498  // detectors can write in round-robin fashion
499  void createHitBuffers()
500  {
501  using VectorHit_t = decltype(static_cast<Det*>(this)->Det::getHits(0));
502  using Hit_t = typename std::remove_pointer<VectorHit_t>::type::value_type;
503  for (int buffer = 0; buffer < NHITBUFFERS; ++buffer) {
504  int probe = 0;
505  bool more{false};
506  do {
507  auto ptr = o2::utils::createSimVector<Hit_t>();
508  more = static_cast<Det*>(this)->Det::setHits(probe, ptr);
509  mCachedPtr[buffer].emplace_back(ptr);
510  probe++;
511  } while (more);
512  }
513  }

515  void initializeLate() final
516  {

531  void BeginEvent() final
532  {
533  if (UseShm<Det>::value) {
534  mCurrentBuffer = (mCurrentBuffer + 1) % NHITBUFFERS;
535  while (mShmBusy[mCurrentBuffer] != nullptr && *mShmBusy[mCurrentBuffer]) {
536  // this should ideally never happen
537  LOG(INFO) << " BUSY WAITING SIZE ";
538  sleep(1);
539  }
540 
541  using Hit_t = decltype(static_cast<Det*>(this)->Det::getHits(0));
542 
543  // now we have to clear the hits before writing again
544  int probe = 0;
545  for (auto bareptr : mCachedPtr[mCurrentBuffer]) {
546  auto hits = static_cast<Hit_t>(bareptr);
547  // assign ..
548  static_cast<Det*>(this)->Det::setHits(probe, hits);
549  hits->clear();
550  probe++;
551  }
552  }
553  }

555  ~DetImpl() override
556  {
557  for (int i = 0; i < NHITBUFFERS; ++i) {
558  if (mShmBusy[i]) {
559  auto& instance = o2::utils::ShmManager::Instance();
560  if (instance.hasSegment()) {
561  instance.freememblock(mShmBusy[i]);
562  } else {
563  delete mShmBusy[i];
564  }
565  }
566  }
567  freeHitBuffers();
568  }
