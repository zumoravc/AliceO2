180 class DataPageWriter
181 {
182  std::vector<char> mBuffer;
183  int mNpacketsInBuffer = 0;
184  std::vector<std::vector<char>> mPages;
185  std::vector<int> mNpackets;
186  static constexpr int MAX_Page_size = 8192;
187 
188  public:
189  o2::header::RAWDataHeader mRDH;
190  void flush(std::ostream& str)
191  {
192  writePage();
193  mRDH.stop = 0;
194  for (int page = 0; page < int(mPages.size()); ++page) {
195  mRDH.memorySize = mPages[page].size() + mRDH.headerSize;
196  mRDH.offsetToNext = mRDH.memorySize;
197  mRDH.packetCounter = mNpackets[page];
198  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
199  str.write(mPages[page].data(), mPages[page].size());
200  mRDH.pageCnt++;
201  }
202  if (!mPages.empty()) {
203  mRDH.memorySize = mRDH.headerSize;
204  mRDH.offsetToNext = mRDH.memorySize;
205  mRDH.stop = 1;
206  mRDH.pageCnt++;
207  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
208  mPages.clear();
209  mNpackets.clear();
210  }
211  }
212 
213  void writePage()
214  {
215  if (mBuffer.size() == 0)
216  return;
217  mPages.emplace_back(std::move(mBuffer));
218  LOG(DEBUG) << " writePage " << mBuffer.size();
219  mNpackets.push_back(mNpacketsInBuffer);
220  mNpacketsInBuffer = 0;
221  mBuffer.clear();
222  }
223 
224  void write(std::vector<char> const& new_data)
225  {
226  if (mBuffer.size() + new_data.size() + mRDH.headerSize > MAX_Page_size) {
227  LOG(DEBUG) << " write rest " << mBuffer.size() << " " << new_data.size() << " " << mRDH.headerSize;
228  writePage();
229  }
230  LOG(DEBUG) << " write vector " << new_data.size() << " buffer " << mBuffer.size() << " RDH " << mRDH.headerSize << " new data " << new_data.data();
231  mBuffer.insert(mBuffer.end(), new_data.begin(), new_data.end());
232  mNpacketsInBuffer++;
233  LOG(DEBUG) << " write vector end mBuffer.size " << mBuffer.size() << " mNpacketsInBuffer " << mNpacketsInBuffer << " newdtata " << new_data.size();
234  }
