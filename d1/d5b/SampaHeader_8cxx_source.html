130 bool isHeartbeat(uint64_t header)
131 {
132  // - bits 7-9 must be zero
133  // - bits 10-19 must be zero
134  // - bits 24,26,28 must be one
135  // - bits 25,27 must be zero
136  // - bit 49 must be zero
137  for (int i = 7; i <= 9; i++) {
138  if (!checkBit(header, i, false)) {
139  return false;
140  }
141  }
142  for (int i = 10; i <= 19; i++) {
143  if (!checkBit(header, i, false)) {
144  return false;
145  }
146  }
147  if (!checkBit(header, 24, true) || !checkBit(header, 26, true) || !checkBit(header, 28, true) || !checkBit(header, 25, false) || !checkBit(header, 27, false) || !checkBit(header, 49, false)) {
148  return false;
149  }
150  return true;
151 }

377 std::string asString(const SampaHeader& sh)
378 {
379  std::stringstream os;
380 
381  uint64_t one = 1;
382  std::vector<int> sep = {5, 6, 9, 19, 23, 28, 48};
383  for (int i = 0; i < 50; i++) {
384  os << ((sh.uint64() & (one << i)) ? "1" : "0");
385  if (std::find(begin(sep), end(sep), i) != end(sep)) {
386  os << "|";
387  }
388  }
389  os << "\n";
390  os << fmt::sprintf("%6x %d %3x %10x %4x %5x %20x %d (0x%x) | %s %s",
391  sh.hammingCode(),
392  sh.headerParity(),
393  static_cast<uint8_t>(sh.packetType()),
394  sh.nof10BitWords(),
395  sh.chipAddress(),
396  sh.channelAddress(),
397  sh.bunchCrossingCounter(),
398  sh.payloadParity(),
399  sh.uint64(),
400  packetTypeName(sh.packetType()),
401  sh.hasError() ? "ERROR" : "")
402  << "\n";
403  return os.str();
404 }

421 int partialOddParity3(uint64_t value, int pos)
422 {
423  // compute the odd parity of all the bits at position x
424  // (where x & (2^pos)) are set
425  //
426 
427  int n{0};
428  uint64_t one{1};
429  const uint64_t test{one << pos};
430 
431  for (uint64_t i = 0; i < 49; i++) {
432  if (conv[i] < 0) {
433  continue;
434  }
435  if ((i + 1) & test) {
436  if (value & (one << conv[i])) {
437  ++n;
438  }
439  }
440  }
441  return (n + 1) % 2 == 0;
442 }

445 {

448  int hamming{0};

451  hamming += partialOddParity3(value, i) * (1 << i);

562 int computeHeaderParity1(uint64_t v)
563 {
564  int n{0};
565  constexpr uint64_t one{1};
566  for (int i = 0; i < 6; i++) {
567  if (v & (one << i)) {
568  n++;
569  }
