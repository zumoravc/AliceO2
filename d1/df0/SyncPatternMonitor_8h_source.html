102 inline void SyncPatternMonitor::checkWord(const short hw, const short pos)
103 {
104  ++mCheckedWords;
105  if (hw == SYNC_PATTERN[mPosition])
106  ++mPosition;
107  else if (!(mPosition == SYNC_START + 2 && hw == SYNC_PATTERN[mPosition - 1]))
108  mPosition = SYNC_START;
109  // Don't reset mPosition at the beginning to avoid missing of start of sync pattern in cases like
110  //
111  //
112  // ... A A A B B ...
113  //
114  // ^ ^ ^
115  // random A _| | |_ would trigger reset
116  // |
117  // real start
118 
119  if (mPosition == 32)
120  patternFound(pos);
121 };

123 inline void SyncPatternMonitor::patternFound(const short hw)
124 {
125  LOG(DEBUG) << "SAMPA " << mSampa << " (" << ((mLowHigh == 0) ? " low" : "high") << "): "

128  LOG(WARNING) << "SAMPA " << mSampa << " (" << ((mLowHigh == 0) ? " low" : "high") << "): "
129  << "SYNC was already found";
130  }
131  mPatternFound = true;
132  mPosition = SYNC_START;
133  mHwWithPattern = hw;
