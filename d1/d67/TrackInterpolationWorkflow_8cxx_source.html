31 framework::WorkflowSpec getTPCInterpolationWorkflow(bool disableRootInp, bool disableRootOut)
32 {
33  framework::WorkflowSpec specs;
34  bool useMC = false;
35  std::vector<int> tpcClusSectors = o2::RangeTokenizer::tokenize<int>("0-35");
36  std::vector<int> tpcClusLanes = tpcClusSectors;
37  if (!disableRootInp) {
38  specs.emplace_back(o2::its::getITSTrackReaderSpec(useMC));
39  specs.emplace_back(o2::tpc::getTPCTrackReaderSpec(useMC));
40  specs.emplace_back(o2::tpc::getPublisherSpec(o2::tpc::PublisherConf{
41  "tpc-native-cluster-reader",
42  "tpcrec",
43  {"clusterbranch", "TPCClusterNative", "Branch with TPC native clusters"},
44  {"clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch"},
45  OutputSpec{"TPC", "CLUSTERNATIVE"},
46  OutputSpec{"TPC", "CLNATIVEMCLBL"},
47  tpcClusSectors,
48  tpcClusLanes},
49  useMC));
50  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(useMC));
51  specs.emplace_back(o2::tof::getClusterReaderSpec(useMC));
52  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(useMC));
53  }
54 
55  specs.emplace_back(o2::tpc::getTPCInterpolationSpec(useMC, tpcClusLanes));
56 
57  if (!disableRootOut) {
58  specs.emplace_back(o2::tpc::getTPCResidualWriterSpec(useMC));
59  }
60  return specs;
61 }
