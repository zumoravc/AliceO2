27 std::string digitIdAsString(const o2::mch::Digit& digit,
28  const Digit2ElecMapper& digit2elec)
29 {
30  auto optElecId = digit2elec(digit);
31  if (!optElecId.has_value()) {
32  return "UNKNOWN";
33  }
34  auto dsElecId = optElecId.value().first;
35  auto dschid = optElecId.value().second;
36  return fmt::format("{}-CH{}", asString(dsElecId), dschid);
37 }
