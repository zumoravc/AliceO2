93  // insert bit at LSB
94  self_type& operator<<=(bool bit)
95  {
96  mCode <<= 1;
97  if (bit) {
98  mCode.set(0);
99  } else {
100  mCode.reset(0);
101  }
102  mCodeLen += 1;
103  return *this;
104  }
105 
106  // insert bit at MSB
107  self_type& operator>>=(bool bit)
108  {
109  if (bit) {
110  mCode.set(mCodeLen);
111  } else {
112  mCode.reset(mCodeLen);
113  }
114  mCodeLen += 1;
115  return *this;
116  }

118  void print(std::ostream& stream = std::cout) const
119  {
120  static int level = 1;
121  stream << "node weight: " << std::setw(9) << mWeight;
122  if (!mLeft) {
123  stream << " leave ";
124  } else {
125  stream << " tree ";
126  }
127  stream << " code length: " << mCodeLen;
128  stream << " code " << mCode;
129  stream << std::endl;
130  level++;
131  if (mLeft) {
132  stream << std::setw(level) << level << ": left: ";
133  mLeft->print(stream);
134  }
135  if (mRight) {
136  stream << std::setw(level) << level << ": right: ";
137  mRight->print(stream);
138  }
139  level--;
140  }

260  code_type Encode(typename _BASE::value_type symbol, uint16_t& codeLength) const
261  {
262  codeLength = 0;
263  auto nodeIndex = _BASE::alphabet_type::getIndex(symbol);
264  if (nodeIndex < mLeaveNodes.size()) {
265  // valid symbol/value
266  codeLength = mLeaveNodes[nodeIndex]->getBinaryCodeLength();
267  return mLeaveNodes[nodeIndex]->getBinaryCode();
268  } else {
269  std::string msg = "symbol ";
270  msg += symbol;
271  msg += " not found in alphapet ";
272  msg += _BASE::getName();
273  throw std::range_error(msg);
274  }
275 
276  static const code_type dummy = 0;
277  return dummy;
278  }

289  value_type Decode(code_type code, uint16_t& codeLength) const
290  {
291  // TODO: need to check if there is a loaded tree, but don't
292  // want to check this every time when calling. Maybe its enough
293  // to let the dereferencing below throw an exception
294  codeLength = 0;
295  typename _BASE::value_type v = 0;
296  // dereference the iterator and get raw pointer from shared pointer
297  // TODO: work on shared pointers here as well
298  // the top node is the only element in the multiset after using the
299  // weighted sort algorithm to build the tree, all nodes are referenced
300  // from their parents in the tree.
301  const node_type* node = (*mTreeNodes.begin()).get();
302  uint16_t codeMSB = code.size() - 1;
303  while (node) {
304  // N.B.: nodes have either both child nodes or none of them
305  if (node->getLeftChild() == nullptr) {
306  // this is a leave node, retrieve value for corresponding index
307  // TODO: validity check for index, this can be done once after
308  // initializing the Huffman configuration, either after training
309  // or loading the configuration
310  return _BASE::alphabet_type::getSymbol(node->getIndex());
311  }
312  if (codeLength > codeMSB) {
313  // the size of the code type is shorter than the Huffman tree length
314  throw std::range_error("code type length insufficient for Huffman tree length");
315  break;
316  }
317  bool bit = false;
318  if (OrderMSB) {
319  bit = code.test(codeMSB - codeLength);
320  } else {
321  bit = code.test(codeLength);
322  }
323  ++codeLength;
324  if (bit) {
325  node = node->getLeftChild();
326  } else {
327  node = node->getRightChild();
328  }
329  }
330  return v;
331  }

375  bool GenerateHuffmanTree()
376  {
377  mLeaveNodes.clear();
378 
379  // probability model provides map of {symbol, weight}-pairs
380  _BASE& model = *this;
381  for (auto i : model) {
382  // create nodes knowing about their index and the symbol weight
383  mLeaveNodes.push_back(std::make_shared<node_type>(i.second, _BASE::alphabet_type::getIndex(i.first)));
384  }
385 
386  // insert pointer to nodes into ordered structure to build tree
387  // since the type is a pointer, a specific 'less' functor needs to
388  // be provided to dereference before applying operator<
389  for (auto& i : mLeaveNodes) {
390  mTreeNodes.insert(i);
391  }
392  while (mTreeNodes.size() > 1) {
393  // create new node combining the two with lowest probability
394  std::shared_ptr<node_type> combinedNode = std::make_shared<node_type>(*mTreeNodes.begin(), *++mTreeNodes.begin());
395  // remove those two nodes from the list
396  mTreeNodes.erase(mTreeNodes.begin());
397  mTreeNodes.erase(mTreeNodes.begin());
398  // insert the new node according to the less functor
399  mTreeNodes.insert(combinedNode);
400  }
401  // assign value, method works on pointer
402  // dereference iterator and shared_ptr to get the raw pointer
403  // TODO: change method to work on shared instead of raw pointers
404  assignCode((*mTreeNodes.begin()).get());
405  return true;
406  }

422  int assignCode(node_type* node)
423  {
424  if (node == nullptr) {
425  return 0;
426  }
427  int codelen = node->getBinaryCodeLength();
428  int retcodelen = codelen;
429  if (node->getLeftChild()) { // bit '1' branch
430  code_type c = node->getBinaryCode();
431  if (OrderMSB) { // note: this is a compile time switch
432  c <<= 1;
433  c.set(0);
434  } else {
435  c.set(codelen);
436  }
437  node->getLeftChild()->setBinaryCode(codelen + 1, c);
438  int branchlen = assignCode(node->getLeftChild());
439  if (retcodelen < branchlen)
440  retcodelen = branchlen;
441  }
442  if (node->getRightChild()) { // bit '0' branch
443  code_type c = node->getBinaryCode();
444  if (OrderMSB) {
445  c <<= 1;
446  c.reset(0);
447  } else {
448  c.reset(codelen);
449  }
450  node->getRightChild()->setBinaryCode(codelen + 1, c);
451  int branchlen = assignCode(node->getRightChild());
452  if (retcodelen < branchlen)
453  retcodelen = branchlen;
454  }
455  return retcodelen;
456  }

505  int read(std::istream& in)
506  {
507  mLeaveNodes.clear();
508  mTreeNodes.clear();
509  int lineNo = -1;
510  std::string node, left, right, parameters;
511  std::set<int> nodeIndices;
512  struct treeNodeConfiguration {
513  treeNodeConfiguration(int _index, int _left, int _right) : index(_index), left(_left), right(_right) {}
514  int index, left, right;
515  bool operator<(const treeNodeConfiguration& other) const { return index < other.index; }
516  };
517  std::set<treeNodeConfiguration> treeNodeConfigurations;
518  char firstChar = 0;
519  std::map<int, int> leaveNodeMap;
520  while ((in.get(firstChar)) && firstChar != '\n' && (in.putback(firstChar)) && (in >> node) && (in >> left) &&
521  (in >> right) && ++lineNo >= 0) {
522  std::getline(in, parameters);
523  if (lineNo != std::stoi(node)) {
524  std::cerr << "format error: expected node no " << lineNo << ", but got " << node << " (" << left << " " << right
525  << " " << parameters << ")" << std::endl;
526  std::cerr << "Note: Huffman table dump has to be terminated by blank line or eof" << std::endl;
527  break;
528  }
529  if (parameters.empty()) {
530  // std::cout << "tree node " << lineNo << " left=" << left << " right=" << right << std::endl;
531  int leftIndex = std::stoi(left);
532  int rightIndex = std::stoi(right);
533  auto it = nodeIndices.find(leftIndex);
534  if (it == nodeIndices.end()) {
535  std::cerr << "Format error: can not find left child node with index " << leftIndex << std::endl;
536  return -1;
537  }
538  nodeIndices.erase(it);
539  it = nodeIndices.find(rightIndex);
540  if (it == nodeIndices.end()) {
541  std::cerr << "Format error: can not find right child node with index " << rightIndex << std::endl;
542  return -1;
543  }
544  nodeIndices.erase(it);
545  treeNodeConfigurations.insert(treeNodeConfiguration(lineNo, leftIndex, rightIndex));
546  } else {
547  std::stringstream vs(left), ws(right);
548  typename _BASE::alphabet_type::value_type symbol;
549  vs >> symbol;
550  typename _BASE::weight_type weight;
551  ws >> weight;
552  int symbolIndex = _BASE::alphabet_type::getIndex(symbol);
553  // grow the vector as operator[] always expects index within range
554  if (mLeaveNodes.size() < symbolIndex + 1) {
555  mLeaveNodes.resize(symbolIndex + 1);
556  }
557  mLeaveNodes[symbolIndex] = std::make_shared<node_type>(weight, symbolIndex);
558  std::stringstream ps(parameters);
559  uint16_t codeLen = 0;
560  ps >> codeLen;
561  code_type code = 0;
562  ps >> code;
563  mLeaveNodes[symbolIndex]->setBinaryCode(codeLen, code);
564  leaveNodeMap[lineNo] = symbolIndex;
565  _BASE::addWeight(symbol, weight);
566  // std::cout << "leave node " << lineNo << " " << " value=" << value << " weight=" << weight << " " << codeLen
567  // << " " << code << std::endl;
568  }
569  nodeIndices.insert(lineNo);
570  }
571  std::map<int, std::shared_ptr<node_type>> treeNodes;
572  for (auto conf : treeNodeConfigurations) {
573  std::shared_ptr<node_type> left_local;
574  auto ln = leaveNodeMap.find(conf.left);
575  if (ln != leaveNodeMap.end()) {
576  left_local = mLeaveNodes[ln->second];
577  leaveNodeMap.erase(ln);
578  } else {
579  auto tn = treeNodes.find(conf.left);
580  if (tn == treeNodes.end()) {
581  std::cerr << "Internal error: can not find left child node with index " << conf.left << std::endl;
582  return -1;
583  }
584  left_local = tn->second;
585  treeNodes.erase(tn);
586  }
587  std::shared_ptr<node_type> right_local;
588  auto rn = leaveNodeMap.find(conf.right);
589  if (rn != leaveNodeMap.end()) {
590  right_local = mLeaveNodes[rn->second];
591  leaveNodeMap.erase(rn);
592  } else {
593  auto tn = treeNodes.find(conf.right);
594  if (tn == treeNodes.end()) {
595  std::cerr << "Internal error: can not find right child node with index " << conf.right << std::endl;
596  return -1;
597  }
598  right_local = tn->second;
599  treeNodes.erase(tn);
600  }
601  // make combined node shared ptr and add to map
602  treeNodes[conf.index] = std::make_shared<node_type>(left_local, right_local);
603  }
604  if (leaveNodeMap.size() != 0 || treeNodes.size() != 1) {
605  std::cerr << "error: " << leaveNodeMap.size() << " unhandled leave node(s)"
606  << "; " << treeNodes.size() << " tree nodes(s), expected 1" << std::endl;
607  }
608  mTreeNodes.insert(treeNodes.begin()->second);
609  return 0;
610  }

649  return rightIndex + 1;
