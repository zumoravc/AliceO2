133 Bool_t
134  Generator::triggerEvent()
135 {
139  if (mTriggers.size() == 0 && mDeepTriggers.size() == 0)
140  return kTRUE;
141 
143  Bool_t triggered;
144  if (mTriggerMode == kTriggerOFF)
145  return kTRUE;
146  else if (mTriggerMode == kTriggerOR)
147  triggered = kFALSE;
148  else if (mTriggerMode == kTriggerAND)
149  triggered = kTRUE;
150  else
151  return kTRUE;
152 
154  for (const auto& trigger : mTriggers) {
155  auto retval = trigger(mParticles);
156  if (mTriggerMode == kTriggerOR)
157  triggered |= retval;
158  if (mTriggerMode == kTriggerAND)
159  triggered &= retval;
160  }
161 
163  for (const auto& trigger : mDeepTriggers) {
164  auto retval = trigger(mInterface, mInterfaceName);
165  if (mTriggerMode == kTriggerOR)
166  triggered |= retval;
167  if (mTriggerMode == kTriggerAND)
168  triggered &= retval;
169  }
170 
172  return triggered;
173 }
