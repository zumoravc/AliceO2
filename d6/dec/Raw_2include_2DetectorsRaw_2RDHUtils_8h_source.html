55  template <typename H>
56  static constexpr int getVersion()
57  {
58 #ifndef __OPENCL__
59  RDHAny::sanityCheckStrict<H>();
60  if (std::is_same<H, RDHv6>::value) {
61  return 6;
62  }
63  if (std::is_same<H, RDHv5>::value) {
64  return 5;
65  }
66  if (std::is_same<H, RDHv4>::value) {
67  return 4;
68  }
69 #else
70  return -1; // dummy value as this method will be used on the CPU only
71 #endif // __OPENCL__
72  }
