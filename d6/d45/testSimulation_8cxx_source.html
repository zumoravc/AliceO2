112 std::vector<Hit> generateHits(size_t nHits, int deId, const Mapping& mapping, const GeometryTransformer geoTrans)
113 {
114  std::vector<Hit> hits;
115  std::random_device rd;
116  std::mt19937 mt(rd());
117  std::uniform_real_distribution<double> distX(-127.5, 127.5);
118  std::uniform_real_distribution<double> distY(-40., 40.);
119  while (hits.size() < nHits) {
120  Point3D<float> point(distX(mt), distY(mt), 0.);
121  if (!mapping.stripByPosition(point.x(), point.y(), 0, deId, false).isValid()) {
122  continue;
123  }
124  Point3D<float> globalPoint = geoTrans.localToGlobal(deId, point);
125  hits.emplace_back(hits.size(), deId, globalPoint, globalPoint);
126  }
127  return hits;
128 }

130 std::vector<GenTrack> generateTracks(int nTracks)
131 {
132  Mapping::MpStripIndex stripIndex;
133  auto tracks = simTracking.trackGen.generate(nTracks);
134  std::vector<GenTrack> genTracks;
135  for (auto& track : tracks) {
136  int trackId = &track - &tracks[0];
137  GenTrack genTrack;
138  genTrack.track = track;
139  for (int ich = 0; ich < 4; ++ich) {
140  auto clusters = simTracking.hitFinder.getLocalPositions(track, ich);
141  bool isFired = false;
142  for (auto& cl : clusters) {
143  stripIndex = simBase.mapping.stripByPosition(cl.xCoor, cl.yCoor, 0, cl.deId, false);
144  if (!stripIndex.isValid()) {
145  continue;
146  }
147  auto pos = simBase.geoTrans.localToGlobal(cl.deId, cl.xCoor, cl.yCoor);
148  genTrack.hits.emplace_back(trackId, cl.deId, pos, pos);
149  isFired = true;
150  }
151  if (isFired) {
152  ++genTrack.nFiredChambers;
153  }
154  }
155  genTracks.emplace_back(genTrack);
156  }
157  return genTracks;
158 }

160 bool checkLabel(const ColumnData& digit, MCLabel& label, std::string& errorMessage)
161 {
162  std::stringstream ss;
163  ss << digit << "\n";
164  bool isOk = true;
165  if (label.getDEId() != digit.deId) {
166  isOk = false;
167  ss << "label deId " << label.getDEId() << " != " << digit.deId << "\n";
168  }
169  if (label.getColumnId() != digit.columnId) {
170  isOk = false;
171  ss << "label columnId " << label.getColumnId() << " != " << digit.columnId << "\n";
172  }
173  int firstStrip = label.getFirstStrip();
174  int lastStrip = label.getLastStrip();
175  int cathode = label.getCathode();
176  int nLines = (cathode == 0) ? 4 : 1;
177  for (int iline = 0; iline < nLines; ++iline) {
178  for (int istrip = 0; istrip < 16; ++istrip) {
179  int currStrip = MCLabel::getStrip(istrip, iline);
180  bool isFired = digit.isStripFired(istrip, cathode, iline);
181  if (isFired != (currStrip >= firstStrip && currStrip <= lastStrip)) {
182  isOk = false;
183  ss << "Cathode: " << cathode << " firstStrip: " << firstStrip << " lastStrip: " << lastStrip;
184  ss << " line: " << iline << " strip: " << istrip << " fired: " << isFired << "\n";
185  }
186  }
187  }
188  errorMessage = ss.str();
189  return isOk;
190 }

192 std::vector<PreCluster> getRelatedPreClusters(const Hit& hit, int cathode, const std::vector<PreCluster>& preClusters, const o2::dataformats::MCTruthContainer<MCCompLabel>& labels, const ROFRecord& rofRecord)
193 {
194  std::vector<PreCluster> sortedPC;
195  for (size_t ipc = rofRecord.firstEntry; ipc < rofRecord.firstEntry + rofRecord.nEntries; ++ipc) {
196  for (auto& label : labels.getLabels(ipc)) {
197  if (label.getTrackID() == hit.GetTrackID() && preClusters[ipc].cathode == cathode) {
198  sortedPC.emplace_back(preClusters[ipc]);
199  }
200  }
201  }
202  std::sort(sortedPC.begin(), sortedPC.end(), [](const PreCluster& pc1, const PreCluster& pc2) { return (pc1.firstColumn <= pc2.firstColumn); });
203  return sortedPC;
204 }

206 bool isContiguous(const std::vector<PreCluster>& sortedPC)
207 {
208  int lastColumn = sortedPC.front().firstColumn;
209  for (auto it = sortedPC.begin() + 1; it != sortedPC.end(); ++it) {
210  if (it->firstColumn - (it - 1)->firstColumn != 1) {
211  return false;
212  }
213  }
214  return true;
215 }

217 bool isInside(double localX, double localY, const std::vector<PreCluster>& sortedPC, std::string& errorMsg)
218 {
219  std::stringstream ss;
220  ss << "Point: (" << localX << ", " << localY << ") outside PC:";
221  for (auto& pc : sortedPC) {
222  MpArea area = simClustering.preClusterHelper.getArea(pc);
223  ss << " PC: " << pc;
224  ss << " area: x (" << area.getXmin() << ", " << area.getXmax() << ")";
225  ss << " y (" << area.getYmin() << ", " << area.getYmax() << ")";
226  if (localX >= area.getXmin() && localX <= area.getXmax() &&
227  localY >= area.getYmin() && localY <= area.getYmax()) {
228  return true;
229  }
230  }
231  errorMsg = ss.str();
232  return false;
233 }

488 BOOST_DATA_TEST_CASE(MID_SimTracks, boost::unit_test::data::make({1, 2, 3, 4, 5, 6, 7, 8, 9}), nTracks)
489 {
490  // In this test, we generate tracks and we reconstruct them.
491  // It is worth noting that in this case we do not use the default digitizer
492  // But rather a digitizer with a cluster size of 0.
493  // The reason is that, with the current implementation of the digitizer,
494  // which has to be further fine-tuned,
495  // we can have very large clusters, of few local-boards/columns.
496  // In this case the resolution is rather bad, and this impacts the tracking performance,
497  // even when the tracking algorithm has no bugs.
498  // The aim of this test is to check that the algorithms work.
499  // The tracking performance and tuning of the MC will be done in the future via dedicated studies.
500 
501  std::vector<ColumnDataMC> digitStoreMC, digitsAccum;
502  o2::dataformats::MCTruthContainer<MCLabel> digitLabelsMC, digitLabelsAccum;
503  std::vector<ROFRecord> digitsROF;
504  std::vector<std::vector<GenTrack>> genTrackCollection;
505 
506  // In the tracking algorithm, if we have two tracks that are compatible within uncertainties
507  // we keep only one of the two. This is done to avoid duplicated tracks.
508  // In this test we can have many tracks in the same event.
509  // If two tracks are close, they can give two reconstructed tracks compatible among each others,
510  // within their uncertainties. One of the two is therefore rejected.
511  // However, the track might not be compatible with the (rejected) generated track that has no uncertainty.
512  // To avoid this, compare adding a factor 2 in the sigma cut.
513  float chi2Cut = simTracking.tracker.getSigmaCut() * simTracking.tracker.getSigmaCut();
514 
515  unsigned long int nGood = 0, nUntagged = 0, nTaggedNonCompatible = 0, nReconstructible = 0;
516 
517  for (size_t ievent = 0; ievent < 100; ++ievent) {
518  auto genTracks = generateTracks(nTracks);
519  std::vector<Hit> hits;
520  for (auto& genTrack : genTracks) {
521  std::copy(genTrack.hits.begin(), genTrack.hits.end(), std::back_inserter(hits));
522  if (genTrack.isReconstructible()) {
523  ++nReconstructible;
524  }
525  }
526  genTrackCollection.emplace_back(genTracks);
527 
528  simDigitizer.digitizerNoClusterSize.process(hits, digitStoreMC, digitLabelsMC);
529  digitsROF.emplace_back(o2::constants::lhc::LHCBunchSpacingNS * ievent, EventType::Standard, digitsAccum.size(), digitStoreMC.size());
530  std::copy(digitStoreMC.begin(), digitStoreMC.end(), std::back_inserter(digitsAccum));
531  digitLabelsAccum.mergeAtBack(digitLabelsMC);
532  }
533 
534  simDigitizer.digitsMerger.process(digitsAccum, digitLabelsAccum, digitsROF);
535  simClustering.preClusterizer.process(simDigitizer.digitsMerger.getColumnData(), simDigitizer.digitsMerger.getROFRecords());
536  simClustering.correlation.clear();
537  simClustering.clusterizer.process(simClustering.preClusterizer.getPreClusters(), simClustering.preClusterizer.getROFRecords());
538  simClustering.preClusterLabeler.process(simClustering.preClusterizer.getPreClusters(), simDigitizer.digitsMerger.getMCContainer(), simClustering.preClusterizer.getROFRecords(), simDigitizer.digitsMerger.getROFRecords());
539  simClustering.clusterLabeler.process(simClustering.preClusterizer.getPreClusters(), simClustering.preClusterLabeler.getContainer(), simClustering.clusterizer.getClusters(), simClustering.correlation);
540  simTracking.tracker.process(simClustering.clusterizer.getClusters(), simClustering.clusterizer.getROFRecords());
541  simTracking.trackLabeler.process(simTracking.tracker.getClusters(), simTracking.tracker.getTracks(), simClustering.clusterLabeler.getContainer());
542 
543  // For the moment we save all clusters
544  BOOST_TEST(simTracking.tracker.getClusters().size() == simClustering.clusterizer.getClusters().size());
545  BOOST_TEST(simTracking.tracker.getTracks().size() == simTracking.trackLabeler.getTracksLabels().getIndexedSize());
546  BOOST_TEST(simTracking.tracker.getClusters().size() == simTracking.trackLabeler.getTrackClustersLabels().getIndexedSize());
547  BOOST_TEST(simTracking.tracker.getTrackROFRecords().size() == digitsROF.size());
548 
549  std::string debugInfo = "";
550  // Test that all reconstructible tracks are reconstructed
551  for (size_t ievent = 0; ievent < genTrackCollection.size(); ++ievent) {
552  auto firstTrack = simTracking.tracker.getTrackROFRecords()[ievent].firstEntry;
553  auto nTracks = simTracking.tracker.getTrackROFRecords()[ievent].nEntries;
554  for (size_t igen = 0; igen < genTrackCollection[ievent].size(); ++igen) {
555  bool isReco = false;
556  for (size_t itrack = firstTrack; itrack < firstTrack + nTracks; ++itrack) {
557  for (auto& label : simTracking.trackLabeler.getTracksLabels().getLabels(itrack)) {
558  if (label.getTrackID() == igen) {
559  if (simTracking.tracker.getTracks()[itrack].isCompatible(genTrackCollection[ievent][igen].track, chi2Cut)) {
560  isReco = true;
561  break;
562  }
563  }
564  }
565  if (isReco) {
566  break;
567  }
568  }
569  std::stringstream ss;
570  ss << "Gen ID: " << igen << " isReconstructible: " << genTrackCollection[ievent][igen].isReconstructible() << " != isReco " << isReco;
571  if (debugInfo.empty()) {
572  debugInfo = getDebugInfo(genTrackCollection[ievent], simTracking.tracker, simTracking.trackLabeler, simTracking.tracker.getTrackROFRecords()[ievent], simTracking.tracker.getClusterROFRecords()[ievent]);
573  }
574  ss << "\n"
575  << debugInfo;
576  BOOST_TEST(isReco == genTrackCollection[ievent][igen].isReconstructible(), ss.str().c_str());
577  } // loop on generated tracks
578 
579  // Perform some statistics
580  for (size_t itrack = firstTrack; itrack < firstTrack + nTracks; ++itrack) {
581  for (auto& label : simTracking.trackLabeler.getTracksLabels().getLabels(itrack)) {
582  if (label.isEmpty()) {
583  ++nUntagged;
584  continue;
585  }
586  const Track& matchedGenTrack(genTrackCollection[ievent][label.getTrackID()].track);
587  if (simTracking.tracker.getTracks()[itrack].isCompatible(matchedGenTrack, chi2Cut)) {
588  ++nGood;
589  } else {
590  ++nTaggedNonCompatible;
591  }
592  int nMatched = 0;
593  for (int ich = 0; ich < 4; ++ich) {
594  int trClusIdx = simTracking.tracker.getTracks()[itrack].getClusterMatched(ich);
595  if (trClusIdx < 0) {
596  continue;
597  }
598  for (auto& trClusterLabel : simTracking.trackLabeler.getTrackClustersLabels().getLabels(trClusIdx)) {
599  if (trClusterLabel.getTrackID() == label.getTrackID()) {
600  ++nMatched;
601  }
602  }
603  }
604  BOOST_TEST(nMatched >= 3);
605  } // lop on track labels
606  } // loop on reconstructed tracks
607  } // loop on event
608  std::stringstream outMsg;
609  outMsg << "Tracks per event: " << nTracks << " fraction of good: " << static_cast<double>(nGood) / static_cast<double>(nReconstructible) << " untagged: " << static_cast<double>(nUntagged) / static_cast<double>(nReconstructible) << " tagged but not compatible: " << static_cast<double>(nTaggedNonCompatible) / static_cast<double>(nReconstructible);
610 
611  BOOST_TEST_MESSAGE(outMsg.str().c_str());
612 }
