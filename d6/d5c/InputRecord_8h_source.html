181  {
182  if (pos >= mSpan.size() || pos < 0) {
183  throw std::runtime_error("Unknown message requested at position " + std::to_string(pos));
184  }
185  if (part > 0 && part >= getNofParts(pos)) {
186  throw std::runtime_error("Invalid message part index at " + std::to_string(pos) + ":" + std::to_string(part));

215  template <typename T = DataRef, typename R>
216  decltype(auto) get(R binding, int part = 0) const
217  {
218  DataRef ref{nullptr, nullptr};
219  // Get the actual dataref
220  if constexpr (std::is_same_v<std::decay_t<R>, char const*> ||
221  std::is_same_v<std::decay_t<R>, char*> ||
222  std::is_same_v<std::decay_t<R>, std::string>) {
223  try {
224  int pos = -1;
225  if constexpr (std::is_same_v<std::decay_t<R>, std::string>) {
226  pos = getPos(binding.c_str());
227  } else {
228  pos = getPos(binding);
229  }
230  if (pos < 0) {
231  throw std::invalid_argument("no matching route found for " + std::string(binding));
232  }
233  ref = this->getByPos(pos, part);
234  } catch (const std::exception& e) {
235  throw std::runtime_error("Unknown argument requested " + std::string(binding) +
236  " - " + e.what());
237  }
238  } else if constexpr (std::is_same_v<std::decay_t<R>, DataRef>) {
239  ref = binding;
240  } else {
241  static_assert(always_static_assert_v<R>, "Unknown binding type");
242  }
243  if constexpr (std::is_same_v<std::decay_t<T>, DataRef>) {
244  return ref;
245  } else if constexpr (std::is_same<T, std::string>::value) {
246  // substitution for std::string
247  // If we ask for a string, we need to duplicate it because we do not want
248  // the buffer to be deleted when it goes out of scope. The string is built
249  // from the data and its lengh, null-termination is not necessary.
250  // return std::string object
251  auto header = header::get<const header::DataHeader*>(ref.header);
252  assert(header);
253  return std::string(ref.payload, header->payloadSize);
254 
255  // implementation (c)
256  } else if constexpr (std::is_same<T, char const*>::value) {
257  // substitution for const char*
258  // If we ask for a char const *, we simply point to the payload. Notice this
259  // is meant for C-style strings which are expected to be null terminated.
260  // If you want to actually get hold of the buffer, use gsl::span<char> as that will
261  // give you the size as well.
262  // return pointer to payload content
263  return reinterpret_cast<char const*>(ref.payload);
264 
265  // implementation (d)
266  } else if constexpr (std::is_same<T, TableConsumer>::value) {
267  // substitution for TableConsumer
268  // For the moment this is dummy, as it requires proper support to
269  // create the RDataSource from the arrow buffer.
270  auto header = header::get<const header::DataHeader*>(ref.header);
271  assert(header);
272  auto data = reinterpret_cast<uint8_t const*>(ref.payload);
273  return std::make_unique<TableConsumer>(data, header->payloadSize);
274 
275  // implementation (e)
276  } else if constexpr (framework::is_boost_serializable<T>::value || is_specialization<T, BoostSerialized>::value) {
277  // substitution for boost-serialized entities
278  // We have to deserialize the ostringstream.
279  // FIXME: check that the string is null terminated.
280  // @return deserialized copy of payload
281  auto header = header::get<const header::DataHeader*>(ref.header);
282  assert(header);
283  auto str = std::string(ref.payload, header->payloadSize);
284  assert(header->payloadSize == sizeof(T));
285  if constexpr (is_specialization<T, BoostSerialized>::value) {
286  return o2::utils::BoostDeserialize<typename T::wrapped_type>(str);
287  } else {
288  return o2::utils::BoostDeserialize<T>(str);
289  }
290 
291  // implementation (f)
292  } else if constexpr (is_span<T>::value) {
293  // substitution for span of messageable objects
294  // FIXME: there will be std::span in C++20
295  static_assert(is_messageable<typename T::value_type>::value, "span can only be created for messageable types");
296  auto header = header::get<const header::DataHeader*>(ref.header);
297  assert(header);
298  if (sizeof(typename T::value_type) > 1 && header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
299  throw std::runtime_error("Inconsistent serialization method for extracting span");
300  }
301  using ValueT = typename T::value_type;
302  if (header->payloadSize % sizeof(ValueT)) {
303  throw std::runtime_error("Inconsistent type and payload size at " + std::string(ref.spec->binding) +
304  ": type size " + std::to_string(sizeof(ValueT)) +
305  " payload size " + std::to_string(header->payloadSize));
306  }
307  return gsl::span<ValueT const>(reinterpret_cast<ValueT const*>(ref.payload), header->payloadSize / sizeof(ValueT));
308 
309  // implementation (g)
310  } else if constexpr (is_container<T>::value) {
311  // currently implemented only for vectors
312  if constexpr (is_specialization<typename std::remove_const<T>::type, std::vector>::value) {
313  auto header = o2::header::get<const DataHeader*>(ref.header);
314  auto method = header->payloadSerializationMethod;
315  if (method == o2::header::gSerializationMethodNone) {
316  // TODO: construct a vector spectator
317  // this is a quick solution now which makes a copy of the plain vector data
318  auto* start = reinterpret_cast<typename T::value_type const*>(ref.payload);
319  auto* end = start + header->payloadSize / sizeof(typename T::value_type);
320  T result(start, end);
321  return result;
322  } else if (method == o2::header::gSerializationMethodROOT) {
328  using NonConstT = typename std::remove_const<T>::type;
329  if constexpr (is_specialization<T, ROOTSerialized>::value == true || has_root_dictionary<T>::value == true) {
330  // we expect the unique_ptr to hold an object, exception should have been thrown
331  // otherwise
332  auto object = DataRefUtils::as<NonConstT>(ref);
333  // need to swap the content of the deserialized container to a local variable to force return
334  // value optimization
335  T container;
336  std::swap(const_cast<NonConstT&>(container), *object);
337  return container;
338  } else {
339  throw std::runtime_error("No supported conversion function for ROOT serialized message");
340  }
341  } else {
342  throw std::runtime_error("Attempt to extract object from message with unsupported serialization type");
343  }
344  } else {
345  static_assert(always_static_assert_v<T>, "unsupported code path");
346  }
347 
348  // implementation (h)
349  } else if constexpr (is_messageable<T>::value) {
350  // extract a messageable type by reference
351  // Cast content of payload bound by @a binding to known type.
352  // we need to check the serialization type, the cast makes only sense for
353  // unserialized objects
354  using DataHeader = o2::header::DataHeader;
355 
356  auto header = o2::header::get<const DataHeader*>(ref.header);
357  auto method = header->payloadSerializationMethod;
358  if (method != o2::header::gSerializationMethodNone) {
359  // FIXME: we could in principle support serialized content here as well if we
360  // store all extracted objects internally and provide cleanup
361  throw std::runtime_error("Can not extract a plain object from serialized message");
362  }
363  return *reinterpret_cast<T const*>(ref.payload);
364 
365  // implementation (i)
366  } else if constexpr (std::is_pointer<T>::value &&
367  (is_messageable<typename std::remove_pointer<T>::type>::value || has_root_dictionary<typename std::remove_pointer<T>::type>::value)) {
368  // extract a messageable type or object with ROOT dictionary by pointer
369  // return unique_ptr to message content with custom deleter
370  using DataHeader = o2::header::DataHeader;
371  using ValueT = typename std::remove_pointer<T>::type;
372 
373  auto header = o2::header::get<const DataHeader*>(ref.header);
374  auto method = header->payloadSerializationMethod;
375  if (method == o2::header::gSerializationMethodNone) {
376  if constexpr (is_messageable<ValueT>::value) {
377  auto const* ptr = reinterpret_cast<ValueT const*>(ref.payload);
378  // return type with non-owning Deleter instance
379  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(ptr, Deleter<ValueT const>(false));
380  return result;
381  } else if constexpr (is_specialization<ValueT, std::vector>::value && has_messageable_value_type<ValueT>::value) {
382  // TODO: construct a vector spectator
383  // this is a quick solution now which makes a copy of the plain vector data
384  auto* start = reinterpret_cast<typename ValueT::value_type const*>(ref.payload);
385  auto* end = start + header->payloadSize / sizeof(typename ValueT::value_type);
386  auto container = std::make_unique<ValueT>(start, end);
387  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(container.release(), Deleter<ValueT const>(true));
388  return result;
389  }
390  throw std::runtime_error("unsupported code path");
391  } else if (method == o2::header::gSerializationMethodROOT) {
392  // This supports the common case of retrieving a root object and getting pointer.
393  // Notice that this will return a copy of the actual contents of the buffer, because
394  // the buffer is actually serialised, for this reason we return a unique_ptr<T>.
395  // FIXME: does it make more sense to keep ownership of all the deserialised
396  // objects in a single place so that we can avoid duplicate deserializations?
397  // explicitely specify serialization method to ROOT-serialized because type T
398  // is messageable and a different method would be deduced in DataRefUtils
399  // return type with owning Deleter instance, forwarding to default_deleter
400  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(DataRefUtils::as<ROOTSerialized<ValueT>>(ref).release());
401  return result;
402  } else {
403  throw std::runtime_error("Attempt to extract object from message with unsupported serialization type");
404  }
405  } else if constexpr (has_root_dictionary<T>::value) {
406  // retrieving ROOT objects follows the pointer approach, i.e. T* has to be specified
407  // as template parameter and a unique_ptr will be returned, std vectors of ROOT serializable
408  // objects can be retrieved by move, this is handled above in the "container" code branch
409  static_assert(always_static_assert_v<T>, "ROOT objects need to be retrieved by pointer");
410  } else {
411  // non-messageable objects for which serialization method can not be derived by type,
412  // the operation depends on the transmitted serialization method
413  using DataHeader = o2::header::DataHeader;
414 
415  auto header = o2::header::get<const DataHeader*>(ref.header);
416  auto method = header->payloadSerializationMethod;
417  if (method == o2::header::gSerializationMethodNone) {
418  // this code path is only selected if the type is non-messageable
419  throw std::runtime_error(
420  "Type mismatch: attempt to extract a non-messagable object "
421  "from message with unserialized data");
422  } else if (method == o2::header::gSerializationMethodROOT) {
423  // explicitely specify serialization method to ROOT-serialized because type T
424  // is messageable and a different method would be deduced in DataRefUtils
425  // return type with owning Deleter instance, forwarding to default_deleter
426  std::unique_ptr<T const, Deleter<T const>> result(DataRefUtils::as<ROOTSerialized<T>>(ref).release());
427  return result;
428  } else {
429  throw std::runtime_error("Attempt to extract object from message with unsupported serialization type");
430  }
431  }

464  class Iterator : public IteratorBase<T>
465  {
466  public:
467  using ParentType = ParentT;
468  using SelfType = Iterator;
469  using value_type = typename IteratorBase<T>::value_type;
470  using reference = typename IteratorBase<T>::reference;

472  using ElementType = typename std::remove_const<value_type>::type;

474  Iterator() = delete;
475 
476  Iterator(ParentType const* parent, size_t position = 0, size_t size = 0)
477  : mParent(parent), mPosition(position), mSize(size > position ? size : position), mElement{nullptr, nullptr, nullptr}
478  {
479  if (mPosition < mSize) {
480  if (mParent->isValid(mPosition)) {
481  mElement = mParent->getByPos(mPosition);
482  } else {
483  ++(*this);
484  }
485  }
486  }
487 
488  ~Iterator() = default;
489 
490  // prefix increment
491  SelfType& operator++()
492  {
493  while (mPosition < mSize && ++mPosition < mSize) {
494  if (!mParent->isValid(mPosition)) {
495  continue;
496  }
497  mElement = mParent->getByPos(mPosition);
498  break;
499  }
500  if (mPosition >= mSize) {
501  // reset the element to the default value of the type
502  mElement = ElementType{};
503  }
504  return *this;
505  }
506  // postfix increment
507  SelfType operator++(int /*unused*/)
508  {
509  SelfType copy(*this);
510  operator++();
511  return copy;
512  }
513  // return reference
514  reference operator*() const
515  {
516  return mElement;
517  }
518  // comparison
519  bool operator==(const SelfType& rh) const
520  {
521  return mPosition == rh.mPosition;
522  }
523  // comparison
524  bool operator!=(const SelfType& rh) const
525  {
526  return mPosition != rh.mPosition;
527  }
528 
529  bool matches(o2::header::DataHeader matcher) const
