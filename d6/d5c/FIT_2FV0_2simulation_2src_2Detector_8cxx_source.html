87 Bool_t Detector::ProcessHits(FairVolume* v)
88 {
89  // This method is called from the MC stepping
90 
91  // Track only charged particles and photons
92  bool isPhotonTrack = false;
93  Int_t particlePdg = fMC->TrackPid();
94  if (particlePdg == 22) { // If particle is standard PDG photon
95  isPhotonTrack = true;
96  }
97  if (!(isPhotonTrack || fMC->TrackCharge())) {
98  return kFALSE;
99  }
100 
101  // Check track status to define when hit is started and when it is stopped
102  bool startHit = false, stopHit = false;
103  if ((fMC->IsTrackEntering()) || (fMC->IsTrackInside() && !mTrackData.mHitStarted)) {
104  startHit = true;
105  } else if ((fMC->IsTrackExiting() || fMC->IsTrackOut() || fMC->IsTrackStop())) {
106  stopHit = true;
107  }
108 
109  // Increment energy loss at all steps except entrance
110  if (!startHit) {
111  mTrackData.mEnergyLoss += fMC->Edep();
112  }
113 
114  // Track is entering or created in the volume
115  // Start registering new hit, defined as the combination of all the steps from a given particle
116  if (startHit) {
117  mTrackData.mHitStarted = true;
118  mTrackData.mEnergyLoss = 0.;
119  fMC->TrackMomentum(mTrackData.mMomentumStart);
120  fMC->TrackPosition(mTrackData.mPositionStart);
121  }
122 
123  // Track is exiting or stopped within the volume, finalize recording of this hit and save it
124  if (stopHit) {
125  TLorentzVector positionStop;
126  fMC->TrackPosition(positionStop);
127  Int_t trackID = fMC->GetStack()->GetCurrentTrackNumber();
128 
129  // Get unique ID of the detector cell (sensitive volume)
130  Int_t cellId = mGeometry->getCurrentCellId(fMC);
131 
132  Point3D<float> posStart(mTrackData.mPositionStart.X(), mTrackData.mPositionStart.Y(), mTrackData.mPositionStart.Z());
133  Point3D<float> posStop(positionStop.X(), positionStop.Y(), positionStop.Z());
134  Vector3D<float> momStart(mTrackData.mMomentumStart.Px(), mTrackData.mMomentumStart.Py(), mTrackData.mMomentumStart.Pz());
135  addHit(trackID, cellId, posStart, posStop, momStart,
136  mTrackData.mMomentumStart.E(), positionStop.T(),
137  mTrackData.mEnergyLoss, particlePdg);
138  } else {
139  return kFALSE; // do noting more
140  }
141 
142  return kTRUE;
143 }

168 void Detector::createMaterials()

172  // Air mixture
173  const Int_t nAir = 4;
174  Float_t aAir[nAir] = {12.0107, 14.0067, 15.9994, 39.948};
175  Float_t zAir[nAir] = {6, 7, 8, 18};
176  Float_t wAir[nAir] = {0.000124, 0.755267, 0.231781, 0.012827};
177  const Float_t dAir = 0.00120479;
178 
179  // EJ-204 scintillator, based on polyvinyltoluene
180  const Int_t nScint = 2;
181  Float_t aScint[nScint] = {1.00784, 12.0107};
182  Float_t zScint[nScint] = {1, 6};
183  // TODO: Verify which of the following 2 lines is correct
184  Float_t wScint[nScint] = {0.07085, 0.92915}; // based on EJ-204 datasheet: n_atoms/cm3
185  //Float_t wScint[nScint] = { 0.08528, 0.91472 }; // based on chemical composition of base: polyvinyltoluene
186  const Float_t dScint = 1.023;
187 
188  // PMMA plastic mixture: (C5O2H8)n, same for plastic fiber support and for the fiber core
189  // Fiber cladding is different, but it comprises only 3% of the fiber volume, so it is not included
190  const Int_t nPlast = 3;
191  Float_t aPlast[nPlast] = {1.00784, 12.0107, 15.999};
192  Float_t zPlast[nPlast] = {1, 6, 8};
193  Float_t wPlast[nPlast] = {0.08054, 0.59985, 0.31961};

244  o2::base::Detector::Medium(FiberInner, "FiberInner$", matId, unsens, fieldType, maxField,
