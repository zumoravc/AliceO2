35 // number of bunches and the 40 MHz clock with 25 ns bunch spacing
36 // gives revolution time of 89.1 us and 11.223345 kHz
37 // this depends on the assumption that the particles are moving effectively
38 // at speed of light. There are also documents specifying the orbit time
39 // to 89.4 us
40 // Note: avoid to define the revolution frequency and use the integral numbers
41 // for bunch places and bunch spacing in nano seconds
42 // TODO: this eventually needs to be configurable
43 static constexpr int gNumberOfBunches = 3564;
44 static constexpr int gBunchSpacingNanoSec = 25;
45 static constexpr int gOrbitTimeNanoSec = std::ratio<gNumberOfBunches * gBunchSpacingNanoSec>::num;
46 
47 using OrbitPrecision = std::integral_constant<int, 0>;
48 using BunchPrecision = std::integral_constant<int, 1>;
49 
50 // the type of the clock tick depends on whether to use also the bunches
51 // as substructure of the orbit.
52 // a trait class to extrat the properties of the clock, namely the type
53 // of the tick and the period
54 template <typename T>
55 struct Property {
56  // the default does not specify anything and is never going to be used
57 };
58 
59 template <>
60 struct Property<OrbitPrecision> {
61  using rep = uint32_t;
62  // avoid rounding errors by using the integral numbers in the std::ratio
63  // template to define the period
64  using period = std::ratio_multiply<std::ratio<gOrbitTimeNanoSec>, std::nano>;
65 };
66 
67 template <>
68 struct Property<BunchPrecision> {
69  using rep = uint64_t;
70  // this is effectively the LHC clock and the ratio is the
71  // bunch spacing
72  using period = std::ratio_multiply<std::ratio<gBunchSpacingNanoSec>, std::nano>;
73 };
74 } // namespace lhc_clock_parameter

76 // a chrono clock implementation
77 // - always relative to run start
78 // - need run start to calculate the epoch
79 // - based on revolution frequency and number of bunches
80 // TODO: the reference time is probably the start of the fill
81 template <typename RefTimePoint, typename Precision = lhc_clock_parameter::OrbitPrecision>
82 class LHCClock
83 {
84  public:
85  LHCClock(const RefTimePoint& start) : mReference(start) {}
87  LHCClock() = delete;
88  ~LHCClock() = default;
89  LHCClock(const LHCClock&) = default;
90  LHCClock& operator=(const LHCClock&) = default;
91 
92  using rep = typename lhc_clock_parameter::Property<Precision>::rep;
93  using period = typename lhc_clock_parameter::Property<Precision>::period;
94  using duration = std::chrono::duration<rep, period>;
95  using time_point = std::chrono::time_point<LHCClock>;
96  // this follows the naming convention of std chrono
97  static const bool is_steady = true;
98 
101  time_point now() noexcept
102  {
103  // tp1 - tp2 results in a duration, we use to create a time_point with characteristics
104  // of the clock.
105  return time_point(std::chrono::duration_cast<duration>(std::chrono::system_clock::now()) - mReference);
106  }
107 
108  private:
110  RefTimePoint mReference;
111 };

136  template <class T, typename Rep = typename T::rep, typename Period = typename T::period>
137  auto get() const
138  {
139  static_assert(std::is_same<typename T::rep, Rep>::value && std::is_same<typename T::period, Period>::value,
140  "only clock and duration types defining the rep and period member types are allowed");
141  using duration = std::chrono::duration<Rep, Period>;
142  if (mUnit == sClockLHC) {
143  // cast each part individually, if the precision of the return type
144  // is smaller the values are simply truncated
145  return std::chrono::duration_cast<duration>(LHCOrbitClock::duration(mPeriod) + LHCBunchClock::duration(mBCNumber));
146  }
147  if (mUnit == sMicroSeconds) {
148  // TODO: is there a better way to mark the subticks invalid for the
149  // micro seconds representation? First step is probably to remove/rename the
150  // variable
151  assert(mSubTicks == 0);
152  return std::chrono::duration_cast<duration>(std::chrono::microseconds(mTicks));
153  }
154  // invalid time unit identifier
155  // TODO: define error policy
156  assert(0);
157  return std::chrono::duration_cast<duration>(std::chrono::seconds(0));
158  }
