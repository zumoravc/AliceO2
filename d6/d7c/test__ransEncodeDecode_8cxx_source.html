68 BOOST_FIXTURE_TEST_CASE_TEMPLATE(test_EncodeDecode, T, Fixtures, T)
69 {
70  // iterate over the message and create PDF and CDF for each symbol in the message
71  o2::rans::SymbolStatistics stats{std::begin(T::source), std::end(T::source)};
72  // For performance reason the distributions must be rescaled to be a power of two.
73  stats.rescaleToNBits(T::probabilityBits);
74 
75  // buffer to write the rANS coded message into
76  std::vector<typename T::stream_t> encoderBuffer(1 << 20, 0);
77  //create a stateful encoder object that builds an encoding table from the given symbol statistics
78  const typename T::encoder_t encoder{stats, T::probabilityBits};
79  // encoder rANS encodes symbols from SOURCE array to encoder Buffer. Since coder and decoder
80  // are mathematical inverse functions on the ANS state, they operate as a stack -
81  // i.e. the decoder outputs the message in reverse order of the encoder.
82  // By convention the encoder runs backwards over the input so that the decoder can return
83  // the data in the expected order. The encoder will start from encoderBuffer.end()
84  // and output symbols backwards returning an iterator to the first encoded item.
85  // This means the encodeded message ranges from encodedMessageStart to encoderBuffer.end()
86  auto encodedMessageStart = encoder.process(encoderBuffer.begin(), encoderBuffer.end(), std::begin(T::source), std::end(T::source));
87 
88  // The decoded message will go into the decoder buffer which will have as many symbols as the original message
89  std::vector<typename T::source_t> decoderBuffer(std::end(T::source) - std::begin(T::source), 0);
90  // create a stateful decoder object that build a decoding table from the given symbol statistics
91  const typename T::decoder_t decoder{stats, T::probabilityBits};
92  // the decoder unwinds the rANS state in the encoder buffer starting at ransBegin and decodes it into the decode buffer;
93  decoder.process(decoderBuffer.begin(), encodedMessageStart, stats.getMessageLength());
94 
95  //the decodeBuffer and the source message have to be identical afterwards.
96  BOOST_REQUIRE(std::memcmp(&(*T::source.begin()), decoderBuffer.data(),
97  decoderBuffer.size() * sizeof(typename T::source_t)) == 0);
98 }
