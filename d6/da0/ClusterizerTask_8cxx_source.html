70 template <class InputType>
71 void ClusterizerTask<InputType>::process(const std::string inputFileName, const std::string outputFileName)
72 {
73  LOG(DEBUG) << "Running clusterization on new event";
74 
75  // Create reader, initilize clusterizer geometry
76  init();
77 
78  // Load output file
79  std::unique_ptr<TFile> outFile(TFile::Open(outputFileName.data(), "recreate"));
80  if (!outFile || outFile->IsZombie()) {
81  LOG(FATAL) << "Failed to open output file " << outputFileName;
82  }
83 
84  // Create output tree
85  std::unique_ptr<TTree> outTree = std::make_unique<TTree>("o2sim", "EMCAL clusters");
86  outTree->Branch("EMCCluster", &mClustersArray);
87  outTree->Branch("EMCClusterInputIndex", &mClustersInputIndices);
88  outTree->Branch("EMCClusterTRGR", &mClusterTriggerRecordsClusters);
89  outTree->Branch("EMCIndicesTRGR", &mClusterTriggerRecordsIndices);
90 
91  mClustersArray->clear();
92  mClustersInputIndices->clear();
93  mClusterTriggerRecordsClusters->clear();
94  mClusterTriggerRecordsIndices->clear();
95 
96  // Loop over entries of the input tree
97  mInputReader->openInput(inputFileName);
98  while (mInputReader->readNextEntry()) {
99 
100  auto InputVector = mInputReader->getInputArray();
101 
102  int currentStartClusters = mClustersArray->size();
103  int currentStartIndices = mClustersInputIndices->size();
104 
105  for (auto iTrgRcrd = mInputReader->getTriggerArray()->begin(); iTrgRcrd != mInputReader->getTriggerArray()->end(); ++iTrgRcrd) {
106 
107  mClusterizer.findClusters(gsl::span<const InputType>(InputVector->data() + iTrgRcrd->getFirstEntry(), iTrgRcrd->getNumberOfObjects())); // Find clusters on cells/digits given in reader::mInputArray (pass by ref)
108 
109  // Get found clusters + cell/digit indices for output
110  auto clusterstmp = mClusterizer.getFoundClusters();
111  auto clusterIndecestmp = mClusterizer.getFoundClustersInputIndices();
112  std::copy(clusterstmp->begin(), clusterstmp->end(), std::back_inserter(*mClustersArray));
113  std::copy(clusterIndecestmp->begin(), clusterIndecestmp->end(), std::back_inserter(*mClustersInputIndices));
114 
115  mClusterTriggerRecordsClusters->emplace_back(iTrgRcrd->getBCData(), currentStartClusters, clusterstmp->size());
116  mClusterTriggerRecordsIndices->emplace_back(iTrgRcrd->getBCData(), currentStartIndices, clusterIndecestmp->size());
117 
118  currentStartClusters = mClustersArray->size();
119  currentStartIndices = mClustersInputIndices->size();
120  }
121  outTree->Fill();
122  }
123 
124  // Write, close, and destroy tree/file
125  outTree->Write();
126  outTree.reset(); // here we reset the unique ptr, not the tree!
127  outFile->Close();
128 }
