71 int DataInputDescriptor::fillInputfiles()
72 {
73  if (getNumberInputfiles() > 0) {
74  // 1. mfilenames
75  return getNumberInputfiles();
76  }
77 
78  auto fileName = getInputfilesFilename();
79  if (!fileName.empty()) {
80  // 2. getFilenamesRegex() @ getInputfilesFilename()
81  try {
82  std::ifstream filelist(fileName);
83  while (std::getline(filelist, fileName)) {
84  if (getFilenamesRegexString().empty() ||
85  std::regex_match(fileName, getFilenamesRegex())) {
86  addFilename(fileName);
87  }
88  }
89  } catch (...) {
90  LOGP(ERROR, "Check the input files file! Unable to process \"{}\"!", getInputfilesFilename());
91  return 0;
92  }
93  } else {
94  // 3. getFilenamesRegex() @ mdefaultFilenamesPtr
95  if (mdefaultFilenamesPtr) {
96  for (auto fileName : *mdefaultFilenamesPtr) {
97  if (getFilenamesRegexString().empty() ||
98  std::regex_match(fileName, getFilenamesRegex())) {
99  addFilename(fileName);
100  }
101  }
102  }
103  }
104 
105  return getNumberInputfiles();
106 }

170 {
171  // open the file
172  FILE* f = fopen(fnjson.c_str(), "r");
173  if (!f) {
174  LOGP(ERROR, "Could not open JSON file \"{}\"!", fnjson);
175  return false;
176  }
177 
178  // create streamer
179  char readBuffer[65536];
180  FileReadStream inputStream(f, readBuffer, sizeof(readBuffer));
181 
182  // parse the json file
183  Document jsonDoc;
184  jsonDoc.ParseStream(inputStream);
185  auto status = readJsonDocument(&jsonDoc);
186 
187  // clean up
188  fclose(f);
189 
190  return status;

193 bool DataInputDirector::readJsonDocument(Document* jsonDoc)
194 {
195  // initialisations
196  std::string fileName("");
197  int ntfm = -1;
198  const char* itemName;
199 
200  // is it a proper json document?
201  if (jsonDoc->HasParseError()) {
202  LOGP(ERROR, "Check the JSON document! There is a problem with the format!");
203  return false;
204  }
205 
206  // InputDirector
207  itemName = "InputDirector";
208  const Value& didirItem = (*jsonDoc)[itemName];
209  if (!didirItem.IsObject()) {
210  LOGP(ERROR, "Check the JSON document! Couldn't find an \"{}\" object!", itemName);
211  return false;
212  }
213 
214  // now read various items
215  itemName = "debugmode";
216  if (didirItem.HasMember(itemName)) {
217  if (didirItem[itemName].IsBool()) {
218  mdebugmode = (didirItem[itemName].GetBool());
219  } else {
220  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a boolean!", itemName);
221  return false;
222  }
223  } else {
224  mdebugmode = false;
225  }
226 
227  if (mdebugmode) {
228  StringBuffer buffer;
229  buffer.Clear();
230  PrettyWriter<StringBuffer> writer(buffer);
231  didirItem.Accept(writer);
232  LOGP(INFO, "InputDirector object: {}", std::string(buffer.GetString()));
233  }
234 
235  itemName = "fileregex";
236  if (didirItem.HasMember(itemName)) {
237  if (didirItem[itemName].IsString()) {
238  setFilenamesRegex(didirItem[itemName].GetString());
239  } else {
240  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
241  return false;
242  }
243  }
244 
245  itemName = "resfiles";
246  if (didirItem.HasMember(itemName)) {
247  if (didirItem[itemName].IsString()) {
248  fileName = didirItem[itemName].GetString();
249  if (fileName.size() && fileName[0] == '@') {
250  fileName.erase(0, 1);
251  setInputfilesFile(fileName);
252  } else {
253  setInputfilesFile("");
254  mdefaultInputFiles.emplace_back(fileName);
255  }
256  } else if (didirItem[itemName].IsArray()) {
257  setInputfilesFile("");
258  auto fns = didirItem[itemName].GetArray();
259  for (auto& fn : fns) {
260  mdefaultInputFiles.emplace_back(fn.GetString());
261  }
262  } else {
263  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string or an array!", itemName);
264  return false;
265  }
266  }
267 
268  itemName = "InputDescriptors";
269  if (didirItem.HasMember(itemName)) {
270  if (!didirItem[itemName].IsArray()) {
271  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be an array!", itemName);
272  return false;
273  }
274 
275  // loop over DataInputDescriptors
276  for (auto& didescItem : didirItem[itemName].GetArray()) {
277  if (!didescItem.IsObject()) {
278  LOGP(ERROR, "Check the JSON document! \"{}\" must be objects!", itemName);
279  return false;
280  }
281  // create a new dataInputDescriptor
282  auto didesc = new DataInputDescriptor();
283  didesc->setDefaultInputfiles(&mdefaultInputFiles);
284 
285  itemName = "table";
286  if (didescItem.HasMember(itemName)) {
287  if (didescItem[itemName].IsString()) {
288  didesc->tablename = didescItem[itemName].GetString();
289  didesc->matcher = DataDescriptorQueryBuilder::buildNode(didesc->tablename);
290  } else {
291  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
292  return false;
293  }
294  } else {
295  LOGP(ERROR, "Check the JSON document! Item \"{}\" is missing!", itemName);
296  return false;
297  }
298 
299  itemName = "treename";
300  if (didescItem.HasMember(itemName)) {
301  if (didescItem[itemName].IsString()) {
302  didesc->treename = didescItem[itemName].GetString();
303  } else {
304  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
305  return false;
306  }
307  } else {
308  auto m = DataDescriptorQueryBuilder::getTokens(didesc->tablename);
309  didesc->treename = m[2];
310  }
311 
312  itemName = "fileregex";
313  if (didescItem.HasMember(itemName)) {
314  if (didescItem[itemName].IsString()) {
315  if (didesc->getNumberInputfiles() == 0) {
316  didesc->setFilenamesRegex(didescItem[itemName].GetString());
317  }
318  } else {
319  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
320  return false;
321  }
322  } else {
323  if (didesc->getNumberInputfiles() == 0) {
324  didesc->setFilenamesRegex(mFilenameRegexPtr);
325  }
326  }
327 
328  itemName = "resfiles";
329  if (didescItem.HasMember(itemName)) {
330  if (didescItem[itemName].IsString()) {
331  fileName = didescItem[itemName].GetString();
332  if (fileName.size() && fileName[0] == '@') {
333  didesc->setInputfilesFile(fileName.erase(0, 1));
334  } else {
335  if (didesc->getFilenamesRegexString().empty() ||
336  std::regex_match(fileName, didesc->getFilenamesRegex())) {
337  didesc->addFilename(fileName);
338  }
339  }
340  } else if (didescItem[itemName].IsArray()) {
341  auto fns = didescItem[itemName].GetArray();
342  for (auto& fn : fns) {
343  if (didesc->getFilenamesRegexString().empty() ||
344  std::regex_match(fn.GetString(), didesc->getFilenamesRegex())) {
345  didesc->addFilename(fn.GetString());
346  }
347  }
348  } else {
349  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string or an array!", itemName);
350  return false;
351  }
352  } else {
353  didesc->setInputfilesFile(minputfilesFilePtr);
354  }
355 
356  // fill mfilenames and add InputDescriptor to InputDirector
357  if (didesc->fillInputfiles() > 0) {
358  mdataInputDescriptors.emplace_back(didesc);
359  } else {
360  didesc->printOut();
361  LOGP(INFO, "This DataInputDescriptor is ignored because its file list is empty!");
362  }
363  }
364  }
365 
366  // add a default DataInputDescriptor
367  createDefaultDataInputDescriptor();
368 
369  // check that all DataInputDescriptors have the same number of input files
370  if (!isValid()) {
371  printOut();
372  return false;
373  }
374 
375  // print the DataIputDirector
376  if (mdebugmode) {
377  printOut();
378  }
379 
380  return true;
381 }

457  LOGP(ERROR, "Couldn't open file \"{}\"", file->GetName());
458  }
459
