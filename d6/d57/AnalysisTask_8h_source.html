68 template <typename... PC>
69 struct WritingCursor<soa::Table<PC...>> {
70  using persistent_table_t = soa::Table<PC...>;
71  using cursor_t = decltype(std::declval<TableBuilder>().cursor<persistent_table_t>());
72 
73  template <typename... T>
74  void operator()(T... args)
75  {
76  static_assert(sizeof...(PC) == sizeof...(T), "Argument number mismatch");
77  ++mCount;
78  cursor(0, extract(args)...);
79  }
80 
82  int64_t lastIndex()
83  {
84  return mCount;
85  }
86 
87  bool resetCursor(TableBuilder& builder)
88  {
89  mBuilder = &builder;
90  cursor = std::move(FFL(builder.cursor<persistent_table_t>()));
91  mCount = -1;
92  return true;
93  }
94 
97  void reserve(int64_t size)
98  {
99  mBuilder->reserve(typename persistent_table_t::columns{}, size);
100  }
101 
102  decltype(FFL(std::declval<cursor_t>())) cursor;
103 
104  private:
105  template <typename T>
106  static decltype(auto) extract(T const& arg)
107  {
108  if constexpr (soa::is_soa_iterator_t<T>::value) {
109  return arg.globalIndex();
110  } else {
111  static_assert(!framework::has_type_v<T, framework::pack<PC...>>, "Argument type mismatch");
112  return arg;
113  }
114  }
115 
119  TableBuilder* mBuilder = nullptr;
120  int64_t mCount = -1;
121 };

351  template <typename T, size_t At>
352  static auto extractSomethingFromRecord(InputRecord& record, std::vector<ExpressionInfo> const infos)
353  {
354  using decayed = std::decay_t<T>;
355 
356  if constexpr (soa::is_soa_filtered_t<decayed>::value) {
357  for (auto& info : infos) {
358  if (info.index == At)
359  return extractFilteredFromRecord<decayed>(record, info, soa::make_originals_from_type<decayed>());
360  }
361  } else if constexpr (soa::is_soa_iterator_t<decayed>::value) {
362  if constexpr (std::is_same_v<typename decayed::policy_t, soa::FilteredIndexPolicy>) {
363  for (auto& info : infos) {
364  if (info.index == At)
365  return extractFilteredFromRecord<decayed>(record, info, soa::make_originals_from_type<decayed>());
366  }
367  } else {
368  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
369  }
370  } else {
371  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
372  }
373  O2_BUILTIN_UNREACHABLE();
374  }

392  template <typename G, typename... A>
393  struct GroupSlicer {
394  using grouping_t = std::decay_t<G>;
395  GroupSlicer(G& gt, std::tuple<A...>& at)
396  : max{gt.size()},
397  mBegin{GroupSlicerIterator(gt, at)}
398  {

401  struct GroupSlicerSentinel {
402  int64_t position;
403  };
404 
405  struct GroupSlicerIterator {
406  using associated_pack_t = framework::pack<A...>;
407 
408  GroupSlicerIterator() = default;
409  GroupSlicerIterator(GroupSlicerIterator const&) = default;
410  GroupSlicerIterator(GroupSlicerIterator&&) = default;
411  GroupSlicerIterator& operator=(GroupSlicerIterator const&) = default;
412  GroupSlicerIterator& operator=(GroupSlicerIterator&&) = default;
413 
414  auto getLabelFromType()
415  {
416  if constexpr (soa::is_type_with_originals_v<std::decay_t<G>>) {
417  using T = typename framework::pack_element_t<0, typename std::decay_t<G>::originals>;
418  using groupingMetadata = typename aod::MetadataTrait<T>::metadata;
419  return std::string("f") + groupingMetadata::tableLabel() + "ID";
420  } else {
421  using groupingMetadata = typename aod::MetadataTrait<std::decay_t<G>>::metadata;
422  return std::string("f") + groupingMetadata::tableLabel() + "ID";
423  }
424  }
425 
426  GroupSlicerIterator(G& gt, std::tuple<A...>& at)
427  : mAt{&at},
428  mGroupingElement{gt.begin()},
429  position{0}
430  {
431  auto indexColumnName = getLabelFromType();
432  arrow::compute::FunctionContext ctx;
436  auto splitter = [&](auto&& x) {
437  using xt = std::decay_t<decltype(x)>;
438  constexpr auto index = framework::has_type_at<std::decay_t<decltype(x)>>(associated_pack_t{});
439  if (hasIndexTo<std::decay_t<G>>(typename xt::persistent_columns_t{})) {
440  auto result = o2::framework::sliceByColumn(&ctx, indexColumnName,
441  static_cast<int32_t>(gt.size()),
442  x.asArrowTable(),
443  &groups[index],
444  &offsets[index]);
445  if (result.ok() == false) {
446  throw std::runtime_error("Cannot split collection");
447  }
448  if (groups[index].size() != gt.size()) {
449  throw std::runtime_error("Splitting collection resulted in different group number than there is rows in the grouping table.");
450  };
451  }
452  };
453 
454  std::apply(
455  [&](auto&&... x) -> void {
456  (splitter(x), ...);
457  },
458  at);
460  auto extractor = [&](auto&& x) {
461  using xt = std::decay_t<decltype(x)>;
462  if constexpr (soa::is_soa_filtered_t<xt>::value) {
463  constexpr auto index = framework::has_type_at<std::decay_t<decltype(x)>>(associated_pack_t{});
464  selections[index] = &x.getSelectedRows();
465  starts[index] = selections[index]->begin();
466  offsets[index].push_back(std::get<xt>(at).tableSize());
467  }
468  };
469  std::apply(
470  [&](auto&&... x) -> void {
471  (extractor(x), ...);
472  },
473  at);
474  }

476  template <typename B, typename... C>

482  template <typename B, typename C>
483  constexpr bool isIndexTo()
484  {
485  if constexpr (soa::is_type_with_binding_v<C>) {
486  if constexpr (soa::is_type_with_originals_v<std::decay_t<B>>) {
487  using TT = typename framework::pack_element_t<0, typename std::decay_t<B>::originals>;
488  return std::is_same_v<typename C::binding_t, TT>;
489  } else {
490  using TT = std::decay_t<B>;
491  return std::is_same_v<typename C::binding_t, TT>;
492  }
493  }
494  return false;
495  }

529  template <typename A1>
530  auto prepareArgument()
531  {
532  constexpr auto index = framework::has_type_at<A1>(associated_pack_t{});
533  if (hasIndexTo<G>(typename std::decay_t<A1>::persistent_columns_t{})) {
534  if constexpr (soa::is_soa_filtered_t<std::decay_t<A1>>::value) {
535  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[position]).value);
536 
537  // for each grouping element we need to slice the selection vector
538  auto start_iterator = std::lower_bound(starts[index], selections[index]->end(), (offsets[index])[position]);
539  auto stop_iterator = std::lower_bound(start_iterator, selections[index]->end(), (offsets[index])[position + 1]);
540  starts[index] = stop_iterator;
541  soa::SelectionVector slicedSelection{start_iterator, stop_iterator};
542  std::transform(slicedSelection.begin(), slicedSelection.end(), slicedSelection.begin(),
543  [&](int64_t idx) {
544  return idx - static_cast<int64_t>((offsets[index])[position]);
545  });
546 
547  std::decay_t<A1> typedTable{{groupedElementsTable}, std::move(slicedSelection), (offsets[index])[position]};
548  return typedTable;
549  } else {
550  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[position]).value);
551  std::decay_t<A1> typedTable{{groupedElementsTable}, (offsets[index])[position]};
552  return typedTable;
553  }
554  } else {
555  return std::get<A1>(*mAt);
556  }
557  O2_BUILTIN_UNREACHABLE();
558  }
559 
560  std::tuple<A...>* mAt;
561  typename grouping_t::iterator mGroupingElement;
562  uint64_t position = 0;
563 
564  std::array<std::vector<arrow::compute::Datum>, sizeof...(A)> groups;
565  std::array<std::vector<uint64_t>, sizeof...(A)> offsets;
566  std::array<soa::SelectionVector const*, sizeof...(A)> selections;
567  std::array<soa::SelectionVector::const_iterator, sizeof...(A)> starts;
568  };

583  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
584  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*)(Grouping, Associated...), std::vector<ExpressionInfo> const& infos)
585  {
586  using G = std::decay_t<Grouping>;
587  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, &C::process, infos);
588  if constexpr (sizeof...(Associated) == 0) {
589  // single argument to process
590  if constexpr (soa::is_soa_iterator_t<G>::value) {
591  for (auto& element : groupingTable) {
592  task.process(*element);
593  }
594  } else {
595  static_assert(soa::is_soa_table_like_t<G>::value,
596  "Single argument of process() should be a table-like or an iterator");
597  task.process(groupingTable);
598  }
599  } else {
600  // multiple arguments to process
601  static_assert(((soa::is_soa_iterator_t<std::decay_t<Associated>>::value == false) && ...),
602  "Associated arguments of process() should not be iterators");
603  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, &C::process, infos);
604  auto binder = [&](auto&& x) {
605  x.bindExternalIndices(&groupingTable, &std::get<std::decay_t<Associated>>(associatedTables)...);
606  };
607  groupingTable.bindExternalIndices(&std::get<std::decay_t<Associated>>(associatedTables)...);
608 
609  if constexpr (soa::is_soa_iterator_t<std::decay_t<G>>::value) {
610  // grouping case
611  auto slicer = GroupSlicer(groupingTable, associatedTables);
612  for (auto& slice : slicer) {
613  auto associatedSlices = slice.associatedTables();
614  std::apply(
615  [&](auto&&... x) {
616  (binder(x), ...);
617  },
618  associatedSlices);
619 
620  invokeProcessWithArgs(task, slice.groupingElement(), associatedSlices);
621  }
622  } else {
623  // non-grouping case
624  std::apply(
625  [&](auto&&... x) {
626  (binder(x), ...);
627  },
628  associatedTables);
629 
630  invokeProcessWithArgs(task, groupingTable, associatedTables);
631  }
632  }
633  }
634 
635  template <typename T, typename G, typename... A>
636  static void invokeProcessWithArgs(T& task, G g, std::tuple<A...>& at)
637  {
638  task.process(g, std::get<A>(at)...);
639  }
640 };

851 template <typename T, typename... Args>
852 DataProcessorSpec adaptAnalysisTask(char const* name, Args&&... args)
853 {
854  TH1::AddDirectory(false);
855  auto task = std::make_shared<T>(std::forward<Args>(args)...);
856  auto hash = compile_time_hash(name);
857 
858  std::vector<OutputSpec> outputs;
859  std::vector<ConfigParamSpec> options;
860 
861  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
862  static_assert(has_process<T>::value || has_run<T>::value || has_init<T>::value,
863  "At least one of process(...), T::run(...), init(...) must be defined");
864 
865  std::vector<InputSpec> inputs;
866  std::vector<ExpressionInfo> expressionInfos;
867 
868  if constexpr (has_process<T>::value) {
869  // this pushes (I,schemaPtr,nullptr) into expressionInfos for arguments that are Filtered/filtered_iterators
870  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, inputs, expressionInfos);
871  // here the FilterManager will prepare the gandiva trees matched to schemas and put the pointers into expressionInfos
872  std::apply([&expressionInfos](auto&... x) {
873  return (FilterManager<std::decay_t<decltype(x)>>::createExpressionTrees(x, expressionInfos), ...);
874  },
875  tupledTask);
876  }
877 
878  std::apply([&outputs, &hash](auto&... x) { return (OutputManager<std::decay_t<decltype(x)>>::appendOutput(outputs, x, hash), ...); }, tupledTask);
879  std::apply([&options, &hash](auto&... x) { return (OptionManager<std::decay_t<decltype(x)>>::appendOption(options, x), ...); }, tupledTask);
880 
881  auto algo = AlgorithmSpec::InitCallback{[task, expressionInfos](InitContext& ic) {
882  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
883  std::apply([&ic](auto&&... x) { return (OptionManager<std::decay_t<decltype(x)>>::prepare(ic, x), ...); }, tupledTask);
884 
885  auto& callbacks = ic.services().get<CallbackService>();
886  auto endofdatacb = [task](EndOfStreamContext& eosContext) {
887  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
888  std::apply([&eosContext](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::postRun(eosContext, x), ...); }, tupledTask);
889  eosContext.services().get<ControlService>().readyToQuit(QuitRequest::Me);
890  };
891  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
892 
893  if constexpr (has_init<T>::value) {
894  task->init(ic);
895  }
896  return [task, expressionInfos](ProcessingContext& pc) {
897  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
898  std::apply([&pc](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::prepare(pc, x), ...); }, tupledTask);
899  if constexpr (has_run<T>::value) {
900  task->run(pc);
901  }
902  if constexpr (has_process<T>::value) {
903  AnalysisDataProcessorBuilder::invokeProcess(*(task.get()), pc.inputs(), &T::process, expressionInfos);
904  }
905  std::apply([&pc](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::finalize(pc, x), ...); }, tupledTask);
906  };
907  }};
908 
909  DataProcessorSpec spec{
910  name,
911  // FIXME: For the moment we hardcode this. We could build
912  // this list from the list of methods actually implemented in the
913  // task itself.
914  inputs,
915  outputs,
916  algo,
917  options};
918  return spec;
919 }
