40 class ClusterizerDeviceDPL

43  void init(o2::framework::InitContext& ic)
44  {
45  if (!mPreClusterizer.init()) {
46  LOG(ERROR) << "Initialization of MID pre-clusterizer device failed";

48 
49  if (!mClusterizer.init()) {
50  LOG(ERROR) << "Initialization of MID clusterizer device failed";

52 
53  auto stop = [this]() {
54  LOG(INFO) << "Capacities: ROFRecords: " << mClusterizer.getROFRecords().capacity() << " preclusters: " << mPreClusterizer.getPreClusters().capacity() << " clusters: " << mClusterizer.getClusters().capacity();
55  double scaleFactor = 1.e6 / mNROFs;
56  LOG(INFO) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us pre-clustering: " << mTimerPreCluster.count() * scaleFactor << " us clustering: " << mTimerCluster.count() * scaleFactor << " us";
57  };
58  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
59  }

61  void run(o2::framework::ProcessingContext& pc)
62  {
63  auto tStart = std::chrono::high_resolution_clock::now();
64 
65  auto msg = pc.inputs().get("mid_data");
66  gsl::span<const ColumnData> patterns = of::DataRefUtils::as<const ColumnData>(msg);
67 
68  auto msgROF = pc.inputs().get("mid_data_rof");
69  gsl::span<const ROFRecord> inROFRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
70 
71  // Pre-clustering
72  auto tAlgoStart = std::chrono::high_resolution_clock::now();
73  mPreClusterizer.process(patterns, inROFRecords);
74  mTimerPreCluster += std::chrono::high_resolution_clock::now() - tAlgoStart;
75  LOG(DEBUG) << "Generated " << mPreClusterizer.getPreClusters().size() << " PreClusters";
76 
77  // Clustering
78  tAlgoStart = std::chrono::high_resolution_clock::now();
79  mClusterizer.process(mPreClusterizer.getPreClusters(), mPreClusterizer.getROFRecords());
80  mTimerCluster += std::chrono::high_resolution_clock::now() - tAlgoStart;
81 
82  pc.outputs().snapshot(of::Output{"MID", "CLUSTERS", 0, of::Lifetime::Timeframe}, mClusterizer.getClusters());
83  LOG(DEBUG) << "Sent " << mClusterizer.getClusters().size() << " clusters";
84  pc.outputs().snapshot(of::Output{"MID", "CLUSTERSROF", 0, of::Lifetime::Timeframe}, mClusterizer.getROFRecords());
85  LOG(DEBUG) << "Sent " << mClusterizer.getROFRecords().size() << " ROF";
86 
87  mTimer += std::chrono::high_resolution_clock::now() - tStart;
88  mNROFs += inROFRecords.size();
89  }
90 
91  private:
92  PreClusterizer mPreClusterizer{};
93  Clusterizer mClusterizer{};
94  std::chrono::duration<double> mTimer{0};
95  std::chrono::duration<double> mTimerPreCluster{0};
96  std::chrono::duration<double> mTimerCluster{0};
97  unsigned long mNROFs{0};
98 };
