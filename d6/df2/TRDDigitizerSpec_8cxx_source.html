51  void run(framework::ProcessingContext& pc)
52  {
53  static bool finished = false;
54  if (finished) {
55  return;
56  }
57  LOG(INFO) << "Doing TRD digitization";
58 
59  bool mctruth = pc.outputs().isAllowed({"TRD", "LABELS", 0});
60 
61  Calibrations simcal;
62  simcal.setCCDBForSimulation(297595);
63  mDigitizer.setCalibrations(&simcal);
64 
65  // read collision context from input
66  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
67  context->initSimChains(o2::detectors::DetID::TRD, mSimChains);
68  auto& irecords = context->getEventRecords();
69 
70  for (auto& record : irecords) {
71  LOG(INFO) << "TRD TIME RECEIVED " << record.timeNS;
72  }
73 
74  auto& eventParts = context->getEventParts();
75  std::vector<o2::trd::Digit> digitsAccum; // accumulator for digits
76  o2::dataformats::MCTruthContainer<o2::trd::MCLabel> labelsAccum;
77  std::vector<TriggerRecord> triggers;
78 
79  std::vector<o2::trd::Digit> digits; // digits which get filled
80  o2::dataformats::MCTruthContainer<o2::trd::MCLabel> labels; // labels which get filled
81 
82  TStopwatch timer;
83  timer.Start();
84 
85  // loop over all composite collisions given from context
86  // (aka loop over all the interaction records)
87  for (int collID = 0; collID < irecords.size(); ++collID) {
88  mDigitizer.setEventTime(irecords[collID].timeNS);
89 
90  // for each collision, loop over the constituents event and source IDs
91  // (background signal merging is basically taking place here)
92  for (auto& part : eventParts[collID]) {
93  mDigitizer.setEventID(part.entryID);
94  mDigitizer.setSrcID(part.sourceID);
95 
96  // get the hits for this event and this source
97  std::vector<o2::trd::HitType> hits;
98  context->retrieveHits(mSimChains, "TRDHit", part.sourceID, part.entryID, &hits);
99  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found TRD " << hits.size() << " hits ";
100 
101  mDigitizer.process(hits, digits, labels);
102 
103  // Add trigger record
104  triggers.emplace_back(irecords[collID], digitsAccum.size(), digits.size());
105 
106  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
107  if (mctruth) {
108  labelsAccum.mergeAtBack(labels);
109  }
110  digits.clear();
111  labels.clear();
112  }
113  }
114 
115  // Force flush of the digits that remain in the digitizer cache
116  mDigitizer.flush(digits, labels);
117  triggers.emplace_back(irecords[irecords.size() - 1], digitsAccum.size(), digits.size());
118  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
119  if (mctruth) {
120  labelsAccum.mergeAtBack(labels);
121  }
122  timer.Stop();
123  LOG(INFO) << "TRD: Digitization took " << timer.RealTime() << "s";
124 
125  LOG(INFO) << "TRD: Sending " << digitsAccum.size() << " digits";
126  pc.outputs().snapshot(Output{"TRD", "DIGITS", 0, Lifetime::Timeframe}, digitsAccum);
127  if (mctruth) {
128  LOG(INFO) << "TRD: Sending " << labelsAccum.getNElements() << " labels";
129  pc.outputs().snapshot(Output{"TRD", "LABELS", 0, Lifetime::Timeframe}, labelsAccum);
130  }
131  LOG(INFO) << "TRD: Sending ROMode= " << mROMode << " to GRPUpdater";
132  pc.outputs().snapshot(Output{"TRD", "ROMode", 0, Lifetime::Timeframe}, mROMode);
133  LOG(INFO) << "TRD: Sending trigger records";
134  pc.outputs().snapshot(Output{"TRD", "TRGRDIG", 0, Lifetime::Timeframe}, triggers);
135 
136  // we should be only called once; tell DPL that this process is ready to exit
137  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
138  finished = true;
139  }
140 
141  private:
142  Digitizer mDigitizer;
143  std::vector<TChain*> mSimChains;
144  // RS: at the moment using hardcoded flag for continuos readout
145  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
146 };
