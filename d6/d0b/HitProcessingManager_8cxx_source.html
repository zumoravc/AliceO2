27 bool HitProcessingManager::setupChain()
28 {
29  if (mBackgroundFileNames.size() == 0 && mSignalFileNames.size() == 0) {
30  // there are no files to be analysed/processed;
31  LOG(WARNING) << "No files to be analysed";
32  return false;
33  }
34 
35  // count max signal id
36  int sourcecounter = 0;
37  int maxsourceid = 0;
38  for (auto& pair : mSignalFileNames) {
39  sourcecounter++;
40  maxsourceid = std::max(pair.first, maxsourceid);
41  }
42  if (maxsourceid != sourcecounter) {
43  LOG(WARNING) << "max source id " << maxsourceid << " vs " << sourcecounter;
44  }
45  LOG(INFO) << "setting up " << maxsourceid + 1 << " chains";
46  mSimChains.resize(maxsourceid + 1);
47 
48  // allocate chains
49  for (int i = 0; i < mSimChains.size(); ++i) {
50  // make o2sim a parameter
51  mSimChains[i] = new TChain("o2sim");
52  }
53 
54  // background chain
55  auto& c = *mSimChains[0];
56  c.Reset();
57  for (auto& filename : mBackgroundFileNames) {
58  c.AddFile(o2::base::NameConf::getMCKinematicsFileName(filename.data()).c_str());
59  }
60 
61  for (auto& pair : mSignalFileNames) {
62  const auto& signalid = pair.first;
63  const auto& filenamevector = pair.second;
64  auto& chain = *mSimChains[signalid];
65  for (auto& filename : filenamevector) {
66  chain.AddFile(o2::base::NameConf::getMCKinematicsFileName(filename.data()).c_str());
67  }
68  }
69 
70  return true;
71 }

73 void HitProcessingManager::setupRun(int ncollisions)
74 {

76  return;
77  }

79  // load geometry

82 
83  //
84  if (ncollisions != -1) {
85  mNumberOfCollisions = ncollisions;

87  mNumberOfCollisions = mSimChains[0]->GetEntries();
88  LOG(INFO) << "Automatic deduction of number of collisions ... will just take number of background entries "

91  mDigitizationContext.setNCollisions(mNumberOfCollisions);
92  sampleCollisionTimes();

131 void HitProcessingManager::sampleCollisionConstituents()
132 {
133  TRandom3 rnd(0); // we don't use the global to be in isolation
134  auto getBackgroundRoundRobin = [this]() {
135  static int bgcounter = 0;
136  int numbg = mSimChains[0]->GetEntries();
137  if (bgcounter == numbg) {
138  bgcounter = 0;
139  }
140  return EventPart(0, bgcounter++);
141  };
142 
143  const int nsignalids = mSimChains.size() - 1;
144  auto getSignalRoundRobin = [this, nsignalids]() {
145  static int bgcounter = 0;
146  static int signalid = 0;
147  static std::vector<int> counter(nsignalids, 0);
148  if (signalid == nsignalids) {
149  signalid = 0;
150  }
151  const auto realsourceid = signalid + 1;
152  int numentries = mSimChains[realsourceid]->GetEntries();
153  if (counter[signalid] == numentries) {
154  counter[signalid] = 0;
155  }
156  EventPart e(realsourceid, counter[signalid]);
157  counter[signalid]++;
158  signalid++;
159  return e;
160  };
161 
162  auto getRandomBackground = [this, &rnd]() {
163  int numbg = mSimChains[0]->GetEntries();
164  const auto eventID = (int)numbg * rnd.Rndm();
165  return EventPart(0, eventID);
166  };
167 
168  auto getRandomSignal = [this, nsignalids, &rnd]() {
169  const auto sourceID = 1 + (int)(rnd.Rndm() * nsignalids);
170  const auto signalID = (int)(rnd.Rndm() * mSimChains[sourceID]->GetEntries());
171  return EventPart(sourceID, signalID);
172  };
173 
174  // we fill mDigitizationContext.mEventParts
175  auto& eventparts = mDigitizationContext.getEventParts();
176  eventparts.clear();
177  eventparts.resize(mDigitizationContext.getEventRecords().size());
178  for (int i = 0; i < mDigitizationContext.getEventRecords().size(); ++i) {
179  eventparts[i].clear();
180  // NOTE: THIS PART WOULD BENEFIT FROM A MAJOR REDESIGN
181  // WISHFUL ITEMS WOULD BE:
182  // - ALLOW COLLISION ENGINEERING FROM OUTSIDE (give wanted sequence as file)
183  // * the outside person can decide what kind of sampling and sequence to use
184  // - CHECK IF VERTEX IS CONSISTENT
185  if (mSampleCollisionsRandomly) {
186  eventparts[i].emplace_back(getRandomBackground());
187  if (mSimChains.size() > 1) {
188  eventparts[i].emplace_back(getRandomSignal());
189  }
190  } else {
191  // push any number of constituents?
192  // for the moment just 2 : one background and one signal
193  eventparts[i].emplace_back(getBackgroundRoundRobin());
194  if (mSimChains.size() > 1) {
195  eventparts[i].emplace_back(getSignalRoundRobin());
196  }
197  }
198  }
199 
200  // push any number of constituents?
201  // for the moment just max 2 : one background and one signal
202  mDigitizationContext.setMaxNumberParts(1);
203  if (mSimChains.size() > 1) {
204  mDigitizationContext.setMaxNumberParts(2);
205  }
206 
207  mDigitizationContext.printCollisionSummary();
208 }
