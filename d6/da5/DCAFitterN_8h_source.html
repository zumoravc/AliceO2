30 struct TrackCovI {
31  float sxx, syy, syz, szz;
32 
33  TrackCovI(const o2::track::TrackParCov& trc) { set(trc); }
34 
35  TrackCovI() = default;
36 
37  void set(const o2::track::TrackParCov& trc)
38  {
39  // we assign Y error to X for DCA calculation
40  // (otherwise for quazi-collinear tracks the X will not be constrained)
41  float cyy = trc.getSigmaY2(), czz = trc.getSigmaZ2(), cyz = trc.getSigmaZY(), cxx = cyy;
42  float detYZ = cyy * czz - cyz * cyz;
43  if (detYZ > 0.) {
44  auto detYZI = 1. / detYZ;
45  sxx = 1. / cxx;
46  syy = czz * detYZI;
47  syz = -cyz * detYZI;
48  szz = cyy * detYZI;
49  } else {
50  throw std::runtime_error("invalid track covariance");
51  }
52  }
53 };

57 struct TrackDeriv {
58  float dydx, dzdx, d2ydx2, d2zdx2;
59  TrackDeriv() = default;
60  TrackDeriv(const o2::track::TrackPar& trc, float bz) { set(trc, bz); }
61  void set(const o2::track::TrackPar& trc, float bz)
62  {
63  float snp = trc.getSnp(), csp = std::sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz) * cspI;
64  dydx = snp * cspI; // = snp/csp
65  dzdx = trc.getTgl() * cspI; // = tgl/csp
66  d2ydx2 = crv2c * cspI * cspI; // = crv/csp^3
67  d2zdx2 = crv2c * dzdx * dydx; // = crv*tgl*snp/csp^3
68  }
69 };

250 template <int N, typename... Args>
251 template <class... Tr>
252 int DCAFitterN<N, Args...>::process(const Tr&... args)
253 {
254  // This is a main entry point: fit PCA of N tracks
255  static_assert(sizeof...(args) == N, "incorrect number of input tracks");
256  assign(0, args...);
257  clear();
258  for (int i = 0; i < N; i++) {
259  mTrAux[i].set(*mOrigTrPtr[i], mBz);
260  }
261  if (!mCrossings.set(mTrAux[0], *mOrigTrPtr[0], mTrAux[1], *mOrigTrPtr[1])) { // even for N>2 it should be enough to test just 1 loop
262  return 0; // no crossing
263  }
264  if (mUseAbsDCA) {
265  calcRMatrices(); // needed for fast residuals derivatives calculation in case of abs. distance minimization
266  }
267  if (mCrossings.nDCA == MAXHYP) { // if there are 2 candidates and they are too close, chose their mean as a starting point
268  auto dst2 = (mCrossings.xDCA[0] - mCrossings.xDCA[1]) * (mCrossings.xDCA[0] - mCrossings.xDCA[1]) +
269  (mCrossings.yDCA[0] - mCrossings.yDCA[1]) * (mCrossings.yDCA[0] - mCrossings.yDCA[1]);
270  if (dst2 < mMaxDist2ToMergeSeeds) {
271  mCrossings.nDCA = 1;
272  mCrossings.xDCA[0] = 0.5 * (mCrossings.xDCA[0] + mCrossings.xDCA[1]);
273  mCrossings.yDCA[0] = 0.5 * (mCrossings.yDCA[0] + mCrossings.yDCA[1]);
274  }
275  }
276  // check all crossings
277  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
278  // check if radius is acceptable
279  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
280  continue;
281  }
282  mCrossIDCur = ic;
283  mCrossIDAlt = (mCrossings.nDCA == 2 && mAllowAltPreference) ? 1 - ic : -1; // works for max 2 crossings
284  mNIters[mCurHyp] = 0;
285  mTrPropDone[mCurHyp] = false;
286  mChi2[mCurHyp] = -1.;
287  mPCA[mCurHyp][0] = mCrossings.xDCA[ic];
288  mPCA[mCurHyp][1] = mCrossings.yDCA[ic];
289 
290  if (mUseAbsDCA ? minimizeChi2NoErr() : minimizeChi2()) {
291  mOrder[mCurHyp] = mCurHyp;
292  if (mPropagateToPCA && !propagateTracksToVertex(mCurHyp)) {
293  return false;
294  }
295  mCurHyp++;
296  }
297  }
298 
299  for (int i = mCurHyp; i--;) { // order in quality
300  for (int j = i; j--;) {
301  if (mChi2[mOrder[i]] < mChi2[mOrder[j]]) {
302  std::swap(mOrder[i], mOrder[j]);
303  }
304  }
305  }
306  return mCurHyp;
307 }

310 template <int N, typename... Args>
311 bool DCAFitterN<N, Args...>::calcPCACoefs()
312 {
313  //< calculate Ti matrices for global vertex decomposition to V = sum_{0<i<N} Ti pi, see EQ.T in the ref
314  if (!calcInverseWeight()) {
315  return false;
316  }
317  for (int i = N; i--;) { // build Mi*Ei matrix
318  const auto& taux = mTrAux[i];
319  const auto& tcov = mTrcEInv[mCurHyp][i];
320  MatStd3D miei;
321  miei[0][0] = taux.c * tcov.sxx;
322  miei[0][1] = -taux.s * tcov.syy;
323  miei[0][2] = -taux.s * tcov.syz;
324  miei[1][0] = taux.s * tcov.sxx;
325  miei[1][1] = taux.c * tcov.syy;
326  miei[1][2] = taux.c * tcov.syz;
327  //miei[2][0] = 0;
328  miei[2][1] = tcov.syz;
329  miei[2][2] = tcov.szz;
330  mTrCFVT[mCurHyp][i] = mWeightInv * miei;
331  }
332  return true;
333 }

336 template <int N, typename... Args>
337 bool DCAFitterN<N, Args...>::calcInverseWeight()
338 {
339  //< calculate [sum_{0<j<N} M_j*E_j*M_j^T]^-1 used for Ti matrices, see EQ.T
340  auto* arrmat = mWeightInv.Array();
341  memset(arrmat, 0, sizeof(mWeightInv));
342  enum { XX,
343  XY,
344  YY,
345  XZ,
346  YZ,
347  ZZ };
348  for (int i = N; i--;) {
349  const auto& taux = mTrAux[i];
350  const auto& tcov = mTrcEInv[mCurHyp][i];
351  arrmat[XX] += taux.cc * tcov.sxx + taux.ss * tcov.syy;
352  arrmat[XY] += taux.cs * (tcov.sxx - tcov.syy);
353  arrmat[XZ] += -taux.s * tcov.syz;
354  arrmat[YY] += taux.cc * tcov.syy + taux.ss * tcov.sxx;
355  arrmat[YZ] += taux.c * tcov.syz;
356  arrmat[ZZ] += tcov.szz;
357  }
358  // invert 3x3 symmetrix matrix
359  return mWeightInv.Invert();
360 }

363 template <int N, typename... Args>
364 void DCAFitterN<N, Args...>::calcResidDerivatives()
365 {
366  //< calculate matrix of derivatives for weighted chi2: residual i vs parameter X of track j
367  MatStd3D matMT;
368  for (int i = N; i--;) { // residual being differentiated
369  const auto& taux = mTrAux[i];
370  for (int j = N; j--;) { // track over which we differentiate
371  const auto& matT = mTrCFVT[mCurHyp][j]; // coefficient matrix for track J
372  const auto& trDx = mTrDer[mCurHyp][j]; // track point derivs over track X param
373  auto& dr1 = mDResidDx[i][j];
374  auto& dr2 = mD2ResidDx2[i][j];
375  // calculate M_i^tr * T_j
376  matMT[0][0] = taux.c * matT[0][0] + taux.s * matT[1][0];
377  matMT[0][1] = taux.c * matT[0][1] + taux.s * matT[1][1];
378  matMT[0][2] = taux.c * matT[0][2] + taux.s * matT[1][2];
379  matMT[1][0] = -taux.s * matT[0][0] + taux.c * matT[1][0];
380  matMT[1][1] = -taux.s * matT[0][1] + taux.c * matT[1][1];
381  matMT[1][2] = -taux.s * matT[0][2] + taux.c * matT[1][2];
382  matMT[2][0] = matT[2][0];
383  matMT[2][1] = matT[2][1];
384  matMT[2][2] = matT[2][2];
385 
386  // calculate DResid_i/Dx_j = (delta_ij - M_i^tr * T_j) * DTrack_k/Dx_k
387  dr1[0] = -(matMT[0][0] + matMT[0][1] * trDx.dydx + matMT[0][2] * trDx.dzdx);
388  dr1[1] = -(matMT[1][0] + matMT[1][1] * trDx.dydx + matMT[1][2] * trDx.dzdx);
389  dr1[2] = -(matMT[2][0] + matMT[2][1] * trDx.dydx + matMT[2][2] * trDx.dzdx);
390 
391  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ijk - M_i^tr * T_j * delta_jk) * D2Track_k/dx_k^2
392  dr2[0] = -(matMT[0][1] * trDx.d2ydx2 + matMT[0][2] * trDx.d2zdx2);
393  dr2[1] = -(matMT[1][1] * trDx.d2ydx2 + matMT[1][2] * trDx.d2zdx2);
394  dr2[2] = -(matMT[2][1] * trDx.d2ydx2 + matMT[2][2] * trDx.d2zdx2);
395 
396  if (i == j) {
397  dr1[0] += 1.;
398  dr1[1] += trDx.dydx;
399  dr1[2] += trDx.dzdx;
400 
401  dr2[1] += trDx.d2ydx2;
402  dr2[2] += trDx.d2zdx2;
403  }
404  } // track over which we differentiate
405  } // residual being differentiated
406 }

409 template <int N, typename... Args>
410 void DCAFitterN<N, Args...>::calcResidDerivativesNoErr()
411 {
412  //< calculate matrix of derivatives for absolute distance chi2: residual i vs parameter X of track j
413  constexpr double NInv1 = 1. - NInv; // profit from Rii = I/Ninv
414  for (int i = N; i--;) { // residual being differentiated
415  const auto& trDxi = mTrDer[mCurHyp][i]; // track point derivs over track X param
416  auto& dr1ii = mDResidDx[i][i];
417  auto& dr2ii = mD2ResidDx2[i][i];
418  dr1ii[0] = NInv1;
419  dr1ii[1] = NInv1 * trDxi.dydx;
420  dr1ii[2] = NInv1 * trDxi.dzdx;
421 
422  dr2ii[0] = 0;
423  dr2ii[1] = NInv1 * trDxi.d2ydx2;
424  dr2ii[2] = NInv1 * trDxi.d2zdx2;
425 
426  for (int j = i; j--;) { // track over which we differentiate
427  auto& dr1ij = mDResidDx[i][j];
428  auto& dr1ji = mDResidDx[j][i];
429  const auto& trDxj = mTrDer[mCurHyp][j]; // track point derivs over track X param
430  auto cij = mCosDif[i][j], sij = mSinDif[i][j]; // M_i^T*M_j / N matrices non-trivial elements = {ci*cj+si*sj , si*cj-ci*sj }, see 5 in ref.
431 
432  // calculate DResid_i/Dx_j = (delta_ij - R_ij) * DTrack_j/Dx_j for j<i
433  dr1ij[0] = -(cij + sij * trDxj.dydx);
434  dr1ij[1] = -(-sij + cij * trDxj.dydx);
435  dr1ij[2] = -trDxj.dzdx * NInv;
436 
437  // calculate DResid_j/Dx_i = (delta_ij - R_ji) * DTrack_i/Dx_i for j<i
438  dr1ji[0] = -(cij - sij * trDxi.dydx);
439  dr1ji[1] = -(sij + cij * trDxi.dydx);
440  dr1ji[2] = -trDxi.dzdx * NInv;
441 
442  auto& dr2ij = mD2ResidDx2[i][j];
443  auto& dr2ji = mD2ResidDx2[j][i];
444  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ij - Rij) * D2Track_j/dx_j^2 * delta_jk for j<i
445  dr2ij[0] = -sij * trDxj.d2ydx2;
446  dr2ij[1] = -cij * trDxj.d2ydx2;
447  dr2ij[2] = -trDxj.d2zdx2 * NInv;
448 
449  // calculate D2Resid_j/(Dx_i Dx_k) = (delta_ij - Rji) * D2Track_i/dx_i^2 * delta_ik for j<i
450  dr2ji[0] = sij * trDxi.d2ydx2;
451  dr2ji[1] = -cij * trDxi.d2ydx2;
452  dr2ji[2] = -trDxi.d2zdx2 * NInv;
453 
454  } // track over which we differentiate
455  } // residual being differentiated
456 }

459 template <int N, typename... Args>
460 void DCAFitterN<N, Args...>::calcRMatrices()
461 {
462  //< calculate Rij = 1/N M_i^T * M_j matrices (rotation from j-th track to i-th track frame)
463  for (int i = N; i--;) {
464  const auto& mi = mTrAux[i];
465  for (int j = i; j--;) {
466  const auto& mj = mTrAux[j];
467  mCosDif[i][j] = (mi.c * mj.c + mi.s * mj.s) * NInv; // cos(alp_i-alp_j) / N
468  mSinDif[i][j] = (mi.s * mj.c - mi.c * mj.s) * NInv; // sin(alp_i-alp_j) / N
469  }
470  }
471 }

475 void DCAFitterN<N, Args...>::calcChi2Derivatives()
476 {
477  //< calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters X, see EQ.Chi2 in the ref
478  std::array<std::array<Vec3D, N>, N> covIDrDx; // tempory vectors of covI_j * dres_j/dx_i
479 
480  // chi2 1st derivative
481  for (int i = N; i--;) {
482  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * covI_j * Dres_j/Dx_i }
483  dchi1 = 0;
484  for (int j = N; j--;) {
485  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
486  const auto& covI = mTrcEInv[mCurHyp][j]; // inverse cov matrix of track j
487  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
488  auto& cidr = covIDrDx[i][j]; // vector covI_j * dres_j/dx_i, save for 2nd derivative calculation
489  cidr[0] = covI.sxx * dr1[0];
490  cidr[1] = covI.syy * dr1[1] + covI.syz * dr1[2];
491  cidr[2] = covI.syz * dr1[1] + covI.szz * dr1[2];
492  // calculate res_i * covI_j * dres_j/dx_i
493  dchi1 += ROOT::Math::Dot(res, cidr);
494  }
495  }
496  // chi2 2nd derivative
497  for (int i = N; i--;) {
498  for (int j = i + 1; j--;) { // symmetric matrix
499  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
500  dchi2 = 0;
501  for (int k = N; k--;) {
502  const auto& dr1j = mDResidDx[k][j]; // vector of k-th residuals 1st derivative over X param of track j
503  const auto& cidrkj = covIDrDx[i][k]; // vector covI_k * dres_k/dx_i
504  dchi2 += ROOT::Math::Dot(dr1j, cidrkj);
505  if (k == j) {
506  const auto& res = mTrRes[mCurHyp][k]; // vector of residuals of track k
507  const auto& covI = mTrcEInv[mCurHyp][k]; // inverse cov matrix of track k
508  const auto& dr2ij = mD2ResidDx2[k][j]; // vector of k-th residuals 2nd derivative over X params of track j
509  dchi2 += res[0] * covI.sxx * dr2ij[0] + res[1] * (covI.syy * dr2ij[1] + covI.syz * dr2ij[2]) + res[2] * (covI.syz * dr2ij[1] + covI.szz * dr2ij[2]);
510  }
511  }
512  }
513  }
514 }

517 template <int N, typename... Args>
518 void DCAFitterN<N, Args...>::calcChi2DerivativesNoErr()
519 {
520  //< calculate 1st and 2nd derivatives of abs DCA (chi2) over track parameters X, see (6) in the ref
521  for (int i = N; i--;) {
522  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * Dres_j/Dx_i }
523  dchi1 = 0; // chi2 1st derivative
524  for (int j = N; j--;) {
525  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
526  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
527  dchi1 += ROOT::Math::Dot(res, dr1);
528  if (i >= j) { // symmetrix matrix
529  // chi2 2nd derivative
530  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
531  dchi2 = ROOT::Math::Dot(mTrRes[mCurHyp][i], mD2ResidDx2[i][j]);
532  for (int k = N; k--;) {
533  dchi2 += ROOT::Math::Dot(mDResidDx[k][i], mDResidDx[k][j]);
534  }
535  }
536  }
537  }

598 template <int N, typename... Args>
599 inline double DCAFitterN<N, Args...>::calcChi2() const
600 {
601  // calculate current chi2
602  double chi2 = 0;
603  for (int i = N; i--;) {
604  const auto& res = mTrRes[mCurHyp][i];
605  const auto& covI = mTrcEInv[mCurHyp][i];
606  chi2 += res[0] * res[0] * covI.sxx + res[1] * res[1] * covI.syy + res[2] * res[2] * covI.szz + 2. * res[1] * res[2] * covI.syz;
607  }

625 template <int N, typename... Args>
626 bool DCAFitterN<N, Args...>::correctTracks(const VecND& corrX)
627 {
628  // propagate tracks to updated X
629  for (int i = N; i--;) {
630  const auto& trDer = mTrDer[mCurHyp][i];
631  auto dx2h = 0.5 * corrX[i] * corrX[i];
632  mTrPos[mCurHyp][i][0] -= corrX[i];
633  mTrPos[mCurHyp][i][1] -= trDer.dydx * corrX[i] - dx2h * trDer.d2ydx2;
634  mTrPos[mCurHyp][i][2] -= trDer.dzdx * corrX[i] - dx2h * trDer.d2zdx2;
635  }
636  return true;
637 }

678 template <int N, typename... Args>
679 bool DCAFitterN<N, Args...>::minimizeChi2()
680 {
681  // find best chi2 (weighted DCA) of N tracks in the vicinity of the seed PCA
682  for (int i = N; i--;) {
683  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];
684  auto x = mTrAux[i].c * mPCA[mCurHyp][0] + mTrAux[i].s * mPCA[mCurHyp][1]; // X of PCA in the track frame
685  if (!mCandTr[mCurHyp][i].propagateTo(x, mBz)) {
686  return false;
687  }
688  setTrackPos(mTrPos[mCurHyp][i], mCandTr[mCurHyp][i]); // prepare positions
689  mTrcEInv[mCurHyp][i].set(mCandTr[mCurHyp][i]); // prepare inverse cov.matrices at starting point
690  }
691 
692  if (mMaxDZIni > 0 && !roughDZCut()) { // apply rough cut on tracks Z difference
693  return false;
694  }
695 
696  if (!calcPCACoefs()) { // prepare tracks contribution matrices to the global PCA
697  return false;
698  }
699  calcPCA(); // current PCA
700  calcTrackResiduals(); // current track residuals
701  float chi2Upd, chi2 = calcChi2();
702  do {
703  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
704  calcResidDerivatives(); // current residals derivatives (1st and 2nd)
705  calcChi2Derivatives(); // current chi2 derivatives (1st and 2nd)
706 
707  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
708  if (!mD2Chi2Dx2.Invert()) {
709  LOG(ERROR) << "InversionFailed";
710  return false;
711  }
712  VecND dx = mD2Chi2Dx2 * mDChi2Dx;
713  if (!correctTracks(dx)) {
714  return false;
715  }
716  calcPCA(); // updated PCA
717  if (mCrossIDAlt >= 0 && closerToAlternative()) {
718  mAllowAltPreference = false;
719  return false;
720  }
721  calcTrackResiduals(); // updated residuals
722  chi2Upd = calcChi2(); // updated chi2
723  if (getAbsMax(dx) < mMinParamChange || chi2Upd > chi2 * mMinRelChi2Change) {
724  chi2 = chi2Upd;
725  break; // converged
726  }
727  chi2 = chi2Upd;
728  } while (++mNIters[mCurHyp] < mMaxIter);
729  //
730  mChi2[mCurHyp] = chi2 * NInv;
731  return mChi2[mCurHyp] < mMaxChi2;
732 }
