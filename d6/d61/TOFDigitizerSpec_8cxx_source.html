47  void initDigitizerTask(framework::InitContext& ic) override
48  {
49  LOG(INFO) << "Initializing TOF digitization";
50 
51  mSimChains = std::move(std::make_unique<std::vector<TChain*>>());
52 
53  // the instance of the actual digitizer
54  mDigitizer = std::move(std::make_unique<o2::tof::Digitizer>());
55  // containers for digits and labels
56  mDigits = std::move(std::make_unique<std::vector<o2::tof::Digit>>());
57  mLabels = std::move(std::make_unique<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>());
58 
59  // init digitizer
60  mDigitizer->init();
61  const bool isContinuous = ic.options().get<int>("pileup");
62  LOG(INFO) << "CONTINUOUS " << isContinuous;
63  mDigitizer->setContinuous(isContinuous);
64  mDigitizer->setMCTruthContainer(mLabels.get());
65  LOG(INFO) << "TOF initialization done";
66  }

68  void run(framework::ProcessingContext& pc)
69  {
70  static bool finished = false;
71  if (finished) {
72  return;
73  }
74  // RS: at the moment using hardcoded flag for continuos readout
75  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
76 
77  // read collision context from input
78  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
79  auto& timesview = context->getEventRecords();
80  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
81 
82  context->initSimChains(o2::detectors::DetID::TOF, *mSimChains.get());
83 
84  // if there is nothing to do ... return
85  if (timesview.size() == 0) {
86  return;
87  }
88 
89  TStopwatch timer;
90  timer.Start();
91 
92  LOG(INFO) << " CALLING TOF DIGITIZATION ";
93  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
94  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObj;
95 
96  if (mUseCCDB) { // read calibration objects from ccdb
97  // check LHC phase
98  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
99  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
100 
101  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
102  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
103 
104  // make a copy in global scope
105  lhcPhaseObj = lhcPhaseObjTmp;
106  channelCalibObj = channelCalibObjTmp;
107  } else { // calibration objects set to zero
108  lhcPhaseObj.addLHCphase(0, 0);
109  lhcPhaseObj.addLHCphase(2000000000, 0);
110 
111  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
112  channelCalibObj.addTimeSlewingInfo(ich, 0, 0);
113  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
114  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
115  channelCalibObj.setFractionUnderPeak(sector, channelInSector, 1);
116  }
117  }
118 
119  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, &channelCalibObj);
120  mDigitizer->setCalibApi(&calibapi);
121 
122  static std::vector<o2::tof::HitType> hits;
123 
124  auto& eventParts = context->getEventParts();
125  // loop over all composite collisions given from context
126  // (aka loop over all the interaction records)
127  for (int collID = 0; collID < timesview.size(); ++collID) {
128  mDigitizer->setEventTime(timesview[collID].timeNS);
129 
130  // for each collision, loop over the constituents event and source IDs
131  // (background signal merging is basically taking place here)
132  for (auto& part : eventParts[collID]) {
133  mDigitizer->setEventID(part.entryID);
134  mDigitizer->setSrcID(part.sourceID);
135 
136  // get the hits for this event and this source
137  hits.clear();
138  context->retrieveHits(*mSimChains.get(), "TOFHit", part.sourceID, part.entryID, &hits);
139 
140  // LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found " << hits.size() << " hits ";
141 
142  // call actual digitization procedure
143  mLabels->clear();
144  mDigits->clear();
145  mDigitizer->process(&hits, mDigits.get());
146  }
147  }
148  if (mDigitizer->isContinuous()) {
149  mDigits->clear();
150  mLabels->clear();
151  mDigitizer->flushOutputContainer(*mDigits.get());
152  }
153 
154  std::vector<Digit>* digitsVector = mDigitizer->getDigitPerTimeFrame();
155  std::vector<ReadoutWindowData>* readoutwindow = mDigitizer->getReadoutWindowData();
156  std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>* mcLabVecOfVec = mDigitizer->getMCTruthPerTimeFrame();
157 
158  LOG(INFO) << "Post " << digitsVector->size() << " digits in " << readoutwindow->size() << " RO windows";
159 
160  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
161  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe}, *digitsVector);
162  if (pc.outputs().isAllowed({o2::header::gDataOriginTOF, "DIGITSMCTR", 0})) {
163  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe}, *mcLabVecOfVec);
164  }
165  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe}, *readoutwindow);
166  LOG(INFO) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
167  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}, roMode);
168 
169  timer.Stop();
170  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
171 
172  // we should be only called once; tell DPL that this process is ready to exit
173  pc.services().get<ControlService>().endOfStream();
174 
175  finished = true;
176  }
177 
178  private:
179  std::unique_ptr<std::vector<TChain*>> mSimChains;
180  std::unique_ptr<o2::tof::Digitizer> mDigitizer;
181  std::unique_ptr<std::vector<o2::tof::Digit>> mDigits;
182  std::unique_ptr<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> mLabels;
183  bool mUseCCDB = false;
184 };

186 DataProcessorSpec getTOFDigitizerSpec(int channel, bool useCCDB, bool mctruth)
187 {
188  // create the full data processor spec using
189  // a name identifier
190  // input description
191  // algorithmic description (here a lambda getting called once to setup the actual processing function)
192  // options that can be used for this processor (here: input file names where to take the hits)
193  std::vector<InputSpec> inputs;
194  inputs.emplace_back("collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
195  if (useCCDB) {
196  inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
197  inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
198  }
199  std::vector<OutputSpec> outputs;
200  outputs.emplace_back(o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
201  outputs.emplace_back(o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
202  if (mctruth) {
203  outputs.emplace_back(o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
204  }
205  outputs.emplace_back(o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe);
206  return DataProcessorSpec{
207  "TOFDigitizer",
208  inputs,
209  outputs,
210  AlgorithmSpec{adaptFromTask<TOFDPLDigitizerTask>(useCCDB)},
211  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}
212  // I can't use VariantType::Bool as it seems to have a problem
213  };
214 }
