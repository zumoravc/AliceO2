25 void ChamberStatus::setStatus(int det, char status)
26 {
27 
28  //
29  // set the chamber status
30  //
31  //
32 
33  switch (status) {
34  case ChamberStatus::kGood:
35  mStatus[det] |= kGoodpat;
36  mStatus[det] &= !kNoDatapat;
37  mStatus[det] &= !kBadCalibratedpat;
38  break;
39  case ChamberStatus::kNoData:
40  mStatus[det] &= !kGoodpat;
41  mStatus[det] |= kNoDatapat;
42  mStatus[det] |= kNoDataHalfChamberSideApat;
43  mStatus[det] |= kNoDataHalfChamberSideBpat;
44  // mStatus[det] |= kBadCalibratedpat;
45  break;
46  case ChamberStatus::kNoDataHalfChamberSideA:
47  mStatus[det] |= kNoDataHalfChamberSideApat;
48  if (mStatus[det] & kNoDataHalfChamberSideB) {
49  mStatus[det] |= kNoDatapat;
50  mStatus[det] &= !kGoodpat;
51  }
52  break;
53  case ChamberStatus::kNoDataHalfChamberSideB:
54  mStatus[det] |= kNoDataHalfChamberSideBpat;
55  if (mStatus[det] & kNoDataHalfChamberSideApat) {
56  mStatus[det] &= !kGoodpat;
57  mStatus[det] |= kNoDatapat;
58  }
59  break;
60  case ChamberStatus::kBadCalibrated:
61  mStatus[det] &= !kGoodpat;
62  mStatus[det] |= kBadCalibratedpat;
63  break;
64  case ChamberStatus::kNotCalibrated:
65  mStatus[det] &= !kGoodpat;
66  mStatus[det] |= kNotCalibratedpat;
67  // mStatus[det] &= !kBadCalibratedpat;
68  break;
69  default:
70  mStatus[det] &= !kGoodpat;
71  mStatus[det] &= !kNoDatapat;
72  mStatus[det] &= !kNoDataHalfChamberSideApat;
73  mStatus[det] &= !kNoDataHalfChamberSideBpat;
74  mStatus[det] &= !kBadCalibratedpat;
75  mStatus[det] &= !kNotCalibratedpat;
76  }
77 }

79 void ChamberStatus::unsetStatusBit(int det, char status)
80 {
81 
82  //
83  // unset the chamber status bit
84  //
85  //
86 
87  switch (status) {
88  case ChamberStatus::kGood:
89  mStatus[det] &= !kGoodpat;
90  break;
91  case ChamberStatus::kNoData:
92  mStatus[det] &= !kNoDatapat;
93  break;
94  case ChamberStatus::kNoDataHalfChamberSideA:
95  mStatus[det] &= !kNoDataHalfChamberSideApat;
96  break;
97  case ChamberStatus::kNoDataHalfChamberSideB:
98  mStatus[det] &= !kNoDataHalfChamberSideBpat;
99  break;
100  case ChamberStatus::kBadCalibrated:
101  mStatus[det] &= !kBadCalibratedpat;
102  break;
103  case ChamberStatus::kNotCalibrated:
104  mStatus[det] &= !kNotCalibratedpat;
105  break;
106  default:
107  mStatus[det] &= !(kGoodpat & kNoDatapat & kNoDataHalfChamberSideApat & kNoDataHalfChamberSideBpat & kBadCalibratedpat & kNotCalibratedpat);
108  }
109 }

111 TH2D* ChamberStatus::plot(int sm, int rphi)
112 {
113  //
114  // Plot chamber status for supermodule and halfchamberside
115  // as a function of layer and stack
116  //
117 
118  TH2D* h2 = new TH2D(Form("sm_%d_rphi_%d", sm, rphi), Form("sm_%d_rphi_%d", sm, rphi), 5, 0.0, 5.0, 6, 0.0, 6.0);
119 
120  h2->SetDirectory(nullptr);
121  h2->SetXTitle("stack");
122  h2->SetYTitle("layer");
123 
124  int start = sm * 30;
125  int end = (sm + 1) * 30;
126 
127  for (int i = start; i < end; i++) {
128  int layer = i % 6;
129  int stackn = static_cast<int>((i - start) / 6.);
130  int status = getStatus(i);
131  h2->Fill(stackn, layer, status);
132  if (rphi == 0) {
133  if (!(mStatus[i] & kNoDataHalfChamberSideBpat))
134  h2->Fill(stackn, layer, status);
135  } else if (rphi == 1) {
136  if (!(mStatus[i] & kNoDataHalfChamberSideApat))
137  h2->Fill(stackn, layer, status);
138  }
139  }
140 
141  return h2;
142 }
