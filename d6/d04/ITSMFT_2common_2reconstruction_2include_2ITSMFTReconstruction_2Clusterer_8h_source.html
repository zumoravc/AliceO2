109  void expandPreCluster(uint32_t ip, uint16_t row, int preClusIndex)
110  {
111  auto& firstIndex = preClusterHeads[preClusterIndices[preClusIndex]];
112  pixels.emplace_back(firstIndex, ip);
113  firstIndex = pixels.size() - 1;
114  curr[row] = preClusIndex;
115  }
116 
118  void addNewPrecluster(uint32_t ip, uint16_t row)
119  {
120  preClusterHeads.push_back(pixels.size());
121  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
122  pixels.emplace_back(-1, ip);
123  int lastIndex = preClusterIndices.size();
124  preClusterIndices.push_back(lastIndex);
125  curr[row] = lastIndex; // store index of the new precluster in the current column buffer
126  }
