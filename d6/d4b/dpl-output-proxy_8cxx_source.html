42 WorkflowSpec defineDataProcessing(ConfigContext const& config)
43 {
44  std::string processorName = config.options().get<std::string>("proxy-name");
45  std::string inputConfig = config.options().get<std::string>("dataspec");
46  int defaultPort = config.options().get<int>("default-port");
47  std::string defaultTransportConfig = config.options().get<std::string>("default-transport");
48  if (defaultTransportConfig == "zeromq") {
49  // nothing to do for the moment
50  } else if (defaultTransportConfig == "shmem") {
51  // nothing to do for the moment
52  } else {
53  throw std::runtime_error("invalid argument for option --default-transport : '" + defaultTransportConfig + "'");
54  }
55 
56  std::vector<InputSpec> inputs = select(inputConfig.c_str());
57  if (inputs.size() == 0) {
58  throw std::runtime_error("invalid dataspec '" + inputConfig + "'");
59  }
60 
61  // we build the default channel configuration from the binding of the first input
62  // in order to have more than one we would need to possibility to have support for
63  // vectored options
64  // use the OutputChannelSpec as a tool to create the default configuration for the out-of-band channel
65  OutputChannelSpec externalChannelSpec;
66  externalChannelSpec.name = inputs[0].binding;
67  externalChannelSpec.type = ChannelType::Push;
68  externalChannelSpec.method = ChannelMethod::Bind;
69  externalChannelSpec.hostname = "localhost";
70  externalChannelSpec.port = defaultPort;
71  externalChannelSpec.listeners = 0;
72  // in principle, protocol and transport are two different things but fur simplicity
73  // we use ipc when shared memory is selected and the normal tcp url whith zeromq,
74  // this is for building the default configuration which can be simply changed from the
75  // command line
76  if (!defaultTransportConfig.empty()) {
77  if (defaultTransportConfig == "zeromq") {
78  externalChannelSpec.protocol = ChannelProtocol::Network;
79  } else if (defaultTransportConfig == "shmem") {
80  externalChannelSpec.protocol = ChannelProtocol::IPC;
81  }
82  }
83  std::string defaultChannelConfig = formatExternalChannelConfiguration(externalChannelSpec);
84  // at some point the formatting tool might add the transport as well so we have to check
85  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
86  defaultChannelConfig += ",transport=" + defaultTransportConfig;
87  }
88 
89  std::vector<DataProcessorSpec> workflow;
90  workflow.emplace_back(std::move(specifyFairMQDeviceOutputProxy(processorName.c_str(), inputs, defaultChannelConfig.c_str())));
91  return workflow;
92 }
