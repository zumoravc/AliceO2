33 struct HashTask {
34  std::vector<float> xBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
35  std::vector<float> yBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
36  Produces<aod::Hashes> hashes;
37 
38  // Calculate hash for an element based on 2 properties and their bins.
39  int getHash(std::vector<float> const& xBins, std::vector<float> const& yBins, float colX, float colY)
40  {
41  if (colX < xBins[0] || colY < yBins[0]) {
42  return -1;
43  }
44  for (int i = 1; i < xBins.size(); i++) {
45  if (colX < xBins[i]) {
46  for (int j = 1; j < yBins.size(); j++) {
47  if (colY < yBins[j]) {
48  return i + j * (xBins.size() + 1);
49  }
50  }
51  return -1;
52  }
53  }
54 
55  return -1;
56  }
57 
58  void process(aod::Collisions const& collisions)
59  {
60  for (auto& collision : collisions) {
61  int hash = getHash(xBins, yBins, collision.posX(), collision.posY());
62  LOGF(info, "Collision: %d (%f, %f, %f) hash: %d", collision.index(), collision.posX(), collision.posY(), collision.posZ(), hash);
63  hashes(hash);
64  }
65  }
66 };

68 // Version 1: Using categorised combinations
69 struct CollisionsCombinationsTask {
70  o2::framework::expressions::Filter trackFilter = (aod::track::x > -0.8f) && (aod::track::x < 0.8f) && (aod::track::y > 1.0f);
71 
72  void process(aod::Hashes const& hashes, aod::Collisions& collisions, soa::Filtered<aod::Tracks>& tracks)
73  {
74  collisions.bindExternalIndices(&tracks);
75  auto tracksTuple = std::make_tuple(tracks);
76  AnalysisDataProcessorBuilder::GroupSlicer slicer(collisions, tracksTuple);
77 
78  // Strictly upper categorised collisions
79  for (auto& [c1, c2] : selfCombinations("fBin", 5, -1, join(hashes, collisions), join(hashes, collisions))) {
80  LOGF(info, "Collisions bin: %d pair: %d (%f, %f, %f), %d (%f, %f, %f)", c1.bin(), c1.index(), c1.posX(), c1.posY(), c1.posZ(), c2.index(), c2.posX(), c2.posY(), c2.posZ());
81 
82  auto it1 = slicer.begin();
83  auto it2 = slicer.begin();
84  for (auto& slice : slicer) {
85  if (slice.groupingElement().index() == c1.index()) {
86  it1 = slice;
87  break;
88  }
89  }
90  for (auto& slice : slicer) {
91  if (slice.groupingElement().index() == c2.index()) {
92  it2 = slice;
93  break;
94  }
95  }
96  auto tracks1 = std::get<soa::Filtered<aod::Tracks>>(it1.associatedTables());
97  tracks1.bindExternalIndices(&collisions);
98  auto tracks2 = std::get<soa::Filtered<aod::Tracks>>(it2.associatedTables());
99  tracks2.bindExternalIndices(&collisions);
100 
101  for (auto& [t1, t2] : combinations(CombinationsFullIndexPolicy(tracks1, tracks2))) {
102  LOGF(info, "Mixed event tracks pair: (%d, %d) from events (%d, %d)", t1.index(), t2.index(), c1.index(), c2.index());
103  }
104  }
105  }
106 };
