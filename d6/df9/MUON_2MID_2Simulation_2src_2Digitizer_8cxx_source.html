28 void Digitizer::addStrip(const Mapping::MpStripIndex& stripIndex, int cathode, int deId)
29 {
31  for (auto& col : mDigits) {
32  if (col.deId == deId && col.columnId == stripIndex.column) {
33  col.addStrip(stripIndex.strip, cathode, stripIndex.line);
34  return;
35  }
36  }
37 
38  mDigits.emplace_back(ColumnDataMC{(uint8_t)deId, (uint8_t)stripIndex.column});
39  mDigits.back().addStrip(stripIndex.strip, cathode, stripIndex.line);

43 bool Digitizer::addBPStrips(double xPos, double yPos, int deId, double prob, double xOffset)
44 {
46 
47  // The offset is used to check the strips in the bending plane
48  // placed in the board close to the fired one
49  // If the response says that the strip is too far away, there is no need to check further
50  if (!mResponse.isFired(prob, std::abs(xOffset), 0, deId)) {
51  return false;
52  }
53 
54  Mapping::MpStripIndex stripIndex = mMapping.stripByPosition(xPos + 1.01 * xOffset, yPos, 0, deId, false);
55  // Check if the point is still inside the RPC
56  if (!stripIndex.isValid()) {
57  return false;
58  }
59  addStrip(stripIndex, 0, deId);
60  MpArea area = mMapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
61  std::array<double, 2> dist = {area.getYmax() - yPos, yPos - area.getYmin()};
62  addNeighbours(stripIndex, 0, deId, prob, dist, xOffset);
63  return true;
64 }

67 bool Digitizer::addNeighbours(const Mapping::MpStripIndex& stripIndex, int cathode, int deId, double prob,
68  const std::array<double, 2>& initialDist, double xOffset)
69 {
71 
72  double xOffset2 = xOffset * xOffset;
73  for (int idir = 0; idir < 2; ++idir) {
74  // Search for neighbours in the two directions
75  // up and down for the BP, right and left for the NBP
76  double dist = initialDist[idir];
77  Mapping::MpStripIndex neigh = mMapping.nextStrip(stripIndex, cathode, deId, idir);
78  while (neigh.isValid() && mResponse.isFired(prob, std::sqrt(dist * dist + xOffset2), cathode, deId)) {
79  addStrip(neigh, cathode, deId);
80  dist += mMapping.getStripSize(neigh.strip, cathode, neigh.column, deId);
81  neigh = mMapping.nextStrip(neigh, cathode, deId, idir);
82  }
83  }
84  return true;
85 }

88 bool Digitizer::hitToDigits(const Hit& hit)
89 {
91 
92  // Clear
93  mDigits.clear();
94 
95  // Convert point from global to local coordinates
96  auto midPt = hit.middlePoint();
97  int deId = hit.GetDetectorID();
98  Point3D<double> localPoint = mTransformer.globalToLocal(deId, (double)midPt.x(), (double)midPt.y(), (double)midPt.z());
99 
100  // First get the touched BP strip
101  Mapping::MpStripIndex stripIndex = mMapping.stripByPosition(localPoint.x(), localPoint.y(), 0, deId);
102 
103  // Check if the chamber was efficient
104  bool isEfficientBP, isEfficientNBP;
105  if (!mEfficiencyResponse.isEfficient(deId, stripIndex.column, stripIndex.line, isEfficientBP, isEfficientNBP)) {
106  return false;
107  }
108 
109  // Digitize if the RPC was efficient
110  double prob = mRandom(mGenerator);
111 
112  if (isEfficientBP) {
113  addStrip(stripIndex, 0, deId);
114  MpArea area = mMapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
115  // This is the distance between the hit point and the edges of the strip along y
116  std::array<double, 2> dist = {area.getYmax() - localPoint.y(), localPoint.y() - area.getYmin()};
117  addNeighbours(stripIndex, 0, deId, prob, dist);
118  // Search for neighbours in the close column toward inside
119  addBPStrips(localPoint.x(), localPoint.y(), deId, prob, area.getXmin() - localPoint.x());
120  // Search for neighbours in the close column toward outside
121  addBPStrips(localPoint.x(), localPoint.y(), deId, prob, area.getXmax() - localPoint.x());
122  }
123 
124  // Then check the touched NBP strip
125  if (isEfficientNBP) {
126  stripIndex = mMapping.stripByPosition(localPoint.x(), localPoint.y(), 1, deId);
127  addStrip(stripIndex, 1, deId);
128  MpArea area = mMapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
129  // This is the distance between the hit point and the edges of the strip along x
130  std::array<double, 2> dist = {area.getXmax() - localPoint.x(), localPoint.x() - area.getXmin()};
131  addNeighbours(stripIndex, 1, deId, prob, dist);
132  }
133 
134  return true;
135 }

138 void Digitizer::process(const std::vector<Hit>& hits, std::vector<ColumnDataMC>& digitStore, o2::dataformats::MCTruthContainer<MCLabel>& mcContainer)
139 {
141  digitStore.clear();
142  mcContainer.clear();
143  int firstStrip = 0, lastStrip = 0;
144 
145  for (auto& hit : hits) {
146  hitToDigits(hit);
147 
148  for (auto& digit : mDigits) {
149  digitStore.emplace_back(digit);
150  for (int icathode = 0; icathode < 2; ++icathode) {
151  if (getLabelLimits(icathode, digit, firstStrip, lastStrip)) {
152  MCLabel label(hit.GetTrackID(), mEventID, mSrcID, digit.deId, digit.columnId, icathode, firstStrip, lastStrip);
153  mcContainer.addElement(digitStore.size() - 1, label);
154  }
155  }
156  }
157  }
158 }

161 bool Digitizer::getLabelLimits(int cathode, const ColumnDataMC& col, int& firstStrip, int& lastStrip) const
162 {
164  int nLines = (cathode == 0) ? 4 : 1;
165  int invalid = 9999;
166  firstStrip = invalid, lastStrip = invalid;
167  for (int iline = 0; iline < nLines; ++iline) {
168  for (int istrip = 0; istrip < 16; ++istrip) {
169  if (col.isStripFired(istrip, cathode, iline)) {
170  lastStrip = MCLabel::getStrip(istrip, iline);
171  if (firstStrip == invalid) {
172  firstStrip = lastStrip;
173  }
174  } else if (firstStrip != invalid) {
175  return true;
176  }
177  }
178  }
179  return (firstStrip != invalid);
180 }
