38 class DigitsReaderDeviceDPL
39 {
40  public:
41  void init(o2::framework::InitContext& ic)
42  {
43  auto filename = ic.options().get<std::string>("mid-digit-infile");
44  mFile = std::make_unique<TFile>(filename.c_str());
45  if (!mFile->IsOpen()) {
46  LOG(ERROR) << "Cannot open the " << filename << " file !";
47  mState = 1;
48  return;
49  }
50  mTree = static_cast<TTree*>(mFile->Get("o2sim"));
51  if (!mTree) {
52  LOG(ERROR) << "Cannot find tree in " << filename;
53  mState = 1;
54  return;
55  }
56  mTree->SetBranchAddress("MIDDigit", &mDigits);
57  mTree->SetBranchAddress("MIDDigitMCLabels", &mMCContainer);
58  mTree->SetBranchAddress("MIDROFRecords", &mROFRecords);
59  mState = 0;
60  }
61 
62  void run(o2::framework::ProcessingContext& pc)
63  {
64  if (mState != 0) {
65  return;
66  }
67 
68  std::vector<ColumnData> digits;
69  o2::dataformats::MCTruthContainer<MCLabel> mcContainer;
70  std::vector<ROFRecord> rofRecords;
71 
72  for (auto ientry = 0; ientry < mTree->GetEntries(); ++ientry) {
73  mTree->GetEntry(ientry);
74  digits.insert(digits.end(), mDigits->begin(), mDigits->end());
75  mcContainer.mergeAtBack(*mMCContainer);
76  rofRecords.insert(rofRecords.end(), mROFRecords->begin(), mROFRecords->end());
77  }
78 
79  LOG(DEBUG) << "MIDDigitsReader pushed " << digits.size() << " merged digits";
80  pc.outputs().snapshot(of::Output{"MID", "DATA", 0, of::Lifetime::Timeframe}, digits);
81  pc.outputs().snapshot(of::Output{"MID", "DATAROF", 0, of::Lifetime::Timeframe}, rofRecords);
82  LOG(DEBUG) << "MIDDigitsReader pushed " << mcContainer.getIndexedSize() << " indexed digits";
83  pc.outputs().snapshot(of::Output{"MID", "DATALABELS", 0, of::Lifetime::Timeframe}, mcContainer);
84 
85  mState = 2;
86  pc.services().get<of::ControlService>().endOfStream();
87  }
88 
89  private:
90  std::unique_ptr<TFile> mFile{nullptr};
91  TTree* mTree{nullptr}; // not owner
92  std::vector<o2::mid::ColumnDataMC>* mDigits{nullptr}; // not owner
93  o2::dataformats::MCTruthContainer<MCLabel>* mMCContainer{nullptr}; // not owner
94  std::vector<o2::mid::ROFRecord>* mROFRecords{nullptr}; // not owner
95  int mState = 0;
96 };
