168 GPUdi() CONSTEXPR17 T GPUCommonMath::nextMultipleOf(T val)
169 {
170  CONSTEXPRIF(I & (I - 1))
171  {
172  T tmp = val % I;
173  if (tmp)
174  val += I - tmp;
175  return val;
176  }
177  else
178  {
179  return (val + I - 1) & ~(T)(I - 1);

328 GPUhdi() float GPUCommonMath::FastInvSqrt(float _x)
329 {
330  // the function calculates fast inverse sqrt
331  union {
332  float f;
333  int i;
334  } x = {_x};
335  const float xhalf = 0.5f * x.f;
336  x.i = 0x5f3759df - (x.i >> 1);
337  x.f = x.f * (1.5f - xhalf * x.f * x.f);
338  return x.f;
339 }
