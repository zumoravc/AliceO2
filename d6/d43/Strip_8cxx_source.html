36 Int_t Strip::addDigit(Int_t channel, Int_t tdc, Int_t tot, Int_t bc, Int_t lbl, Int_t triggerorbit, Int_t triggerbunch)
37 {
38 
39  // return the MC label. We pass it also as argument, but it can change in
40  // case the digit was merged
41 
42  auto key = Digit::getOrderingKey(channel, bc, tdc); // the digits are ordered first per channel, then inside the channel per BC, then per time
43  auto dig = findDigit(key);
44  if (dig) {
45  lbl = dig->getLabel(); // getting the label from the already existing digit
46  dig->merge(tdc, tot); // merging to the existing digit
47  mDigitMerged++;
48  } else {
49  mDigits.emplace(std::make_pair(key, Digit(channel, tdc, tot, bc, lbl, triggerorbit, triggerbunch)));
50  }
51 
52  return lbl;
53 }

56 void Strip::fillOutputContainer(std::vector<Digit>& digits)
57 {
58  // transfer digits that belong to the strip to the output array of digits
59  // we assume that the Strip has stored inside only digits from one readout
60  // window --> we flush them all
61 
62  if (mDigits.empty())
63  return;
64  auto itBeg = mDigits.begin();
65  auto iter = itBeg;
66  for (; iter != mDigits.end(); ++iter) {
67  Digit& dig = iter->second;
68  digits.emplace_back(dig);
69  }
70 
71  // if (iter!=mDigits.end()) iter--;
72  mDigits.erase(itBeg, iter);
73 }
