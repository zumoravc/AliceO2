82  {
83  TrackInserter inserter = [&tracks, &clusIdx, this](const TrackITSExt& t) -> int {
84  // convert internal track to output format
85  auto& trackNew = tracks.emplace_back(t);
86  int noc = t.getNumberOfClusters();
87  int clEntry = clusIdx.size();
88  for (int i = 0; i < noc; i++) {
89  const Cluster* c = this->getCluster(t.getClusterIndex(i));
90  Int_t idx = c - &mClusterCache[0]; // Index of this cluster in event
91  clusIdx.emplace_back(idx);
92  }
93  trackNew.setClusterRefs(clEntry, noc);
94  return tracks.size();
95  };
96  process(clusters, it, dict, inserter, rof);
97  }
