32  BOOST_CHECK_CLOSE(bz0, nomBz, 0.1);
33 
34  const int ntst = 10000;

36  double xyz[ntst][3] = {}, bxyz[ntst][3] = {}, dummyAcc = 0.;

39  gRandom->RndmArray(3, rnd);
40  xyz[it][0] = rnd[0] * 400. * TMath::Cos(rnd[1] * TMath::Pi() * 2);
41  xyz[it][1] = rnd[1] * 400. * TMath::Sin(rnd[1] * TMath::Pi() * 2);
42  xyz[it][2] = (rnd[0] - 0.5) * 250;
43  }
44

47  TStopwatch swSlow;

58

63  for (int ii = repFactor; ii--;) {
64  for (int it = ntst; it--;) {
65  fld->Field(xyz[it], bfast);
66  }
67  }
68  swFast.Stop();

72  double rat = sF > 0. ? sS / sF : -1;
73  LOG(INFO) << "Timing: Exact param: " << sS << " Fast param: " << sF
74  << "s/call -> factor " << rat;
75 
76  // compare slow/fast param precision

78  const char comp[] = "XYZ";

96  BOOST_CHECK(TMath::Abs(mean[i] / nomBz) < 1.e-3);
