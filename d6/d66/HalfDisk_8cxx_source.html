45 HalfDisk::HalfDisk(HalfDiskSegmentation* segmentation)
46  : TNamed(segmentation->GetName(), segmentation->GetName()),
47  mPCBSupport(nullptr),
48  mSupport(nullptr),
49  mHeatExchanger(nullptr),
50  mSegmentation(segmentation)
51 {
52  Geometry* mftGeom = Geometry::instance();
53  SetUniqueID(mSegmentation->GetUniqueID());
54  auto& mftBaseParam = MFTBaseParam::Instance();
55 
56  LOG(DEBUG1) << "HalfDisk " << Form("creating half-disk: %s Unique ID = %d ", GetName(), mSegmentation->GetUniqueID());
57 
58  mHalfDiskVolume = new TGeoVolumeAssembly(GetName());
59 
60  // Building Heat Exchanger Between faces
61  if (mftBaseParam.buildHeatExchanger) {
62  TGeoVolumeAssembly* heatExchangerVol = createHeatExchanger();
63  mHalfDiskVolume->AddNode(heatExchangerVol, 1);
64  }
65 
66  if (!mftBaseParam.minimal && mftBaseParam.buildPCBSupports) {
67  // Building Support
68  TGeoVolumeAssembly* supportVol = createSupport();
69  mHalfDiskVolume->AddNode(supportVol, 1);
70  }
71 
72  if (!mftBaseParam.minimal && mftBaseParam.buildPCBs) {
73  // Building PCB
74  TGeoVolumeAssembly* PCBVol = createPCBSupport();
75  mHalfDiskVolume->AddNode(PCBVol, 1);
76  }
77  // Building Front Face of the Half Disk
78  createLadders();
79 }

137 void HalfDisk::createLadders()
138 {
139 
140  LOG(DEBUG1) << "CreateLadders: start building ladders";
141  for (Int_t iLadder = 0; iLadder < mSegmentation->getNLadders(); iLadder++) {
142 
143  LadderSegmentation* ladderSeg = mSegmentation->getLadder(iLadder);
144  if (!ladderSeg)
145  Fatal("CreateLadders", Form("No Segmentation found for ladder %d ", iLadder), 0, 0);
146  auto* ladder = new Ladder(ladderSeg);
147  TGeoVolume* ladVol = ladder->createVolume();
148 
149  // Position of the center on the ladder volume in the ladder coordinate system
150  TGeoBBox* shape = (TGeoBBox*)ladVol->GetShape();
151  Double_t center[3];
152  center[0] = shape->GetDX();
153  center[1] = shape->GetDY();
154  center[2] = shape->GetDZ();
155 
156  Double_t master[3];
157  ladderSeg->getTransformation()->LocalToMaster(center, master);
158  Int_t ladderId = Geometry::instance()->getLadderID(ladderSeg->GetUniqueID());
159 
160  mHalfDiskVolume->AddNode(ladVol, ladderId, new TGeoCombiTrans(master[0], master[1], master[2], ladderSeg->getTransformation()->GetRotation()));
161 
162  delete ladder;
163  }
164 }
