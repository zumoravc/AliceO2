37 const int gTreeSize = 10; // elements in the test tree
38 DataProcessorSpec getSourceSpec()
39 {
40  auto initFct = [](InitContext& ic) {
41  // create a test tree in a temporary file
42  std::string fileName = gSystem->TempDirectory();
43  fileName += "/test_RootTreeReader.root";
44 
45  {
46  std::unique_ptr<TFile> testFile(TFile::Open(fileName.c_str(), "RECREATE"));
47  std::unique_ptr<TTree> testTree = std::make_unique<TTree>("testtree", "testtree");
48 
49  std::vector<o2::test::TriviallyCopyable> msgblarray;
50  std::vector<o2::test::Polymorphic> valarray;
51  auto* branch1 = testTree->Branch("msgblarray", &msgblarray);
52  auto* branch2 = testTree->Branch("dataarray", &valarray);
53 
54  for (int entry = 0; entry < gTreeSize; entry++) {
55  msgblarray.clear();
56  valarray.clear();
57  for (int idx = 0; idx < entry + 1; ++idx) {
58  msgblarray.emplace_back((entry * 10) + idx, 0, 0);
59  valarray.emplace_back((entry * 10) + idx);
60  }
61  testTree->Fill();
62  }
63  testTree->Write();
64  testTree->SetDirectory(nullptr);
65  testFile->Close();
66  }
67 
68  constexpr auto persistency = Lifetime::Transient;
69  auto reader = std::make_shared<RootTreeReader>("testtree", // tree name
70  fileName.c_str(), // input file name
71  RootTreeReader::BranchDefinition<std::vector<o2::test::TriviallyCopyable>>{Output{"TST", "ARRAYOFMSGBL", 0, persistency}, "msgblarray"},
72  Output{"TST", "ARRAYOFDATA", 0, persistency},
73  "dataarray",
74  RootTreeReader::PublishingMode::Single);
75 
76  auto processingFct = [reader](ProcessingContext& pc) {
77  if (reader->getCount() >= gTreeSize) {
78  return;
79  }
80  if (reader->getCount() == 0) {
81  // add two additional headers on the stack in the first entry
82  o2::header::NameHeader<16> auxHeader("extended_info");
83  o2::header::DataHeader dummyheader;
84  (++(*reader))(pc, auxHeader, dummyheader);
85  } else {
86  // test signature without headers for the rest of the entries
87  (++(*reader))(pc);
88  }
89  if ((reader->getCount()) >= gTreeSize) {
90  pc.services().get<ControlService>().endOfStream();
91  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
92  }
93  };
94 
95  return processingFct;
96  };
97 
98  return DataProcessorSpec{"source", // name of the processor
99  {},
100  {OutputSpec{"TST", "ARRAYOFDATA"},
101  OutputSpec{"TST", "ARRAYOFMSGBL"}},
102  AlgorithmSpec(initFct)};
103 }

105 DataProcessorSpec getSinkSpec()
106 {
107  auto processingFct = [](ProcessingContext& pc) {
108  static int counter = 0;
109  using DataHeader = o2::header::DataHeader;
110  for (auto& input : pc.inputs()) {
111  auto dh = o2::header::get<const DataHeader*>(input.header);
112  LOG(INFO) << dh->dataOrigin.str << " " << dh->dataDescription.str << " " << dh->payloadSize;
113  }
114  auto data = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input1");
115  if (counter == 0) {
116  // the first entry comes together with additional headers on the stack, test those ...
117  auto auxHeader = DataRefUtils::getHeader<o2::header::NameHeader<16>*>(pc.inputs().get("input1"));
118  ASSERT_ERROR(auxHeader != nullptr);
119  if (auxHeader != nullptr) {
120  ASSERT_ERROR(std::string("extended_info") == auxHeader->getName());
121  o2::header::hexDump("", auxHeader, auxHeader->headerSize);
122  auto dummyheader = auxHeader->next();
123  ASSERT_ERROR(dummyheader != nullptr && dummyheader->size() == sizeof(o2::header::DataHeader));
124  }
125  }
126 
127  LOG(INFO) << "count: " << counter << " data elements:" << data.size();
128  ASSERT_ERROR(counter + 1 == data.size());
129 
130  // retrieving the unserialized message as vector
131  auto msgblvec = pc.inputs().get<std::vector<o2::test::TriviallyCopyable>*>("input2");
132  ASSERT_ERROR(counter + 1 == msgblvec->size());
133 
134  // retrieving the unserialized message as span
135  auto msgblspan = pc.inputs().get<gsl::span<o2::test::TriviallyCopyable>>("input2");
136  ASSERT_ERROR(counter + 1 == msgblspan.size());
137 
138  for (unsigned int idx = 0; idx < data.size(); idx++) {
139  LOG(INFO) << data[idx].get();
140  auto expected = 10 * counter + idx;
141  ASSERT_ERROR(data[idx].get() == expected);
142  ASSERT_ERROR(((*msgblvec)[idx] == o2::test::TriviallyCopyable{expected, 0, 0}));
143  ASSERT_ERROR((msgblspan[idx] == o2::test::TriviallyCopyable{expected, 0, 0}));
144  }
145 
146  ++counter;
147  };
148 
149  return DataProcessorSpec{"sink", // name of the processor
150  {InputSpec{"input1", "TST", "ARRAYOFDATA"},
151  InputSpec{"input2", "TST", "ARRAYOFMSGBL"}},
152  Outputs{},
153  AlgorithmSpec(processingFct)};
154 }
