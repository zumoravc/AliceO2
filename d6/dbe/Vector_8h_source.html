199 template <typename T>
200 void Vector<T>::reset(const T* const source, const int size, const int initialSize)
201 {
202  if (size > mCapacity) {
203  if (mArrayPointer != nullptr) {
204  Utils::Host::gpuFree(mArrayPointer);
205  }
206  Utils::Host::gpuMalloc(reinterpret_cast<void**>(&mArrayPointer), size * sizeof(T));
207  mCapacity = size;
208  }
209 
210  if (source != nullptr) {
211  Utils::Host::gpuMemcpyHostToDevice(mArrayPointer, source, size * sizeof(T));
212  Utils::Host::gpuMemcpyHostToDevice(mDeviceSize, &size, sizeof(int));
213 
214  } else {
215  if (mDeviceSize == nullptr) {
216  Utils::Host::gpuMalloc(reinterpret_cast<void**>(&mDeviceSize), sizeof(int));
217  }
218  Utils::Host::gpuMemcpyHostToDevice(mDeviceSize, &initialSize, sizeof(int));
219  }
220 }
