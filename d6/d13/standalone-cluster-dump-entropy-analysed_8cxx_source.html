164 INode* BuildTree(const double* frequencies, unsigned int UniqueSymbols)
165 {
166  std::priority_queue<INode*, std::vector<INode*>, NodeCmp> trees;
167 
168  for (int i = 0; i < UniqueSymbols; ++i) {
169  if (frequencies[i] != 0) {
170  trees.push(new LeafNode(frequencies[i], i));
171  }
172  }
173  while (trees.size() > 1) {
174  INode* childR = trees.top();
175  trees.pop();
176 
177  INode* childL = trees.top();
178  trees.pop();
179 
180  INode* parent = new InternalNode(childR, childL);
181  trees.push(parent);
182  }
183  return trees.top();
184 }

186 void GenerateCodes(const INode* node, const HuffCode& prefix, HuffCodeMap& outCodes)
187 {
188  if (const LeafNode* lf = dynamic_cast<const LeafNode*>(node)) {
189  outCodes[lf->c] = prefix;
190  } else if (const InternalNode* in = dynamic_cast<const InternalNode*>(node)) {
191  HuffCode leftPrefix = prefix;
192  leftPrefix.push_back(false);
193  GenerateCodes(in->left, leftPrefix, outCodes);
194 
195  HuffCode rightPrefix = prefix;
196  rightPrefix.push_back(true);
197  GenerateCodes(in->right, rightPrefix, outCodes);
198  }
199 }

213 void do_diff(unsigned int& val, int& last, unsigned int bits, unsigned int maxval = 0)
214 {
215  int tmp = val;
216  val -= last;
217  if (maxval && optimized_negative_values) {
218  while ((signed)val < 0) {
219  val += maxval;
220  }
221  } else {
222  val &= (1 << bits) - 1;
223  }
224  last = tmp;
225 }

227 unsigned int truncate(int j, unsigned int val)
228 {
229  if (truncate_bits && field_bits[j] != significant_bits[j] && val) {
230  int ldz = sizeof(unsigned int) * 8 - __builtin_clz(val);
231  if (ldz > significant_bits[j]) {
232  val &= ((1 << ldz) - 1) ^ ((1 << (ldz - significant_bits[j])) - 1);
233  }
234  }
235  return (val);
236 }
