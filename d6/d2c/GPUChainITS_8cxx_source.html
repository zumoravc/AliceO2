79 int GPUChainITS::RunITSTrackFit(std::vector<Road>& roads, std::array<const Cluster*, 7> clusters, std::array<const Cell*, 5> cells, const std::array<std::vector<TrackingFrameInfo>, 7>& tf, std::vector<TrackITSExt>& tracks)
80 {
81  auto threadContext = GetThreadContext();
82  bool doGPU = GetRecoStepsGPU() & RecoStep::ITSTracking;
83  GPUITSFitter& Fitter = processors()->itsFitter;
84  GPUITSFitter& FitterShadow = doGPU ? processorsShadow()->itsFitter : Fitter;
85 
86  Fitter.clearMemory();
87  Fitter.SetNumberOfRoads(roads.size());
88  for (int i = 0; i < 7; i++) {
89  Fitter.SetNumberTF(i, tf[i].size());
90  }
91  Fitter.SetMaxData(processors()->ioPtrs);
92  std::copy(clusters.begin(), clusters.end(), Fitter.clusters());
93  std::copy(cells.begin(), cells.end(), Fitter.cells());
94  SetupGPUProcessor(&Fitter, true);
95  std::copy(roads.begin(), roads.end(), Fitter.roads());
96  for (int i = 0; i < 7; i++) {
97  std::copy(tf[i].begin(), tf[i].end(), Fitter.trackingFrame()[i]);
98  }
99 
100  WriteToConstantMemory(RecoStep::ITSTracking, (char*)&processors()->itsFitter - (char*)processors(), &FitterShadow, sizeof(FitterShadow), 0);
101  TransferMemoryResourcesToGPU(RecoStep::ITSTracking, &Fitter, 0);
102  runKernel<GPUITSFitterKernel>(GetGridBlk(BlockCount(), 0), krnlRunRangeNone, krnlEventNone);
103  TransferMemoryResourcesToHost(RecoStep::ITSTracking, &Fitter, 0);
104 
105  SynchronizeGPU();
106 
107  for (unsigned int i = 0; i < Fitter.NumberOfTracks(); i++) {
108  auto& trkin = Fitter.tracks()[i];
109 
110  tracks.emplace_back(TrackITSExt{{trkin.X(),
111  trkin.mAlpha,
112  {trkin.Par()[0], trkin.Par()[1], trkin.Par()[2], trkin.Par()[3], trkin.Par()[4]},
113  {trkin.Cov()[0], trkin.Cov()[1], trkin.Cov()[2], trkin.Cov()[3], trkin.Cov()[4], trkin.Cov()[5], trkin.Cov()[6], trkin.Cov()[7], trkin.Cov()[8], trkin.Cov()[9], trkin.Cov()[10], trkin.Cov()[11], trkin.Cov()[12], trkin.Cov()[13], trkin.Cov()[14]}},
114  (short int)((trkin.NDF() + 5) / 2),
115  trkin.Chi2(),
116  0,
117  {trkin.mOuterParam.X,
118  trkin.mOuterParam.alpha,
119  {trkin.mOuterParam.P[0], trkin.mOuterParam.P[1], trkin.mOuterParam.P[2], trkin.mOuterParam.P[3], trkin.mOuterParam.P[4]},
120  {trkin.mOuterParam.C[0], trkin.mOuterParam.C[1], trkin.mOuterParam.C[2], trkin.mOuterParam.C[3], trkin.mOuterParam.C[4], trkin.mOuterParam.C[5], trkin.mOuterParam.C[6], trkin.mOuterParam.C[7], trkin.mOuterParam.C[8], trkin.mOuterParam.C[9],
121  trkin.mOuterParam.C[10], trkin.mOuterParam.C[11], trkin.mOuterParam.C[12], trkin.mOuterParam.C[13], trkin.mOuterParam.C[14]}},
122  {{trkin.mClusters[0], trkin.mClusters[1], trkin.mClusters[2], trkin.mClusters[3], trkin.mClusters[4], trkin.mClusters[5], trkin.mClusters[6]}}});
123  }
124  return 0;
125 }
