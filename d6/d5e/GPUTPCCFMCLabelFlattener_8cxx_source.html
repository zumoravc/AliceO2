30 
31  for (Row row = 0; row < GPUCA_ROW_COUNT; row++) {
32  cls.mPlabelHeaderOffset[row] = headerOffset;
33  headerOffset += cls.mPclusterInRow[row];
34 
35  cls.mPlabelDataOffset[row] = dataOffset;
36  if (cls.mPclusterInRow[row] > 0) {
37  auto& lastInterim = cls.mPlabelsByRow[cls.mNMaxClusterPerRow * row + cls.mPclusterInRow[row] - 1];
38  uint labelsInRow = lastInterim.offset + lastInterim.labels.size();
39  dataOffset += labelsInRow;
40  }
41  }
42 
43  labels.header.resize(headerOffset);
44  labels.data.resize(dataOffset);
45 }
