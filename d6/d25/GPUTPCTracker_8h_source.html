213  {
214  const float chi2_suppress = 6.f;
215  float weight = (((float)NHits * (chi2_suppress - chi2 / 500.f)) * (1e9f / chi2_suppress / 160.f));
216  if (weight < 0.f || weight > 2e9f) {
217  return 0;
218  }
219  return ((int)weight);
