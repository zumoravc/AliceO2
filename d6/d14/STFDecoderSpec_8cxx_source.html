56 template <class Mapping>
57 void STFDecoder<Mapping>::init(InitContext& ic)
58 {
59  mDecoder = std::make_unique<RawPixelDecoder<Mapping>>();
60  mDecoder->init();
61 
62  auto detID = Mapping::getDetID();
63  mNThreads = ic.options().get<int>("nthreads");
64  mDecoder->setNThreads(mNThreads);
65  mDecoder->setFormat(ic.options().get<bool>("old-format") ? GBTLink::OldFormat : GBTLink::NewFormat);
66  mDecoder->setVerbosity(ic.options().get<int>("decoder-verbosity"));
67  if (mDoClusters) {
68  o2::base::GeometryManager::loadGeometry(); // for generating full clusters
69  GeometryTGeo* geom = nullptr;
70  if (detID == o2::detectors::DetID::ITS) {
71  geom = o2::its::GeometryTGeo::Instance();
72  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L));
73  } else {
74  geom = o2::mft::GeometryTGeo::Instance();
75  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L));
76  }
77  mClusterer = std::make_unique<Clusterer>();
78  mClusterer->setGeometry(geom);
79  mClusterer->setNChips(Mapping::getNChips());
80 
81  // settings for the fired pixel overflow masking
82  const auto& alpParams = DPLAlpideParam<Mapping::getDetID()>::Instance();
83  const auto& clParams = ClustererParam<Mapping::getDetID()>::Instance();
84  mClusterer->setMaxBCSeparationToMask(alpParams.roFrameLength / o2::constants::lhc::LHCBunchSpacingNS + clParams.maxBCDiffToMaskBias);
85  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
86 
87  std::string dictFile = o2::base::NameConf::getDictionaryFileName(detID, mDictName, ".bin");
88  if (o2::base::NameConf::pathExists(dictFile)) {
89  mClusterer->loadDictionary(dictFile);
90  LOG(INFO) << mSelfName << " clusterer running with a provided dictionary: " << dictFile;
91  } else {
92  LOG(INFO) << mSelfName << " Dictionary " << dictFile << " is absent, " << Mapping::getName() << " clusterer expects cluster patterns";
93  }
94  mClusterer->print();
95  }

99 template <class Mapping>
100 void STFDecoder<Mapping>::run(ProcessingContext& pc)
101 {
102  int nSlots = pc.inputs().getNofParts(0);
103  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
104  mTimer.Start(false);
105  mDecoder->startNewTF(pc.inputs());
106  auto orig = o2::header::gDataOriginITS;
107  using CLUSVECDUMMY = std::vector<Cluster>;
108  std::vector<o2::itsmft::Cluster> clusVec;
109  std::vector<o2::itsmft::CompClusterExt> clusCompVec;
110  std::vector<o2::itsmft::ROFRecord> clusROFVec;
111  std::vector<unsigned char> clusPattVec;
112  std::vector<Digit> digVec;
113  std::vector<ROFRecord> digROFVec;
114  CLUSVECDUMMY* clusVecDUMMY = nullptr;
115  mDecoder->setDecodeNextAuto(false);
116  while (mDecoder->decodeNextTrigger()) {
117  if (mDoDigits) { // call before clusterization, since the latter will hide the digits
118  mDecoder->fillDecodedDigits(digVec, digROFVec); // lot of copying involved
119  }
120  if (mDoClusters) { // !!! THREADS !!!
121  mClusterer->process(mNThreads, *mDecoder.get(), (CLUSVECDUMMY*)nullptr, &clusCompVec, mDoPatterns ? &clusPattVec : nullptr, &clusROFVec);
122  }
123  }
124 
125  if (mDoDigits) {
126  pc.outputs().snapshot(Output{orig, "DIGITS", 0, Lifetime::Timeframe}, digVec);
127  pc.outputs().snapshot(Output{orig, "ITSDigitROF", 0, Lifetime::Timeframe}, digROFVec);
128  }
129  if (mDoClusters) { // we are not obliged to create vectors which are not requested, but other devices might not know the options of this one
130  pc.outputs().snapshot(Output{orig, "CLUSTERS", 0, Lifetime::Timeframe}, clusVec); // DUMMY!!!
131  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
132  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
133  pc.outputs().snapshot(Output{orig, "ITSClusterROF", 0, Lifetime::Timeframe}, clusROFVec);
134  }
135 
136  if (mDoClusters) {
137  LOG(INFO) << mSelfName << " Built " << clusCompVec.size() << " clusters in " << clusROFVec.size() << " ROFs";
138  }
139  if (mDoDigits) {
140  LOG(INFO) << mSelfName << " Decoded " << digVec.size() << " Digits in " << digROFVec.size() << " ROFs";
141  }
142  mTimer.Stop();
143  LOG(INFO) << mSelfName << " Total time for TF " << mTFCounter << " : CPU: " << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0;
144  mTFCounter++;
145 }

164 std::unique_ptr<Clusterer> STFDecoder<Mapping>::setupClusterer(const std::string& dictName)
165 {
166  bool isContinuous = true;
167  const auto grp = o2::parameters::GRPObject::loadFrom(o2::base::NameConf::getGRPFileName());
168  if (grp) {
169  isContinuous = grp->isDetContinuousReadOut(Mapping::getName());
170  } else {
171  throw std::runtime_error("failed to retrieve GRP");
172  }
173  // settings for the fired pixel overflow masking
174  const auto& alpParams = DPLAlpideParam<Mapping::getDetID()>::Instance();
175  int maskNBC = alpParams.roFrameLength / o2::constants::lhc::LHCBunchSpacingNS + 10;
176  mClusterer = createClusterer(Mapping::getNChips(), maskNBC, isContinuous, "");
177  mClusterer->print();

180 DataProcessorSpec getSTFDecoderITSSpec(bool doClusters, bool doPatterns, bool doDigits, const std::string& dict)
181 {
182  std::vector<OutputSpec> outputs;
183  auto orig = o2::header::gDataOriginITS;
184 
185  if (doDigits) {
186  outputs.emplace_back(orig, "DIGITS", 0, Lifetime::Timeframe);
187  outputs.emplace_back(orig, "ITSDigitROF", 0, Lifetime::Timeframe);
188  }
189  if (doClusters) {
190  outputs.emplace_back(orig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
191  outputs.emplace_back(orig, "ITSClusterROF", 0, Lifetime::Timeframe);
192  // in principle, we don't need to open this input if we don't need to send real data,
193  // but other devices expecting it do not know about options of this device: problem?
194  // if (doClusters && doPatterns)
195  outputs.emplace_back(orig, "PATTERNS", 0, Lifetime::Timeframe); // RSTODO: DUMMY, FULL CLUSTERS ARE BEING ELIMINATED
196  //
197  outputs.emplace_back(orig, "CLUSTERS", 0, Lifetime::Timeframe); // RSTODO: DUMMY, FULL CLUSTERS ARE BEING ELIMINATED
198  }
199 
200  return DataProcessorSpec{
201  "its-stf-decoder",
202  Inputs{{"stf", ConcreteDataTypeMatcher{orig, "RAWDATA"}, Lifetime::Timeframe}},
203  outputs,
204  AlgorithmSpec{adaptFromTask<STFDecoder<ChipMappingITS>>(doClusters, doPatterns, doDigits, dict)},
205  Options{
206  {"nthreads", VariantType::Int, 0, {"Number of decoding/clustering threads (<1: rely on openMP default)"}},
207  {"old-format", VariantType::Bool, false, {"Use old format (1 trigger per CRU page)"}},
208  {"decoder-verbosity", VariantType::Int, 0, {"Verbosity level (-1: silent, 0: errors, 1: headers, 2: data)"}}}};
