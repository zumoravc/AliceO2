100 void DigitizerTask::FinishTask()
101 {
102  // finalize digitization, if needed, flash remaining digits
103 
104  if (!mDigitizer.getParams().isContinuous()) {
105  return;
106  }
107 
108  // is there QED backgroung provided? Fill QED slots up to the end of reserved ROFrames
109  if (mQEDBranch) {
110  processQEDBackground(mDigitizer.getEndTimeOfROFMax());
111  }
112 
113  FairRootManager* mgr = FairRootManager::Instance();
114  mgr->SetLastFill(kTRUE);
115  mDigitsArray.clear();
116  mMCTruthArray.clear();
117  mDigitizer.fillOutputContainer();

121 void DigitizerTask::processQEDBackground(double tMax)
122 {
123  // process QED time-slots until provided collision time (in ns)
124 
125  double tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS;
126 
127  while (tQEDNext < tMax) {
128  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
129  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
130  if (++mLastQEDEntry >= mQEDBranch->GetEntries()) {
131  mLastQEDEntry = 0; // wrapp if needed
132  }
133  mQEDBranch->GetEntry(mLastQEDEntry);
134  mDigitizer.setEventTime(mLastQEDTimeNS);
135  mDigitizer.process(mHitsArrayQED, mLastQEDEntry, mQEDSourceID);
136  //
137  }
138 }
