96 BOOST_AUTO_TEST_CASE(FastTransform_test_setSpaceChargeCorrection)
97 {
98 
99  std::unique_ptr<TPCFastTransform> fastTransform0(TPCFastTransformHelperO2::instance()->create(0));
100 
101  auto correctionXUV = [&](int roc, const float /*x*/, const float u, const float v, float& dX, float& dU, float& dV) {
102  dX = 1. + 1 * u + 0.1 * u * u;
103  dU = 2. + 0.2 * u + 0.002 * u * u; // + 0.001 * u * u * u;
104  dV = 3. + 0.1 * v + 0.01 * v * v; //+ 0.0001 * v * v * v;
105  };
106 
107  auto correctionGlobal = [&](int roc, const double XYZ[3], double dXdYdZ[3]) {
108  const TPCFastTransformGeo& geo = fastTransform0->getGeometry();
109  float lx, ly, lz, u, v, dx, du, dv, gx, gy, gz;
110  geo.convGlobalToLocal(roc, XYZ[0], XYZ[1], XYZ[2], lx, ly, lz);
111  geo.convLocalToUV(roc, ly, lz, u, v);
112  correctionXUV(roc, lx, u, v, dx, du, dv);
113  lx += dx;
114  geo.convUVtoLocal(roc, u + du, v + dv, ly, lz);
115  geo.convLocalToGlobal(roc, lx, ly, lz, gx, gy, gz);
116  dXdYdZ[0] = gx - XYZ[0];
117  dXdYdZ[1] = gy - XYZ[1];
118  dXdYdZ[2] = gz - XYZ[2];
119  };
120 
121  TPCFastTransformHelperO2::instance()->setSpaceChargeCorrection(correctionGlobal);
122 
123  std::unique_ptr<TPCFastTransform> fastTransform(TPCFastTransformHelperO2::instance()->create(0));
124

126
