26 void ColumnDataToLocalBoard::process(gsl::span<const ColumnData> data)
27 {
29  mLocalBoardsMap.clear();
30  mGBTMap.clear();
31 
32  // First fill the map with the active local boards.
33  // Each local board gets a unique id.
34  for (auto& col : data) {
35  for (int iline = mMapping.getFirstBoardBP(col.columnId, col.deId); iline <= mMapping.getLastBoardBP(col.columnId, col.deId); ++iline) {
36  if (col.getBendPattern(iline) || col.getNonBendPattern()) {
37  auto uniqueLocId = mCrateMapper.deLocalBoardToRO(col.deId, col.columnId, iline);
38  auto& roData = mLocalBoardsMap[uniqueLocId];
39  roData.statusWord = raw::sSTARTBIT | raw::sCARDTYPE;
40  roData.boardId = crateparams::getLocId(uniqueLocId);
41  int ich = detparams::getChamber(col.deId);
42  roData.firedChambers |= (1 << ich);
43  roData.patternsBP[ich] = col.getBendPattern(iline);
44  roData.patternsNBP[ich] = col.getNonBendPattern();
45  }
46  }
47  }
48 
49  // Then group the boards belonging to the same GBT link
50  for (auto& item : mLocalBoardsMap) {
51  auto crateId = crateparams::getCrateId(item.first);
52  auto feeId = crateparams::makeROId(crateId, crateparams::getGBTIdFromBoardInCrate(item.second.boardId));
53  mGBTMap[feeId].emplace_back(item.second);
54  }
55 }
