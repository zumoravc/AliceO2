136 bool areClustersEqual(const Cluster2D& cl1, const Cluster2D& cl2)
137 {
138  int nBad = 0;
139  float precision = 1.e-3;
140  if (cl1.deId != cl2.deId) {
141  std::cerr << "Id: " << (int)cl1.deId << " != " << (int)cl2.deId << std::endl;
142  ++nBad;
143  }
144 
145  if (std::abs(cl1.xCoor - cl2.xCoor) > precision) {
146  std::cerr << "xCoor: " << cl1.xCoor << " != " << cl2.xCoor << std::endl;
147  ++nBad;
148  }
149 
150  if (std::abs(cl1.yCoor - cl2.yCoor) > precision) {
151  std::cerr << "yCoor: " << cl1.yCoor << " != " << cl2.yCoor << std::endl;
152  ++nBad;
153  }
154 
155  if (std::abs(cl1.sigmaX2 - cl2.sigmaX2) > precision) {
156  std::cerr << "sigmaX2: " << cl1.sigmaX2 << " != " << cl2.sigmaX2 << std::endl;
157  ++nBad;
158  }
159 
160  if (std::abs(cl1.sigmaY2 - cl2.sigmaY2) > precision) {
161  std::cerr << "sigmaY2: " << cl1.sigmaY2 << " != " << cl2.sigmaY2 << std::endl;
162  ++nBad;
163  }
164 
165  return (nBad == 0);
166 }

198 bool isWithinUncertainties(float xPos, float yPos, const Cluster2D& cl)
199 {
200  std::string str[2] = {"x", "y"};
201  float inputPos[2] = {xPos, yPos};
202  float recoPos[2] = {cl.xCoor, cl.yCoor};
203  float sigma2[2] = {cl.sigmaX2, cl.sigmaY2};
204  bool isOk = true;
205  for (int icoor = 0; icoor < 2; ++icoor) {
206  float sigma = std::sqrt(sigma2[icoor]);
207  if (sigma > 5. || std::abs(recoPos[icoor] - inputPos[icoor]) > 5 * sigma) {
208  std::cerr << str[icoor] << " position input " << inputPos[icoor] << " reco " << recoPos[icoor] << " sigma "
209  << sigma << std::endl;
210  isOk = false;
211  }
212  }
213  return isOk;
214 }

216 std::vector<ColumnData> getFiredStrips(float xPos, float yPos, int deId, Mapping& mapping)
217 {
218  // This is a quite simple case just for testing purposes.
219  // The fired strips are simply the fired strip itself + its neighbours.
220  // However, in the bending plane, this also consists of strips with no overlap
221  // with the non-bending plane
222  std::vector<ColumnData> columns;
223  for (int cathode = 1; cathode >= 0; --cathode) {
224  Mapping::MpStripIndex stripIndex = mapping.stripByPosition(xPos, yPos, cathode, deId, false);
225  if (!stripIndex.isValid()) {
226  continue;
227  }
228  std::vector<Mapping::MpStripIndex> neighbours = mapping.getNeighbours(stripIndex, cathode, deId);
229  neighbours.push_back(stripIndex);
230  for (auto& neigh : neighbours) {
231  ColumnData* columnStruct = nullptr;
232  for (auto& currCol : columns) {
233  if (currCol.columnId == neigh.column) {
234  columnStruct = &currCol;
235  break;
236  }
237  }
238  if (!columnStruct) {
239  if (cathode == 0) {
240  // For the sake of simplicity we reject the neighbour bending strips
241  // that do not overlap to the non-bending plane
242  continue;
243  }
244  columns.emplace_back(ColumnData());
245  columnStruct = &columns.back();
246  columnStruct->deId = deId;
247  columnStruct->columnId = neigh.column;
248  }
249  columnStruct->addStrip(neigh.strip, cathode, neigh.line);
250  }
251  }
252  return columns;
253 }
