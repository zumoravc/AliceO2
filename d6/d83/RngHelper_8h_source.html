43  // static function to get a true random number from /dev/urandom
44  template <typename T>
45  static T readURandom()
46  {
47  int randomDataHandle = open("/dev/urandom", O_RDONLY);
48  if (randomDataHandle < 0) {
49  // something went wrong
50  } else {
51  T seed;
52  auto result = read(randomDataHandle, &seed, sizeof(T));
53  if (result < 0) {
54  // something went wrong
55  }
56  close(randomDataHandle);
57  return seed;
58  }
59  return T(0);
60  }
