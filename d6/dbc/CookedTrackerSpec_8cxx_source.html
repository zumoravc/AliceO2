51 void CookedTrackerDPL::init(InitContext& ic)
52 {
53  mTimer.Stop();
54  mTimer.Reset();
55  auto nthreads = ic.options().get<int>("nthreads");
56  mTracker.setNumberOfThreads(nthreads);
57  auto filename = ic.options().get<std::string>("grp-file");
58  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
59  if (grp) {
60  mGRP.reset(grp);
61  o2::base::Propagator::initFieldFromGRP(grp);
62  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
63 
64  o2::base::GeometryManager::loadGeometry();
65  o2::its::GeometryTGeo* geom = o2::its::GeometryTGeo::Instance();
66  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L, o2::TransformType::T2GRot,
67  o2::TransformType::T2G));
68  mTracker.setGeometry(geom);
69 
70  double origD[3] = {0., 0., 0.};
71  mTracker.setBz(field->getBz(origD));
72 
73  bool continuous = mGRP->isDetContinuousReadOut("ITS");
74  LOG(INFO) << "ITSCookedTracker RO: continuous=" << continuous;
75  mTracker.setContinuousMode(continuous);
76  } else {
77  throw std::runtime_error(o2::utils::concat_string("Cannot retrieve GRP from the ", filename));
78  }
79 
80  std::string dictPath = ic.options().get<std::string>("its-dictionary-path");
81  std::string dictFile = o2::base::NameConf::getDictionaryFileName(o2::detectors::DetID::ITS, dictPath, ".bin");
82  if (o2::base::NameConf::pathExists(dictFile)) {
83  mDict.readBinaryFile(dictFile);
84  LOG(INFO) << "Tracker running with a provided dictionary: " << dictFile;
85  } else {
86  LOG(INFO) << "Dictionary " << dictFile << " is absent, Tracker expects cluster patterns";
87  }
88 }

90 void CookedTrackerDPL::run(ProcessingContext& pc)
91 {
92  mTimer.Start(false);
93  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
94  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
95  auto clusters = pc.inputs().get<gsl::span<o2::itsmft::Cluster>>("clusters");
96 
97  // code further down does assignment to the rofs and the altered object is used for output
98  // we therefore need a copy of the vector rather than an object created directly on the input data,
99  // the output vector however is created directly inside the message memory thus avoiding copy by
100  // snapshot
101  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
102  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
103 
104  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> labels;
105  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
106  if (mUseMC) {
107  labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
108  // get the array as read-onlt span, a snapshot is send forward
109  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
110  }
111  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
112  FastMultEst multEst; // mult estimator
113 
114  LOG(INFO) << "ITSCookedTracker pulled " << compClusters.size() << " clusters, in " << rofs.size() << " RO frames";
115 
116  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackLabels;
117  if (mUseMC) {
118  mTracker.setMCTruthContainers(labels.get(), &trackLabels);
119  }
120 
121  o2::its::VertexerTraits vertexerTraits;
122  o2::its::Vertexer vertexer(&vertexerTraits);
123  o2::its::ROframe event(0);
124 
125  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0, Lifetime::Timeframe});
126  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0, Lifetime::Timeframe});
127  auto& tracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0, Lifetime::Timeframe});
128  auto& clusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0, Lifetime::Timeframe});
129 
130  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
131  for (auto& rof : rofs) {
132  auto& vtxROF = vertROFvec.emplace_back(rof); // register entry and number of vertices in the
133  vtxROF.setFirstEntry(vertices.size());
134  vtxROF.setNEntries(0);
135 
136  auto it = pattIt;
137  o2::its::ioutils::loadROFrameData(rof, event, compClusters, pattIt, mDict, labels.get());
138 
139  // fast cluster mult. cut if asked (e.g. sync. mode)
140  if (rof.getNEntries() && (multEstConf.cutMultClusLow > 0 || multEstConf.cutMultClusHigh > 0)) { // cut was requested
141  auto mult = multEst.process(rof.getROFData(compClusters));
142  if (mult < multEstConf.cutMultClusLow || (multEstConf.cutMultClusHigh > 0 && mult > multEstConf.cutMultClusHigh)) {
143  LOG(INFO) << "Estimated cluster mult. " << mult << " is outside of requested range "
144  << multEstConf.cutMultClusLow << " : " << multEstConf.cutMultClusHigh << " | ROF " << rof.getBCData();
145  rof.setFirstEntry(tracks.size());
146  rof.setNEntries(0);
147  continue;
148  }
149  }
150 
151  vertexer.clustersToVertices(event);
152  auto vtxVecLoc = vertexer.exportVertices();
153 
154  if (multEstConf.cutMultVtxLow > 0 || multEstConf.cutMultVtxHigh > 0) { // cut was requested
155  std::vector<o2::dataformats::Vertex<o2::dataformats::TimeStamp<int>>> vtxVecSel;
156  vtxVecSel.swap(vtxVecLoc);
157  for (const auto& vtx : vtxVecSel) {
158  if (vtx.getNContributors() < multEstConf.cutMultVtxLow || (multEstConf.cutMultVtxHigh > 0 && vtx.getNContributors() > multEstConf.cutMultVtxHigh)) {
159  LOG(INFO) << "Found vertex mult. " << vtx.getNContributors() << " is outside of requested range "
160  << multEstConf.cutMultVtxLow << " : " << multEstConf.cutMultVtxHigh << " | ROF " << rof.getBCData();
161  continue; // skip vertex of unwanted multiplicity
162  }
163  vtxVecLoc.push_back(vtx);
164  }
165  }
166  if (vtxVecLoc.empty()) {
167  if (multEstConf.cutMultVtxLow < 1) { // do blind search only if there is no cut on the low mult vertices
168  vtxVecLoc.emplace_back();
169  } else {
170  rof.setFirstEntry(tracks.size());
171  rof.setNEntries(0);
172  continue;
173  }
174  } else { // save vetrices
175  vtxROF.setNEntries(vtxVecLoc.size());
176  for (const auto& vtx : vtxVecLoc) {
177  vertices.push_back(vtx);
178  }
179  }
180  mTracker.setVertices(vtxVecLoc);
181  mTracker.process(compClusters, it, mDict, tracks, clusIdx, rof);
182  }
183 
184  LOG(INFO) << "ITSCookedTracker pushed " << tracks.size() << " tracks";
185 
186  if (mUseMC) {
187  pc.outputs().snapshot(Output{"ITS", "TRACKSMCTR", 0, Lifetime::Timeframe}, trackLabels);
188  pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
189  }
190  mTimer.Stop();
191 }

199 DataProcessorSpec getCookedTrackerSpec(bool useMC)
200 {
201  std::vector<InputSpec> inputs;
202  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
203  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
204  inputs.emplace_back("clusters", "ITS", "CLUSTERS", 0, Lifetime::Timeframe);
205  inputs.emplace_back("ROframes", "ITS", "ITSClusterROF", 0, Lifetime::Timeframe);
206 
207  std::vector<OutputSpec> outputs;
208  outputs.emplace_back("ITS", "TRACKS", 0, Lifetime::Timeframe);
209  outputs.emplace_back("ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
210  outputs.emplace_back("ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
211  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
212  outputs.emplace_back("ITS", "VERTICESROF", 0, Lifetime::Timeframe);
213 
214  if (useMC) {
215  inputs.emplace_back("labels", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
216  inputs.emplace_back("MC2ROframes", "ITS", "ITSClusterMC2ROF", 0, Lifetime::Timeframe);
217  outputs.emplace_back("ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
218  outputs.emplace_back("ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
219  }
220 
221  return DataProcessorSpec{
222  "its-cooked-tracker",
223  inputs,
224  outputs,
225  AlgorithmSpec{adaptFromTask<CookedTrackerDPL>(useMC)},
226  Options{
227  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
228  {"its-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
229  {"nthreads", VariantType::Int, 1, {"Number of threads"}}}};
230 }
