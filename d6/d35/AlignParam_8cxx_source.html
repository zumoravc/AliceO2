57 void AlignParam::setMatrixRotation(double psi, double theta, double phi, TGeoHMatrix& dest) const
58 {
60  double rot[9] = {};
61  Double_t sinpsi = std::sin(psi);
62  Double_t cospsi = std::cos(psi);
63  Double_t sinthe = std::sin(theta);
64  Double_t costhe = std::cos(theta);
65  Double_t sinphi = std::sin(phi);
66  Double_t cosphi = std::cos(phi);
67  rot[0] = costhe * cosphi;
68  rot[1] = -costhe * sinphi;
69  rot[2] = sinthe;
70  rot[3] = sinpsi * sinthe * cosphi + cospsi * sinphi;
71  rot[4] = -sinpsi * sinthe * sinphi + cospsi * cosphi;
72  rot[5] = -costhe * sinpsi;
73  rot[6] = -cospsi * sinthe * cosphi + sinpsi * sinphi;
74  rot[7] = cospsi * sinthe * sinphi + sinpsi * cosphi;
75  rot[8] = costhe * cospsi;
76  dest.SetRotation(rot);
77 }

115 bool AlignParam::matrixToAngles(const double* rot, double& psi, double& theta, double& phi)
116 {
121  //
122  if (std::abs(rot[0]) < 1e-7 || std::abs(rot[8]) < 1e-7) {
123  LOG(ERROR) << "Failed to extract roll-pitch-yall angles!";
124  return false;
125  }
126  psi = std::atan2(-rot[5], rot[8]);
127  theta = std::asin(rot[2]);
128  phi = std::atan2(-rot[1], rot[0]);
129  return true;

149 bool AlignParam::setLocalParams(const TGeoMatrix& m)
150 {
151  // Set the global delta transformation by passing the TGeo matrix
152  // for the local delta transformation.
153  // In case that the TGeo was not initialized or not closed,
154  // returns false and the object parameters are not set.
155  //
156  if (!gGeoManager || !gGeoManager->IsClosed()) {
157  LOG(ERROR) << "Can't set the local alignment object parameters! gGeoManager doesn't exist or it is still open!";
158  return false;
159  }
160 
161  const char* symname = getSymName();
162  TGeoHMatrix gprime, gprimeinv;
163  TGeoPhysicalNode* pn = nullptr;
164  TGeoPNEntry* pne = gGeoManager->GetAlignableEntry(symname);
165  if (pne) {
166  pn = pne->GetPhysicalNode();
167  if (pn) {
168  if (pn->IsAligned()) {
169  LOG(WARNING) << "Volume " << symname << " has been misaligned already!";
170  }
171  gprime = *pn->GetMatrix();
172  } else {
173  gprime = pne->GetGlobalOrig();
174  }
175  } else {
176  LOG(WARNING) << "The symbolic volume name " << symname
177  << " does not correspond to a physical entry. Using it as volume path!";
178  if (!gGeoManager->cd(symname)) {
179  LOG(ERROR) << "Volume name or path " << symname << " is not valid!";
180  return false;
181  }
182  gprime = *gGeoManager->GetCurrentMatrix();
183  }
184 
185  TGeoHMatrix m1; // the TGeoHMatrix copy of the local delta "m"
186  m1.SetTranslation(m.GetTranslation());
187  m1.SetRotation(m.GetRotationMatrix());
188 
189  gprimeinv = gprime.Inverse();
190  m1.Multiply(&gprimeinv);
191  m1.MultiplyLeft(&gprime);
192 
193  return setLocalParams(m1);
194 }

197 bool AlignParam::createLocalMatrix(TGeoHMatrix& m) const
198 {
199  // Get the matrix for the local delta transformation.
200  // In case that the TGeo was not initialized or not closed,
201  // returns false and the object parameters are not set.
202  //
203  if (!gGeoManager || !gGeoManager->IsClosed()) {
204  LOG(ERROR) << "Can't get the local alignment object parameters! gGeoManager doesn't exist or it is still open!";
205  return false;
206  }
207 
208  const char* symname = getSymName();
209  TGeoPhysicalNode* node;
210  TGeoPNEntry* pne = gGeoManager->GetAlignableEntry(symname);
211  if (pne) {
212  if (!pne->GetPhysicalNode()) {
213  node = gGeoManager->MakeAlignablePN(pne);
214  } else {
215  node = pne->GetPhysicalNode();
216  }
217  } else {
218  LOG(WARNING) << "The symbolic volume name " << symname
219  << " does not correspond to a physical entry. Using it as volume path!";
220  node = (TGeoPhysicalNode*)gGeoManager->MakePhysicalNode(symname);
221  }
222 
223  if (!node) {
224  LOG(ERROR) << "Volume name or path " << symname << " is not valid!";
225  return false;
226  }
227  m = createMatrix();
228  TGeoHMatrix gprime, gprimeinv;
229  gprime = *node->GetMatrix();
230  gprimeinv = gprime.Inverse();
231  m.Multiply(&gprime);
232  m.MultiplyLeft(&gprimeinv);
233 
234  return true;
235 }

238 bool AlignParam::applyToGeometry(bool ovlpcheck, double ovlToler)
239 {
243  //
244  if (!gGeoManager || !gGeoManager->IsClosed()) {
245  LOG(ERROR) << "Can't apply the alignment object! gGeoManager doesn't exist or it is still open!";
246  return false;
247  }
248 
249  if (gGeoManager->IsLocked()) {
250  LOG(ERROR) << "Can't apply the alignment object! Geometry is locked!";
251  return false;
252  }
253 
254  const char* symname = getSymName();
255  const char* path;
256  TGeoPhysicalNode* node;
257  TGeoPNEntry* pne = gGeoManager->GetAlignableEntry(symname);
258  if (pne) {
259  path = pne->GetTitle();
260  node = gGeoManager->MakeAlignablePN(pne);
261  } else {
262  LOG(DEBUG) << "The symbolic volume name " << symname
263  << " does not correspond to a physical entry. Using it as a volume path!";
264  path = symname;
265  if (!gGeoManager->CheckPath(path)) {
266  LOG(ERROR) << "Volume path " << path << " is not valid";
267  return false;
268  }
269  if (gGeoManager->GetListOfPhysicalNodes()->FindObject(path)) {
270  LOG(ERROR) << "Volume path " << path << " has been misaligned already!";
271  return false;
272  }
273  node = (TGeoPhysicalNode*)gGeoManager->MakePhysicalNode(path);
274  }
275 
276  if (!node) {
277  LOG(ERROR) << "Volume path " << path << " is not valid";
278  return false;
279  }
280 
281  // Double_t threshold = 0.001;
282 
283  TGeoHMatrix gprime = *node->GetMatrix();
284  TGeoHMatrix align = createMatrix();
285  gprime.MultiplyLeft(&align);
286  TGeoHMatrix* ginv = new TGeoHMatrix; // TGeoPhysicalNode takes and manages raw pointer, need naked new!
287  TGeoHMatrix* g = node->GetMatrix(node->GetLevel() - 1);
288  *ginv = g->Inverse();
289  *ginv *= gprime;
290 
291  LOG(DEBUG) << "Aligning volume " << symname;
292 
293  if (ovlpcheck) {
294  node->Align(ginv, nullptr, true, ovlToler);
295  TObjArray* ovlpArray = gGeoManager->GetListOfOverlaps();
296  Int_t nOvlp = ovlpArray->GetEntriesFast();
297  if (nOvlp) {
298  LOG(INFO) << "Misalignment of node " << node->GetName() << " generated the following " << nOvlp
299  << "overlaps/extrusions:";
300  for (int i = 0; i < nOvlp; i++) {
301  ((TGeoOverlap*)ovlpArray->UncheckedAt(i))->PrintInfo();
302  }
303  }
304  } else {
305  node->Align(ginv, nullptr, false, ovlToler);
306  }
307 
308  return true;
309 }
