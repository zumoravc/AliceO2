107 static void BM_mergingCollectionsTHNSparse(benchmark::State& state)
108 {
109  size_t collectionSize = state.range(0);
110 
111  const Double_t min = 0.0;
112  const Double_t max = 1000000.0;
113  const size_t dim = 10;
114  const Int_t bins = 250;
115  const Int_t binsDims[dim] = {bins, bins, bins, bins, bins, bins, bins, bins, bins, bins};
116  const Double_t mins[dim] = {min, min, min, min, min, min, min, min, min, min};
117  const Double_t maxs[dim] = {max, max, max, max, max, max, max, max, max, max};
118 
119  TRandomMixMax gen;
120  gen.SetSeed(std::random_device()());
121  Double_t randomArray[dim];
122 
123  for (auto _ : state) {
124 
125  state.PauseTiming();
126  TCollection* collection = new TObjArray();
127  collection->SetOwner(true);
128  for (size_t i = 0; i < collectionSize; i++) {
129 
130  auto* h = new THnSparseI(("test" + std::to_string(i)).c_str(), "test", dim, binsDims, mins, maxs);
131  for (size_t entry = 0; entry < 50000; entry++) {
132  gen.RndmArray(dim, randomArray);
133  for (double& r : randomArray) {
134  r *= max;
135  }
136  h->Fill(randomArray);
137  }
138  collection->Add(h);
139  }
140  auto* m = new THnSparseI("merged", "merged", dim, binsDims, mins, maxs);
141 
142  state.ResumeTiming();
143  m->Merge(collection);
144 
145  state.PauseTiming();
146 
147  delete collection;
148  delete m;
149  }

152 static void BM_mergingCollectionsTTree(benchmark::State& state)
153 {
154  size_t collectionSize = state.range(0);
155 
156  struct format1 {
157  Int_t a;
158  Long64_t b;
159  Float_t c;
160  Double_t d;
161  } branch1;
162  ULong64_t branch2;
163 
164  auto createTree = [&](std::string name) -> TTree* {
165  TTree* tree = new TTree();
166  tree->SetName(name.c_str());
167  tree->Branch("b1", &branch1, "a/I:b/L:c/F:d/D");
168  tree->Branch("b2", &branch2);
169  return tree;
170  };
171 
172  TRandomMixMax gen;
173  gen.SetSeed(std::random_device()());
174  Double_t randomArray[5];
175 
176  for (auto _ : state) {
177  state.PauseTiming();
178  TCollection* collection = new TObjArray();
179  collection->SetOwner(true);
180  for (size_t i = 0; i < collectionSize; i++) {
181  TTree* t = createTree(std::to_string(i));
182  for (size_t entry = 0; entry < 6250; entry++) {
183  gen.RndmArray(5, randomArray);
184  branch1 = {static_cast<Int_t>(randomArray[0]), static_cast<Long64_t>(randomArray[1]), static_cast<Float_t>(randomArray[2]), randomArray[3]};
185  branch2 = randomArray[4];
186  t->Fill();
187  }
188  collection->Add(t);
189  }
190  TTree* merged = createTree("merged");
191 
192  state.ResumeTiming();
193  merged->Merge(collection);
194 
195  state.PauseTiming();
196 
197  delete collection;
198  delete merged;
199  }

202 static void BM_mergingPODCollections(benchmark::State& state)
203 {
204  size_t collectionSize = state.range(0);
205  size_t bins = 62500; // makes 250kB
206 
207  std::vector<std::vector<float>*> collection;
208  TF1* uni = new TF1("uni", "1", 0, 1000000);
209 
210  const size_t randoms = 50000;
211  TRandomMixMax gen;
212  gen.SetSeed(std::random_device()());
213  Double_t randomArray[randoms];
214 
215  for (size_t i = 0; i < collectionSize; i++) {
216  auto* v = new std::vector<float>(bins, 0);
217  gen.RndmArray(randoms, randomArray);
218  for (double r : randomArray) {
219  size_t idx = r * bins;
220  if (idx != bins) {
221  (*v)[idx] += 1;
222  }
223  }
224  collection.push_back(v);
225  }
226 
227  auto* m = new std::vector<float>(bins, 0);
228 
229  auto merge = [&](size_t i) {
230  auto* v = collection[i];
231  for (size_t b = 0; b < bins; b++) {
232  (*m)[b] += (*v)[b];
233  }
234  };
235 
236  for (auto _ : state) {
237  for (size_t i = 0; i < collectionSize; i++) {
238  merge(i);
239  }
240  }
241 
242  for (size_t i = 0; i < collectionSize; i++) {
243  delete collection[i];
244  }
245  delete m;
246  delete uni;
247 }
