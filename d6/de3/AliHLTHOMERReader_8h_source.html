90  1: homer_uint32

103  /* Return the origin of the data in the current event's data
104  block with the given block index (starting at 0). */
105  virtual homer_uint32 GetBlockDataOrigin(unsigned long ndx) const = 0;
106  /* Return a specification of the data in the current event's data
107  block with the given block index (starting at 0). */
108  virtual homer_uint32 GetBlockDataSpec(unsigned long ndx) const = 0;
109
