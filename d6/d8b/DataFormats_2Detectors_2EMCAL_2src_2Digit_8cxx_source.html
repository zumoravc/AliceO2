22 Digit& Digit::operator+=(const Digit& other)
23 {
24  if (canAdd(other)) {
25  Int_t a = getAmplitudeADC() + other.getAmplitudeADC();
26  Short_t s = (Short_t)(a);
27 
28  if (a < -0x8000)
29  setAmplitudeADC(-0x8000);
30  else if (a <= constants::EMCAL_HGLGTRANSITION)
31  setAmplitudeADC(s);
32  else if (a <= 0x7fff * constants::EMCAL_HGLGFACTOR)
33  setAmplitudeADC(s / constants::EMCAL_HGLGFACTOR, ChannelType_t::LOW_GAIN);
34  else
35  setAmplitudeADC(0x7fff, ChannelType_t::LOW_GAIN);
36  }
37  // Does nothing if the digits are in different towers or have incompatible times.
38  return *this;
39 }

41 void Digit::setAmplitudeADC(Short_t amplitude, ChannelType_t ctype)
42 {
43  if (ctype == ChannelType_t::LOW_GAIN) {
44  if (amplitude > constants::EMCAL_HGLGTRANSITION / constants::EMCAL_HGLGFACTOR) {
45  mAmplitude = amplitude;
46  mChannelType = ChannelType_t::LOW_GAIN;
47  } else if (amplitude < -0x8000 / constants::EMCAL_HGLGFACTOR) {
48  mAmplitude = -0x8000;
49  mChannelType = ChannelType_t::HIGH_GAIN;
50  } else {
51  mAmplitude = amplitude * constants::EMCAL_HGLGFACTOR;
52  mChannelType = ChannelType_t::HIGH_GAIN;
53  }
54  } else {
55  if (amplitude > constants::EMCAL_HGLGTRANSITION) {
56  mAmplitude = amplitude / constants::EMCAL_HGLGFACTOR;
57  mChannelType = ChannelType_t::LOW_GAIN;
58  } else {
59  mAmplitude = amplitude;
60  mChannelType = ctype;
61  }
62  }
63 }

65 Int_t Digit::getAmplitudeADC(ChannelType_t ctype) const
66 {
67  if (ctype == ChannelType_t::LOW_GAIN) {
68  // return in units of Low-Gain ADC counts
69  if (mChannelType == ChannelType_t::LOW_GAIN)
70  return (Int_t)(mAmplitude);
71  else
72  return mAmplitude / constants::EMCAL_HGLGFACTOR;
73  }
74 
75  // return in units of High-Gain ADC counts (default)
76  if (mChannelType == ChannelType_t::LOW_GAIN)
77  return mAmplitude * constants::EMCAL_HGLGFACTOR;
78 
79  return (Int_t)(mAmplitude);
80 }

82 void Digit::setAmplitude(Double_t amplitude)
83 {
84  if (amplitude < -0x8000 * constants::EMCAL_ADCENERGY) {
85  setAmplitudeADC(-0x8000, ChannelType_t::HIGH_GAIN);
86  } else if (amplitude <= constants::EMCAL_HGLGTRANSITION * constants::EMCAL_ADCENERGY) {
87  Short_t a = (Short_t)(std::floor(amplitude / constants::EMCAL_ADCENERGY));
88  setAmplitudeADC(a, ChannelType_t::HIGH_GAIN);
89  } else if (amplitude <= 0x7fff * constants::EMCAL_ADCENERGY * constants::EMCAL_HGLGFACTOR) {
90  Short_t a = (Short_t)(std::floor(amplitude / (constants::EMCAL_ADCENERGY * constants::EMCAL_HGLGFACTOR)));
91  setAmplitudeADC(a, ChannelType_t::LOW_GAIN);
92  } else {
93  setAmplitudeADC(0x7fff, ChannelType_t::LOW_GAIN);
94  }
95 }
