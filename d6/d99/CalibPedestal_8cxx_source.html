58 Int_t CalibPedestal::updateROC(const Int_t roc, const Int_t row, const Int_t pad,
59  const Int_t timeBin, const Float_t signal)
60 {
61  Int_t adcValue = Int_t(signal);
62  if (timeBin < mFirstTimeBin || timeBin > mLastTimeBin)
63  return 0;
64  if (adcValue < mADCMin || adcValue > mADCMax)
65  return 0;
66 
67  const GlobalPadNumber padInROC = mMapper.getPadNumberInROC(PadROCPos(roc, row, pad));
68  Int_t bin = padInROC * mNumberOfADCs + (adcValue - mADCMin);
69  vectorType& adcVec = *getVector(ROC(roc), kTRUE);
70  ++(adcVec[bin]);
71 
72  //printf("bin: %5d, val: %.2f\n", bin, adcVec[bin]);
73 
74  return 0;
75 }

94 //______________________________________________________________________________
95 void CalibPedestal::analyse()
96 {
97  ROC roc;
98 
99  std::vector<float> fitValues;
100 
101  for (auto& vecPtr : mADCdata) {
102  auto vec = vecPtr.get();
103  if (!vec) {
104  ++roc;
105  continue;
106  }
107 
108  CalROC& calROCPedestal = mPedestal.getCalArray(roc);
109  CalROC& calROCNoise = mNoise.getCalArray(roc);
110 
111  float* array = vec->data();
112 
113  const size_t numberOfPads = (roc.rocType() == RocType::IROC) ? mMapper.getPadsInIROC() : mMapper.getPadsInOROC();
114 
115  float pedestal{};
116  float noise{};
117 
118  TF1 fg("fg", "gaus");
119  fg.SetRange(mADCMin - 0.5f, mADCMax + 1.5f);
120 
121  for (Int_t ichannel = 0; ichannel < numberOfPads; ++ichannel) {
122  size_t offset = ichannel * mNumberOfADCs;
123  if (mStatisticsType == StatisticsType::GausFit) {
124  fit(mNumberOfADCs, array + offset, float(mADCMin) - 0.5f, float(mADCMax + 1) - 0.5f, fg); // -0.5 since ADC values are discrete
125  pedestal = fg.GetParameter(1);
126  noise = fg.GetParameter(2);
127  } else if (mStatisticsType == StatisticsType::GausFitFast) {
128  fitGaus(mNumberOfADCs, array + offset, float(mADCMin) - 0.5f, float(mADCMax + 1) - 0.5f, fitValues); // -0.5 since ADC values are discrete
129  pedestal = fitValues[1];
130  noise = fitValues[2];
131  } else if (mStatisticsType == StatisticsType::MeanStdDev) {
132  StatisticsData data = getStatisticsData(array + offset, mNumberOfADCs, double(mADCMin) - 0.5, double(mADCMax) - 0.5); // -0.5 since ADC values are discrete
133  pedestal = data.mCOG;
134  noise = data.mStdDev;
135  }
136  noise = std::abs(noise); // noise can be negative in gaus fit
137 
138  calROCPedestal.setValue(ichannel, pedestal);
139  calROCNoise.setValue(ichannel, noise);
140 
141  //printf("roc: %2d, channel: %4d, pedestal: %.2f, noise: %.2f\n", roc.getRoc(), ichannel, pedestal, noise);
142  }
143 
144  ++roc;
145  }
146 }

174 TH2* CalibPedestal::createControlHistogram(ROC roc)
175 {
176  auto* data = mADCdata[roc.getRoc()]->data();
177 
178  const size_t numberOfPads = (roc.rocType() == RocType::IROC) ? mMapper.getPadsInIROC() : mMapper.getPadsInOROC();
179  TH2F* h2 = new TH2F(fmt::format("hADCValues_ROC{:02}", roc.getRoc()).data(), fmt::format("ADC values of ROC {:02}", roc.getRoc()).data(), numberOfPads, 0, numberOfPads, mNumberOfADCs, mADCMin, mADCMax);
180  h2->SetDirectory(nullptr);
181  for (int ichannel = 0; ichannel < numberOfPads; ++ichannel) {
182  size_t offset = ichannel * mNumberOfADCs;
183 
184  for (int iADC = 0; iADC < mNumberOfADCs; ++iADC) {
185  h2->Fill(ichannel, mADCMin + iADC, (data + offset)[iADC]);
186  }
187  }
188 
189  return h2;
190 }
