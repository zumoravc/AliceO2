28 class PixelData
29 {
30 
31  public:
32  PixelData(const Digit* dig) : mRow(dig->getRow()), mCol(dig->getColumn()) {}
33  PixelData(uint16_t r = 0, uint16_t c = 0) : mRow(r), mCol(c) {}
34  uint16_t getRow() const { return mRow & RowMask; }
35  uint16_t getCol() const { return mCol; }
36  bool isMasked() const { return mRow & MaskBit; }
37  void setMask() { mRow |= MaskBit; }
38  void unsetMask() { mRow &= RowMask; }
39 
41  uint16_t getRowDirect() const { return mRow; }
42 
43  bool operator==(const PixelData& dt) const
44  {
46  return (getCol() == dt.getCol()) && (getRow() == dt.getRow());
47  }
48 
49  bool operator>(const PixelData& dt) const
50  {
52  if (getCol() == dt.getCol()) {
53  return getRow() > dt.getRow();
54  }
55  return getCol() > dt.getCol();
56  }
57 
58  bool operator<(const PixelData& dt) const
59  {
61  if (getCol() == dt.getCol()) {
62  return getRow() < dt.getRow();
63  }
64  return getCol() < dt.getCol();
65  }
66 
67  bool isNeighbour(const PixelData& dt, int maxDist) const
68  {
70  return (std::abs(static_cast<int>(getCol()) - static_cast<int>(dt.getCol())) <= maxDist &&
71  std::abs(static_cast<int>(getRow()) - static_cast<int>(dt.getRow())) <= maxDist);
72  }
73 
74  int compare(const PixelData& dt) const
75  {
77  return operator==(dt) ? 0 : (operator>(dt) ? 1 : -1);
78  }
79 
80  static constexpr uint32_t DummyROF = 0xffffffff;
81  static constexpr uint32_t DummyChipID = 0xffff;
82 
83  private:
84  void sanityCheck() const;
85  static constexpr int RowMask = 0x1ff;
86  static constexpr int MaskBit = 0x200;
87  uint16_t mRow = 0;
88  uint16_t mCol = 0;
89 
90  ClassDefNV(PixelData, 1);
91 };

140  void maskFiredInSample(const ChipPixelData& sample)
141  {
143  const auto& pixelsS = sample.getData();
144  uint32_t nC = mPixels.size();
145  if (!nC) {
146  return;
147  }
148  uint32_t nS = pixelsS.size();
149  if (!nS) {
150  return;
151  }
152  uint32_t itC = 0, itS = 0;
153  while (itC < nC && itS < nS) {
154  auto& pix0 = mPixels[itC];
155  const auto& pixC = pixelsS[itS];
156  if (pix0 == pixC) { // same
157  pix0.setMask();
158  if (mFirstUnmasked == itC++) { // mFirstUnmasked should flag 1st unmasked pixel entry
159  mFirstUnmasked = itC;
160  }
161  itS++;
162  } else if (pix0 < pixC) {
163  itC++;
164  } else {
165  itS++;
166  }
167  }
168  }

170  void maskFiredInSample(const ChipPixelData& sample, int maxDist)
171  {
173  const auto& pixelsS = sample.getData();
174  int nC = mPixels.size();
175  if (!nC) {
176  return;
177  }
178  int nS = pixelsS.size();
179  if (!nS) {
180  return;
181  }
182  for (int itC = 0, itS = 0; itC < nC; itC++) {
183  auto& pix0 = mPixels[itC];
184 
185  // seek to itS which is inferior than itC - maxDist
186  auto mincol = pix0.getCol() > maxDist ? pix0.getCol() - maxDist : 0;
187  auto minrow = pix0.getRowDirect() > maxDist ? pix0.getRowDirect() - maxDist : 0;
188  if (itS == nS) { // in case itS lool below reached the end
189  itS--;
190  }
191  while ((pixelsS[itS].getCol() > mincol || pixelsS[itS].getRow() > minrow) && itS > 0) {
192  itS--;
193  }
194  for (; itS < nS; itS++) {
195  const auto& pixC = pixelsS[itS];
196 
197  auto drow = static_cast<int>(pixC.getRow()) - static_cast<int>(pix0.getRowDirect());
198  auto dcol = static_cast<int>(pixC.getCol()) - static_cast<int>(pix0.getCol());
199 
200  if (dcol > maxDist || (dcol == maxDist && drow > maxDist)) {
201  break; // all higher itS will not match to this itC also
202  }
203  if (dcol < -maxDist || (drow > maxDist || drow < -maxDist)) {
204  continue;
205  } else {
206  pix0.setMask();
207  if (int(mFirstUnmasked) == itC) { // mFirstUnmasked should flag 1st unmasked pixel entry
208  mFirstUnmasked = itC + 1;
209  }
210  break;
211  }
212  }
213  }
214  }
