28  DataProcessorSpec{
29  "producer", //
30  Inputs{}, //
31  {OutputSpec{"TES", "STATEFUL", 0, Lifetime::Timeframe}}, //
32  // The producer is stateful, we use a static for the state in this
33  // particular case, but a Singleton or a captured new object would
34  // work as well.
35  AlgorithmSpec{
36  adaptStateful(
37  [](CallbackService& callbacks) {
38  static int foo = 0;
39  static int step = 0; // incremented in registered callbacks
40  auto startcb = []() {
41  ++step;
42  LOG(INFO) << "start " << step;
43  };
44  auto stopcb = []() {
45  ++step;
46  LOG(INFO) << "stop " << step;
47  };
48  auto resetcb = []() {
49  ++step;
50  LOG(INFO) << "reset " << step;
51  };
52  callbacks.set(CallbackService::Id::Start, startcb);
53  callbacks.set(CallbackService::Id::Stop, stopcb);
54  callbacks.set(CallbackService::Id::Reset, resetcb);
55  return adaptStateless([](DataAllocator& outputs, ControlService& control) {
56  auto& out = outputs.newChunk({"TES", "STATEFUL", 0}, sizeof(int));
57  auto outI = reinterpret_cast<int*>(out.data());
58  LOG(INFO) << "foo " << foo;
59  outI[0] = foo++;
60  control.endOfStream();
61  control.readyToQuit(QuitRequest::Me);
62  });
63  }) //
64  } //
65  }, //

67  "consumer", //
68  {InputSpec{"test", "TES", "STATEFUL", 0, Lifetime::Timeframe}}, //
69  Outputs{}, //
70  AlgorithmSpec{

72  []() {
