38 bool TimeframeReaderDevice::ConditionalRun()
39 {
40  auto addPartFn = [this](FairMQParts& parts, char* buffer, size_t size) {
41  parts.AddPart(this->NewMessage(
42  buffer,
43  size,
44  [](void* data, void* hint) { delete[](char*) data; },
45  nullptr));
46  };
47  auto sendFn = [this](FairMQParts& parts) { this->Send(parts, this->mOutChannelName); };
48 
49  // FIXME: For the moment we support a single file. This should really be a glob. We
50  // should also have a strategy for watching directories.
51  std::vector<std::string> files;
52  files.push_back(mInFileName);
53  for (auto&& fn : files) {
54  mFile.open(fn, std::ofstream::in | std::ofstream::binary);
55  try {
56  streamTimeframe(mFile,
57  addPartFn,
58  sendFn);
59  } catch (std::runtime_error& e) {
60  LOG(ERROR) << e.what() << "\n";
61  }
62  mSeen.push_back(fn);
63  }
64 
65  return false;
66 }
