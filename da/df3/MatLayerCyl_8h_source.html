32 /**********************************************************************
33  * *
34  * Material data on the cells of cylindrical layer *
35  * Cell is limited by 2 radial planes at phiMin,phiMax, *
36  * radii radMin,radMax and XY planes at zMin, zMax. *
37  * This limits are defined in the cell container class *
38  * *
39  * *
40  * Cylindrical material layer *
41  * *
42  * Logical division is given by mNZBins and mNPhiBins *
43  * but the actual number phi slices might be less if neighbouring *
44  * phi bins with similar properties are merged together. The actual *
45  * phi slice should be accessed via mPhiBin2Slice map. *
46  * *
47  **********************************************************************/

169  GPUd() int getEdgePhiBinOfSlice(int phiBin, int dir) const
170  {
171  // Get edge bin (in direction dir) of the slice, to which phiBin belongs
172  // No check for phiBin validity is done
173  auto slice = phiBin2Slice(phiBin);
174  while (slice == phiBin2Slice((phiBin += dir)))
175  ;
176  return phiBin - dir;
177  }
