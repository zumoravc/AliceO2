25 InjectorFunction readoutAdapter(OutputSpec const& spec)
26 {
27  auto counter = std::make_shared<uint64_t>(0);
28 
29  return [spec, counter](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
30  for (size_t i = 0; i < parts.Size(); ++i) {
31  DataHeader dh;
32  // FIXME: this will have to change and extract the actual subspec from
33  // the data.
34  ConcreteDataMatcher concrete = DataSpecUtils::asConcreteDataMatcher(spec);
35  dh.dataOrigin = concrete.origin;
36  dh.dataDescription = concrete.description;
37  dh.subSpecification = concrete.subSpec;
38  dh.payloadSize = parts.At(i)->GetSize();
39  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
40 
41  DataProcessingHeader dph{*counter, 0};
42  (*counter) += 1UL;
43  o2::header::Stack headerStack{dh, dph};
44  sendOnChannel(device, std::move(headerStack), std::move(parts.At(i)), spec, channelRetriever);
45  }
46  };
47 }
