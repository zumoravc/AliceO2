85 void digi2raw(const std::string& inpName, const std::string& outDir, int verbosity, bool filePerLink, uint32_t rdhV, bool noEmptyHBF, int superPageSizeInB)
86 {
87  TStopwatch swTot;
88  swTot.Start();
89  o2::ft0::Digits2Raw m2r;
90  m2r.setFilePerLink(filePerLink);
91  m2r.setVerbosity(verbosity);
92  auto& wr = m2r.getWriter();
93  wr.setSuperPageSize(superPageSizeInB);
94  wr.useRDHVersion(rdhV);
95  wr.setDontFillEmptyHBF(noEmptyHBF);
96 
97  std::string outDirName(outDir);
98  if (outDirName.back() != '/') {
99  outDirName += '/';
100  }
101  // if needed, create output directory
102  if (gSystem->AccessPathName(outDirName.c_str())) {
103  if (gSystem->mkdir(outDirName.c_str(), kTRUE)) {
104  LOG(FATAL) << "could not create output directory " << outDirName;
105  } else {
106  LOG(INFO) << "created output directory " << outDirName;
107  }
108  }
109 
110  m2r.readDigits(outDirName, inpName);
111  wr.writeConfFile(wr.getOrigin().str, "RAWDATA", o2::utils::concat_string(outDirName, wr.getOrigin().str, "raw.cfg"));
112  //
113  swTot.Stop();
114  swTot.Print();
115 }
