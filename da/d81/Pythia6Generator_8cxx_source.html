65 Bool_t Pythia6Generator::ReadEvent(FairPrimaryGenerator* primGen)
66 {
67 
68  // Check for input file
69  if (!mInputFile) {
70  // if ( ! mInputFile->is_open() ) {
71  cout << "-E Pythia6Generator: Input file not open!" << endl;
72  return kFALSE;
73  }
74 
75  // Define event variable to be read from file
76  Int_t ntracks = 0, eventID = 0, ncols = 0;
77 
78  // Define track variables to be read from file
79  Int_t nLev = 0, pdgID = 0, nM1 = -1, nM2 = -1, nDF = -1, nDL = -1;
80  Float_t fPx = 0., fPy = 0., fPz = 0., fM = 0., fE = 0.;
81  Float_t fVx = 0., fVy = 0., fVz = 0., fT = 0.;
82 
83  // Read event header line from input file
84 
85  Int_t max_nr = 0;
86 
87  Text_t buffer[200];
88  ncols = fscanf(mInputFile, "%d\t%d", &eventID, &ntracks);
89 
90  if (ncols && ntracks > 0) {
91 
92  if (mVerbose > 0)
93  cout << "Event number: " << eventID << "\tNtracks: " << ntracks << endl;
94 
95  for (Int_t ll = 0; ll < ntracks; ll++) {
96  ncols = fscanf(mInputFile, "%d %d %d %d %d %d %f %f %f %f %f %f %f %f %f", &nLev, &pdgID, &nM1, &nM2, &nDF, &nDL, &fPx, &fPy, &fPz, &fE, &fM, &fVx, &fVy, &fVz, &fT);
97  if (mVerbose > 0)
98  cout << nLev << "\t" << pdgID << "\t" << nM1 << "\t" << nM2 << "\t" << nDF << "\t" << nDL << "\t" << fPx << "\t" << fPy << "\t" << fPz << "\t" << fE << "\t" << fM << "\t" << fVx << "\t" << fVy << "\t" << fVz << "\t" << fT << endl;
99  if (nLev == 1)
100  primGen->AddTrack(pdgID, fPx, fPy, fPz, fVx, fVy, fVz);
101  }
102  } else {
103  cout << "-I Pythia6Generator: End of input file reached " << endl;
104  CloseInput();
105  return kFALSE;
106  }
107 
108  // If end of input file is reached : close it and abort run
109  if (feof(mInputFile)) {
110  cout << "-I Pythia6Generator: End of input file reached " << endl;
111  CloseInput();
112  return kFALSE;
113  }
114 
115  /*
116  cout << "-I Pythia6Generator: Event " << eventID << ", vertex = ("
117  << vx << "," << vy << "," << vz << ") cm, multiplicity "
118  << ntracks << endl;
119  */
120 
121  return kTRUE;
122 }
