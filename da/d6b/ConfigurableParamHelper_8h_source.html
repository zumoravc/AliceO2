107  // Grab the list of ConfigurableParam data members
108  // Returns a nullptr if the TClass of the P template class cannot be created.
109  std::vector<ParamDataMember>* getDataMembers() const
110  {
111  // just a helper line to make sure P::sInstance is looked-up
112  // and that compiler complains about missing static sInstance of type P
113  // volatile void* ptr = (void*)&P::sInstance;
114  // static assert on type of sInstance:
115  static_assert(std::is_same<decltype(P::sInstance), P>::value,
116  "static instance must of same type as class");
117 
118  // obtain the TClass for P and delegate further
119  auto cl = TClass::GetClass(typeid(P));
120  if (!cl) {
121  _ParamHelper::printWarning(typeid(P));
122  return nullptr;
123  }
124 
125  return _ParamHelper::getDataMembersImpl(getName(), cl, (void*)this, sValueProvenanceMap);
126  }
