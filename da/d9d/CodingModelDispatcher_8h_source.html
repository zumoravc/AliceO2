76  template <typename ValueType, typename WeightType>
77  class addWeightFctr
78  {
79  public:
80  addWeightFctr(ValueType _v, WeightType _w) : value(_v), weight(_w) {}
81  ~addWeightFctr() {}
82 
83  using return_type = bool;
84 
85  template <typename T>
86  return_type operator()(T& stage)
87  {
88  // the addWeight function belongs to the probability model as base
89  // of the specific model; funcions of the base can be accessed by
90  // static casting. This avoids an extra level of function calls.
91  return static_cast<typename T::wrapped_type::base_type&>(*stage).addWeight(value, weight);
92  }
93 
94  private:
95  ValueType value;
96  WeightType weight;
97  };
98 
105  template <typename ValueType, typename WeightType>
106  bool addWeight(ValueType v, WeightType w, bool switchToNextModel = true)
107  {
108  bool result = mContainer.apply(mPosition, addWeightFctr<ValueType, WeightType>(v, w));
109  if (switchToNextModel && ++mPosition >= getNumberOfModels()) {
110  mPosition = 0;
111  }
112  return result;
113  }

320  class readFctr
321  {
322  public:
323  readFctr(std::istream& in, container_type& container) : mIn(in), mContainer(container) {}
324  ~readFctr() {}
325 
326  using return_type = bool;
327 
328  template <typename T>
329  return_type operator()(boost::type<T>)
330  {
331  T& stage = static_cast<T&>(mContainer);
332  std::string level, name, remaining;
333  mIn >> level;
334  mIn >> name;
335  if (!mIn) {
336  return false;
337  }
338  if (std::stoi(level) != T::level::value || name.compare((*stage).getName())) {
339  std::cerr << "Format error: expecting level '" << T::level::value << "' and name '" << (*stage).getName()
340  << "', got: " << level << " " << name << std::endl;
341  }
342  std::cout << "reading configuration for model " << name << std::endl;
343  std::getline(mIn, remaining); // flush the current line
344  (*stage).read(mIn);
345  return true;
346  }
347 
348  private:
349  std::istream& mIn;
350  container_type& mContainer;
351  };
352 
358  int read(const char* filename)
359  {
360  std::ifstream input(filename);
361  if (!input.good()) {
362  return -1;
363  }
364  // TODO: probably need mpl fold here to propagate the return value
365  boost::mpl::for_each<typename container_type::types, boost::type<boost::mpl::_>>(readFctr(input, mContainer));
366  return 0;
367  }

371  int mPosition;
373  container_type mContainer;
