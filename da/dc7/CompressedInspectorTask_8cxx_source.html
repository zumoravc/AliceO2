34 template <typename RAWDataHeader>
35 void CompressedInspectorTask<RAWDataHeader>::init(InitContext& ic)
36 {
37  LOG(INFO) << "CompressedInspector init";
38  auto filename = ic.options().get<std::string>("tof-compressed-inspector-filename");
39 
41  if (mFile && mFile->IsOpen()) {
42  LOG(WARNING) << "a file was already open, closing";
43  mFile->Close();

46  mFile = TFile::Open(filename.c_str(), "RECREATE");
47  if (!mFile || !mFile->IsOpen()) {
48  LOG(ERROR) << "cannot open output file: " << filename;
49  mStatus = true;
50  return;
51  }
52 
53  mHistos1D["hHisto"] = new TH1F("hHisto", "", 1000, 0., 1000.);
54  mHistos1D["time"] = new TH1F("hTime", ";time (24.4 ps)", 2097152, 0., 2097152.);
55  mHistos1D["timebc"] = new TH1F("hTimeBC", ";time (24.4 ps)", 1024, 0., 1024.);
56  mHistos1D["tot"] = new TH1F("hTOT", ";ToT (48.8 ps)", 2048, 0., 2048.);
57  mHistos1D["indexE"] = new TH1F("hIndexE", ";index EO", 172800, 0., 172800.);
58  mHistos2D["slotPartMask"] = new TH2F("hSlotPartMask", ";crate;slot", 72, 0., 72., 12, 1., 13.);
59  mHistos2D["diagnostic"] = new TH2F("hDiagnostic", ";crate;slot", 72, 0., 72., 13, 0., 13.);
60  mHistos1D["Nerror"] = new TH1F("hNError", ";number of error", 1000, 0., 1000.);
61  mHistos1D["Ntest"] = new TH1F("hNTest", ";number of test", 1000, 0., 1000.);
62  mHistos1D["errorBit"] = new TH1F("hErrorBit", ";TDC error bit", 15, 0., 15.);
63  mHistos2D["error"] = new TH2F("hError", ";slot;TDC", 24, 1., 13., 15, 0., 15.);
64  mHistos2D["test"] = new TH2F("hTest", ";slot;TDC", 24, 1., 13., 15, 0., 15.);
65 
66  auto finishFunction = [this]() {
67  LOG(INFO) << "CompressedInspector finish";
68  for (auto& histo : mHistos1D)
69  histo.second->Write();
70  for (auto& histo : mHistos2D)
71  histo.second->Write();
72  mFile->Close();
73  };
74  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishFunction);
75 }

77 template <typename RAWDataHeader>
78 void CompressedInspectorTask<RAWDataHeader>::run(ProcessingContext& pc)
79 {
80  LOG(DEBUG) << "CompressedInspector run";
81 
83  if (mStatus) {
84  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
85  return;
86  }
87 
89  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
90  if (!iit.isValid())
91  continue;
92 
94  for (auto const& ref : iit) {
95 
96  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
97  auto payloadIn = ref.payload;
98  auto payloadInSize = headerIn->payloadSize;
99 
100  DecoderBaseT<RAWDataHeader>::setDecoderBuffer(payloadIn);
101  DecoderBaseT<RAWDataHeader>::setDecoderBufferSize(payloadInSize);
102  DecoderBaseT<RAWDataHeader>::run();
103  }
104  }

115 template <typename RAWDataHeader>
116 void CompressedInspectorTask<RAWDataHeader>::frameHandler(const CrateHeader_t* crateHeader, const CrateOrbit_t* crateOrbit,
117  const FrameHeader_t* frameHeader, const PackedHit_t* packedHits)
118 {
119  mHistos1D["hHisto"]->Fill(frameHeader->numberOfHits);
120  for (int i = 0; i < frameHeader->numberOfHits; ++i) {
121  auto packedHit = packedHits + i;
122  auto indexE = packedHit->channel +
123  8 * packedHit->tdcID +
124  120 * packedHit->chain +
125  240 * (frameHeader->trmID - 3) +
126  2400 * crateHeader->drmID;
127  int time = packedHit->time;
128  int timebc = time % 1024;
129  time += (frameHeader->frameID << 13);
130 
131  mHistos1D["indexE"]->Fill(indexE);
132  mHistos1D["time"]->Fill(time);
133  mHistos1D["timebc"]->Fill(timebc);
134  mHistos1D["tot"]->Fill(packedHit->tot);
135  }
136 };

138 template <typename RAWDataHeader>
139 void CompressedInspectorTask<RAWDataHeader>::trailerHandler(const CrateHeader_t* crateHeader, const CrateOrbit_t* crateOrbit,
140  const CrateTrailer_t* crateTrailer, const Diagnostic_t* diagnostics,
141  const Error_t* errors)
142 {
143  mHistos2D["diagnostic"]->Fill(crateHeader->drmID, 0);
144  for (int i = 0; i < crateTrailer->numberOfDiagnostics; ++i) {
145  auto diagnostic = diagnostics + i;
146  mHistos2D["diagnostic"]->Fill(crateHeader->drmID, diagnostic->slotID);
147  }
148  int nError = 0, nTest = 0;
149  for (int i = 0; i < crateTrailer->numberOfErrors; ++i) {
150  auto error = errors + i;
151  if (error->undefined) {
152  nTest++;
153  mHistos2D["test"]->Fill(error->slotID + 0.5 * error->chain, error->tdcID);
154  } else {
155  nError++;
156  mHistos2D["error"]->Fill(error->slotID + 0.5 * error->chain, error->tdcID);
157  for (int ibit = 0; ibit < 15; ++ibit) {
158  if (error->errorFlags & (1 << ibit))
159  mHistos1D["errorBit"]->Fill(ibit);
160  }
161  }
162  }
163  mHistos1D["Nerror"]->Fill(nError);
164  mHistos1D["Ntest"]->Fill(nTest);
165 };
