190 template <typename T>
191 inline T Detector::BetheBlochAleph(T bg, T kp1, T kp2, T kp3, T kp4, T kp5)
192 {
193  T beta = bg / std::sqrt(static_cast<T>(1.) + bg * bg);
194 
195  T aa = std::pow(beta, kp4);
196  T bb = std::pow(static_cast<T>(1.) / bg, kp5);
197  bb = std::log(kp3 + bb);
198 
199  return (kp2 - aa - bb) * kp1 / aa;
200 }
