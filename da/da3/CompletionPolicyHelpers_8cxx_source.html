24 CompletionPolicy CompletionPolicyHelpers::defineByName(std::string const& name, CompletionPolicy::CompletionOp op)
25 {
26  auto matcher = [name](DeviceSpec const& device) -> bool {
27  return std::regex_match(device.name.begin(), device.name.end(), std::regex(name));
28  };
29  auto callback = [op](CompletionPolicy::InputSet) -> CompletionPolicy::CompletionOp {
30  return op;
31  };
32  switch (op) {
33  case CompletionPolicy::CompletionOp::Consume:
34  return CompletionPolicy{"always-consume", matcher, callback};
35  break;
36  case CompletionPolicy::CompletionOp::Process:
37  return CompletionPolicy{"always-process", matcher, callback};
38  break;
39  case CompletionPolicy::CompletionOp::Wait:
40  return CompletionPolicy{"always-wait", matcher, callback};
41  break;
42  case CompletionPolicy::CompletionOp::Discard:
43  return CompletionPolicy{"always-discard", matcher, callback};
44  break;
45  }
46  O2_BUILTIN_UNREACHABLE();
47 }

49 CompletionPolicy CompletionPolicyHelpers::consumeWhenAll(const char* name, CompletionPolicy::Matcher matcher)
50 {
51  auto callback = [](CompletionPolicy::InputSet inputs) -> CompletionPolicy::CompletionOp {
52  for (auto& input : inputs) {
53  if (input.header == nullptr && input.payload == nullptr) {
54  return CompletionPolicy::CompletionOp::Wait;
55  }
56  }
57  return CompletionPolicy::CompletionOp::Consume;
58  };
59  return CompletionPolicy{name, matcher, callback};

75 CompletionPolicy CompletionPolicyHelpers::processWhenAny(const char* name, CompletionPolicy::Matcher matcher)
76 {
77  auto callback = [](CompletionPolicy::InputSet inputs) -> CompletionPolicy::CompletionOp {
78  size_t present = 0;
79  for (auto& input : inputs) {
80  if (input.header != nullptr) {
81  present++;
82  }
83  }
84  if (present == inputs.size()) {
85  return CompletionPolicy::CompletionOp::Consume;
86  } else if (present == 0) {
87  return CompletionPolicy::CompletionOp::Wait;
88  }
89  return CompletionPolicy::CompletionOp::Process;
90  };
91  return CompletionPolicy{name, matcher, callback};
92 }
