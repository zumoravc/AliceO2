35 void Cave::createMaterials()

40  Float_t sxmgmx;

43  // AIR

47  Float_t dAir = 1.20479E-3 * 960. / 1014.;

49  //
50  matmgr.Mixture("CAVE", 2, "Air", aAir, zAir, dAir, 4, wAir);
51  //
52  matmgr.Medium("CAVE", 2, "Air", 2, 0, isxfld, sxmgmx, 10, -1, -0.1, 0.1, -10);
53 }

101 // we set the random number generator for each
102 // primary in order to be reproducible in a multi-processing sub-event
103 // setting
104 void Cave::BeginPrimary()
105 {
106  static int primcounter = 0;
107 
108  // only do it, if not in pure trackSeeding mode
109  if (!o2::conf::SimCutParams::Instance().trackSeed) {
110  auto& conf = o2::conf::SimConfig::Instance();
111  auto chunks = conf.getInternalChunkSize();
112  if (chunks != -1) {
113  if (primcounter % chunks == 0) {
114  static int counter = 1;
115  auto seed = counter + 10;
116  gRandom->SetSeed(seed);
117  counter++;
118  }
119  }
120  }
121  primcounter++;
122 }
