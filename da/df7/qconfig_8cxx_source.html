111 template <class T>
112 inline int qConfigTypeSpecialized<T>::qAddOptionMain(qConfigSettings<T>& settings, T& ref, int& i, const char** argv, const int argc, T def)
113 {
114  int retVal = 0;
115  int iOrg = i;
116  if (settings.doSet) {
117  ref = settings.set;
118  } else if ((retVal = qAddOptionType<T>(settings, ref, i, argv, argc, def))) {
119  return (retVal);
120  }
121  if ((retVal = qAddOptionMinMax<T>(settings, ref, iOrg < argc ? argv[iOrg] : ""))) {
122  return (retVal);
123  }
124  qAddOptionMessage<T>(settings, ref);
125  return (0);
126 }

202  template <typename... Args>
203  static inline int qAddOptionVec(std::vector<T>& ref, int& i, const char** argv, const int argc, const char* /*help*/, Args&&... args)
204  {
205  auto settings = qConfigGetSettings(args...);
206  int iFirst = i, iLast;
207  do {
208  iLast = i;
209  T tmp = T(), def = T();
210  settings.allowEmpty = (i != iFirst);
211  int retVal = qConfigTypeSpecialized<T>::qAddOptionMain(settings, tmp, i, argv, argc, def);
212  if (retVal) {
213  return (retVal != qcrArgMissing || i == iFirst ? retVal : 0);
214  }
215  if (i == iFirst || i != iLast) {
216  ref.push_back(tmp);
217  }
218  } while (i != iLast);

246  template <typename... Args>
247  static inline void qConfigHelpOption(const char* name, const char* type, const char* def, const char* optname, char optnameshort, const char* preopt, char preoptshort, int optionType, const char* help, Args&&... args)
248  {
249  auto settings = qConfigGetSettings(args...);
250  const bool boolType = optionType != 1 && std::is_same<T, bool>::value;
251  const char* arguments = settings.doSet ? " (" : (settings.doDefault || optionType == 1 || boolType) ? " [arg] (" : optionType == 2 ? " [...] (" : " arg (";
252  char argBuffer[4] = {0};
253  unsigned int argBufferPos = 0;
254  if (optnameshort && preoptshort) {
255  argBuffer[argBufferPos++] = '-';
256  }
257  if (optnameshort) {
258  argBuffer[argBufferPos++] = preoptshort == 0 ? '-' : preoptshort;
259  argBuffer[argBufferPos++] = optnameshort;
260  }
261  std::cout << "\t" << name << ": " << argBuffer << (optnameshort == 0 ? "" : arguments) << "--" << preopt << optname << (optnameshort == 0 ? arguments : ", ") << "type: " << type;
262  if (optionType == 0) {
263  std::cout << ", default: " << def;
264  }
265  if (optionType == 1) {
266  std::cout << ", sets " << name << " to " << def;
267  }
268  if (settings.checkMin) {
269  std::cout << ", minimum: " << settings.min;
270  }
271  if (settings.checkMax) {
272  std::cout << ", maximum: " << settings.max;
273  }
274  std::cout << ")\n\t\t" << help << ".\n";
275  if (settings.doSet) {
276  std::cout << "\t\tSets " << name << " to " << settings.set << ".\n";
277  } else if (settings.doDefault) {
278  std::cout << "\t\tIf no argument is supplied, " << name << " is set to " << settings.set << ".\n";
279  } else if (boolType) {
280  std::cout << "\t\tIf no argument is supplied, " << name << " is set to true.\n";
281  }
282  if (optionType == 2) {
283  std::cout << "\t\tCan be set multiple times, accepts multiple arguments.\n";
284  }
285  std::cout << "\n";
286  }
