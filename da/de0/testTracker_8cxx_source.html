63 TrackClusters getTrackClusters(const Track& track)
64 {
65  TrackClusters trCl;
66  trCl.track = track;
67  trCl.nFiredChambers = 0;
68  Mapping::MpStripIndex stripIndex;
69  MpArea area;
70  Cluster2D cl;
71  for (int ich = 0; ich < 4; ++ich) {
72  auto hits = helper.hitFinder.getLocalPositions(track, ich);
73  bool isFired = false;
74  for (auto& hit : hits) {
75  int deId = hit.deId;
76  float xPos = hit.xCoor;
77  float yPos = hit.yCoor;
78  stripIndex = helper.mapping.stripByPosition(xPos, yPos, 0, deId, false);
79  if (!stripIndex.isValid()) {
80  continue;
81  }
82  cl.deId = deId;
83  area = helper.mapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
84  cl.yCoor = area.getCenterY();
85  float halfSize = area.getHalfSizeY();
86  cl.sigmaY2 = halfSize * halfSize / 3.;
87  stripIndex = helper.mapping.stripByPosition(xPos, yPos, 1, deId, false);
88  area = helper.mapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
89  cl.xCoor = area.getCenterX();
90  halfSize = area.getHalfSizeX();
91  cl.sigmaX2 = halfSize * halfSize / 3.;
92  trCl.clusters.push_back(cl);
93  isFired = true;
94  } // loop on fired pos
95  if (isFired) {
96  ++(trCl.nFiredChambers);
97  }
98  }
99  return trCl;
100 }

122 BOOST_DATA_TEST_CASE(TestMultipleTracks, boost::unit_test::data::xrange(1, 9), nTracksPerEvent)
123 {
124  float chi2Cut = helper.tracker.getSigmaCut() * helper.tracker.getSigmaCut();
125  int nTotFakes = 0, nTotReconstructible = 0;
126  for (int ievt = 0; ievt < 1000; ++ievt) {
127  std::vector<TrackClusters> trackClusters = getTrackClusters(nTracksPerEvent);
128  std::vector<Cluster2D> clusters;
129 
130  // Fill string for debugging
131  std::stringstream ss;
132  ss << "\n";
133  int itr = -1;
134  for (auto& trCl : trackClusters) {
135  ++itr;
136  ss << "Track " << itr << ": " << trCl.track << "\n";
137  for (auto& cl : trCl.clusters) {
138  clusters.push_back(cl);
139  ss << " deId " << (int)cl.deId << " pos: (" << cl.xCoor << ", " << cl.yCoor << ")";
140  }
141  ss << "\n";
142  }
143 
144  // Run tracker algorithm
145  helper.tracker.process(clusters);
146 
147  // Further strings for debugging
148  ss << " Reconstructed tracks:\n";
149  for (size_t ireco = 0; ireco < helper.tracker.getTracks().size(); ++ireco) {
150  ss << " " << helper.tracker.getTracks()[ireco] << "\n";
151  }
152 
153  // Check that all reconstructible tracks are reconstructed
154  size_t nReconstructible = 0;
155  itr = -1;
156  for (auto& trCl : trackClusters) {
157  ++itr;
158  bool isReco = false;
159  for (size_t ireco = 0; ireco < helper.tracker.getTracks().size(); ++ireco) {
160  if (helper.tracker.getTracks()[ireco].isCompatible(trCl.track, chi2Cut)) {
161  isReco = true;
162  break;
163  }
164  }
165  bool isReconstructible = trCl.isReconstructible();
166  // If the number of tracks is small, we can check that:
167  // 1) all reconstructible tracks are reconstructed
168  // 2) all non-reconstructible tracks are not
169  // Case 2), however, is not always valid when we have many tracks,
170  // since we can combine clusters from different tracks to build fakes
171  bool testTrack = (isReconstructible || nTracksPerEvent < 4);
172  if (testTrack) {
173  BOOST_TEST(isReco == isReconstructible, ss.str() << " track " << itr << " reco " << isReco
174  << " != reconstructible " << isReconstructible);
175  }
176  if (isReconstructible) {
177  ++nReconstructible;
178  }
179  } // loop on input tracks
180  nTotReconstructible += nReconstructible;
181  int nFakes = helper.tracker.getTracks().size() - nReconstructible;
182  if (nFakes > 0) {
183  ++nTotFakes;
184  }
185  } // loop on events
186  // To show the following message, run the test with: --log_level=message
187  BOOST_TEST_MESSAGE("Fraction of fake tracks: " << (double)nTotFakes / (double)nTotReconstructible);
188 }
