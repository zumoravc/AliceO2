75 bool Track::propagateToZ(float zPosition)
76 {
79 
80  // Nothing to be done if we're already at zPosition
81  // Notice that the z position is typically the z of the cluster,
82  // which is provided in float precision as well.
83  // So it is typically a well defined value, hence the strict equality
84  if (mPosition[2] == zPosition) {
85  return false;
86  }
87 
88  // Propagate the track position parameters
89  float dZ = zPosition - mPosition[2];
90  float newX = mPosition[0] + mDirection[0] * dZ;
91  float newY = mPosition[1] + mDirection[1] * dZ;
92  setPosition(newX, newY, zPosition);
93 
94  // Propagate the covariance matrix
95  std::array<float, 6> newCovParams;
96  float dZ2 = dZ * dZ;
97  for (int idx = 0; idx < 2; ++idx) {
98  int slopeIdx = idx + 2;
99  int covIdx = idx + 4;
100  // s_x^2 -> s_x^2 + 2*cov(x,slopeX)*dZ + s_slopeX^2*dZ^2
101  newCovParams[idx] =
102  mCovarianceParameters[idx] + 2. * mCovarianceParameters[covIdx] * dZ + mCovarianceParameters[slopeIdx] * dZ2;
103  // cov(x,slopeX) -> cov(x,slopeX) + s_slopeX^2*dZ
104  newCovParams[covIdx] = mCovarianceParameters[covIdx] + mCovarianceParameters[slopeIdx] * dZ;
105  // s_slopeX^2 -> s_slopeX^2
106  newCovParams[slopeIdx] = mCovarianceParameters[slopeIdx];
107  }
108 
109  mCovarianceParameters.swap(newCovParams);
110 
111  return true;
112 }

115 bool Track::isCompatible(const Track& track, float chi2Cut) const
116 {
118  if (track.mPosition[2] != mPosition[2]) {
119  Track copyTrack(track);
120  copyTrack.propagateToZ(mPosition[2]);
121  return isCompatible(copyTrack, chi2Cut);
122  }
123 
124  // // method 1: chi2 calculation accounting for covariance between slope and position.
125  // // This is the full calculation. However, if we have two parallel tracks with same x
126  // // but different ( > Nsigmas) y position, the algorithm will return true since the
127  // // difference in one of the parameters is compensated by the similarity of the others.
128  // // This is probably not what we need.
129  // double chi2 = 0.;
130  // double pos1[2] = { getPosition().x(), getPosition().y() };
131  // double dir1[2] = { getDirection().x(), getDirection().y() };
132  // double pos2[2] = { track.getPosition().x(), track.getPosition().y() };
133  // double dir2[2] = { track.getDirection().x(), track.getDirection().y() };
134  //
135  // for (int icoor = 0; icoor < 2; ++icoor) {
136  // double diffPos = pos1[icoor] - pos2[icoor];
137  // double diffSlope = dir1[icoor] - dir2[icoor];
138  // double varPos = mCovarianceParameters[icoor] + track.mCovarianceParameters[icoor];
139  // double varSlope = mCovarianceParameters[icoor + 2] + track.mCovarianceParameters[icoor + 2];
140  // double cov = mCovarianceParameters[icoor + 4] + track.mCovarianceParameters[icoor + 4];
141  // chi2 += (diffPos * diffPos * varSlope + diffSlope * diffSlope * varPos - 2. * diffPos * diffSlope * cov) /
142  // (varPos * varSlope - cov * cov);
143  // }
144  //
145  // return (chi2 / 4.) < chi2Cut;
146 
147  // method 2: apply the cut on each parameter
148  // This method avoids the issue of method 1
149  double p1[4] = {mPosition[0], mPosition[1], mDirection[0], mDirection[1]};
150  double p2[4] = {track.mPosition[0], track.mPosition[1], track.mDirection[0],
151  track.mDirection[1]};
152  for (int ipar = 0; ipar < 4; ++ipar) {
153  double diff = p1[ipar] - p2[ipar];
154  if (diff * diff / (mCovarianceParameters[ipar] + track.mCovarianceParameters[ipar]) > chi2Cut) {
155  return false;
156  };
157  }
158  return true;
159 }
