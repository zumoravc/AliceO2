58  using TestDescriptorT = Descriptor<descriptorSize>;

60  BOOST_CHECK(TestDescriptorT::bitcount == descriptorSize * 8);
61  BOOST_CHECK(sizeof(TestDescriptorT::ItgType) * TestDescriptorT::arraySize == descriptorSize);
62  static_assert(TestDescriptorT::size == sizeof(TestDescriptorT),
63  "Descriptor must have size of the underlying data member");
64 
65  // the Descriptor allows to define an integral value from
66  // a char sequence so that the in-memory representation shows
67  // a readable pattern
68  constexpr char readable[] = "DESCRPTR";
69  // the inverse sequence as integer value: R T P R C S E D
70  constexpr TestDescriptorT::ItgType itgvalue = 0x5254505243534544;
71 
72  TestDescriptorT testDescriptor(readable);
73  TestDescriptorT anotherDescriptor("ANOTHER");
74  BOOST_CHECK(TestDescriptorT::size == sizeof(testDescriptor));
75 
76  // copy constructor and comparison operator
77  BOOST_CHECK(testDescriptor == TestDescriptorT(testDescriptor));
78 
79  // comparison operator
80  BOOST_CHECK(testDescriptor != anotherDescriptor);
81 
82  // type cast operator
83  BOOST_CHECK(itgvalue == testDescriptor);
84 
85  // assignment operator
86  anotherDescriptor = testDescriptor;
87  BOOST_CHECK(testDescriptor == anotherDescriptor);
88  anotherDescriptor = TestDescriptorT("SOMEMORE");
89  BOOST_CHECK(itgvalue != anotherDescriptor);
90 
91  // assignment using implicit conversion with constructor
92  anotherDescriptor = itgvalue;
93  BOOST_CHECK(testDescriptor == anotherDescriptor);
94 
95  // check with runtime string
96  std::string runtimeString = "RUNTIMES";
97  TestDescriptorT runtimeDescriptor;
98  runtimeDescriptor.runtimeInit(runtimeString.c_str());
99  BOOST_CHECK(runtimeDescriptor == TestDescriptorT("RUNTIMES"));

144 BOOST_AUTO_TEST_CASE(DataOrigin_test)
145 {
146  // test for the templated Descriptor struct
147  constexpr int descriptorSize = 4;
148  using TestDescriptorT = Descriptor<descriptorSize>;
149  BOOST_CHECK(TestDescriptorT::size == descriptorSize);
150  BOOST_CHECK(TestDescriptorT::bitcount == descriptorSize * 8);
151  BOOST_CHECK(sizeof(TestDescriptorT::ItgType) * TestDescriptorT::arraySize == descriptorSize);
152  BOOST_CHECK(TestDescriptorT::size == sizeof(DataOrigin));
153 
154  // we want to explicitely have the size of DataOrigin to be 4
155  static_assert(sizeof(DataOrigin) == 4,
156  "DataOrigin struct must be of size 4");
157 
158  // Check that ordering works.
159  BOOST_CHECK(DataOrigin("A") < DataOrigin("B"));
160  BOOST_CHECK(DataOrigin("AA") < DataOrigin("AB"));
161  BOOST_CHECK(DataOrigin("AAA") < DataOrigin("AAB"));
162  BOOST_CHECK(DataOrigin("AAA") < DataOrigin("ABA"));
163  std::vector<DataOrigin> v1 = {DataOrigin("B"), DataOrigin("C"), DataOrigin("A")};
164  std::sort(v1.begin(), v1.end());
165  BOOST_CHECK_EQUAL(v1[0], DataOrigin("A"));
166  BOOST_CHECK_EQUAL(v1[1], DataOrigin("B"));
167  BOOST_CHECK_EQUAL(v1[2], DataOrigin("C"));
168  std::vector<DataOrigin> v2 = {DataOrigin("A"), DataOrigin("B")};
169  std::sort(v2.begin(), v2.end());
170  BOOST_CHECK_EQUAL(v2[0], DataOrigin("A"));
171  BOOST_CHECK_EQUAL(v2[1], DataOrigin("B"));
172 
173  using CustomHeader = std::tuple<DataOrigin, DataDescription>;
174  std::vector<CustomHeader> v3{CustomHeader{"TST", "B"}, CustomHeader{"TST", "A"}};
175  std::sort(v3.begin(), v3.end());
176  auto h0 = CustomHeader{"TST", "A"};
177  auto h1 = CustomHeader{"TST", "B"};
178  BOOST_CHECK(v3[0] == h0);
179  BOOST_CHECK(v3[1] == h1);
180 
181  using CustomHeader2 = std::tuple<DataOrigin, DataDescription, int>;
182  std::vector<CustomHeader2> v4{CustomHeader2{"TST", "A", 1}, CustomHeader2{"TST", "A", 0}};
183  std::sort(v4.begin(), v4.end());
184  auto hh0 = CustomHeader2{"TST", "A", 0};
185  auto hh1 = CustomHeader2{"TST", "A", 1};
186  BOOST_CHECK(v4[0] == hh0);
187  BOOST_CHECK(v4[1] == hh1);
188 
189  struct CustomHeader3 {
190  DataOrigin origin;
191  DataDescription desc;
192  uint64_t subSpec;
193  int isOut;
194  };
195  std::vector<CustomHeader3> v5{CustomHeader3{"TST", "A", 0, 1}, CustomHeader3{"TST", "A", 0, 0}};
196  std::sort(v5.begin(), v5.end(), [](CustomHeader3 const& lhs, CustomHeader3 const& rhs) {
197  return std::tie(lhs.origin, lhs.desc, rhs.subSpec, lhs.isOut) < std::tie(rhs.origin, rhs.desc, rhs.subSpec, rhs.isOut);
198  });
199  BOOST_CHECK(v5[0].isOut == 0);
200  BOOST_CHECK(v5[1].isOut == 1);
201 }

254

261  BOOST_CHECK(h1 != nullptr);

265  BOOST_CHECK(h2 != nullptr);
266  BOOST_CHECK(0 == std::strcmp(h2->getName(), "somename"));

268  BOOST_CHECK(h2->serialization == gSerializationMethodNone);
269  BOOST_CHECK(h2->size() == sizeof(NameHeader<9>));
270  BOOST_CHECK(h2->getNameLength() == 9);
271 
272  // create new stack from stack and additional header
273  auto meta = test::MetaHeader{42};
274  Stack s2{s1, meta};
275  BOOST_CHECK(s2.size() == s1.size() + sizeof(decltype(meta)));
276 
277  //check dynamic construction - where we don't have the type information and need to
278  //work with BaseHeader pointers
279  const test::MetaHeader thead{2};
280  o2::header::BaseHeader const* bname = reinterpret_cast<BaseHeader const*>(&thead);
281  Stack ds2(s1, *bname);
282  BOOST_CHECK(ds2.size() == s1.size() + sizeof(thead));
283  BOOST_CHECK(std::memcmp(get<test::MetaHeader*>(ds2.data()), &thead, sizeof(thead)) == 0);
284 
285  auto* h3 = get<test::MetaHeader*>(s1.data());
286  BOOST_CHECK(h3 == nullptr);
287  h3 = get<test::MetaHeader*>(s2.data());
288  BOOST_REQUIRE(h3 != nullptr);
289  BOOST_CHECK(h3->flagsNextHeader == false);
290  h1 = get<DataHeader*>(s2.data());
291  BOOST_REQUIRE(h1 != nullptr);
292  BOOST_CHECK(h1->flagsNextHeader == true);
293 
294  // create stack from header and empty stack
295  Stack s3{meta, Stack{}};
296  BOOST_CHECK(s3.size() == sizeof(meta));
297  h3 = get<test::MetaHeader*>(s3.data());
298  BOOST_REQUIRE(h3 != nullptr);
299  // no next header to be flagged as the stack was empty
300  BOOST_CHECK(h3->flagsNextHeader == false);
301 
302  // create new stack from stack, empty stack, and header

304  BOOST_CHECK(s4.size() == s1.size() + sizeof(meta));

306  h3 = get<test::MetaHeader*>(s4.data());
