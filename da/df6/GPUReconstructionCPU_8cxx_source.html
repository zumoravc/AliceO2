61 template <class T, int I, typename... Args>
62 int GPUReconstructionCPUBackend::runKernelBackend(krnlSetup& _xyz, const Args&... args)
63 {
64  auto& x = _xyz.x;
65  auto& y = _xyz.y;
66  if (x.device == krnlDeviceType::Device) {
67  throw std::runtime_error("Cannot run device kernel on host");
68  }
69  if (x.nThreads != 1) {
70  throw std::runtime_error("Cannot run device kernel on host with nThreads != 1");
71  }
72  unsigned int num = y.num == 0 || y.num == -1 ? 1 : y.num;
73  for (unsigned int k = 0; k < num; k++) {
74  if (mDeviceProcessingSettings.ompKernels) {
75  GPUCA_OPENMP(parallel for num_threads(mDeviceProcessingSettings.nThreads))
76  for (unsigned int iB = 0; iB < x.nBlocks; iB++) {
77  typename T::GPUSharedMemory smem;
78  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
79  }
80  } else {
81  for (unsigned int iB = 0; iB < x.nBlocks; iB++) {
82  typename T::GPUSharedMemory smem;
83  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
84  }
85  }
86  }
87  return 0;
88 }

112 size_t GPUReconstructionCPU::TransferMemoryResourcesHelper(GPUProcessor* proc, int stream, bool all, bool toGPU)
113 {
114  int inc = toGPU ? GPUMemoryResource::MEMORY_INPUT_FLAG : GPUMemoryResource::MEMORY_OUTPUT_FLAG;
115  int exc = toGPU ? GPUMemoryResource::MEMORY_OUTPUT_FLAG : GPUMemoryResource::MEMORY_INPUT_FLAG;
116  size_t n = 0;
117  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
118  GPUMemoryResource& res = mMemoryResources[i];
119  if (res.mPtr == nullptr) {
120  continue;
121  }
122  if (proc && res.mProcessor != proc) {
123  continue;
124  }
125  if (!(res.mType & GPUMemoryResource::MEMORY_GPU) || (res.mType & GPUMemoryResource::MEMORY_CUSTOM_TRANSFER)) {
126  continue;
127  }
128  if (!mDeviceProcessingSettings.keepAllMemory && !all && (res.mType & exc) && !(res.mType & inc)) {
129  continue;
130  }
131  if (toGPU) {
132  n += TransferMemoryResourceToGPU(&mMemoryResources[i], stream);
133  } else {
134  n += TransferMemoryResourceToHost(&mMemoryResources[i], stream);
135  }
136  }
137  return n;
138 }

195 int GPUReconstructionCPU::RunChains()
196 {
197  mStatNEvents++;
198  mNEventsProcessed++;
199 
200  if (mThreadId != GetThread()) {
201  if (mDeviceProcessingSettings.debugLevel >= 2) {
202  GPUInfo("Thread changed, migrating context, Previous Thread: %d, New Thread: %d", mThreadId, GetThread());
203  }
204  mThreadId = GetThread();
205  }
206 
207  timerTotal.Start();
208  if (mSlaves.size() || mMaster) {
209  WriteConstantParams(); // Reinitialize
210  }
211  for (unsigned int i = 0; i < mChains.size(); i++) {
212  int retVal = mChains[i]->RunChain();
213  if (retVal) {
214  return retVal;
215  }
216  }
217  timerTotal.Stop();
218 
219  mStatWallTime = (timerTotal.GetElapsedTime() * 1000000. / mStatNEvents);
220  if (GetDeviceProcessingSettings().debugLevel >= 1) {
221  double kernelTotal = 0;
222  std::vector<double> kernelStepTimes(N_RECO_STEPS);
223 
224  for (unsigned int i = 0; i < mTimers.size(); i++) {
225  double time = 0;
226  if (mTimers[i] == nullptr) {
227  continue;
228  }
229  for (int j = 0; j < mTimers[i]->num; j++) {
230  HighResTimer& timer = mTimers[i]->timer[j];
231  time += timer.GetElapsedTime();
232  if (mDeviceProcessingSettings.resetTimers) {
233  timer.Reset();
234  }
235  }
236 
237  char type = mTimers[i]->type;
238  if (type == 0) {
239  kernelTotal += time;
240  int stepNum = getRecoStepNum(mTimers[i]->step);
241  kernelStepTimes[stepNum] += time;
242  }
243  type = type == 0 ? 'K' : 'C';
244  char bandwidth[256] = "";
245  if (mTimers[i]->memSize && mStatNEvents && time != 0.) {
246  snprintf(bandwidth, 256, " (%6.3f GB/s - %'14lu bytes)", mTimers[i]->memSize / time * 1e-9, (unsigned long)(mTimers[i]->memSize / mStatNEvents));
247  }
248  printf("Execution Time: Task (%c %8ux): %50s Time: %'10d us%s\n", type, mTimers[i]->count, mTimers[i]->name.c_str(), (int)(time * 1000000 / mStatNEvents), bandwidth);
249  if (mDeviceProcessingSettings.resetTimers) {
250  mTimers[i]->count = 0;
251  mTimers[i]->memSize = 0;
252  }
253  }
254  for (int i = 0; i < N_RECO_STEPS; i++) {
255  if (kernelStepTimes[i] != 0.) {
256  printf("Execution Time: Step : %50s Time: %'10d us\n", GPUDataTypes::RECO_STEP_NAMES[i], (int)(kernelStepTimes[i] * 1000000 / mStatNEvents));
257  }
258  if (mTimersRecoSteps[i].bytesToGPU) {
259  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10d us (%6.3f GB/s - %'14lu bytes - %'14lu per call)\n", mTimersRecoSteps[i].countToGPU, "DMA to GPU", GPUDataTypes::RECO_STEP_NAMES[i], (int)(mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1000000 / mStatNEvents),
260  mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToGPU / mStatNEvents, mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].countToGPU);
261  }
262  if (mTimersRecoSteps[i].bytesToHost) {
263  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10d us (%6.3f GB/s - %'14lu bytes - %'14lu per call)\n", mTimersRecoSteps[i].countToHost, "DMA to Host", GPUDataTypes::RECO_STEP_NAMES[i], (int)(mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1000000 / mStatNEvents),
264  mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToHost / mStatNEvents, mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].countToHost);
265  }
266  if (mDeviceProcessingSettings.resetTimers) {
267  mTimersRecoSteps[i].bytesToGPU = mTimersRecoSteps[i].bytesToHost = 0;
268  mTimersRecoSteps[i].timerToGPU.Reset();
269  mTimersRecoSteps[i].timerToHost.Reset();
270  mTimersRecoSteps[i].timer.Reset();
271  mTimersRecoSteps[i].countToGPU = 0;
272  mTimersRecoSteps[i].countToHost = 0;
273  }
274  }
275  mStatKernelTime = kernelTotal * 1000000 / mStatNEvents;
276  printf("Execution Time: Total : %50s Time: %'10d us\n", "Total Kernel", (int)mStatKernelTime);
277  printf("Execution Time: Total : %50s Time: %'10d us\n", "Total Wall", (int)mStatWallTime);
278  } else if (GetDeviceProcessingSettings().debugLevel >= 0) {
279  printf("Total Wall Time: %'d us\n", (int)mStatWallTime);
280  }
281  if (mDeviceProcessingSettings.resetTimers) {
282  mStatNEvents = 0;
283  timerTotal.Reset();
284  }
285 
286  return 0;
287 }
