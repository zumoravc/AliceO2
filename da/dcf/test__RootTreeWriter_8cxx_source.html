58 template <typename T>
59 bool checkBranch(TTree& tree, BranchContent<T>&& content)
60 {
61  TBranch* branch = tree.GetBranch(content.branchName);
62  BOOST_REQUIRE(branch != nullptr);
63  T store;
64  T* pointer = &store;
65  // in general, pointer to pointer has to be used for setting the branch
66  // address to a store object, however this does not work for fundamental
67  // types, there the address to the variable has to be used in order
68  // to read back the value. Why? no clue.
69  if (std::is_fundamental<T>::value) {
70  branch->SetAddress(&store);
71  } else {
72  branch->SetAddress(&pointer);
73  }
74  branch->GetEntry(0);
75  BOOST_CHECK_MESSAGE(store == content.reference, "mismatch for branch " << content.branchName);
76  return store == content.reference;
77 }

79 template <typename T, typename... Args>
80 bool checkBranch(TTree& tree, BranchContent<T>&& content, Args&&... args)
81 {
82  return checkBranch(tree, std::forward<BranchContent<T>>(content)) && checkBranch(tree, std::forward<Args>(args)...);

85 template <typename... Args>
86 bool checkTree(const char* filename, const char* treename, Args&&... args)
87 {
88  TFile* file = TFile::Open(filename);
89  BOOST_REQUIRE(file != nullptr);
90  TTree* tree = reinterpret_cast<TTree*>(file->GetObjectChecked(treename, "TTree"));
91  BOOST_REQUIRE(tree != nullptr);
92  return checkBranch(*tree, std::forward<Args>(args)...);
93 }

100  using Container = std::vector<o2::test::Polymorphic>;
101  // setting up the writer with two branch definitions
102  // first definition is for a single input and simple type written to one branch
103  // second branch handles two inputs of the same data type, the mapping of the
104  // input data to the target branch is taken from the sub specification
105  auto getIndex = [](o2::framework::DataRef const& ref) -> size_t {
106  auto const* dataHeader = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
107  return dataHeader->subSpecification;
108  };
109  auto getName = [](std::string base, size_t i) -> std::string {
110  return base + "_" + std::to_string(i);
111  };
112  auto customClose = [](TFile* file, TTree* tree) {
113  // branches are filled independently of the tree, so the tree state needs to be
114  // synchronized with the branch states
115  tree->SetEntries();
116  std::cout << "Custom close, tree has " << tree->GetEntries() << " entries" << std::endl;
117  // there was one write cycle and each branch should have one entry
118  BOOST_CHECK(tree->GetEntries() == 1);
119  tree->Write();
120  file->Close();
121  };
122  RootTreeWriter writer(filename.c_str(), treename, // file and tree name
123  customClose,
124  RootTreeWriter::BranchDef<int>{"input1", "intbranch"},
125  RootTreeWriter::BranchDef<Container>{
126  std::vector<std::string>({"input2", "input3"}), "containerbranch",
127  // define two target branches (this matches the input list)
128  2,
129  // the callback extracts the sub specification from the DataHeader as index
130  getIndex,
131  // the branch names are simply built by adding the index
132  getName},
133  RootTreeWriter::BranchDef<const char*>{"input4", "binarybranch"},
134  RootTreeWriter::BranchDef<o2::test::TriviallyCopyable>{"input6", "msgablebranch"},
135  RootTreeWriter::BranchDef<std::vector<int>>{"input6", "intvecbranch"},
136  RootTreeWriter::BranchDef<std::vector<o2::test::TriviallyCopyable>>{"input7", "trivvecbranch"},
137  // TriviallyCopyable can be sent with either serialization methods NONE or ROOT
138  RootTreeWriter::BranchDef<std::vector<o2::test::TriviallyCopyable>>{"input8", "srlzdvecbranch"});
139 
140  BOOST_CHECK(writer.getStoreSize() == 7);
141 
142  // need to mimic a context to actually call the processing
143  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
144  std::vector<FairMQMessagePtr> store;
145 
146  auto createPlainMessage = [&transport, &store](DataHeader&& dh, auto& data) {
147  dh.payloadSize = sizeof(data);
148  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
149  DataProcessingHeader dph{0, 1};
150  o2::header::Stack stack{dh, dph};
151  FairMQMessagePtr header = transport->CreateMessage(stack.size());
152  FairMQMessagePtr payload = transport->CreateMessage(sizeof(data));
153  memcpy(header->GetData(), stack.data(), stack.size());
154  memcpy(payload->GetData(), &data, sizeof(data));
155  store.emplace_back(std::move(header));
156  store.emplace_back(std::move(payload));
157  };
158 
159  auto createVectorMessage = [&transport, &store](DataHeader&& dh, auto& data) {
160  dh.payloadSize = data.size() * sizeof(typename std::remove_reference<decltype(data)>::type::value_type);
161  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
162  DataProcessingHeader dph{0, 1};
163  o2::header::Stack stack{dh, dph};
164  FairMQMessagePtr header = transport->CreateMessage(stack.size());
165  FairMQMessagePtr payload = transport->CreateMessage(dh.payloadSize);
166  memcpy(header->GetData(), stack.data(), stack.size());
167  memcpy(payload->GetData(), data.data(), dh.payloadSize);
168  store.emplace_back(std::move(header));
169  store.emplace_back(std::move(payload));
170  };
171 
172  auto createSerializedMessage = [&transport, &store](DataHeader&& dh, auto& data) {
173  FairMQMessagePtr payload = transport->CreateMessage();
174  auto* cl = TClass::GetClass(typeid(decltype(data)));
175  TMessageSerializer().Serialize(*payload, &data, cl);
176  dh.payloadSize = payload->GetSize();
177  dh.payloadSerializationMethod = o2::header::gSerializationMethodROOT;
178  DataProcessingHeader dph{0, 1};
179  o2::header::Stack stack{dh, dph};
180  FairMQMessagePtr header = transport->CreateMessage(stack.size());
181  memcpy(header->GetData(), stack.data(), stack.size());
182  store.emplace_back(std::move(header));
183  store.emplace_back(std::move(payload));
184  };

201  // Note: InputRecord works on references to the schema and the message vector

203  // the InputRecord. Intrestingly enough, the compiler does not complain about
204  // getting reference to temporary rvalue argument. So it might work if the

212  {InputSpec{"input5", "TST", "MSGABLE"}, 5, "input5", 0}, //
213  {InputSpec{"input6", "TST", "FDMTLVEC"}, 6, "input6", 0}, //
214  {InputSpec{"input7", "TST", "TRIV_VEC"}, 7, "input7", 0}, //
215  {InputSpec{"input8", "TST", "SRLZDVEC"}, 8, "input8", 0}, //
216  };
217 
218  auto getter = [&store](size_t i) -> DataRef {
219  return DataRef{nullptr, static_cast<char const*>(store[2 * i]->GetData()), static_cast<char const*>(store[2 * i + 1]->GetData())};
220  };
221 
222  InputRecord inputs{
223  schema,
224  InputSpan{getter, store.size() / 2}};
225 
226  writer(inputs);
