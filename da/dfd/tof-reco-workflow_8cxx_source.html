47 {
48  workflowOptions.push_back(ConfigParamSpec{"input-type", o2::framework::VariantType::String, "digits", {"digits, raw, clusters, TBI"}});
49  workflowOptions.push_back(ConfigParamSpec{"output-type", o2::framework::VariantType::String, "clusters,matching-info,calib-info", {"digits,clusters, matching-info, calib-info, TBI"}});
50  workflowOptions.push_back(ConfigParamSpec{"disable-mc", o2::framework::VariantType::Bool, false, {"disable sending of MC information, TBI"}});
51  workflowOptions.push_back(ConfigParamSpec{"tof-sectors", o2::framework::VariantType::String, "0-17", {"TOF sector range, e.g. 5-7,8,9 ,TBI"}});
52  workflowOptions.push_back(ConfigParamSpec{"tof-lanes", o2::framework::VariantType::Int, 1, {"number of parallel lanes up to the matcher, TBI"}});
53  workflowOptions.push_back(ConfigParamSpec{"use-ccdb", o2::framework::VariantType::Bool, false, {"enable access to ccdb tof calibration objects"}});
54  workflowOptions.push_back(ConfigParamSpec{"use-fit", o2::framework::VariantType::Bool, false, {"enable access to fit info for calibration"}});
55  workflowOptions.push_back(ConfigParamSpec{"input-desc", o2::framework::VariantType::String, "CRAWDATA", {"Input specs description string"}});
56  workflowOptions.push_back(ConfigParamSpec{"disable-root-input", o2::framework::VariantType::Bool, false, {"disable root-files input readers"}});
57  workflowOptions.push_back(ConfigParamSpec{"disable-root-output", o2::framework::VariantType::Bool, false, {"disable root-files output writers"}});
58  workflowOptions.push_back(ConfigParamSpec{"configKeyValues", o2::framework::VariantType::String, "", {"Semicolon separated key=value strings ..."}});
59 }

79 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
80 {
81  WorkflowSpec specs;
82 
83  o2::base::Propagator::initFieldFromGRP("o2sim_grp.root");
84  auto grp = o2::parameters::GRPObject::loadFrom("o2sim_grp.root");
85 
86  if (!grp) {
87  LOG(ERROR) << "This workflow needs a valid GRP file to start";
88  return specs;
89  }
90  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
91  // o2::conf::ConfigurableParam::writeINI("o2tofrecoflow_configuration.ini");
92 
93  // the lane configuration defines the subspecification ids to be distributed among the lanes.
94  // auto tofSectors = o2::RangeTokenizer::tokenize<int>(cfgc.options().get<std::string>("tof-sectors"));
95  // std::vector<int> laneConfiguration = tofSectors;
96  auto nLanes = cfgc.options().get<int>("tof-lanes");
97  auto inputType = cfgc.options().get<std::string>("input-type");
98  auto outputType = cfgc.options().get<std::string>("output-type");
99 
100  bool writecluster = 0;
101  bool writematching = 0;
102  bool writecalib = 0;
103  bool writedigit = 0;
104  bool writeraw = 0;
105 
106  if (outputType.rfind("clusters") < outputType.size())
107  writecluster = 1;
108  if (outputType.rfind("matching-info") < outputType.size())
109  writematching = 1;
110  if (outputType.rfind("calib-info") < outputType.size())
111  writecalib = 1;
112  if (outputType.rfind("digits") < outputType.size())
113  writedigit = 1;
114  if (outputType.rfind("raw") < outputType.size())
115  writeraw = 1;
116 
117  bool dgtinput = 0;
118  if (inputType == "digits") {
119  dgtinput = 1;
120  }
121  bool clusterinput = 0;
122  if (inputType == "clusters") {
123  clusterinput = 1;
124  }
125  bool rawinput = 0;
126  if (inputType == "raw") {
127  rawinput = 1;
128  }
129 
130  auto useMC = !cfgc.options().get<bool>("disable-mc");
131  auto useCCDB = cfgc.options().get<bool>("use-ccdb");
132  auto useFIT = cfgc.options().get<bool>("use-fit");
133  bool disableRootInput = cfgc.options().get<bool>("disable-root-input") || rawinput;
134  bool disableRootOutput = cfgc.options().get<bool>("disable-root-output");
135 
136  LOG(INFO) << "TOF RECO WORKFLOW configuration";
137  LOG(INFO) << "TOF input = " << cfgc.options().get<std::string>("input-type");
138  LOG(INFO) << "TOF output = " << cfgc.options().get<std::string>("output-type");
139  LOG(INFO) << "TOF sectors = " << cfgc.options().get<std::string>("tof-sectors");
140  LOG(INFO) << "TOF disable-mc = " << cfgc.options().get<std::string>("disable-mc");
141  LOG(INFO) << "TOF lanes = " << cfgc.options().get<std::string>("tof-lanes");
142  LOG(INFO) << "TOF use-ccdb = " << cfgc.options().get<std::string>("use-ccdb");
143  LOG(INFO) << "TOF use-fit = " << cfgc.options().get<std::string>("use-fit");
144  LOG(INFO) << "TOF disable-root-input = " << disableRootInput;
145  LOG(INFO) << "TOF disable-root-output = " << disableRootOutput;
146 
147  if (clusterinput) {
148  LOG(INFO) << "Insert TOF Cluster Reader";
149  specs.emplace_back(o2::tof::getClusterReaderSpec(useMC));
150  } else if (dgtinput) {
151  // TOF clusterizer
152  LOG(INFO) << "Insert TOF Digit reader from file";
153  specs.emplace_back(o2::tof::getDigitReaderSpec(useMC));
154 
155  if (writeraw) {
156  LOG(INFO) << "Insert TOF Raw writer";
157  specs.emplace_back(o2::tof::getTOFRawWriterSpec());
158  }
159  } else if (rawinput) {
160  LOG(INFO) << "Insert TOF Compressed Raw Decoder";
161  auto inputDesc = cfgc.options().get<std::string>("input-desc");
162  specs.emplace_back(o2::tof::getCompressedDecodingSpec(inputDesc));
163  useMC = 0;
164 
165  if (writedigit && !disableRootOutput) {
166  // add TOF digit writer without mc labels
167  LOG(INFO) << "Insert TOF Digit Writer";
168  specs.emplace_back(o2::tof::getTOFDigitWriterSpec(0));
169  }
170  }
171 
172  if (!clusterinput && writecluster) {
173  LOG(INFO) << "Insert TOF Clusterizer";
174  specs.emplace_back(o2::tof::getTOFClusterizerSpec(useMC, useCCDB));
175  if (writecluster && !disableRootOutput) {
176  LOG(INFO) << "Insert TOF Cluster Writer";
177  specs.emplace_back(o2::tof::getTOFClusterWriterSpec(useMC));
178  }
179  }
180 
181  if (useFIT) {
182  specs.emplace_back(o2::ft0::getFT0RecPointReaderSpec(useMC));
183  }
184 
185  if (writematching || writecalib) {
186  if (!disableRootInput) {
187  LOG(INFO) << "Insert ITS-TPC Track Reader";
188  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(useMC));
189  }
190  LOG(INFO) << "Insert TOF Matching";
191  specs.emplace_back(o2::tof::getTOFRecoWorkflowSpec(useMC, useFIT));
192 
193  if (writematching && !disableRootOutput) {
194  LOG(INFO) << "Insert TOF Matched Info Writer";
195  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC));
196  }
197  if (writecalib) {
198  LOG(INFO) << "Insert TOF Calib Info Writer";
199  specs.emplace_back(o2::tof::getTOFCalibWriterSpec());
200  }
201  }
202  LOG(INFO) << "Number of active devices = " << specs.size();
203 
204  return std::move(specs);
205 }
