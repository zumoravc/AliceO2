50 void Hall::createMaterials()
51 {
52  auto& matmgr = o2::base::MaterialManager::Instance();
53 
54  //
55  // Create materials for the experimental hall
56  //
57  Int_t isxfld = 2.;
58  Float_t sxmgmx = 10.;
59  o2::base::Detector::initFieldTrackingParams(isxfld, sxmgmx);
60  isxfld = 0;
61 
62  Float_t aconc[10] = {1., 12.01, 15.994, 22.99, 24.305, 26.98, 28.086, 39.1, 40.08, 55.85};
63  Float_t zconc[10] = {1., 6., 8., 11., 12., 13., 14., 19., 20., 26.};
64  Float_t wconc[10] = {.01, .001, .529107, .016, .002, .033872, .337021, .013, .044, .014};
65 
66  Float_t aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
67  Float_t zAir[4] = {6., 7., 8., 18.};
68  Float_t wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
69  Float_t dAir = 1.20479E-3;
70 
71  // Steel
72  Float_t asteel[4] = {55.847, 51.9961, 58.6934, 28.0855};
73  Float_t zsteel[4] = {26., 24., 28., 14.};
74  Float_t wsteel[4] = {.715, .18, .1, .005};
75 
76  // ****************
77  // Defines tracking media parameters.
78  // Les valeurs sont commentees pour laisser le defaut
79  // a GEANT (version 3-21, page CONS200), f.m.
80  Float_t epsil, stmin, deemax, tmaxfd, stemax;
81  epsil = .001; // Tracking precision,
82  stemax = -1.; // Maximum displacement for multiple scat
83  tmaxfd = -20.; // Maximum angle due to field deflection
84  deemax = -.3; // Maximum fractional energy loss, DLS
85  stmin = -.8;
86  // ***************
87 
88  // only media needed for geometry are created
89 
90  // Stainless Steel
91  matmgr.Mixture("HALL", kSTST_C2, "STAINLESS STEEL3", asteel, zsteel, 7.88, 4, wsteel);
92  matmgr.Medium("HALL", kSTST_C2, "STST_C2", kSTST_C2, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
93 
94  // Air
95  matmgr.Mixture("HALL", kAIR_C2, "AIR2", aAir, zAir, dAir, 4, wAir);
96  matmgr.Medium("HALL", kAIR_C2, "AIR_C2", kAIR_C2, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
97 
98  // Concrete
99  matmgr.Mixture("HALL", kCC_C2, "CONCRETE2", aconc, zconc, 2.35, 10, wconc);
100  matmgr.Medium("HALL", kCC_C2, "CC_C2", kCC_C2, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
101 
102  matmgr.Mixture("HALL", kCC_C0, "CONCRETE0", aconc, zconc, 2.35, 10, wconc);
103  matmgr.Medium("HALL", kCC_C0, "CC_C0", kCC_C0, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
104  // Iron
105  matmgr.Material("HALL", kFE_C2, "IRON", 55.85, 26., 7.87, 1.76, 17.1);
106  matmgr.Medium("HALL", kFE_C2, "FE_C2", kFE_C2, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
107 }
