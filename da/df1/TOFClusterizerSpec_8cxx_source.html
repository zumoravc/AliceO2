38 // use the tasking system of DPL
39 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
40 class TOFDPLClustererTask
41 {
42  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
43  bool mUseMC = true;
44  bool mUseCCDB = false;
45 
46  public:
47  explicit TOFDPLClustererTask(bool useMC, bool useCCDB) : mUseMC(useMC), mUseCCDB(useCCDB) {}
48  void init(framework::InitContext& ic)
49  {
50  // nothing special to be set up
51  mTimer.Stop();
52  mTimer.Reset();
53  }

55  void run(framework::ProcessingContext& pc)
56  {
57  mTimer.Start(false);
58  // get digit data
59  auto digits = pc.inputs().get<gsl::span<o2::tof::Digit>>("tofdigits");
60  auto row = pc.inputs().get<std::vector<o2::tof::ReadoutWindowData>*>("readoutwin");
61 
62  auto labelvector = std::make_shared<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>>();
63  if (mUseMC) {
64  auto digitlabels = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
65  *labelvector.get() = std::move(*digitlabels);
66  mClusterer.setMCTruthContainer(&mClsLabels);
67  mClsLabels.clear();
68  }
69 
70  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
71  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObj;
72 
73  if (mUseCCDB) { // read calibration objects from ccdb
74  // check LHC phase
75  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
76  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
77 
78  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
79  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
80 
81  // make a copy in global scope
82  lhcPhaseObj = lhcPhaseObjTmp;
83  channelCalibObj = channelCalibObjTmp;
84  } else { // calibration objects set to zero
85  lhcPhaseObj.addLHCphase(0, 0);
86  lhcPhaseObj.addLHCphase(2000000000, 0);
87 
88  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
89  channelCalibObj.addTimeSlewingInfo(ich, 0, 0);
90  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
91  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
92  channelCalibObj.setFractionUnderPeak(sector, channelInSector, 1);
93  }
94  }
95 
96  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, &channelCalibObj);
97 
98  mClusterer.setCalibApi(&calibapi);
99 
100  // call actual clustering routine
101  mClustersArray.clear();
102 
103  for (int i = 0; i < row->size(); i++) {
104  printf("# TOF readout window for clusterization = %d/%lu (N digits = %d)\n", i, row->size(), row->at(i).size());
105  auto digitsRO = row->at(i).getBunchChannelData(digits);
106 
107  mReader.setDigitArray(&digitsRO);
108  if (mUseMC) {
109  mClusterer.process(mReader, mClustersArray, &(labelvector->at(i)));
110  } else
111  mClusterer.process(mReader, mClustersArray, nullptr);
112  }
113  LOG(INFO) << "TOF CLUSTERER : TRANSFORMED " << digits.size()
114  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
115 
116  // send clusters
117  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe}, mClustersArray);
118  // send labels
119  if (mUseMC)
120  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mClsLabels);
121 
122  mTimer.Stop();
123  }
124 
125  void endOfStream(EndOfStreamContext& ec)
126  {
127  LOGF(INFO, "TOF Clusterer total timing: Cpu: %.3e Real: %.3e s in %d slots",
128  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
129  }
130 
131  private:
132  DigitDataReader mReader;
133  Clusterer mClusterer;
134  TStopwatch mTimer;
135 
136  std::vector<Cluster> mClustersArray;
137  MCLabelContainer mClsLabels;
138 };

140 o2::framework::DataProcessorSpec getTOFClusterizerSpec(bool useMC, bool useCCDB)
141 {
142  std::vector<InputSpec> inputs;
143  inputs.emplace_back("tofdigits", o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
144  inputs.emplace_back("readoutwin", o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
145  if (useCCDB) {
146  inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
147  inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
148  }
149  if (useMC)
150  inputs.emplace_back("tofdigitlabels", o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
151 
152  return DataProcessorSpec{
153  "TOFClusterer",
154  inputs,
155  Outputs{OutputSpec{o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe},
156  OutputSpec{o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe}},
157  AlgorithmSpec{adaptFromTask<TOFDPLClustererTask>(useMC, useCCDB)},
158  Options{/* for the moment no options */}};
159 }
