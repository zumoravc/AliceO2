29 bool o2::data_flow::HeartbeatSampler::ConditionalRun()
30 {
31  std::this_thread::sleep_for(std::chrono::nanoseconds(mPeriod));
32 
33  o2::header::HeartbeatStatistics hbfPayload;
34 
35  o2::header::DataHeader dh;
36  dh.dataDescription = o2::header::gDataDescriptionHeartbeatFrame;
37  dh.dataOrigin = o2::header::DataOrigin("SMPL");
38  dh.subSpecification = 0;
39  dh.payloadSize = sizeof(hbfPayload);
40 
41  // Note: the block type of both header an trailer members of the envelope
42  // structure are autmatically initialized to the appropriate block type
43  // and size '1' (i.e. only one 64bit word)
44  o2::header::HeartbeatFrameEnvelope specificHeader;
45  specificHeader.header.orbit = mCount;
46  specificHeader.trailer.hbAccept = 1;
47 
48  O2Message outgoing;
49 
50  // build multipart message from header and payload
51  o2::base::addDataBlock(outgoing, {dh, specificHeader}, NewSimpleMessage(hbfPayload));
52 
53  // send message
54  Send(outgoing, mOutputChannelName.c_str());
55  outgoing.fParts.clear();
56 
57  mCount++;
58  return true;
59 }
