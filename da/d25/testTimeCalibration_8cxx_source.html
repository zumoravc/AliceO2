30 BOOST_AUTO_TEST_CASE(testTimeCalibration)
31 {
32 
33  // Single channel test
34  //
35  // For each channel test set and update, and compare read value
36  // against set value
37  for (unsigned short c = 0; c < 17664; c++) {
38  TimeCalibrationParams singletest;
39  singletest.addTimeCalibParam(c, 0, 0); //HG
40  BOOST_CHECK_EQUAL(singletest.getTimeCalibParam(c, 0), 0);
41 
42  singletest.addTimeCalibParam(c, 600, 0); //HG
43  BOOST_CHECK_EQUAL(singletest.getTimeCalibParam(c, 0), 600);
44 
45  singletest.addTimeCalibParam(c, 0, 1); //LG
46  BOOST_CHECK_EQUAL(singletest.getTimeCalibParam(c, 1), 0);
47 
48  singletest.addTimeCalibParam(c, 600, 1); //LG
49  BOOST_CHECK_EQUAL(singletest.getTimeCalibParam(c, 1), 600);
50  }
51 
52  // Pattern test
53  //
54  // For each channel check the wheather the time calibration coeffcient is correctly written
55  // Test data obtained from LHC18b
56  TimeCalibrationParams parameter;
57 
58  const char* aliceO2env = std::getenv("O2_ROOT");
59  std::string inputDir = " ";
60  if (aliceO2env)
61  inputDir = aliceO2env;
62  inputDir += "/share/Detectors/EMC/files/";
63 
64  std::string fileHG = inputDir + "TimeCalibCoeffHG.txt";
65  std::ifstream allTimeAvHG(fileHG, std::ifstream::in);
66  if (!allTimeAvHG.is_open())
67  std::cout << "The file TimeCalibCoeffHG was not opened\n";
68 
69  std::string fileLG = inputDir + "TimeCalibCoeffLG.txt";
70  std::ifstream allTimeAvLG(fileLG, std::ifstream::in);
71  if (!allTimeAvLG.is_open())
72  std::cout << "The file TimeCalibCoeffLG was not opened\n";
73 
74  unsigned short TimeHG[17664], TimeLG[17664];
75 
76  unsigned short icell = 0;
77  while (1) {
78  allTimeAvHG >> TimeHG[icell];
79  if (!allTimeAvHG.good())
80  break;
81  parameter.addTimeCalibParam(icell, TimeHG[icell], 0); //HG
82  icell++;
83  }
84 
85  icell = 0;
86  while (1) {
87  allTimeAvLG >> TimeLG[icell];
88  if (!allTimeAvLG.good())
89  break;
90  parameter.addTimeCalibParam(icell, TimeLG[icell], 1); //LG
91  icell++;
92  }
93 
94  for (int icell = 0; icell < 17664; icell++) {
95  BOOST_CHECK_EQUAL(parameter.getTimeCalibParam(icell, 0), TimeHG[icell]);
96 
97  BOOST_CHECK_EQUAL(parameter.getTimeCalibParam(icell, 1), TimeLG[icell]);
98  }
99 
100  // Writing to the other container which will be used for comparison
101  TimeCalibrationParams parameterLHC17g;
102 
103  std::string fileHG_LHC17g = inputDir + "TimeCalibCoeffHG_LHC17g.txt";
104  std::ifstream allTimeAvHG_LHC17g(fileHG_LHC17g, std::ifstream::in);

107

110  if (!allTimeAvLG_LHC17g.is_open())

112

114 
115  icell = 0;

117  allTimeAvHG_LHC17g >> TimeHG_LHC17g;
118  if (!allTimeAvHG_LHC17g.good())
119  break;
120  parameterLHC17g.addTimeCalibParam(icell, TimeHG_LHC17g, 0); //HG

122  }
123 
124  icell = 0;

126  allTimeAvLG_LHC17g >> TimeLG_LHC17g;
127  if (!allTimeAvLG_LHC17g.good())
128  break;
129  parameterLHC17g.addTimeCalibParam(icell, TimeLG_LHC17g, 1); //LG
130  icell++;
131  }
132 
133  // Equal
134  //
135  // - Compare time calibration for LHC18b with itself. The result must be true.
136  // - Compare time calibration for LHC18b with time calibration for LHC17g. The result must be false
137  BOOST_CHECK_EQUAL(parameter == parameter, true);
138  BOOST_CHECK_EQUAL(parameter == parameterLHC17g, false);
139 
140  allTimeAvHG.close();
141  allTimeAvLG.close();
142  allTimeAvHG_LHC17g.close();
143  allTimeAvLG_LHC17g.close();
