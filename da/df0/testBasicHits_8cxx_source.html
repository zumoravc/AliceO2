22 BOOST_AUTO_TEST_CASE(BasicXYZHit)
23 {
24  using HitType = BasicXYZEHit<float>;
25  HitType hit(1., 2., 3., 0.01, -1.1, -1, 1);
26 
27  BOOST_CHECK_CLOSE(hit.GetX(), 1., 1E-4);
28  BOOST_CHECK_CLOSE(hit.GetY(), 2., 1E-4);
29  BOOST_CHECK_CLOSE(hit.GetZ(), 3., 1E-4);
30  BOOST_CHECK_CLOSE(hit.GetEnergyLoss(), -1.1, 1E-4);
31  BOOST_CHECK_CLOSE(hit.GetTime(), 0.01, 1E-4);
32 
33  hit.SetX(0.);
34  BOOST_CHECK_CLOSE(hit.GetX(), 0., 1E-4);
35 
36  // check coordinate transformation of the hit coordinate
37  // check that is works with float + double
38  // note that ROOT transformations are always double valued
39  using ROOT::Math::Transform3D;
40  Transform3D idtransf; // defaults to identity transformation
41 
42  auto transformed = idtransf(hit.GetPos());
43  BOOST_CHECK_CLOSE(transformed.Y(), hit.GetY(), 1E-4);
44 }
