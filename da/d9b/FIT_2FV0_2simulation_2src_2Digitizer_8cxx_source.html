32 void Digitizer::init()
33 {
34  LOG(INFO) << "V0Digitizer::init -> start = ";
35 
36  mNBins = FV0DigParam::Instance().waveformNbins; //Will be computed using detector set-up from CDB
37  mBinSize = FV0DigParam::Instance().waveformBinWidth; //Will be set-up from CDB
38 
39  for (Int_t i = 0; i < DP::NCHANNELS; i++) {
40  mPmtChargeVsTime[i].resize(mNBins);
41  }
42 
43  auto const roundVc = [&](int i) -> int {
44  return (i / Vc::float_v::Size) * Vc::float_v::Size;
45  };
46  // set up PMT response tables
47  Float_t offset = -0.5f * mBinSize; // offset \in [-0.5..0.5] * mBinSize
48  Int_t const nBins = roundVc(std::lround(4.0f * FV0DigParam::Instance().pmtTransitTime / mBinSize));
49  for (auto& table : mPmtResponseTables) {
50  table.resize(nBins);
51  Float_t t = -2.0f * FV0DigParam::Instance().pmtTransitTime + offset; // t \in offset + [-2 2] * FV0DigParam::Instance().mPmtTransitTime
52  for (Int_t j = 0; j < nBins; ++j) {
53  table[j] = Digitizer::PmtResponse(t);
54  t += mBinSize;
55  }
56  offset += mBinSize / Float_t(DP::NUM_PMT_RESPONSE_TABLES - 1);
57  }
58 
59  TF1 scintDelayFn("fScintDelay", "gaus",
60  -6.0f * FV0DigParam::Instance().intrinsicTimeRes,
61  +6.0f * FV0DigParam::Instance().intrinsicTimeRes);
62  scintDelayFn.SetParameters(1, 0, FV0DigParam::Instance().intrinsicTimeRes);
63  mRndScintDelay.initialize(scintDelayFn);
64 
65  // Initialize function describing the PMT time response
66  TF1 pmtResponseFn("mPmtResponse",
67  &Digitizer::PmtResponse,
68  -1.0f * FV0DigParam::Instance().pmtTransitTime,
69  +2.0f * FV0DigParam::Instance().pmtTransitTime, 0);
70  pmtResponseFn.SetNpx(100);
71  mPmtTimeIntegral = pmtResponseFn.Integral(-1.0f * FV0DigParam::Instance().pmtTransitTime,
72  +2.0f * FV0DigParam::Instance().pmtTransitTime);
73 
74  // Initialize function describing PMT response to the single photoelectron
75  TF1 singlePhESpectrumFn("mSinglePhESpectrum",
76  &Digitizer::SinglePhESpectrum,
77  FV0DigParam::Instance().photoelMin,
78  FV0DigParam::Instance().photoelMax, 0);
79  Float_t const meansPhE = singlePhESpectrumFn.Mean(FV0DigParam::Instance().photoelMin, FV0DigParam::Instance().photoelMax);
80  mRndGainVar.initialize([&]() -> float {
81  return singlePhESpectrumFn.GetRandom(FV0DigParam::Instance().photoelMin, FV0DigParam::Instance().photoelMax) / meansPhE;
82  });
83 
84  TF1 signalShapeFn("signalShape", "crystalball", 0, 200);
85  signalShapeFn.SetParameters(FV0DigParam::Instance().shapeConst,
86  FV0DigParam::Instance().shapeMean,
87  FV0DigParam::Instance().shapeSigma,
88  FV0DigParam::Instance().shapeAlpha,
89  FV0DigParam::Instance().shapeN);
90  mRndSignalShape.initialize([&]() -> float {
91  return signalShapeFn.GetRandom(0, mBinSize * Float_t(mNBins));
92  });
93 
94  LOG(INFO) << "V0Digitizer::init -> finished";
95 }

97 void Digitizer::process(const std::vector<o2::fv0::Hit>& hits)
98 {
99  LOG(INFO) << "[FV0] Digitizer::process(): begin with " << hits.size() << " hits";
100 
101  std::vector<int> hitIdx(hits.size());
102  std::iota(std::begin(hitIdx), std::end(hitIdx), 0);
103  std::sort(std::begin(hitIdx), std::end(hitIdx), [&hits](int a, int b) { return hits[a].GetTrackID() < hits[b].GetTrackID(); });
104 
105  auto const roundVc = [&](int i) -> int {
106  return (i / Vc::float_v::Size) * Vc::float_v::Size;
107  };
108  Int_t parentIdPrev = -10;
109  // use ordered hits
110  for (auto ids : hitIdx) {
111  const auto& hit = hits[ids];
112  Int_t detId = hit.GetDetectorID();
113  Double_t hitEdep = hit.GetHitValue() * 1e3; //convert to MeV
114 
115  // TODO: check how big is inaccuracy if more than 1 'below-threshold' particles hit the same detector cell
116  if (hitEdep < FV0DigParam::Instance().singleMipThreshold) {
117  continue;
118  }
119 
120  float distanceFromXc = 0;
121  if (Geometry::instance()->isRing5(detId)) {
122  distanceFromXc = getDistFromCellCenter(detId, hit.GetX(), hit.GetY());
123  }
124 
125  int iChannelPerCell = 0;
126  while (iChannelPerCell < 2) { // loop over 2 channels, into which signal from each cell in ring 5 is split
127  if (Geometry::instance()->isRing5(detId)) {
128  // The first channel number is located counter-clockwise from the cell center
129  // and remains identical to the detector number, the second one is clockwise and incremented by 8
130  if (iChannelPerCell == 1) {
131  detId += 8;
132  }
133 
134  // Split signal magnitude to fractions depending on the distance of the hit from the cell center
135  hitEdep = (hit.GetHitValue() * 1e3) * getSignalFraction(distanceFromXc, iChannelPerCell == 0);
136  LOG(INFO) << " detId: " << detId << "-" << iChannelPerCell << " hitEdep: " << hitEdep << " distanceFromXc: " << distanceFromXc;
137  ++iChannelPerCell;
138  } else {
139  iChannelPerCell = 2; // not a ring 5 cell -> don't repeat the loop
140  }
141 
142  Double_t const nPhotons = hitEdep * DP::N_PHOTONS_PER_MEV;
143  Int_t const nPhE = SimulateLightYield(detId, nPhotons);
144  Float_t const t = hit.GetTime() * 1e9 + FV0DigParam::Instance().pmtTransitTime;
145  Float_t const charge = TMath::Qe() * FV0DigParam::Instance().pmtGain * mBinSize / mPmtTimeIntegral;
146 
147  auto& analogSignal = mPmtChargeVsTime[detId];
148  for (Int_t iPhE = 0; iPhE < nPhE; ++iPhE) {
149  Float_t const tPhE = t + mRndSignalShape.getNextValue();
150  Int_t const firstBin = roundVc(
151  TMath::Max((Int_t)0, (Int_t)((tPhE - FV0DigParam::Instance().pmtTransitTime) / mBinSize)));
152  Int_t const lastBin = TMath::Min((Int_t)mNBins - 1,
153  (Int_t)((tPhE + 2. * FV0DigParam::Instance().pmtTransitTime) / mBinSize));
154  Float_t const tempT = mBinSize * (0.5f + firstBin) - tPhE;
155  Float_t* p = analogSignal.data() + firstBin;
156  long iStart = std::lround((tempT + 2.0f * FV0DigParam::Instance().pmtTransitTime) / mBinSize);
157  float const offset = tempT + 2.0f * FV0DigParam::Instance().pmtTransitTime - Float_t(iStart) * mBinSize;
158  long const iOffset = std::lround(offset / mBinSize * Float_t(DP::NUM_PMT_RESPONSE_TABLES - 1));
159  if (iStart < 0) { // this should not happen
160  LOG(ERROR) << "V0Digitizer: table lookup failure";
161  }
162  iStart = roundVc(std::max(long(0), iStart));
163 
164  Vc::float_v workVc;
165  Vc::float_v pmtVc;
166  Float_t const* q = mPmtResponseTables[DP::NUM_PMT_RESPONSE_TABLES / 2 + iOffset].data() + iStart;
167  Float_t const* qEnd = &mPmtResponseTables[DP::NUM_PMT_RESPONSE_TABLES / 2 + iOffset].back();
168  for (Int_t i = firstBin, iEnd = roundVc(lastBin); q < qEnd && i < iEnd; i += Vc::float_v::Size) {
169  pmtVc.load(q);
170  q += Vc::float_v::Size;
171  Vc::prefetchForOneRead(q);
172  workVc.load(p);
173  workVc += mRndGainVar.getNextValueVc() * charge * pmtVc;
174  workVc.store(p);
175  p += Vc::float_v::Size;
176  Vc::prefetchForOneRead(p);
177  }
178  } //photo electron loop
179 
180  // Charged particles in MCLabel
181  Int_t const parentId = hit.GetTrackID();
182  if (parentId != parentIdPrev) {
183  mMCLabels.emplace_back(parentId, mEventId, mSrcId, detId);
184  parentIdPrev = parentId;
185  }
186  }
187  } //hit loop
188 }

190 void Digitizer::analyseWaveformsAndStore(std::vector<fv0::BCData>& digitsBC,
191  std::vector<fv0::ChannelData>& digitsCh,
192  dataformats::MCTruthContainer<fv0::MCLabel>& labels)
193 {
194  // Sum charge of all time bins to get total charge collected for a given channel
195  size_t const first = digitsCh.size();
196  size_t nStored = 0;
197  for (Int_t ipmt = 0; ipmt < DP::NCHANNELS; ++ipmt) {
198  Float_t totalCharge = 0.0f;
199  auto const& analogSignal = mPmtChargeVsTime[ipmt];
200  for (Int_t iTimeBin = 0; iTimeBin < mNBins; ++iTimeBin) {
201  Float_t const timeBinCharge = mPmtChargeVsTime[ipmt][iTimeBin];
202  totalCharge += timeBinCharge;
203  }
204  totalCharge *= DP::INV_CHARGE_PER_ADC;
205  digitsCh.emplace_back(ipmt, SimulateTimeCfd(ipmt), std::lround(totalCharge));
206  ++nStored;
207  }
208 
209  // Send MClabels and digitsBC to storage
210  size_t const nBC = digitsBC.size();
211  digitsBC.emplace_back(first, nStored, mIntRecord);
212  for (auto const& lbl : mMCLabels) {
213  labels.addElement(nBC, lbl);
214  }
215  mMCLabels.clear();
216 }

222 Int_t Digitizer::SimulateLightYield(Int_t pmt, Int_t nPhot) const
223 {
224  const Float_t epsilon = 0.0001f;
225  const Float_t p = FV0DigParam::Instance().lightYield * FV0DigParam::Instance().photoCathodeEfficiency;
226  if ((fabs(1.0f - p) < epsilon) || nPhot == 0) {
227  return nPhot;
228  }
229  const Int_t n = Int_t(nPhot < 100
230  ? gRandom->Binomial(nPhot, p)
231  : gRandom->Gaus(p * nPhot + 0.5, TMath::Sqrt(p * (1 - p) * nPhot)));
232  return n;
233 }

235 Float_t Digitizer::SimulateTimeCfd(Int_t channel) const
236 {
237  Float_t timeCfd = -1024.0f;
238  Int_t const binShift = TMath::Nint(FV0DigParam::Instance().timeShiftCfd / mBinSize);
239  Float_t sigPrev = -mPmtChargeVsTime[channel][0];
240  for (Int_t iTimeBin = 1; iTimeBin < mNBins; ++iTimeBin) {
241  Float_t const sigCurrent = (iTimeBin >= binShift
242  ? 5.0f * mPmtChargeVsTime[channel][iTimeBin - binShift] - mPmtChargeVsTime[channel][iTimeBin]
243  : -mPmtChargeVsTime[channel][iTimeBin]);
244  if (sigPrev < 0.0f && sigCurrent >= 0.0f) {
245  timeCfd = Float_t(iTimeBin) * mBinSize;
246  break;
247  }
248  sigPrev = sigCurrent;
249  }
250  return timeCfd;
251 }

258 Double_t Digitizer::PmtResponse(Double_t x)
259 {
260  // this function describes the PMT time response to a single photoelectron
261  if (x > 2 * FV0DigParam::Instance().pmtTransitTime)
262  return 0.0;
263  if (x < -FV0DigParam::Instance().pmtTransitTime)
264  return 0.0;
265  x += FV0DigParam::Instance().pmtTransitTime;
266  Double_t const x2 = x * x;
267  return x2 * std::exp(-x2 * FV0DigParam::Instance().oneOverPmtTransitTime2);
268 }

270 Double_t Digitizer::SinglePhESpectrum(Double_t* x, Double_t*)
271 {
272  // x -- number of photo-electrons emitted from the first dynode
273  // this function describes the PMT amplitude response to a single photoelectron
274  if (x[0] < 0.0)
275  return 0.0;
276  return (TMath::Poisson(x[0], FV0DigParam::Instance().pmtNbOfSecElec) +
277  FV0DigParam::Instance().pmtTransparency * TMath::Poisson(x[0], 1.0));
278 }

280 // The Distance is positive for top half-sectors (when the hit position is above the cell center (has higher y))
281 // TODO: performance check needed
282 float Digitizer::getDistFromCellCenter(UInt_t cellId, double hitx, double hity)
283 {
284  Geometry* geo = Geometry::instance();
285 
286  // Parametrize the line (ax+by+c=0) that crosses the detector center and the cell's middle point
287  Point3D<float>* pCell = &geo->getCellCenter(cellId);
288  float x0, y0, z0;
289  geo->getGlobalPosition(x0, y0, z0);
290  double a = -(y0 - pCell->Y()) / (x0 - pCell->X());
291  double b = 1;
292  double c = -(y0 - a * x0);
293  // Return the distance from hit to this line
294  return (a * hitx + b * hity + c) / TMath::Sqrt(a * a + b * b);
295 }
