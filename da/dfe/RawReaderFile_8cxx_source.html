42 template <class RawHeader>
43 void RawReaderFile<RawHeader>::init()
44 {
45  mDataFile.open(mInputFileName, std::ifstream::binary);
46  if (!mDataFile.good())
47  throw std::runtime_error("Unable to open or access file " + mInputFileName);
48  // get length of file in bytes
49  mDataFile.seekg(0, mDataFile.end);
50  mFileSize = mDataFile.tellg();
51  mDataFile.seekg(0, mDataFile.beg);
52  // the file is supposed to contain N x 8kB packets. So the number of packets
53  // can be determined by the file-size. Ideally, this is not required but the
54  // information is derived directly from the header size and payload size.
55  // *** to be adapted to header info ***
56  mNumData = mFileSize / (8 * 1024);
57 }

59 template <class RawHeader>
60 void RawReaderFile<RawHeader>::next()
61 {
62  mRawPayload.reset();
63  bool isDataTerminated = false;
64  do {
65  nextPage(false);
66  // check if we find a valid RCU trailer
67  // the payload must be at the end of the buffer
68  // if not present and error will be thrown
69  try {
70  RCUTrailer::constructFromPayloadWords(mRawBuffer.getDataWords());
71  isDataTerminated = true;
72  } catch (...) {
73  }
74  } while (!isDataTerminated);
75 }

77 template <class RawHeader>
78 void RawReaderFile<RawHeader>::nextPage(bool doResetPayload)
79 {
80  if (mCurrentPosition >= mNumData)
81  throw RawDecodingError(RawDecodingError::ErrorType_t::PAGE_NOTFOUND);
82  if (doResetPayload)
83  mRawPayload.reset();
84  auto start = mDataFile.tellg();
85  readHeader();
86  readPayload();
87  mRawPayload.appendPayloadWords(mRawBuffer.getDataWords());
88  mRawPayload.increasePageCount();
89  mDataFile.seekg(int(start) + mRawHeader.offsetToNext);
90  mCurrentPosition++;
91 }

93 template <class RawHeader>
94 void RawReaderFile<RawHeader>::readPage(int page)
95 {
96  mRawHeaderInitialized = false;
97  mPayloadInitialized = false;
98  if (page >= mNumData)
99  throw RawDecodingError(RawDecodingError::ErrorType_t::PAGE_NOTFOUND);
100  mDataFile.seekg(page * 8192);
101  auto start = mDataFile.tellg();
102  readHeader();
103  readPayload();
104  mDataFile.seekg(int(start) + mRawHeader.offsetToNext);
105  mCurrentPosition = page;
106 }
