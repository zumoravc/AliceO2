82 void DigitizerSpec::run(framework::ProcessingContext& pc)
83 {
84  if (mFinished)
85  return;
86 
87  // read collision context from input
88  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
89  context->initSimChains(o2::detectors::DetID::CPV, mSimChains);
90  auto& timesview = context->getEventRecords();
91  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
92 
93  // if there is nothing to do ... return
94  if (timesview.size() == 0) {
95  return;
96  }
97 
98  TStopwatch timer;
99  timer.Start();
100 
101  LOG(INFO) << " CALLING CPV DIGITIZATION ";
102  std::vector<TriggerRecord> triggers;
103  static std::vector<o2::cpv::Hit> hits;
104 
105  mLabels.clear();
106  mDigits.clear();
107  int indexStart = mDigits.size();
108  auto& eventParts = context->getEventParts();
109  // loop over all composite collisions given from context
110  // (aka loop over all the interaction records)
111  for (int collID = 0; collID < timesview.size(); ++collID) {
112  mDigitizer.setEventTime(timesview[collID].timeNS);
113 
114  // for each collision, loop over the constituents event and source IDs
115  // (background signal merging is basically taking place here)
116  for (auto& part : eventParts[collID]) {
117 
118  // get the hits for this event and this source
119  retrieveHits("CPVHit", part.sourceID, part.entryID);
120  mDigitizer.setCurrEvID(part.entryID);
121  }
122 
123  LOG(DEBUG) << "Found " << mHitsBg->size() << " BG hits and " << mHitsS->size() << "signal hits";
124 
125  // call actual digitization procedure
126  mDigitizer.process(mHitsBg, mHitsS, mDigits, mLabels);
127 
128  // Add trigger record
129  triggers.emplace_back(timesview[collID], indexStart, mDigits.size() - indexStart);
130  indexStart = mDigits.size();
131 
132  LOG(DEBUG) << "Have " << mDigits.size() << " digits ";
133  }
134 
135  LOG(DEBUG) << "Have " << mLabels.getNElements() << " CPV labels ";
136  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
137  pc.outputs().snapshot(Output{"CPV", "DIGITS", 0, Lifetime::Timeframe}, mDigits);
138  pc.outputs().snapshot(Output{"CPV", "DIGITTRIGREC", 0, Lifetime::Timeframe}, triggers);
139  if (pc.outputs().isAllowed({"CPV", "DIGITSMCTR", 0})) {
140  pc.outputs().snapshot(Output{"CPV", "DIGITSMCTR", 0, Lifetime::Timeframe}, mLabels);
141  }
142  // CPV is always a triggering detector
143  const o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::TRIGGERING;
144  LOG(DEBUG) << "CPV: Sending ROMode= " << roMode << " to GRPUpdater";
145  pc.outputs().snapshot(Output{"CPV", "ROMode", 0, Lifetime::Timeframe}, roMode);
146 
147  timer.Stop();
148  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
149 
150  pc.services().get<o2::framework::ControlService>().endOfStream();
151  // we should be only called once; tell DPL that this process is ready to exit
152  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
153  mFinished = true;
154 }

156 DataProcessorSpec getCPVDigitizerSpec(int channel, bool mctruth)
157 {
158 
159  // create the full data processor spec using
160  // a name identifier
161  // input description
162  // algorithmic description (here a lambda getting called once to setup the actual processing function)
163  // options that can be used for this processor (here: input file names where to take the hits)
164  std::vector<OutputSpec> outputs;
165  outputs.emplace_back("CPV", "DIGITS", 0, Lifetime::Timeframe);
166  outputs.emplace_back("CPV", "DIGITTRIGREC", 0, Lifetime::Timeframe);
167  if (mctruth) {
168  outputs.emplace_back("CPV", "DIGITSMCTR", 0, Lifetime::Timeframe);
169  }
170  outputs.emplace_back("CPV", "ROMode", 0, Lifetime::Timeframe);
171 
172  return DataProcessorSpec{
173  "CPVDigitizer", Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
174  outputs,
175  AlgorithmSpec{o2::framework::adaptFromTask<DigitizerSpec>()},
176  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}};
177 }
