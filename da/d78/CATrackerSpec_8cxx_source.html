133  // Parse the config string
134  const char* opt = options.c_str();
135  if (opt && *opt) {
136  printf("Received options %s\n", opt);
137  const char* optPtr = opt;
138  while (optPtr && *optPtr) {
139  while (*optPtr == ' ') {
140  optPtr++;
141  }
142  const char* nextPtr = strstr(optPtr, " ");
143  const int optLen = nextPtr ? nextPtr - optPtr : strlen(optPtr);
144  if (strncmp(optPtr, "cont", optLen) == 0) {
145  continuous = true;
146  printf("Continuous tracking mode enabled\n");
147  } else if (strncmp(optPtr, "dump", optLen) == 0) {
148  dump = 1;
149  printf("Dumping of input events enabled\n");
150  } else if (strncmp(optPtr, "dumponly", optLen) == 0) {
151  dump = 2;
152  printf("Dumping of input events enabled, processing disabled\n");
153  } else if (strncmp(optPtr, "display", optLen) == 0) {
154 #ifdef GPUCA_BUILD_EVENT_DISPLAY
155  processAttributes->displayBackend.reset(new GPUDisplayBackendGlfw);
156  display = processAttributes->displayBackend.get();
157  printf("Event display enabled\n");
158 #else
159  printf("Standalone Event Display not enabled at build time!\n");
160 #endif
161  } else if (strncmp(optPtr, "qa", optLen) == 0) {
162  qa = true;
163  printf("Enabling TPC Standalone QA\n");
164  } else if (optLen > 3 && strncmp(optPtr, "bz=", 3) == 0) {
165  sscanf(optPtr + 3, "%f", &solenoidBz);
166  printf("Using solenoid field %f\n", solenoidBz);
167  } else if (optLen > 5 && strncmp(optPtr, "refX=", 5) == 0) {
168  sscanf(optPtr + 5, "%f", &refX);
169  printf("Propagating to reference X %f\n", refX);
170  } else if (optLen > 5 && strncmp(optPtr, "debug=", 6) == 0) {
171  sscanf(optPtr + 6, "%d", &debugLevel);
172  printf("Debug level set to %d\n", debugLevel);
173  } else if (optLen > 8 && strncmp(optPtr, "threads=", 8) == 0) {
174  sscanf(optPtr + 8, "%d", &nThreads);
175  printf("Using %d threads\n", nThreads);
176  } else if (optLen > 8 && strncmp(optPtr, "gpuType=", 8) == 0) {
177  int len = std::min(optLen - 8, 1023);
178  memcpy(gpuType, optPtr + 8, len);
179  gpuType[len] = 0;
180  useGPU = true;
181  printf("Using GPU Type %s\n", gpuType);
182  } else if (optLen > 8 && strncmp(optPtr, "matBudFile=", 8) == 0) {
183  int len = std::min(optLen - 11, 1023);
184  memcpy(matBudFileName, optPtr + 11, len);
185  matBudFileName[len] = 0;
186  } else if (optLen > 8 && strncmp(optPtr, "dEdxFile=", 9) == 0) {
187  int len = std::min(optLen - 9, 1023);
188  memcpy(dEdxSplinesFile, optPtr + 9, len);
189  dEdxSplinesFile[len] = 0;
190  } else if (optLen > 15 && strncmp(optPtr, "transformation=", 15) == 0) {
191  int len = std::min(optLen - 15, 1023);
192  memcpy(tpcTransformationFileName, optPtr + 15, len);
193  tpcTransformationFileName[len] = 0;
194  readTransformationFromFile = true;
195  printf("Read TPC transformation from the file \"%s\"\n", tpcTransformationFileName);
196  } else {
197  printf("Unknown option: %s\n", optPtr);
198  throw std::invalid_argument("Unknown config string option");
199  }
200  optPtr = nextPtr;
201  }
202  }

279

306  auto processingFct = [processAttributes, specconfig](ProcessingContext& pc) {
307  if (processAttributes->readyToQuit) {
308  return;
309  }
310  auto& parser = processAttributes->parser;
311  auto& tracker = processAttributes->tracker;
312  uint64_t activeSectors = 0;
313  auto& verbosity = processAttributes->verbosity;
314  // FIXME cleanup almost duplicated code
315  auto& validMcInputs = processAttributes->validMcInputs;
316  using CachedMCLabelContainer = decltype(std::declval<InputRecord>().get<MCLabelContainer*>(DataRef{nullptr, nullptr, nullptr}));
317  std::vector<CachedMCLabelContainer> mcInputs;
318  std::vector<gsl::span<const char>> inputs;
319  struct InputRef {
320  DataRef data;
321  DataRef labels;
322  };
323  std::map<int, InputRef> inputrefs;
324  o2::gpu::GPUTrackingInOutZS tpcZS;
325  std::vector<const void*> tpcZSmetaPointers[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
326  std::vector<unsigned int> tpcZSmetaSizes[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
327  const void** tpcZSmetaPointers2[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
328  const unsigned int* tpcZSmetaSizes2[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
329  std::array<gsl::span<const o2::tpc::Digit>, NSectors> inputDigits;
330  std::array<std::unique_ptr<const MCLabelContainer>, NSectors> inputDigitsMC;
331  std::array<unsigned int, NEndpoints * NSectors> tpcZSonTheFlySizes;
332  gsl::span<const ZeroSuppressedContainer8kb> inputZS;
333 
334  // unsigned int totalZSPages = 0;
335  if (specconfig.processMC) {
336  std::vector<InputSpec> filter = {
337  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "DIGITSMCTR"}, Lifetime::Timeframe},
338  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLNATIVEMCLBL"}, Lifetime::Timeframe},
339  };
340  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
341  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(ref);
342  if (sectorHeader == nullptr) {
343  // FIXME: think about error policy
344  LOG(ERROR) << "sector header missing on header stack";
345  return;
346  }
347  const int sector = sectorHeader->sector();
348  if (sector < 0) {
349  continue;
350  }
351  std::bitset<NSectors> sectorMask(sectorHeader->sectorBits);
352  if ((validMcInputs & sectorMask).any()) {
353  // have already data for this sector, this should not happen in the current
354  // sequential implementation, for parallel path merged at the tracker stage
355  // multiple buffers need to be handled
356  throw std::runtime_error("can only have one MC data set per sector");
357  }
358  inputrefs[sector].labels = ref;
359  if (specconfig.caClusterer) {
360  inputDigitsMC[sector] = std::move(pc.inputs().get<const MCLabelContainer*>(ref));
361  } else {
362  }
363  validMcInputs |= sectorMask;
364  activeSectors |= sectorHeader->activeSectors;
365  if (verbosity > 1) {
366  LOG(INFO) << "received " << *(ref.spec) << " MC label containers"
367  << " for sectors " << sectorMask //
368  << std::endl //
369  << " mc input status: " << validMcInputs //
370  << std::endl //
371  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
372  }
373  }
374  }
375 
376  auto& validInputs = processAttributes->validInputs;
377  int operation = 0;
378  std::vector<InputSpec> filter = {
379  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "DIGITS"}, Lifetime::Timeframe},
380  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERNATIVE"}, Lifetime::Timeframe},
381  };
382 
383  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
384  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(ref);
385  if (sectorHeader == nullptr) {
386  throw std::runtime_error("sector header missing on header stack");
387  }
388  const int sector = sectorHeader->sector();
389  if (sector < 0) {
390  continue;
391  }
392  std::bitset<NSectors> sectorMask(sectorHeader->sectorBits);
393  if ((validInputs & sectorMask).any()) {
394  // have already data for this sector, this should not happen in the current
395  // sequential implementation, for parallel path merged at the tracker stage
396  // multiple buffers need to be handled
397  throw std::runtime_error("can only have one cluster data set per sector");
398  }
399  activeSectors |= sectorHeader->activeSectors;
400  validInputs |= sectorMask;
401  inputrefs[sector].data = ref;
402  if (specconfig.caClusterer && !specconfig.zsOnTheFly) {
403  inputDigits[sector] = pc.inputs().get<gsl::span<o2::tpc::Digit>>(ref);
404  LOG(INFO) << "GOT SPAN FOR SECTOR " << sector << " -> " << inputDigits[sector].size();
405  }
406  }
407  if (specconfig.zsOnTheFly) {
408  tpcZSonTheFlySizes = {0};
409  // tpcZSonTheFlySizes: #zs pages per endpoint:
410  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "ZSSIZES"}, Lifetime::Timeframe}};
411  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
412  tpcZSonTheFlySizes = pc.inputs().get<std::array<unsigned int, NEndpoints * NSectors>>(ref);
413  }
414  // zs pages
415  std::vector<InputSpec> filter2 = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "TPCZS"}, Lifetime::Timeframe}};
416  for (auto const& ref : InputRecordWalker(pc.inputs(), filter2)) {
417  inputZS = pc.inputs().get<gsl::span<ZeroSuppressedContainer8kb>>(ref);
418  }
419  //set all sectors as active and as valid inputs
420  for (int s = 0; s < NSectors; s++) {
421  activeSectors |= 1 << s;
422  validInputs.set(s);
423  }
424  for (unsigned int i = 0; i < GPUTrackingInOutZS::NSLICES; i++) {
425  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
426  tpcZSmetaPointers[i][j].clear();
427  tpcZSmetaSizes[i][j].clear();
428  }
429  }
430 
431  unsigned int offset = 0;
432  for (unsigned int i = 0; i < NSectors; i++) {
433  unsigned int pageSector = 0;
434  for (unsigned int j = 0; j < NEndpoints; j++) {
435  pageSector += tpcZSonTheFlySizes[i * NEndpoints + j];
436  offset += tpcZSonTheFlySizes[i * NEndpoints + j];
437  }
438  LOG(INFO) << "GOT ZS pages FOR SECTOR " << i << " -> pages: " << pageSector;
439  }
440  }
441  if (specconfig.zsDecoder) {
442  for (unsigned int i = 0; i < GPUTrackingInOutZS::NSLICES; i++) {
443  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
444  tpcZSmetaPointers[i][j].clear();
445  tpcZSmetaSizes[i][j].clear();
446  }
447  }
448  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}};
449  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
450  const o2::header::DataHeader* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
451  const gsl::span<const char> raw = pc.inputs().get<gsl::span<char>>(ref);
452  o2::framework::RawParser parser(raw.data(), raw.size());
453 
454  const unsigned char* ptr = nullptr;
455  int count = 0;
456  rdh_utils::FEEIDType lastFEE = -1;
457  int rawcru = 0;
458  int rawendpoint = 0;
459  size_t totalSize = 0;
460  for (auto it = parser.begin(); it != parser.end(); it++) {
461  const unsigned char* current = it.raw();
462  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)current;
463  if (current == nullptr || it.size() == 0 || (current - ptr) % TPCZSHDR::TPC_ZS_PAGE_SIZE || o2::raw::RDHUtils::getFEEID(*rdh) != lastFEE) {
464  if (count) {
465  tpcZSmetaPointers[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(ptr);
466  tpcZSmetaSizes[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(count);
467  }
468  count = 0;
469  //lastFEE = o2::raw::RDHUtils::getFEEID(*rdh);
470  lastFEE = int(rdh->feeId);
471  rawcru = int(rdh->cruID);
472  rawendpoint = int(rdh->endPointID);
473  if (it.size() == 0 && tpcZSmetaPointers[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].size()) {
474  ptr = nullptr;
475  continue;
476  }
477  ptr = current;
478  }
479  count++;
480  }
481  if (count) {
482  tpcZSmetaPointers[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(ptr);
483  tpcZSmetaSizes[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(count);
484  }
485  }
