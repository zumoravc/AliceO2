133 int GPUTPCTrackerComponent::ReadConfigurationString(const char* arguments)
134 {
135  // Set configuration parameters for the CA tracker component from the string
136 
137  int iResult = 0;
138  if (!arguments) {
139  return iResult;
140  }
141 
142  TString allArgs = arguments;
143  TString argument;
144  int bMissingParam = 0;
145 
146  TObjArray* pTokens = allArgs.Tokenize(" ");
147 
148  int nArgs = pTokens ? pTokens->GetEntries() : 0;
149 
150  for (int i = 0; i < nArgs; i++) {
151  argument = ((TObjString*)pTokens->At(i))->GetString();
152  if (argument.IsNull()) {
153  continue;
154  }
155 
156  if (argument.CompareTo("-solenoidBz") == 0) {
157  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
158  break;
159  }
160  HLTWarning("argument -solenoidBz is deprecated, magnetic field set up globally (%f)", GetBz());
161  continue;
162  }
163 
164  if (argument.CompareTo("-minNClustersOnTrack") == 0) {
165  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
166  break;
167  }
168  fMinNTrackClusters = ((TObjString*)pTokens->At(i))->GetString().Atoi();
169  HLTInfo("minNClustersOnTrack set to: %d", fMinNTrackClusters);
170  continue;
171  }
172 
173  if (argument.CompareTo("-minTrackPt") == 0) {
174  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
175  break;
176  }
177  fMinTrackPt = ((TObjString*)pTokens->At(i))->GetString().Atof();
178  HLTInfo("minTrackPt set to: %f", fMinTrackPt);
179  continue;
180  }
181 
182  if (argument.CompareTo("-clusterZCut") == 0) {
183  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
184  break;
185  }
186  fClusterZCut = TMath::Abs(((TObjString*)pTokens->At(i))->GetString().Atof());
187  HLTInfo("ClusterZCut set to: %f", fClusterZCut);
188  continue;
189  }
190 
191  if (argument.CompareTo("-neighboursSearchArea") == 0) {
192  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
193  break;
194  }
195  mNeighboursSearchArea = TMath::Abs(((TObjString*)pTokens->At(i))->GetString().Atof());
196  HLTInfo("NeighboursSearchArea set to: %f", mNeighboursSearchArea);
197  continue;
198  }
199 
200  if (argument.CompareTo("-errorCorrectionY") == 0) {
201  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
202  break;
203  }
204  fClusterErrorCorrectionY = ((TObjString*)pTokens->At(i))->GetString().Atof();
205  HLTInfo("Cluster Y error correction factor set to: %f", fClusterErrorCorrectionY);
206  continue;
207  }
208 
209  if (argument.CompareTo("-errorCorrectionZ") == 0) {
210  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
211  break;
212  }
213  fClusterErrorCorrectionZ = ((TObjString*)pTokens->At(i))->GetString().Atof();
214  HLTInfo("Cluster Z error correction factor set to: %f", fClusterErrorCorrectionZ);
215  continue;
216  }
217 
218  if (argument.CompareTo("-allowGPU") == 0) {
219  fAllowGPU = 1;
220  HLTInfo("Will try to run tracker on GPU");
221  continue;
222  }
223 
224  if (argument.CompareTo("-GlobalTracking") == 0) {
225  fGlobalTracking = 1;
226  HLTInfo("Global Tracking Activated");
227  continue;
228  }
229 
230  if (argument.CompareTo("-GPUHelperThreads") == 0) {
231  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
232  break;
233  }
234  fGPUHelperThreads = ((TObjString*)pTokens->At(i))->GetString().Atoi();
235  HLTInfo("Number of GPU Helper Threads set to: %d", fGPUHelperThreads);
236  continue;
237  }
238 
239  if (argument.CompareTo("-CPUTrackers") == 0) {
240  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
241  break;
242  }
243  fCPUTrackers = ((TObjString*)pTokens->At(i))->GetString().Atoi();
244  HLTInfo("Number of CPU Trackers set to: %d", fCPUTrackers);
245  continue;
246  }
247 
248  if (argument.CompareTo("-SearchWindowDZDR") == 0) {
249  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
250  break;
251  }
252  fSearchWindowDZDR = ((TObjString*)pTokens->At(i))->GetString().Atof();
253  HLTInfo("Search Window DZDR set to: %f", fSearchWindowDZDR);
254  continue;
255  }
256 
257  if (argument.CompareTo("-GPUDeviceNum") == 0) {
258  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
259  break;
260  }
261  fGPUDeviceNum = ((TObjString*)pTokens->At(i))->GetString().Atoi();
262  HLTInfo("Using GPU Device Number %d", fGPUDeviceNum);
263  continue;
264  }
265 
266  if (argument.CompareTo("-GPUType") == 0) {
267  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
268  break;
269  }
270  fGPUType = ((TObjString*)pTokens->At(i))->GetString();
271  continue;
272  }
273 
274  if (argument.CompareTo("-GPUStuckProtection") == 0) {
275  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
276  break;
277  }
278  fGPUStuckProtection = ((TObjString*)pTokens->At(i))->GetString().Atoi();
279  continue;
280  }
281 
282  if (argument.CompareTo("-AsyncGPUStuckProtection") == 0) {
283  if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
284  break;
285  }
286  fAsync = ((TObjString*)pTokens->At(i))->GetString().Atoi();
287  continue;
288  }
289 
290  HLTError("Unknown option \"%s\"", argument.Data());
291  iResult = -EINVAL;
292  }
293  delete pTokens;
294 
295  if (bMissingParam) {
296  HLTError("Specifier missed for parameter \"%s\"", argument.Data());
297  iResult = -EINVAL;
298  }
299 
300  return iResult;
301 }

304 {

307  const char* defaultNotify = "";

335 int GPUTPCTrackerComponent::Configure(const char* cdbEntry, const char* chainId, const char* commandLine)
336 {
337  // Configure the component
338  // There are few levels of configuration,
339  // parameters which are set on one step can be overwritten on the next step
340 
341  //* read hard-coded values
342  SetDefaultConfiguration();
343 
344  //* read the default CDB entry
345  int iResult1 = ReadCDBEntry(nullptr, chainId);
346 
347  //* read magnetic field
348  fSolenoidBz = GetBz();
349 
350  //* read the actual CDB entry if required
351  int iResult2 = (cdbEntry) ? ReadCDBEntry(cdbEntry, chainId) : 0;
352 
353  //* read extra parameters from input (if they are)
354  int iResult3 = 0;
355 
356  if (commandLine && commandLine[0] != '\0') {
357  HLTInfo("received configuration string from HLT framework: \"%s\"", commandLine);
358  iResult3 = ReadConfigurationString(commandLine);
359  }
360 
361  if (fRec) {
362  ConfigureSlices();
363  }
364 
365  return iResult1 ? iResult1 : (iResult2 ? iResult2 : iResult3);
366 }

368 int GPUTPCTrackerComponent::ConfigureSlices()
369 {
370  // Initialize the tracker slices
371  GPUSettingsRec rec;
372  GPUSettingsEvent ev;
373  GPUSettingsDeviceProcessing devProc;
374 
375  ev.solenoidBz = fSolenoidBz;
376  ev.continuousMaxTimeBin = 0; // triggered events
377  if (mNeighboursSearchArea > 0) {
378  rec.NeighboursSearchArea = mNeighboursSearchArea;
379  }
380  if (fClusterErrorCorrectionY > 1.e-4) {
381  rec.ClusterError2CorrectionY = fClusterErrorCorrectionY * fClusterErrorCorrectionY;
382  }
383  if (fClusterErrorCorrectionZ > 1.e-4) {
384  rec.ClusterError2CorrectionZ = fClusterErrorCorrectionZ * fClusterErrorCorrectionZ;
385  }
386  rec.MinNTrackClusters = fMinNTrackClusters;
387  rec.SetMinTrackPt(fMinTrackPt);
388  rec.SearchWindowDZDR = fSearchWindowDZDR;
389  devProc.nDeviceHelperThreads = fGPUHelperThreads;
390  rec.GlobalTracking = fGlobalTracking;
391  devProc.stuckProtection = fGPUStuckProtection;
392  rec.NonConsecutiveIDs = true;
393  rec.mergerReadFromTrackerDirectly = false;
394 
395  GPURecoStepConfiguration steps;
396  steps.steps.set(GPUDataTypes::RecoStep::TPCSliceTracking);
397  steps.inputs.set(GPUDataTypes::InOutType::TPCClusters);
398  steps.outputs.set(GPUDataTypes::InOutType::TPCSectorTracks);
399 
400  fRec->SetSettings(&ev, &rec, &devProc, &steps);
401  fChain->LoadClusterErrors();
402  return fRec->Init();
403 }

420 int GPUTPCTrackerComponent::DoInit(int argc, const char** argv)
421 {
422  if (fRec) {
423  return EINPROGRESS;
424  }
425 
426  // Configure the CA tracker component
427  TString arguments = "";
428  for (int i = 0; i < argc; i++) {
429  if (!arguments.IsNull()) {
430  arguments += " ";
431  }
432  arguments += argv[i];
433  }
434 
435  int retVal = Configure(nullptr, nullptr, arguments.Data());
436  if (retVal == 0) {
437  if (fAsync) {
438  if (fAsyncProcessor.Initialize(1)) {
439  return (-ENODEV);
440  }
441  void* initRetVal;
442  if (fAsyncProcessor.InitializeAsyncMemberTask(this, &GPUTPCTrackerComponent::TrackerInit, nullptr, &initRetVal) != 0) {
443  return (-ENODEV);
444  }
445  if (initRetVal) {
446  return (-ENODEV);
447  }
448  } else {
449  if (TrackerInit(nullptr) != nullptr) {
450  return (-ENODEV);
451  }
452  }
453  }
454 
455  return (retVal);
456 }

486 int GPUTPCTrackerComponent::DoEvent(const AliHLTComponentEventData& evtData, const AliHLTComponentBlockData* blocks, AliHLTComponentTriggerData& /*trigData*/, AliHLTUInt8_t* outputPtr, AliHLTUInt32_t& size, vector<AliHLTComponentBlockData>& outputBlocks)
487 {
488  //* process event
489  if (!fRec) {
490  HLTError("CATracker not initialized properly");
491  return -ENOENT;
492  }
493 
494  AliHLTTPCTrackerWrapperData tmpPar;
495  tmpPar.fEvtData = &evtData;
496  tmpPar.fBlocks = blocks;
497  tmpPar.mOutputPtr = outputPtr;
498  tmpPar.fSize = &size;
499  tmpPar.mOutputBlocks = &outputBlocks;
500 
501  static int trackerTimeout = 0;
502  if (trackerTimeout) {
503  size = 0;
504  return (0);
505  }
506 
507  int retVal;
508  if (fAsync) {
509  void* asyncRetVal = nullptr;
510  if (fAsyncProcessor.InitializeAsyncMemberTask(this, &GPUTPCTrackerComponent::TrackerDoEvent, &tmpPar, &asyncRetVal, fAsync) != 0) {
511  HLTError("Tracking timed out, disabling this tracker instance");
512  trackerTimeout = 1;
513  size = 0;
514  return (-ENODEV);
515  } else {
516  retVal = (int)(size_t)asyncRetVal;
517  }
518  } else {
519  retVal = (int)(size_t)TrackerDoEvent(&tmpPar);
520  }
521  return (retVal);
522 }

524 void* GPUTPCTrackerComponent::TrackerDoEvent(void* par)
525 {
526  AliHLTTPCTrackerWrapperData* tmpPar = reinterpret_cast<AliHLTTPCTrackerWrapperData*>(par);
527 
528  const AliHLTComponentEventData& evtData = *(tmpPar->fEvtData);
529  const AliHLTComponentBlockData* blocks = tmpPar->fBlocks;
530  AliHLTUInt8_t* outputPtr = tmpPar->mOutputPtr;
531  AliHLTUInt32_t& size = *(tmpPar->fSize);
532  vector<AliHLTComponentBlockData>& outputBlocks = *(tmpPar->mOutputBlocks);
533 
534  AliHLTUInt32_t maxBufferSize = size;
535  size = 0; // output size
536 
537  if (GetFirstInputBlock(kAliHLTDataTypeSOR) || GetFirstInputBlock(kAliHLTDataTypeEOR)) {
538  return 0;
539  }
540 
541  fBenchmark.StartNewEvent();
542  fBenchmark.Start(0);
543 
544  // Logging( kHLTLogWarning, "HLT::TPCCATracker::DoEvent", "DoEvent", "CA::DoEvent()" );
545  if (evtData.fBlockCnt <= 0) {
546  HLTWarning("no blocks in event");
547  return 0;
548  }
549 
550  // Prepare everything for all slices
551  const AliHLTTPCClusterXYZData* clustersXYZ[NSLICES][fgkNPatches] = {nullptr};
552  const AliHLTTPCRawClusterData* clustersRaw[NSLICES][fgkNPatches] = {nullptr};
553 
554  for (unsigned long ndx = 0; ndx < evtData.fBlockCnt; ndx++) {
555  const AliHLTComponentBlockData& pBlock = blocks[ndx];
556  int slice = AliHLTTPCDefinitions::GetMinSliceNr(pBlock);
557  int patch = AliHLTTPCDefinitions::GetMinPatchNr(pBlock);
558  if (pBlock.fDataType == AliHLTTPCDefinitions::RawClustersDataType()) {
559  clustersRaw[slice][patch] = (const AliHLTTPCRawClusterData*)pBlock.fPtr;
560  } else if (pBlock.fDataType == AliHLTTPCDefinitions::ClustersXYZDataType()) {
561  clustersXYZ[slice][patch] = (const AliHLTTPCClusterXYZData*)pBlock.fPtr;
562  }
563  }
564 
565  GPUTPCClusterData* clusterData[NSLICES] = {nullptr};
566  int nClusters[NSLICES] = {0};
567 
568  int nClustersTotal = 0;
569  for (int slice = 0; slice < NSLICES; slice++) {
570  int nClustersSliceTotal = 0;
571  for (int patch = 0; patch < 6; patch++) {
572  if (clustersXYZ[slice][patch]) {
573  nClustersSliceTotal += clustersXYZ[slice][patch]->fCount;
574  }
575  }
576  if (nClustersSliceTotal > 500000) {
577  HLTWarning("Too many clusters in tracker input: Slice %d, Number of Clusters %d, slice not included in tracking", slice, nClustersSliceTotal);
578  nClusters[slice] = nClustersSliceTotal;
579  } else if (nClustersSliceTotal == 0) {
580  nClusters[slice] = nClustersSliceTotal;
581  } else {
582  clusterData[slice] = new GPUTPCClusterData[nClustersSliceTotal];
583  nClusters[slice] = nClustersSliceTotal;
584  GPUTPCClusterData* pCluster = clusterData[slice];
585  for (int patch = 0; patch < 6; patch++) {
586  if (clustersXYZ[slice][patch] != nullptr && clustersRaw[slice][patch] != nullptr) {
587  const AliHLTTPCClusterXYZData& clXYZ = *clustersXYZ[slice][patch];
588  const AliHLTTPCRawClusterData& clRaw = *clustersRaw[slice][patch];
589 
590  if (clXYZ.fCount != clRaw.fCount) {
591  HLTError("Number of entries in raw and xyz clusters are not mached %d vs %d", clXYZ.fCount, clRaw.fCount);
592  continue;
593  }
594 
595  const int firstRow = AliHLTTPCGeometry::GetFirstRow(patch);
596  for (int ic = 0; ic < clXYZ.fCount; ic++) {
597  const AliHLTTPCClusterXYZ& c = clXYZ.fClusters[ic];
598  const AliHLTTPCRawCluster& cRaw = clRaw.fClusters[ic];
599  if (c.GetZ() > fClusterZCut || c.GetZ() < -fClusterZCut) {
600  continue;
601  }
602  if (c.GetX() < 1.f) {
603  continue; // cluster xyz position was not calculated for whatever reason
604  }
605  pCluster->id = AliHLTTPCGeometry::CreateClusterID(slice, patch, ic);
606  pCluster->x = c.GetX();
607  pCluster->y = c.GetY();
608  pCluster->z = c.GetZ();
609  pCluster->row = firstRow + cRaw.GetPadRow();
610  pCluster->flags = cRaw.GetFlags();
611  if (cRaw.GetSigmaPad2() < kAlmost0 || cRaw.GetSigmaTime2() < kAlmost0) {
612  pCluster->flags |= GPUTPCGMMergedTrackHit::flagSingle;
613  }
614  pCluster->amp = cRaw.GetCharge();
615 #ifdef GPUCA_FULL_CLUSTERDATA
616  pCluster->pad = cRaw.GetPad();
617  pCluster->time = cRaw.GetTime();
618  pCluster->ampMax = cRaw.GetQMax();
619  pCluster->sigmaPad2 = cRaw.GetSigmaPad2();
