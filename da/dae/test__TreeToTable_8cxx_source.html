24 BOOST_AUTO_TEST_CASE(TreeToTableConversion)
25 {
26  using namespace o2::framework;
28  Int_t ndp = 1000;
29 
30  TFile f1("tree2table.root", "RECREATE");
31  TTree t1("t1", "a simple Tree with simple variables");
32  Bool_t ok;
33  Float_t px, py, pz;
34  Double_t random;
35  Int_t ev;
36  t1.Branch("ok", &ok, "ok/O");
37  t1.Branch("px", &px, "px/F");
38  t1.Branch("py", &py, "py/F");
39  t1.Branch("pz", &pz, "pz/F");
40  t1.Branch("random", &random, "random/D");
41  t1.Branch("ev", &ev, "ev/I");
42 
43  //fill the tree
44  int ntruein = 0;
45  for (int i = 0; i < ndp; i++) {
46  ok = (i % 2) == 0;
47  if (ok) {
48  ntruein++;
49  }
50  gRandom->Rannor(px, py);
51  pz = px * px + py * py;
52  random = gRandom->Rndm();
53  ev = i + 1;
54  t1.Fill();
55  }
56  t1.Write();
57  auto br = (TBranch*)t1.GetBranch("ok");
58 
59  // Create an arrow table from this.
60  TreeToTable tr2ta(&t1);
61  auto stat = tr2ta.addAllColumns();
62  if (!stat) {
63  LOG(ERROR) << "Table was not created!";
64  return;
65  }
66  auto table = tr2ta.process();
67 
68  // count number of rows with ok==true
69  int ntrueout = 0;
70  auto chunks = table->column(0);
71  BOOST_REQUIRE_NE(chunks.get(), nullptr);
72  auto oks =
73  std::dynamic_pointer_cast<arrow::BooleanArray>(chunks->chunk(0));
74  BOOST_REQUIRE_NE(oks.get(), nullptr);
75  for (int ii = 0; ii < table->num_rows(); ii++) {
76  ntrueout += oks->Value(ii) ? 1 : 0;
77  }
78 
79  // test result
80  BOOST_REQUIRE_EQUAL(table->Validate().ok(), true);
81  BOOST_REQUIRE_EQUAL(table->num_rows(), ndp);
82  BOOST_REQUIRE_EQUAL(table->num_columns(), 6);
83  BOOST_REQUIRE_EQUAL(table->column(0)->type()->id(), arrow::boolean()->id());
84  BOOST_REQUIRE_EQUAL(ntruein, ntrueout);
85  BOOST_REQUIRE_EQUAL(table->column(1)->type()->id(), arrow::float32()->id());
86  BOOST_REQUIRE_EQUAL(table->column(2)->type()->id(), arrow::float32()->id());
87  BOOST_REQUIRE_EQUAL(table->column(3)->type()->id(), arrow::float32()->id());
88  BOOST_REQUIRE_EQUAL(table->column(4)->type()->id(), arrow::float64()->id());
89  BOOST_REQUIRE_EQUAL(table->column(5)->type()->id(), arrow::int32()->id());
90 
91  f1.Close();
92 
93  TFile* f2 = new TFile("table2tree.root", "RECREATE");
94  TableToTree ta2tr(table, f2, "mytree");
95  stat = ta2tr.addAllBranches();
96  auto t2 = ta2tr.process();
97  br = (TBranch*)t2->GetBranch("ok");
98 
99  BOOST_REQUIRE_EQUAL(t2->GetEntries(), ndp);
100  BOOST_REQUIRE_EQUAL(br->GetEntries(), ndp);
101 
102  f2->Close();
103 }
