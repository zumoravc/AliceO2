24 void PropertyTreeHelpers::populateDefaults(std::vector<ConfigParamSpec> const& schema,
25  boost::property_tree::ptree& pt,
26  boost::property_tree::ptree& provenance)
27 {
28  for (auto& spec : schema) {
29  std::string key = spec.name.substr(0, spec.name.find(","));
30  try {
31  if (spec.defaultValue.type() == VariantType::Empty) {
32  continue;
33  }
34  switch (spec.type) {
35  case VariantType::Int:
36  pt.put(key, spec.defaultValue.get<int>());
37  break;
38  case VariantType::Int64:
39  pt.put(key, spec.defaultValue.get<int64_t>());
40  break;
41  case VariantType::Float:
42  pt.put(key, spec.defaultValue.get<float>());
43  break;
44  case VariantType::Double:
45  pt.put(key, spec.defaultValue.get<double>());
46  break;
47  case VariantType::String:
48  pt.put(key, spec.defaultValue.get<std::string>());
49  break;
50  case VariantType::Bool:
51  pt.put(key, spec.defaultValue.get<bool>());
52  break;
53  case VariantType::Unknown:
54  case VariantType::Empty:
55  default:
56  throw std::runtime_error("Unknown variant type");
57  }
58  provenance.put(key, "default");
59  } catch (std::runtime_error& re) {
60  throw re;
61  } catch (std::exception& e) {
62  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
63  } catch (...) {
64  throw std::invalid_argument(std::string("missing option: ") + key);
65  }
66  }
67 }

69 void PropertyTreeHelpers::populate(std::vector<ConfigParamSpec> const& schema,
70  boost::property_tree::ptree& pt,
71  boost::program_options::variables_map const& vmap,
72  boost::property_tree::ptree& provenance)
73 {
74  for (auto& spec : schema) {
75  // strip short version to get the correct key
76  std::string key = spec.name.substr(0, spec.name.find(","));
77  if (vmap.count(key) == 0) {
78  continue;
79  }
80  try {
81  switch (spec.type) {
82  case VariantType::Int:
83  pt.put(key, vmap[key].as<int>());
84  break;
85  case VariantType::Int64:
86  pt.put(key, vmap[key].as<int64_t>());
87  break;
88  case VariantType::Float:
89  pt.put(key, vmap[key].as<float>());
90  break;
91  case VariantType::Double:
92  pt.put(key, vmap[key].as<double>());
93  break;
94  case VariantType::String:
95  if (auto v = boost::any_cast<std::string>(&vmap[key].value())) {
96  pt.put(key, *v);
97  }
98  break;
99  case VariantType::Bool:
100  pt.put(key, vmap[key].as<bool>());
101  break;
102  case VariantType::Unknown:
103  case VariantType::Empty:
104  default:
105  throw std::runtime_error("Unknown variant type");
106  }
107  provenance.put(key, "fairmq");
108  } catch (std::runtime_error& re) {
109  throw re;
110  } catch (std::exception& e) {
111  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
112  } catch (...) {
113  throw std::invalid_argument(std::string("missing option: ") + key);
114  }
115  }

169 void traverseRecursive(const boost::property_tree::ptree& parent,
170  const boost::property_tree::ptree::path_type& childPath,
171  const boost::property_tree::ptree& child,
172  PropertyTreeHelpers::WalkerFunction& method)
173 {
174  using boost::property_tree::ptree;
175 
176  method(parent, childPath, child);
177  for (ptree::const_iterator it = child.begin(); it != child.end(); ++it) {
178  ptree::path_type curPath = childPath / ptree::path_type(it->first);
179  traverseRecursive(parent, curPath, it->second, method);
180  }
181 }

189 void PropertyTreeHelpers::merge(boost::property_tree::ptree& dest,
190  boost::property_tree::ptree const& source,
191  boost::property_tree::ptree::path_type const& mergePoint)

194  WalkerFunction merge = [&dest, &mergePoint](ptree const& parent, ptree::path_type childPath, ptree const& child) {

197  traverse(source, merge);
