112 // ----- Passing the event ---------------------------------------------
113 Bool_t Pythia8Generator::ReadEvent(FairPrimaryGenerator* cpg)
114 {
115  const double mm2cm = 0.1;
116  const double clight = 2.997924580e10; //cm/c
117  Int_t npart = 0;
118  while (npart == 0) {
119  mPythia->next();
120  for (int i = 0; i < mPythia->event.size(); i++) {
121  if (mPythia->event[i].isFinal()) {
122  // only send HNL decay products to G4
123  if (mHNL != 0) {
124  Int_t im = mPythia->event[i].mother1();
125  if (mPythia->event[im].id() == mHNL) {
126  // for the moment, hardcode 110m is maximum decay length
127  Double_t z = mPythia->event[i].zProd();
128  Double_t x = abs(mPythia->event[i].xProd());
129  Double_t y = abs(mPythia->event[i].yProd());
130  // cout<<"debug HNL decay pos "<<x<<" "<< y<<" "<< z <<endl;
131  if (z < 11000. && z > 7000. && x < 250. && y < 250.) {
132  npart++;
133  }
134  }
135  } else {
136  npart++;
137  }
138  };
139  };
140  // happens if a charm particle being produced which does decay without producing a HNL. Try another event.
141  // if (npart == 0){ mPythia->event.list();}
142  };
143  // cout<<"debug p8 event 0 " << mPythia->event[0].id()<< " "<< mPythia->event[1].id()<< " "
144  // << mPythia->event[2].id()<< " "<< npart <<endl;
145  for (Int_t ii = 0; ii < mPythia->event.size(); ii++) {
146  Bool_t wanttracking = true;
147  if (!mPythia->event[ii].isFinal())
148  wanttracking = false;
149  Double_t z = mPythia->event[ii].zProd();
150  Double_t x = mPythia->event[ii].xProd();
151  Double_t y = mPythia->event[ii].yProd();
152  Double_t pz = mPythia->event[ii].pz();
153  Double_t px = mPythia->event[ii].px();
154  Double_t py = mPythia->event[ii].py();
155  Double_t t = mPythia->event[ii].tProd();
156  x *= mm2cm;
157  y *= mm2cm;
158  z *= mm2cm;
159  t *= mm2cm / clight;
160  Double_t e = -9e9;
161  // handle system entry separately
162  if (!wanttracking)
163  e = mPythia->event[ii].e();
164  //
165  printf("AddTrack from Pythia8 %5d \n", (Int_t)mPythia->event[ii].id());
166  cpg->AddTrack((Int_t)mPythia->event[ii].id(), px, py, pz, x, y, z,
167  (Int_t)mPythia->event[ii].mother1(), wanttracking, e, t);
168  } // particle lloop
169 
170  return kTRUE;
171 }
