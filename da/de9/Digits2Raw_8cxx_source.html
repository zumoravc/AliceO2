10 /*
11  Digits to RAW data coding. RAW data format - class DataFormat/Detectors/FIT/FT0/RawEventData
12  18 PMs (GBT links) 12 MCPs each and 1 TCM, each stream transmit separately
13 Event header - 80bits
14  uint startDescriptor : 4;
15  uint nGBTWords : 4;
16  uint reservedField : 32;
17  uint orbit : 32;
18  uint bc : 12;
19 
20  Event data 40bits
21  short int time : 12;
22  short int charge : 12;
23  unsigned short int numberADC : 1;
24  bool isDoubleEvent : 1;
25  bool is1TimeLostEvent : 1;
26  bool is2TimeLostEvent : 1;
27  bool isADCinGate : 1;
28  bool isTimeInfoLate : 1;
29  bool isAmpHigh : 1;
30  bool isEventInTVDC : 1;
31  bool isTimeInfoLost : 1;
32  uint channelID : 4;
33 GBT packet:
34 RDH + Event header + event data, 2 channels per 1 GBT word;
35 if no data for this PM - only headers.
36 
37 Trigger mode : detector sends data to FLP at each trigger;
38 Continueous mode : for only bunches with data at least in 1 channel.
39 */

71 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
72 {
73  LOG(INFO) << "**********Digits2Raw::convertDigits" << std::endl;
74 
75  o2::ft0::LookUpTable lut{o2::ft0::Digits2Raw::linear()};
76  LOG(DEBUG) << " ##### LookUp set ";
77 
78  std::string outd = outDir;
79  if (outd.back() != '/') {
80  outd += '/';
81  }
82  using namespace o2::raw;
83  for (int ilink = 0; ilink < NPMs; ++ilink) {
84  mLinkID = uint32_t(ilink);
85  mFeeID = uint64_t(ilink);
86  mCruID = uint16_t(0);
87  mEndPointID = uint32_t(0);
88  std::string outFileLink = mOutputPerLink ? o2::utils::concat_string(outDir, "ft0_link", std::to_string(ilink), ".raw") : o2::utils::concat_string(outDir, "ft0.raw");
89  mWriter.registerLink(mFeeID, mCruID, mLinkID, mEndPointID, outFileLink);
90  }
91 
92  TFile* fdig = TFile::Open(fileDigitsName.data());
93  assert(fdig != nullptr);
94  LOG(INFO) << " Open digits file " << fileDigitsName.data();
95  TTree* digTree = (TTree*)fdig->Get("o2sim");
96 
97  std::vector<o2::ft0::Digit> digitsBC, *ft0BCDataPtr = &digitsBC;
98  std::vector<o2::ft0::ChannelData> digitsCh, *ft0ChDataPtr = &digitsCh;
99 
100  digTree->SetBranchAddress("FT0DIGITSBC", &ft0BCDataPtr);
101  digTree->SetBranchAddress("FT0DIGITSCH", &ft0ChDataPtr);
102 
103  uint32_t old_orbit = ~0;
104  o2::InteractionRecord intRecord;
105  o2::InteractionRecord lastIR = mSampler.getFirstIR();
106  std::vector<o2::InteractionRecord> HBIRVec;
107 
108  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
109  digTree->GetEntry(ient);
110 
111  int nbc = digitsBC.size();
112  LOG(DEBUG) << "Entry " << ient << " : " << nbc << " BCs stored";
113  for (int ibc = 0; ibc < nbc; ibc++) {
114  auto& bcd = digitsBC[ibc];
115  intRecord = bcd.getIntRecord();
116  auto channels = bcd.getBunchChannelData(digitsCh);
117  if (!channels.empty())
118  convertDigits(bcd, channels, lut, intRecord);
119  }
120  }
121 }

123 void Digits2Raw::convertDigits(o2::ft0::Digit bcdigits,
124  gsl::span<const ChannelData> pmchannels,
125  const o2::ft0::LookUpTable& lut,
126  o2::InteractionRecord const& intRecord)
127 {
128 
129  // check empty event
130  int oldlink = -1;
131  int nchannels = 0;
132  int nch = pmchannels.size();
133  for (int ich = 0; ich < nch; ich++) {
134  int nlink = lut.getLink(pmchannels[ich].ChId);
135  if (nlink != oldlink) {
136  if (oldlink >= 0) {
137  uint nGBTWords = uint((nchannels + 1) / 2);
138  LOG(DEBUG) << " oldlink " << oldlink << " nGBTWords " << nGBTWords;
139  if ((nchannels % 2) == 1)
140  mRawEventData.mEventData[nchannels] = {};
141  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
142  auto data = mRawEventData.to_vector(false);
143  mLinkID = uint32_t(oldlink);
144  mFeeID = uint64_t(oldlink);
145  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, intRecord, data);
146  }
147  oldlink = nlink;
148  mRawEventData.mEventHeader = makeGBTHeader(nlink, intRecord);
149  nchannels = 0;
150  // LOG(INFO) << " switch to new link " << nlink;
151  }
152  auto& newData = mRawEventData.mEventData[nchannels];
153  bool isAside = (pmchannels[ich].ChId < 96);
154  newData.charge = pmchannels[ich].QTCAmpl;
155 
156  newData.time = pmchannels[ich].CFDTime;
157  newData.is1TimeLostEvent = 0;
158  newData.is2TimeLostEvent = 0;
159  newData.isADCinGate = 1;
160  newData.isAmpHigh = 0;
161  newData.isDoubleEvent = 0;
162  newData.isEventInTVDC = 1;
163  newData.isTimeInfoLate = 0;
164  newData.isTimeInfoLost = 0;
165  int chain = std::rand() % 2;
166  newData.numberADC = chain ? 1 : 0;
167  newData.channelID = lut.getMCP(pmchannels[ich].ChId);
168  // LOG(INFO) << "packed GBT " << nlink << " channelID " << (int)newData.channelID << " charge " << newData.charge << " time " << newData.time << " chain " << int(newData.numberADC) << " size " << sizeof(newData);
169  nchannels++;
170  }
171  // fill mEventData[nchannels] with 0s to flag that this is a dummy data
172  uint nGBTWords = uint((nchannels + 1) / 2);
173  if ((nchannels % 2) == 1)
174  mRawEventData.mEventData[nchannels] = {};
175  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
176  LOG(DEBUG) << " last " << oldlink;
177  //TCM
178  mRawEventData.mEventHeader = makeGBTHeader(LinkTCM, intRecord); //TCM
179  mRawEventData.mEventHeader.nGBTWords = 1;
180  auto& tcmdata = mRawEventData.mTCMdata;
181  mTriggers = bcdigits.getTriggers();
182 
183  float ampA = mTriggers.amplA;
184  float ampC = mTriggers.amplC;
185  if (ampA > 131071)
186  ampA = 131071; //2^17
187  if (ampC > 131071)
188  ampC = 131071; //2^17
189  tcmdata.vertex = mTriggers.getVertex();
190  tcmdata.orA = mTriggers.getOrA();
191  tcmdata.orC = mTriggers.getOrC();
192  tcmdata.sCen = mTriggers.getSCen();
193  tcmdata.cen = mTriggers.getCen();
194  tcmdata.nChanA = mTriggers.nChanA;
195  tcmdata.nChanC = mTriggers.nChanC;
196  tcmdata.amplA = ampA;
197  tcmdata.amplC = ampC;
198  tcmdata.timeA = mTriggers.timeA;
199  tcmdata.timeC = mTriggers.timeC;
200  if (mVerbosity > 0) {
201  LOG(INFO) << " triggers read "
202  << " time A " << mTriggers.timeA << " time C " << mTriggers.timeC
203  << " amp A " << ampA << " amp C " << ampC
204  << " N A " << int(mTriggers.nChanA) << " N C " << int(mTriggers.nChanC)
205  << " trig "
206  << " ver " << mTriggers.getVertex() << " A " << mTriggers.getOrA() << " C " << mTriggers.getOrC();
207 
208  LOG(INFO) << "TCMdata"
209  << " time A " << tcmdata.timeA << " time C " << tcmdata.timeC
210  << " amp A " << tcmdata.amplA << " amp C " << tcmdata.amplC
211  << " N A " << int(tcmdata.nChanA) << " N C " << int(tcmdata.nChanC)
212  << " trig "
213  << " ver " << tcmdata.vertex << " A " << tcmdata.orA << " C " << tcmdata.orC
214  << " size " << sizeof(tcmdata);
215  }
216  auto data = mRawEventData.to_vector(1);
217  mLinkID = uint32_t(LinkTCM);
218  mFeeID = uint64_t(LinkTCM);
219  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, intRecord, data);
220 }
