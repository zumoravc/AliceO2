96 Operations createOperations(Filter const& expression)
97 {
98  Operations OperationSpecs;
99  std::stack<NodeRecord> path;
100  auto isLeaf = [](Node const* const node) {
101  return ((node->left == nullptr) && (node->right == nullptr));
102  };
103 
104  auto processLeaf = [](Node const* const node) {
105  return std::visit(
106  overloaded{
107  [lh = LiteralNodeHelper{}](LiteralNode node) { return lh(node); },
108  [bh = BindingNodeHelper{}](BindingNode node) { return bh(node); },
109  [](auto&&) { return DatumSpec{}; }},
110  node->self);
111  };
112 
113  size_t index = 0;
114  // insert the top node into stack
115  path.emplace(expression.node.get(), index++);
116 
117  // while the stack is not empty
118  while (path.empty() == false) {
119  auto& top = path.top();
120 
121  // create operation spec, pop the node and add its children
122  auto operationSpec =
123  std::visit(
124  overloaded{
125  [](OpNode node) { return ColumnOperationSpec{node.op}; },
126  [](auto&&) { return ColumnOperationSpec{}; }},
127  top.node_ptr->self);
128 
129  operationSpec.result = DatumSpec{top.index, operationSpec.type};
130  path.pop();
131 
132  auto left = top.node_ptr->left.get();
133  bool leftLeaf = isLeaf(left);
134  size_t li = 0;
135  if (leftLeaf) {
136  operationSpec.left = processLeaf(left);
137  } else {
138  li = index;
139  operationSpec.left = DatumSpec{index++, atype::NA};
140  }
141 
142  decltype(left) right = nullptr;
143  if (top.node_ptr->right != nullptr)
144  right = top.node_ptr->right.get();
145  bool rightLeaf = true;
146  if (right != nullptr)
147  rightLeaf = isLeaf(right);
148  size_t ri = 0;
149  auto isUnary = false;
150  if (top.node_ptr->right == nullptr) {
151  operationSpec.right = DatumSpec{};
152  isUnary = true;
153  } else {
154  if (rightLeaf) {
155  operationSpec.right = processLeaf(right);
156  } else {
157  ri = index;
158  operationSpec.right = DatumSpec{index++, atype::NA};
159  }
160  }
161 
162  OperationSpecs.push_back(std::move(operationSpec));
163  if (!leftLeaf)
164  path.emplace(left, li);
165  if (!isUnary && !rightLeaf)
166  path.emplace(right, ri);
167  }
168  // at this stage the operations vector is created, but the field types are
169  // only set for the logical operations and leaf nodes
170  std::vector<atype::type> resultTypes;
171  resultTypes.resize(OperationSpecs.size());
172 
173  auto inferResultType = [&resultTypes](DatumSpec& left, DatumSpec& right) {
174  // if the left datum is monostate (error)
175  if (left.datum.index() == 0) {
176  throw std::runtime_error("Malformed operation spec: empty left datum");
177  }
178 
179  // check if the datums are references
180  if (left.datum.index() == 1)
181  left.type = resultTypes[std::get<size_t>(left.datum)];
182 
183  if (right.datum.index() == 1)
184  right.type = resultTypes[std::get<size_t>(right.datum)];
185 
186  auto t1 = left.type;
187  auto t2 = right.type;
188  // if the right datum is monostate (unary op)
189  if (right.datum.index() == 0) {
190  if (t1 == atype::DOUBLE)
191  return atype::DOUBLE;
192  return atype::FLOAT;
193  }
194 
195  if (t1 == t2)
196  return t1;
197 
198  if (t1 == atype::INT32) {
199  if (t2 == atype::FLOAT)
200  return atype::FLOAT;
201  if (t2 == atype::DOUBLE)
202  return atype::DOUBLE;
203  }
204  if (t1 == atype::FLOAT) {
205  if (t2 == atype::INT32)
206  return atype::FLOAT;
207  if (t2 == atype::DOUBLE)
208  return atype::DOUBLE;
209  }
210  if (t1 == atype::DOUBLE) {
211  return atype::DOUBLE;
212  }
213  throw std::runtime_error(fmt::format("Invalid combination of argument types {} and {}", t1, t2));
214  };
215 
216  for (auto it = OperationSpecs.rbegin(); it != OperationSpecs.rend(); ++it) {
217  auto type = inferResultType(it->left, it->right);
218  if (it->type == atype::NA) {
219  it->type = type;
220  }
221  it->result.type = it->type;
222  resultTypes[std::get<size_t>(it->result.datum)] = it->type;
223  }
224 
225  return OperationSpecs;
226 }

257 Selection createSelection(std::shared_ptr<arrow::Table> table, std::shared_ptr<gandiva::Filter> gfilter)
258 {
259  Selection selection;
260  auto s = gandiva::SelectionVector::MakeInt64(table->num_rows(),
261  arrow::default_memory_pool(),
262  &selection);
263  if (!s.ok())
264  throw std::runtime_error(fmt::format("Cannot allocate selection vector {}", s));
265  arrow::TableBatchReader reader(*table);
266  std::shared_ptr<arrow::RecordBatch> batch;
267  while (true) {
268  s = reader.ReadNext(&batch);
269  if (!s.ok()) {
270  throw std::runtime_error(fmt::format("Cannot read batches from table {}", s));
271  }
272  if (batch == nullptr) {
273  break;
274  }
275  s = gfilter->Evaluate(*batch, selection);
276  if (!s.ok())
277  throw std::runtime_error(fmt::format("Cannot apply filter {}", s));
278  }
279 
280  return selection;
281 }

289 gandiva::NodePtr createExpressionTree(Operations const& opSpecs,
290  gandiva::SchemaPtr const& Schema)
291 {
292  std::vector<gandiva::NodePtr> opNodes;
293  opNodes.resize(opSpecs.size());
294  std::fill(opNodes.begin(), opNodes.end(), nullptr);
295  std::unordered_map<std::string, gandiva::NodePtr> fieldNodes;
296 
297  auto datumNode = [Schema, &opNodes, &fieldNodes](DatumSpec const& spec) {
298  if (spec.datum.index() == 1) {
299  return opNodes[std::get<size_t>(spec.datum)];
300  }
301 
302  if (spec.datum.index() == 2) {
303  auto content = std::get<LiteralNode::var_t>(spec.datum);
304  if (content.index() == 0)
305  return gandiva::TreeExprBuilder::MakeLiteral(static_cast<int32_t>(std::get<int>(content)));
306  if (content.index() == 1)
307  return gandiva::TreeExprBuilder::MakeLiteral(std::get<bool>(content));
308  if (content.index() == 2)
309  return gandiva::TreeExprBuilder::MakeLiteral(std::get<float>(content));
310  if (content.index() == 3)
311  return gandiva::TreeExprBuilder::MakeLiteral(std::get<double>(content));
312  throw std::runtime_error("Malformed LiteralNode");
313  }
314 
315  if (spec.datum.index() == 3) {
316  auto name = std::get<std::string>(spec.datum);
317  auto lookup = fieldNodes.find(name);
318  if (lookup != fieldNodes.end())
319  return lookup->second;
320  auto node = gandiva::TreeExprBuilder::MakeField(Schema->GetFieldByName(name));
321  fieldNodes.insert({name, node});
322  return node;
323  }
324  throw std::runtime_error("Malformed DatumSpec");
325  };
326 
327  gandiva::NodePtr tree = nullptr;
328  for (auto it = opSpecs.rbegin(); it != opSpecs.rend(); ++it) {
329  auto leftNode = datumNode(it->left);
330  auto rightNode = datumNode(it->right);
331  switch (it->op) {
332  case BasicOp::LogicalOr:
333  tree = gandiva::TreeExprBuilder::MakeOr({leftNode, rightNode});
334  break;
335  case BasicOp::LogicalAnd:
336  tree = gandiva::TreeExprBuilder::MakeAnd({leftNode, rightNode});
337  break;
338  default:
339  if (it->op <= BasicOp::Exp) {
340  tree = gandiva::TreeExprBuilder::MakeFunction(binaryOperationsMap[it->op], {leftNode, rightNode}, concreteArrowType(it->type));
341  } else {
342  tree = gandiva::TreeExprBuilder::MakeFunction(binaryOperationsMap[it->op], {leftNode}, concreteArrowType(it->type));
343  }
344  break;
345  }
346  opNodes[std::get<size_t>(it->result.datum)] = tree;
347  }
348 
349  return tree;
350 }

352 bool isSchemaCompatible(gandiva::SchemaPtr const& Schema, Operations const& opSpecs)
353 {
354  std::set<std::string> opFieldNames;
355  for (auto& spec : opSpecs) {
356  if (spec.left.datum.index() == 3)
357  opFieldNames.insert(std::get<std::string>(spec.left.datum));
358  if (spec.right.datum.index() == 3)
359  opFieldNames.insert(std::get<std::string>(spec.right.datum));
360  }
361 
362  std::set<std::string> schemaFieldNames;
363  for (auto& field : Schema->fields()) {
364  schemaFieldNames.insert(field->name());
365  }
366 
367  return std::includes(schemaFieldNames.begin(), schemaFieldNames.end(),
368  opFieldNames.begin(), opFieldNames.end());
369 }

371 void updateExpressionInfos(expressions::Filter const& filter, std::vector<ExpressionInfo>& eInfos)
372 {
373  if (eInfos.empty()) {

375  }
376  Operations ops = createOperations(filter);
377  for (auto& info : eInfos) {
378  if (isSchemaCompatible(info.schema, ops)) {
379  auto tree = createExpressionTree(ops, info.schema);
381  if (info.tree != nullptr) {
382  info.tree = gandiva::TreeExprBuilder::MakeAnd({info.tree, tree});
383  } else {
384  info.tree = tree;
385  }
386  }
