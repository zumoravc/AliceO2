82 bool Detector::ProcessHits(FairVolume* v)
83 {
84  // If not charged track or already stopped or disappeared, just return.
85  if ((!fMC->TrackCharge()) || fMC->IsTrackDisappeared()) {
86  return false;
87  }
88 
89  // Inside sensitive volume ?
90  bool drRegion = false;
91  bool amRegion = false;
92  const TString cIdSensDr = "J";
93  const TString cIdSensAm = "K";
94  const TString cIdCurrent = fMC->CurrentVolName();
95  if (cIdCurrent[1] == cIdSensDr) {
96  drRegion = true;
97  }
98  if (cIdCurrent[1] == cIdSensAm) {
99  amRegion = true;
100  }
101  if (!drRegion && !amRegion) {
102  return false;
103  }
104 
105  // Determine the dectector number
106  int sector, det;
107  // The plane number and chamber number
108  char cIdChamber[3];
109  cIdChamber[0] = cIdCurrent[2];
110  cIdChamber[1] = cIdCurrent[3];
111  cIdChamber[2] = 0;
112  // The det-sec number (0 – 29)
113  const int idChamber = mGeom->getDetectorSec(atoi(cIdChamber));
114  // The sector number (0 - 17), according to the standard coordinate system
115  TString cIdPath = fMC->CurrentVolPath();
116  char cIdSector[3];
117  cIdSector[0] = cIdPath[21];
118  cIdSector[1] = cIdPath[22];
119  cIdSector[2] = 0;
120  sector = atoi(cIdSector);
121  // The detector number (0 – 539)
122  det = mGeom->getDetector(mGeom->getLayer(idChamber), mGeom->getStack(idChamber), sector);
123 
124  // 0: InFlight 1: Entering 2: Exiting
125  int trkStat = 0;
126 
127  o2::data::Stack* stack = (o2::data::Stack*)fMC->GetStack();
128  float xp, yp, zp;
129  float px, py, pz, etot;
130  float trackLength = fMC->TrackLength(); // Return the length of the current track from its origin (in cm)
131  float tof = fMC->TrackTime(); // Return the current time of flight of the track being transported (in s).
132 
133  // Special hits if track is entering
134  if (drRegion && fMC->IsTrackEntering()) {
135  // Create a track reference at the entrance of each
136  // chamber that contains the momentum components of the particle
137  fMC->TrackMomentum(px, py, pz, etot);
138  fMC->TrackPosition(xp, yp, zp);
139  stack->addTrackReference(o2::TrackReference(xp, yp, zp, px, py, pz,
140  trackLength,
141  tof,
142  stack->GetCurrentTrackNumber(),
143  GetDetId()));
144  // Update track status
145  trkStat = 1;
146  // Create the hits from TR photons if electron/positron is entering the drift volume
147  const bool ele = (std::fabs(fMC->TrackPid()) == 11); // electron PDG code.
148  if (mTRon && ele) {
149  createTRhit(det);
150  }
151  } else if (amRegion && fMC->IsTrackExiting()) {
152  // Create a track reference at the exit of each
153  // chamber that contains the momentum components of the particle
154  fMC->TrackMomentum(px, py, pz, etot);
155  fMC->TrackPosition(xp, yp, zp);
156  stack->addTrackReference(o2::TrackReference(xp, yp, zp, px, py, pz,
157  trackLength,
158  tof,
159  stack->GetCurrentTrackNumber(),
160  GetDetId()));
161  // Update track status
162  trkStat = 2;
163  }
164 
165  // Calculate the charge according to GEANT Edep
166  // Create a new dEdx hit
167  const float enDep = std::max(fMC->Edep(), 0.0) * 1e9; // Energy in eV
168  const int totalChargeDep = (int)(enDep / mWion); // Total charge
169 
170  // Store those hits with enDep bigger than the ionization potential of the gas mixture for in-flight tracks
171  // or store hits of tracks that are entering or exiting
172  if (totalChargeDep || trkStat) {
173  fMC->TrackPosition(xp, yp, zp);
174  tof = tof * 1e6; // The time of flight in micro-seconds
175  const int trackID = stack->GetCurrentTrackNumber();
176  double pos[3] = {xp, yp, zp};
177  double loc[3] = {-99, -99, -99};
178  gGeoManager->MasterToLocal(pos, loc); // Go to the local coordinate system (locR, locC, locT)
179  float locC = loc[0], locR = loc[1], locT = loc[2];
180  if (drRegion) {
181  locT = locT - 0.5 * (TRDGeometry::drThick() + TRDGeometry::amThick()); // Relative to middle of amplification region
182  }
183  addHit(xp, yp, zp, locC, locR, locT, tof, totalChargeDep, trackID, det, drRegion);
184  stack->addHit(GetDetId());
185  return true;
186  }
187  return false;
188 }

190 void Detector::createTRhit(int det)
191 {
192  //
193  // Creates an electron cluster from a TR photon.
194  // The photon is assumed to be created a the end of the radiator. The
195  // distance after which it deposits its energy takes into account the
196  // absorbtion of the entrance window and of the gas mixture in drift
197  // volume.
198  //
199 
200  // Maximum number of TR photons per track
201  constexpr int mMaxNumberOfTRPhotons = 50; // Make this a class member?
202 
203  float px, py, pz, etot;
204  fMC->TrackMomentum(px, py, pz, etot);
205  float pTot = std::sqrt(px * px + py * py + pz * pz);
206  std::vector<float> photonEnergyContainer; // energy in keV
207  mTR->createPhotons(11, pTot, photonEnergyContainer); // Create TR photons
208  if (photonEnergyContainer.size() > mMaxNumberOfTRPhotons) {
209  LOG(ERROR) << "Boundary error: nTR = " << photonEnergyContainer.size() << ", mMaxNumberOfTRPhotons = " << mMaxNumberOfTRPhotons;
210  }
211 
212  // Loop through the TR photons
213  for (const float& photonEnergy : photonEnergyContainer) {
214  const double energyMeV = photonEnergy * 1e-3;
215  const double energyeV = photonEnergy * 1e3;
216  double absLength = 0.0;
217  double sigma = 0.0;
218  // Take the absorbtion in the entrance window into account
219  double muMy = mTR->getMuMy(energyMeV);
220  sigma = muMy * mFoilDensity;
221  if (sigma > 0.0) {
222  absLength = gRandom->Exp(1.0 / sigma);
223  if (absLength < TRDGeometry::myThick()) {
224  continue;
225  }
226  } else {
227  continue;
228  }
229  // The absorbtion cross sections in the drift gas
230  // Gas-mixture (Xe/CO2)
231  double muNo = 0.0;
232  if (TRDCommonParam::Instance()->IsXenon()) {
233  muNo = mTR->getMuXe(energyMeV);
234  } else if (TRDCommonParam::Instance()->IsArgon()) {
235  muNo = mTR->getMuAr(energyMeV);
236  }
237  double muCO = mTR->getMuCO(energyMeV);
238  double fGasNobleFraction = 1;
239  double fGasDensity = 1;
240  sigma = (fGasNobleFraction * muNo + (1.0 - fGasNobleFraction) * muCO) * fGasDensity * mTR->getTemp();
241 
242  // The distance after which the energy of the TR photon
243  // is deposited.
244  if (sigma > 0.0) {
245  absLength = gRandom->Exp(1.0 / sigma);
246  if (absLength > (TRDGeometry::drThick() + TRDGeometry::amThick())) {
247  continue;
248  }
249  } else {
250  continue;
251  }
252 
253  // The position of the absorbtion
254  float xp, yp, zp;
255  fMC->TrackPosition(xp, yp, zp);
256  float invpTot = 1. / pTot;
257  float x = xp + px * invpTot * absLength;
258  float y = yp + py * invpTot * absLength;
259  float z = zp + pz * invpTot * absLength;
260 
261  // Add the hit to the array. TR photon hits are marked by negative energy (and not by charge)
262  float tof = fMC->TrackTime() * 1e6; // The time of flight in micro-seconds
263  o2::data::Stack* stack = (o2::data::Stack*)fMC->GetStack();
264  const int trackID = stack->GetCurrentTrackNumber();
265  const int totalChargeDep = -1 * (int)(energyeV / mWion); // Negative charge for tagging TR photon hits
266  // prepare local coordinates
267  double pos[3] = {x, y, z};
268  double loc[3] = {-99, -99, -99};
269  gGeoManager->MasterToLocal(pos, loc); // Go to the local coordinate system (locR, locC, locT)
270  float locC = loc[0], locR = loc[1], locT = loc[2];
271  locT = locT - 0.5 * (TRDGeometry::drThick() + TRDGeometry::amThick()); // Relative to middle of amplification region
272  addHit(x, y, z, locC, locR, locT, tof, totalChargeDep, trackID, det, true); // All TR hits are in drift region
273  stack->addHit(GetDetId());
274  }
275 }
