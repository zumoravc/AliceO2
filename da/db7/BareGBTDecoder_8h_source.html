92 template <typename CHARGESUM>
93 size_t BareGBTDecoder<CHARGESUM>::append(Payload bytes)
94 {
95  if (bytes.size() % 16 != 0) {
96  throw std::invalid_argument("can only bytes by group of 16 (i.e. 128 bits)");
97  }
98  size_t n{0};
99  for (int j = 0; j < bytes.size(); j += 16) {
100  ++mNofGbtWordsSeens;
101  int elinkIndex = 0;
102  for (auto b : bytes.subspan(j, 10)) {
103  const uint8_t b8 = std::to_integer<uint8_t>(b);
104  mElinks[elinkIndex++].append(b8 & 2, b8 & 1);
105  mElinks[elinkIndex++].append(b8 & 8, b8 & 4);
106  mElinks[elinkIndex++].append(b8 & 32, b8 & 16);
107  mElinks[elinkIndex++].append(b8 & 128, b8 & 64);
108  }
109  n += 10;
110  }
111  return n;
112 }
