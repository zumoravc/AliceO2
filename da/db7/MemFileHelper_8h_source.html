88  static std::unique_ptr<TMemFile> createTMemFile(const TObject& rootObject, const std::string& fName, const std::string& objName)
89  {
90  std::unique_ptr<TMemFile> uptr;
91 #if ROOT_VERSION_CODE < ROOT_VERSION(6, 18, 0)
92  uptr = std::make_unique<TMemFile>(fName.c_str(), "RECREATE");
93 #else
94  uptr = std::make_unique<TMemFile>(fName.c_str(), "RECREATE", "", ROOT::RCompressionSetting::EDefaults::kUseGeneralPurpose, 1024);
95 #endif
96  if (uptr->IsZombie()) {
97  uptr->Close();
98  uptr.reset();
99  throw std::runtime_error(std::string("Error opening memory file ") + fName.c_str());
100  } else {
101  rootObject.Write(objName.c_str());
102  uptr->Close();
103  }
104  return uptr;
105  }

107  //________________________________________________________________
109  static std::unique_ptr<TMemFile> createTMemFile(const void* obj, const std::type_info& tinfo, const std::string& fName, const std::string& optName = "")
110  {
111  std::unique_ptr<TMemFile> uptr;
112  std::string clsName = getClassName(tinfo);
113  if (clsName.empty()) {
114  return uptr;
115  }
116  std::string objName = optName.empty() ? clsName : optName;
117  std::string fileName = fName.empty() ? (objName + ".root") : fName;
118 #if ROOT_VERSION_CODE < ROOT_VERSION(6, 18, 0)
119  uptr = std::make_unique<TMemFile>(fileName.c_str(), "RECREATE");
120 #else
121  uptr = std::make_unique<TMemFile>(fileName.c_str(), "RECREATE", "", ROOT::RCompressionSetting::EDefaults::kUseGeneralPurpose, 1024);
122 #endif
123  if (uptr->IsZombie()) {
124  uptr->Close();
125  uptr.reset();
126  throw std::runtime_error(std::string("Error opening memory file ") + fileName.c_str());
127  } else {
128  uptr->WriteObjectAny(obj, clsName.c_str(), objName.c_str());
129  uptr->Close();
130  }
131  return uptr;
132  }
