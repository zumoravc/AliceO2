51 BOOST_AUTO_TEST_CASE(test_forwardparser_header_and_trailer)
52 {
53  using FrameT = o2::algorithm::Composite<Header, Trailer>;
54  // note: the length of the data is set in the header word
55  using TestFrame = o2::algorithm::StaticSequenceAllocator;
56  TestFrame tf(FrameT(16, "lotsofsillydata", 0xaa),
57  FrameT(5, "test", 0xcc),
58  FrameT(10, "dummydata", 0x33));
59 
60  using ParserT = o2::algorithm::ForwardParser<typename FrameT::HeaderType,
61  typename FrameT::TrailerType>;
62 
63  auto checkHeader = [](const typename FrameT::HeaderType& header) {
64  return header.identifier == 0xdeadbeef;
65  };
66  auto checkTrailer = [](const typename FrameT::TrailerType& trailer) {
67  return trailer.identifier == 0xaaffee00;
68  };
69  auto getFrameSize = [](const typename ParserT::HeaderType& header) {
70  // frame size includes total offset from header and trailer
71  return header.payloadSize + ParserT::totalOffset;
72  };
73 
74  std::vector<typename ParserT::FrameInfo> frames;
75  auto insert = [&frames](typename ParserT::FrameInfo& info) {
76  frames.emplace_back(info);
77  return true;
78  };
79 
80  ParserT parser;
81  auto result = parser.parse(tf.buffer.get(), tf.size(),
82  checkHeader,
83  checkTrailer,
84  getFrameSize,
85  insert);
86 
87  BOOST_REQUIRE(result == 3);
88  BOOST_REQUIRE(frames.size() == 3);
89 
90  BOOST_CHECK(memcmp(frames[0].payload, "lotsofsillydata", frames[0].length) == 0);
91  BOOST_CHECK(memcmp(frames[1].payload, "test", frames[1].length) == 0);
92  BOOST_CHECK(memcmp(frames[2].payload, "dummydata", frames[2].length) == 0);
93 }
