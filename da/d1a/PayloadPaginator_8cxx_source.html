19 PayloadPaginator::PayloadPaginator(o2::raw::RawFileWriter& fw,
20  const std::string outputFileName,
21  Solar2FeeLinkMapper solar2feelink,
22  bool userLogic,
23  bool chargeSumMode) : mRawFileWriter(fw),
24  mOutputFileName{outputFileName},
25  mSolar2FeeLink{solar2feelink},
26  mExtraFeeIdMask{chargeSumMode ? static_cast<uint16_t>(0x100) : static_cast<uint16_t>(0)}
27 {
28  if (userLogic) {
29  mSolar2FeeLink = [solar2feelink](uint16_t solarId) -> std::optional<FeeLinkId> {
30  static auto s2f = solar2feelink;
31  auto f = s2f(solarId);
32  if (!f.has_value()) {
33  return std::nullopt;
34  }
35  return FeeLinkId(f->feeId(), 15);
36  };
37  }
38 }

40 void PayloadPaginator::operator()(gsl::span<const std::byte> buffer)
41 {
42  std::set<DataBlockRef> dataBlockRefs;
43  forEachDataBlockRef(
44  buffer, [&](const DataBlockRef& ref) {
45  dataBlockRefs.insert(ref);
46  });
47 
48  for (auto r : dataBlockRefs) {
49  auto& b = r.block;
50  auto& h = b.header;
51  auto feelink = mSolar2FeeLink(r.block.header.solarId).value();
52  int endpoint = feelink.feeId() % 2;
53  int cru = (feelink.feeId() - endpoint) / 2;
54  auto feeId = feelink.feeId() | mExtraFeeIdMask;
55  if (mFeeLinkIds.find(feelink) == mFeeLinkIds.end()) {
56  mRawFileWriter.registerLink(feeId, cru, feelink.linkId(), endpoint, mOutputFileName);
57  mFeeLinkIds.insert(feelink);
58  }
59  mRawFileWriter.addData(feeId, cru, feelink.linkId(), endpoint,
60  {h.bc, h.orbit},
61  gsl::span<char>(const_cast<char*>(reinterpret_cast<const char*>(&b.payload[0])),
62  b.payload.size()));
63  }
64 }

66 std::vector<std::byte> paginate(gsl::span<const std::byte> buffer, bool userLogic,
67  bool chargeSumMode, const std::string& tmpfilename)
68 {
69  fair::Logger::SetConsoleSeverity("nolog");
70  o2::raw::RawFileWriter fw;
71 
72  fw.setVerbosity(1);
73  fw.setDontFillEmptyHBF(true);
74 
75  Solar2FeeLinkMapper solar2feelink = createSolar2FeeLinkMapper<ElectronicMapperGenerated>();
76 
77  {
78  PayloadPaginator p(fw, tmpfilename, solar2feelink, userLogic, chargeSumMode);
79  p(buffer);
80  fw.close();
81  }
82 
83  std::ifstream in(tmpfilename, std::ifstream::binary);
84  // get length of file:
85  in.seekg(0, in.end);
86  int length = in.tellg();
87  in.seekg(0, in.beg);
88  std::vector<std::byte> pages(length);
89 
90  // read data as a block:
91  in.read(reinterpret_cast<char*>(&pages[0]), length);
92 
93  return pages;
94 }
