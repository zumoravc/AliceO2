39  static void getPadAndTotalCharge(HitType const& hit, int& chamber, int& pc, int& px, int& py, float& totalcharge)
40  {
41  float localX;
42  float localY;
43  chamber = hit.GetDetectorID();
44  double tmp[3] = {hit.GetX(), hit.GetY(), hit.GetZ()};

51  static float getFractionalContributionForPad(HitType const& hit, int somepad)

60  Param::Instance()->Mars2Lors(chamber, tmp, localX, localY);

79  static float IntPartMathiX(float x, int pad)
80  {
81  // Integration of Mathieson.
82  // This is the answer to electrostatic problem of charge distrubution in MWPC described elsewhere. (NIM A370(1988)602-603)
83  // Arguments: x,y- position of the center of Mathieson distribution
84  // Returns: a charge fraction [0-1] imposed into the pad
85  auto shift1 = -LorsX(pad) + 0.5 * Param::SizePadX();
86  auto shift2 = -LorsX(pad) - 0.5 * Param::SizePadX();
87 
88  auto ux1 = Param::SqrtK3x() * TMath::TanH(Param::K2x() * (x + shift1) / Param::PitchAnodeCathode());
89  auto ux2 = Param::SqrtK3x() * TMath::TanH(Param::K2x() * (x + shift2) / o2::hmpid::Param::PitchAnodeCathode());
90 
91  return o2::hmpid::Param::K4x() * (TMath::ATan(ux2) - TMath::ATan(ux1));
92  }
93  //++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
94 
95  static Double_t IntPartMathiY(Double_t y, int pad)
96  {
97  // Integration of Mathieson.
98  // This is the answer to electrostatic problem of charge distrubution in MWPC described elsewhere. (NIM A370(1988)602-603)
99  // Arguments: x,y- position of the center of Mathieson distribution
100  // Returns: a charge fraction [0-1] imposed into the pad
101  Double_t shift1 = -LorsY(pad) + 0.5 * o2::hmpid::Param::SizePadY();
102  Double_t shift2 = -LorsY(pad) - 0.5 * o2::hmpid::Param::SizePadY();
103 
104  Double_t uy1 = Param::SqrtK3y() * TMath::TanH(Param::K2y() * (y + shift1) / Param::PitchAnodeCathode());
105  Double_t uy2 = Param::SqrtK3y() * TMath::TanH(Param::K2y() * (y + shift2) / Param::PitchAnodeCathode());
106 
107  return Param::K4y() * (TMath::ATan(uy2) - TMath::ATan(uy1));
108  }
109 
110  static float InMathieson(float localX, float localY, int pad)
111  {
112  return 4. * Digit::IntPartMathiX(localX, pad) * Digit::IntPartMathiY(localY, pad);
113  }
