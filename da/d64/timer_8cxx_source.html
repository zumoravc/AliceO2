25 inline double HighResTimer::GetTime()
26 {
27 #ifdef _WIN32
28  __int64 istart;
29  QueryPerformanceCounter((LARGE_INTEGER*)&istart);
30  return ((double)istart);
31 #elif defined(__MACH__) || defined(__APPLE__)
32  clock_serv_t cclock;
33  mach_timespec_t mts;
34  host_get_clock_service(mach_host_self(), CALENDAR_CLOCK, &cclock);
35  clock_get_time(cclock, &mts);
36  mach_port_deallocate(mach_task_self(), cclock);
37  return ((double)mts.tv_sec * 1.0e9 + (double)mts.tv_nsec);
38 #else
39  timespec tv;
40  clock_gettime(CLOCK_REALTIME, &tv);

101 double HighResTimer::GetCurrentElapsedTime(bool reset)
102 {
103  if (running == 0) {

105  Reset();
106  return (retVal);

108  double CurrentTime = GetTime();
109  double retVal = (CurrentTime - StartTime + ElapsedTime) / Frequency;
110  if (reset) {
