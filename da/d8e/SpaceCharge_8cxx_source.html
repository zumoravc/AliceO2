170 float SpaceCharge::calculateLookupTables()
171 {
172  // Potential, E field and electron distortion and correction lookup tables
173  TStopwatch timer;
174  mLookUpTableCalculator.ForceInitSpaceCharge3DPoissonIntegralDz(mNR, mNZ, mNPhi, 300, 1e-8);
175  float tRealCalc = static_cast<float>(timer.RealTime());
176 
177  // Lookup tables for local ion drift along E field
178  if (mSCDistortionType == SCDistortionType::SCDistortionsRealistic) {
179  if (!mMemoryAllocated) {
180  allocateMemory();
181  }
182  TMatrixD* matrixDriftZ = nullptr;
183  TMatrixD* matrixDriftRPhi = nullptr;
184  TMatrixD* matrixDriftR = nullptr;
185  for (int iside = 0; iside < 2; ++iside) {
186  const int sign = 1 - iside * 2;
187  for (int iphi = 0; iphi < mNPhi; ++iphi) {
188  const float phi = static_cast<float>(mCoordPhi[iphi]);
189  if (iside == 0) {
190  matrixDriftZ = mMatrixIonDriftZA[iphi].get();
191  matrixDriftRPhi = mMatrixIonDriftRPhiA[iphi].get();
192  matrixDriftR = mMatrixIonDriftRA[iphi].get();
193  } else {
194  matrixDriftZ = mMatrixIonDriftZC[iphi].get();
195  matrixDriftRPhi = mMatrixIonDriftRPhiC[iphi].get();
196  matrixDriftR = mMatrixIonDriftRC[iphi].get();
197  }
198  int roc = iside == 0 ? o2::utils::Angle2Sector(phi) : o2::utils::Angle2Sector(phi) + 18;
199  for (int ir = 0; ir < mNR; ++ir) {
200  const float radius = static_cast<float>(mCoordR[ir]);
202  for (int iz = 0; iz < mNZ; ++iz) {
203  const float z = static_cast<float>(mCoordZ[iz]);
204  float x0[3] = {radius, phi, sign * (z + static_cast<float>(mVoxelSizeZ))}; // iphi, ir, iz+1
205  float x1[3] = {radius, phi, sign * z}; // iphi, ir, iz
206  if (iside == 1) {
207  x0[2] *= -1;
208  x1[2] *= -1;
209  }
210  double eVector0[3] = {0., 0., 0.};
211  double eVector1[3] = {0., 0., 0.};
212  mLookUpTableCalculator.GetElectricFieldCyl(x0, roc, eVector0); // returns correct sign for Ez
213  mLookUpTableCalculator.GetElectricFieldCyl(x1, roc, eVector1); // returns correct sign for Ez
214 
215  // drift of ions along E field
216  (*matrixDriftR)(ir, iz) = -1 * sign * mVoxelSizeZ * 0.5 * (eVector0[0] + eVector1[0]) / (sign * sEzField + eVector0[2]);
217  (*matrixDriftRPhi)(ir, iz) = -1 * sign * mVoxelSizeZ * 0.5 * (eVector0[1] + eVector1[1]) / (sign * sEzField + eVector0[2]);
218  (*matrixDriftZ)(ir, iz) = -1 * sign * mVoxelSizeZ + DvDEoverv0 * mVoxelSizeZ * 0.5 * (eVector0[2] + eVector1[2]);
219  }
220  }
221  }
222  if (iside == 0) {
223  mLookUpIonDriftA->CopyFromMatricesToInterpolator();
224  } else {
225  mLookUpIonDriftC->CopyFromMatricesToInterpolator();
226  }
227  }
228 
229  // TODO: Propagate current SC density along E field by one time bin for next update
230  // propagateSpaceCharge();
231  }
232 
233  mInitLookUpTables = true;
234  return tRealCalc;
235 }

307 void SpaceCharge::fillPrimaryIons(double r, double phi, double z, int nIons)
308 {
309  Side side = z > 0 ? Side::A : Side::C;
310  double dr = 0.;
311  double drphi = 0.;
312  double dz = 0.;
313  getIonDrift(side, r, phi, z, dr, drphi, dz);
314  double rdist = r + dr;
315  double phidist = phi + drphi / r;
316  double zdist = z + dz;
317  const int zBin = TMath::BinarySearch(mNZ, mCoordZ.data(), std::abs(zdist));
318  const int phiBin = TMath::BinarySearch(mNPhi, mCoordPhi.data(), phidist);
319  const int rBin = TMath::BinarySearch(mNR, mCoordR.data(), rdist);
321  if (z > 0 && zdist > 0) {
322  mSpaceChargeDensityA[phiBin * mNR * mNZ + rBin * mNZ + zBin] += ions2Charge(rBin, nIons);
323  } else if (z < 0 && zdist < 0) {
324  mSpaceChargeDensityC[phiBin * mNR * mNZ + rBin * mNZ + zBin] += ions2Charge(rBin, nIons);
325  }
326 }

377 void SpaceCharge::propagateSpaceCharge()
378 {
379  //
380  // Continuity equation, assuming that the charge is conserved:
381  // delta(rho_0) / delta(t) = - div(flux) = - div(rho_0 * u) = - ( rho_0 * div(u) + u * grad(rho_0) )
382  // u: velocity vector of space-charge density
383  // rho_0: space-charge density at time t0
384  //
385  //
386  // Calculate the change of the space-charge density rho_0 at time t0 after time delta(t) = mVoxelSizeZ / v_driftIon:
387  // delta(rho_0) = - ( rho_0 * div(u) + u * grad(rho_0) ) * delta(t)
388  // = - ( rho_0 * div(d) + d * grad(rho_0) )
389  // d: drift vector (dr, dphi, mVoxelSizeZ + dz)
390  //
391  // div(d) = 1/r del(r*d_r)/del(r) + 1/r del(d_phi)/del(phi) + del(d_z)/del(z)
392  // = d_r/r + del(d_r)/del(r) + 1/r del(d_phi)/del(phi) + del(d_z)/del(z)
393  //
394  // grad(rho_0) = del(rho_0)/del(r) * e_r + 1/r * del(rho_0)/del(phi) * e_phi + del(rho_0)/del(z) * e_z
395  // e_i: unit vectors in i = {r, phi, z} direction
396  //
397 
398  // Finite difference interpolation coefficients
399  const float coeffCent40 = 1.f / 12.f;
400  const float coeffCent41 = -2.f / 3.f;
401  const float coeffCent42 = -1.f * coeffCent41;
402  const float coeffCent43 = -1.f * coeffCent40;
403  const float coeffFwd30 = -11.f / 6.f;
404  const float coeffFwd31 = 3.f;
405  const float coeffFwd32 = -1.5;
406  const float coeffFwd33 = 1.f / 3.f;
407  const float coeffFwd20 = -1.5;
408  const float coeffFwd21 = 2.f;
409  const float coeffFwd22 = -0.5;
410 
411  std::vector<float>* scDensity = nullptr;
412  TMatrixD** matrixDriftZ = nullptr;
413  TMatrixD** matrixDriftRPhi = nullptr;
414  TMatrixD** matrixDriftR = nullptr;
415  for (int iside = 0; iside < 2; ++iside) {
416  const int signZ = iside == 0 ? 1 : -1;
417  if (iside == 0) {
418  scDensity = &mSpaceChargeDensityA;
419  matrixDriftZ = (TMatrixD**)mMatrixIonDriftZA.get();
420  matrixDriftRPhi = (TMatrixD**)mMatrixIonDriftRPhiA.get();
421  matrixDriftR = (TMatrixD**)mMatrixIonDriftRA.get();
422  } else {
423  scDensity = &mSpaceChargeDensityC;
424  matrixDriftZ = (TMatrixD**)mMatrixIonDriftZC.get();
425  matrixDriftRPhi = (TMatrixD**)mMatrixIonDriftRPhiC.get();
426  matrixDriftR = (TMatrixD**)mMatrixIonDriftRC.get();
427  }
429  std::vector<float> newSCDensity(mNPhi * mNR * mNZ);
430  for (int iphi0 = 0; iphi0 < mNPhi; ++iphi0) {
431 
432  for (int ir0 = 0; ir0 < mNR; ++ir0) {
433  const float r0 = static_cast<float>(mCoordR[ir0]);
434 
435  for (int iz0 = 0; iz0 < mNZ; ++iz0) {
436 
437  // rho_0 * div(d)
438  float ddrdr = 0.f;
439  if (ir0 > 1 && ir0 < mNR - 2) {
440  ddrdr = (coeffCent40 * (*matrixDriftR[iphi0])(ir0 - 2, iz0) + coeffCent41 * (*matrixDriftR[iphi0])(ir0 - 1, iz0) + coeffCent42 * (*matrixDriftR[iphi0])(ir0 + 1, iz0) + coeffCent43 * (*matrixDriftR[iphi0])(ir0 + 2, iz0)) / static_cast<float>(mVoxelSizeR);
441  } else if (ir0 < 2) {
442  ddrdr = (coeffFwd30 * (*matrixDriftR[iphi0])(ir0, iz0) + coeffFwd31 * (*matrixDriftR[iphi0])(ir0 + 1, iz0) + coeffFwd32 * (*matrixDriftR[iphi0])(ir0 + 2, iz0) + coeffFwd33 * (*matrixDriftR[iphi0])(ir0 + 3, iz0)) / static_cast<float>(mVoxelSizeR);
443  } else if (ir0 > (mNR - 3)) {
444  ddrdr = -1 * (coeffFwd30 * (*matrixDriftR[iphi0])(ir0, iz0) + coeffFwd31 * (*matrixDriftR[iphi0])(ir0 - 1, iz0) + coeffFwd32 * (*matrixDriftR[iphi0])(ir0 - 2, iz0) + coeffFwd33 * (*matrixDriftR[iphi0])(ir0 - 3, iz0)) / static_cast<float>(mVoxelSizeR);
445  }
446 
447  const int iphiCent0 = iphi0 - 2 + (mNPhi) * (iphi0 < 2);
448  const int iphiCent1 = iphi0 - 1 + (mNPhi) * (iphi0 < 1);
449  const int iphiCent3 = iphi0 + 1 - (mNPhi) * (iphi0 > (mNPhi - 2));
450  const int iphiCent4 = iphi0 + 2 - (mNPhi) * (iphi0 > (mNPhi - 3));
451  const float ddphidphi = (coeffCent40 * (*matrixDriftRPhi[iphiCent0])(ir0, iz0) + coeffCent41 * (*matrixDriftRPhi[iphiCent1])(ir0, iz0) + coeffCent42 * (*matrixDriftRPhi[iphiCent3])(ir0, iz0) + coeffCent43 * (*matrixDriftRPhi[iphiCent4])(ir0, iz0)) / static_cast<float>(mVoxelSizePhi);
452 
453  float ddzdz = 0.f;
454  if (iz0 > 1 && iz0 < mNZ - 2) {
455  ddzdz = signZ * (coeffCent40 * (*matrixDriftZ[iphi0])(ir0, iz0 - 2) + coeffCent41 * (*matrixDriftZ[iphi0])(ir0, iz0 - 1) + coeffCent42 * (*matrixDriftZ[iphi0])(ir0, iz0 + 1) + coeffCent43 * (*matrixDriftZ[iphi0])(ir0, iz0 + 2)) / static_cast<float>(mVoxelSizeR);
456  } else if (iz0 == 1 || iz0 == (mNZ - 2)) {
457  ddzdz = signZ * (-0.5 * (*matrixDriftZ[iphi0])(ir0, iz0 - 1) + 0.5 * (*matrixDriftZ[iphi0])(ir0, iz0 + 1)) / static_cast<float>(mVoxelSizeR);
458  } else if (iz0 == 0) {
459  ddzdz = signZ * (coeffFwd20 * (*matrixDriftZ[iphi0])(ir0, iz0) + coeffFwd21 * (*matrixDriftZ[iphi0])(ir0, iz0 + 1) + coeffFwd22 * (*matrixDriftZ[iphi0])(ir0, iz0 + 2)) / static_cast<float>(mVoxelSizeR);
460  } else if (iz0 == (mNZ - 1)) {
461  ddzdz = -1 * signZ * (coeffFwd20 * (*matrixDriftZ[iphi0])(ir0, iz0) + coeffFwd21 * (*matrixDriftZ[iphi0])(ir0, iz0 - 1) + coeffFwd22 * (*matrixDriftZ[iphi0])(ir0, iz0 - 2)) / static_cast<float>(mVoxelSizeR);
462  }
463 
464  const float qdivd = (*scDensity)[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] * (((*matrixDriftR[iphi0])(ir0, iz0) + ddphidphi) / r0 + ddrdr + ddzdz);
465 
466  // - d * grad(rho_0) = - d_drift * grad(rho_0(x)) + d_dist * grad(rho_0(x+d_drift)) = (charge0 - charge1) + d_dist * grad(rho_0(x-d_drift))
467  if (iz0 < (mNZ - 1)) {
468  const float dr = (*matrixDriftR[iphi0])(ir0, iz0);
469  const float drphi = (*matrixDriftRPhi[iphi0])(ir0, iz0);
470  const float dz = (*matrixDriftZ[iphi0])(ir0, iz0) + mVoxelSizeZ * signZ;
471 
472  const int ir1 = dr < 0 ? ir0 - 1 * (ir0 == (mNR - 1)) : ir0 - 1 + 1 * (ir0 == 0);
473  const int ir2 = dr < 0 ? ir0 + 1 - 1 * (ir0 == (mNR - 1)) : ir0 + 1 * (ir0 == 0);
474  const int iphi1 = drphi < 0 ? iphi0 : iphi0 - 1 + (mNPhi) * (iphi0 == 0);
475  const int iphi2 = drphi < 0 ? iphi0 + 1 - (mNPhi) * (iphi0 == (mNPhi - 1)) : iphi0;
476  const int iz1 = dz < 0 ? iz0 + 1 - 1 * (iside == 0) * (iz0 == (mNZ - 2)) : iz0 + 2 * (iside == 1) - 1 * (iside == 1) * (iz0 == (mNZ - 2));
477  const int iz2 = dz < 0 ? iz0 + 2 - 2 * (iside == 1) - 1 * (iside == 0) * (iz0 == (mNZ - 2)) : iz0 + 1 - 1 * (iside == 1) * (iz0 == (mNZ - 2));
478 
479  const float dqdr = ((*scDensity)[iphi0 * mNR * mNZ + ir2 * mNZ + (iz0 + 1)] - (*scDensity)[iphi0 * mNR * mNZ + ir1 * mNZ + (iz0 + 1)]) / static_cast<float>(mVoxelSizeR);
480  const float dqdphi = ((*scDensity)[iphi2 * mNR * mNZ + ir0 * mNZ + (iz0 + 1)] - (*scDensity)[iphi1 * mNR * mNZ + ir0 * mNZ + (iz0 + 1)]) / static_cast<float>(mVoxelSizePhi);
481  const float dqdz = ((*scDensity)[iphi0 * mNR * mNZ + ir0 * mNZ + iz2] - (*scDensity)[iphi0 * mNR * mNZ + ir0 * mNZ + iz1]) / static_cast<float>(mVoxelSizeZ);
482 
483  const float dgradq = dr * dqdr + drphi / r0 * dqdphi + dz * dqdz;
484 
485  newSCDensity[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] = (*scDensity)[iphi0 * mNR * mNZ + ir0 * mNZ + (iz0 + 1)] - (qdivd + dgradq);
486  } else {
487  newSCDensity[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] = -qdivd;
488  }
489 
490  if (newSCDensity[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] < 0.f) {
491  newSCDensity[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] = 0.f;
492  }
493  }
494  }
495  }
496  (*scDensity).swap(newSCDensity);
497  }

500 void SpaceCharge::propagateIons()
501 {
502  std::vector<float>* scDensity = nullptr;
503  AliTPCLookUpTable3DInterpolatorD* lookUpIonDrift = nullptr;
504  for (int iside = 0; iside < 2; ++iside) {
505  const int signZ = iside == 0 ? 1 : -1;
506  if (iside == 0) {
507  scDensity = &mSpaceChargeDensityA;
508  lookUpIonDrift = mLookUpIonDriftA.get();
509  } else {
510  scDensity = &mSpaceChargeDensityC;
511  lookUpIonDrift = mLookUpIonDriftC.get();
512  }
513  std::vector<float> newSCDensity(mNPhi * mNR * mNZ);
514 
515  for (int iphi0 = 0; iphi0 < mNPhi; ++iphi0) {
516  const double phi0 = static_cast<float>(mCoordPhi[iphi0]);
517 
518  for (int ir0 = 0; ir0 < mNR; ++ir0) {
519  const double r0 = static_cast<float>(mCoordR[ir0]);
520  const double r1 = r0 + mVoxelSizeR;
521 
522  for (int iz0 = 0; iz0 < mNZ; ++iz0) {
523  const double z0 = mCoordZ[iz0] * signZ;
524 
525  const float ionDensity = (*scDensity)[iphi0 * mNR * mNZ + ir0 * mNZ + iz0] * AliTPCPoissonSolver::fgke0 / TMath::Qe(); // #ions / cm^3
526  const float nIons = std::round(ionDensity * mVoxelSizeZ * 0.5 * mVoxelSizePhi * (r1 * r1 - r0 * r0)); // absolute #ions in the voxel
527 
528  for (int iion = 0; iion < nIons; ++iion) {
529  double phiIon = phi0 + mRandomFlat.getNextValue() * mVoxelSizePhi;
530  double rIon = r0 + mRandomFlat.getNextValue() * mVoxelSizeR;
531  double zIon = z0 + mRandomFlat.getNextValue() * mVoxelSizeZ * signZ;
532 
533  double drphi = 0.f;
534  double dr = 0.f;
535  double dz = 0.f;
536  lookUpIonDrift->GetValue(rIon, phiIon, std::abs(zIon), dr, drphi, dz);
537  float phiIonF = static_cast<float>(phiIon + (drphi / rIon));
538  o2::utils::BringTo02PiGen(phiIonF);
539  rIon += dr;
540  zIon += dz;
541 
542  // continue if ion is outside the TPC boundaries in r or z
543  if (rIon > (mCoordR[mNR - 1] + mVoxelSizeR) || rIon < mCoordR[0]) {
544  continue;
545  }
546  if ((zIon * signZ) > (mCoordZ[mNZ - 1] + mVoxelSizeZ) || (zIon * signZ) < mCoordZ[0]) {
547  continue;
548  }
549 
550  const int iphiDrift = TMath::BinarySearch(mNPhi, mCoordPhi.data(), static_cast<double>(phiIonF));
551  const int irDrift = TMath::BinarySearch(mNR, mCoordR.data(), rIon);
552  const int izDrift = TMath::BinarySearch(mNZ, mCoordZ.data(), std::abs(zIon));
553  newSCDensity[iphiDrift * mNR * mNZ + irDrift * mNZ + izDrift] += ions2Charge(irDrift, 1);
554  }
555  }
556  }
557  }
558  (*scDensity).swap(newSCDensity);
559  }
