41 MultiView::MultiView()
42 {
43  // set scene names and descriptions
44  mSceneNames[Scene3dGeom] = "3D Geometry Scene";
45  mSceneNames[SceneRphiGeom] = "R-Phi Geometry Scene";
46  mSceneNames[SceneZrhoGeom] = "Rho-Z Geometry Scene";
47  mSceneNames[Scene3dEvent] = "3D Event Scene";
48  mSceneNames[SceneRphiEvent] = "R-Phi Event Scene";
49  mSceneNames[SceneZrhoEvent] = "Rho-Z Event Scene";
50 
51  mSceneDescriptions[Scene3dGeom] = "Scene holding 3D geometry.";
52  mSceneDescriptions[SceneRphiGeom] = "Scene holding projected geometry for the R-Phi view.";
53  mSceneDescriptions[SceneZrhoGeom] = "Scene holding projected geometry for the Rho-Z view.";
54  mSceneDescriptions[Scene3dEvent] = "Scene holding 3D event.";
55  mSceneDescriptions[SceneRphiEvent] = "Scene holding projected event for the R-Phi view.";
56  mSceneDescriptions[SceneZrhoEvent] = "Scene holding projected event for the Rho-Z view.";
57 
58  // spawn scenes
59  mScenes[Scene3dGeom] = gEve->GetGlobalScene();
60  mScenes[Scene3dGeom]->SetNameTitle(mSceneNames[Scene3dGeom].c_str(), mSceneDescriptions[Scene3dGeom].c_str());
61 
62  mScenes[Scene3dEvent] = gEve->GetEventScene();
63  mScenes[Scene3dEvent]->SetNameTitle(mSceneNames[Scene3dEvent].c_str(), mSceneDescriptions[Scene3dEvent].c_str());
64 
65  for (int i = SceneRphiGeom; i < NumberOfScenes; ++i) {
66  mScenes[i] = gEve->SpawnNewScene(mSceneNames[i].c_str(), mSceneDescriptions[i].c_str());
67  }
68 
69  // Projection managers
70  mProjections[ProjectionRphi] = new TEveProjectionManager();
71  mProjections[ProjectionZrho] = new TEveProjectionManager();
72 
73  mProjections[ProjectionRphi]->SetProjection(TEveProjection::kPT_RPhi);
74  mProjections[ProjectionZrho]->SetProjection(TEveProjection::kPT_RhoZ);
75 
76  gEve->AddToListTree(static_cast<TEveElement*>(mProjections[ProjectionRphi]), false);
77  gEve->AddToListTree(static_cast<TEveElement*>(mProjections[ProjectionZrho]), false);
78 
79  // add axes
80  TEnv settings;
81  ConfigurationManager::getInstance().getConfig(settings);
82  const bool showAxes = settings.GetValue("axes.show", false);
83  if (showAxes) {
84  for (int i = 0; i < NumberOfProjections; ++i) {
85  TEveProjectionAxes axes(mProjections[static_cast<EProjections>(i)]);
86  axes.SetMainColor(kWhite);
87  axes.SetTitle("R-Phi");
88  axes.SetTitleSize(0.05);
89  axes.SetTitleFont(102);
90  axes.SetLabelSize(0.025);
91  axes.SetLabelFont(102);
92  mScenes[getSceneOfProjection(static_cast<EProjections>(i))]->AddElement(&axes);
93  }
94  }
95 
96  setupMultiview();
97  sInstance = this;
98 }

113 void MultiView::setupMultiview()
114 {
115  // Split window in packs for 3D and projections, create viewers and add scenes to them
116  TEveWindowSlot* slot = TEveWindow::CreateWindowInTab(gEve->GetBrowser()->GetTabRight());
117  TEveWindowPack* pack = slot->MakePack();
118 
119  pack->SetElementName("Multi View");
120  pack->SetHorizontal();
121  pack->SetShowTitleBar(kFALSE);
122 
123  pack->NewSlotWithWeight(2)->MakeCurrent(); // new slot is created from pack
124  mViews[View3d] = gEve->SpawnNewViewer("3D View", "");
125  mViews[View3d]->AddScene(mScenes[Scene3dGeom]);
126  mViews[View3d]->AddScene(mScenes[Scene3dEvent]);
127 
128  pack = pack->NewSlot()->MakePack();
129  pack->SetNameTitle("2D Views", "");
130  pack->SetShowTitleBar(kFALSE);
131  pack->NewSlot()->MakeCurrent();
132  mViews[ViewRphi] = gEve->SpawnNewViewer("R-Phi View", "");
133  mViews[ViewRphi]->GetGLViewer()->SetCurrentCamera(TGLViewer::kCameraOrthoXOY);
134  mViews[ViewRphi]->AddScene(mScenes[SceneRphiGeom]);
135  mViews[ViewRphi]->AddScene(mScenes[SceneRphiEvent]);
136 
137  pack->NewSlot()->MakeCurrent();
138  mViews[ViewZrho] = gEve->SpawnNewViewer("Rho-Z View", "");
139  mViews[ViewZrho]->GetGLViewer()->SetCurrentCamera(TGLViewer::kCameraOrthoXOY);
140  mViews[ViewZrho]->AddScene(mScenes[SceneZrhoGeom]);
141  mViews[ViewZrho]->AddScene(mScenes[SceneZrhoEvent]);
142 }

161 void MultiView::registerGeometry(TEveGeoShape* geom, bool threeD, bool rPhi, bool zRho)
162 {
163  if (!geom) {
164  LOG(ERROR) << "MultiView::registerGeometry -- geometry is NULL!";
165  exit(-1);
166  }
167  //mGeomVector.push_back(geom);
168 
169  TEveProjectionManager* projection;
170 
171  if (threeD) {
172  gEve->AddElement(geom, getScene(Scene3dGeom));
173  }
174  if (rPhi) {
175  projection = getProjection(ProjectionRphi);
176  projection->SetCurrentDepth(-10);
177  projection->ImportElements(geom, getScene(SceneRphiGeom));
178  projection->SetCurrentDepth(0);
179  }
180  if (zRho) {
181  projection = getProjection(ProjectionZrho);
182  projection->SetCurrentDepth(-10);
183  projection->ImportElements(geom, getScene(SceneZrhoGeom));
184  projection->SetCurrentDepth(0);
185  }
186 }
