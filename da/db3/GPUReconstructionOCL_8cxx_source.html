60 int GPUReconstructionOCL::InitDevice_Runtime()
61 {
62  if (mMaster == nullptr) {
63  cl_int ocl_error;
64  cl_uint num_platforms;
65  if (GPUFailedMsgI(clGetPlatformIDs(0, nullptr, &num_platforms))) {
66  quit("Error getting OpenCL Platform Count");
67  }
68  if (num_platforms == 0) {
69  quit("No OpenCL Platform found");
70  }
71  if (mDeviceProcessingSettings.debugLevel >= 2) {
72  GPUInfo("%d OpenCL Platforms found", num_platforms);
73  }
74 
75  // Query platforms
76  mInternals->platforms.reset(new cl_platform_id[num_platforms]);
77  if (GPUFailedMsgI(clGetPlatformIDs(num_platforms, mInternals->platforms.get(), nullptr))) {
78  quit("Error getting OpenCL Platforms");
79  }
80 
81  bool found = false;
82  if (mDeviceProcessingSettings.platformNum >= 0) {
83  if (mDeviceProcessingSettings.platformNum >= (int)num_platforms) {
84  quit("Invalid platform specified");
85  }
86  mInternals->platform = mInternals->platforms[mDeviceProcessingSettings.platformNum];
87  found = true;
88  } else {
89  for (unsigned int i_platform = 0; i_platform < num_platforms; i_platform++) {
90  char platform_profile[64] = {}, platform_version[64] = {}, platform_name[64] = {}, platform_vendor[64] = {};
91  clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_PROFILE, sizeof(platform_profile), platform_profile, nullptr);
92  clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
93  clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_NAME, sizeof(platform_name), platform_name, nullptr);
94  clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
95  if (mDeviceProcessingSettings.debugLevel >= 2) {
96  GPUInfo("Available Platform %d: (%s %s) %s %s", i_platform, platform_profile, platform_version, platform_vendor, platform_name);
97  }
98  if (CheckPlatform(i_platform)) {
99  found = true;
100  mInternals->platform = mInternals->platforms[i_platform];
101  }
102  }
103  }
104 
105  if (found == false) {
106  quit("Did not find compatible OpenCL Platform");
107  }
108 
109  cl_uint count, bestDevice = (cl_uint)-1;
110  if (GPUFailedMsgI(clGetDeviceIDs(mInternals->platform, CL_DEVICE_TYPE_ALL, 0, nullptr, &count))) {
111  quit("Error getting OPENCL Device Count");
112  }
113 
114  // Query devices
115  mInternals->devices.reset(new cl_device_id[count]);
116  if (GPUFailedMsgI(clGetDeviceIDs(mInternals->platform, CL_DEVICE_TYPE_ALL, count, mInternals->devices.get(), nullptr))) {
117  quit("Error getting OpenCL devices");
118  }
119 
120  char device_vendor[64], device_name[64];
121  cl_device_type device_type;
122  cl_uint freq, shaders;
123 
124  if (mDeviceProcessingSettings.debugLevel >= 2) {
125  GPUInfo("Available OPENCL devices:");
126  }
127  std::vector<bool> devicesOK(count, false);
128  for (unsigned int i = 0; i < count; i++) {
129  if (mDeviceProcessingSettings.debugLevel >= 3) {
130  GPUInfo("Examining device %d", i);
131  }
132  cl_uint nbits;
133  cl_bool endian;
134 
135  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_NAME, 64, device_name, nullptr);
136  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_VENDOR, 64, device_vendor, nullptr);
137  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_TYPE, sizeof(cl_device_type), &device_type, nullptr);
138  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_MAX_CLOCK_FREQUENCY, sizeof(freq), &freq, nullptr);
139  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(shaders), &shaders, nullptr);
140  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_ADDRESS_BITS, sizeof(nbits), &nbits, nullptr);
141  clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_ENDIAN_LITTLE, sizeof(endian), &endian, nullptr);
142  int deviceOK = true;
143  const char* deviceFailure = "";
144  if (mDeviceProcessingSettings.gpuDeviceOnly && ((device_type & CL_DEVICE_TYPE_CPU) || !(device_type & CL_DEVICE_TYPE_GPU))) {
145  deviceOK = false;
146  deviceFailure = "No GPU device";
147  }
148  if (nbits / 8 != sizeof(void*)) {
149  deviceOK = false;
150  deviceFailure = "No 64 bit device";
151  }
152  if (!endian) {
153  deviceOK = false;
154  deviceFailure = "No Little Endian Mode";
155  }
156 
157  double bestDeviceSpeed = -1, deviceSpeed = (double)freq * (double)shaders;
158  if (mDeviceProcessingSettings.debugLevel >= 2) {
159  GPUImportant("Device %s%2d: %s %s (Frequency %d, Shaders %d, %d bit) (Speed Value: %lld)%s %s", deviceOK ? " " : "[", i, device_vendor, device_name, (int)freq, (int)shaders, (int)nbits, (long long int)deviceSpeed, deviceOK ? " " : " ]", deviceOK ? "" : deviceFailure);
160  }
161  if (!deviceOK) {
162  continue;
163  }
164  devicesOK[i] = true;
165  if (deviceSpeed > bestDeviceSpeed) {
166  bestDevice = i;
167  bestDeviceSpeed = deviceSpeed;
168  } else {
169  if (mDeviceProcessingSettings.debugLevel >= 2) {
170  GPUInfo("Skipping: Speed %f < %f", deviceSpeed, bestDeviceSpeed);
171  }
172  }
173  }
174  if (bestDevice == (cl_uint)-1) {
175  quit("No %sOPENCL Device available, aborting OPENCL Initialisation", count ? "appropriate " : "");
176  }
177 
178  if (mDeviceProcessingSettings.deviceNum > -1) {
179  if (mDeviceProcessingSettings.deviceNum >= (signed)count) {
180  quit("Requested device ID %d does not exist", mDeviceProcessingSettings.deviceNum);
181  } else if (!devicesOK[mDeviceProcessingSettings.deviceNum]) {
182  quit("Unsupported device requested (%d)", mDeviceProcessingSettings.deviceNum);
183  } else {
184  bestDevice = mDeviceProcessingSettings.deviceNum;
185  }
186  }
187  mInternals->device = mInternals->devices[bestDevice];
188 
189  cl_ulong constantBuffer, globalMem, localMem;
190  char deviceVersion[64];
191  size_t maxWorkGroup, maxWorkItems[3];
192  clGetDeviceInfo(mInternals->device, CL_DEVICE_NAME, 64, device_name, nullptr);
193  clGetDeviceInfo(mInternals->device, CL_DEVICE_VENDOR, 64, device_vendor, nullptr);
194  clGetDeviceInfo(mInternals->device, CL_DEVICE_TYPE, sizeof(cl_device_type), &device_type, nullptr);
195  clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_CLOCK_FREQUENCY, sizeof(freq), &freq, nullptr);
196  clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(shaders), &shaders, nullptr);
197  clGetDeviceInfo(mInternals->device, CL_DEVICE_GLOBAL_MEM_SIZE, sizeof(globalMem), &globalMem, nullptr);
198  clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_CONSTANT_BUFFER_SIZE, sizeof(constantBuffer), &constantBuffer, nullptr);
199  clGetDeviceInfo(mInternals->device, CL_DEVICE_LOCAL_MEM_SIZE, sizeof(localMem), &localMem, nullptr);
200  clGetDeviceInfo(mInternals->device, CL_DEVICE_VERSION, sizeof(deviceVersion) - 1, deviceVersion, nullptr);
201  clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_WORK_GROUP_SIZE, sizeof(maxWorkGroup), &maxWorkGroup, nullptr);
202  clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_WORK_ITEM_SIZES, sizeof(maxWorkItems), maxWorkItems, nullptr);
203  if (mDeviceProcessingSettings.debugLevel >= 2) {
204  GPUInfo("Using OpenCL device %d: %s %s with properties:", bestDevice, device_vendor, device_name);
205  GPUInfo("\tVersion = %s", deviceVersion);
206  GPUInfo("\tFrequency = %d", (int)freq);
207  GPUInfo("\tShaders = %d", (int)shaders);
208  GPUInfo("\tGLobalMemory = %lld", (long long int)globalMem);
209  GPUInfo("\tContantMemoryBuffer = %lld", (long long int)constantBuffer);
210  GPUInfo("\tLocalMemory = %lld", (long long int)localMem);
211  GPUInfo("\tmaxThreadsPerBlock = %lld", (long long int)maxWorkGroup);
212  GPUInfo("\tmaxThreadsDim = %lld %lld %lld", (long long int)maxWorkItems[0], (long long int)maxWorkItems[1], (long long int)maxWorkItems[2]);
213  GPUInfo(" ");
214  }
215 #ifndef GPUCA_NO_CONSTANT_MEMORY
216  if (gGPUConstantMemBufferSize > constantBuffer) {
217  quit("Insufficient constant memory available on GPU %d < %d!", (int)constantBuffer, (int)gGPUConstantMemBufferSize);
218  }
219 #endif
220 
221  mDeviceName = device_name;
222  mDeviceName += " (OpenCL)";
223  mBlockCount = shaders;
224  mWarpSize = 32;
225  mMaxThreads = std::max<int>(mMaxThreads, maxWorkGroup * mBlockCount);
226 
227  mInternals->context = clCreateContext(nullptr, ContextForAllPlatforms() ? count : 1, ContextForAllPlatforms() ? mInternals->devices.get() : &mInternals->device, nullptr, nullptr, &ocl_error);
228  if (GPUFailedMsgI(ocl_error)) {
229  quit("Could not create OPENCL Device Context!");
230  }
231 
232  if (GetOCLPrograms()) {
233  return 1;
234  }
235 
236  if (mDeviceProcessingSettings.debugLevel >= 2) {
237  GPUInfo("OpenCL program and kernels loaded successfully");
238  }
239 
240  mInternals->mem_gpu = clCreateBuffer(mInternals->context, CL_MEM_READ_WRITE, mDeviceMemorySize, nullptr, &ocl_error);
241  if (GPUFailedMsgI(ocl_error)) {
242  clReleaseContext(mInternals->context);
243  quit("OPENCL Memory Allocation Error");
244  }
245 
246  mInternals->mem_constant = clCreateBuffer(mInternals->context, CL_MEM_READ_ONLY, gGPUConstantMemBufferSize, nullptr, &ocl_error);
247  if (GPUFailedMsgI(ocl_error)) {
248  clReleaseMemObject(mInternals->mem_gpu);
249  clReleaseContext(mInternals->context);
250  quit("OPENCL Constant Memory Allocation Error");
251  }
252 
253  for (int i = 0; i < mNStreams; i++) {
254 #ifdef CL_VERSION_2_0
255  cl_queue_properties prop = 0;
256  if (mDeviceProcessingSettings.deviceTimers) {
257  prop |= CL_QUEUE_PROFILING_ENABLE;
258  }
259  mInternals->command_queue[i] = clCreateCommandQueueWithProperties(mInternals->context, mInternals->device, &prop, &ocl_error);
260 #else
261  mInternals->command_queue[i] = clCreateCommandQueue(mInternals->context, mInternals->device, 0, &ocl_error);
262 #endif
263  if (GPUFailedMsgI(ocl_error)) {
264  quit("Error creating OpenCL command queue");
265  }
266  }
267  if (GPUFailedMsgI(clEnqueueMigrateMemObjects(mInternals->command_queue[0], 1, &mInternals->mem_gpu, 0, 0, nullptr, nullptr))) {
268  quit("Error migrating buffer");
269  }
270  if (GPUFailedMsgI(clEnqueueMigrateMemObjects(mInternals->command_queue[0], 1, &mInternals->mem_constant, 0, 0, nullptr, nullptr))) {
271  quit("Error migrating buffer");
272  }
273 
274  mInternals->mem_host = clCreateBuffer(mInternals->context, CL_MEM_READ_WRITE | CL_MEM_ALLOC_HOST_PTR, mHostMemorySize, nullptr, &ocl_error);
275  if (GPUFailedMsgI(ocl_error)) {
276  quit("Error allocating pinned host memory");
277  }
278 
279  const char* krnlGetPtr = "__kernel void krnlGetPtr(__global char* gpu_mem, __global char* constant_mem, __global size_t* host_mem) {if (get_global_id(0) == 0) {host_mem[0] = (size_t) gpu_mem; host_mem[1] = (size_t) constant_mem;}}";
280  cl_program program = clCreateProgramWithSource(mInternals->context, 1, (const char**)&krnlGetPtr, nullptr, &ocl_error);
281  if (GPUFailedMsgI(ocl_error)) {
282  quit("Error creating program object");
283  }
284  ocl_error = clBuildProgram(program, 1, &mInternals->device, "", nullptr, nullptr);
285  if (GPUFailedMsgI(ocl_error)) {
286  char build_log[16384];
287  clGetProgramBuildInfo(program, mInternals->device, CL_PROGRAM_BUILD_LOG, 16384, build_log, nullptr);
288  GPUImportant("Build Log:\n\n%s\n\n", build_log);
289  quit("Error compiling program");
290  }
291  cl_kernel kernel = clCreateKernel(program, "krnlGetPtr", &ocl_error);
292  if (GPUFailedMsgI(ocl_error)) {
293  quit("Error creating kernel");
294  }
295  OCLsetKernelParameters(kernel, mInternals->mem_gpu, mInternals->mem_constant, mInternals->mem_host);
296  if (GPUFailedMsgI(clExecuteKernelA(mInternals->command_queue[0], kernel, 16, 16, nullptr)) ||
297  GPUFailedMsgI(clFinish(mInternals->command_queue[0])) ||
298  GPUFailedMsgI(clReleaseKernel(kernel)) ||
299  GPUFailedMsgI(clReleaseProgram(program))) {
300  quit("Error obtaining device memory ptr");
301  }
302 
303  if (mDeviceProcessingSettings.debugLevel >= 2) {
304  GPUInfo("Mapping hostmemory");
305  }
306  mHostMemoryBase = clEnqueueMapBuffer(mInternals->command_queue[0], mInternals->mem_host, CL_TRUE, CL_MAP_READ | CL_MAP_WRITE, 0, mHostMemorySize, 0, nullptr, nullptr, &ocl_error);
307  if (GPUFailedMsgI(ocl_error)) {
308  quit("Error allocating Page Locked Host Memory");
309  }
310 
311  mDeviceMemoryBase = ((void**)mHostMemoryBase)[0];
312  mDeviceConstantMem = (GPUConstantMem*)((void**)mHostMemoryBase)[1];
313 
314  if (mDeviceProcessingSettings.debugLevel >= 1) {
315  GPUInfo("Memory ptrs: GPU (%lld bytes): %p - Host (%lld bytes): %p", (long long int)mDeviceMemorySize, mDeviceMemoryBase, (long long int)mHostMemorySize, mHostMemoryBase);
316  memset(mHostMemoryBase, 0xDD, mHostMemorySize);
317  }
318 
319  GPUInfo("OPENCL Initialisation successfull (%d: %s %s (Frequency %d, Shaders %d), %lld / %lld bytes host / global memory, Stack frame %d, Constant memory %lld)", bestDevice, device_vendor, device_name, (int)freq, (int)shaders, (long long int)mDeviceMemorySize,
320  (long long int)mHostMemorySize, -1, (long long int)gGPUConstantMemBufferSize);
321  } else {
322  GPUReconstructionOCL* master = dynamic_cast<GPUReconstructionOCL*>(mMaster);
323  mBlockCount = master->mBlockCount;
324  mWarpSize = master->mWarpSize;
325  mMaxThreads = master->mMaxThreads;
326  mDeviceName = master->mDeviceName;
327  mDeviceConstantMem = master->mDeviceConstantMem;
328  mInternals = master->mInternals;
329  }
330 
331  return (0);
332 }

334 int GPUReconstructionOCL::ExitDevice_Runtime()
335 {
336  // Uninitialize OPENCL
337  SynchronizeGPU();
338 
339  if (mMaster == nullptr) {
340  if (mDeviceMemoryBase) {
341  clReleaseMemObject(mInternals->mem_gpu);
342  clReleaseMemObject(mInternals->mem_constant);
343  for (unsigned int i = 0; i < mInternals->kernels.size(); i++) {
344  clReleaseKernel(mInternals->kernels[i].first);
345  }
346  mInternals->kernels.clear();
347  }
348  if (mHostMemoryBase) {
349  clEnqueueUnmapMemObject(mInternals->command_queue[0], mInternals->mem_host, mHostMemoryBase, 0, nullptr, nullptr);
350  for (int i = 0; i < mNStreams; i++) {
351  clReleaseCommandQueue(mInternals->command_queue[i]);
352  }
353  clReleaseMemObject(mInternals->mem_host);
354  }
355 
356  clReleaseProgram(mInternals->program);
357  clReleaseContext(mInternals->context);
358  GPUInfo("OPENCL Uninitialized");
359  }
360  mDeviceMemoryBase = nullptr;
361  mHostMemoryBase = nullptr;
362 
363  return (0);
364 }

366 size_t GPUReconstructionOCL::GPUMemCpy(void* dst, const void* src, size_t size, int stream, int toGPU, deviceEvent* ev, deviceEvent* evList, int nEvents)
367 {
368  if (evList == nullptr) {
369  nEvents = 0;
370  }
371  if (mDeviceProcessingSettings.debugLevel >= 3) {
372  stream = -1;
373  }
374  if (stream == -1) {
375  SynchronizeGPU();
376  }
377  if (toGPU == -2) {
378  GPUFailedMsg(clEnqueueCopyBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, mInternals->mem_gpu, (char*)src - (char*)mDeviceMemoryBase, (char*)dst - (char*)mDeviceMemoryBase, size, nEvents, (cl_event*)evList, (cl_event*)ev));
379  } else if (toGPU) {
380  GPUFailedMsg(clEnqueueWriteBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, stream == -1, (char*)dst - (char*)mDeviceMemoryBase, size, src, nEvents, (cl_event*)evList, (cl_event*)ev));
381  } else {
382  GPUFailedMsg(clEnqueueReadBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, stream == -1, (char*)src - (char*)mDeviceMemoryBase, size, dst, nEvents, (cl_event*)evList, (cl_event*)ev));
383  }

414 int GPUReconstructionOCL::DoStuckProtection(int stream, void* event)
415 {
416  if (mDeviceProcessingSettings.stuckProtection) {
417  cl_int tmp = 0;
418  for (int i = 0; i <= mDeviceProcessingSettings.stuckProtection / 50; i++) {
419  usleep(50);
420  clGetEventInfo(*(cl_event*)event, CL_EVENT_COMMAND_EXECUTION_STATUS, sizeof(tmp), &tmp, nullptr);
421  if (tmp == CL_COMPLETE) {
422  break;
423  }
424  }
425  if (tmp != CL_COMPLETE) {
426  mGPUStuck = 1;
427  quit("GPU Stuck, future processing in this component is disabled, skipping event (GPU Event State %d)", (int)tmp);
428  }
429  } else {
430  clFinish(mInternals->command_queue[stream]);
431  }
432  return 0;
433 }
