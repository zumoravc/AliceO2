647 //_____________________________________________________
648 inline float TrackResiduals::getDXI(int ix) const
649 {
650  if (mUniformBins[VoxX]) {
651  return mDXI;
652  } else {
653  if (ix < param::NRowsPerROC[0]) {
654  // we are in the IROC
655  return 1.f / param::RowDX[0];
656  } else if (ix > param::NRowsAccumulated[param::NROCTypes - 1]) {
657  // we are in the last OROC
658  return 1.f / param::RowDX[param::NROCTypes - 1];
659  }
660 #ifdef TPC_RUN2
661  else {
662  // we are in OROC1
663  return 1.f / param::RowDX[1];
664  }
665 #else
666  else if (ix < param::NRowsAccumulated[2]) {
667  // OROC1
668  return 1.f / param::RowDX[1];
669  } else {
670  // OROC2
671  return 1.f / param::RowDX[2];
672  }
673 #endif
674  }
675 }

683 //_____________________________________________________
684 inline float TrackResiduals::getY2X(int ix, int ip) const
685 {
686  if (mUniformBins[VoxF]) {
687  return (0.5f + ip) * mDY2X[ix] - mMaxY2X[ix];
688  }
689  return mMaxY2X[ix] * (mY2XBinsCenter[ip] - mY2XBinsDH[ip]);
690 }

694 {
695  if (mUniformBins[VoxZ]) {
696  return (0.5f + iz) * getDZ2X();
697  }
698  return mZ2XBinsCenter[iz];
699 }

736 //_____________________________________________________
737 inline int TrackResiduals::getXBinExact(float x) const
738 {
739  if (mUniformBins[VoxX]) {
740  int ix = (x - param::MinX) * mDXI;
741  return (ix < 0 || ix >= mNXBins) ? -1 : ix;
742  }
743  return getRowID(x);
744 }

746 //_____________________________________________________
747 inline int TrackResiduals::getXBin(float x) const
748 {
749  int bx = getXBinExact(x);
750  return bx > -1 ? (bx < mNXBins ? bx : mNXBins - 1) : 0;
751 }
752 
753 //_____________________________________________________
754 inline int TrackResiduals::getY2XBinExact(float y2x, int ix) const
755 {
756  if (y2x < -mMaxY2X[ix]) {
757  return -1;
758  }
759  if (y2x > mMaxY2X[ix]) {
760  return mNY2XBins;
761  }
762  if (mUniformBins[VoxF]) {
763  return static_cast<int>((y2x + mMaxY2X[ix]) * getDY2XI(ix));
764  }
765  y2x /= mMaxY2X[ix];
766  for (int iBin = 0; iBin < mNY2XBins; ++iBin) {
767  if (y2x < mY2XBinsCenter[iBin] + mY2XBinsDH[iBin]) {
768  return iBin;
769  }
770  }
771  return mNY2XBins;
772 }

774 //_____________________________________________________
775 inline int TrackResiduals::getY2XBin(float y2x, int ix) const
776 {
777  int bp = getY2XBinExact(y2x, ix);
778  return bp > -1 ? (bp < mNY2XBins ? bp : mNY2XBins - 1) : 0;
779 }

781 //_____________________________________________________
782 inline int TrackResiduals::getZ2XBinExact(float z2x) const
783 {
784  if (mUniformBins[VoxZ]) {
785  float bz = z2x * getDZ2XI();
786  if (bz >= mNZ2XBins) {
787  return -1;
788  }
789  if (bz < 0) {
790  // accounting for clusters which were moved to the wrong side
791  // TODO: how can this happen?
792  bz = 0;
793  }
794  return static_cast<int>(nearbyint(bz));
795  }
796  for (int iBin = 0; iBin < mNZ2XBins; ++iBin) {
797  if (z2x < mZ2XBinsCenter[iBin] + mZ2XBinsDH[iBin]) {
798  return iBin;
