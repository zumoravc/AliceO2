91 class HistogramRegistry
92 {
93  public:
94  HistogramRegistry(char const* const name_, bool enable, std::vector<HistogramSpec> specs)
95  : name(name_),
96  enabled(enable),
97  mRegistryKey(),
98  mRegistryValue()
99  {
100  mRegistryKey.fill(0u);
101  for (auto& spec : specs) {
102  insert(spec);
103  }
104  }
105 
106  auto& get(char const* const name) const
107  {
108  const uint32_t id = compile_time_hash(name);
109  const uint32_t i = imask(id);
110  if (O2_BUILTIN_LIKELY(id == mRegistryKey[i])) {
111  return mRegistryValue[i];
112  }
113  for (auto j = 1u; j < MAX_REGISTRY_SIZE; ++j) {
114  if (id == mRegistryKey[imask(j + i)]) {
115  return mRegistryValue[imask(j + i)];
116  }
117  }
118  throw std::runtime_error("No match found!");
119  }
120 
121  // @return the associated OutputSpec
122  OutputSpec const spec()
123  {
124  ConcreteDataMatcher matcher{"HIST", "\0", 0};
125  strncpy(matcher.description.str, this->name.data(), 16);
126  return OutputSpec{OutputLabel{this->name}, matcher};
127  }
128 
129  OutputRef ref()
130  {
131  return OutputRef{this->name, 0};
132  }
133 
135  mutable uint32_t lookup = 0;
136 
137  private:
138  void insert(HistogramSpec& spec)
139  {
140  uint32_t i = imask(spec.id);
141  for (auto j = 0u; j < MAX_REGISTRY_SIZE; ++j) {
142  if (mRegistryValue[imask(j + i)].get() == nullptr) {
143  mRegistryKey[imask(j + i)] = spec.id;
144  mRegistryValue[imask(j + i)] = {std::make_unique<TH1F>(spec.name.data(), spec.readableName.data(), spec.config.nBins, spec.config.xmin, spec.config.xmax)};
145  lookup += j;
146  return;
147  }
148  }
149  throw std::runtime_error("Internal array is full.");
150  }
151 
152  inline constexpr uint32_t imask(uint32_t i) const
153  {
154  return i & mask;
155  }
156  std::string name;
157  bool enabled;
158 
161  static constexpr uint32_t mask = 0x1FF;
162  static constexpr uint32_t MAX_REGISTRY_SIZE = mask + 1;
163  std::array<uint32_t, MAX_REGISTRY_SIZE> mRegistryKey;
164  std::array<std::unique_ptr<TH1>, MAX_REGISTRY_SIZE> mRegistryValue;
165 };
