95  {
96  T value;
97  {
98  std::unique_lock<std::mutex> lock(mMutex);
99  if (_BASE::empty() && !mStop) {
100  // TODO: proper implementation with timeout with wait_for, return
101  // value will be std::cv_status::no_timeout. The condition variable
102  // is allowed to wake up spuriously, also then the return value is
103  // std::cv_status::no_timeout, so the queue needs to be checked again
104  mFillStatus.wait(lock);
105  }
106  value = _BASE::front();
107  }
108  bool keepGoing = processor(value);
109  {
110  std::unique_lock<std::mutex> lock(mMutex);
111  _BASE::pop();
112  if (mStop && _BASE::empty()) {
113  keepGoing = false;
114  mStop = false;
115  }
116  }
117  return keepGoing;
118  }
