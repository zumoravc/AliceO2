95  inline float signalForm(float x) const
96  { // table lookup for the signal shape
97  if (x <= 0.0f)
98  return 0.0f;
99  float const y = x / mParameters.bunchWidth * DP::SIGNAL_TABLE_SIZE;
100  int const index = std::floor(y);
101  if (index + 1 >= DP::SIGNAL_TABLE_SIZE) {
102  return mSignalTable.back();
103  }
104  float const rem = y - index;
105  return mSignalTable[index] + rem * (mSignalTable[index + 1] - mSignalTable[index]);
106  }
107 
108  inline Vc::float_v signalFormVc(Vc::float_v x) const
109  { // table lookup for the signal shape (SIMD version)
110  auto const y = x / mParameters.bunchWidth * DP::SIGNAL_TABLE_SIZE;
111  Vc::float_v::IndexType const index = Vc::floor(y);
112  auto const rem = y - index;
113  Vc::float_v val(0);
114  for (size_t i = 0; i < float_v::size(); ++i) {
115  if (y[i] < 0.0f)
116  continue;
117  if (index[i] + 1 < DP::SIGNAL_TABLE_SIZE) {
118  val[i] = mSignalTable[index[i]] + rem[i] * (mSignalTable[index[i] + 1] - mSignalTable[index[i]]);
119  } else {
120  val[i] = mSignalTable.back();
121  }
122  }
123  return val;
124  }

156 // signal shape function
157 template <typename Float>
158 Float signalForm_i(Float x)
159 {
160  using namespace std;
161  Float const a = -0.45458;
162  Float const b = -0.83344945;
163  return x > Float(0) ? -(exp(b * x) - exp(a * x)) / Float(7.8446501) : Float(0);
164  //return -(exp(-0.83344945 * x) - exp(-0.45458 * x)) * (x >= 0) / 7.8446501; // Maximum should be 7.0/250 mV
165 };
166 
167 // integrated signal shape function
168 inline float signalForm_integral(float x)
169 {
170  using namespace std;
171  double const a = -0.45458;
172  double const b = -0.83344945;
173  if (x < 0)
174  x = 0;
175  return -(exp(b * x) / b - exp(a * x) / a) / 7.8446501;
176 };
177 
178 // SIMD version of the integrated signal shape function
179 inline Vc::float_v signalForm_integralVc(Vc::float_v x)
180 {
181  auto const mask = (x >= 0.0f);
182  Vc::float_v arg(0);
183  arg.assign(x, mask); // branchless if
184  Vc::float_v const a(-0.45458f);
185  Vc::float_v const b(-0.83344945f);
186  Vc::float_v result = -(Vc::exp(b * arg) / b - Vc::exp(a * arg) / a) / 7.8446501f;
187  return result;
188 };
