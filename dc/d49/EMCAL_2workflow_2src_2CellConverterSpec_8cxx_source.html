27 void CellConverterSpec::run(framework::ProcessingContext& ctx)
28 {
29  LOG(DEBUG) << "[EMCALCellConverter - run] called";
30  mOutputCells.clear();
31  mOutputTriggers.clear();
32  auto digitsAll = ctx.inputs().get<gsl::span<o2::emcal::Digit>>("digits");
33  auto triggers = ctx.inputs().get<gsl::span<o2::emcal::TriggerRecord>>("triggers");
34  LOG(DEBUG) << "[EMCALCellConverter - run] Received " << digitsAll.size() << " digits from " << triggers.size() << " trigger ...";
35  int currentstart = mOutputCells.size(), ncellsTrigger = 0;
36  for (const auto& trg : triggers) {
37  if (!trg.getNumberOfObjects()) {
38  mOutputTriggers.emplace_back(trg.getBCData(), currentstart, ncellsTrigger);
39  continue;
40  }
41  gsl::span<const o2::emcal::Digit> digits(digitsAll.data() + trg.getFirstEntry(), trg.getNumberOfObjects());
42  for (const auto& dig : digits) {
43  ChannelType_t chantype;
44  if (dig.getHighGain())
45  chantype = ChannelType_t::HIGH_GAIN;
46  else if (dig.getLowGain())
47  chantype = ChannelType_t::LOW_GAIN;
48  else if (dig.getTRU())
49  chantype = ChannelType_t::TRU;
50  else if (dig.getLEDMon())
51  chantype = ChannelType_t::LEDMON;
52  mOutputCells.emplace_back(dig.getTower(), dig.getEnergy(), dig.getTimeStamp(), chantype);
53  ncellsTrigger++;
54  }
55  mOutputTriggers.emplace_back(trg.getBCData(), currentstart, ncellsTrigger);
56  currentstart = mOutputCells.size();
57  ncellsTrigger = 0;
58  }
59  LOG(DEBUG) << "[EMCALCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
60  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
61  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggers);
62  if (mPropagateMC) {
63  // copy mc truth container without modification
64  // as indexing doesn't change
65  auto truthcont = ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("digitsmctr");
66  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe}, *truthcont);
67  }
68 }

70 o2::framework::DataProcessorSpec o2::emcal::reco_workflow::getCellConverterSpec(bool propagateMC)
71 {
72  std::vector<o2::framework::InputSpec> inputs;
73  std::vector<o2::framework::OutputSpec> outputs;
74  inputs.emplace_back("digits", o2::header::gDataOriginEMC, "DIGITS", 0, o2::framework::Lifetime::Timeframe);
75  inputs.emplace_back("triggers", "EMC", "DIGITSTRGR", 0, o2::framework::Lifetime::Timeframe);
76  outputs.emplace_back("EMC", "CELLS", 0, o2::framework::Lifetime::Timeframe);
77  outputs.emplace_back("EMC", "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe);
78  if (propagateMC) {
79  inputs.emplace_back("digitsmctr", "EMC", "DIGITSMCTR", 0, o2::framework::Lifetime::Timeframe);
80  outputs.emplace_back("EMC", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe);
81  }
82  return o2::framework::DataProcessorSpec{"EMCALCellConverterSpec",
83  inputs,
84  outputs,
85  o2::framework::adaptFromTask<o2::emcal::reco_workflow::CellConverterSpec>(propagateMC)};
86 }
