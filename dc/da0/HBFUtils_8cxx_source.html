23 int64_t HBFUtils::getHBF(const IR& rec) const
24 {
26  auto diff = rec.differenceInBC(getFirstIR());
27  if (diff < 0) {
28  LOG(ERROR) << "IR " << rec.bc << '/' << rec.orbit << " is ahead of the reference IR "
29  << bcFirst << '/' << orbitFirst;
30  throw std::runtime_error("Requested IR is ahead of the reference IR");
31  }
32  return diff / o2::constants::lhc::LHCMaxBunches;

36 int HBFUtils::fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const
37 {
38  // Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
39  // BCs between interaction records "fromIR" and "toIR" (inclusive).
40  dst.clear();
41  int hb0 = getHBF(fromIR), hb1 = getHBF(toIR);
42  if (fromIR.bc != bcFirst) { // unless we are just starting the HBF of fromIR, it was already counted
43  hb0++;
44  }
45  for (int ihb = hb0; ihb <= hb1; ihb++) {
46  dst.emplace_back(getIRHBF(ihb));
47  }
48  return dst.size();
49 }
