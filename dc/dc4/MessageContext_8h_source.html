59  // this is the virtual interface for context objects
60  class ContextObject
61  {
62  public:
63  ContextObject() = delete;
64  ContextObject(FairMQMessagePtr&& headerMsg, FairMQMessagePtr&& payloadMsg, const std::string& bindingChannel)
65  : mParts{}, mChannel{bindingChannel}
66  {
67  mParts.AddPart(std::move(headerMsg));
68  mParts.AddPart(std::move(payloadMsg));
69  }
70  ContextObject(FairMQMessagePtr&& headerMsg, const std::string& bindingChannel)
71  : mParts{}, mChannel{bindingChannel}
72  {
73  mParts.AddPart(std::move(headerMsg));
74  }
75  virtual ~ContextObject() = default;
76 
80  virtual FairMQParts finalize()
81  {
82  FairMQParts parts = std::move(mParts);
83  assert(parts.Size() == 2);
84  auto* header = o2::header::get<o2::header::DataHeader*>(parts.At(0)->GetData());
85  if (header == nullptr) {
86  throw std::logic_error("No valid header message found");
87  } else {
88  // o2::header::get returns const pointer, but here we can change the message
89  const_cast<o2::header::DataHeader*>(header)->payloadSize = parts.At(1)->GetSize();
90  }
91  // return value optimization returns by move
92  return parts;
93  }
94 
96  const std::string& channel() const
97  {
98  return mChannel;
99  }
100 
101  bool empty() const
102  {
103  return mParts.Size() == 0;
104  }
105 
106  o2::header::DataHeader const* header()
107  {
108  // we would expect this function to be const but the FairMQParts API does not allow this
109  if (empty() || mParts.At(0) == nullptr) {
110  return nullptr;
111  }
112  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData());
113  }
114 
115  protected:
116  FairMQParts mParts;
117  std::string const& mChannel;
118  };

165  : ContextObject(std::forward<FairMQMessagePtr>(headerMsg), context->getChannelRef(bindingChannel)),
166  // the transport factory
167  mFactory{context->proxy().getTransport(bindingChannel, index)},
168  // the memory resource takes ownership of the message
169  mResource{mFactory ? mFactory->GetMemoryResource() : nullptr},
170  // create the vector with apropriate underlying memory resource for the message
171  mData{std::forward<Args>(args)..., pmr::polymorphic_allocator<value_type>(mResource)}
172  {
173  // FIXME: drop this repeated check and make sure at initial setup of devices that everything is fine
174  // introduce error policy
175  if (mFactory == nullptr) {
176  throw std::runtime_error(std::string("failed to get transport factory for channel ") + bindingChannel);
177  }
178  if (mResource == nullptr) {
179  throw std::runtime_error(std::string("no memory resource for channel ") + bindingChannel);
180  }
181  }
182  ~ContainerRefObject() override = default;

188  assert(mParts.Size() == 1);

401  void schedule(Messages::value_type&& message)
402  {
403  auto const* header = message->header();
404  if (header == nullptr) {
405  throw std::logic_error("No valid header message found");
406  }
407  mScheduledMessages.emplace_back(std::move(message));
408  if (mDispatchControl.dispatch != nullptr) {
409  // send all scheduled messages if there is no trigger callback or its result is true
410  if (mDispatchControl.trigger == nullptr || mDispatchControl.trigger(*header)) {
411  std::unordered_map<std::string const*, FairMQParts> outputs;
412  for (auto& message : mScheduledMessages) {
413  FairMQParts parts = std::move(message->finalize());
414  assert(message->empty());
415  assert(parts.Size() == 2);
416  for (auto& part : parts) {
417  outputs[&(message->channel())].AddPart(std::move(part));
418  }
419  }
420  for (auto& [channel, parts] : outputs) {
421  mDispatchControl.dispatch(std::move(parts), *channel, DefaultChannelIndex);
422  }
423  mScheduledMessages.clear();
424  }
425  }
426  }
427
