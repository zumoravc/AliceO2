37 static void BM_TreeToTable(benchmark::State& state)
38 {
39 
40  // initialize a random generator
41  std::default_random_engine e1(1234567891);
42  std::uniform_real_distribution<double> rd(0, 1);
43  std::normal_distribution<float> rf(5., 2.);
44  std::discrete_distribution<ULong64_t> rl({10, 20, 30, 30, 5, 5});
45  std::discrete_distribution<int> ri({10, 20, 30, 30, 5, 5});
46 
47  // create a table and fill the columns with random numbers
48  TableBuilder builder;
49  auto rowWriter =
50  builder.persist<double, float, ULong64_t, int>({"a", "b", "c", "d"});
51  for (auto i = 0; i < state.range(0); ++i) {
52  rowWriter(0, rd(e1), rf(e1), rl(e1), ri(e1));
53  }
54  auto table = builder.finalize();
55 
56  // now convert the table to a tree
57  TFile fout("tree2table.root", "RECREATE");
58  TableToTree ta2tr(table, &fout, "tree2table");
59  ta2tr.addAllBranches();
60  ta2tr.process();
61  fout.Close();
62 
63  // read tree and convert to table again
64  TFile* f = nullptr;
65  TreeToTable* tr2ta = nullptr;
66  for (auto _ : state) {
67 
68  // Open file and create tree
69  f = new TFile("tree2table.root", "READ");
70  auto tr = (TTree*)f->Get("tree2table");
71 
72  // benchmark TreeToTable
73  if (tr) {
74  tr2ta = new TreeToTable(tr);
75  if (tr2ta->addAllColumns()) {
76  auto ta = tr2ta->process();
77  }
78 
79  } else {
80  LOG(INFO) << "tree is empty!";
81  }
82 
83  // clean up
84  if (tr2ta)
85  delete tr2ta;
86  f->Close();
87  if (f)
88  delete f;
89  }
90 
91  state.SetBytesProcessed(state.iterations() * state.range(0) * 24);
92 }
