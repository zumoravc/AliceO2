43 std::vector<TopoIndexInfo>
44  WorkflowHelpers::topologicalSort(size_t nodeCount,
45  int const* edgeIn,
46  int const* edgeOut,
47  size_t byteStride,
48  size_t edgesCount)
49 {
50  size_t stride = byteStride / sizeof(int);
51  using EdgeIndex = int;
52  // Create the index which will be returned.
53  std::vector<TopoIndexInfo> index(nodeCount);
54  for (int wi = 0; wi < nodeCount; ++wi) {
55  index[wi] = {wi, 0};
56  }
57  std::vector<EdgeIndex> remainingEdgesIndex(edgesCount);
58  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
59  remainingEdgesIndex[ei] = ei;
60  }
61 
62  // Create a vector where at each position we have true
63  // if the vector has dependencies, false otherwise
64  std::vector<bool> nodeDeps(nodeCount, false);
65  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
66  nodeDeps[*(edgeOut + ei * stride)] = true;
67  }
68 
69  // We start with all those which do not have any dependencies
70  // They are layer 0.
71  std::list<TopoIndexInfo> L;
72  for (int ii = 0; ii < index.size(); ++ii) {
73  if (nodeDeps[ii] == false) {
74  L.push_back({ii, 0});
75  }
76  }
77 
78  // The final result.
79  std::vector<TopoIndexInfo> S;
80  // The set of vertices which can be reached by the current node
81  std::set<TopoIndexInfo> nextVertex;
82  // The set of edges which are not related to the current node.
83  std::vector<EdgeIndex> nextEdges;
84  while (!L.empty()) {
85  auto node = L.front();
86  S.push_back(node);
87  L.pop_front();
88  nextVertex.clear();
89  nextEdges.clear();
90 
91  // After this, nextVertex will contain all the vertices
92  // which have the current node as incoming.
93  // nextEdges will contain all the edges which are not related
94  // to the current node.
95  for (auto& ei : remainingEdgesIndex) {
96  if (*(edgeIn + ei * stride) == node.index) {
97  nextVertex.insert({*(edgeOut + ei * stride), node.layer + 1});
98  } else {
99  nextEdges.push_back(ei);
100  }
101  }
102  remainingEdgesIndex.swap(nextEdges);
103 
104  // Of all the vertices which have node as incoming,
105  // check if there is any other incoming node.
106  std::set<TopoIndexInfo> hasPredecessors;
107  for (auto& ei : remainingEdgesIndex) {
108  for (auto& m : nextVertex) {
109  if (m.index == *(edgeOut + ei * stride)) {
110  hasPredecessors.insert({m.index, m.layer});
111  }
112  }
113  }
114  std::vector<TopoIndexInfo> withPredecessor;
115  std::set_difference(nextVertex.begin(), nextVertex.end(),
116  hasPredecessors.begin(), hasPredecessors.end(),
117  std::back_inserter(withPredecessor));
118  std::copy(withPredecessor.begin(), withPredecessor.end(), std::back_inserter(L));
119  }
120  return S;
121 }

123 void addMissingOutputsToReader(std::vector<OutputSpec> const& providedOutputs,
124  std::vector<InputSpec> requestedInputs,
125  DataProcessorSpec& publisher)
126 {
127  auto matchingOutputFor = [](InputSpec const& requested) {
128  return [&requested](OutputSpec const& provided) {
129  return DataSpecUtils::match(requested, provided);
130  };
131  };
132  auto last = std::unique(requestedInputs.begin(), requestedInputs.end());
133  requestedInputs.erase(last, requestedInputs.end());
134  for (InputSpec const& requested : requestedInputs) {
135  auto provided = std::find_if(providedOutputs.begin(),
136  providedOutputs.end(),
137  matchingOutputFor(requested));
138 
139  if (provided != providedOutputs.end()) {
140  continue;
141  }
142  auto concrete = DataSpecUtils::asConcreteDataMatcher(requested);
143  publisher.outputs.emplace_back(OutputSpec{concrete.origin, concrete.description, concrete.subSpec});
144  }

203  std::vector<OutputSpec> providedCCDBs;
204  std::vector<OutputSpec> providedOutputObj;

233  }

235  processor.options.push_back(ConfigParamSpec{"period-" + input.binding, VariantType::Int, 1000, {"period of the timer"}});
236  }

238  } break;

240  auto concrete = DataSpecUtils::asConcreteDataMatcher(input);
241  timer.outputs.emplace_back(OutputSpec{concrete.origin, concrete.description, concrete.subSpec, Lifetime::Enumeration});

243  case Lifetime::Condition: {

252  case Lifetime::Transient:

263  auto& output = processor.outputs[oi];
264  if (DataSpecUtils::partialMatch(output, header::DataOrigin{"AOD"})) {
265  providedAODs.emplace_back(output);

272  } else {
273  it->second.push_back(output.binding.value);

275  }
276  if (output.lifetime == Lifetime::Condition) {
277  providedCCDBs.push_back(output);
278  }
279  }
280  }
281

283  addMissingOutputsToReader(providedCCDBs, requestedCCDBs, ccdbBackend);

286

291  extraSpecs.push_back(transientStore);
292  }

295  }
296  if (aodReader.outputs.empty() == false) {
297  extraSpecs.push_back(timePipeline(aodReader, ctx.options().get<int64_t>("readers")));

303  extraSpecs.push_back(timer);
304  }
305 
306  // This is to inject a file sink so that any dangling ATSK object is written
307  // to a ROOT file.
308  if (providedOutputObj.size() != 0) {
309  auto rootSink = CommonDataProcessors::getOutputObjSink(outObjMap, outTskMap);
310  extraSpecs.push_back(rootSink);
311  }
312 
313  workflow.insert(workflow.end(), extraSpecs.begin(), extraSpecs.end());
314 
319  // First analyze all ouputs
320  // outputtypes = isAOD*2 + isdangling*1 + 0
321  auto [OutputsInputs, outputtypes] = analyzeOutputs(workflow);
322 
323  // file sink for any AOD output
324  extraSpecs.clear();
325 
326  // select outputs of type AOD
327  std::vector<InputSpec> OutputsInputsAOD;
328  std::vector<bool> isdangling;
329  for (int ii = 0; ii < OutputsInputs.size(); ii++) {
330  if ((outputtypes[ii] & 2) == 2) {
331 
332  // temporarily also request to be dangling
333  if ((outputtypes[ii] & 1) == 1) {
334  OutputsInputsAOD.emplace_back(OutputsInputs[ii]);
335  isdangling.emplace_back((outputtypes[ii] & 1) == 1);
336  }
337  }
338  }
339 
340  if (OutputsInputsAOD.size() > 0) {
341  auto fileSink = CommonDataProcessors::getGlobalAODSink(OutputsInputsAOD,
342  isdangling);
343  extraSpecs.push_back(fileSink);
344  }
345  workflow.insert(workflow.end(), extraSpecs.begin(), extraSpecs.end());
346 
347  // file sink for notAOD dangling outputs
348  extraSpecs.clear();
349 
350  // select dangling outputs which are not of type AOD
351  std::vector<InputSpec> OutputsInputsDangling;
352  for (int ii = 0; ii < OutputsInputs.size(); ii++) {
353  if ((outputtypes[ii] & 1) == 1 && (outputtypes[ii] & 2) == 0)
354  OutputsInputsDangling.emplace_back(OutputsInputs[ii]);
355  }
356 
357  std::vector<InputSpec> unmatched;
358  if (OutputsInputsDangling.size() > 0) {
359  auto fileSink = CommonDataProcessors::getGlobalFileSink(OutputsInputsDangling, unmatched);
360  if (unmatched.size() != OutputsInputsDangling.size()) {
361  extraSpecs.push_back(fileSink);
362  }
363  }
364  if (unmatched.size() > 0) {
365  extraSpecs.push_back(CommonDataProcessors::getDummySink(unmatched));
366  }
367  workflow.insert(workflow.end(), extraSpecs.begin(), extraSpecs.end());
368 }

370 void WorkflowHelpers::constructGraph(const WorkflowSpec& workflow,
371  std::vector<DeviceConnectionEdge>& logicalEdges,
372  std::vector<OutputSpec>& outputs,
373  std::vector<LogicalForwardInfo>& forwardedInputsInfo)
374 {
375  assert(!workflow.empty());
376  // This is the state. Oif is the iterator I use for the searches.
377  std::list<LogicalOutputInfo> availableOutputsInfo;
378  auto const& constOutputs = outputs; // const version of the outputs
379  decltype(availableOutputsInfo.begin()) oif;
380  // Forwards is a local cache to avoid adding forwards before time.
381  std::vector<LogicalOutputInfo> forwards;
382 
383  // Notice that availableOutputsInfo MUST be updated first, since it relies on
384  // the size of outputs to be the one before the update.
385  auto enumerateAvailableOutputs = [&workflow, &outputs, &availableOutputsInfo]() {
386  for (size_t wi = 0; wi < workflow.size(); ++wi) {
387  auto& producer = workflow[wi];
388 
389  for (size_t oi = 0; oi < producer.outputs.size(); ++oi) {
390  auto& out = producer.outputs[oi];
391  auto uniqueOutputId = outputs.size();
392  availableOutputsInfo.emplace_back(LogicalOutputInfo{wi, uniqueOutputId, false});
393  outputs.push_back(out);
394  }
395  }
396  };
397 
398  // Notice that if the output is actually a forward, we need to store that
399  // information so that when we add it at device level we know which output
400  // channel we need to connect it too.
401  auto hasMatchingOutputFor = [&workflow, &constOutputs,
402  &availableOutputsInfo, &oif,
403  &forwardedInputsInfo](size_t ci, size_t ii) {
404  assert(ci < workflow.size());
405  assert(ii < workflow[ci].inputs.size());
406  auto& input = workflow[ci].inputs[ii];
407  auto matcher = [&input, &constOutputs](const LogicalOutputInfo& outputInfo) -> bool {
408  auto& output = constOutputs[outputInfo.outputGlobalIndex];
409  return DataSpecUtils::match(input, output);
410  };
411  oif = std::find_if(availableOutputsInfo.begin(),
412  availableOutputsInfo.end(),
413  matcher);
414  if (oif != availableOutputsInfo.end() && oif->forward) {
415  LogicalForwardInfo forward;
416  forward.consumer = ci;
417  forward.inputLocalIndex = ii;
418  forward.outputGlobalIndex = oif->outputGlobalIndex;
419  forwardedInputsInfo.emplace_back(LogicalForwardInfo{ci, ii, oif->outputGlobalIndex});
420  }
421  return oif != availableOutputsInfo.end();
422  };
423 
424  // We have consumed the input, therefore we remove it from the list.
425  // We will insert the forwarded inputs only at the end of the iteration.
426  auto findNextOutputFor = [&availableOutputsInfo, &constOutputs, &oif, &workflow](
427  size_t ci, size_t& ii) {
428  auto& input = workflow[ci].inputs[ii];
429  auto matcher = [&input, &constOutputs](const LogicalOutputInfo& outputInfo) -> bool {
430  auto& output = constOutputs[outputInfo.outputGlobalIndex];
431  return DataSpecUtils::match(input, output);
432  };
433  oif = availableOutputsInfo.erase(oif);
434  oif = std::find_if(oif, availableOutputsInfo.end(), matcher);
435  return oif;
436  };
437 
438  auto numberOfInputsFor = [&workflow](size_t ci) {
439  auto& consumer = workflow[ci];
440  return consumer.inputs.size();
441  };
442 
443  auto maxInputTimeslicesFor = [&workflow](size_t pi) {
444  auto& processor = workflow[pi];
445  return processor.maxInputTimeslices;
446  };
447 
448  // Trivial, but they make reading easier..
449  auto getOutputAssociatedProducer = [&oif]() {
450  return oif->specIndex;
451  };
452 
453  // Trivial, but they make reading easier..
454  auto getAssociateOutput = [&oif]() {
455  return oif->outputGlobalIndex;
456  };
457 
458  auto isForward = [&oif]() {
459  return oif->forward;
460  };
461 
462  // Trivial but makes reasing easier in the outer loop.
463  auto createEdge = [&logicalEdges](size_t producer,
464  size_t consumer,
465  size_t tpi,
466  size_t ptpi,
467  size_t uniqueOutputId,
468  size_t matchingInputInConsumer,
469  bool doForward) {
470  logicalEdges.emplace_back(
471  DeviceConnectionEdge{producer,
472  consumer,
473  tpi,
474  ptpi,
475  uniqueOutputId,
476  matchingInputInConsumer,
477  doForward});
478  };
479 
480  auto errorDueToMissingOutputFor = [&workflow, &constOutputs](size_t ci, size_t ii) {
481  auto input = workflow[ci].inputs[ii];
482  std::ostringstream str;
483  str << "No matching output found for "
484  << DataSpecUtils::describe(input) << ". Candidates:\n";
485 
486  for (auto& output : constOutputs) {
487  str << "-" << DataSpecUtils::describe(output) << "\n";
488  }
489 
490  throw std::runtime_error(str.str());
491  };
492 
493  // Whenever we have a set of forwards, we need to append it
494  // the the global list of outputs, so that they can be matched
495  // and we need to add a ForwardRoute for the current consumer
496  // because it is the one who will actually do the forwarding.
497  auto appendForwardsToPossibleOutputs = [&availableOutputsInfo, &forwards]() {
498  for (auto& forward : forwards) {
499  availableOutputsInfo.push_back(forward);
500  }
501  };
502 
503  // Given we create a forward every time we match and input and an
504  // output, having no forwards means we did not find any matching.
505  auto noMatchingOutputFound = [&forwards]() {
506  return forwards.empty();
507  };
508 
509  // Forwards is basically a cache to record
510  auto newEdgeBetweenDevices = [&forwards]() {
511  forwards.clear();
512  };
513 
514  auto forwardOutputFrom = [&forwards](size_t consumer, size_t uniqueOutputId) {
515  forwards.push_back(LogicalOutputInfo{consumer, uniqueOutputId, true});
516  };
517 
518  // This is the outer loop
519  //
520  // Here we iterate over dataprocessor items in workflow and we consider them
521  // as consumer, since we are interested in their inputs.
522  // Notice also we need to search for all the matching inputs, since
523  // we could have more than one source that matches (e.g. in the
524  // case of a time merger).
525  // Once consumed, an output is not actually used anymore, however
526  // we append it as a forward.
527  // Finally, If a device has n-way pipelining, we need to create one node per
528  // parallel pipeline and add an edge for each.
529  enumerateAvailableOutputs();
530 
531  for (size_t consumer = 0; consumer < workflow.size(); ++consumer) {
532  for (size_t input = 0; input < numberOfInputsFor(consumer); ++input) {
533  newEdgeBetweenDevices();
534 
535  while (hasMatchingOutputFor(consumer, input)) {
536  auto producer = getOutputAssociatedProducer();
537  auto uniqueOutputId = getAssociateOutput();
538  for (size_t tpi = 0; tpi < maxInputTimeslicesFor(consumer); ++tpi) {
539  for (size_t ptpi = 0; ptpi < maxInputTimeslicesFor(producer); ++ptpi) {
540  createEdge(producer, consumer, tpi, ptpi, uniqueOutputId, input, isForward());
541  }
542  forwardOutputFrom(consumer, uniqueOutputId);
543  }
544  findNextOutputFor(consumer, input);
545  }
546  if (noMatchingOutputFound()) {
547  errorDueToMissingOutputFor(consumer, input);
548  }
549  appendForwardsToPossibleOutputs();
550  }
551  }
552 }

554 std::vector<EdgeAction>
555  WorkflowHelpers::computeOutEdgeActions(
556  const std::vector<DeviceConnectionEdge>& edges,
557  const std::vector<size_t>& index)
558 {
559  DeviceConnectionEdge last{ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX};
560 
561  assert(edges.size() == index.size());
562  std::vector<EdgeAction> actions(edges.size(), EdgeAction{false, false});
563  for (size_t i : index) {
564  auto& edge = edges[i];
565  auto& action = actions[i];
566  action.requiresNewDevice = last.producer != edge.producer || last.producerTimeIndex != edge.producerTimeIndex;
567  action.requiresNewChannel = last.consumer != edge.consumer || last.producer != edge.producer || last.timeIndex != edge.timeIndex || last.producerTimeIndex != edge.producerTimeIndex;
568  last = edge;
569  }
570  return actions;
571 }

573 std::vector<EdgeAction>
574  WorkflowHelpers::computeInEdgeActions(
575  const std::vector<DeviceConnectionEdge>& edges,
576  const std::vector<size_t>& index)
577 {
578  DeviceConnectionEdge last{ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX, ULONG_MAX};
579 
580  assert(edges.size() == index.size());
581  std::vector<EdgeAction> actions(edges.size(), EdgeAction{false, false});
582  for (size_t i : index) {
583  auto& edge = edges[i];
584  auto& action = actions[i];
585  // Calculate which actions need to be taken for this edge.
586  action.requiresNewDevice = last.consumer != edge.consumer || last.timeIndex != edge.timeIndex;
587  action.requiresNewChannel =
588  last.consumer != edge.consumer || last.timeIndex != edge.timeIndex || last.producer != edge.producer || last.producerTimeIndex != edge.producerTimeIndex;
589 
590  last = edge;
591  }
592  return actions;
593 }

595 void WorkflowHelpers::sortEdges(std::vector<size_t>& inEdgeIndex,
596  std::vector<size_t>& outEdgeIndex,
597  const std::vector<DeviceConnectionEdge>& edges)
598 {
599  inEdgeIndex.resize(edges.size());
600  outEdgeIndex.resize(edges.size());
601  std::iota(inEdgeIndex.begin(), inEdgeIndex.end(), 0);
602  std::iota(outEdgeIndex.begin(), outEdgeIndex.end(), 0);
603 
604  // Two indexes, one to bind the outputs, the other
605  // one to connect the inputs. The
606  auto outSorter = [&edges](size_t i, size_t j) {
607  auto& a = edges[i];
608  auto& b = edges[j];
609  return std::tie(a.producer, a.producerTimeIndex, a.timeIndex, a.consumer) < std::tie(b.producer, b.producerTimeIndex, b.timeIndex, b.consumer);
610  };
611  auto inSorter = [&edges](size_t i, size_t j) {
612  auto& a = edges[i];
613  auto& b = edges[j];
614  return std::tie(a.consumer, a.timeIndex, a.producer, a.producerTimeIndex) < std::tie(b.consumer, b.timeIndex, b.producer, b.producerTimeIndex);
615  };
616 
617  std::sort(inEdgeIndex.begin(), inEdgeIndex.end(), inSorter);
618  std::sort(outEdgeIndex.begin(), outEdgeIndex.end(), outSorter);
619 }

672 std::tuple<std::vector<InputSpec>, std::vector<unsigned char>> WorkflowHelpers::analyzeOutputs(WorkflowSpec const& workflow)
673 {
674  // compute total number of input/output
675  size_t totalInputs = 0;
676  size_t totalOutputs = 0;
677  for (auto& spec : workflow) {
678  totalInputs += spec.inputs.size();
679  totalOutputs += spec.outputs.size();
680  }
681 
682  std::vector<DataMatcherId> inputs;
683  std::vector<DataMatcherId> outputs;
684  inputs.reserve(totalInputs);
685  outputs.reserve(totalOutputs);
686 
687  std::vector<InputSpec> results;
688  std::vector<unsigned char> outputtypes;
689  results.reserve(totalOutputs);
690  outputtypes.reserve(totalOutputs);
691 
693  for (size_t wi = 0, we = workflow.size(); wi != we; ++wi) {
694  auto& spec = workflow[wi];
695  for (size_t ii = 0, ie = spec.inputs.size(); ii != ie; ++ii) {
696  inputs.emplace_back(DataMatcherId{wi, ii});
697  }
698  for (size_t oi = 0, oe = spec.outputs.size(); oi != oe; ++oi) {
699  outputs.emplace_back(DataMatcherId{wi, oi});
700  }
701  }
702 
703  for (size_t oi = 0, oe = outputs.size(); oi != oe; ++oi) {
704  auto& output = outputs[oi];
705  auto& outputSpec = workflow[output.workflowId].outputs[output.id];
706 
707  // compute output type
708  unsigned char outputtype = 0;
709 
710  // is AOD?
711  if (DataSpecUtils::partialMatch(outputSpec, header::DataOrigin("AOD"))) {
712  outputtype += 2;
713  }
714 
715  // is dangling output?
716  bool matched = false;
717  for (size_t ii = 0, ie = inputs.size(); ii != ie; ++ii) {
718  auto& input = inputs[ii];
719  // Inputs of the same workflow cannot match outputs
720  if (output.workflowId == input.workflowId) {
721  continue;
722  }
723  auto& inputSpec = workflow[input.workflowId].inputs[input.id];
724  if (DataSpecUtils::match(inputSpec, outputSpec)) {
725  matched = true;
726  break;
727  }
728  }
729  if (!matched) {
730  outputtype += 1;
731  }
732 
733  // update results and outputtypes
734  auto input = DataSpecUtils::matchingInput(outputSpec);
735  char buf[64];
736  input.binding = (snprintf(buf, 63, "output_%zu_%zu", output.workflowId, output.id), buf);
737 
738  // make sure that entries are unique
739  if (std::find(results.begin(), results.end(), input) == results.end()) {
740  results.emplace_back(input);
741  outputtypes.emplace_back(outputtype);
742  }
743  }
744 
745  // make sure that results is unique
746 
747  return std::make_tuple(results, outputtypes);
748 }
