186 int GPUTPCGMPolynomialFieldManager::GetPolynomialField(float nominalFieldkG, GPUTPCGMPolynomialField& field)
187 {
188  //
189  // get closest pre-calculated polynomial field approximation of the TPC region for the given field value nominalFieldkG
190  // returns !=0 in case of error
191  //
192  // check which field map is in use
193 
194  field.Reset();
195 
196  StoredField_t type = kUnknown;
197 
198  if (fabsf(fabsf(nominalFieldkG) - 5.00668f) <= fabsf(fabsf(nominalFieldkG) - 2.f)) {
199  type = k5kG;
200  } else {
201  type = k2kG;
202  }
203 
204  return GetPolynomialField(type, nominalFieldkG, field);
205 }

266 int GPUTPCGMPolynomialFieldManager::FitFieldTpc(AliMagF* inputFld, GPUTPCGMPolynomialField& polyField, double step)
267 {
268  //
269  // Fit magnetic field with polynoms
270  //
271 
272  const double kCLight = 0.000299792458;
273  const double kAlmost0Field = 1.e-13;
274 
275  AliMagF* fld = inputFld;
276 
277  if (!fld) {
278  // fld = new AliMagF("Fit", "Fit", 1., 1., AliMagF::k2kG);
279  fld = (AliMagF*)TGeoGlobalMagField::Instance()->GetField();
280  }
281  if (!fld) {
282  return -1;
283  }
284 
285  const double sectorAngleShift = 10. / 180. * TMath::Pi();
286  const double sectorAngle = 20. / 180. * TMath::Pi();
287  const int nRows = AliHLTTPCGeometry::GetNRows();
288 
289  double xMin = AliHLTTPCGeometry::Row2X(0);
290  double xMax = AliHLTTPCGeometry::Row2X(nRows - 1);
291  double rMin = xMin;
292  double rMax = xMax / TMath::Cos(sectorAngle / 2.);
293 
294  double dA = 1. / rMax; // angular step == 1 cm at outer radius
295  dA *= step;
296  int nSectorParticles = (int)(sectorAngle / dA);
297  if (nSectorParticles < 1) {
298  nSectorParticles = 1;
299  }
300  dA = sectorAngle / nSectorParticles;
301 
302  double dZ = 1. * step; // step in z == 1 cm
303 
304  double zMin = -AliHLTTPCGeometry::GetZLength();
305  double zMax = AliHLTTPCGeometry::GetZLength();
306 
307  double alMin = -sectorAngle / 2.;
308  double alMax = sectorAngle / 2. - 0.5 * dA;
309 
310  Double_t solenoidBzkG = fld->SolenoidField();
311  Double_t solenoidBzkGInv = (TMath::Abs(solenoidBzkG) > kAlmost0Field) ? 1. / solenoidBzkG : 0.;
312 
313  std::cout << "solenoidBz = " << solenoidBzkG << " kG" << std::endl;
314 
315  const int M = GPUTPCGMPolynomialField::NTPCM;
316  AliHLTTPCPolynomFit fitBx(M);
317  AliHLTTPCPolynomFit fitBy(M);
318  AliHLTTPCPolynomFit fitBz(M);
319 
320  for (int sector = 0; sector < 18; sector++) {
321  std::cout << "sector = " << sector << std::endl;
322  double asec = sectorAngleShift + sector * sectorAngle;
323  double cs = TMath::Cos(asec);
324  double ss = TMath::Sin(asec);
325  for (double al = alMin; al < alMax; al += dA) {
326  std::cout << "angle " << al / TMath::Pi() * 180. << " grad " << std::endl;
327  double tg = TMath::Tan(al);
328  for (int row = 0; row < AliHLTTPCGeometry::GetNRows(); row++) {
329  double xl = AliHLTTPCGeometry::Row2X(row);
330  double yl = xl * tg;
331  double x = xl * cs - yl * ss;
332  double y = xl * ss + yl * cs;
333  // std::cout<<"sector = "<<sector<<" al = "<<al/TMath::Pi()*180.<<" xl "<<xl<<" yl "<<yl<<std::endl;
334 
335  for (double z = zMin; z <= zMax; z += dZ) { // 1 cm step in Z
336  Double_t xyz[3] = {x, y, z};
337  Double_t B[3] = {0., 0., 0.};
338  if (fld->IsUniform()) {
339  B[0] = B[1] = 0.;
340  B[2] = fld->SolenoidField();
341  } else {
342  fld->Field(xyz, B);
343  }
344  B[0] *= solenoidBzkGInv;
345  B[1] *= solenoidBzkGInv;
346  B[2] *= solenoidBzkGInv;
347 
348  float f[M];
349  GPUTPCGMPolynomialField::GetPolynomsTpc(x, y, z, f);
350  fitBx.AddMeasurement(f, B[0]);
351  fitBy.AddMeasurement(f, B[1]);
352  fitBz.AddMeasurement(f, B[2]);
353  }
354  }
355  }
356  }
357 
358  // field coefficients
359  float cX[M];
360  float cY[M];
361  float cZ[M];
362 
363  int errX = fitBx.Fit(cX);
364  int errY = fitBy.Fit(cY);
365  int errZ = fitBz.Fit(cZ);
366 
367  if (errX != 0 || errY != 0 || errZ != 0) {
368  std::cout << "Fit of polynamial field failed!!!: errX " << errX << " errY " << errY << " errZ " << errZ << std::endl;
369  if (fld != inputFld) {
370  delete fld;
371  }
372  return -1;
373  }
374 
375  GPUTPCGMPolynomialField fittedField;
376  fittedField.SetFieldTpc(cX, cY, cZ);
377 
378  // scale result
379  double nominalBz = solenoidBzkG * kCLight;
380 
381  for (int i = 0; i < M; i++) {
382  cX[i] = nominalBz * cX[i];
383  cY[i] = nominalBz * cY[i];
384  cZ[i] = nominalBz * cZ[i];
385  }
386  polyField.SetFieldNominal(nominalBz);
387  polyField.SetFieldTpc(cX, cY, cZ);
388 
389  gStyle->SetOptStat(1111111);
390 
391  TH1F histBx("Performance B_x", "Error B_x", 1000, -0.005, 0.005);
392  TH1F histBy("Performance B_y", "Error B_y", 1000, -0.005, 0.005);
393  TH1F histBz("Performance B_z", "Error B_z", 1000, -0.005, 0.005);
394 
395  for (int sector = 0; sector < 18; sector++) {
396  std::cout << "check quality: sector = " << sector << std::endl;
397  double asec = sectorAngleShift + sector * sectorAngle;
398  double cs = TMath::Cos(asec);
399  double ss = TMath::Sin(asec);
400  for (double al = alMin; al < alMax; al += dA) {
401  std::cout << "check quality: angle " << al / TMath::Pi() * 180. << " grad " << std::endl;
402  double tg = TMath::Tan(al);
403  for (int row = 0; row < AliHLTTPCGeometry::GetNRows(); row++) {
404  double xl = AliHLTTPCGeometry::Row2X(row);
405  double yl = xl * tg;
406  double x = xl * cs - yl * ss;
407  double y = xl * ss + yl * cs;
408  for (double z = zMin; z <= zMax; z += dZ) {
409  Double_t xyz[3] = {x, y, z};
410  Double_t B[3];
411  if (fld->IsUniform()) {
412  B[0] = B[1] = 0.;
413  B[2] = fld->SolenoidField();
414  } else {
415  fld->Field(xyz, B);
416  }
