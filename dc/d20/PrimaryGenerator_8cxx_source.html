74 Bool_t PrimaryGenerator::GenerateEvent(FairGenericStack* pStack)
75 {
79  if (!mEmbedTree)
80  return FairPrimaryGenerator::GenerateEvent(pStack);
81 
85  mEmbedTree->GetEntry(mEmbedIndex);
86  setInteractionVertex(mEmbedEvent);
87 
89  auto genList = GetListOfGenerators();
90  for (int igen = 0; igen < genList->GetEntries(); ++igen) {
91  auto o2gen = dynamic_cast<Generator*>(genList->At(igen));
92  if (o2gen)
93  o2gen->notifyEmbedding(mEmbedEvent);
94  }
95 
97  if (!FairPrimaryGenerator::GenerateEvent(pStack))
98  return kFALSE;
99 
101  auto o2event = dynamic_cast<MCEventHeader*>(fEvent);
102  if (o2event) {
103  o2event->setEmbeddingFileName(mEmbedFile->GetName());
104  o2event->setEmbeddingEventIndex(mEmbedIndex);
105  }
106 
108  mEmbedIndex++;
109  mEmbedIndex %= mEmbedEntries;
110 
112  return kTRUE;
113 }

117 void PrimaryGenerator::AddTrack(Int_t pdgid, Double_t px, Double_t py, Double_t pz,
118  Double_t vx, Double_t vy, Double_t vz,
119  Int_t parent, Bool_t wanttracking,
120  Double_t e, Double_t tof,
121  Double_t weight, TMCProcess proc)
122 {
126  vx += fVertex.X();
127  vy += fVertex.Y();
128  vz += fVertex.Z();
129 
131  if (wanttracking && !TDatabasePDG::Instance()->GetParticle(pdgid)) {
132  LOG(WARN) << "Particle to be tracked is not defined in PDG: pdg = " << pdgid;
133  wanttracking = false;
134  }
135 
137  Int_t doTracking = 0; // Go to tracking
138  if (fdoTracking && wanttracking) {
139  doTracking = 1;
140  }
141  Int_t dummyparent = -1; // Primary particle (now the value is -1 by default)
142  Double_t polx = 0.; // Polarisation
143  Double_t poly = 0.;
144  Double_t polz = 0.;
145  Int_t ntr = 0; // Track number; to be filled by the stack
146  Int_t status = 0; // Generation status
147 
148  if (parent != -1) {
149  parent += fMCIndexOffset;
150  } // correct for tracks which are in list before generator is called
151 
160  if (abs(pdgid) == 311 && doTracking) {
161  LOG(WARN) << "K0/antiK0 requested for tracking: converting into K0s/K0L";
162  pdgid = gRandom->Uniform() < 0.5 ? 310 : 130;
163  }
164 
166  fStack->PushTrack(doTracking, parent, pdgid, px, py, pz,
167  e, vx, vy, vz, tof, polx, poly, polz, proc, ntr,
168  weight, status, parent);
169 
170  fNTracks++;
171 }

175 void PrimaryGenerator::setInteractionDiamond(const Double_t* xyz, const Double_t* sigmaxyz)
176 {
179  LOG(INFO) << "Setting interaction diamond: position = {"
180  << xyz[0] << "," << xyz[1] << "," << xyz[2] << "} cm";
181  LOG(INFO) << "Setting interaction diamond: width = {"
182  << sigmaxyz[0] << "," << sigmaxyz[1] << "," << sigmaxyz[2] << "} cm";
183  SetBeam(xyz[0], xyz[1], sigmaxyz[0], sigmaxyz[1]);
184  SetTarget(xyz[2], sigmaxyz[2]);
185 
186  auto const& param = InteractionDiamondParam::Instance();
187  SmearVertexXY(false);
188  SmearVertexZ(false);
189  SmearGausVertexXY(false);
190  SmearGausVertexZ(false);
191  if (param.distribution == o2::eventgen::EVertexDistribution::kFlat) {
192  SmearVertexXY(true);
193  SmearVertexZ(true);
194  } else if (param.distribution == o2::eventgen::EVertexDistribution::kGaus) {
195  SmearGausVertexXY(true);
196  SmearGausVertexZ(true);
197  } else {
198  LOG(ERROR) << "PrimaryGenerator: Unsupported vertex distribution";
199  }
200 }

219 Bool_t PrimaryGenerator::embedInto(TString fname)
220 {
224  if (mEmbedFile && mEmbedFile->IsOpen()) {
225  LOG(ERROR) << "Another embedding file is currently open";
226  return kFALSE;
227  }
228 
230  mEmbedFile = TFile::Open(fname);
231  if (!mEmbedFile || !mEmbedFile->IsOpen()) {
232  LOG(ERROR) << "Cannot open file for embedding: " << fname;
233  return kFALSE;
234  }
235 
237  mEmbedTree = (TTree*)mEmbedFile->Get("o2sim");
238  if (!mEmbedTree) {
239  LOG(ERROR) << R"(Cannot find "o2sim" tree for embedding in )" << fname;
240  return kFALSE;
241  }
242 
244  mEmbedEntries = mEmbedTree->GetEntries();
245  if (mEmbedEntries <= 0) {
246  LOG(ERROR) << "Invalid number of entries found in tree for embedding: " << mEmbedEntries;
247  return kFALSE;
248  }
249 
251  mEmbedEvent = new MCEventHeader;
252  mEmbedTree->SetBranchAddress("MCEventHeader.", &mEmbedEvent);
253 
255  return kTRUE;
256 }
