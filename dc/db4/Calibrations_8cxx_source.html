43 void Calibrations::setCCDB(int calibrationobjecttype, long timestamp)
44 {
45  int donothingfornow = 1;
46  auto& ccdbmgr = o2::ccdb::BasicCCDBManager::instance();
47  ccdbmgr.setTimestamp(timestamp); // set which time stamp of data we want this is called per timeframe, and removes the need to call it when querying a value.
48  switch (calibrationobjecttype) {
49  case 1: //simulation
50  mChamberCalibrations = ccdbmgr.get<o2::trd::ChamberCalibrations>("TRD_test/ChamberCalibrations");
51  mLocalVDrift = ccdbmgr.get<o2::trd::LocalVDrift>("TRD_test/LocalVDrift");
52  mLocalT0 = ccdbmgr.get<o2::trd::LocalT0>("TRD_test/LocalT0");
53  mLocalGainFactor = ccdbmgr.get<o2::trd::LocalGainFactor>("TRD_test/LocalGainFactor");
54  mPadNoise = ccdbmgr.get<o2::trd::PadNoise>("TRD_test/PadNoise");
55  mChamberStatus = ccdbmgr.get<o2::trd::ChamberStatus>("TRD_test/ChamberStatus");
56  mPadStatus = ccdbmgr.get<o2::trd::PadStatus>("TRD_test/PadStatus");
57  mChamberNoise = ccdbmgr.get<o2::trd::ChamberNoise>("TRD_test/ChamberNoise");
58  //std::shared_ptr<TrapConfig> mTrapConfig;
59  //std::shared_ptr<PRFWidth> mPRDWidth
60  //std::shared_ptr<OnlineGainFactors> mOnlineGainFactors;
61  break;
62  case 2: //reconstruction
63  donothingfornow = 1;
64  break;
65  case 3: // calibration
66  donothingfornow = 2;
67  break;
68  default:
69  LOG(fatal) << "unknown calibration type coming into setCCDB for TRD Calibrations";
70  }
71 }
