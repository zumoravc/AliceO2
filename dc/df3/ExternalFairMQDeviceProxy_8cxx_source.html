56 void sendOnChannel(FairMQDevice& device, FairMQParts& messages, std::string const& channel)
57 {
58  // Note: DPL is only setting up one instance of a channel while FairMQ allows to have an
59  // array of channels, the index is 0 in the call
60  constexpr auto index = 0;
61  LOG(DEBUG) << "sending " << messages.Size() << " messages on " << channel;
62  // TODO: we can make this configurable
63  const int maxTimeout = 10000;
64  int timeout = 0;
65  // try dispatch with increasing timeout in order to also drop a warning if the dispatching
66  // has been tried multiple times within max timeout
67  // since we do not want any messages to be dropped at this stage, we stay in the loop until
68  // the downstream congestion is resolved
69  // TODO: we might want to treat this error condition some levels higher up, but for
70  // the moment its an appropriate solution. The important thing is not to drop
71  // messages and to be informed about the congestion.
72  while (device.Send(messages, channel, index, timeout) < 0) {
73  if (timeout == 0) {
74  timeout = 1;
75  } else if (timeout < maxTimeout) {
76  timeout *= 10;
77  } else {
78  LOG(ERROR) << "failed to dispatch messages on channel " << channel << ", downstream queue might be full\n"
79  << "or unconnected. No data is dropped, keep on trying, but this will hold the reading from\n"
80  << "the input and expose back-pressure upstream. RESOLVE DOWNSTREAM CONGESTION to continue";
81  if (timeout == maxTimeout) {
82  // we add 1ms to disable the warning below
83  timeout += 1;
84  }
85  }
86  }
87  if (timeout > 0 && timeout <= maxTimeout) {
88  LOG(WARNING) << "dispatching on channel " << channel << " was delayed by " << timeout << " ms";
89  }
90  // TODO: feeling this is a bit awkward, but the interface of FairMQParts does not provide a
91  // method to clear the content.
92  // Maybe the FairMQ API can be improved at some point. Actually the ownership of all messages should be passed
93  // on to the transport and the messages should be empty after sending and the parts content can be cleared.
94  //assert(std::accumulate(messages.begin(), messages.end(), true, [](bool a, auto const& msg) {return a && (msg.get() == nullptr);}));
95  messages.fParts.clear();
96 }

99 {
100  // Note: DPL is only setting up one instance of a channel while FairMQ allows to have an

110 void sendOnChannel(FairMQDevice& device, o2::header::Stack&& headerStack, FairMQMessagePtr&& payloadMessage, OutputSpec const& spec, ChannelRetriever& channelRetriever)
111 {
112  auto channelName = channelRetriever(spec);
113  constexpr auto index = 0;
114  if (channelName.empty()) {
115  LOG(WARNING) << "can not find matching channel for " << DataSpecUtils::describe(spec);
116  return;
117  }
118  for (auto& channelInfo : device.fChannels) {
119  if (channelInfo.first != channelName) {
120  continue;
121  }
122  assert(channelInfo.second.size() == 1);
123  // allocate the header message using the underlying transport of the channel
124  auto channelAlloc = o2::pmr::getTransportAllocator(channelInfo.second[index].Transport());
125  FairMQMessagePtr headerMessage = o2::pmr::getMessage(std::move(headerStack), channelAlloc);
126 
127  FairMQParts out;
128  out.AddPart(std::move(headerMessage));
129  out.AddPart(std::move(payloadMessage));
130  sendOnChannel(device, out, channelName);
131  return;
132  }
133  LOG(ERROR) << "internal mismatch, can not find channel " << channelName << " in the list of channel infos of the device";
134 }

160 std::tuple<std::vector<size_t>, size_t> findSplitParts(FairMQParts& parts, size_t start, std::vector<bool>& indicesDone)
161 {
162  std::optional<ConcreteDataMatcher> matcher = std::nullopt;
163  size_t nofParts = 0;
164  bool isGood = true;
165  std::vector<size_t> indexList;
166  size_t payloadSize = 0;
167  size_t index = start;
168  do {
169  auto dh = o2::header::get<DataHeader*>(parts.At(index * 2)->GetData());
170  if (!dh) {
171  LOG(ERROR) << "data on input " << index << " does not follow the O2 data model, DataHeader missing";
172  break;
173  }
174 
175  if (matcher == std::nullopt) {
176  matcher = ConcreteDataMatcher{dh->dataOrigin, dh->dataDescription, dh->subSpecification};
177  nofParts = dh->splitPayloadParts;
178  } else if (*matcher != ConcreteDataMatcher{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) {
179  continue;
180  }
181  LOG(DEBUG) << "matched " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts;
182  indicesDone[index] = true;
183 
184  if (isGood && (dh->splitPayloadIndex != indexList.size())) {
185  LOG(ERROR) << "invalid sequence of split payload parts, expecting index " << indexList.size() << " but got " << dh->splitPayloadIndex;
186  isGood = false;
187  } else if (isGood && (dh->payloadSize != parts.At(index * 2 + 1)->GetSize())) {
188  LOG(ERROR) << "Mismatch of payload size in DataHeader and size of payload message: " << dh->payloadSize << "/"
189  << parts.At(index * 2 + 1)->GetSize();
190  isGood = false;
191  }
192  if (!isGood) {
193  continue;
194  }
195  indexList.emplace_back(index);
196  payloadSize += dh->payloadSize;
197  LOG(DEBUG) << "processed index " << index << ", cached " << indexList.size() << ", total payload size " << payloadSize;
198  } while ((++index) < parts.Size() / 2);
199 
200  if (indexList.size() != nofParts || !isGood) {
201  payloadSize = 0;
202  indexList.clear();
203  }
204  return std::make_tuple(std::move(indexList), payloadSize);
205 }

207 FairMQMessagePtr mergePayloads(FairMQDevice& device, FairMQParts& parts, std::vector<size_t> const& indexList, size_t payloadSize, std::string const channel)
208 {
209  auto message = device.NewMessageFor(channel, 0, payloadSize);
210  if (!message) {
211  return message;
212  }
213  size_t copied = 0;
214  char* data = reinterpret_cast<char*>(message->GetData());
215  for (auto const& index : indexList) {
216  size_t partSize = parts.At(index * 2 + 1)->GetSize();
217  memcpy(data, parts.At(index * 2 + 1)->GetData(), partSize);
218  data += partSize;
219  copied += partSize;
220  }
221  return message;
222 }

224 InjectorFunction dplModelAdaptor(std::vector<OutputSpec> const& filterSpecs, bool throwOnUnmatchedInputs)
225 {
226  // structure to hald information on the unmatch ed data and print a warning at cleanup
227  class DroppedDataSpecs
228  {
229  public:
230  DroppedDataSpecs() = default;
231  ~DroppedDataSpecs()
232  {
233  warning();
234  }
235 
236  bool find(std::string const& desc) const
237  {
238  return descriptions.find(desc) != std::string::npos;
239  }
240 
241  void add(std::string const& desc)
242  {
243  descriptions += "\n " + desc;
244  }
245 
246  void warning() const
247  {
248  if (not descriptions.empty()) {
249  LOG(WARNING) << "Some input data are not matched by filter rules " << descriptions << "\n"
250  << "DROPPING OF THESE MESSAGES HAS BEEN ENABLED BY CONFIGURATION";
251  }
252  }
253 
254  private:
255  std::string descriptions;
256  };
257 
258  return [filterSpecs = std::move(filterSpecs), throwOnUnmatchedInputs, droppedDataSpecs = std::make_shared<DroppedDataSpecs>()](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
259  std::unordered_map<std::string, FairMQParts> outputs;
260  std::vector<bool> indicesDone(parts.Size() / 2, false);
261  std::vector<std::string> unmatchedDescriptions;
262  for (size_t msgidx = 0; msgidx < parts.Size() / 2; ++msgidx) {
263  if (indicesDone[msgidx]) {
264  continue;
265  }
266 
267  auto dh = o2::header::get<DataHeader*>(parts.At(msgidx * 2)->GetData());
268  if (!dh) {
269  LOG(ERROR) << "data on input " << msgidx << " does not follow the O2 data model, DataHeader missing";
270  continue;
271  }
272  LOG(DEBUG) << msgidx << ": " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts << " payload " << parts.At(msgidx * 2 + 1)->GetSize();
273 
274  OutputSpec query{dh->dataOrigin, dh->dataDescription, dh->subSpecification};
275  LOG(DEBUG) << "processing " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts;
276  for (auto const& spec : filterSpecs) {
277  // filter on the specified OutputSpecs, the default value is a ConcreteDataTypeMatcher with origin and description 'any'
278  if (DataSpecUtils::match(spec, OutputSpec{{header::gDataOriginAny, header::gDataDescriptionAny}}) ||
279  DataSpecUtils::match(spec, query)) {
280  auto channelName = channelRetriever(query);
281  if (channelName.empty()) {
282  LOG(WARNING) << "can not find matching channel, not able to adopt " << DataSpecUtils::describe(query);
283  break;
284  }
285  if (dh->splitPayloadParts > 1 && dh->splitPayloadParts != std::numeric_limits<decltype(dh->splitPayloadParts)>::max()) {
286  auto [indexList, payloadSize] = findSplitParts(parts, msgidx, indicesDone);
287  for (auto const& partidx : indexList) {
288  if (partidx == msgidx) {
289  continue;
290  }
291  auto dh = o2::header::get<DataHeader*>(parts.At(partidx * 2)->GetData());
292  if (dh) {
293  LOG(DEBUG) << partidx << ": " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts << " payload " << parts.At(partidx * 2 + 1)->GetSize();
294  }
295  }
296  if (payloadSize > 0) {
297  size_t headerSize = parts.At(msgidx * 2)->GetSize();
298  auto headerMessage = device.NewMessageFor(channelName, 0, headerSize);
299  memcpy(headerMessage->GetData(), parts.At(msgidx * 2)->GetData(), headerSize);
300  auto clonedh = const_cast<DataHeader*>(o2::header::get<DataHeader*>(headerMessage->GetData()));
301  clonedh->splitPayloadParts = 0;
302  clonedh->splitPayloadIndex = 0;
303  clonedh->payloadSize = payloadSize;
304  outputs[channelName].AddPart(std::move(headerMessage));
305  outputs[channelName].AddPart(std::move(mergePayloads(device, parts, indexList, payloadSize, channelName)));
306  LOG(DEBUG) << "merged " << indexList.size() << " split payload parts in new message of size " << payloadSize << " on channel " << channelName << " (" << outputs[channelName].Size() << " parts)";
307  }
308  } else {
309  outputs[channelName].AddPart(std::move(parts.At(msgidx * 2)));
310  outputs[channelName].AddPart(std::move(parts.At(msgidx * 2 + 1)));
311  LOG(DEBUG) << "associating part with index " << msgidx << " to channel " << channelName << " (" << outputs[channelName].Size() << ")";
312  }
313  // TODO: implement configurable dispatch policy preferably via the workflow spec config
314  sendOnChannel(device, outputs[channelName], channelName);
315  indicesDone[msgidx] = true;
316  break;
317  }
318  }
319  if (indicesDone[msgidx] == false) {
320  unmatchedDescriptions.emplace_back(DataSpecUtils::describe(query));
321  }
322  }
323  for (auto& [channelName, channelParts] : outputs) {
324  if (channelParts.Size() == 0) {
325  continue;
326  }
327  sendOnChannel(device, channelParts, channelName);
328  }
329  if (not unmatchedDescriptions.empty()) {
330  if (throwOnUnmatchedInputs) {
331  std::string descriptions;
332  for (auto const& desc : unmatchedDescriptions) {
333  descriptions += "\n " + desc;
334  }
335  throw std::runtime_error("No matching filter rule for input data " + descriptions +
336  "\n Add appropriate matcher(s) to dataspec definition or allow to drop unmatched data");
337  } else {
338  bool changed = false;
339  for (auto const& desc : unmatchedDescriptions) {
340  if (not droppedDataSpecs->find(desc)) {
341  // a new description
342  droppedDataSpecs->add(desc);
343  changed = true;
344  }
345  }
346  if (changed) {
347  droppedDataSpecs->warning();
348  }
349  }
350  }
351  };
352 }

354 InjectorFunction incrementalConverter(OutputSpec const& spec, uint64_t startTime, uint64_t step)
355 {
356  auto timesliceId = std::make_shared<size_t>(startTime);
357 
358  return [timesliceId, spec, step](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
359  // We iterate on all the parts and we send them two by two,
360  // adding the appropriate O2 header.
361  for (size_t i = 0; i < parts.Size(); ++i) {
362  DataHeader dh;
363 
364  // FIXME: this only supports fully specified output specs...
365  ConcreteDataMatcher matcher = DataSpecUtils::asConcreteDataMatcher(spec);
366  dh.dataOrigin = matcher.origin;
367  dh.dataDescription = matcher.description;
368  dh.subSpecification = matcher.subSpec;
369  dh.payloadSize = parts.At(i)->GetSize();
370 
371  DataProcessingHeader dph{*timesliceId, 0};
372  *timesliceId += step;
373  //we have to move the incoming data
374  o2::header::Stack headerStack{dh, dph};
375 
376  sendOnChannel(device, std::move(headerStack), std::move(parts.At(i)), spec, channelRetriever);
377  }
378  };

381 DataProcessorSpec specifyExternalFairMQDeviceProxy(char const* name,
382  std::vector<OutputSpec> const& outputs,
383  char const* defaultChannelConfig,
384  std::function<void(FairMQDevice&,
385  FairMQParts&,
386  std::function<std::string(OutputSpec const&)>)>
387  converter)
388 {
389  DataProcessorSpec spec;
390  spec.name = strdup(name);
391  spec.inputs = {};
392  spec.outputs = outputs;
393  // The Init method will register a new "Out of band" channel and
394  // attach an OnData to it which is responsible for converting incoming
395  // messages into DPL messages.
396  spec.algorithm = AlgorithmSpec{[converter, channel = spec.name](InitContext& ctx) {
397  auto device = ctx.services().get<RawDeviceService>().device();
398  // make a copy of the output routes and pass to the lambda by move
399  auto outputRoutes = ctx.services().get<RawDeviceService>().spec().outputs;
400  assert(device);
401 
402  // check that the name used for registering the OnData callback corresponds
403  // to the configured output channel, unfortunately we can not automatically
404  // deduce this from list of channels without knowing the name, because there
405  // will be multiple channels. At least we throw a more informative exception.
406  // FairMQDevice calls the custom init before the channels have been configured
407  // so we do the check before starting in a dedicated callback
408  auto channelConfigurationChecker = [channel, device]() {
409  if (device->fChannels.count(channel) == 0) {
410  throw std::runtime_error("the required out-of-band channel '" + channel + "' has not been configured, please check the name in the channel configuration");
411  }
412  };
413  ctx.services().get<CallbackService>().set(CallbackService::Id::Start, channelConfigurationChecker);
414 
415  // Converter should pump messages
416  auto handler = [device, converter, outputRoutes = std::move(outputRoutes)](FairMQParts& inputs, int) {
417  auto channelRetriever = [outputRoutes = std::move(outputRoutes)](OutputSpec const& query) -> std::string {
418  for (auto& route : outputRoutes) {
419  LOG(DEBUG) << "matching: " << DataSpecUtils::describe(query) << " to route " << DataSpecUtils::describe(route.matcher);
420  if (DataSpecUtils::match(route.matcher, query)) {
421  return route.channel;
422  }
423  }
424  return std::string("");
425  };
426  converter(*device, inputs, channelRetriever);
427  return true;
428  };
429 
430  device->OnData(channel, handler);
431  return [](ProcessingContext&) {};
432  }};
433  const char* d = strdup(((std::string(defaultChannelConfig).find("name=") == std::string::npos ? (std::string("name=") + name + ",") : "") + std::string(defaultChannelConfig)).c_str());
434  spec.options = {
435  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}}};
436  return spec;
437 }

439 DataProcessorSpec specifyFairMQDeviceOutputProxy(char const* name,
440  Inputs const& inputSpecs,
441  const char* defaultChannelConfig)
442 {
443  DataProcessorSpec spec;
444  spec.name = name;
445  spec.inputs = inputSpecs;
446  spec.outputs = {};
447  spec.algorithm = adaptStateful([inputSpecs](CallbackService& callbacks, RawDeviceService& rds) {
448  auto device = rds.device();
449  // check that the input spec bindings have corresponding output channels
450  // FairMQDevice calls the custom init before the channels have been configured
451  // so we do the check before starting in a dedicated callback
452  auto channelConfigurationChecker = [inputSpecs = std::move(inputSpecs), device]() {
453  LOG(INFO) << "checking channel configuration";
454  for (auto const& spec : inputSpecs) {
455  if (device->fChannels.count(spec.binding) == 0) {
456  throw std::runtime_error("no corresponding output channel found for input '" + spec.binding + "'");
457  }
458  }
459  };
460  callbacks.set(CallbackService::Id::Start, channelConfigurationChecker);
461  return adaptStateless([](RawDeviceService& rds, InputRecord& inputs) {
462  std::unordered_map<std::string, FairMQParts> outputs;
463  auto& device = *rds.device();
464  for (auto& input : inputs) {
465  // TODO: we need to make a copy of the messages, maybe we can implement functionality in
466  // the RawDeviceService to forward messages, but this also needs to take into account that
467  // other consumers might exist
468  size_t headerMsgSize = o2::header::Stack::headerStackSize(reinterpret_cast<o2::byte const*>(input.header));
469  auto* dh = o2::header::get<DataHeader*>(input.header);
470  if (!dh) {
471  std::stringstream errorMessage;
472  errorMessage << "no data header in " << *input.spec;
473  throw std::runtime_error(errorMessage.str());
474  }
475  size_t payloadMsgSize = dh->payloadSize;
476  // we could probably do something like this but we do not know when the message is going to be sent
477  // and if DPL is still owning a valid copy.
478  //auto headerMessage = device.NewMessageFor(input.spec->binding, input.header, headerMsgSize, [](void*, void*) {});
479 
480  // Note: DPL is only setting up one instance of a channel while FairMQ allows to have an
481  // array of channels, the index is 0 in the call
482  constexpr auto index = 0;
483  auto headerMessage = device.NewMessageFor(input.spec->binding, index, headerMsgSize);
484  memcpy(headerMessage->GetData(), input.header, headerMsgSize);
485  auto payloadMessage = device.NewMessageFor(input.spec->binding, index, payloadMsgSize);
486  memcpy(payloadMessage->GetData(), input.payload, payloadMsgSize);
487  outputs[input.spec->binding].AddPart(std::move(headerMessage));
488  outputs[input.spec->binding].AddPart(std::move(payloadMessage));
489  }
490  for (auto& [channelName, channelParts] : outputs) {
491  if (channelParts.Size() == 0) {
492  continue;
493  }
494  sendOnChannel(device, channelParts, channelName);
495  }
496  });
497  });
498  const char* d = strdup(((std::string(defaultChannelConfig).find("name=") == std::string::npos ? (std::string("name=") + name + ",") : "") + std::string(defaultChannelConfig)).c_str());
499  spec.options = {
500  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}}};
501 
502  return spec;
503 }
