116 template <typename T>
117 Node<T>* buildNode(const std::vector<T>& values, int b, int e)
118 {
119  Interval<T> i{values[b], values[e]};
120  int mid((b + e) / 2);
121  Node<T>* node = new Node<T>(i, values[mid]);
122  if (e - b == 1) {
123  return node;
124  }
125  node->setLeft(buildNode<T>(values, b, mid)).setRight(buildNode<T>(values, mid, e));
126  return node;
127 }

160 template <typename T>
161 void Node<T>::contribution(Interval<T> i, std::vector<o2::mch::contour::impl::Interval<T>>& edgeStack)
162 {
164  if (cardinality()) {
165  return;
166  }
167  if (interval().isFullyContainedIn(i) && !isPotent()) {
168  if (edgeStack.empty()) {
169  edgeStack.push_back(interval());
170  } else {
171  auto& back = edgeStack.back();
172  if (!back.extend(interval())) {
173  // add a new segment if it can not be merged with current one
174  edgeStack.push_back(interval());
175  }
176  }
177  } else {
178  if (goLeft(i)) {
179  left()->contribution(i, edgeStack);
180  }
181  if (goRight(i)) {
182  right()->contribution(i, edgeStack);
183  }
184  }
185 }

202 template <typename T>

220 {

222  decreaseCardinality();

237 template <typename T>
238 void Node<T>::update()
239 {
240  if (left() == nullptr) {
241  potent(false);
242  } else {
243  if (left()->cardinality() > 0 && right()->cardinality() > 0) {
244  promote();
245  }
246  potent(!(!isActive(*left()) && !isActive(*right())));
247  }
248 }

267 template <typename T>
268 std::vector<const Node<T>*> Node<T>::getNodeList() const
269 {

271  return {this};
272  }

274  auto r = right()->getNodeList();

276  return l;
