36 TBranch* getOrMakeBranch(TTree* tree, const char* brname, T* ptr)
37 {
38  if (auto br = tree->GetBranch(brname)) {
39  br->SetAddress(static_cast<void*>(ptr));
40  return br;

55 void TrackWriter::run(ProcessingContext& pc)
56 {
57  auto tracks = pc.inputs().get<const std::vector<o2::its::TrackITS>>("tracks");
58  auto clusIdx = pc.inputs().get<gsl::span<int>>("trackClIdx");
59  auto rofs = pc.inputs().get<const std::vector<o2::itsmft::ROFRecord>>("ROframes");
60  auto vertices = pc.inputs().get<const std::vector<Vertex>>("vertices");
61  auto verticesROF = pc.inputs().get<const std::vector<o2::itsmft::ROFRecord>>("verticesROF");
62 
63  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> labels;
64  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* plabels = nullptr;
65  std::vector<int> clusIdxOut;
66  clusIdxOut.reserve(clusIdx.size());
67  std::copy(clusIdx.begin(), clusIdx.end(), std::back_inserter(clusIdxOut));
68 
69  LOG(INFO) << "ITSTrackWriter pulled " << tracks.size() << " tracks, in " << rofs.size() << " RO frames";
70 
71  auto tracksPtr = &tracks;
72  getOrMakeBranch(mTree.get(), "ITSTrack", &tracksPtr);
73  auto clusIdxOutPtr = &clusIdxOut;
74  getOrMakeBranch(mTree.get(), "ITSTrackClusIdx", &clusIdxOutPtr);
75  auto verticesPtr = &vertices;
76  getOrMakeBranch(mTree.get(), "Vertices", &verticesPtr);
77  auto verticesROFPtr = &verticesROF;
78  getOrMakeBranch(mTree.get(), "VerticesROF", &verticesROFPtr);
79  auto rofsPtr = &rofs;
80  getOrMakeBranch(mTree.get(), "ITSTracksROF", &rofsPtr); // write ROFrecords vector to a tree
81 
82  std::vector<o2::itsmft::MC2ROFRecord> mc2rofs, *mc2rofsPtr = &mc2rofs;
83  if (mUseMC) {
84  labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
85  plabels = labels.get();
86  getOrMakeBranch(mTree.get(), "ITSTrackMCTruth", &plabels);
87 
88  const auto m2rvec = pc.inputs().get<gsl::span<o2::itsmft::MC2ROFRecord>>("MC2ROframes");
89  mc2rofs.reserve(m2rvec.size());
90  std::copy(m2rvec.begin(), m2rvec.end(), std::back_inserter(mc2rofs));
91  getOrMakeBranch(mTree.get(), "ITSTracksMC2ROF", &mc2rofsPtr);
92  }
93  mTree->Fill();
94 }

104 DataProcessorSpec getTrackWriterSpec(bool useMC)
105 {
106  std::vector<InputSpec> inputs;
107  inputs.emplace_back("tracks", "ITS", "TRACKS", 0, Lifetime::Timeframe);
108  inputs.emplace_back("trackClIdx", "ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
109  inputs.emplace_back("ROframes", "ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
110  inputs.emplace_back("vertices", "ITS", "VERTICES", 0, Lifetime::Timeframe);
111  inputs.emplace_back("verticesROF", "ITS", "VERTICESROF", 0, Lifetime::Timeframe);
112  if (useMC) {
113  inputs.emplace_back("labels", "ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
114  inputs.emplace_back("MC2ROframes", "ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
115  }
116 
117  return DataProcessorSpec{
118  "its-track-writer",
119  inputs,
120  Outputs{},
121  AlgorithmSpec{adaptFromTask<TrackWriter>(useMC)},
122  Options{
123  {"its-track-outfile", VariantType::String, "o2trac_its.root", {"Name of the output file"}}}};
124 }
