198 void CDBInterface::createDefaultPedestals()
199 {
200  // ===| create random pedestals |=============================================
201  mPedestals = std::make_unique<CalPad>("Pedestals");
202 
203  // distribution based on test beam data
204  const float meanPedestal = 72.5;
205  const float sigmaPedestal = 9.0;
206 
207  // set a minimum and maximum for the value
208  const float minPedestal = meanPedestal - 4 * sigmaPedestal;
209  const float maxPedestal = meanPedestal + 4 * sigmaPedestal;
210 
211  for (auto& calArray : mPedestals->getData()) {
212  for (auto& val : calArray.getData()) {
213  float random = gRandom->Gaus(meanPedestal, sigmaPedestal);
214  if (random < minPedestal) {
215  random = minPedestal;
216  }
217  if (random > maxPedestal) {
218  random = maxPedestal;
219  }
220  val = random;
221  }
222  }
223 }

282 bool CDBStorage::checkMetaData(MetaData_t metaData) const
283 {
284  if (!metaData.size()) {
285  LOGP(error, "no meta data set");
286  }
287 
288  const std::array<std::string_view, 3> requirements{
289  "Required",
290  "Recommended",
291  "Optional"};
292 
293  const std::array<std::vector<std::string_view>, 3> tests{{
294  {"Responsible", "Reason", "Intervention"}, //errors
295  {"JIRA"}, //warnings
296  {"Comment"} //infos
297  }};
298 
299  std::array<int, 3> counts{};
300 
301  for (size_t i = 0; i < requirements.size(); ++i) {
302  for (const auto& test : tests[i]) {
303  if (!metaData[test.data()].size()) {
304  const auto message = fmt::format("{} field {} not set in meta data", requirements[i], test);
305  if (i == 0) {
306  LOG(error) << message;
307  } else if (i == 1) {
308  LOG(warning) << message;
309  } else {
310  LOG(info) << message;
311  }
312  ++counts[i];
313  } else {
314  LOGP(debug, "{} field '{}' set to '{}'", requirements[i], test, metaData[test.data()]);
315  }
316  }
317  }
318 
319  return counts[0] == 0;
320 }

368  }
369 
370  std::unordered_map<std::string, CalDet<float>> pulserCalib;
371  pulserCalib["T0"] = *t0;
372  pulserCalib["Width"] = *width;
373  pulserCalib["Qtot"] = *qtot;
374 
375  storeObject(&pulserCalib, CDBType::CalPulser, first, last);
