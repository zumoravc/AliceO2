26 /*
27  * AlpideRespSimMat : class to access the response: probability to collect electron
28  * in MNPix*MNPix cells.
29  */
30 class AlpideRespSimMat
31 {
32  public:
33  static int constexpr NPix = 5;
34  static int constexpr MatSize = NPix * NPix;
35  static int constexpr getNPix() { return NPix; }
36 
37  AlpideRespSimMat() = default;
38  ~AlpideRespSimMat() = default;
39 
40  void adopt(const AlpideRespSimMat& src, bool flipRow = false, bool flipCol = false)
41  {
42  // copy constructor with option of channels flipping
43  for (int iRow = NPix; iRow--;) {
44  int rw = flipRow ? NPix - 1 - iRow : iRow;
45  for (int iCol = NPix; iCol--;) {
46  int bDest = rw * NPix + (flipCol ? NPix - 1 - iCol : iCol);
47  data[bDest] = src.data[iRow * NPix + iCol];
48  }
49  }
50  }
51 
53  float getValue(int iRow, int iCol) const { return data[iRow * NPix + iCol]; }
54  float getValue(int iRow, int iCol, bool flipRow, bool flipCol) const
55  {
56  int bin = (flipRow ? NPix - 1 - iRow : iRow) * NPix + (flipCol ? NPix - 1 - iCol : iCol);
57  return data[bin];
58  }
59 
61  std::array<float, MatSize>* getArray() { return &data; }
62 
64  void print(bool flipRow = false, bool flipCol = false) const;
65 
66  private:
67  std::array<float, MatSize> data;
68 
69  ClassDefNV(AlpideRespSimMat, 1);
70 };

146 {

154 {

158 }

160 //-----------------------------------------------------
161 inline int AlpideSimResponse::getDepthBin(float pos) const
162 {
165  int i = (mDptMax - pos) * mStepInvDpt;
