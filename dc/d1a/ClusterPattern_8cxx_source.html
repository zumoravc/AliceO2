102 int ClusterPattern::getCOG(int rowSpan, int colSpan, const unsigned char patt[Cluster::kMaxPatternBytes], float& xCOG, float& zCOG)
103 {
104  int tempxCOG = 0, tempzCOG = 0, tempFiredPixels = 0, ic = 0, ir = 0;
105  int nBits = rowSpan * colSpan;
106  int nBytes = nBits / 8;
107  if (nBits % 8 != 0) {
108  nBytes++;
109  }
110  for (unsigned int i = 0; i < nBytes; i++) {
111  unsigned char tempChar = patt[i];
112  int s = 128; // 0b10000000
113  while (s > 0) {
114  if ((tempChar & s) != 0) {
115  tempFiredPixels++;
116  tempxCOG += ir;
117  tempzCOG += ic;
118  }
119  ic++;
120  s /= 2;
121  if ((ir + 1) * ic == nBits) {
122  break;
123  }
124  if (ic == colSpan) {
125  ic = 0;
126  ir++;
127  }
128  }
129  if ((ir + 1) * ic == nBits) {
130  break;
131  }
132  }
133  xCOG = float(tempxCOG) / tempFiredPixels;
134  zCOG = float(tempzCOG) / tempFiredPixels;
135 
136  return tempFiredPixels;
137 }
