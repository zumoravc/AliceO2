167 template <typename T, typename... Targs>
168 constexpr T String2(char c, Targs... Fargs)
169 {
170  // number of arguments has either to match the size of the type, or the last element is treated
171  // as '0' if missing
172  // TODO: this is a bit of a question, do we allow the number of characters to
173  // be smaller than the size of the type and just pad with 0 like in the case
174  // of a char array argument?
175  static_assert(sizeof...(Targs) == sizeof(T) + 1 ||
176  sizeof...(Targs) == sizeof(T),
177  "number of arguments does not match the uint type width");
178  return internal::String2__<T>(c, Fargs...);
179 }

185 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
186 constexpr T String2(const char (&str)[N])
187 {
188  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
189  static_assert(N >= pos, "Position is larger than the length of string");
190  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
191  "String parameter is longer than the size of the data type");
192  // clang-format off
193  return ((T)str[0 + pos] |
194  (str[0 + pos] && sizeof(T) >= 2 ? ((T)str[1 + pos] << (sizeof(T) >= 2 ? 8 : 0) |
195  (str[1 + pos] && sizeof(T) >= 4 ? ((T)str[2 + pos] << (sizeof(T) >= 4 ? 16 : 0) |
196  (str[2 + pos] && sizeof(T) >= 4 ? ((T)str[3 + pos] << (sizeof(T) >= 4 ? 24 : 0) |
197  (str[3 + pos] && sizeof(T) >= 8 ? ((T)str[4 + pos] << (sizeof(T) >= 8 ? 32 : 0) |
198  (str[4 + pos] && sizeof(T) >= 8 ? ((T)str[5 + pos] << (sizeof(T) >= 8 ? 40 : 0) |
199  (str[5 + pos] && sizeof(T) >= 8 ? ((T)str[6 + pos] << (sizeof(T) >= 8 ? 48 : 0) |
200  (str[6 + pos] && sizeof(T) >= 8 ? ((T)str[7 + pos] << (sizeof(T) >= 8 ? 56 : 0))
201  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
202  // clang-format on
203 }

210 struct DescriptorCompareTraits {
211  template <typename T, typename Length>
212  static bool compare(const T& lh, const T& rh, Length N)
213  {
214  return std::memcmp(lh.str, rh.str, N) == 0;
215  }
216  template <typename T, typename Length>
217  static bool lessThen(const T& lh, const T& rh, Length N)
218  {
219  return std::memcmp(lh.str, rh.str, N) < 0;
220  }
221 };
222 template <>
223 struct DescriptorCompareTraits<1> {
224  template <typename T, typename Length>
225  static bool compare(const T& lh, const T& rh, Length)
226  {
227  return lh.itg[0] == rh.itg[0];
228  }
229  template <typename T, typename Length>
230  static bool lessThen(const T& lh, const T& rh, Length)
231  {
232  return lh.itg[0] < rh.itg[0];
233  }
234 };
235 template <>
236 struct DescriptorCompareTraits<2> {
237  template <typename T, typename Length>
238  static bool compare(const T& lh, const T& rh, Length)
239  {
240  return (lh.itg[0] == rh.itg[0]) && (lh.itg[1] == rh.itg[1]);
241  }
242  template <typename T, typename Length>
243  static bool lessThen(const T& lh, const T& rh, Length)
244  {
245  return std::tie(lh.itg[0], lh.itg[1]) < std::tie(rh.itg[0], rh.itg[1]);
246  }

257 struct Descriptor {
258  static_assert(internal::NumberOfActiveBits<N>::value == 1,
259  "Descriptor size is required to be a power of 2");

264  using ItgType = typename internal::TraitsIntType<N>::Type;

269  };

273  Descriptor& operator=(const Descriptor& other) = default;

276  // can use Descriptor(ItgType initializer) for conversion

304  void runtimeInit(const char* string, short length = -1)
305  {
306  char* target = str;
307  char* targetEnd = target;
308  if (length >= 0 && length < (int)N)
309  targetEnd += length;
310  else
311  targetEnd += N;
312  const char* source = string;
313  for (; source != nullptr && target < targetEnd && *source != 0; ++target, ++source)
314  *target = *source;
315  targetEnd = str + N;
316  for (; target < targetEnd; ++target)
317  *target = 0;
318  // require the string to be not longer than the descriptor size
319  if (source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N))) {
320  } else {
321  throw std::invalid_argument("argument must not be longer than descriptor size");
322  }
323  }
324 
325  bool operator==(const Descriptor& other) const { return DescriptorCompareTraits<arraySize>::compare(*this, other, N); }
326  bool operator<(const Descriptor& other) const { return DescriptorCompareTraits<arraySize>::lessThen(*this, other, N); }
327  bool operator!=(const Descriptor& other) const { return not this->operator==(other); }

338  template <typename T>
339  std::enable_if_t<std::is_same<T, std::string>::value == true, T> as() const
340  {
341  // backward search to find first non-zero char
342  // FIXME: can optimize this by using the int value to start at e.g. size/2
343  // if the upper part of the string is just zeros.
344  size_t len = size;
345  while (len > 1 && str[len - 1] == 0) {
346  --len;
347  }

349  return std::move(ret);
350  }

352  void print() const

354  // eventually terminate string before printing
355  PrinterPolicy printer;
356  printer(str);
357  };

496 template <typename HeaderType, typename std::enable_if_t<std::is_pointer<HeaderType>::value, int> = 0>
497 auto get(const o2::byte* buffer, size_t /*len*/ = 0)
498 {
499  using HeaderConstPtrType = const typename std::remove_pointer<HeaderType>::type*;
500  using HeaderValueType = typename std::remove_pointer<HeaderType>::type;
501 
502  const BaseHeader* current = BaseHeader::get(buffer);
503  if (!current) {
504  return HeaderConstPtrType{nullptr};
505  }
506  if (current->description == HeaderValueType::sHeaderType) {
507  // FIXME: We need to think how to handle multiple versions.
508  // For the moment we require the version to exactly match the expected version, which
509  // is part of the header definition. The check function 'sanityCheck' will throw
510  // otherwise, we keep the code related to the exception outside the header file.
511  // Note: Can not check on size because the O2 data model requires variable size headers
512  // to be supported.
513  if (current->sanityCheck(HeaderValueType::sVersion)) {
514  return reinterpret_cast<HeaderConstPtrType>(current);
515  }
516  }
517  auto* prev = current;
518  while ((current = current->next())) {
519  prev = current;
520  if (current->description == HeaderValueType::sHeaderType) {
521  if (current->sanityCheck(HeaderValueType::sVersion)) {
522  return reinterpret_cast<HeaderConstPtrType>(current);
523  }
524  }
525  }
526  // if a next header was expected but not found, the header stack is inconsistent
527  // either the flag is wrongly set, or the size of the current header is wrong so
528  // that the BaseHeader information is not found at the expected offset.
529  if (prev->flagsNextHeader) {
530  prev->throwInconsistentStackError();
531  }
532  return HeaderConstPtrType{nullptr};
533 }

542 // utilities for converting strings to ints - tries to mimick the stadard API, but the result is
543 // strongly typed - can be used for fixed width and short unsigned integer types unlike the standard
544 // utilities.
545 template <typename T>
546 T strtoui(const char* str, char** str_end, int base) noexcept
547 {
548  static_assert(std::numeric_limits<T>::is_integer && !std::numeric_limits<T>::is_signed,
549  "Target must be an unsigned integer type");
550  // standard lib gives us 2 specializations of strtouX, so let's use them both where appropriate
551  if constexpr (sizeof(T) <= sizeof(uint32_t)) {
552  unsigned long res = strtoul(str, str_end, base);
553  if (res > std::numeric_limits<T>::max()) {
554  errno = ERANGE;
555  return std::numeric_limits<T>::max();
556  } else {
557  return static_cast<T>(res);
558  };
559  } else {
560  unsigned long long res = strtoull(str, str_end, base);
561  if (res > std::numeric_limits<T>::max()) {
562  errno = ERANGE;
563  return std::numeric_limits<T>::max();
564  } else {
565  return static_cast<T>(res);
566  };
567  }
568 };

570 template <typename T>
571 T stoui(const std::string& str, size_t* pos = nullptr, int base = 10)
572 {
573  static_assert(std::numeric_limits<T>::is_integer && !std::numeric_limits<T>::is_signed,
574  "Target must be an unsigned integer type");
575  // standard lib gives us 2 specializations of stouX, so let's use them both where appropriate
576  if constexpr (sizeof(T) <= sizeof(uint32_t)) {
577  unsigned long res = std::stoul(str, pos, base);
578  if (res > std::numeric_limits<T>::max()) {
579  throw std::out_of_range("result does not fit in target type");
580  } else {
581  return static_cast<T>(res);
582  };
583  } else {
584  unsigned long long res = std::stoull(str, pos, base);
585  if (res > std::numeric_limits<T>::max()) {
586  throw std::out_of_range("result does not fit in target type");
587  } else {
588  return static_cast<T>(res);
589  };
590  }
