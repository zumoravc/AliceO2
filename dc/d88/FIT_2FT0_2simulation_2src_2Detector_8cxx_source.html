80 void Detector::ConstructGeometry()
81 {
82  LOG(DEBUG) << "Creating FT0 geometry\n";
83  CreateMaterials();
84 
85  Float_t zdetA = Geometry::ZdetA;
86  Float_t zdetC = Geometry::ZdetC;
87 
88  Int_t idrotm[999];
89  Double_t x, y, z;
90  Float_t pstartC[3] = {20., 20, 5};
91  Float_t pstartA[3] = {20, 20, 5};
92  Float_t pinstart[3] = {2.9491, 2.9491, 2.5};
93  Float_t pmcp[3] = {2.949, 2.949, 1.}; // MCP
94 
95  int nCellsA = Geometry::NCellsA;
96  int nCellsC = Geometry::NCellsC;
97 
98  Geometry geometry;
99  TVector3 centerMCP = geometry.centerMCP(2);
100  Matrix(idrotm[901], 90., 0., 90., 90., 180., 0.);
101 
102  // C side Concave Geometry
103 
104  Double_t crad = Geometry::ZdetC; // define concave c-side radius here
105 
106  Double_t dP = pmcp[0]; // side length of mcp divided by 2
107 
108  // uniform angle between detector faces==
109  Double_t btta = 2 * TMath::ATan(dP / crad);
110 
111  // get noncompensated translation data
112  Double_t grdin[6] = {-3, -2, -1, 1, 2, 3};
113  Double_t gridpoints[6];
114  for (Int_t i = 0; i < 6; i++) {
115  gridpoints[i] = crad * TMath::Sin((1 - 1 / (2 * TMath::Abs(grdin[i]))) * grdin[i] * btta);
116  }
117 
118  Double_t xi[Geometry::NCellsC] = {gridpoints[1], gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[0],
119  gridpoints[1], gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[5],
120  gridpoints[0], gridpoints[1], gridpoints[4], gridpoints[5], gridpoints[0],
121  gridpoints[1], gridpoints[4], gridpoints[5], gridpoints[0], gridpoints[1],
122  gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[5], gridpoints[1],
123  gridpoints[2], gridpoints[3], gridpoints[4]};
124  Double_t yi[Geometry::NCellsC] = {gridpoints[5], gridpoints[5], gridpoints[5], gridpoints[5], gridpoints[4],
125  gridpoints[4], gridpoints[4], gridpoints[4], gridpoints[4], gridpoints[4],
126  gridpoints[3], gridpoints[3], gridpoints[3], gridpoints[3], gridpoints[2],
127  gridpoints[2], gridpoints[2], gridpoints[2], gridpoints[1], gridpoints[1],
128  gridpoints[1], gridpoints[1], gridpoints[1], gridpoints[1], gridpoints[0],
129  gridpoints[0], gridpoints[0], gridpoints[0]};
130  Double_t zi[Geometry::NCellsC];
131  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
132  zi[i] = TMath::Sqrt(TMath::Power(crad, 2) - TMath::Power(xi[i], 2) - TMath::Power(yi[i], 2));
133  }
134 
135  // get rotation data
136  Double_t ac[Geometry::NCellsC], bc[Geometry::NCellsC], gc[Geometry::NCellsC];
137  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
138  ac[i] = TMath::ATan(yi[i] / xi[i]) - TMath::Pi() / 2 + 2 * TMath::Pi();
139  if (xi[i] < 0) {
140  bc[i] = TMath::ACos(zi[i] / crad);
141  } else {
142  bc[i] = -1 * TMath::ACos(zi[i] / crad);
143  }
144  }
145  Double_t xc2[Geometry::NCellsC], yc2[Geometry::NCellsC], zc2[Geometry::NCellsC];
146 
147  // compensation based on node position within individual detector geometries
148  // determine compensated radius
149  Double_t rcomp = crad + pstartC[2] / 2.0; //
150  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
151  // Get compensated translation data
152  xc2[i] = rcomp * TMath::Cos(ac[i] + TMath::Pi() / 2) * TMath::Sin(-1 * bc[i]);
153  yc2[i] = rcomp * TMath::Sin(ac[i] + TMath::Pi() / 2) * TMath::Sin(-1 * bc[i]);
154  zc2[i] = rcomp * TMath::Cos(bc[i]);
155 
156  // Convert angles to degrees
157  ac[i] *= 180 / TMath::Pi();
158  bc[i] *= 180 / TMath::Pi();
159  gc[i] = -1 * ac[i];
160  }
161  // A Side
162 
163  Float_t xa[Geometry::NCellsA] = {-11.8, -5.9, 0, 5.9, 11.8, -11.8, -5.9, 0, 5.9, 11.8, -12.8, -6.9,
164  6.9, 12.8, -11.8, -5.9, 0, 5.9, 11.8, -11.8, -5.9, 0, 5.9, 11.8};
165 
166  Float_t ya[Geometry::NCellsA] = {11.9, 11.9, 12.9, 11.9, 11.9, 6.0, 6.0, 7.0, 6.0, 6.0, -0.1, -0.1,
167  0.1, 0.1, -6.0, -6.0, -7.0, -6.0, -6.0, -11.9, -11.9, -12.9, -11.9, -11.9};
168 
169  TGeoVolumeAssembly* stlinA = new TGeoVolumeAssembly("0STL"); // A side mother
170  TGeoVolumeAssembly* stlinC = new TGeoVolumeAssembly("0STR"); // C side mother
171 
172  // FIT interior
173  TVirtualMC::GetMC()->Gsvolu("0INS", "BOX", getMediumID(kAir), pinstart, 3);
174  TGeoVolume* ins = gGeoManager->GetVolume("0INS");
175  //
176  TGeoTranslation* tr[Geometry::NCellsA + Geometry::NCellsC];
177  TString nameTr;
178 
179  // A side Translations
180  for (Int_t itr = 0; itr < Geometry::NCellsA; itr++) {
181  nameTr = Form("0TR%i", itr + 1);
182  z = -pstartA[2] + pinstart[2];
183  tr[itr] = new TGeoTranslation(nameTr.Data(), xa[itr], ya[itr], z);
184  tr[itr]->RegisterYourself();
185  stlinA->AddNode(ins, itr, tr[itr]);
186  }
187 
188  TGeoRotation* rot[Geometry::NCellsC];
189  TString nameRot;
190 
191  TGeoCombiTrans* com[Geometry::NCellsC];
192  TString nameCom;
193 
194  // C Side Transformations
195  for (Int_t itr = Geometry::NCellsA; itr < Geometry::NCellsA + Geometry::NCellsC; itr++) {
196  nameTr = Form("0TR%i", itr + 1);
197  nameRot = Form("0Rot%i", itr + 1);
198  int ic = itr - Geometry::NCellsA;
199  // nameCom = Form("0Com%i",itr+1);
200  rot[ic] = new TGeoRotation(nameRot.Data(), ac[ic], bc[ic], gc[ic]);
201  rot[ic]->RegisterYourself();
202 
203  tr[itr] = new TGeoTranslation(nameTr.Data(), xc2[ic], yc2[ic], (zc2[ic] - 80.));
204  tr[itr]->RegisterYourself();
205 
206  // com[itr-Geometry::NCellsA] = new TGeoCombiTrans(tr[itr],rot[itr-Geometry::NCellsA]);
207  com[ic] = new TGeoCombiTrans(xc2[ic], yc2[ic], (zc2[ic] - 80), rot[ic]);
208  std::cout << ic << " " << xc2[ic] << " " << yc2[ic] << std::endl;
209  TGeoHMatrix hm = *com[ic];
210  TGeoHMatrix* ph = new TGeoHMatrix(hm);
211  stlinC->AddNode(ins, itr, ph);
212  }
213 
214  TGeoVolume* alice = gGeoManager->GetVolume("cave");
215  alice->AddNode(stlinA, 1, new TGeoTranslation(0, 0, zdetA));
216  TGeoRotation* rotC = new TGeoRotation("rotC", 90., 0., 90., 90., 180., 0.);
217  alice->AddNode(stlinC, 1, new TGeoCombiTrans(0., 0., -zdetC, rotC));
218 
219  // MCP + 4 x wrapped radiator + 4xphotocathod + MCP + Al top in front of radiators
220  SetOneMCP(ins);
221 }

231 void Detector::SetOneMCP(TGeoVolume* ins)
232 {
233 
234  Double_t x, y, z;
235 
236  Float_t pinstart[3] = {2.9491, 2.9491, 2.5};
237  Float_t ptop[3] = {1.324, 1.324, 1.}; // Cherenkov radiator
238  // Float_t ptopblack[3] = {1.324, 1.324, 0.0002}; // black paper on the top on radiator
239  Float_t ptopref[3] = {1.3241, 1.3241, 1.}; // Cherenkov radiator wrapped with reflector
240  Double_t prfv[3] = {0.0002, 1.323, 1.}; // Vertical refracting layer bettwen radiators and between radiator and not optical Air
241  Double_t prfh[3] = {1.323, 0.0002, 1.}; // Horizontal refracting layer bettwen radiators and ...
242  Float_t pmcp[3] = {2.949, 2.949, 1.}; // MCP
243  Float_t pmcpinner[3] = {2.749, 2.749, 0.1};
244  Float_t pmcpside[3] = {0.1, 2.949, 1};
245  Float_t pmcpbase[3] = {2.949, 2.949, 0.1};
246  Float_t pmcptopglass[3] = {2.949, 2.949, 0.1}; // MCP top glass optical
247 
248  Float_t preg[3] = {1.324, 1.324, 0.005}; // Photcathode
249  Double_t pal[3] = {2.648, 2.648, 0.25}; // 5mm Al on top of each radiator
250  // Entry window (glass)
251  TVirtualMC::GetMC()->Gsvolu("0TOP", "BOX", getMediumID(kOpGlass), ptop, 3); // Glass radiator
252  TGeoVolume* top = gGeoManager->GetVolume("0TOP");
253  // TVirtualMC::GetMC()->Gsvolu("0TBL", "BOX", getMediumID(kOptBlack), ptopblack, 3); // Glass radiator
254  // TGeoVolume* topblack = gGeoManager->GetVolume("0TBL");
255  TVirtualMC::GetMC()->Gsvolu("0TRE", "BOX", getMediumID(kAir), ptopref, 3); // Air: wrapped radiator
256  TGeoVolume* topref = gGeoManager->GetVolume("0TRE");
257  TVirtualMC::GetMC()->Gsvolu("0RFV", "BOX", getMediumID(kOptAl), prfv, 3); // Optical Air vertical
258  TGeoVolume* rfv = gGeoManager->GetVolume("0RFV");
259  TVirtualMC::GetMC()->Gsvolu("0RFH", "BOX", getMediumID(kOptAl), prfh, 3); // Optical Air horizontal
260  TGeoVolume* rfh = gGeoManager->GetVolume("0RFH");
261 
262  TVirtualMC::GetMC()->Gsvolu("0PAL", "BOX", getMediumID(kAl), pal, 3); // 5mm Al on top of the radiator
263  TGeoVolume* altop = gGeoManager->GetVolume("0PAL");
264 
265  TVirtualMC::GetMC()->Gsvolu("0REG", "BOX", getMediumID(kOpGlassCathode), preg, 3);
266  TGeoVolume* cat = gGeoManager->GetVolume("0REG");
267 
268  //wrapped radiator + reflecting layers
269 
270  Int_t ntops = 0, nrfvs = 0, nrfhs = 0;
271  // Float_t yin = 0, xinv = 0, yinv = 0;
272  x = y = z = 0;
273  topref->AddNode(top, 1, new TGeoTranslation(0, 0, 0));
274  float xinv = -ptop[0] - prfv[0];
275  topref->AddNode(rfv, 1, new TGeoTranslation(xinv, 0, 0));
276  printf(" GEOGEO refv %f , 0,0 \n", xinv);
277  xinv = ptop[0] + prfv[0];
278  topref->AddNode(rfv, 2, new TGeoTranslation(xinv, 0, 0));
279  printf(" GEOGEO refv %f , 0,0 \n", xinv);
280  float yinv = -ptop[1] - prfh[1];
281  topref->AddNode(rfh, 1, new TGeoTranslation(0, yinv, 0));
282  printf(" GEOGEO refh , 0, %f, 0 \n", yinv);
283  yinv = ptop[1] + prfh[1];
284  topref->AddNode(rfh, 2, new TGeoTranslation(0, yinv, 0));
285  // zin = -ptop[2] - ptopblack[2];
286  // printf(" GEOGEO refh , 0, 0, %f \n", zin);
287  // topref->AddNode(topblack, 1, new TGeoTranslation(0, 0, zin));
288 
289  //container for radiator, cathode
290  for (Int_t ix = 0; ix < 2; ix++) {
291  float xin = -pinstart[0] + 0.3 + (ix + 0.5) * 2 * ptopref[0];
292  for (Int_t iy = 0; iy < 2; iy++) {
293  z = -pinstart[2] + 2 * pal[2] + ptopref[2];
294  float yin = -pinstart[1] + 0.3 + (iy + 0.5) * 2 * ptopref[1];
295  ntops++;
296  ins->AddNode(topref, ntops, new TGeoTranslation(xin, yin, z));
297  z += ptopref[2] + 2. * pmcptopglass[2] + preg[2];
298  ins->AddNode(cat, ntops, new TGeoTranslation(xin, yin, z));
299  }
300  }
301  //Al top
302  z = -pinstart[2] + pal[2];
303  ins->AddNode(altop, 1, new TGeoTranslation(0, 0, z));
304 
305  // MCP
306  TVirtualMC::GetMC()->Gsvolu("0MTO", "BOX", getMediumID(kOpGlass), pmcptopglass, 3); //Op Glass
307  TGeoVolume* mcptop = gGeoManager->GetVolume("0MTO");
308  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + pmcptopglass[2];
309  ins->AddNode(mcptop, 1, new TGeoTranslation(0, 0, z));
310 
311  TVirtualMC::GetMC()->Gsvolu("0MCP", "BOX", getMediumID(kAir), pmcp, 3); //glass
312  TGeoVolume* mcp = gGeoManager->GetVolume("0MCP");
313  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + 2 * pmcptopglass[2] + 2 * preg[2] + pmcp[2];
314  ins->AddNode(mcp, 1, new TGeoTranslation(0, 0, z));
315  TVirtualMC::GetMC()->Gsvolu("0MIN", "BOX", getMediumID(kGlass), pmcpinner, 3); //glass
316  TGeoVolume* mcpinner = gGeoManager->GetVolume("0MIN");
317  mcp->AddNode(mcpinner, 1, new TGeoTranslation(0, 0, 0));
318 
319  TVirtualMC::GetMC()->Gsvolu("0MSI", "BOX", getMediumID(kGlass), pmcpside, 3); //glass
320  TGeoVolume* mcpside = gGeoManager->GetVolume("0MSI");
321  x = -pmcp[0] + pmcpside[0];
322  y = -pmcp[1] + pmcpside[1];
323  mcp->AddNode(mcpside, 1, new TGeoTranslation(x, y, 0));
324  x = pmcp[0] - pmcpside[0];
325  y = pmcp[1] - pmcpside[1];
326  mcp->AddNode(mcpside, 2, new TGeoTranslation(x, y, 0));
327  x = -pmcp[1] + pmcpside[1];
328  y = -pmcp[0] + pmcpside[0];
329  mcp->AddNode(mcpside, 3, new TGeoCombiTrans(x, y, 0, new TGeoRotation("R2", 90, 0, 0)));
330  x = pmcp[1] - pmcpside[1];
331  y = pmcp[0] - pmcpside[0];
332  mcp->AddNode(mcpside, 4, new TGeoCombiTrans(x, y, 0, new TGeoRotation("R2", 90, 0, 0)));
333 
334  TVirtualMC::GetMC()->Gsvolu("0MBA", "BOX", getMediumID(kCeramic), pmcpbase, 3); //glass
335  TGeoVolume* mcpbase = gGeoManager->GetVolume("0MBA");
336  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + pmcptopglass[2] + 2 * pmcp[2] + pmcpbase[2];
337  ins->AddNode(mcpbase, 1, new TGeoTranslation(0, 0, z));
338 }

340 Bool_t Detector::ProcessHits(FairVolume* v)
341 {
342 
343  TString volname = fMC->CurrentVolName();
344 
345  TVirtualMCStack* stack = fMC->GetStack();
346  Int_t quadrant, mcp;
347  if (fMC->IsTrackEntering()) {
348  float x, y, z;
349  fMC->TrackPosition(x, y, z);
350  fMC->CurrentVolID(quadrant);
351  fMC->CurrentVolOffID(1, mcp);
352  float time = fMC->TrackTime() * 1.0e9; //time from seconds to ns
353  int trackID = stack->GetCurrentTrackNumber();
354  int detID = 4 * mcp + quadrant - 1;
355  float etot = fMC->Etot();
356  int iPart = fMC->TrackPid();
357  float enDep = fMC->Edep();
358  Int_t parentID = stack->GetCurrentTrack()->GetMother(0);
359  if (fMC->TrackCharge()) { //charge particles for MCtrue
360  AddHit(x, y, z, time, 10, trackID, detID);
361  }
362  if (iPart == 50000050) { // If particles is photon then ...
363  if (etot > 7.46e-9 || etot < 2.4e-9) {
364  fMC->StopTrack();
365  return kFALSE;
366  }
367  if (volname.Contains("0TOP")) {
368  if (!RegisterPhotoE(etot)) {
369  fMC->StopTrack();
370  return kFALSE;
371  }
372  mTrackIdTop = trackID;
373  }
374 
375  if (volname.Contains("0MTO")) {
376  if (trackID != mTrackIdTop) {
377  if (!RegisterPhotoE(etot)) {
378  // std::cout<<" "<<etot<<" track "<<trackID<<" parentID "<<parentID<<" "<<volname<<std::endl;
379  fMC->StopTrack();
380  return kFALSE;
381  }
382  mTrackIdMCPtop = trackID;
383  }
384  }
385 
386  if (volname.Contains("0REG")) {
387  if (trackID != mTrackIdTop && trackID != mTrackIdMCPtop) {
388  if (RegisterPhotoE(etot)) {
389  AddHit(x, y, z, time, enDep, parentID, detID);
390  }
391  }
392  if (trackID == mTrackIdTop || trackID == mTrackIdMCPtop) {
393  AddHit(x, y, z, time, enDep, parentID, detID);
394  }
395  }
396  }
397 
398  return kTRUE;
399  }
400  return kFALSE;
401 }

431 void Detector::CreateMaterials()
432 {
433  Int_t isxfld = 2; // magneticField->Integ();
434  Float_t sxmgmx = 10.; // magneticField->Max();
435  // FIXME: use o2::base::Detector::initFieldTrack to init mag field params
436 
437  // Float_t a,z,d,radl,absl,buf[1];
438  // Int_t nbuf;
439  // AIR
440 
441  Float_t aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
442  Float_t zAir[4] = {6., 7., 8., 18.};
443  Float_t wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
444  Float_t dAir = 1.20479E-3;
445  Float_t dAir1 = 1.20479E-11;
446  // Radiator glass SiO2
447  Float_t aglass[2] = {28.0855, 15.9994};
448  Float_t zglass[2] = {14., 8.};
449  Float_t wglass[2] = {1., 2.};
450  Float_t dglass = 2.65;
451  // MCP glass SiO2
452  Float_t dglass_mcp = 1.3;
453  // Ceramic 97.2% Al2O3 , 2.8% SiO2
454  Float_t aCeramic[2] = {26.981539, 15.9994};
455  Float_t zCeramic[2] = {13., 8.};
456  Float_t wCeramic[2] = {2., 3.};
457  Float_t denscer = 3.6;
458  //*** Definition Of avaible FIT materials ***
459  Material(11, "Aliminium$", 26.98, 13.0, 2.7, 8.9, 999);
460  Mixture(1, "Vacuum$", aAir, zAir, dAir1, 4, wAir);
461  Mixture(2, "Air$", aAir, zAir, dAir, 4, wAir);
462  Mixture(4, "MCP glass $", aglass, zglass, dglass_mcp, -2, wglass);
463  Mixture(24, "Radiator Optical glass$", aglass, zglass, dglass, -2, wglass);
464  Mixture(3, "Ceramic $", aCeramic, zCeramic, denscer, -2, wCeramic);

573 Int_t Detector::ReadOptProperties(const std::string filePath)
574 {
575  std::ifstream infile;
576  infile.open(filePath.c_str());
577  LOG(INFO) << " file " << filePath.c_str();
578  // Check if file is opened correctly
579  if (infile.fail() == true) {
580  // AliFatal(Form("Error opening ascii file: %s", filePath.c_str()));
581  return -1;
582  }
583 
584  std::string comment; // dummy, used just to read 4 first lines and move the cursor to the 5th, otherwise unused
585  if (!getline(infile, comment)) { // first comment line
586  LOG(ERROR) << "Error opening ascii file (it is probably a folder!): " << filePath.c_str();
587  return -2;
588  }
589  getline(infile, comment); // 2nd comment line
590 
591  // Get number of elements required for the array
592  Int_t nLines;
593  infile >> nLines;
594  if (nLines < 0 || nLines > 1e4) {
595  return -4;
596  }
597 
598  getline(infile, comment); // finish 3rd line after the nEntries are read
599  getline(infile, comment); // 4th comment line
600 
601  // read the main body of the file (table of values: energy, absorption length and refractive index)
602  Int_t iLine = 0;
603  std::string sLine;
604  getline(infile, sLine);
605  while (!infile.eof()) {
606  if (iLine >= nLines) {
607  // LOG(ERROR) << "Line number: " << iLine << " reaches range of declared arraySize:" << kNbins << " Check input file:" << filePath.c_str();
608  return -5;
609  }
610  std::stringstream ssLine(sLine);
611  // First column:
612  Double_t energy;
613  ssLine >> energy;
614  energy *= 1e-9; // Convert eV -> GeV immediately
615  mPhotonEnergyD.push_back(energy);
616  // Second column:
617  Double_t absorption;
618  ssLine >> absorption;
619  mAbsorptionLength.push_back(absorption);
620  // Third column:
621  Double_t refraction;
622  ssLine >> refraction;
623  mRefractionIndex.push_back(refraction);
624  // Fourth column:
625  Double_t efficiency;
626  ssLine >> efficiency;
627  mQuantumEfficiency.push_back(efficiency);
628  if (!(ssLine.good() || ssLine.eof())) { // check if there were problems with numbers conversion
629  // AliFatal(Form("Error while reading line %i: %s", iLine, ssLine.str().c_str()));
630 
631  return -6;
632  }
633  getline(infile, sLine);
634  iLine++;
635  }
636  if (iLine != mPhotonEnergyD.size()) {
637  // LOG(ERROR)(Form("Total number of lines %i is different than declared %i. Check input file: %s", iLine, kNbins,
638  // filePath.c_str()));
639  return -7;
640  }
641 
642  LOG(INFO) << "Optical properties taken from the file: " << filePath.c_str() << " Number of lines read: " << iLine;
643  return 0;
644 }
