68  bool isManhattan() const
69  {
70  for (auto i = 0; i < mVertices.size() - 1; ++i) {
71  if (!isVertical(mVertices[i], mVertices[i + 1]) && !isHorizontal(mVertices[i], mVertices[i + 1])) {
72  return false;
73  }
74  }
75  return true;

82  double signedArea() const
83  {
87  double area{0.0};
88  for (auto i = 0; i < mVertices.size() - 1; ++i) {
89  auto& current = mVertices[i];
90  auto& next = mVertices[i + 1];
91  area += current.x * next.y - next.x * current.y;
92  }
93  return area * 0.5;
94  }

124 template <typename T>
125 Polygon<T> close(Polygon<T> polygon)
126 {
127  if (!polygon.isClosed()) {
128  auto vertices = getVertices(polygon);
129  vertices.push_back(polygon.firstVertex());
130  Polygon<T> pol(vertices.begin(), vertices.end());
131  if (!pol.isManhattan()) {
132  throw std::logic_error("closing resulted in non Manhattan polygon");
133  }
134  return pol;
135  }
136  return polygon;
137 }

149 template <typename T>
150 bool operator==(const Polygon<T>& lhs, const Polygon<T>& rhs)
151 {
152  if (lhs.size() != rhs.size()) {
153  return false;
154  }
155 
156  auto l = getSortedVertices(lhs);
157  auto r = getSortedVertices(rhs);
158 
159  if (l.size() != r.size()) {
160  return false;
161  }
162 
163  for (auto i = 0; i < l.size(); ++i) {
164  if (l[i] != r[i]) {
165  return false;
166  }
167  }
168  return true;
169 }

171 template <typename T>
172 bool Polygon<T>::contains(T xp, T yp) const
173 {
174  // Note that this algorithm yields unpredicatable result if the point xp,yp
175  // is on one edge of the polygon. Should not generally matters, except when comparing
176  // two different implementations maybe.
177  //
178  // TODO : look e.g. to http://alienryderflex.com/polygon/ for some possible optimizations
179  // (e.g. pre-computation)
180  //
181  if (!isClosed()) {
182  throw std::invalid_argument("contains method can only work with closed polygons");
183  }
184 
185  auto j = mVertices.size() - 1;
186  bool oddNodes{false};
187  for (auto i = 0; i < mVertices.size(); i++) {
188  if ((mVertices[i].y < yp && mVertices[j].y >= yp) || (mVertices[j].y < yp && mVertices[i].y >= yp)) {
189  if (mVertices[i].x +
190  (yp - mVertices[i].y) / (mVertices[j].y - mVertices[i].y) * (mVertices[j].x - mVertices[i].x) <
191  xp) {
192  oddNodes = !oddNodes;
193  }
194  }
195  j = i;
196  }
197  return oddNodes;
198 }

211 template <typename T>
212 std::vector<o2::mch::contour::Vertex<T>> getSortedVertices(const Polygon<T>& polygon)
213 {
214  std::vector<o2::mch::contour::Vertex<T>> vertices;
215  auto size = polygon.size();
216  if (polygon.isClosed()) {
217  --size;
218  }
219  vertices.reserve(size);
220  for (auto i = 0; i < size; ++i) {
221  vertices.push_back(polygon[i]);
222  }
223  std::sort(vertices.begin(), vertices.end());
224  return vertices;
225 }

227 template <typename T>
228 BBox<T> getBBox(const std::vector<Vertex<T>>& vertices)
229 {
230 
231  T xmin{std::numeric_limits<T>::max()};
232  T xmax{std::numeric_limits<T>::lowest()};
233  T ymin{std::numeric_limits<T>::max()};
234  T ymax{std::numeric_limits<T>::lowest()};
235 
236  for (const auto& v : vertices) {
237  xmin = std::min(xmin, v.x);
238  xmax = std::max(xmax, v.x);
239  ymin = std::min(ymin, v.y);
240  ymax = std::max(ymax, v.y);
241  }
242  return {xmin, ymin, xmax, ymax};
243 }

276 template <typename T>
277 auto squaredDistancePointToPolygon(const Vertex<T>& point, const Polygon<T>& polygon) -> decltype(point.x * point.x)
278 {
279  T d{std::numeric_limits<T>::max()};
280  for (auto i = 0; i < polygon.size() - 1; ++i) {
281  auto s0 = polygon[i];
282  auto s1 = polygon[i + 1];
283  auto d2 = squaredDistanceOfPointToSegment(point, s0, s1);
284  d = std::min(d2, d);
285  }
286  return d;
287 }
