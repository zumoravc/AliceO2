21 BOOST_AUTO_TEST_CASE(TestTMessageSerializer)
22 {
23  const char* testname = "testname";
24  const char* testtitle = "testtitle";
25  using namespace o2::framework;
26 
27  TObjArray array;
28  array.SetOwner();
29  array.Add(new TNamed(testname, testtitle));
30 
31  FairTMessage msg;
32  TMessageSerializer::serialize(msg, &array);
33 
34  auto buf = as_span(msg);
35  BOOST_CHECK_EQUAL(buf.size(), msg.BufferSize());
36  BOOST_CHECK_EQUAL(static_cast<void*>(buf.data()), static_cast<void*>(msg.Buffer()));
37  // test deserialization with TObject as target class (default)
38  auto out = TMessageSerializer::deserialize(buf);
39 
40  TObjArray* outarr = dynamic_cast<TObjArray*>(out.get());
41  BOOST_CHECK_EQUAL(out.get(), outarr);
42  TNamed* named = dynamic_cast<TNamed*>(outarr->At(0));
43  BOOST_CHECK_EQUAL(static_cast<void*>(named), static_cast<void*>(outarr->At(0)));
44  BOOST_CHECK_EQUAL(named->GetName(), testname);
45  BOOST_CHECK_EQUAL(named->GetTitle(), testtitle);
46 
47  // test deserialization with a wrong target class and check the exception
48  try {
49  auto out2 = TMessageSerializer::deserialize<TNamed>(buf);
50  BOOST_FAIL("here we should never get, the function call must fail with exception");
51  } catch (std::exception& e) {
52  std::string expected("can not convert serialized class TObjArray into target class TNamed");
53  BOOST_CHECK_MESSAGE(expected == e.what(), e.what());
54  }
55 }
