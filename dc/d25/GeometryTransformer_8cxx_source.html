48 ROOT::Math::Transform3D getDefaultRPCTransform(bool isRight, int chamber, int rpc)
49 {
51  const double degToRad = TMath::DegToRad();
52  double angle = isRight ? 0. : 180.;
53  ROOT::Math::Rotation3D rot(ROOT::Math::RotationY(angle * degToRad));
54  double xSign = isRight ? 1. : -1.;
55  double xPos = xSign * geoparams::getRPCCenterPosX(chamber, rpc);
56  double sign = (rpc % 2 == 0) ? 1. : -1;
57  if (!isRight) {
58  sign *= -1.;
59  }
60  double zPos = sign * geoparams::RPCZShift;
61  double newZ = geoparams::DefaultChamberZ[0] + zPos;
62  double oldZ = geoparams::DefaultChamberZ[0] - zPos;
63  double yPos = geoparams::getRPCHalfHeight(chamber) * (rpc - 4) * (1. + newZ / oldZ);
64  ROOT::Math::Translation3D trans(xPos, yPos, zPos);
65  return trans * rot;
66 }

68 GeometryTransformer createDefaultTransformer()
69 {
71  GeometryTransformer geoTrans;
72  for (int ich = 0; ich < detparams::NChambers; ++ich) {
73  for (int iside = 0; iside < 2; ++iside) {
74  bool isRight = (iside == 0);
75  for (int irpc = 0; irpc < detparams::NRPCLines; ++irpc) {
76  int deId = detparams::getDEId(isRight, ich, irpc);
77  ROOT::Math::Transform3D matrix = getDefaultChamberTransform(ich) * getDefaultRPCTransform(isRight, ich, irpc);
78  geoTrans.setMatrix(deId, matrix);
79  }
80  }
81  }
82  return geoTrans;
83 }

85 GeometryTransformer createTransformationFromManager(const TGeoManager* geoManager)
86 {
88  GeometryTransformer geoTrans;
89  TGeoNavigator* navig = geoManager->GetCurrentNavigator();
90  for (int ide = 0; ide < detparams::NDetectionElements; ++ide) {
91  int ichamber = detparams::getChamber(ide);
92  std::stringstream volPath;
93  volPath << geoManager->GetTopVolume()->GetName() << "/";
94  if (geoManager->GetVolume("YOUT2")) {
95  volPath << "YOUT2_1/";
96  }
97  volPath << geoparams::getChamberVolumeName(ichamber) << "_1/" << geoparams::getRPCVolumeName(geoparams::getRPCType(ide), ichamber) << "_" << std::to_string(ide);
98  if (!navig->cd(volPath.str().c_str())) {
99  throw std::runtime_error("Could not get to volPathName=" + volPath.str());
100  }
101  geoTrans.setMatrix(ide, o2::Transform3D{*(navig->GetCurrentMatrix())});
102  }
103  return geoTrans;
104 }
