67 TGeoVolume* Ladder::createVolume()
68 {
69  Int_t nChips = mSegmentation->getNSensors();
70 
71  // Create the flex
72  mFlex = new Flex(mSegmentation);
73  Double_t flexLength = nChips * (SegmentationAlpide::SensorSizeCols + Geometry::sSensorInterspace) +
74  Geometry::sLadderOffsetToEnd + Geometry::sSensorSideOffset;
75  Double_t shiftY =
76  4 * Geometry::sSensorTopOffset + SegmentationAlpide::SensorSizeRows - Geometry::sFlexHeight / 2; // to be verified!!
77 
78  auto& mftBaseParam = MFTBaseParam::Instance();
79  if (mftBaseParam.buildFlex) {
80  TGeoVolumeAssembly* flexVol = mFlex->makeFlex(mSegmentation->getNSensors(), flexLength);
81  mLadderVolume->AddNode(flexVol, 1, new TGeoTranslation(flexLength / 2 + Geometry::sSensorSideOffset / 2, shiftY, Geometry::sFlexThickness / 2 - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness)));
82  }
83  // Create the CMOS Sensors
84  createSensors();
85 
86  return mLadderVolume;
87 }

92 void Ladder::createSensors()
93 {
94 
95  Geometry* mftGeom = Geometry::instance();
96 
97  // Create Shapes
98 
99  // sensor = sensitive volume
100  TString namePrefixS = "MFTSensor";
101 
102  // chip = sensor + readout
103  TString namePrefixC =
104  Form("MFT_C_%d_%d_%d", mftGeom->getHalfID(mSegmentation->GetUniqueID()),
105  mftGeom->getDiskID(mSegmentation->GetUniqueID()), mftGeom->getLadderID(mSegmentation->GetUniqueID()));
106 
107  // the MFT glue
108  TString namePrefixG =
109  Form("MFT_G_%d_%d_%d", mftGeom->getHalfID(mSegmentation->GetUniqueID()),
110  mftGeom->getDiskID(mSegmentation->GetUniqueID()), mftGeom->getLadderID(mSegmentation->GetUniqueID()));
111 
112  TGeoMedium* kMedGlue = gGeoManager->GetMedium("MFT_SE4445$");
113 
114  TGeoVolume* glue = gGeoManager->MakeBox(
115  namePrefixG.Data(), kMedGlue, (SegmentationAlpide::SensorSizeCols - Geometry::sGlueEdge) / 2.,
116  (SegmentationAlpide::SensorSizeRows - Geometry::sGlueEdge) / 2., Geometry::sGlueThickness / 2.);
117  glue->SetVisibility(kTRUE);
118  glue->SetLineColor(kRed - 10);
119  glue->SetLineWidth(1);
120  glue->SetFillColor(glue->GetLineColor());
121  glue->SetFillStyle(4000); // 0% transparent
122 
123  // common with ITS
124  TGeoVolume* chipVol = AlpideChip::createChip(Geometry::sChipThickness / 2., Geometry::sSensorThickness / 2.,
125  namePrefixC, namePrefixS, kFALSE);
126 
127  // chipVol->Print();
128 
129  for (int ichip = 0; ichip < mSegmentation->getNSensors(); ichip++) {
130 
131  ChipSegmentation* chipSeg = mSegmentation->getSensor(ichip);
132  TGeoCombiTrans* chipPos = chipSeg->getTransformation();
133  TGeoCombiTrans* chipPosGlue = chipSeg->getTransformation();
134 
135  // Position of the center on the chip in the chip coordinate system
136  Double_t pos[3] = {SegmentationAlpide::SensorSizeCols / 2., SegmentationAlpide::SensorSizeRows / 2.,
137  Geometry::sChipThickness / 2. - Geometry::sGlueThickness - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness)};
138 
139  Double_t posglue[3] = {SegmentationAlpide::SensorSizeCols / 2., SegmentationAlpide::SensorSizeRows / 2.,
140  Geometry::sGlueThickness / 2 - Geometry::sChipThickness - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness)};
141 
142  Double_t master[3];
143  Double_t masterglue[3];
144  chipPos->LocalToMaster(pos, master);
145  chipPosGlue->LocalToMaster(posglue, masterglue);
146 
147  TGeoBBox* shape = (TGeoBBox*)mLadderVolume->GetShape();
148  master[0] -= shape->GetDX();
149  master[1] -= shape->GetDY();
150  master[2] -= shape->GetDZ();
151 
152  masterglue[0] -= shape->GetDX();
153  masterglue[1] -= shape->GetDY();
154  masterglue[2] -= shape->GetDZ();
155 
156  LOG(DEBUG1) << "CreateSensors " << Form("adding chip %s_%d ", namePrefixS.Data(), ichip);
157  // chipPos->Print();
158 
159  TGeoTranslation* trans = new TGeoTranslation(master[0], master[1], master[2]);
160  TGeoHMatrix* final = new TGeoHMatrix((*trans) * (Geometry::sTransMFT2ITS).Inverse());
161  mLadderVolume->AddNode(chipVol, ichip, final);
162  mLadderVolume->AddNode(glue, ichip, new TGeoTranslation(masterglue[0], masterglue[1], masterglue[2]));
163  }
164 }
