66 template <typename T>
67 void DBScan<T>::classifyVertices(std::function<unsigned char(std::vector<Edge>& edges)> classFunction)
68 {
69  mClassFunction = classFunction;
70  const size_t size = {this->mVertices->size()};
71  mStates.resize(size);
72 
73  if (!this->isMultiThreading()) {
74  for (size_t iVertex{0}; iVertex < size; ++iVertex) {
75  mStates[iVertex] = std::make_pair<int, unsigned char>(iVertex, classFunction(this->getEdges()[iVertex]));
76  }
77  } else {
78  const size_t stride{static_cast<size_t>(std::ceil(this->mVertices->size() / static_cast<size_t>(this->mExecutors.size())))};
79  for (size_t iExecutor{0}; iExecutor < this->mExecutors.size(); ++iExecutor) {
80  // We cannot pass a template function to std::thread(), using lambda instead
81  this->mExecutors[iExecutor] = std::thread(
82  [iExecutor, stride, this](const auto& classFunction) {
83  for (size_t iVertex{iExecutor * stride}; iVertex < stride * (iExecutor + 1) && iVertex < this->mVertices->size(); ++iVertex) {
84  mStates[iVertex] = std::make_pair<int, unsigned char>(iVertex, classFunction(this->getEdges()[iVertex]));
85  }
86  },
87  mClassFunction);
88  }
89  }
90  for (auto&& thread : this->mExecutors) {
91  thread.join();
92  }
93 }

102 template <typename T>
103 std::vector<int> DBScan<T>::getCores()
104 {
105  std::vector<State> cores;
106  std::vector<int> coreIndices;
107  std::copy_if(mStates.begin(), mStates.end(), std::back_inserter(cores), [](const State& state) { return state.second == 2; });
108  std::transform(cores.begin(), cores.end(), std::back_inserter(coreIndices), [](const State& state) -> int { return state.first; });
109  return coreIndices;
110 }

112 template <typename T>
113 std::vector<std::vector<int>> DBScan<T>::computeClusters()
114 {
115  std::vector<std::vector<int>> clusters;
116  std::vector<int> cores = getCores();
117  std::vector<unsigned char> usedVertices(this->mVertices->size(), false);
118 
119  for (size_t core{0}; core < cores.size(); ++core) {
120  if (!usedVertices[cores[core]]) {
121  std::vector<unsigned char> clusterFlags = this->getCluster(cores[core]);
122  std::transform(usedVertices.begin(), usedVertices.end(), clusterFlags.begin(), usedVertices.begin(), std::logical_or<>());
123  clusters.emplace_back(std::move(this->getClusterIndices(clusterFlags)));
124  }
125  }
126  return clusters;
127 }
