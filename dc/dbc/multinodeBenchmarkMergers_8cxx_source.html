53 WorkflowSpec defineDataProcessing(ConfigContext const& config)
54 {
55  int mergersLayers = config.options().get<int>("mergers-layers");
56  PublicationDecision mergersPublicationDecision = PublicationDecision::EachNSeconds;
57  double mergersPublicationInterval = config.options().get<double>("mergers-publication-interval");
58  InputObjectsTimespan mergersOwnershipMode =
59  config.options().get<std::string>("mergers-ownership-mode") == "full" ? InputObjectsTimespan::FullHistory : InputObjectsTimespan::LastDifference;
60  std::string inputChannelConfig = config.options().get<std::string>("input-channel-config");
61 
62  WorkflowSpec specs;
63 
64  specs.emplace_back(std::move(specifyExternalFairMQDeviceProxy(
65  "histo",
66  {{{"histo"}, {"TST", "HISTO"}}},
67  inputChannelConfig.c_str(),
68  dplModelAdaptor())));
69 
70  MergerInfrastructureBuilder mergersBuilder;
71  mergersBuilder.setInfrastructureName("histos");
72  mergersBuilder.setInputSpecs({{"histo", {"TST", "HISTO"}}});
73  mergersBuilder.setOutputSpec({{"main"}, "TST", "FULLHISTO", 0});
74  MergerConfig mergerConfig;
75  mergerConfig.inputObjectTimespan = {mergersOwnershipMode};
76  mergerConfig.publicationDecision = {mergersPublicationDecision, mergersPublicationDecision == PublicationDecision::EachNSeconds ? mergersPublicationInterval : 1.0};
77  mergerConfig.mergedObjectTimespan = {MergedObjectTimespan::FullHistory};
78  mergerConfig.topologySize = {TopologySize::NumberOfLayers, mergersLayers};
79  mergersBuilder.setConfig(mergerConfig);
80 
81  mergersBuilder.generateInfrastructure(specs);
82 
83  auto printHisto = [](const TH1* histo) {
84  if (histo) {
85  std::string bins = "BINS:";
86  for (int i = 1; i <= histo->GetNbinsX(); i++) {
87  bins += " " + std::to_string((int)histo->GetBinContent(i));
88  if (i >= 100) {
89  LOG(INFO) << "Trimming the output to 100 entries, total is: " << histo->GetNbinsX();
90  break;
91  }
92  }
93  LOG(INFO) << bins;
94  } else {
95  LOG(INFO) << "they asked me to print a nullptr";
96  }
97  };
98 
99  // clang-format off
100  DataProcessorSpec printer{
101  "printer-bins",
102  Inputs{
103  { "histo", "TST", "FULLHISTO", 0 }
104  },
105  Outputs{},
106  AlgorithmSpec{
107  (AlgorithmSpec::InitCallback) [&](InitContext&) {
108  return (AlgorithmSpec::ProcessCallback) [&](ProcessingContext& processingContext) mutable {
109  auto ref = processingContext.inputs().get<DataRef>("histo");
110  auto tobject = DataRefUtils::as<TObject>(ref);
111  if (auto histo = dynamic_cast<const TH1F*>(tobject.get())) {
112  printHisto(histo);
113  } else if (auto collection = dynamic_cast<TCollection*>(tobject.get())) {
114  LOG(INFO) << "Received a collection, printing the first and the last histogram, total is: " << std::to_string(collection->GetEntries());
115  printHisto(dynamic_cast<TH1*>(collection->begin()()));
116  printHisto(dynamic_cast<TH1*>(collection->FindObject(std::to_string(collection->GetEntries() - 1).c_str())));
117  collection->SetOwner(true);
118  }
119  };
120  }
121  }
122  };
123  specs.push_back(printer);
124 
125 
126  return specs;
127 }
