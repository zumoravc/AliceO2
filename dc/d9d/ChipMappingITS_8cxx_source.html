43 ChipMappingITS::ChipMappingITS()
44 {
45  // init chips info
46 
47  /*
48  FEE ID field in RDH is 16-bit wide
49  Each RU has 10-bit DIPSWITCH, use bits 9:2 as an 8bit ID field to be mapped to FEE ID field in RDH
50 
51  | Lr |Stave Count| Bin. pref | Range of Binary Addresses
52  | |= RU Count | Stave addr|
53  |----------------------------------------------------------
54  | L0 | 12 |b 0000xxxx | 0000_0000 : 0000_1011 (0 – 11)
55  | L1 | 16 |b 0001xxxx | 0001_0000 : 0001_1111 (0 – 15)
56  | L2 | 20 |b 001xxxxx | 001_00000 : 001_10011 (0 – 19)
57  | L3 | 24 |b 010xxxxx | 010_00000 : 010_10111 (0 – 23)
58  | L4 | 30 |b 011xxxxx | 011_00000 : 011_11101 (0 – 29)
59  | L5 | 42 |b 10xxxxxx | 10_000000 : 10_101001 (0 – 41)
60  | L6 | 48 |b 11xxxxxx | 11_000000 : 11_101111 (0 – 47)
61 
62  FEEId format:
63  15|14 12|11 10 |9 8|7 6|5 0|
64  0| Layer | Reserve | Fiber | 00 | StaveNumber |
65 
66  */
67 
68  uint32_t maxRUHW = composeFEEId(NLayers - 1, NStavesOnLr[NLayers - 1], NLinks - 1); // Max possible FEE ID
69  assert(maxRUHW < 0xffff);
70  mFEEId2RUSW.resize(maxRUHW + 1, 0xff);
71 
72  // IB: single cable per chip
73  int ctrChip = 0;
74  mChipInfoEntrySB[IB] = ctrChip;
75  mCableHW2SW[IB].resize(NChipsPerStaveSB[IB], 0xff);
76  mCableHWFirstChip[IB].resize(NChipsPerStaveSB[IB], 0xff);
77  for (int i = 0; i < NChipsPerStaveSB[IB]; i++) {
78  auto& cInfo = mChipsInfo[ctrChip++];
79  cInfo.id = i;
80  cInfo.moduleHW = 0;
81  cInfo.moduleSW = 0;
82  cInfo.chipOnModuleSW = i;
83  cInfo.chipOnModuleHW = i;
84  cInfo.cableHW = i;
85  cInfo.cableSW = i;
86  cInfo.chipOnCable = 0; // every chip is master
87  mCableHW2SW[IB][cInfo.cableHW] = cInfo.cableSW; //1-to-1 mapping
88  mCableHWFirstChip[IB][i] = 0; // stave and module are the same
89  }
90 
91  for (int bid = MB; bid <= OB; bid++) { // MB and OB staves have similar layout
92  mChipInfoEntrySB[bid] = ctrChip;
93  mCableHW2SW[bid].resize(0xff, 0xff);
94  mCableHWFirstChip[bid].resize(0xff, 0xff);
95 
96  for (int i = 0; i < NChipsPerStaveSB[bid]; i++) {
97  auto& cInfo = mChipsInfo[ctrChip++];
98  int hstave = i / (NChipsPerStaveSB[bid] / 2);
99  cInfo.id = i;
100  cInfo.moduleSW = i / NChipsPerModuleSB[bid]; // SW module ID (within the stave)
101  cInfo.moduleHW = 1 + cInfo.moduleSW % (NModulesPerStaveSB[bid] / 2); // identification within the HS
102 
103  cInfo.chipOnModuleSW = i % NChipsPerModuleSB[bid];
104  cInfo.chipOnModuleHW = ChipOBModSW2HW[cInfo.chipOnModuleSW];
105 
106  uint8_t connector = (hstave << 1) + (cInfo.chipOnModuleSW < (NChipsPerModuleSB[bid] / 2) ? 0 : 1);
107  cInfo.cableHW = (connector << 3) + (cInfo.moduleHW - 1);
108  cInfo.cableSW = (cInfo.moduleHW - 1) + connector * (NModulesPerStaveSB[bid] / 2);
109  cInfo.chipOnCable = cInfo.chipOnModuleSW % (NChipsPerModuleSB[bid] / 2); // each cable serves half module
110  mCableHW2SW[bid][cInfo.cableHW] = cInfo.cableSW;
111  if (cInfo.chipOnCable == 0) {
112  mCableHWFirstChip[bid][cInfo.cableHW] = cInfo.moduleSW * NChipsPerModuleSB[bid];
113  }
114  }
115  }
116 
117  int ctrStv = 0;
118  uint16_t chipCount = 0;
119  for (int ilr = 0; ilr < NLayers; ilr++) {
120  for (int ist = 0; ist < NStavesOnLr[ilr]; ist++) {
121  auto& sInfo = mStavesInfo[ctrStv];
122  sInfo.idSW = ctrStv++;
123 
124  // map FEEIds (RU read out by at most 3 GBT links) to SW ID
125  sInfo.idHW = composeFEEId(ilr, ist, 0); // FEEId for link 0
126  mFEEId2RUSW[sInfo.idHW] = sInfo.idSW;
127  for (int lnk = 1; lnk < NLinks; lnk++) {
128  mFEEId2RUSW[composeFEEId(ilr, ist, lnk)] = sInfo.idSW;
129  }
130  sInfo.layer = ilr;
131  sInfo.ruType = RUTypeLr[ilr];
132  sInfo.nCables = NCablesPerStaveSB[sInfo.ruType];
133  sInfo.firstChipIDSW = chipCount;
134  chipCount += NChipsPerStaveSB[sInfo.ruType];
135  }
136  }
137  assert(ctrStv == getNRUs());
138 }

159 void ChipMappingITS::expandChipInfoSW(int idSW, int& lay, int& sta, int& ssta, int& mod, int& chipInMod) const
160 {
161  // convert SW chip ID to detailed info SW info
162  ChipInfo chi;
163  getChipInfoSW(idSW, chi);
164  const auto staveInfo = getRUInfoSW(chi.ru);
165  lay = staveInfo->layer;
166  sta = staveInfo->idSW;
167  mod = chi.chOnRU->moduleSW;
168  chipInMod = chi.chOnRU->chipOnModuleSW;
169  ssta = lay < 3 || (mod < (NModulesPerStaveSB[chi.ruType] >> 2)) ? 0 : 1;
170 }

172 //______________________________________________
173 void ChipMappingITS::expandChipInfoHW(int idSW, int& lay, int& sta, int& ssta, int& mod, int& chipInMod) const
174 {
175  // convert SW chip ID to detailed info HW info
176  ChipInfo chi;
177  getChipInfoSW(idSW, chi);
178  const auto staveInfo = getRUInfoSW(chi.ru);
179  lay = staveInfo->layer;
180  sta = staveInfo->idSW - getFirstStavesOnLr(lay); // stave relative to layer
181  mod = lay < 3 ? 0 : 1 + (chi.chOnRU->moduleSW % NModulesAlongStaveSB[chi.ruType]); // module on the substave
182  ssta = chi.chOnRU->moduleSW / NModulesAlongStaveSB[chi.ruType];
183  chipInMod = chi.chOnRU->chipOnModuleHW;
184 }

187 std::string ChipMappingITS::getChipNameHW(int idSW) const
188 {
189  // convert global SW chip ID to name in HW conventions
190  int lay, sta, ssta, mod, cinmod;
191  expandChipInfoHW(idSW, lay, sta, ssta, mod, cinmod);
192  std::stringstream strs;
193  strs << 'L' << lay << '_' << std::setfill('0') << std::setw(2) << sta;
194  if (lay > 2) {
195  strs << (ssta ? 'U' : 'L') << "_M" << mod;
196  }
197  strs << "_C" << std::setfill('0') << std::setw(2) << cinmod;
198 
199  return strs.str();
200 }

203 void ChipMappingITS::imposeFEEId2RUSW(uint16_t feeID, uint16_t ruSW)
204 {
205  // test if it is legitimate
206  uint16_t lr, ruOnLr, link;
207  expandFEEId(feeID, lr, ruOnLr, link);
208  if (lr >= NLayers || ruOnLr >= NStavesOnLr[lr] || link >= NLinks) {
209  LOG(FATAL) << "Invalid FEE#0x" << std::hex << feeID << std::dec << ": corresponds to Lr#" << lr
210  << " StaveOnLr#" << ruOnLr << " GBTLinkOnRU#" << link;
211  }
212  if (ruSW >= getNRUs()) {
213  LOG(FATAL) << "Invalid SW RUid " << ruSW << " (cannot exceed " << getNRUs() << ")";
214  }
215  mFEEId2RUSW[feeID] = ruSW;
216 }
