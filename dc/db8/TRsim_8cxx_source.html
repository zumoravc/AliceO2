68 void TRsim::init()
69 {
70  //
71  // Initialization
72  // The default radiator are prolypropilene foils of 10 mu thickness
73  // with gaps of 80 mu filled with N2.
74  //
75 
76  mNFoils[0] = 170;
77  mNFoils[1] = 225;
78  mNFoils[2] = 275;
79  mNFoils[3] = 305;
80  mNFoils[4] = 325;
81  mNFoils[5] = 340;
82  mNFoils[6] = 350;
83 
84  mNFoilsUp[0] = 1.25;
85  mNFoilsUp[1] = 1.75;
86  mNFoilsUp[2] = 2.50;
87  mNFoilsUp[3] = 3.50;
88  mNFoilsUp[4] = 4.50;
89  mNFoilsUp[5] = 5.50;
90  mNFoilsUp[6] = 10000.0;
91 
92  mFoilThick = 0.0013;
93  mGapThick = 0.0060;
94  mFoilDens = 0.92;
95  mGapDens = 0.00125;
96 
97  mFoilZ = 5.28571;
98  mGapZ = 7.0;
99  mFoilA = 10.4286;
100  mGapA = 14.00674;
101  mTemp = 293.16;
102 
103  mFoilOmega = getOmega(mFoilDens, mFoilZ, mFoilA);
104  mGapOmega = getOmega(mGapDens, mGapZ, mGapA);
105 
106  mSpBinWidth = mSpRange / mSpNBins;
107  mSpLower = 1.0 - 0.5 * mSpBinWidth;
108  mSpUpper = mSpLower + mSpRange;
109 
110  if (mSpectrum) {
111  delete mSpectrum;
112  }
113  mSpectrum = new TH1D("TRspectrum", "TR spectrum", mSpNBins, mSpLower, mSpUpper);
114  mSpectrum->SetDirectory(nullptr);
115 
116  // Set the sigma values
117  setSigma();
118 }

120 //_____________________________________________________________________________
121 int TRsim::createPhotons(int pdg, float p, std::vector<float>& ePhoton)
122 {
123  //
124  // Create TRD photons for a charged particle of type <pdg> with the total
125  // momentum <p>.
126  // Energies of the produced TR photons: <ePhoton>
127  //
128 
129  // PDG codes
130  const int kPdgEle = 11;
131  const int kPdgMuon = 13;
132  const int kPdgPion = 211;
133  const int kPdgKaon = 321;
134 
135  float mass = 0;
136  switch (TMath::Abs(pdg)) {
137  case kPdgEle:
138  mass = 5.11e-4;
139  break;
140  case kPdgMuon:
141  mass = 0.10566;
142  break;
143  case kPdgPion:
144  mass = 0.13957;
145  break;
146  case kPdgKaon:
147  mass = 0.4937;
148  break;
149  default:
150  return 0;
151  break;
152  };
153 
154  // Calculate the TR photons
155  return calculatePhotons(p, mass, ePhoton);
156 }

158 //_____________________________________________________________________________
159 int TRsim::calculatePhotons(float p, float mass, std::vector<float>& ePhoton)
160 {
161  //
162  // Produces TR photons using a parametric model for regular radiator. Photons
163  // with energy larger than 15 keV are included in the MC stack and tracked by VMC
164  // machinary.
165  //
166  // Input parameters:
167  // p - parent momentum (GeV/c)
168  // mass - parent mass
169  //
170  // Output :
171  // ePhoton - energy container of this photons in keV.
172  //
173 
174  const double kAlpha = 0.0072973;
175  const int kSumMax = 30;
176  double tau = mGapThick / mFoilThick;
177  // Calculate gamma
178  double gamma = TMath::Sqrt(p * p + mass * mass) / mass;
179  // Select the number of foils corresponding to momentum
180  int foils = selectNFoils(p);
181 
182  // The TR spectrum
183  double csi1;
184  double csi2;
185  double rho1;
186  double rho2;
187  double sigma;
188  double sum;
189  double nEqu;
190  double thetaN;
191  double aux;
192  double energyeV;
193  double energykeV;
194 
195  mSpectrum->Reset();
196  for (int iBin = 1; iBin <= mSpNBins; iBin++) {
197 
198  energykeV = mSpectrum->GetBinCenter(iBin);
199  energyeV = energykeV * 1.0e3;
200 
201  sigma = getSigma(energykeV);
202 
203  csi1 = mFoilOmega / energyeV;
204  csi2 = mGapOmega / energyeV;
205 
206  rho1 = 2.5 * energyeV * mFoilThick * 1.0e4 * (1.0 / (gamma * gamma) + csi1 * csi1);
207  rho2 = 2.5 * energyeV * mFoilThick * 1.0e4 * (1.0 / (gamma * gamma) + csi2 * csi2);
208 
209  // Calculate the sum
210  sum = 0.0;
211  for (int n = 1; n <= kSumMax; n++) {
212  thetaN = (TMath::Pi() * 2.0 * n - (rho1 + tau * rho2)) / (1.0 + tau);
213  if (thetaN < 0.0) {
214  thetaN = 0.0;
215  }
216  aux = 1.0 / (rho1 + thetaN) - 1.0 / (rho2 + thetaN);
217  sum += thetaN * (aux * aux) * (1.0 - TMath::Cos(rho1 + thetaN));
218  }
219 
220  // Equivalent number of foils
221  nEqu = (1.0 - TMath::Exp(-foils * sigma)) / (1.0 - TMath::Exp(-sigma));
222 
223  // dN / domega
224  mSpectrum->SetBinContent(iBin, 4.0 * kAlpha * nEqu * sum / (energykeV * (1.0 + tau)));
225  }
226 
227  // <nTR> (binsize corr.)
228  float nTr = mSpBinWidth * mSpectrum->Integral();
229  // Number of TR photons from Poisson distribution with mean <nTr>
230  int nPhCand = gRandom->Poisson(nTr);
231 
232  // Link the MC stack and get info about parent electron
233  TVirtualMCStack* stack = TVirtualMC::GetMC()->GetStack();
234  int track = stack->GetCurrentTrackNumber();
235  double px, py, pz, ptot;
236  TVirtualMC::GetMC()->TrackMomentum(px, py, pz, ptot);
237  ptot = TMath::Sqrt(px * px + py * py + pz * pz);
238  px /= ptot;
239  py /= ptot;
240  pz /= ptot;
241  // Current position of electron
242  double x, y, z;
243  TVirtualMC::GetMC()->TrackPosition(x, y, z);
244  double t = TVirtualMC::GetMC()->TrackTime();
245  for (int iPhoton = 0; iPhoton < nPhCand; ++iPhoton) {
246  double e = mSpectrum->GetRandom(); // Energy of the TR photon
247  // Put TR photon on particle stack
248  if (e > 15) {
249  e *= 1e-6; // Convert it to GeV
250  int phtrack;
251  stack->PushTrack(1, // Must be 1
252  track, // Identifier of the parent track, -1 for a primary
253  22, // Particle code.
254  px * e, // 4 momentum (The photon is generated on the same
255  py * e, // direction as the parent. For irregular radiator one
256  pz * e, // can calculate also the angle but this is a second
257  e, // order effect)
258  x, y, z, t, // 4 vertex
259  0.0, 0.0, 0.0, // Polarisation
260  kPFeedBackPhoton, // Production mechanism (there is no TR in G3 so one has to make some convention)
261  phtrack, // On output the number of the track stored
262  1.0,
263  1);
264  }
265  // Custom treatment of TR photons
266  else {
267  ePhoton.push_back(e);
268  }
269  }
270  return 1;
271 }

273 //_____________________________________________________________________________
274 void TRsim::setSigma()
275 {
276  //
277  // Sets the absorbtion crosssection for the energies of the TR spectrum
278  //
279 
280  for (int iBin = 0; iBin < mSpNBins; iBin++) {
281  double energykeV = iBin * mSpBinWidth + 1.0;
282  mSigma[iBin] = getSigma(energykeV);
283  }
284 }

286 //_____________________________________________________________________________
287 double TRsim::getSigma(double energykeV)
288 {
289  //
290  // Calculates the absorbtion crosssection for a one-foil-one-gap-radiator
291  //
292 
293  // keV -> MeV
294  double energyMeV = energykeV * 0.001;
295  if (energyMeV >= 0.001) {
296  return (getMuPo(energyMeV) * mFoilDens * mFoilThick +
297  getMuAi(energyMeV) * mGapDens * mGapThick * getTemp());
298  } else {
299  return 1.0e6;
300  }
301 }

632 double TRsim::interpolate(double energyMeV, double* en, const double* const mu, int n)
633 {
634  //
635  // interpolates the photon absorbtion cross section
636  // for a given energy <energyMeV>.
637  //
638 
639  double de = 0;
640  int index = 0;
641  int istat = locate(en, n, energyMeV, index, de);
642  if (istat == 0) {
643  return (mu[index] - de * (mu[index] - mu[index + 1]) / (en[index + 1] - en[index]));
644  } else {
645  return 0.0;
646  }
647 }

649 //_____________________________________________________________________________
650 int TRsim::locate(double* xv, int n, double xval, int& kl, double& dx)
651 {
652  //
653  // locates a point (xval) in a 1-dim grid (xv(n))
654  //
655 
656  if (xval >= xv[n - 1]) {
657  return 1;
658  }
659  if (xval < xv[0]) {
660  return -1;
661  }
662  int km;
663  int kh = n - 1;
664  kl = 0;
665  while (kh - kl > 1) {
666  if (xval < xv[km = (kl + kh) / 2]) {
667  kh = km;
668  } else {
669  kl = km;
670  }
671  }
672  if ((xval < xv[kl]) ||
673  (xval > xv[kl + 1]) ||
674  (kl >= n - 1)) {
675  LOG(FATAL) << Form("locate failed xv[%d] %f xval %f xv[%d] %f!!!\n", kl, xv[kl], xval, kl + 1, xv[kl + 1]);
676  exit(1);
677  }
678  dx = xval - xv[kl];
679  return 0;
680 }
