66 Double_t AliTPC3DCylindricalInterpolator::InterpolateCylindrical(Double_t r, Double_t z, Double_t phi) const
67 {
68  Int_t iLow = 0, jLow = 0, m = 0;
69  Int_t kLow = 0;
70  Int_t index;
71 
72  // tri cubic points
73  Double_t saveArray[fOrder + 1];
74  Double_t savedArray[fOrder + 1];
75  Double_t zListM1[fOrder + 1];
76  Double_t valueM1[fOrder + 1];
77 
78  Bool_t neg = kFALSE;
79 
80  // check phi
81  while (phi < 0.0) {
82  phi = TMath::TwoPi() + phi;
83  }
84  while (phi > TMath::TwoPi()) {
85  phi = phi - TMath::TwoPi();
86  }
87 
88  // search lowest index related to r,z and phi
89  Search(fNR, fRList, r, iLow);
90  Search(fNZ, fZList, z, jLow);
91  Search(fNPhi, fPhiList, phi, kLow);
92 
93  // order >= 3
94  kLow -= (fOrder / 2);
95  iLow -= (fOrder / 2);
96  jLow -= (fOrder / 2);
97 
98  // check if out of range
99  if (iLow < 0) {
100  iLow = 0;
101  }
102  if (jLow < 0) {
103  jLow = 0;
104  }
105  if (kLow < 0) {
106  kLow = fNPhi + kLow;
107  }
108  // check if out of range
109  if (iLow + fOrder >= fNR - 1) {
110  iLow = fNR - 1 - fOrder;
111  }
112  if (jLow + fOrder >= fNZ - 1) {
113  jLow = fNZ - 1 - fOrder;
114  }
115 
116  // do for each
117  for (Int_t k = 0; k < fOrder + 1; k++) {
118  m = (kLow + k) % fNPhi;
119  // interpolate
120  for (Int_t i = iLow; i < iLow + fOrder + 1; i++) {
121  if (fOrder < 3) {
122  if (jLow >= 0) {
123  index = m * (fNZ * fNR) + i * (fNZ) + jLow;
124  saveArray[i - iLow] = Interpolate(&fZList[jLow], &fValue[index], z);
125  } else {
126  index = m * (fNZ * fNR) + i * (fNZ);
127  zListM1[0] = fZList[0] - (fZList[1] - fZList[0]);
128  zListM1[1] = fZList[0];
129  zListM1[2] = fZList[1];
130  valueM1[0] = fValue[index] - (fValue[index + 1] - fValue[index]);
131  valueM1[1] = fValue[index];
132  valueM1[2] = fValue[index + 1];
133  saveArray[i - iLow] = Interpolate(&zListM1[0], &valueM1[0], z);
134  }
135 
136  } else {
137  index = m * (fNZ * fNR) + i * (fNZ);
138  saveArray[i - iLow] = InterpolateCubicSpline(fZList, &fValue[index], &fSecondDerZ[index], fNZ, fNZ, fNZ,
139  z, 1);
140  }
141  }
142  savedArray[k] = Interpolate(&fRList[iLow], saveArray, r);
143  }
144  return (InterpolatePhi(&fPhiList[0], kLow, fNPhi, savedArray, phi));
145 }

154 Double_t AliTPC3DCylindricalInterpolator::Interpolate(Double_t xArray[], Double_t yArray[], Double_t x) const
155 {
156  Double_t y;
157 
158  // if cubic spline
159  if (fOrder > 2) {
160  Double_t y2Array[fOrder + 1];
161  InitCubicSpline(xArray, yArray, fOrder + 1, y2Array, 1);
162  y = InterpolateCubicSpline(xArray, yArray, y2Array, fOrder + 1, fOrder + 1, fOrder + 1, x, 1);
163  } else if (fOrder == 2) {
164  // Quadratic Interpolation = 2
165  y = (x - xArray[1]) * (x - xArray[2]) * yArray[0] / ((xArray[0] - xArray[1]) * (xArray[0] - xArray[2]));
166  y += (x - xArray[2]) * (x - xArray[0]) * yArray[1] / ((xArray[1] - xArray[2]) * (xArray[1] - xArray[0]));
167  y += (x - xArray[0]) * (x - xArray[1]) * yArray[2] / ((xArray[2] - xArray[0]) * (xArray[2] - xArray[1]));
168  } else {
169  // Linear Interpolation = 1
170  y = yArray[0] + (yArray[1] - yArray[0]) * (x - xArray[0]) / (xArray[1] - xArray[0]);
171  }
172  return (y);
173 }

182 Double_t AliTPC3DCylindricalInterpolator::InterpolatePhi(
183  Double_t xArray[], const Int_t iLow, const Int_t lenX, Double_t yArray[], Double_t x) const
184 {
185  Int_t i0 = iLow;
186  Double_t xi0 = xArray[iLow];
187  Int_t i1 = (iLow + 1) % lenX;
188  Double_t xi1 = xArray[i1];
189  Int_t i2 = (iLow + 2) % lenX;
190  Double_t xi2 = xArray[i2];
191 
192  if (fOrder <= 2) {
193  if (xi1 < xi0) {
194  xi1 = TMath::TwoPi() + xi1;
195  }
196  if (xi2 < xi1) {
197  xi2 = TMath::TwoPi() + xi2;
198  }
199  if (x < xi0) {
200  x = TMath::TwoPi() + x;
201  }
202  }
203 
204  Double_t y;
205  if (fOrder > 2) {
206  Double_t y2Array[fOrder + 1];
207  Double_t xArrayTemp[fOrder + 1];
208  Double_t dPhi = xArray[1] - xArray[0];
209  // make list phi ascending order
210  for (Int_t i = 0; i < fOrder + 1; i++) {
211  xArrayTemp[i] = xArray[iLow] + (dPhi * i);
212  }
213  if (x < xArrayTemp[0]) {
214  x = TMath::TwoPi() + x;
215  }
216  if (x < xArrayTemp[0] || x > xArrayTemp[fOrder]) {
217  printf("x (%f) is outside of interpolation box (%f,%f)\n", x, xArrayTemp[0], xArrayTemp[fOrder]);
218  }
219 
220  InitCubicSpline(xArrayTemp, yArray, fOrder + 1, y2Array, 1);
221  y = InterpolateCubicSpline(xArrayTemp, yArray, y2Array, fOrder + 1, fOrder + 1, fOrder + 1, x, 1);
222  } else if (fOrder == 2) { // Quadratic Interpolation = 2
223  y = (x - xi1) * (x - xi2) * yArray[0] / ((xi0 - xi1) * (xi0 - xi2));
224  y += (x - xi2) * (x - xi0) * yArray[1] / ((xi1 - xi2) * (xi1 - xi0));
225  y += (x - xi0) * (x - xi1) * yArray[2] / ((xi2 - xi0) * (xi2 - xi1));
226  } else { // Li2near Interpolation = 1
227  y = yArray[0] + (yArray[1] - yArray[0]) * (x - xi0) / (xi1 - xi0);
228  }
229  return (y);
230 }

240 void AliTPC3DCylindricalInterpolator::InitCubicSpline(Double_t* xArray, Double_t* yArray, const Int_t n, Double_t* y2Array,
241  const Int_t skip) const
242 {
243  Double_t u[n];
244  Double_t sig, p, qn, un;
245 
246  y2Array[0] = 0.0;
247  u[0] = 0.0; //natural condition
248 
249  for (Int_t i = 1; i <= n - 2; i++) {
250  sig = (xArray[i] - xArray[i - 1]) / (xArray[i + 1] - xArray[i - 1]);
251  p = sig * y2Array[(i - 1) * skip] + 2.0;
252  y2Array[i * skip] = (sig - 1.0) / p;
253  u[i] = (yArray[(i + 1) * skip] - yArray[i * skip]) / (xArray[i + 1] - xArray[i]) -
254  (yArray[i * skip] - yArray[(i - 1) * skip]) / (xArray[i] - xArray[i - 1]);
255  u[i] = (6.0 * u[i] / (xArray[i + 1] - xArray[i - 1]) - sig * u[i - 1]) / p;
256  }
257 
258  qn = un = 0.0;
259 
260  y2Array[(n - 1) * skip] = (un - qn * u[n - 2]) / (qn * y2Array[(n - 2) * skip] + 1.0);
261  for (Int_t k = n - 2; k >= 0; k--) {
262  y2Array[k * skip] = y2Array[k * skip] * y2Array[(k + 1) * skip] + u[k];
263  }
264 }

274 void AliTPC3DCylindricalInterpolator::InitCubicSpline(Double_t* xArray, Double_t* yArray, const Int_t n, Double_t* y2Array,
275  const Int_t skip, Double_t yp0, Double_t ypn1) const
276 {
277  Double_t u[n];
278  Double_t sig, p, qn, un;
279 
280  y2Array[0] = 0.0;
281  u[0] = 0.0; //natural condition
282 
283  for (Int_t i = 1; i <= n - 2; i++) {
284  sig = (xArray[i] - xArray[i - 1]) / (xArray[i + 1] - xArray[i - 1]);
285  p = sig * y2Array[(i - 1) * skip] + 2.0;
286  y2Array[i * skip] = (sig - 1.0) / p;
287  u[i] = (yArray[(i + 1) * skip] - yArray[i * skip]) / (xArray[i + 1] - xArray[i]) -
288  (yArray[i * skip] - yArray[(i - 1) * skip]) / (xArray[i] - xArray[i - 1]);
289  u[i] = (6.0 * u[i] / (xArray[i + 1] - xArray[i - 1]) - sig * u[i - 1]) / p;
290  }
291 
292  qn = un = 0.0;
293  y2Array[(n - 1) * skip] = (un - qn * u[n - 2]) / (qn * y2Array[(n - 2) * skip] + 1.0);
294  for (Int_t k = n - 2; k >= 0; k--) {
295  y2Array[k * skip] = y2Array[k * skip] * y2Array[(k + 1) * skip] + u[k];
296  }
297 }

310 Double_t AliTPC3DCylindricalInterpolator::InterpolateCubicSpline(Double_t* xArray, Double_t* yArray, Double_t* y2Array,
311  const Int_t nxArray, const Int_t nyArray,
312  const Int_t ny2Array, Double_t x, Int_t skip) const
313 {
314  Int_t klo, khi, k;
315  Float_t h, b, a;
316  klo = 0;
317  khi = nxArray - 1;
318 
319  while (khi - klo > 1) {
320  k = (khi + klo) >> 1;
321  if (xArray[k] > x) {
322  khi = k;
323  } else {
324  klo = k;
325  }
326  }
327 
328  h = xArray[khi] - xArray[klo];
329 
330  if (TMath::Abs(h) < 1e-10) {
331  return 0.0;
332  }
333 
334  a = (xArray[khi] - x) / h;
335  b = (x - xArray[klo]) / h;
336 
337  Double_t y = a * yArray[klo] + b * yArray[khi] +
338  ((a * a * a - a) * y2Array[klo * skip] + (b * b * b - b) * y2Array[khi * skip]) * (h * h) / 6.0;
339 
340  return y;
341 }

345 void AliTPC3DCylindricalInterpolator::InitCubicSpline()
346 {
347 
348  Double_t yp0, ypn1;
349  if (fIsInitCubic != kTRUE) {
350  fSecondDerZ = new Double_t[fNR * fNZ * fNPhi];
351 
352  // Init at Z direction
353  for (Int_t m = 0; m < fNPhi; m++) {
354  for (Int_t i = 0; i < fNR; i++) {
355  yp0 = (-(11.0 / 6.0) * fValue[(m * (fNZ * fNR) + i * fNZ)] +
356  (3.0 * fValue[(m * (fNZ * fNR) + i * fNZ) + 1]) -
357  (1.5 * fValue[(m * (fNZ * fNR) + i * fNZ) + 2]) +

364  (fZList[0] - fZList[1]);

380 void AliTPC3DCylindricalInterpolator::Search(Int_t n, const Double_t xArray[], Double_t x, Int_t& low) const
381 {
383 
384  Long_t middle, high;
385  Int_t ascend = 0, increment = 1;
386 
387  if (xArray[n - 1] > xArray[0]) {
388  ascend = 1; // Ascending ordered table if true
389  }
390  if (low < 0 || low > n - 1) {
391  low = -1;
392  high = n;
393  } else { // Ordered Search phase
394  if ((Int_t)(x > xArray[low]) == ascend) {
395  if (low == n - 1) {
396  return;
397  }
398  high = low + 1;
399  while ((Int_t)(x > xArray[high]) == ascend) {
400  low = high;
401  increment *= 2;
402  high = low + increment;
403  if (high > n - 1) {
404  high = n;
405  break;
406  }
407  }
408  } else {
409  if (low == 0) {
410  low = -1;
411  return;
412  }
413  high = low - 1;
414  while ((Int_t)(x < xArray[low]) == ascend) {
415  high = low;
416  increment *= 2;
417  if (increment >= high) {
418  low = -1;
419  break;
420  } else {
421  low = high - increment;
422  }
423  }
424  }
425  }
426 
427  while ((high - low) != 1) { // Binary Search Phase
428  middle = (high + low) / 2;
429  if ((Int_t)(x > xArray[middle]) == ascend) {
430  low = middle;
431  } else {
432  high = middle;
433  }
434  }
435 
436  if (x > xArray[n - 1]) {
437  low = n;
438  }
439  if (x < xArray[0]) {
440  low = -1;
441  }
442 }
