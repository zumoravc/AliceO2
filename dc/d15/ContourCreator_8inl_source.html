35 template <typename T>
36 void sortVerticalEdges(std::vector<VerticalEdge<T>>& edges)
37 {
38  // sort vertical edges in ascending x order
39  // if same x, insure that left edges are before right edges
40  // within same x, order by increasing bottommost y
41  // Mind your steps ! This sorting is critical to the contour merging algorithm !
42 
43  std::sort(edges.begin(), edges.end(), [](const VerticalEdge<T>& e1, const VerticalEdge<T>& e2) {
44 
45  auto x1 = e1.begin().x;
46  auto x2 = e2.begin().x;
47 
48  auto y1 = bottom(e1);
49  auto y2 = bottom(e2);
50 
51  if (areEqual(x1, x2)) {
52  if (isLeftEdge(e1) && isRightEdge(e2)) {
53  return true;
54  }
55  if (isRightEdge(e1) && isLeftEdge(e2)) {
56  return false;
57  }
58  return y1 < y2;
59  } else if (x1 < x2) {
60  return true;
61  } else {
62  return false;
63  }
64  });
65 }

67 template <typename T>
68 Interval<T> interval(const VerticalEdge<T>& edge)
69 {
70  auto y1 = edge.begin().y;
71  auto y2 = edge.end().y;
72  return y2 > y1 ? Interval<T>(y1, y2) : Interval<T>(y2, y1);
73 }

75 template <typename T>
76 std::vector<VerticalEdge<T>> getVerticalEdges(const Polygon<T>& polygon)
77 {
79  std::vector<VerticalEdge<T>> edges;
80  for (auto i = 0; i < polygon.size() - 1; ++i) {
81  auto current = polygon[i];
82  auto next = polygon[i + 1];
83  if (current.x == next.x) {
84  edges.push_back({ current.x, current.y, next.y });
85  }
86  }
87  return edges;
88 }

116 template <typename T>
117 std::vector<T> getPositions(const std::vector<Polygon<T>>& polygons, GetVertexPosFunc<T> func)
118 {
119  std::vector<T> ypos;
120  for (auto i = 0; i < polygons.size(); ++i) {
121  for (auto j = 0; j < polygons[i].size(); ++j) {
122  ypos.push_back(func(polygons[i][j]));
123  }
124  }
125  std::sort(ypos.begin(), ypos.end());
126  auto last = std::unique(ypos.begin(), ypos.end(), [](const T& a, const T& b) { return areEqual(a, b); });
127  ypos.erase(last, ypos.end());
128  return ypos;
129 }

143 template <typename T>
144 std::vector<VerticalEdge<T>> sweep(Node<T>* segmentTree, const std::vector<VerticalEdge<T>>& polygonVerticalEdges)
145 {
146  std::vector<VerticalEdge<T>> contourVerticalEdges;
147 
148  std::vector<Interval<T>> edgeStack;
149 
150  for (auto i = 0; i < polygonVerticalEdges.size(); ++i) {
151 
152  const auto& edge = polygonVerticalEdges[i];
153  auto ival = interval(edge);
154 
155  if (isLeftEdge(edge)) {
156  segmentTree->contribution(ival, edgeStack);
157  segmentTree->insertInterval(ival);
158  } else {
159  segmentTree->deleteInterval(ival);
160  segmentTree->contribution(ival, edgeStack);
161  }
162 
163  auto e1{ edge };
164 
165  if (i < polygonVerticalEdges.size() - 1) {
166  e1 = polygonVerticalEdges[i + 1];
167  }
168 
169  if ((isLeftEdge(edge) != isLeftEdge(e1)) || (!areEqual(edge.begin().x, e1.begin().x)) ||
170  (i == polygonVerticalEdges.size() - 1)) {
171  for (auto es : edgeStack) {
172  contourVerticalEdges.push_back(isRightEdge(edge) ? VerticalEdge<T>{ edge.begin().x, es.begin(), es.end() }
173  : VerticalEdge<T>{ edge.begin().x, es.end(), es.begin() });
174  }
175  edgeStack.clear();
176  }
177  }
178 
179  return contourVerticalEdges;
180 }

190 template <typename T>
191 std::vector<HorizontalEdge<T>> verticalsToHorizontals(const std::vector<VerticalEdge<T>>& verticals)
192 {
193  std::vector<HorizontalEdge<T>> horizontals(verticals.size());
194 
195  using VertexWithRef = std::pair<Vertex<T>, T>;
196  std::vector<VertexWithRef> vertices;
197 
198  for (auto i = 0; i < verticals.size(); ++i) {
199  const auto& edge = verticals[i];
200  vertices.push_back({ edge.begin(), i });
201  vertices.push_back({ edge.end(), i });
202  }
203 
204  std::sort(vertices.begin(), vertices.end(),
205  [](const VertexWithRef& v1, const VertexWithRef& v2) { return v1.first < v2.first; });
206 
207  for (auto i = 0; i < vertices.size() / 2; ++i) {
208  const auto& p1 = vertices[2 * i];
209  const auto& p2 = vertices[2 * i + 1];
210  const VerticalEdge<T>& refEdge = verticals[p1.second];
211  auto e = p1.first.x;
212  auto b = p2.first.x;
213  if ((areEqual(p1.first.y, bottom(refEdge)) && isLeftEdge(refEdge)) ||
214  (areEqual(p1.first.y, top(refEdge)) && isRightEdge(refEdge))) {
215  std::swap(b, e);
216  }
217  HorizontalEdge<T> h{ p1.first.y, b, e };
218  // which vertical edge is preceding this horizontal ?
219  int preceding = p1.second;
220  int next = p2.second;
221  if (b > e) {
222  std::swap(preceding, next);
223  }
224  horizontals[preceding] = h;
225  }
226  return horizontals;
227 }

229 template <typename T>
230 Contour<T> finalizeContour(const std::vector<VerticalEdge<T>>& verticals,
231  const std::vector<HorizontalEdge<T>>& horizontals)
232 {
233  if (verticals.size() != horizontals.size()) {
234  throw std::invalid_argument("should get the same number of verticals and horizontals");
235  }
236 
237  for (auto i = 0; i < verticals.size(); ++i) {
238  if (horizontals[i].begin() != verticals[i].end()) {
239  throw std::invalid_argument("got an horizontal edge not connected to its (supposedly) preceding vertical edge");
240  }
241  }
242 
243  std::vector<ManhattanEdge<T>> all;
244 
245  for (auto i = 0; i < verticals.size(); ++i) {
246  all.push_back(verticals[i]);
247  all.push_back(horizontals[i]);
248  }
249 
250  Contour<T> contour;
251 
252  std::vector<bool> alreadyAdded(all.size(), false);
253  std::vector<int> inorder;
254 
255  int nofUsed{ 0 };
256  int iCurrent{ 0 };
257 
258  ManhattanEdge<T> startSegment{ all[iCurrent] };
259 
260  while (nofUsed < all.size()) {
261 
262  const ManhattanEdge<T>& currentSegment{ all[iCurrent] };
263  inorder.push_back(iCurrent);
264  alreadyAdded[iCurrent] = true;
265  ++nofUsed;
266  if (currentSegment.end() == startSegment.begin()) {
267  if (inorder.empty()) {
268  throw std::runtime_error("got an empty polygon");
269  }
270  std::vector<Vertex<T>> vertices;
271  vertices.reserve(inorder.size());
272  for (auto i : inorder) {
273  vertices.push_back(all[i].begin());
274  }
275  Polygon<T> polygon(vertices.begin(), vertices.end());
276  contour.addPolygon(close(polygon));
277  iCurrent = std::distance(alreadyAdded.begin(), std::find_if(alreadyAdded.begin(), alreadyAdded.end(),
278  [](bool a) { return a == false; }));
279  if (iCurrent<all.size()) {
280  startSegment = all[iCurrent];
281  inorder.clear();
282  }
283  }
284 
285  for (auto i = 0; i < alreadyAdded.size(); ++i) {
286  if (i != iCurrent && alreadyAdded[i] == false) {
287  if (currentSegment.end() == all[i].begin()) {
288  iCurrent = i;
289  break;
290  }
291  }
292  }
293  }
294 
295  return contour;
296 }
