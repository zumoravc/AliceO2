27 int roundUp(const int numToRound, const int multiple)
28 {
29  if (multiple == 0) {
30  return numToRound;
31  }
32  int remainder{numToRound % multiple};
33  if (remainder == 0) {
34  return numToRound;
35  }
36  return numToRound + multiple - remainder;
37 }

39 int findNearestDivisor(const int numToRound, const int divisor)
40 {
41  if (numToRound > divisor) {
42  return divisor;
43  }
44  int result = numToRound;
45  while (divisor % result != 0) {
46  ++result;
47  }
48 
49  return result;
50 }

82 dim3 Utils::HostHIP::getBlockSize(const int colsNum, const int rowsNum, const int maxThreadsPerBlock)
83 {
84  const DeviceProperties& deviceProperties = ContextHIP::getInstance().getDeviceProperties();
85  int xThreads = std::max(std::min(colsNum, static_cast<int>(deviceProperties.maxThreadsDim.x)), 1);
86  int yThreads = std::max(std::min(rowsNum, static_cast<int>(deviceProperties.maxThreadsDim.y)), 1);
87  const int totalThreads = roundUp(std::min(xThreads * yThreads, maxThreadsPerBlock),
88  static_cast<int>(deviceProperties.warpSize));
89  if (xThreads > yThreads) {
90  xThreads = findNearestDivisor(xThreads, totalThreads);
91  yThreads = totalThreads / xThreads;
92 
93  } else {
94  yThreads = findNearestDivisor(yThreads, totalThreads);
95  xThreads = totalThreads / yThreads;
96  }
97 
98  return dim3{static_cast<unsigned int>(xThreads), static_cast<unsigned int>(yThreads)};
99 }
