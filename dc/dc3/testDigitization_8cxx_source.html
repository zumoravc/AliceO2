59

71  std::vector<o2::mch::Hit> hits(2);
72  hits.at(0) = o2::mch::Hit(trackId1, detElemId1, entrancePoint1, exitPoint1, eloss1, length, tof);

78  o2::mch::mapping::Segmentation seg2{detElemId2};

80 
81  int digitcounter1 = 0;
82  int digitcounter2 = 0;
83  int count = 0;
84 
85  for (auto& digit : digits) {
86 
87  int padid = digit.getPadID();
88  auto label = (mctruthcontainer.getLabels(count))[0];
89  int trackID = label.getTrackID();
90  ++count;
91 
92  if (trackID == trackId1) {
93  bool check = seg1.isValid(digit.getPadID());
94  if (!check)
95  BOOST_FAIL(" digit-pad not belonging to hit det-element-ID ");
96  double padposX = seg1.padPositionX(padid);
97  double padsizeX = seg1.padSizeX(padid);
98  double padposY = seg1.padPositionY(padid);

104  t.MasterToLocal(pos, lpos);
105 
106  BOOST_CHECK_CLOSE(lpos.x(), padposX, padsizeX * 4.0);
107  BOOST_CHECK_CLOSE(lpos.y(), padposY, padsizeY * 10.0);
108  //non uniform pad sizes?
109  digitcounter1++;
110  } else if (trackID == trackId2) {

144 BOOST_AUTO_TEST_CASE(mergingDigitizer)
145 {
146  //merging
147  o2::mch::Digitizer digitizer;
148  int trackId1 = 0;
149  int trackId2 = 1;
150  short detElemId1 = 101;
151  short detElemId2 = 1012;
152  float eloss1 = 1e-6;
153  float eloss2 = 1e-6;
154  float length = 0.f;
155  float tof = 0.0;
156 
157  std::vector<o2::mch::Hit> hits(2);
158  hits.at(0) = o2::mch::Hit(trackId1, detElemId1, entrancePoint1, exitPoint1, eloss1, length, tof);
159  hits.at(1) = o2::mch::Hit(trackId2, detElemId2, entrancePoint2, exitPoint2, eloss2, length, tof);
160 
161  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mctruthcontainer;
162  std::vector<o2::mch::Digit> digits;
163 
164  digitizer.process(hits, digits, mctruthcontainer);
165 
166  std::vector<o2::MCCompLabel> labels = digitizer.getTrackLabels();
167 
168  int rep1 = 9;
169  int rep2 = 9;
170 
171  for (int i = 0; i < rep1; i++) {
172  digits.emplace_back(digits.at(0).getDetID(), digits.at(0).getPadID(), digits.at(0).getADC(), digits.at(0).getTime());
173  labels.emplace_back(labels.at(0).getTrackID(), labels.at(0).getEventID(), labels.at(0).getSourceID(), false);
174  }
175  for (int i = 0; i < rep2; i++) {
176  digits.emplace_back(digits.at(0).getDetID(), digits.at(1).getPadID(), digits.at(1).getADC(), digits.at(1).getTime());
177  labels.emplace_back(labels.at(1).getTrackID(), labels.at(1).getEventID(), labels.at(1).getSourceID(), false);
178  }
179 
180  o2::dataformats::MCTruthContainer<o2::MCCompLabel> container;
181 
182  for (int index = 0; index < labels.size(); ++index) {
183  container.addElement(index, labels.at(index));
184  }
185 
186  digitizer.mergeDigits(digits, container);
187 
188  std::vector<o2::mch::Digit> mergeddigits = digitizer.getDigits();
189  std::vector<o2::MCCompLabel> mergedlabels = digitizer.getTrackLabels();
190 
191  BOOST_CHECK_CLOSE(mergeddigits.at(0).getADC(), digits.at(0).getADC() * (float)(rep1 + 1), digits.at(0).getADC() / 10000.);
192  BOOST_CHECK_CLOSE(mergeddigits.at(1).getADC(), digits.at(1).getADC() * (float)(rep2 + 1), digits.at(1).getADC() / 10000.);
193  BOOST_CHECK_CLOSE((float)mergedlabels.at(0).getTrackID(), (float)labels.at(0).getTrackID(), 0.1);
194  BOOST_CHECK_CLOSE((float)mergedlabels.at(1).getTrackID(), (float)labels.at(1).getTrackID(), 0.1);
195 
196  BOOST_CHECK_CLOSE((float)(digits.size() - rep1 - rep2), (float)mergeddigits.size(), 0.1);
197 
198 } //testing
