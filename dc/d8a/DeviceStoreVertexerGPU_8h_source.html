177 inline std::vector<Tracklet> DeviceStoreVertexerGPU::getDupletsFromGPU(const Order order)
178 {
179  // Careful: this might lead to large allocations, use debug-purpose only
180  std::vector<int> sizes;
181  sizes.resize(constants::its::LayersNumberVertexer);
182  mSizes.copyIntoSizedVector(sizes);
183  std::vector<Tracklet> tmpDuplets;
184  tmpDuplets.resize(static_cast<size_t>(mGPUConf.dupletsCapacity));
185  std::vector<int> nFoundDuplets;
186  nFoundDuplets.resize(sizes[1]);
187  std::vector<Tracklet> shrinkedDuplets;
188 
189  if (order == GPU::Order::fromInnermostToMiddleLayer) {
190  mNFoundDuplets[0].copyIntoSizedVector(nFoundDuplets);
191  mDuplets01.copyIntoSizedVector(tmpDuplets);
192  } else {
193  mDuplets12.copyIntoSizedVector(tmpDuplets);
194  mNFoundDuplets[1].copyIntoSizedVector(nFoundDuplets);
195  }
196  for (int iCluster{0}; iCluster < sizes[1]; ++iCluster) {
197  const int stride{iCluster * mGPUConf.maxTrackletsPerCluster};
198  for (int iDuplet{0}; iDuplet < nFoundDuplets[iCluster]; ++iDuplet) {
199  shrinkedDuplets.push_back(tmpDuplets[stride + iDuplet]);
200  }
201  }
202  return shrinkedDuplets;
203 }

205 inline std::vector<std::array<int, 2>> DeviceStoreVertexerGPU::getDupletIndicesFromGPU()
206 {
207  // Careful: this might lead to large allocations, use debug-purpose only.
208  std::array<std::vector<int>, 2> allowedLines;
209  std::vector<std::array<int, 2>> allowedPairIndices;
210  int nLines = getNExclusiveFoundLines().getElementFromDevice(mClusters[1].getSizeFromDevice() - 1) + getNFoundLines().getElementFromDevice(mClusters[1].getSizeFromDevice() - 1);
211  allowedPairIndices.reserve(nLines);
212  for (int iAllowed{0}; iAllowed < 2; ++iAllowed) {
213  allowedLines[iAllowed].resize(nLines);
214  mDupletIndices[iAllowed].resize(nLines);
215  mDupletIndices[iAllowed].copyIntoSizedVector(allowedLines[iAllowed]);
216  }
217  for (size_t iPair{0}; iPair < allowedLines[0].size(); ++iPair) {
218  allowedPairIndices.emplace_back(std::array<int, 2>{allowedLines[0][iPair], allowedLines[1][iPair]});
219  }
220  return allowedPairIndices;
221 }
