291 std::string bitNumberScale(int n, int nspaces, bool right2left)
292 {
293  std::string line1;
294  std::string line2;
295  for (int i = 0; i < n; i++) {
296  if (i > 0 && i % 10 == 0) {
297  line1 += std::to_string(i / 10);
298  } else {
299  line1 += " ";
300  }
301  line2 += std::to_string(i % 10);
302  }
303 
304  if (right2left) {
305  std::reverse(begin(line1), end(line1));
306  std::reverse(begin(line2), end(line2));
307  }
308  std::string spaces(nspaces, ' ');
309  std::string rv;
310 
311  if (n > 10) {
312  rv = spaces + line1 + "\n";
313  }
314  rv += spaces + line2;
315  return rv;
316 }

391 void compare(std::string_view s1, std::string_view s2)
392 {
393  if (s1.size() != s2.size()) {
394  std::cout << "sizes differ\n";
395  return;
396  }
397  std::vector<int> dif;
398 
399  for (auto i = 0; i < s1.size(); i++) {
400  if (s1[i] != s2[i]) {
401  dif.push_back(i);
402  }
403  }
404  if (dif.size()) {
405  std::cout << "indices of " << dif.size() << " differences:\n";
406  for (auto d : dif) {
407  std::cout << d << " ";
408  }
409  std::cout << "\n";
410  }
411 }
