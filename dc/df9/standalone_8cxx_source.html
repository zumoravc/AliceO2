100 int ReadConfiguration(int argc, char** argv)
101 {
102  int qcRet = qConfigParse(argc, (const char**)argv);
103  if (qcRet) {
104  if (qcRet != qConfig::qcrHelp) {
105  printf("Error parsing command line parameters\n");
106  }
107  return 1;
108  }
109  if (configStandalone.printSettings) {
110  qConfigPrint();
111  }
112 
113 #ifndef _WIN32
114  setlocale(LC_ALL, "");
115  setlocale(LC_NUMERIC, "");
116  if (configStandalone.affinity != -1) {
117  cpu_set_t mask;
118  CPU_ZERO(&mask);
119  CPU_SET(configStandalone.affinity, &mask);
120 
121  printf("Setting affinitiy to restrict on CPU core %d\n", configStandalone.affinity);
122  if (0 != sched_setaffinity(0, sizeof(mask), &mask)) {
123  printf("Error setting CPU affinity\n");
124  return 1;
125  }
126  }
127  if (configStandalone.fifo) {
128  printf("Setting FIFO scheduler\n");
129  sched_param param;
130  sched_getparam(0, &param);
131  param.sched_priority = 1;
132  if (0 != sched_setscheduler(0, SCHED_FIFO, &param)) {
133  printf("Error setting scheduler\n");
134  return 1;
135  }
136  }
137  if (configStandalone.fpe) {
138  feenableexcept(FE_INVALID | FE_DIVBYZERO | FE_OVERFLOW);
139  }
140  if (configStandalone.flushDenormals) {
141  disable_denormals();
142  }
143 
144 #else
145  if (configStandalone.affinity != -1) {
146  printf("Affinity setting not supported on Windows\n");
147  return 1;
148  }
149  if (configStandalone.fifo) {
150  printf("FIFO Scheduler setting not supported on Windows\n");
151  return 1;
152  }
153  if (configStandalone.fpe) {
154  printf("FPE not supported on Windows\n");
155  return 1;
156  }
157 #endif
158 #ifndef HAVE_O2HEADERS
159  configStandalone.configRec.runTRD = configStandalone.configRec.rundEdx = configStandalone.configRec.runCompression = configStandalone.configRec.runTransformation = configStandalone.testSyncAsync = configStandalone.testSync = 0;
160  configStandalone.configRec.ForceEarlyTPCTransform = 1;
161 #endif
162 #ifndef GPUCA_TPC_GEOMETRY_O2
163  configStandalone.configRec.mergerReadFromTrackerDirectly = 0;
164 #endif
165 #ifndef GPUCA_BUILD_QA
166  if (configStandalone.qa || configStandalone.eventGenerator) {
167  printf("QA not enabled in build\n");
168  return 1;
169  }
170 #endif
171  if (configStandalone.qa) {
172  if (getenv("LC_NUMERIC")) {
173  printf("Please unset the LC_NUMERIC env variable, otherwise ROOT will not be able to fit correctly\n"); // BUG: ROOT Problem
174  return 1;
175  }
176  }
177 #ifndef GPUCA_BUILD_EVENT_DISPLAY
178  if (configStandalone.eventDisplay) {
179  printf("EventDisplay not enabled in build\n");
180  return 1;
181  }
182 #endif
183  if (configStandalone.configTF.bunchSim && configStandalone.configTF.nMerge) {
184  printf("Cannot run --MERGE and --SIMBUNCHES togeterh\n");
185  return 1;
186  }
187  if (configStandalone.configTF.bunchSim > 1) {
188  configStandalone.configTF.timeFrameLen = 1.e9 * configStandalone.configTF.bunchSim / configStandalone.configTF.interactionRate;
189  }
190  if (configStandalone.configTF.nMerge) {
191  double len = configStandalone.configTF.nMerge - 1;
192  if (configStandalone.configTF.randomizeDistance) {
193  len += 0.5;
194  }
195  if (configStandalone.configTF.shiftFirstEvent) {
196  len += 0.5;
197  }
198  configStandalone.configTF.timeFrameLen = (len * configStandalone.configTF.averageDistance / GPUReconstructionTimeframe::TPCZ + 1) * GPUReconstructionTimeframe::DRIFT_TIME;
199  }
200  if (configStandalone.configQA.inputHistogramsOnly && configStandalone.configQA.compareInputs.size() == 0) {
201  printf("Can only produce QA pdf output when input files are specified!\n");
202  return 1;
203  }
204  if (configStandalone.eventDisplay) {
205  configStandalone.noprompt = 1;
206  }
207  if (configStandalone.DebugLevel >= 4) {
208  configStandalone.OMPThreads = 1;
209  }
210 
211 #ifdef WITH_OPENMP
212  if (configStandalone.OMPThreads != -1) {
213  omp_set_num_threads(configStandalone.OMPThreads);
214  } else {
215  configStandalone.OMPThreads = omp_get_max_threads();
216  }
217  if (configStandalone.OMPThreads != omp_get_max_threads()) {
218  printf("Cannot set number of OMP threads!\n");
219  return 1;
220  }
221 #else
222  configStandalone.OMPThreads = 1;
223 #endif
224  if (configStandalone.outputcontrolmem) {
225  outputmemory.reset(new char[configStandalone.outputcontrolmem]);
226  }
227 
228 #if !(defined(CUDA_ENABLED) || defined(OPENCL1_ENABLED) || defined(HIP_ENABLED))
229  if (configStandalone.runGPU) {
230  printf("GPU disables at build time!\n");
231  printf("Press a key to exit!\n");
232  getchar();
233  return 1;
234  }
235 #endif
236  return (0);
237 }

492 void OutputStat(GPUChainTracking* t, long long int* nTracksTotal = nullptr, long long int* nClustersTotal = nullptr)
493 {
494  int nTracks = 0, nClusters = 0, nAttachedClusters = 0, nAttachedClustersFitted = 0, nAdjacentClusters = 0;
495  for (int k = 0; k < t->GetTPCMerger().NOutputTracks(); k++) {
496  if (t->GetTPCMerger().OutputTracks()[k].OK()) {
497  nTracks++;
498  nAttachedClusters += t->GetTPCMerger().OutputTracks()[k].NClusters();
499  nAttachedClustersFitted += t->GetTPCMerger().OutputTracks()[k].NClustersFitted();
500  }
501  }
502  for (unsigned int k = 0; k < t->GetTPCMerger().NMaxClusters(); k++) {
503  int attach = t->GetTPCMerger().ClusterAttachment()[k];
504  if (attach & GPUTPCGMMergerTypes::attachFlagMask) {
505  nAdjacentClusters++;
506  }
507  }
508 
509  nClusters = t->GetTPCMerger().NClusters();
510  if (nTracksTotal && nClustersTotal) {
511  *nTracksTotal += nTracks;
512  *nClustersTotal += nClusters;
513  }
514 
515  char trdText[1024] = "";
516  if (t->GetRecoSteps() & GPUReconstruction::RecoStep::TRDTracking) {
517  int nTracklets = 0;
518  for (int k = 0; k < t->GetTRDTracker()->NTracks(); k++) {
519  auto& trk = t->GetTRDTracker()->Tracks()[k];
520  nTracklets += trk.GetNtracklets();
521  }
522  snprintf(trdText, 1024, " - TRD Tracker reconstructed %d tracks (%d tracklets)", t->GetTRDTracker()->NTracks(), nTracklets);
523  }
524  printf("Output Tracks: %d (%d / %d / %d / %d clusters (fitted / attached / adjacent / total))%s\n", nTracks, nAttachedClustersFitted, nAttachedClusters, nAdjacentClusters, t->GetTPCMerger().NMaxClusters(), trdText);
525 }
