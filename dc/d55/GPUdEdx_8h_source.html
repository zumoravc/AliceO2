69 GPUdi() void GPUdEdx::checkSubThresh(int roc)
70 {
71  if (roc != mLastROC) {
72  if (mNSubThresh && mCount + mNSubThresh <= MAX_NCL) {
73  for (int i = 0; i < mNSubThresh; i++) {
74  mChargeTot[mCount] = mSubThreshMinTot;
75  mChargeMax[mCount++] = mSubThreshMinMax;
76  }
77  mNClsROC[mLastROC] += mNSubThresh;
78  mNClsROCSubThresh[mLastROC] += mNSubThresh;
79  }
80  mNSubThresh = 0;
81  mSubThreshMinTot = 1e10f;
82  mSubThreshMinMax = 1e10f;
83  }
84 
85  mLastROC = roc;
86 }

88 GPUdi() void GPUdEdx::fillCluster(float qtot, float qmax, int padRow, float trackSnp, float trackTgl, const GPUParam& GPUrestrict() param, const TPCdEdxCalibrationSplines* splines, float z)
89 {
90  if (mCount >= MAX_NCL) {
91  return;
92  }
93  const int roc = param.tpcGeometry.GetROC(padRow);
94  checkSubThresh(roc);
95  float snp2 = trackSnp * trackSnp;
96  if (snp2 > GPUCA_MAX_SIN_PHI_LOW) {
97  snp2 = GPUCA_MAX_SIN_PHI_LOW;
98  }
99  const float tgl2 = trackTgl * trackTgl;
100  float factor = CAMath::Sqrt((1 - snp2) / (1 + tgl2));
101  factor /= param.tpcGeometry.PadHeight(padRow);
102  qtot *= factor;
103  qmax *= factor;
104 
105  const float sec2 = 1.f / (1.f - snp2);
106  // angleZ local dip angle: z angle - dz/dx (cm/cm)
107  float angleZ = CAMath::Sqrt(tgl2 * sec2);
108  if (angleZ > 3) {
109  angleZ = 3;
110  }
111 
112  const int region = param.tpcGeometry.GetRegion(padRow);
113  z = CAMath::Abs(z);
114 
115  // get the correction for qMax and qTot from the splines for given angle and drift length
116  const float qMaxCorr = splines->interpolateqMax(region, angleZ, z);
117  const float qTotCorr = splines->interpolateqTot(region, angleZ, z);
118  qmax /= qMaxCorr;
119  qtot /= qTotCorr;
120 
121  mChargeTot[mCount] = qtot;
122  mChargeMax[mCount++] = qmax;
123  mNClsROC[roc]++;
124  if (qtot < mSubThreshMinTot) {
125  mSubThreshMinTot = qtot;
126  }
127  if (qmax < mSubThreshMinMax) {
128  mSubThreshMinMax = qmax;
129  }
130 }
