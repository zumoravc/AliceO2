25 bool Clusterizer::loadPreClusters(gsl::span<const PreCluster>& preClusters)
26 {
28 
29  // Loop on pre-clusters in the BP
30  PreClustersDE* de = nullptr;
31  mPreClusters = &preClusters;
32  for (auto& pc : preClusters) {
33  int deIndex = pc.deId;
34  de = &mPreClustersDE[deIndex];
35  de->setDEId(deIndex);
36  size_t idx = &pc - &preClusters[0];
37  if (pc.cathode == 0) {
38  de->getPreClustersBP(pc.firstColumn).push_back({idx, 0, mPreClusterHelper.getArea(pc)});
39  } else {
40  de->getPreClustersNBP().push_back({idx, 0});
41  for (int icolumn = pc.firstColumn; icolumn <= pc.lastColumn; ++icolumn) {
42  de->getPreClustersNBP().back().area[icolumn] = mPreClusterHelper.getArea(icolumn, pc);
43  }
44  }
45 
46  mActiveDEs[deIndex] = true;
47  }
48 
49  return (preClusters.size() > 0);
50 }

72 void Clusterizer::process(gsl::span<const PreCluster> preClusters, gsl::span<const ROFRecord> rofRecords)
73 {
78  mClusters.clear();
79  mROFRecords.clear();
80  for (auto& rofRecord : rofRecords) {
81  mPreClusterOffset = rofRecord.firstEntry;
82  auto firstEntry = mClusters.size();
83  process(preClusters.subspan(rofRecord.firstEntry, rofRecord.nEntries), true);
84  auto nEntries = mClusters.size() - firstEntry;

90 bool Clusterizer::makeClusters(PreClustersDE& pcs)
91 {
93  int deIndex = pcs.getDEId();
94 
95  // loop over pre-clusters in the non-bending plane
96  for (int inb = 0; inb < pcs.getNPreClustersNBP(); ++inb) {
97  auto& pcNB = pcs.getPreClusterNBP(inb);
98  // Try to find matching pre-clusters in the bending plane
99  int icolumn = (*mPreClusters)[pcNB.index].firstColumn;
100  // bool isNBPpaired = false;
101  if (icolumn == (*mPreClusters)[pcNB.index].lastColumn) {
102  // This is the most simple and common case: the NBP pre-cluster is on
103  // one single column. So it can be easily matched with the BP
104  // since the corresponding contours are both rectangles
105  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
106  auto& pcB = pcs.getPreClusterBP(icolumn, ib);
107  makeCluster(pcB.area, pcNB.area[icolumn], icolumn, deIndex);
108  pcB.paired = 1;
109  pcNB.paired = 1;
110  mFunction(mClusters.size() - 1, pcB.index + mPreClusterOffset);
111  mFunction(mClusters.size() - 1, pcNB.index + mPreClusterOffset);
112  }
113  } else {
114  // The NBP pre-cluster spans different columns.
115  // The BP contour is therefore a series of neighbour rectangles
116  // Here we could build a giant cluster with all neighbour pre-clusters
117  // in the bending plane.
118  // However, the pre-cluster in the BP might be close to more than one pre-clusters
119  // in the BP of the neighbour column, thus leading to a pattern with non-contiguous
120  // pre-clusters along the y axis...which are treated as separate clusters
121  // in the standard configuration.
122  // All in all, since the strips are huge, we decided to pair the clusters only between
123  // two adjacent columns.
124  // In this way, a huge cluster spanning three columns will be transformed in
125  // two clusters centered at the border between two columns.
126  // This leads to a redundancy. Still, the sigma of the resulting cluster will be so large
127  // to have no impact whatsoever in the track reconstruction.
128  int pairId = 10 + inb;
129  for (; icolumn <= (*mPreClusters)[pcNB.index].lastColumn; ++icolumn) {
130  int neighColumn = icolumn + 1;
131  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
132  PreClustersDE::BP& pcB = pcs.getPreClusterBP(icolumn, ib);
133  // This function checks for the neighbours only in icolumn+1
134  std::vector<int> neighbours = pcs.getNeighbours(icolumn, ib);
135  // It can happen that the NBP spans two columns...but there are fired strips only on
136  // one column of the BP. In this case we only consider the column with a hit in the BP.
137  // Of course, we need to check that the current pre-cluster was not already paired
138  // with the pre-cluster in the previous column.
139  if (neighbours.empty() && pcB.paired != pairId) {
140  makeCluster(pcB.area, pcNB.area[icolumn], icolumn, deIndex);
141  mFunction(mClusters.size() - 1, pcB.index + mPreClusterOffset);
142  mFunction(mClusters.size() - 1, pcNB.index + mPreClusterOffset);
143  } else {
144  for (auto& jb : neighbours) {
145  PreClustersDE::BP& pcBneigh = pcs.getPreClusterBP(neighColumn, jb);
146  makeCluster(pcB, pcBneigh, pcNB, deIndex);
147  // Here we set the paired flag to a custom ID of the pre-cluster in the NBP
148  // So that, when we move to the next column, we do not add it twice.
149  pcBneigh.paired = pairId;
150  // setPairedFlag(neighColumn, jb, pairId);
151  mFunction(mClusters.size() - 1, pcB.index + mPreClusterOffset);
152  mFunction(mClusters.size() - 1, pcBneigh.index + mPreClusterOffset);
153  mFunction(mClusters.size() - 1, pcNB.index + mPreClusterOffset);
154  }
155  }
156  pcB.paired = 1;
157  pcNB.paired = 1;
158  } // loop on pre-clusters in the BP
159  } // loop on columns
160  }
161 
162  if (pcNB.paired == 0) {
163  // If it is not paired, it means that we have
164  // a monocathodic cluster in the NBP
165  for (int icolumn = (*mPreClusters)[pcNB.index].firstColumn; icolumn <= (*mPreClusters)[pcNB.index].lastColumn; ++icolumn) {
166  makeCluster(pcNB.area[icolumn], pcNB.area[icolumn], icolumn, deIndex);
167  mFunction(mClusters.size() - 1, pcNB.index + mPreClusterOffset);
168  }
169  }
170  } // loop on pre-clusters in the NBP
171 
173  for (int icolumn = 0; icolumn <= 6; ++icolumn) {
174  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
175  PreClustersDE::BP& pcB = pcs.getPreClusterBP(icolumn, ib);
176  if (pcB.paired == 0) {
177  makeCluster(pcB.area, pcB.area, icolumn, deIndex);
178  mFunction(mClusters.size() - 1, pcB.index + mPreClusterOffset);
179  }
180  }
181  }
182 
183  // Reset the pre-clusters before exiting
184  pcs.reset();
185 
186  return true;
187 }

202 void Clusterizer::makeCluster(const MpArea& areaBP, const MpArea& areaNBP, const int& icolumn, const int& deIndex)
203 {
205  float xCoor = 0.5 * (areaNBP.getXmax() + areaNBP.getXmin());
206  float yCoor = 0.5 * (areaBP.getYmax() + areaBP.getYmin());
207  double deltaX = areaNBP.getXmax() - areaNBP.getXmin();
208  double deltaY = areaBP.getYmax() - areaBP.getYmin();
209  float sigmaX2 = deltaX * deltaX / 12;
210  float sigmaY2 = deltaY * deltaY / 12;
211  mClusters.push_back({static_cast<uint8_t>(deIndex), xCoor, yCoor, sigmaX2, sigmaY2});
212 }

215 void Clusterizer::makeCluster(const PreClustersDE::BP& pcBP, const PreClustersDE::BP& pcBPNeigh, const PreClustersDE::NBP& pcNBP, const int& deIndex)
216 {
218  // This is the general case:
219  // perform the full calculation assuming a uniform charge distribution
220 
221  double x2[2][2] = {{0., 0.}, {0., 0.}};
222  double x3[2][2] = {{0., 0.}, {0., 0.}};
223  double dim[2][2];
224  double delta[2];
225  double sumArea = 0.;
226 
227  std::vector<const PreClustersDE::BP*> pcBlist = {&pcBP, &pcBPNeigh};
228 
229  for (auto* pc : pcBlist) {
230  int icolumn = (*mPreClusters)[pc->index].firstColumn;
231  dim[0][0] = pcNBP.area[icolumn].getXmin();
232  dim[0][1] = pcNBP.area[icolumn].getXmax();
233  dim[1][0] = pc->area.getYmin();
234  dim[1][1] = pc->area.getYmax();
235  for (int iplane = 0; iplane < 2; ++iplane) {
236  delta[iplane] = dim[iplane][1] - dim[iplane][0];
237  }
238  // area = dx * dy
239  sumArea += delta[0] * delta[1];
240  for (int iplane = 0; iplane < 2; ++iplane) {
241  for (int ip = 0; ip < 2; ++ip) {
242  // second momentum = x_i * x_i * dy
243  double currX2 = dim[iplane][ip] * dim[iplane][ip] * delta[1 - iplane];
244  x2[iplane][ip] += currX2;
245  // third momentum = x_i * x_i * x_i * dy
246  x3[iplane][ip] += currX2 * dim[iplane][ip];
247  }
248  }
249  } // loop on column
250 
251  double coor[2], sigma2[2];
252  for (int iplane = 0; iplane < 2; ++iplane) {
253  coor[iplane] = (x2[iplane][1] - x2[iplane][0]) / sumArea / 2.;
254  sigma2[iplane] = (x3[iplane][1] - x3[iplane][0]) / sumArea / 3. - coor[iplane] * coor[iplane];
255  }
256 
257  mClusters.push_back({static_cast<uint8_t>(deIndex), static_cast<float>(coor[0]), static_cast<float>(coor[1]), static_cast<float>(sigma2[0]), static_cast<float>(sigma2[1])});
258 }
