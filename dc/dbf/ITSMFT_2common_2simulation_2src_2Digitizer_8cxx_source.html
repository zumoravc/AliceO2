50 void Digitizer::process(const std::vector<Hit>* hits, int evID, int srcID)
51 {
52  // digitize single event, the time must have been set beforehand
53 
54  LOG(INFO) << "Digitizing " << mGeometry->getName() << " hits of entry " << evID << " from source "
55  << srcID << " at time " << mEventTime + mParams.getTimeOffset() << " (TOff.= "
56  << mParams.getTimeOffset() << " ROFrame= " << mNewROFrame << ")"
57  << " cont.mode: " << isContinuous()
58  << " Min/Max ROFrames " << mROFrameMin << "/" << mROFrameMax;
59 
60  // is there something to flush ?
61  if (mNewROFrame > mROFrameMin) {
62  fillOutputContainer(mNewROFrame - 1); // flush out all frame preceding the new one
63  }
64 
65  int nHits = hits->size();
66  std::vector<int> hitIdx(nHits);
67  std::iota(std::begin(hitIdx), std::end(hitIdx), 0);
68  // sort hits to improve memory access
69  std::sort(hitIdx.begin(), hitIdx.end(),
70  [hits](auto lhs, auto rhs) {
71  return (*hits)[lhs].GetDetectorID() < (*hits)[rhs].GetDetectorID();
72  });
73  for (int i : hitIdx) {
74  processHit((*hits)[i], mROFrameMax, evID, srcID);
75  }
76  // in the triggered mode store digits after every MC event
77  // TODO: in the real triggered mode this will not be needed, this is actually for the
78  // single event processing only
79  if (!mParams.isContinuous()) {
80  fillOutputContainer(mROFrameMax);
81  }
82 }

85 void Digitizer::setEventTime(double t)
86 {
87  // assign event time in ns
88  mEventTime = t;
89  // to randomize the RO phase wrt the event time we use a random offset
90  if (mParams.isContinuous()) { // in continuous mode we set the offset only in the very beginning
91  if (!mParams.isTimeOffsetSet()) { // offset is initially at -inf
92  mParams.setTimeOffset(0);
93  }
94  } else { // in the triggered mode we start from 0 ROFrame in every event, is this correct?
95  mParams.setTimeOffset(mEventTime); // + mParams.getROFrameLength() * (gRandom->Rndm() - 0.5));
96  mROFrameMin = 0; // so we reset the frame counters
97  mROFrameMax = 0;
98  }
99 
100  mEventTime -= mParams.getTimeOffset(); // subtract common offset
101  if (mEventTime < 0.) {
102  mEventTime = 0.;
103  } else if (mEventTime > UINT_MAX * mParams.getROFrameLength()) {
104  LOG(FATAL) << "ROFrame for event time " << t << " exceeds allowe maximum " << UINT_MAX;
105  }
106 
107  // RO frame corresponding to provided time
108  mNewROFrame = static_cast<UInt_t>(mEventTime * mParams.getROFrameLengthInv());
109 
110  if (mNewROFrame < mROFrameMin) {
111  LOG(FATAL) << "New ROFrame (time=" << t << ") precedes currently cashed " << mROFrameMin;
112  }
113 
114  if (mParams.isContinuous() && mROFrameMax < mNewROFrame) {
115  mROFrameMax = mNewROFrame - 1; // all frames up to this are finished
116  }
117 }

120 void Digitizer::fillOutputContainer(UInt_t frameLast)
121 {
122  // fill output with digits from min.cached up to requested frame, generating the noise beforehand
123  if (frameLast > mROFrameMax) {
124  frameLast = mROFrameMax;
125  }
126  // make sure all buffers for extra digits are created up to the maxFrame
127  getExtraDigBuffer(mROFrameMax);
128 
129  LOG(INFO) << "Filling " << mGeometry->getName() << " digits output for RO frames " << mROFrameMin << ":"
130  << frameLast;
131 
132  o2::itsmft::ROFRecord rcROF;
133 
134  // we have to write chips in RO increasing order, therefore have to loop over the frames here
135  for (; mROFrameMin <= frameLast; mROFrameMin++) {
136  rcROF.setROFrame(mROFrameMin);
137  rcROF.setFirstEntry(mDigits->size()); // start of current ROF in digits
138 
139  auto& extra = *(mExtraBuff.front().get());
140  for (auto& chip : mChips) {
141  chip.addNoise(mROFrameMin, mROFrameMin, &mParams);
142  auto& buffer = chip.getPreDigits();
143  if (buffer.empty()) {
144  continue;
145  }
146  auto itBeg = buffer.begin();
147  auto iter = itBeg;
148  ULong64_t maxKey = chip.getOrderingKey(mROFrameMin + 1, 0, 0) - 1; // fetch digits with key below that
149  for (; iter != buffer.end(); ++iter) {
150  if (iter->first > maxKey) {
151  break; // is the digit ROFrame from the key > the max requested frame
152  }
153  auto& preDig = iter->second; // preDigit
154  if (preDig.charge >= mParams.getChargeThreshold()) {
155  int digID = mDigits->size();
156  mDigits->emplace_back(chip.getChipIndex(), preDig.row, preDig.col, preDig.charge);
157  mMCLabels->addElement(digID, preDig.labelRef.label);
158  auto& nextRef = preDig.labelRef; // extra contributors are in extra array
159  while (nextRef.next >= 0) {
160  nextRef = extra[nextRef.next];
161  mMCLabels->addElement(digID, nextRef.label);
162  }
163  }
164  }
165  buffer.erase(itBeg, iter);
166  }
167  // finalize ROF record
168  rcROF.setNEntries(mDigits->size() - rcROF.getFirstEntry()); // number of digits
169  rcROF.getBCData().setFromNS(mROFrameMin * mParams.getROFrameLength() + mParams.getTimeOffset());
170  if (mROFRecords) {
171  mROFRecords->push_back(rcROF);
172  }
173  extra.clear(); // clear container for extra digits of the mROFrameMin ROFrame
174  // and move it as a new slot in the end
175  mExtraBuff.emplace_back(mExtraBuff.front().release());
176  mExtraBuff.pop_front();
177  }
178 }

181 void Digitizer::processHit(const o2::itsmft::Hit& hit, UInt_t& maxFr, int evID, int srcID)
182 {
183  // convert single hit to digits
184  double hTime0 = hit.GetTime() * sec2ns;
185  if (hTime0 > 20e3) {
186  const int maxWarn = 10;
187  static int warnNo = 0;
188  if (warnNo < maxWarn) {
189  LOG(WARNING) << "Ignoring hit with time_in_event = " << hTime0 << " ns"
190  << ((++warnNo < maxWarn) ? "" : " (suppressing further warnings)");
191  }
192  return;
193  }
194  hTime0 += mEventTime; // time from the RO start, in ns
195 
196  // calculate RO Frame for this hit
197  if (hTime0 < 0) {
198  hTime0 = 0.;
199  }
200  float tTot = mParams.getSignalShape().getMaxDuration();
201  // frame of the hit signal start
202  UInt_t roFrame = UInt_t(hTime0 * mParams.getROFrameLengthInv());
203  // frame of the hit signal end: in the triggered mode we read just 1 frame
204  UInt_t roFrameMax = mParams.isContinuous() ? UInt_t((hTime0 + tTot) * mParams.getROFrameLengthInv()) : roFrame;
205  int nFrames = roFrameMax + 1 - roFrame;
206  if (roFrameMax > maxFr) {
207  maxFr = roFrameMax; // if signal extends beyond current maxFrame, increase the latter
208  }
209  // delay of the signal start wrt 1st ROF start
210  float timeInROF = float(hTime0 - (roFrame * mParams.getROFrameLength()));
211 
212  // here we start stepping in the depth of the sensor to generate charge diffision
213  float nStepsInv = mParams.getNSimStepsInv();
214  int nSteps = mParams.getNSimSteps();
215  const auto& matrix = mGeometry->getMatrixL2G(hit.GetDetectorID());
216  Vector3D<float> xyzLocS(matrix ^ (hit.GetPosStart())); // start position in sensor frame
217  Vector3D<float> xyzLocE(matrix ^ (hit.GetPos())); // end position in sensor frame
218 
219  Vector3D<float> step(xyzLocE);
220  step -= xyzLocS;
221  step *= nStepsInv; // position increment at each step
222  // the electrons will injected in the middle of each step
223  Vector3D<float> stepH(step * 0.5);
224  xyzLocS += stepH;
225  xyzLocE -= stepH;
226 
227  int rowS = -1, colS = -1, rowE = -1, colE = -1, nSkip = 0;
228  // get entrance pixel row and col
229  while (!Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), rowS, colS)) { // guard-ring ?
230  if (++nSkip >= nSteps) {
231  return; // did not enter to sensitive matrix
232  }
233  xyzLocS += step;
234  }
235  // get exit pixel row and col
236  while (!Segmentation::localToDetector(xyzLocE.X(), xyzLocE.Z(), rowE, colE)) { // guard-ring ?
237  if (++nSkip >= nSteps) {
238  return; // did not enter to sensitive matrix
239  }
240  xyzLocE -= step;
241  }
242  // estimate the limiting min/max row and col where the non-0 response is possible
243  if (rowS > rowE) {
244  std::swap(rowS, rowE);
245  }
246  if (colS > colE) {
247  std::swap(colS, colE);
248  }
249  rowS -= AlpideRespSimMat::NPix / 2;
250  rowE += AlpideRespSimMat::NPix / 2;
251  if (rowS < 0) {
252  rowS = 0;
253  }
254  if (rowE >= Segmentation::NRows) {
255  rowE = Segmentation::NRows - 1;
256  }
257  colS -= AlpideRespSimMat::NPix / 2;
258  colE += AlpideRespSimMat::NPix / 2;
259  if (colS < 0) {
260  colS = 0;
261  }
262  if (colE >= Segmentation::NCols) {
263  colE = Segmentation::NCols - 1;
264  }
265  int rowSpan = rowE - rowS + 1, colSpan = colE - colS + 1; // size of plaquet where some response is expected
266 
267  float respMatrix[rowSpan][colSpan]; // response accumulated here
268  std::fill(&respMatrix[0][0], &respMatrix[0][0] + rowSpan * colSpan, 0.f);
269 
270  float nElectrons = hit.GetEnergyLoss() * mParams.getEnergyToNElectrons(); // total number of deposited electrons
271  nElectrons *= nStepsInv; // N electrons injected per step
272  if (nSkip) {
273  nSteps -= nSkip;
274  }
275  //
276  int rowPrev = -1, colPrev = -1, row, col;
277  float cRowPix = 0.f, cColPix = 0.f; // local coordinated of the current pixel center
278 
279  const o2::itsmft::AlpideSimResponse* resp = mParams.getAlpSimResponse();
280 
281  // take into account that the AlpideSimResponse depth defintion has different min/max boundaries
282  // although the max should coincide with the surface of the epitaxial layer, which in the chip
283  // local coordinates has Y = +SensorLayerThickness/2
284 
285  xyzLocS.SetY(xyzLocS.Y() + resp->getDepthMax() - Segmentation::SensorLayerThickness / 2.);
286 
287  // collect charge in evey pixel which might be affected by the hit
288  for (int iStep = nSteps; iStep--;) {
289  // Get the pixel ID
290  Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), row, col);
291  if (row != rowPrev || col != colPrev) { // update pixel and coordinates of its center
292  if (!Segmentation::detectorToLocal(row, col, cRowPix, cColPix)) {
293  continue; // should not happen
294  }
295  rowPrev = row;
296  colPrev = col;
297  }
298  bool flipCol, flipRow;
299  // note that response needs coordinates along column row (locX) (locZ) then depth (locY)
300  auto rspmat = resp->getResponse(xyzLocS.X() - cRowPix, xyzLocS.Z() - cColPix, xyzLocS.Y(), flipRow, flipCol);
301 
302  xyzLocS += step;
303  if (!rspmat) {
304  continue;
305  }
306 
307  for (int irow = AlpideRespSimMat::NPix; irow--;) {
308  int rowDest = row + irow - AlpideRespSimMat::NPix / 2 - rowS; // destination row in the respMatrix
309  if (rowDest < 0 || rowDest >= rowSpan) {
310  continue;
311  }
312  for (int icol = AlpideRespSimMat::NPix; icol--;) {
313  int colDest = col + icol - AlpideRespSimMat::NPix / 2 - colS; // destination column in the respMatrix
314  if (colDest < 0 || colDest >= colSpan) {
315  continue;
316  }
317  respMatrix[rowDest][colDest] += rspmat->getValue(irow, icol, flipRow, flipCol);
318  }
319  }
320  }
321 
322  // fire the pixels assuming Poisson(n_response_electrons)
323  o2::MCCompLabel lbl(hit.GetTrackID(), evID, srcID, false);
324  auto& chip = mChips[hit.GetDetectorID()];
325 
326  for (int irow = rowSpan; irow--;) {
327  UShort_t rowIS = irow + rowS;
328  for (int icol = colSpan; icol--;) {
329  float nEleResp = respMatrix[irow][icol];
330  if (!nEleResp) {
331  continue;
332  }
333  int nEle = gRandom->Poisson(nElectrons * nEleResp); // total charge in given pixel
334  // ignore charge which have no chance to fire the pixel
335  if (nEle < mParams.getMinChargeToAccount()) {
336  continue;
337  }
338  UShort_t colIS = icol + colS;
339  //
340  registerDigits(chip, roFrame, timeInROF, nFrames, rowIS, colIS, nEle, lbl);
341  }
342  }
343 }

346 void Digitizer::registerDigits(ChipDigitsContainer& chip, UInt_t roFrame, float tInROF, int nROF,
347  UShort_t row, UShort_t col, int nEle, o2::MCCompLabel& lbl)
348 {
349  // Register digits for given pixel, accounting for the possible signal contribution to
350  // multiple ROFrame. The signal starts at time tInROF wrt the start of provided roFrame
351  // In every ROFrame we check the collected signal during strobe
352 
353  float tStrobe = mParams.getStrobeDelay() - tInROF; // strobe start wrt signal start
354  for (int i = 0; i < nROF; i++) {
355  UInt_t roFr = roFrame + i;
356  int nEleROF = mParams.getSignalShape().getCollectedCharge(nEle, tStrobe, tStrobe + mParams.getStrobeLength());
357  tStrobe += mParams.getROFrameLength(); // for the next ROF
358 
359  // discard too small contributions, they have no chance to produce a digit
360  if (nEleROF < mParams.getMinChargeToAccount()) {
361  continue;
362  }
363  if (roFr > mEventROFrameMax)
364  mEventROFrameMax = roFr;
365  if (roFr < mEventROFrameMin)
366  mEventROFrameMin = roFr;
367  auto key = chip.getOrderingKey(roFr, row, col);
368  PreDigit* pd = chip.findDigit(key);
369  if (!pd) {
370  chip.addDigit(key, roFr, row, col, nEleROF, lbl);
371  } else { // there is already a digit at this slot, account as PreDigitExtra contribution
372  pd->charge += nEleROF;
373  if (pd->labelRef.label == lbl) { // don't store the same label twice
374  continue;
375  }
376  ExtraDig* extra = getExtraDigBuffer(roFr);
377  int& nxt = pd->labelRef.next;
378  bool skip = false;
379  while (nxt >= 0) {
380  if ((*extra)[nxt].label == lbl) { // don't store the same label twice
381  skip = true;
382  break;
383  }
384  nxt = (*extra)[nxt].next;
385  }
386  if (skip) {
387  continue;
388  }
389  // new predigit will be added in the end of the chain
390  nxt = extra->size();
391  extra->emplace_back(lbl);
392  }
393  }
