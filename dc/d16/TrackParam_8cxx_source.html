140 Double_t TrackParam::px() const
141 {
143  Double_t pZ;
144  if (TMath::Abs(mParameters(4, 0)) > 0) {
145  Double_t pYZ = (TMath::Abs(mParameters(4, 0)) > 0) ? TMath::Abs(1.0 / mParameters(4, 0)) : FLT_MAX;
146  pZ = -pYZ / (TMath::Sqrt(1.0 + mParameters(3, 0) * mParameters(3, 0))); // spectro. (z<0)
147  } else {
148  pZ = -FLT_MAX / TMath::Sqrt(1.0 + mParameters(3, 0) * mParameters(3, 0) + mParameters(1, 0) * mParameters(1, 0));
149  }
150  return pZ * mParameters(1, 0);
151 }

353 Bool_t TrackParam::isCompatibleTrackParam(const TrackParam& trackParam, Double_t sigma2Cut, Double_t& chi2) const
354 {
359 
360  // reset chi2 value
361  chi2 = 0.;
362 
363  // ckeck covariance matrices
364  if (!mCovariances && !trackParam.mCovariances) {
365  LOG(ERROR) << "Covariance matrix must exist for at least one set of parameters";
366  return kFALSE;
367  }
368 
369  Double_t maxChi2 = 5. * sigma2Cut * sigma2Cut; // 5 degrees of freedom
370 
371  // check Z parameters
372  if (mZ != trackParam.mZ)
373  LOG(WARN) << "Parameters are given at different Z position (" << mZ << " : " << trackParam.mZ
374  << "): results are meaningless";
375 
376  // compute the parameter residuals
377  TMatrixD deltaParam(mParameters, TMatrixD::kMinus, trackParam.mParameters);
378 
379  // build the error matrix
380  TMatrixD weight(5, 5);
381  if (mCovariances)
382  weight += *mCovariances;
383  if (trackParam.mCovariances)
384  weight += *(trackParam.mCovariances);
385 
386  // invert the error matrix to get the parameter weights if possible
387  if (weight.Determinant() == 0) {
388  LOG(ERROR) << "Cannot compute the compatibility chi2";
389  return kFALSE;
390  }
391  weight.Invert();
392 
393  // compute the compatibility chi2
394  TMatrixD tmp(deltaParam, TMatrixD::kTransposeMult, weight);
395  TMatrixD mChi2(tmp, TMatrixD::kMult, deltaParam);
396 
397  // set chi2 value
398  chi2 = mChi2(0, 0);
399 
400  // check compatibility
401  if (chi2 > maxChi2)
402  return kFALSE;
403 
404  return kTRUE;
405 }
