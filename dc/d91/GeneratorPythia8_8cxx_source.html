53  Generator::Init();
54 
56  if (!mConfig.empty()) {
57  if (!mPythia.readFile(mConfig, true)) {
58  LOG(FATAL) << "Failed to init \'Pythia8\': problems with configuration file "

60  return false;

62  }
63 
65  if (!mHooksFileName.empty()) {
66  auto hooks = GetFromMacro<Pythia8::UserHooks*>(mHooksFileName, mHooksFuncName, "Pythia8::UserHooks*", "pythia8_user_hooks");

68  LOG(FATAL) << "Failed to init \'Pythia8\': problem with user hooks configuration ";
69  return false;
70  }
71 #if PYTHIA_VERSION_INTEGER < 8300
72  mPythia.setUserHooksPtr(hooks);
73 #else
74  mPythia.setUserHooksPtr(std::shared_ptr<Pythia8::UserHooks>(hooks));
75 #endif
76  }
77 
78 #if PYTHIA_VERSION_INTEGER < 8300
79 
87  mPythia.readString("HadronLevel:Decay off");
88 #endif
89 
91  if (!mPythia.init()) {
92  LOG(FATAL) << "Failed to init \'Pythia8\': init returned with error";
93  return false;
94  }
95

146 Bool_t
147  GeneratorPythia8::importParticles()
148 {
151  /* loop over particles */
152  // auto weight = mPythia.info.weight(); // TBD: use weights
153  auto nParticles = mPythia.event.size();
154  for (Int_t iparticle = 0; iparticle < nParticles; iparticle++) { // first particle is system
155  auto particle = mPythia.event[iparticle];
156  auto pdg = particle.id();
157  auto st = particle.statusHepMC();
158  auto px = particle.px();
159  auto py = particle.py();
160  auto pz = particle.pz();
161  auto et = particle.e();
162  auto vx = particle.xProd();
163  auto vy = particle.yProd();
164  auto vz = particle.zProd();
165  auto vt = particle.tProd();
166  auto m1 = particle.mother1();
167  auto m2 = particle.mother2();
168  auto d1 = particle.daughter1();
169  auto d2 = particle.daughter2();
170  mParticles.push_back(TParticle(pdg, st, m1, m2, d1, d2, px, py, pz, et, vx, vy, vz, vt));
171  }
172 
174  return kTRUE;
175 }
