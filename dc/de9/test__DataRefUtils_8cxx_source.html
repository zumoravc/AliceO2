52 // Simple test for ROOT container deserialization.
53 BOOST_AUTO_TEST_CASE(TestRootContainerSerialization)
54 {
55  DataRef ref;
56  TMessage* tm = new TMessage(kMESS_OBJECT);
57  TObjArray container;
58  // the original container is explicitly non-owning, its owning
59  // flag is preserved during serialization
60  container.SetOwner(false);
61  auto object = std::make_unique<TObjString>("test");
62  container.Add(object.get());
63  tm->WriteObject(&container);
64  o2::header::DataHeader dh;
65  dh.payloadSerializationMethod = o2::header::gSerializationMethodROOT;
66  ref.payload = tm->Buffer();
67  dh.payloadSize = tm->BufferSize();
68  ref.header = reinterpret_cast<char const*>(&dh);
69 
70  auto s = DataRefUtils::as<TObjArray>(ref);
71  BOOST_REQUIRE(s.get() != nullptr);
72  BOOST_REQUIRE(s->GetEntries() == 1);
73  BOOST_CHECK_EQUAL(std::string(s->At(0)->GetName()), "test");
74  // the extracted object must be owning to avoid memory leaks
75  // the get method takes care of this
76  BOOST_CHECK(s->IsOwner());
77 }
