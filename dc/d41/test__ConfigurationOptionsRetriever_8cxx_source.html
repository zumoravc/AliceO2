42  ConfigParamSpec{"section.key_int", VariantType::Int64, 1ll, {"an int64_t option"}},
43  ConfigParamSpec{"section.key_float", VariantType::Float, 2.0f, {"a float option"}},
44  ConfigParamSpec{"section.key_string", VariantType::String, "foo", {"a string option"}},
45  };
46

48  std::unique_ptr<ParamRetriever> fairmqRetriver{new ConfigurationOptionsRetriever(conf.get(), ".")};

50 
51  ConfigParamStore store{specs, std::move(retrievers)};

59  // We can get nested objects also via their top-level ptree.
60  auto pt = store.store().get_child("section");
