40 o2::framework::DataProcessorSpec getLinkZSToDigitsSpec(int channel, const std::string_view inputDef, std::vector<int> const& tpcSectors)
41 {
42 
43  static constexpr int MaxNumberOfBunches = 3564;
44 
45  struct ProcessAttributes {
46  uint32_t lastOrbit{0};
47  uint32_t firstOrbit{0};
48  uint32_t firstBC{0};
49  uint32_t maxEvents{100};
50  uint32_t processedEvents{0};
51  uint64_t activeSectors{0};
52  bool isContinuous{false};
53  bool quit{false};
54  std::vector<int> tpcSectors{};
55  std::array<std::vector<Digit>, Sector::MAXSECTOR> digitsAll{};
56 
58  void clearDigits()
59  {
60  for (auto& digits : digitsAll) {
61  digits.clear();
62  }
63  }
64 
66  void sortDigits()
67  {
68  // sort digits
69  for (auto& digits : digitsAll) {
70  std::sort(digits.begin(), digits.end(), [](const auto& a, const auto& b) {
71  if (a.getTimeStamp() < b.getTimeStamp())
72  return true;
73  if ((a.getTimeStamp() == b.getTimeStamp()) && (a.getRow() < b.getRow()))
74  return true;
75  return false;
76  });
77  }
78  }
79  };
80 
81  // ===| stateful initialization |=============================================
82  //
83  auto initFunction = [channel, tpcSectors](InitContext& ic) {
84  auto processAttributes = std::make_shared<ProcessAttributes>();
85  // ===| create and set up processing attributes |===
86  {
87  processAttributes->maxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
88  processAttributes->tpcSectors = tpcSectors;
89  }
90 
91  // ===| data processor |====================================================
92  //
93  auto processingFct = [processAttributes, channel](ProcessingContext& pc) {
94  if (processAttributes->quit) {
95  return;
96  }
97 
98  // ===| digit snapshot |===
99  //
100  // lambda that snapshots digits to be sent out;
101  // prepares and attaches header with sector information
102  //
103  auto snapshotDigits = [&pc, processAttributes, channel](std::vector<o2::tpc::Digit> const& digits, int sector) {
104  o2::tpc::TPCSectorHeader header{sector};
105  header.activeSectors = processAttributes->activeSectors;
106  // digit for now are transported per sector, not per lane
107  // pc.outputs().snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe, header},
108  pc.outputs().snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(sector), Lifetime::Timeframe, header},
109  const_cast<std::vector<o2::tpc::Digit>&>(digits));
110  };
111 
112  auto& mapper = Mapper::instance();
113 
114  // loop over all inputs
115  for (auto& input : pc.inputs()) {
116  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
117 
118  // select only RAW data
119  if (dh->dataDescription != o2::header::gDataDescriptionRawData) {
120  continue;
121  }
122 
123  // ===| extract electronics mapping information |===
124  const auto subSpecification = dh->subSpecification;
125  const auto cruID = subSpecification >> 16;
126  const auto linkID = ((subSpecification + (subSpecification >> 8)) & 0xFF) - 1;
127  const auto dataWrapperID = ((subSpecification >> 8) & 0xFF) > 0;
128  const auto globalLinkID = linkID + dataWrapperID * 12;
129  const auto sector = cruID / 10;
130  const CRU cru(cruID);
131  const int fecLinkOffsetCRU = (mapper.getPartitionInfo(cru.partition()).getNumberOfFECs() + 1) / 2;
132  const int fecInPartition = (globalLinkID % 12) + (globalLinkID > 11) * fecLinkOffsetCRU;
133  const int regionIter = cruID % 2;
134 
135  const int sampaMapping[10] = {0, 0, 1, 1, 2, 3, 3, 4, 4, 2};
136  const int channelOffset[10] = {0, 16, 0, 16, 0, 0, 16, 0, 16, 16};
137 
138  processAttributes->activeSectors |= (0x1 << sector);
139 
140  LOGP(debug, "Specifier: {}/{}/{}", dh->dataOrigin.as<std::string>(), dh->dataDescription.as<std::string>(), dh->subSpecification);
141  LOGP(debug, "Payload size: {}", dh->payloadSize);
142  LOGP(debug, "CRU: {}; linkID: {}; dataWrapperID: {}; globalLinkID: {}", cruID, linkID, dataWrapperID, globalLinkID);
143 
144  try {
145  o2::framework::RawParser parser(input.payload, dh->payloadSize);
146 
147  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
148  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV4>();
149  if (!rdhPtr) {
150  break;
151  }
152 
153  const auto& rdh = *rdhPtr;
154 
155  // ===| only accept physics triggers |===
156  if (rdh.triggerType != 0x10) {
157  continue;
158  }
159 
160  // ===| event handling |===
161  //
162  // really ugly, better treatment required extension in DPL
163  // events are are detected by close by orbit numbers
164  //
165  const auto hbOrbit = rdh.heartbeatOrbit;
166  const auto lastOrbit = processAttributes->lastOrbit;
167 
168  if (!processAttributes->firstOrbit) {
169  processAttributes->firstOrbit = hbOrbit;
170  if (!processAttributes->isContinuous) {
171  processAttributes->firstBC = rdh.heartbeatBC;
172  }
173  }
174 
175  const auto globalBCoffset = ((hbOrbit - processAttributes->firstOrbit) * MaxNumberOfBunches - processAttributes->firstBC); // To be calculated
176 
177  if ((lastOrbit > 0) && (hbOrbit > (lastOrbit + 3))) {
178  ++processAttributes->processedEvents;
179  LOG(INFO) << fmt::format("Number of processed events: {} ({})", processAttributes->processedEvents, processAttributes->maxEvents);
180  processAttributes->sortDigits();
181 
182  // publish digits of all configured sectors
183  for (auto isector : processAttributes->tpcSectors) {
184  snapshotDigits(processAttributes->digitsAll[isector], isector);
185  }
186  processAttributes->clearDigits();
187 
188  processAttributes->activeSectors = 0;
189  if (processAttributes->processedEvents >= processAttributes->maxEvents) {
190  LOGP(info, "Maximum number of events reached ({}), no more processing will be done", processAttributes->maxEvents);
191  processAttributes->quit = true;
192  pc.services().get<ControlService>().endOfStream();
193  //pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
194  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
195  break;
196  }
197  }
198 
199  processAttributes->lastOrbit = hbOrbit;
200  const auto size = it.size();
201  auto data = it.data();
202  LOGP(debug, "Raw data block payload size: {}", size);
203 
204  // cast raw data pointer to link based zero suppression definition
205  zerosupp_link_based::ContainerZS* zsdata = (zerosupp_link_based::ContainerZS*)data;
206  const zerosupp_link_based::ContainerZS* const zsdataEnd = (zerosupp_link_based::ContainerZS*)(data + size);
207 
208  while (zsdata < zsdataEnd) {
209  const auto channelBits = zsdata->cont.header.getChannelBits();
210  const uint32_t numberOfWords = zsdata->cont.header.numWordsPayload;
211  assert((channelBits.count() - 1) / 10 == numberOfWords - 1);
212 
213  std::size_t processedChannels = 0;
214  for (std::size_t ichannel = 0; ichannel < channelBits.size(); ++ichannel) {
215  if (!channelBits[ichannel]) {
216  continue;
217  }
218 
219  // adc value
220  const auto adcValue = zsdata->getADCValueFloat(processedChannels);
221 
222  // pad mapping
223  // TODO: verify the assumptions of the channel mapping!
224  // assumes the following sorting (s_chn is the channel on the sampa),
225  // in this case for even regiona (lower half fec)
226  // chn# SAMPA s_chn
227  // 0 0 0
228  // 1 0 1
229  // 2 0 16
230  // 3 0 17
231  // 4 1 0
232  // 5 1 1
233  // 6 1 16
234  // 7 1 17
235  // 8 2 0
236  // 9 2 1
237  //
238  // 10 0 2
239  // 11 0 3
240  // 12 0 18
241  // 13 0 19
242  // 14 1 2
243  // 15 1 3
244  // 16 1 18
245  // 17 1 19
246  // 18 2 2
247  // 19 2 3
248  //
249  // 20 0 4
250  // 21 0 5
251  // 22 0 20
252  // 23 0 21
253  // ...
254  // For the uneven regions (upper half fec), the sampa ordering
255  // is 3, 3, 3, 3, 4, 4, 4, 4, 2, 2
256  const int istreamm = ((ichannel % 10) / 2);
257  const int partitionStream = istreamm + regionIter * 5;
258  const int sampaOnFEC = sampaMapping[partitionStream];
259  const int channel = (ichannel % 2) + 2 * (ichannel / 10);
260  const int channelOnSAMPA = channel + channelOffset[partitionStream];
261 
262  const auto padSecPos = mapper.padSecPos(cru, fecInPartition, sampaOnFEC, channelOnSAMPA);
263  const auto& padPos = padSecPos.getPadPos();
264  int timebin = (globalBCoffset + zsdata->cont.header.bunchCrossing) / 8; // To be calculated
265 
266  // add digit
267  processAttributes->digitsAll[sector].emplace_back(cruID, adcValue, padPos.getRow(), padPos.getPad(), timebin);
268  ++processedChannels;
269  }
270 
271  // go to next time bin
272  zsdata = zsdata->next();
273  }
274  }
275 
276  } catch (const std::runtime_error& e) {
277  LOG(ERROR) << "can not create raw parser form input data";
278  o2::header::hexDump("payload", input.payload, dh->payloadSize, 64);
279  LOG(ERROR) << e.what();
280  }
281  }
282  };
283 
284  return processingFct;
285  };
286 
287  std::stringstream id;
288  id << "TPCDigitizer" << channel;
289 
290  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
291  for (auto isector : tpcSectors) {
292  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
293  }
294 
295  return DataProcessorSpec{
296  id.str().c_str(),
297  select(inputDef.data()),
298  outputs,
299  AlgorithmSpec{initFunction},
300  Options{
301  {"max-events", VariantType::Int, 100, {"maximum number of events to process"}},
302  {"pedestal-file", VariantType::String, "", {"file with pedestals and noise for zero suppression"}}}};
303 }
304 } // namespace tpc
305 } // namespace o2
