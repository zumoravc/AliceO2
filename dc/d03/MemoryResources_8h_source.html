75  MessageResource& operator=(MessageResource&&) = default;
76  MessageResource(FairMQMessagePtr message)
77  : mUpstream{message->GetTransport()->GetMemoryResource()},
78  mMessageSize{message->GetSize()},
79  mMessageData{mUpstream ? mUpstream->setMessage(std::move(message))
80  : throw std::runtime_error("MessageResource::MessageResource upstream is nullptr")}
81  {
82  }
83  FairMQMessagePtr getMessage(void* p) override { return mUpstream->getMessage(p); }
84  void* setMessage(FairMQMessagePtr message) override { return mUpstream->setMessage(std::move(message)); }
85  FairMQTransportFactory* getTransportFactory() noexcept override { return nullptr; }
86  size_t getNumberOfMessages() const noexcept override { return mMessageData ? 1 : 0; }
87 
88  protected:
89  FairMQMemoryResource* mUpstream{nullptr};
90  size_t mMessageSize{0};
91  void* mMessageData{nullptr};
92  bool initialImport{true};
93 
94  void* do_allocate(std::size_t bytes, std::size_t alignment) override
95  {
96  if (initialImport) {
97  if (bytes > mMessageSize) {
98  throw std::bad_alloc();
99  }
100  initialImport = false;
101  return mMessageData;
102  } else {
103  return mUpstream->allocate(bytes, alignment);
104  }
105  }
106  void do_deallocate(void* p, std::size_t bytes, std::size_t alignment) override
107  {
108  mUpstream->deallocate(p, bytes, alignment);
109  return;
110  }
111  bool do_is_equal(const memory_resource& other) const noexcept override
112  {
113  // since this uniquely owns the message it can never be equal to anybody else
114  return false;
115  }
116 };

119 // A spectator pmr memory resource which only watches the memory of the underlying buffer, does not
120 // carry out real allocation. It owns the underlying buffer which is destroyed on deallocation.
121 template <typename BufferType>
122 class SpectatorMemoryResource : public boost::container::pmr::memory_resource
123 {
124  public:
125  using buffer_type = BufferType;
126 
127  SpectatorMemoryResource() noexcept = delete;
128  SpectatorMemoryResource(const SpectatorMemoryResource&) noexcept = delete;
129  SpectatorMemoryResource(SpectatorMemoryResource&&) noexcept = default;
130  SpectatorMemoryResource& operator=(const SpectatorMemoryResource&) = delete;
131  SpectatorMemoryResource& operator=(SpectatorMemoryResource&&) = default;
132  ~SpectatorMemoryResource() noexcept override = default;
133 
134  // the resource is the pointer managed by unique_ptr
135  template <typename T>
136  SpectatorMemoryResource(std::unique_ptr<T, typename buffer_type::deleter_type>&& buffer, size_t size)
137  : mBuffer{std::move(buffer)}, mPointer{mBuffer.get()}, mSize{size}
138  {
139  }
140 
141  // the resource is the data of the vector managed by unique ptr
142  template <typename T>
143  SpectatorMemoryResource(std::unique_ptr<std::vector<T>, typename buffer_type::deleter_type>&& buffer)
144  : mBuffer{std::move(buffer)}, mPointer{mBuffer->data()}, mSize{mBuffer->size() * sizeof(T)}
145  {
146  }
147 
148  // TODO: the underlying resource can be directly the vector or the read only buffer
149  protected:
150  void* do_allocate(std::size_t bytes, std::size_t alignment) override
151  {
152  if (mSize > 0) {
153  if (bytes > mSize) {
154  throw std::bad_alloc();
155  }
156  mSize = 0;
157  return mPointer;
158  }
159  throw std::runtime_error("Can not allocate: this memory resource is only supposed to provide spectator access to external buffer");
160  }
161 
162  void do_deallocate(void* p, std::size_t bytes, std::size_t alignment) override
163  {
164  if (p == mPointer) {
165  mBuffer.reset();
166  mPointer = nullptr;
167  } else if (mPointer == nullptr) {
168  // there is an error in the logic flow, this should never be called more than once
169  throw std::logic_error("underlying controlled resource has been released already");
170  } else {
171  throw std::logic_error("this resource can only deallocate the controlled resource pointer");
172  }
173  }
174  bool do_is_equal(const memory_resource& other) const noexcept override
175  {
176  // uniquely owns the underlying resource, can never be equal to any other instance
177  return false;
178  }
179 
180  private:
181  buffer_type mBuffer;
182  void* mPointer = nullptr;
183  size_t mSize = 0;
184 };

187 // This in general (as in STL) is a bad idea, but here it is safe to inherit from an allocator since we
188 // have no additional data and only override some methods so we don't get into slicing and other problems.
189 template <typename T>
190 class SpectatorAllocator : public boost::container::pmr::polymorphic_allocator<T>
191 {
192  public:
193  using boost::container::pmr::polymorphic_allocator<T>::polymorphic_allocator;
194  using propagate_on_container_move_assignment = std::true_type;
195 
196  // skip default construction of empty elements
197  // this is important for two reasons: one: it allows us to adopt an existing buffer (e.g. incoming message) and
198  // quickly construct large vectors while skipping the element initialization.
199  template <class U>
200  void construct(U*)
201  {
202  }
203 
204  // dont try to call destructors, makes no sense since resource is managed externally AND allowed
205  // types cannot have side effects
206  template <typename U>
207  void destroy(U*)
208  {
209  }
210 
211  T* allocate(size_t size) { return reinterpret_cast<T*>(this->resource()->allocate(size * sizeof(T), 0)); }
212  void deallocate(T* ptr, size_t size)
213  {
214  this->resource()->deallocate(const_cast<typename std::remove_cv<T>::type*>(ptr), size);
215  }
216 };
