151 TGeoCompositeShape* getRoundedShape(const char* name, float halfThickness, float xPos, float yPos, float radius)
152 {
154 
155  // create the position where the hole will be created
156  const char* shiftName = Form("%sX%.1fY%.1fShift", name, TMath::Abs(xPos), TMath::Abs(yPos));
157  auto shift = new TGeoTranslation(shiftName, xPos, yPos, 0.);
158  shift->RegisterYourself();
159 
160  // create the tube that create the hole
161  const char* tubeName = Form("%sR%.1fHole", name, radius);
162  new TGeoTube(tubeName, 0., radius, halfThickness);
163 
164  const char* shapeName = Form("%sX%.1fY%.1fR%.1fShape", name, TMath::Abs(xPos), TMath::Abs(yPos), radius);
165 
166  // create the hole in the box and return the built shape
167  return new TGeoCompositeShape(shapeName, Form("%sBox-%s:%s", name, tubeName, shiftName));
168 }

191 void createPCBs()
192 {
194 
199 
200  // Define some necessary variables
201  string bendName, nonbendName;
202  float y = 0., shift = 0., length = 0.; // useful parameters for dimensions and positions
203  int nDualSampas = 0;
204 
205  // get the DualSampa volume
206  auto* dualSampaVol = getDualSampa();
207 
208  for (const auto& [pcbName, dualSampas] : kPcbTypes) { // loop over the PCB types of the array
209 
210  auto name = (const char*)pcbName.data();
211 
212  auto pcb = new TGeoVolumeAssembly(name);
213 
214  // Reset shift variables
215  float gasShift = 0., pcbShift = 0.;
216 
217  float gasLength = kGasLength;
218  float pcbLength = kPCBLength;
219 
220  int numb = pcbName[1] - '0'; // char -> int conversion
221 
222  // change the variables according to the PCB shape if necessary
223  switch (pcbName.front()) {
224  case 'R': // rounded
225  numb = pcbName.back() - '0';
226  gasLength = (numb == 1) ? kR1PCBLength : kGasLength;
227  pcbLength = (numb == 1) ? kR1PCBLength : kRoundedPCBLength;
228  gasShift = gasLength - kGasLength;
229  pcbShift = pcbLength - kPCBLength;
230 
231  bendName = Form("%sB", name);
232  nonbendName = Form("%sN", name);
233  break;
234  case 'S': // shortened
235  gasLength = kShortPCBLength;
236  pcbLength = kShortPCBLength;
237  bendName = Form("S2B%c", pcbName.back());
238  nonbendName = Form("S2N%c", pcbName.back());
239  break;
240  default: // normal
241  bendName = (numb == 3) ? pcbName.substr(0, 3) : pcbName.substr(0, 2);
242  nonbendName = (numb == 3) ? pcbName.substr(3) : pcbName.substr(2);
243  }
244 
245  float borderLength = pcbLength;
246 
247  // create the volume of each material (a box by default)
248  // sensitive gas
249  auto gas = new TGeoVolume(Form("%s gas", name), new TGeoBBox(Form("%sGasBox", name), gasLength / 2., kGasHalfHeight, kGasHalfThickness), assertMedium(Medium::Gas));
250 
251  float x = pcbLength / 2;
252  float halfHeight = kPCBHalfHeight;
253 
254  // bending cathode
255  auto bend = new TGeoVolume(bendName.data(), new TGeoBBox(Form("%sBendBox", name), x, halfHeight, kCathodeHalfThickness), assertMedium(Medium::Copper));
256 
257  // non-bending cathode
258  auto nonbend = new TGeoVolume(nonbendName.data(), new TGeoBBox(Form("%sNonBendBox", name), x, halfHeight, kCathodeHalfThickness), assertMedium(Medium::Copper));
259 
260  // insulating material
261  auto insu = new TGeoVolume(Form("%s insulator", name), new TGeoBBox(Form("%sInsuBox", name), x, halfHeight, kInsuHalfThickness), assertMedium(Medium::FR4));
262 
263  // bottom spacer (noryl)
264  auto spacer = new TGeoVolume(Form("%s bottom spacer", name), new TGeoBBox(Form("%sBottomSpacerBox", name), x, kHoriSpacerHalfHeight, kSpacerHalfThickness), assertMedium(Medium::Noryl));
265 
266  // change the volume shape if we are creating a rounded PCB
267  if (isRounded(pcbName)) {
268  // LHC beam pipe radius ("R3" -> it is a slat of a station 4 or 5)
269  float radius = (numb == 3) ? kSt45Radius : kSt3Radius;
270  // y position of the PCB center w.r.t the beam pipe shape
271  switch (numb) {
272  case 1:
273  y = 0.; // central for "R1"
274  break;
275  case 2:
276  y = kSt3RoundedSlatYPos; // "R2" -> station 3
277  break;
278  default:
279  y = kSt45RoundedSlatYPos; // "R3" -> station 4 or 5
280  break;
281  }
282  // compute the radius of curvature of the PCB we want to create
283  float curvRad = radius + 2 * kRoundedSpacerHalfLength;
284 
285  x = kRoundedPCBLength - gasLength / 2 - kVertSpacerHalfLength;
286  gas->SetShape(getRoundedShape(Form("%sGas", name), kGasHalfThickness, x, -y, curvRad));
287 
288  x = kRoundedPCBLength - pcbLength / 2 - kVertSpacerHalfLength;
289 
290  bend->SetShape(getRoundedShape(Form("%sBend", name), kCathodeHalfThickness, x, -y, curvRad));
291 
292  nonbend->SetShape(getRoundedShape(Form("%sNonBend", name), kCathodeHalfThickness, x, -y, curvRad));
293 
294  insu->SetShape(getRoundedShape(Form("%sInsu", name), kInsuHalfThickness, x, -y, curvRad));
295 
296  if (pcbName.back() != '1') { // change the bottom spacer and border shape for "R2" and "R3" PCBs
297 
298  spacer->SetShape(getRoundedShape(Form("%sBottomSpacer", name), kSpacerHalfThickness, x, -y + kGasHalfHeight + kHoriSpacerHalfHeight, radius));
299 
300  borderLength -= (pcbName.back() == '3') ? curvRad : curvRad + kRoundedSpacerHalfLength;
301  }
302  }
303 
305  float halfThickness = kGasHalfThickness;
306  pcb->AddNode(gas, 1, new TGeoTranslation(gasShift / 2, 0., 0.));
307  float z = halfThickness;
308 
309  halfThickness = kCathodeHalfThickness;
310  x = pcbShift / 2;
311  z += halfThickness;
312  pcb->AddNode(bend, 1, new TGeoTranslation(x, 0., z));
313  pcb->AddNode(nonbend, 2, new TGeoTranslation(x, 0., -z));
314  z += halfThickness;
315 
316  halfThickness = kInsuHalfThickness;
317  z += halfThickness;
318  pcb->AddNode(insu, 1, new TGeoTranslation(x, 0., z));
319  pcb->AddNode(insu, 2, new TGeoTranslation(x, 0., -z));
320  z += halfThickness;
321 
322  // the horizontal spacers
323  y = kGasHalfHeight + kHoriSpacerHalfHeight;
324  pcb->AddNode(gGeoManager->GetVolume(Form("Top spacer %.2f long", pcbLength)), 1, new TGeoTranslation(x, y, 0.));
325  pcb->AddNode(spacer, 1, new TGeoTranslation(x, -y, 0.));
326 
327  // the borders
328  y = kPCBHalfHeight - kBorderHalfHeight;
329  pcb->AddNode(gGeoManager->GetVolume(Form("Top border %.2f long", pcbLength)), 1, new TGeoTranslation(x, y, 0.));
330  x = (pcbShift - pcbLength + borderLength) / 2;
331  pcb->AddNode(gGeoManager->MakeBox(Form("%s bottom border", name), assertMedium(Medium::Rohacell),
332  borderLength / 2, kBorderHalfHeight, kBorderHalfThickness),
333  1, new TGeoTranslation(x, -y, 0.));
334 
335  // the DualSampa read-out cards
336  halfThickness = kDualSampaHalfThickness;
337  z += halfThickness;
338  for (int i = 0; i < dualSampas.size(); i++) {
339 
340  nDualSampas = dualSampas[i];
341  length = (i % 2) ? borderLength : pcbLength;
342  shift = (i % 2) ? TMath::Power(-1, isRounded(pcbName)) * (pcbLength - borderLength) : 0.;
343  y = TMath::Power(-1, i % 2) * kDualSampaYPos;
344  z = TMath::Power(-1, i / 2) * TMath::Abs(z);
345 
346  for (int j = 0; j < nDualSampas; j++) {
347  pcb->AddNode(dualSampaVol, 100 * i + j,
348  new TGeoTranslation((j - nDualSampas / 2) * (length / nDualSampas) - (nDualSampas % 2 - 1) * (length / (2 * nDualSampas)) +
349  (pcbShift + shift) / 2,
350  y, z));
351  }
352  } // end of the MANUs loop
353  } // end of the PCBs loop
354 }

357 void createSlats()
358 {
361 
362  const auto kSpacerMed = assertMedium(Medium::Noryl);
363  auto leftSpacer = gGeoManager->GetVolume("Left spacer");
364 
365  // Mirror rotation for the slat panel on the non-bending side
366  auto mirror = new TGeoRotation();
367  mirror->ReflectZ(true);
368 
369  for (const auto& [typeName, pcbVector] : kSlatTypes) {
370 
371  auto name = (const char*)typeName.data(); // slat name (easier to name volumes)
372 
373  // create the slat volume assembly
374  auto slat = new TGeoVolumeAssembly(name);
375 
376  // Reset slat variables
377  float direction = TMath::Power(-1, isRounded(typeName)); // rounded and "normal" slats are built in opposite x-axis direction (mapping convention)
378  float halfLength = 2 * 2 * kVertSpacerHalfLength; // vertical spacers
379  float center = (pcbVector.size() - 1) * kGasLength / 2;
380  float panelShift = 0.;
381  int iVol = 0;
382 
383  // loop over the number of PCBs in the current slat
384  for (const auto& pcb : pcbVector) {
385 
386  float gasLength = kGasLength;
387  float pcbLength = kPCBLength;
388 
389  switch (pcb.front()) {
390  case 'R':
391  pcbLength = (pcb.back() == '1') ? kR1PCBLength : kRoundedPCBLength;
392  panelShift += pcbLength - kRoundedPCBLength;
393  break;
394  case 'S':
395  pcbLength = kShortPCBLength;
396  gasLength = kShortPCBLength;
397  panelShift += direction * (pcbLength - kPCBLength);
398  break;
399  }
400 
401  halfLength += pcbLength;
402 
403  // place the corresponding PCB volume in the slat and correct the origin of the slat
404  slat->AddNode(gGeoManager->GetVolume(pcb.data()), iVol + 1,
405  new TGeoTranslation(iVol * kPCBLength - direction * (kPCBLength - gasLength) / 2 - center, 0, 0));
406  iVol++;
407 
408  } // end of the PCBs loop
409  panelShift /= 2;
410  halfLength /= 2;
411 
412  // compute the LV cable length
413  float cableHalfLength = (typeName.find('3') < typeName.size()) ? kSt45SupportHalfLength : kCh5SupportHalfLength;
414  if (typeName == "122200N")
415  cableHalfLength = kCh6SupportHalfLength;
416  cableHalfLength -= halfLength;
417  if (typeName == "122330N")
418  cableHalfLength -= kGasLength / 2;
419 
420  float rightSpacerHalfHeight = kSlatPanelHalfHeight;
421  if (isRounded(typeName)) {
422  halfLength -= kVertSpacerHalfLength; // don't count the vertical spacer length twice in the case of rounded slat
423  rightSpacerHalfHeight = kGasHalfHeight; // to avoid overlaps with the horizontal spacers
424  }
425 
426  // right vertical spacer
427  auto rightSpacer = new TGeoVolume(Form("%s right spacer", name),
428  new TGeoBBox(Form("%sRightSpacerBox", name), kVertSpacerHalfLength, rightSpacerHalfHeight, kSpacerHalfThickness), kSpacerMed);
429 
430  // glue a slat panel on each side of the PCBs
431  auto panel = new TGeoVolumeAssembly(Form("%s panel", name));
432 
433  float x = halfLength;
434  float halfHeight = kSlatPanelHalfHeight;
435 
436  // glue
437  auto glue = new TGeoVolume(Form("%s panel glue", name), new TGeoBBox(Form("%sGlueBox", name), x, halfHeight, kGlueHalfThickness), assertMedium(Medium::Glue));
438 
439  // nomex (bulk)
440  auto nomexBulk = new TGeoVolume(Form("%s panel nomex (bulk)", name), new TGeoBBox(Form("%sNomexBulkBox", name), x, halfHeight, kNomexBulkHalfThickness), assertMedium(Medium::BulkNomex));
441 
442  // carbon fiber
443  auto carbon = new TGeoVolume(Form("%s panel carbon fiber", name), new TGeoBBox(Form("%sCarbonBox", name), x, halfHeight, kCarbonHalfThickness), assertMedium(Medium::Carbon));
444 
445  // nomex (honeycomb)
446  auto nomex = new TGeoVolume(Form("%s panel nomex (honeycomb)", name), new TGeoBBox(Form("%sNomexBox", name), x, halfHeight, kNomexHalfThickness), assertMedium(Medium::HoneyNomex));
447 
448  // change the volume shape if we are creating a rounded slat
449  if (isRounded(typeName)) {
450 
451  // LHC beam pipe radius ("NR3" -> it is a slat of a station 4 or 5)
452  float radius = (typeName.back() == '3') ? kSt45Radius : kSt3Radius;
453 
454  // extreme angle values for the rounded spacer
455  float angMin = 90., angMax = 180.;
456 
457  // position of the slat center w.r.t the beam pipe center
458  x = halfLength - kVertSpacerHalfLength;
459  float y = 0.;
460  float xRoundedPos = x + panelShift;
461 
462  // change the LV cable length for st.3 slats
463  cableHalfLength = (isShort(typeName)) ? kCh5SupportHalfLength : kCh6SupportHalfLength;
464 
465  // change the above values if necessary
466  switch (typeName.back()) { // get the last character
467  case '1': // central for "S(N)R1"
468  y = 0.;
469  x = 2 * kPCBLength - panelShift + kVertSpacerHalfLength;
470  xRoundedPos = kRoundedPCBLength + kPCBLength - kRoundedSpacerHalfLength;
471  angMin = angMax - TMath::RadToDeg() * TMath::ACos((kRoundedPCBLength - kR1PCBLength - kRoundedSpacerHalfLength) / radius);
472  angMax = -angMin;
473  break;
474  case '2': // "S(N)R2" -> station 3
475  y = kSt3RoundedSlatYPos;
476  angMax = angMax - TMath::RadToDeg() * TMath::ASin((y - kSlatPanelHalfHeight) / (radius + kRoundedSpacerHalfLength));
477  break;
478  default: // "NR3" -> station 4 or 5
479  y = kSt45RoundedSlatYPos;
480  angMax = angMax - TMath::RadToDeg() * TMath::ASin((y - kSlatPanelHalfHeight) / (radius + kRoundedSpacerHalfLength));
481  cableHalfLength = kSt45SupportHalfLength;
482  break;
483  }
484 
485  cableHalfLength -= (x + halfLength) / 2;
486 
487  // create and place the rounded spacer
488  slat->AddNode(gGeoManager->MakeTubs(Form("%s rounded spacer", name), kSpacerMed, radius,
489  radius + 2 * kRoundedSpacerHalfLength, kSpacerHalfThickness, angMin, angMax),
490  1, new TGeoTranslation(xRoundedPos, -y, 0.));
491 
492  glue->SetShape(getRoundedShape(Form("%sGlue", name), kGlueHalfThickness, x, -y, radius));
493 
494  nomexBulk->SetShape(getRoundedShape(Form("%sNomexBulk", name), kNomexBulkHalfThickness, x, -y, radius));
495 
496  carbon->SetShape(getRoundedShape(Form("%sCarbon", name), kCarbonHalfThickness, x, -y, radius));
497 
498  nomex->SetShape(getRoundedShape(Form("%sNomex", name), kNomexHalfThickness, x, -y, radius));
499 
500  rightSpacer->SetShape(getRoundedShape(Form("%sRightSpacer", name), kSpacerHalfThickness, 0., -y, radius));
501 
502  } // end of the "rounded" condition
503 
504  // place all the layers in the slat panel volume assembly (the panel origin is on the glue edge !)
505 
506  float halfThickness = kGlueHalfThickness;
507  float z = halfThickness;
508  panel->AddNode(glue, 1, new TGeoTranslation(0., 0., z));
509  z += halfThickness;
510 
511  halfThickness = kNomexBulkHalfThickness;
512  z += halfThickness;
513  panel->AddNode(nomexBulk, 1, new TGeoTranslation(0., 0., z));
514  z += halfThickness;
515 
516  halfThickness = kGlueHalfThickness;
517  z += halfThickness;
518  panel->AddNode(glue, 2, new TGeoTranslation(0., 0., z));
519  z += halfThickness;
520

522  z += halfThickness;

626 void buildHalfChambers(TGeoVolume& topVolume)
627 {
630 
631  // read the json containing all the necessary parameters to place the slat volumes in the half-chambers
632  StringStream is(jsonSlatDescription.c_str());
633 
634  Document doc;
635  doc.ParseStream(is);
636 
637  // get the "half-chambers" array
638  Value& hChs = doc["HalfChambers"];
639  assert(hChs.IsArray());
640 
641  // mother volume to place the half-chambers
642  auto motherVolume = &topVolume;
643 
644  // loop over the objects (half-chambers) of the array
645  for (const auto& halfCh : hChs.GetArray()) {
646  // check that "halfCh" is an object
647  if (!halfCh.IsObject())
648  throw runtime_error("Can't create the half-chambers : wrong Value input");
649 
650  int moduleID = halfCh["moduleID"].GetInt();
651  const string name = halfCh["name"].GetString();
652  // get the chamber number (if the chamber name has a '0' at the 3rd digit, take the number after; otherwise it's the
653  // chamber 10)
654  int nCh = (name.find('0') == 2) ? name[3] - '0' : 10;
655 
656  auto halfChVol = new TGeoVolumeAssembly(name.data());
657 
658  // place the support panel corresponding to the chamber number
659  auto supRot = new TGeoRotation();
660  if (moduleID % 2)
661  supRot->RotateY(180.);
662  halfChVol->AddNode(gGeoManager->GetVolume(Form("Chamber %d support panel", nCh)), moduleID, supRot);
663 
664  // place the slat volumes on the different nodes of the half-chamber
665  for (const auto& slat : halfCh["nodes"].GetArray()) {
666  // check that "slat" is an object
667  if (!slat.IsObject())
668  throw runtime_error("Can't create the slat : wrong Value input");
669 
670  int detID = slat["detID"].GetInt();
671 
672  // place the slat on the half-chamber volume
673  halfChVol->AddNode(
674  gGeoManager->GetVolume(slat["type"].GetString()), detID,
675  new TGeoCombiTrans(slat["position"][0].GetDouble(), slat["position"][1].GetDouble(),
676  slat["position"][2].GetDouble(),
677  new TGeoRotation(Form("Slat%drotation", detID), slat["rotation"][0].GetDouble(),
678  slat["rotation"][1].GetDouble(), slat["rotation"][2].GetDouble(),
679  slat["rotation"][3].GetDouble(), slat["rotation"][4].GetDouble(),
680  slat["rotation"][5].GetDouble())));
681 
682  } // end of the node loop
683 
684  Double_t zpos = halfCh["position"][2].GetDouble();
685 
686  // change the mother volume if the Dipole and/or 'YOUT2' are present in the geometry
687  if ((nCh == 5 || nCh == 6) && gGeoManager->GetVolume("Dipole")) {
688  motherVolume = gGeoManager->GetVolume("DDIP");
689  zpos *= -1.; // z > 0 convention for dipole volumes
690  }
691 
692  if (nCh > 6 && gGeoManager->GetVolume("YOUT2")) {
693  motherVolume = gGeoManager->GetVolume("YOUT2");
694  }
695 
696  // place the half-chamber in the mother volume
697  motherVolume->AddNode(
698  halfChVol, moduleID,
699  new TGeoCombiTrans(halfCh["position"][0].GetDouble(), halfCh["position"][1].GetDouble(), zpos,
700  new TGeoRotation(Form("%srotation", name.data()), halfCh["rotation"][0].GetDouble(),
701  halfCh["rotation"][1].GetDouble(), halfCh["rotation"][2].GetDouble(),
702  halfCh["rotation"][3].GetDouble(), halfCh["rotation"][4].GetDouble(),
703  halfCh["rotation"][5].GetDouble())));
704 
705  } // end of the half-chambers loop
706 }
