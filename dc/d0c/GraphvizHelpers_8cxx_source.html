39 void GraphvizHelpers::dumpDeviceSpec2Graphviz(std::ostream& out, const std::vector<DeviceSpec>& specs)
40 {
41  out << R"GRAPHVIZ(digraph structs {
42  node[shape=record]
43 )GRAPHVIZ";
44  std::map<std::string, std::string> outputChannel2Device;
45  std::map<std::string, unsigned int> outputChannel2Port;
46 
47  for (auto& spec : specs) {
48  auto id = spec.id;
49  out << " " << quote(id) << R"( [label="{{)";
50  bool firstInput = true;
51  for (auto&& input : spec.inputChannels) {
52  if (firstInput == false) {
53  out << "|";
54  }
55  firstInput = false;
56  out << "<" << input.name << ">" << input.name;
57  }
58  out << "}|";
59  auto totalChannels = spec.inputChannels.size() +
60  spec.outputChannels.size();
61  out << id << "(" << totalChannels << ")";
62  out << "|{";
63  bool firstOutput = true;
64  for (auto&& output : spec.outputChannels) {
65  outputChannel2Device.insert(std::make_pair(output.name, id));
66  outputChannel2Port.insert(std::make_pair(output.name, output.port));
67  if (firstOutput == false) {
68  out << "|";
69  }
70  firstOutput = false;
71  out << "<" << output.name << ">" << output.name;
72  }
73  out << R"(}}"];)"
74  << "\n";
75  }
76  for (auto& spec : specs) {
77  for (auto& input : spec.inputChannels) {
78  // input and output name are now the same
79  auto outputName = input.name;
80  out << " " << quote(outputChannel2Device[outputName]) << ":" << quote(outputName) << "-> " << quote(spec.id)
81  << ":" << quote(input.name) << R"( [label=")" << input.port << R"(")"
82  << "]\n";
83  }
84  }
85  out << "}\n";
86 }
