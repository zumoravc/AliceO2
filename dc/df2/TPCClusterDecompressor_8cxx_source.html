32  std::vector<ClusterNative> clusters[NSLICES][GPUCA_ROW_COUNT];
33  unsigned int offset = 0;
34  for (unsigned int i = 0; i < clustersCompressed->nTracks; i++) {
35  unsigned int slice = clustersCompressed->sliceA[i];
36  unsigned int row = clustersCompressed->rowA[i];
37  GPUTPCCompressionTrackModel track;
38  for (unsigned int j = 0; j < clustersCompressed->nTrackClusters[i]; j++) {
39  unsigned int pad = 0, time = 0;
40  if (j) {
41  unsigned char tmpSlice = clustersCompressed->sliceLegDiffA[offset - i - 1];
42  bool changeLeg = (tmpSlice >= NSLICES);
43  if (changeLeg) {
44  tmpSlice -= NSLICES;
45  }
46  if (clustersCompressed->nComppressionModes & 2) {
47  slice += tmpSlice;
48  if (slice >= NSLICES) {
49  slice -= NSLICES;
50  }
51  row += clustersCompressed->rowDiffA[offset - i - 1];
52  if (row >= GPUCA_ROW_COUNT) {
53  row -= GPUCA_ROW_COUNT;
54  }
55  } else {
56  slice = tmpSlice;
57  row = clustersCompressed->rowDiffA[offset - i - 1];
58  }
59  if (changeLeg && track.Mirror()) {
60  offset += clustersCompressed->nTrackClusters[i] - j;
61  break;
62  }
63  if (track.Propagate(param.tpcGeometry.Row2X(row), param.SliceParam[slice].Alpha)) {
64  offset += clustersCompressed->nTrackClusters[i] - j;
65  break;
66  }
67  unsigned int timeTmp = clustersCompressed->timeResA[offset - i - 1];
68  if (timeTmp & 800000) {
69  timeTmp |= 0xFF000000;
70  }
71  time = timeTmp + ClusterNative::packTime(CAMath::Max(0.f, param.tpcGeometry.LinearZ2Time(slice, track.Z())));
72  float tmpPad = CAMath::Max(0.f, CAMath::Min((float)param.tpcGeometry.NPads(GPUCA_ROW_COUNT - 1), param.tpcGeometry.LinearY2Pad(slice, row, track.Y())));
73  pad = clustersCompressed->padResA[offset - i - 1] + ClusterNative::packPad(tmpPad);
74  } else {
75  time = clustersCompressed->timeA[i];
76  pad = clustersCompressed->padA[i];
77  }
78  std::vector<ClusterNative>& clusterVector = clusters[slice][row];
79  clusterVector.emplace_back(time, clustersCompressed->flagsA[offset], pad, clustersCompressed->sigmaTimeA[offset], clustersCompressed->sigmaPadA[offset], clustersCompressed->qMaxA[offset], clustersCompressed->qTotA[offset]);
80  float y = param.tpcGeometry.LinearPad2Y(slice, row, clusterVector.back().getPad());
81  float z = param.tpcGeometry.LinearTime2Z(slice, clusterVector.back().getTime());
82  if (j == 0) {
83  track.Init(param.tpcGeometry.Row2X(row), y, z, param.SliceParam[slice].Alpha, clustersCompressed->qPtA[i], param);
84  }
85  if (j + 1 < clustersCompressed->nTrackClusters[i] && track.Filter(y, z, row)) {
86  offset += clustersCompressed->nTrackClusters[i] - j;
87  break;
88  }
89  offset++;
90  }
91  }
92  clusterBuffer.resize(clustersCompressed->nAttachedClusters + clustersCompressed->nUnattachedClusters);
93  unsigned int offsets[NSLICES][GPUCA_ROW_COUNT];
94  offset = 0;
95  for (unsigned int i = 0; i < NSLICES; i++) {
96  for (unsigned int j = 0; j < GPUCA_ROW_COUNT; j++) {
97  clustersNative.nClusters[i][j] = clusters[i][j].size() + clustersCompressed->nSliceRowClusters[i * GPUCA_ROW_COUNT + j];
98  offsets[i][j] = offset;
99  offset += clustersCompressed->nSliceRowClusters[i * GPUCA_ROW_COUNT + j];
100  }
101  }

104  GPUCA_OPENMP(parallel for)
105  for (unsigned int i = 0; i < NSLICES; i++) {

107  int offset2 = offsets[i][j];
108  ClusterNative* buffer = &clusterBuffer[clustersNative.clusterOffset[i][j]];
109  if (clusters[i][j].size()) {

111  }
112  unsigned int time = 0;
113  unsigned short pad = 0;
114  ClusterNative* cl = buffer + clusters[i][j].size();
115  for (unsigned int k = 0; k < clustersCompressed->nSliceRowClusters[i * GPUCA_ROW_COUNT + j]; k++) {
116  if (clustersCompressed->nComppressionModes & 2) {
117  unsigned int timeTmp = clustersCompressed->timeDiffU[offset2];
118  if (timeTmp & 800000) {
119  timeTmp |= 0xFF000000;
120  }
121  time += timeTmp;
122  pad += clustersCompressed->padDiffU[offset2];
123  } else {
124  time = clustersCompressed->timeDiffU[offset2];
125  pad = clustersCompressed->padDiffU[offset2];
126  }
127  *(cl++) = ClusterNative(time, clustersCompressed->flagsU[offset2], pad, clustersCompressed->sigmaTimeU[offset2], clustersCompressed->sigmaPadU[offset2], clustersCompressed->qMaxU[offset2], clustersCompressed->qTotU[offset2]);
128  }
129  std::sort(buffer, buffer + clustersNative.nClusters[i][j]);
130  }
131  }
132
