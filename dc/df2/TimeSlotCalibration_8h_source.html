82 template <typename Input, typename Container>
83 bool TimeSlotCalibration<Input, Container>::process(TFType tf, const gsl::span<const Input> data)
84 {
85  int maxDelay = mMaxSlotsDelay * mSlotLength;
86  // if (tf<mLastClosedTF || (!mSlots.empty() && getSlot(0).getTFStart() > tf + maxDelay)) { // ignore TF
87  if (tf < mLastClosedTF || (!mSlots.empty() && getLastSlot().getTFStart() > tf + maxDelay)) { // ignore TF
88  LOG(INFO) << "Ignoring TF " << tf;
89  return false;
90  }
91 
92  // check if some slots are done
93  checkSlotsToFinalize(tf, maxDelay);
94 
95  // process current TF
96  auto& slotTF = getSlotForTF(tf);
97  slotTF.getContainer()->fill(data);
98 
99  return true;
100 }

103 template <typename Input, typename Container>
104 void TimeSlotCalibration<Input, Container>::checkSlotsToFinalize(TFType tf, int maxDelay)
105 {
106  // Check which slots can be finalized, provided the newly arrived TF is tf
107  // check if some slots are done
108  for (auto slot = mSlots.begin(); slot != mSlots.end(); slot++) {
109  if ((slot->getTFEnd() + maxDelay) < tf) {
110  if (hasEnoughData(*slot)) {
111  finalizeSlot(*slot); // will be removed after finalization
112  } else if ((slot + 1) != mSlots.end()) {
113  LOG(INFO) << "Merging underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd()
114  << " to slot " << (slot + 1)->getTFStart() << " <= TF <= " << (slot + 1)->getTFEnd();
115  (slot + 1)->mergeToPrevious(*slot);
116  } else {
117  LOG(INFO) << "Discard underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
118  break; // slot has no enough stat. and there is no other slot to merge it to
119  }
120  mLastClosedTF = slot->getTFEnd() + 1; // do not accept any TF below this
121  LOG(INFO) << "closing slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
122  mSlots.erase(slot);
123  } else {
124  break;
125  }
126  if (mSlots.empty()) { // since erasing the very last entry may invalidate mSlots.end()
127  break;
128  }
129  }
130 }

133 template <typename Input, typename Container>
134 inline TFType TimeSlotCalibration<Input, Container>::tf2SlotMin(TFType tf) const
135 {
136  if (tf < mFirstTF) {
137  throw std::runtime_error("invalide TF");
138  }
139  return TFType((tf - mFirstTF) / mSlotLength) * mSlotLength + mFirstTF;
140 }

143 template <typename Input, typename Container>
144 TimeSlot<Container>& TimeSlotCalibration<Input, Container>::getSlotForTF(TFType tf)
145 {
146  if (!mSlots.empty() && mSlots.front().getTFStart() > tf) {
147  auto tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1);
148  auto tftgt = tf2SlotMin(tf);
149  while (tfmn >= tftgt) {
150  LOG(INFO) << "Adding new slot for " << tfmn << " <= TF <= " << tfmn + mSlotLength - 1;
151  emplaceNewSlot(true, tfmn, tfmn + mSlotLength - 1);
152  if (!tfmn) {
153  break;
154  }
155  tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1);
156  }
157  return mSlots[0];
158  }
159  for (auto it = mSlots.begin(); it != mSlots.end(); it++) {
160  auto rel = (*it).relateToTF(tf);
161  if (rel == 0) {
162  return (*it);
163  }
164  }
165  // need to add in the end
166  auto tfmn = mSlots.empty() ? tf2SlotMin(tf) : tf2SlotMin(mSlots.back().getTFEnd() + 1);
167  do {
168  LOG(INFO) << "Adding new slot for " << tfmn << " <= TF <= " << tfmn + mSlotLength - 1;
169  emplaceNewSlot(false, tfmn, tfmn + mSlotLength - 1);
170  tfmn = tf2SlotMin(mSlots.back().getTFEnd() + 1);
171  } while (tf > mSlots.back().getTFEnd());
172 
173  return mSlots.back();
174 }
