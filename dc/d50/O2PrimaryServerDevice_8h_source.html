154  bool ConditionalRun() override
155  {
156  auto& channel = fChannels.at("primary-get").at(0);
157  std::unique_ptr<FairMQMessage> request(channel.NewMessage());
158  auto bytes = channel.Receive(request);
159  if (bytes < 0) {
160  LOG(ERROR) << "Some error occurred on socket during receive";
161  return true; // keep going
162  }
163  return HandleRequest(request, 0);

167  bool HandleRequest(FairMQMessagePtr& request, int /*index*/)
168  {
169  LOG(INFO) << "GOT A REQUEST WITH SIZE " << request->GetSize();
170  std::string requeststring(static_cast<char*>(request->GetData()), request->GetSize());
171 
172  if (requeststring.compare("configrequest") == 0) {
173  return HandleConfigRequest(request);
174  }
175 
176  else if (requeststring.compare("primrequest") != 0) {
177  LOG(INFO) << "unknown request\n";
178  return true;
179  }
180 
181  static int counter = 0;
182  if (counter >= mMaxEvents && mNeedNewEvent) {
183  return false;
184  }
185 
186  LOG(INFO) << "Received request for work ";
187  if (mNeedNewEvent) {
188  // we need a newly generated event now
189  if (mGeneratorThread.joinable()) {
190  mGeneratorThread.join();
191  }
192  mNeedNewEvent = false;
193  mPartCounter = 0;
194  counter++;
195  }
196 
197  auto& prims = mStack.getPrimaries();
198  auto numberofparts = (int)std::ceil(prims.size() / (1. * mChunkGranularity));
199  // number of parts should be at least 1 (even if empty)
200  numberofparts = std::max(1, numberofparts);
201 
202  o2::data::PrimaryChunk m;
203  o2::data::SubEventInfo i;
204  i.eventID = counter;
205  i.maxEvents = mMaxEvents;
206  i.part = mPartCounter + 1;
207  i.nparts = numberofparts;
208  i.seed = counter + mInitialSeed;
209  i.index = m.mParticles.size();
210  i.mMCEventHeader = mEventHeader;
211  m.mSubEventInfo = i;
212 
213  int endindex = prims.size() - mPartCounter * mChunkGranularity;
214  int startindex = prims.size() - (mPartCounter + 1) * mChunkGranularity;
215  if (startindex < 0) {
216  startindex = 0;
217  }
218  if (endindex < 0) {
219  endindex = 0;
220  }
221 
222  for (int index = startindex; index < endindex; ++index) {
223  m.mParticles.emplace_back(prims[index]);
224  }
225 
226  LOG(INFO) << "Sending " << m.mParticles.size() << " particles";
227  LOG(INFO) << "treating ev " << counter << " part " << i.part << " out of " << i.nparts;
228 
229  // feedback to driver if new event started
230  if (mPipeToDriver != -1 && i.part == 1) {
231  if (write(mPipeToDriver, &counter, sizeof(counter))) {
232  }
233  }
234 
235  mPartCounter++;
236  if (mPartCounter == numberofparts) {
237  mNeedNewEvent = true;
238  // start generation of a new event
239  mGeneratorThread = std::thread(&O2PrimaryServerDevice::generateEvent, this);
240  }
241 
242  TMessage* tmsg = new TMessage(kMESS_OBJECT);
243  tmsg->WriteObjectAny((void*)&m, TClass::GetClass("o2::data::PrimaryChunk"));
244 
245  auto free_tmessage = [](void* data, void* hint) { delete static_cast<TMessage*>(hint); };
246 
247  std::unique_ptr<FairMQMessage> message(
248  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
249 
250  // send answer
251  TStopwatch timer;
252  timer.Start();
253  auto code = Send(message, "primary-get", 0, 5000); // we introduce timeout in order not to block other requests
254  timer.Stop();
255  auto time = timer.CpuTime();
256  if (code > 0) {
257  LOG(INFO) << "Reply send in " << time << "s";
258  return true;
259  } else {
260  LOG(WARN) << "Sending process had problems. Return code : " << code << " time " << time << "s";
261  }
262  return true;
263  }
