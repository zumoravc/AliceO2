59 void MatchTPCITS::printABTracksTree(const ABTrackLinksList& llist) const
60 {
61  // dump all hypotheses
62  if (llist.lowestLayer == NITSLayers) {
63  printf("No matches\n");
64  return;
65  }
66  o2::MCCompLabel lblTrc;
67  if (mTPCTrkLabels) {
68  lblTrc = mTPCTrkLabels->getLabels(mTPCWork[llist.trackID].sourceID)[0];
69  }
70  LOG(INFO) << "Matches for track " << llist.trackID << " lowest lr: " << int(llist.lowestLayer) << " " << lblTrc
71  << " pT= " << mTPCWork[llist.trackID].getPt();
72 
73  auto printHyp = [this, &lblTrc](int nextHyp, int cnt) {
74  printf("#%d Lr/IC/Lnk/ClID/Chi2/Chi2Nrm/{MC}:%c ", cnt++, mTPCTrkLabels ? (lblTrc.isFake() ? 'F' : 'C') : '-');
75  int nFakeClus = 0, nTotClus = 0, parID = nextHyp; // print particular hypothesis
76  while (1) {
77  const auto& lnk = mABTrackLinks[parID];
78  int mcEv = -1, mcTr = -1;
79  if (lnk.clID > MinusOne && mITSClsLabels) {
80  nTotClus++;
81  const auto lab = mITSClsLabels->getLabels(lnk.clID)[0];
82  if (lab.isValid()) {
83  mcEv = lab.getEventID();
84  mcTr = lab.getTrackID();
85  if (mcEv != lblTrc.getEventID() || mcTr != lblTrc.getTrackID()) {
86  nFakeClus++;
87  }
88  } else {
89  mcEv = mcTr = -999; // noise
90  nFakeClus++;
91  }
92  } else if (lnk.isDummyTop() && mTPCTrkLabels) { // top layer, use TPC MC lbl
93  mcEv = lblTrc.getEventID();
94  mcTr = lblTrc.getTrackID();
95  }
96  printf("[%d/%d/%d/%d/%6.2f/%6.2f/{%d/%d}]", lnk.layerID, lnk.icCandID, parID, lnk.clID, lnk.chi2, lnk.chi2Norm(), mcEv, mcTr);
97  if (lnk.isDummyTop()) { // reached dummy seed on the dummy layer above the last ITS layer
98  break;
99  }
100  parID = lnk.parentID;
101  }
102  printf(" NTot:%d NFakes:%d\n", nTotClus, nFakeClus);
103  };
104 
105  int cnt = 0; // tmp
106  for (int lowest = llist.lowestLayer; lowest <= mParams->ABRequireToReachLayer; lowest++) {
107  int nextHyp = llist.firstInLr[lowest];
108  while (nextHyp > MinusOne) {
109  if (mABTrackLinks[nextHyp].nDaughters == 0) { // select only head links, w/o daughters
110  printHyp(nextHyp, cnt++);
111  }
112  nextHyp = mABTrackLinks[nextHyp].nextOnLr;
113  }
114  }
115 
116  if (llist.bestOrdLinkID > MinusOne) { // print best matches list
117  int next = llist.bestOrdLinkID;
118  LOG(INFO) << "Best matches:";
119  int cnt = 0;
120  while (next > MinusOne) {
121  const auto& lnkOrd = mABBestLinks[next];
122  int nextHyp = lnkOrd.trackLinkID;
123  printHyp(nextHyp, cnt++);
124  next = lnkOrd.nextLinkID;
125  refitABTrack(nextHyp);
126  }
127  }
128 }

131 void MatchTPCITS::dumpABTracksDebugTree(const ABTrackLinksList& llist)
132 {
133  // dump all hypotheses
134  static int entCnt = 0;
135  if (llist.lowestLayer == NITSLayers) {
136  return;
137  }
138  LOG(INFO) << "Dump AB Matches for track " << llist.trackID;
139  o2::MCCompLabel lblTrc;
140  if (mTPCTrkLabels) {
141  lblTrc = mTPCTrkLabels->getLabels(mTPCWork[llist.trackID].sourceID)[0]; // tmp
142  }
143  int ord = 0;
144  for (int lowest = llist.lowestLayer; lowest <= mParams->ABRequireToReachLayer; lowest++) {
145  int nextHyp = llist.firstInLr[lowest];
146  while (nextHyp > MinusOne) {
147  if (mABTrackLinks[nextHyp].nDaughters != 0) { // select only head links, w/o daughters
148  nextHyp = mABTrackLinks[nextHyp].nextOnLr;
149  continue;
150  }
151  // fill debug AB track
152  ABDebugTrack dbgTrack;
153  int parID = nextHyp; // print particular hypothesis
154  while (1) {
155  const auto& lnk = mABTrackLinks[parID];
156  if (lnk.clID > MinusOne) {
157  auto& dbgLnk = dbgTrack.links.emplace_back();
158 #ifdef _ALLOW_DEBUG_AB_
159  dbgLnk.seed = lnk.seed; // seed before update
160 #endif
161  dbgLnk.clLabel = mITSClsLabels->getLabels(lnk.clID)[0];
162  dbgLnk.chi2 = lnk.chi2;
163  dbgLnk.lr = lnk.layerID;
164  (o2::BaseCluster<float>&)dbgLnk = mITSClustersArray[lnk.clID];
165  dbgTrack.nClusITS++;
166  if (lblTrc.getEventID() == dbgLnk.clLabel.getEventID() && std::abs(lblTrc.getTrackID()) == dbgLnk.clLabel.getTrackID()) {
167  dbgTrack.nClusITSCorr++;
168  }
169  }
170  if (lnk.isDummyTop()) { // reached dummy seed on the dummy layer above the last ITS layer
171  dbgTrack.tpcSeed = lnk; // tpc seed
172  dbgTrack.trackID = llist.trackID;
173  dbgTrack.tpcLabel = lblTrc;
174  dbgTrack.icCand = lnk.icCandID;
175  dbgTrack.icTimeBin = mInteractions[lnk.icCandID].timeBins;
176  const auto& tpcTrOrig = mTPCTracksArray[mTPCWork[llist.trackID].sourceID];
177  unsigned nclTPC = tpcTrOrig.getNClusterReferences();
178  dbgTrack.nClusTPC = nclTPC > 0xff ? 0xff : nclTPC;
179  dbgTrack.sideAC = tpcTrOrig.hasASideClusters() + (tpcTrOrig.hasCSideClusters() ? 2 : 0);
180  break;
181  }
182  parID = lnk.parentID;
183  }
184  dbgTrack.chi2 = dbgTrack.links.front().chi2;
185  // at the moment links contain cumulative chi2, convert to track-to-cluster chi2
186  for (int i = 0; i < dbgTrack.nClusITS - 1; i++) {
187  dbgTrack.links[i].chi2 -= dbgTrack.links[i + 1].chi2;
188  }
189  // at the moment the links are stored from inner to outer layers, invert this order
190  for (int i = dbgTrack.nClusITS / 2; i--;) {
191  std::swap(dbgTrack.links[i], dbgTrack.links[dbgTrack.nClusITS - i - 1]);
192  }
193  dbgTrack.valid = ord == 0 && llist.isValidated();
194  dbgTrack.order = ord++;
195  // dump debug track
196  (*mDBGOut) << "abtree"
197  << "trc=" << dbgTrack << "\n";
198  entCnt++;
199  nextHyp = mABTrackLinks[nextHyp].nextOnLr;
200  }
201  } // loop over layers
202 }

260  return;

264  doMatching(sec);
265  }
266  mTimer[SWDoMatching].Stop();
267  if (0) { // enabling this creates very verbose output
268  mTimer[SWTot].Stop();
269  printCandidatesTPC();
270  printCandidatesITS();

272  }
273 
274  selectBestMatches();
275 
276  refitWinners();
277 
278  if (Params::Instance().runAfterBurner) {
279  runAfterBurner();
280  }
281 
282 #ifdef _ALLOW_DEBUG_TREES_
283  if (mDBGOut && isDebugFlag(WinnerMatchesTree)) {
284  dumpWinnerMatches();
285  }
286  mDBGOut.reset();
287 #endif
288 
289  gSystem->GetProcInfo(&procInfoStop);
290  mTimer[SWTot].Stop();
291 
292  for (int i = 0; i < NStopWatches; i++) {
293  LOGF(INFO, "Timing for %15s: Cpu: %.3e Real: %.3e s in %d slots", TimerName[i], mTimer[i].CpuTime(), mTimer[i].RealTime(), mTimer[i].Counter() - 1);

405 //______________________________________________
406 bool MatchTPCITS::validateTPCMatch(int iTPC)
407 {
408  const auto& tTPC = mTPCWork[iTPC];
409  auto& rcTPC = mMatchRecordsTPC[tTPC.matchID]; // best TPC->ITS match
410  /* // should never happen
411  if (rcTPC.nextRecID == Validated) {
412  LOG(WARNING) << "TPC->ITS was already validated";
413  return false; // RS do we need this
414  }
415  */
416  // check if it is consistent with corresponding ITS->TPC match
417  auto& tITS = mITSWork[rcTPC.partnerID]; // partner ITS track
418  auto& rcITS = mMatchRecordsITS[tITS.matchID]; // best ITS->TPC match record
419  if (rcITS.nextRecID == Validated) {
420  return false;
421  }
422  if (rcITS.partnerID == iTPC) { // is best matching TPC track for this ITS track actually iTPC?
423  // unlink winner TPC track from all ITS candidates except winning one
424  int nextTPC = rcTPC.nextRecID;
425  while (nextTPC > MinusOne) {
426  auto& rcTPCrem = mMatchRecordsTPC[nextTPC];
427  removeTPCfromITS(iTPC, rcTPCrem.partnerID); // remove references on mtID from ITS match=rcTPCrem.partnerID
428  nextTPC = rcTPCrem.nextRecID;
429  }
430  rcTPC.nextRecID = Validated;
431  int itsWinID = rcTPC.partnerID;
432 
433  // unlink winner ITS match from all TPC matches using it
434  int nextITS = rcITS.nextRecID;
435  while (nextITS > MinusOne) {
436  auto& rcITSrem = mMatchRecordsITS[nextITS];
437  removeITSfromTPC(itsWinID, rcITSrem.partnerID); // remove references on itsWinID from TPC match=rcITSrem.partnerID
438  nextITS = rcITSrem.nextRecID;
439  }
440  rcITS.nextRecID = Validated;
441  return true;
442  }
443  return false;
444 }

471 //______________________________________________
472 bool MatchTPCITS::prepareTPCTracks()
473 {
475  mTimer[SWPrepTPC].Start(false);
476  mMatchRecordsTPC.clear();
477 
478  int ntr = mTPCTracksArray.size();
479  // number of records might be actually more than N tracks!
480  mMatchRecordsTPC.reserve(mMatchRecordsTPC.size() + mParams->maxMatchCandidates * ntr);
481 
482  // copy the track params, propagate to reference X and build sector tables
483  mTPCWork.clear();
484  mTPCWork.reserve(ntr);
485  if (mMCTruthON) {
486  mTPCLblWork.clear();
487  mTPCLblWork.reserve(ntr);
488  }
489  for (int sec = o2::constants::math::NSectors; sec--;) {
490  mTPCSectIndexCache[sec].clear();
491  mTPCSectIndexCache[sec].reserve(100 + 1.2 * ntr / o2::constants::math::NSectors);
492  mTPCTimeBinStart[sec].clear();
493  }
494 
495  for (int it = 0; it < ntr; it++) {
496 
497  const auto& trcOrig = mTPCTracksArray[it];
498 
499  // make sure the track was propagated to inner TPC radius at the ref. radius
500  if (trcOrig.getX() > XTPCInnerRef + 0.1 || trcOrig.getPt() < mParams->minTPCPt) {
501  continue;
502  }
503  // discard tracks with too few clusters
504  if (trcOrig.getNClusterReferences() < mParams->minTPCClusters) {
505  continue;
506  }
507  // discard tracks w/o certain number of innermost pads (last cluster is innermost one
508  {
509  uint8_t clSect = 0, clRow = 0;
510  uint32_t clIdx = 0;
511  trcOrig.getClusterReference(mTPCTrackClusIdx, trcOrig.getNClusterReferences() - 1, clSect, clRow, clIdx);
512  if (clRow > mParams->askMinTPCRow) {
513  continue;
514  }
515  }
516 
517  // create working copy of track param
518  mTPCWork.emplace_back(static_cast<const o2::track::TrackParCov&>(trcOrig), it);
519  auto& trc = mTPCWork.back();
520  // propagate to matching Xref
521  if (!propagateToRefX(trc)) {
522  mTPCWork.pop_back(); // discard track whose propagation to XMatchingRef failed
523  continue;
524  }
525  if (mMCTruthON) {
526  mTPCLblWork.emplace_back(mTPCTrkLabels->getLabels(it)[0]);
527  }
528 
529  float time0 = trcOrig.getTime0();
530  trc.timeBins.set(time0 - trcOrig.getDeltaTBwd() - mTPCTimeEdgeTSafeMargin,
531  time0 + trcOrig.getDeltaTFwd() + mTPCTimeEdgeTSafeMargin);
532  // assign min max possible Z for this track which still respects the clusters A/C side
533  if (trcOrig.hasASideClustersOnly()) {
534  trc.zMin = trc.getZ() - trcOrig.getDeltaTBwd() * mTPCBin2Z;
535  trc.zMax = trc.getZ() + trcOrig.getDeltaTFwd() * mTPCBin2Z;
536  } else if (trcOrig.hasCSideClustersOnly()) {
537  trc.zMin = trc.getZ() - trcOrig.getDeltaTFwd() * mTPCBin2Z;
538  trc.zMax = trc.getZ() + trcOrig.getDeltaTBwd() * mTPCBin2Z;
539  }
540  // TODO : special treatment of tracks crossing the CE
541 
542  // cache work track index
543  mTPCSectIndexCache[o2::utils::Angle2Sector(trc.getAlpha())].push_back(mTPCWork.size() - 1);
544  }
545  float maxTimeBin = 0;
546  int nITSROFs = mITSROFTimes.size();
547  // sort tracks in each sector according to their timeMax
548  for (int sec = o2::constants::math::NSectors; sec--;) {
549  auto& indexCache = mTPCSectIndexCache[sec];
550  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " TPC tracks";
551  if (!indexCache.size())
552  continue;
553  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
554  auto& trcA = mTPCWork[a];
555  auto& trcB = mTPCWork[b];
556  return (trcA.timeBins.max() - trcB.timeBins.max()) < 0.;
557  });
558 
559  // build array of 1st entries with tmax corresponding to each ITS ROF (or trigger)

608  // Do preparatory work for matching

610

612  LOG(INFO) << "Empty TF";
613  return false;
614  }
615  mTimer[SWPrepITS].Start(false);
616  mITSWork.clear();
617  mITSROFTimes.clear();
618  // number of records might be actually more than N tracks!
619  mMatchRecordsITS.clear(); // RS TODO reserve(mMatchRecordsITS.size() + mParams->maxMatchCandidates*ntr);
620  if (mMCTruthON) {
621  mITSLblWork.clear();
622  }
623

627  mABClusterLinkIndex.clear();
628  mABClusterLinkIndex.resize(nITSClus, MinusOne);
629 
630  for (int sec = o2::constants::math::NSectors; sec--;) {
631  mITSSectIndexCache[sec].clear();
632  mITSTimeBinStart[sec].clear();
633  mITSTimeBinStart[sec].resize(nROFs, -1); // start of ITS work tracks in every sector
634  }
635  setStartIR(mITSTrackROFRec[0].getBCData());
636  for (int irof = 0; irof < nROFs; irof++) {
637  const auto& rofRec = mITSTrackROFRec[irof];

639  float tmn = intRecord2TPCTimeBin(rofRec.getBCData()); // ITS track min time in TPC time-bins

643  mITSTimeBinStart[sec][irof] = mITSSectIndexCache[sec].size(); // The sorting does not affect this
644  }
645 
646  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
647  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
648  const auto& trcOrig = mITSTracksArray[it];
649  if (mParams->runAfterBurner) {

659  if (!trc.rotate(o2::utils::Angle2Alpha(trc.getPhiPos()))) {

678  // For a track with Yr and Phir at Xr the distance^2 between the poisition of this track in the neighb. sector
679  // when propagated to Xr (in this neighbouring sector) and the edge will be (neglecting the curvature)
680  // [(Xr*tg(10)-Yr)/(tgPhir+tg70)]^2 / cos(70)^2 // for the next sector

689  addLastTrackCloneForNeighbourSector(sector < (o2::constants::math::NSectors - 1) ? sector + 1 : 0);
690  }
691  // sector down
692  float dy2Dn = (YMaxAtXMatchingRef + trcY) / (tgp - Tan70);
693  if ((dy2Dn * dy2Dn * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector down
694  addLastTrackCloneForNeighbourSector(sector > 1 ? sector - 1 : o2::constants::math::NSectors - 1);
695  }
696  }
697  }
698 
699  // sort tracks in each sector according to their time, then tgl
700  for (int sec = o2::constants::math::NSectors; sec--;) {
701  auto& indexCache = mITSSectIndexCache[sec];
702  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " ITS tracks";
703  if (!indexCache.size()) {
704  continue;
705  }
706  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
707  auto& trackA = mITSWork[a];
708  auto& trackB = mITSWork[b];
709  if (trackA.roFrame < trackB.roFrame) { // ITS tracks have the same time coverage
710  return true;
711  } else if (trackA.roFrame > trackB.roFrame) {
712  return false;
713  }
714  return trackA.getTgl() < trackB.getTgl();
715  });
716 
717  /* TOREMOVE?
718  // build array of 1st entries with of each ITS ROFrame
719  int nbins = 1 + mITSWork[indexCache.back()].roFrame;
720  auto& tbinStart = mITSTimeBinStart[sec];
721  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
722  tbinStart[0] = 0;
723  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
724  auto& trc = mITSWork[indexCache[itr]];
725  if (tbinStart[trc.roFrame] == -1) {
726  tbinStart[trc.roFrame] = itr;
727  }
728  }
729  for (int i = 1; i < nbins; i++) {
730  if (tbinStart[i] == -1) { // fill gaps with preceding indices. TODO ? Shall we do this for cont.readout in ITS only?
731  tbinStart[i] = tbinStart[i - 1];
732  }
733  }
734  */
735  } // loop over tracks of single sector
736  mMatchRecordsITS.reserve(mITSWork.size() * mParams->maxMatchCandidates);
737  mTimer[SWPrepITS].Stop();

750 //_____________________________________________________
751 void MatchTPCITS::doMatching(int sec)
752 {
754  auto& cacheITS = mITSSectIndexCache[sec]; // array of cached ITS track indices for this sector
755  auto& cacheTPC = mTPCSectIndexCache[sec]; // array of cached ITS track indices for this sector
756  auto& tbinStartTPC = mTPCTimeBinStart[sec]; // array of 1st TPC track with timeMax in ITS ROFrame
757  auto& tbinStartITS = mITSTimeBinStart[sec];
758  int nTracksTPC = cacheTPC.size(), nTracksITS = cacheITS.size();
759  if (!nTracksTPC || !nTracksITS) {
760  LOG(INFO) << "Matchng sector " << sec << " : N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS << " in sector "
761  << sec;
762  return;
763  }
764 
766  float maxTDriftSafe = (mNTPCBinsFullDrift + mParams->TPCITSTimeBinSafeMargin + mTPCTimeEdgeTSafeMargin);
767 
768  // get min ROFrame (in TPC time-bins) of ITS tracks currently in cache
769  auto minROFITS = mITSWork[cacheITS.front()].roFrame;
770 
771  if (minROFITS >= int(tbinStartTPC.size())) {
772  LOG(INFO) << "ITS min ROFrame " << minROFITS << " exceeds all cached TPC track ROF eqiuvalent "
773  << cacheTPC.size() - 1;
774  return;
775  }
776 
777  int nCheckTPCControl = 0, nCheckITSControl = 0, nMatchesControl = 0; // temporary
778 
779  int idxMinTPC = tbinStartTPC[minROFITS]; // index of 1st cached TPC track within cached ITS ROFrames
780  for (int itpc = idxMinTPC; itpc < nTracksTPC; itpc++) {
781  auto& trefTPC = mTPCWork[cacheTPC[itpc]];
782  // estimate ITS 1st ROframe bin this track may match to: TPC track are sorted according to their
783  // timeMax, hence the timeMax - MaxmNTPCBinsFullDrift are non-decreasing
784  int itsROBin = tpcTimeBin2ITSROFrame(trefTPC.timeBins.max() - maxTDriftSafe);
785  if (itsROBin >= int(tbinStartITS.size())) { // time of TPC track exceeds the max time of ITS in the cache
786  break;
787  }
788  int iits0 = tbinStartITS[itsROBin];
789  nCheckTPCControl++;
790  for (auto iits = iits0; iits < nTracksITS; iits++) {
791  auto& trefITS = mITSWork[cacheITS[iits]];
792  const auto& timeITS = mITSROFTimes[trefITS.roFrame];
793  // compare if the ITS and TPC tracks may overlap in time
794  if (trefTPC.timeBins < timeITS) {
795  // since TPC tracks are sorted in timeMax and ITS tracks are sorted in timeMin
796  // all following ITS tracks also will not match
797  break;
798  }
799  if (trefTPC.timeBins > timeITS) { // its bracket precedes TPC bracket
800  continue;
801  }
802  nCheckITSControl++;
803  float chi2 = -1;
804  int rejFlag = compareTPCITSTracks(trefITS, trefTPC, chi2);
805 
806 #ifdef _ALLOW_DEBUG_TREES_
807  if (mDBGOut && ((rejFlag == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
808  fillTPCITSmatchTree(cacheITS[iits], cacheTPC[itpc], rejFlag, chi2);
809  }
810 #endif
811 
812  if (rejFlag == RejectOnTgl) {
813  // ITS tracks in each ROFrame are ordered in Tgl, hence if this check failed on Tgl check
814  // (i.e. tgl_its>tgl_tpc+tolerance), tnem all other ITS tracks in this ROFrame will also have tgl too large.
815  // Jump on the 1st ITS track of the next ROFrame
816  int rof = trefITS.roFrame;
817  bool stop = false;
818  do {
819  if (++rof >= int(tbinStartITS.size())) {
820  stop = true;
821  break; // no more ITS ROFrames in cache
822  }
823  iits = tbinStartITS[rof] - 1; // next track to be checked -1
824  } while (iits <= tbinStartITS[trefITS.roFrame]); // skip empty bins
825  if (stop) {
826  break;
827  }
828  continue;
829  }
830  if (rejFlag != Accept) {
831  continue;
832  }
833  registerMatchRecordTPC(cacheITS[iits], cacheTPC[itpc], chi2); // register matching candidate
834  nMatchesControl++;
835  }
836  }
837 
838  LOG(INFO) << "Match sector " << sec << " N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS
839  << " N TPC tracks checked: " << nCheckTPCControl << " (starting from " << idxMinTPC
840  << "), checks: " << nCheckITSControl << ", matches:" << nMatchesControl;
841 }

864 //______________________________________________
865 bool MatchTPCITS::registerMatchRecordTPC(int iITS, int iTPC, float chi2)
866 {
869  auto& tTPC = mTPCWork[iTPC]; // get matchRecord structure of this TPC track, create if none
870  if (tTPC.matchID < 0) { // no matches yet, just add new record
871  registerMatchRecordITS(iITS, iTPC, chi2); // register TPC track in the ITS records
872  tTPC.matchID = mMatchRecordsTPC.size(); // new record will be added in the end
873  mMatchRecordsTPC.emplace_back(iITS, chi2); // create new record with empty reference on next match
874  return true;
875  }
876 
877  int count = 0, nextID = tTPC.matchID, topID = MinusOne;
878  do {
879  auto& nextMatchRec = mMatchRecordsTPC[nextID];
880  count++;
881  if (chi2 < nextMatchRec.chi2) { // need to insert new record before nextMatchRec?
882  if (count < mParams->maxMatchCandidates) {
883  break; // will insert in front of nextID
884  } else { // max number of candidates reached, will overwrite the last one
885  nextMatchRec.chi2 = chi2;
886  suppressMatchRecordITS(nextMatchRec.partnerID, iTPC); // flag as disabled the overriden ITS match
887  registerMatchRecordITS(iITS, tTPC.matchID, chi2); // register TPC track entry in the ITS records
888  nextMatchRec.partnerID = iITS; // reuse the record of suppressed ITS match to store better one
889  return true;
890  }
891  }
892  topID = nextID; // check next match record
893  nextID = nextMatchRec.nextRecID;
894  } while (nextID > MinusOne);
895 
896  // if count == mParams->maxMatchCandidates, the max number of candidates was already reached, and the
897  // new candidated was either discarded (if its chi2 is worst one) or has overwritten worst
898  // existing candidate. Otherwise, we need to add new entry
899  if (count < mParams->maxMatchCandidates) {
900  if (topID < 0) { // the new match is top candidate
901  topID = tTPC.matchID = mMatchRecordsTPC.size(); // register new record as top one
902  } else { // there are better candidates
903  topID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC.size(); // register to his parent
904  }
905  // nextID==-1 will mean that the while loop run over all candidates->the new one is the worst (goes to the end)
906  registerMatchRecordITS(iITS, tTPC.matchID, chi2); // register TPC track in the ITS records
907  mMatchRecordsTPC.emplace_back(iITS, chi2, nextID); // create new record with empty reference on next match
908  // make sure that after addition the number of candidates don't exceed allowed number
909  count++;
910  while (nextID > MinusOne) {
911  if (count > mParams->maxMatchCandidates) {
912  suppressMatchRecordITS(mMatchRecordsTPC[nextID].partnerID, iTPC);
913  // exclude nextID record, w/o changing topID (which becomes the last record)
914  nextID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC[nextID].nextRecID;
915  continue;
916  }
917  count++;
918  topID = nextID;
919  nextID = mMatchRecordsTPC[nextID].nextRecID;
920  }
921  return true;
922  } else {
923  return false; // unless nextID was assigned OverrideExisting, new candidate was discarded
924  }
925 }

928 void MatchTPCITS::registerMatchRecordITS(int iITS, int iTPC, float chi2)
929 {
931  auto& tITS = mITSWork[iITS];
932  int idnew = mMatchRecordsITS.size();
933  mMatchRecordsITS.emplace_back(iTPC, chi2); // associate iTPC with this record

935  tITS.matchID = idnew;
936  return;
937  }
938  // there are other matches for this ITS track, insert the new record preserving chi2 order
939  // navigate till last record or the one with worse chi2
940  int topID = MinusOne, nextRecord = tITS.matchID;
941  mMatchRecordsITS.emplace_back(iTPC, chi2); // associate iTPC with this record
942  auto& newRecord = mMatchRecordsITS.back();

944  auto& recITS = mMatchRecordsITS[nextRecord];
945  if (chi2 < recITS.chi2) { // insert before this one
946  newRecord.nextRecID = nextRecord; // new one will refer to old one it overtook
947  if (topID < 0) {
948  tITS.matchID = idnew; // the new one is the best match, track will refer to it
949  } else {
950  mMatchRecordsITS[topID].nextRecID = idnew; // new record will follow existing better one
951  }
952  return;
953  }
954  topID = nextRecord;
955  nextRecord = mMatchRecordsITS[nextRecord].nextRecID;
956  } while (nextRecord > MinusOne);
957 
958  // if we reached here, the new record should be added in the end
959  mMatchRecordsITS[topID].nextRecID = idnew; // register new link

963 int MatchTPCITS::compareTPCITSTracks(const TrackLocITS& tITS, const TrackLocTPC& tTPC, float& chi2) const
964 {
966  auto& trackTPC = tTPC;
967  auto& trackITS = tITS;
968  chi2 = -1.f;
969  int rejFlag = Accept;
970  float diff; // make rough check differences and their nsigmas
971 
972  // start with check on Tgl, since rjection on it will allow to profit from sorting
973  diff = trackITS.getParam(o2::track::kTgl) - trackTPC.getParam(o2::track::kTgl);
974  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kTgl], RejectOnTgl))) {
975  return rejFlag;
976  }
977  diff *= diff / (trackITS.getDiagError2(o2::track::kTgl) + trackTPC.getDiagError2(o2::track::kTgl));
978  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kTgl], RejectOnTgl + NSigmaShift))) {
979  return rejFlag;
980  }
981 
982  diff = trackITS.getParam(o2::track::kY) - trackTPC.getParam(o2::track::kY);
983  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kY], RejectOnY))) {
984  return rejFlag;
985  }
986  diff *= diff / (trackITS.getDiagError2(o2::track::kY) + trackTPC.getDiagError2(o2::track::kY));
987  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kY], RejectOnY + NSigmaShift))) {
988  return rejFlag;
989  }
990 
991  if (mCompareTracksDZ) { // in continuous mode we usually don't use DZ
992  diff = trackITS.getParam(o2::track::kZ) - trackTPC.getParam(o2::track::kZ);
993  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kZ], RejectOnZ))) {
994  return rejFlag;
995  }
996  diff *= diff / (trackITS.getDiagError2(o2::track::kZ) + trackTPC.getDiagError2(o2::track::kZ));
997  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kZ], RejectOnZ + NSigmaShift))) {
998  return rejFlag;
999  }
1000  } else { // in continuous mode we use special check on allowed Z range
1001  if (trackITS.getParam(o2::track::kZ) - tTPC.zMax > mParams->crudeAbsDiffCut[o2::track::kZ])
1002  return RejectOnZ;
1003  if (trackITS.getParam(o2::track::kZ) - tTPC.zMin < -mParams->crudeAbsDiffCut[o2::track::kZ])
1004  return -RejectOnZ;
1005  }
1006 
1007  diff = trackITS.getParam(o2::track::kSnp) - trackTPC.getParam(o2::track::kSnp);
1008  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kSnp], RejectOnSnp))) {
1009  return rejFlag;
1010  }
1011  diff *= diff / (trackITS.getDiagError2(o2::track::kSnp) + trackTPC.getDiagError2(o2::track::kSnp));
1012  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kSnp], RejectOnSnp + NSigmaShift))) {
1013  return rejFlag;
1014  }
1015 
1016  diff = trackITS.getParam(o2::track::kQ2Pt) - trackTPC.getParam(o2::track::kQ2Pt);
1017  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kQ2Pt], RejectOnQ2Pt))) {
1018  return rejFlag;
1019  }
1020  diff *= diff / (trackITS.getDiagError2(o2::track::kQ2Pt) + trackTPC.getDiagError2(o2::track::kQ2Pt));
1021  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kQ2Pt], RejectOnQ2Pt + NSigmaShift))) {
1022  return rejFlag;
1023  }
1024  // calculate mutual chi2 excluding Z in continuos mode
1025  chi2 = getPredictedChi2NoZ(tITS, tTPC);
1026  if (chi2 > mParams->cutMatchingChi2)
1027  return RejectOnChi2;
1028 
1029  return Accept;
1030 }
