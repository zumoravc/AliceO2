32 // simple helper struct to keep the InputRecord and ownership of messages
33 struct DataSet {
34  // not nice with the double vector but for quick unit test ok
35  using MessageSet = std::vector<std::unique_ptr<std::vector<char>>>;
36  using TaggedSet = std::pair<o2::header::DataOrigin, MessageSet>;
37  using Messages = std::vector<TaggedSet>;
38  using CheckType = std::vector<std::string>;
39  DataSet(std::vector<InputRoute>&& s, Messages&& m, CheckType&& v)
40  : schema{std::move(s)}, messages{std::move(m)}, record{schema, {[this](size_t i, size_t part) {
41  BOOST_REQUIRE(i < this->messages.size());
42  BOOST_REQUIRE(part < this->messages[i].second.size() / 2);
43  auto header = static_cast<char const*>(this->messages[i].second.at(2 * part)->data());
44  auto payload = static_cast<char const*>(this->messages[i].second.at(2 * part + 1)->data());
45  return DataRef{nullptr, header, payload};
46  },
47  [this](size_t i) { return i < this->messages.size() ? messages[i].second.size() / 2 : 0; }, this->messages.size()}},
48  values{std::move(v)}
49  {
50  BOOST_REQUIRE(messages.size() == schema.size());
51  }
52 
53  std::vector<InputRoute> schema;
54  Messages messages;
55  InputRecord record;
56  CheckType values;
57 };

59 DataSet createData()
60 {
61  // Create the routes we want for the InputRecord
62  std::vector<InputSpec> inputspecs = {
63  InputSpec{"tpc", "TPC", "SOMEDATA", 0, Lifetime::Timeframe},
64  InputSpec{"its", ConcreteDataTypeMatcher{"ITS", "SOMEDATA"}, Lifetime::Timeframe},
65  InputSpec{"tof", "TOF", "SOMEDATA", 1, Lifetime::Timeframe}};
66 
67  size_t i = 0;
68  auto createRoute = [&i](const char* source, InputSpec& spec) {
69  return InputRoute{
70  spec,
71  i++,
72  source};
73  };
74 
75  std::vector<InputRoute> schema = {
76  createRoute("tpc_source", inputspecs[0]),
77  createRoute("its_source", inputspecs[1]),
78  createRoute("tof_source", inputspecs[2])};
79 
80  decltype(DataSet::values) checkValues;
81  DataSet::Messages messages;
82 
83  auto createMessage = [&messages, &checkValues](DataHeader dh) {
84  checkValues.emplace_back(dh.dataOrigin.as<std::string>() + "_" + dh.dataDescription.as<std::string>() + "_" + std::to_string(dh.subSpecification));
85  std::string const& data = checkValues.back();
86  dh.payloadSize = data.size();
87  DataProcessingHeader dph{0, 1};
88  Stack stack{dh, dph};
89  auto it = messages.begin(), end = messages.end();
90  for (; it != end; ++it) {
91  if (it->first == dh.dataOrigin) {
92  break;
93  }
94  }
95  if (it == end) {
96  messages.resize(messages.size() + 1);
97  it = messages.end() - 1;
98  it->first = dh.dataOrigin;
99  }
100  auto& routemessages = it->second;
101  routemessages.emplace_back(std::make_unique<std::vector<char>>(stack.size()));
102  memcpy(routemessages.back()->data(), stack.data(), routemessages.back()->size());
103  routemessages.emplace_back(std::make_unique<std::vector<char>>(dh.payloadSize));
104  memcpy(routemessages.back()->data(), data.data(), routemessages.back()->size());
105  };
106 
107  // we create message for the 3 input routes, the messages have different page size
108  // and the second messages has 3 parts, each with the same page size
109  // the test value is written as payload after the RDH and all values are cached for
110  // later checking when parsing the data set
111  DataHeader dh1;
112  dh1.dataDescription = "SOMEDATA";
113  dh1.dataOrigin = "TPC";
114  dh1.subSpecification = 0;
115  dh1.payloadSerializationMethod = o2::header::gSerializationMethodNone;
116  DataHeader dh2;
117  dh2.dataDescription = "SOMEDATA";
118  dh2.dataOrigin = "ITS";
119  dh2.subSpecification = 0;
120  dh2.payloadSerializationMethod = o2::header::gSerializationMethodNone;
121  DataHeader dh3;
122  dh3.dataDescription = "SOMEDATA";
123  dh3.dataOrigin = "ITS";
124  dh3.subSpecification = 1;
125  dh3.payloadSerializationMethod = o2::header::gSerializationMethodNone;
126  DataHeader dh4;
127  dh4.dataDescription = "SOMEDATA";
128  dh4.dataOrigin = "TOF";
129  dh4.subSpecification = 255;
130  dh4.payloadSerializationMethod = o2::header::gSerializationMethodNone;
131  createMessage(dh1);
132  createMessage(dh2);
133  createMessage(dh3);
134  createMessage(dh4);
135 
136  return {std::move(schema), std::move(messages), std::move(checkValues)};
137 }

139 BOOST_AUTO_TEST_CASE(test_DPLRawParser)
140 {
141  auto dataset = createData();
142  InputRecord& inputs = dataset.record;
143  BOOST_REQUIRE(dataset.messages.size() > 0);
144  BOOST_REQUIRE(dataset.messages[0].second.at(0) != nullptr);
145  BOOST_REQUIRE(inputs.size() == 3);
146  BOOST_CHECK((*inputs.begin()).header == dataset.messages[0].second.at(0)->data());
147 
148  int count = 0;
149  for (auto const& ref : InputRecordWalker(inputs)) {
150  auto const* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
151  auto const data = inputs.get<std::string>(ref);
152  BOOST_CHECK(data == dataset.values[count]);
153  count++;
154  }
155  BOOST_REQUIRE(count == 4);
156

158  {"tpc", "TPC", "SOMEDATA", 0, Lifetime::Timeframe},
159  {"its", ConcreteDataTypeMatcher{"ITS", "SOMEDATA"}, Lifetime::Timeframe},
160  };

168  BOOST_REQUIRE(count == 3);
