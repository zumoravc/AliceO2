55 void Decoder<GBTDECODER>::process(gsl::span<const uint8_t> bytes)
56 {
58  clear();
59  o2::framework::RawParser parser(bytes.data(), bytes.size());
60  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
61  if (it.size() == 0) {
62  continue;
63  }
64  gsl::span<const uint8_t> payload(it.data(), it.size());
65  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(it.raw());
66  process(payload, *rdhPtr);
67  }
68  flush();
69 }

71 template <typename GBTDECODER>
72 void Decoder<GBTDECODER>::flush()
73 {
75  for (auto& gbtDec : mGBTDecoders) {
76  if (!gbtDec.getData().empty()) {
77  size_t firstEntry = mData.size();
78  mData.insert(mData.end(), gbtDec.getData().begin(), gbtDec.getData().end());
79  size_t lastRof = mROFRecords.size();
80  mROFRecords.insert(mROFRecords.end(), gbtDec.getROFRecords().begin(), gbtDec.getROFRecords().end());
81  for (auto rofIt = mROFRecords.begin() + lastRof; rofIt != mROFRecords.end(); ++rofIt) {
82  rofIt->firstEntry += firstEntry;
83  }
84  gbtDec.clear();
85  }
86  }
87 }
