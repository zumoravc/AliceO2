33 std::vector<Track> TrackGenerator::generate(int nTracks)
34 {
37  std::vector<Track> tracks;
38  Track track;
39  std::array<float, 3> pos;
40  std::array<float, 2> dir;
41  for (int itrack = 0; itrack < nTracks; ++itrack) {
42  for (int ipos = 0; ipos < 3; ++ipos) {
43  pos[ipos] =
44  std::uniform_real_distribution<float>{mPositionLimits[2 * ipos], mPositionLimits[2 * ipos + 1]}(mGenerator);
45  }
46  track.setPosition(pos[0], pos[1], pos[2]);
47 
48  std::array<float, 4> limits = getLimitsForAcceptance(pos);
49 
50  for (int idir = 0; idir < 2; ++idir) {
51  dir[idir] = std::uniform_real_distribution<float>{limits[2 * idir], limits[2 * idir + 1]}(mGenerator);
52  }
53  track.setDirection(dir[0], dir[1], 1.);
54  tracks.push_back(track);
55  }
56  return tracks;

60 std::array<float, 4> TrackGenerator::getLimitsForAcceptance(std::array<float, 3> pos)
61 {
64  std::array<float, 4> limits = mSlopeLimits;
65  float dZ = -1600. - pos[2];
66  if (dZ == 0.) {
67  dZ = 0.0001;
68  }
69 
70  // These are the maximum x and y position of MT11 with the standard alignment
71  std::array<float, 2> maxValues{{257., 306.7}};
72  for (int icoor = 0; icoor < 2; ++icoor) {
73  int imin = 2 * icoor;
74  int imax = 2 * icoor + 1;
75  float minSlope = (pos[icoor] - maxValues[icoor]) / dZ;
76  float maxSlope = (maxValues[icoor] - pos[icoor]) / dZ;
77  if (minSlope > limits[imin]) {
78  limits[imin] = minSlope;
79  }
80  if (maxSlope < limits[imax]) {
81  limits[imax] = maxSlope;
82  }
83  }
84  return limits;
