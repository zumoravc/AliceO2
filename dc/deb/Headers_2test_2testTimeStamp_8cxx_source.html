46 BOOST_AUTO_TEST_CASE(TimeStamp_test)
47 {
48  // check the TimeStamp member layout with respect to manually
49  // assembled 64 bit field
50  // 'AC' (accelerator) specifies the LHCClock
51  // 40404040 orbits are almost 1h
52  // together with 1512 bunches we pass 1h
53  // TODO: extend from fixed values to random values
54  uint64_t bunches = 1512;
55  uint64_t orbits = 40404040;
56  uint64_t ts64 = String2<uint16_t>("AC") | bunches << 16 | orbits << 32;
57  TimeStamp timestamp("AC", orbits, bunches);
58 
59  // using the type cast operator
60  BOOST_CHECK(ts64 == timestamp);
61 
62  // checking cast to LHCClock
63  auto timeInLHCOrbitClock = timestamp.get<LHCOrbitClock>();
64  auto timeInLHCBunchClock = timestamp.get<LHCBunchClock>();
65  BOOST_CHECK(timeInLHCOrbitClock.count() == orbits);
66  BOOST_CHECK(timeInLHCOrbitClock == LHCOrbitClock::duration(orbits));
67 
68  // time in LHCOrbitClock cuts off the bunches
69  BOOST_CHECK(timeInLHCOrbitClock <= timeInLHCBunchClock);
70 
71  // get explicitely the time ignoring bunch counter and cast to seconds
72  // that must be less then 1h
73  auto timeInSeconds = std::chrono::duration_cast<std::chrono::seconds>(timestamp.get<LHCOrbitClock>());
74  BOOST_CHECK(timeInSeconds < std::chrono::hours(1));
75 
76  // directly retrieving time in hours takes both orbits and bunches for
77  // internal calculation before casting to hours
78  auto timeInHours = timestamp.get<std::chrono::hours>();
79  BOOST_CHECK(timeInHours.count() == 1);
80 
81  // setting timestamp with a value in unit mico seconds
82  uint64_t tenSeconds = 10000000;
83  timestamp = String2<uint16_t>("US") | tenSeconds << 32;
84 
85  // check conversion of the us value to LHCClock
86  auto timeInOrbitPrecision = timestamp.get<LHCOrbitClock>();
87  auto timeInBunchPrecision = timestamp.get<LHCBunchClock>();
88  uint64_t expectedOrbits = tenSeconds * 1000 / (lhc_clock_parameter::gNumberOfBunches * lhc_clock_parameter::gBunchSpacingNanoSec);
89  BOOST_CHECK(timeInOrbitPrecision.count() == expectedOrbits);
90 
91  // conversion in orbit precision ignores bunches
92  BOOST_CHECK(timeInOrbitPrecision <= timeInBunchPrecision);
93 }
