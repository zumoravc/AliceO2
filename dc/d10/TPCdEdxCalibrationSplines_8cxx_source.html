59 void TPCdEdxCalibrationSplines::recreate(int nKnotsU1[], int nKnotsU2[])
60 {
62 
63  FlatObject::startConstruction();
64 
65  int buffSize = 0;
66  int offsets1[FSplines];
67  int offsets2[FSplines];
68  for (unsigned int i = 0; i < FSplines; i++) {
69  mCalibSplinesqMax[i].recreate(nKnotsU1[i], nKnotsU2[i]);
70  buffSize = alignSize(buffSize, mCalibSplinesqMax[i].getBufferAlignmentBytes());
71  offsets1[i] = buffSize;
72  buffSize += mCalibSplinesqMax[i].getFlatBufferSize();
73  }
74  for (unsigned int i = 0; i < FSplines; i++) {
75  mCalibSplinesqTot[i].recreate(nKnotsU1[i], nKnotsU2[i]);
76  buffSize = alignSize(buffSize, mCalibSplinesqTot[i].getBufferAlignmentBytes());
77  offsets2[i] = buffSize;
78  buffSize += mCalibSplinesqTot[i].getFlatBufferSize();
79  }
80 
81  FlatObject::finishConstruction(buffSize);
82 
83  for (unsigned int i = 0; i < FSplines; i++) {
84  mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
85  }
86  for (unsigned int i = 0; i < FSplines; i++) {
87  mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
88  }
89 }

179 {

184  int offsets2[FSplines];

186  auto defaultFnd2D = [&](float x1, float x2, float f[]) {
187  f[0] = 1.f;

189

191  o2::gpu::Spline2D<float, 1> splineTmpqMax;
192  splineTmpqMax.approximateFunction(0., 1., 0., 1., defaultFnd2D);
193  mCalibSplinesqMax[ireg] = splineTmpqMax;
194  buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
195  offsets1[ireg] = buffSize;
196  buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
197  }

201  splineTmpqTot.approximateFunction(0., 1., 0., 1., defaultFnd2D);
202  mCalibSplinesqTot[ireg] = splineTmpqTot;
203  buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
204  offsets2[ireg] = buffSize;
205  buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
206  }
