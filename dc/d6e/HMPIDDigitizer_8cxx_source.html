25 void HMPIDDigitizer::zeroSuppress(std::vector<o2::hmpid::Digit> const& digits, std::vector<o2::hmpid::Digit>& newdigits,
26  o2::dataformats::MCTruthContainer<o2::MCCompLabel> const& labels,
27  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* newlabels)
28 {
29  int index = 0;
30  for (auto& digit : digits) {

32  newdigits.push_back(digit);
33 
34  if (newlabels) {
35  // copy the labels to the new place with the right new index
36  newlabels->addElements(newdigits.size() - 1, labels.getLabels(index));
37  }
38  }
39  index++;
40  }
41 }

58 // this will process hits and fill the digit vector with digits which are finalized
59 void HMPIDDigitizer::process(std::vector<o2::hmpid::HitType> const& hits, std::vector<o2::hmpid::Digit>& digits)
60 {
61  for (auto& hit : hits) {
62  int chamber, pc, px, py;
63  float totalQ;
64  // retrieves center pad and the total charge
65  Digit::getPadAndTotalCharge(hit, chamber, pc, px, py, totalQ);
66 
67  if (px < 0 || py < 0) {
68  continue;
69  }
70 
71  // determine which pads to loop over
72  std::array<int, 9> allpads;
73  int counter = 0;
74  for (int nx = -1; nx <= 1; ++nx) {
75  for (int ny = -1; ny <= 1; ++ny) {
76  allpads[counter] = Param::Abs(chamber, pc, px + nx, py + ny);
77  counter++;
78  }
79  }
80 
81  for (auto& pad : allpads) {
82  auto iter = mIndexForPad.find(pad);
83  int index = -1;
84  if (iter != mIndexForPad.end()) {
85  index = iter->second;
86  }
87  // auto index = mIndexForPad[pad];
88  float fraction = Digit::getFractionalContributionForPad(hit, pad);
89  // LOG(INFO) << "FRACTION ON PAD " << pad << " IS " << fraction;
90  if (index != -1) {
91  // digit exists ... reuse
92  auto& digit = mDigits[index];
93  digit.addCharge(totalQ * fraction);
94 
95  if (mRegisteredLabelContainer) {
96  auto labels = mTmpLabelContainer.getLabels(index);
97  o2::MCCompLabel newlabel(hit.GetTrackID(), mEventID, mSrcID, false);
98  bool newlabelneeded = true;
99  for (auto& l : labels) {
100  if (l == newlabel) {
101  newlabelneeded = false;
102  break;
103  }
104  }
105  if (newlabelneeded) {
106  mTmpLabelContainer.addElementRandomAccess(index, newlabel);
107  }
108  }
109  } else {
110  // create digit ... and register
111  mDigits.emplace_back(mCurrentTriggerTime, pad, totalQ * fraction);
112  mIndexForPad[pad] = mDigits.size() - 1;
113  mInvolvedPads.emplace_back(pad);
114 
115  if (mRegisteredLabelContainer) {
116  // add label for this digit
117  mTmpLabelContainer.addElement(mDigits.size() - 1, o2::MCCompLabel(hit.GetTrackID(), mEventID, mSrcID, false));
118  }
119  }
120  }
121  }
122 }
