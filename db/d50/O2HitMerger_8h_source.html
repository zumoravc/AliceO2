145  template <typename T, typename V>
146  V insertAdd(std::map<T, V>& m, T const& key, V value)
147  {
148  const auto iter = m.find(key);
149  V accum{0};
150  if (iter != m.end()) {
151  iter->second += value;
152  accum = iter->second;
153  } else {
154  m.insert(std::make_pair(key, value));
155  accum = value;
156  }
157  return accum;
158  }

246  bool handleSimData(FairMQParts& data, int /*index*/)
247  {
248  bool expectmore = true;
249  int index = 0;
250  auto infoptr = o2::base::decodeTMessage<o2::data::SubEventInfo*>(data, index++);
251  o2::data::SubEventInfo& info = *infoptr;
252  auto accum = insertAdd<uint32_t, uint32_t>(mPartsCheckSum, info.eventID, (uint32_t)info.part);
253 
254  LOG(INFO) << "SIMDATA channel got " << data.Size() << " parts for event " << info.eventID << " part " << info.part << " out of " << info.nparts;
255 
256  fillSubEventInfoEntry(info);
257  consumeData<std::vector<o2::MCTrack>>(info.eventID, "MCTrack", data, index);
258  consumeData<std::vector<o2::TrackReference>>(info.eventID, "TrackRefs", data, index);
259  consumeData<o2::dataformats::MCTruthContainer<o2::TrackReference>>(info.eventID, "IndexedTrackRefs", data, index);
260  while (index < data.Size()) {
261  consumeHits(info.eventID, data, index);
262  }
263  // set the number of entries in the tree
264  auto tree = mEventToTTreeMap[info.eventID];
265  auto memfile = mEventToTMemFileMap[info.eventID];
266  tree->SetEntries(tree->GetEntries() + 1);
267  LOG(INFO) << "tree has file " << tree->GetDirectory()->GetFile()->GetName();
268  mEntries++;
269 
270  if (isDataComplete<uint32_t>(accum, info.nparts)) {
271  LOG(INFO) << "EVERYTHING IS HERE FOR EVENT " << info.eventID << "\n";
272 
273  // check if previous flush finished
274  if (mMergerIOThread.joinable()) {
275  mMergerIOThread.join();
276  }
277 
278  // start hit merging and flushing in a separate thread in order not to block
279  mMergerIOThread = std::thread([info, this]() { mergeAndFlushData(info.eventID); });
280 
281  mEventChecksum += info.eventID;
282  // we also need to check if we have all events
283  if (isDataComplete<uint32_t>(mEventChecksum, info.maxEvents)) {
284  LOG(INFO) << "ALL EVENTS HERE; CHECKSUM " << mEventChecksum;
285 
286  // flush remaining data and close file
287  if (mMergerIOThread.joinable()) {
288  mMergerIOThread.join();
289  }
290 
291  expectmore = false;
292  }
293 
294  if (mPipeToDriver != -1) {
295  if (write(mPipeToDriver, &info.eventID, sizeof(info.eventID)) == -1) {
296  LOG(ERROR) << "FAILED WRITING TO PIPE";
297  };
298  }
299  }
300  if (!expectmore) {
301  // somehow FairMQ has difficulties shutting down; helping manually
302  raise(SIGINT);
303  }
304  return expectmore;
305  }

320  void reorderAndMergeMCTRacks(TTree& origin, TTree& target, const std::vector<int>& nprimaries, const std::vector<int>& nsubevents)
321  {
322  std::vector<MCTrack>* incomingdata = nullptr;
323  auto targetdata = new std::vector<MCTrack>;
324  auto originbr = origin.GetBranch("MCTrack");
325  originbr->SetAddress(&incomingdata);
326  const auto entries = origin.GetEntries();
327  //
328  // loop over subevents to store the primary events
329  //
330  Int_t nprimTot = 0;
331  for (auto entry = entries - 1; entry >= 0; --entry) {
332  int index = nsubevents[entry];
333  nprimTot += nprimaries[index];
334  printf("merge %lld %5d %5d %5d \n", entry, index, nsubevents[entry], nsubevents[index]);
335  originbr->GetEntry(index);
336  for (Int_t i = 0; i < nprimaries[index]; i++) {
337  auto& track = incomingdata->at(i);
338  track.SetFirstDaughterTrackId(-1);
339  targetdata->push_back(track);
340  }
341  incomingdata->clear();
342  delete incomingdata;
343  incomingdata = nullptr;
344  }
345  //
346  // loop a second time to store the secondaries and fix the mother track IDs
347  //
348  Int_t idelta1 = nprimTot;
349  Int_t idelta0 = 0;
350  for (auto entry = entries - 1; entry >= 0; --entry) {
351  int index = nsubevents[entry];
352 
353  originbr->GetEntry(index);
354 
355  Int_t npart = (int)(incomingdata->size());
356  Int_t nprim = nprimaries[index];
357  idelta1 -= nprim;
358 
359  for (Int_t i = nprim; i < npart; i++) {
360  auto& track = incomingdata->at(i);
361  Int_t cId = track.getMotherTrackId();
362  if (cId >= nprim) {
363  cId += idelta1;
364  } else {
365  cId += idelta0;
366  }
367  track.SetMotherTrackId(cId);
368  track.SetFirstDaughterTrackId(-1);
369 
370  Int_t hwm = (int)(targetdata->size());
371  auto& mother = targetdata->at(cId);
372  if (mother.getFirstDaughterTrackId() == -1) {
373  mother.SetFirstDaughterTrackId(hwm);
374  }
375  mother.SetLastDaughterTrackId(hwm);
376 
377  targetdata->push_back(track);
378  }
379  idelta0 += nprim;
380  idelta1 += npart;
381  incomingdata->clear();
382  delete incomingdata;
383  incomingdata = nullptr;
384  }
385 
386  //
387  // write to output
388  auto targetbr = o2::base::getOrMakeBranch(target, "MCTrack", &targetdata);
389  targetbr->SetAddress(&targetdata);
390  targetbr->Fill();
391  targetbr->ResetAddress();
392  targetdata->clear();
393  }

395  template <typename T>
396  void remapTrackIdsAndMerge(std::string brname, TTree& origin, TTree& target,
397  const std::vector<int>& trackoffsets, const std::vector<int>& nprimaries, const std::vector<int>& subevOrdered)
398  {
399  //
400  // Remap the mother track IDs by adding an offset.
401  // The offset calculated as the sum of the number of entries in the particle list of the previous subevents.
402  // This method is called by O2HitMerger::mergeAndFlushData(int)
403  //
404  T* incomingdata = nullptr;
405  auto targetdata = new T;
406  auto originbr = origin.GetBranch(brname.c_str());
407  originbr->SetAddress(&incomingdata);
408  const auto entries = origin.GetEntries();
409 
410  if (entries == 1) {
411  // nothing to do in case there is only one entry
412  originbr->GetEntry(0);
413  targetdata = incomingdata;
414  } else {
415  // loop over subevents
416  Int_t nprimTot = 0;
417  for (auto entry = 0; entry < entries; entry++)
418  nprimTot += nprimaries[entry];
419  Int_t idelta0 = 0;
420  Int_t idelta1 = nprimTot;
421  for (auto entry = entries - 1; entry >= 0; --entry) {
422  Int_t index = subevOrdered[entry];
423  Int_t nprim = nprimaries[index];
424  originbr->GetEntry(index);
425  idelta1 -= nprim;
426  for (auto& data : *incomingdata) {
427  updateTrackIdWithOffset(data, nprim, idelta0, idelta1);
428  targetdata->push_back(data);
429  }
430  incomingdata->clear();
431  idelta0 += nprim;
432  idelta1 += trackoffsets[index];
433  delete incomingdata;
434  incomingdata = nullptr;
435  }
436  }
437  auto targetbr = o2::base::getOrMakeBranch(target, brname.c_str(), &targetdata);
438  targetbr->SetAddress(&targetdata);
439  targetbr->Fill();
440  targetbr->ResetAddress();
441  targetdata->clear();
442  }

444  void updateTrackIdWithOffset(MCTrack& track, Int_t nprim, Int_t idelta0, Int_t idelta1)
445  {
446  Int_t cId = track.getMotherTrackId();
447  Int_t ioffset = (cId < nprim) ? idelta0 : idelta1;
448  if (cId != -1)
449  track.SetMotherTrackId(cId + ioffset);
450  }

452  void updateTrackIdWithOffset(TrackReference& ref, Int_t nprim, Int_t idelta0, Int_t idelta1)
453  {
454  Int_t cId = ref.getTrackID();
455  Int_t ioffset = (cId < nprim) ? idelta0 : idelta1;
456  ref.setTrackID(cId + ioffset);
457  }

459  // this merges all entries from the TBranch brname from the origin TTree (containing one event only)
460  // into a single entry in a target TTree / same branch
461  // (assuming T is typically a vector; merging is simply done by appending)
462  template <typename T>
463  void merge(std::string brname, TTree& origin, TTree& target)
464  {
465  auto originbr = origin.GetBranch(brname.c_str());
466  auto targetdata = new T;
467  T* incomingdata = nullptr;
468  originbr->SetAddress(&incomingdata);
469 
470  const auto entries = origin.GetEntries();
471 
472  T* filladdress = nullptr;
473  if (entries == 1) {
474  // this avoids useless copy in case there was no sub-event splitting; we just use the original data
475  originbr->GetEntry(0);
476  filladdress = incomingdata;
477  } else {
478  filladdress = targetdata;
479  for (auto entry = 0; entry < entries; ++entry) {
480  originbr->GetEntry(entry);
481  backInsert(*incomingdata, *targetdata);
482  delete incomingdata;
483  incomingdata = nullptr;
484  }
485  }
486 
487  // fill target for this event
488  auto targetbr = o2::base::getOrMakeBranch(target, brname.c_str(), &filladdress);
489  targetbr->SetAddress(&filladdress);
490  targetbr->Fill();
491  targetbr->ResetAddress();
492  targetdata->clear();
493  if (incomingdata) {
494  delete incomingdata;
495  incomingdata = nullptr;
496  }
497 
498  delete targetdata;
499  }

514  // This method goes over the tree containing data for a given event; potentially merges
515  // it and flushes it into the actual output file.
516  // The method can be called asynchronously to data collection
517  bool mergeAndFlushData(int eventID)
518  {
519  LOG(INFO) << "ENTERING MERGING/FLUSHING HITS STAGE FOR EVENT " << eventID;
520 
521  auto tree = mEventToTTreeMap[eventID];
522  if (!tree) {
523  LOG(INFO) << "NO TTREE FOUND FOR EVENT " << eventID;
524  return false;
525  }
526 
527  if (tree->GetEntries() == 0 || mNExpectedEvents == 0) {
528  LOG(INFO) << "NO ENTRY IN TTREE FOUND FOR EVENT " << eventID;
529  return false;
530  }
531 
532  TStopwatch timer;
533  timer.Start();
534 
535  // calculate trackoffsets
536  auto infobr = tree->GetBranch("SubEventInfo");
537 
538  auto& confref = o2::conf::SimConfig::Instance();
539 
540  std::vector<int> trackoffsets; // collecting trackoffsets to be applied to correct
541  std::vector<int> nprimaries; // collecting primary particles in each subevent
542  std::vector<int> nsubevents; // collecting of subevent numbers
543 
544  std::unique_ptr<o2::dataformats::MCEventHeader> eventheader; // The event header
545 
546  // the MC labels (trackID) for hits
547  o2::data::SubEventInfo* info = nullptr;
548  infobr->SetAddress(&info);
549  for (int i = 0; i < infobr->GetEntries(); ++i) {
550  infobr->GetEntry(i);
551  assert(info->npersistenttracks >= 0);
552  trackoffsets.emplace_back(info->npersistenttracks);
553  nprimaries.emplace_back(info->nprimarytracks);
554  nsubevents.emplace_back(info->part);
555  if (eventheader == nullptr) {
556  eventheader = std::unique_ptr<dataformats::MCEventHeader>(
557  new dataformats::MCEventHeader(info->mMCEventHeader));
558  } else {
559  eventheader->getMCEventStats().add(info->mMCEventHeader.getMCEventStats());
560  }
561  }
562 
563  // now see which events can be discarded in any case due to no hits
564  if (confref.isFilterOutNoHitEvents()) {
565  if (eventheader && eventheader->getMCEventStats().getNHits() == 0) {
566  LOG(INFO) << " Taking out event " << eventID << " due to no hits ";
567 
568  return false;
569  }
570  }
571 
572  // put the event headers into the new TTree
573  o2::dataformats::MCEventHeader* headerptr = eventheader.get();
574  auto headerbr = o2::base::getOrMakeBranch(*mOutTree, "MCEventHeader.", &headerptr);
575  headerbr->SetAddress(&headerptr);
576  headerbr->Fill();
577  headerbr->ResetAddress();
578 
579  // attention: We need to make sure that we write everything in the same event order
580  // but iteration over keys of a standard map in C++ is ordered
581 
582  // b) merge the general data
583  //
584  // for MCTrack remap the motherIds and merge at the same go
585  const auto entries = tree->GetEntries();
586  std::vector<int> subevOrdered((int)(nsubevents.size()));
587  for (auto entry = entries - 1; entry >= 0; --entry) {
588  subevOrdered[nsubevents[entry] - 1] = entry;
589  printf("HitMerger entry: %lld nprimry: %5d trackoffset: %5d \n", entry, nprimaries[entry], trackoffsets[entry]);
590  }
591 
592  reorderAndMergeMCTRacks(*tree, *mOutTree, nprimaries, subevOrdered);
593  Int_t ioffset = 0;
594  remapTrackIdsAndMerge<std::vector<o2::TrackReference>>("TrackRefs", *tree, *mOutTree, trackoffsets, nprimaries, subevOrdered);
595  merge<o2::dataformats::MCTruthContainer<o2::TrackReference>>("IndexedTrackRefs", *tree, *mOutTree);
596 
597  // c) do the merge procedure for all hits ... delegate this to detector specific functions
598  // since they know about types; number of branches; etc.
599  // this will also fix the trackIDs inside the hits
600  for (int id = 0; id < mDetectorInstances.size(); ++id) {
601  auto& det = mDetectorInstances[id];
602  if (det) {
603  auto hittree = mDetectorToTTreeMap[id];
604  det->mergeHitEntries(*tree, *hittree, trackoffsets, nprimaries, subevOrdered);
605  hittree->SetEntries(hittree->GetEntries() + 1);
606  LOG(INFO) << "flushing tree to file " << hittree->GetDirectory()->GetFile()->GetName();
607  mDetectorOutFiles[id]->Write("", TObject::kOverwrite);
608  }
609  }
610 
611  // increase the entry count in the tree
612  mOutTree->SetEntries(mOutTree->GetEntries() + 1);
613  LOG(INFO) << "outtree has file " << mOutTree->GetDirectory()->GetFile()->GetName();
614  mOutFile->Write("", TObject::kOverwrite);
615 
616  // remove tree for that eventID
617  delete mEventToTTreeMap[eventID];
618  mEventToTTreeMap.erase(eventID);
619  // remove memfile
