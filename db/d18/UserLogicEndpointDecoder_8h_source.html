97 template <typename CHARGESUM>
98 size_t UserLogicEndpointDecoder<CHARGESUM>::append(Payload buffer)
99 {
100  if (buffer.size() % 8) {
101  throw std::invalid_argument("buffer size should be a multiple of 8");
102  }
103  size_t n{0};
104 
105  for (size_t i = 0; i < buffer.size(); i += 8) {
106  uint64_t word = (static_cast<uint64_t>(buffer[i + 0])) |
107  (static_cast<uint64_t>(buffer[i + 1]) << 8) |
108  (static_cast<uint64_t>(buffer[i + 2]) << 16) |
109  (static_cast<uint64_t>(buffer[i + 3]) << 24) |
110  (static_cast<uint64_t>(buffer[i + 4]) << 32) |
111  (static_cast<uint64_t>(buffer[i + 5]) << 40) |
112  (static_cast<uint64_t>(buffer[i + 6]) << 48) |
113  (static_cast<uint64_t>(buffer[i + 7]) << 56);
114 
115  if (word == 0) {
116  continue;
117  }
118  if (word == 0xFEEDDEEDFEEDDEED) {
119  continue;
120  }
121 
122  // Get the GBT link associated to this word
123  int gbt = (word >> 59) & 0x1F;
124 
125  if (gbt < 0 || gbt > 11) {
126  throw fmt::format("warning : out-of-range gbt {} word={:08X}\n", gbt, word);
127  }
128 
129  // Get the corresponding decoders array, or allocate it if does not exist yet
130  auto d = mElinkDecoders.find(gbt);
131  if (d == mElinkDecoders.end()) {
132 
133  // Compute the (feeId, linkId) pair...
134  FeeLinkId feeLinkId(mFeeId, gbt);
135  // ... and get the solar from it
136  auto solarId = mFee2SolarMapper(feeLinkId);
137 
138  if (!solarId.has_value()) {
139  throw std::logic_error(fmt::format("{} Could not get solarId from feeLinkId={}\n", __PRETTY_FUNCTION__, asString(feeLinkId)));
140  }
141 
142  mElinkDecoders.emplace(static_cast<uint16_t>(gbt),
143  impl::makeArray<40>([=](size_t i) {
144  DsElecId dselec{solarId.value(), static_cast<uint8_t>(i / 5), static_cast<uint8_t>(i % 5)};
145  return ElinkDecoder(dselec, mChannelHandler);
146  }));
147  d = mElinkDecoders.find(gbt);
148  }
149 
150  // in the 14 MSB(Most Significant Bits) 6 are used to specify the Dual Sampa index (0..39)
151  uint16_t dsid = (word >> 53) & 0x3F;
152 
153  // bits 50..52 are error bits
154  int8_t error = static_cast<uint8_t>((word >> 50) & 0x7);
155 
156  // the remaining (LSB) 50 bits represents the actual data to passed to the ElinkDecoder
157  uint64_t data50 = word & FIFTYBITSATONE;
158  d->second.at(dsid).append(data50, error);
159  n += 8;
160  }
161  return n;
162 }
