30 // a label container with both contiguous and non-contiguous modes
31 // for label storage
32 // NOTE: the current specialization for contiguous storage is not
33 // yet on par with MCTruthContainer ... but eventually we might unify
34 // everything and keep only the more generic and configurable version
35 template <typename LabelType, bool isContiguousStorage = true>
36 class LabelContainer
37 {
38  public:
39  struct HeaderElementContinuous {
40  HeaderElementContinuous() = default; // for ROOT IO
41  HeaderElementContinuous(ushort s, uint i) : size(s), index(i) {}
42  uint index = 0; // index of first label in the actual label storage
43  ushort size = 0; // total number of labels
44  ClassDefNV(HeaderElementContinuous, 1);
45  };
46 
47  struct HeaderElementLinked {
48  HeaderElementLinked() = default; // for ROOT IO
49  HeaderElementLinked(uint i, int l, ushort s) : index(i), lastindex(l), size(s) {}
50  uint index = 0; // index of first label in the actual label storage
51  uint lastindex = 0; // index of last label in the actual label storage
52  ushort size = 0; // total number of labels
53  ClassDefNV(HeaderElementLinked, 1);
54  };
55 
56  using HeaderElement = typename std::conditional<isContiguousStorage, HeaderElementContinuous, HeaderElementLinked>::type;
57  using StoredLabelType = typename std::conditional<isContiguousStorage, LabelType, std::pair<LabelType, int>>::type;
58 
59  // internal functions allowing the iterator implementation to be completely generic
60  static uint getNextIndex(uint index, std::vector<LabelType> const& /*labels*/)
61  {
62  return index + 1;
63  }
64  static uint getNextIndex(uint index, std::vector<std::pair<LabelType, int>> const& labels)
65  {
66  return labels[index].second;
67  }
68  static LabelType& dereference(std::vector<LabelType>& v, int index)
69  {
70  return v[index];
71  }
72  static LabelType& dereference(std::vector<std::pair<LabelType, int>>& v, int index)
73  {
74  return v[index].first;
75  }
76  static int lastIndex(HeaderElementContinuous const& h)
77  {
78  return h.index + h.size;
79  }
80  static int lastIndex(HeaderElementLinked const& h)
81  {
82  // -1 since this is indication of end of linked list
83  return -1;
84  }

86  // an iterator class to iterate over truthelements
87  class Iterator : public std::iterator<std::input_iterator_tag, LabelType>
88  {
89  private:
90  std::vector<StoredLabelType>& mLabelsRef; // reference to labels vector
91  int index; // startindex
92  public:
93  Iterator(std::vector<StoredLabelType>& v, int i) : mLabelsRef(v), index(i) {}
94  Iterator(const Iterator& it) : mLabelsRef(it.mLabelsRef), index(it.index) {}
95  Iterator& operator=(const Iterator& it)
96  {
97  mLabelsRef = it.mLabelsRef;
98  index = it.index;
99  return *this;
100  }
101 
102  // go to the next element as indicated by second entry of StoredLabelType
103  Iterator& operator++()
104  {
105  index = getNextIndex(index, mLabelsRef);
106  return *this;
107  }
108 
109  bool operator==(const Iterator& rhs) const { return index == rhs.index; }
110  bool operator!=(const Iterator& rhs) const { return index != rhs.index; }
111  LabelType& operator*() { return dereference(mLabelsRef, index); }
112  };

114  // a proxy class offering a (non-owning) container view on labels of a given data index
115  // container offers basic forward iterator functionality
116  class LabelView
117  {
118  private:
119  int dataindex;
120  std::vector<HeaderElement>& mHeaderArrayRef;
121  std::vector<StoredLabelType>& mLabelArrayRef;
122 
123  public:
124  constexpr LabelView(int i, std::vector<HeaderElement>& v1, std::vector<StoredLabelType>& v2) : dataindex(i), mHeaderArrayRef(v1), mLabelArrayRef(v2) {}
125 
126  // begin + end iterators to loop over the labels
127  Iterator begin()
128  {
129  return dataindex < mHeaderArrayRef.size() ? Iterator(mLabelArrayRef, mHeaderArrayRef[dataindex].index) : Iterator(mLabelArrayRef, 0);
130  }
131 
132  Iterator end()
133  {
134  return dataindex < mHeaderArrayRef.size() ? Iterator(mLabelArrayRef, lastIndex(mHeaderArrayRef[dataindex])) : Iterator(mLabelArrayRef, 0);
135  }
136 
137  // get number of labels
138  size_t size() const { return dataindex < mHeaderArrayRef.size() ? mHeaderArrayRef[dataindex].size : 0; }
139  };

141  static void addLabelImpl(int dataindex, std::vector<HeaderElementContinuous>& headerv, std::vector<LabelType>& labelv, LabelType const& label)
142  {
143  if (dataindex < headerv.size()) {
144  // look if we have something for this dataindex already
145  // must currently be the last one
146  if (dataindex != (headerv.size() - 1)) {
147  throw std::runtime_error("LabelContainer: unsupported code path");
148  }
149  } else {
150  assert(dataindex == headerv.size());
151  // add a new one
152  headerv.emplace_back(0, labelv.size());
153  }
154  auto& header = headerv[dataindex];
155  header.size++;
156  labelv.emplace_back(label);
157  }
158 
159  static void addLabelImpl(int dataindex, std::vector<HeaderElementLinked>& headerv, std::vector<std::pair<LabelType, int>>& labelv, LabelType const& label)
160  {
161  labelv.emplace_back(std::make_pair(label, -1));
162  // something exists for this dataindex already
163  if (dataindex < headerv.size()) {
164  auto& header = headerv[dataindex];
165  // increase size
166  header.size++;
167  const auto lastindex = labelv.size() - 1;
168  // fix link at previous last index
169  labelv[header.lastindex].second = lastindex;
170  // new last index
171  header.lastindex = lastindex;
172  } else {
173  // we support only appending in order?
174  assert(dataindex == headerv.size());
175  // add a new header element; pointing to last slot in mTruthArray
176  const auto lastpos = labelv.size() - 1;
177  headerv.emplace_back(lastpos, lastpos, 1);
178  }
179  }

196  }
197

202  }
