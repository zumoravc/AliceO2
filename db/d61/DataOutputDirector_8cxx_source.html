23 DataOutputDescriptor::DataOutputDescriptor(std::string inString)
24 {
25  // inString is an item consisting of 4 parts which are separated by a ':'
26  // "origin/description/subSpec:treename:col1/col2/col3:filename"
27  // the 1st part is used to create a DataDescriptorMatcher
28  // the other parts are used to fill treename, colnames, and filename
29  // remove all spaces
30  auto cleanString = remove_ws(inString);
31 
32  // reset
33  treename = "";
34  colnames.clear();
35  mfilenameBase = "";
36 
37  // analyze the parts of the input string
38  static const std::regex delim1(":");
39  std::sregex_token_iterator end;
40  std::sregex_token_iterator iter1(cleanString.begin(),
41  cleanString.end(),
42  delim1,
43  -1);
44 
45  // create the DataDescriptorMatcher
46  if (iter1 == end) {
47  return;
48  }
49  auto tableString = iter1->str();
50  matcher = DataDescriptorQueryBuilder::buildNode(tableString);
51 
52  // get the table name
53  auto tableItems = DataDescriptorQueryBuilder::getTokens(tableString);
54  if (!std::string(tableItems[2]).empty()) {
55  tablename = tableItems[2];
56  }
57 
58  // get the tree name
59  // defaul tree name is the table name
60  treename = tablename;
61  ++iter1;
62  if (iter1 == end) {
63  return;
64  }
65  if (!iter1->str().empty()) {
66  treename = iter1->str();
67  }
68 
69  // get column names
70  ++iter1;
71  if (iter1 == end) {
72  return;
73  }
74  if (!iter1->str().empty()) {
75  auto cns = iter1->str();
76 
77  static const std::regex delim2("/");
78  std::sregex_token_iterator iter2(cns.begin(),
79  cns.end(),
80  delim2,
81  -1);
82  for (; iter2 != end; ++iter2) {
83  if (!iter2->str().empty()) {
84  colnames.emplace_back(iter2->str());
85  }
86  }
87  }
88 
89  // get the file name base
90  ++iter1;
91  if (iter1 == end) {
92  return;
93  }
94  if (!iter1->str().empty()) {
95  mfilenameBase = iter1->str();
96  }
97 }

146 void DataOutputDirector::readString(std::string const& keepString)
147 {
148  // the keep-string keepString consists of ','-separated items
149  // create for each item a corresponding DataOutputDescriptor
150  static const std::regex delim(",");
151  std::sregex_token_iterator end;
152  std::sregex_token_iterator iter(keepString.begin(),
153  keepString.end(),
154  delim,
155  -1);
156 
157  // loop over ','-separated items
158  for (; iter != end; ++iter) {
159  auto itemString = iter->str();
160 
161  // create a new DataOutputDescriptor and add it to the list
162  auto dodesc = new DataOutputDescriptor(itemString);
163  if (dodesc->getFilenameBase().empty()) {
164  dodesc->setFilenameBase(mfilenameBasePtr);
165  }
166  mDataOutputDescriptors.emplace_back(dodesc);
167  mfilenameBases.emplace_back(dodesc->getFilenameBase());
168  mtreeFilenames.emplace_back(dodesc->treename + dodesc->getFilenameBase());
169  }
170 
171  // the combination [tree name/file name] must be unique
172  // throw exception if this is not the case
173  auto it = std::unique(mtreeFilenames.begin(), mtreeFilenames.end());
174  if (it != mtreeFilenames.end()) {
175  printOut();
176  LOGP(FATAL, "Dublicate tree names in a file!");
177  }
178 
179  // make unique/sorted list of filenameBases
180  std::sort(mfilenameBases.begin(), mfilenameBases.end());
181  auto last = std::unique(mfilenameBases.begin(), mfilenameBases.end());
182  mfilenameBases.erase(last, mfilenameBases.end());
183 
184  // prepare list mfilePtrs of TFile and mfileCounts
185  for (auto fn : mfilenameBases) {
186  mfilePtrs.emplace_back(new TFile());
187  mfileCounts.emplace_back(-1);
188  }
189 }

247 std::tuple<std::string, std::string, int> DataOutputDirector::readJsonDocument(Document* jsonDocument)
248 {
249  std::string smc(":");
250  std::string slh("/");
251  const char* itemName;
252 
253  // initialisations
254  std::string dfn("");
255  std::string fmode("");
256  int ntfm = -1;
257 
258  // is it a proper json document?
259  if (jsonDocument->HasParseError()) {
260  LOGP(ERROR, "Check the JSON document! There is a problem with the format!");
261  return memptyanswer;
262  }
263 
264  // OutputDirector
265  itemName = "OutputDirector";
266  const Value& dodirItem = (*jsonDocument)[itemName];
267  if (!dodirItem.IsObject()) {
268  LOGP(ERROR, "Check the JSON document! Couldn't find an \"{}\" object!", itemName);
269  return memptyanswer;
270  }
271 
272  // now read various items
273  itemName = "debugmode";
274  if (dodirItem.HasMember(itemName)) {
275  if (dodirItem[itemName].IsBool()) {
276  mdebugmode = dodirItem[itemName].GetBool();
277  } else {
278  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a boolean!", itemName);
279  return memptyanswer;
280  }
281  } else {
282  mdebugmode = false;
283  }
284 
285  if (mdebugmode) {
286  StringBuffer buffer;
287  buffer.Clear();
288  Writer<rapidjson::StringBuffer> writer(buffer);
289  dodirItem.Accept(writer);
290  LOGP(INFO, "OutputDirector object: {}", std::string(buffer.GetString()));
291  }
292 
293  itemName = "resfile";
294  if (dodirItem.HasMember(itemName)) {
295  if (dodirItem[itemName].IsString()) {
296  dfn = dodirItem[itemName].GetString();
297  setFilenameBase(dfn);
298  } else {
299  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
300  return memptyanswer;
301  }
302  }
303 
304  itemName = "resfilemode";
305  if (dodirItem.HasMember(itemName)) {
306  if (dodirItem[itemName].IsString()) {
307  fmode = dodirItem[itemName].GetString();
308  } else {
309  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
310  return memptyanswer;
311  }
312  }
313 
314  itemName = "ntfmerge";
315  if (dodirItem.HasMember(itemName)) {
316  if (dodirItem[itemName].IsNumber()) {
317  ntfm = dodirItem[itemName].GetInt();
318  } else {
319  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a number!", itemName);
320  return memptyanswer;
321  }
322  }
323 
324  itemName = "OutputDescriptors";
325  if (dodirItem.HasMember(itemName)) {
326  if (!dodirItem[itemName].IsArray()) {
327  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be an array!", itemName);
328  return memptyanswer;
329  }
330 
331  // loop over DataOutputDescriptors
332  for (auto& dodescItem : dodirItem[itemName].GetArray()) {
333  if (!dodescItem.IsObject()) {
334  LOGP(ERROR, "Check the JSON document! \"{}\" must be objects!", itemName);
335  return memptyanswer;
336  }
337 
338  std::string dodString = "";
339  itemName = "table";
340  if (dodescItem.HasMember(itemName)) {
341  if (dodescItem[itemName].IsString()) {
342  dodString += dodescItem[itemName].GetString();
343  } else {
344  LOGP(ERROR, "Check the JSON document! \"{}\" must be a string!", itemName);
345  return memptyanswer;
346  }
347  }
348  dodString += smc;
349  itemName = "treename";
350  if (dodescItem.HasMember(itemName)) {
351  if (dodescItem[itemName].IsString()) {
352  dodString += dodescItem[itemName].GetString();
353  } else {
354  LOGP(ERROR, "Check the JSON document! \"{}\" must be a string!", itemName);
355  return memptyanswer;
356  }
357  }
358  dodString += smc;
359  itemName = "columns";
360  if (dodescItem.HasMember(itemName)) {
361  if (dodescItem[itemName].IsArray()) {
362  auto columnNames = dodescItem[itemName].GetArray();
363  for (auto& c : columnNames) {
364  dodString += (c == columnNames[0]) ? c.GetString() : slh + c.GetString();
365  }
366  } else {
367  LOGP(ERROR, "Check the JSON document! \"{}\" must be an array!", itemName);
368  return memptyanswer;
369  }
370  }
371  dodString += smc;
372  itemName = "filename";
373  if (dodescItem.HasMember(itemName)) {
374  if (dodescItem[itemName].IsString()) {
375  dodString += dodescItem[itemName].GetString();
376  } else {
377  LOGP(ERROR, "Check the JSON document! \"{}\" must be a string!", itemName);
378  return memptyanswer;
379  }
380  }
381 
382  // convert s to DataOutputDescription object
383  readString(dodString);
384  }
385  }
386 
387  // print the DataOutputDirector
388  if (mdebugmode) {
389  printOut();
390  }
391 
392  return std::make_tuple(dfn, fmode, ntfm);
393 }

428 TFile* DataOutputDirector::getDataOutputFile(DataOutputDescriptor* dodesc,
429  int ntf, int ntfmerge,
430  std::string filemode)
431 {
432  // initialisation
433  TFile* filePtr = nullptr;
434 
435  // search dodesc->filename in mfilenameBases and return corresponding filePtr
436  auto it = std::find(mfilenameBases.begin(), mfilenameBases.end(), dodesc->getFilenameBase());
437  if (it != mfilenameBases.end()) {
438  int ind = std::distance(mfilenameBases.begin(), it);
439 
440  // check if new version of file needs to be opened
441  int fcnt = (int)(ntf / ntfmerge);
442  if ((ntf % ntfmerge) == 0 && fcnt > mfileCounts[ind]) {
443  if (mfilePtrs[ind]) {
444  mfilePtrs[ind]->Close();
445  }
446 
447  mfileCounts[ind] = fcnt;
448  auto fn = mfilenameBases[ind] + "_" + std::to_string(mfileCounts[ind]) + ".root";
449  mfilePtrs[ind] = new TFile(fn.c_str(), filemode.c_str());
450  }
451  filePtr = mfilePtrs[ind];
452  filePtr->cd();
453  }
454 
455  return filePtr;
456 }
