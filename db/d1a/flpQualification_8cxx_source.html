45 DataProcessorSpec templateProcessor(std::string const& inputType)
46 {
47  std::vector<InputSpec> inputs;
48  if (inputType == "readout") {
49  inputs.emplace_back("x", ConcreteDataTypeMatcher{"ITS", "RAWDATA"}, Lifetime::Timeframe);
50  } else {
51  inputs.emplace_back("x", ConcreteDataTypeMatcher{"FLP", "RAWDATA"}, Lifetime::Timeframe);
52  }
53  return DataProcessorSpec{"some-processor",
54  inputs,
55  {
56  OutputSpec{"TST", "P", 0, Lifetime::Timeframe},
57  },
58  // The processor is stateful. We want to call srand only
59  // once, and then return the callback to be invoked for every message.
60  AlgorithmSpec{[](InitContext& setup) {
61  srand(setup.services().get<ParallelContext>().index1D());
62  return adaptStateless([counter = std::make_shared<int>(0)](ParallelContext& parallelInfo, InputRecord& inputs, DataAllocator& outputs) {
63  auto values = inputs.get("x");
64  // Create a single output.
65  size_t index = parallelInfo.index1D();
66  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(values);
67  if (dh) {
68  LOG(INFO) << "some-processor" << index << ": "
69  << dh->dataOrigin.as<std::string>() << "/" << dh->dataDescription.as<std::string>() << "/"
70  << dh->subSpecification << " payload size " << dh->payloadSize;
71  }
72  auto aData =
73  outputs.make<int>(Output{"TST", "P", static_cast<o2::header::DataHeader::SubSpecificationType>(index)}, 1);
74  aData[0] = (*counter)++;
75  });
76  }}};
77 }

89 WorkflowSpec defineDataProcessing(ConfigContext const& config)
90 {
91  size_t jobs = config.options().get<int>("2-layer-jobs");
92  size_t stages = config.options().get<int>("3-layer-pipelining");
93  std::string inputType = config.options().get<std::string>("input-type");
94  if (inputType != "readout" && inputType != "stfb") {
95  throw std::runtime_error("Unknown input type " + inputType + ". Available options are `readout' and `stfb'.");
96  }
97 
107  Outputs readoutProxyOutput;
108  if (inputType == "readout") {
109  // we keep the hardcoded value of ITS RAWDATA as it was before
110  // note that this will translate into ConcreteDataMatcher with subspec 0
111  readoutProxyOutput.emplace_back("ITS", "RAWDATA");
112  } else {
113  // need one output per job in the 2nd level, but use subSpec-agnostic matcher
114  // if there is only one job
115  if (jobs > 1) {
116  for (auto i = 0; i < jobs; i++) {
117  readoutProxyOutput.emplace_back("FLP", "RAWDATA", i);
118  }
119  } else {
120  readoutProxyOutput.emplace_back(ConcreteDataTypeMatcher{"FLP", "RAWDATA"});
121  }
122  readoutProxyOutput.emplace_back("FLP", "DISTSUBTIMEFRAME", 0);
123  }
124  DataProcessorSpec readoutProxy = specifyExternalFairMQDeviceProxy(
125  "readout-proxy",
126  std::move(readoutProxyOutput),
127  "type=pair,method=connect,address=ipc:///tmp/readout-pipe-0,rateLogging=1,transport=shmem",
128  inputType == "readout" ? readoutAdapter({"ITS", "RAWDATA"}) : dplModelAdaptor({ConcreteDataTypeMatcher{"FLP", "RAWDATA"}, {"FLP", "DISTSUBTIMEFRAME", 0}}));
129 
130  // This is an example of how we can parallelize by subSpec.
131  // templatedProcessor will be instanciated N times and the lambda function
132  // passed to the parallel statement will be applied to each one of the
133  // instances in order to modify it. Parallel will also make sure the name of
134  // the instance is amended from "some-processor" to "some-processor-<index>".
135  // This is to simulate processing of different input channels in parallel on
136  // the FLP.
137  auto dataParallelLayer = parallel(templateProcessor(inputType), jobs, [](DataProcessorSpec& spec, size_t index) {
138  DataSpecUtils::updateMatchingSubspec(spec.inputs[0], index);
139  DataSpecUtils::updateMatchingSubspec(spec.outputs[0], index);
140  });
141 
142  // This is a set of processor which will be able to process consistent
143  // timeslices on the FLP. I.e. time units where all the channels are
144  // present.
145  DataProcessorSpec timeParallelProcessor = timePipeline(
146  DataProcessorSpec{
147  "merger",
148  mergeInputs(InputSpec{"x", "TST", "P"},
149  jobs,
150  [](InputSpec& input, size_t index) {
151  DataSpecUtils::updateMatchingSubspec(input, index);
152  }),
153  {OutputSpec{{"label"}, "TST", "merger_output"}},
154  AlgorithmSpec{[](InitContext& setup) {
155  return [](ProcessingContext& ctx) {
156  ctx.outputs().make<int>(OutputRef("label", 0), 1);
157  };
158  }}},
159  stages);
160 
161  // proxyOut is the component which will forward things to be
162  // sent to the EPN to the data distribution device.
163  // FIXME: actually connect to DataDistribution to push data somewhere else.
164  Inputs proxyInputs{InputSpec{"x", "TST", "merger_output"}};
165  if (inputType == "stfb") {
166  proxyInputs.emplace_back("y", "FLP", "DISTSUBTIMEFRAME", 0);
167  }
168  DataProcessorSpec proxyOut{
169  "proxyout",
170  std::move(proxyInputs),
171  Outputs{},
172  AlgorithmSpec{[](InitContext& setup) {
173  return [](ProcessingContext& ctx) {
174  };
175  }}};
176 
177  WorkflowSpec workflow;
178  workflow.emplace_back(readoutProxy);
179  // The OutputSpec definition of the template processor ignores the subSpec, when creating
180  // parallel instances, the subSpec is set per intance. If we only have one instance, the
181  // unspecified subSpec needs to be kept for the matcher rather than creating one
182  // "parallel" instance with concrete subSpec.
183  if (jobs > 1) {
184  workflow.insert(workflow.end(), dataParallelLayer.begin(), dataParallelLayer.end());
185  } else {
186  workflow.emplace_back(templateProcessor(inputType));
187  }
188  workflow.emplace_back(timeParallelProcessor);
189  workflow.emplace_back(proxyOut);
190 
191  return workflow;
192 }
