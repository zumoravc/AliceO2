41 class ClusterSamplerTask
42 {
43  public:
44  //_________________________________________________________________________________________________
45  void init(framework::InitContext& ic)
46  {
48  LOG(INFO) << "initializing cluster sampler";
49 
50  auto inputFileName = ic.options().get<std::string>("infile");
51  mInputFile.open(inputFileName, ios::binary);
52  if (!mInputFile.is_open()) {
53  throw invalid_argument("Cannot open input file" + inputFileName);
54  }
55 
56  auto stop = [this]() {
58  LOG(INFO) << "stop cluster sampler";
59  this->mInputFile.close();
60  };
61  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
62  }
63 
64  //_________________________________________________________________________________________________
65  void run(framework::ProcessingContext& pc)
66  {
68 
69  int event(0);
70  mInputFile.read(reinterpret_cast<char*>(&event), SSizeOfInt);
71  if (mInputFile.fail()) {
72  pc.services().get<ControlService>().endOfStream();
73  return; // probably reached eof
74  }
75 
76  int nClusters(0);
77  mInputFile.read(reinterpret_cast<char*>(&nClusters), SSizeOfInt);
78  if (nClusters == 0) {
79  return; // skip empty event
80  }
81 
82  // create the output message
83  auto size = nClusters * SSizeOfClusterStruct;
84  auto msgOut = pc.outputs().make<char>(Output{"MCH", "CLUSTERS", 0, Lifetime::Timeframe}, SHeaderSize + size);
85  if (msgOut.size() != SHeaderSize + size) {
86  throw length_error("incorrect message payload");
87  }
88 
89  auto bufferPtr = msgOut.data();
90 
91  // fill header info
92  memcpy(bufferPtr, &event, SSizeOfInt);
93  bufferPtr += SSizeOfInt;
94  memcpy(bufferPtr, &nClusters, SSizeOfInt);
95  bufferPtr += SSizeOfInt;
96 
97  // fill tracks and clusters info
98  mInputFile.read(bufferPtr, size);
99  }
100 
101  private:
102  static constexpr int SSizeOfInt = sizeof(int);
103  static constexpr int SHeaderSize = 2 * SSizeOfInt;
104  static constexpr int SSizeOfClusterStruct = sizeof(ClusterStruct);
105 
106  std::ifstream mInputFile{};
107 };
