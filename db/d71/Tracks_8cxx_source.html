59 bool Tracks::processTrack(const o2::tpc::TrackTPC& track)
60 {
61  // ===| variables required for cutting and filling |===
62  const auto eta = track.getEta();
63  const auto phi = track.getPhi();
64  const auto pt = track.getPt();
65  const auto sign = track.getSign();
66  const auto nCls = track.getNClusterReferences();
67 
68  // ===| fill one histogram before any cuts |===
69  mHist1D[0].Fill(nCls);
70 
71  // ===| cuts |===
72  // hard coded cuts. Should be more configural in future
73  if (nCls < 20) {
74  return false;
75  }
76 
77  // ===| 1D histogram filling |===
78  mHist1D[1].Fill(nCls);
79  mHist1D[2].Fill(eta);
80 
81  if (eta > 0.) {
82  mHist1D[3].Fill(phi);
83  } else {
84  mHist1D[4].Fill(phi);
85  }
86 
87  mHist1D[5].Fill(pt);
88  mHist1D[6].Fill(sign);
89 
90  // ===| 2D histogram filling |===
91  mHist2D[0].Fill(eta, nCls);
92 
93  if (eta > 0.) {
94  mHist2D[1].Fill(phi, nCls);
95  } else {
96  mHist2D[2].Fill(phi, nCls);
97  }
98 
99  mHist2D[3].Fill(pt, nCls);
100  mHist2D[4].Fill(phi, eta);
101 
102  if (sign < 0.) {
103  mHist2D[5].Fill(phi, eta);
104  } else {
105  mHist2D[6].Fill(phi, eta);
106  }
107 
108  return true;
109 }
