106 void GPUTPCTracker::RegisterMemoryAllocation()
107 {
108  AllocateAndInitializeLate();
109  bool reuseCondition = !mRec->GetDeviceProcessingSettings().keepDisplayMemory && mRec->GetDeviceProcessingSettings().trackletSelectorInPipeline && ((mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCSliceTracking) || mRec->GetDeviceProcessingSettings().nThreads == 1);
110  GPUMemoryReuse reLinks{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataLinks, (unsigned short)(mISlice % mRec->GetDeviceProcessingSettings().nStreams)};
111  mMemoryResLinks = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataLinks, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceLinks", reLinks);
112  mMemoryResSliceScratch = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceScratch");
113  mMemoryResSliceInput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataInput, GPUMemoryResource::MEMORY_INPUT | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceInput");
114  GPUMemoryReuse reWeights{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataWeights, (unsigned short)(mISlice % mRec->GetDeviceProcessingSettings().nStreams)};
115  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataWeights, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceWeights", reWeights);
116  GPUMemoryReuse reScratch{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerScratch, (unsigned short)(mISlice % mRec->GetDeviceProcessingSettings().nStreams)};
117  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCTrackerScratch", reScratch);
118  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratchHost, GPUMemoryResource::MEMORY_SCRATCH_HOST, "TPCTrackerHost");
119  mMemoryResCommon = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersCommon, GPUMemoryResource::MEMORY_PERMANENT, "TPCTrackerCommon");
120  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataRows, GPUMemoryResource::MEMORY_PERMANENT, "TPCSliceRows");
121 
122  auto type = GPUMemoryResource::MEMORY_OUTPUT;
123  if (mRec->GetDeviceProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) { // For individual scheme, we allocate tracklets separately, and change the type for the following allocations to custom
124  type = GPUMemoryResource::MEMORY_CUSTOM;
125  mMemoryResTracklets = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersTracklets, type, "TPCTrackerTracklets");
126  }
127  mMemoryResOutput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersOutput, type, "TPCTrackerTracks");
128 }

196 GPUh() void GPUTPCTracker::WriteOutput()
197 {
198  mOutput->SetNTracks(0);
199  mOutput->SetNLocalTracks(0);
200  mOutput->SetNTrackClusters(0);
201 
202  if (mCommonMem->nTracks == 0) {
203  return;
204  }
205  if (mCommonMem->nTracks > GPUCA_MAX_SLICE_NTRACK) {
206  GPUError("Maximum number of tracks exceeded, cannot store");
207  return;
208  }
209 
210  int nStoredHits = 0;
211  int nStoredTracks = 0;
212  int nStoredLocalTracks = 0;
213 
214  GPUTPCTrack* out = mOutput->FirstTrack();
215 
216  trackSortData* trackOrder = new trackSortData[mCommonMem->nTracks];
217  for (unsigned int i = 0; i < mCommonMem->nTracks; i++) {
218  trackOrder[i].fTtrack = i;
219  trackOrder[i].fSortVal = mTracks[trackOrder[i].fTtrack].NHits() / 1000.f + mTracks[trackOrder[i].fTtrack].Param().GetZ() * 100.f + mTracks[trackOrder[i].fTtrack].Param().GetY();
220  }
221  std::sort(trackOrder, trackOrder + mCommonMem->nLocalTracks, SortComparison<trackSortData>); // TODO: Check why this sorting affects the merging efficiency!
222  std::sort(trackOrder + mCommonMem->nLocalTracks, trackOrder + mCommonMem->nTracks, SortComparison<trackSortData>);
223 
224  for (unsigned int iTrTmp = 0; iTrTmp < mCommonMem->nTracks; iTrTmp++) {
225  const int iTr = trackOrder[iTrTmp].fTtrack;
226  GPUTPCTrack& iTrack = mTracks[iTr];
227 
228  *out = iTrack;
229  int nClu = 0;
230  int iID = iTrack.FirstHitID();
231 
232  for (int ith = 0; ith < iTrack.NHits(); ith++) {
233  const GPUTPCHitId& ic = mTrackHits[iID + ith];
234  int iRow = ic.RowIndex();
235  int ih = ic.HitIndex();
236 
237  const GPUTPCRow& row = mData.Row(iRow);
238  int clusterIndex = mData.ClusterDataIndex(row, ih);
239 #ifdef GPUCA_ARRAY_BOUNDS_CHECKS
240  if (ih >= row.NHits() || ih < 0) {
241  GPUError("Array out of bounds access (Sector Row) (Hit %d / %d - NumC %d): Sector %d Row %d Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih);
242  fflush(stdout);
243  continue;
244  }
245  if (clusterIndex >= NHitsTotal() || clusterIndex < 0) {
246  GPUError("Array out of bounds access (Cluster Data) (Hit %d / %d - NumC %d): Sector %d Row %d Hit %d, Clusterdata Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih, clusterIndex);
247  fflush(stdout);
248  continue;
249  }
250 #endif
251 
252  float origX, origY, origZ;
253  unsigned char flags;
254  unsigned short amp;
255  int id;
256  if (Param().earlyTpcTransform) {
257  origX = mData.ClusterData()[clusterIndex].x;
258  origY = mData.ClusterData()[clusterIndex].y;
259  origZ = mData.ClusterData()[clusterIndex].z;
260  flags = mData.ClusterData()[clusterIndex].flags;
261  amp = mData.ClusterData()[clusterIndex].amp;
262  id = mData.ClusterData()[clusterIndex].id;
263  } else {
264  const ClusterNativeAccess& cls = *mConstantMem->ioPtrs.clustersNative;
265  id = clusterIndex + cls.clusterOffset[mISlice][0];
266  GPUTPCConvertImpl::convert(*mConstantMem, mISlice, iRow, cls.clustersLinear[id].getPad(), cls.clustersLinear[id].getTime(), origX, origY, origZ);
267  flags = cls.clustersLinear[id].getFlags();
268  amp = cls.clustersLinear[id].qTot;
269  }
270  GPUTPCSliceOutCluster c;
271  c.Set(id, iRow, flags, amp, origX, origY, origZ);
272 #ifdef GPUCA_TPC_RAW_PROPAGATE_PAD_ROW_TIME
273  c.mPad = mData.ClusterData()[clusterIndex].pad;
274  c.mTime = mData.ClusterData()[clusterIndex].time;
275 #endif
276  out->SetOutTrackCluster(nClu, c);
277  nClu++;
278  }
279 
280  nStoredTracks++;
281  if (iTr < mCommonMem->nLocalTracks) {
282  nStoredLocalTracks++;
283  }
284  nStoredHits += nClu;
285  out->SetNHits(nClu);
286  out = out->NextTrack();
287  }
288  delete[] trackOrder;
289 
290  mOutput->SetNTracks(nStoredTracks);
291  mOutput->SetNLocalTracks(nStoredLocalTracks);
292  mOutput->SetNTrackClusters(nStoredHits);
293  if (Param().debugLevel >= 3) {
294  GPUInfo("Slice %d, Output: Tracks %d, local tracks %d, hits %d", mISlice, nStoredTracks, nStoredLocalTracks, nStoredHits);
295  }
296 }
