93 int GPUReconstruction::Init()
94 {
95  if (mMaster) {
96  throw std::runtime_error("Must not call init on slave!");
97  }
98  int retVal = InitPhaseBeforeDevice();
99  if (retVal) {
100  return retVal;
101  }
102  for (unsigned int i = 0; i < mSlaves.size(); i++) {
103  retVal = mSlaves[i]->InitPhaseBeforeDevice();
104  if (retVal) {
105  GPUError("Error initialization slave (before deviceinit)");
106  return retVal;
107  }
108  mNStreams = std::max(mNStreams, mSlaves[i]->mNStreams);
109  mHostMemorySize = std::max(mHostMemorySize, mSlaves[i]->mHostMemorySize);
110  mDeviceMemorySize = std::max(mDeviceMemorySize, mSlaves[i]->mDeviceMemorySize);
111  }
112  if (InitDevice()) {
113  return 1;
114  }
115  mHostMemoryPoolEnd = (char*)mHostMemoryBase + mHostMemorySize;
116  mDeviceMemoryPoolEnd = (char*)mDeviceMemoryBase + mDeviceMemorySize;
117  if (InitPhasePermanentMemory()) {
118  return 1;
119  }
120  for (unsigned int i = 0; i < mSlaves.size(); i++) {
121  mSlaves[i]->mDeviceMemorySize = mDeviceMemorySize;
122  mSlaves[i]->mHostMemorySize = mHostMemorySize;
123  mSlaves[i]->mDeviceMemoryBase = mDeviceMemoryPermanent;
124  mSlaves[i]->mHostMemoryBase = mHostMemoryPermanent;
125  mSlaves[i]->mHostMemoryPoolEnd = mHostMemoryPoolEnd;
126  mSlaves[i]->mDeviceMemoryPoolEnd = mDeviceMemoryPoolEnd;
127  if (mSlaves[i]->InitDevice()) {
128  GPUError("Error initialization slave (deviceinit)");
129  return 1;
130  }
131  if (mSlaves[i]->InitPhasePermanentMemory()) {
132  GPUError("Error initialization slave (permanent memory)");
133  return 1;
134  }
135  mDeviceMemoryPermanent = mSlaves[i]->mDeviceMemoryPermanent;
136  mHostMemoryPermanent = mSlaves[i]->mHostMemoryPermanent;
137  }
138  retVal = InitPhaseAfterDevice();
139  if (retVal) {
140  return retVal;
141  }
142  for (unsigned int i = 0; i < mSlaves.size(); i++) {
143  mSlaves[i]->mDeviceMemoryPermanent = mDeviceMemoryPermanent;
144  mSlaves[i]->mHostMemoryPermanent = mHostMemoryPermanent;
145  retVal = mSlaves[i]->InitPhaseAfterDevice();
146  if (retVal) {
147  GPUError("Error initialization slave (after device init)");
148  return retVal;
149  }
150  }
151  return 0;
152 }

154 int GPUReconstruction::InitPhaseBeforeDevice()
155 {
156  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_AUTO) {
157  mDeviceProcessingSettings.memoryAllocationStrategy = IsGPU() ? GPUMemoryResource::ALLOCATION_GLOBAL : GPUMemoryResource::ALLOCATION_INDIVIDUAL;
158  }
159  if (mDeviceProcessingSettings.debugLevel >= 4) {
160  mDeviceProcessingSettings.keepAllMemory = true;
161  }
162  if (mDeviceProcessingSettings.eventDisplay || mDeviceProcessingSettings.keepAllMemory) {
163  mDeviceProcessingSettings.keepDisplayMemory = true;
164  }
165  if (mDeviceProcessingSettings.debugLevel < 6) {
166  mDeviceProcessingSettings.debugMask = 0;
167  }
168  if (mDeviceProcessingSettings.debugLevel < 1) {
169  mDeviceProcessingSettings.deviceTimers = false;
170  }
171  if (mDeviceProcessingSettings.debugLevel >= 6 && mDeviceProcessingSettings.comparableDebutOutput) {
172  mDeviceProcessingSettings.nTPCClustererLanes = 1;
173  if (mDeviceProcessingSettings.trackletConstructorInPipeline < 0) {
174  mDeviceProcessingSettings.trackletConstructorInPipeline = 1;
175  }
176  if (mDeviceProcessingSettings.trackletSelectorInPipeline < 0) {
177  mDeviceProcessingSettings.trackletSelectorInPipeline = 1;
178  }
179  if (mDeviceProcessingSettings.trackletSelectorSlices < 0) {
180  mDeviceProcessingSettings.trackletSelectorSlices = 1;
181  }
182  }
183  if (mDeviceProcessingSettings.tpcCompressionGatherMode < 0) {
184  mDeviceProcessingSettings.tpcCompressionGatherMode = (mRecoStepsGPU & GPUDataTypes::RecoStep::TPCCompression) ? 2 : 0;
185  }
186  if (!(mRecoStepsGPU & GPUDataTypes::RecoStep::TPCMerging)) {
187  mDeviceProcessingSettings.mergerSortTracks = false;
188  }
189  if (!IsGPU()) {
190  mDeviceProcessingSettings.nDeviceHelperThreads = 0;
191  }
192 
193 #ifndef HAVE_O2HEADERS
194  mRecoSteps.setBits(RecoStep::ITSTracking, false);
195  mRecoSteps.setBits(RecoStep::TRDTracking, false);
196  mRecoSteps.setBits(RecoStep::TPCConversion, false);
197  mRecoSteps.setBits(RecoStep::TPCCompression, false);
198  mRecoSteps.setBits(RecoStep::TPCdEdx, false);
199 #endif
200  mRecoStepsGPU &= mRecoSteps;
201  mRecoStepsGPU &= AvailableRecoSteps();
202  if (!IsGPU()) {
203  mRecoStepsGPU.set((unsigned char)0);
204  }
205  if (param().rec.NonConsecutiveIDs) {
206  param().rec.DisableRefitAttachment = 0xFF;
207  }
208  if (!(mRecoStepsGPU & RecoStep::TPCMerging)) {
209  mDeviceProcessingSettings.fullMergerOnGPU = false;
210  }
211  if (mDeviceProcessingSettings.debugLevel || !mDeviceProcessingSettings.fullMergerOnGPU) {
212  mDeviceProcessingSettings.delayedOutput = false;
213  }
214 
215  UpdateSettings();
216  GPUCA_GPUReconstructionUpdateDefailts();
217  if (!mDeviceProcessingSettings.trackletConstructorInPipeline) {
218  mDeviceProcessingSettings.trackletSelectorInPipeline = false;
219  }
220 
221  mMemoryScalers->factor = mDeviceProcessingSettings.memoryScalingFactor;
222 
223 #ifdef WITH_OPENMP
224  if (mDeviceProcessingSettings.nThreads <= 0) {
225  mDeviceProcessingSettings.nThreads = omp_get_max_threads();
226  } else {
227  omp_set_num_threads(mDeviceProcessingSettings.nThreads);
228  }
229 #else
230  mDeviceProcessingSettings.nThreads = 1;
231 #endif
232  mMaxThreads = std::max(mMaxThreads, mDeviceProcessingSettings.nThreads);
233  if (IsGPU()) {
234  mNStreams = std::max(mDeviceProcessingSettings.nStreams, 3);
235  }
236 
237  mDeviceMemorySize = mHostMemorySize = 0;
238  for (unsigned int i = 0; i < mChains.size(); i++) {
239  mChains[i]->RegisterPermanentMemoryAndProcessors();
240  size_t memGpu, memHost;
241  mChains[i]->MemorySize(memGpu, memHost);
242  mDeviceMemorySize += memGpu;
243  mHostMemorySize += memHost;
244  }
245  if (mDeviceProcessingSettings.forceMemoryPoolSize && mDeviceProcessingSettings.forceMemoryPoolSize <= 2 && CanQueryMaxMemory()) {
246  mDeviceMemorySize = mDeviceProcessingSettings.forceMemoryPoolSize;
247  } else if (mDeviceProcessingSettings.forceMemoryPoolSize > 2) {
248  mDeviceMemorySize = mHostMemorySize = mDeviceProcessingSettings.forceMemoryPoolSize;
249  }
250 
251  for (unsigned int i = 0; i < mProcessors.size(); i++) {
252  (mProcessors[i].proc->*(mProcessors[i].RegisterMemoryAllocation))();
253  }
254 
255  return 0;
256 }

302 int GPUReconstruction::Exit()
303 {
304  if (!mInitialized) {
305  return 1;
306  }
307  for (unsigned int i = 0; i < mSlaves.size(); i++) {
308  if (mSlaves[i]->Exit()) {
309  GPUError("Error exiting slave");
310  }
311  }
312 
313  mChains.clear(); // Make sure we destroy a possible ITS GPU tracker before we call the destructors
314  mHostConstantMem.reset(); // Reset these explicitly before the destruction of other members unloads the library
315  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
316  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
317  if (mMemoryResources[i].mReuse >= 0) {
318  continue;
319  }
320  operator delete(mMemoryResources[i].mPtrDevice GPUCA_OPERATOR_NEW_ALIGNMENT);
321  mMemoryResources[i].mPtr = mMemoryResources[i].mPtrDevice = nullptr;
322  }
323  }
324  mMemoryResources.clear();
325  if (mInitialized) {
326  ExitDevice();
327  }
328  mInitialized = false;
329  return 0;

386 size_t GPUReconstruction::AllocateRegisteredMemoryHelper(GPUMemoryResource* res, void*& ptr, void*& memorypool, void* memorybase, size_t memorysize, void* (GPUMemoryResource::*setPtr)(void*), void*& memorypoolend)
387 {
388  if (res->mReuse >= 0) {
389  ptr = (&ptr == &res->mPtrDevice) ? mMemoryResources[res->mReuse].mPtrDevice : mMemoryResources[res->mReuse].mPtr;
390  size_t retVal = (char*)((res->*setPtr)(ptr)) - (char*)(ptr);
391  if (retVal > mMemoryResources[res->mReuse].mSize) {
392  throw std::bad_alloc();
393  }
394  if (mDeviceProcessingSettings.debugLevel >= 5) {
395  std::cout << "Reused " << res->mName << ": " << retVal << "\n";
396  }
397  return retVal;
398  }
399  if (memorypool == nullptr) {
400  GPUInfo("Memory pool uninitialized");
401  throw std::bad_alloc();
402  }
403  size_t retVal;
404  size_t minSize = res->mOverrideSize;
405  if (IsGPU() && minSize < GPUCA_BUFFER_ALIGNMENT) {
406  minSize = GPUCA_BUFFER_ALIGNMENT;
407  }
408  if ((res->mType & GPUMemoryResource::MEMORY_STACK) && memorypoolend) {
409  retVal = (char*)((res->*setPtr)((char*)1)) - (char*)(1);
410  memorypoolend = (void*)((char*)memorypoolend - GPUProcessor::getAlignmentMod<GPUCA_MEMALIGN>(memorypoolend));
411  if (retVal < minSize) {
412  retVal = minSize;
413  }
414  retVal += GPUProcessor::getAlignment<GPUCA_MEMALIGN>(retVal);
415  memorypoolend = (char*)memorypoolend - retVal;
416  ptr = memorypoolend;
417  (res->*setPtr)(ptr);
418  } else {
419  ptr = memorypool;
420  memorypool = (char*)((res->*setPtr)(ptr));
421  retVal = (char*)memorypool - (char*)ptr;
422  if (retVal < minSize) {
423  retVal = minSize;
424  memorypool = (char*)ptr + minSize;
425  }
426  memorypool = (void*)((char*)memorypool + GPUProcessor::getAlignment<GPUCA_MEMALIGN>(memorypool));
427  }
428  if (memorypoolend ? (memorypool > memorypoolend) : ((size_t)((char*)memorypool - (char*)memorybase) > memorysize)) {
429  std::cout << "Memory pool size exceeded (" << res->mName << ": " << (memorypoolend ? (memorysize + ((char*)memorypool - (char*)memorypoolend)) : (char*)memorypool - (char*)memorybase) << " < " << memorysize << "\n";
430  throw std::bad_alloc();
431  }
432  if (mDeviceProcessingSettings.debugLevel >= 5) {
433  std::cout << "Allocated " << res->mName << ": " << retVal << " - available: " << (memorypoolend ? ((char*)memorypoolend - (char*)memorypool) : (memorysize - ((char*)memorypool - (char*)memorybase))) << "\n";
434  }
435  return retVal;
436 }

438 size_t GPUReconstruction::AllocateRegisteredMemory(short ires, GPUOutputControl* control)
439 {
440  GPUMemoryResource* res = &mMemoryResources[ires];
441  if ((res->mType & GPUMemoryResource::MEMORY_PERMANENT) && res->mPtr != nullptr) {
442  ResetRegisteredMemoryPointers(ires);
443  } else if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL && (control == nullptr || control->OutputType == GPUOutputControl::AllocateInternal)) {
444  if (!(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
445  if (res->mPtrDevice && res->mReuse < 0) {
446  operator delete(res->mPtrDevice GPUCA_OPERATOR_NEW_ALIGNMENT);
447  }
448  res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
449  if (res->mReuse >= 0) {
450  if (res->mSize > mMemoryResources[res->mReuse].mSize) {
451  GPUError("Invalid reuse, insufficient size: %lld < %lld", (long long int)mMemoryResources[res->mReuse].mSize, (long long int)res->mSize);
452  throw std::bad_alloc();
453  }
454  res->mPtrDevice = mMemoryResources[res->mReuse].mPtrDevice;
455  } else {
456  res->mPtrDevice = operator new(res->mSize + GPUCA_BUFFER_ALIGNMENT GPUCA_OPERATOR_NEW_ALIGNMENT);
457  }
458  res->mPtr = GPUProcessor::alignPointer<GPUCA_BUFFER_ALIGNMENT>(res->mPtrDevice);
459  res->SetPointers(res->mPtr);
460  if (mDeviceProcessingSettings.debugLevel >= 5) {
461  std::cout << (res->mReuse >= 0 ? "Reused " : "Allocated ") << res->mName << ": " << res->mSize << "\n";
462  }
463  }
464  } else {
465  if (res->mPtr != nullptr) {
466  GPUError("Double allocation! (%s)", res->mName);
467  throw std::bad_alloc();
468  }
469  if ((!IsGPU() || (res->mType & GPUMemoryResource::MEMORY_HOST) || mDeviceProcessingSettings.keepDisplayMemory) && !(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) { // keepAllMemory --> keepDisplayMemory
470  if (control && control->OutputType == GPUOutputControl::UseExternalBuffer) {
471  if (control->OutputAllocator) {
472  res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
473  res->mPtr = control->OutputAllocator(res->mSize);
474  res->SetPointers(res->mPtr);
475  } else {
476  void* dummy = nullptr;
477  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, control->OutputPtr, control->OutputBase, control->OutputMaxSize, &GPUMemoryResource::SetPointers, dummy);
478  }
479  } else {
480  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, mHostMemoryPool, mHostMemoryBase, mHostMemorySize, &GPUMemoryResource::SetPointers, mHostMemoryPoolEnd);
481  }
482  }
483  if (IsGPU() && (res->mType & GPUMemoryResource::MEMORY_GPU)) {
484  if (res->mProcessor->mDeviceProcessor == nullptr) {
485  GPUError("Device Processor not set (%s)", res->mName);
486  throw std::bad_alloc();
487  }
488  size_t size = AllocateRegisteredMemoryHelper(res, res->mPtrDevice, mDeviceMemoryPool, mDeviceMemoryBase, mDeviceMemorySize, &GPUMemoryResource::SetDevicePointers, mDeviceMemoryPoolEnd);
489 
490  if (!(res->mType & GPUMemoryResource::MEMORY_HOST) || (res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
491  res->mSize = size;
492  } else if (size != res->mSize) {
493  GPUError("Inconsistent device memory allocation (%s)", res->mName);
494  throw std::bad_alloc();
495  }
496  }
497  UpdateMaxMemoryUsed();
498  }
499  return res->mReuse >= 0 ? 0 : res->mSize;
500 }
