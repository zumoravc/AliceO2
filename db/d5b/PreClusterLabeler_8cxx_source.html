23 {

26  auto dataROFIt = rofRecordsData.begin();
27  for (auto pcROFIt = rofRecordsPC.begin(); pcROFIt != rofRecordsPC.end(); ++pcROFIt) {
28  for (size_t ipc = pcROFIt->firstEntry; ipc < pcROFIt->firstEntry + pcROFIt->nEntries; ++ipc) {
29  for (size_t idata = dataROFIt->firstEntry; idata < dataROFIt->firstEntry + dataROFIt->nEntries; ++idata) {
30  auto labels = inMCContainer.getLabels(idata);
31  for (auto& label : labels) {
32  if (label.getCathode() != preClusters[ipc].cathode) {
33  continue;
34  }
35  if (label.getDEId() != preClusters[ipc].deId) {
36  continue;
37  }
38  int columnId = label.getColumnId();
39  if (columnId < preClusters[ipc].firstColumn || columnId > preClusters[ipc].lastColumn) {
40  continue;
41  }
42  int firstStrip = MCLabel::getStrip(preClusters[ipc].firstStrip, preClusters[ipc].firstLine);
43  int lastStrip = MCLabel::getStrip(preClusters[ipc].lastStrip, preClusters[ipc].lastLine);
44  if ((columnId == preClusters[ipc].firstColumn && label.getLastStrip() < firstStrip) || (columnId == preClusters[ipc].lastColumn && label.getFirstStrip() > lastStrip)) {
