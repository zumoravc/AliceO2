105 int GPUTRDTrackerComponent::ReadConfigurationString(const char* arguments)
106 {
107  // Set configuration parameters for the TRD tracker component from the string
108 
109  int iResult = 0;
110  if (!arguments) {
111  return iResult;
112  }
113 
114  TString allArgs = arguments;
115  TString argument;
116 
117  TObjArray* pTokens = allArgs.Tokenize(" ");
118 
119  int nArgs = pTokens ? pTokens->GetEntries() : 0;
120 
121  for (int i = 0; i < nArgs; i++) {
122  argument = ((TObjString*)pTokens->At(i))->GetString();
123  if (argument.IsNull()) {
124  continue;
125  }
126 
127  if (argument.CompareTo("-debugOutput") == 0) {
128  fDebugTrackOutput = true;
129  fVerboseDebugOutput = true;
130  HLTInfo("Tracks are dumped in the GPUTRDTrack format");
131  continue;
132  }
133 
134  if (argument.CompareTo("-requireITStrack") == 0) {
135  fRequireITStrack = true;
136  HLTInfo("TRD tracker requires seeds (TPC tracks) to have an ITS match");
137  continue;
138  }
139 
140  HLTError("Unknown option \"%s\"", argument.Data());
141  iResult = -EINVAL;
142  }
143  delete pTokens;
144 
145  return iResult;
146 }

149 int GPUTRDTrackerComponent::DoInit(int argc, const char** argv)
150 {
151  // see header file for class documentation
152 
153  int iResult = 0;
154  if (fTracker) {
155  return -EINPROGRESS;
156  }
157 
158  fBenchmark.Reset();
159  fBenchmark.SetTimer(0, "total");
160  fBenchmark.SetTimer(1, "reco");
161 
162  if (AliGeomManager::GetGeometry() == nullptr) {
163  AliGeomManager::LoadGeometry();
164  }
165 
166  fTrackList = new TList();
167  if (!fTrackList) {
168  return -ENOMEM;
169  }
170  fTrackList->SetOwner(kFALSE);
171 
172  TString arguments = "";
173  for (int i = 0; i < argc; i++) {
174  if (!arguments.IsNull()) {
175  arguments += " ";
176  }
177  arguments += argv[i];
178  }
179 
180  iResult = ReadConfigurationString(arguments.Data());
181 
182  fGeo = new GPUTRDGeometry();
183  if (!fGeo) {
184  return -ENOMEM;
185  }
186  if (!GPUTRDGeometry::CheckGeometryAvailable()) {
187  HLTError("TRD geometry not available");
188  return -EINVAL;
189  }
190  fTracker = new GPUTRDTracker();
191  if (!fTracker) {
192  return -ENOMEM;
193  }
194  if (fVerboseDebugOutput) {
195  fTracker->EnableDebugOutput();
196  }
197  fTracker->Init(fGeo);
198 
199  return iResult;
200 }

214 int GPUTRDTrackerComponent::DoEvent(const AliHLTComponentEventData& evtData, const AliHLTComponentBlockData* blocks, AliHLTComponentTriggerData& /*trigData*/, AliHLTUInt8_t* outputPtr, AliHLTUInt32_t& size, std::vector<AliHLTComponentBlockData>& outputBlocks)
215 {
216  // process event
217 
218  if (!IsDataEvent()) {
219  return 0;
220  }
221 
222  if (evtData.fBlockCnt <= 0) {
223  HLTWarning("no blocks in event");
224  return 0;
225  }
226 
227  fBenchmark.StartNewEvent();
228  fBenchmark.Start(0);
229 
230  AliHLTUInt32_t maxBufferSize = size;
231  size = 0; // output size
232 
233  int iResult = 0;
234 
235  if (fTrackList->GetEntries() != 0) {
236  fTrackList->Clear(); // tracks are owned by GPUTRDTracker
237  }
238 
239  int nBlocks = evtData.fBlockCnt;
240 
241  AliHLTTracksData* tpcData = nullptr;
242  AliHLTTracksData* itsData = nullptr;
243  AliHLTTrackMCData* tpcDataMC = nullptr;
244 
245  std::vector<GPUTRDTrack> tracksTPC;
246  std::vector<int> tracksTPCLab;
247  std::vector<int> tracksTPCId;
248 
249  bool hasMCtracklets = false;
250 
251  int nTrackletsTotal = 0;
252  int nTrackletsTotalMC = 0;
253  GPUTRDTrackletWord* tracklets = nullptr;
254  GPUTRDTrackletLabels* trackletsMC = nullptr;
255 
256  for (int iBlock = 0; iBlock < nBlocks; iBlock++) {
257  if (blocks[iBlock].fDataType == (kAliHLTDataTypeTrack | kAliHLTDataOriginITS) && fRequireITStrack) {
258  itsData = (AliHLTTracksData*)blocks[iBlock].fPtr;
259  fBenchmark.AddInput(blocks[iBlock].fSize);
260  } else if (blocks[iBlock].fDataType == (AliHLTTPCDefinitions::TracksOuterDataType() | kAliHLTDataOriginTPC)) {
261  tpcData = (AliHLTTracksData*)blocks[iBlock].fPtr;
262  fBenchmark.AddInput(blocks[iBlock].fSize);
263  } else if (blocks[iBlock].fDataType == (kAliHLTDataTypeTrackMC | kAliHLTDataOriginTPC)) {
264  tpcDataMC = (AliHLTTrackMCData*)blocks[iBlock].fPtr;
265  fBenchmark.AddInput(blocks[iBlock].fSize);
266  } else if (blocks[iBlock].fDataType == (AliHLTTRDDefinitions::fgkTRDTrackletDataType)) {
267  tracklets = reinterpret_cast<GPUTRDTrackletWord*>(blocks[iBlock].fPtr);
268  nTrackletsTotal = blocks[iBlock].fSize / sizeof(GPUTRDTrackletWord);
269  fBenchmark.AddInput(blocks[iBlock].fSize);
270  } else if (blocks[iBlock].fDataType == (AliHLTTRDDefinitions::fgkTRDMCTrackletDataType)) {
271  hasMCtracklets = true;
272  trackletsMC = reinterpret_cast<GPUTRDTrackletLabels*>(blocks[iBlock].fPtr);
273  nTrackletsTotalMC = blocks[iBlock].fSize / sizeof(GPUTRDTrackletLabels);
274  fBenchmark.AddInput(blocks[iBlock].fSize);
275  }
276  }
277 
278  if (tpcData == nullptr) {
279  HLTInfo("did not receive any TPC tracks. Skipping event");
280  return 0;
281  }
282 
283  if (nTrackletsTotal == 0) {
284  HLTInfo("did not receive any TRD tracklets. Skipping event");
285  return 0;
286  }
287 
288  if (hasMCtracklets && nTrackletsTotal != nTrackletsTotalMC) {
289  HLTError("the numbers of input tracklets does not match the number of input MC labels for them");
290  return -EINVAL;
291  }
292 
293  int nTPCtracks = tpcData->fCount;
294  std::vector<bool> itsAvail(nTPCtracks, false);
295  if (itsData) {
296  // look for ITS tracks with >= 2 hits
297  int nITStracks = itsData->fCount;
298  AliHLTExternalTrackParam* currITStrack = itsData->fTracklets;
299  for (int iTrkITS = 0; iTrkITS < nITStracks; iTrkITS++) {
300  if (currITStrack->fNPoints >= 2) {
301  itsAvail.at(currITStrack->fTrackID) = true;
302  }
303  unsigned int dSize = sizeof(AliHLTExternalTrackParam) + currITStrack->fNPoints * sizeof(unsigned int);
304  currITStrack = (AliHLTExternalTrackParam*)(((Byte_t*)currITStrack) + dSize);
305  }
306  }
307  std::map<int, int> mcLabels;
308  if (tpcDataMC) {
309  // look for TPC track MC labels
310  int nMCtracks = tpcDataMC->fCount;
311  for (int iMC = 0; iMC < nMCtracks; iMC++) {
312  AliHLTTrackMCLabel& lab = tpcDataMC->fLabels[iMC];
313  mcLabels[lab.fTrackID] = lab.fMCLabel;
314  }
315  }
316  AliHLTExternalTrackParam* currOutTrackTPC = tpcData->fTracklets;
317  for (int iTrk = 0; iTrk < nTPCtracks; iTrk++) {
318  // store TPC tracks (if required only the ones with >=2 ITS hits)
319  if (itsData != nullptr && !itsAvail.at(currOutTrackTPC->fTrackID)) {
320  continue;
321  }
322  GPUTRDTrack t(*currOutTrackTPC);
323  int mcLabel = -1;
324  if (tpcDataMC) {
325  if (mcLabels.find(currOutTrackTPC->fTrackID) != mcLabels.end()) {
326  mcLabel = mcLabels[currOutTrackTPC->fTrackID];
327  }
328  }
329  tracksTPC.push_back(t);
330  tracksTPCId.push_back(currOutTrackTPC->fTrackID);
331  tracksTPCLab.push_back(mcLabel);
332  unsigned int dSize = sizeof(AliHLTExternalTrackParam) + currOutTrackTPC->fNPoints * sizeof(unsigned int);
333  currOutTrackTPC = (AliHLTExternalTrackParam*)+(((Byte_t*)currOutTrackTPC) + dSize);
334  }
335 
336  if (fVerboseDebugOutput) {
337  HLTInfo("TRDTrackerComponent received %i tracklets\n", nTrackletsTotal);
338  }
339 
340  fTracker->Reset();
341 
342  // loop over all tracklets
343  for (int iTracklet = 0; iTracklet < nTrackletsTotal; ++iTracklet) {
344  if (!hasMCtracklets) {
345  if (fTracker->LoadTracklet(tracklets[iTracklet])) {
346  return -EINVAL;
347  }
348  } else {
349  if (fTracker->LoadTracklet(tracklets[iTracklet], trackletsMC[iTracklet].mLabel)) {
350  return -EINVAL;
351  }
352  }
353  }
354  // loop over all tracks
355  for (unsigned int iTrack = 0; iTrack < tracksTPC.size(); ++iTrack) {
356  fTracker->LoadTrack(tracksTPC[iTrack], tracksTPCLab[iTrack]);
357  }
358 
359  fBenchmark.Start(1);
360  fTracker->DoTracking(NULL);
361  fBenchmark.Stop(1);
362 
363  GPUTRDTrack* trackArray = fTracker->Tracks();
364  int nTracks = fTracker->NTracks();
365  GPUTRDTracker::GPUTRDSpacePointInternal* spacePoints = fTracker->SpacePoints();
366 
367  // TODO delete fTrackList since it only works for TObjects (or use compiler flag after tests with GPU track type)
368  // for (int iTrack=0; iTrack<nTracks; ++iTrack) {
369  // fTrackList->AddLast(&trackArray[iTrack]);
370  //}
371 
372  // push back GPUTRDTracks for debugging purposes
373  if (fDebugTrackOutput) {
374  PushBack(fTrackList, (kAliHLTDataTypeTObject | kAliHLTDataOriginTRD), 0x3fffff);
375  }
376  // push back AliHLTExternalTrackParam (default)
377  else {
378 
379  AliHLTUInt32_t blockSize = GPUTRDTrackData::GetSize(nTracks);
380  if (size + blockSize > maxBufferSize) {
381  HLTWarning("Output buffer exceeded for tracks");
382  return -ENOSPC;
383  }
384 
385  GPUTRDTrackData* outTracks = (GPUTRDTrackData*)(outputPtr);
386  outTracks->fCount = 0;
387 
388  for (int iTrk = 0; iTrk < nTracks; ++iTrk) {
389  GPUTRDTrack& t = trackArray[iTrk];
390  if (t.GetNtracklets() == 0) {
391  continue;
392  }
393  GPUTRDTrackDataRecord& currOutTrack = outTracks->fTracks[outTracks->fCount];
394  t.ConvertTo(currOutTrack);
395  outTracks->fCount++;
396  }
397 
398  AliHLTComponentBlockData resultData;
399  FillBlockData(resultData);
400  resultData.fOffset = size;
401  resultData.fSize = blockSize;
402  resultData.fDataType = AliHLTTRDDefinitions::fgkTRDTrackDataType;
403  outputBlocks.push_back(resultData);
404  fBenchmark.AddOutput(resultData.fSize);
405 
406  size += blockSize;
407  outputPtr += resultData.fSize;
408 
409  blockSize = 0;
410 
411  // space points calculated from tracklets
412 
413  blockSize = sizeof(GPUTRDTrackPointData) + sizeof(GPUTRDTrackPoint) * nTrackletsTotal;
414 
415  if (size + blockSize > maxBufferSize) {
416  HLTWarning("Output buffer exceeded for space points");
417  return -ENOSPC;
418  }
419 
420  GPUTRDTrackPointData* outTrackPoints = (GPUTRDTrackPointData*)(outputPtr);
421  outTrackPoints->fCount = nTrackletsTotal;
422 
423  { // fill array with 0 for a case..
424  GPUTRDTrackPoint empty;
425  empty.fX[0] = 0;
426  empty.fX[1] = 0;
427  empty.fX[2] = 0;
428  empty.fVolumeId = 0;
429  for (int i = 0; i < nTrackletsTotal; ++i) {
430  outTrackPoints->fPoints[i] = empty;
431  }
432  }
433 
434  for (int i = 0; i < nTrackletsTotal; ++i) {
435  const GPUTRDTracker::GPUTRDSpacePointInternal& sp = spacePoints[i];
436  int id = sp.mId;
437  if (id < 0 || id >= nTrackletsTotal) {
438  HLTError("Internal error: wrong space point index %d", id);
439  }
440  GPUTRDTrackPoint* currOutPoint = &outTrackPoints->fPoints[id];
441  currOutPoint->fX[0] = sp.mR; // x in sector coordinates
442  currOutPoint->fX[1] = sp.mX[0]; // y in sector coordinates
443  currOutPoint->fX[2] = sp.mX[1]; // z in sector coordinates
444  currOutPoint->fVolumeId = sp.mVolumeId;
445  }
446  AliHLTComponentBlockData resultDataSP;
447  FillBlockData(resultDataSP);
448  resultDataSP.fOffset = size;
449  resultDataSP.fSize = blockSize;
450  resultDataSP.fDataType = AliHLTTRDDefinitions::fgkTRDTrackPointDataType | kAliHLTDataOriginTRD;
451  outputBlocks.push_back(resultDataSP);
452  fBenchmark.AddOutput(resultData.fSize);
453  size += blockSize;
454  outputPtr += resultDataSP.fSize;
455 
456  HLTInfo("TRD tracker: output %d tracks and %d track points", outTracks->fCount, outTrackPoints->fCount);
457  }
458 
459  fBenchmark.Stop(0);
460  HLTInfo(fBenchmark.GetStatistics());
461 
462  return iResult;
463 }
