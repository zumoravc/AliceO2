227 bool GeometryTGeo::getChipId(int index, int& lay, int& sta, int& hsta, int& mod, int& chip) const
228 {
229  lay = getLayer(index);
230  index -= getFirstChipIndex(lay);
231  sta = index / mNumberOfChipsPerStave[lay];
232  index %= mNumberOfChipsPerStave[lay];
233  hsta = mNumberOfHalfStaves[lay] > 0 ? index / mNumberOfChipsPerHalfStave[lay] : -1;
234  index %= mNumberOfChipsPerHalfStave[lay];
235  mod = mNumberOfModules[lay] > 0 ? index / mNumberOfChipsPerModule[lay] : -1;
236  chip = index % mNumberOfChipsPerModule[lay];
237 
238  return kTRUE;
239 }

273 //__________________________________________________________________________
274 TGeoHMatrix* GeometryTGeo::extractMatrixSensor(int index) const
275 {
276  // extract matrix transforming from the PHYSICAL sensor frame to global one
277  // Note, the if the effective sensitive layer thickness is smaller than the
278  // total physical sensor tickness, this matrix is biased and connot be used
279  // directly for transformation from sensor frame to global one.
280  //
281  // Therefore we need to add a shift
282 
283  int lay, stav, sstav, mod, chipInMod;
284  getChipId(index, lay, stav, sstav, mod, chipInMod);
285 
286  int wrID = mLayerToWrapper[lay];
287 
288  TString path = Form("/cave_1/%s_2/", GeometryTGeo::getITSVolPattern());
289 
290  if (wrID >= 0) {
291  path += Form("%s%d_1/", getITSWrapVolPattern(), wrID);
292  }
293 
294  path +=
295  Form("%s%d_1/%s%d_%d/", GeometryTGeo::getITSLayerPattern(), lay, GeometryTGeo::getITSStavePattern(), lay, stav);
296 
297  if (mNumberOfHalfStaves[lay] > 0) {
298  path += Form("%s%d_%d/", GeometryTGeo::getITSHalfStavePattern(), lay, sstav);
299  }
300  if (mNumberOfModules[lay] > 0) {
301  path += Form("%s%d_%d/", GeometryTGeo::getITSModulePattern(), lay, mod);
302  }
303  path +=
304  Form("%s%d_%d/%s%d_1", GeometryTGeo::getITSChipPattern(), lay, chipInMod, GeometryTGeo::getITSSensorPattern(), lay);
305 
306  static TGeoHMatrix matTmp;
307  gGeoManager->PushPath();
308 
309  if (!gGeoManager->cd(path.Data())) {
310  gGeoManager->PopPath();
311  LOG(ERROR) << "Error in cd-ing to " << path.Data();
312  return nullptr;
313  } // end if !gGeoManager
314 
315  matTmp = *gGeoManager->GetCurrentMatrix(); // matrix may change after cd
316  // RSS
317  // printf("%d/%d/%d %s\n",lay,stav,detInSta,path.Data());
318  // mat->Print();
319  // Restore the modeler state.
320  gGeoManager->PopPath();
321 
322  // account for the difference between physical sensitive layer (where charge collection is simulated) and effective sensor ticknesses
323  static TGeoTranslation tra(0., 0.5 * (Segmentation::SensorLayerThickness - Segmentation::SensorLayerThicknessEff), 0.);
324 
325  matTmp *= tra;
326 
327  return &matTmp;
328 }

330 //__________________________________________________________________________
331 void GeometryTGeo::Build(int loadTrans)
332 {
333  if (isBuilt()) {
334  LOG(WARNING) << "Already built";
335  return; // already initialized
336  }
337 
338  if (!gGeoManager) {
339  // RSTODO: in future there will be a method to load matrices from the CDB
340  LOG(FATAL) << "Geometry is not loaded";
341  }
342 
343  mNumberOfLayers = extractNumberOfLayers();
344  if (!mNumberOfLayers) {
345  return;
346  }
347 
348  mNumberOfStaves.resize(mNumberOfLayers);
349  mNumberOfHalfStaves.resize(mNumberOfLayers);
350  mNumberOfModules.resize(mNumberOfLayers);
351  mNumberOfChipsPerModule.resize(mNumberOfLayers);
352  mNumberOfChipRowsPerModule.resize(mNumberOfLayers);
353  mNumberOfChipsPerHalfStave.resize(mNumberOfLayers);
354  mNumberOfChipsPerStave.resize(mNumberOfLayers);
355  mNumberOfChipsPerLayer.resize(mNumberOfLayers);
356  mLastChipIndex.resize(mNumberOfLayers);
357  int numberOfChips = 0;
358 
359  for (int i = 0; i < mNumberOfLayers; i++) {
360  mNumberOfStaves[i] = extractNumberOfStaves(i);
361  mNumberOfHalfStaves[i] = extractNumberOfHalfStaves(i);
362  mNumberOfModules[i] = extractNumberOfModules(i);
363  mNumberOfChipsPerModule[i] = extractNumberOfChipsPerModule(i, mNumberOfChipRowsPerModule[i]);
364  mNumberOfChipsPerHalfStave[i] = mNumberOfChipsPerModule[i] * Max(1, mNumberOfModules[i]);
365  mNumberOfChipsPerStave[i] = mNumberOfChipsPerHalfStave[i] * Max(1, mNumberOfHalfStaves[i]);
366  mNumberOfChipsPerLayer[i] = mNumberOfChipsPerStave[i] * mNumberOfStaves[i];
367  numberOfChips += mNumberOfChipsPerLayer[i];
368  mLastChipIndex[i] = numberOfChips - 1;
369  }
370  setSize(numberOfChips);
371  fillTrackingFramesCache();
372  //
373  fillMatrixCache(loadTrans);
374 }

376 //__________________________________________________________________________
377 void GeometryTGeo::fillMatrixCache(int mask)
378 {
379  // populate matrix cache for requested transformations
380  //
381  if (mSize < 1) {
382  LOG(WARNING) << "The method Build was not called yet";
383  Build(mask);
384  return;
385  }
386 
387  // build matrices
388  if ((mask & o2::utils::bit2Mask(o2::TransformType::L2G)) && !getCacheL2G().isFilled()) {
389  // Matrices for Local (Sensor!!! rather than the full chip) to Global frame transformation
390  LOG(INFO) << "Loading ITS L2G matrices from TGeo";
391  auto& cacheL2G = getCacheL2G();
392  cacheL2G.setSize(mSize);
393 
394  for (int i = 0; i < mSize; i++) {
395  TGeoHMatrix* hm = extractMatrixSensor(i);
396  cacheL2G.setMatrix(Mat3D(*hm), i);
397  }
398  }
399 
400  if ((mask & o2::utils::bit2Mask(o2::TransformType::T2L)) && !getCacheT2L().isFilled()) {
401  // matrices for Tracking to Local (Sensor!!! rather than the full chip) frame transformation
402  LOG(INFO) << "Loading ITS T2L matrices from TGeo";
403  auto& cacheT2L = getCacheT2L();
404  cacheT2L.setSize(mSize);
405  for (int i = 0; i < mSize; i++) {
406  TGeoHMatrix& hm = createT2LMatrix(i);
407  cacheT2L.setMatrix(Mat3D(hm), i);
408  }
409  }
410 
411  if ((mask & o2::utils::bit2Mask(o2::TransformType::T2G)) && !getCacheT2G().isFilled()) {
412  LOG(WARNING) << "It is faster to use 2D rotation for T2G instead of full Transform3D matrices";
413  // matrices for Tracking to Global frame transformation
414  LOG(INFO) << "Loading ITS T2G matrices from TGeo";
415  auto& cacheT2G = getCacheT2G();
416  cacheT2G.setSize(mSize);
417 
418  for (int i = 0; i < mSize; i++) {
419  TGeoHMatrix& mat = createT2LMatrix(i);
420  mat.MultiplyLeft(extractMatrixSensor(i));
421  cacheT2G.setMatrix(Mat3D(mat), i);
422  }
423  }
424 
425  if ((mask & o2::utils::bit2Mask(o2::TransformType::T2GRot)) && !getCacheT2GRot().isFilled()) {
426  // 2D rotation matrices for Tracking frame to Global rotations
427  LOG(INFO) << "Loading ITS T2G rotation 2D matrices";
428  auto& cacheT2Gr = getCacheT2GRot();
429  cacheT2Gr.setSize(mSize);
430  for (int i = 0; i < mSize; i++) {
431  cacheT2Gr.setMatrix(Rot2D(getSensorRefAlpha(i)), i);
432  }
433  }
434 }

450 //__________________________________________________________________________
451 int GeometryTGeo::extractNumberOfLayers()
452 {
453  int numberOfLayers = 0;
454 
455  TGeoVolume* itsV = gGeoManager->GetVolume(getITSVolPattern());
456  if (!itsV) {
457  LOG(FATAL) << "ITS volume " << getITSVolPattern() << " is not in the geometry";
458  }
459 
460  // Loop on all ITSV nodes, count Layer volumes by checking names
461  // Build on the fly layer - wrapper correspondence
462  TObjArray* nodes = itsV->GetNodes();
463  int nNodes = nodes->GetEntriesFast();
464 
465  for (int j = 0; j < nNodes; j++) {
466  int lrID = -1;
467  TGeoNode* nd = (TGeoNode*)nodes->At(j);
468  const char* name = nd->GetName();
469 
470  if (strstr(name, getITSLayerPattern())) {
471  numberOfLayers++;
472  if ((lrID = extractVolumeCopy(name, GeometryTGeo::getITSLayerPattern())) < 0) {
473  LOG(FATAL) << "Failed to extract layer ID from the " << name;
474  exit(1);
475  }
476 
477  mLayerToWrapper[lrID] = -1; // not wrapped
478  } else if (strstr(name, getITSWrapVolPattern())) { // this is a wrapper volume, may cointain layers
479  int wrID = -1;
480  if ((wrID = extractVolumeCopy(name, GeometryTGeo::getITSWrapVolPattern())) < 0) {
481  LOG(FATAL) << "Failed to extract wrapper ID from the " << name;
482  exit(1);
483  }
484 
485  TObjArray* nodesW = nd->GetNodes();
486  int nNodesW = nodesW->GetEntriesFast();
487 
488  for (int jw = 0; jw < nNodesW; jw++) {
489  TGeoNode* ndW = (TGeoNode*)nodesW->At(jw);
490  if (strstr(ndW->GetName(), getITSLayerPattern())) {
491  if ((lrID = extractVolumeCopy(ndW->GetName(), GeometryTGeo::getITSLayerPattern())) < 0) {
492  LOG(FATAL) << "Failed to extract layer ID from the " << name;
493  exit(1);
494  }
495  numberOfLayers++;
496  mLayerToWrapper[lrID] = wrID;
497  }
498  }
499  }
500  }
501  return numberOfLayers;
502 }

587 //__________________________________________________________________________
588 int GeometryTGeo::extractNumberOfChipsPerModule(int lay, int& nrow) const
589 {
590  int numberOfChips = 0;
591  char stavnam[30];
592  TGeoVolume* volLd = nullptr;
593 
594  if (!sModuleName.empty()) {
595  snprintf(stavnam, 30, "%s%d", getITSModulePattern(), lay);
596  volLd = gGeoManager->GetVolume(stavnam);
597  }
598  if (!volLd) { // no modules on this layer, check substaves
599  if (!sHalfStaveName.empty()) {
600  snprintf(stavnam, 30, "%s%d", getITSHalfStavePattern(), lay);
601  volLd = gGeoManager->GetVolume(stavnam);
602  }
603  }
604  if (!volLd) { // no substaves on this layer, check staves
605  snprintf(stavnam, 30, "%s%d", getITSStavePattern(), lay);
606  volLd = gGeoManager->GetVolume(stavnam);
607  }
608  if (!volLd) {
609  LOG(FATAL) << "can't find volume containing chips on layer " << lay;
610  }
611 
612  // Loop on all stave nodes, count Chip volumes by checking names
613  int nNodes = volLd->GetNodes()->GetEntries();
614 
615  double xmin = 1e9, xmax = -1e9, zmin = 1e9, zmax = -1e9;
616  double lab[3], loc[3] = {0, 0, 0};
617  double dx = -1, dz = -1;
618 
619  for (int j = 0; j < nNodes; j++) {
620  // AliInfo(Form("L%d %d of %d %s %s ->
621  // %d",lay,j,nNodes,volLd->GetNodes()->At(j)->GetName(),GetITSChipPattern(),numberOfChips));
622  TGeoNodeMatrix* node = (TGeoNodeMatrix*)volLd->GetNodes()->At(j);
623  if (!strstr(node->GetName(), getITSChipPattern())) {
624  continue;
625  }
626  node->LocalToMaster(loc, lab);
627  if (lab[0] > xmax) {
628  xmax = lab[0];
629  }
630  if (lab[0] < xmin) {
631  xmin = lab[0];
632  }
633  if (lab[2] > zmax) {
634  zmax = lab[2];
635  }
636  if (lab[2] < zmin) {
637  zmin = lab[2];
638  }
639 
640  numberOfChips++;
641 
642  if (dx < 0) {
643  TGeoShape* chShape = node->GetVolume()->GetShape();
644  TGeoBBox* bbox = dynamic_cast<TGeoBBox*>(chShape);
645  if (!bbox) {
646  LOG(FATAL) << "Chip " << node->GetName() << " volume is of unprocessed shape " << chShape->IsA()->GetName();
647  } else {
648  dx = 2 * bbox->GetDX();
649  dz = 2 * bbox->GetDZ();
650  }
651  }
652  }
653 
654  double spanX = xmax - xmin;
655  double spanZ = zmax - zmin;
656  nrow = TMath::Nint(spanX / dx + 1);
657  int ncol = TMath::Nint(spanZ / dz + 1);
658  if (nrow * ncol != numberOfChips) {
659  LOG(ERROR) << "Inconsistency between Nchips=" << numberOfChips << " and Nrow*Ncol=" << nrow << "*" << ncol << "->"
660  << nrow * ncol << "\n"
661  << "Extracted chip dimensions (x,z): " << dx << " " << dz << " Module Span: " << spanX << " " << spanZ;
662  }
663  return numberOfChips;
664 }

698 void GeometryTGeo::extractSensorXAlpha(int isn, float& x, float& alp)
699 {
700  // calculate r and phi of the impact of the normal on the sensor
701  // (i.e. phi of the tracking frame alpha and X of the sensor in this frame)
702  double locA[3] = {-100., 0., 0.}, locB[3] = {100., 0., 0.}, gloA[3], gloB[3];
703  const TGeoHMatrix* matL2G = extractMatrixSensor(isn);
704 
705  matL2G->LocalToMaster(locA, gloA);
706  matL2G->LocalToMaster(locB, gloB);
707  double dx = gloB[0] - gloA[0], dy = gloB[1] - gloA[1];
708  double t = (gloB[0] * dx + gloB[1] * dy) / (dx * dx + dy * dy);
709  double xp = gloB[0] - dx * t, yp = gloB[1] - dy * t;
710  x = Sqrt(xp * xp + yp * yp);
711  alp = ATan2(yp, xp);
712  BringTo02Pi(alp);
713 }
