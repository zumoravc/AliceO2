44 void TrapConfigHandler::init()
45 {
46  if (!mTrapConfig) {
47  LOG(error) << "No TRAPconfig given";
48  return;
49  }
50 
51  // setup of register allocation
52  // I/O configuration which we don't care about
53  mTrapConfig->setTrapRegAlloc(TrapConfig::kSEBDOU, TrapConfig::kAllocNone);
54  // position look-up table by layer
55  for (int iBin = 0; iBin < 128; iBin++)
56  mTrapConfig->setTrapRegAlloc((TrapConfig::TrapReg_t)(TrapConfig::kTPL00 + iBin), TrapConfig::kAllocByLayer);
57  // ... individual
58  mTrapConfig->setTrapRegAlloc(TrapConfig::kC14CPUA, TrapConfig::kAllocByMCM);
59  mTrapConfig->setTrapRegAlloc(TrapConfig::kC15CPUA, TrapConfig::kAllocByMCM);
60 
61  // setup of DMEM allocation
62  for (int iAddr = TrapConfig::mgkDmemStartAddress;
63  iAddr < (TrapConfig::mgkDmemWords + TrapConfig::mgkDmemStartAddress); iAddr++) {
64 
65  if (iAddr == TrapSimulator::mgkDmemAddrDeflCorr)
66  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByMCMinSM);
67 
68  else if (iAddr == TrapSimulator::mgkDmemAddrNdrift)
69  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByDetector);
70 
71  else if ((iAddr >= TrapSimulator::mgkDmemAddrDeflCutStart) && (iAddr <= TrapSimulator::mgkDmemAddrDeflCutEnd))
72  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByMCMinSM);
73 
74  else if ((iAddr >= TrapSimulator::mgkDmemAddrTrackletStart) && (iAddr <= TrapSimulator::mgkDmemAddrTrackletEnd))
75  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByMCM);
76 
77  else if ((iAddr >= TrapSimulator::mgkDmemAddrLUTStart) && (iAddr <= TrapSimulator::mgkDmemAddrLUTEnd))
78  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocGlobal);
79 
80  else if (iAddr == TrapSimulator::mgkDmemAddrLUTcor0)
81  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByMCMinSM);
82 
83  else if (iAddr == TrapSimulator::mgkDmemAddrLUTcor1)
84  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocByMCMinSM);
85 
86  else if (iAddr == TrapSimulator::mgkDmemAddrLUTnbins)
87  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocGlobal);
88 
89  else if (iAddr == TrapSimulator::mgkDmemAddrLUTLength)
90  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocGlobal);
91 
92  else
93  mTrapConfig->setDmemAlloc(iAddr, TrapConfig::kAllocGlobal);
94  }
95  mFeeParam = FeeParam::instance();
96 }

113 int TrapConfigHandler::loadConfig()
114 {
115  // load a default configuration which is suitable for simulation
116  // for a detailed description of the registers see the TRAP manual
117  // if you want to resimulate tracklets on real data use the appropriate config instead
118 
119  if (!mTrapConfig) {
120  LOG(error) << "No TRAPconfig given";
121  return -1;
122  }
123 
124  // prepare mFeeParam
125  // ndrift (+ 5 binary digits)
126  mFeeParam->setNtimebins(20 << 5);
127  // deflection + tilt correction
128  mFeeParam->setRawOmegaTau(0.16133);
129  // deflection range table
130  mFeeParam->setRawPtMin(0.1);
131  // magnetic field
132  mFeeParam->setRawMagField(0.0);
133  // scaling factors for q0, q1
134  mFeeParam->setRawScaleQ0(0);
135  mFeeParam->setRawScaleQ1(0);
136  // disable length correction and tilting correction
137  mFeeParam->setRawLengthCorrectionEnable(false);
138  mFeeParam->setRawTiltCorrectionEnable(false);
139 
140  for (int iDet = 0; iDet < 540; iDet++) {
141  // HC header configuration bits
142  mTrapConfig->setTrapReg(TrapConfig::kC15CPUA, 0x2102, iDet); // zs, deh
143 
144  // no. of timebins
145  mTrapConfig->setTrapReg(TrapConfig::kC13CPUA, 24, iDet);
146 
147  // pedestal filter
148  mTrapConfig->setTrapReg(TrapConfig::kFPNP, 4 * 10, iDet);
149  mTrapConfig->setTrapReg(TrapConfig::kFPTC, 0, iDet);
150  mTrapConfig->setTrapReg(TrapConfig::kFPBY, 0, iDet); // bypassed!
151 
152  // gain filter
153  for (int adc = 0; adc < 20; adc++) {
154  mTrapConfig->setTrapReg(TrapConfig::TrapReg_t(TrapConfig::kFGA0 + adc), 40, iDet);
155  mTrapConfig->setTrapReg(TrapConfig::TrapReg_t(TrapConfig::kFGF0 + adc), 15, iDet);
156  }
157  mTrapConfig->setTrapReg(TrapConfig::kFGTA, 20, iDet);
158  mTrapConfig->setTrapReg(TrapConfig::kFGTB, 2060, iDet);
159  mTrapConfig->setTrapReg(TrapConfig::kFGBY, 0, iDet); // bypassed!
160 
161  // tail cancellation
162  mTrapConfig->setTrapReg(TrapConfig::kFTAL, 200, iDet);
163  mTrapConfig->setTrapReg(TrapConfig::kFTLL, 0, iDet);
164  mTrapConfig->setTrapReg(TrapConfig::kFTLS, 200, iDet);
165  mTrapConfig->setTrapReg(TrapConfig::kFTBY, 0, iDet);
166 
167  // tracklet calculation
168  mTrapConfig->setTrapReg(TrapConfig::kTPQS0, 5, iDet);
169  mTrapConfig->setTrapReg(TrapConfig::kTPQE0, 10, iDet);
170  mTrapConfig->setTrapReg(TrapConfig::kTPQS1, 11, iDet);
171  mTrapConfig->setTrapReg(TrapConfig::kTPQE1, 20, iDet);
172  mTrapConfig->setTrapReg(TrapConfig::kTPFS, 5, iDet);
173  mTrapConfig->setTrapReg(TrapConfig::kTPFE, 20, iDet);
174  mTrapConfig->setTrapReg(TrapConfig::kTPVBY, 0, iDet);
175  mTrapConfig->setTrapReg(TrapConfig::kTPVT, 10, iDet);
176  mTrapConfig->setTrapReg(TrapConfig::kTPHT, 150, iDet);
177  mTrapConfig->setTrapReg(TrapConfig::kTPFP, 40, iDet);
178  mTrapConfig->setTrapReg(TrapConfig::kTPCL, 1, iDet);
179  mTrapConfig->setTrapReg(TrapConfig::kTPCT, 10, iDet);
180 
181  // apply mFeeParams
182  configureDyCorr(iDet);
183  configureDRange(iDet); // deflection range
184  configureNTimebins(iDet); // timebins in the drift region
185  configurePIDcorr(iDet); // scaling parameters for the PID
186 
187  // event buffer
188  mTrapConfig->setTrapReg(TrapConfig::kEBSF, 1, iDet); // 0: store filtered; 1: store unfiltered
189 
190  // zs applied to data stored in event buffer (sel. by EBSF)
191  mTrapConfig->setTrapReg(TrapConfig::kEBIS, 15, iDet); // single indicator threshold (plus two digits)
192  mTrapConfig->setTrapReg(TrapConfig::kEBIT, 30, iDet); // sum indicator threshold (plus two digits)
193  mTrapConfig->setTrapReg(TrapConfig::kEBIL, 0xf0, iDet); // lookup table
194  mTrapConfig->setTrapReg(TrapConfig::kEBIN, 0, iDet); // neighbour sensitivity
195 
196  // raw data
197  mTrapConfig->setTrapReg(TrapConfig::kNES, (0x0000 << 16) | 0x1000, iDet);
198  }
199 
200  // ****** hit position LUT
201 
202  // now calculate it from PRF
203  PadResponse padresponse;
204  double padResponse[3]; // pad response left, central, right
205  double padResponseR[3]; // pad response left, central, right
206  double padResponseL[3]; // pad response left, central, right
207 
208  for (int iLayer = 0; iLayer < 6; iLayer++) {
209  TGraph gr(128);
210  for (int iBin = 0; iBin < 256 * 0.5; iBin++) {
211  padresponse.getPRF(1., iBin * 1. / 256., iLayer, padResponse);
212  padresponse.getPRF(1., iBin * 1. / 256. - 1., iLayer, padResponseR);
213  padresponse.getPRF(1., iBin * 1. / 256. + 1., iLayer, padResponseL);
214  gr.SetPoint(iBin, (0.5 * (padResponseR[1] - padResponseL[1]) / padResponse[1] * 256), iBin);
215  }
216  for (int iBin = 0; iBin < 128; iBin++) {
217  int corr = (int)(gr.Eval(iBin)) - iBin;
218  if (corr < 0)
219  corr = 0;
220  else if (corr > 31)
221  corr = 31;
222  for (int iStack = 0; iStack < 540 / 6; iStack++) {
223  mTrapConfig->setTrapReg((TrapConfig::TrapReg_t)(TrapConfig::kTPL00 + iBin), corr, 6 * iStack + iLayer);
224  }
225  }
226  }
227  // ****** hit position LUT configuration end
228 
229  return 0;
230 }

232 int TrapConfigHandler::loadConfig(std::string filename)
233 {
234  //
235  // load a TRAP configuration from a file
236  // The file format is the format created by the standalone
237  // command coder scc / show_cfdat but without the running number
238  // scc /show_cfdat adds as the first column
239  // which are two tools to inspect/export configurations from wingDB
240  //
241 
242  if (!mTrapConfig) {
243  LOG(error) << "No TRAPconfig given";
244  return -1;
245  }
246 
247  int ignoredLines = 0;
248  int ignoredCmds = 0;
249  int readLines = 0;
250 
251  LOG(debug3) << "Processing file " << filename.c_str();
252  std::ifstream infile;
253  infile.open(filename.c_str(), std::ifstream::in);
254  if (!infile.is_open()) {
255  LOG(error) << "Can not open MCM configuration file " << filename;
256  return false;
257  }
258 
259  int addr;
260 
261  // reset restrictive mask
262  mRestrictiveMask = (0x3ffff << 11) | (0x1f << 6) | 0x3f;
263  char linebuffer[512];
264  istringstream line;
265 
266  while (infile.getline(linebuffer, 512) && infile.good()) {
267  unsigned int cmd = 999;
268  int data = -1;
269  int extali = -1;
270  line.clear();
271  line.str(linebuffer);
272  data = -1;
273  line >> std::skipws >> cmd >> addr >> data >> extali; // the lines read from config file can contain additional columns.
274  // Therefore the detour via istringstream
275 
276  if (cmd != 999 && addr != -1 && extali != -1) {
277 
278  if (cmd == mgkScsnCmdWrite) {
279  for (int det = 0; det < kNdet; det++) {
280  unsigned int rocpos = (1 << (TRDGeometry::getSector(det) + 11)) | (1 << (TRDGeometry::getStack(det) + 6)) | (1 << TRDGeometry::getLayer(det));
281  LOG(debug) << "checking restriction: mask=0x" << hex << std::setw(8) << mRestrictiveMask << " rocpos=0x" << hex << std::setw(8) << rocpos;
282  if ((mRestrictiveMask & rocpos) == rocpos) {
283  LOG(debug) << "match:"
284  << " " << cmd << " " << extali << " " << addr << " " << data;
285  addValues(det, cmd, extali, addr, data);
286  }
287  }
288  }
289 
290  else if (cmd == mgkScsnLTUparam) {
291  // processLTUparam(extali, addr, data);
292  }
293 
294  else if (cmd == mgkScsnCmdRestr) {
295  mRestrictiveMask = data;
296  LOG(debug) << "updated restrictive mask to 0x" << hex << std::setw(8) << mRestrictiveMask;
297  }
298 
299  else if ((cmd == mgkScsnCmdReset) ||
300  (cmd == mgkScsnCmdRobReset)) {
301  mTrapConfig->resetRegs();
302  }
303 
304  else if (cmd == mgkScsnCmdSetHC) {
305  int fullVersion = ((data & 0x7F00) >> 1) | (data & 0x7f);
306 
307  for (int iDet = 0; iDet < kNdet; iDet++) {
308  int smls = (TRDGeometry::getSector(iDet) << 6) | (TRDGeometry::getLayer(iDet) << 3) | TRDGeometry::getStack(iDet);
309 
310  for (int iRob = 0; iRob < 8; iRob++) {
311  // HC mergers
312  mTrapConfig->setTrapReg(TrapConfig::kC14CPUA, 0xc << 16, iDet, iRob, 17);
313  mTrapConfig->setTrapReg(TrapConfig::kC15CPUA, ((1 << 29) | (fullVersion << 15) | (1 << 12) | (smls << 1) | (iRob % 2)), iDet, iRob, 17);
314 
315  // board mergers
316  mTrapConfig->setTrapReg(TrapConfig::kC14CPUA, 0, iDet, iRob, 16);
317  mTrapConfig->setTrapReg(TrapConfig::kC15CPUA, ((1 << 29) | (fullVersion << 15) | (1 << 12) | (smls << 1) | (iRob % 2)), iDet, iRob, 16);
318 
319  // and now for the others
320  for (int iMcm = 0; iMcm < 16; iMcm++) {
321  mTrapConfig->setTrapReg(TrapConfig::kC14CPUA, iMcm | (iRob << 4) | (3 << 16), iDet, iRob, iMcm);
322  mTrapConfig->setTrapReg(TrapConfig::kC15CPUA, ((1 << 29) | (fullVersion << 15) | (1 << 12) | (smls << 1) | (iRob % 2)), iDet, iRob, iMcm);
323  }
324  }
325  }
326  }
327 
328  else if ((cmd == mgkScsnCmdRead) ||
329  (cmd == mgkScsnCmdPause) ||
330  (cmd == mgkScsnCmdPtrg) ||
331  (cmd == mgkScsnCmdHwPtrg) ||
332  (cmd == mgkScsnCmdRobPower) ||
333  (cmd == mgkScsnCmdTtcRx) ||
334  (cmd == mgkScsnCmdMcmTemp) ||
335  (cmd == mgkScsnCmdOri) ||
336  (cmd == mgkScsnCmdPM)) {
337  LOG(debug2) << "ignored SCSN command: " << cmd << " " << addr << " " << data << " " << extali;
338  }
339 
340  else {
341  LOG(warn) << "unknown SCSN command: " << cmd << " " << addr << " " << data << " " << extali;
342  ignoredCmds++;
343  }
344 
345  readLines++;
346  }
347 
348  else if (!infile.eof() && !infile.good()) {
349  infile.clear();
350  infile.ignore(256, '\n');
351  ignoredLines++;
352  }
353 
354  if (!infile.eof())
355  infile.clear();
356  }
357 
358  infile.close();
359 
360  LOG(debug3) << "Ignored lines: " << ignoredLines << ", ignored cmds: " << ignoredCmds;
361 
362  if (ignoredLines > readLines)
363  LOG(error) << "More than 50 %% of the input file could not be processed. Perhaps you should check the input file %s", filename.c_str();
364 
365  return true;
366 }

374 void TrapConfigHandler::processLTUparam(int dest, int addr, unsigned int data)
375 {
376  //
377  // Process the LTU parameters and stores them in internal class variables
378  // or transfer the stored values to TrapConfig, depending on the dest parameter
379  //
380 
381  switch (dest) {
382 
383  case 0: // set the parameters in TrapConfig
384  for (int det = 0; det < kNdet; det++) {
385  configureDyCorr(det);
386  configureDRange(det); // deflection range
387  configureNTimebins(det); // timebins in the drift region
388  configurePIDcorr(det); // scaling parameters for the PID
389  }
390  break;
391 
392  case 1: // set variables
393  switch (addr) {
394 
395  case 0:
396  mFeeParam->setPtMin(data);
397  break; // pt_min in GeV/c (*1000)
398  case 1:
399  mFeeParam->setMagField(data);
400  break; // B in T (*1000)
401  case 2:
402  mFeeParam->setOmegaTau(data);
403  break; // omega*tau
404  case 3:
405  mFeeParam->setNtimebins(data);
406  break;
407  // ntimbins: drift time (for 3 cm) in timebins (5 add. bin. digits)
408  case 4:
409  mFeeParam->setScaleQ0(data);
410  break;
411  case 5:
412  mFeeParam->setScaleQ1(data);
413  break;
414  case 6:
415  mFeeParam->setLengthCorrectionEnable(data);
416  break;
417  case 7:
418  mFeeParam->setTiltCorrectionEnable(data);
419  break;
420  }
421  break;
422 
423  default:
424  LOG(error) << "dest " << dest << " not implemented";
425  }
426 }

442 void TrapConfigHandler::configureDyCorr(int det)
443 {
444  //
445  // Deflection length correction
446  // due to Lorentz angle and tilted pad correction
447  // This correction is in units of padwidth / (256*32)
448  //
449 
450  if (!mTrapConfig) {
451  LOG(error) << "No TRAPconfig given";
452  return;
453  }
454 
455  int nRobs = TRDGeometry::getStack(det) == 2 ? 6 : 8;
456 
457  for (int r = 0; r < nRobs; r++) {
458  for (int m = 0; m < 16; m++) {
459  int dest = 1 << 10 | r << 7 | m;
460  int dyCorrInt = mFeeParam->getDyCorrection(det, r, m);
461  addValues(det, mgkScsnCmdWrite, dest, TrapSimulator::mgkDmemAddrDeflCorr, dyCorrInt);
462  }
463  }
464 }

466 void TrapConfigHandler::configureDRange(int det)
467 {
468  //
469  // deflection range LUT
470  // range calculated according to B-field (in T) and pt_min (in GeV/c)
471  // if pt_min < 0.1 GeV/c the maximal allowed range for the tracklet
472  // deflection (-64..63) is used
473  //
474 
475  if (!mTrapConfig) {
476  LOG(error) << "No TRAPconfig given";
477  return;
478  }
479 
480  int nRobs = TRDGeometry::getStack(det) == 2 ? 6 : 8;
481 
482  int dyMinInt;
483  int dyMaxInt;
484 
485  for (int r = 0; r < nRobs; r++) {
486  for (int m = 0; m < 16; m++) {
487  for (int c = 0; c < 18; c++) {
488 
489  // cout << "maxdefl: " << maxDeflAngle << ", localPhi " << localPhi << endl;
490  // cout << "r " << r << ", m" << m << ", c " << c << ", min angle: " << localPhi-maxDeflAngle << ", max: " << localPhi+maxDeflAngle
491  // << ", min int: " << dyMinInt << ", max int: " << dyMaxInt << endl;
492  int dest = 1 << 10 | r << 7 | m;
493  int lutAddr = TrapSimulator::mgkDmemAddrDeflCutStart + 2 * c;
494  mFeeParam->getDyRange(det, r, m, c, dyMinInt, dyMaxInt);
495  addValues(det, mgkScsnCmdWrite, dest, lutAddr + 0, dyMinInt);
496  addValues(det, mgkScsnCmdWrite, dest, lutAddr + 1, dyMaxInt);
497  }
498  }
499  }

529 void TrapConfigHandler::configurePIDcorr(int det)
530 {
531  //
532  // Calculate the MCM individual correction factors for the PID
533  // and transfer them to TrapConfig
534  //
535 
536  if (!mTrapConfig) {
537  LOG(error) << "No TRAPconfig given";
538  return;
539  }
540 
541  static const int addrLUTcor0 = TrapSimulator::mgkDmemAddrLUTcor0;
542  static const int addrLUTcor1 = TrapSimulator::mgkDmemAddrLUTcor1;
543 
544  unsigned int cor0;
545  unsigned int cor1;
546  int readoutboard;
547  int mcm;
548  // int nRobs = TRDGeometry::getStack(det) == 2 ? 6 : 8;
549  int MaxRows;
550  FeeParam* feeparam = FeeParam::instance();
551  if (TRDGeometry::getStack(det) == 2)
552  MaxRows = FeeParam::mgkNrowC0;
553  else
554  MaxRows = FeeParam::mgkNrowC1;
555  int MaxCols = FeeParam::mgkNcol;
556  for (int row = 0; row < MaxRows; row++) { //TODO put this back to rob/mcm and not row/col as done in TrapSimulator
557  for (int col = 0; col < MaxCols; col++) {
558  readoutboard = feeparam->getROBfromPad(row, col);
559  mcm = feeparam->getMCMfromPad(row, col);
560  int dest = 1 << 10 | readoutboard << 7 | mcm;
561  //TODO impelment a method for determining if gaintables are valid, used to be if pointer was valid.
562  if (mGtbl.getMCMGain(det, row, col) != 0.0) //TODO check this logic there might be problems here.
563  mFeeParam->getCorrectionFactors(det, readoutboard, mcm, 9, cor0, cor1, mGtbl.getMCMGain(det, row, col));
564  else
565  mFeeParam->getCorrectionFactors(det, readoutboard, mcm, 9, cor0, cor1);
566  addValues(det, mgkScsnCmdWrite, dest, addrLUTcor0, cor0);
567  addValues(det, mgkScsnCmdWrite, dest, addrLUTcor1, cor1);
568  }
569  }

572 bool TrapConfigHandler::addValues(unsigned int det, unsigned int cmd, unsigned int extali, int addr, unsigned int data)
573 {
574  // transfer the informations provided by LoadConfig to the internal class variables
575 
576  if (!mTrapConfig) {
577  LOG(error) << "No TRAPconfig given";
578  return false;
579  }
580 
581  if (cmd != mgkScsnCmdWrite) {
582  LOG(error) << "Invalid command received: " << cmd;
583  return false;
584  }
585 
586  TrapConfig::TrapReg_t mcmReg = mTrapConfig->getRegByAddress(addr);
587  int rocType = TRDGeometry::getStack(det) == 2 ? 0 : 1;
588 
589  static const int mcmListSize = 40; // 40 is more or less arbitrary
590  int mcmList[mcmListSize];
591 
592  // configuration registers
593  if (mcmReg >= 0 && mcmReg < TrapConfig::kLastReg) {
594 
595  for (int linkPair = 0; linkPair < mgkMaxLinkPairs; linkPair++) {
596  if (FeeParam::extAliToAli(extali, linkPair, rocType, mcmList, mcmListSize) != 0) {
597  int i = 0;
598  while ((i < mcmListSize) && (mcmList[i] != -1)) {
599  if (mcmList[i] == 127) {
600  LOG(debug) << "broadcast write to " << mTrapConfig->getRegName((TrapConfig::TrapReg_t)mcmReg) << ": 0x" << hex << std::setw(8) << data;
601  mTrapConfig->setTrapReg((TrapConfig::TrapReg_t)mcmReg, data, det);
602  } else {
603  LOG(debug) << "individual write to " << mTrapConfig->getRegName((TrapConfig::TrapReg_t)mcmReg) << " (" << (mcmList[i] >> 7) << "," << (mcmList[i] & 0x7F) << "): 0x" << hex << std::setw(8) << data;
604  mTrapConfig->setTrapReg((TrapConfig::TrapReg_t)mcmReg, data, det, (mcmList[i] >> 7) & 0x7, (mcmList[i] & 0x7F));
605  }
606  i++;
607  }
608  }
609  }
610  return true;
611  }
612  // DMEM
613  else if ((addr >= TrapConfig::mgkDmemStartAddress) &&
614  (addr < (TrapConfig::mgkDmemStartAddress + TrapConfig::mgkDmemWords))) {
615  for (int linkPair = 0; linkPair < mgkMaxLinkPairs; linkPair++) {
616  if (FeeParam::extAliToAli(extali, linkPair, rocType, mcmList, mcmListSize) != 0) {
617  int i = 0;
618  while (i < mcmListSize && mcmList[i] != -1) {
619  if (mcmList[i] == 127)
620  mTrapConfig->setDmem(addr, data, det, 0, 127);
621  else
622  mTrapConfig->setDmem(addr, data, det, mcmList[i] >> 7, mcmList[i] & 0x7f);
623  i++;
624  }
625  }
626  }
627  return true;
628  } else if ((addr >= TrapConfig::mgkImemStartAddress) &&
629  (addr < (TrapConfig::mgkImemStartAddress + TrapConfig::mgkImemWords))) {
630  // IMEM is ignored for now
631  return true;
632  } else if ((addr >= TrapConfig::mgkDbankStartAddress) &&
633  (addr < (TrapConfig::mgkDbankStartAddress + TrapConfig::mgkImemWords))) {
634  // DBANK is ignored for now
635  return true;
636  } else {
637  LOG(error) << "Writing to unhandled address 0x" << hex << std::setw(4) << addr;
638  return false;
639  }
640 }
