115 void DataProcessingDevice::Init()
116 {
117  // For some reason passing rateLogging does not work anymore.
118  // This makes sure we never have more than one notification per minute.
119  for (auto& x : fChannels) {
120  for (auto& c : x.second) {
121  if (c.GetRateLogging() < MIN_RATE_LOGGING) {
122  c.UpdateRateLogging(MIN_RATE_LOGGING);
123  }
124  }
125  }
126  // If available use the ConfigurationInterface, otherwise go for
127  // the command line options.
128  bool hasConfiguration = false;
129  bool hasOverrides = false;
130  if (mServiceRegistry.active<ConfigurationInterface>()) {
131  auto& cfg = mServiceRegistry.get<ConfigurationInterface>();
132  hasConfiguration = true;
133  try {
134  cfg.getRecursive(mSpec.name);
135  hasOverrides = true;
136  } catch (...) {
137  // No overrides...
138  }
139  }
140  // We only use the configuration file if we have a stanza for the given
141  // dataprocessor
142  std::vector<std::unique_ptr<ParamRetriever>> retrievers;
143  if (hasConfiguration && hasOverrides) {
144  auto& cfg = mServiceRegistry.get<ConfigurationInterface>();
145  retrievers.emplace_back(std::make_unique<ConfigurationOptionsRetriever>(&cfg, mSpec.name));
146  } else {
147  retrievers.emplace_back(std::make_unique<FairOptionsRetriever>(GetConfig()));
148  }
149  auto configStore = std::move(std::make_unique<ConfigParamStore>(mSpec.options, std::move(retrievers)));
150  configStore->preload();
151  configStore->activate();
153  for (auto& entry : configStore->store()) {
154  LOG(INFO) << "[CONFIG] " << entry.first << "=" << configStore->store().get<std::string>(entry.first) << " 1 " << configStore->provenance(entry.first.c_str());
155  }
156  mConfigRegistry = std::make_unique<ConfigParamRegistry>(std::move(configStore));
157 
158  mExpirationHandlers.clear();
159 
160  auto distinct = DataRelayerHelpers::createDistinctRouteIndex(mSpec.inputs);
161  int i = 0;
162  for (auto& di : distinct) {
163  auto& route = mSpec.inputs[di];
164  if (route.configurator.has_value() == false) {
165  i++;
166  continue;
167  }
168  ExpirationHandler handler{
169  RouteIndex{i++},
170  route.matcher.lifetime,
171  route.configurator->creatorConfigurator(*mConfigRegistry),
172  route.configurator->danglingConfigurator(*mConfigRegistry),
173  route.configurator->expirationConfigurator(*mConfigRegistry)};
174  mExpirationHandlers.emplace_back(std::move(handler));
175  }
176 
177  auto& monitoring = mServiceRegistry.get<Monitoring>();
178  monitoring.enableBuffering(MONITORING_QUEUE_SIZE);
179  static const std::string dataProcessorIdMetric = "dataprocessor_id";
180  static const std::string dataProcessorIdValue = mSpec.name;
181  monitoring.addGlobalTag("dataprocessor_id", dataProcessorIdValue);
182 
183  if (mInit) {
184  InitContext initContext{*mConfigRegistry, mServiceRegistry};
185  mStatefulProcess = mInit(initContext);
186  }
187  mState.inputChannelInfos.resize(mSpec.inputChannels.size());
191  for (size_t ci = 0; ci < mSpec.inputChannels.size(); ++ci) {
192  auto& name = mSpec.inputChannels[ci].name;
193  if (name.find("from_internal-dpl-clock") == 0) {
194  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
195  } else if (name.find("from_internal-dpl-ccdb-backend") == 0) {
196  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
197  }
198  }
199 }

257  .addTag(Key::Subsystem, Value::DPL));
258  monitoring.send(Metric{(stats.lastTotalProcessedSize / (stats.lastLatency.maxLatency ? stats.lastLatency.maxLatency : 1) / 1000), "input_rate_mb_s"}
259  .addTag(Key::Subsystem, Value::DPL));
260  monitoring.send(Metric{(int)currentBackoff, "current_backoff"}.addTag(Key::Subsystem, Value::DPL));
261 
262  lastSent = currentTime;
263  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
264  };

267  auto flushMetrics = [& stats = mStats,
268  &relayer = mRelayer,
269  &lastFlushed = mLastMetricFlushedTimestamp,
270  &currentTime = mBeginIterationTimestamp,
271  &monitoring = mServiceRegistry.get<Monitoring>()]()
272  -> void {
273  if (currentTime - lastFlushed < 1000) {
274  return;
275  }
276 
277  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);

280  // sent...

282  auto state = stats.relayerState[si];
283  monitoring.send({state, "data_relayer/" + std::to_string(si)});
284  }
285  relayer.sendContextState();
286  monitoring.flushBuffer();
287  lastFlushed = currentTime;
288  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);

291  auto switchState = [& control = mServiceRegistry.get<ControlService>(),
292  &state = mState.streaming](StreamingState newState) {
293  state = newState;
294  control.notifyStreamingState(state);
295  };
296 
297  auto now = std::chrono::high_resolution_clock::now();
298  mBeginIterationTimestamp = (uint64_t)std::chrono::duration<double, std::milli>(now.time_since_epoch()).count();
299 
300  if (mPendingRegionInfos.empty() == false) {
301  std::vector<FairMQRegionInfo> toBeNotified;
302  toBeNotified.swap(mPendingRegionInfos); // avoid any MT issue.
303  for (auto const& info : toBeNotified) {
304  mServiceRegistry.get<CallbackService>()(CallbackService::Id::RegionInfoCallback, info);
305  }
306  }
307  mServiceRegistry.get<CallbackService>()(CallbackService::Id::ClockTick);
308  // Whether or not we had something to do.
309  bool active = false;
310 
311  // Notice that fake input channels (InputChannelState::Pull) cannot possibly
312  // expect to receive an EndOfStream signal. Thus we do not wait for these
313  // to be completed. In the case of data source devices, as they do not have
314  // real data input channels, they have to signal EndOfStream themselves.
315  bool allDone = std::any_of(mState.inputChannelInfos.begin(), mState.inputChannelInfos.end(), [](const auto& info) {
316  return info.state != InputChannelState::Pull;
317  });
318  // Whether or not all the channels are completed
319  for (size_t ci = 0; ci < mSpec.inputChannels.size(); ++ci) {
320  auto& channel = mSpec.inputChannels[ci];
321  auto& info = mState.inputChannelInfos[ci];
322 
323  if (info.state != InputChannelState::Completed && info.state != InputChannelState::Pull) {
324  allDone = false;
325  }
326  if (info.state != InputChannelState::Running) {
327  continue;
328  }
329  FairMQParts parts;
330  auto result = this->Receive(parts, channel.name, 0, 0);
331  if (result > 0) {
332  this->handleData(parts, info);
333  active |= this->tryDispatchComputation();
334  }
335  }
336  if (active == false) {
337  mServiceRegistry.get<CallbackService>()(CallbackService::Id::Idle);
338  }
339  active |= mRelayer.processDanglingInputs(mExpirationHandlers, mServiceRegistry);
340  this->tryDispatchComputation();
341 
342  sendRelayerMetrics();
343  flushMetrics();
344 
345  // If we got notified that all the sources are done, we call the EndOfStream
346  // callback and return false. Notice that what happens next is actually
347  // dependent on the callback, not something which is controlled by the
348  // framework itself.
349  if (allDone == true && mState.streaming == StreamingState::Streaming) {
350  switchState(StreamingState::EndOfStreaming);
351  }
352 
353  if (mState.streaming == StreamingState::EndOfStreaming) {
354  // We keep processing data until we are Idle.
355  // FIXME: not sure this is the correct way to drain the queues, but
356  // I guess we will see.
357  while (this->tryDispatchComputation()) {
358  mRelayer.processDanglingInputs(mExpirationHandlers, mServiceRegistry);
359  }
360  sendRelayerMetrics();
361  flushMetrics();
362  mContextRegistry.get<MessageContext>()->clear();
363  mContextRegistry.get<StringContext>()->clear();
364  mContextRegistry.get<ArrowContext>()->clear();
365  mContextRegistry.get<RawBufferContext>()->clear();
366  EndOfStreamContext eosContext{mServiceRegistry, mAllocator};
367  mServiceRegistry.get<CallbackService>()(CallbackService::Id::EndOfStream, eosContext);
368  DataProcessor::doSend(*this, *mContextRegistry.get<MessageContext>());
369  DataProcessor::doSend(*this, *mContextRegistry.get<StringContext>());
370  DataProcessor::doSend(*this, *mContextRegistry.get<ArrowContext>());
371  DataProcessor::doSend(*this, *mContextRegistry.get<RawBufferContext>());
372  for (auto& channel : mSpec.outputChannels) {
373  DataProcessingHelpers::sendEndOfStream(*this, channel);
374  }
375  // This is needed because the transport is deleted before the device.
376  mRelayer.clear();
377  switchState(StreamingState::Idle);
378  mCurrentBackoff = 10;
379  return true;
380  }
381  // Update the backoff factor
382  //
383  // In principle we should use 1/rate for MIN_BACKOFF_DELAY and (1/maxRate -
384  // 1/minRate)/ 2^MAX_BACKOFF for BACKOFF_DELAY_STEP. We hardcode the values
385  // for the moment to some sensible default.
386  if (active && mState.streaming != StreamingState::Idle) {
387  mCurrentBackoff = std::max(0, mCurrentBackoff - 1);
388  } else {
389  mCurrentBackoff = std::min(MAX_BACKOFF, mCurrentBackoff + 1);
390  }
391 
392  if (mCurrentBackoff != 0) {
393  auto delay = (rand() % ((1 << mCurrentBackoff) - 1)) * BACKOFF_DELAY_STEP;
394  if (delay > MIN_BACKOFF_DELAY) {
395  WaitFor(std::chrono::microseconds(delay - MIN_BACKOFF_DELAY));
396  }
397  }
398  return true;
399 }

410 bool DataProcessingDevice::handleData(FairMQParts& parts, InputChannelInfo& info)
411 {
412  assert(mSpec.inputChannels.empty() == false);
413  assert(parts.Size() > 0);
414 
415  // Initial part. Let's hide all the unnecessary and have
416  // simple lambdas for each of the steps I am planning to have.
417  assert(!mSpec.inputs.empty());
418 
419  // These duplicate references are created so that each function
420  // does not need to know about the whole class state, but I can
421  // fine grain control what is exposed at each state.
422  auto& monitoringService = mServiceRegistry.get<Monitoring>();
423  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
424  ScopedExit metricFlusher([&monitoringService] {
425  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
426  });
427 
428  enum struct InputType {
429  Invalid,
430  Data,
431  SourceInfo
432  };
433 
434  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
435  // and we do a few stats. We bind parts as a lambda captured variable, rather
436  // than an input, because we do not want the outer loop actually be exposed
437  // to the implementation details of the messaging layer.
438  auto getInputTypes = [& stats = mStats, &parts, &info]() -> std::optional<std::vector<InputType>> {
439  stats.inputParts = parts.Size();
440 
441  if (parts.Size() % 2) {
442  return std::nullopt;
443  }
444  std::vector<InputType> results(parts.Size() / 2, InputType::Invalid);
445 
446  for (size_t hi = 0; hi < parts.Size() / 2; ++hi) {
447  auto pi = hi * 2;
448  auto sih = o2::header::get<SourceInfoHeader*>(parts.At(pi)->GetData());
449  if (sih) {
450  info.state = sih->state;
451  results[hi] = InputType::SourceInfo;
452  continue;
453  }
454  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
455  if (!dh) {
456  results[hi] = InputType::Invalid;
457  LOGP(error, "Header is not a DataHeader?");
458  continue;
459  }
460  if (dh->payloadSize != parts.At(pi + 1)->GetSize()) {
461  results[hi] = InputType::Invalid;
462  LOGP(error, "DataHeader payloadSize mismatch");
463  continue;
464  }
465  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
466  if (!dph) {
467  results[hi] = InputType::Invalid;
468  LOGP(error, "Header stack does not contain DataProcessingHeader");
469  continue;
470  }
471  results[hi] = InputType::Data;
472  }
473  return results;
474  };

476  auto reportError = [& device = *this](const char* message) {
477  device.error(message);
478  };

480  auto handleValidMessages = [&parts, &relayer = mRelayer, &reportError](std::vector<InputType> const& types) {
481  // We relay execution to make sure we have a complete set of parts
482  // available.
483  for (size_t pi = 0; pi < (parts.Size() / 2); ++pi) {
484  switch (types[pi]) {
485  case InputType::Data: {
486  auto headerIndex = 2 * pi;
487  auto payloadIndex = 2 * pi + 1;
488  assert(payloadIndex < parts.Size());
489  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
490  std::move(parts.At(payloadIndex)));
491  if (relayed == DataRelayer::WillNotRelay) {
492  reportError("Unable to relay part.");
493  }
494  } break;
495  case InputType::SourceInfo: {
496 
497  } break;
498  case InputType::Invalid: {
499  reportError("Invalid part found.");
500  } break;
501  }
502  }
503  };
504 
505  // Second part. This is the actual outer loop we want to obtain, with
506  // implementation details which can be read. Notice how most of the state
507  // is actually hidden. For example we do not expose what "input" is. This
508  // will allow us to keep the same toplevel logic even if the actual meaning
509  // of input is changed (for example we might move away from multipart
510  // messages). Notice also that we need to act diffently depending on the
511  // actual CompletionOp we want to perform. In particular forwarding inputs
512  // also gets rid of them from the cache.
513  auto inputTypes = getInputTypes();
514  if (bool(inputTypes) == false) {
515  reportError("Parts should come in couples. Dropping it.");
516  return true;
517  }
518  handleValidMessages(*inputTypes);
519  return true;
520 }

522 bool DataProcessingDevice::tryDispatchComputation()
523 {
524  // This is the actual hidden state for the outer loop. In case we decide we
525  // want to support multithreaded dispatching of operations, I can simply
526  // move these to some thread local store and the rest of the lambdas
527  // should work just fine.
528  std::vector<DataRelayer::RecordAction> completed;
529  std::vector<MessageSet> currentSetOfInputs;
530 
531  auto& allocator = mAllocator;
532  auto& context = *mContextRegistry.get<MessageContext>();
533  auto& device = *this;
534  auto& errorCallback = mError;
535  auto& errorCount = mErrorCount;
536  auto& forwards = mSpec.forwards;
537  auto& inputsSchema = mSpec.inputs;
538  auto& processingCount = mProcessingCount;
539  auto& rdfContext = *mContextRegistry.get<ArrowContext>();
540  auto& relayer = mRelayer;
541  auto& serviceRegistry = mServiceRegistry;
542  auto& statefulProcess = mStatefulProcess;
543  auto& statelessProcess = mStatelessProcess;
544  auto& stringContext = *mContextRegistry.get<StringContext>();
545  auto& timingInfo = mTimingInfo;
546  auto& timesliceIndex = mServiceRegistry.get<TimesliceIndex>();
547  auto& rawContext = *mContextRegistry.get<RawBufferContext>();
548 
549  // These duplicate references are created so that each function
550  // does not need to know about the whole class state, but I can
551  // fine grain control what is exposed at each state.
552  // FIXME: I should use a different id for this state.
553  auto& monitoringService = mServiceRegistry.get<Monitoring>();
554  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
555  ScopedExit metricFlusher([&monitoringService] {
556  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
557  });
558 
559  auto reportError = [&device](const char* message) {
560  device.error(message);
561  };
562 
563  // For the moment we have a simple "immediately dispatch" policy for stuff
564  // in the cache. This could be controlled from the outside e.g. by waiting
565  // for a few sets of inputs to arrive before we actually dispatch the
566  // computation, however this can be defined at a later stage.
567  auto canDispatchSomeComputation = [&completed, &relayer]() -> bool {
568  completed = relayer.getReadyToProcess();
569  return completed.empty() == false;
570  };
571 
572  // We use this to get a list with the actual indexes in the cache which
573  // indicate a complete set of inputs. Notice how I fill the completed
574  // vector and return it, so that I can have a nice for loop iteration later
575  // on.
576  auto getReadyActions = [&relayer, &completed, &stats = mStats]() -> std::vector<DataRelayer::RecordAction> {
577  stats.pendingInputs = (int)relayer.getParallelTimeslices() - completed.size();
578  stats.incomplete = completed.empty() ? 1 : 0;
579  return completed;
580  };
581 
582  // This is needed to convert from a pair of pointers to an actual DataRef
583  // and to make sure the ownership is moved from the cache in the relayer to
584  // the execution.
585  auto fillInputs = [&relayer, &inputsSchema, &currentSetOfInputs](TimesliceSlot slot) -> InputRecord {
586  currentSetOfInputs = std::move(relayer.getInputsForTimeslice(slot));
587  auto getter = [&currentSetOfInputs](size_t i, size_t partindex) -> DataRef {
588  if (currentSetOfInputs[i].size() > partindex) {
589  return DataRef{nullptr,
590  static_cast<char const*>(currentSetOfInputs[i].at(partindex).header->GetData()),
591  static_cast<char const*>(currentSetOfInputs[i].at(partindex).payload->GetData())};
592  }
593  return DataRef{nullptr, nullptr, nullptr};
594  };
595  auto nofPartsGetter = [&currentSetOfInputs](size_t i) -> size_t {
596  return currentSetOfInputs[i].size();
597  };
598  InputSpan span{getter, nofPartsGetter, currentSetOfInputs.size()};
599  return InputRecord{inputsSchema, std::move(span)};
600  };

654  // When processing them, timers will have to be cleaned up
655  // to avoid double counting them.
656  // This was actually the easiest solution we could find for
657  // O2-646.
658  auto cleanTimers = [&currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
659  assert(record.size() == currentSetOfInputs.size());
660  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
661  DataRef input = record.getByPos(ii);
662  if (input.spec->lifetime != Lifetime::Timer) {
663  continue;
664  }
665  if (input.header == nullptr || input.payload == nullptr) {
666  continue;
667  }
668  // This will hopefully delete the message.
669  currentSetOfInputs[ii].clear();
670  }

673  // This is how we do the forwarding, i.e. we push
674  // the inputs which are shared between this device and others
675  // to the next one in the daisy chain.
676  // FIXME: do it in a smarter way than O(N^2)
677  auto forwardInputs = [&reportError, &forwards, &device, &currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
678  assert(record.size() == currentSetOfInputs.size());
679  // we collect all messages per forward in a map and send them together
680  std::unordered_map<std::string, FairMQParts> forwardedParts;
681  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
682  DataRef input = record.getByPos(ii);
683 
684  // If is now possible that the record is not complete when
685  // we forward it, because of a custom completion policy.
686  // this means that we need to skip the empty entries in the
687  // record for being forwarded.
688  if (input.header == nullptr || input.payload == nullptr) {
689  continue;
690  }
691  auto sih = o2::header::get<SourceInfoHeader*>(input.header);
692  if (sih) {
693  continue;
694  }
695 
696  auto dh = o2::header::get<DataHeader*>(input.header);
697  if (!dh) {
698  reportError("Header is not a DataHeader?");
699  continue;
700  }
701  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
702  if (!dph) {
703  reportError("Header stack does not contain DataProcessingHeader");
704  continue;
705  }
706 
707  for (auto& part : currentSetOfInputs[ii]) {
708  for (auto const& forward : forwards) {
709  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification) == false || (dph->startTime % forward.maxTimeslices) != forward.timeslice) {
710  continue;
711  }
712  auto& header = part.header;
713  auto& payload = part.payload;
714 
715  if (header.get() == nullptr) {
716  // FIXME: this should not happen, however it's actually harmless and
717  // we can simply discard it for the moment.
718  // LOG(ERROR) << "Missing header! " << dh->dataDescription.as<std::string>();
719  continue;
720  }
721  auto fdph = o2::header::get<DataProcessingHeader*>(header.get()->GetData());
722  if (fdph == nullptr) {
723  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
724  continue;
725  }
726  auto fdh = o2::header::get<DataHeader*>(header.get()->GetData());
727  if (fdh == nullptr) {
728  LOG(ERROR) << "Forwarded data does not have a DataHeader";
729  continue;
730  }
731  forwardedParts[forward.channel].AddPart(std::move(header));
732  forwardedParts[forward.channel].AddPart(std::move(payload));
733  }
734  }
735  }
736  for (auto& [channelName, channelParts] : forwardedParts) {
737  if (channelParts.Size() == 0) {
738  continue;
739  }
740  assert(channelParts.Size() % 2 == 0);
741  assert(o2::header::get<DataProcessingHeader*>(channelParts.At(0)->GetData()));
742  // in DPL we are using subchannel 0 only
743  device.Send(channelParts, channelName, 0);
744  }
745  };

747  auto calculateInputRecordLatency = [](InputRecord const& record, auto now) -> DataProcessingStats::InputLatency {
748  DataProcessingStats::InputLatency result{static_cast<int>(-1), 0};
749 
750  auto currentTime = (uint64_t)std::chrono::duration<double, std::milli>(now.time_since_epoch()).count();
751  for (auto& item : record) {
752  auto* header = o2::header::get<DataProcessingHeader*>(item.header);
753  if (header == nullptr) {
754  continue;
755  }
756  int partLatency = currentTime - header->creation;
757  result.minLatency = std::min(result.minLatency, partLatency);
758  result.maxLatency = std::max(result.maxLatency, partLatency);
759  }
760  return result;
761  };

763  auto calculateTotalInputRecordSize = [](InputRecord const& record) -> int {
764  size_t totalInputSize = 0;
765  for (auto& item : record) {
766  auto* header = o2::header::get<DataHeader*>(item.header);
767  if (header == nullptr) {
768  continue;
769  }
770  totalInputSize += header->payloadSize;
771  }

775  auto switchState = [& control = mServiceRegistry.get<ControlService>(),
776  &state = mState.streaming](StreamingState newState) {
777  state = newState;
778  control.notifyStreamingState(state);
779  };
780 
781  if (canDispatchSomeComputation() == false) {
782  return false;
783  }
784 
785  for (auto action : getReadyActions()) {
786  if (action.op == CompletionPolicy::CompletionOp::Wait) {
787  continue;
788  }
789 
790  prepareAllocatorForCurrentTimeSlice(TimesliceSlot{action.slot});
791  InputRecord record = fillInputs(action.slot);
792  if (action.op == CompletionPolicy::CompletionOp::Discard) {
793  if (forwards.empty() == false) {
794  forwardInputs(action.slot, record);
795  continue;
796  }
797  }
798  auto tStart = std::chrono::high_resolution_clock::now();
799  for (size_t ai = 0; ai != record.size(); ai++) {
800  auto cacheId = action.slot.index * record.size() + ai;
801  auto state = record.isValid(ai) ? 2 : 0;
802  mStats.relayerState.resize(std::max(cacheId + 1, mStats.relayerState.size()), 0);
803  mStats.relayerState[cacheId] = state;
804  }
805  try {
806  if (mState.quitRequested == false) {
807  dispatchProcessing(action.slot, record);
808  }
809  } catch (std::exception& e) {
810  errorHandling(e, record);
811  }
812  for (size_t ai = 0; ai != record.size(); ai++) {
813  auto cacheId = action.slot.index * record.size() + ai;
814  auto state = record.isValid(ai) ? 3 : 0;
815  mStats.relayerState.resize(std::max(cacheId + 1, mStats.relayerState.size()), 0);
816  mStats.relayerState[cacheId] = state;
817  }
818  auto tEnd = std::chrono::high_resolution_clock::now();
819  mStats.lastElapsedTimeMs = std::chrono::duration<double, std::milli>(tEnd - tStart).count();
820  mStats.lastTotalProcessedSize = calculateTotalInputRecordSize(record);
821  mStats.lastLatency = calculateInputRecordLatency(record, tStart);
822  // We forward inputs only when we consume them. If we simply Process them,
823  // we keep them for next message arriving.
824  if (action.op == CompletionPolicy::CompletionOp::Consume) {
825  if (forwards.empty() == false) {
826  forwardInputs(action.slot, record);
827  }
828  } else if (action.op == CompletionPolicy::CompletionOp::Process) {
829  cleanTimers(action.slot, record);
830  }
831  }
832  // We now broadcast the end of stream if it was requested
833  if (mState.streaming == StreamingState::EndOfStreaming) {
834  for (auto& channel : mSpec.outputChannels) {
835  DataProcessingHelpers::sendEndOfStream(*this, channel);
836  }
837  switchState(StreamingState::Idle);
838  }
839 
840  return true;
841 }
