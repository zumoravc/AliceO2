46 void TrackFitterTask::init(InitContext& ic)
47 {
49  LOG(INFO) << "initializing track fitter";
50  mTrackFitter = std::make_unique<o2::mft::TrackFitter>();
51 
52  auto filename = ic.options().get<std::string>("grp-file");
53  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
54  if (grp) {
55  mGRP.reset(grp);
56  o2::base::Propagator::initFieldFromGRP(grp);
57  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
58 
59  double centerMFT[3] = {0, 0, -61.4}; // Field at center of MFT
60  mTrackFitter->setBz(field->getBz(centerMFT));
61 
62  } else {
63  LOG(ERROR) << "Cannot retrieve GRP from the " << filename.c_str() << " file !";
64  mState = 0;
65  }
66  mState = 1;
67 }

70 void TrackFitterTask::run(ProcessingContext& pc)
71 {
72 
73  if (mState != 1)
74  return;
75 
76  auto tracksLTF = pc.inputs().get<gsl::span<o2::mft::TrackLTF>>("tracksltf");
77  auto tracksCA = pc.inputs().get<gsl::span<o2::mft::TrackCA>>("tracksca");
78 
79  int nTracksCA = 0;
80  int nFailedTracksCA = 0;
81  int nTracksLTF = 0;
82  int nFailedTracksLTF = 0;
83  std::vector<o2::mft::FitterTrackMFT> fittertracks(tracksLTF.size() + tracksCA.size());
84  std::list<o2::itsmft::Cluster> clusters;
85 
86  // Fit LTF tracks
87  for (const auto& track : tracksLTF) {
88  auto& temptrack = fittertracks.at(nTracksLTF + nFailedTracksLTF);
89  convertTrack(track, temptrack, clusters);
90  mTrackFitter->fit(temptrack, false) ? nTracksLTF++ : nFailedTracksLTF++;
91  } // end fit LTF tracks
92 
93  // Fit CA tracks
94  for (const auto& track : tracksCA) {
95  auto& temptrack = fittertracks.at(nTracksLTF + nFailedTracksLTF + nTracksCA + nFailedTracksCA);
96  convertTrack(track, temptrack, clusters);
97  mTrackFitter->fit(temptrack, false) ? nTracksCA++ : nFailedTracksCA++;
98  } // end fit CA tracks
99 
100  auto& finalMFTtracks = pc.outputs().make<std::vector<o2::mft::TrackMFT>>(Output{"MFT", "TRACKS", 0, Lifetime::Timeframe});
101  finalMFTtracks.resize(nTracksLTF + nTracksCA);
102 
103  auto nTotalTracks = 0;
104  // Convert fitter tracks to the final Standalone MFT Track
105  for (const auto& track : fittertracks) {
106  if (!track.isRemovable()) {
107  auto& temptrack = finalMFTtracks.at(nTotalTracks);
108  temptrack.setZ(track.first().getZ());
109  temptrack.setParameters(TtoSMatrix5(track.first().getParameters()));
110  temptrack.setCovariances(TtoSMatrixSym55(track.first().getCovariances()));
111  temptrack.setTrackChi2(track.first().getTrackChi2());
112  temptrack.setMCCompLabels(track.getMCCompLabels(), track.getNPoints());
113  temptrack.setInvQPtQuadtratic(track.getInvQPtQuadtratic());
114  temptrack.setChi2QPtQuadtratic(track.getChi2QPtQuadtratic());
115  //finalMFTtracks.back().printMCCompLabels();
116  nTotalTracks++;
117  }
118  }
119 
120  LOG(INFO) << "MFTFitter loaded " << tracksLTF.size() << " LTF tracks";
121  LOG(INFO) << "MFTFitter loaded " << tracksCA.size() << " CA tracks";
122  LOG(INFO) << "MFTFitter pushed " << fittertracks.size() << " tracks";
123  LOG(INFO) << "MFTFitter dropped " << nFailedTracksLTF << " LTF tracks";
124  LOG(INFO) << "MFTFitter dropped " << nFailedTracksCA << " CA tracks";
125 
126  mState = 2;
127  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
128 }

152 void convertTrack(const T& inTrack, O& outTrack, C& clusters)
153 {
154  //auto fittedTrack = FitterTrackMFT();
155  //TMatrixD covariances(5,5);
156  auto xpos = inTrack.getXCoordinates();
157  auto ypos = inTrack.getYCoordinates();
158  auto zpos = inTrack.getZCoordinates();
159  auto clusterIDs = inTrack.getClustersId();
160  auto nClusters = inTrack.getNPoints();
161  static int ntrack = 0;
162 
163  // Add clusters to Tracker's cluster vector & set fittedTrack cluster range.
164  // TODO: get rid of this cluster vector
165  for (auto cls = 0; cls < nClusters; cls++) {
166  o2::itsmft::Cluster& tempcluster = clusters.emplace_back(clusterIDs[cls], xpos[cls], ypos[cls], zpos[cls]);
167  tempcluster.setSigmaY2(5.43e-4); // FIXME:
168  tempcluster.setSigmaZ2(5.0e-4); // FIXME: Use clusters errors once available
169  outTrack.createParamAtCluster(tempcluster);
170  }
171  outTrack.setMCCompLabels(inTrack.getMCCompLabels(), nClusters);
172 }

175 SMatrix55 TtoSMatrixSym55(TMatrixD inMatrix)
176 {
177  // TMatrix to sym SMatrix
178  SMatrix55 outMatrix;
179  for (auto i = 5; i--;) {
180  outMatrix(i, i) = inMatrix(i, i);
181  for (auto j = i; j--;) {
182  outMatrix(i, j) = inMatrix(i, j);
183  }
184  }
185  return outMatrix;
186 }
