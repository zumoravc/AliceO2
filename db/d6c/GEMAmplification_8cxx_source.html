36  const float sigmaOverMu = mGasParam->SigmaOverMu;
37  const float kappa = 1 / (sigmaOverMu * sigmaOverMu);
38  boost::format polya("1/(TMath::Gamma(%1%)*%2%) * TMath::Power(x/%3%, %4%) * TMath::Exp(-x/%5%)");
39 
40  // to be replaced by std::filesystem::exists once we have C++17
41  auto fileexists = [](const char* fileName) -> bool {

46  const char* polyaFileName = "tpc_polya.root";
47  TFile* outfile;
48  auto cacheexists = fileexists(polyaFileName);
49  if (cacheexists) {
50  LOG(INFO) << "TPC: GEM setup from existing cache";
51  outfile = TFile::Open(polyaFileName);
52  } else {
53  LOG(INFO) << "TPC: GEM setup - initialization from scratch";
54  outfile = TFile::Open(polyaFileName, "CREATE");
55  }

58  for (int i = 0; i < 4; ++i) {
59  float s = mGEMParam->AbsoluteGain[i] / kappa;
60  polya % kappa % s % s % (kappa - 1) % s;
61  std::string name = polya.str();
62  o2::base::CachingTF1* polyaDistribution = nullptr;
63  if (!cacheexists) {
64  polyaDistribution = new o2::base::CachingTF1("polya", name.c_str(), 0, 10.f * mGEMParam->AbsoluteGain[i]);
67  polyaDistribution->SetNpx(100000);
68  } else {
69  polyaDistribution = (o2::base::CachingTF1*)outfile->Get(TString::Format("func%d", i).Data());
70  // FIXME: verify that distribution corresponds to the parameters used here
71  }
72  mGain[i].initialize(*polyaDistribution);
73 
74  if (!cacheexists) {
75  outfile->WriteTObject(polyaDistribution, TString::Format("func%d", i).Data());
76  }
77  delete polyaDistribution;
78  }
79 
81  const float gainStack = mGEMParam->TotalGainStack;
82  const float kappaStack = mGEMParam->KappaStack;
83  const float effStack = mGEMParam->EfficiencyStack;
84  const float sStack = gainStack / (kappaStack * (1.f - effStack));
85  polya % kappaStack % sStack % sStack % (kappaStack - 1) % sStack;
86  std::string name = polya.str();
87  o2::base::CachingTF1* polyaDistribution = nullptr;
88  if (!cacheexists) {
89  polyaDistribution = new o2::base::CachingTF1("polya", name.c_str(), 0, 50000);
90  polyaDistribution->SetNpx(50000);
91  } else {
92  polyaDistribution = (o2::base::CachingTF1*)outfile->Get("polyaStack");
93  }
94  mGainFullStack.initialize(*polyaDistribution);
95 
96  if (!cacheexists) {
97  outfile->WriteTObject(polyaDistribution, "polyaStack");
98  }
99  delete polyaDistribution;
100 
101  if (outfile)
102  outfile->Close();
103  watch.Stop();
104  LOG(INFO) << "TPC: GEM setup (polya) took " << watch.CpuTime();
105 }

159 int GEMAmplification::getGEMMultiplication(int nElectrons, int GEM)
160 {
163  if (nElectrons < 1) {
165  return 0;
166  } else if (nElectrons > 500) {
170  return ((mRandomGaus.getNextValue() * std::sqrt(static_cast<float>(nElectrons)) *
171  mGasParam->SigmaOverMu) +
172  nElectrons) *
173  mGEMParam->AbsoluteGain[GEM];
174  } else {
177  int electronsOut = 0;
178  for (int i = 0; i < nElectrons; ++i) {
179  electronsOut += mGain[GEM].getNextValue();
180  }
181  return electronsOut;
182  }
183 }

185 int GEMAmplification::getElectronLosses(int nElectrons, float probability)
186 {
188  float electronsFloat = static_cast<float>(nElectrons);
189  if (nElectrons < 1 || probability < 0.00001) {
191  return 0;
192  } else if (probability > 0.99999) {
194  return nElectrons;
195  } else if (electronsFloat * probability >= 5.f && electronsFloat * (1.f - probability) >= 5.f) {
198  return (mRandomGaus.getNextValue() * std::sqrt(electronsFloat * probability * (1 - probability))) +
199  electronsFloat * probability + 0.5;
200  } else {
202  int nElectronsOut = 0;
203  for (int i = 0; i < nElectrons; ++i) {
204  if (mRandomFlat.getNextValue() < probability) {
205  ++nElectronsOut;
206  }
207  }
208  return nElectronsOut;
209  }
