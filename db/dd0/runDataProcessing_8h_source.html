57 // This template magic allow users to customize the behavior of the process
58 // by (optionally) implementing a `configure` method which modifies one of the
59 // objects in question.
60 //
61 // For example it can be optionally implemented by the user to specify the
62 // channel policies for your setup. Use this if you want to customize the way
63 // your devices communicate between themself, e.g. if you want to use REQ/REP
64 // in place of PUB/SUB.
65 //
66 // The advantage of this approach is that we do not need to expose the
67 // configurability / configuration object to the user, unless he really wants to
68 // modify it. The drawback is that we need to declare the `customize` method
69 // before include this file.

115 int main(int argc, char** argv)
116 {
117  using namespace o2::framework;
118  using namespace boost::program_options;
119 
120  int result = 1;
121  try {
122  // The 0 here is an int, therefore having the template matching in the
123  // SFINAE expression above fit better the version which invokes user code over
124  // the default one.
125  // The default policy is a catch all pub/sub setup to be consistent with the past.
126  std::vector<o2::framework::ConfigParamSpec> workflowOptions;
127  UserCustomizationsHelper::userDefinedCustomization(workflowOptions, 0);
128  workflowOptions.push_back(ConfigParamSpec{"readers", VariantType::Int64, 1ll, {"number of parallel readers to use"}});
129  workflowOptions.push_back(ConfigParamSpec{"pipeline", VariantType::String, "", {"override default pipeline size"}});
130  std::vector<ChannelConfigurationPolicy> channelPolicies;
131  UserCustomizationsHelper::userDefinedCustomization(channelPolicies, 0);
132  auto defaultChannelPolicies = ChannelConfigurationPolicy::createDefaultPolicies();
133  channelPolicies.insert(std::end(channelPolicies), std::begin(defaultChannelPolicies), std::end(defaultChannelPolicies));
134 
135  std::vector<CompletionPolicy> completionPolicies;
136  UserCustomizationsHelper::userDefinedCustomization(completionPolicies, 0);
137  auto defaultCompletionPolicies = CompletionPolicy::createDefaultPolicies();
138  completionPolicies.insert(std::end(completionPolicies), std::begin(defaultCompletionPolicies), std::end(defaultCompletionPolicies));
139 
140  std::vector<DispatchPolicy> dispatchPolicies;
141  UserCustomizationsHelper::userDefinedCustomization(dispatchPolicies, 0);
142  auto defaultDispatchPolicies = DispatchPolicy::createDefaultPolicies();
143  dispatchPolicies.insert(std::end(dispatchPolicies), std::begin(defaultDispatchPolicies), std::end(defaultDispatchPolicies));
144 
145  std::vector<std::unique_ptr<ParamRetriever>> retrievers;
146  std::unique_ptr<ParamRetriever> retriever{new BoostOptionsRetriever(true, argc, argv)};
147  retrievers.emplace_back(std::move(retriever));
148  auto workflowOptionsStore = std::make_unique<ConfigParamStore>(workflowOptions, std::move(retrievers));
149  workflowOptionsStore->preload();
150  workflowOptionsStore->activate();
151  ConfigParamRegistry workflowOptionsRegistry(std::move(workflowOptionsStore));
152  ConfigContext configContext(workflowOptionsRegistry, argc, argv);
153  o2::framework::WorkflowSpec specs = defineDataProcessing(configContext);
154  overridePipeline(configContext, specs);
155  result = doMain(argc, argv, specs, channelPolicies, completionPolicies, dispatchPolicies, workflowOptions, configContext);
156  } catch (boost::exception& e) {
157  doBoostException(e);
158  } catch (std::exception const& error) {
159  LOG(ERROR) << "error while setting up workflow: " << error.what();
160  } catch (...) {
161  LOG(ERROR) << "Unknown error while setting up workflow.";
162  }
163 
164  char* idstring = nullptr;
165  for (int argi = 0; argi < argc; argi++) {
166  if (strcmp(argv[argi], "--id") == 0 && argi + 1 < argc) {
167  idstring = argv[argi + 1];
168  break;
169  }
170  }
171  o2::framework::OnWorkflowTerminationHook onWorkflowTerminationHook;
172  UserCustomizationsHelper::userDefinedCustomization(onWorkflowTerminationHook, 0);
173  onWorkflowTerminationHook(idstring);
174  LOG(INFO) << "Process " << getpid() << " is exiting.";
175  return result;
176 }
