55 {
56  // for the TPC it is useful to take at most half of the available (logical) cores due to memory requirements
57  int defaultlanes = std::max(1u, std::thread::hardware_concurrency() / 2);
58  const std::string laneshelp("Number of tpc processing lanes. A lane is a pipeline of algorithms.");
59 
60  const std::string sectorshelp("List of TPC sectors, comma separated ranges, e.g. 0-3,7,9-15");

62  const std::string tpcrthelp("Run TPC reco workflow to specified output type, currently supported: 'digits,clusters,tracks'");
63  const std::string decoderHelp("Decoder type to use: 'GBT,LinkZS'");
64 
65  std::vector<ConfigParamSpec> options{
66  {"input-spec", VariantType::String, "A:TPC/RAWDATA", {"selection string input specs"}},
67  {"tpc-lanes", VariantType::Int, defaultlanes, {laneshelp}},
68  {"tpc-sectors", VariantType::String, sectorDefault.c_str(), {sectorshelp}},
69  {"tpc-reco-output", VariantType::String, "", {tpcrthelp}},
70  {"decoder-type", VariantType::String, "GBT", {decoderHelp}},
71  {"configKeyValues", VariantType::String, "", {"Semicolon separated key=value strings (e.g.: 'TPCCalibPedestal.FirstTimeBin=10;...')"}},
72  {"configFile", VariantType::String, "", {"configuration file for configurable parameters"}}};
73 
74  std::swap(workflowOptions, options);

79 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)
80 // by default this will be std::max(the number of physical cores, numberofsectors)
81 // as a temporary means to fully use a machine and as a way to play with different topologies
82 int getNumTPCLanes(std::vector<int> const& sectors, ConfigContext const& configcontext)
83 {
84  auto lanes = configcontext.options().get<int>("tpc-lanes");
85  if (lanes < 0) {
86  LOG(FATAL) << "tpc-lanes needs to be positive\n";
87  return 0;
88  }
89  // crosscheck with sectors
90  return std::min(lanes, (int)sectors.size());
91 }

93 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
94 {
95 
96  // set up configuration
97  o2::conf::ConfigurableParam::updateFromFile(configcontext.options().get<std::string>("configFile"));
98  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
99  o2::conf::ConfigurableParam::writeINI("o2-tpc-raw-reco-workflow_configuration.ini");
100 
101  WorkflowSpec specs;
102 
103  // for the moment no parallel workflow, so just one lane hard coded
104  auto tpcSectors = o2::RangeTokenizer::tokenize<int>(configcontext.options().get<std::string>("tpc-sectors"));
105  auto lanes = 1; //getNumTPCLanes(tpcSectors, config);
106 
107  const auto decoderType = DecoderTypeMap.at(configcontext.options().get<std::string>("decoder-type"));
108 
109  int fanoutsize = 0;
110  for (int l = 0; l < lanes; ++l) {
111  if (decoderType == DecoderType::GBT) {
112  specs.emplace_back(o2::tpc::getRawToDigitsSpec(fanoutsize, configcontext.options().get<std::string>("input-spec"), tpcSectors));
113  } else if (decoderType == DecoderType::LinkZS) {
114  specs.emplace_back(o2::tpc::getLinkZSToDigitsSpec(fanoutsize, configcontext.options().get<std::string>("input-spec"), tpcSectors));
115  } else {
116  LOG(FATAL) << "bad decoder type";
117  }
118  fanoutsize++;
119  }
120 
121  // ===| attach the TPC reco workflow |========================================
122  // default is to dump digits
123  std::string_view recoOuput = "digits";
124  const auto tpcRecoOutputType = configcontext.options().get<std::string>("tpc-reco-output");
125  if (!tpcRecoOutputType.empty()) {
126  recoOuput = tpcRecoOutputType.c_str();
127  }
128 
129  auto tpcRecoWorkflow = o2::tpc::reco_workflow::getWorkflow(tpcSectors, tpcSectors, false, lanes, "digitizer", recoOuput.data());
130  specs.insert(specs.end(), tpcRecoWorkflow.begin(), tpcRecoWorkflow.end());
131 
132  return specs;
133 }
