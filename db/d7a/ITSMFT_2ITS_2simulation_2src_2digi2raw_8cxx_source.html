94 void digi2raw(std::string_view inpName, std::string_view outDir, bool filePerCRU, int verbosity, uint32_t rdhV, bool noEmptyHBF, int superPageSizeInB)
95 {
96  TStopwatch swTot;
97  swTot.Start();
98  using ROFR = o2::itsmft::ROFRecord;
99  using ROFRVEC = std::vector<o2::itsmft::ROFRecord>;
100  const uint8_t ruSWMin = 0, ruSWMax = 0xff; // seq.ID of 1st and last RU (stave) to convert
101 
102  LOG(INFO) << "HBFUtil settings:";
103  o2::raw::HBFUtils::Instance().print();
104 
105  // if needed, create output directory
106  if (gSystem->AccessPathName(outDir.data())) {
107  if (gSystem->mkdir(outDir.data(), kTRUE)) {
108  LOG(FATAL) << "could not create output directory " << outDir;
109  } else {
110  LOG(INFO) << "created output directory " << outDir;
111  }
112  }
114  std::string digTreeName{o2::base::NameConf::MCTTREENAME.data()};
115  TChain digTree(digTreeName.c_str());
116  digTree.AddFile(inpName.data());
117  digTree.SetBranchStatus("*MCTruth*", 0); // ignore MC info
118 
119  std::vector<o2::itsmft::Digit> digiVec, *digiVecP = &digiVec;
120  std::string digBranchName = o2::utils::concat_string(MAP::getName(), "Digit");
121  if (!digTree.GetBranch(digBranchName.c_str())) {
122  LOG(FATAL) << "Failed to find the branch " << digBranchName << " in the tree " << digTreeName;
123  }
124  digTree.SetBranchAddress(digBranchName.c_str(), &digiVecP);
125 
126  // ROF record entries in the digit tree
127  ROFRVEC rofRecVec, *rofRecVecP = &rofRecVec;
128  std::string rofRecName = o2::utils::concat_string(MAP::getName(), "DigitROF");
129  if (!digTree.GetBranch(rofRecName.c_str())) {
130  LOG(FATAL) << "Failed to find the branch " << rofRecName << " in the tree " << digTreeName;
131  }
132  digTree.SetBranchAddress(rofRecName.c_str(), &rofRecVecP);
134  std::string inputGRP = o2::base::NameConf::getGRPFileName();
135  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
136 
137  o2::itsmft::MC2RawEncoder<MAP> m2r;
138  m2r.setVerbosity(verbosity);
139  m2r.setContinuousReadout(grp->isDetContinuousReadOut(MAP::getDetID())); // must be set explicitly
140  m2r.setDefaultSinkName(o2::utils::concat_string(MAP::getName(), ".raw"));
141  m2r.setMinMaxRUSW(ruSWMin, ruSWMax);
142  m2r.getWriter().setSuperPageSize(superPageSizeInB);
143  m2r.getWriter().useRDHVersion(rdhV);
144  m2r.getWriter().setDontFillEmptyHBF(noEmptyHBF);
145 
146  m2r.setVerbosity(verbosity);
147  setupLinks(m2r, outDir, MAP::getName(), filePerCRU);
148  //-------------------------------------------------------------------------------<<<<
149  int lastTreeID = -1;
150  long offs = 0, nEntProc = 0;
151  for (int i = 0; i < digTree.GetEntries(); i++) {
152  digTree.GetEntry(i);
153  for (const auto& rofRec : rofRecVec) {
154  int nDigROF = rofRec.getNEntries();
155  if (verbosity) {
156  LOG(INFO) << "Processing ROF:" << rofRec.getROFrame() << " with " << nDigROF << " entries";
157  rofRec.print();
158  }
159  if (!nDigROF) {
160  if (verbosity) {
161  LOG(INFO) << "Frame is empty"; // ??
162  }
163  continue;
164  }
165  nEntProc++;
166  auto dgs = nDigROF ? gsl::span<const o2::itsmft::Digit>(&digiVec[rofRec.getFirstEntry()], nDigROF) : gsl::span<const o2::itsmft::Digit>();
167  m2r.digits2raw(dgs, rofRec.getBCData());
168  }
169  } // loop over multiple ROFvectors (in case of chaining)
170 
171  m2r.getWriter().writeConfFile(MAP::getName(), "RAWDATA", o2::utils::concat_string(outDir, '/', MAP::getName(), "raw.cfg"));
172  m2r.finalize(); // finish TF and flush data
173  //
174  swTot.Stop();
175  swTot.Print();

178 void setupLinks(o2::itsmft::MC2RawEncoder<MAP>& m2r, std::string_view outDir, std::string_view outPrefix, bool filePerCRU)
179 {
180  //------------------------------------------------------------------------------->>>>
181  // just as an example, we require here that the staves are read via 3 links, with partitioning according to lnkXB below
182  // while OB staves use only 1 link.
183  // Note, that if the RU container is not defined, it will be created automatically
184  // during encoding.
185  // If the links of the container are not defined, a single link readout will be assigned
186 
187  constexpr int MaxLinksPerRU = 3;
188  constexpr int MaxLinksPerCRU = 16;
189  const auto& mp = m2r.getMapping();
190  int lnkAssign[3][MaxLinksPerRU] = {
191  // requested link cabling for IB, MB and OB
192  /* // uncomment this to have 1 link per RU
193  {9, 0, 0}, // IB
194  {16, 0, 0}, // MB
195  {28, 0, 0} // OB
196  */
197  /* // uncomment this to have 3 link per RU
198  {3, 3, 3}, // IB
199  {5, 5, 6}, // MB
200  {9, 9, 10} // OB
201  */
202  // partition according to https://indico.cern.ch/event/907685/contributions/3818967/attachments/2019436/3376161/20200411_SOX_EOX_Proposal.pdf
203  {3, 3, 3}, // IB
204  {8, 8, 0}, // MB
205  {14, 14, 0} // OB
206  };
207  std::vector<std::vector<int>> defRU{// number of RUs per CRU at each layer
208  {6, 6},
209  {8, 8},
210  {5, 5, 5, 5},
211  {12, 12},
212  {8, 7, 8, 7},
213  {11, 10, 11, 10},
214  {12, 12, 12, 12}};
215  // this is an arbitrary mapping
216  int nCRU = 0, nRUtot = 0, nRU = 0, nLinks = 0;
217  int linkID = 0, cruID = o2::detectors::DetID::ITS << 10; // this will be the lowest CRUID
218  std::string outFileLink;
219 
220  for (int ilr = 0; ilr < mp.NLayers; ilr++) {
221  int nruLr = mp.getNStavesOnLr(ilr);
222  int ruType = mp.getRUType(nRUtot); // IB, MB or OB
223  int* lnkAs = lnkAssign[ruType];
224  // count requested number of links per RU
225  int nlk = 0;
226  for (int i = 3; i--;) {
227  nlk += lnkAs[i] ? 1 : 0;
228  }
229  const auto& ruList = defRU[ilr];
230  // sanity check
231  if (std::accumulate(ruList.begin(), ruList.end(), 0) != nruLr) {
232  throw std::runtime_error("Declared RU partitioning does not add up to expected N staves per layer");
233  }
234  int ruInLr = 0; // counter for RUs within the layer
235  int nCRUlr = ruList.size();
236  for (int iCRU = 0; iCRU < nCRUlr; iCRU++) {
237  for (int irc = 0; irc < ruList[iCRU]; irc++) {
238  int ruID = nRUtot++;
239  bool acceptRU = !(ruID < m2r.getRUSWMin() || ruID > m2r.getRUSWMax()); // ignored RUs ?
240  if (acceptRU) {
241  m2r.getCreateRUDecode(ruID); // create RU container
242  nRU++;
243  }
244  int accL = 0;
245  for (int il = 0; il < MaxLinksPerRU; il++) { // create links
246  if (acceptRU && lnkAs[il]) {
247  nLinks++;
248  auto& ru = *m2r.getRUDecode(ruID);
249  uint32_t lanes = mp.getCablesOnRUType(ru.ruInfo->ruType); // lanes patter of this RU
250  ru.links[il] = m2r.addGBTLink();
251  auto link = m2r.getGBTLink(ru.links[il]);
252  link->lanes = lanes & ((0x1 << lnkAs[il]) - 1) << (accL);
253  link->idInCRU = linkID;
254  link->cruID = cruID;
255  link->feeID = mp.RUSW2FEEId(ruID, il);
256  link->endPointID = 0; // 0 or 1
257  accL += lnkAs[il];
258  if (m2r.getVerbosity()) {
259  LOG(INFO) << "RU" << ruID << '(' << ruInLr << " on lr " << ilr << ") " << link->describe()
260  << " -> " << outFileLink;
261  }
262  // register the link in the writer, if not done here, its data will be dumped to common default file
263  outFileLink = filePerCRU ? o2::utils::concat_string(outDir, "/", outPrefix, "_cru", std::to_string(nCRU), ".raw") : o2::utils::concat_string(outDir, "/", outPrefix, "_lr", std::to_string(ilr), ".raw");
264  m2r.getWriter().registerLink(link->feeID, link->cruID, link->idInCRU, link->endPointID, outFileLink);
265  //
266  linkID++;
267  }
268  } // links of RU
269  ruInLr++;
270  } // RUs of CRU
271  cruID++;
272  nCRU++;
273  } // CRUs of the layer
274  } // layers
275  LOG(INFO) << "Distributed " << nLinks << " links on " << nRU << " RUs in " << nCRU << " CRUs";
276 }
