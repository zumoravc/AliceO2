37 // Default track selection requiring one hit in the SPD
38 TrackSelection getGlobalTrackSelection()
39 {
40  TrackSelection selectedTracks;
41  selectedTracks.SetMinPt(0.1f);
42  selectedTracks.SetMaxPt(1e10f);
43  selectedTracks.SetMinEta(-0.8f);
44  selectedTracks.SetMaxEta(0.8f);
45  selectedTracks.SetRequireITSRefit(true);
46  selectedTracks.SetRequireTPCRefit(true);
47  selectedTracks.SetMinNCrossedRowsTPC(70);
48  selectedTracks.SetMinNCrossedRowsOverFindableClustersTPC(0.8f);
49  selectedTracks.SetMaxChi2PerClusterTPC(4.f);
50  selectedTracks.SetMaxChi2PerClusterITS(36.f);
51  selectedTracks.SetRequireHitsInITSLayers(1,
52  {0, 1}); // one hit in any SPD layer
53  selectedTracks.SetMaxDcaXY(2.4f);
54  selectedTracks.SetMaxDcaZ(2.f);
55  return selectedTracks;
56 }

58 // Default track selection requiring no hit in the SPD and one in the innermost
59 // SDD
60 TrackSelection getGlobalTrackSelectionSDD()
61 {
62  TrackSelection selectedTracks = getGlobalTrackSelection();
63  selectedTracks.ResetITSRequirements();
64  selectedTracks.SetRequireNoHitsInITSLayers({0, 1}); // no hit in SPD layers
65  selectedTracks.SetRequireHitsInITSLayers(1,
66  {2}); // one hit in first SDD layer
67  return selectedTracks;
68 }

75 struct TrackExtensionTask {
76 
77  Produces<aod::TracksExtended> extendedTrackQuantities;
78 
79  void process(aod::Collision const& collision,
80  soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksCov> const& tracks)
81  {
82  float sinAlpha = 0.f;
83  float cosAlpha = 0.f;
84  float globalX = 0.f;
85  float globalY = 0.f;
86  float dcaXY = 0.f;
87  float dcaZ = 0.f;
88 
89  for (auto& track : tracks) {
90 
91  sinAlpha = sin(track.alpha());
92  cosAlpha = cos(track.alpha());
93  globalX = track.x() * cosAlpha - track.y() * sinAlpha;
94  globalY = track.x() * sinAlpha + track.y() * cosAlpha;
95 
96  dcaXY = track.charge() * sqrt(pow((globalX - collision.posX()), 2) +
97  pow((globalY - collision.posY()), 2));
98  dcaZ = track.charge() * sqrt(pow(track.z() - collision.posZ(), 2));
99 
100  extendedTrackQuantities(dcaXY, dcaZ);
101  }
102  }
103 };

245  void process(aod::Collision const& collision,
246  soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksCov, aod::TracksExtended,
247  aod::TrackSelection> const& tracks)
248  {
249 
250  collisionPos->Fill(collision.posX(), collision.posY());
251 
252  for (auto& track : tracks) {
253 
254  if (selectedTracks == 1 && !track.isGlobalTrack())
255  continue; // FIXME: this should be replaced by framework filter
256  else if (selectedTracks == 2 && !track.isGlobalTrackSDD())
257  continue;
258 
259  // TPC
260  tpcFindableClusters->Fill(track.tpcNClsFindable());
261  tpcFoundClusters->Fill(track.tpcNClsFound());
262  tpcSharedClusters->Fill(track.tpcNClsShared());
263  tpcCrossedRows->Fill(track.tpcNClsCrossedRows());
264 
265  tpcCrossedRowsOverFindableClusters->Fill(
266  track.tpcCrossedRowsOverFindableCls());
267  tpcFractionSharedClusters->Fill(track.tpcFractionSharedCls());
268  tpcChi2PerCluster->Fill(
269  track.tpcChi2NCl());
270 
271  // ITS
272  itsFoundClusters->Fill(track.itsNCls());
273  itsChi2PerCluster->Fill(track.itsChi2NCl());
274  for (unsigned int i = 0; i < 7; i++) {
275  if (track.itsClusterMap() & (1 << i))
276  itsHits->Fill(i);
277  }
278 
279  // Tracks
280  alpha->Fill(track.alpha());
281  x->Fill(track.x());
282  y->Fill(track.y());
283  z->Fill(track.z());
284 
285  dcaxy->Fill(track.dcaXY());
286  dcaz->Fill(track.dcaZ());
287 
288  signed1Pt->Fill(track.signed1Pt());
289  snp->Fill(track.snp());
290  tgl->Fill(track.tgl());
291 
292  for (unsigned int i = 0; i < 64; i++) {
293  if (track.flags() & (1 << i))
294  flag->Fill(i);
295  }
296 
297  pt->Fill(track.pt());
298  eta->Fill(track.eta());
299  phi->Fill(track.phi());
300  }
301  }
