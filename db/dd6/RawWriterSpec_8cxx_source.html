42  {
43  auto filename = ic.options().get<std::string>("mid-raw-outfile");
44  mEncoder.init(filename.c_str());
45 
46  auto stop = [this]() {
47  mEncoder.finalize();
48  };
49  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
50 
51  // Write basic config files to be used with raw data reader workflow
52  auto cfgFilename = filename;
53  auto pos = filename.find_last_of('.');
54  if (pos != std::string::npos) {
55  cfgFilename.erase(pos);
56  }
57  cfgFilename += ".cfg";
58  std::ofstream outCfgFile(cfgFilename.c_str());
59  outCfgFile << "[defaults]\n";
60  outCfgFile << "dataOrigin = " << header::gDataOriginMID.as<std::string>() << "\n\n";
61  outCfgFile << "[input-file]\n";
62  outCfgFile << "filePath = " << filename << "\n";
63  outCfgFile.close();
64  }

66  void run(o2::framework::ProcessingContext& pc)
67  {
68  auto msg = pc.inputs().get("mid_data");
69  gsl::span<const ColumnData> data = of::DataRefUtils::as<const ColumnData>(msg);
70 
71  auto msgROF = pc.inputs().get("mid_data_rof");
72  gsl::span<const ROFRecord> rofRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
73 
74  for (auto& rofRecord : rofRecords) {
75  auto eventData = data.subspan(rofRecord.firstEntry, rofRecord.nEntries);
76  mEncoder.process(eventData, rofRecord.interactionRecord, rofRecord.eventType);
77  }
78  }
79 
80  private:
81  Encoder mEncoder{};
