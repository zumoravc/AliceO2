258 inline void GBTFrameContainer::addGBTFrame(unsigned word3, unsigned word2, unsigned word1, unsigned word0)
259 {
260  if (!mEnableStoreGBTFrames && (mGBTFrames.size() > 1)) {
261  mGBTFrames[0] = mGBTFrames[1];
262  mGBTFrames[1].setData(word3, word2, word1, word0);
263  } else {
264  mGBTFrames.emplace_back(word3, word2, word1, word0);
265  }
266  processFrame(mGBTFrames.end() - 1);
267 };
