69 GPUd() MEM_CLASS_PRE2() inline GPUTPCTrackLinearisation::GPUTPCTrackLinearisation(const MEM_LG2(GPUTPCTrackParam) & GPUrestrict() t) : mSinPhi(t.SinPhi()), mCosPhi(0), mDzDs(t.DzDs()), mQPt(t.QPt())
70 {
71  if (mSinPhi > GPUCA_MAX_SIN_PHI) {
72  mSinPhi = GPUCA_MAX_SIN_PHI;
73  } else if (mSinPhi < -GPUCA_MAX_SIN_PHI) {
74  mSinPhi = -GPUCA_MAX_SIN_PHI;
75  }
76  mCosPhi = CAMath::Sqrt(1 - mSinPhi * mSinPhi);
77  if (t.SignCosPhi() < 0) {
78  mCosPhi = -mCosPhi;
79  }
80 }
