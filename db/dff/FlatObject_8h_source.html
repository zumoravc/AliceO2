121 template <typename T>
122 T* resizeArray(T*& ptr, int oldSize, int newSize, T* newPtr = nullptr)
123 {
124  // Resize array pointed by ptr. T must be a POD class.
125  // If the non-null newPtr is provided, use it instead of allocating a new one.
126  // In this case it is up to the user to ensure that it has at least newSize slots allocated.
127  // Return original array pointer, so that the user can manage previously allocate memory
128  if (oldSize < 0) {
129  oldSize = 0;
130  }
131  if (newSize > 0) {
132  if (!newPtr) {
133  newPtr = new T[newSize];
134  }
135  int mcp = std::min(newSize, oldSize);
136  std::memmove(newPtr, ptr, mcp * sizeof(T));
137  if (newSize > oldSize) {
138  std::memset(newPtr + mcp, 0, (newSize - oldSize) * sizeof(T));
139  }
140  }
141  T* oldPtr = ptr;
142  ptr = newPtr;
143  return oldPtr;
144 }

367 inline void FlatObject::cloneFromObject(const FlatObject& obj, char* newFlatBufferPtr)
368 {
372  // that we want to relocate the obj to external buffer)
373 
374  assert(obj.isConstructed());
375 
376  // providing *this with internal buffer as obj makes sens only if we want to conver it to object with PROVIDED external buffer
377  assert(!(!newFlatBufferPtr && obj.mFlatBufferPtr == mFlatBufferPtr && obj.isBufferInternal()));
378 
379  char* oldPtr = resizeArray(mFlatBufferPtr, mFlatBufferSize, obj.mFlatBufferSize, newFlatBufferPtr);
380 
381  if (isBufferInternal()) {
382  delete[] oldPtr; // delete old buffer if owned
383  }
384  mFlatBufferSize = obj.mFlatBufferSize;
385  mFlatBufferContainer = newFlatBufferPtr ? nullptr : mFlatBufferPtr; // external buffer is not provided, make object to own the buffer
386  std::memcpy(mFlatBufferPtr, obj.mFlatBufferPtr, obj.mFlatBufferSize);
387  mConstructionMask = (unsigned int)ConstructionState::Constructed;
388 }

455 inline int FlatObject::writeToFile(T& obj, TFile& outf, const char* name)
456 {
458  assert(obj.isConstructed());
459 
460  if (outf.IsZombie()) {
461  LOG(ERROR) << "Failed to write to file " << outf.GetName();
462  return -1;
463  }
464 
465  bool isBufferExternal = !obj.isBufferInternal();
466  if (isBufferExternal) {
467  obj.adoptInternalBuffer(obj.mFlatBufferPtr);
468  }
469  outf.WriteObjectAny(&obj, T::Class(), name);
470  if (isBufferExternal) {
471  obj.clearInternalBufferPtr();
472  }
473  return 0;
474 }

476 template <class T, class TFile>
477 inline T* FlatObject::readFromFile(TFile& inpf, const char* name)
478 {
480 
481  if (inpf.IsZombie()) {
482  LOG(ERROR) << "Failed to read from file " << inpf.GetName();
483  return nullptr;
484  }
485  T* pobj = reinterpret_cast<T*>(inpf.GetObjectChecked(name, T::Class()));
486  if (!pobj) {
487  LOG(ERROR) << "Failed to load " << name << " from " << inpf.GetName();
488  return nullptr;
489  }
490  if (pobj->mFlatBufferSize > 0 && pobj->mFlatBufferContainer == nullptr) {
491  LOG(ERROR) << "Failed to load " << name << " from " << inpf.GetName() << ": empty flat buffer container";
492  return nullptr;
493  }
494  pobj->setActualBufferAddress(pobj->mFlatBufferContainer);
495  return pobj;
496 }

501 template <class T>
502 inline std::string FlatObject::stressTest(T& obj)
503 {
507  std::string err;
508 
509  if (!obj.isConstructed())
510  return "tested object is not constructed!";
511 
512  T tst;
513  tst.cloneFromObject(obj, nullptr);
514  if (!tst.isConstructed() || !tst.isBufferInternal())
515  return "error at cloneFromObject()!";
516 
517  obj.destroy();
518 
519  char* buf0 = tst.releaseInternalBuffer();
520  char* buf1 = new char[tst.getFlatBufferSize()];
521  char* buf2 = new char[tst.getFlatBufferSize()];
522  std::memcpy(buf1, tst.getFlatBufferPtr(), tst.getFlatBufferSize());
523  tst.setActualBufferAddress(buf1);
524  delete[] buf0;
525 
526  tst.setFutureBufferAddress(buf2);
527  std::memcpy(buf2, buf1, tst.getFlatBufferSize());
528  delete[] buf1;
529 
530  if (tst.isBufferInternal())
531  return err = "error, buffer should be external!";
532 
533  tst.adoptInternalBuffer(buf2);
534  if (!tst.isBufferInternal())
535  return err = "error, buffer should be internal!";
536 
537  obj.cloneFromObject(tst, nullptr);
538  if (!obj.isBufferInternal())
539  return err = "error, buffer should be internal!";
540 
541  return err;
542 }
