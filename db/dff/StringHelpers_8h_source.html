62 constexpr uint32_t crc32(char const* str, int length)
63 {
64  uint32_t crc = 0xFFFFFFFF;
65  for (auto j = 0; j <= length; ++j) {
66  crc = (crc >> 8) ^ crc_table[(crc ^ static_cast<unsigned int>(str[j])) & 0x000000FF];
67  }
68  return crc;
69 }
