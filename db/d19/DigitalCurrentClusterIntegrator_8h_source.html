37  {
38  if (mIntegratedCurrents[sector][row] == nullptr) {
39  int nPads = o2::tpc::Mapper::instance().getNumberOfPadsInRowSector(row);
40  mIntegratedCurrents[sector][row].reset(new unsigned long long int[nPads]);

45  {
46  int ipad = ipad + 0.5;
47  if (ipad < 0)
48  ipad = 0;
49  int maxPad = o2::tpc::Mapper::instance().getNumberOfPadsInRowSector(row);
50  if (ipad >= maxPad)
51  ipad = maxPad - 1;
52  mIntegratedCurrents[sector][row][ipad] += charge;
53  }
