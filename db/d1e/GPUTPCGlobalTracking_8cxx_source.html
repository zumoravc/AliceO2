116 GPUd() void GPUTPCGlobalTracking::PerformGlobalTracking(int nBlocks, int nThreads, int iBlock, int iThread, const GPUTPCTracker& tracker, GPUsharedref() MEM_LOCAL(GPUSharedMemory) & GPUrestrict() smem, GPUTPCTracker& GPUrestrict() sliceTarget, bool right)
117 {
118  for (int i = iBlock * nThreads + iThread; i < tracker.CommonMemory()->nLocalTracks; i += nThreads * nBlocks) {
119  {
120  const int tmpHit = tracker.Tracks()[i].FirstHitID();
121  if (tracker.TrackHits()[tmpHit].RowIndex() >= GPUCA_GLOBAL_TRACKING_MIN_ROWS && tracker.TrackHits()[tmpHit].RowIndex() < GPUCA_GLOBAL_TRACKING_RANGE) {
122  int rowIndex = tracker.TrackHits()[tmpHit].RowIndex();
123  const GPUTPCRow& GPUrestrict() row = tracker.Row(rowIndex);
124  float Y = (float)tracker.Data().HitDataY(row, tracker.TrackHits()[tmpHit].HitIndex()) * row.HstepY() + row.Grid().YMin();
125  if (sliceTarget.CommonMemory()->nTracks >= sliceTarget.NMaxTracks()) { // >= since will increase by 1
126  sliceTarget.CommonMemory()->kernelError = GPUCA_ERROR_GLOBAL_TRACKING_TRACK_OVERFLOW;
127  return;
128  }
129  if (!right && Y < -row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_LOWER) {
130  // GPUInfo("Track %d, lower row %d, left border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, -row.MaxY());
131  PerformGlobalTrackingRun(sliceTarget, smem, tracker, i, rowIndex, -tracker.Param().DAlpha, -1);
132  }
133  if (right && Y > row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_LOWER) {
134  // GPUInfo("Track %d, lower row %d, right border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, row.MaxY());
135  PerformGlobalTrackingRun(sliceTarget, smem, tracker, i, rowIndex, tracker.Param().DAlpha, -1);
136  }
137  }
138  }
139 
140  {
141  const int tmpHit = tracker.Tracks()[i].FirstHitID() + tracker.Tracks()[i].NHits() - 1;
142  if (tracker.TrackHits()[tmpHit].RowIndex() < GPUCA_ROW_COUNT - GPUCA_GLOBAL_TRACKING_MIN_ROWS && tracker.TrackHits()[tmpHit].RowIndex() >= GPUCA_ROW_COUNT - GPUCA_GLOBAL_TRACKING_RANGE) {
143  int rowIndex = tracker.TrackHits()[tmpHit].RowIndex();
144  const GPUTPCRow& GPUrestrict() row = tracker.Row(rowIndex);
145  float Y = (float)tracker.Data().HitDataY(row, tracker.TrackHits()[tmpHit].HitIndex()) * row.HstepY() + row.Grid().YMin();
146  if (sliceTarget.CommonMemory()->nTracks >= sliceTarget.NMaxTracks()) { // >= since will increase by 14
147  sliceTarget.CommonMemory()->kernelError = GPUCA_ERROR_GLOBAL_TRACKING_TRACK_OVERFLOW;
148  return;
149  }
150  if (!right && Y < -row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_UPPER) {
151  // GPUInfo("Track %d, upper row %d, left border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, -row.MaxY());
152  PerformGlobalTrackingRun(sliceTarget, smem, tracker, i, rowIndex, -tracker.Param().DAlpha, 1);
153  }
154  if (right && Y > row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_UPPER) {
155  // GPUInfo("Track %d, upper row %d, right border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, row.MaxY());
