23 {
24  LOG(INFO) << "Init trigger \'particle\' with following parameters";
25  LOG(INFO) << param;
26  return [&param](const std::vector<TParticle>& particles) -> bool {
27  for (const auto& particle : particles) {
28  if (particle.GetPdgCode() != param.pdg)
29  continue;
30  if (particle.Pt() < param.ptMin || particle.Pt() > param.ptMax)
31  continue;
32  if (particle.Eta() < param.etaMin || particle.Eta() > param.etaMax)
33  continue;
34  if (particle.Phi() < param.phiMin || particle.Phi() > param.phiMax)
35  continue;
36  if (particle.Y() < param.yMin || particle.Y() > param.yMax)
37  continue;
38  return true;
39  }
40  return false;
41  };
