34 GPUd() int GPUTPCGMPhysicalTrackModel::PropagateToXBzLightNoUpdate(float x, float Bz, float& GPUrestrict() dLp)
35 {
36  //
37  // transport the track to X=x in magnetic field B = ( 0, 0, Bz[kG*0.000299792458] )
38  // dLp is a return value == path length / track momentum [cm/(GeV/c)]
39  // the method returns error code (0 == no error)
40  //
41  // Additional values are not recalculated, UpdateValues() has to be called afterwards!!
42  //
43  float b = mQ * Bz;
44  float pt2 = mPx * mPx + mPy * mPy;
45  float dx = x - mX;
46  float pye = mPy - dx * b; // extrapolated py
47  float pxe2 = pt2 - pye * pye;
48 
49  if (mPx < (1.f - GPUCA_MAX_SIN_PHI) || pxe2 < (1.f - GPUCA_MAX_SIN_PHI) * (1.f - GPUCA_MAX_SIN_PHI)) {
50  return -1; // can not transport to x=x
51  }
52  float pxe = CAMath::Sqrt(pxe2); // extrapolated px
53  float pti = 1.f / CAMath::Sqrt(pt2);
54 
55  float ty = (mPy + pye) / (mPx + pxe);
56  float dy = dx * ty;
57  float dS; // path in XY
58  {
59  float chord = dx * CAMath::Sqrt(1.f + ty * ty); // chord to the extrapolated point == sqrt(dx^2+dy^2)*sign(dx)
60  float sa = 0.5f * chord * b * pti; // sin( half of the rotation angle ) == (chord/2) / radius
61 
62  // dS = (Pt/b)*2*arcsin( sa )
63  // = (Pt/b)*2*sa*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
64  // = chord*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
65 
66  float sa2 = sa * sa;
67  const float k2 = 1.f / 6.f;
68  const float k4 = 3.f / 40.f;
69  // const float k6 = 5.f/112.f;
70  dS = chord + chord * sa2 * (k2 + k4 * sa2);
71  // dS = sqrt(pt2)/b*2.*CAMath::ASin( sa );
72  }
73 
74  dLp = pti * dS; // path in XYZ / p == path in XY / pt
75 
76  float dz = mPz * dLp;
77 
78  mX = x;
79  mY += dy;
80  mZ += dz;
81  mPx = pxe;
82  mPy = pye;
83  // mPz = mPz;
84  // mQ = mQ;
85  return 0;
86 }

88 GPUd() int GPUTPCGMPhysicalTrackModel::PropagateToXBxByBz(float x, float Bx, float By, float Bz, float& GPUrestrict() dLp)
89 {
90  //
91  // transport the track to X=x in magnetic field B = ( Bx, By, Bz )[kG*0.000299792458]
92  // xyzPxPyPz as well as all the additional values will change. No need to call UpdateValues() afterwards.
93  // the method returns error code (0 == no error)
94  //
95 
96  if (0) { // simple transport in Bz for test proposes
97  return PropagateToXBzLight(x, Bz, dLp);
98  }
99  dLp = 0.f;
100 
101  GPUTPCGMPhysicalTrackModel t = *this;
102 
103  // Rotate to the system where Bx=By=0.
104 
105  float bt = CAMath::Sqrt(Bz * Bz + By * By);
106  float bb = CAMath::Sqrt(Bx * Bx + By * By + Bz * Bz);
107 
108  float c1 = 1.f, s1 = 0.f;
109  float c2 = 1.f, s2 = 0.f;
110 
111  if (bt > 1.e-4f) {
112  c1 = Bz / bt;
113  s1 = By / bt;
114  c2 = bt / bb;
115  s2 = -Bx / bb;
116  }
117 
118  // rotation matrix: first around x, then around y'
119  // after the first rotation: Bx'==Bx, By'==0, Bz'==Bt, X'==X
120  // after the second rotation: Bx''==0, By''==0, Bz''==B, X'' axis is as close as possible to the original X
121 
122  //
123  // ( c2 0 s2 ) ( 1 0 0 )
124  // R = ( 0 1 0 ) X ( 0 c1 -s1 )
125  // (-s2 0 c2 ) ( 0 s1 c1 )
126  //
127 
128  float R0[3] = {c2, s1 * s2, c1 * s2};
129  float R1[3] = {0, c1, -s1};
130  float R2[3] = {-s2, s1 * c2, c1 * c2};
131 
132  // parameters and the extrapolation point in the rotated coordinate system
133  {
134  float lx = t.X(), ly = t.Y(), lz = t.Z(), lpx = t.Px(), lpy = t.Py(), lpz = t.Pz();
135 
136  t.X() = R0[0] * lx + R0[1] * ly + R0[2] * lz;
137  t.Y() = R1[0] * lx + R1[1] * ly + R1[2] * lz;
138  t.Z() = R2[0] * lx + R2[1] * ly + R2[2] * lz;
139 
140  t.Px() = R0[0] * lpx + R0[1] * lpy + R0[2] * lpz;
141  t.Py() = R1[0] * lpx + R1[1] * lpy + R1[2] * lpz;
142  t.Pz() = R2[0] * lpx + R2[1] * lpy + R2[2] * lpz;
143  }
144 
145  float dx = x - mX;
146  float xe = t.X() + dx; // propagate on same dx in rotated system
147 
148  // transport in rotated coordinate system to X''=xe:
149 
150  if (t.Px() < (1.f - GPUCA_MAX_SIN_PHI)) {
151  t.Px() = 1.f - GPUCA_MAX_SIN_PHI;
152  }
153  if (t.PropagateToXBzLightNoUpdate(xe, bb, dLp) != 0) {
154  return -1;
155  }
156 
157  // rotate coordinate system back to the original R{-1}==R{T}
158  {
159  float lx = t.X(), ly = t.Y(), lz = t.Z(), lpx = t.Px(), lpy = t.Py(), lpz = t.Pz();
160 
161  t.X() = R0[0] * lx + R1[0] * ly + R2[0] * lz;
162  t.Y() = R0[1] * lx + R1[1] * ly + R2[1] * lz;
163  t.Z() = R0[2] * lx + R1[2] * ly + R2[2] * lz;
164 
165  t.Px() = R0[0] * lpx + R1[0] * lpy + R2[0] * lpz;
166  t.Py() = R0[1] * lpx + R1[1] * lpy + R2[1] * lpz;
167  t.Pz() = R0[2] * lpx + R1[2] * lpy + R2[2] * lpz;
168  }
169 
170  // a small (hopefully) additional step to X=x. Perhaps it may be replaced by linear extrapolation.
171 
172  float ddLp = 0;
173  if (t.Px() < (1.f - GPUCA_MAX_SIN_PHI)) {
174  t.Px() = 1.f - GPUCA_MAX_SIN_PHI;
175  }
176  if (t.PropagateToXBzLightNoUpdate(x, Bz, ddLp) != 0) {
177  return -1;
178  }
179 
180  dLp += ddLp;
181 
182  t.UpdateValues();
183  *this = t;
184  return 0;
185 }

187 GPUd() int GPUTPCGMPhysicalTrackModel::PropagateToLpBz(float Lp, float Bz)
188 {
189  // Lp is path length L over track momentum p in [cm/GeV], Bz in kG*clight
190  //
191  // it is a copy of AliExternalTrackParam: ghelix3 routine.
192  //
193  // the method returns error code (0 == no error)
194  //
195 
196  float qfield = mQ * Bz;
197 
198  float step = Lp;
199 
200  const float kOvSqSix = CAMath::Sqrt(1.f / 6.f);
201 
202  float px = mPx;
203  float py = mPy;
204  float pz = mPz;
205 
206  float tet = qfield * step;
207 
208  float tsint, sintt, sint, cos1t;
209  if (CAMath::Abs(tet) > 0.03f) {
210  sint = CAMath::Sin(tet);
211  sintt = sint / tet;
212  tsint = (tet - sint) / tet;
213  float t = CAMath::Sin(0.5f * tet);
214  cos1t = 2.f * t * t / tet;
215  } else {
216  tsint = tet * tet / 6.f;
217  sintt = (1.f - tet * kOvSqSix) * (1.f + tet * kOvSqSix); // 1.- tsint;
218  sint = tet * sintt;
219  cos1t = 0.5f * tet;
220  }
221 
222  float f1 = step * sintt;
223  float f2 = step * cos1t;
224  float f3 = step * tsint;
225  float f4 = -tet * cos1t;
226  float f5 = sint;
227 
228  mX += f1 * px - f2 * py;
229  mY += f1 * py + f2 * px;
230  mZ += f1 * pz + f3 * pz;
231 
232  mPx += f4 * px - f5 * py;
233  mPy += f4 * py + f5 * px;
234 
235  UpdateValues();
236 
237  return 0;
238 }
