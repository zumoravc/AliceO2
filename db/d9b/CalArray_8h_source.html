149 // ===| pad region etc. initialisation |========================================
150 template <class T>
151 void CalArray<T>::initData()
152 {
153  const auto& mapper = Mapper::instance();
154 
155  switch (mPadSubset) {
156  case PadSubset::ROC: {
157  mData.resize(ROC(mPadSubsetNumber).rocType() == RocType::IROC ? mapper.getPadsInIROC() : mapper.getPadsInOROC());
158  if (mName.empty()) {
159  setName(boost::str(format("ROC_%1$02d") % mPadSubsetNumber));
160  }
161  break;
162  }
163  case PadSubset::Partition: {
164  mData.resize(mapper.getPartitionInfo(mPadSubsetNumber % mapper.getNumberOfPartitions()).getNumberOfPads());
165  if (mName.empty()) {
166  setName(boost::str(format("Partition_%1$03d") % mPadSubsetNumber));
167  }
168  break;
169  }
170  case PadSubset::Region: {
171  mData.resize(mapper.getPadRegionInfo(mPadSubsetNumber % mapper.getNumberOfPadRegions()).getNumberOfPads());
172  if (mName.empty()) {
173  setName(boost::str(format("Region_%1$03d") % mPadSubsetNumber));
174  }
175  break;
176  }
177  }
178 }

202 {
203  if (!((mPadSubset == other.mPadSubset) && (mPadSubsetNumber == other.mPadSubsetNumber))) {
204  LOG(ERROR) << "You are trying to operate on incompatible objects: Pad subset type and number must be the same on both objects";
205  return *this;
206  }
207  for (size_t i = 0; i < mData.size(); ++i) {
208  mData[i] += other.getValue(i);
