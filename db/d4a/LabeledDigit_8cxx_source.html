28 LabeledDigit& LabeledDigit::operator+=(const LabeledDigit& other)
29 {
30  if (canAdd(other)) {
31  Int_t a1 = getAmplitudeADC();
32  Int_t a2 = other.getAmplitudeADC();
33  Double_t r = ((a1 + a2) != 0) ? 1.0 / (a1 + a2) : 0.0;
34  mDigit += other.getDigit();
35 
36  for (int j = 0; j < mLabels.size(); j++) {
37  mLabels.at(j).setAmplitudeFraction(mLabels.at(j).getAmplitudeFraction() * a1 * r);
38  }
39 
40  for (auto label : other.getLabels()) {
41  label.setAmplitudeFraction(label.getAmplitudeFraction() * a2 * r);
42  mLabels.push_back(label);
43  }
44  }
45  // Does nothing if the digits are in different towers or have incompatible times.
46  return *this;
47 }
