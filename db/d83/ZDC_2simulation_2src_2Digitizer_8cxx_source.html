43 void Digitizer::process(const std::vector<o2::zdc::Hit>& hits,
44  std::vector<o2::zdc::BCData>& digitsBC,
45  std::vector<o2::zdc::ChannelData>& digitsCh,
46  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
47 {
48  // loop over all hits and produce digits
49  LOG(DEBUG) << "Processing IR = " << mIR << " | NHits = " << hits.size();
50 
51  flush(digitsBC, digitsCh, labels); // flush cached signal which cannot be affect by new event
52 
53  for (auto& hit : hits) {
54 
55  std::array<o2::InteractionRecord, NBC2Cache> cachedIR;
56  // hit of given IR can con
57  // for each hit find out sector + detector information
58  int detID = hit.GetDetectorID();
59  int secID = hit.getSector();
60  float nPhotons;
61  if (detID == ZEM) { // TODO: ZEMCh1 and Common are both 0, could skip the check for detID
62  nPhotons = (secID == ZEMCh1) ? hit.getPMCLightYield() : hit.getPMQLightYield();
63  } else {
64  nPhotons = (secID == Common) ? hit.getPMCLightYield() : hit.getPMQLightYield();
65  }
66  if (!nPhotons) {
67  continue;
68  }
69  if (nPhotons < 0 || nPhotons > 1e6) {
70  int chan = toChannel(detID, secID);
71  LOG(ERROR) << "Anomalous number of photons " << nPhotons << " for channel " << chan << '(' << channelName(chan) << ')';
72  continue;
73  }
74 
75  double hTime = hit.GetTime() - getTOFCorrection(detID); // account for TOF to detector
76  hTime += mIR.timeNS;
77  //
78  o2::InteractionRecord irHit(hTime); // BC in which the hit appears (might be different from interaction BC for slow particles)
79 
80  // nominal time of the BC to which the hit will be attributed
81  double bcTime = o2::InteractionRecord::bc2ns(irHit.bc, irHit.orbit);
82 
83  int nCachedIR = 0;
84  for (int i = BCCacheMin; i < BCCacheMax + 1; i++) {
85  double tNS = hTime + o2::constants::lhc::LHCBunchSpacingNS * i;
86  cachedIR[nCachedIR].setFromNS(tNS);
87  if (tNS < 0 && cachedIR[nCachedIR] > irHit) {
88  continue; // don't go to negative BC/orbit (it will wrap)
89  }
90  getCreateBCCache(cachedIR[nCachedIR++]); // ensure existence of cached container
91  }
92  auto channel = toChannel(detID, secID);
93  phe2Sample(nPhotons, hit.getParentID(), hTime, cachedIR, nCachedIR, channel);
94  // if digit for this sector does not exist, create one otherwise add to it
95  }
96 }

98 void Digitizer::flush(std::vector<o2::zdc::BCData>& digitsBC,
99  std::vector<o2::zdc::ChannelData>& digitsCh,
100  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
101 {
102  // do we have something to flush? We can do this only for cached BC data which is distanced from currently processed BC by NBCReadOut
103  int lastDoneBCid = -1, diff2Last = 0;
104  int nCached = mCache.size();
105  if (nCached < 1) {
106  return;
107  }
108  if (mIR.differenceInBC(mCache.back()) > -BCCacheMin) {
109  LOG(DEBUG) << "Generating new pedestal BL fluct. for BC range " << mCache.front() << " : " << mCache.back();
110  generatePedestal();
111  } else {
112  return;
113  }
114  o2::InteractionRecord ir0(mCache.front());
115  int cacheSpan = 1 + mCache.back().differenceInBC(ir0);
116  LOG(DEBUG) << "Cache spans " << cacheSpan << " with " << nCached << " BCs cached";
117 
118  mFastCache.clear();
119  mFastCache.resize(cacheSpan, nullptr);
120  mStoreChanMask.clear();
121  mStoreChanMask.resize(cacheSpan + mNBCAHead, 0);
122 
123  for (int ibc = nCached; ibc--;) { // digitize BCs which might not be affected by future events
124  auto& bc = mCache[ibc];
125  lastDoneBCid = ibc;
126  if (!bc.digitized) {
127  digitizeBC(bc);
128  }
129  int bcSlot = mCache[ibc].differenceInBC(ir0);
130  mFastCache[bcSlot] = &mCache[ibc]; // add to fast access cache
131  }
132  mDummyBC.clear();
133  digitizeBC(mDummyBC);
134 
135  // check trigger condition for digitized BCs
136  for (int ibc = 0; ibc < cacheSpan; ibc++) {
137  if (mFastCache[ibc] && !mFastCache[ibc]->triggerChecked) {
138  triggerBC(ibc);
139  }
140  } // all allowed BCs are checked for trigger
141 
142  // store triggered BC with requested few BCs ahead
143  for (int ibc = -mNBCAHead; ibc < cacheSpan; ibc++) {
144  auto bcr = mStoreChanMask[ibc + mNBCAHead];
145  if (!bcr) {
146  continue; // nothing to store for this BC
147  }
148  BCCache* bcPtr = nullptr;
149  if (ibc < 0 || !mFastCache[ibc]) {
150  mDummyBC = ir0 + ibc; // fix the IR of the dummy BC
151  bcPtr = &mDummyBC;
152  } else {
153  bcPtr = mFastCache[ibc];
154  }
155  storeBC(*bcPtr, bcr, digitsBC, digitsCh, labels);
156  } // all allowed BCs are checked for trigger
157 
158  // clean cache for BCs which are not needed anymore
159  LOG(DEBUG) << "Cleaning cache";
160  mCache.erase(mCache.begin(), mCache.end());
161 }

163 void Digitizer::generatePedestal()
164 {
165  for (int idet : {ZNA, ZPA, ZNC, ZPC}) {
166  int chanSum = toChannel(idet, Sum);
167  mPedestalBLFluct[chanSum] = 0.;
168  int comm = toChannel(idet, Common);
169  mPedestalBLFluct[comm] = gRandom->Gaus(0, mSimCondition->channels[comm].pedestalFluct);
170  for (int ic : {Ch1, Ch2, Ch3, Ch4}) {
171  int chan = toChannel(idet, ic);
172  mPedestalBLFluct[chanSum] += mPedestalBLFluct[chan] = gRandom->Gaus(0, mSimCondition->channels[chan].pedestalFluct);
173  }
174  }
175  mPedestalBLFluct[IdZEM1] = gRandom->Gaus(0, mSimCondition->channels[IdZEM1].pedestalFluct);
176  mPedestalBLFluct[IdZEM2] = gRandom->Gaus(0, mSimCondition->channels[IdZEM2].pedestalFluct);
177 }

179 void Digitizer::digitizeBC(BCCache& bc)
180 {
181  auto& bcdata = bc.data;
182  // apply gain
183  for (int idet : {ZNA, ZPA, ZNC, ZPC}) {
184  for (int ic : {Ch1, Ch2, Ch3, Ch4}) {
185  int chan = toChannel(idet, ic);
186  auto gain = mSimCondition->channels[chan].gain;
187  for (int ib = NTimeBinsPerBC; ib--;) {
188  bcdata[chan][ib] *= gain;
189  }
190  }
191  }
192  for (int ib = NTimeBinsPerBC; ib--;) {
193  bcdata[IdZEM1][ib] *= mSimCondition->channels[IdZEM1].gain;
194  bcdata[IdZEM2][ib] *= mSimCondition->channels[IdZEM2].gain;
195  }
196  //
197  for (int ib = NTimeBinsPerBC; ib--;) {
198  bcdata[IdZNASum][ib] = mSimCondition->channels[IdZNASum].gain *
199  (bcdata[IdZNA1][ib] + bcdata[IdZNA2][ib] + bcdata[IdZNA3][ib] + bcdata[IdZNA4][ib]);
200  bcdata[IdZPASum][ib] = mSimCondition->channels[IdZPASum].gain *
201  (bcdata[IdZPA1][ib] + bcdata[IdZPA2][ib] + bcdata[IdZPA3][ib] + bcdata[IdZPA4][ib]);
202  bcdata[IdZNCSum][ib] = mSimCondition->channels[IdZNCSum].gain *
203  (bcdata[IdZNC1][ib] + bcdata[IdZNC2][ib] + bcdata[IdZNC3][ib] + bcdata[IdZNC4][ib]);
204  bcdata[IdZPCSum][ib] = mSimCondition->channels[IdZPCSum].gain *
205  (bcdata[IdZPC1][ib] + bcdata[IdZPC2][ib] + bcdata[IdZPC3][ib] + bcdata[IdZPC4][ib]);
206  }
207  for (int chan = NChannels; chan--;) {
208  const auto& chanConf = mSimCondition->channels[chan];
209  auto pedBaseLine = mSimCondition->channels[chan].pedestal;
210  for (int ib = NTimeBinsPerBC; ib--;) {
211  bcdata[chan][ib] += gRandom->Gaus(pedBaseLine + mPedestalBLFluct[chan], chanConf.pedestalNoise);
212  int adc = std::nearbyint(bcdata[chan][ib]);
213  bcdata[chan][ib] = adc < ADCMax ? (adc > ADCMin ? adc : ADCMin) : ADCMax;
214  }
215  }
216  bc.digitized = true;
217 }

219 bool Digitizer::triggerBC(int ibc)
220 {
221  // check trigger for the cached BC in the position ibc
222  auto& bcCached = *mFastCache[ibc];
223 
224  LOG(DEBUG) << "CHECK TRIGGER " << ibc << " IR=" << bcCached;
225  if (mIsContinuous) {
226  for (int ic = mTriggerConfig.size(); ic--;) {
227  const auto& trigCh = mTriggerConfig[ic];
228  bool okPrev = false;
229  int last1 = trigCh.last + 2;
230  // look for 2 consecutive bins (the 1st one spanning trigCh.first : trigCh.last range) so that
231  // signal[bin]-signal[bin+trigCh.shift] > trigCh.threshold
232  for (int ib = trigCh.first; ib < last1; ib++) { // ib may be negative, so we shift by offs and look in the ADC cache
233  int binF, bcFidx = ibc + binHelper(ib, binF);
234  int binL, bcLidx = ibc + binHelper(ib + trigCh.shift, binL);
235  const auto& bcF = (bcFidx < 0 || !mFastCache[bcFidx]) ? mDummyBC : *mFastCache[bcFidx];
236  const auto& bcL = (bcLidx < 0 || !mFastCache[bcLidx]) ? mDummyBC : *mFastCache[bcLidx];
237  bool ok = bcF.data[trigCh.id][binF] - bcL.data[trigCh.id][binL] > trigCh.threshold;
238  if (ok && okPrev) { // trigger ok!
239  bcCached.trigChanMask |= 0x1 << trigCh.id; // register trigger mask
240  LOG(DEBUG) << " triggering channel " << int(trigCh.id) << " => " << bcCached.trigChanMask;
241  break;
242  }
243  okPrev = ok;
244  }
245  } // loop over trigger channels
246  } else {
247  // just check if this BC IR corresponds to externall trigger
248  if (!mIRExternalTrigger.empty() && mIRExternalTrigger.front() == bcCached) {
249  bcCached.trigChanMask = AllChannelsMask;
250  mIRExternalTrigger.pop_front(); // suppress accounted external trigger
251  }
252  }
253 
254  if (bcCached.trigChanMask) { // there are triggered channels, flag modules/channels to read
255  for (int ibcr = ibc - mNBCAHead; ibcr <= ibc; ibcr++) {
256  auto& bcr = mStoreChanMask[ibcr + mNBCAHead];
257  for (const auto& mdh : mModConfAux) {
258  if (bcCached.trigChanMask & mdh.trigChannels) { // are there triggered channels in this module?
259  bcr |= mdh.readChannels; // flag channels to store
260  }
261  }
262  }
263  }
264  bcCached.triggerChecked = true;
265  return bcCached.trigChanMask;
266 }

269  std::vector<o2::zdc::BCData>& digitsBC, std::vector<o2::zdc::ChannelData>& digitsCh,
270  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
271 {
272  // store selected data of selected BC
273  if (!chan2Store) {

275  }

295 void Digitizer::phe2Sample(int nphe, int parID, double timeHit, std::array<o2::InteractionRecord, NBC2Cache> const& cachedIR, int nCachedIR, int channel)
296 {
297  //function to simulate the waveform from no. of photoelectrons seen in a given sample
298  // for electrons at timeInSample wrt beginning of the sample
299 
300  double time0 = cachedIR[0].bc2ns(); // start time of the 1st cashed BC
301  const auto& chanConfig = mSimCondition->channels[channel];
302 
303  float timeDiff = time0 - timeHit;
304  int sample = (timeDiff - gRandom->Gaus(chanConfig.timePosition, chanConfig.timeJitter)) * ChannelSimCondition::ShapeBinWidthInv + chanConfig.ampMinID;
305  int ir = 0;
306  bool stop = false;
307 
308  do {
309  auto bcCache = getBCCache(cachedIR[ir]);
310  bool added = false;
311  for (int ib = 0; ib < NTimeBinsPerBC; ib++) {
312  if (sample >= chanConfig.shape.size()) {
313  stop = true;
314  break;
315  }
316  if (sample >= 0) {
317  auto signal = chanConfig.shape[sample] * nphe; // signal accounting for the gain
318  (*bcCache).data[channel][ib] += signal;
319  added = true;
320  }
321  sample += o2::constants::lhc::LHCBunchSpacingNS / NTimeBinsPerBC * ChannelSimCondition::ShapeBinWidthInv; // sample increment for next bin
322  }
323  if (added) {
324  (*bcCache).labels.emplace_back(parID, mEventID, mSrcID, channel);
325  }
326  } while (++ir < nCachedIR && !stop);
327 }

329 o2::zdc::Digitizer::BCCache& Digitizer::getCreateBCCache(const o2::InteractionRecord& ir)
330 {
331  if (mCache.empty() || mCache.back() < ir) {
332  mCache.emplace_back();
333  auto& cb = mCache.back();
334  cb = ir;
335  return cb;
336  }
337  if (mCache.front() > ir) {
338  mCache.emplace_front();
339  auto& cb = mCache.front();
340  cb = ir;
341  return cb;
342  }
343 
344  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
345  if ((*cb) == ir) {
346  return *cb;
347  }
348  if (ir < (*cb)) {
349  auto cbnew = mCache.emplace(cb); // insert new element before cb
350  (*cbnew) = ir;
351  return (*cbnew);
352  }
353  }
354  return mCache.front();
355 }

383 void Digitizer::refreshCCDB()
384 {
385  // fetch ccdb objects. TODO: decide if this stays here or goes to the Spec
386  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
387  if (mTimeStamp == mgr.getTimestamp()) {
388  return;
389  }
390  mgr.setTimestamp(mTimeStamp);
391 
392  if (!mModuleConfig) { // load this only once
393  mModuleConfig = mgr.get<ModuleConfig>(CCDBPathConfigModule);
394  LOG(INFO) << "Loaded module configuration for timestamp " << mTimeStamp;
395  // fetch trigger info
396  mTriggerConfig.clear();
397  mModConfAux.clear();
398  for (const auto& md : mModuleConfig->modules) {
399  if (md.id >= 0) {
400  mModConfAux.emplace_back(md);
401  //
402  for (int ic = Module::MaxChannels; ic--;) {
403  if (md.trigChannel[ic]) { // check if this triggering channel was already registered
404  bool skip = false;
405  for (int is = mTriggerConfig.size(); is--;) {
406  if (mTriggerConfig[is].id == md.channelID[ic]) {
407  skip = true;
408  break;
409  }
410  }
411  if (!skip) {
412  const auto& trgChanConf = md.trigChannelConf[ic];
413  if (trgChanConf.last + trgChanConf.shift + 1 >= NTimeBinsPerBC) {
414  LOG(FATAL) << "Wrong trigger settings";
415  }
416  mTriggerConfig.emplace_back(trgChanConf);
417  LOG(INFO) << "Adding channel " << int(trgChanConf.id) << '(' << channelName(trgChanConf.id) << ") as triggering one";
418  if (trgChanConf.first < mTrigBinMin) {
419  mTrigBinMin = trgChanConf.first;
420  }
421  if (trgChanConf.last + trgChanConf.shift > mTrigBinMax) {
422  mTrigBinMax = trgChanConf.last + trgChanConf.shift;
423  }
424  }
425  }
426  }
427  }
428  }
429  if (int(mTriggerConfig.size()) > MaxTriggerChannels) {
430  LOG(FATAL) << "Too many triggering channels (" << mTriggerConfig.size() << ')';
431  }
432  mModuleConfig->print();
433  //
434  }
435 
436  if (!mSimCondition) { // load this only once
437  mSimCondition = mgr.get<SimCondition>(CCDBPathConfigSim);
438  LOG(INFO) << "Loaded simulation configuration for timestamp " << mTimeStamp;
439  mSimCondition->print();
440  }
