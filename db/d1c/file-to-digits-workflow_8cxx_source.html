106  void decodeBuffer(gsl::span<const std::byte> page, std::vector<o2::mch::Digit>& digits)
107  {
108  size_t ndigits{0};
109 
110  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
111  auto s = asString(dsElecId);
112  channel = ds2manu(int(channel));
113  if (mPrint) {
114  auto ch = fmt::format("{}-CH{} samples={}", s, channel, sc.samples.size());
115  std::cout << ch << std::endl;
116  }
117  double digitadc(0);
118  //for (auto d = 0; d < sc.nofSamples(); d++) {
119  for (auto d = 0; d < sc.samples.size(); d++) {
120  digitadc += sc.samples[d];
121  }
122 
123  int deId = -1;
124  int dsIddet = -1;
125  if (auto opt = Elec2Det(dsElecId); opt.has_value()) {
126  DsDetId dsDetId = opt.value();
127  dsIddet = dsDetId.dsId();
128  deId = dsDetId.deId();
129  }
130  if (dsIddet < 0 || deId < 0) {
131  std::cout << "SOLAR " << (int)dsElecId.solarId()
132  << " DS " << (int)dsElecId.elinkId() << " (" << (int)dsElecId.elinkGroupId() << "," << (int)dsElecId.elinkIndexInGroup() << ")"
133  << " CHIP " << ((int)channel) / 32 << " CH " << ((int)channel) % 32 << " ADC " << digitadc << " DE# " << deId << " DSid " << dsIddet << std::endl;
134  return;
135  }
136 
137  int padId = -1;
138  try {
139  const Segmentation& segment = segmentation(deId);
140  //Segmentation segment(deId);
141 
142  padId = segment.findPadByFEE(dsIddet, int(channel));
143  if (mPrint)
144  std::cout << "DS " << (int)dsElecId.elinkId() << " CHIP " << ((int)channel) / 32 << " CH " << ((int)channel) % 32 << " ADC " << digitadc << " DE# " << deId << " DSid " << dsIddet << " PadId " << padId << std::endl;
145  } catch (const std::exception& e) {
146  return;
147  }
148 
149  o2::mch::Digit::Time time;
150 
151  digits.emplace_back(o2::mch::Digit(deId, padId, digitadc, time));
152  //o2::mch::Digit& mchdigit = digits.back();
153  //mchdigit.setDetID(deId);
154  //mchdigit.setPadID(padId);
155  //mchdigit.setADC(digitadc);
156  //mchdigit.setTimeStamp(time);
157 
158  if (mPrint)
159  std::cout << "DIGIT STORED:\nADC " << digits.back().getADC() << " DE# " << digits.back().getDetID() << " PadId " << digits.back().getPadID() << " time " << digits.back().getTime().sampaTime << std::endl;
160  ++ndigits;
161  };
162 
163  const auto patchPage = [&](gsl::span<const std::byte> rdhBuffer) {
164  auto rdhPtr = const_cast<void*>(reinterpret_cast<const void*>(rdhBuffer.data()));
165  nrdhs++;
166  auto cruId = o2::raw::RDHUtils::getCRUID(rdhPtr);
167  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhPtr);
168  o2::raw::RDHUtils::setFEEID(rdhPtr, cruId * 2 + endpoint);
169  if (mPrint) {
170  std::cout << nrdhs << "--\n";
171  o2::raw::RDHUtils::printRDH(rdhPtr);
172  }
173  };
174 
175  patchPage(page);
176  if (!decoder.has_value())
177  decoder = o2::mch::raw::createPageDecoder(page, channelHandler);
178  decoder.value()(page);
179  }

219  void run(framework::ProcessingContext& pc)
220  {
221  std::vector<o2::mch::Digit> digits;
222 
223  uint32_t CRUbuf[4 * 4];
224  CRUheader CRUh;
226 
227  int RDH_BLOCK_SIZE = 8192;
228 
229  if (mFrameMax == 0)
230  return;
231  //printf("mFrameMax: %d\n", mFrameMax);
232  if (mFrameMax > 0)
233  mFrameMax -= 1;
234 
235  mInputFile.read((char*)(&CRUbuf), sizeof(CRUbuf));
236  memcpy(&CRUh, CRUbuf, sizeof(CRUheader));
237  if (CRUh.header_version != 4 || CRUh.header_size != 64)
238  return;
239 
240  RDH_BLOCK_SIZE = CRUh.next_packet_offset;
241 
242  char* buf = (char*)malloc(RDH_BLOCK_SIZE);
243  memcpy(buf, CRUbuf, CRUh.header_size);
244 
245  mInputFile.read(buf + CRUh.header_size, RDH_BLOCK_SIZE - CRUh.header_size);
246  if (mInputFile.fail()) {
247  if (mPrint) {
248  LOG(INFO) << "end of file reached";
249  }
250  free(buf);
251  pc.services().get<ControlService>().endOfStream();
252  return; // probably reached eof
253  }
254 
255  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(buf), RDH_BLOCK_SIZE);
256  decodeBuffer(buffer, digits);
257 
258  if (mPrint) {
259  for (auto d : digits) {
260  std::cout << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC() << " time " << d.getTime().sampaTime << std::endl;
261  }
262  }
263 
264  const size_t OUT_SIZE = sizeof(o2::mch::Digit) * digits.size();
265 
267  char* outbuffer = nullptr;
268  outbuffer = (char*)realloc(outbuffer, OUT_SIZE);
269  memcpy(outbuffer, digits.data(), OUT_SIZE);
270 
271  // create the output message
272  auto freefct = [](void* data, void*) { free(data); };
273  pc.outputs().adoptChunk(Output{"MCH", "DIGITS", 0}, outbuffer, OUT_SIZE, freefct, nullptr);
274  }
