29 void BunchFilling::setBCTrain(int nBC, int bcSpacing, int firstBC)
30 {
31  // add interacting BC train with given spacing starting at given place, i.e.
32  // train with 25ns spacing should have bcSpacing = 1
33  for (int i = 0; i < nBC; i++) {
34  setBC(firstBC);
35  firstBC += bcSpacing;
36  }
37 }

40 void BunchFilling::setBCTrains(int nTrains, int trainSpacingInBC, int nBC, int bcSpacing, int firstBC)
41 {
42  // add nTrains trains of interacting BCs with bcSpacing within the train and trainSpacingInBC empty slots
43  // between the trains
44  for (int it = 0; it < nTrains; it++) {
45  setBCTrain(nBC, bcSpacing, firstBC);
46  firstBC += nBC * bcSpacing + trainSpacingInBC;
47  }
48 }
