109 template <typename CHARGESUM>
110 void UserLogicElinkDecoder<CHARGESUM>::append(uint64_t data50, uint8_t error)
111 {
112 #ifdef ULDEBUG
113  debugHeader() << (*this) << fmt::format(" --> append50 {:013x} error {} data10={:d} {:d} {:d} {:d}\n", data50, error, static_cast<uint10_t>(data50 & 0x3FF), static_cast<uint10_t>((data50 & 0xFFC00) >> 10), static_cast<uint10_t>((data50 & 0x3FF00000) >> 20), static_cast<uint10_t>((data50 & 0xFFC0000000) >> 30), static_cast<uint10_t>((data50 & 0x3FF0000000000) >> 40));
114 #endif
115 
116  if (isSync(data50)) {
117  clear();
118  transition(State::WaitingHeader);
119  return;
120  }
121 
122  auto data = data50;
123 
124  for (auto i = 0; i < 5; i++) {
125  append10(static_cast<uint10_t>(data & 0x3FF));
126  data >>= 10;
127  if (hasError()) {
128 #ifdef ULDEBUG
129  debugHeader() << (*this) << " reset due to hasError\n";
130 #endif
131  reset();
132  break;
133  }
134  if (mState == State::WaitingHeader && isIncomplete(error)) {
135 #ifdef ULDEBUG
136  debugHeader() << (*this) << " reset due to isIncomplete\n";
137 #endif
138  reset();
139  return;
140  }
141  }

157 template <typename CHARGESUM>
158 void UserLogicElinkDecoder<CHARGESUM>::append10(uint10_t data10)
159 {
160 #ifdef ULDEBUG
161  debugHeader() << (*this) << fmt::format(" --> data10 {:d}\n", data10);
162 #endif
163  switch (mState) {
164  case State::WaitingHeader:
165  setHeaderPart(data10);
166  if (isHeaderComplete()) {
167  completeHeader();
168  if (isSync(mSampaHeader.uint64()) ||
169  mSampaHeader.packetType() == SampaPacketType::HeartBeat ||
170  mSampaHeader.nof10BitWords() == 0) {
171  reset();
172  } else {
173  transition(State::WaitingSize);
174  }
175  }
176  break;
177  case State::WaitingSize:
178  if (moreWordsToRead()) {
179  auto factor = DataFormatSizeFactor<CHARGESUM>::value;
180  auto value = factor * data10;
181  setClusterSize(value);
182  if (hasError()) {
183  return;
184  }
185  transition(State::WaitingTime);
186  } else {
187  mErrorMessage = "WaitingSize but no more words";
188  return;
189  }
190  break;
191  case State::WaitingTime:
192  if (moreWordsToRead()) {
193  setClusterTime(data10);
194  transition(State::WaitingSample);
195  } else {
196  mErrorMessage = "WaitingTime but no more words";
197  return;
198  }
199  break;
200  case State::WaitingSample:
201  if (moreSampleToRead()) {
202  setSample(data10);
203  } else if (moreWordsToRead()) {
204  transition(State::WaitingSize);
205  append10(data10);
206  } else {
207  transition(State::WaitingHeader);
208  append10(data10);
209  }
210  break;
211  default:
212  break;
213  };
214 }

248 template <typename CHARGESUM>
249 void UserLogicElinkDecoder<CHARGESUM>::completeHeader()
250 {
251  uint64_t header{0};
252  for (auto i = 0; i < mHeaderParts.size(); i++) {
253  header += (static_cast<uint64_t>(mHeaderParts[i]) << (10 * i));
254  }
255 
256  mSampaHeader = SampaHeader(header);
257  mNof10BitWords = mSampaHeader.nof10BitWords();
258 
259 #ifdef ULDEBUG
260  debugHeader() << (*this) << fmt::format(" --> completeHeader {:013X}\n", header);
261  debugHeader() << "\n";
262  std::stringstream s(o2::mch::raw::asString(mSampaHeader));
263  std::string part;
264  while (std::getline(s, part, '\n')) {
265  debugHeader() << (*this) << part << "\n";
266  }
267  debugHeader() << "\n";
268 #endif
269 
270  mHeaderParts.clear();
271 }

383 template <>
384 void UserLogicElinkDecoder<ChargeSumMode>::prepareAndSendCluster()
385 {
386  if (mSamples.size() != 2) {
387  throw std::invalid_argument(fmt::format("expected sample size to be 2 but it is {}", mSamples.size()));
388  }
389  uint32_t q = (((static_cast<uint32_t>(mSamples[1]) & 0x3FF) << 10) | (static_cast<uint32_t>(mSamples[0]) & 0x3FF));
390  SampaCluster sc(mClusterTime, mSampaHeader.bunchCrossingCounter(), q);
391  sendCluster(sc);
392  mSamples.clear();
