29 {
31  mL += dL;
32  float p2 = track.getP2Inv();
33  float dTns = dL * 1000.f / o2::constants::physics::LightSpeedCm2NS; // time change in ps for beta = 1 particle

36  float betaInv = std::sqrt(1.f + m2z * m2z * p2);

38  }
