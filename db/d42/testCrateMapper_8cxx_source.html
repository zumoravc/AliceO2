75 BOOST_AUTO_TEST_CASE(Consistency)
76 {
77  o2::mid::Mapping mapping;

79  for (int ide = 0; ide < o2::mid::detparams::NDetectionElements; ++ide) {
80  for (int icol = mapping.getFirstColumn(ide); icol < 7; ++icol) {
81  for (int iline = mapping.getFirstBoardBP(icol, ide); iline <= mapping.getLastBoardBP(icol, ide); ++iline) {
82  auto uniqueLocId = crateMapper.deLocalBoardToRO(ide, icol, iline);
83  auto crateId = o2::mid::crateparams::getCrateId(uniqueLocId);
84  auto deBoardId = crateMapper.roLocalBoardToDE(crateId, o2::mid::crateparams::getLocId(uniqueLocId));
85  BOOST_TEST(static_cast<int>(crateMapper.getColumnId(deBoardId)) == icol);
86  BOOST_TEST(static_cast<int>(crateMapper.getLineId(deBoardId)) == iline);
87  int rpcLineId = crateMapper.getRPCLine(deBoardId);
88  BOOST_TEST(rpcLineId == o2::mid::detparams::getRPCLine(ide));
89  int ich = o2::mid::detparams::getChamber(ide);
90  BOOST_TEST(o2::mid::detparams::getDEId(o2::mid::crateparams::isRightSide(crateId), ich, rpcLineId) == ide);
