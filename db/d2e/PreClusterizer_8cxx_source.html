81 bool PreClusterizer::loadPatterns(gsl::span<const ColumnData>& stripPatterns)
82 {
84 
85  // Loop on stripPatterns
86  for (auto& col : stripPatterns) {
87  int deIndex = col.deId;
88  assert(deIndex < detparams::NDetectionElements);
89 
90  auto search = mMpDEs.find(deIndex);
91  PatternStruct* de = nullptr;
92  if (search == mMpDEs.end()) {
93  de = &mMpDEs[deIndex];
94  de->deId = col.deId;
95  } else {
96  de = &(search->second);
97  }
98 
99  mActiveDEs[deIndex] = true;
100 
101  de->firedColumns |= (1 << col.columnId);
102  de->columns[col.columnId] = col;
103  }
104 
105  return (stripPatterns.size() > 0);
106 }

109 void PreClusterizer::preClusterizeNBP(PatternStruct& de)
110 {
112  PreCluster* pc = nullptr;
113  for (int icolumn = 0; icolumn < 7; ++icolumn) {
114  if (de.columns[icolumn].getNonBendPattern() == 0) {
115  continue;
116  }
117  int nStripsNBP = mMapping.getNStripsNBP(icolumn, de.deId);
118  for (int istrip = 0; istrip < nStripsNBP; ++istrip) {
119  if (de.columns[icolumn].isNBPStripFired(istrip)) {
120  if (!pc) {
121  mPreClusters.push_back({static_cast<uint8_t>(de.deId), 1, static_cast<uint8_t>(icolumn), static_cast<uint8_t>(icolumn), 0, 0, static_cast<uint8_t>(istrip), static_cast<uint8_t>(istrip)});
122  pc = &mPreClusters.back();
123  }
124  pc->lastColumn = icolumn;
125  pc->lastStrip = istrip;
126  } else {
127  pc = nullptr;
128  }
129  }
130  de.columns[icolumn].setNonBendPattern(0); // Reset pattern
131  }
132 }

135 void PreClusterizer::preClusterizeBP(PatternStruct& de)
136 {
138  for (int icolumn = mMapping.getFirstColumn(de.deId); icolumn < 7; ++icolumn) {
139  if ((de.firedColumns & (1 << icolumn)) == 0) {
140  continue;
141  }
142  PreCluster* pc = nullptr;
143  int firstLine = mMapping.getFirstBoardBP(icolumn, de.deId);
144  int lastLine = mMapping.getLastBoardBP(icolumn, de.deId);
145  for (int iline = firstLine; iline <= lastLine; ++iline) {
146  if (de.columns[icolumn].getBendPattern(iline) == 0) {
147  continue;
148  }
149  for (int istrip = 0; istrip < 16; ++istrip) {
150  if (de.columns[icolumn].isBPStripFired(istrip, iline)) {
151  if (!pc) {
152  mPreClusters.push_back({static_cast<uint8_t>(de.deId), 0, static_cast<uint8_t>(icolumn), static_cast<uint8_t>(icolumn), static_cast<uint8_t>(iline), static_cast<uint8_t>(iline), static_cast<uint8_t>(istrip), static_cast<uint8_t>(istrip)});
153  pc = &mPreClusters.back();
154  }
155  pc->lastLine = iline;
156  pc->lastStrip = istrip;
157  } else {
158  pc = nullptr;
159  }
160  }
161  de.columns[icolumn].setBendPattern(0, iline); // Reset pattern
162 
163  } // loop on lines
164  } // loop on columns
165 }
