24 void checkResults(const DCAFitter& fitter, const std::array<float, 3> xyz)
25 {
26  int nCand = fitter.getNCandidates();
27  BOOST_CHECK(nCand > 0);
28  for (int ic = 0; ic < nCand; ic++) {
29  const auto& vtx = fitter.getPCACandidate(ic);
30  float dx = vtx.x - xyz[0], dy = vtx.y - xyz[1], dz = vtx.z - xyz[2];
31  float dst = TMath::Sqrt(dx * dx + dy * dy + dz * dz);
32 
33  const auto &trc0 = fitter.getTrack0(ic), &trc1 = fitter.getTrack1(ic); // track parameters at V0
34  printf("Candidate %d: DCA:%+e Vtx: %+e %+e %+e [Diff to true: %+e %+e %+e -> %+e]\n",
35  ic, fitter.getChi2AtPCACandidate(ic), vtx.x, vtx.y, vtx.z, dx, dy, dz, dst);
36  printf("Track X-parameters at PCA: %+e %+e\n", trc0.getX(), trc1.getX());
37  trc0.print();
38  trc1.print();
39  BOOST_CHECK(dst < 5e-2);
40  }
41 }
