64 template <class Mapping>
65 int RUDecodeData::decodeROF(const Mapping& mp)
66 {
67  nChipsFired = 0;
68  lastChipChecked = 0;
69  int ntot = 0;
70  auto* chipData = &chipsData[0];
71  for (int icab = 0; icab < nCables; icab++) {
72  if (!cableData[icab].getSize()) {
73  continue;
74  }
75  int nhits = 0;
76  while ((nhits = AlpideCoder::decodeChip(*chipData, cableData[icab]))) { // we register only chips with hits or errors flags set
77  if (nhits > 0) {
78  // convert HW chip id within the module to absolute chip id
79  chipData->setChipID(mp.getGlobalChipID(chipData->getChipID(), cableHWID[icab], *ruInfo));
80  setROFInfo(chipData, cableLinkPtr[icab]);
81  ntot += nhits;
82  if (++nChipsFired < chipsData.size()) { // fetch next free chip
83  chipData = &chipsData[nChipsFired];
84  } else {
85  break; // last chip decoded
86  }
87  }
88  }
89  }
90 
91  return ntot;
92 }
