28 ObjectStore extractObjectFrom(const framework::DataRef& ref)
29 {
30  // We do extraction on the low level to efficiently determine if the message
31  // contains an object inheriting MergeInterface or TObject. If we did it the
32  // the following way and catch an exception:
33  // framework::DataRefUtils::as<MergeInterface>(ref)
34  // it could cause a memory leak if `ref` contained a non-owning TCollection.
35  // This way we also avoid doing most of the checks twice.
36  const static std::string errorPrefix = "Could not extract object to be merged: ";
37 
38  using DataHeader = o2::header::DataHeader;
39  auto header = o2::header::get<const DataHeader*>(ref.header);
40  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
41  throw std::runtime_error(errorPrefix + "It is not ROOT-serialized");
42  }
43 
44  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
45  auto* storedClass = ftm.GetClass();
46  if (storedClass == nullptr) {
47  throw std::runtime_error(errorPrefix + "Unknown stored class");
48  }
49 
50  auto* mergeInterfaceClass = TClass::GetClass(typeid(MergeInterface));
51  auto* tObjectClass = TClass::GetClass(typeid(TObject));
52 
53  bool inheritsFromMergeInterface = storedClass->InheritsFrom(mergeInterfaceClass);
54  bool inheritsFromTObject = storedClass->InheritsFrom(tObjectClass);
55 
56  if (!inheritsFromMergeInterface && !inheritsFromTObject) {
57  throw std::runtime_error(
58  errorPrefix + "Class '" + storedClass->GetName() + "'does not inherit from MergeInterface nor TObject");
59  }
60 
61  auto* object = ftm.ReadObjectAny(storedClass);
62  if (object == nullptr) {
63  throw std::runtime_error(
64  errorPrefix + "Failed to read object with name '" + storedClass->GetName() + "' from message using ROOT serialization.");
65  }
66 
67  if (inheritsFromTObject) {
68  return TObjectPtr(static_cast<TObject*>(object), algorithm::deleteTCollections);
69  } else {
70  return MergeInterfacePtr(static_cast<MergeInterface*>(object));
71  }
72 }
