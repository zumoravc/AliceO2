22 float CalibLHCphaseTOF::getLHCphase(int timestamp) const
23 {
24  int n = 0;
25  while (n < mLHCphase.size() && mLHCphase[n].first <= timestamp)
26  n++;
27  n--;
28 
29  if (n < 0) { // timestamp is before of the first available value
30  return 0;
31  }
32  return mLHCphase[n].second;
33 }

36 void CalibLHCphaseTOF::addLHCphase(int timestamp, float phaseLHC)
37 {
38  // optimized if timestamp are given in increasing order
39  int n = mLHCphase.size();
40  mLHCphase.emplace_back(timestamp, phaseLHC);
41 
42  if (n && mLHCphase[n].first < mLHCphase[n - 1].first) { // in the wrong order sort!
43  std::sort(mLHCphase.begin(), mLHCphase.end(), [](const auto& lhs, const auto& rhs) {
44  return lhs.first < rhs.first;
45  });
46  }
47 }
