19 int dump(const o2::mch::raw::DsElecId& dsElecId,
20  const o2::mch::raw::DsDetId& dsDetId)
21 {
22  auto solar2fee = o2::mch::raw::createSolar2FeeLinkMapper<o2::mch::raw::ElectronicMapperGenerated>();
23  auto feelink = solar2fee(dsElecId.solarId());
24  if (!feelink.has_value()) {
25  std::cout << "Could not get FeeLinkId for solarId " << dsElecId.solarId() << "\n";
26  return 4;
27  }
28  std::cout << fmt::format("{} (elinkId {:d}) [ {} ] {}\n",
29  o2::mch::raw::asString(dsElecId),
30  dsElecId.elinkId(),
31  o2::mch::raw::asString(feelink.value()),
32  o2::mch::raw::asString(dsDetId));
33  return 0;
34 }

36 int convertElec2Det(uint16_t solarId, uint8_t groupId, uint8_t indexId)
37 {
38  try {
39  std::cout << fmt::format("solarId {} groupId {} indexId {}\n",
40  solarId, groupId, indexId);
41  o2::mch::raw::DsElecId dsElecId(solarId, groupId, indexId);
42  auto elec2det = o2::mch::raw::createElec2DetMapper<o2::mch::raw::ElectronicMapperGenerated>();
43  auto dsDetId = elec2det(dsElecId);
44  if (!dsDetId.has_value()) {
45  std::cout << o2::mch::raw::asString(dsElecId) << " is not (yet?) known to the electronic mapper\n";
46  return 3;
47  }
48  return dump(dsElecId, dsDetId.value());
49  } catch (const std::exception& e) {
50  std::cout << e.what() << "\n";
51  return 4;
52  }
53 }

81 int main(int argc, char** argv)
82 {
83  int solarId;
84  int groupId;
85  int indexId;
86  int deId;
87  int dsId;
88 
89  po::variables_map vm;
90  po::options_description generic("Generic options");
91 
92  // clang-format off
93  generic.add_options()
94  ("help,h", "produce help message")
95  ("solarId,s",po::value<int>(&solarId),"solar id")
96  ("groupId,g",po::value<int>(&groupId),"group id")
97  ("indexId,i",po::value<int>(&indexId),"index id")
98  ("dsId,d",po::value<int>(&dsId),"dual sampa id")
99  ("deId,e",po::value<int>(&deId),"detection element id")
100  ;
101  // clang-format on
102 
103  po::options_description cmdline;
104  cmdline.add(generic);
105 
106  po::store(po::command_line_parser(argc, argv).options(cmdline).run(), vm);
107 
108  if (vm.count("help")) {
109  return usage(generic);
110  }
111 
112  try {
113  po::notify(vm);
114  } catch (boost::program_options::error& e) {
115  std::cout << "Error: " << e.what() << "\n";
116  exit(1);
117  }
118 
119  if (vm.count("solarId") && vm.count("groupId") && vm.count("indexId")) {
120  return convertElec2Det(static_cast<uint16_t>(solarId),
121  static_cast<uint8_t>(groupId),
122  static_cast<uint8_t>(indexId));
123  } else if (vm.count("deId") && vm.count("dsId")) {
124  return convertDet2Elec(deId, dsId);
125  } else {
126  std::cout << "Incorrect mix of options\n";
127  return usage(generic);
128  }
129 
130  return 0;
131 }
