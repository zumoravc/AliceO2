31 // Base class for a configurable parameter.
32 //
33 // A configurable parameter (ConfigurableParameter) is a simple class, defining
34 // a few (pod) properties/members which are registered
35 // in a global (boost) property tree / structure.
36 //
37 // The features that we provide here are:
38 // *) Automatic translation from C++ data description to INI/JSON/XML
39 // format via ROOT introspection and boost property trees and
40 // the possibility to readably save the configuration
41 // *) Serialization/Deserialization into ROOT binary blobs (for the purpose
42 // of writing/retrieving parameters from CCDB and to pass parameters along the processing chain)
43 // *) Automatic integration of sub-classes into a common configuration
44 // *) Be able to query properties from high level interfaces (just knowing
45 // *) Be able to set properties from high-level interfaces (and modifying the underlying
46 // C++ object)
47 // *) Automatic ability to modify parameters from the command-line
48 // *) Keeping track of the provenance of individual parameter values; The user is able
49 // to see whether is parameter is defaulted-code-value/coming-from-CCDB/coming-from-comandline
50 //
51 // Note that concrete parameter sub-classes **must** be implemented
52 // by inheriting from ConfigurableParamHelper and not from this class.
53 //
54 // ---------------------
55 // Example: To define a parameter class TPCGasParameters, one does the following:
56 //
57 // class TPCGasParamer : public ConfigurableParamHelper<TPCGasParameter>
58 // {
59 // public:
60 // double getGasDensity() const { return mGasDensity; }
61 // private: // define properties AND default values
62 // double mGasDensity = 1.23;
63 // int mGasMaterialID = 1;
64 //
65 // O2ParamDef(TPCGasParameter, TPCGas); // a macro implementing some magic
66 // }
67 //
68 //
69 // We can now query the parameters in various ways
70 // - All parameter classes are singletons and we can say: TPCGasParameter::Instance().getGasDensity();
71 // - We can query by key (using classname + parameter name) from the global registry:
72 // - ConfigurableParameter::getValueAs<double>("TPCGas", "mGasDensity");
73 //
74 // We can modify the parameters via the global registry together with an automatic syncing
75 // of the underlying C++ object:
76 // - ConfigurableParameter::setValue("TPCGas.mGasDensity", "0.5");
77 //
78 // - TPCGasParameter::Instance().getGasParameter() will now return 0.5;
79 //
80 // This feature allows to easily modify parameters at run-time via a textual representation
81 // (for example by giving strings on the command line)
82 //
83 // The collection of all parameter keys and values can be stored to a human/machine readable
84 // file
85 // - ConfigurableParameter::writeJSON("thisconfiguration.json")

87 struct EnumLegalValues {

90  bool isLegal(const std::string& value) const
91  {
92  for (auto& v : vvalues) {

180  template <typename T>
181  static void setValue(std::string const& mainkey, std::string const& subkey, T x)
182  {
183  if (!sIsFullyInitialized) {
184  initialize();
185  }
186  assert(sPtree);
187  try {
188  auto key = mainkey + "." + subkey;
189  if (sPtree->get_optional<std::string>(key).is_initialized()) {
190  sPtree->put(key, x);
191  auto changed = updateThroughStorageMap(mainkey, subkey, typeid(T), (void*)&x);
192  if (changed) {
193  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
194  }
195  }
196  } catch (std::exception const& e) {
197  std::cerr << "Error in setValue (T) " << e.what() << "\n";
198  }
199  }
200 
201  // specialized for std::string
202  // which means that the type will be converted internally
203  static void setValue(std::string const& key, std::string const& valuestring)
204  {
205  if (!sIsFullyInitialized) {
206  initialize();
207  }
208  assert(sPtree);
209  try {
210  if (sPtree->get_optional<std::string>(key).is_initialized()) {
211  sPtree->put(key, valuestring);
212  auto changed = updateThroughStorageMapWithConversion(key, valuestring);

214  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
215  }
216  }
