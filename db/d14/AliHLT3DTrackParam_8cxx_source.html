21 double AliHLT3DTrackParam::GetDStoPoint(double Bz, const double xyz[3], const double* T0) const
22 {
23  //* Get DS = Path/Momentum to a certain space point for Bz field
24 
25  double q = fSignQ;
26  if (!T0) {
27  T0 = mParam;
28  } else {
29  q = T0[6];
30  }
31 
32  const double kCLight = 0.000299792458;
33  double bq = Bz * q * kCLight;
34  double pt2 = T0[3] * T0[3] + T0[4] * T0[4];
35  if (pt2 < 1.e-4) {
36  return 0;
37  }
38  double dx = xyz[0] - T0[0];
39  double dy = xyz[1] - T0[1];
40  double a = dx * T0[3] + dy * T0[4];
41  double dS = 0;
42  if (TMath::Abs(bq) < 1.e-8) {
43  dS = a / pt2;
44  } else {
45  dS = TMath::ATan2(bq * a, pt2 + bq * (dy * T0[3] - dx * T0[4])) / bq;
46  }
47  return dS;
48 }

50 void AliHLT3DTrackParam::TransportToDS(double Bz, double DS, double* T0)
51 {
52  //* Transport the particle on DS = Path/Momentum, for Bz field
53 
54  double tmp[7];
55  if (!T0) {
56  T0 = tmp;
57  T0[0] = mParam[0];
58  T0[1] = mParam[1];
59  T0[2] = mParam[2];
60  T0[3] = mParam[3];
61  T0[4] = mParam[4];
62  T0[5] = mParam[5];
63  T0[6] = fSignQ;
64  }
65  const double kCLight = 0.000299792458;
66  Bz = Bz * T0[6] * kCLight;
67  double bs = Bz * DS;
68  double s = TMath::Sin(bs), c = TMath::Cos(bs);
69  double sB, cB;
70  if (TMath::Abs(bs) > 1.e-10) {
71  sB = s / Bz;
72  cB = (1 - c) / Bz;
73  } else {
74  const Double_t kOvSqr6 = 1. / TMath::Sqrt(6.);
75  sB = (1. - bs * kOvSqr6) * (1. + bs * kOvSqr6) * DS;
76  cB = .5 * sB * bs;
77  }
78 
79  double px = T0[3];
80  double py = T0[4];
81  double pz = T0[5];
82 
83  double d[6] = {mParam[0] - T0[0], mParam[1] - T0[1], mParam[2] - T0[2], mParam[3] - T0[3], mParam[4] - T0[4], mParam[5] - T0[5]};
84 
85  T0[0] = T0[0] + sB * px + cB * py;
86  T0[1] = T0[1] - cB * px + sB * py;
87  T0[2] = T0[2] + DS * pz;
88  T0[3] = c * px + s * py;
89  T0[4] = -s * px + c * py;
90  T0[5] = T0[5];
91 
92  // clang-format off
93  double mJ[6][6] = { {1, 0, 0, sB, cB, 0, },
94  {0, 1, 0, -cB, sB, 0, },
95  {0, 0, 1, 0, 0, DS, },
96  {0, 0, 0, c, s, 0, },
97  {0, 0, 0, -s, c, 0, },
98  {0, 0, 0, 0, 0, 1, }};
99  // clang-format on
100 
101  for (int i = 0; i < 6; i++) {
102  mParam[i] = T0[i];
103  for (int j = 0; j < 6; j++) {
104  mParam[i] += mJ[i][j] * d[j];
105  }
106  }
107 
108  double mA[6][6];
109  for (int k = 0, i = 0; i < 6; i++) {
110  for (int j = 0; j <= i; j++, k++) {
111  mA[i][j] = mA[j][i] = fCov[k];
112  }
113  }
114 
115  double mJC[6][6];
116  for (int i = 0; i < 6; i++) {
117  for (int j = 0; j < 6; j++) {
118  mJC[i][j] = 0;
119  for (int k = 0; k < 6; k++) {
120  mJC[i][j] += mJ[i][k] * mA[k][j];
121  }
122  }
123  }
124 
125  for (int k = 0, i = 0; i < 6; i++) {
126  for (int j = 0; j <= i; j++, k++) {
127  fCov[k] = 0;
128  for (int l = 0; l < 6; l++) {
129  fCov[k] += mJC[i][l] * mJ[j][l];
130  }
131  }
132  }
133 }

151 void AliHLT3DTrackParam::GetGlueMatrix(const double xyz[3], double G[6], const double* T0) const
152 {
153  //* !
154 
155  if (!T0) {
156  T0 = mParam;
157  }
158 
159  double dx = xyz[0] - T0[0], dy = xyz[1] - T0[1], dz = xyz[2] - T0[2];
160  double px2 = T0[3] * T0[3], py2 = T0[4] * T0[4], pz2 = T0[5] * T0[5];
161  double s2 = (dx * dx + dy * dy + dz * dz);
162  double p2 = px2 + py2 + pz2;
163  if (p2 > 1.e-4) {
164  s2 /= p2;
165  }
166  double x = T0[3] * s2;
167  double xx = px2 * s2, xy = x * T0[4], xz = x * T0[5], yy = py2 * s2, yz = T0[4] * T0[5] * s2;
168  G[0] = xx;
169  G[1] = xy;
170  G[2] = yy;
171  G[3] = xz;
172  G[4] = yz;
173  G[5] = pz2 * s2;
174 }

176 void AliHLT3DTrackParam::Filter(const double m[3], const double V[6], const double G[6])
177 {
178  //* !
179 
180  // clang-format off
181  double
182  c00 = fCov[0],
183  c10 = fCov[1], c11 = fCov[2],
184  c20 = fCov[3], c21 = fCov[4], c22 = fCov[5],
185  c30 = fCov[6], c31 = fCov[7], c32 = fCov[8],
186  c40 = fCov[10], c41 = fCov[11], c42 = fCov[12],
187  c50 = fCov[15], c51 = fCov[16], c52 = fCov[17];
188  // clang-format on
189 
190  double z0 = m[0] - mParam[0], z1 = m[1] - mParam[1], z2 = m[2] - mParam[2];
191 
192  double mS[6] = {c00 + V[0] + G[0], c10 + V[1] + G[1], c11 + V[2] + G[2], c20 + V[3] + G[3], c21 + V[4] + G[4], c22 + V[5] + G[5]};
193  double mSi[6];
194  mSi[0] = mS[4] * mS[4] - mS[2] * mS[5];
195  mSi[1] = mS[1] * mS[5] - mS[3] * mS[4];
196  mSi[3] = mS[2] * mS[3] - mS[1] * mS[4];
197  double det = 1. / (mS[0] * mSi[0] + mS[1] * mSi[1] + mS[3] * mSi[3]);
198  mSi[0] *= det;
199  mSi[1] *= det;
200  mSi[3] *= det;
201  mSi[2] = (mS[3] * mS[3] - mS[0] * mS[5]) * det;
202  mSi[4] = (mS[0] * mS[4] - mS[1] * mS[3]) * det;
203  mSi[5] = (mS[1] * mS[1] - mS[0] * mS[2]) * det;
204 
205  fNDF += 2;
206  fChi2 += (+(mSi[0] * z0 + mSi[1] * z1 + mSi[3] * z2) * z0 + (mSi[1] * z0 + mSi[2] * z1 + mSi[4] * z2) * z1 + (mSi[3] * z0 + mSi[4] * z1 + mSi[5] * z2) * z2);
207 
208  double k0, k1, k2; // k = CHtS
209 
210  k0 = c00 * mSi[0] + c10 * mSi[1] + c20 * mSi[3];
211  k1 = c00 * mSi[1] + c10 * mSi[2] + c20 * mSi[4];
212  k2 = c00 * mSi[3] + c10 * mSi[4] + c20 * mSi[5];
213 
214  mParam[0] += k0 * z0 + k1 * z1 + k2 * z2;
215  fCov[0] -= k0 * c00 + k1 * c10 + k2 * c20;
216 
217  k0 = c10 * mSi[0] + c11 * mSi[1] + c21 * mSi[3];
218  k1 = c10 * mSi[1] + c11 * mSi[2] + c21 * mSi[4];
219  k2 = c10 * mSi[3] + c11 * mSi[4] + c21 * mSi[5];
220 
221  mParam[1] += k0 * z0 + k1 * z1 + k2 * z2;
222  fCov[1] -= k0 * c00 + k1 * c10 + k2 * c20;
223  fCov[2] -= k0 * c10 + k1 * c11 + k2 * c21;
224 
225  k0 = c20 * mSi[0] + c21 * mSi[1] + c22 * mSi[3];
226  k1 = c20 * mSi[1] + c21 * mSi[2] + c22 * mSi[4];
227  k2 = c20 * mSi[3] + c21 * mSi[4] + c22 * mSi[5];
228 
229  mParam[2] += k0 * z0 + k1 * z1 + k2 * z2;
230  fCov[3] -= k0 * c00 + k1 * c10 + k2 * c20;
231  fCov[4] -= k0 * c10 + k1 * c11 + k2 * c21;
232  fCov[5] -= k0 * c20 + k1 * c21 + k2 * c22;
233 
234  k0 = c30 * mSi[0] + c31 * mSi[1] + c32 * mSi[3];
235  k1 = c30 * mSi[1] + c31 * mSi[2] + c32 * mSi[4];
236  k2 = c30 * mSi[3] + c31 * mSi[4] + c32 * mSi[5];
237 
238  mParam[3] += k0 * z0 + k1 * z1 + k2 * z2;
239  fCov[6] -= k0 * c00 + k1 * c10 + k2 * c20;
240  fCov[7] -= k0 * c10 + k1 * c11 + k2 * c21;
241  fCov[8] -= k0 * c20 + k1 * c21 + k2 * c22;
242  fCov[9] -= k0 * c30 + k1 * c31 + k2 * c32;
243 
244  k0 = c40 * mSi[0] + c41 * mSi[1] + c42 * mSi[3];
245  k1 = c40 * mSi[1] + c41 * mSi[2] + c42 * mSi[4];
246  k2 = c40 * mSi[3] + c41 * mSi[4] + c42 * mSi[5];
247 
248  mParam[4] += k0 * z0 + k1 * z1 + k2 * z2;
249  fCov[10] -= k0 * c00 + k1 * c10 + k2 * c20;
250  fCov[11] -= k0 * c10 + k1 * c11 + k2 * c21;
251  fCov[12] -= k0 * c20 + k1 * c21 + k2 * c22;
252  fCov[13] -= k0 * c30 + k1 * c31 + k2 * c32;
253  fCov[14] -= k0 * c40 + k1 * c41 + k2 * c42;
254 
255  k0 = c50 * mSi[0] + c51 * mSi[1] + c52 * mSi[3];
256  k1 = c50 * mSi[1] + c51 * mSi[2] + c52 * mSi[4];
257  k2 = c50 * mSi[3] + c51 * mSi[4] + c52 * mSi[5];
258 
259  mParam[5] += k0 * z0 + k1 * z1 + k2 * z2;
260  fCov[15] -= k0 * c00 + k1 * c10 + k2 * c20;
261  fCov[16] -= k0 * c10 + k1 * c11 + k2 * c21;
262  fCov[17] -= k0 * c20 + k1 * c21 + k2 * c22;
263  fCov[18] -= k0 * c30 + k1 * c31 + k2 * c32;
264  fCov[19] -= k0 * c40 + k1 * c41 + k2 * c42;
265  fCov[20] -= k0 * c50 + k1 * c51 + k2 * c52;
266 
267  // fit charge
268 
269  double px = mParam[3];
270  double py = mParam[4];
271  double pz = mParam[5];
272 
273  double p = TMath::Sqrt(px * px + py * py + pz * pz);
274  double pi = 1. / p;
275  double qp = fSignQ * pi;
276  double qp3 = qp * pi * pi;
277  double c60 = qp3 * (c30 + c40 + c50), c61 = qp3 * (c31 + c41 + c51), c62 = qp3 * (c32 + c42 + c52);
278 
279  k0 = c60 * mSi[0] + c61 * mSi[1] + c62 * mSi[3];
280  k1 = c60 * mSi[1] + c61 * mSi[2] + c62 * mSi[4];
281  k2 = c60 * mSi[3] + c61 * mSi[4] + c62 * mSi[5];
282 
283  qp += k0 * z0 + k1 * z1 + k2 * z2;
284  if (qp > 0) {
285  fSignQ = 1;
286  } else if (qp < 0) {
287  fSignQ = -1;
288  } else {
289  fSignQ = 0;
290  }
291 }

319 void AliHLT3DTrackParam::RotateCoordinateSystem(double alpha)
320 {
321  //* !
322 
323  double cA = TMath::Cos(alpha);
324  double sA = TMath::Sin(alpha);
325  double x = mParam[0], y = mParam[1], px = mParam[3], py = mParam[4];
326  mParam[0] = x * cA + y * sA;
327  mParam[1] = -x * sA + y * cA;
328  mParam[2] = mParam[2];
329  mParam[3] = px * cA + py * sA;
330  mParam[4] = -px * sA + py * cA;
331  mParam[5] = mParam[5];
332 
333  // clang-format off
334  double mJ[6][6] = { { cA, sA, 0, 0, 0, 0 },
335  { -sA, cA, 0, 0, 0, 0 },
336  { 0, 0, 1, 0, 0, 0 },
337  { 0, 0, 0, cA, sA, 0 },
338  { 0, 0, 0, -sA, cA, 0 },
339  { 0, 0, 0, 0, 0, 1 }};
340  // clang-format on
341 
342  double mA[6][6];
343  for (int k = 0, i = 0; i < 6; i++) {
344  for (int j = 0; j <= i; j++, k++) {
345  mA[i][j] = mA[j][i] = fCov[k];

369 void AliHLT3DTrackParam::Get5Parameters(double alpha, double T[6], double C[15]) const
370 {
371  //* !
372 
373  AliHLT3DTrackParam t = *this;
374  t.RotateCoordinateSystem(alpha);
375  double x = t.mParam[0], y = t.mParam[1], z = t.mParam[2], px = t.mParam[3], py = t.mParam[4], pz = t.mParam[5], q = t.fSignQ;
376 
377  double p2 = px * px + py * py + pz * pz;
378  if (p2 < 1.e-8) {
379  p2 = 1;
380  }
381  double n2 = 1. / p2;
382  double n = sqrt(n2);
383 
384  T[5] = x;
385  T[0] = y;
386  T[1] = z;
387  T[2] = py / px;
388  T[3] = pz / px;
389  T[4] = q * n;
390 
391  // clang-format off
392  double mJ[5][6] = { { -T[2], 1, 0, 0, 0, 0 },
393  { -T[3], 0, 1, 0, 0, 0 },
394  { 0, 0, 0, -T[2] / px, 1. / px, 0 },
395  { 0, 0, 0, -T[3] / px, 0, 1. / px },
396  { 0, 0, 0, -T[4]*n2*px, -T[4]*n2*py, -T[4]*n2*pz}};
397  // clang-format on
398 
399  double mA[6][6];
400  for (int k = 0, i = 0; i < 6; i++) {
401  for (int j = 0; j <= i; j++, k++) {
402  mA[i][j] = mA[j][i] = t.fCov[k];
403  }
404  }
405 
406  double mJC[5][6];
407  for (int i = 0; i < 5; i++) {
408  for (int j = 0; j < 6; j++) {
409  mJC[i][j] = 0;
410  for (int k = 0; k < 6; k++) {
411  mJC[i][j] += mJ[i][k] * mA[k][j];
412  }
413  }
414  }
415

417  for (int j = 0; j <= i; j++, k++) {
418  C[k] = 0;
419  for (int l = 0; l < 6; l++) {
420  C[k] += mJC[i][l] * mJ[j][l];
