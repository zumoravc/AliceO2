168  // add element for a particular dataindex
169  // at the moment only strictly consecutive modes are supported
170  void addElement(uint32_t dataindex, TruthElement const& element)
171  {
172  if (dataindex < mHeaderArray.size()) {
173  // look if we have something for this dataindex already
174  // must currently be the last one
175  if (dataindex != (mHeaderArray.size() - 1)) {
176  throw std::runtime_error("MCTruthContainer: unsupported code path");
177  }
178  } else {
179  // assert(dataindex == mHeaderArray.size());
180 
181  // add empty holes
182  int holes = dataindex - mHeaderArray.size();
183  assert(holes >= 0);
184  for (int i = 0; i < holes; ++i) {
185  mHeaderArray.emplace_back(mTruthArray.size());
186  }
187  // add a new one
188  mHeaderArray.emplace_back(mTruthArray.size());
189  }
190  mTruthArray.emplace_back(element);
191  }

215  // Add element at last position or for a previous index
216  // (at random access position).
217  // This might be a slow process since data has to be moved internally
218  // so this function should be used with care.
219  void addElementRandomAccess(uint32_t dataindex, TruthElement const& element)
220  {
221  if (dataindex >= mHeaderArray.size()) {
222  // a new dataindex -> push element at back
223 
224  // we still forbid to leave holes
225  assert(dataindex == mHeaderArray.size());
226 
227  mHeaderArray.resize(dataindex + 1);
228  mHeaderArray[dataindex] = mTruthArray.size();
229  mTruthArray.emplace_back(element);
230  } else {
231  // if appending at end use fast function
232  if (dataindex == mHeaderArray.size() - 1) {
233  addElement(dataindex, element);
234  return;
235  }
236 
237  // existing dataindex
238  // have to:
239  // a) move data;
240  // b) insert new element;
241  // c) adjust indices of all headers right to this
242  auto currentindex = mHeaderArray[dataindex].index;
243  auto lastindex = currentindex + getSize(dataindex);
244  assert(currentindex >= 0);
245 
246  // resize truth array
247  mTruthArray.resize(mTruthArray.size() + 1);
248  // move data (have to do from right to left)
249  for (int i = mTruthArray.size() - 1; i > lastindex; --i) {
250  mTruthArray[i] = mTruthArray[i - 1];
251  }
252  // insert new element
253  mTruthArray[lastindex] = element;
254 
255  // fix headers
256  for (uint32_t i = dataindex + 1; i < mHeaderArray.size(); ++i) {
257  auto oldindex = mHeaderArray[i].index;
258  mHeaderArray[i].index = (oldindex != -1) ? oldindex + 1 : oldindex;
259  }
260  }
261  }

270  // merge another container to the back of this one
271  void mergeAtBack(MCTruthContainer<TruthElement> const& other)
272  {

274  const auto oldheadersize = mHeaderArray.size();
275 
276  // copy from other

280  // adjust information of newly attached part
281  for (uint32_t i = oldheadersize; i < mHeaderArray.size(); ++i) {

290  template <typename ContainerType>
291  size_t flatten_to(ContainerType& container)
292  {
293  size_t bufferSize = sizeof(FlatHeader) + sizeof(MCTruthHeaderElement) * mHeaderArray.size() + sizeof(TruthElement) * mTruthArray.size();
294  container.resize((bufferSize / sizeof(typename ContainerType::value_type)) + ((bufferSize % sizeof(typename ContainerType::value_type)) > 0 ? 1 : 0));
295  char* target = reinterpret_cast<char*>(container.data());
296  auto& flatheader = *reinterpret_cast<FlatHeader*>(target);
297  target += sizeof(FlatHeader);
298  flatheader.version = 1;
299  flatheader.sizeofHeaderElement = sizeof(MCTruthHeaderElement);
300  flatheader.sizeofTruthElement = sizeof(TruthElement);
301  flatheader.reserved = 0;
302  flatheader.nofHeaderElements = mHeaderArray.size();
303  flatheader.nofTruthElements = mTruthArray.size();
304  size_t copySize = flatheader.sizeofHeaderElement * flatheader.nofHeaderElements;
305  memcpy(target, mHeaderArray.data(), copySize);
306  target += copySize;
307  copySize = flatheader.sizeofTruthElement * flatheader.nofTruthElements;
308  memcpy(target, mTruthArray.data(), copySize);
309  return bufferSize;
310  }

315  void restore_from(const char* buffer, size_t bufferSize)
316  {
317  if (buffer == nullptr || bufferSize < sizeof(FlatHeader)) {
318  return;
319  }
320  auto* source = buffer;
321  auto& flatheader = *reinterpret_cast<FlatHeader const*>(source);
322  source += sizeof(FlatHeader);
323  if (bufferSize < sizeof(FlatHeader) + flatheader.sizeofHeaderElement * flatheader.nofHeaderElements + flatheader.sizeofTruthElement * flatheader.nofTruthElements) {
324  throw std::runtime_error("inconsistent buffer size: too small");
325  return;
326  }
327  if (flatheader.sizeofHeaderElement != sizeof(MCTruthHeaderElement) || flatheader.sizeofTruthElement != sizeof(TruthElement)) {
328  // not yet handled
329  throw std::runtime_error("member element sizes don't match");
330  }
331  // TODO: with a spectator memory ressource the vectors can be built directly
332  // over the original buffer, there is the implementation for a memory ressource
333  // working on a FairMQ message, here we would need two memory resources over
334  // the two ranges of the input buffer
335  // for now doing a copy
336  mHeaderArray.resize(flatheader.nofHeaderElements);
337  mTruthArray.resize(flatheader.nofTruthElements);
338  size_t copySize = flatheader.sizeofHeaderElement * flatheader.nofHeaderElements;
339  memcpy(mHeaderArray.data(), source, copySize);
340  source += copySize;
341  copySize = flatheader.sizeofTruthElement * flatheader.nofTruthElements;
342  memcpy(mTruthArray.data(), source, copySize);
343  }
