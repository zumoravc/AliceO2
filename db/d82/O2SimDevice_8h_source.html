83  // should go into a helper
84  // this function queries the sim config data and initializes the SimConfig singleton
85  // returns true if successful / false if not
86  static bool querySimConfig(FairMQChannel& channel)
87  {
88  auto text = new std::string("configrequest");
89  std::unique_ptr<FairMQMessage> request(channel.NewMessage(const_cast<char*>(text->c_str()),
90  text->length(), CustomCleanup, text));
91  std::unique_ptr<FairMQMessage> reply(channel.NewMessage());
92 
93  int timeoutinMS = 100000; // wait for 100s max
94  if (channel.Send(request, timeoutinMS) > 0) {
95  LOG(INFO) << "Waiting for configuration answer ";
96  if (channel.Receive(reply, timeoutinMS) > 0) {
97  LOG(INFO) << "Configuration answer received, containing " << reply->GetSize() << " bytes ";
98 
99  // the answer is a TMessage containing the simulation Configuration
100  auto message = std::make_unique<o2::devices::TMessageWrapper>(reply->GetData(), reply->GetSize());
101  auto config = static_cast<o2::conf::SimConfigData*>(message.get()->ReadObjectAny(message.get()->GetClass()));
102  if (!config) {
103  return false;
104  }
105 
106  LOG(INFO) << "COMMUNICATED ENGINE " << config->mMCEngine;
107 
108  auto& conf = o2::conf::SimConfig::Instance();
109  conf.resetFromConfigData(*config);
110  FairLogger::GetLogger()->SetLogVerbosityLevel(conf.getLogVerbosity().c_str());
111  delete config;
112  } else {
113  LOG(ERROR) << "No configuration received within " << timeoutinMS << "ms\n";
114  return false;
115  }
116  } else {
117  LOG(ERROR) << "Could not send configuration request within " << timeoutinMS << "ms\n";
118  return false;
119  }
120  return true;
121  }

123  // initializes the simulation classes; queries the configuration on a given channel
124  static bool initSim(FairMQChannel& channel, std::unique_ptr<FairRunSim>& simptr)
125  {
126  if (!querySimConfig(channel)) {
127  return false;
128  }
129 
130  LOG(INFO) << "Setting up the simulation ...";
131  simptr = std::move(std::unique_ptr<FairRunSim>(o2sim_init(true)));
132  FairSystemInfo sysinfo;
133 
134  // to finish initialization (trigger further cross section table building etc) -- which especially
135  // G4 is doing at the first ProcessRun
136  // The goal is to have everything setup before we fork
137  TVirtualMC::GetMC()->ProcessRun(0);
138 
139  LOG(INFO) << "MEM-STAMP END OF SIM INIT" << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "
140  << sysinfo.GetMaxMemory() << " MB\n";
141 
142  return true;
143  }

145  bool Kernel(FairMQChannel& requestchannel, FairMQChannel& dataoutchannel)
146  {
147  auto text = new std::string("primrequest");
148 
149  // create message object with a pointer to the data buffer,
150  // its size,
151  // custom deletion function (called when transfer is done),
152  // and pointer to the object managing the data buffer
153  FairMQMessagePtr request(requestchannel.NewMessage(const_cast<char*>(text->c_str()), // data
154  text->length(), // size
155  CustomCleanup,
156  text));
157  FairMQMessagePtr reply(dataoutchannel.NewMessage());
158 
159  mVMCApp->setSimDataChannel(&dataoutchannel);
160 
161  LOG(INFO) << "Requesting work ";
162  int timeoutinMS = 1000000; // wait for 1000s max -- we should have a more robust solution
163  // this should be mostly driven by the time to setup the particle generator in the server
164  auto sendcode = requestchannel.Send(request, timeoutinMS);
165  if (sendcode > 0) {
166  LOG(INFO) << "Waiting for answer ";
167  // asking for primary generation
168 
169  int trial = 0;
170  int code = -1;
171  do {
172  code = requestchannel.Receive(reply, timeoutinMS);
173  trial++;
174  if (code > 0) {
175  LOG(INFO) << "Answer received, containing " << reply->GetSize() << " bytes ";
176 
177  // wrap incoming bytes as a TMessageWrapper which offers "adoption" of a buffer
178  auto message = new TMessageWrapper(reply->GetData(), reply->GetSize());
179  auto chunk = static_cast<o2::data::PrimaryChunk*>(message->ReadObjectAny(message->GetClass()));
180 
181  mVMCApp->setPrimaries(chunk->mParticles);
182 
183  auto info = chunk->mSubEventInfo;
184  mVMCApp->setSubEventInfo(&info);
185 
186  LOG(INFO) << "Processing " << chunk->mParticles.size() << " primary particles "
187  << "for event " << info.eventID << "/" << info.maxEvents << " "
188  << "part " << info.part << "/" << info.nparts;
189  gRandom->SetSeed(chunk->mSubEventInfo.seed);
190 
191  auto& conf = o2::conf::SimConfig::Instance();
192  if (strcmp(conf.getMCEngine().c_str(), "TGeant4") == 0) {
193  mVMC->ProcessEvent();
194  } else {
195  // for Geant3 at least calling ProcessEvent is not enough
196  // as some hooks are not called
197  mVMC->ProcessRun(1);
198  }
199 
200  FairSystemInfo sysinfo;
201  LOG(INFO) << "TIME-STAMP " << mTimer.RealTime() << "\t";
202  mTimer.Continue();
203  LOG(INFO) << "MEM-STAMP " << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "
204  << sysinfo.GetMaxMemory() << " MB\n";
205  delete message;
206  delete chunk;
207  } else {
208  LOG(INFO) << " No answer reveived from server. Return code " << code;
209  }
210  } while (code == -1 && trial <= 2);
211  } else {
212  LOG(INFO) << " Requesting work from server not possible. Return code " << sendcode;
213  return false;
214  }
215  return true;
216  }
