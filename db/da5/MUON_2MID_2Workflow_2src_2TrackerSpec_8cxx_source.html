38  public:
39  void init(o2::framework::InitContext& ic)
40  {
41 
42  auto geoFilename = ic.options().get<std::string>("geometry-filename");
43  if (!gGeoManager) {
44  o2::base::GeometryManager::loadGeometry(geoFilename);
45  }
46 
47  mTracker = std::make_unique<Tracker>(createTransformationFromManager(gGeoManager));
48 
49  if (!mTracker->init()) {
50  LOG(ERROR) << "Initialization of MID tracker device failed";
51  }
52 
53  auto stop = [this]() {
54  LOG(INFO) << "Capacities: ROFRecords: " << mTracker->getTrackROFRecords().capacity() << " tracks: " << mTracker->getTracks().capacity() << " clusters: " << mTracker->getClusters().capacity();
55  double scaleFactor = 1.e6 / mNROFs;
56  LOG(INFO) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us tracking: " << mTimerAlgo.count() * scaleFactor << " us";
57  };
58  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
59  }

61  void run(o2::framework::ProcessingContext& pc)
62  {
63  auto tStart = std::chrono::high_resolution_clock::now();
64 
65  auto msg = pc.inputs().get("mid_clusters");
66  gsl::span<const Cluster2D> clusters = of::DataRefUtils::as<const Cluster2D>(msg);
67 
68  auto msgROF = pc.inputs().get("mid_clusters_rof");
69  gsl::span<const ROFRecord> inROFRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
70 
71  auto tAlgoStart = std::chrono::high_resolution_clock::now();
72  mTracker->process(clusters, inROFRecords);
73  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
74 
75  pc.outputs().snapshot(of::Output{"MID", "TRACKS", 0, of::Lifetime::Timeframe}, mTracker->getTracks());
76  LOG(DEBUG) << "Sent " << mTracker->getTracks().size() << " tracks.";
77  pc.outputs().snapshot(of::Output{"MID", "TRACKCLUSTERS", 0, of::Lifetime::Timeframe}, mTracker->getClusters());
78  LOG(DEBUG) << "Sent " << mTracker->getClusters().size() << " track clusters.";
79 
80  pc.outputs().snapshot(of::Output{"MID", "TRACKSROF", 0, of::Lifetime::Timeframe}, mTracker->getTrackROFRecords());
81  LOG(DEBUG) << "Sent " << mTracker->getTrackROFRecords().size() << " ROFs.";
82  pc.outputs().snapshot(of::Output{"MID", "TRCLUSROF", 0, of::Lifetime::Timeframe}, mTracker->getClusterROFRecords());
83  LOG(DEBUG) << "Sent " << mTracker->getClusterROFRecords().size() << " ROFs.";
84 
85  mTimer += std::chrono::high_resolution_clock::now() - tStart;
86  mNROFs += inROFRecords.size();
87  }
88 
89  private:
90  std::unique_ptr<Tracker> mTracker{nullptr};
91  std::chrono::duration<double> mTimer{0};
92  std::chrono::duration<double> mTimerAlgo{0};
93  unsigned int mNROFs{0};
94 };
