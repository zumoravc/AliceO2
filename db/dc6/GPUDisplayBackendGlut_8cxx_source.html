171 void GPUDisplayBackendGlut::mouseFunc(int button, int state, int x, int y)
172 {
173  if (button == 3) {
174  me->mMouseWheel += 100;
175  } else if (button == 4) {
176  me->mMouseWheel -= 100;
177  } else if (state == GLUT_DOWN) {
178  if (button == GLUT_LEFT_BUTTON) {
179  me->mMouseDn = true;
180  } else if (button == GLUT_RIGHT_BUTTON) {
181  me->mMouseDnR = true;
182  }
183  me->mMouseDnX = x;

186  if (button == GLUT_LEFT_BUTTON) {
187  me->mMouseDn = false;

189  me->mMouseDnR = false;
190  }
191  }

202 int GPUDisplayBackendGlut::OpenGLMain()
203 {
204  me = this;
205  int nopts = 2;
206  char opt1[] = "progname";
207  char opt2[] = "-direct";
208  char* opts[] = {opt1, opt2};
209  glutInit(&nopts, opts);
210  glutInitContextVersion(GL_MIN_VERSION_MAJOR, GL_MIN_VERSION_MINOR);
211  glutInitContextProfile(GPUCA_DISPLAY_OPENGL_CORE_FLAGS ? GLUT_CORE_PROFILE : GLUT_COMPATIBILITY_PROFILE);
212  glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
213  glutInitWindowSize(INIT_WIDTH, INIT_HEIGHT);
214  glutCreateWindow(GL_WINDOW_NAME);
215  glutSetOption(GLUT_ACTION_ON_WINDOW_CLOSE, GLUT_ACTION_GLUTMAINLOOP_RETURNS);
216 
217  if (GPUDisplayExtInit()) {
218  fprintf(stderr, "Error initializing GL extension wrapper\n");
219  return (-1);
220  }
221  if (InitGL()) {
222  fprintf(stderr, "Error in OpenGL initialization\n");
223  return (1);
224  }
225 
226  glutDisplayFunc(displayFunc);
227  glutIdleFunc(glutLoopFunc);
228  glutReshapeFunc(ReSizeGLSceneWrapper);
229  glutKeyboardFunc(keyboardDownFunc);
230  glutKeyboardUpFunc(keyboardUpFunc);
231  glutSpecialFunc(specialDownFunc);
232  glutSpecialUpFunc(specialUpFunc);
233  glutMouseFunc(mouseFunc);
234  glutMotionFunc(mouseMoveFunc);
235  glutPassiveMotionFunc(mouseMoveFunc);
236  glutMouseWheelFunc(mMouseWheelFunc);
237  ToggleMaximized(true);
238 
239  pthread_mutex_lock(&mSemLockExit);
240  mGlutRunning = true;
241  pthread_mutex_unlock(&mSemLockExit);
242  glutMainLoop();
243  mDisplayControl = 2;
244  pthread_mutex_lock(&mSemLockExit);
245  mGlutRunning = false;
246  pthread_mutex_unlock(&mSemLockExit);
247 
248  return 0;
249 }
