42 void o2::data_flow::TimeframeValidatorDevice::Run()
43 {
44  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this)) {
45  FairMQParts timeframeParts;
46  if (Receive(timeframeParts, mInChannelName, 0, 100) <= 0)
47  continue;
48 
49  if (timeframeParts.Size() < 2)
50  LOG(ERROR) << "Expecting at least 2 parts\n";
51 
52  auto indexHeader = o2::header::get<header::DataHeader*>(timeframeParts.At(timeframeParts.Size() - 2)->GetData());
53  // FIXME: Provide iterator pair API for the index
54  // Index should really be something which provides an
55  // iterator pair API so that we can sort / find / lower_bound
56  // easily. Right now we simply use it a C-style array.
57  auto index = reinterpret_cast<IndexElement*>(timeframeParts.At(timeframeParts.Size() - 1)->GetData());
58 
59  // TODO: fill this with checks on time frame
60  LOG(INFO) << "This time frame has " << timeframeParts.Size() << " parts.\n";
61  auto indexEntries = indexHeader->payloadSize / sizeof(DataHeader);
62  if (indexHeader->dataDescription != DataDescription("TIMEFRAMEINDEX"))
63  LOG(ERROR) << "Could not find a valid index header\n";
64  LOG(INFO) << indexHeader->dataDescription.str << "\n";
65  LOG(INFO) << "This time frame has " << indexEntries << "entries in the index.\n";
66  if ((indexEntries * 2 + 2) != (timeframeParts.Size()))
67  LOG(ERROR) << "Mismatched index and received parts\n";
68 
69  // - Use the index to find out if we have TPC data
70  // - Get the part with the TPC data
71  // - Validate TPCCluster dummy data
72  // - Validate ITSRaw dummy data
73  int tpcIndex = -1;
74  int itsIndex = -1;
75 
76  for (int ii = 0; ii < indexEntries; ++ii) {
77  IndexElement& ie = index[ii];
78  assert(ie.second >= 0);
79  LOG(DEBUG) << ie.first.dataDescription.str << " "
80  << ie.first.dataOrigin.str << std::endl;
81  if ((ie.first.dataOrigin == header::gDataOriginTPC) && (ie.first.dataDescription == header::gDataDescriptionClusters)) {
82  tpcIndex = ie.second;
83  }
84  if ((ie.first.dataOrigin == header::gDataOriginITS) && (ie.first.dataDescription == header::gDataDescriptionClusters)) {
85  itsIndex = ie.second;
86  }
87  }
88 
89  if (tpcIndex < 0) {
90  LOG(ERROR) << "Could not find expected TPC payload\n";
91  continue;
92  }
93  if (itsIndex < 0) {
94  LOG(ERROR) << "Could not find expected ITS payload\n";
95  continue;
96  }
97  LOG(DEBUG) << "TPC Index " << tpcIndex << "\n";
98  LOG(DEBUG) << "ITS Index " << itsIndex << "\n";
99 
100  // Data header it at position - 1
101  auto tpcHeader = reinterpret_cast<DataHeader*>(timeframeParts.At(tpcIndex)->GetData());
102  if ((tpcHeader->dataDescription != header::gDataDescriptionClusters) ||
103  (tpcHeader->dataOrigin != header::gDataOriginTPC)) {
104  LOG(ERROR) << "Wrong data description. Expecting TPC - CLUSTERS, found "
105  << tpcHeader->dataOrigin.str << " - "
106  << tpcHeader->dataDescription.str << "\n";
107  continue;
108  }
109  auto tpcPayload = reinterpret_cast<TPCTestCluster*>(timeframeParts.At(tpcIndex + 1)->GetData());
110  if (tpcHeader->payloadSize % sizeof(TPCTestCluster))
111  LOG(ERROR) << "TPC - CLUSTERS Size Mismatch\n";
112  auto numOfClusters = tpcHeader->payloadSize / sizeof(TPCTestCluster);
113  for (size_t ci = 0; ci < numOfClusters; ++ci) {
114  TPCTestCluster& cluster = tpcPayload[ci];
115  if (cluster.z != 1.5) {
116  LOG(ERROR) << "TPC Data mismatch. Expecting z = 1.5 got " << cluster.z << "\n";
117  break;
118  }
119  if (cluster.timeStamp != ci) {
120  LOG(ERROR) << "TPC Data mismatch. Expecting " << ci << " got " << cluster.timeStamp << "\n";
121  break;
122  }
123  }
124 
125  // Data header it at position - 1
126  auto itsHeader = reinterpret_cast<DataHeader*>(timeframeParts.At(itsIndex)->GetData());
127  if ((itsHeader->dataDescription != header::gDataDescriptionClusters) || (itsHeader->dataOrigin != header::gDataOriginITS)) {
128  LOG(ERROR) << "Wrong data description. Expecting ITS - CLUSTERS, found "
129  << itsHeader->dataOrigin.str << " - " << itsHeader->dataDescription.str << "\n";
130  continue;
131  }
132  auto itsPayload = reinterpret_cast<ITSRawData*>(timeframeParts.At(itsIndex + 1)->GetData());
133  if (itsHeader->payloadSize % sizeof(ITSRawData))
134  LOG(ERROR) << "ITS - CLUSTERS Size Mismatch.\n";
135  numOfClusters = itsHeader->payloadSize / sizeof(ITSRawData);
136  for (size_t ci = 0; ci < numOfClusters; ++ci) {
137  ITSRawData& cluster = itsPayload[ci];
138  if (cluster.timeStamp != ci) {
139  LOG(ERROR) << "ITS Data mismatch. Expecting " << ci
140  << " got " << cluster.timeStamp << "\n";
141  break;
142  }
143  }
144  LOG(INFO) << "Everything is fine with received timeframe\n";
145  }
146 }
