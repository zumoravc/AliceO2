38  BOOST_CHECK_CLOSE(vtx.getCov()[i], cov[i], 1e-5);
39  }
40  BOOST_CHECK_CLOSE(vtx.getChi2(), chi2, 1e-5);
41  BOOST_CHECK(vtx.getNContributors() == nc);
42  BOOST_CHECK(vtx.getTimeStamp() == ts);
