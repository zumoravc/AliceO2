74 size_t RawFileReader::LinkData::getNextHBFSize() const
75 {
76  // estimate the memory size of the next HBF to read
77  // The blocks are guaranteed to not cover more than 1 HB
78  size_t sz = 0;
79  int ibl = nextBlock2Read, nbl = blocks.size();
80  while (ibl < nbl && (blocks[ibl].orbit == blocks[nextBlock2Read].orbit)) {
81  sz += blocks[ibl].size;
82  ibl++;
83  }
84  return sz;
85 }

87 //____________________________________________
88 size_t RawFileReader::LinkData::readNextHBF(char* buff)
89 {
90  // read data of the next complete HB, buffer of getNextHBFSize() must be allocated in advance
91  size_t sz = 0;
92  int ibl = nextBlock2Read, nbl = blocks.size();
93  bool error = false;
94  while (ibl < nbl) {
95  const auto& blc = blocks[ibl];
96  if (blc.orbit != blocks[nextBlock2Read].orbit) {
97  break;
98  }
99  ibl++;
100  auto fl = reader->mFiles[blc.fileID];
101  if (fseek(fl, blc.offset, SEEK_SET) || fread(buff + sz, 1, blc.size, fl) != blc.size) {
102  LOGF(ERROR, "Failed to read for the %s a bloc:", describe());
103  blc.print();
104  error = true;
105  }
106  sz += blc.size;
107  }
108  nextBlock2Read = ibl;
109  return error ? 0 : sz; // in case of the error we ignore the data
110 }

144 //_____________________________________________________________________
145 size_t RawFileReader::LinkData::readNextTF(char* buff)
146 {
147  // read next complete TF, buffer of getNextTFSize() must be allocated in advance
148  size_t sz = 0;
149  int ibl0 = nextBlock2Read, nbl = blocks.size();
150  bool error = false;
151  while (nextBlock2Read < nbl && (blocks[nextBlock2Read].tfID == blocks[ibl0].tfID)) { // nextBlock2Read is incremented by the readNextHBF!
152  auto szb = readNextHBF(buff + sz);
153  if (!szb) {
154  error = true;
155  }
156  sz += szb;
157  }
158  return error ? 0 : sz; // in case of the error we ignore the data
159 }

202 //____________________________________________
203 size_t RawFileReader::LinkData::getLargestSuperPage() const
204 {
205  // estimate largest super page size
206  size_t szMax = 0, szLast = 0;
207  for (const auto& bl : blocks) {
208  if (bl.testFlag(LinkBlock::StartSP)) { // account previous SPage and start accumulation of the next one
209  if (szLast > szMax) {
210  szMax = szLast;
211  }
212  szLast = 0;
213  }
214  szLast += bl.size;
215  }
216  return szLast > szMax ? szLast : szMax;
217 }

236 //_____________________________________________________________________
237 bool RawFileReader::LinkData::preprocessCRUPage(const RDHAny& rdh, bool newSPage)
238 {
239  // account RDH in statistics
240  bool ok = true;
241  bool newTF = false, newHB = false;
242 
243  if (RDHUtils::getFEEID(rdh) != RDHUtils::getFEEID(rdhl)) { // make sure links with different FEEID were not assigned same subspec
244  LOGF(ERROR, "Same SubSpec is found for %s with different RDH.feeId", describe());
245  LOGF(ERROR, "old RDH assigned SubSpec=0x%-8d:", subspec);
246  RDHUtils::dumpRDH(rdhl);
247  LOGF(ERROR, "new RDH assigned SubSpec=0x%-8d:", subspec);
248  RDHUtils::dumpRDH(rdh);
249  throw std::runtime_error("Conflicting SubSpecs are provided");
250  ok = false;
251  nErrors++;
252  }
253 
254  auto pageCnt = RDHUtils::getPageCounter(rdh);
255  if (pageCnt == 0) {
256  auto triggerType = RDHUtils::getTriggerType(rdh);
257  newTF = (triggerType & o2::trigger::TF);
258  newHB = (triggerType & (o2::trigger::ORBIT | o2::trigger::HB)) == (o2::trigger::ORBIT | o2::trigger::HB);
259  } else if (reader->mCheckErrors & (0x1 << ErrWrongPageCounterIncrement)) {
260  // check increasing pageCnt
261  if (nCRUPages && (pageCnt != (RDHUtils::getPageCounter(rdhl) + 1))) { // skip for very 1st page
262  LOG(ERROR) << ErrNames[ErrWrongPageCounterIncrement]
263  << " old=" << int(pageCnt) << " new=" << int(RDHUtils::getPageCounter(rdhl));
264  ok = false;
265  nErrors++;
266  }
267  }
268 
269  if (reader->mCheckErrors) {
270  if (nCRUPages) {
271  // check increasing (or wrapping) packetCounter
272  auto packetCounter = RDHUtils::getPacketCounter(rdh);
273  auto packetCounterL = RDHUtils::getPacketCounter(rdhl);
274  if ((packetCounter != ((packetCounterL + 1) & 0xff)) &&
275  (reader->mCheckErrors & (0x1 << ErrWrongPacketCounterIncrement))) { // skip for very 1st page
276  LOG(ERROR) << ErrNames[ErrWrongPacketCounterIncrement]
277  << " new=" << int(packetCounter) << " old=" << int(packetCounterL);
278  ok = false;
279  nErrors++;
280  }
281  // check if number of HBFs in the TF is as expected
282  if (newTF) {
283  if (nHBFinTF != HBFUtils::Instance().getNOrbitsPerTF() &&
284  (reader->mCheckErrors & (0x1 << ErrWrongHBFsPerTF))) {
285  LOG(ERROR) << ErrNames[ErrWrongHBFsPerTF] << ": "
286  << nHBFinTF << " instead of " << HBFUtils::Instance().getNOrbitsPerTF();
287  ok = false;
288  nErrors++;
289  }
290  nHBFinTF = 0; // reset
291  }
292 
293  } else { // make sure data starts with TF and HBF
294  if ((!newTF || !newHB || pageCnt != 0) &&
295  (reader->mCheckErrors & (0x1 << ErrWrongFirstPage))) {
296  LOG(ERROR) << ErrNames[ErrWrongFirstPage];
297  ok = false;
298  nErrors++;
299  }
300  }
301  }
302  auto stop = RDHUtils::getStop(rdh);
303  auto hbIR = RDHUtils::getHeartBeatIR(rdh), hblIR = RDHUtils::getHeartBeatIR(rdhl);
304  if (newHB) {
305  if (reader->mCheckErrors) {
306  nHBFinTF++;
307  if (stop && (reader->mCheckErrors & (0x1 << ErrHBFStopOnFirstPage))) {
308  LOG(ERROR) << ErrNames[ErrHBFStopOnFirstPage] << " @ HBF#" << nHBFrames;
309  ok = false;
310  nErrors++;
311  }
312  if (openHB && (reader->mCheckErrors & (0x1 << ErrHBFNoStop))) {
313  LOG(ERROR) << ErrNames[ErrHBFNoStop] << " @ HBF#" << nHBFrames;
314  ok = false;
315  nErrors++;
316  }
317  if ((reader->mCheckErrors & (0x1 << ErrHBFJump)) &&
318  (nCRUPages && // skip this check for the very 1st RDH
319  !(hbIR.bc == hblIR.bc && hbIR.orbit == hblIR.orbit + 1))) {
320  LOG(ERROR) << ErrNames[ErrHBFJump] << " @ HBF#" << nHBFrames << " New HB orbit/bc=" << hbIR.orbit << '/' << int(hbIR.bc)
321  << " is not incremented by 1 orbit wrt Old HB orbit/bc=" << hblIR.orbit << '/' << int(hblIR.bc);
322  ok = false;
323  nErrors++;
324  }
325  } // end of check errors
326  openHB = true;
327  nHBFrames++;
328  }
329  if (stop) {
330  openHB = false;
331  }
332 
333  if (newTF || newSPage || newHB) {
334  auto& bl = blocks.emplace_back(reader->mCurrentFileID, reader->mPosInFile);
335  if (newTF) {
336  nTimeFrames++;
337  bl.setFlag(LinkBlock::StartTF);
338  if (reader->mCheckErrors & (0x1 << ErrNoSuperPageForTF)) {
339  if (reader->mMultiLinkFile && !newSPage) {
340  LOG(ERROR) << ErrNames[ErrNoSuperPageForTF] << " @ TF#" << nTimeFrames;
341  ok = false;
342  nErrors++;
343  }
344  } // end of check errors
345  }
346  bl.orbit = hbIR.orbit;
347  bl.tfID = nTimeFrames - 1;
348 
349  if (newSPage) {
350  nSPages++;
351  bl.setFlag(LinkBlock::StartSP);
352  }
353  if (newHB) {
354  bl.setFlag(LinkBlock::StartHB);
355  }
356  }
357  blocks.back().setFlag(LinkBlock::EndHB, stop); // last processed RDH defines this flag
358  blocks.back().size += RDHUtils::getOffsetToNext(rdh);
359  rdhl = rdh;
360  nCRUPages++;
361  if (!ok) {
362  LOG(ERROR) << " ^^^Problem(s) was encountered at offset " << reader->mPosInFile << " of file " << reader->mCurrentFileID;
363  RDHUtils::printRDH(rdh);
364  } else if (reader->mVerbosity > 1) {
365  if (reader->mVerbosity > 2) {
366  RDHUtils::dumpRDH(rdh);
367  } else {
368  RDHUtils::printRDH(rdh);
369  }
370  }
371  return true;
372 }

386 int RawFileReader::getLinkLocalID(const RDHAny& rdh, const o2::header::DataOrigin orig)
387 {
388  // get id of the link subspec. in the parser (create entry if new)
389  LinkSubSpec_t subspec = RDHUtils::getSubSpec(rdh);
390  LinkSpec_t spec = createSpec(orig, subspec);
391  auto entryMap = mLinkEntries.find(spec);
392  if (entryMap == mLinkEntries.end()) { // need to register a new link
393  int n = mLinkEntries.size();
394  mLinkEntries[spec] = n;
395  auto& lnk = mLinksData.emplace_back(rdh, this);
396  lnk.subspec = subspec;
397  lnk.origin = mDataSpecs[mCurrentFileID].first;
398  lnk.description = mDataSpecs[mCurrentFileID].second;
399  lnk.spec = spec;
400  return n;
401  }
402  return entryMap->second;
403 }

405 //_____________________________________________________________________
406 bool RawFileReader::preprocessFile(int ifl)
407 {
408  // preprocess file, check RDH data, build statistics
409  std::unique_ptr<char[]> buffer = std::make_unique<char[]>(mBufferSize);
410  FILE* fl = mFiles[ifl];
411  mCurrentFileID = ifl;
412  LinkSpec_t specPrev = 0xffffffffffffffff;
413  int lIDPrev = -1;
414  mMultiLinkFile = false;
415  rewind(fl);
416  long int nr = 0;
417  mPosInFile = 0;
418  size_t nRDHread = 0, boffs;
419  bool ok = true, readMore = true;
420  while (readMore && (nr = fread(buffer.get(), 1, mBufferSize, fl))) {
421  boffs = 0;
422  while (1) {
423  auto& rdh = *reinterpret_cast<RDHUtils::RDHAny*>(&buffer[boffs]);
424  nRDHread++;
425  LinkSpec_t spec = createSpec(mDataSpecs[mCurrentFileID].first, RDHUtils::getSubSpec(rdh));
426  int lID = lIDPrev;
427  if (spec != specPrev) { // link has changed
428  specPrev = spec;
429  if (lIDPrev != -1) {
430  mMultiLinkFile = true;
431  }
432  lID = getLinkLocalID(rdh, mDataSpecs[mCurrentFileID].first);
433  }
434  bool newSPage = lID != lIDPrev;
435  mLinksData[lID].preprocessCRUPage(rdh, newSPage);
436  if (mLinksData[lID].nTimeFrames && (mLinksData[lID].nTimeFrames - 1 > mMaxTFToRead)) { // limit reached, discard the last read
437  mLinksData[lID].nTimeFrames--;
438  mLinksData[lID].blocks.pop_back();
439  if (mLinksData[lID].nHBFrames > 0) {
440  mLinksData[lID].nHBFrames--;
441  }
442  if (mLinksData[lID].nCRUPages > 0) {
443  mLinksData[lID].nCRUPages--;
444  }
445  lIDPrev = -1; // last block is closed
446  readMore = false;
447  break;
448  }
449  boffs += RDHUtils::getOffsetToNext(rdh);
450  mPosInFile += RDHUtils::getOffsetToNext(rdh);
451  lIDPrev = lID;
452  if (boffs + sizeof(RDHUtils::RDHAny) >= nr) {
453  if (fseek(fl, mPosInFile, SEEK_SET)) {
454  readMore = false;
455  break;
456  }
457  break;
458  }
459  }
460  }
461  LOGF(INFO, "File %3d : %9li bytes scanned, %6d RDH read for %4d links from %s",
462  mCurrentFileID, mPosInFile, nRDHread, int(mLinkEntries.size()), mFileNames[mCurrentFileID]);
463  return ok;
464 }

496 bool RawFileReader::addFile(const std::string& sname, o2::header::DataOrigin origin, o2::header::DataDescription desc)
497 {
498  if (mInitDone) {
499  LOG(ERROR) << "Cannot add new files after initialization";
500  return false;
501  }
502  auto inFile = fopen(sname.c_str(), "rb");
503  bool ok = true;
504  if (!inFile) {
505  LOG(ERROR) << "Failed to open input file " << sname;
506  ok = false;
507  }
508  if (origin == o2h::gDataOriginInvalid) {
509  LOG(ERROR) << "Invalid data origin " << origin.as<std::string>() << " for file " << sname;
510  ok = false;
511  }
512  if (desc == o2h::gDataDescriptionInvalid) {
513  LOG(ERROR) << "Invalid data description " << desc.as<std::string>() << " for file " << sname;
514  ok = false;
515  }
516  if (!ok) {
517  fclose(inFile);
518  }
519 
520  mFileNames.push_back(sname);
521  mFiles.push_back(inFile);
522  mDataSpecs.emplace_back(origin, desc);
523  return true;
524 }

526 //_____________________________________________________________________
527 bool RawFileReader::init()
528 {
529  // make initialization, preprocess files and chack for errors if asked
530 
531  for (int i = 0; i < NErrorsDefined; i++) {
532  if (mCheckErrors & (0x1 << i))
533  LOGF(INFO, "%s check for /%s/", (mCheckErrors & (0x1 << i)) ? "perform" : "ignore ", ErrNames[i].data());
534  }
535  if (mMaxTFToRead < 0xffffffff) {
536  LOGF(INFO, "at most %u TF will be processed", mMaxTFToRead);
537  }
538 
539  int nf = mFiles.size();
540  bool ok = true;
541  for (int i = 0; i < nf; i++) {
542  ok &= preprocessFile(i);
543  }
544  mOrderedIDs.resize(mLinksData.size());
545  for (int i = mLinksData.size(); i--;) {
546  mOrderedIDs[i] = i;
547  if (mNTimeFrames < mLinksData[i].nTimeFrames) {
548  mNTimeFrames = mLinksData[i].nTimeFrames;
549  }
550  }
551  std::sort(mOrderedIDs.begin(), mOrderedIDs.end(),
552  [& links = mLinksData](int a, int b) { return links[a].spec < links[b].spec; });
553 
554  size_t maxSP = 0, maxTF = 0;
555 
556  LOGF(INFO, "Summary of preprocessing:");
557  for (int i = 0; i < int(mLinksData.size()); i++) {
558  const auto& link = getLink(i);
559  auto msp = link.getLargestSuperPage();
560  auto mtf = link.getLargestTF();
561  if (maxSP < msp) {
562  maxSP = msp;
563  }
564  if (maxTF < mtf) {
565  maxTF = mtf;
566  }
567  std::stringstream counts;
568  counts << "Lnk" << std::setw(4) << std::left << i << "| ";
569  link.print(mVerbosity, counts.str());
570  if (msp > mNominalSPageSize) {
571  LOGF(DEBUG, " Attention: largest superpage %zu B exceeds expected %d B",
572  msp, mNominalSPageSize);
573  }
574  // min max orbits
575  if (link.blocks.front().orbit < mOrbitMin) {
576  mOrbitMin = link.blocks.front().orbit;
577  }
578  if (link.blocks.back().orbit > mOrbitMax) {
579  mOrbitMax = link.blocks.back().orbit;
580  }
581  if ((mCheckErrors & (0x1 << ErrWrongNumberOfTF)) && (mNTimeFrames != link.nTimeFrames)) {
582  LOGF(ERROR, "%s for %s: %u TFs while %u were seen for other links", ErrNames[ErrWrongNumberOfTF],
583  link.describe(), link.nTimeFrames, mNTimeFrames);
584  }
585  }
586  LOGF(INFO, "First orbit: %d, Last orbit: %d", mOrbitMin, mOrbitMax);
587  LOGF(INFO, "Largest super-page: %zu B, largest TF: %zu B", maxSP, maxTF);
588  if (!mCheckErrors) {
589  LOGF(INFO, "Detailed data format check was disabled");
590  }
591  mInitDone = true;
592 
593  return ok;
594 }

597 o2h::DataOrigin RawFileReader::getDataOrigin(const std::string& ors)
598 {
599  constexpr int NGoodOrigins = 18;
600  constexpr std::array<o2h::DataOrigin, NGoodOrigins> goodOrigins{

648 RawFileReader::InputsMap RawFileReader::parseInput(const std::string& confUri)
649 {
650  // read input files from configuration
651  std::map<std::pair<o2h::DataOrigin, o2h::DataDescription>, std::vector<std::string>> entries;
652 
653  if (confUri.empty()) {
654  throw std::runtime_error("Input configuration file is not provided");
655  }
656  std::string confFile = confUri;
657  ConfigFile cfg;
658  // ConfigFile file requires input as file:///absolute_path or file:local
659  if (confFile.rfind("file:", 0) != 0) {
660  confFile.insert(0, "file:");
661  }
662  try {
663  cfg.load(confFile);
664  } catch (std::string& e) { // unfortunately, the ConfigFile::load throws a string rather than the std::exception
665  throw std::runtime_error(std::string("Failed to parse configuration ") + confFile + " : " + e);
666  }
667  //
668  try {
669  std::string origStr, descStr, defstr = "defaults";
670  cfg.getOptionalValue<std::string>(defstr + ".dataOrigin", origStr, DEFDataOrigin.as<std::string>());
671  auto defDataOrigin = getDataOrigin(origStr);
672  if (defDataOrigin == o2h::gDataOriginInvalid) {
673  throw std::runtime_error(std::string("Invalid default data origin ") + origStr);
674  }
675  cfg.getOptionalValue<std::string>(defstr + ".dataDescription", descStr, DEFDataDescription.as<std::string>());
676  auto defDataDescription = getDataDescription(descStr);
677  if (defDataDescription == o2h::gDataDescriptionInvalid) {
678  throw std::runtime_error(std::string("Invalid default data description ") + descStr);
679  }
680  entries[{defDataOrigin, defDataDescription}]; // insert
681  LOG(DEBUG) << "Setting default dataOrigin/Description " << defDataOrigin.as<std::string>() << '/' << defDataDescription.as<std::string>();
682 
683  for (auto flsect : ConfigFileBrowser(&cfg, "input-")) {
684  std::string flNameStr, defs{""};
685  cfg.getOptionalValue<std::string>(flsect + ".dataOrigin", origStr, defDataOrigin.as<std::string>());
686  cfg.getOptionalValue<std::string>(flsect + ".dataDescription", descStr, defDataDescription.as<std::string>());
687  cfg.getOptionalValue<std::string>(flsect + ".filePath", flNameStr, defs);
688  if (flNameStr.empty()) {
689  LOG(DEBUG) << "Skipping incomplete input " << flsect;
690  continue;
691  }
692  auto dataOrigin = getDataOrigin(origStr);
693  if (dataOrigin == o2h::gDataOriginInvalid) {
694  throw std::runtime_error(std::string("Invalid data origin ") + origStr + " for " + flsect);
695  }
696  auto dataDescription = getDataDescription(descStr);
697  if (dataDescription == o2h::gDataDescriptionInvalid) {
698  throw std::runtime_error(std::string("Invalid data description ") + descStr + " for " + flsect);
699  }
700  entries[{dataOrigin, dataDescription}].push_back(flNameStr);
701  LOG(DEBUG) << "adding file " << flNameStr << " to dataOrigin/Description " << dataOrigin.as<std::string>() << '/' << dataDescription.as<std::string>();
702  }
703  } catch (std::string& e) { // to catch exceptions from the parser
704  throw std::runtime_error(std::string("Aborting due to the exception: ") + e);
705  }
706 
707  return entries;
708 }
