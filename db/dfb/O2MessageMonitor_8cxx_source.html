54 void O2MessageMonitor::Run()
55 {
56  // check socket type of data channel
57  std::string type;
58  std::vector<FairMQChannel>& subChannels = fChannels["data"];
59  if (subChannels.size() > 0) {
60  type = subChannels[0].GetType();
61  }
62 
63  auto dataResource = o2::pmr::getTransportAllocator(subChannels[0].Transport());
64 
65  while (FairMQ13<FairMQDevice>::IsRunning(this) && (--mIterations) != 0) {
66  this_thread::sleep_for(chrono::milliseconds(mDelay));
67 
68  O2Message message;
69 
70  // maybe send a request
71  if (type == "req") {
72  addDataBlock(message, {dataResource, DataHeader{gDataDescriptionInfo, gDataOriginAny, DataHeader::SubSpecificationType{0}}},
73  NewSimpleMessageFor("data", 0, mPayload));
74  Send(message, "data");
75  message.fParts.clear();
76  }
77 
78  // message in;
79  Receive(message, "data");
80  LOG(INFO) << "== New message=============================";
81  o2::base::forEach(message, [&](auto header, auto data) {
82  hexDump("headerBuffer", header.data(), header.size());
83  hexDump("dataBuffer", data.data(), data.size(), mLimitOutputCharacters);
84  });
85  message.fParts.clear();
86 
87  // maybe a reply message
88  if (type == "rep") {
89  o2::base::addDataBlock(message,
90  {dataResource, DataHeader{gDataDescriptionInfo, gDataOriginAny, DataHeader::SubSpecificationType{0}}},
91  NewSimpleMessageFor("data", 0, ""));
92  Send(message, "data");
93  }
94  }
95 }
