46 void RawWriter::run(ProcessingContext& pc)
47 {
48  if (mFinished) {
49  return;
50  }
51 
52  auto digits = pc.inputs().get<std::vector<o2::tof::Digit>*>("tofdigits");
53  auto row = pc.inputs().get<std::vector<o2::tof::ReadoutWindowData>*>("readoutwin");
54  int nwindow = row->size();
55  LOG(INFO) << "Encoding " << nwindow << " TOF readout windows";
56 
57  int cache = 1024 * 1024; // 1 MB
58  int verbosity = 0;
59 
60  o2::tof::raw::Encoder encoder;
61  encoder.setVerbose(verbosity);
62 
63  encoder.open(mOutFileName, mOutDirName);
64  encoder.alloc(cache);
65 
66  int nwindowperorbit = Geo::NWINDOW_IN_ORBIT;
67  int nwindowintimeframe = o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * nwindowperorbit;
68  int nwindowFilled = nwindow;
69  if (nwindowFilled % nwindowintimeframe) {
70  nwindowFilled = (nwindowFilled / nwindowintimeframe + 1) * nwindowintimeframe;
71  }
72 
73  std::vector<o2::tof::Digit> emptyWindow;
74 
75  std::vector<o2::tof::Digit> digitRO;
76 
77  std::vector<std::vector<o2::tof::Digit>> digitWindows;
78 
79  for (int i = 0; i < nwindow; i += nwindowperorbit) { // encode 3 tof windows (1 orbit)
80  if (verbosity)
81  printf("----------\nwindow = %d - %d\n----------\n", i, i + nwindowperorbit - 1);
82 
83  digitWindows.clear();
84 
85  // push all windows in the current orbit in the structure
86  for (int j = i; j < i + nwindowperorbit; j++) {
87  if (j < nwindow) {
88  digitRO.clear();
89  for (int id = 0; id < row->at(j).size(); id++)
90  digitRO.push_back((*digits)[row->at(j).first() + id]);
91  digitWindows.push_back(digitRO);
92  } else {
93  digitWindows.push_back(emptyWindow);
94  }
95  }
96 
97  encoder.encode(digitWindows, i);
98  }
99 
100  encoder.close();
101 
102  // create configuration file for rawreader
103  std::ofstream fconfig;
104  fconfig.open("TOFraw.cfg");
105 
106  for (int icru = 0; icru < encoder.getNCRU(); icru++) {
107  fconfig << "[input-tof-" << icru << "]\n";
108  fconfig << "dataOrigin = TOF\n";
109  fconfig << "dataDescription = RAWDATA\n";
110  fconfig << "filePath = " << mOutDirName << "/cru" << std::setfill('0') << std::setw(2) << icru << mOutFileName << "\n\n";
111  }
112  fconfig.close();
113 
114  mFinished = true;
115  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
116 }
