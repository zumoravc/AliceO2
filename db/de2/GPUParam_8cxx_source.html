123 void GPUParam::UpdateEventSettings(const GPUSettingsEvent* e, const GPUSettingsDeviceProcessing* p)
124 {
125  if (e) {
126  AssumeConstantBz = e->constBz;
127  ToyMCEventsFlag = e->homemadeEvents;
128  ContinuousTracking = e->continuousMaxTimeBin != 0;
129  continuousMaxTimeBin = e->continuousMaxTimeBin == -1 ? (0.023 * 5e6) : e->continuousMaxTimeBin;
130  polynomialField.Reset();
131  if (AssumeConstantBz) {
132  GPUTPCGMPolynomialFieldManager::GetPolynomialField(GPUTPCGMPolynomialFieldManager::kUniform, BzkG, polynomialField);
133  } else {
134  GPUTPCGMPolynomialFieldManager::GetPolynomialField(BzkG, polynomialField);
135  }
136  }
137  if (p) {
138  debugLevel = p->debugLevel;
139  resetTimers = p->resetTimers;
140  }
141  earlyTpcTransform = rec.ForceEarlyTPCTransform == -1 ? (!ContinuousTracking) : rec.ForceEarlyTPCTransform;
142 }
