98 void SemiregularSpline2D3D::construct(const int numberOfRowsInput, const int numbersOfKnots[])
99 {
104 
105  int numberOfRows = numberOfRowsInput;
106  if (numberOfRows < 5) {
107  numberOfRows = 5;
108  }
109 
110  FlatObject::startConstruction();
111 
112  //construct regular grid for v
113  mGridV.construct(numberOfRows);
114 
115  // For each x element numbersOfKnots may be a single RegularSpline1D with x knots.
116  // so first create the array
117  RegularSpline1D splineArray[numberOfRows];
118 
119  // And construct them
120  for (int i = 0; i < numberOfRowsInput; i++) {
121  splineArray[i].construct(numbersOfKnots[i]);
122  }
123  for (int i = numberOfRowsInput; i < numberOfRows; i++) {
124  splineArray[i].construct(5);
125  }
126 
127  // this is the space which is taken just by the RegularSpline1D's
128  mDataIndexMapOffset = numberOfRows * sizeof(RegularSpline1D);
129 
130  //The buffer size is the size of the array
131  FlatObject::finishConstruction(mDataIndexMapOffset + numberOfRows * sizeof(int));
132 
133  // Array for the 1D-Splines inside the buffer
134  RegularSpline1D* bufferSplines = getSplineArrayNonConst();
135 
136  // paste local splineArray to the buffer
137  for (int i = 0; i < numberOfRows; i++) {
138  bufferSplines[i] = splineArray[i];
139  }
140 
141  // Just calculating the total number of knots in this 2D3D spline.
142  int numberOfKnots = 0;
143  for (int i = 0; i < numberOfRows; i++) {
144  int knotsU = getGridU(i).getNumberOfKnots();
145  numberOfKnots += knotsU;
146  }
147 
148  //save the numberOfRows and numberOfKnots
149  mNumberOfRows = numberOfRows;
150  mNumberOfKnots = numberOfKnots;
151 
152  // map to save the starting data index for each v-coordinate
153  int* dataIndexMap = getDataIndexMapNonConst();
154 
155  // this will count the amount of u-knots "under" a v-coordinate
156  int uSum = 0;
157 
158  //count the amount of knots which are in gridU's lower than i
159  for (int dv = 0; dv < mNumberOfRows; dv++) {
160  dataIndexMap[dv] = uSum;
161  uSum += numbersOfKnots[dv];
162  }
163 }
