30 // Base primary vertex class, with position, error, N candidates and flags field
31 // The Stamp template parameter allows to define vertex (time)stamp in different
32 // formats (ITS ROFrame ID, real time + error etc)
33 
34 template <typename Stamp = o2::dataformats::TimeStamp<int>>
35 class Vertex
36 {
37  using ushort = unsigned short;
38 
39  public:
40  enum CovElems : int { kCovXX,
41  kCovXY,
42  kCovYY,
43  kCovXZ,
44  kCovYZ,
45  kCovZZ,
46  kNCov };
47  static ushort constexpr FlagsMask = 0xffff;
48 
49  Vertex() = default;
50  ~Vertex() = default;
51  Vertex(const Point3D<float>& pos, const std::array<float, kNCov>& cov, ushort nCont, float chi2)
52  : mPos(pos), mCov(cov), mNContributors(nCont), mChi2(chi2)
53  {
54  }
55 
56  void print() const;
57 
58  // getting the cartesian coordinates and errors
59  float getX() const { return mPos.X(); }
60  float getY() const { return mPos.Y(); }
61  float getZ() const { return mPos.Z(); }
62  float getSigmaX2() const { return mCov[kCovXX]; }
63  float getSigmaY2() const { return mCov[kCovYY]; }
64  float getSigmaZ2() const { return mCov[kCovZZ]; }
65  float getSigmaXY() const { return mCov[kCovXY]; }
66  float getSigmaXZ() const { return mCov[kCovXZ]; }
67  float getSigmaYZ() const { return mCov[kCovYZ]; }
68  const std::array<float, kNCov>& getCov() const { return mCov; }
69 
70  Point3D<float> getXYZ() const { return mPos; }
71  Point3D<float>& getXYZ() { return mPos; }
72 
73  void setX(float x) { mPos.SetX(x); }
74  void setY(float y) { mPos.SetY(y); }
75  void setZ(float z) { mPos.SetZ(z); }
76 
77  void setXYZ(float x, float y, float z)
78  {
79  setX(x);
80  setY(y);
81  setZ(z);
82  }
83  void setPos(const Point3D<float>& p) { mPos = p; }
84 
85  void setSigmaX2(float v) { mCov[kCovXX] = v; }
86  void setSigmaY2(float v) { mCov[kCovYY] = v; }
87  void setSigmaZ2(float v) { mCov[kCovZZ] = v; }
88  void setSigmaXY(float v) { mCov[kCovXY] = v; }
89  void setSigmaXZ(float v) { mCov[kCovXZ] = v; }
90  void setSigmaYZ(float v) { mCov[kCovYZ] = v; }
91  void setCov(float sxx, float sxy, float syy, float sxz, float syz, float szz)
92  {
93  setSigmaX2(sxx);
94  setSigmaY2(syy);
95  setSigmaZ2(szz);
96  setSigmaXY(sxy);
97  setSigmaXZ(sxz);
98  setSigmaYZ(syz);
99  }
100  void setCov(const std::array<float, kNCov>& cov) { mCov = cov; }
101 
102  ushort getNContributors() const { return mNContributors; }
103  void setNContributors(ushort v) { mNContributors = v; }
104 
105  ushort getBits() const { return mBits; }
106  bool isBitSet(int bit) const { return mBits & (FlagsMask & (0x1 << bit)); }
107  void setBits(ushort b) { mBits = b; }
108  void setBit(int bit) { mBits |= FlagsMask & (0x1 << bit); }
109  void resetBit(int bit) { mBits &= ~(FlagsMask & (0x1 << bit)); }
110 
111  void setChi2(float v) { mChi2 = v; }
112  float getChi2() const { return mChi2; }
113 
114  const Stamp& getTimeStamp() const { return mTimeStamp; }
115  Stamp& getTimeStamp() { return mTimeStamp; }
116  void setTimeStamp(const Stamp& v) { mTimeStamp = v; }
117 
118  private:
119  Point3D<float> mPos;
120  std::array<float, kNCov> mCov;
121  float mChi2 = 0;
122  ushort mNContributors = 0;
123  ushort mBits = 0;
124  Stamp mTimeStamp;
125 
126  ClassDefNV(Vertex, 1);
127 };
