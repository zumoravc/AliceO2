47 std::vector<InputSpec> DataDescriptorQueryBuilder::parse(char const* config)
48 {
49  std::vector<QueryBuilderState> states;
50  states.push_back(IN_BEGIN_QUERY);
51 
52  std::vector<InputSpec> result;
53  char const* next = config;
54  char const* cur = config;
55  char const* expectedSeparators;
56  char* endptr;
57  std::string errorString;
58  std::unique_ptr<DataDescriptorMatcher> currentMatcher;
59  std::vector<data_matcher::Node> nodes;
60  std::optional<std::string> currentBinding;
61  std::optional<std::string> currentOrigin;
62  std::optional<std::string> currentDescription;
63  std::optional<header::DataHeader::SubSpecificationType> currentSubSpec;
64  std::optional<uint64_t> currentTimeModulo;
65  size_t currentNumber;
66 
67  auto error = [&errorString, &states](std::string const& s) {
68  errorString = s;
69  states.push_back(IN_ERROR);
70  };
71 
72  auto pushState = [&states](QueryBuilderState state) {
73  states.push_back(state);
74  };
75 
76  auto token = [&states, &expectedSeparators](QueryBuilderState state, char const* sep) {
77  states.push_back(state);
78  expectedSeparators = sep;
79  };

81  auto assignLastStringMatch = [&next, &cur, &error, &pushState, &nodes](std::string const& what, size_t maxSize, std::optional<std::string>& s, QueryBuilderState nextState) {
82  if ((next - cur == 0) || (next - cur > maxSize)) {
83  error(what + " needs to be between 1 and " + std::to_string(maxSize) + " char long");
84  return false;
85  }
86  s = std::string(cur, next - cur);
87  if (memchr("\0;", *next, 2)) {
88  pushState(IN_END_MATCHER);
89  cur = next;
90  return true;
91  }
92  pushState(nextState);
93  cur = next + 1;
94  return true;
95  };
96 
97  auto assignLastNumericMatch = [&next, &cur, &error, &pushState, &currentNumber](std::string const& what, auto& value,
98  QueryBuilderState nextState) {
99  if ((next - cur == 0)) {
100  error("number expected");
101  return false;
102  }
103  value = currentNumber;
104  if (memchr("\0;", *next, 2)) {
105  pushState(IN_END_MATCHER);
106  cur = next;
107  return true;
108  }
109  pushState(nextState);
110  cur = next + 1;
111  return true;
112  };
113 
114  auto buildMatchingTree = [&nodes](std::string const& binding) -> InputSpec {
115  auto lastMatcher =
116  std::make_unique<DataDescriptorMatcher>(DataDescriptorMatcher::Op::Just,
117  StartTimeValueMatcher(ContextRef{0}));
118  for (size_t ni = 0, ne = nodes.size(); ni < ne; ++ni) {
119  auto& node = nodes[nodes.size() - 1 - ni];
120  auto tmp = std::make_unique<DataDescriptorMatcher>(DataDescriptorMatcher::Op::And,
121  std::move(node),
122  std::move(lastMatcher));
123  assert(lastMatcher.get() == nullptr);
124  lastMatcher = std::move(tmp);
125  }
126  return InputSpec{binding, std::move(*lastMatcher.release())};
127  };
128 
129  while (states.empty() == false) {
130  auto const state = states.back();
131  states.pop_back();
132 
133  switch (state) {
134  case IN_STRING: {
135  next = strpbrk(cur, expectedSeparators);
136  if (next == nullptr) {
137  next = cur + strlen(cur);
138  }
139  } break;
140  case IN_NUMBER: {
141  currentNumber = strtoll(cur, &endptr, 10);
142  if (endptr == cur) {
143  error("Expected a number");
144  }
145  next = endptr;
146  } break;
147  case IN_BEGIN_QUERY: {
148  (*cur == 0) ? pushState(IN_END_QUERY)
149  : pushState(IN_BEGIN_MATCHER);
150  } break;
151  case IN_BEGIN_MATCHER: {
152  nodes.clear();
153  pushState(IN_BEGIN_BINDING);
154  } break;
155  case IN_BEGIN_BINDING: {
156  pushState(IN_END_BINDING);
157  token(IN_STRING, ":/;,");
158  } break;
159  case IN_END_BINDING: {
160  // We are at the end of the string already.
161  // This is really an origin...
162  if (strchr("\0/;", *next)) {
163  pushState(IN_END_ORIGIN);
164  continue;
165  }
166  if (next - cur == 0) {
167  error("empty binding string");
168  continue;
169  }
170  currentBinding = std::string(cur, next - cur);
171  pushState(IN_BEGIN_ORIGIN);
172  cur = next + 1;
173  } break;
174  case IN_BEGIN_ORIGIN: {
175  pushState(IN_END_ORIGIN);
176  token(IN_STRING, "/;");
177  } break;
178  case IN_END_ORIGIN: {
179  if (assignLastStringMatch("origin", 4, currentOrigin, IN_BEGIN_DESCRIPTION)) {
180  nodes.push_back(OriginValueMatcher{*currentOrigin});
181  }
182  } break;
183  case IN_BEGIN_DESCRIPTION: {
184  pushState(IN_END_DESCRIPTION);
185  token(IN_STRING, "/;");
186  } break;
187  case IN_END_DESCRIPTION: {
188  if (assignLastStringMatch("description", 16, currentDescription, IN_BEGIN_SUBSPEC)) {
189  nodes.push_back(DescriptionValueMatcher{*currentDescription});
190  }
191  } break;
192  case IN_BEGIN_SUBSPEC: {
193  pushState(IN_END_SUBSPEC);
194  token(IN_NUMBER, ";%");
195  } break;
196  case IN_END_SUBSPEC: {
197  if (assignLastNumericMatch("subspec", currentSubSpec, IN_BEGIN_TIMEMODULO)) {
198  nodes.push_back(SubSpecificationTypeValueMatcher{*currentSubSpec});
199  }
200  } break;
201  case IN_BEGIN_TIMEMODULO: {
202  pushState(IN_END_TIMEMODULO);
203  token(IN_NUMBER, ";");
204  } break;
205  case IN_END_TIMEMODULO: {
206  assignLastNumericMatch("timemodulo", currentTimeModulo, IN_ERROR);
207  } break;
208  case IN_END_MATCHER: {
209  if (*cur == ';' && *(cur + 1) == '\0') {
210  error("Remove trailing ;");
211  continue;
212  }
213  result.push_back(buildMatchingTree(*currentBinding));
214  if (*cur == '\0') {
215  pushState(IN_END_QUERY);
216  } else if (*cur == ';') {
217  cur += 1;
218  pushState(IN_BEGIN_MATCHER);
219  } else {
220  error("Unexpected character" + std::string(cur, 1));
221  }
222  } break;
223  case IN_ERROR: {
224  throw std::runtime_error("Parse error: " + errorString);
225  } break;
226  case IN_END_QUERY: {
227  } break;
228  default: {
229  error("Unhandled state");
230  } break;
231  }
232  }
233  return std::move(result);
234 }

237 {

239 
240  std::sregex_token_iterator end;

242  config.end(),
243  delim,
244  -1);
245 
246  std::unique_ptr<DataDescriptorMatcher> next, result;
247 
248  for (; iter != end; ++iter) {
249  std::smatch m;
250  auto s = iter->str();
251  auto newNode = buildNode(s);
252 
253  if (result.get() == nullptr) {
254  result = std::move(newNode);
255  } else {
256  next = std::move(std::make_unique<DataDescriptorMatcher>(DataDescriptorMatcher::Op::Or,

260  }
261  }

309 std::unique_ptr<DataDescriptorMatcher> DataDescriptorQueryBuilder::buildNode(std::string const& nodeString)
310 {
311 
312  std::smatch m = getTokens(nodeString);
313 
314  std::unique_ptr<DataDescriptorMatcher> next;
315  auto newNode = std::make_unique<DataDescriptorMatcher>(
316  DataDescriptorMatcher::Op::And,
317  OriginValueMatcher{m[1]},
318  std::make_unique<DataDescriptorMatcher>(

321  SubSpecificationTypeValueMatcher{m[3]}));
322 
323  return newNode;
