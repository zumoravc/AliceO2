62 ImVec4 colorForLogLevel(LogParsingHelpers::LogLevel logLevel)
63 {
64  switch (logLevel) {
65  case LogParsingHelpers::LogLevel::Info:
66  return PaletteHelpers::GREEN;
67  case LogParsingHelpers::LogLevel::Debug:

80 void displayHistory(const DeviceInfo& info, DeviceControl& control)
81 {
82  if (info.history.empty()) {
83  return;
84  }
85  int startPos = info.historyPos;
86  const int historySize = info.history.size();
87 
88  int triggerStartPos = startPos + 1 % historySize;
89  int triggerStopPos = startPos % historySize;
90 
91  int j = startPos;
92  // We look for a stop trigger, so that we know where to stop the search for
93  // out start search. If no stop trigger is found, we search until the end
94  if (control.logStopTrigger[0]) {
95  while ((j % historySize) != ((startPos + 1) % historySize)) {
96  assert(j >= 0);
97  assert(j < historySize);
98  auto& line = info.history[j];
99  if (strstr(line.c_str(), control.logStopTrigger)) {
100  triggerStopPos = (j + 1) % historySize;
101  break;
102  }
103  // Wrap in case we end up below 0
104  j = (j == 0) ? historySize - 1 : j - 1;
105  }
106  }
107 
108  // Look for the last instance of the start trigger before the
109  // last stop trigger.
110  j = startPos + 1;
111  if (control.logStartTrigger[0]) {
112  while ((j % historySize) != triggerStopPos) {
113  assert(historySize > j);
114  assert(historySize == 1000);
115  auto& line = info.history[j];
116  if (strstr(line.c_str(), control.logStartTrigger)) {
117  triggerStartPos = j;
118  }
119  j = (j + 1) % historySize;
120  }
121  }
122 
123  // We start from the last trigger found. Eventually this is the first
124  // line in the ring buffer, if no trigger is specified.
125  size_t ji = triggerStartPos % historySize;
126  size_t je = triggerStopPos % historySize;
127  size_t iterations = 0;
128  while (historySize && ((ji % historySize) != je)) {
129  assert(iterations < historySize);
130  iterations++;
131  assert(historySize == 1000);
132  assert(ji < historySize);
133  auto& line = info.history[ji];
134  auto logLevel = info.historyLevel[ji];
135 
136  // Skip empty lines
137  if (line.empty()) {
138  ji = (ji + 1) % historySize;
139  continue;
140  }
141  // Print matching lines
142  if (strstr(line.c_str(), control.logFilter) != nullptr) {
143  auto color = colorForLogLevel(logLevel);
144  // We filter twice, once on input, to reduce the
145  // stream, a second time at display time, to avoid
146  // showing unrelevant messages from past.
147  if (logLevel >= control.logLevel) {
148  ImGui::TextColored(color, line.c_str(), line.c_str() + line.size());
149  }
150  }
151  ji = (ji + 1) % historySize;
152  }
153 }

170 void displayDeviceMetrics(const char* label, ImVec2 canvasSize, std::string const& selectedMetricName,
171  size_t rangeBegin, size_t rangeEnd, size_t bins, MetricsDisplayStyle displayType,
172  std::vector<DeviceSpec> const& specs, std::vector<DeviceMetricsInfo> const& metricsInfos)
173 {
174  static std::vector<ImColor> palette = {
175  ImColor{218, 124, 48},
176  ImColor{62, 150, 81},
177  ImColor{204, 37, 41},
178  ImColor{83, 81, 84},
179  ImColor{107, 76, 154},
180  ImColor{146, 36, 40},
181  ImColor{148, 139, 61}};
182  std::vector<void const*> metricsToDisplay;
183  std::vector<const char*> deviceNames;
184  std::vector<MultiplotData> userData;
185  std::vector<ImColor> colors;
186  MetricType metricType;
187  size_t metricSize = 0;
188  assert(specs.size() == metricsInfos.size());
189  float maxValue = std::numeric_limits<float>::lowest();
190  float minValue = 0;
191  size_t maxDomain = std::numeric_limits<size_t>::lowest();
192  size_t minDomain = std::numeric_limits<size_t>::max();
193 
194  for (int mi = 0; mi < metricsInfos.size(); ++mi) {
195  auto vi = DeviceMetricsHelper::metricIdxByName(selectedMetricName, metricsInfos[mi]);
196  if (vi == metricsInfos[mi].metricLabelsIdx.size()) {
197  continue;
198  }
199  auto& metric = metricsInfos[mi].metrics[vi];
200  deviceNames.push_back(specs[mi].name.c_str());
201  colors.push_back(palette[mi % palette.size()]);
202  metricType = metric.type;
203  MultiplotData data;
204  data.mod = metricsInfos[mi].timestamps[vi].size();
205  data.first = metric.pos - data.mod;
206  data.X = metricsInfos[mi].timestamps[vi].data();
207  minValue = std::min(minValue, metricsInfos[mi].min[vi]);
208  maxValue = std::max(maxValue, metricsInfos[mi].max[vi]);
209  minDomain = std::min(minDomain, metricsInfos[mi].minDomain[vi]);
210  maxDomain = std::max(maxDomain, metricsInfos[mi].maxDomain[vi]);
211  switch (metric.type) {
212  case MetricType::Int: {
213  data.size = metricsInfos[mi].intMetrics[metric.storeIdx].size();
214  data.Y = metricsInfos[mi].intMetrics[metric.storeIdx].data();
215  data.type = MetricType::Int;
216  metricType = MetricType::Int;
217  metricSize = metricsInfos[mi].intMetrics[metric.storeIdx].size();
218  } break;
219  case MetricType::Float: {
220  data.size = metricsInfos[mi].floatMetrics[metric.storeIdx].size();
221  data.Y = metricsInfos[mi].floatMetrics[metric.storeIdx].data();
222  data.type = MetricType::Float;
223  metricType = MetricType::Float;
224  metricSize = metricsInfos[mi].floatMetrics[metric.storeIdx].size();
225  } break;
226  case MetricType::Unknown:
227  case MetricType::String: {
228  data.size = metricsInfos[mi].stringMetrics[metric.storeIdx].size();
229  data.Y = nullptr;
230  data.type = MetricType::String;
231  metricType = MetricType::String;
232  metricSize = metricsInfos[mi].stringMetrics[metric.storeIdx].size();
233  } break;
234  }
235  userData.emplace_back(data);
236  }
237 
238  maxDomain = std::max(minDomain + 1024, maxDomain);
239 
240  for (size_t ui = 0; ui < userData.size(); ++ui) {
241  metricsToDisplay.push_back(&(userData[ui]));
242  }
243  auto getterY = [](const void* hData, int idx) -> float {
244  auto histoData = reinterpret_cast<const MultiplotData*>(hData);
245  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
246  // size_t pos = (static_cast<size_t>(idx)) % histoData->mod;
247  assert(pos >= 0 && pos < 1024);
248  if (histoData->type == MetricType::Int) {
249  return static_cast<const int*>(histoData->Y)[pos];
250  } else if (histoData->type == MetricType::Float) {
251  return static_cast<const float*>(histoData->Y)[pos];
252  } else {
253  return 0;
254  }
255  };
256  auto getterX = [](const void* hData, int idx) -> size_t {
257  auto histoData = reinterpret_cast<const MultiplotData*>(hData);
258  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
259  //size_t pos = (static_cast<size_t>(idx)) % histoData->mod;
260  assert(pos >= 0 && pos < 1024);
261  return static_cast<const size_t*>(histoData->X)[pos];
262  };
263  switch (displayType) {
264  case MetricsDisplayStyle::Histos:
265  ImGui::PlotMultiHistograms(
266  label,
267  userData.size(),
268  deviceNames.data(),
269  colors.data(),
270  getterY,
271  getterX,
272  metricsToDisplay.data(),
273  metricSize,
274  minValue,
275  maxValue * 1.2f,
276  minDomain,
277  maxDomain,
278  canvasSize);
279  break;
280  case MetricsDisplayStyle::Lines:
281  ImGui::PlotMultiLines(
282  label,
283  userData.size(),
284  deviceNames.data(),
285  colors.data(),
286  getterY,
287  getterX,
288  metricsToDisplay.data(),
289  metricSize,
290  minValue,
291  maxValue * 1.2f,
292  minDomain,
293  maxDomain,
294  canvasSize);
295  break;
296  default:
297  break;
298  }
299 }

306 void metricsTableRow(std::vector<ColumnInfo> columnInfos,
307  std::vector<DeviceMetricsInfo> const& metricsInfos,
308  int row)
309 {
310  ImGui::Text("%d", row);
311  ImGui::NextColumn();
312 
313  for (size_t j = 0; j < columnInfos.size(); j++) {
314  auto& info = columnInfos[j];
315  auto& metricsInfo = metricsInfos[j];
316 
317  if (info.index == -1) {
318  ImGui::TextUnformatted("-");
319  ImGui::NextColumn();
320  continue;
321  }
322  switch (info.type) {
323  case MetricType::Int: {
324  ImGui::Text("%i (%i)", metricsInfo.intMetrics[info.index][row], info.index);
325  ImGui::NextColumn();
326  } break;
327  case MetricType::Float: {
328  ImGui::Text("%f (%i)", metricsInfo.floatMetrics[info.index][row], info.index);
329  ImGui::NextColumn();
330  } break;
331  case MetricType::String: {
332  ImGui::Text("%s (%i)", metricsInfo.stringMetrics[info.index][row].data, info.index);
333  ImGui::NextColumn();
334  } break;
335  default:
336  ImGui::NextColumn();
337  break;
338  }

342 void historyBar(gui::WorkspaceGUIState& globalGUIState,
343  size_t rangeBegin, size_t rangeEnd,
344  gui::DeviceGUIState& state,
345  DriverInfo const& driverInfo,
346  DeviceSpec const& spec,
347  DeviceMetricsInfo const& metricsInfo)
348 {
349  bool open = ImGui::TreeNode(state.label.c_str());
350  if (open) {
351  ImGui::Text("# channels: %lu", spec.outputChannels.size() + spec.inputChannels.size());
352  ImGui::TreePop();
353  }
354  ImGui::NextColumn();
355 
356  if (globalGUIState.selectedMetric == -1) {
357  ImGui::NextColumn();
358  return;
359  }
360 
361  auto currentMetricName = driverInfo.availableMetrics[globalGUIState.selectedMetric];
362 
363  size_t i = DeviceMetricsHelper::metricIdxByName(currentMetricName, metricsInfo);
364  // We did not find any plot, skipping this.
365  if (i == metricsInfo.metricLabelsIdx.size()) {
366  ImGui::NextColumn();
367  return;
368  }
369  auto& metric = metricsInfo.metrics[i];
370 
371  switch (metric.type) {
372  case MetricType::Int: {
373  HistoData<int> data;
374  data.mod = metricsInfo.timestamps[i].size();
375  data.first = metric.pos - data.mod;
376  data.size = metricsInfo.intMetrics[metric.storeIdx].size();
377  data.points = metricsInfo.intMetrics[metric.storeIdx].data();
378 
379  auto getter = [](void* hData, int idx) -> float {
380  auto histoData = reinterpret_cast<HistoData<int>*>(hData);
381  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
382  assert(pos >= 0 && pos < 1024);
383  return histoData->points[pos];
384  };
385  ImGui::PlotLines(("##" + currentMetricName).c_str(), getter, &data, data.size);
386  ImGui::NextColumn();
387  } break;
388  case MetricType::Float: {
389  HistoData<float> data;
390  data.mod = metricsInfo.timestamps[i].size();
391  data.first = metric.pos - data.mod;
392  data.size = metricsInfo.floatMetrics[metric.storeIdx].size();
393  data.points = metricsInfo.floatMetrics[metric.storeIdx].data();
394 
395  auto getter = [](void* hData, int idx) -> float {
396  auto histoData = reinterpret_cast<HistoData<float>*>(hData);
397  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
398  assert(pos >= 0 && pos < 1024);
399  return histoData->points[pos];
400  };
401  ImGui::PlotLines(("##" + currentMetricName).c_str(), getter, &data, data.size);
402  ImGui::NextColumn();
403  } break;
404  default:
405  ImGui::NextColumn();
406  return;
407  break;
408  }
409 }

412 std::vector<ColumnInfo> calculateTableIndex(gui::WorkspaceGUIState& globalGUIState,
413  int selectedMetric,
414  DriverInfo const& driverInfo,
415  std::vector<DeviceMetricsInfo> const& metricsInfos)
416 {
417  std::vector<ColumnInfo> columns;
418  for (size_t j = 0; j < globalGUIState.devices.size(); ++j) {
419  const DeviceMetricsInfo& metricsInfo = metricsInfos[j];
421  if (selectedMetric == -1) {
422  columns.push_back({MetricType::Int, -1});
423  continue;
424  }
425  auto currentMetricName = driverInfo.availableMetrics[selectedMetric];
426  size_t idx = DeviceMetricsHelper::metricIdxByName(currentMetricName, metricsInfo);
427 
428  // We did not find any plot, skipping this.
429  if (idx == metricsInfo.metricLabelsIdx.size()) {
430  columns.push_back({MetricType::Int, -1});
431  continue;
432  }
433  auto metric = metricsInfos[j].metrics[idx];
434  columns.push_back({metric.type, static_cast<int>(metric.storeIdx)});
435  }
436  return columns;
437 };

440  DriverInfo const& driverInfo,
441  std::vector<DeviceInfo> const& infos,
442  std::vector<DeviceSpec> const& devices,
443  std::vector<DataProcessorInfo> const& metadata,

476  static char const* plotStyles[] = {
477  "lines",
478  "histograms",
479  "sparks",
480  "table"};
481  ImGui::SameLine();
482  static enum MetricsDisplayStyle currentStyle = MetricsDisplayStyle::Lines;
483  ImGui::Combo("##Select style", reinterpret_cast<int*>(&currentStyle), plotStyles, IM_ARRAYSIZE(plotStyles));
484 
485  // Calculate the full timestamp range for the selected metric
486  size_t minTime = -1;
487  size_t maxTime = 0;
488  std::string currentMetricName;
489  if (state.selectedMetric >= 0) {
490  currentMetricName = driverInfo.availableMetrics[state.selectedMetric];
491  for (auto& metricInfo : metricsInfos) {
492  size_t mi = DeviceMetricsHelper::metricIdxByName(currentMetricName, metricInfo);
493  if (mi == metricInfo.metricLabelsIdx.size()) {
494  continue;
495  }
496  auto& metric = metricInfo.metrics[mi];
497  auto& timestamps = metricInfo.timestamps[mi];
498 
499  for (size_t ti = 0; ti != metricInfo.timestamps.size(); ++ti) {
500  size_t minRangePos = (metric.pos + ti) % metricInfo.timestamps.size();
501  size_t curMinTime = timestamps[minRangePos];
502  if (curMinTime == 0) {
503  continue;
504  }
505  minTime = minTime < curMinTime ? minTime : curMinTime;
506  if (minTime != 0 && minTime != -1) {
507  break;
508  }
509  }
510  size_t maxRangePos = (size_t)(metric.pos) - 1 % metricInfo.timestamps.size();
511  size_t curMaxTime = timestamps[maxRangePos];
512  maxTime = maxTime > curMaxTime ? maxTime : curMaxTime;
513  }
514  }
515  if (minTime != -1) {
516  ImGui::Text("min timestamp: %zu, max timestamp: %zu", minTime, maxTime);
517  }
518  ImGui::EndGroup();
519  if (!currentMetricName.empty()) {
520  switch (currentStyle) {
521  case MetricsDisplayStyle::Histos:
522  case MetricsDisplayStyle::Lines: {
523  displayDeviceMetrics("Metrics",
524  ImVec2(ImGui::GetIO().DisplaySize.x - 10, state.bottomPaneSize - ImGui::GetItemRectSize().y - 20), currentMetricName, minTime, maxTime, 1024,
525  currentStyle, devices, metricsInfos);
526  } break;
527  case MetricsDisplayStyle::Sparks: {
528  ImGui::BeginChild("##ScrollingRegion", ImVec2(ImGui::GetIO().DisplaySize.x + state.leftPaneSize + state.rightPaneSize - 10, -ImGui::GetItemsLineHeightWithSpacing()), false,
529  ImGuiWindowFlags_HorizontalScrollbar);
530  ImGui::Columns(2);
531  ImGui::SetColumnOffset(1, 300);
532  for (size_t i = 0; i < state.devices.size(); ++i) {
533  gui::DeviceGUIState& deviceGUIState = state.devices[i];
534  const DeviceSpec& spec = devices[i];
535  const DeviceMetricsInfo& metricsInfo = metricsInfos[i];
536 
537  historyBar(state, minTime, maxTime, deviceGUIState, driverInfo, spec, metricsInfo);
538  }
539  ImGui::Columns(1);
540  ImGui::EndChild();
541  } break;
542  case MetricsDisplayStyle::Table: {
543  ImGui::BeginChild("##ScrollingRegion", ImVec2(ImGui::GetIO().DisplaySize.x + state.leftPaneSize + state.rightPaneSize - 10, -ImGui::GetItemsLineHeightWithSpacing()), false,
544  ImGuiWindowFlags_HorizontalScrollbar);
545 
546  // The +1 is for the timestamp column
547  ImGui::Columns(state.devices.size() + 1);
548  ImGui::TextUnformatted("entry");
549  ImGui::NextColumn();
550  ImVec2 textsize = ImGui::CalcTextSize("extry", nullptr, true);
551  float offset = 0.f;
552  offset += std::max(100.f, textsize.x);
553  for (size_t j = 0; j < state.devices.size(); ++j) {
554  gui::DeviceGUIState& deviceGUIState = state.devices[j];
555  const DeviceSpec& spec = devices[j];
556 
557  ImGui::SetColumnOffset(-1, offset);
558  textsize = ImGui::CalcTextSize(spec.name.c_str(), nullptr, true);
559  offset += std::max(100.f, textsize.x);
560  ImGui::TextUnformatted(spec.name.c_str());
561  ImGui::NextColumn();
562  }
563  ImGui::Separator();
564 
565  auto columns = calculateTableIndex(state, state.selectedMetric, driverInfo, metricsInfos);
566 
567  // Calculate which columns we want to see.
568  // FIXME: only one column for now.
569  for (size_t i = 0; i < 10; ++i) {
570  metricsTableRow(columns, metricsInfos, i);
571  }
572  ImGui::Columns(1);
573 
574  ImGui::EndChild();
575  } break;
576  }
577  }
578  ImGui::End();
579 }
