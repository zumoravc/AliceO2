36 void RCUTrailer::constructFromRawPayload(const gsl::span<const uint32_t> payloadwords)
37 {
38  reset();
39  int index = payloadwords.size() - 1;
40  auto word = payloadwords[index];
41  if ((word >> 30) != 3)
42  throw Error(Error::ErrorType_t::DECODING_INVALID, "Last RCU trailer word not found!");
43  mFirmwareVersion = (word >> 16) & 0xFF;
44 
45  mRCUId = (int)((word >> 7) & 0x1FF);
46  int trailerSize = (word & 0x7F);
47 
48  if (trailerSize < 2)
49  throw Error(Error::ErrorType_t::SIZE_INVALID, fmt::format("Invalid trailer size found (%d bytes) !", trailerSize * 4).data());
50  mTrailerSize = trailerSize;
51 
52  for (; trailerSize > 0; trailerSize--) {
53  word = payloadwords[--index];
54  if ((word >> 30) != 2) {
55  std::cerr << "Missing RCU trailer identifier pattern!\n";
56  continue;
57  }
58  int parCode = (word >> 26) & 0xF;
59  int parData = word & 0x3FFFFFF;
60  switch (parCode) {
61  case 1:
62  // ERR_REG1
63  mFECERRA = ((parData >> 13) & 0x1FFF) << 7;
64  mFECERRB = ((parData & 0x1FFF)) << 7;
65  break;
66  case 2:
67  // ERR_REG2
68  mERRREG2 = parData & 0x1FF;
69  break;
70  case 3:
71  // ERR_REG3
72  mERRREG3 = parData & 0x1FFFFFF;
73  break;
74  case 4:
75  // FEC_RO_A
76  mActiveFECsA = parData & 0xFFFF;
77  break;
78  case 5:
79  // FEC_RO_B
80  mActiveFECsB = parData & 0xFFFF;
81  break;
82  case 6:
83  // RDO_CFG1
84  mAltroCFG1 = parData & 0xFFFFF;
85  break;
86  case 7:
87  // RDO_CFG2
88  mAltroCFG2 = parData & 0x1FFFFFF;
89  break;
90  default:
91  std::cerr << "Undefined parameter code " << parCode << ", ignore it !\n";
92  break;
93  }
94  }
95  mPayloadSize = payloadwords[--index] & 0x3FFFFFF;
96  mIsInitialized = true;
97 }

99 double RCUTrailer::getTimeSample() const
100 {
101  unsigned char fq = (mAltroCFG2 >> 5) & 0xF;
102  double tSample;
103  switch (fq) {
104  case 0:
105  // 20 MHz
106  tSample = 2.0;
107  break;
108  case 1:
109  // 10 Mhz
110  tSample = 4.0;
111  break;
112  case 2:
113  // 5 MHz
114  tSample = 8.;
115  break;
116  default:
117  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("Invalid sampling frequency value %d !", int(fq)).data());
118  }
119 
120  return tSample * o2::constants::lhc::LHCBunchSpacingNS * 1.e-9;
121 }

123 void RCUTrailer::setTimeSample(double timesample)
124 {
125  int fq = 0;
126  if (std::abs(timesample - 50) < DBL_EPSILON)
127  fq = 0;
128  else if (std::abs(timesample - 100) < DBL_EPSILON)
129  fq = 1;
130  else if (std::abs(timesample - 200) < DBL_EPSILON)
131  fq = 2;
132  else
133  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("invalid time sample: %f", timesample).data());
134  mAltroCFG2 = (mAltroCFG2 & 0x1F) | fq << 5;
135 }

137 double RCUTrailer::getL1Phase() const
138 {
139  double tSample = getTimeSample(),
140  phase = ((double)(mAltroCFG2 & 0x1F)) * o2::constants::lhc::LHCBunchSpacingNS * 1.e-9;
141  if (phase >= tSample) {
142  throw Error(Error::ErrorType_t::L1PHASE_INVALID, fmt::format("Invalid L1 trigger phase (%e s (phase) >= %e s (sampling time)) !", phase, tSample).data());
143  }
144  return phase;
145 }

153 std::vector<uint32_t> RCUTrailer::encode() const
154 {
155  std::vector<uint32_t> encoded;
156  encoded.emplace_back(mAltroCFG2 | 7 << 26);
157  encoded.emplace_back(mAltroCFG1 | 6 << 26);
158  encoded.emplace_back(mActiveFECsB | 5 << 26);
159  encoded.emplace_back(mActiveFECsA | 4 << 26);
160  encoded.emplace_back(mERRREG3 | 3 << 26);
161  encoded.emplace_back(mERRREG2 | 2 << 26);
162  encoded.emplace_back(mFECERRB >> 7 | (mFECERRA >> 7) << 13 | 1 << 26);
163 
164  uint32_t lasttrailerword = 3 << 30 | mFirmwareVersion << 16 | mRCUId << 7 | encoded.size();
165  encoded.emplace_back(lasttrailerword);
166 
167  return encoded;
168 }

170 void RCUTrailer::printStream(std::ostream& stream) const
171 {
172  std::vector<std::string> errors;
173  double timesample = -1., l1phase = -1.;
174  try {
175  timesample = getTimeSample();
176  } catch (Error& e) {
177  errors.push_back(e.what());
178  }
179  try {
180  l1phase = getL1Phase();
181  } catch (Error& e) {
182  errors.push_back(e.what());
183  }
184 
185  stream << "RCU trailer (Format version 2):\n"
186  << "==================================================\n"
187  << "RCU ID: " << mRCUId << "\n"
188  << "Firmware version: " << int(mFirmwareVersion) << "\n"
189  << "Trailer size: " << mTrailerSize << "\n"
190  << "Payload size: " << mPayloadSize << "\n"
191  << "FECERRA: 0x" << std::hex << mFECERRA << "\n"
192  << "FECERRB: 0x" << std::hex << mFECERRB << "\n"
193  << "ERRREG2: 0x" << std::hex << mERRREG2 << "\n"
194  << "#channels skipped due to address mismatch: " << std::dec << getNumberOfChannelAddressMismatch() << "\n"
195  << "#channels skipped due to bad block length: " << std::dec << getNumberOfChannelLengthMismatch() << "\n"
196  << "Active FECs (branch A): 0x" << std::hex << mActiveFECsA << "\n"
197  << "Active FECs (branch B): 0x" << std::hex << mActiveFECsB << "\n"
198  << "Baseline corr: 0x" << std::hex << int(getBaselineCorrection()) << "\n"
199  << "Number of presamples: " << std::dec << int(getNumberOfPresamples()) << "\n"
200  << "Number of postsamples: " << std::dec << int(getNumberOfPostsamples()) << "\n"
201  << "Second baseline corr: " << (hasSecondBaselineCorr() ? "yes" : "no") << "\n"
202  << "GlitchFilter: " << std::dec << int(getGlitchFilter()) << "\n"
203  << "Number of non-ZS postsamples: " << std::dec << int(getNumberOfNonZeroSuppressedPostsamples()) << "\n"
204  << "Number of non-ZS presamples: " << std::dec << int(getNumberOfNonZeroSuppressedPresamples()) << "\n"
205  << "Number of ALTRO buffers: " << std::dec << getNumberOfAltroBuffers() << "\n"
206  << "Number of pretrigger samples: " << std::dec << int(getNumberOfPretriggerSamples()) << "\n"
207  << "Number of samples per channel: " << std::dec << getNumberOfSamplesPerChannel() << "\n"
208  << "Sparse readout: " << (isSparseReadout() ? "yes" : "no") << "\n"
209  << "AltroCFG1: 0x" << std::hex << mAltroCFG1 << "\n"
210  << "AltroCFG2: 0x" << std::hex << mAltroCFG2 << "\n"
211  << "Sampling time: " << std::scientific << timesample << " s\n"
212  << "L1 Phase: " << std::scientific << l1phase << " s\n"
213  << std::dec << std::fixed;
214  if (errors.size()) {
215  stream << "Errors: \n"
216  << "-------------------------------------------------\n";
217  for (const auto& e : errors) {
218  stream << e << "\n";
219  }
220  }
221  stream << "==================================================\n";
222 }
