24 TrackPar::TrackPar(const array<float, 3>& xyz, const array<float, 3>& pxpypz, int charge, bool sectorAlpha)
25  : mX{0.f}, mAlpha{0.f}, mP{0.f}
26 {
27  // construct track param from kinematics
28 
29  // Alpha of the frame is defined as:
30  // sectorAlpha == false : -> angle of pt direction
31  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
32  // angle of pt direction for r==0
33  //
34  //
35  constexpr float kSafe = 1e-5f;
36  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
37  float alp = 0;
38  if (sectorAlpha || radPos2 < 1) {
39  alp = atan2f(pxpypz[1], pxpypz[0]);
40  } else {
41  alp = atan2f(xyz[1], xyz[0]);
42  }
43  if (sectorAlpha) {
44  alp = utils::Angle2Alpha(alp);
45  }
46  //
47  float sn, cs;
48  utils::sincosf(alp, sn, cs);
49  // protection: avoid alpha being too close to 0 or +-pi/2
50  if (fabs(sn) < 2 * kSafe) {
51  if (alp > 0) {
52  alp += alp < PIHalf ? 2 * kSafe : -2 * kSafe;
53  } else {
54  alp += alp > -PIHalf ? -2 * kSafe : 2 * kSafe;
55  }
56  utils::sincosf(alp, sn, cs);
57  } else if (fabs(cs) < 2 * kSafe) {
58  if (alp > 0) {
59  alp += alp > PIHalf ? 2 * kSafe : -2 * kSafe;
60  } else {
61  alp += alp > -PIHalf ? 2 * kSafe : -2 * kSafe;
62  }
63  utils::sincosf(alp, sn, cs);
64  }
65  // get the vertex of origin and the momentum
66  array<float, 3> ver{xyz[0], xyz[1], xyz[2]};
67  array<float, 3> mom{pxpypz[0], pxpypz[1], pxpypz[2]};
68  //
69  // Rotate to the local coordinate system
70  utils::RotateZ(ver, -alp);
71  utils::RotateZ(mom, -alp);
72  //
73  float ptI = 1.f / sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
74  mX = ver[0];
75  mAlpha = alp;
76  mP[kY] = ver[1];
77  mP[kZ] = ver[2];
78  mP[kSnp] = mom[1] * ptI;
79  mP[kTgl] = mom[2] * ptI;
80  mP[kQ2Pt] = ptI * charge;
81  //
82  if (fabs(1 - getSnp()) < kSafe) {
83  mP[kSnp] = 1. - kSafe; // Protection
84  } else if (fabs(-1 - getSnp()) < kSafe) {
85  mP[kSnp] = -1. + kSafe; // Protection
86  }
87  //
88 }

91 bool TrackPar::getPxPyPzGlo(array<float, 3>& pxyz) const
92 {
93  // track momentum
94  if (fabs(getQ2Pt()) < Almost0 || fabs(getSnp()) > Almost1) {
95  return false;
96  }
97  float cs, sn, pt = fabs(1.f / getQ2Pt());
98  float r = sqrtf((1.f - getSnp()) * (1.f + getSnp()));
99  utils::sincosf(getAlpha(), sn, cs);
100  pxyz[0] = pt * (r * cs - getSnp() * sn);
101  pxyz[1] = pt * (getSnp() * cs + r * sn);
102  pxyz[2] = pt * getTgl();
103  return true;
104 }

107 bool TrackPar::getPosDirGlo(array<float, 9>& posdirp) const
108 {
109  // fill vector with lab x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha
110  float ptI = fabs(getQ2Pt());
111  float snp = getSnp();
112  if (ptI < Almost0 || fabs(snp) > Almost1) {
113  return false;
114  }
115  float &sn = posdirp[7], &cs = posdirp[8];
116  float csp = sqrtf((1.f - snp) * (1.f + snp));
117  float cstht = sqrtf(1.f + getTgl() * getTgl());
118  float csthti = 1.f / cstht;
119  utils::sincosf(getAlpha(), sn, cs);
120  posdirp[0] = getX() * cs - getY() * sn;
121  posdirp[1] = getX() * sn + getY() * cs;
122  posdirp[2] = getZ();
123  posdirp[3] = (csp * cs - snp * sn) * csthti; // px/p
124  posdirp[4] = (snp * cs + csp * sn) * csthti; // py/p
125  posdirp[5] = getTgl() * csthti; // pz/p
126  posdirp[6] = cstht / ptI; // p
127  return true;
128 }

131 bool TrackPar::rotateParam(float alpha)
132 {
133  // rotate to alpha frame
134  if (fabs(getSnp()) > Almost1) {
135  LOGF(WARNING, "Precondition is not satisfied: |sin(phi)|>1 ! {:f}", getSnp());
136  return false;
137  }
138  //
139  utils::BringToPMPi(alpha);
140  //
141  float ca = 0, sa = 0;
142  utils::sincosf(alpha - getAlpha(), sa, ca);
143  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
144  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
145  // direction in local frame is along the X axis
146  if ((csp * ca + snp * sa) < 0) {
147  //LOGF(WARNING,"Rotation failed: local cos(phi) would become {:.2f}", csp * ca + snp * sa);
148  return false;
149  }
150  //
151  float tmp = snp * ca - csp * sa;
152  if (fabs(tmp) > Almost1) {
153  LOGF(WARNING, "Rotation failed: new snp {:.2f}", tmp);
154  return false;
155  }
156  float xold = getX(), yold = getY();
157  mAlpha = alpha;
158  mX = xold * ca + yold * sa;
159  mP[kY] = -xold * sa + yold * ca;
160  mP[kSnp] = tmp;
161  return true;
162 }

165 bool TrackPar::propagateParamTo(float xk, const array<float, 3>& b)
166 {
167  //----------------------------------------------------------------
168  // Extrapolate this track params (w/o cov matrix) to the plane X=xk in the field b[].
169  //
170  // X [cm] is in the "tracking coordinate system" of this track.
171  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
172  //----------------------------------------------------------------
173 
174  float dx = xk - getX();
175  if (fabs(dx) < Almost0) {
176  return true;
177  }
178  // Do not propagate tracks outside the ALICE detector
179  if (fabs(dx) > 1e5 || fabs(getY()) > 1e5 || fabs(getZ()) > 1e5) {
180  LOGF(WARNING, "Anomalous track, target X:{:f}", xk);
181  // print();
182  return false;
183  }
184  float crv = (fabs(b[2]) < Almost0) ? 0.f : getCurvature(b[2]);
185  float x2r = crv * dx;
186  float f1 = getSnp(), f2 = f1 + x2r;
187  if (fabs(f1) > Almost1 || fabs(f2) > Almost1) {
188  return false;
189  }
190  float r1 = sqrtf((1.f - f1) * (1.f + f1));
191  if (fabs(r1) < Almost0) {
192  return false;
193  }
194  float r2 = sqrtf((1.f - f2) * (1.f + f2));
195  if (fabs(r2) < Almost0) {
196  return false;
197  }
198  float dy2dx = (f1 + f2) / (r1 + r2);
199  float step = (fabs(x2r) < 0.05f) ? dx * fabs(r2 + f2 * dy2dx) // chord
200  : 2.f * asinf(0.5f * dx * sqrtf(1.f + dy2dx * dy2dx) * crv) / crv; // arc
201  step *= sqrtf(1.f + getTgl() * getTgl());
202  //
203  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
204  array<float, 9> vecLab{0.f};
205  if (!getPosDirGlo(vecLab)) {
206  return false;
207  }
208 
209  // rotate to the system where Bx=By=0.
210  float bxy2 = b[0] * b[0] + b[1] * b[1];
211  float bt = sqrtf(bxy2);
212  float cosphi = 1.f, sinphi = 0.f;
213  if (bt > Almost0) {
214  cosphi = b[0] / bt;
215  sinphi = b[1] / bt;
216  }
217  float bb = sqrtf(bxy2 + b[2] * b[2]);
218  float costet = 1., sintet = 0.;
219  if (bb > Almost0) {
220  costet = b[2] / bb;
221  sintet = bt / bb;
222  }
223  array<float, 7> vect{costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
224  -sinphi * vecLab[0] + cosphi * vecLab[1],
225  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
226  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
227  -sinphi * vecLab[3] + cosphi * vecLab[4],
228  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
229  vecLab[6]};
230 
231  // Do the helix step
232  float sgn = getSign();
233  g3helx3(sgn * bb, step, vect);
234 
235  // rotate back to the Global System
236  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
237  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
238  vecLab[2] = -sintet * vect[0] + costet * vect[2];
239 
240  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
241  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
242  vecLab[5] = -sintet * vect[3] + costet * vect[5];
243 
244  // rotate back to the Tracking System
245  float sinalp = -vecLab[7], cosalp = vecLab[8];
246  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
247  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
248  vecLab[0] = t;
249  t = cosalp * vecLab[3] - sinalp * vecLab[4];
250  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
251  vecLab[3] = t;
252 
253  // Do the final correcting step to the target plane (linear approximation)
254  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
255  if (fabs(dx) > Almost0) {
256  if (fabs(vecLab[3]) < Almost0) {
257  return false;
258  }
259  dx = xk - vecLab[0];
260  x += dx;
261  y += vecLab[4] / vecLab[3] * dx;
262  z += vecLab[5] / vecLab[3] * dx;
263  }
264 
265  // Calculate the track parameters
266  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
267  mX = x;
268  mP[kY] = y;
269  mP[kZ] = z;
270  mP[kSnp] = vecLab[4] * t;
271  mP[kTgl] = vecLab[5] * t;
272  mP[kQ2Pt] = sgn * t / vecLab[6];
273 
274  return true;
275 }

278 bool TrackPar::propagateParamTo(float xk, float b)
279 {
280  //----------------------------------------------------------------
281  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
282  // Only parameters are propagated, not the matrix. To be used for small
283  // distances only (<mm, i.e. misalignment)
284  //----------------------------------------------------------------
285  float dx = xk - getX();
286  if (fabs(dx) < Almost0) {
287  return true;
288  }
289  float crv = (fabs(b) < Almost0) ? 0.f : getCurvature(b);
290  float x2r = crv * dx;
291  float f1 = getSnp(), f2 = f1 + x2r;
292  if ((fabs(f1) > Almost1) || (fabs(f2) > Almost1)) {
293  return false;
294  }
295  float r1 = sqrtf((1.f - f1) * (1.f + f1));
296  if (fabs(r1) < Almost0) {
297  return false;
298  }
299  float r2 = sqrtf((1.f - f2) * (1.f + f2));
300  if (fabs(r2) < Almost0) {
301  return false;
302  }
303  mX = xk;
304  double dy2dx = (f1 + f2) / (r1 + r2);
305  mP[kY] += dx * dy2dx;
306  mP[kSnp] += x2r;
307  if (fabs(x2r) < 0.05f) {
308  mP[kZ] += dx * (r2 + f2 * dy2dx) * getTgl();
309  } else {
310  // for small dx/R the linear apporximation of the arc by the segment is OK,
311  // but at large dx/R the error is very large and leads to incorrect Z propagation
312  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
313  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
314  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
315  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
316  // track1 += rot/crv*track3;
317  //
318  float rot = asinf(r1 * f2 - r2 * f1); // more economic version from Yura.
319  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
320  if (f2 > 0.f) {
321  rot = PI - rot; //
322  } else {
323  rot = -PI - rot;
324  }
325  }
326  mP[kZ] += getTgl() / crv * rot;
327  }
328  return true;
329 }

332 bool TrackPar::getYZAt(float xk, float b, float& y, float& z) const
333 {
334  //----------------------------------------------------------------
335  // estimate Y,Z in tracking frame at given X
336  //----------------------------------------------------------------
337  float dx = xk - getX();
338  if (fabs(dx) < Almost0) {
339  return true;
340  }
341  float crv = (fabs(b) < Almost0) ? 0.f : getCurvature(b);
342  float x2r = crv * dx;
343  float f1 = getSnp(), f2 = f1 + x2r;
344  if ((fabs(f1) > Almost1) || (fabs(f2) > Almost1)) {
345  return false;
346  }
347  if (fabs(getQ2Pt()) < Almost0) {
348  return false;
349  }
350  float r1 = sqrtf((1.f - f1) * (1.f + f1));
351  if (fabs(r1) < Almost0) {
352  return false;
353  }
354  float r2 = sqrtf((1.f - f2) * (1.f + f2));
355  if (fabs(r2) < Almost0) {
356  return false;
357  }
358  double dy2dx = (f1 + f2) / (r1 + r2);
359  y = mP[kY] + dx * dy2dx;
360  z = mP[kZ];
361  if (fabs(x2r) < 0.05f) {
362  z += dx * (r2 + f2 * dy2dx) * getTgl();
363  } else {
364  // for small dx/R the linear apporximation of the arc by the segment is OK,
365  // but at large dx/R the error is very large and leads to incorrect Z propagation
366  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
367  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
368  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
369  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
370  // track1 += rot/crv*track3;
371  //
372  float rot = asinf(r1 * f2 - r2 * f1); // more economic version from Yura.

375  rot = PI - rot; //
376  } else {
377  rot = -PI - rot;
378  }
379  }
380  z += getTgl() / crv * rot;
381  }
382  return true;
383 }

432 bool TrackPar::getXatLabR(float r, float& x, float bz, o2::track::DirType dir) const
433 {
434  // Get local X of the track position estimated at the radius lab radius r.
435  // The track curvature is accounted exactly
436  //
437  // The flag "dir" can be used to remove the ambiguity of which intersection to take (out of 2 possible)
438  // DirAuto (==0) - take the intersection closest to the current track position
439  // DirOutward (==1) - go along the track (increasing mX)
440  // DirInward (==-1) - go backward (decreasing mX)
441  //
442  const auto fy = mP[0], sn = mP[2];
443  const float kEps = 1.e-6;
444  //
445  auto crv = getCurvature(bz);
446  if (fabs(crv) > o2::constants::math::Almost0) { // helix
447  // get center of the track circle
448  o2::utils::CircleXY circle;
449  getCircleParamsLoc(bz, circle);
450  auto r0 = sqrtf(circle.getCenterD2());
451  if (r0 <= o2::constants::math::Almost0) {
452  return false; // the track is concentric to circle
453  }
454  float tR2r0 = 1., g = 0., tmp = 0.;
455  if (fabs(circle.rC - r0) > kEps) {
456  tR2r0 = circle.rC / r0;
457  g = 0.5 * (r * r / (r0 * circle.rC) - tR2r0 - 1. / tR2r0);
458  tmp = 1. + g * tR2r0;
459  } else {
460  tR2r0 = 1.0;
461  g = 0.5 * r * r / (r0 * circle.rC) - 1.;
462  tmp = 0.5 * r * r / (r0 * r0);
463  }
464  auto det = (1. - g) * (1. + g);
465  if (det < 0.) {
466  return false; // does not reach raduis r
467  }
468  det = sqrtf(det);
469  //
470  // the intersection happens in 2 points: {circle.xC+tR*C,circle.yC+tR*S}
471  // with C=f*c0+-|s0|*det and S=f*s0-+c0 sign(s0)*det
472  // where s0 and c0 make direction for the circle center (=circle.xC/r0 and circle.yC/r0)
473  //
474  x = circle.xC * tmp;
475  auto y = circle.yC * tmp;
476  if (fabs(circle.yC) > o2::constants::math::Almost0) { // when circle.yC==0 the x,y is unique
477  auto dfx = tR2r0 * fabs(circle.yC) * det;
478  auto dfy = tR2r0 * circle.xC * (circle.yC > 0. ? det : -det);
479  if (dir == DirAuto) { // chose the one which corresponds to smallest step
480  auto delta = (x - mX) * dfx - (y - fy) * dfy; // the choice of + in C will lead to smaller step if delta<0
481  x += delta < 0. ? dfx : -dfx;
482  } else if (dir == DirOutward) { // along track direction: x must be > mX
483  x -= dfx; // try the smallest step (dfx is positive)
484  auto dfeps = mX - x; // handle special case of very small step
485  if (dfeps < -kEps) {
486  return true;
487  }
488  if (fabs(dfeps) < kEps && fabs(mX * mX + fy * fy - r * r) < kEps) { // are we already in right r?
489  return mX;
490  }
491  x += dfx + dfx;
492  auto dxm = x - mX;
493  if (dxm > 0.) {
494  return true;
495  } else if (dxm < -kEps) {
496  return false;
497  }
498  x = mX; // don't move
499  } else { // backward: x must be < mX
500  x += dfx; // try the smallest step (dfx is positive)
501  auto dfeps = x - mX; // handle special case of very small step
502  if (dfeps < -kEps) {
503  return true;
504  }
505  if (fabs(dfeps) < kEps && fabs(mX * mX + fy * fy - r * r) < kEps) { // are we already in right r?
506  return mX;
507  }
508  x -= dfx + dfx;
509  auto dxm = x - mX;
510  if (dxm < 0.) {
511  return true;
512  }
513  if (dxm > kEps) {
514  return false;
515  }
516  x = mX; // don't move
517  }
518  } else { // special case: track touching the circle just in 1 point
519  if ((dir == DirOutward && x < mX) || (dir == DirInward && x > mX)) {
520  return false;
521  }
522  }
523  } else { // this is a straight track
524  if (fabs(sn) >= o2::constants::math::Almost1) { // || to Y axis
525  auto det = (r - mX) * (r + mX);
526  if (det < 0.) {
527  return false; // does not reach raduis r
528  }
529  x = mX;
530  if (dir == DirAuto) {
531  return true;
532  }
533  det = sqrtf(det);
534  if (dir == DirOutward) { // along the track direction
535  if (sn > 0.) {
536  if (fy > det) {
537  return false; // track is along Y axis and above the circle
538  }
539  } else {
540  if (fy < -det) {
541  return false; // track is against Y axis amd belo the circle
542  }
543  }
544  } else if (dir == DirInward) { // against track direction
545  if (sn > 0.) {
546  if (fy < -det) {
547  return false; // track is along Y axis
548  }
549  } else if (fy > det) {
550  return false; // track is against Y axis
551  }
552  }
553  } else if (fabs(sn) <= o2::constants::math::Almost0) { // || to X axis
554  auto det = (r - fy) * (r + fy);
555  if (det < 0.) {
556  return false; // does not reach raduis r
557  }
558  det = sqrtf(det);
559  if (dir == DirAuto) {
560  x = mX > 0. ? det : -det; // choose the solution requiring the smalest step
561  return true;
562  } else if (dir == DirOutward) { // along the track direction
563  if (mX > det) {
564  return false; // current point is in on the right from the circle
565  } else {
566  x = (mX < -det) ? -det : det; // on the left : within the circle
567  }
568  } else { // against the track direction
569  if (mX < -det) {
570  return false;
571  } else {
572  x = mX > det ? det : -det;
573  }
574  }
575  } else { // general case of straight line
576  auto cs = sqrtf((1 - sn) * (1 + sn));
577  auto xsyc = mX * sn - fy * cs;
578  auto det = (r - xsyc) * (r + xsyc);
579  if (det < 0.) {
580  return false; // does not reach raduis r
581  }
582  det = sqrtf(det);
583  auto xcys = mX * cs + fy * sn;
584  auto t = -xcys;
585  if (dir == DirAuto) {
586  t += t > 0. ? -det : det; // chose the solution requiring the smalest step
587  } else if (dir > 0) { // go in increasing mX direction. ( t+-det > 0)
588  if (t >= -det) {
589  t += -det; // take minimal step giving t>0
590  } else {
591  return false; // both solutions have negative t
592  }
593  } else { // go in increasing mX direction. (t+-det < 0)
594  if (t < det) {
595  t -= det; // take minimal step giving t<0
596  } else {
597  return false; // both solutions have positive t
598  }
599  }
600  x = mX + cs * t;
601  }
602  }
603  //
604  return true;
605 }

622 bool TrackParCov::propagateTo(float xk, float b)
623 {
624  //----------------------------------------------------------------
625  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
626  //----------------------------------------------------------------
627  float dx = xk - getX();
628  if (fabs(dx) < Almost0) {
629  return true;
630  }
631  float crv = (fabs(b) < Almost0) ? 0.f : getCurvature(b);
632  float x2r = crv * dx;
633  float f1 = getSnp(), f2 = f1 + x2r;
634  if ((fabs(f1) > Almost1) || (fabs(f2) > Almost1)) {
635  return false;
636  }
637  float r1 = sqrtf((1.f - f1) * (1.f + f1));
638  if (fabs(r1) < Almost0) {
639  return false;
640  }
641  float r2 = sqrtf((1.f - f2) * (1.f + f2));
642  if (fabs(r2) < Almost0) {
643  return false;
644  }
645  setX(xk);
646  double dy2dx = (f1 + f2) / (r1 + r2);
647  float dP[kNParams] = {0.f};
648  dP[kY] = dx * dy2dx;
649  dP[kSnp] = x2r;
650  if (fabs(x2r) < 0.05f) {
651  dP[kZ] = dx * (r2 + f2 * dy2dx) * getTgl();
652  } else {
653  // for small dx/R the linear apporximation of the arc by the segment is OK,
654  // but at large dx/R the error is very large and leads to incorrect Z propagation
655  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
656  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
657  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
658  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
659  // mP1 += rot/crv*mP3;
660  //
661  float rot = asinf(r1 * f2 - r2 * f1); // more economic version from Yura.
662  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
663  if (f2 > 0.f) {
664  rot = PI - rot; //
665  } else {
666  rot = -PI - rot;
667  }
668  }
669  dP[kZ] = getTgl() / crv * rot;
670  }
671 
672  updateParams(dP); // apply corrections
673 
674  float &c00 = mC[kSigY2], &c10 = mC[kSigZY], &c11 = mC[kSigZ2], &c20 = mC[kSigSnpY], &c21 = mC[kSigSnpZ],
675  &c22 = mC[kSigSnp2], &c30 = mC[kSigTglY], &c31 = mC[kSigTglZ], &c32 = mC[kSigTglSnp], &c33 = mC[kSigTgl2],
676  &c40 = mC[kSigQ2PtY], &c41 = mC[kSigQ2PtZ], &c42 = mC[kSigQ2PtSnp], &c43 = mC[kSigQ2PtTgl],
677  &c44 = mC[kSigQ2Pt2];
678 
679  // evaluate matrix in double prec.
680  double rinv = 1. / r1;
681  double r3inv = rinv * rinv * rinv;
682  double f24 = dx * b * B2C; // x2r/mP[kQ2Pt];
683  double f02 = dx * r3inv;
684  double f04 = 0.5 * f24 * f02;
685  double f12 = f02 * getTgl() * f1;
686  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
687  double f13 = dx * rinv;
688 
689  // b = C*ft
690  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
691  double b02 = f24 * c40;
692  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
693  double b12 = f24 * c41;
694  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
695  double b22 = f24 * c42;
696  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
697  double b42 = f24 * c44;
698  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
699  double b32 = f24 * c43;
700 
701  // a = f*b = f*C*ft
702  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
703  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
704  double a22 = f24 * b42;
705 
706  // F*C*Ft = C + (b + bt + a)
707  c00 += b00 + b00 + a00;
708  c10 += b10 + b01 + a01;
709  c20 += b20 + b02 + a02;
710  c30 += b30;
711  c40 += b40;
712  c11 += b11 + b11 + a11;
713  c21 += b21 + b12 + a12;
714  c31 += b31;
715  c41 += b41;
716  c22 += b22 + b22 + a22;
717  c32 += b32;
718  c42 += b42;
719 
720  checkCovariance();
721 
722  return true;
723 }

726 bool TrackParCov::rotate(float alpha)
727 {
728  // rotate to alpha frame
729  if (fabs(getSnp()) > Almost1) {
730  LOGF(ERROR, "Precondition is not satisfied: |sin(phi)|>1 ! {:f}", getSnp());
731  return false;
732  }
733  //
734  utils::BringToPMPi(alpha);
735  //
736  float ca = 0, sa = 0;
737  utils::sincosf(alpha - getAlpha(), sa, ca);
738  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
739  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
740  // direction in local frame is along the X axis
741  if ((csp * ca + snp * sa) < 0) {
742  //LOGF(WARNING,"Rotation failed: local cos(phi) would become {:.2f}", csp * ca + snp * sa);
743  return false;
744  }
745  //
746 
747  float updSnp = snp * ca - csp * sa;
748  if (fabs(updSnp) > Almost1) {
749  LOGF(WARNING, "Rotation failed: new snp {:.2f}", updSnp);
750  return false;
751  }
752  float xold = getX(), yold = getY();
753  setAlpha(alpha);
754  setX(xold * ca + yold * sa);
755  setY(-xold * sa + yold * ca);
756  setSnp(updSnp);
757 
758  if (fabs(csp) < Almost0) {
759  LOGF(WARNING, "Too small cosine value {:f}", csp);
760  csp = Almost0;
761  }
762 
763  float rr = (ca + snp / csp * sa);
764 
765  mC[kSigY2] *= (ca * ca);
766  mC[kSigZY] *= ca;
767  mC[kSigSnpY] *= ca * rr;
768  mC[kSigSnpZ] *= rr;
769  mC[kSigSnp2] *= rr * rr;
770  mC[kSigTglY] *= ca;
771  mC[kSigTglSnp] *= rr;
772  mC[kSigQ2PtY] *= ca;
773  mC[kSigQ2PtSnp] *= rr;
774 
775  checkCovariance();
776  return true;
777 }

780 TrackParCov::TrackParCov(const array<float, 3>& xyz, const array<float, 3>& pxpypz,
781  const array<float, kLabCovMatSize>& cv, int charge, bool sectorAlpha)
782 {
783  // construct track param and covariance from kinematics and lab errors
784 
785  // Alpha of the frame is defined as:
786  // sectorAlpha == false : -> angle of pt direction
787  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
788  // angle of pt direction for r==0
789  //
790  //
791  constexpr float kSafe = 1e-5f;
792  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
793  float alp = 0;
794  if (sectorAlpha || radPos2 < 1) {
795  alp = atan2f(pxpypz[1], pxpypz[0]);
796  } else {
797  alp = atan2f(xyz[1], xyz[0]);
798  }
799  if (sectorAlpha) {
800  alp = utils::Angle2Alpha(alp);
801  }
802  //
803  float sn, cs;
804  utils::sincosf(alp, sn, cs);
805  // protection: avoid alpha being too close to 0 or +-pi/2
806  if (fabs(sn) < 2.f * kSafe) {
807  if (alp > 0) {
808  alp += alp < PIHalf ? 2.f * kSafe : -2.f * kSafe;
809  } else {
810  alp += alp > -PIHalf ? -2.f * kSafe : 2.f * kSafe;
811  }
812  utils::sincosf(alp, sn, cs);
813  } else if (fabs(cs) < 2.f * kSafe) {
814  if (alp > 0) {
815  alp += alp > PIHalf ? 2.f * kSafe : -2.f * kSafe;
816  } else {
817  alp += alp > -PIHalf ? 2.f * kSafe : -2.f * kSafe;
818  }
819  utils::sincosf(alp, sn, cs);
820  }
821  // get the vertex of origin and the momentum
822  array<float, 3> ver{xyz[0], xyz[1], xyz[2]};
823  array<float, 3> mom{pxpypz[0], pxpypz[1], pxpypz[2]};
824  //
825  // Rotate to the local coordinate system
826  utils::RotateZ(ver, -alp);
827  utils::RotateZ(mom, -alp);
828  //
829  float pt = sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
830  float ptI = 1.f / pt;
831  setX(ver[0]);
832  setAlpha(alp);
833  setY(ver[1]);
834  setZ(ver[2]);
835  setSnp(mom[1] * ptI); // cos(phi)
836  setTgl(mom[2] * ptI); // tg(lambda)
837  setQ2Pt(ptI * charge);
838  //
839  if (fabs(1.f - getSnp()) < kSafe) {
840  setSnp(1.f - kSafe); // Protection
841  } else if (fabs(-1.f - getSnp()) < kSafe) {
842  setSnp(-1.f + kSafe); // Protection
843  }
844  //
845  // Covariance matrix (formulas to be simplified)
846  float r = mom[0] * ptI; // cos(phi)
847  float cv34 = sqrtf(cv[3] * cv[3] + cv[4] * cv[4]);
848  //
849  int special = 0;
850  float sgcheck = r * sn + getSnp() * cs;
851  if (fabs(sgcheck) > 1 - kSafe) { // special case: lab phi is +-pi/2
852  special = 1;
853  sgcheck = sgcheck < 0 ? -1.f : 1.f;
854  } else if (fabs(sgcheck) < kSafe) {
855  sgcheck = cs < 0 ? -1.0f : 1.0f;
856  special = 2; // special case: lab phi is 0
857  }
858  //
859  mC[kSigY2] = cv[0] + cv[2];
860  mC[kSigZY] = (-cv[3] * sn) < 0 ? -cv34 : cv34;
861  mC[kSigZ2] = cv[5];
862  //
863  float ptI2 = ptI * ptI;
864  float tgl2 = getTgl() * getTgl();
865  if (special == 1) {
866  mC[kSigSnpY] = cv[6] * ptI;
867  mC[kSigSnpZ] = -sgcheck * cv[8] * r * ptI;
868  mC[kSigSnp2] = fabs(cv[9] * r * r * ptI2);
869  mC[kSigTglY] = (cv[10] * getTgl() - sgcheck * cv[15]) * ptI / r;
870  mC[kSigTglZ] = (cv[17] - sgcheck * cv[12] * getTgl()) * ptI;
871  mC[kSigTglSnp] = (-sgcheck * cv[18] + cv[13] * getTgl()) * r * ptI2;
872  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[19] * mC[4] + cv[14] * tgl2) * ptI2;
873  mC[kSigQ2PtY] = cv[10] * ptI2 / r * charge;
874  mC[kSigQ2PtZ] = -sgcheck * cv[12] * ptI2 * charge;
875  mC[kSigQ2PtSnp] = cv[13] * r * ptI * ptI2 * charge;
876  mC[kSigQ2PtTgl] = (-sgcheck * cv[19] + cv[14] * getTgl()) * r * ptI2 * ptI;
877  mC[kSigQ2Pt2] = fabs(cv[14] * ptI2 * ptI2);
878  } else if (special == 2) {
879  mC[kSigSnpY] = -cv[10] * ptI * cs / sn;
880  mC[kSigSnpZ] = cv[12] * cs * ptI;
881  mC[kSigSnp2] = fabs(cv[14] * cs * cs * ptI2);
882  mC[kSigTglY] = (sgcheck * cv[6] * getTgl() - cv[15]) * ptI / sn;
883  mC[kSigTglZ] = (cv[17] - sgcheck * cv[8] * getTgl()) * ptI;
884  mC[kSigTglSnp] = (cv[19] - sgcheck * cv[13] * getTgl()) * cs * ptI2;
885  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[18] * getTgl() + cv[9] * tgl2) * ptI2;
886  mC[kSigQ2PtY] = sgcheck * cv[6] * ptI2 / sn * charge;
887  mC[kSigQ2PtZ] = -sgcheck * cv[8] * ptI2 * charge;
888  mC[kSigQ2PtSnp] = -sgcheck * cv[13] * cs * ptI * ptI2 * charge;
889  mC[kSigQ2PtTgl] = (-sgcheck * cv[18] + cv[9] * getTgl()) * ptI2 * ptI * charge;
890  mC[kSigQ2Pt2] = fabs(cv[9] * ptI2 * ptI2);
891  } else {
892  double m00 = -sn; // m10=cs;
893  double m23 = -pt * (sn + getSnp() * cs / r), m43 = -pt * pt * (r * cs - getSnp() * sn);
894  double m24 = pt * (cs - getSnp() * sn / r), m44 = -pt * pt * (r * sn + getSnp() * cs);
895  double m35 = pt, m45 = -pt * pt * getTgl();
896  //
897  m43 *= charge;
898  m44 *= charge;
899  m45 *= charge;
900  //
901  double a1 = cv[13] - cv[9] * (m23 * m44 + m43 * m24) / m23 / m43;
902  double a2 = m23 * m24 - m23 * (m23 * m44 + m43 * m24) / m43;
903  double a3 = m43 * m44 - m43 * (m23 * m44 + m43 * m24) / m23;
904  double a4 = cv[14] + 2. * cv[9];
905  double a5 = m24 * m24 - 2. * m24 * m44 * m23 / m43;
906  double a6 = m44 * m44 - 2. * m24 * m44 * m43 / m23;
907  //
908  mC[kSigSnpY] = (cv[10] * m43 - cv[6] * m44) / (m24 * m43 - m23 * m44) / m00;
909  mC[kSigQ2PtY] = (cv[6] / m00 - mC[kSigSnpY] * m23) / m43;
910  mC[kSigTglY] = (cv[15] / m00 - mC[kSigQ2PtY] * m45) / m35;
911  mC[kSigSnpZ] = (cv[12] * m43 - cv[8] * m44) / (m24 * m43 - m23 * m44);
912  mC[kSigQ2PtZ] = (cv[8] - mC[kSigSnpZ] * m23) / m43;
913  mC[kSigTglZ] = cv[17] / m35 - mC[kSigQ2PtZ] * m45 / m35;
914  mC[kSigSnp2] = fabs((a4 * a3 - a6 * a1) / (a5 * a3 - a6 * a2));
915  mC[kSigQ2Pt2] = fabs((a1 - a2 * mC[kSigSnp2]) / a3);
916  mC[kSigQ2PtSnp] = (cv[9] - mC[kSigSnp2] * m23 * m23 - mC[kSigQ2Pt2] * m43 * m43) / m23 / m43;
917  double b1 = cv[18] - mC[kSigQ2PtSnp] * m23 * m45 - mC[kSigQ2Pt2] * m43 * m45;
918  double b2 = m23 * m35;
919  double b3 = m43 * m35;
920  double b4 = cv[19] - mC[kSigQ2PtSnp] * m24 * m45 - mC[kSigQ2Pt2] * m44 * m45;
921  double b5 = m24 * m35;
922  double b6 = m44 * m35;
923  mC[kSigTglSnp] = (b4 - b6 * b1 / b3) / (b5 - b6 * b2 / b3);
924  mC[kSigQ2PtTgl] = b1 / b3 - b2 * mC[kSigTglSnp] / b3;
925  mC[kSigTgl2] = fabs((cv[20] - mC[kSigQ2Pt2] * (m45 * m45) - mC[kSigQ2PtTgl] * 2. * m35 * m45) / (m35 * m35));
926  }
927  checkCovariance();
928 }

983  double r3inv = rinv * rinv * rinv;
984  double f24 = dx * b[2] * B2C; // x2r/track[kQ2Pt];
985  double f02 = dx * r3inv;
986  double f04 = 0.5 * f24 * f02;
987  double f12 = f02 * getTgl() * f1;
988  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
989  double f13 = dx * rinv;
990 
991  // b = C*ft
992  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
993  double b02 = f24 * c40;
994  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
995  double b12 = f24 * c41;
996  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
997  double b22 = f24 * c42;
998  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
999  double b42 = f24 * c44;
1000  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
1001  double b32 = f24 * c43;
1002 
1003  // a = f*b = f*C*ft
1004  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
1005  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
1006  double a22 = f24 * b42;
1007 
1008  // F*C*Ft = C + (b + bt + a)
1009  c00 += b00 + b00 + a00;
1010  c10 += b10 + b01 + a01;
1011  c20 += b20 + b02 + a02;
1012  c30 += b30;
1013  c40 += b40;
1014  c11 += b11 + b11 + a11;
1015  c21 += b21 + b12 + a12;
1016  c31 += b31;
1017  c41 += b41;
1018  c22 += b22 + b22 + a22;
1019  c32 += b32;
1020  c42 += b42;
1021 
1022  checkCovariance();
1023 
1024  // Rotate to the system where Bx=By=0.
1025  float bxy2 = b[0] * b[0] + b[1] * b[1];
1026  float bt = sqrtf(bxy2);
1027  float cosphi = 1.f, sinphi = 0.f;
1028  if (bt > Almost0) {
1029  cosphi = b[0] / bt;
1030  sinphi = b[1] / bt;
1031  }
1032  float bb = sqrtf(bxy2 + b[2] * b[2]);
1033  float costet = 1., sintet = 0.;
1034  if (bb > Almost0) {
1035  costet = b[2] / bb;
1036  sintet = bt / bb;
1037  }
1038  array<float, 7> vect{costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
1039  -sinphi * vecLab[0] + cosphi * vecLab[1],
1040  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
1041  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
1042  -sinphi * vecLab[3] + cosphi * vecLab[4],
1043  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
1044  vecLab[6]};
1045 
1046  // Do the helix step
1047  float sgn = getSign();
1048  g3helx3(sgn * bb, step, vect);
1049 
1050  // Rotate back to the Global System
1051  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
1052  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
1053  vecLab[2] = -sintet * vect[0] + costet * vect[2];
1054 
1055  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
1056  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
1057  vecLab[5] = -sintet * vect[3] + costet * vect[5];
1058 
1059  // Rotate back to the Tracking System
1060  float sinalp = -vecLab[7], cosalp = vecLab[8];
1061  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
1062  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
1063  vecLab[0] = t;
1064  t = cosalp * vecLab[3] - sinalp * vecLab[4];
1065  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
1066  vecLab[3] = t;
1067 
1068  // Do the final correcting step to the target plane (linear approximation)
1069  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
1070  if (fabs(dx) > Almost0) {
1071  if (fabs(vecLab[3]) < Almost0) {
1072  return false;
1073  }
1074  dx = xk - vecLab[0];
1075  x += dx;
1076  y += vecLab[4] / vecLab[3] * dx;
1077  z += vecLab[5] / vecLab[3] * dx;
1078  }
1079 
1080  // Calculate the track parameters
1081  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
1082  setX(x);
1083  setY(y);
1084  setZ(z);
1085  setSnp(vecLab[4] * t);
1086  setTgl(vecLab[5] * t);
1087  setQ2Pt(sgn * t / vecLab[6]);
1088 
1089  return true;
1090 }

1182 float TrackParCov::getPredictedChi2(const array<float, 2>& p, const array<float, 3>& cov) const
1183 {

1191  return VeryBig;

1195  float z = getZ() - p[1];
1196

1198 }

1261 bool TrackParCov::update(const TrackParCov& rhs, const MatrixDSym5& covInv)
1262 {

1264

1272  return false;
1273  }
1274 
1275  // gain matrix K = Cov0*H*(Cov0+Cov0)^-1 (for measurement matrix H=I)
1276  MatrixDSym5 matC0;
1277  matC0(kY, kY) = getSigmaY2();
1278  matC0(kZ, kY) = getSigmaZY();
1279  matC0(kZ, kZ) = getSigmaZ2();

1293

1295  // RS: why SMatix, SVector does not provide multiplication operators ???
1296  double diff[kNParams];
1297  for (int i = kNParams; i--;) {

1315  mC[kSigTglZ] -= matK(kTgl, kZ);

1319  mC[kSigQ2PtZ] -= matK(kQ2Pt, kZ);
1320  mC[kSigQ2PtSnp] -= matK(kQ2Pt, kSnp);
1321  mC[kSigQ2PtTgl] -= matK(kQ2Pt, kTgl);
1322  mC[kSigQ2Pt2] -= matK(kQ2Pt, kQ2Pt);

1324  return true;
1325 }

1435  if (anglecorr) {

1437  x2x0 *= angle;
1438  xrho *= angle;
1439  }
1440

1443  p += p; // q=2 particle
1444  }

1462  cC33 = t2c2I * cst2I;

1479  }
1480  }

1483  float e = sqrtf(e2);
1484  if (fabs(dE) > kMaxELossFrac * e) {
1485  return false; // 30% energy loss is too much!
1486  }

1488  float pupd2 = eupd * eupd - mass2;
1489  if (pupd2 < kMinP * kMinP) {
1490  return false;
1491  }
1492  cP4 = p / sqrtf(pupd2);

1494  // Approximate energy loss fluctuation (M.Ivanov)
1495  constexpr float knst = 0.07f; // To be tuned.
1496  float sigmadE = knst * sqrtf(fabs(dE)) * e / p2 * getQ2Pt();

1498  }
1499

1501  fC22 += cC22;
1502  fC33 += cC33;
1503  fC43 += cC43;
1504  fC44 += cC44;
1505  setQ2Pt(getQ2Pt() * cP4);

1542 void o2::track::g3helx3(float qfield, float step, array<float, 7>& vect)
1543 {
1544  /******************************************************************
1545  * *
1546  * GEANT3 tracking routine in a constant field oriented *
1547  * along axis 3 *
1548  * Tracking is performed with a conventional *
1549  * helix step method *
1550  * *
1551  * Authors R.Brun, M.Hansroul ********* *
1552  * Rewritten V.Perevoztchikov *
1553  * *
1554  * Rewritten in C++ by I.Belikov *
1555  * *
1556  * qfield (kG) - particle charge times magnetic field *
1557  * step (cm) - step length along the helix *
1558  * vect[7](cm,GeV/c) - input/output x, y, z, px/p, py/p ,pz/p, p *
1559  * *
1560  ******************************************************************/
1561  const int ix = 0, iy = 1, iz = 2, ipx = 3, ipy = 4, ipz = 5, ipp = 6;
1562  constexpr float kOvSqSix = 0.408248f; // sqrtf(1./6.);
1563 
1564  float cosx = vect[ipx], cosy = vect[ipy], cosz = vect[ipz];
1565 
1566  float rho = qfield * B2C / vect[ipp];
1567  float tet = rho * step;
1568 
1569  float tsint, sintt, sint, cos1t;
1570  if (fabs(tet) > 0.03f) {
1571  sint = sinf(tet);
1572  sintt = sint / tet;
1573  tsint = (tet - sint) / tet;
1574  float t = sinf(0.5f * tet);
1575  cos1t = 2 * t * t / tet;
1576  } else {
1577  tsint = tet * tet / 6.f;
1578  sintt = (1.f - tet * kOvSqSix) * (1.f + tet * kOvSqSix); // 1.- tsint;
1579  sint = tet * sintt;
1580  cos1t = 0.5f * tet;
1581  }
1582 
1583  float f1 = step * sintt;
1584  float f2 = step * cos1t;
1585  float f3 = step * tsint * cosz;
1586  float f4 = -tet * cos1t;
1587  float f5 = sint;
1588 
1589  vect[ix] += f1 * cosx - f2 * cosy;
1590  vect[iy] += f1 * cosy + f2 * cosx;
1591  vect[iz] += f1 * cosz + f3;
1592 
1593  vect[ipx] += f4 * cosx - f5 * cosy;
1594  vect[ipy] += f4 * cosy + f5 * cosx;
1595 }

1598 float o2::track::BetheBlochSolid(float bg, float rho, float kp1, float kp2, float meanI, float meanZA)
1599 {
1600  //
1601  // This is the parameterization of the Bethe-Bloch formula inspired by Geant.
1602  //
1603  // bg - beta*gamma
1604  // rho - density [g/cm^3]
1605  // kp1 - density effect first junction point
1606  // kp2 - density effect second junction point
1607  // meanI - mean excitation energy [GeV]
1608  // meanZA - mean Z/A
1609  //
1610  // The default values for the kp* parameters are for silicon.
1611  // The returned value is in [GeV/(g/cm^2)].
1612  //
1613  constexpr float mK = 0.307075e-3f; // [GeV*cm^2/g]
1614  constexpr float me = 0.511e-3f; // [GeV/c^2]
1615  kp1 *= 2.303f;
1616  kp2 *= 2.303f;
1617  float bg2 = bg * bg;
1618  float maxT = 2.f * me * bg2; // neglecting the electron mass
1619 
1620  //*** Density effect
1621  float d2 = 0.;
1622  const float x = log(bg);
1623  const float lhwI = log(28.816 * 1e-9 * sqrtf(rho * meanZA) / meanI);
1624  if (x > kp2) {
1625  d2 = lhwI + x - 0.5;
1626  } else if (x > kp1) {
1627  double r = (kp2 - x) / (kp2 - kp1);
1628  d2 = lhwI + x - 0.5 + (0.5 - lhwI - kp1) * r * r * r;
1629  }
1630  return mK * meanZA * (1 + bg2) / bg2 * (0.5 * log(2 * me * bg2 * maxT / (meanI * meanI)) - bg2 / (1 + bg2) - d2);
1631 }
