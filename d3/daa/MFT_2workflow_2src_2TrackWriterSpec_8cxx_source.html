46 void TrackWriter::run(ProcessingContext& pc)
47 {
48  if (mState != 1)
49  return;
50 
51  auto tracks = pc.inputs().get<const std::vector<o2::mft::TrackMFT>>("tracks");
52  auto tracksltf = pc.inputs().get<const std::vector<o2::mft::TrackLTF>>("tracksltf");
53  auto tracksca = pc.inputs().get<const std::vector<o2::mft::TrackCA>>("tracksca");
54  auto rofs = pc.inputs().get<const std::vector<o2::itsmft::ROFRecord>>("ROframes");
55 
56  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> labels;
57  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* plabels = nullptr;
58 
59  LOG(INFO) << "MFTTrackWriter pulled "
60  << tracks.size() << " tracks, "
61  << tracksltf.size() << " tracks LTF, "
62  << tracksca.size() << " tracks CA, in "
63  << rofs.size() << " RO frames";
64 
65  TTree tree("o2sim", "Tree with MFT tracks");
66  tree.Branch("MFTTrack", &tracks);
67  tree.Branch("MFTTrackLTF", &tracksltf);
68  tree.Branch("MFTTrackCA", &tracksca);
69  if (mUseMC) {
70  labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
71  plabels = labels.get();
72  tree.Branch("MFTTrackMCTruth", &plabels);
73  }
74  // write ROFrecords vector to a tree
75  auto* rofsPtr = &rofs;
76  tree.Branch("MFTTracksROF", &rofsPtr);
77 
78  std::vector<o2::itsmft::MC2ROFRecord> mc2rofs, *mc2rofsPtr = &mc2rofs;
79  if (mUseMC) {
80  // write MC2ROFrecord vector (directly inherited from digits input) to a tree
81  const auto m2rvec = pc.inputs().get<gsl::span<o2::itsmft::MC2ROFRecord>>("MC2ROframes");
82  mc2rofs.reserve(m2rvec.size());
83  for (const auto& m2rv : m2rvec) {
84  mc2rofs.push_back(m2rv);
85  }
86  tree.Branch("MFTTracksMC2ROF", &mc2rofsPtr);
87  }
88 
89  tree.Fill();
90  tree.Write();
91  mFile->Close();
92 
93  mState = 2;
94  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
95 }

97 DataProcessorSpec getTrackWriterSpec(bool useMC)
98 {
99  std::vector<InputSpec> inputs;
100  inputs.emplace_back("tracks", "MFT", "TRACKS", 0, Lifetime::Timeframe);
101  inputs.emplace_back("tracksltf", "MFT", "TRACKSLTF", 0, Lifetime::Timeframe);
102  inputs.emplace_back("tracksca", "MFT", "TRACKSCA", 0, Lifetime::Timeframe);
103  inputs.emplace_back("ROframes", "MFT", "MFTTrackROF", 0, Lifetime::Timeframe);
104  if (useMC) {
105  inputs.emplace_back("labels", "MFT", "TRACKSMCTR", 0, Lifetime::Timeframe);
106  inputs.emplace_back("MC2ROframes", "MFT", "MFTTrackMC2ROF", 0, Lifetime::Timeframe);
107  }
108 
109  return DataProcessorSpec{
110  "mft-track-writer",
111  inputs,
112  Outputs{},
113  AlgorithmSpec{adaptFromTask<TrackWriter>(useMC)},
114  Options{
115  {"mft-track-outfile", VariantType::String, "mfttracks.root", {"Name of the output file"}}}};
116 }
