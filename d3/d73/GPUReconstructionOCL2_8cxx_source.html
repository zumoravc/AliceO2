59 int GPUReconstructionOCL2Backend::GetOCLPrograms()
60 {
61  char platform_version[64] = {}, platform_vendor[64] = {};
62  clGetPlatformInfo(mInternals->platform, CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
63  clGetPlatformInfo(mInternals->platform, CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
64  float ver = 0;
65  sscanf(platform_version, "OpenCL %f", &ver);
66 
67  cl_int return_status[1] = {CL_SUCCESS};
68  cl_int ocl_error;
69 #ifdef OPENCL2_ENABLED_AMD
70  if (strcmp(platform_vendor, "Advanced Micro Devices, Inc.") == 0) {
71  size_t program_sizes[1] = {_makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_amd_size};
72  char* program_binaries[1] = {_makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_amd};
73  mInternals->program = clCreateProgramWithBinary(mInternals->context, 1, &mInternals->device, program_sizes, (const unsigned char**)program_binaries, return_status, &ocl_error);
74  } else
75 #endif
76 
77 #ifdef OPENCL2_ENABLED_SPIRV // clang-format off
78  if (ver >= 2.2) {
79  mInternals->program = clCreateProgramWithIL(mInternals->context, _makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_spirv, _makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_spirv_size, &ocl_error);
80  } else
81 #endif // clang-format on
82 
83  {
84  size_t program_sizes[1] = {_makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_src_size};
85  char* programs_sources[1] = {_makefile_opencl_program_Base_opencl_GPUReconstructionOCL2_cl_src};
86  mInternals->program = clCreateProgramWithSource(mInternals->context, (cl_uint)1, (const char**)&programs_sources, program_sizes, &ocl_error);
87  }
88 
89  if (GPUFailedMsgI(ocl_error)) {
90  GPUError("Error creating OpenCL program from binary");
91  return 1;
92  }
93  if (GPUFailedMsgI(return_status[0])) {
94  GPUError("Error creating OpenCL program from binary (device status)");
95  return 1;
96  }
97 
98  if (GPUFailedMsgI(clBuildProgram(mInternals->program, 1, &mInternals->device, GPUCA_M_STR(OCL_FLAGS), nullptr, nullptr))) {
99  cl_build_status status;
100  if (GPUFailedMsgI(clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_STATUS, sizeof(status), &status, nullptr)) == 0 && status == CL_BUILD_ERROR) {
101  size_t log_size;
102  clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_LOG, 0, nullptr, &log_size);
103  std::unique_ptr<char[]> build_log(new char[log_size + 1]);
104  clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_LOG, log_size, build_log.get(), nullptr);
105  build_log[log_size] = 0;
106  GPUError("Build Log:\n\n%s\n", build_log.get());
107  }
108  return 1;
109  }

129 bool GPUReconstructionOCL2Backend::CheckPlatform(unsigned int i)
130 {
131  char platform_version[64] = {}, platform_vendor[64] = {};
132  clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
133  clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
134  float ver1 = 0;
135  sscanf(platform_version, "OpenCL %f", &ver1);
136  if (ver1 >= 2.2f) {
137  if (mDeviceProcessingSettings.debugLevel >= 2) {
138  GPUInfo("OpenCL 2.2 capable platform found");
139  }
140  return true;
141  }
142 
143  if (strcmp(platform_vendor, "Advanced Micro Devices, Inc.") == 0 && ver1 >= 2.0f) {
144  float ver2 = 0;
145  const char* pos = strchr(platform_version, '(');
146  if (pos) {
147  sscanf(pos, "(%f)", &ver2);
148  }
149  if ((ver1 >= 2.f && ver2 >= 2000.f) || ver1 >= 2.1f) {
150  if (mDeviceProcessingSettings.debugLevel >= 2) {
151  GPUInfo("AMD ROCm OpenCL Platform found");
152  }
153  return true;
154  }
155  }
156  return false;
157 }
