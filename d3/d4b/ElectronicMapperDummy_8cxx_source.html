23 // build the map to go from electronic ds id to detector ds id
24 std::map<uint32_t, uint32_t> buildDsElecId2DsDetIdMap()
25 {
26  std::map<uint32_t, uint32_t> e2d;
27 
28  auto dslist = createDualSampaMapper();
29 
30  uint16_t n{0};
31  uint16_t solarId{0};
32  uint8_t groupId{0};
33  uint8_t index{0};
34 
35  for (auto deId : o2::mch::raw::deIdsForAllMCH) {
36  // assign a tuple (solarId,groupId,index) to the pair (deId,dsId)
37  for (auto dsId : dslist(deId)) {
38  // index 0..4
39  // groupId 0..7
40  // solarId 0..nsolars
41  if (n % 5 == 0) {
42  index = 0;
43  if (n % 8 == 0) {
44  groupId = 0;
45  } else {
46  groupId++;
47  }
48  } else {
49  index++;
50  }
51  if (n % 40 == 0) {
52  solarId++;
53  }
54  o2::mch::raw::DsElecId dsElecId(solarId, groupId, index);
55  o2::mch::raw::DsDetId dsDetId(deId, dsId);
56  e2d.emplace(o2::mch::raw::encode(dsElecId),
57  o2::mch::raw::encode(dsDetId));
58  n++;
59  };
60  };
61  return e2d;
62 }

64 std::map<uint32_t, uint16_t> buildFeeLinkId2SolarIdMap()
65 {
66  std::map<uint32_t, uint16_t> c2s;
67 
68  uint16_t n{0};
69  uint16_t solarId{0};
70 
71  auto dslist = createDualSampaMapper();
72 
73  for (auto deId : o2::mch::raw::deIdsForAllMCH) {
74  // assign a tuple (fee,link) to each solarId
75  for (auto dsId : dslist(deId)) {
76  if (n % 40 == 0) {
77  solarId++;
78  auto feeId = solarId / 12;
79  auto linkId = solarId - feeId * 12;
80  c2s[encode(o2::mch::raw::FeeLinkId(feeId, linkId))] = solarId;
81  }
82  n++;
83  };
84  };
85  auto feeId = solarId / 12;
86  auto linkId = solarId - feeId * 12;
87  c2s[encode(o2::mch::raw::FeeLinkId(feeId, linkId))] = solarId;
88  return c2s;
89 }
