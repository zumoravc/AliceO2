37 DataProcessorSpec getSimReaderSpec(SubspecRange range, const std::vector<std::string>& simprefixes, const std::vector<int>& tpcsectors)
38 {
39  uint64_t activeSectors = 0;
40  for (const auto& tpcsector : tpcsectors) {
41  activeSectors |= (uint64_t)0x1 << tpcsector;
42  }
43 
44  auto doit = [range, tpcsectors, activeSectors](ProcessingContext& pc) {
45  auto& mgr = steer::HitProcessingManager::instance();
46  auto eventrecords = mgr.getDigitizationContext().getEventRecords();
47  const auto& context = mgr.getDigitizationContext();
48 
49  for (auto const& sector : tpcsectors) {
50  // Note: the TPC sector header was serving the sector to lane mapping before
51  // now the only remaining purpose is to propagate the mask of valid sectors
52  // in principle even that is not necessary any more because every sector has
53  // a dedicated route bound to the sector numbers as subspecification and the
54  // merging can be done based on that. However if we in the future go over to
55  // a multipart scheme again, we again will need the information, so we keep it
56  // For the moment, sector member in the sector header is in sync with
57  // subspecification of the route
58  o2::tpc::TPCSectorHeader header{sector};
59  header.activeSectors = activeSectors;
60  pc.outputs().snapshot(OutputRef{"collisioncontext", static_cast<SubSpecificationType>(sector), {header}},
61  context);
62  }
63 
64  // the first 36 channel numbers are reserved for the TPC, now publish the remaining
65  // channels
66  for (int subchannel = range.min; subchannel < range.max; ++subchannel) {
67  LOG(INFO) << "SENDING SOMETHING TO OTHERS";
68  pc.outputs().snapshot(
69  OutputRef{"collisioncontext", static_cast<SubSpecificationType>(subchannel)},
70  context);
71  }
72 
73  // digitizer workflow runs only once
74  // send endOfData control event and mark the reader as ready to finish
75  pc.services().get<ControlService>().endOfStream();
76  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
77  };
78 
79  // init function return a lambda taking a ProcessingContext
80  auto initIt = [simprefixes, doit](InitContext& ctx) {
81  // initialize fundamental objects
82  auto& mgr = steer::HitProcessingManager::instance();
83 
84  if (simprefixes.size() == 0) {
85  LOG(ERROR) << "No simulation prefix available";
86  } else {
87  LOG(INFO) << "adding " << simprefixes[0] << "\n";
88  mgr.addInputFile(simprefixes[0]);
89  for (int part = 1; part < simprefixes.size(); ++part) {
90  mgr.addInputSignalFile(simprefixes[part]);
91  }
92  }
93 
94  // do we start from an existing context
95  auto incontextstring = ctx.options().get<std::string>("incontext");
96  LOG(INFO) << "INCONTEXTSTRING " << incontextstring;
97  if (incontextstring.size() > 0) {
98  auto success = mgr.setupRunFromExistingContext(incontextstring.c_str());
99  if (!success) {
100  LOG(FATAL) << "Could not read collision context from " << incontextstring;
101  }
102  } else {
103 
104  auto intRate = ctx.options().get<float>("interactionRate"); // is interaction rate requested?
105  if (intRate < 1.f) {
106  intRate = 1.f;
107  }
108  LOG(INFO) << "Imposing hadronic interaction rate " << intRate << "Hz";
109  mgr.getInteractionSampler().setInteractionRate(intRate);
110 
111  auto bcPatternFile = ctx.options().get<std::string>("bcPatternFile");
112  if (!bcPatternFile.empty()) {
113  mgr.getInteractionSampler().setBunchFilling(bcPatternFile);
114  }
115 
116  mgr.getInteractionSampler().init();
117  // doing a random event selection/subsampling?
118  mgr.setRandomEventSequence(ctx.options().get<int>("randomsample") > 0);
119 
120  // number of collisions asked?
121  auto col = ctx.options().get<int>("ncollisions");
122  if (col != 0) {
123  mgr.setupRun(col);
124  } else {
125  mgr.setupRun();
126  }
127  LOG(INFO) << "Initializing Spec ... have " << mgr.getDigitizationContext().getEventRecords().size() << " times ";
128  LOG(INFO) << "Serializing Context for later reuse";
129  mgr.writeDigitizationContext(ctx.options().get<std::string>("outcontext").c_str());
130  }
131 
132  return doit;
133  };
134 
135  std::vector<OutputSpec> outputs;
136  for (auto const& tpcsector : tpcsectors) {
137  outputs.emplace_back(
138  OutputSpec{{"collisioncontext"}, "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(tpcsector), Lifetime::Timeframe});
139  }
140  for (int subchannel = range.min; subchannel < range.max; ++subchannel) {
141  outputs.emplace_back(
142  OutputSpec{{"collisioncontext"}, "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(subchannel), Lifetime::Timeframe});
143  }
144 
145  return DataProcessorSpec{
146  /*ID*/ "SimReader",
147  /*INPUT CHANNELS*/ Inputs{}, outputs,
148  /* ALGORITHM */
149  AlgorithmSpec{initIt},
150  /* OPTIONS */
151  Options{
152  {"interactionRate", VariantType::Float, 50000.0f, {"Total hadronic interaction rate (Hz)"}},
153  {"bcPatternFile", VariantType::String, "", {"Interacting BC pattern file (e.g. from CreateBCPattern.C)"}},
154  {"simFileQED", VariantType::String, "", {"Sim (QED) input filename"}},
155  {"outcontext", VariantType::String, "collisioncontext.root", {"Output file for collision context"}},
156  {"incontext", VariantType::String, "", {"Take collision context from this file"}},
157  {"ncollisions,n",
158  VariantType::Int,
159  0,
160  {"number of collisions to sample (default is given by number of entries in chain"}},
161  {"randomsample", VariantType::Int, 0, {"Draw collisions random instead of linear sequence. (Default no = 0)"}}}};
162 }
