35 // a generic looper of data members of a TClass; calling a callback
36 // reused in various functions below
37 void loopOverMembers(TClass* cl, void* obj,
38  std::function<void(const TDataMember*, int, int)>&& callback)
39 {
40  auto memberlist = cl->GetListOfDataMembers();
41  for (int i = 0; i < memberlist->GetEntries(); ++i) {
42  auto dm = (TDataMember*)memberlist->At(i);
43 
44  auto isValidComplex = [dm]() {
45  return isString(*dm) || dm->IsEnum();
46  };
47 
48  // filter out static members for now
49  if (dm->Property() & kIsStatic) {
50  continue;
51  }
52 
53  if (dm->IsaPointer()) {
54  continue;
55  }
56  if (!dm->IsBasic() && !isValidComplex()) {
57  continue;
58  }
59 
60  const auto dim = dm->GetArrayDim();
61  // we support very simple vectored data in 1D for now
62  if (dim > 1) {
63  continue;
64  }
65 
66  const auto size = (dim == 1) ? dm->GetMaxIndex(dim - 1) : 1; // size of array (1 if scalar)
67  for (int index = 0; index < size; ++index) {
68  callback(dm, index, size);
69  }
70  }
71 }

224 std::vector<ConfigParamSpec>
225  RootConfigParamHelpers::asConfigParamSpecsImpl(std::string const& mainKey, TClass* cl, void* obj)
226 {
227  std::vector<ConfigParamSpec> specs;
228 
229  auto toDataMember = [&mainKey, &specs, obj](const TDataMember* dm, int index, int size) {
230  auto dt = dm->GetDataType();
231  auto TS = dt ? dt->Size() : 0;
232  char* ptr = ((char*)obj) + dm->GetOffset() + index * TS;
233  const std::string name = mainKey + "." + getName(dm, index, size);
234 
235  specs.push_back(memberToConfigParamSpec(name.c_str(), dm, ptr));
236  };
237 
238  loopOverMembers(cl, obj, toDataMember);
239  return specs;
240 }

244 void RootConfigParamHelpers::fillFromPtree(TClass* cl, void* obj, boost::property_tree::ptree const& pt)
245 {
246  auto toDataMember = [obj, &pt](const TDataMember* dm, int index, int size) {
247  auto dt = dm->GetDataType();
248  auto TS = dt ? dt->Size() : 0;
249  char* ptr = ((char*)obj) + dm->GetOffset() + index * TS;
250  const std::string name = getName(dm, index, size);
251  auto it = pt.get_child_optional(name);
252  if (!it) {
253  return;
254  }
255  ptreeToMember(*it, dm->GetTrueTypeName(), dm, ptr);
256  };
257 
258  loopOverMembers(cl, obj, toDataMember);
259 }
