76  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator collision, soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra>> const& tracks)
77  {
78 
79  if (!collision.sel7())
80  return;
81 
82  centV0M->Fill(collision.centV0M());
83  vtxZ->Fill(collision.posZ());
84 
85  for (auto& track : tracks) {
86  //if (track.pt() < ptlow)
87  // continue;
88  ptH->Fill(track.pt());
89  trZ->Fill(track.z());
90  itsChi2->Fill(track.itsChi2NCl());
91  for (int i = 0; i < 6; i++) {
92  if (track.itsClusterMap() & (uint8_t(1) << i))
93  itsHits->Fill(i);
94  if (track.itsClusterMap() & (uint8_t(1) << i))
95  itsHitsVsPt->Fill(i, track.pt());
96  }
97  tpcDedx->Fill(track.tpcInnerParam(), track.tpcSignal());
98  tpcChi2->Fill(track.tpcChi2NCl());
99  tpcCls->Fill(track.tpcNClsFound());
100  for (int i = 0; i < 64; i++) {
101  if (track.flags() & (uint64_t(1) << i))
102  flagsHist->Fill(i);
103  }
104  }
105  for (auto& [t0, t1] : combinations(tracks, tracks)) {
106  ptCorr->Fill(t0.pt(), t1.pt());
107  if (!((t0.itsClusterMap() & (uint8_t(1) << 0)) || (t0.itsClusterMap() & (uint8_t(1) << 1))))
108  continue;
109  if (!((t1.itsClusterMap() & (uint8_t(1) << 0)) || (t1.itsClusterMap() & (uint8_t(1) << 1))))
110  continue;
111 
112  TLorentzVector p1, p2, p;
113  p1.SetXYZM(t0.px(), t0.py(), t0.pz(), gkMass);
114  p2.SetXYZM(t1.px(), t1.py(), t1.pz(), gkMass);
115  p = p1 + p2;
116 
117  if (t0.charge() * t1.charge() < 0) {
118  invMassPM->Fill(p.M());
119  invMassVsPt->Fill(p.M(), p.Pt());
120  invMassVsCentrality->Fill(p.M(), collision.centV0M());
121  } else {
122  if (t0.charge() > 0)
123  invMassPP->Fill(p.M());
124  if (t0.charge() < 0)
125  invMassMM->Fill(p.M());
126  }
127  }
128  }
129 };
