59  switch (op) {

63  case Operation::ZSDecoder:
64  zsDecoder = true;
65  break;
66  case Operation::ZSOnTheFly:
67  zsOnTheFly = true;
68  break;
69  case Operation::OutputTracks:
70  outputTracks = true;
71  break;
72  case Operation::OutputCompClusters:
73  outputCompClusters = true;
74  break;
75  case Operation::OutputCompClustersFlat:
76  outputCompClustersFlat = true;
77  break;
78  case Operation::OutputCAClusters:
79  outputCAClusters = true;
80  break;
81  case Operation::ProcessMC:
82  processMC = true;
83  break;
84  case Operation::Noop:
85  break;
86  default:
87  throw std::runtime_error("invalid CATracker operation");
88  }
89  if constexpr (sizeof...(args) > 0) {
90  init(std::forward<Args>(args)...);
