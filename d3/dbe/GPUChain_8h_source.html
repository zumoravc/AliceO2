232 template <bool Always, class T, class S, typename... Args>
233 inline void GPUChain::timeCpy(RecoStep step, int toGPU, S T::*func, Args... args)
234 {
235  if (!Always && step != RecoStep::NoRecoStep && !(GetRecoStepsGPU() & step)) {
236  return;
237  }
238  HighResTimer* timer = nullptr;
239  size_t* bytes = nullptr;
240  if (mRec->mDeviceProcessingSettings.debugLevel >= 1 && toGPU >= 0) { // Todo: time special cases toGPU < 0
241  int id = mRec->getRecoStepNum(step, false);
242  if (id != -1) {
243  auto& tmp = mRec->mTimersRecoSteps[id];
244  timer = toGPU ? &tmp.timerToGPU : &tmp.timerToHost;
245  bytes = toGPU ? &tmp.bytesToGPU : &tmp.bytesToHost;
246  (toGPU ? tmp.countToGPU : tmp.countToHost)++;
247  timer->Start();
248  }
249  }
250  size_t n = (mRec->*func)(args...);
251  if (timer) {
252  SynchronizeGPU();
253  timer->Stop();
254  *bytes += n;
255  }
256 }

258 template <class T, class S, typename... Args>
259 bool GPUChain::DoDebugAndDump(GPUChain::RecoStep step, int mask, bool transfer, T& processor, S T::*func, Args&&... args)
260 {
261  if (GetDeviceProcessingSettings().keepAllMemory) {
262  if (transfer) {
263  TransferMemoryResourcesToHost(step, &processor, -1, true);
264  }
265  if (GetDeviceProcessingSettings().debugLevel >= 6 && (mask == 0 || (GetDeviceProcessingSettings().debugMask & mask))) {
266  (processor.*func)(args...);
267  return true;
268  }
269  }
270  return false;
271 }
