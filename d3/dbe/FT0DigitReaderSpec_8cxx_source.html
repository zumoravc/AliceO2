39 void DigitReader::run(ProcessingContext& pc)
40 {
41  if (mFinished) {
42  return;
43  }
44 
45  { // load data from files
46  TFile digFile(mInputFileName.c_str(), "read");
47  if (digFile.IsZombie()) {
48  LOG(FATAL) << "Failed to open FT0 digits file " << mInputFileName;
49  }
50  LOG(INFO) << " file read " << mInputFileName;
51  TTree* digTree = (TTree*)digFile.Get(mDigitTreeName.c_str());
52  // digTree->Print();
53  if (!digTree) {
54  LOG(FATAL) << "Failed to load FT0 digits tree " << mDigitTreeName << " from " << mInputFileName;
55  }
56  if (!digTree) {
57  LOG(FATAL) << "Failed to load FT0 digits tree " << mDigitTreeName << " from " << mInputFileName;
58  }
59  LOG(INFO) << "Loaded FT0 digits tree " << mDigitTreeName << " from " << mInputFileName;
60 
61  digTree->SetBranchAddress("FT0DIGITSBC", &mDigitsBC);
62  digTree->SetBranchAddress("FT0DIGITSCH", &mDigitsCH);
63 
64  if (mUseMC) {
65  if (digTree->GetBranch(mDigitMCTruthBranchName.c_str())) {
66  digTree->SetBranchAddress(mDigitMCTruthBranchName.c_str(), &mMCTruth);
67  LOG(INFO) << "Will use MC-truth from " << mDigitMCTruthBranchName;
68  } else {
69  LOG(INFO) << "MC-truth is missing";
70  mUseMC = false;
71  }
72  }
73 
74  digTree->GetEntry(0);
75  delete digTree;
76  digFile.Close();
77  }
78 
79  pc.outputs().snapshot(Output{mOrigin, "DIGITSBC", 0, Lifetime::Timeframe}, *mDigitsBC);
80  pc.outputs().snapshot(Output{mOrigin, "DIGITSCH", 0, Lifetime::Timeframe}, *mDigitsCH);
81  if (mUseMC) {
82  pc.outputs().snapshot(Output{mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe}, *mMCTruth);
83  }
84 
85  mFinished = true;
86  pc.services().get<ControlService>().endOfStream();
87  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
88 }
