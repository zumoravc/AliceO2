96  // clang-format off
97  DataProcessorSpec simpleQcTask{
98  "simpleQcTask",
99  Inputs{
100  { "TPC_CLUSTERS_S", { "DS", "simpleQcTask-0" } },
101  { "TPC_CLUSTERS_P_S", { "DS", "simpleQcTask-1" } }
102  },
103  Outputs{},
104  AlgorithmSpec{
105  (AlgorithmSpec::ProcessCallback) [](ProcessingContext& ctx){
106  auto inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("TPC_CLUSTERS_S").payload);
107  auto inputDataTpcProcessed = reinterpret_cast<const FakeCluster*>(ctx.inputs().get(
108  "TPC_CLUSTERS_P_S").payload);
109 
110  const auto* header = o2::header::get<DataHeader*>(ctx.inputs().get("TPC_CLUSTERS_S").header);
111 
112  bool dataGood = true;
113  for (int j = 0; j < header->payloadSize / sizeof(FakeCluster); ++j) {
114  float diff = std::abs(-inputDataTpc[j].x - inputDataTpcProcessed[j].x) +
115  std::abs(2 * inputDataTpc[j].y - inputDataTpcProcessed[j].y) +
116  std::abs(inputDataTpc[j].z * inputDataTpc[j].q - inputDataTpcProcessed[j].z) +
117  std::abs(inputDataTpc[j].q - inputDataTpcProcessed[j].q);
118  if (diff > 1) {
119  dataGood = false;
120  break;
121  }
122  }
123 
124  LOG(INFO) << "simpleQcTask - received data is " << (dataGood ? "correct" : "wrong");
125  }
126  }
127  };

156 void someDataProducerAlgorithm(ProcessingContext& ctx)
157 {
158  size_t index = ctx.services().get<ParallelContext>().index1D();
159  std::this_thread::sleep_for(std::chrono::seconds(1));
160  // Creates a new message of size collectionChunkSize which
161  // has "TPC" as data origin and "CLUSTERS" as data description.
162  auto& tpcClusters = ctx.outputs().make<FakeCluster>(
163  Output{"TPC", "CLUSTERS", static_cast<o2::header::DataHeader::SubSpecificationType>(index)}, collectionChunkSize);
164  int i = 0;
165 
166  for (auto& cluster : tpcClusters) {
167  assert(i < collectionChunkSize);
168  cluster.x = index;
169  cluster.y = i;
170  cluster.z = i;
171  cluster.q = rand() % 1000;
172  i++;
173  }
174 }

176 void someProcessingStageAlgorithm(ProcessingContext& ctx)
177 {
178  size_t index = ctx.services().get<ParallelContext>().index1D();
179 
180  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
181  auto& processedTpcClusters = ctx.outputs().make<FakeCluster>(
182  Output{"TPC", "CLUSTERS_P", static_cast<o2::header::DataHeader::SubSpecificationType>(index)},
183  collectionChunkSize);
184 
185  int i = 0;
186  for (auto& cluster : processedTpcClusters) {
187  assert(i < collectionChunkSize);
188  cluster.x = -inputDataTpc[i].x;
189  cluster.y = 2 * inputDataTpc[i].y;
190  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
191  cluster.q = inputDataTpc[i].q;
192  i++;
193  }
194 };
