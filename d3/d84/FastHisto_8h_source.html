187 inline void FastHisto<T>::print(const int prec) const
188 {
189  const math_utils::math_base::StatisticsData data = getStatisticsData();
190  LOGP(info, "\n Entries: {}", std::accumulate(mBinCont.begin(), mBinCont.end(), 0));
191  LOGP(info, "Truncated Mean: {}", data.mCOG);
192  LOGP(info, "Standard Deviation: {}", data.mStdDev);
193  LOGP(info, "sum of content: {}", data.mSum);
194 
195  const int maxEle = *std::max_element(mBinCont.begin(), mBinCont.end()); // get the maximum element in the
196  // histogram needed for printing
197  // this loop prints the histogram
198  // starting from the largest value in the array and go backwards to 0
199  std::stringstream stream;
200  stream.width(10);
201 
202  // getting the x axis label
203  stream << " 1";
204  for (int i = 2; i <= maxEle; ++i) {
205  stream.width(2);
206  stream << std::right << i;
207  }
208  LOGP(info, "{}", stream.str());
209  stream.str(std::string());
210  stream << "---------";
211  for (int i = 0; i <= maxEle; ++i) {
212  stream << "--";
213  }
214  LOGP(info, "{}", stream.str());
215 
216  const float binWidth = getBinWidth();
217  const int totalBins = mBinCont.size();
218 
219  for (int i = 0; i < totalBins; ++i) {
220  stream.str(std::string());
221  if (i == 0 && mUseUnderflow) {
222  stream.width(9);
223  stream << std::right << "U_Flow | ";
224  } else if (i == totalBins - 1 && mUseOverflow) {
225  stream.width(9);
226  stream << std::right << "O_Flow | ";
227  } else {
228  stream.width(6);
229  const float xPos = mXmin + binWidth * (i - mUseUnderflow);
230  stream << std::fixed << std::setprecision(prec) << xPos;
231  const std::string sIn = " | ";
232  stream << sIn;
233  }
234 
235  for (int j = 1; j <= getBinContent(i); ++j) {
236  const std::string sIn = "x ";
237  stream << sIn;
238  }
239  std::string xPosString = stream.str();
240  LOGP(info, "{}", xPosString);
241  }
242  LOGP(info, "");
243 }

245 template <class T>
246 inline math_utils::math_base::StatisticsData FastHisto<T>::getStatisticsData(const float low, const float high) const
247 {
248  math_utils::math_base::StatisticsData data{};
249  // in case something went wrong the COG is the histogram lower limit
250  data.mCOG = mXmin;
251  if (mBinCont.size() == 0) {
252  return data;
253  }
254  if (low > high) {
255  return data;
256  }
257 
258  // total histogram content
259  const T integral = std::accumulate(mBinCont.begin(), mBinCont.end(), 0);
260  if (integral == 0) {
261  return data;
262  }
263  const float lowerBound = integral * low;
264  const float upperBound = integral * high;
265  const unsigned int lastBin = mBinCont.size() - 1;
266 
267  float mean = 0;
268  float sum = 0;
269  float rms2 = 0;
270 
271  // add fractional values
272  const float binWidth = getBinWidth();
273  const int shiftBin = mUseUnderflow ? -1 : 0; // shift the bin position in case of additional bin on the left
274 
275  T countContent = 0;
276  bool isLowerFrac = false;
277  unsigned int bin = 0;
278  for (bin = 0; bin <= lastBin; ++bin) {
279  countContent += mBinCont[bin];
280 
281  if (countContent - lowerBound <= 0) { // lower truncation cut
282  continue;
283  } else if (countContent >= upperBound) { // upper truncation cut
284  break;
285  }
286 
287  const float xcenter = mXmin + (bin + 0.5f + shiftBin) * binWidth;
288  const float tmpBinCont = isLowerFrac ? mBinCont[bin] : countContent - lowerBound; // set bincontent to countcontent for first time only
289  const float tmpMean = tmpBinCont * xcenter;
290  mean += tmpMean;
291  sum += tmpBinCont;
292  rms2 += tmpMean * xcenter;
293  isLowerFrac = true;
294  }
295 
296  if (!checkBin(bin)) {
297  return data;
298  }
299 
300  //set last bin
301  //TODO move to upper loop
302  const float xcenter = mXmin + (bin + 0.5f + shiftBin) * binWidth;
303  const T upFrac = mBinCont[bin] - (static_cast<float>(countContent) - upperBound);
304  const float tmpMean = upFrac * xcenter;
305  mean += tmpMean;
306  sum += upFrac;
307  rms2 += tmpMean * xcenter;
308 
309  if (sum == 0) {
310  return data;
311  }
312 
313  mean /= sum;
314  data.mCOG = mean;
315  rms2 /= sum;
316  data.mStdDev = std::sqrt(std::abs(rms2 - mean * mean));
317  data.mSum = sum;
318  return data;
319 }

321 template <class T>
322 inline int FastHisto<T>::findBin(const T val) const
323 {
324  if (val < mXmin) {
325  if (!mUseUnderflow) {
326  LOGP(warning, "findBin: UNDERFLOW BIN: BIN IS NOT IN HISTOGRAM RANGE!");
327  return -1; // if undeflow bin is not used BUT value is in underflow bin return -1
328  } else {
329  return 0; // if underflow bin is used return 0 as index
330  }
331  }
332 
333  if (val >= mXmax) {
334  if (!mUseOverflow) {
335  LOGP(warning, "findBin: OVERFLOW BIN: BIN IS NOT IN HISTOGRAM RANGE!");
336  return -1; // if overflow bin is not used BUT value is in overflow bin return -1
337  } else {
338  return mBinCont.size() - 1; // if overflow bin is used return the last index in the vector
339  }
340  }
341 
342  const float binWidth = getBinWidth();
343  const int bin = (val - mXmin) / binWidth + mUseUnderflow;
344  return bin;
