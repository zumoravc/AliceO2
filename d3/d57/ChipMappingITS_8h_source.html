102  void getChipInfoSW(int chipSW, ChipInfo& chInfo) const
103  {
104  chInfo.id = chipSW;
105  if (chipSW > NChipsSB[IB] + NChipsSB[MB] - 1) { // OB
106  chipSW -= NChipsSB[IB] + NChipsSB[MB];
107  chInfo.ruType = OB;
108  auto dvRU = std::div(chipSW, NChipsPerStaveSB[OB]);
109  chInfo.ru = NStavesSB[IB] + NStavesSB[MB] + dvRU.quot; // RU ID == stave ID
110  chInfo.chOnRU = getChipOnRUInfo(OB, dvRU.rem);
111  } else if (chipSW > NChipsSB[IB] - 1) { // MB
112  chipSW -= NChipsSB[IB];
113  chInfo.ruType = MB;
114  auto dvRU = std::div(chipSW, NChipsPerStaveSB[MB]);
115  chInfo.ru = NStavesSB[IB] + dvRU.quot; // RU ID == stave ID
116  chInfo.chOnRU = getChipOnRUInfo(MB, dvRU.rem);
117  } else { // IB
118  chInfo.ruType = IB;
119  auto dvRU = std::div(chipSW, NChipsPerStaveSB[IB]);
120  chInfo.ru = dvRU.quot; // RU ID == stave ID = module ID
121  chInfo.chOnRU = getChipOnRUInfo(IB, dvRU.rem);
122  }
123  }

197  int getRUIDSW(int lr, int ruOnLr) const
198  {
199  int sid = 0;
200  for (int i = 0; i < NLayers; i++) {
201  if (i >= lr)
202  break;
203  sid += NStavesOnLr[i];
204  }
205  return sid + ruOnLr;
206  }
