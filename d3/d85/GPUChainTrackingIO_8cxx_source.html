71 void GPUChainTracking::DumpData(const char* filename)
72 {
73  FILE* fp = fopen(filename, "w+b");
74  if (fp == nullptr) {
75  return;
76  }
77  fwrite(DUMP_HEADER, 1, DUMP_HEADER_SIZE, fp);
78  fwrite(&GPUReconstruction::geometryType, sizeof(GPUReconstruction::geometryType), 1, fp);
79  DumpData(fp, mIOPtrs.clusterData, mIOPtrs.nClusterData, InOutPointerType::CLUSTER_DATA);
80  DumpData(fp, mIOPtrs.rawClusters, mIOPtrs.nRawClusters, InOutPointerType::RAW_CLUSTERS);
81  if (mIOPtrs.clustersNative) {
82  DumpData(fp, &mIOPtrs.clustersNative->clustersLinear, &mIOPtrs.clustersNative->nClustersTotal, InOutPointerType::CLUSTERS_NATIVE);
83  fwrite(&mIOPtrs.clustersNative->nClusters[0][0], sizeof(mIOPtrs.clustersNative->nClusters[0][0]), NSLICES * GPUCA_ROW_COUNT, fp);
84  }
85  if (mIOPtrs.tpcPackedDigits) {
86  DumpData(fp, mIOPtrs.tpcPackedDigits->tpcDigits, mIOPtrs.tpcPackedDigits->nTPCDigits, InOutPointerType::TPC_DIGIT);
87  }
88  if (mIOPtrs.tpcZS) {
89  size_t total = 0;
90  for (int i = 0; i < NSLICES; i++) {
91  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
92  for (unsigned int k = 0; k < mIOPtrs.tpcZS->slice[i].count[j]; k++) {
93  total += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
94  }
95  }
96  }
97  std::vector<std::array<char, TPCZSHDR::TPC_ZS_PAGE_SIZE>> pages(total);
98  char* ptr = pages[0].data();
99  GPUTrackingInOutZS::GPUTrackingInOutZSCounts counts;
100  total = 0;
101  for (int i = 0; i < NSLICES; i++) {
102  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
103  for (unsigned int k = 0; k < mIOPtrs.tpcZS->slice[i].count[j]; k++) {
104  memcpy(&ptr[total * TPCZSHDR::TPC_ZS_PAGE_SIZE], mIOPtrs.tpcZS->slice[i].zsPtr[j][k], mIOPtrs.tpcZS->slice[i].nZSPtr[j][k] * TPCZSHDR::TPC_ZS_PAGE_SIZE);
105  counts.count[i][j] += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
106  total += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
107  }
108  }
109  }
110  total *= TPCZSHDR::TPC_ZS_PAGE_SIZE;
111  DumpData(fp, &ptr, &total, InOutPointerType::TPC_ZS);
112  fwrite(&counts, sizeof(counts), 1, fp);
113  }
114  DumpData(fp, mIOPtrs.sliceTracks, mIOPtrs.nSliceTracks, InOutPointerType::SLICE_OUT_TRACK);
115  DumpData(fp, mIOPtrs.sliceClusters, mIOPtrs.nSliceClusters, InOutPointerType::SLICE_OUT_CLUSTER);
116  DumpData(fp, &mIOPtrs.mcLabelsTPC, &mIOPtrs.nMCLabelsTPC, InOutPointerType::MC_LABEL_TPC);
117  DumpData(fp, &mIOPtrs.mcInfosTPC, &mIOPtrs.nMCInfosTPC, InOutPointerType::MC_INFO_TPC);
118  DumpData(fp, &mIOPtrs.mergedTracks, &mIOPtrs.nMergedTracks, InOutPointerType::MERGED_TRACK);
119  DumpData(fp, &mIOPtrs.mergedTrackHits, &mIOPtrs.nMergedTrackHits, InOutPointerType::MERGED_TRACK_HIT);
120  DumpData(fp, &mIOPtrs.trdTracks, &mIOPtrs.nTRDTracks, InOutPointerType::TRD_TRACK);
121  DumpData(fp, &mIOPtrs.trdTracklets, &mIOPtrs.nTRDTracklets, InOutPointerType::TRD_TRACKLET);
122  DumpData(fp, &mIOPtrs.trdTrackletsMC, &mIOPtrs.nTRDTrackletsMC, InOutPointerType::TRD_TRACKLET_MC);
123  fclose(fp);
124 }
