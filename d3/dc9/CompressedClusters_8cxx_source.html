50 void CompressedClustersROOT::Streamer(TBuffer& R__b)
51 {
52  // the custom streamer for CompressedClustersROOT
53  if (R__b.IsReading()) {
54  R__b.ReadClassBuffer(CompressedClustersROOT::Class(), this);
55  gsl::span flatdata{this->flatdata, this->flatdataSize};
56  CompressedClustersHelpers::restoreFrom(flatdata, *this);
57  } else {
58  std::vector<char> flatdata;
59  // member flatdata is nullptr unless it is an already extracted object which
60  // is streamed again. Overriding the existing flatbuffer is a potential
61  // memory leak
62  // Note: lots of cornercases here which are diffucult to catch with the design
63  // of CompClusters container (which is optimized to be used with GPU). Main concern
64  // is if an extracted object is not read-only and the counters are changed.
65  // TODO: we can add a consistency check whether the size of the flatbuffer
66  // and the pointers match the object.
67  bool isflat = this->flatdata != nullptr;
68  if (!isflat) {
69  CompressedClustersHelpers::flattenTo(flatdata, *this);
70  this->flatdataSize = flatdata.size();
71  this->flatdata = flatdata.data();
72  }
73  R__b.WriteClassBuffer(CompressedClustersROOT::Class(), this);
74  if (!isflat) {
75  this->flatdataSize = 0;
76  this->flatdata = nullptr;
77  }
78  }
79 }
