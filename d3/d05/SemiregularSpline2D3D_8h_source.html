200 inline void SemiregularSpline2D3D::getKnotUV(int iKnot, float& u, float& v) const
201 {
202 
203  // iterate through all RegularSpline1D's
204  for (int i = 0; i < mNumberOfRows; i++) {
205  const RegularSpline1D& gridU = getGridU(i);
206  const int nk = gridU.getNumberOfKnots();
207 
208  // if the searched index is less or equal as the number of knots in the current spline
209  // the searched u-v-coordinates have to be in this spline.
210  if (iKnot <= nk - 1) {
211 
212  //in that case v is the current index
213  v = mGridV.knotIndexToU(i);
214 
215  //and u the coordinate of the given index
216  u = gridU.knotIndexToU(iKnot);
217  break;
218  }
219 
220  //if iKnot is greater than number of knots the searched u-v cannot be in the current gridU
221  //so we search for nk less indizes and continue with the next v-coordinate
222  iKnot -= nk;
223  }
224 }

227 inline void SemiregularSpline2D3D::correctEdges(T* data) const
228 {
229  //Regular v-Grid (vertical)
230  const RegularSpline1D& gridV = getGridV();
231 
232  int nv = mNumberOfRows;
233 
234  //EIGENTLICH V VOR U!!!
235  //Wegen Splines aber U vor V
236 
237  { // ==== left edge of U ====
238  //loop through all gridUs
239  for (int iv = 1; iv < mNumberOfRows - 1; iv++) {
240  T* f0 = data + getDataIndex(0, iv);
241  T* f1 = f0 + 3;
242  T* f2 = f0 + 6;
243  T* f3 = f0 + 9;
244  for (int idim = 0; idim < 3; idim++) {
245  f0[idim] = (T)(0.5 * f0[idim] + 1.5 * f1[idim] - 1.5 * f2[idim] + 0.5 * f3[idim]);
246  }
247  }
248  }
249 
250  { // ==== right edge of U ====
251  //loop through all gridUs
252  for (int iv = 1; iv < mNumberOfRows - 1; iv++) {
253  const RegularSpline1D& gridU = getGridU(iv);
254  int nu = gridU.getNumberOfKnots();
255  T* f0 = data + getDataIndex(nu - 4, iv);
256  T* f1 = f0 + 3;
257  T* f2 = f0 + 6;
258  T* f3 = f0 + 9;
259  for (int idim = 0; idim < 3; idim++) {
260  f3[idim] = (T)(0.5 * f0[idim] - 1.5 * f1[idim] + 1.5 * f2[idim] + 0.5 * f3[idim]);
261  }
262  }
263  }
264 
265  { // ==== low edge of V ====
266  const RegularSpline1D& gridU = getGridU(0);
267  int nu = gridU.getNumberOfKnots();
268 
269  for (int iu = 0; iu < nu; iu++) {
270  //f0 to f3 are the x,y,z values of 4 points in the grid along the v axis.
271  //Since there are no knots because of the irregularity you can get this by using the getSplineMethod.
272  T* f0 = data + getDataIndex(iu, 0);
273  float u = gridU.knotIndexToU(iu);
274 
275  float x1 = 0, y1 = 0, z1 = 0, x2 = 0, y2 = 0, z2 = 0, x3 = 0, y3 = 0, z3 = 0;
276  getSpline(data, u, gridV.knotIndexToU(1), x1, y1, z1);
277  getSpline(data, u, gridV.knotIndexToU(2), x2, y2, z2);
278  getSpline(data, u, gridV.knotIndexToU(3), x3, y3, z3);
279 
280  T f1[3] = {x1, y1, z1};
281  T f2[3] = {x2, y2, z2};
282  T f3[3] = {x3, y3, z3};
283  for (int idim = 0; idim < 3; idim++) {
284  f0[idim] = (T)(0.5 * f0[idim] + 1.5 * f1[idim] - 1.5 * f2[idim] + 0.5 * f3[idim]);
285  }
286  }
287  }
288 
289  { // ==== high edge of V ====
290  const RegularSpline1D& gridU = getGridU(nv - 4);
291  int nu = getGridU(nv - 1).getNumberOfKnots();
292 
293  for (int iu = 0; iu < nu; iu++) {
294 
295  float u = getGridU(nv - 1).knotIndexToU(iu);
296 
297  float x1 = 0, y1 = 0, z1 = 0, x2 = 0, y2 = 0, z2 = 0, x3 = 0, y3 = 0, z3 = 0;
298  getSpline(data, u, gridV.knotIndexToU(nv - 4), x1, y1, z1);
299  getSpline(data, u, gridV.knotIndexToU(nv - 3), x2, y2, z2);
300  getSpline(data, u, gridV.knotIndexToU(nv - 2), x3, y3, z3);
301 
302  T f0[3] = {x1, y1, z1};
303  T f1[3] = {x2, y2, z2};
304  T f2[3] = {x3, y3, z3};
305  T* f3 = data + getDataIndex(iu, nv - 1);
306  for (int idim = 0; idim < 3; idim++) {
307  f3[idim] = (T)(0.5 * f0[idim] - 1.5 * f1[idim] + 1.5 * f2[idim] + 0.5 * f3[idim]);
308  }
309  }
310  }
311 
312  // =============== CORRECT CORNERS ==============
313 
314  { // === Lower left corner with u-direction ===
315  T* f0 = data;
316  T* f1 = f0 + 3;
317  T* f2 = f0 + 6;
318  T* f3 = f0 + 9;
319  for (int idim = 0; idim < 3; idim++) {
320  f0[idim] = (T)(0.5 * f0[idim] + 1.5 * f1[idim] - 1.5 * f2[idim] + 0.5 * f3[idim]);
321  }
322  }
323 
324  { // ==== Lower right corner with u-direction ===
325 
326  const RegularSpline1D& gridU = getGridU(0);
327  int nu = gridU.getNumberOfKnots();
328  T* f0 = data + getDataIndex(nu - 4, 0);
329  T* f1 = f0 + 3;
330  T* f2 = f0 + 6;
331  T* f3 = f0 + 9;
332  for (int idim = 0; idim < 3; idim++) {
333  f3[idim] = (T)(0.5 * f0[idim] - 1.5 * f1[idim] + 1.5 * f2[idim] + 0.5 * f3[idim]);
334  }
335  }
336 
337  { // === upper left corner with u-direction ===
338  T* f0 = data + getDataIndex(0, nv - 1);
339  T* f1 = f0 + 3;
340  T* f2 = f0 + 6;
341  T* f3 = f0 + 9;
342  for (int idim = 0; idim < 3; idim++) {
343  f0[idim] = (T)(0.5 * f0[idim] + 1.5 * f1[idim] - 1.5 * f2[idim] + 0.5 * f3[idim]);
344  }
345  }
346 
347  { // ==== upper right corner with u-direction ===
348  const RegularSpline1D& gridU = getGridU(nv - 1);
349  int nu = gridU.getNumberOfKnots();
350  T* f0 = data + getDataIndex(nu - 4, nv - 1);
351  T* f1 = f0 + 3;
352  T* f2 = f0 + 6;
353  T* f3 = f0 + 9;
354  for (int idim = 0; idim < 3; idim++) {
355  f3[idim] = (T)(0.5 * f0[idim] - 1.5 * f1[idim] + 1.5 * f2[idim] + 0.5 * f3[idim]);
356  }
357  }

361 inline void SemiregularSpline2D3D::getSpline(const T* correctedData, float u, float v, T& x, T& y, T& z) const
362 {
363  // Get interpolated value for f(u,v) using data array correctedData[getNumberOfKnots()] with corrected edges
364 
365  // find the v indizes of the u-splines that are needed.
366  int iknotv = mGridV.getKnotIndex(v);
367 
368  // to save the index positions of u-coordinates we create an array
369  T dataVx[12];
370  //int dataOffset0 = getDataIndex0(0, iknotv-1); //index of the very left point in the vi-1-th gridU
371 
372  // we loop through the 4 needed u-Splines
373  int vxIndex = 0;
374  for (int vi = 0; vi < 4; vi++, vxIndex += 3) {
375 
376  const int vDelta = iknotv + vi - 1;
377  const RegularSpline1D& gridU = getGridU(vDelta);
378  // and find at which index in that specific spline the u-coordinate must lay.
379 
380  const int ui = gridU.getKnotIndex(u);
381  const int dataOffset = getDataIndex(ui - 1, vDelta); //(dataOffset0 + (ui-1))*3;
382 
383  dataVx[vxIndex + 0] = gridU.getSpline(ui, correctedData[dataOffset], correctedData[dataOffset + 3], correctedData[dataOffset + 6], correctedData[dataOffset + 9], u);
384  dataVx[vxIndex + 1] = gridU.getSpline(ui, correctedData[dataOffset + 1], correctedData[dataOffset + 4], correctedData[dataOffset + 7], correctedData[dataOffset + 10], u);
385  dataVx[vxIndex + 2] = gridU.getSpline(ui, correctedData[dataOffset + 2], correctedData[dataOffset + 5], correctedData[dataOffset + 8], correctedData[dataOffset + 11], u);
386  }
387 
388  //return results
389  x = mGridV.getSpline(iknotv, dataVx[0], dataVx[3], dataVx[6], dataVx[9], v);
390  y = mGridV.getSpline(iknotv, dataVx[1], dataVx[4], dataVx[7], dataVx[10], v);
391  z = mGridV.getSpline(iknotv, dataVx[2], dataVx[5], dataVx[8], dataVx[11], v);
392 }

394 inline void SemiregularSpline2D3D::getSplineVec(const float* correctedData, float u, float v, float& x, float& y, float& z) const
395 {
396  // Same as getSpline, but using vectorized calculation.
397  // \param correctedData should be at least 128-bit aligned
398 
399 #if !defined(__CINT__) && !defined(__ROOTCINT__) && !defined(GPUCA_GPUCODE) && !defined(GPUCA_NO_VC)
400  //&& !defined(__CLING__)
401  /*
402  Idea: There are 16 knots important for (u, v).
403  a,b,c,d := Knots in first u-grid
404  e,f,g,h := Knots in second u-grid
405  i,j,k,l := Knots in third u-grid
406  m,n,o,p := Knots in fourth u-grid.
407  It could be possible to calculate the spline in 3 dimentions for a,b,c,d at the same time as e,f,g,h etc.
408  3 of the 4 parallel threads of the vector would calculate x,y,z for one row and the last task already calculates x for the next one.
409  => 4x faster
410 
411  Problem:To do this, we need vectors where every i-th element is used for the calculation. So what we need is:
412  [a,e,i,m]
413  [b,f,j,n]
414  [c,g,k,o]
415  [d,h,l,p]
416  This is barely possible to do with a good performance because e.g. a,e,i,m do not lay beside each other in data.
417 
418  Work around 1:
419  Don't calculate knots parrallel but the dimensions. But you can only be 3x faster this way because the 4th thread would be the x-dimension of the next point.
420 
421  Work around 2:
422  Try to create a matrix as it was mentioned earlier ([a,e,i,m][b,f,..]...) by copying data.
423  This may be less efficient than Work around 1 but needs to be measured.
424 
425  */
426 
427  //workaround 1:
428  int vGridi = mGridV.getKnotIndex(v);
429 
430  float dataU[12];
431  int vOffset = 0;
432  for (int vi = 0; vi < 4; vi++, vOffset += 3) {
433  const RegularSpline1D& gridU = getGridU(vi + vGridi - 1);
434 
435  // and find at which index in that specific spline the u-coordinate must lay.
436  int ui = gridU.getKnotIndex(u);
437 
438  // using getDataIndex we know at which position knot (ui, vi) is saved.
439  // dataU0 to U3 are 4 points along the u-spline surrounding the u coordinate.
440  const float* dataU0 = correctedData + getDataIndex(ui - 1, vGridi + vi - 1);
441 
442  Vc::float_v dt0(dataU0 + 0);
443  Vc::float_v dt1(dataU0 + 3);
444  Vc::float_v dt2(dataU0 + 6);
445  Vc::float_v dt3(dataU0 + 9);
446 
447  Vc::float_v resU = gridU.getSpline(ui, dt0, dt1, dt2, dt3, u);
448 
449  // save the results in dataVx array
450  dataU[vOffset + 0] = resU[0];
451  dataU[vOffset + 1] = resU[1];
452  dataU[vOffset + 2] = resU[2];
453  }
454 
455  Vc::float_v dataV0(dataU + 0);
456  Vc::float_v dataV1(dataU + 3);
457  Vc::float_v dataV2(dataU + 6);
458  Vc::float_v dataV3(dataU + 9);
459 
460  Vc::float_v res = mGridV.getSpline(vGridi, dataV0, dataV1, dataV2, dataV3, v);
461  x = res[0];
462  y = res[1];
463  z = res[2];
464 
465 //getSpline( correctedData, u, v, x, y, z );
466 #else
467  getSpline(correctedData, u, v, x, y, z);
468 #endif
469 }
