100 inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
101 {
102  // convert to row/col w/o over/underflow check
103  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
104  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
105  iRow = int(xRow / PitchRow);
106  iCol = int(zCol / PitchCol);
107  if (xRow < 0)
108  iRow -= 1;
109  if (zCol < 0)
110  iCol -= 1;
111 }
112 
113 //_________________________________________________________________________________________________
114 inline bool SegmentationAlpide::localToDetector(float xRow, float zCol, int& iRow, int& iCol)
115 {
116  // convert to row/col
117  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt left edge of Active matrix
118  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
119  if (xRow < 0 || xRow >= ActiveMatrixSizeRows || zCol < 0 || zCol >= ActiveMatrixSizeCols) {
120  iRow = iCol = -1;
121  return false;
122  }
123  iRow = int(xRow / PitchRow);
124  iCol = int(zCol / PitchCol);
125  return true;
126 }
