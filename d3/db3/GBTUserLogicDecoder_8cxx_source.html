33 void GBTUserLogicDecoder::process(gsl::span<const uint8_t> bytes, uint16_t bc, uint32_t orbit, uint16_t pageCnt)
34 {
36  mOutputHandler.setIR(bc, orbit, pageCnt);
37 
38  bool isFeeding = false;
39  for (auto& byte : bytes) {
40  if (mELinkDecoder.getNBytes() == 0 && (byte & raw::sSTARTBIT) == 0) {
41  // The e-link decoder is empty, meaning that we expect a new board.
42  // The first byte of the board should have the STARTBIT on.
43  // If this is not the case, it means that:
44  // a) there was a problem in the decoding
45  // b) we reached the end of the payload (and we have zeros until the end of the 256 bits word)
46  // In both cases, we need to stop
47  break;
48  }
49  mELinkDecoder.add(byte);
50  if (mELinkDecoder.isComplete()) {
51  if (raw::isLoc(mELinkDecoder.getStatusWord())) {
52  std::invoke(mOnDoneLoc, mOutputHandler, mELinkDecoder.getId() % 8, mELinkDecoder);
53  } else {
54  size_t ilink = 8 + mELinkDecoder.getId() % 8;
55  if (ilink > 9) {
56  continue;
57  }
58  std::invoke(mOnDoneReg, mOutputHandler, 8 + mELinkDecoder.getId() % 8, mELinkDecoder);
59  }
60  mELinkDecoder.reset();
61  }
62  }
63 }
