27 void GPUTPCGMTracksToTPCSeeds::CreateSeedsFromHLTTracks(TObjArray* seeds, AliTPCtracker* tpctracker)
28 {
29  const GPUTPCGMMerger* merger = GPUTPCGlobalMergerComponent::GetCurrentMerger();
30  if (merger == nullptr) {
31  return;
32  }
33  seeds->Clear();
34  int index = 0;
35  for (int i = 0; i < merger->NOutputTracks(); i++) {
36  const GPUTPCGMMergedTrack& track = merger->OutputTracks()[i];
37  if (!track.OK()) {
38  continue;
39  }
40 
41  AliTPCtrack tr;
42  tr.Set(track.GetParam().GetX(), track.GetAlpha(), track.GetParam().GetPar(), track.GetParam().GetCov());
43  AliTPCseed* seed = new (tpctracker->NextFreeSeed()) AliTPCseed(tr);
44  for (int j = 0; j < GPUCA_ROW_COUNT; j++) {
45  seed->SetClusterPointer(j, nullptr);
46  seed->SetClusterIndex(j, -1);
47  }
48  int ncls = 0;
49  int lastrow = -1;
50  int lastleg = -1;
51  for (int j = track.NClusters() - 1; j >= 0; j--) {
52  const GPUTPCGMMergedTrackHit& cls = merger->Clusters()[track.FirstClusterRef() + j];
53  if (cls.state & GPUTPCGMMergedTrackHit::flagReject) {
54  continue;
55  }
56  if (lastrow != -1 && (cls.row < lastrow || cls.leg != lastleg)) {
57  break;
58  }
59  if (cls.row == lastrow) {
60  continue;
61  }
62 
63  AliTPCtrackerRow& row = tpctracker->GetRow(cls.slice % 18, cls.row);
64  unsigned int clIndexOffline = 0;
65  AliTPCclusterMI* clOffline = row.FindNearest2(cls.y, cls.z, 0.01f, 0.01f, clIndexOffline);
66  if (!clOffline) {
67  continue;
68  }
69  clIndexOffline = row.GetIndex(clIndexOffline);
70 
71  clOffline->Use(10);
72  seed->SetClusterPointer(cls.row, clOffline);
73  seed->SetClusterIndex2(cls.row, clIndexOffline);
74 
75  lastrow = cls.row;
76  lastleg = cls.leg;
77  ncls++;
78  }
79 
80  seed->SetRelativeSector(track.GetAlpha() / (M_PI / 9.f));
81  seed->SetNumberOfClusters(ncls);
82  seed->SetNFoundable(ncls);
83  seed->SetChi2(track.GetParam().GetChi2());
84 
85  float alpha = seed->GetAlpha();
86  if (alpha >= 2 * M_PI) {
87  alpha -= 2. * M_PI;
88  }
89  if (alpha < 0) {
90  alpha += 2. * M_PI;
91  }
92  seed->SetRelativeSector(track.GetAlpha() / (M_PI / 9.f));
93 
94  seed->SetPoolID(tpctracker->GetLastSeedId());
95  seed->SetIsSeeding(kTRUE);
96  seed->SetSeed1(GPUCA_ROW_COUNT - 1);
97  seed->SetSeed2(GPUCA_ROW_COUNT - 2);
98  seed->SetSeedType(0);
99  seed->SetFirstPoint(-1);
100  seed->SetLastPoint(-1);
101  seeds->AddLast(seed); // note, track is seed, don't free the seed
102  index++;
103  }
104 }

106 void GPUTPCGMTracksToTPCSeeds::UpdateParamsOuter(TObjArray* seeds)
107 {
108  const GPUTPCGMMerger* merger = GPUTPCGlobalMergerComponent::GetCurrentMerger();
109  if (merger == nullptr) {
110  return;
111  }
112  int index = 0;
113  for (int i = 0; i < merger->NOutputTracks(); i++) {
114  const GPUTPCGMMergedTrack& track = merger->OutputTracks()[i];
115  if (!track.OK()) {
116  continue;
117  }
118  if (index > seeds->GetEntriesFast()) {
119  GPUError("Invalid number of offline seeds");
120  return;
121  }
122  AliTPCseed* seed = (AliTPCseed*)seeds->UncheckedAt(index++);
123  const GPUTPCGMTrackParam::GPUTPCGMTrackParam::GPUTPCOuterParam& param = track.OuterParam();
124  seed->Set(param.X, param.alpha, param.P, param.C);
125  }
