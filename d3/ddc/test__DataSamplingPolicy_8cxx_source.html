54 BOOST_AUTO_TEST_CASE(DataSamplingPolicyConfiguration)
55 {
56  using boost::property_tree::ptree;
57  DataSamplingPolicy policy;
58 
59  ptree config;
60  config.put("id", "my_policy");
61  config.put("active", "true");
62  config.put("query", "c:TST/CHLEB/33;m:TST/MLEKO/33");
63  ptree samplingConditions;
64  ptree conditionRandom;
65  conditionRandom.put("condition", "random");
66  conditionRandom.put("fraction", "0.1");
67  conditionRandom.put("seed", "2137");
68  samplingConditions.push_back(std::make_pair("", conditionRandom));
69  config.add_child("samplingConditions", samplingConditions);
70  config.put("blocking", "false");
71 
72  policy.configure(config);
73 
74  BOOST_CHECK_EQUAL(policy.getName(), "my_policy");
75  BOOST_CHECK((policy.prepareOutput(ConcreteDataMatcher{"TST", "CHLEB", 33})) == (Output{"DS", "my_policy-0", 33}));
76  BOOST_CHECK((policy.prepareOutput(ConcreteDataMatcher{"TST", "MLEKO", 33})) == (Output{"DS", "my_policy-1", 33}));
77  const auto& map = policy.getPathMap();
78  BOOST_CHECK((*map.find(ConcreteDataMatcher{"TST", "CHLEB", 33})).second == (OutputSpec{"DS", "my_policy-0", 33}));
79  BOOST_CHECK((*map.find(ConcreteDataMatcher{"TST", "MLEKO", 33})).second == (OutputSpec{"DS", "my_policy-1", 33}));
80  BOOST_CHECK_EQUAL(map.size(), 2);
81 
82  BOOST_CHECK(policy.match(ConcreteDataMatcher{"TST", "CHLEB", 33}));
83  BOOST_CHECK(!policy.match(ConcreteDataMatcher{"TST", "SZYNKA", 33}));
84 
85  DataProcessingHeader dph{555, 0};
86  o2::header::Stack headerStack{dph};
87  DataRef dr{nullptr, reinterpret_cast<const char*>(headerStack.data()), nullptr};
88  policy.decide(dr); // just make sure it does not crash
89 
90  config.put("id", "too-long-policy-name");
91  policy.configure(config);
92  BOOST_CHECK_EQUAL(policy.getName(), "too-long-polic");
93  BOOST_CHECK((policy.prepareOutput(ConcreteDataMatcher{"TST", "CHLEB", 33})) == (Output{"DS", "too-long-polic-0", 33}));
94  BOOST_CHECK((policy.prepareOutput(ConcreteDataMatcher{"TST", "MLEKO", 33})) == (Output{"DS", "too-long-polic-1", 33}));
95  BOOST_CHECK_EQUAL(policy.getPathMap().size(), 2); // previous paths should be cleared
96 }
