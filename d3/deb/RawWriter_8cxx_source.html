46 void RawWriter::processNextTrigger()
47 {
48  // initialize page handler when processing the first trigger
49  mPageHandler->initTrigger(mCurrentTrigger->getBCData());
50  for (auto srucont : mSRUdata)
51  srucont.mChannels.clear();
52  std::vector<o2::emcal::Digit*>* bunchDigits;
53  int lasttower = -1;
54  for (auto& dig : gsl::span(&mDigits->data()[mCurrentTrigger->getFirstEntry()], mCurrentTrigger->getNumberOfObjects())) {
55  auto tower = dig.getTower();
56  if (tower != lasttower) {
57  lasttower = tower;
58  auto onlineindices = getOnlineID(tower);
59  int sruID = std::get<0>(onlineindices);
60  auto towerdata = mSRUdata[sruID].mChannels.find(tower);
61  if (towerdata == mSRUdata[sruID].mChannels.end()) {
62  mSRUdata[sruID].mChannels[tower] = {std::get<1>(onlineindices), std::get<2>(onlineindices), std::vector<o2::emcal::Digit*>(mNADCSamples)};
63  bunchDigits = &(mSRUdata[sruID].mChannels[tower].mDigits);
64  memset(bunchDigits->data(), 0, sizeof(o2::emcal::Digit*) * mNADCSamples);
65  } else {
66  bunchDigits = &(towerdata->second.mDigits);
67  }
68  (*bunchDigits)[int(dig.getTimeStamp())] = &dig;
69  }
70  }
71 
72  // Create and fill DMA pages for each channel
73  std::vector<char> payload;
74  for (auto srucont : mSRUdata) {
75  // EMCAL does not set HBF triggers, only set trigger BC and orbit
76  o2::header::RAWDataHeaderV4 rawheader;
77  rawheader.triggerBC = mCurrentTrigger->getBCData().bc;
78  rawheader.triggerOrbit = mCurrentTrigger->getBCData().orbit;
79  // @TODO: Set trigger type
80  rawheader.feeId = srucont.mSRUid;
81  rawheader.linkID = srucont.mSRUid;
82 
83  for (const auto& [tower, channel] : srucont.mChannels) {
84  // Find out hardware address of the channel
85  auto hwaddress = mMappers[srucont.mSRUid].getHardwareAddress(channel.mRow, channel.mCol, ChannelType_t::HIGH_GAIN); // @TODO distinguish between high- and low-gain cells
86 
87  std::vector<int> rawbunches;
88  for (auto& bunch : findBunches(channel.mDigits)) {
89  rawbunches.push_back(bunch.mStarttime);
90  rawbunches.push_back(bunch.mADCs.size());
91  for (auto adc : bunch.mADCs) {
92  rawbunches.push_back(adc);
93  }
94  }
95  auto encodedbunches = encodeBunchData(rawbunches);
96  auto chanhead = createChannelHeader(hwaddress, encodedbunches.size() * 3 - 2, false);
97  char* chanheadwords = reinterpret_cast<char*>(&chanhead);
98  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
99  payload.emplace_back(chanheadwords[iword]);
100  }
101  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
102  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
103  payload.emplace_back(channelwords[iword]);
104  }
105  }
106 
107  // Create RCU trailer
108  auto trailerwords = createRCUTrailer(payload.size(), 16, 16, 100., 0.);
109  for (auto word : trailerwords)
110  payload.emplace_back(word);
111 
112  // write DMA page to stream
113  mPageHandler->addPageForLink(srucont.mSRUid, rawheader, payload);
114  }
115 }

117 std::vector<AltroBunch> RawWriter::findBunches(const std::vector<o2::emcal::Digit*>& channelDigits)
118 {
119  std::vector<AltroBunch> result;
120  AltroBunch* currentBunch = nullptr;
121  int starttime = 0;
122  for (auto ien = channelDigits.size() - 1;; ien--) {
123  auto dig = channelDigits[ien];
124  if (!dig) {
125  starttime++;
126  continue;
127  }
128  int adc = dig->getEnergy() / constants::EMCAL_ADCENERGY;
129  if (adc < mPedestal) {
130  // Stop bunch
131  currentBunch = nullptr;
132  starttime++;
133  continue;
134  }
135  if (!currentBunch) {
136  // start new bunch
137  AltroBunch bunch;
138  bunch.mStarttime = starttime;
139  result.push_back(bunch);
140  currentBunch = &(result.back());
141  }
142  currentBunch->mADCs.emplace_back(adc);
143  starttime++;
144  }
145  return result;
146 }

148 std::tuple<int, int, int> RawWriter::getOnlineID(int towerID)
149 {
150  auto cellindex = mGeometry->GetCellIndex(towerID);
151  auto supermoduleID = std::get<0>(cellindex);
152  auto etaphi = mGeometry->GetCellPhiEtaIndexInSModule(supermoduleID, std::get<1>(cellindex), std::get<2>(cellindex), std::get<3>(cellindex));
153  auto etaphishift = mGeometry->ShiftOfflineToOnlineCellIndexes(supermoduleID, std::get<0>(etaphi), std::get<1>(etaphi));
154  int row = std::get<0>(etaphishift), col = std::get<1>(etaphishift);
155 
156  int sruID = -1;
157  if (0 <= row && row < 8)
158  sruID = 0; // first cable row
159  else if (8 <= row && row < 16 && 0 <= col && col < 24)
160  sruID = 0; // first half;
161  else if (8 <= row && row < 16 && 24 <= col && col < 48)
162  sruID = 1; // second half;
163  else if (16 <= row && row < 24)
164  sruID = 1; // third cable row
165  if (supermoduleID % 2 == 1)
166  sruID = 1 - sruID; // swap for odd=C side, to allow us to cable both sides the same
167 
168  return std::make_tuple(sruID, row, col);
169 }

171 std::vector<int> RawWriter::encodeBunchData(const std::vector<int>& data)
172 {
173  std::vector<int> encoded;
174  CaloBunchWord currentword;
175  int wordnumber = 0;
176  for (auto adc : data) {
177  switch (wordnumber) {
178  case 0:
179  currentword.mWord0 = adc;
180  break;
181  case 1:
182  currentword.mWord1 = adc;
183  break;
184  case 2:
185  currentword.mWord2 = adc;
186  break;
187  };
188  if (wordnumber == 2) {
189  // start new word;
190  encoded.push_back(currentword.mDataWord);
191  currentword.mDataWord = 0;
192  wordnumber = 0;
193  } else {
194  wordnumber++;
195  }
196  }
197  return encoded;
198 }
