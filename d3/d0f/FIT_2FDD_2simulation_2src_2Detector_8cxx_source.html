102 Bool_t Detector::ProcessHits(FairVolume* vol)
103 {
104  // This is step manager
105  // do not track neutral particles
106  if (!(fMC->TrackCharge())) {
107  return kFALSE;
108  }
109 
110  Int_t copy;
111  Int_t current_volid = fMC->CurrentVolID(copy);
112 
113  // Get sensitive volumes id (scintillator pads)
114  static Int_t idFDA = fMC->VolId("FDApad");
115  static Int_t idFDC = fMC->VolId("FDCpad");
116 
117  // Get sector copy (1,2,3,4) ( 1 level up from pad )
118  Int_t sect;
119  fMC->CurrentVolOffID(1, sect);
120 
121  // Get Detector copy (1,2) ( 2 levels up from pad )
122  Int_t detc;
123  fMC->CurrentVolOffID(2, detc);
124 
125  // Set detector type: FDA or FDC
126  Int_t ADlayer = (current_volid == idFDC) ? 0 : 2;
127 
128  sect--; // sector within layer [0-3]
129  detc--; // detector copy [0-1]
130  ADlayer += detc; // global layer number [0-3]
131 
132  Int_t ADsector = ADlayer * 4 + sect; // Global AD sector number [0-15]
133  // Layer Sector Number
134  // FDC 0 = 0- 3
135  // FDC 1 = 4- 7
136  // FDA 2 = 8-11
137  // FDA 3 = 12-15
138 
139  Float_t fFDDLightYield(12.8e6);
140  //BC420 yield is 0.64 of antracene which is 20k photons/MeV = 12800/MeV = 12.8e6/GeV
141 
142  Float_t destep_ad = fMC->Edep();
143  Int_t nPhotonsInStep_ad = Int_t(fFDDLightYield * destep_ad);
144  nPhotonsInStep_ad = gRandom->Poisson(nPhotonsInStep_ad);
145 
146  static Float_t eloss_ad = 0.;
147  static Float_t tlength_ad = 0.;
148  static Int_t nPhotons_ad = 0;
149 
150  Float_t x, y, z;
151 
152  eloss_ad += destep_ad;
153 
154  if (fMC->IsTrackEntering()) {
155  nPhotons_ad = nPhotonsInStep_ad;
156  fMC->TrackPosition(x, y, z);
157 
158  eloss_ad = 0.0;
159  return kFALSE;
160  }
161  nPhotons_ad += nPhotonsInStep_ad;
162 
163  if (fMC->IsTrackExiting() || fMC->IsTrackStop() || fMC->IsTrackDisappeared()) {
164  Int_t trackID = fMC->GetStack()->GetCurrentTrackNumber();
165 
166  Float_t time = fMC->TrackTime() * 1.0e9; //time from seconds to ns
167  TVector3 vPos(x, y, z);
168 
169  addHit(trackID, ADsector, vPos, time, eloss_ad, nPhotons_ad);
170  return kTRUE;
171  }
172 
173  return kFALSE;
174 }

177 Hit* Detector::addHit(int trackID, unsigned short detID, const TVector3& Pos, double Time, double eLoss, int nPhot)
178 {
179  //LOG(INFO) << "FDD hit "<<trackID<<" "<<detID<<" "<<Time<<" "<<eLoss<<" "<<nPhot;
