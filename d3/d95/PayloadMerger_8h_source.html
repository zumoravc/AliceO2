70  size_t finalise(char** out, MergeableId& id)
71  {
72  *out = nullptr;
73  if (mCheckIfComplete(id, mPartsMap) == false) {
74  return 0;
75  }
76  // If we are here, it means we can send the messages that belong
77  // to some predefined class of equivalence, identified by the MERGEABLE_ID,
78  // to the receiver. This is done by the following process:
79  //
80  // - Extract what we actually want to send (this might be data embedded inside the message itself)
81  // - Calculate the aggregate size of all the payloads.
82  // - Copy all the parts into a final payload
83  // - Create the header part
84  // - Create the payload part
85  // - Send
86  std::vector<std::pair<char*, size_t>> parts;
87 
88  size_t sum = 0;
89  auto range = mPartsMap.equal_range(id);
90  for (auto hi = range.first, he = range.second; hi != he; ++hi) {
91  std::unique_ptr<FairMQMessage>& payload = hi->second;
92  std::pair<char*, size_t> part;
93  part.second = mExtractPayload(&part.first, reinterpret_cast<char*>(payload->GetData()), payload->GetSize());
94  parts.push_back(part);
95  sum += part.second;
96  }
97 
98  auto* payload = new char[sum]();
99  size_t offset = 0;
100  for (auto& part : parts) {
101  // Right now this does a copy. In principle this could be done with some sort of
102  // vectorized I/O
103  memcpy(payload + offset, part.first, part.second);
104  offset += part.second;
105  }
106 
107  mPartsMap.erase(id);
108  *out = payload;
109  return sum;
110  }
111

113  static int64_t fullPayloadExtractor(char** payload,

115  size_t bufferSize)

117  *payload = buffer;
118  return bufferSize;
119  }
