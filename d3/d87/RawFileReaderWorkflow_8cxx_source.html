44 class rawReaderSpecs : public o2f::Task
45 {
46  public:
47  explicit rawReaderSpecs(const std::string& config, bool tfAsMessage = false, bool outPerRoute = true, int loop = 1, uint32_t delay_us = 0,
48  uint32_t errmap = 0xffffffff, uint32_t minTF = 0, uint32_t maxTF = 0xffffffff, size_t buffSize = 1024L * 1024L)
49  : mLoop(loop < 1 ? 1 : loop), mHBFPerMessage(!tfAsMessage), mOutPerRoute(outPerRoute), mDelayUSec(delay_us), mMinTFID(minTF), mMaxTFID(maxTF), mReader(std::make_unique<o2::raw::RawFileReader>(config))
50  {
51  mReader->setCheckErrors(errmap);
52  mReader->setMaxTFToRead(maxTF);
53  mReader->setBufferSize(buffSize);
54  LOG(INFO) << "Will preprocess files with buffer size of " << buffSize << " bytes";
55  LOG(INFO) << "Number of loops over whole data requested: " << mLoop;
56  if (mHBFPerMessage) {
57  LOG(INFO) << "Every link TF will be sent as multipart of HBF messages";
58  } else {
59  LOG(INFO) << "HBF of single TF of each link will be sent as a single message";
60  }
61  LOG(INFO) << "A message per " << (mOutPerRoute ? "route" : "link") << " will be sent";
62  LOG(INFO) << "Delay of " << mDelayUSec << " microseconds will be added between TFs";
63  }

65  void init(o2f::InitContext& ic) final
66  {
67  assert(mReader);
68  mReader->init();
69  if (mMaxTFID >= mReader->getNTimeFrames()) {
70  mMaxTFID = mReader->getNTimeFrames() - 1;
71  }
72  }

74  void run(o2f::ProcessingContext& ctx) final
75  {
76  assert(mReader);
77  static size_t loopsDone = 0, sentSize = 0, sentMessages = 0;
78  if (mDone) {
79  return;
80  }
81  int nhbexp = HBFUtils::Instance().getNOrbitsPerTF();
82  auto device = ctx.services().get<o2f::RawDeviceService>().device();
83  assert(device);
84 
85  auto findOutputChannel = [&ctx](RawFileReader::LinkData& link) {
86  auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
87  for (auto& oroute : outputRoutes) {
88  LOG(INFO) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
89  if (o2f::DataSpecUtils::match(oroute.matcher, link.origin, link.description, link.subspec)) {
90  link.fairMQChannel = oroute.channel;
91  LOG(INFO) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
92  return true;
93  }
94  }
95  LOGF(ERROR, "Failed to find output channel for %s/%s/0x%x", link.origin.as<std::string>(),
96  link.description.as<std::string>(), link.subspec);
97  return false;
98  };
99 
100  // clean-up before reading next TF
101  auto tfID = mReader->getNextTFToRead();
102  int nlinks = mReader->getNLinks();
103 
104  std::unordered_map<std::string, std::unique_ptr<FairMQParts>> messagesPerRoute;
105  std::vector<std::unique_ptr<FairMQParts>> messagesPerLink;
106  if (!mOutPerRoute) {
107  messagesPerLink.resize(nlinks);
108  }
109 
110  if (tfID > mMaxTFID) {
111  if (mReader->getNTimeFrames() && --mLoop) {
112  loopsDone++;
113  tfID = 0;
114  LOG(INFO) << "Starting new loop " << loopsDone << " from the beginning of data";
115  } else {
116  LOGF(INFO, "Finished: payload of %zu bytes in %zu messages sent for %d TFs", sentSize, sentMessages, mTFIDaccum);
117  ctx.services().get<o2f::ControlService>().endOfStream();
118  ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);
119  mDone = true;
120  return;
121  }
122  }
123 
124  if (tfID < mMinTFID) {
125  tfID = mMinTFID;
126  mReader->setNextTFToRead(tfID);
127  for (int il = 0; il < nlinks; il++) {
128  mReader->getLink(il).rewindToTF(tfID);
129  }
130  }
131 
132  // read next time frame
133  size_t tfNParts = 0, tfSize = 0;
134  LOG(INFO) << "Reading TF#" << mTFIDaccum << " (" << tfID << " at iteration " << loopsDone << ')';
135 
136  for (int il = 0; il < nlinks; il++) {
137  auto& link = mReader->getLink(il);
138 
139  if (link.fairMQChannel.empty() && !findOutputChannel(link)) { // no output channel
140  continue;
141  }
142 
143  o2h::DataHeader hdrTmpl(link.description, link.origin, link.subspec); // template with 0 size
144  int nhb = link.getNHBFinTF();
145  hdrTmpl.payloadSerializationMethod = o2h::gSerializationMethodNone;
146  hdrTmpl.splitPayloadParts = mHBFPerMessage ? nhb : 1;
147 
148  while (hdrTmpl.splitPayloadIndex < hdrTmpl.splitPayloadParts) {
149 
150  tfSize += hdrTmpl.payloadSize = mHBFPerMessage ? link.getNextHBFSize() : link.getNextTFSize();
151  o2::header::Stack headerStack{hdrTmpl, o2::framework::DataProcessingHeader{mTFIDaccum}};
152 
153  auto hdMessage = device->NewMessage(headerStack.size());
154  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
155 
156  auto plMessage = device->NewMessage(hdrTmpl.payloadSize);
157  auto bread = mHBFPerMessage ? link.readNextHBF(reinterpret_cast<char*>(plMessage->GetData())) : link.readNextTF(reinterpret_cast<char*>(plMessage->GetData()));
158  if (bread != hdrTmpl.payloadSize) {
159  LOG(ERROR) << "Link " << il << " read " << bread << " bytes instead of " << hdrTmpl.payloadSize
160  << " expected in TF=" << mTFIDaccum << " part=" << hdrTmpl.splitPayloadIndex;
161  }
162  // check if the RDH to send corresponds to expected orbit
163  if (hdrTmpl.splitPayloadIndex == 0) {
164  uint32_t hbOrbExpected = mReader->getOrbitMin() + tfID * nhbexp;
165  uint32_t hbOrbRead = o2::raw::RDHUtils::getHeartBeatOrbit(plMessage->GetData());
166  if (hbOrbExpected != hbOrbRead) {
167  LOGF(ERROR, "Expected orbit=%u but got %u for %d-th HBF in TF#%d of %s/%s/0x%u",
168  hbOrbExpected, hbOrbRead, hdrTmpl.splitPayloadIndex, tfID,
169  link.origin.as<std::string>(), link.description.as<std::string>(), link.subspec);
170  }
171  hdrTmpl.firstTForbit = hbOrbRead + loopsDone * nhbexp; // for next parts
172  reinterpret_cast<o2::header::DataHeader*>(hdMessage->GetData())->firstTForbit = hdrTmpl.firstTForbit;
173  }
174  FairMQParts* parts = nullptr;
175  if (mOutPerRoute) {
176  parts = messagesPerRoute[link.fairMQChannel].get(); // FairMQParts*
177  if (!parts) {
178  messagesPerRoute[link.fairMQChannel] = std::make_unique<FairMQParts>();
179  parts = messagesPerRoute[link.fairMQChannel].get();
180  }
181  } else { // message per link
182  parts = messagesPerLink[il].get(); // FairMQParts*
183  if (!parts) {
184  messagesPerLink[il] = std::make_unique<FairMQParts>();
185  parts = messagesPerLink[il].get();
186  }
187  }
188  parts->AddPart(std::move(hdMessage));
189  parts->AddPart(std::move(plMessage));
190  hdrTmpl.splitPayloadIndex++; // prepare for next
191  tfNParts++;
192  }
193  LOGF(DEBUG, "Added %d parts for TF#%d(%d in iteration %d) of %s/%s/0x%u", hdrTmpl.splitPayloadParts, mTFIDaccum, tfID,
194  loopsDone, link.origin.as<std::string>(), link.description.as<std::string>(), link.subspec);
195  }
196 
197  if (mTFIDaccum) { // delay sending
198  usleep(mDelayUSec);
199  }
200 
201  if (mOutPerRoute) {
202  for (auto& msgIt : messagesPerRoute) {
203  LOG(INFO) << "Sending " << msgIt.second->Size() / 2 << " parts to channel " << msgIt.first;
204  device->Send(*msgIt.second.get(), msgIt.first);
205  }
206  } else {
207  for (int il = 0; il < nlinks; il++) {
208  auto& link = mReader->getLink(il);
209  auto* parts = messagesPerLink[il].get(); // FairMQParts*
210  if (parts) {
211  LOG(INFO) << "Sending " << parts->Size() / 2 << " parts to channel " << link.fairMQChannel << " for " << link.describe();
212  device->Send(*parts, link.fairMQChannel);
213  }
214  }
215  }
216 
217  LOGF(INFO, "Sent payload of %zu bytes in %zu parts in %zu messages for TF %d", tfSize, tfNParts,
218  (mOutPerRoute ? messagesPerRoute.size() : messagesPerLink.size()), mTFIDaccum);
219  sentSize += tfSize;
220  sentMessages += tfNParts;
221 
222  mReader->setNextTFToRead(++tfID);
223  ++mTFIDaccum;
224  }
225 
226  uint32_t getMinTFID() const { return mMinTFID; }
227  uint32_t getMaxTFID() const { return mMaxTFID; }
228  void setMinMaxTFID(uint32_t mn, uint32_t mx)
229  {
230  mMinTFID = mn;
231  mMaxTFID = mx >= mn ? mx : mn;
232  }
233 
234  private:
235  int mLoop = 0; // once last TF reached, loop while mLoop>=0
236  size_t mTFIDaccum = 0; // TFId accumulator (accounts for looping)
237  uint32_t mDelayUSec = 0; // Delay in microseconds between TFs
238  uint32_t mMinTFID = 0; // 1st TF to extract
239  uint32_t mMaxTFID = 0xffffffff; // last TF to extrct
240  bool mHBFPerMessage = true; // true: send TF as multipart of HBFs, false: single message per TF
241  bool mOutPerRoute = true; // true: send 1 large output route, otherwise 1 outpur per link
242  bool mDone = false; // processing is over or not
243  std::unique_ptr<o2::raw::RawFileReader> mReader; // matching engine
244 };

246 o2f::DataProcessorSpec getReaderSpec(std::string config, bool tfAsMessage, bool outPerRoute, int loop, uint32_t delay_us, uint32_t errmap,
247  uint32_t minTF, uint32_t maxTF, size_t buffSize)
248 {
249  // check which inputs are present in files to read
250  o2f::Outputs outputs;
251  if (!config.empty()) {
252  auto conf = o2::raw::RawFileReader::parseInput(config);
253  for (const auto& entry : conf) {
254  const auto& ordesc = entry.first;
255  if (!entry.second.empty()) { // origin and decription for files to process
256  outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{ordesc.first, ordesc.second}));
257  }
258  }
259  }
260  return o2f::DataProcessorSpec{
261  "raw-file-reader",
262  o2f::Inputs{},
263  outputs,
264  o2f::AlgorithmSpec{o2f::adaptFromTask<rawReaderSpecs>(config, tfAsMessage, outPerRoute, loop, delay_us, errmap, minTF, maxTF, buffSize)},
265  o2f::Options{}};
266 }
