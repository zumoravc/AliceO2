31 void merge(TObject* const target, TObject* const other)
32 {
33  if (target == nullptr) {
34  throw std::runtime_error("Merging target is nullptr");
35  }
36  if (other == nullptr) {
37  throw std::runtime_error("Object to be merged in is nullptr");
38  }
39  if (other == target) {
40  throw std::runtime_error("Merging target and the other object point to the same address");
41  }
42  // fixme: should we check if names match?
43 
44  // We expect that both objects follow the same structure, but we allow to add missing objects to TCollections.
45  // First we check if an object contains a MergeInterface, as it should overlap default Merge() methods of TObject.
46  if (auto custom = dynamic_cast<MergeInterface*>(target)) {
47 
48  custom->merge(dynamic_cast<MergeInterface* const>(other));
49 
50  } else if (auto targetCollection = dynamic_cast<TCollection*>(target)) {
51 
52  auto otherCollection = dynamic_cast<TCollection*>(other);
53  if (otherCollection == nullptr) {
54  throw std::runtime_error(std::string("The target object '") + target->GetName() +
55  "' is a TCollection, while the other object '" + other->GetName() + "' is not.");
56  }
57 
58  auto otherIterator = otherCollection->MakeIterator();
59  while (auto otherObject = otherIterator->Next()) {
60  TObject* targetObject = targetCollection->FindObject(otherObject->GetName());
61  if (targetObject) {
62  // That might be another collection or a concrete object to be merged, we walk on the collection recursively.
63  merge(targetObject, otherObject);
64  } else {
65  // We prefer to clone instead of passing the pointer in order to simplify deleting the `other`.
66  targetCollection->Add(otherObject->Clone());
67  }
68  }
69  delete otherIterator;
70  } else {
71  Long64_t errorCode = 0;
72  TObjArray otherCollection;
73  otherCollection.SetOwner(false);
74  otherCollection.Add(other);
75 
76  if (target->InheritsFrom(TH1::Class())) {
77  // this includes TH1, TH2, TH3
78  errorCode = reinterpret_cast<TH1*>(target)->Merge(&otherCollection);
79  } else if (target->InheritsFrom(THnBase::Class())) {
80  // this includes THn and THnSparse
81  errorCode = reinterpret_cast<THnBase*>(target)->Merge(&otherCollection);
82  } else if (target->InheritsFrom(TTree::Class())) {
83  errorCode = reinterpret_cast<TTree*>(target)->Merge(&otherCollection);
84  } else {
85  throw std::runtime_error("Object with type '" + std::string(target->ClassName()) + "' is not one of the mergeable types.");
86  }
87  if (errorCode == -1) {
88  throw std::runtime_error("Merging object of type '" + std::string(target->ClassName()) + "' failed.");
89  }
90  }
91 }

93 void deleteTCollections(TObject* obj)
94 {
95  if (auto c = dynamic_cast<TCollection*>(obj)) {
96  c->SetOwner(false);
97  auto iter = c->MakeIterator();
98  while (auto element = iter->Next()) {
99  deleteTCollections(element);
100  }
101  delete iter;
102  delete c;
103  } else {
104  delete obj;
105  }
