50 void Digitizer::process(const std::vector<Hit>* hitsBg, const std::vector<Hit>* hitsS, std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& labels)
51 {
52  // Convert list of hits to digits:
53  // Add hits with ampl deposition in same pad and same time
54  // Add ampl corrections
55  // Apply time smearing
56 
57  std::vector<Hit>::const_iterator hitBg = hitsBg->cbegin();
58  std::vector<Hit>::const_iterator hitS = hitsS->cbegin();
59  std::vector<Hit>::const_iterator hit;
60  const short kBigAbsID = 32767; //Far above maximal CPV absId
61  short hitAbsId = kBigAbsID;
62  short hitBgAbsId = kBigAbsID;
63  short hitSAbsId = kBigAbsID;
64 
65  if (hitBg != hitsBg->end()) {
66  hitBgAbsId = hitBg->GetDetectorID();
67  }
68  if (hitS != hitsS->end()) {
69  hitSAbsId = hitS->GetDetectorID();
70  }
71  if (hitBgAbsId < hitSAbsId) { // Bg hit exists and smaller than signal
72  hitAbsId = hitBgAbsId;
73  hit = hitBg;
74  mCurrSrcID = 0;
75  ++hitBg;
76  } else {
77  if (hitSAbsId < kBigAbsID) { //Signal hit exists and smaller than Bg
78  hitAbsId = hitSAbsId;
79  hit = hitS;
80  mCurrSrcID = 1;
81  ++hitS;
82  }
83  }
84 
85  short nTotPads = Geometry::getTotalNPads();
86  for (short absId = 1; absId < nTotPads; absId++) {
87 
88  // If signal exist in this pad, add noise to it, otherwise just create noise digit
89  if (absId == hitAbsId) {
90  int labelIndex = labels.getIndexedSize();
91  //Add primary info: create new MCLabels entry
92  o2::MCCompLabel label(hit->GetTrackID(), mCurrEvID, mCurrSrcID, true);
93  labels.addElement(labelIndex, label);
94 
95  Digit digit((*hit), labelIndex);
96 
97  //May be add more hits to this digit?
98  if (hitBg == hitsBg->end()) {
99  hitBgAbsId = kBigAbsID;
100  } else {
101  hitBgAbsId = hitBg->GetDetectorID();
102  }
103  if (hitS == hitsS->end()) {
104  hitSAbsId = kBigAbsID;
105  } else {
106  hitSAbsId = hitS->GetDetectorID();
107  }
108  if (hitBgAbsId < hitSAbsId) { // Bg hit exists and smaller than signal
109  hitAbsId = hitBgAbsId;
110  hit = hitBg;
111  mCurrSrcID = 0;
112  ++hitBg;
113  } else {
114  if (hitSAbsId < kBigAbsID) { //Signal hit exists and smaller than Bg
115  hitAbsId = hitSAbsId;
116  hit = hitS;
117  mCurrSrcID = 1;
118  ++hitS;
119  } else { //no hits left
120  hitAbsId = kBigAbsID;
121  continue;
122  }
123  }
124 
125  while (absId == hitAbsId) {
126  Digit digitNext((*hit), labelIndex); //Use same MCTruth entry so far
127  digit += digitNext;
128 
129  //add MCLabel to list (add energy if same primary or add another label)
130  o2::MCCompLabel label(hit->GetTrackID(), mCurrEvID, mCurrSrcID, true);
131  labels.addElement(labelIndex, label);
132 
133  //next hit?
134  if (hitBg == hitsBg->end()) {
135  hitBgAbsId = kBigAbsID;
136  } else {
137  hitBgAbsId = hitBg->GetDetectorID();
138  }
139  if (hitS == hitsS->end()) {
140  hitSAbsId = kBigAbsID;
141  } else {
142  hitSAbsId = hitS->GetDetectorID();
143  }
144  if (hitBgAbsId < hitSAbsId) { // Bg hit exists and smaller than signal
145  hitAbsId = hitBgAbsId;
146  hit = hitBg;
147  mCurrSrcID = 0;
148  ++hitBg;
149  } else {
150  if (hitSAbsId < kBigAbsID) { //Signal hit exists and smaller than Bg
151  hitAbsId = hitSAbsId;
152  hit = hitS;
153  mCurrSrcID = 1;
154  ++hitS;
155  } else { //no hits left
156  hitAbsId = kBigAbsID;
157  digitNext.setAbsId(kBigAbsID);
158  continue;
159  }
160  }
161 
162  digitNext.fillFromHit(*hit);
163  }
164  // //Current digit finished, sort MCLabels according to eDeposited
165  // auto lbls = labels.getLabels(labelIndex);
166  // std::sort(lbls.begin(), lbls.end(),
167  // [](o2::MCCompLabel a, o2::MCCompLabel b) { return a.getEdep() > b.getEdep(); });
168 
169  // Add Electroinc noise, apply non-linearity, digitize, de-calibrate, time resolution
170  float ampl = digit.getAmplitude();
171  // Simulate electronic noise
172  ampl += simulateNoise();
173 
174  ampl = uncalibrate(ampl, absId);
175 
176  if (ampl < o2::cpv::CPVSimParams::Instance().mZSthreshold) {
177  continue;
178  }
179  digit.setAmplitude(ampl);
180 
181  digits.push_back(digit);
182  } else { // No signal in this pad,
183  if (!Geometry::IsPadExists(absId)) {
184  continue;
185  }
186  // Simulate noise
187  float ampl = simulateNoise();
188  ampl = uncalibrate(ampl, absId);
189  if (ampl > o2::cpv::CPVSimParams::Instance().mZSthreshold) {
190  digits.emplace_back(absId, ampl, -1); // current AbsId, energy, no primary
191  }
192  }
193  }
194 }

199 float Digitizer::uncalibrate(const float e, const int absId)
200 {
201  // Decalibrate CPV digit, i.e. transform from amplitude to ADC counts a factor read from CDB
202  float calib = mCalibParams->getGain(absId);
203  if (calib > 0) {
204  return floor(e / calib);
205  } else {
206  return 0; // TODO apply de-calibration from OCDB
207  }
208 }
