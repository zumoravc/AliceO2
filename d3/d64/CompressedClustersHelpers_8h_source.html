51  template <typename ContainerType>
52  static size_t flattenTo(ContainerType& container, CompressedClustersROOT& clusters)
53  {
54  static_assert(sizeof(typename ContainerType::value_type) == 1);
55  char* dummyptr = nullptr;
56  auto calc_size = [](auto&... args) { return flatten::calc_size(args...); };
57  auto copy_to = [](auto&... args) { return flatten::copy_to(args...); };
58  size_t size = apply(calc_size, dummyptr, clusters);
59  container.resize(size);
60  auto wrtptr = container.data();
61  size_t copySize = apply(copy_to, wrtptr, clusters);
62  assert(copySize == size);
63  return copySize;
64  }

67  template <typename ContainerType>
68  static size_t restoreFrom(ContainerType& container, CompressedClustersROOT& clusters)
69  {
70  static_assert(sizeof(typename ContainerType::value_type) == 1);
71  char* dummyptr = nullptr;
72  auto calc_size = [](auto&... args) { return flatten::calc_size(args...); };
73  auto set_from = [](auto&... args) { return flatten::set_from(args...); };
74  size_t size = apply(calc_size, dummyptr, clusters);
75  if (container.size() != size) {
76  // for the moment we don not support changes in the member layout, we can implement the
77  // handling in the custom streamer
78  throw std::runtime_error("mismatch between raw buffer size and counters, schema evolution is not yet supported");
79  }
80  auto readptr = container.data();
81  size_t checkSize = apply(set_from, readptr, clusters);
82  assert(checkSize == size);
83  return checkSize;
84  }
