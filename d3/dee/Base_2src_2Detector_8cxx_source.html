119 void Detector::SetSpecialPhysicsCuts()
120 {
121  // default implementation for physics cuts setting (might still be overriden by detectors)
122  // we try to read an external text file supposed to be installed
123  // in a standard directory
124  // ${O2_ROOT}/share/Detectors/DETECTORNAME/simulation/data/simcuts.dat
125  LOG(INFO) << "Setting special cuts for " << GetName();
126  const char* aliceO2env = std::getenv("O2_ROOT");
127  std::string inputFile;
128  if (aliceO2env) {
129  inputFile = std::string(aliceO2env);
130  }
131  inputFile += "/share/Detectors/" + std::string(GetName()) + "/simulation/data/simcuts.dat";
132  auto& matmgr = o2::base::MaterialManager::Instance();
133  matmgr.loadCutsAndProcessesFromFile(GetName(), inputFile.c_str());

135  // TODO:

137  // via command line

140 void Detector::initFieldTrackingParams(int& integration, float& maxfield)
141 {
142  // set reasonable default values
143  integration = 2;
144  maxfield = 10;
145  auto vmc = TVirtualMC::GetMC();
146  if (vmc) {
147  auto field = vmc->GetMagField();
148  // see if we can query the o2 field
149  if (auto o2field = dynamic_cast<o2::field::MagneticField*>(field)) {
150  integration = o2field->Integral(); // default integration method?
151  maxfield = o2field->Max();
152  return;
153  }
154  }
155  LOG(INFO) << "No magnetic field found; using default tracking values " << integration << " " << maxfield
156  << " to initialize media\n";
157 }

171 int Detector::registerSensitiveVolumeAndGetVolID(TGeoVolume const* vol)
172 {
173  // register this volume with FairRoot
174  this->FairModule::AddSensitiveVolume(const_cast<TGeoVolume*>(vol));
175  // retrieve the VMC Monte Carlo ID for this volume
176  const int volid = TVirtualMC::GetMC()->VolId(vol->GetName());
177  if (volid <= 0) {
178  LOG(ERROR) << "Could not retrieve VMC volume ID for " << vol->GetName();
179  }
180  return volid;
181 }
