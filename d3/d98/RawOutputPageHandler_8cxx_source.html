36 void RawOutputPageHandler::initTrigger(const o2::InteractionRecord& irtrigger)
37 {
38  auto currenttimeframe = raw::HBFUtils::Instance().getTF(irtrigger);
39  if (currenttimeframe != mCurrentTimeframe) {
40  // write pages to file
41  for (auto& [linkID, pagebuffer] : mRawPages) {
42  // only write pages for links which send data
43  if (pagebuffer.getPages().size())
44  writeTimeframe(linkID, pagebuffer);
45  pagebuffer.flush();
46  }
47  // set the new timeframe
48  mCurrentTimeframe = currenttimeframe;
49  }
50 }

52 void RawOutputPageHandler::writeTimeframe(int linkID, const RawPageBuffer& pagebuffer)
53 {
54  // write pages to file
55  // Write timeframe open
56  RawHeader timeframeheader;
57  timeframeheader.linkID = linkID;
58  mOutputStream.writeSingleHeader(timeframeheader);
59  int pagecounter = 1;
60  for (const auto& page : pagebuffer.getPages()) {
61  auto header = page.mHeader;
62  header.pageCnt = pagecounter;
63  pagecounter = mOutputStream.writeData(header, gsl::span<const char>(page.mPage.data(), page.mPage.size()));
64  }
65 
66  // end of timeframe
67  timeframeheader.pageCnt = pagecounter;
68  timeframeheader.stop = 1;
69  timeframeheader.triggerType = o2::trigger::TF;
70  mOutputStream.writeSingleHeader(timeframeheader);
71 }
