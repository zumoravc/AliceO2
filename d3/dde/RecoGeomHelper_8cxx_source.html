38 void RecoGeomHelper::RecoLadder::updateLimits(const Point3D<float>& pntGlo)
39 {
40  // update limits from the point in Global frame
41  float phi = pntGlo.phi(); // -pi:pi range
42  o2::utils::BringTo02Pi(phi); // temporary bring to 0:2pi range
43  o2::utils::BringTo02Pi(phiRange.min());
44  o2::utils::BringTo02Pi(phiRange.max());
45  phiRange.update(phi);
46  phiMean = phiRange.mean();
47  dphiH = 0.5 * phiRange.delta();
48  if (phiRange.delta() > o2::constants::math::PI) { // wrapping, swap
49  phiRange.set(phiRange.max(), phiRange.min()); // swap
50  phiMean -= o2::constants::math::PI;
51  dphiH = o2::constants::math::PI - dphiH;
52  }
53  o2::utils::BringToPMPi(phiRange.min()); // -pi:pi range
54  o2::utils::BringToPMPi(phiRange.max());
55  o2::utils::BringToPMPi(phiMean);
56  //
57  zRange.update(pntGlo.Z());
58 }

61 void RecoGeomHelper::RecoLadder::init()
62 {
63  auto& chip = chips[0].xyEdges;
64  float x0(chip.getX0()), y0(chip.getY0()), x1(chip.getX1()), y1(chip.getY1());
65  for (int i = 1; i < (int)chips.size(); i++) {
66  chip = chips[i].xyEdges;
67  x0 = chip.getDX() > 0 ? std::min(x0, chip.getX0()) : std::max(x0, chip.getX0());
68  x1 = chip.getDX() > 0 ? std::max(x1, chip.getX1()) : std::min(x1, chip.getX1());
69  y0 = chip.getDY() > 0 ? std::min(y0, chip.getY0()) : std::max(y0, chip.getY0());
70  y1 = chip.getDY() > 0 ? std::max(y1, chip.getY1()) : std::min(y1, chip.getY1());
71  }
72  xyEdges.setEdges(x0, y0, x1, y1);
73 }

89 void RecoGeomHelper::RecoLayer::init()
90 {
91  auto gm = o2::its::GeometryTGeo::Instance();
92  gm->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2GRot, o2::TransformType::T2L)); // more matrices ?
93 
94  int nHStaves = gm->getNumberOfHalfStaves(id);
95  int nStaves = gm->getNumberOfStaves(id);
96  float dxH = o2::itsmft::SegmentationAlpide::SensorSizeRows / 2; // half width in rphi
97  float dzH = o2::itsmft::SegmentationAlpide::SensorSizeCols / 2; // half width in Z
98  int nCh = gm->getNumberOfChipsPerLayer(id), chip0 = gm->getFirstChipIndex(id);
99  int nChMod = gm->getNumberOfChipsPerModule(id), nChModH = nChMod / 2;
100  nLadders = nStaves * nHStaves * (id > 2 ? 2 : 1); // 2 ladders per h-stave for OB
101  ladders.resize(nLadders);
102 
103  for (auto lad : ladders) {
104  lad.chips.reserve(gm->getNumberOfChipsPerHalfStave(id) / (id > 2 ? 2 : 1)); // 2 ladders per h-stave for OB
105  }
106  for (int ich = 0; ich < nCh; ich++) {
107  int chipID = chip0 + ich, lay, sta, ssta, mod, chipInMod;
108  gm->getChipId(chipID, lay, sta, ssta, mod, chipInMod);
109  int ladID = sta, chipInLadder = nChMod - chipInMod - 1; // count from negative to positive Z, contrary to official chips numbering
110  if (nHStaves > 1) { // OB
111  int modUpper = chipInMod / nChModH;
112  ladID = sta * 4 + ssta * 2 + modUpper; // OB module covers 2 "ladders"
113  }
114  auto& ladder = ladders[ladID];
115  auto& chip = ladder.chips.emplace_back();
116  chip.id = chipID;
117  gm->getSensorXAlphaRefPlane(chipID, chip.xRef, chip.alp);
118  o2::utils::sincosf(chip.alp, chip.snAlp, chip.csAlp);
119 
120  Point3D<float> edgeLoc(-dxH, 0.f, -dzH);
121  auto edgeTra = gm->getMatrixT2L(chipID) ^ (edgeLoc); // edge in tracking frame
122  chip.updateLimits(edgeTra);
123  auto edgeGloM = gm->getMatrixT2GRot(chipID)(edgeTra); // edge in global frame
124  updateLimits(edgeGloM);
125  ladder.updateLimits(edgeGloM);
126 
127  edgeLoc.SetXYZ(dxH, 0.f, dzH);
128  edgeTra = gm->getMatrixT2L(chipID) ^ (edgeLoc); // edge in tracking frame
129  chip.updateLimits(edgeTra);
130  auto edgeGloP = gm->getMatrixT2GRot(chipID)(edgeTra); // edge in globalframe
131  updateLimits(edgeGloP);
132  ladder.updateLimits(edgeGloP);
133  chip.xyEdges.setEdges(edgeGloM.X(), edgeGloM.Y(), edgeGloP.X(), edgeGloP.Y());
134  }
135 
136  // sort according to mean phi (in -pi:pi range!!!)
137  std::sort(ladders.begin(), ladders.end(), [](auto& a, auto& b) {
138  float pha = a.phiMean, phb = b.phiMean;
139  o2::utils::BringTo02Pi(pha);
140  o2::utils::BringTo02Pi(phb);
141  return pha < phb;
142  });
143 
144  // make sure chips within the ladder are ordered in Z, renumber ladders
145  for (int i = nLadders; i--;) {
146  auto& lad = ladders[i];
147  std::sort(lad.chips.begin(), lad.chips.end(), [](auto& a, auto& b) { return a.zRange.min() < b.zRange.min(); });
148  lad.id = i;
149  lad.init();
150  }
151  // relate to its neighbour: is the egde at higher R than the edge of the next ladder and do we check for overlaps ?
152  for (int i = nLadders; i--;) {
153  auto& lad = ladders[i];
154  auto& plad = i > 0 ? ladders[i - 1] : ladders[nLadders - 1];
155  if (nHStaves == 2) { // on the OB the ladders of the same halfstave do not overlap
156  int tlay, tsta, tssta, tmod, tchipInMod;
157  int play, psta, pssta, pmod, pchipInMod;
158  gm->getChipId(lad.chips.front().id, tlay, tsta, tssta, tmod, tchipInMod);
159  gm->getChipId(plad.chips.front().id, play, psta, pssta, pmod, pchipInMod);
160  if (tsta == psta && tssta == pssta) { // these are 2 ladders of the same halfstave, no overlap
161  plad.overlapWithNext = RecoLadder::NoOverlap;
162  continue;
163  }
164  }
165  // need to compare the radius of the edge at lowest phi for this ladder with radius at highest phi of prev ladder
166  // find the radius of the edge at low phi
167  float phi0 = std::atan2(lad.xyEdges.getY0(), lad.xyEdges.getX0());
168  float phi1 = std::atan2(lad.xyEdges.getY1(), lad.xyEdges.getX1());
169  o2::utils::BringTo02Pi(phi0); // we don't know a priori if edge0/1 corresponds to low/high phi or vice versa
170  o2::utils::BringTo02Pi(phi1);
171  float r2This = (phi0 < phi1 && phi1 - phi0 < o2::constants::math::PI) ? // pick R of lowest angle
172  lad.xyEdges.getX0() * lad.xyEdges.getX0() + lad.xyEdges.getY0() * lad.xyEdges.getY0()
173  : lad.xyEdges.getX1() * lad.xyEdges.getX1() + lad.xyEdges.getY1() * lad.xyEdges.getY1();
174  //
175  phi0 = std::atan2(plad.xyEdges.getY0(), plad.xyEdges.getX0());
176  phi1 = std::atan2(plad.xyEdges.getY1(), plad.xyEdges.getX1());
177  o2::utils::BringTo02Pi(phi0); // we don't know a priori if edge0/1 corresponds to low/high phi or vice versa
178  o2::utils::BringTo02Pi(phi1);
179  float r2Prev = (phi0 < phi1 && phi1 - phi0 < o2::constants::math::PI) ? // pick R of highest angle
180  plad.xyEdges.getX1() * plad.xyEdges.getX1() + plad.xyEdges.getY1() * plad.xyEdges.getY1()
181  : plad.xyEdges.getX0() * plad.xyEdges.getX0() + plad.xyEdges.getY0() * plad.xyEdges.getY0();
182  //
183  plad.overlapWithNext = r2Prev > r2This ? RecoLadder::Above : RecoLadder::Below;
184  }
185 
186  int ndiv = nLadders * 3; // number of bins for mapping
187  phi2ladder.resize(ndiv);
188  float dphi = o2::constants::math::TwoPI / ndiv;
189  int laddId = 0;
190  for (int i = 0; i < ndiv; i++) {
191  float phi = (0.5 + i) * dphi;
192  o2::utils::BringToPMPi(phi);
193  while (laddId < nLadders) {
194  const auto& lad = ladders[laddId];
195  auto rel = lad.isPhiOutside(phi);
196  if (rel != RecoGeomHelper::Above) {
197  break;
198  }
199  laddId++; // laddId was below phi, catch up
200  }
201  phi2ladder[i] = laddId % nLadders;
202  }
203  lastChipInLadder = ladders[0].chips.size();
204  z2chipID = lastChipInLadder / zRange.delta();
205  lastChipInLadder--;
206  rInv = 1. / rRange.mean();
207 }
