78  value_type operator()()
79  {
80  value_type v;

82  while ((v = mModel(mGenerator)) < min || v >= max) {
83  if (trials++ > 1000) {

180 template <class RealType = double, class _BASE = std::normal_distribution<RealType>>
181 class normal_distribution : public _BASE
182 {
183  public:
184  using result_type = typename _BASE::result_type;
185 
186  normal_distribution(result_type _mean, result_type _stddev) : _BASE(_mean, _stddev), mean(_mean), stddev(_stddev) {}
187 
188  const double sqrt2pi = 2.5066283;
189  const result_type mean;
190  const result_type stddev;
191 
193  // if value_type is an integral type we want to have the probability
194  // that the result value is in the range [v, v+1) whereas the step
195  // can be something else than 1
196  // also the values outside the specified range should be excluded
197  // and the probability for intervals in the range has to be scaled
198  template <typename value_type>
199  double getProbability(value_type v) const
200  {
201  return (exp(-(v - mean) * (v - mean) / (2 * stddev * stddev))) / (stddev * sqrt2pi);
202  }
203 };

220 
221  int factorial(unsigned int n) const { return (n <= 1) ? 1 : factorial(n - 1) * n; }
222 
224  template <typename value_type>
225  double getProbability(value_type v) const
226  {
227  if (v < 0) {
228  return 0.;
229  }
230  return pow(mean, v) * exp(-mean) / factorial(v);
231  }
