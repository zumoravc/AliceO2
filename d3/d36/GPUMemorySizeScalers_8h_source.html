52  double NTPCTracks(double tpcHits) { return offset + tpcHits * tpcTracksPerHit * factor; }
53  double NTPCTrackHits(double tpcHits) { return offset + tpcHits * tpcTrackHitsPerHit * factor; }
54  double NTPCMaxRowStartHits(double tpcHits) { return offset + NTPCStartHits(tpcHits) / GPUCA_ROW_COUNT * 3. * factor; }
55 };
