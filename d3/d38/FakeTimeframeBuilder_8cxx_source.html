26 o2::header::DataDescription lookupDataDescription(const char* key)
27 {
28  if (strcmp(key, "RAWDATA") == 0)
29  return o2::header::gDataDescriptionRawData;
30  else if (strcmp(key, "CLUSTERS") == 0)
31  return o2::header::gDataDescriptionClusters;
32  else if (strcmp(key, "TRACKS") == 0)
33  return o2::header::gDataDescriptionTracks;
34  else if (strcmp(key, "CONFIG") == 0)
35  return o2::header::gDataDescriptionConfig;
36  else if (strcmp(key, "INFO") == 0)
37  return o2::header::gDataDescriptionInfo;
38  return o2::header::gDataDescriptionInvalid;
39 }

61 std::unique_ptr<char[]> fakeTimeframeGenerator(std::vector<FakeTimeframeSpec>& specs, std::size_t& totalSize)
62 {
63  // Calculate the total size of your timeframe. This is
64  // given by:
65  // - N*The size of the data header (this should actually depend on the
66  // kind of data as different dataDescriptions will probably have
67  // different headers).
68  // - Sum_N(The size of the buffer_i)
69  // - The size of the index header
70  // - N*sizeof(dataheader)
71  // Assuming all the data header
72  size_t sizeOfHeaders = specs.size() * sizeof(DataHeader);
73  size_t sizeOfBuffers = 0;
74  for (auto&& spec : specs) {
75  sizeOfBuffers += spec.bufferSize;
76  }
77  size_t sizeOfIndexHeader = sizeof(DataHeader);
78  size_t sizeOfIndex = sizeof(IndexElement) * specs.size();
79  totalSize = sizeOfHeaders + sizeOfBuffers + sizeOfIndexHeader + sizeOfIndex;
80 
81  // Add the actual - data
82  auto buffer = std::make_unique<char[]>(totalSize);
83  char* bi = buffer.get();
84  std::vector<IndexElement> headers;
85  int count = 0;
86  for (auto&& spec : specs) {
87  IndexElement el;
88  el.first.dataDescription = lookupDataDescription(spec.dataDescription);
89  el.first.dataOrigin = lookupDataOrigin(spec.origin);
90  el.first.payloadSize = spec.bufferSize;
91  el.first.headerSize = sizeof(el.first);
92  el.second = count++;
93  // Let's zero at least the header...
94  memset(bi, 0, sizeof(el.first));
95  memcpy(bi, &el, sizeof(el.first));
96  headers.push_back(el);
97  bi += sizeof(el.first);
98  spec.bufferFiller(bi, spec.bufferSize);
99  bi += spec.bufferSize;
100  }
101 
102  // Add the index
103  DataHeader index;
104  index.dataDescription = DataDescription("TIMEFRAMEINDEX");
105  index.dataOrigin = DataOrigin("FKE");
106  index.headerSize = sizeOfIndexHeader;
107  index.payloadSize = sizeOfIndex;
108  memcpy(bi, &index, sizeof(index));
109  memcpy(bi + sizeof(index), headers.data(), headers.size() * sizeof(IndexElement));
110  return std::move(buffer);
111 }
