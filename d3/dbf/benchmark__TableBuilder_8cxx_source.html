80 static void BM_TableBuilderScalarBulk(benchmark::State& state)
81 {
82  using namespace o2::framework;
83  auto chunkSize = state.range(0) / 256;
84  std::vector<float> buffer(chunkSize, 0.); // We assume data is chunked in blocks 256th of the total size
85  for (auto _ : state) {
86  TableBuilder builder;
87  auto bulkWriter = builder.bulkPersist<float>({"x"}, state.range(0));
88  for (size_t i = 0; i < state.range(0) / chunkSize; ++i) {
89  bulkWriter(0, chunkSize, buffer.data());
90  }
91  auto table = builder.finalize();
92  }
93 }
