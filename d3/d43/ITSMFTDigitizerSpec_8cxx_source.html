49  void initDigitizerTask(framework::InitContext& ic) override
50  {
51  // init optional QED chain
52  auto qedfilename = ic.options().get<std::string>("simFileQED");
53  if (qedfilename.size() > 0) {
54  mQEDChain.AddFile(qedfilename.c_str());
55  LOG(INFO) << "Attach QED Tree: " << mQEDChain.GetEntries();
56  }
57 
58  setDigitizationOptions(); // set options provided via configKeyValues mechanism
59  auto& digipar = mDigitizer.getParams();
60 
61  mROMode = digipar.isContinuous() ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
62  LOG(INFO) << mID.getName() << " simulated in "
63  << ((mROMode == o2::parameters::GRPObject::CONTINUOUS) ? "CONTINUOUS" : "TRIGGERED")
64  << " RO mode";
65 
66  // configure digitizer
67  o2::itsmft::GeometryTGeo* geom = nullptr;
68  if (mID == o2::detectors::DetID::ITS) {
69  geom = o2::its::GeometryTGeo::Instance();
70  } else {
71  geom = o2::mft::GeometryTGeo::Instance();
72  }
73  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::L2G)); // make sure L2G matrices are loaded
74  mDigitizer.setGeometry(geom);
75 
76  // init digitizer
77  mDigitizer.init();
78  }

82  void run(framework::ProcessingContext& pc)
83  {
84  if (mFinished) {
85  return;
86  }
87  std::string detStr = mID.getName();
88  // read collision context from input
89  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
90  context->initSimChains(mID, mSimChains);
91  auto& timesview = context->getEventRecords();
92  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
93  // if there is nothing to do ... return
94  if (timesview.size() == 0) {
95  return;
96  }
97  TStopwatch timer;
98  timer.Start();
99  LOG(INFO) << " CALLING ITS DIGITIZATION ";
100 
101  mDigitizer.setDigits(&mDigits);
102  mDigitizer.setROFRecords(&mROFRecords);
103  mDigitizer.setMCLabels(&mLabels);
104 
105  // attach optional QED digits branch
106  setupQEDChain();
107 
108  auto& eventParts = context->getEventParts();
109  // loop over all composite collisions given from context (aka loop over all the interaction records)
110  for (int collID = 0; collID < timesview.size(); ++collID) {
111  auto eventTime = timesview[collID].timeNS;
112 
113  if (mQEDChain.GetEntries()) { // QED must be processed before other inputs since done in small time steps
114  processQED(eventTime);
115  }
116 
117  mDigitizer.setEventTime(eventTime);
118  mDigitizer.resetEventROFrames(); // to estimate min/max ROF for this collID
119  // for each collision, loop over the constituents event and source IDs
120  // (background signal merging is basically taking place here)
121  for (auto& part : eventParts[collID]) {
122 
123  // get the hits for this event and this source
124  mHits.clear();
125  context->retrieveHits(mSimChains, o2::detectors::SimTraits::DETECTORBRANCHNAMES[mID][0].c_str(), part.sourceID, part.entryID, &mHits);
126 
127  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID
128  << " found " << mHits.size() << " hits ";
129 
130  mDigitizer.process(&mHits, part.entryID, part.sourceID); // call actual digitization procedure
131  }
132  mMC2ROFRecordsAccum.emplace_back(collID, -1, mDigitizer.getEventROFrameMin(), mDigitizer.getEventROFrameMax());
133  accumulate();
134  }
135  // finish digitization ... stream any remaining digits/labels
136  if (mQEDChain.GetEntries()) { // fill last slots from QED input
137  processQED(mDigitizer.getEndTimeOfROFMax());
138  }
139 
140  mDigitizer.fillOutputContainer();
141  accumulate();
142 
143  // here we have all digits and labels and we can send them to consumer (aka snapshot it onto output)
144  pc.outputs().snapshot(Output{mOrigin, "DIGITS", 0, Lifetime::Timeframe}, mDigitsAccum);
145  pc.outputs().snapshot(Output{mOrigin, "DIGITSROF", 0, Lifetime::Timeframe}, mROFRecordsAccum);
146  if (mWithMCTruth) {
147  pc.outputs().snapshot(Output{mOrigin, "DIGITSMC2ROF", 0, Lifetime::Timeframe}, mMC2ROFRecordsAccum);
148  pc.outputs().snapshot(Output{mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe}, mLabelsAccum);
149  }
150  LOG(INFO) << mID.getName() << ": Sending ROMode= " << mROMode << " to GRPUpdater";
151  pc.outputs().snapshot(Output{mOrigin, "ROMode", 0, Lifetime::Timeframe}, mROMode);
152 
153  timer.Stop();
154  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
155 
156  // we should be only called once; tell DPL that this process is ready to exit
157  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
158 
159  mFinished = true;
160  }

165  void processQED(double tMax)
166  {
167  auto tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS; // timeslice to retrieve
168  std::string detStr = mID.getName();
169  auto br = mQEDChain.GetBranch((detStr + "Hit").c_str());
170  while (tQEDNext < tMax) {
171  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
172  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
173  if (++mLastQEDEntry >= mQEDChain.GetEntries()) {
174  mLastQEDEntry = 0; // wrapp if needed
175  }
176  br->GetEntry(mLastQEDEntry);
177  mDigitizer.setEventTime(mLastQEDTimeNS);
178  mDigitizer.process(&mHits, mLastQEDEntry, QEDSourceID);
179  //
180  }
181  }

183  void setupQEDChain()
184  {
185  if (!mQEDChain.GetEntries()) {
186  return;
187  }
188  std::string detStr = mID.getName();
189  auto qedBranch = mQEDChain.GetBranch((detStr + "Hit").c_str());
190  assert(qedBranch != nullptr);
191  assert(mQEDEntryTimeBinNS >= 1.0);
192  assert(QEDSourceID < o2::MCCompLabel::maxSourceID());
193  mLastQEDTimeNS = -mQEDEntryTimeBinNS / 2; // time will be assigned to the middle of the bin
194  qedBranch->SetAddress(&mHitsP);
195  LOG(INFO) << "Attaching QED ITS hits as sourceID=" << int(QEDSourceID) << ", entry integrates "
196  << mQEDEntryTimeBinNS << " ns";
197  }

199  void accumulate()
200  {
201  // accumulate result of single event processing, called after processing every event supplied
202  // AND after the final flushing via digitizer::fillOutputContainer
203  if (!mDigits.size()) {
204  return; // no digits were flushed, nothing to accumulate
205  }
206  static int fixMC2ROF = 0; // 1st entry in mc2rofRecordsAccum to be fixed for ROFRecordID
207  auto ndigAcc = mDigitsAccum.size();
208  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(mDigitsAccum));
209 
210  // fix ROFrecords references on ROF entries
211  auto nROFRecsOld = mROFRecordsAccum.size();
212 
213  for (int i = 0; i < mROFRecords.size(); i++) {
214  auto& rof = mROFRecords[i];
215  rof.setFirstEntry(ndigAcc + rof.getFirstEntry());
216  rof.print();
217 
218  if (mFixMC2ROF < mMC2ROFRecordsAccum.size()) { // fix ROFRecord entry in MC2ROF records
219  for (int m2rid = mFixMC2ROF; m2rid < mMC2ROFRecordsAccum.size(); m2rid++) {
220  // need to register the ROFRecors entry for MC event starting from this entry
221  auto& mc2rof = mMC2ROFRecordsAccum[m2rid];
222  if (rof.getROFrame() == mc2rof.minROF) {
223  mFixMC2ROF++;
224  mc2rof.rofRecordID = nROFRecsOld + i;
225  mc2rof.print();
226  }
227  }
228  }
229  }
230  std::copy(mROFRecords.begin(), mROFRecords.end(), std::back_inserter(mROFRecordsAccum));
231  if (mWithMCTruth) {
232  mLabelsAccum.mergeAtBack(mLabels);
233  }
234  LOG(INFO) << "Added " << mDigits.size() << " digits ";
235  // clean containers from already accumulated stuff
236  mLabels.clear();
237  mDigits.clear();
238  mROFRecords.clear();
239  }
240 
241  bool mWithMCTruth = true;
242  bool mFinished = false;
243  o2::detectors::DetID mID;
244  o2::header::DataOrigin mOrigin = o2::header::gDataOriginInvalid;
245  o2::itsmft::Digitizer mDigitizer;
246  std::vector<o2::itsmft::Digit> mDigits;
247  std::vector<o2::itsmft::Digit> mDigitsAccum;
248  std::vector<o2::itsmft::ROFRecord> mROFRecords;
249  std::vector<o2::itsmft::ROFRecord> mROFRecordsAccum;
250  std::vector<o2::itsmft::Hit> mHits;
251  std::vector<o2::itsmft::Hit>* mHitsP = &mHits;
252  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabels;
253  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabelsAccum;
254  std::vector<o2::itsmft::MC2ROFRecord> mMC2ROFRecordsAccum;
255  std::vector<TChain*> mSimChains;
256  TChain mQEDChain = {"o2sim"};
257 
258  double mQEDEntryTimeBinNS = 1000; // time-coverage of single QED tree entry in ns (TODO: make it settable)
259  double mLastQEDTimeNS = 0; // time assingned to last QED entry
260  int mLastQEDEntry = -1; // last used QED entry
261  int mFixMC2ROF = 0; // 1st entry in mc2rofRecordsAccum to be fixed for ROFRecordID
262  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::PRESENT; // readout mode
263 
264  const int QEDSourceID = 99; // unique source ID for the QED (TODO: move it as a const to general class?)
