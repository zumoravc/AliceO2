143  uint8_t nDaughters = 0;
144  int8_t layerID = -1;
145  uint8_t ladderID = 0xff;
146  float chi2 = 0.f;
147 #ifdef _ALLOW_DEBUG_AB_
148  o2::track::TrackParCov seed; // seed before update
149 #endif
150  ABTrackLink() = default;
151  ABTrackLink(const o2::track::TrackParCov& src, int ic, int lr, int parid = MinusOne, int clid = MinusOne, float ch2 = 0.f)
152  : o2::track::TrackParCov(src), clID(clid), parentID(parid), icCandID(ic), layerID(lr), chi2(ch2) {}
153  bool isDisabled() const { return clID == Disabled; }
154  void disable() { clID = Disabled; }
155  bool isDummyTop() const { return clID == MinusTen; }
156  float chi2Norm() const { return layerID < o2::its::RecoGeomHelper::getNLayers() ? chi2 / (o2::its::RecoGeomHelper::getNLayers() - layerID) : 999.; }
157  float chi2NormPredict(float chi2cl) const { return (chi2 + chi2cl) / (1 + o2::its::RecoGeomHelper::getNLayers() - layerID); }
158 };

741  float trackY, float trackZ, float tolerY, float tolerZ,
742  const o2::MCCompLabel& lblTrc) const // TODO lbl is not needed

746  for (int icl = clRange.getEntries(); icl--;) { // note: clusters within a chip are sorted in Z
747  int clID = clRefs.clusterID[icID++]; // so, we go in clusterID increasing direction
