56 void TextControlService::notifyStreamingState(StreamingState state)
57 {
58  switch (state) {
59  case StreamingState::Idle:
60  LOG(INFO) << "CONTROL_ACTION: NOTIFY_STREAMING_STATE IDLE";
61  break;
62  case StreamingState::Streaming:
63  LOG(INFO) << "CONTROL_ACTION: NOTIFY_STREAMING_STATE STREAMING";
64  break;
65  case StreamingState::EndOfStreaming:
66  LOG(INFO) << "CONTROL_ACTION: NOTIFY_STREAMING_STATE EOS";
67  break;
68  default:
69  throw std::runtime_error("Unknown streaming state");
70  }
71 }

73 bool parseControl(std::string const& s, std::smatch& match)
74 {
75  const static std::regex controlRE1(".*CONTROL_ACTION: READY_TO_(QUIT)_(ME|ALL)", std::regex::optimize);
76  const static std::regex controlRE2(".*CONTROL_ACTION: (NOTIFY_STREAMING_STATE) (IDLE|STREAMING|EOS)", std::regex::optimize);
77  return std::regex_search(s, match, controlRE1) || std::regex_search(s, match, controlRE2);
