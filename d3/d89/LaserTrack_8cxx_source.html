29 void LaserTrackContainer::loadTracksFromFile()
30 {

33 
34  const std::string o2Root(std::getenv("O2_ROOT"));
35  const std::string file = o2Root + "/share/Detectors/TPC/files/laserTrackData.txt";
36 
37  std::string line;
38  std::ifstream infile(file, std::ifstream::in);
39 
40  if (!infile.is_open()) {
41  LOG(ERROR) << "Could not open laser track file " << file;
42  return;
43  }
44 
45  while (std::getline(infile, line)) {

48 
49  //printf("%3d: %f %f %f %f %f %f %f \n", id, x, alpha, p0, p1, p2, p3, p4);
50  mLaserTracks[id] = LaserTrack(id, x, alpha, {p0, p1, p2, p3, p4});
51  }
52 }
