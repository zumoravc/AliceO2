174 constexpr const char* channelName(int channel)
175 {
176 
177  // sanity check
178  static_assert(NChannels == sizeof(ChannelNames) / sizeof(std::string_view), "Channels definition is not complete");
179 
180  if (channel >= 0 && channel < NChannels) {
181  return ChannelNames[channel].data();
182  } else if (channel == IdDummy) {
183  return DummyName.data();
184  }
185  return VoidName.data();
186 }

188 //< map channelID to detector/tower
189 constexpr int toDet(int channel, int& tower)
190 {
191  if (channel < toChannel(ZNC, 0)) {
192  tower = channel % NChannelsZN;
193  return DetIDOffs + channel / NChannelsZP;
194  } else {
195  channel -= toChannel(ZNC, 0);
196  tower = channel % NChannelsZN;
197  return ZNC + channel / NChannelsZP;
198  }
199 }
