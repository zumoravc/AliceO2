26 TPCFastTransformGeo::TPCFastTransformGeo()
27 {
28  // Default Constructor: creates an empty uninitialized object
29  double dAlpha = 2. * M_PI / (NumberOfSlicesA);
30  for (int i = 0; i < NumberOfSlices; i++) {
31  SliceInfo& s = mSliceInfos[i];
32  double alpha = dAlpha * (i + 0.5);
33  s.sinAlpha = sin(alpha);
34  s.cosAlpha = cos(alpha);
35  }
36  mSliceInfos[NumberOfSlices] = SliceInfo{0.f, 0.f};
37 
38  for (int i = 0; i < MaxNumberOfRows + 1; i++) {
39  mRowInfos[i] = RowInfo{0.f, -1, 0.f, 0.f, 0.f, 0.f};
40  }
41 }

91 void TPCFastTransformGeo::setTPCrow(int iRow, float x, int nPads, float padWidth)
92 {
94  assert(mConstructionMask & ConstructionState::InProgress);
95  assert(iRow >= 0 && iRow < mNumberOfRows);
96  assert(nPads > 1);
97  assert(padWidth > 0.);
98 
99  // Make scaled U = area between centers of the first and the last pad
100 
101  // double uWidth = (nPads - 1) * padWidth;
102 
103  // Make scaled U = area between the geometrical sector borders
104 
105  const double sectorAngle = 2. * M_PI / NumberOfSlicesA;
106  const double scaleXtoRowWidth = 2. * tan(0.5 * sectorAngle);
107  double uWidth = x * scaleXtoRowWidth; // distance to the sector border
108 
109  RowInfo& row = mRowInfos[iRow];
110  row.x = x;
111  row.maxPad = nPads - 1;
112  row.padWidth = padWidth;
113  row.u0 = -uWidth / 2.;
114  row.scaleUtoSU = 1. / uWidth;
115  row.scaleSUtoU = uWidth;
116 }

149 int TPCFastTransformGeo::test(int slice, int row, float ly, float lz) const
150 {
152 
153  int error = 0;
154 
155  if (!isConstructed())
156  error = -1;
157 
158  if (mNumberOfRows <= 0 || mNumberOfRows >= MaxNumberOfRows)
159  error = -2;
160 
161  float lx = getRowInfo(row).x;
162  float lx1 = 0.f, ly1 = 0.f, lz1 = 0.f;
163  float gx = 0.f, gy = 0.f, gz = 0.f;
164 
165  convLocalToGlobal(slice, lx, ly, lz, gx, gy, gz);
166  convGlobalToLocal(slice, gx, gy, gz, lx1, ly1, lz1);
167 
168  if (fabs(lx1 - lx) > 1.e-4 || fabs(ly1 - ly) > 1.e-4 || fabs(lz1 - lz) > 1.e-7) {
169  std::cout << "Error local <-> global: x " << lx << " dx " << lx1 - lx << " y " << ly << " dy " << ly1 - ly << " z " << lz << " dz " << lz1 - lz << std::endl;
170  error = -3;
171  }
172  float u = 0.f, v = 0.f;
173  convLocalToUV(slice, ly, lz, u, v);
174  convUVtoLocal(slice, u, v, ly1, lz1);
175 
176  if (fabs(ly1 - ly) + fabs(lz1 - lz) > 1.e-6) {
177  std::cout << "Error local <-> UV: y " << ly << " dy " << ly1 - ly << " z " << lz << " dz " << lz1 - lz << std::endl;
178  error = -4;
179  }
180 
181  float su = 0.f, sv = 0.f;
182 
183  convUVtoScaledUV(slice, row, u, v, su, sv);
184 
185  if (su < 0.f || su > 1.f) {
186  std::cout << "Error scaled U range: u " << u << " su " << su << std::endl;
187  error = -5;
188  }
189 
190  float u1 = 0.f, v1 = 0.f;
191  convScaledUVtoUV(slice, row, su, sv, u1, v1);
192 
193  if (fabs(u1 - u) > 1.e-4 || fabs(v1 - v) > 1.e-4) {
194  std::cout << "Error UV<->scaled UV: u " << u << " du " << u1 - u << " v " << v << " dv " << v1 - v << std::endl;
195  error = -6;
196  }
197 
198  float pad = 0.f;
199  convUtoPad(row, u, pad);
200  convPadToU(row, pad, u1);
201 
202  if (fabs(u1 - u) > 1.e-5) {
203  std::cout << "Error U<->Pad: u " << u << " pad " << pad << " du " << u1 - u << std::endl;
204  error = -7;
205  }
206 
207 #if !defined(GPUCA_GPUCODE)
208  if (error != 0) {
209  std::cout << "TPC Fast Transformation Geometry: Internal ERROR " << error << std::endl;
210  }
211 #endif
212  return error;
213 }
