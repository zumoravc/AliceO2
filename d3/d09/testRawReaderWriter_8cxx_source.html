53 // ========================= simple detector data writer ================================
54 //
55 struct TestRawWriter { // simple class to create detector payload for multiple links
56 
57  // suppose detector puts in front and end of every trigger payload some header and trailer
58 
59  RawFileWriter writer{"TST"};
60 
61  //_________________________________________________________________
62  void init()
63  {
64  // init writer
65  writer.useRDHVersion(6);
66 
67  // register links
68  for (int icru = 0; icru < NCRU; icru++) {
69  std::string outFileName = "testdata_cru" + std::to_string(icru) + ".raw";
70  for (int il = 0; il < NLinkPerCRU; il++) {
71  auto& link = writer.registerLink((icru << 8) + il, icru, il, 0, outFileName);
72  RDHUtils::setDetectorField(link.rdhCopy, 0xff << icru); // if needed, set extra link info, will be copied to all RDHs
73  }
74  }
75 
76  writer.setContinuousReadout(); // in case we want to issue StartOfContinuous trigger in the beginning
77  writer.setCarryOverCallBack(this); // we want that writer to ask the detector code how to split large payloads
78  writer.setEmptyPageCallBack(this); // we want the writer to ask the detector code what to put in empty HBFs
79  }
80 
81  //_________________________________________________________________
82  void run()
83  {
84  // write payload and close outputs
85 
86  // generate interaction records for triggers to write
87  std::vector<o2::InteractionTimeRecord> irs(1000);
88  o2::steer::InteractionSampler irSampler;
89  irSampler.setInteractionRate(12000); // ~1.5 interactions per orbit
90  irSampler.init();
91  irSampler.generateCollisionTimes(irs);
92 
93  std::vector<char> buffer;
94 
95  // create payload for every interaction and push it to writer
96  for (const auto& ir : irs) {
97  for (int icru = 0; icru < NCRU - 1; icru++) {
98  // we will create non-0 payload for all but 1st link of every CRU, the writer should take care
99  // of creating empty HBFs for the links w/o data
100  for (int il = 0; il < NLinkPerCRU; il++) {
101  buffer.clear();
102  int nGBT = gRandom->Poisson(RDHUtils::MAXCRUPage / RDHUtils::GBTWord * (il));
103  if (nGBT) {
104  buffer.resize((nGBT + 2) * RDHUtils::GBTWord, icru * NLinkPerCRU + il); // reserve 16B words accounting for the Header and Trailer
105  std::memcpy(buffer.data(), PLHeader.c_str(), RDHUtils::GBTWord);
106  std::memcpy(buffer.data() + buffer.size() - RDHUtils::GBTWord, PLTrailer.c_str(), RDHUtils::GBTWord);
107  // we don't care here about the content of the payload, except the presence of header and trailer
108  }
109  writer.addData((icru << 8) + il, icru, il, 0, ir, buffer);
110  }
111  }
112  }
113  // fill special CRU with preformatted pages
114  auto irHB = HBFUtils::Instance().getFirstIR(); // IR of the TF0/HBF0
115  int cruID = NCRU - 1;
116  while (irHB < irs.back()) {
117  for (int il = 0; il < NLinkPerCRU; il++) {
118  buffer.clear();
119  int pgSize = SpecSize[il] - sizeof(RDHAny);
120  buffer.resize(pgSize);
121  for (int ipg = 2 * (NLinkPerCRU - il); ipg--;) { // just to enforce writing multiple pages per selected HBFs
122  writer.addData((cruID << 8) + il, cruID, il, 0, irHB, buffer, true); // last argument is there to enforce a special "preformatted" mode
123  nPreformatPages++;
124  }
125  }
126  irHB.orbit += HBFUtils::Instance().getNOrbitsPerTF() / NPreformHBFPerTF; // we will write 32 such HBFs per TF
127  }
128 
129  writer.writeConfFile("FLP", "RAWDATA", CFGName); // for further use we write the configuration file
130  writer.close(); // flush buffers and close outputs
131  }
132 
133  // optional callback functions to register in the RawFileWriter
134  //_________________________________________________________________
135  void emptyHBFMethod(const RDHAny* rdh, std::vector<char>& toAdd) const
136  {
137  // what we want to add for every empty page
138  toAdd.resize(RDHUtils::GBTWord);
139  std::memcpy(toAdd.data(), HBFEmpty.c_str(), RDHUtils::GBTWord);
140  }
141 
142  //_________________________________________________________________
143  int carryOverMethod(const RDHAny* rdh, const gsl::span<char> data, const char* ptr, int maxSize, int splitID,
144  std::vector<char>& trailer, std::vector<char>& header) const
145  {
146  // how we want to split the large payloads. The data is the full payload which was sent for writing and
147  // it is already equiped with header and trailer
148  if (maxSize <= RDHUtils::GBTWord) { // do not carry over trailer or header only
149  return 0;
150  }
151 
152  // here we simply copy the header/trailer of the payload to every CRU page of this payload
153  header.resize(RDHUtils::GBTWord);
154  std::memcpy(header.data(), &data[0], RDHUtils::GBTWord);
155  trailer.resize(RDHUtils::GBTWord);
156  std::memcpy(trailer.data(), &data[data.size() - RDHUtils::GBTWord], RDHUtils::GBTWord);
157  // since we write an extra GBT word (trailer) in the end of the CRU page, we ask to write
158  // not the block ptr : ptr+maxSize, but ptr : ptr+maxSize - GBTWord;
159  int sz = maxSize - RDHUtils::GBTWord;
160  return sz;
161  }
162 };

164 struct TestRawReader { // simple class to read detector raw data for multiple links
165 
166  std::unique_ptr<RawFileReader> reader;
167 
168  //_________________________________________________________________
169  void init()
170  {
171  reader = std::make_unique<RawFileReader>(CFGName); // init from configuration file
172  uint32_t errCheck = 0xffffffff;
173  errCheck ^= 0x1 << RawFileReader::ErrNoSuperPageForTF; // makes no sense for superpages not interleaved by others
174  reader->setCheckErrors(errCheck);
175  reader->init();
176  }
177 
178  //_________________________________________________________________
179  void run()
180  {
181  // read data of all links
182  if (!reader) {
183  init();
184  }
185  int nLinks = reader->getNLinks();
186  BOOST_CHECK(nLinks == NCRU * NLinkPerCRU);
187 
188  // make sure no errors detected after initialization
189  for (int il = 0; il < nLinks; il++) {
190  auto& lnk = reader->getLink(il);
191  BOOST_CHECK(lnk.nErrors == 0);
192  }
193 
194  std::vector<std::vector<char>> buffers;
195  std::vector<bool> firstHBF;
196  std::string testStr;
197  testStr.resize(RDHUtils::GBTWord);
198  buffers.resize(nLinks); // 1 buffer per link
199  firstHBF.resize(nLinks, true);
200 
201  int nLinksRead = 0, nPreformatRead = 0;
202  do {
203  nLinksRead = 0;
204  for (int il = 0; il < nLinks; il++) {
205  auto& buff = buffers[il];
206  buff.clear();
207  auto& lnk = reader->getLink(il);
208  auto sz = lnk.getNextHBFSize();
209  if (!sz) {
210  continue;
211  }
212  buff.resize(sz);
213  BOOST_CHECK(lnk.readNextHBF(buff.data()) == sz);
214  nLinksRead++;
215  }
216  if (nLinksRead) {
217  BOOST_CHECK(nLinksRead == nLinks); // all links should have the same number of HBFs
218 
219  const auto rdhRef = *reinterpret_cast<RDHAny*>(buffers[0].data());
220 
221  for (int il = 0; il < nLinks; il++) {
222  auto& buff = buffers[il];
223  int hbsize = buff.size();
224  char* ptr = buff.data();
225  while (ptr < &buff.back()) { // check all RDH open/close and optional headers and trailers
226  const auto rdhi = *reinterpret_cast<RDHAny*>(ptr);
227  if (firstHBF[il]) { // make sure SOT or SOC is there
228  BOOST_CHECK(RDHUtils::getTriggerType(rdhi) & (o2::trigger::SOC | o2::trigger::SOT));
229  }
230  auto memSize = RDHUtils::getMemorySize(rdhi);
231  auto rdhSize = RDHUtils::getHeaderSize(rdhi);
232  BOOST_CHECK(RDHUtils::checkRDH(rdhi)); // check RDH validity
233  BOOST_CHECK(RDHUtils::getHeartBeatIR(rdhRef) == RDHUtils::getHeartBeatIR(rdhi)); // make sure the RDH of each link corresponds to the same BC
234  if (RDHUtils::getStop(rdhi)) { // closing page must be empty
235  BOOST_CHECK(memSize == rdhSize);
236  } else {
237  if (RDHUtils::getCRUID(rdhi) < NCRU - 1) { // these are not special CRUs
238  BOOST_CHECK(memSize > rdhSize); // in this model all non-closing pages must contain something
239  if (memSize - rdhSize == RDHUtils::GBTWord) { // empty HBF will contain just a status word
240  testStr.assign(ptr + rdhSize, RDHUtils::GBTWord);
241  BOOST_CHECK(testStr == HBFEmpty);
242  } else {
243  // pages with real payload should have at least header + trailer + some payload
244  BOOST_CHECK(memSize - rdhSize > 2 * RDHUtils::GBTWord);
245  testStr.assign(ptr + rdhSize, RDHUtils::GBTWord);
246  BOOST_CHECK(testStr == PLHeader);
247  testStr.assign(ptr + memSize - RDHUtils::GBTWord, RDHUtils::GBTWord);
248  BOOST_CHECK(testStr == PLTrailer);
249  }
250  } else { // for the special CRU with preformatted data make sure the page sizes were not modified
251  if (memSize > rdhSize + RDHUtils::GBTWord) {
252  auto tfhb = HBFUtils::Instance().getTFandHBinTF({RDHUtils::getHeartBeatBC(rdhi), RDHUtils::getHeartBeatOrbit(rdhi)}); // TF and HBF relative to TF
253  BOOST_CHECK(tfhb.second % (HBFUtils::Instance().getNOrbitsPerTF() / NPreformHBFPerTF) == 0); // we were filling only every NPreformHBFPerTF-th HBF
254  BOOST_CHECK(memSize == SpecSize[RDHUtils::getLinkID(rdhi)]); // check if the size is correct
255  nPreformatRead++;
256  }
257  }
258  }
259  ptr += RDHUtils::getOffsetToNext(rdhi);
260  }
261  firstHBF[il] = false;
262  }
263  }
264  } while (nLinksRead); // read until there is something to read
265 
266  BOOST_CHECK(nPreformatRead == nPreformatPages); // make sure no preformatted page was lost
267 
268  } // run
269 };
