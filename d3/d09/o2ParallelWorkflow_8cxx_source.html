66 // This is a simple consumer / producer workflow where both are
67 // stateful, i.e. they have context which comes from their initialization.
68 WorkflowSpec defineDataProcessing(ConfigContext const& config)
69 {
70  size_t jobs = config.options().get<int>("2-layer-jobs");
71  size_t stages = config.options().get<int>("3-layer-pipelining");
72 
73  // This is an example of how we can parallelize by subSpec.
74  // templatedProducer will be instanciated 32 times and the lambda function
75  // passed to the parallel statement will be applied to each one of the
76  // instances in order to modify it. Parallel will also make sure the name of
77  // the instance is amended from "some-producer" to "some-producer-<index>".
78  WorkflowSpec workflow = parallel(templateProcessor(), jobs, [](DataProcessorSpec& spec, size_t index) {
79  DataSpecUtils::updateMatchingSubspec(spec.inputs[0], index);
80  DataSpecUtils::updateMatchingSubspec(spec.outputs[0], index);
81  });
82 
83  std::vector<OutputSpec> outputSpecs;
84  for (size_t ssi = 0; ssi < jobs; ++ssi) {
85  outputSpecs.emplace_back("TST", "A", ssi);
86  }
87 
88  workflow.push_back(DataProcessorSpec{"reader", {}, outputSpecs, AlgorithmSpec{[jobs](InitContext& initCtx) {
89  return [jobs](ProcessingContext& ctx) {
90  for (size_t ji = 0; ji < jobs; ++ji) {
91  ctx.outputs().make<int>(Output{"TST", "A", static_cast<o2::header::DataHeader::SubSpecificationType>(ji)},
92  1);
93  }
94  };
95  }}});
96  workflow.push_back(timePipeline(DataProcessorSpec{
97  "merger",
98  mergeInputs(InputSpec{"x", "TST", "P"},
99  jobs,
100  [](InputSpec& input, size_t index) {
101  DataSpecUtils::updateMatchingSubspec(input, index);
102  }),
103  {OutputSpec{{"out"}, "TST", "M"}},
104  AlgorithmSpec{[](InitContext& setup) {
105  return [](ProcessingContext& ctx) {
106  ctx.outputs().make<int>(OutputRef("out", 0), 1);
107  };
108  }}},
109  stages));
110 
111  workflow.push_back(DataProcessorSpec{
112  "writer",
113  {InputSpec{"x", "TST", "M"}},
114  {},
115  AlgorithmSpec{[](InitContext& setup) {
116  return [](ProcessingContext& ctx) {
117  };
118  }}});
119  return workflow;
120 }
