25 std::unique_ptr<EncodedClusters> TPCEntropyEncoder::encode(const CompressedClusters& clusters)
26 {

28

31 
32  //counters
33  encodedClusters->counters = new o2::tpc::EncodedClusters::Counters{clusters.nTracks,
34  clusters.nAttachedClusters,

49  compress("timeResA", clusters.timeResA, clusters.timeResA + clusters.nAttachedClustersReduced, sProbabilityBits25Bit, *encodedClusters);
50  compress("sigmaPadA", clusters.sigmaPadA, clusters.sigmaPadA + clusters.nAttachedClusters, sProbabilityBits8Bit, *encodedClusters);

63  compress("sigmaTimeU", clusters.sigmaTimeU, clusters.sigmaTimeU + clusters.nUnattachedClusters, sProbabilityBits8Bit, *encodedClusters);
64  compress("nTrackClusters", clusters.nTrackClusters, clusters.nTrackClusters + clusters.nTracks, sProbabilityBits16Bit, *encodedClusters);
65  compress("nSliceRowClusters", clusters.nSliceRowClusters, clusters.nSliceRowClusters + clusters.nSliceRows, sProbabilityBits25Bit, *encodedClusters);

70 void TPCEntropyEncoder::appendToTTree(TTree& tree, EncodedClusters& encodedClusters)
71 {
72  tree.Branch("Header", &encodedClusters.header);
73  tree.Branch("Counters", &encodedClusters.counters);
74  tree.Branch("Metadata", &encodedClusters.metadata);
75 
76  for (size_t i = 0; i < encodedClusters.NUM_ARRAYS; i++) {
77  const auto name = encodedClusters.NAMES[i];
78  const auto dictName = std::string(name) + "Dict";
79 
80  auto& dicts = encodedClusters.dicts;
81  auto& buffers = encodedClusters.buffers;
82 
83  assert(dicts[i]);
84  assert(buffers[i]);
85 
86  tree.Branch(dictName.c_str(), &dicts[i]);
87  auto branch = tree.Branch(name, &buffers[i]);
88  branch->SetCompressionLevel(ROOT::RCompressionSetting::ELevel::kUncompressed);
89  }
90 
91  // fill and write
92  tree.Fill();
93 }
