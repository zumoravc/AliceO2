41 class ClusterizerMCDeviceDPL
42 {
43  public:
44  void init(o2::framework::InitContext& ic)
45  {
46  if (!mPreClusterizer.init()) {
47  LOG(ERROR) << "Initialization of MID pre-clusterizer device failed";
48  }
49 
50  mCorrelation.clear();
51 
52  bool isClusterizerInit = mClusterizer.init([&](size_t baseIndex, size_t relatedIndex) { mCorrelation.push_back({baseIndex, relatedIndex}); });
53 
54  if (!isClusterizerInit) {
55  LOG(ERROR) << "Initialization of MID clusterizer device failed";
56  }
57  }
58  void run(o2::framework::ProcessingContext& pc)
59  {
60  auto msg = pc.inputs().get("mid_data");
61  gsl::span<const ColumnData> patterns = of::DataRefUtils::as<const ColumnData>(msg);
62 
63  auto msgROF = pc.inputs().get("mid_data_rof");
64  gsl::span<const ROFRecord> inROFRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
65 
66  std::unique_ptr<const o2::dataformats::MCTruthContainer<MCLabel>> labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<MCLabel>*>("mid_data_labels");
67 
68  // Pre-clustering
69  mPreClusterizer.process(patterns, inROFRecords);
70  LOG(DEBUG) << "Generated " << mPreClusterizer.getPreClusters().size() << " PreClusters";
71 
72  // Clustering
73  mClusterizer.process(mPreClusterizer.getPreClusters(), mPreClusterizer.getROFRecords());
74 
75  // Labelling
76  mPreClusterLabeler.process(mPreClusterizer.getPreClusters(), *labels, mPreClusterizer.getROFRecords(), inROFRecords);
77  mClusterLabeler.process(mPreClusterizer.getPreClusters(), mPreClusterLabeler.getContainer(), mClusterizer.getClusters(), mCorrelation);
78  // Clear the index correlations that will be used in the next cluster processing
79  mCorrelation.clear();
80 
81  pc.outputs().snapshot(of::Output{"MID", "CLUSTERS", 0, of::Lifetime::Timeframe}, mClusterizer.getClusters());
82  LOG(DEBUG) << "Sent " << mClusterizer.getClusters().size() << " clusters";
83  pc.outputs().snapshot(of::Output{"MID", "CLUSTERSROF", 0, of::Lifetime::Timeframe}, mClusterizer.getROFRecords());
84  LOG(DEBUG) << "Sent " << mClusterizer.getROFRecords().size() << " ROF";
85 
86  pc.outputs().snapshot(of::Output{"MID", "CLUSTERSLABELS", 0, of::Lifetime::Timeframe}, mClusterLabeler.getContainer());
87  LOG(DEBUG) << "Sent " << mClusterLabeler.getContainer().getIndexedSize() << " indexed clusters";
88  }
89 
90  private:
91  PreClusterizer mPreClusterizer{};
92  Clusterizer mClusterizer{};
93  PreClusterLabeler mPreClusterLabeler{};
94  ClusterLabeler mClusterLabeler{};
95  std::vector<std::array<size_t, 2>> mCorrelation{};
96 };
