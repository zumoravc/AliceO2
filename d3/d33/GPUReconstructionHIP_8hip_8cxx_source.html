117 template <class T, int I, typename... Args>
118 int GPUReconstructionHIPBackend::runKernelBackend(krnlSetup& _xyz, Args... args)
119 {
120  auto& x = _xyz.x;
121  auto& z = _xyz.z;
122  if (z.evList) {
123  for (int k = 0; k < z.nEvents; k++) {
124  GPUFailedMsg(hipStreamWaitEvent(mInternals->Streams[x.stream], ((hipEvent_t*)z.evList)[k], 0));
125  }
126  }
127  if (mDeviceProcessingSettings.deviceTimers) {
128 #ifdef __CUDACC__
129  GPUFailedMsg(hipEventRecord((hipEvent_t)mDebugEvents->DebugStart, mInternals->Streams[x.stream]));
130 #endif
131  backendInternal<T, I>::runKernelBackendMacro(_xyz, this, (hipEvent_t*)&mDebugEvents->DebugStart, (hipEvent_t*)&mDebugEvents->DebugStop, args...);
132 #ifdef __CUDACC__
133  GPUFailedMsg(hipEventRecord((hipEvent_t)mDebugEvents->DebugStop, mInternals->Streams[x.stream]));
134 #endif
135  GPUFailedMsg(hipEventSynchronize((hipEvent_t)mDebugEvents->DebugStop));
136  float v;
137  GPUFailedMsg(hipEventElapsedTime(&v, (hipEvent_t)mDebugEvents->DebugStart, (hipEvent_t)mDebugEvents->DebugStop));
138  _xyz.t = v * 1.e-3;
139  } else {
140  backendInternal<T, I>::runKernelBackendMacro(_xyz, this, nullptr, nullptr, args...);
141  }
142  GPUFailedMsg(hipGetLastError());
143  if (z.ev) {
144  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)z.ev, mInternals->Streams[x.stream]));
145  }
146  return 0;
147 }

182 int GPUReconstructionHIPBackend::InitDevice_Runtime()
183 {
184  if (mMaster == nullptr) {
185  hipDeviceProp_t hipDeviceProp;
186  int count, bestDevice = -1;
187  double bestDeviceSpeed = -1, deviceSpeed;
188  if (GPUFailedMsgI(hipGetDeviceCount(&count))) {
189  GPUError("Error getting HIP Device Count");
190  return (1);
191  }
192  if (mDeviceProcessingSettings.debugLevel >= 2) {
193  GPUInfo("Available HIP devices:");
194  }
195  const int reqVerMaj = 2;
196  const int reqVerMin = 0;
197  std::vector<bool> devicesOK(count, false);
198  for (int i = 0; i < count; i++) {
199  if (mDeviceProcessingSettings.debugLevel >= 4) {
200  GPUInfo("Examining device %d", i);
201  }
202  if (mDeviceProcessingSettings.debugLevel >= 4) {
203  GPUInfo("Obtained current memory usage for device %d", i);
204  }
205  if (GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp, i))) {
206  continue;
207  }
208  if (mDeviceProcessingSettings.debugLevel >= 4) {
209  GPUInfo("Obtained device properties for device %d", i);
210  }
211  int deviceOK = true;
212  const char* deviceFailure = "";
213  if (hipDeviceProp.major >= 9) {
214  deviceOK = false;
215  deviceFailure = "Invalid Revision";
216  } else if (hipDeviceProp.major < reqVerMaj || (hipDeviceProp.major == reqVerMaj && hipDeviceProp.minor < reqVerMin)) {
217  deviceOK = false;
218  deviceFailure = "Too low device revision";
219  }
220 
221  deviceSpeed = (double)hipDeviceProp.multiProcessorCount * (double)hipDeviceProp.clockRate * (double)hipDeviceProp.warpSize * (double)hipDeviceProp.major * (double)hipDeviceProp.major;
222  if (mDeviceProcessingSettings.debugLevel >= 2) {
223  GPUImportant("Device %s%2d: %s (Rev: %d.%d - Mem %lld)%s %s", deviceOK ? " " : "[", i, hipDeviceProp.name, hipDeviceProp.major, hipDeviceProp.minor, (long long int)hipDeviceProp.totalGlobalMem, deviceOK ? " " : " ]", deviceOK ? "" : deviceFailure);
224  }
225  if (!deviceOK) {
226  continue;
227  }
228  devicesOK[i] = true;
229  if (deviceSpeed > bestDeviceSpeed) {
230  bestDevice = i;
231  bestDeviceSpeed = deviceSpeed;
232  } else {
233  if (mDeviceProcessingSettings.debugLevel >= 2 && mDeviceProcessingSettings.deviceNum < 0) {
234  GPUInfo("Skipping: Speed %f < %f\n", deviceSpeed, bestDeviceSpeed);
235  }
236  }
237  }
238  if (bestDevice == -1) {
239  GPUWarning("No %sHIP Device available, aborting HIP Initialisation", count ? "appropriate " : "");
240  GPUImportant("Requiring Revision %d.%d, Mem: %lld", reqVerMaj, reqVerMin, (long long int)mDeviceMemorySize);
241  return (1);
242  }
243 
244  if (mDeviceProcessingSettings.deviceNum > -1) {
245  if (mDeviceProcessingSettings.deviceNum >= (signed)count) {
246  GPUWarning("Requested device ID %d does not exist", mDeviceProcessingSettings.deviceNum);
247  return (1);
248  } else if (!devicesOK[mDeviceProcessingSettings.deviceNum]) {
249  GPUWarning("Unsupported device requested (%d)", mDeviceProcessingSettings.deviceNum);
250  return (1);
251  } else {
252  bestDevice = mDeviceProcessingSettings.deviceNum;
253  }
254  }
255  mDeviceId = bestDevice;
256 
257  GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp, mDeviceId));
258  hipDeviceProp.totalConstMem = 65536; // TODO: Remove workaround, fixes incorrectly reported HIP constant memory
259 
260  if (mDeviceProcessingSettings.debugLevel >= 2) {
261  GPUInfo("Using HIP Device %s with Properties:", hipDeviceProp.name);
262  GPUInfo("\ttotalGlobalMem = %lld", (unsigned long long int)hipDeviceProp.totalGlobalMem);
263  GPUInfo("\tsharedMemPerBlock = %lld", (unsigned long long int)hipDeviceProp.sharedMemPerBlock);
264  GPUInfo("\tregsPerBlock = %d", hipDeviceProp.regsPerBlock);
265  GPUInfo("\twarpSize = %d", hipDeviceProp.warpSize);
266  GPUInfo("\tmaxThreadsPerBlock = %d", hipDeviceProp.maxThreadsPerBlock);
267  GPUInfo("\tmaxThreadsDim = %d %d %d", hipDeviceProp.maxThreadsDim[0], hipDeviceProp.maxThreadsDim[1], hipDeviceProp.maxThreadsDim[2]);
268  GPUInfo("\tmaxGridSize = %d %d %d", hipDeviceProp.maxGridSize[0], hipDeviceProp.maxGridSize[1], hipDeviceProp.maxGridSize[2]);
269  GPUInfo("\ttotalConstMem = %lld", (unsigned long long int)hipDeviceProp.totalConstMem);
270  GPUInfo("\tmajor = %d", hipDeviceProp.major);
271  GPUInfo("\tminor = %d", hipDeviceProp.minor);
272  GPUInfo("\tclockRate = %d", hipDeviceProp.clockRate);
273  GPUInfo("\tmemoryClockRate = %d", hipDeviceProp.memoryClockRate);
274  GPUInfo("\tmultiProcessorCount = %d", hipDeviceProp.multiProcessorCount);
275  GPUInfo(" ");
276  }
277  mBlockCount = hipDeviceProp.multiProcessorCount;
278  mMaxThreads = std::max<int>(mMaxThreads, hipDeviceProp.maxThreadsPerBlock * mBlockCount);
279  mWarpSize = 64;
280  mDeviceName = hipDeviceProp.name;
281  mDeviceName += " (HIP GPU)";
282 
283  if (hipDeviceProp.major < 3) {
284  GPUError("Unsupported HIP Device");
285  return (1);
286  }
287 #ifndef GPUCA_NO_CONSTANT_MEMORY
288  if (gGPUConstantMemBufferSize > hipDeviceProp.totalConstMem) {
289  GPUError("Insufficient constant memory available on GPU %d < %d!", (int)hipDeviceProp.totalConstMem, (int)gGPUConstantMemBufferSize);
290  return (1);
291  }
292 #endif
293 
294  if (GPUFailedMsgI(hipSetDevice(mDeviceId))) {
295  GPUError("Could not set HIP Device!");
296  return (1);
297  }
298 
299  /*if (GPUFailedMsgI(hipDeviceSetLimit(hipLimitStackSize, GPUCA_GPU_STACK_SIZE)))
300  {
301  GPUError("Error setting HIP stack size");
302  GPUFailedMsgI(hipDeviceReset());
303  return(1);
304  }*/
305 
306  if (mDeviceMemorySize > hipDeviceProp.totalGlobalMem || GPUFailedMsgI(hipMalloc(&mDeviceMemoryBase, mDeviceMemorySize))) {
307  GPUError("HIP Memory Allocation Error (trying %lld bytes, %lld available)", (long long int)mDeviceMemorySize, (long long int)hipDeviceProp.totalGlobalMem);
308  GPUFailedMsgI(hipDeviceReset());
309  return (1);
310  }
311  if (GPUFailedMsgI(hipHostMalloc(&mHostMemoryBase, mHostMemorySize))) {
312  GPUError("Error allocating Page Locked Host Memory (trying %lld bytes)", (long long int)mHostMemorySize);
313  GPUFailedMsgI(hipDeviceReset());
314  return (1);
315  }
316  if (mDeviceProcessingSettings.debugLevel >= 1) {
317  GPUInfo("Memory ptrs: GPU (%lld bytes): %p - Host (%lld bytes): %p", (long long int)mDeviceMemorySize, mDeviceMemoryBase, (long long int)mHostMemorySize, mHostMemoryBase);
318  memset(mHostMemoryBase, 0, mHostMemorySize);
319  if (GPUFailedMsgI(hipMemset(mDeviceMemoryBase, 0xDD, mDeviceMemorySize))) {
320  GPUError("Error during HIP memset, trying workaround with kernel");
321  hipLaunchKernelGGL(HIP_KERNEL_NAME(gHIPMemSetWorkaround), dim3(mBlockCount), dim3(256), 0, 0, (char*)mDeviceMemoryBase, 0xDD, mDeviceMemorySize);
322  if (GPUFailedMsgI(hipGetLastError()) || GPUFailedMsgI(hipDeviceSynchronize())) {
323  GPUError("Error during HIP memset");
324  GPUFailedMsgI(hipDeviceReset());
325  return (1);
326  }
327  }
328  }
329 
330  for (int i = 0; i < mNStreams; i++) {
331  if (GPUFailedMsgI(hipStreamCreate(&mInternals->Streams[i]))) {
332  GPUError("Error creating HIP Stream");
333  GPUFailedMsgI(hipDeviceReset());
334  return (1);
335  }
336  }
337 
338  void* devPtrConstantMem;
339 #ifndef GPUCA_NO_CONSTANT_MEMORY
340  if (GPUFailedMsgI(hipGetSymbolAddress(&devPtrConstantMem, HIP_SYMBOL(gGPUConstantMemBuffer)))) {
341  GPUError("Error getting ptr to constant memory");
342  GPUFailedMsgI(hipDeviceReset());
343  return 1;
344  }
345 #else
346  if (GPUFailedMsgI(hipMalloc(&devPtrConstantMem, gGPUConstantMemBufferSize))) {
347  GPUError("HIP Memory Allocation Error");
348  GPUFailedMsgI(hipDeviceReset());
349  return (1);
350  }
351 #endif
352  mDeviceConstantMem = (GPUConstantMem*)devPtrConstantMem;
353 
354  hipLaunchKernelGGL(HIP_KERNEL_NAME(dummyInitKernel), dim3(mBlockCount), dim3(256), 0, 0, mDeviceMemoryBase);
355  GPUInfo("HIP Initialisation successfull (Device %d: %s (Frequency %d, Cores %d), %lld / %lld bytes host / global memory, Stack frame %d, Constant memory %lld)", mDeviceId, hipDeviceProp.name, hipDeviceProp.clockRate, hipDeviceProp.multiProcessorCount, (long long int)mHostMemorySize,
356  (long long int)mDeviceMemorySize, (int)GPUCA_GPU_STACK_SIZE, (long long int)gGPUConstantMemBufferSize);
357  } else {
358  GPUReconstructionHIPBackend* master = dynamic_cast<GPUReconstructionHIPBackend*>(mMaster);
359  mDeviceId = master->mDeviceId;
360  mBlockCount = master->mBlockCount;
361  mWarpSize = master->mWarpSize;
362  mMaxThreads = master->mMaxThreads;
363  mDeviceName = master->mDeviceName;
364  mDeviceConstantMem = master->mDeviceConstantMem;
365  mInternals = master->mInternals;
366  }
367  for (unsigned int i = 0; i < mEvents.size(); i++) {
368  hipEvent_t* events = (hipEvent_t*)mEvents[i].data();
369  for (unsigned int j = 0; j < mEvents[i].size(); j++) {
370  if (GPUFailedMsgI(hipEventCreate(&events[j]))) {
371  GPUError("Error creating event");
372  GPUFailedMsgI(hipDeviceReset());
373  return 1;
374  }
375  }
376  }
377 
378  return (0);
379 }

418 size_t GPUReconstructionHIPBackend::GPUMemCpy(void* dst, const void* src, size_t size, int stream, int toGPU, deviceEvent* ev, deviceEvent* evList, int nEvents)
419 {
420  if (mDeviceProcessingSettings.debugLevel >= 3) {
421  stream = -1;
422  }
423  if (stream == -1) {
424  SynchronizeGPU();
425  GPUFailedMsg(hipMemcpy(dst, src, size, toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost));
426  } else {
427  if (evList == nullptr) {
428  nEvents = 0;
429  }
430  for (int k = 0; k < nEvents; k++) {
431  GPUFailedMsg(hipStreamWaitEvent(mInternals->Streams[stream], ((hipEvent_t*)evList)[k], 0));
432  }
433  GPUFailedMsg(hipMemcpyAsync(dst, src, size, toGPU == -2 ? hipMemcpyDeviceToDevice : toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost, mInternals->Streams[stream]));
434  }
435  if (ev) {
436  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)ev, mInternals->Streams[stream == -1 ? 0 : stream]));
437  }
438  return size;

513 int GPUReconstructionHIPBackend::GPUDebug(const char* state, int stream)
514 {
515  // Wait for HIP-Kernel to finish and check for HIP errors afterwards, in case of debugmode
516  if (mDeviceProcessingSettings.debugLevel == 0) {
517  return (0);
518  }
519  hipError_t cuErr;
520  cuErr = hipGetLastError();
