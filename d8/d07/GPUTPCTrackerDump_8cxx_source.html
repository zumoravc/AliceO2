101 int GPUTPCTracker::StarthitSortComparison(const void* a, const void* b)
102 {
103  // qsort helper function to sort start hits
104  const GPUTPCHitId* aa = reinterpret_cast<const GPUTPCHitId*>(a);
105  const GPUTPCHitId* bb = reinterpret_cast<const GPUTPCHitId*>(b);
106 
107  if (aa->RowIndex() != bb->RowIndex()) {
108  return (aa->RowIndex() - bb->RowIndex());
109  }
110  return (aa->HitIndex() - bb->HitIndex());
111 }

126 void GPUTPCTracker::DumpTrackHits(std::ostream& out)
127 {
128  // dump tracks to file
129  out << "Tracks: (Slice" << mISlice << ") (" << *NTracks() << ")" << std::endl;
130  for (int k = 0; k < GPUCA_ROW_COUNT; k++) {
131  for (int l = 0; l < Row(k).NHits(); l++) {
132  for (unsigned int j = 0; j < *NTracks(); j++) {
133  if (Tracks()[j].NHits() == 0) {
134  continue;
135  }
136  if (TrackHits()[Tracks()[j].FirstHitID()].RowIndex() == k && TrackHits()[Tracks()[j].FirstHitID()].HitIndex() == l) {
137  for (int i = 0; i < Tracks()[j].NHits(); i++) {
138  out << TrackHits()[Tracks()[j].FirstHitID() + i].RowIndex() << "-" << TrackHits()[Tracks()[j].FirstHitID() + i].HitIndex() << ", ";
139  }
140  if (!mRec->GetDeviceProcessingSettings().comparableDebutOutput) {
141  out << "(Track: " << j << ")";
142  }
143  out << std::endl;
144  }
145  }

150 void GPUTPCTracker::DumpTrackletHits(std::ostream& out)
151 {
152  // dump tracklets to file
153  int nTracklets = *NTracklets();
154  if (nTracklets < 0) {
155  nTracklets = 0;
156  }
157  out << "Tracklets: (Slice" << mISlice << ") (" << nTracklets << ")" << std::endl;
158  std::vector<int> Ids(nTracklets);
159  std::iota(Ids.begin(), Ids.end(), 0);
160  if (mRec->GetDeviceProcessingSettings().comparableDebutOutput) {
161  std::sort(Ids.begin(), Ids.end(), [this](const int& a, const int& b) {
162  if (this->Tracklets()[a].FirstRow() == this->Tracklets()[b].FirstRow()) {
163  return this->Tracklets()[a].Param().Y() > this->Tracklets()[b].Param().Y();
164  }
165  return this->Tracklets()[a].FirstRow() > this->Tracklets()[b].FirstRow();
166  });
167  }
168  for (int jj = 0; jj < nTracklets; jj++) {
169  const int j = Ids[jj];
170  const auto& tracklet = Tracklets()[j];
171  out << "Tracklet " << std::setw(4) << jj << " (Hits: " << std::setw(3) << Tracklets()[j].NHits() << ", Rows: " << (Tracklets()[j].NHits() ? Tracklets()[j].FirstRow() : -1)
172  << " - " << (tracklet.NHits() ? tracklet.LastRow() : -1) << ") ";
173  if (tracklet.NHits() == 0) {
174  ;
175  } else if (tracklet.LastRow() > tracklet.FirstRow() && (tracklet.FirstRow() >= GPUCA_ROW_COUNT || tracklet.LastRow() >= GPUCA_ROW_COUNT)) {
176  GPUError("Error: Tracklet %d First %d Last %d Hits %d", j, tracklet.FirstRow(), tracklet.LastRow(), tracklet.NHits());
177  out << " (Error: Tracklet " << j << " First " << tracklet.FirstRow() << " Last " << tracklet.LastRow() << " Hits " << tracklet.NHits() << ") ";
178  for (int i = 0; i < GPUCA_ROW_COUNT; i++) {
179  // if (tracklet.RowHit(i) != CALINK_INVAL)
180  out << i << "-" << mTrackletRowHits[tracklet.FirstHit() + (i - tracklet.FirstRow())] << ", ";
181  }
182  } else if (tracklet.NHits() && tracklet.LastRow() >= tracklet.FirstRow()) {
183  int nHits = 0;
184  for (int i = tracklet.FirstRow(); i <= tracklet.LastRow(); i++) {
185  calink ih = mTrackletRowHits[tracklet.FirstHit() + (i - tracklet.FirstRow())];
186  if (ih != CALINK_INVAL) {
187  nHits++;
188  }
189  out << i << "-" << mTrackletRowHits[tracklet.FirstHit() + (i - tracklet.FirstRow())] << ", ";
190  }
191  if (nHits != tracklet.NHits()) {
192  std::cout << std::endl
193  << "Wrong NHits!: Expected " << tracklet.NHits() << ", found " << nHits;
194  out << std::endl
195  << "Wrong NHits!: Expected " << tracklet.NHits() << ", found " << nHits;
196  }
197  }
198  out << std::endl;
199  }
