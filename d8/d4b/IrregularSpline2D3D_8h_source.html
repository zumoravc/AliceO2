201 GPUdi() void IrregularSpline2D3D::getKnotUV(int iKnot, float& u, float& v) const
202 {
204  const IrregularSpline1D& gridU = getGridU();
205  const IrregularSpline1D& gridV = getGridV();
206  int nu = gridU.getNumberOfKnots();
207  int iv = iKnot / nu;
208  int iu = iKnot % nu;
209  u = gridU.getKnot(iu).u;
210  v = gridV.getKnot(iv).u;

213 template <typename T>
214 GPUd() void IrregularSpline2D3D::correctEdges(T* data) const
215 {
216  const IrregularSpline1D& gridU = getGridU();
217  const IrregularSpline1D& gridV = getGridV();
218  int nu = gridU.getNumberOfKnots();
219  int nv = gridV.getNumberOfKnots();
220 
221  { // left edge of U
222  int iu = 0;
223  const IrregularSpline1D::Knot* s = gridU.getKnots() + iu;
224  double c0, c1, c2, c3;
225  gridU.getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
226  for (int iv = 0; iv < nv; iv++) {
227  T* f0 = data + (nu * (iv) + iu) * 3;
228  T* f1 = f0 + 3;
229  T* f2 = f0 + 6;
230  T* f3 = f0 + 9;
231  for (int idim = 0; idim < 3; idim++) {
232  f0[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
233  }
234  }
235  }
236 
237  { // right edge of U
238  int iu = nu - 4;
239  const IrregularSpline1D::Knot* s = gridU.getKnots() + iu;
240  double c0, c1, c2, c3;
241  gridU.getEdgeCorrectionCoefficients(s[3].u, s[2].u, s[1].u, s[0].u, c3, c2, c1, c0);
242  for (int iv = 0; iv < nv; iv++) {
243  T* f0 = data + (nu * (iv) + iu) * 3;
244  T* f1 = f0 + 3;
245  T* f2 = f0 + 6;
246  T* f3 = f0 + 9;
247  for (int idim = 0; idim < 3; idim++) {
248  f3[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
249  }
250  }
251  }
252 
253  { // low edge of V
254  int iv = 0;
255  const IrregularSpline1D::Knot* s = gridV.getKnots() + iv;
256  double c0, c1, c2, c3;
257  gridV.getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
258  for (int iu = 0; iu < nu; iu++) {
259  T* f0 = data + (nu * iv + iu) * 3;
260  T* f1 = f0 + nu * 3;
261  T* f2 = f0 + nu * 6;
262  T* f3 = f0 + nu * 9;
263  for (int idim = 0; idim < 3; idim++) {
264  f0[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
265  }
266  }
267  }
268 
269  { // high edge of V
270  int iv = nv - 4;
271  const IrregularSpline1D::Knot* s = gridV.getKnots() + iv;
272  double c0, c1, c2, c3;
273  gridV.getEdgeCorrectionCoefficients(s[3].u, s[2].u, s[1].u, s[0].u, c3, c2, c1, c0);
274  for (int iu = 0; iu < nu; iu++) {
275  T* f0 = data + (nu * iv + iu) * 3;
276  T* f1 = f0 + nu * 3;
277  T* f2 = f0 + nu * 6;
278  T* f3 = f0 + nu * 9;
279  for (int idim = 0; idim < 3; idim++) {
280  f3[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
281  }
282  }
283  }

287 GPUdi() void IrregularSpline2D3D::getSpline(GPUgeneric() const T* correctedData, float u, float v, GPUgeneric() T& x, GPUgeneric() T& y, GPUgeneric() T& z) const
288 {
289  // Get interpolated value for f(u,v) using data array correctedData[getNumberOfKnots()] with corrected edges
290 
291  const IrregularSpline1D& gridU = getGridU();
292  const IrregularSpline1D& gridV = getGridV();
293  int nu = gridU.getNumberOfKnots();
294  int iu = gridU.getKnotIndex(u);
295  int iv = gridV.getKnotIndex(v);
296 
297  const IrregularSpline1D::Knot& knotU = gridU.getKnot(iu);
298  const IrregularSpline1D::Knot& knotV = gridV.getKnot(iv);
299 
300  const T* dataV0 = correctedData + (nu * (iv - 1) + iu - 1) * 3;
301  const T* dataV1 = dataV0 + 3 * nu;
302  const T* dataV2 = dataV0 + 6 * nu;
303  const T* dataV3 = dataV0 + 9 * nu;
304 
305  T dataV[12];
306  for (int i = 0; i < 12; i++) {
307  dataV[i] = gridV.getSpline(knotV, dataV0[i], dataV1[i], dataV2[i], dataV3[i], v);
308  }
309 
310  T* dataU0 = dataV + 0;
311  T* dataU1 = dataV + 3;
312  T* dataU2 = dataV + 6;
313  T* dataU3 = dataV + 9;
314 
315  T res[3];
316  for (int i = 0; i < 3; i++) {
317  res[i] = gridU.getSpline(knotU, dataU0[i], dataU1[i], dataU2[i], dataU3[i], u);
318  }
319  x = res[0];
320  y = res[1];
321  z = res[2];
322 }

324 GPUdi() void IrregularSpline2D3D::getSplineVec(const float* correctedData, float u, float v, float& x, float& y, float& z) const
325 {
326  // Same as getSpline, but using vectorized calculation.
327  // \param correctedData should be at least 128-bit aligned
328 
329 #if !defined(__CINT__) && !defined(__ROOTCINT__) && !defined(GPUCA_GPUCODE) && !defined(GPUCA_NO_VC) && defined(__cplusplus) && __cplusplus >= 201703L
330  const IrregularSpline1D& gridU = getGridU();
331  const IrregularSpline1D& gridV = getGridV();
332  int nu = gridU.getNumberOfKnots();
333  int iu = gridU.getKnotIndex(u);
334  int iv = gridV.getKnotIndex(v);
335  const IrregularSpline1D::Knot& knotU = gridU.getKnot(iu);
336  const IrregularSpline1D::Knot& knotV = gridV.getKnot(iv);
337 
338  const float* dataV0 = correctedData + (nu * (iv - 1) + iu - 1) * 3;
339  const float* dataV1 = dataV0 + 3 * nu;
340  const float* dataV2 = dataV0 + 6 * nu;
341  const float* dataV3 = dataV0 + 9 * nu;
342 
343  // calculate F values at V==v and U == Ui of knots
344 
345  Vc::SimdArray<float, 12> dataV0vec(dataV0), dataV1vec(dataV1), dataV2vec(dataV2), dataV3vec(dataV3), dataVvec = gridV.getSpline(knotV, dataV0vec, dataV1vec, dataV2vec, dataV3vec, v);
346 
347  using V = std::conditional_t<(Vc::float_v::size() >= 4),
348  Vc::float_v,
349  Vc::SimdArray<float, 3>>;
350 
351  float dataV[9 + V::size()];
352  // dataVvec.scatter( dataV, Vc::SimdArray<float,12>::IndexType::IndexesFromZero() );
353  //dataVvec.scatter(dataV, Vc::SimdArray<uint, 12>(Vc::IndexesFromZero));
354  dataVvec.store(dataV, Vc::Unaligned);
355 
356  for (unsigned int i = 12; i < 9 + V::size(); i++) // fill not used part of the vector with 0
357  dataV[i] = 0.f;
358 
359  // calculate F values at V==v and U == u
360  V dataU0vec(dataV), dataU1vec(dataV + 3), dataU2vec(dataV + 6), dataU3vec(dataV + 9);
361 
362  V dataUVvec = gridU.getSpline(knotU, dataU0vec, dataU1vec, dataU2vec, dataU3vec, u);
363 
364  x = dataUVvec[0];
365  y = dataUVvec[1];
366  z = dataUVvec[2];
367 #else
368  getSpline(correctedData, u, v, x, y, z);
369 #endif
370 }
