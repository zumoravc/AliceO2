29 void DataReaderITS::open()
30 {
31  TString clusterFile = "o2clus_its.root";
32  TString trackFile = "o2trac_its.root";
33 
34  this->mTracFile = TFile::Open(trackFile);
35  this->mClusFile = TFile::Open(clusterFile);
36 
37  TTree* roft = (TTree*)this->mTracFile->Get("ITSTracksROF");
38  TTree* rofc = (TTree*)this->mClusFile->Get("ITSClustersROF");
39 
40  if (roft != nullptr && rofc != nullptr) {
41  // TTree* tracks = (TTree*)this->mTracFile->Get("o2sim");
42  TTree* tracksRof = (TTree*)this->mTracFile->Get("ITSTracksROF");
43 
44  // TTree* clusters = (TTree*)this->mClusFile->Get("o2sim");
45  TTree* clustersRof = (TTree*)this->mClusFile->Get("ITSClustersROF");
46 
47  //Read all track RO frames to a buffer
48  std::vector<o2::itsmft::ROFRecord>* trackROFrames = nullptr;
49  tracksRof->SetBranchAddress("ITSTracksROF", &trackROFrames);
50  tracksRof->GetEntry(0);
51 
52  //Read all cluster RO frames to a buffer
53  std::vector<o2::itsmft::ROFRecord>* clusterROFrames = nullptr;
54  clustersRof->SetBranchAddress("ITSClustersROF", &clusterROFrames);
55  clustersRof->GetEntry(0);
56 
57  if (trackROFrames->size() == clusterROFrames->size()) {
58  mMaxEv = trackROFrames->size();
59  } else {
60  Error("DataReaderITS", "Inconsistent number of readout frames in files");
61  exit(1);
62  }
63  }
