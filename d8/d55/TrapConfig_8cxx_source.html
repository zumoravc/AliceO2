36 TrapConfig::TrapConfig(std::string configname)
37 {
38  // default constructor
39 
40  // initialize and reset the TRAP registers
41  initRegs();
42  resetRegs();
43 
44  for (int iWord = 0; iWord < mgkDmemWords; ++iWord) {
45  mDmem[iWord].setAddress(iWord + mgkDmemStartAddress);
46  }
47 
48  // initialize the map from address to register
49  if (!mgRegAddressMapInitialized) {
50  for (int iReg = 0; iReg < kLastReg; iReg++) {
51  int addr = mRegisterValue[iReg].getAddr();
52  if (addr < mgkRegisterAddressBlockStart[0]) {
53  LOG(error) << "Register address 0x" << hex << std::setw(4) << addr << " not handled in register map";
54  } else if (addr < mgkRegisterAddressBlockStart[0] + mgkRegisterAddressBlockSize[0]) {
55  mgRegAddressMap[addr - mgkRegisterAddressBlockStart[0]] = (TrapReg_t)iReg;
56  } else if (addr < mgkRegisterAddressBlockStart[1]) {
57  LOG(error) << "Register address 0x" << hex << setw(4) << addr << " not handled in register map";
58  } else if (addr < mgkRegisterAddressBlockStart[1] + mgkRegisterAddressBlockSize[1]) {
59  mgRegAddressMap[addr - mgkRegisterAddressBlockStart[1] + mgkRegisterAddressBlockSize[0]] = (TrapReg_t)iReg;
60  } else if (addr < mgkRegisterAddressBlockStart[2]) {
61  LOG(error) << "Register address 0x" << hex << setw(4) << addr << " not handled in register map";
62  } else if (addr < mgkRegisterAddressBlockStart[2] + mgkRegisterAddressBlockSize[2]) {
63  int ind = addr - mgkRegisterAddressBlockStart[2] + mgkRegisterAddressBlockSize[1] + mgkRegisterAddressBlockSize[0];
64  mgRegAddressMap[ind] = (TrapReg_t)iReg;
65  } else {
66  LOG(error) << "Register address 0x" << hex << setw(4) << addr << " not handled in register map";
67  }
68  }
69  mgRegAddressMapInitialized = true;
70  }
71 }
