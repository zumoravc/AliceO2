99 template <typename RAWDataHeader, bool verbose>
100 bool Compressor<RAWDataHeader, verbose>::processHBF()
101 {
102 
103  if (verbose && mDecoderVerbose) {
104  std::cout << colorBlue
105  << "--- PROCESS HBF"
106  << colorReset
107  << std::endl;
108  }
109 
110  mDecoderRDH = reinterpret_cast<const RAWDataHeader*>(mDecoderPointer);
111  mEncoderRDH = reinterpret_cast<RAWDataHeader*>(mEncoderPointer);
112  auto rdh = mDecoderRDH;
113 
115  while (!rdh->stop) {
116 
117  if (verbose && mDecoderVerbose) {
118  std::cout << colorBlue
119  << "--- RDH open/continue detected"
120  << colorReset
121  << std::endl;
122  o2::raw::RDHUtils::printRDH(*rdh);
123  }
124 
125  auto headerSize = rdh->headerSize;
126  auto memorySize = rdh->memorySize;
127  auto offsetToNext = rdh->offsetToNext;
128  auto drmPayload = memorySize - headerSize;
129 
131  std::memcpy(mDecoderSaveBuffer + mDecoderSaveBufferDataSize, reinterpret_cast<const char*>(rdh) + headerSize, drmPayload);
132  mDecoderSaveBufferDataSize += drmPayload;
133 
135  rdh = reinterpret_cast<const RAWDataHeader*>(reinterpret_cast<const char*>(rdh) + offsetToNext);
136 
138  if (reinterpret_cast<const char*>(rdh) < mDecoderBuffer + mDecoderBufferSize)
139  continue;
140 
142  return true;
143  }
144 
145  if (verbose && mDecoderVerbose) {
146  std::cout << colorBlue
147  << "--- RDH close detected"
148  << colorReset
149  << std::endl;
150  o2::raw::RDHUtils::printRDH(*rdh);
151  }
152 
154  std::memcpy(mEncoderPointer, mDecoderRDH, mDecoderRDH->headerSize);
155  mEncoderPointer = reinterpret_cast<uint32_t*>(reinterpret_cast<char*>(mEncoderPointer) + rdh->headerSize);
156 
158  mDecoderPointer = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer);
159  mDecoderPointerMax = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer + mDecoderSaveBufferDataSize);
160  while (mDecoderPointer < mDecoderPointerMax) {
161  mEventCounter++;
162  if (processDRM()) { // if this breaks, we did not run the checker and the summary is not reset!
163  mDecoderSummary = {nullptr}; // reset it like this, perhaps a better way can be found
164  break;
165  }
166  }
167  mDecoderSaveBufferDataSize = 0;
168 
170  if (mDecoderFatal) {
171  mFatalCounter++;
172  mEncoderPointer = mEncoderPointerStart;
173  }
174 
175  if (mDecoderError)
176  mErrorCounter++;
177 
179  mEncoderRDH->memorySize = reinterpret_cast<char*>(mEncoderPointer) - reinterpret_cast<char*>(mEncoderRDH);
180  mEncoderRDH->offsetToNext = mEncoderRDH->memorySize;
181 
184  mEncoderRDH = reinterpret_cast<RAWDataHeader*>(mEncoderPointer);
185  std::memcpy(mEncoderRDH, rdh, rdh->headerSize);
186  mEncoderRDH->memorySize = rdh->headerSize;
187  mEncoderRDH->offsetToNext = mEncoderRDH->memorySize;
188  mEncoderPointer = reinterpret_cast<uint32_t*>(reinterpret_cast<char*>(mEncoderPointer) + rdh->headerSize);
189 
190  if (verbose && mDecoderVerbose) {
191  std::cout << colorBlue
192  << "--- END PROCESS HBF"
193  << colorReset
194  << std::endl;
195  }
196 
198  mDecoderPointer = reinterpret_cast<const uint32_t*>(reinterpret_cast<const char*>(rdh) + rdh->offsetToNext);
199 
201  if (reinterpret_cast<const char*>(mDecoderPointer) < mDecoderBuffer + mDecoderBufferSize)
202  return false;
203 
205  return true;
206 }

208 template <typename RAWDataHeader, bool verbose>
209 bool Compressor<RAWDataHeader, verbose>::processDRM()
210 {
211 
212  if (verbose && mDecoderVerbose) {
213  std::cout << colorBlue << "--- PROCESS DRM"
214  << colorReset
215  << std::endl;
216  }
217 
219  mDecoderNextWord = 1;
220  mDecoderError = false;
221  mDecoderFatal = false;
222  mEncoderPointerStart = mEncoderPointer;
223 
225  if (!IS_DRM_COMMON_HEADER(*mDecoderPointer)) {
226  if (verbose) {
227  printf("%s %08x [ERROR] fatal error %s \n", colorRed, *mDecoderPointer, colorReset);
228  }
229  mDecoderFatal = true;
230  return true;
231  }
232  mDecoderSummary.tofDataHeader = mDecoderPointer;
233  if (verbose && mDecoderVerbose) {
234  auto tofDataHeader = reinterpret_cast<const raw::TOFDataHeader_t*>(mDecoderPointer);
235  auto bytePayload = tofDataHeader->bytePayload;
236  printf(" %08x TOF Data Header (bytePayload=%d) \n", *mDecoderPointer, bytePayload);
237  }
238 #ifdef DECODER_PARANOID
239  if (decoderParanoid())
240  return true;
241 #endif
242  decoderNext();
243 
245  mDecoderSummary.tofOrbit = mDecoderPointer;
246  if (verbose && mDecoderVerbose) {
247  auto tofOrbit = reinterpret_cast<const raw::TOFOrbit_t*>(mDecoderPointer);
248  auto orbit = tofOrbit->orbit;
249  printf(" %08x TOF Orbit (orbit=%u) \n", *mDecoderPointer, orbit);
250  }
251 #ifdef DECODER_PARANOID
252  if (decoderParanoid())
253  return true;
254 #endif
255  decoderNext();
256 
258  if (!IS_DRM_GLOBAL_HEADER(*mDecoderPointer)) {
259  if (verbose) {
260  printf("%s %08x [ERROR] fatal error %s \n", colorRed, *mDecoderPointer, colorReset);
261  }
262  mDecoderFatal = true;
263  return true;
264  }
265  mDecoderSummary.drmDataHeader = mDecoderPointer;
266  if (verbose && mDecoderVerbose) {
267  auto drmDataHeader = reinterpret_cast<const raw::DRMDataHeader_t*>(mDecoderPointer);
268  auto drmId = drmDataHeader->drmId;
269  auto eventWords = drmDataHeader->eventWords;
270  printf(" %08x DRM Data Header (drmId=%d, eventWords=%d) \n", *mDecoderPointer, drmId, eventWords);
271  }
272 #ifdef DECODER_PARANOID
273  if (decoderParanoid())
274  return true;
275 #endif
276  decoderNext();
277 
279  mDecoderSummary.drmHeadW1 = mDecoderPointer;
280  if (verbose && mDecoderVerbose) {
281  auto drmHeadW1 = reinterpret_cast<const raw::DRMHeadW1_t*>(mDecoderPointer);
282  auto partSlotMask = drmHeadW1->partSlotMask;
283  auto clockStatus = drmHeadW1->clockStatus;
284  auto drmHSize = drmHeadW1->drmHSize;
285  printf(" %08x DRM Header Word 1 (partSlotMask=0x%03x, clockStatus=%d, drmHSize=%d) \n", *mDecoderPointer, partSlotMask, clockStatus, drmHSize);
286  }
287 #ifdef DECODER_PARANOID
288  if (decoderParanoid())
289  return true;
290 #endif
291  decoderNext();
292 
294  mDecoderSummary.drmHeadW2 = mDecoderPointer;
295  if (verbose && mDecoderVerbose) {
296  auto drmHeadW2 = reinterpret_cast<const raw::DRMHeadW2_t*>(mDecoderPointer);
297  auto enaSlotMask = drmHeadW2->enaSlotMask;
298  auto faultSlotMask = drmHeadW2->faultSlotMask;
299  auto readoutTimeOut = drmHeadW2->readoutTimeOut;
300  printf(" %08x DRM Header Word 2 (enaSlotMask=0x%03x, faultSlotMask=0x%03x, readoutTimeOut=%d) \n", *mDecoderPointer, enaSlotMask, faultSlotMask, readoutTimeOut);
301  }
302 #ifdef DECODER_PARANOID
303  if (decoderParanoid())
304  return true;
305 #endif
306  decoderNext();
307 
309  mDecoderSummary.drmHeadW3 = mDecoderPointer;
310  if (verbose && mDecoderVerbose) {
311  auto drmHeadW3 = reinterpret_cast<const raw::DRMHeadW3_t*>(mDecoderPointer);
312  auto gbtBunchCnt = drmHeadW3->gbtBunchCnt;
313  auto locBunchCnt = drmHeadW3->locBunchCnt;
314  printf(" %08x DRM Header Word 3 (gbtBunchCnt=%d, locBunchCnt=%d) \n", *mDecoderPointer, gbtBunchCnt, locBunchCnt);
315  }
316 #ifdef DECODER_PARANOID
317  if (decoderParanoid())
318  return true;
319 #endif
320  decoderNext();
321 
323  mDecoderSummary.drmHeadW4 = mDecoderPointer;
324  if (verbose && mDecoderVerbose) {
325  printf(" %08x DRM Header Word 4 \n", *mDecoderPointer);
326  }
327 #ifdef DECODER_PARANOID
328  if (decoderParanoid())
329  return true;
330 #endif
331  decoderNext();
332 
334  mDecoderSummary.drmHeadW5 = mDecoderPointer;
335  if (verbose && mDecoderVerbose) {
336  printf(" %08x DRM Header Word 5 \n", *mDecoderPointer);
337  }
338 #ifdef DECODER_PARANOID
339  if (decoderParanoid())
340  return true;
341 #endif
342  decoderNext();
343 
345  *mEncoderPointer = 0x80000000;
346  *mEncoderPointer |= GET_DRMHEADW1_PARTSLOTMASK(*mDecoderSummary.drmHeadW1) << 12;
347  *mEncoderPointer |= GET_DRMDATAHEADER_DRMID(*mDecoderSummary.drmDataHeader) << 24;
348  *mEncoderPointer |= GET_DRMHEADW3_GBTBUNCHCNT(*mDecoderSummary.drmHeadW3);
349  if (verbose && mEncoderVerbose) {
350  auto crateHeader = reinterpret_cast<compressed::CrateHeader_t*>(mEncoderPointer);
351  auto bunchID = crateHeader->bunchID;
352  auto drmID = crateHeader->drmID;
353  auto slotPartMask = crateHeader->slotPartMask;
354  printf("%s %08x Crate header (drmID=%d, bunchID=%d, slotPartMask=0x%x) %s \n", colorGreen, *mEncoderPointer, drmID, bunchID, slotPartMask, colorReset);
355  }
356  encoderNext();
357 
359  *mEncoderPointer = *mDecoderSummary.tofOrbit;
360  if (verbose && mEncoderVerbose) {
361  auto crateOrbit = reinterpret_cast<compressed::CrateOrbit_t*>(mEncoderPointer);
362  auto orbitID = crateOrbit->orbitID;
363  printf("%s %08x Crate orbit (orbitID=%u) %s \n", colorGreen, *mEncoderPointer, orbitID, colorReset);
364  }
365  encoderNext();
366 
368  while (true) {
369 
371  if (IS_LTM_GLOBAL_HEADER(*mDecoderPointer)) {
372  if (processLTM())
373  return true;
374  }
375 
377  if (IS_TRM_GLOBAL_HEADER(*mDecoderPointer) && GET_TRMDATAHEADER_SLOTID(*mDecoderPointer) > 2) {
378  if (processTRM())
379  return true;
380  continue;
381  }
382 
384  if (IS_DRM_GLOBAL_TRAILER(*mDecoderPointer)) {
385  mDecoderSummary.drmDataTrailer = mDecoderPointer;
386  if (verbose && mDecoderVerbose) {
387  auto drmDataTrailer = reinterpret_cast<const raw::DRMDataTrailer_t*>(mDecoderPointer);
388  auto locEvCnt = drmDataTrailer->locEvCnt;
389  printf(" %08x DRM Data Trailer (locEvCnt=%d) \n", *mDecoderPointer, locEvCnt);
390  }
391 #ifdef DECODER_PARANOID
392  if (decoderParanoid())
393  return true;
394 #endif
395  decoderNext();
396 
398  if (IS_FILLER(*mDecoderPointer)) {
399  if (verbose && mDecoderVerbose) {
400  printf(" %08x Filler \n", *mDecoderPointer);
401  }
402 #ifdef DECODER_PARANOID
403  if (decoderParanoid())
404  return true;
405 #endif
406  decoderNext();
407  }
408 
410  *mEncoderPointer = 0x80000000;
411  *mEncoderPointer |= GET_DRMDATATRAILER_LOCEVCNT(*mDecoderSummary.drmDataTrailer) << 4;
412 
414  checkerCheck();
415  *mEncoderPointer |= mCheckerSummary.nDiagnosticWords;
416  *mEncoderPointer |= (mCheckerSummary.nTDCErrors << 16);
417 
418  if (verbose && mEncoderVerbose) {
419  auto CrateTrailer = reinterpret_cast<compressed::CrateTrailer_t*>(mEncoderPointer);
420  auto EventCounter = CrateTrailer->eventCounter;
421  auto NumberOfDiagnostics = CrateTrailer->numberOfDiagnostics;
422  auto NumberOfErrors = CrateTrailer->numberOfErrors;
423  printf("%s %08x Crate trailer (EventCounter=%d, NumberOfDiagnostics=%d, NumberOfErrors=%d) %s \n", colorGreen, *mEncoderPointer, EventCounter, NumberOfDiagnostics, NumberOfErrors, colorReset);
424  }
425  encoderNext();
426 
428  for (int iword = 0; iword < mCheckerSummary.nDiagnosticWords; ++iword) {
429  auto itrm = (mCheckerSummary.DiagnosticWord[iword] & 0xF) - 3;
430  *mEncoderPointer = mCheckerSummary.DiagnosticWord[iword];
431  if (verbose && mEncoderVerbose) {
432  auto Diagnostic = reinterpret_cast<compressed::Diagnostic_t*>(mEncoderPointer);
433  auto slotId = Diagnostic->slotID;
434  auto faultBits = Diagnostic->faultBits;
435  printf("%s %08x Diagnostic (slotId=%d, faultBits=0x%x) %s \n", colorGreen, *mEncoderPointer, slotId, faultBits, colorReset);
436  }
437  encoderNext();
438  }
439 
441  for (int itrm = 0; itrm < 10; ++itrm) {
442  for (int ichain = 0; ichain < 2; ++ichain) {
443  for (int ierror = 0; ierror < mDecoderSummary.trmErrors[itrm][ichain]; ++ierror) {
444  *mEncoderPointer = *mDecoderSummary.trmError[itrm][ichain][ierror];
445  *mEncoderPointer &= 0xFF07FFFF;
446  *mEncoderPointer |= ((itrm + 3) << 19);
447  *mEncoderPointer |= (ichain << 23);
448  if (verbose && mEncoderVerbose) {
449  auto Error = reinterpret_cast<compressed::Error_t*>(mEncoderPointer);
450  auto errorFlags = Error->errorFlags;
451  auto slotID = Error->slotID;
452  auto chain = Error->chain;
453  auto tdcID = Error->tdcID;
454  printf("%s %08x Error (slotId=%d, chain=%d, tdcId=%d, errorFlags=0x%x) %s \n", colorGreen, *mEncoderPointer, slotID, chain, tdcID, errorFlags, colorReset);
455  }
456  encoderNext();
457  }
458  mDecoderSummary.trmErrors[itrm][ichain] = 0;
459  }
460  }
461 
462  mCheckerSummary.nDiagnosticWords = 0;
463  mCheckerSummary.nTDCErrors = 0;
464 
465  break;
466  }
467 
469  if (IS_DRM_TEST_WORD(*mDecoderPointer)) {
470  if (verbose && mDecoderVerbose) {
471  printf(" %08x DRM Test Word \n", *mDecoderPointer);
472  }
473  decoderNext();
474  continue;
475  }
476 
478  mDecoderError = true;
479 
480  if (verbose && mDecoderVerbose) {
481  printf("%s %08x [ERROR] trying to recover DRM decode stream %s \n", colorRed, *mDecoderPointer, colorReset);
482  }
483 
485  if (decoderParanoid())
486  return true;
487 
488  decoderNext();
489 
490  }
492  mIntegratedBytes += getDecoderByteCounter();
493 
494  if (verbose && mDecoderVerbose) {
495  std::cout << colorBlue
496  << "--- END PROCESS DRM"
497  << colorReset
498  << std::endl;
499  }
500 
501  return false;
502 }

547 template <typename RAWDataHeader, bool verbose>
548 bool Compressor<RAWDataHeader, verbose>::processTRM()
549 {
552  uint32_t slotId = GET_TRMDATAHEADER_SLOTID(*mDecoderPointer);
553  int itrm = slotId - 3;
554  mDecoderSummary.trmDataHeader[itrm] = mDecoderPointer;
555  if (verbose && mDecoderVerbose) {
556  auto trmDataHeader = reinterpret_cast<const raw::TRMDataHeader_t*>(mDecoderPointer);
557  auto eventWords = trmDataHeader->eventWords;
558  auto eventCnt = trmDataHeader->eventCnt;
559  auto emptyBit = trmDataHeader->emptyBit;
560  printf(" %08x TRM Data Header (slotId=%u, eventWords=%d, eventCnt=%d, emptyBit=%01x) \n", *mDecoderPointer, slotId, eventWords, eventCnt, emptyBit);
561  }
562 #ifdef DECODER_PARANOID
563  if (decoderParanoid())
564  return true;
565 #endif
566  decoderNext();
567 
569  while (true) {
570 
572  if (IS_TRM_CHAINA_HEADER(*mDecoderPointer) && GET_TRMCHAINHEADER_SLOTID(*mDecoderPointer) == slotId) {
573  if (processTRMchain(itrm, 0))
574  return true;
575  }
576 
578  if (IS_TRM_CHAINB_HEADER(*mDecoderPointer) && GET_TRMCHAINHEADER_SLOTID(*mDecoderPointer) == slotId) {
579  if (processTRMchain(itrm, 1))
580  return true;
581  }
582 
584  if (IS_TRM_GLOBAL_TRAILER(*mDecoderPointer)) {
585  mDecoderSummary.trmDataTrailer[itrm] = mDecoderPointer;
586  if (verbose && mDecoderVerbose) {
587  auto trmDataTrailer = reinterpret_cast<const raw::TRMDataTrailer_t*>(mDecoderPointer);
588  auto eventCRC = trmDataTrailer->eventCRC;
589  auto lutErrorBit = trmDataTrailer->lutErrorBit;
590  printf(" %08x TRM Data Trailer (slotId=%u, eventCRC=%d, lutErrorBit=%d) \n", *mDecoderPointer, slotId, eventCRC, lutErrorBit);
591  }
592 #ifdef DECODER_PARANOID
593  if (decoderParanoid())
594  return true;
595 #endif
596  decoderNext();
597 
599  if (IS_FILLER(*mDecoderPointer)) {
600  if (verbose && mDecoderVerbose) {
601  printf(" %08x Filler \n", *mDecoderPointer);
602  }
603 #ifdef DECODER_PARANOID
604  if (decoderParanoid())
605  return true;
606 #endif
607  decoderNext();
608  }
609 
611  if (mDecoderSummary.hasHits[itrm][0] || mDecoderSummary.hasHits[itrm][1])
612  encoderSpider(itrm);
613 
615  return false;
616  }
617 
619  mDecoderError = true;
620  if (verbose && mDecoderVerbose) {
621  printf("%s %08x [ERROR] breaking TRM decode stream %s \n", colorRed, *mDecoderPointer, colorReset);
622  }
624  if (decoderParanoid())
625  return true;
626 
627  decoderNext();
628  return false;
629 
630  }
633  return false;
634 }

636 template <typename RAWDataHeader, bool verbose>
637 bool Compressor<RAWDataHeader, verbose>::processTRMchain(int itrm, int ichain)
638 {
641  int slotId = itrm + 3;

831  }

833

836  *mEncoderPointer = mSpiderSummary.FramePackedHit[iframe][ihit];

845  }
846  encoderNext();
