24 {

34 template <typename CHARGESUM>
35 void align(gsl::span<ElinkEncoder<BareFormat, CHARGESUM>> elinks)
36 {
37  if (areElinksAligned(elinks)) {
38  return;
39  }
40  auto e = std::max_element(elinks.begin(), elinks.end(),
41  [](const ElinkEncoder<BareFormat, CHARGESUM>& a, const ElinkEncoder<BareFormat, CHARGESUM>& b) {
42  return a.len() < b.len();
43  });
44 
45  // align all elink sizes by adding sync bits
46  for (auto& elink : elinks) {
47  elink.fillWithSync(e->len());
48  }
49 }

51 template <typename CHARGESUM>
52 uint64_t aggregate(gsl::span<ElinkEncoder<BareFormat, CHARGESUM>> elinks, int jstart, int jend, int i)
53 {
54  uint64_t w{0};
55  for (int j = jstart; j < jend; j += 2) {
56  for (int k = 0; k <= 1; k++) {
57  bool v = elinks[j / 2].get(i + 1 - k);
58  uint64_t mask = static_cast<uint64_t>(1) << (j + k);
59  if (v) {
60  w |= mask;
61  } else {
62  w &= ~mask;
63  }
64  }
65  }
66  return w;
67 }

69 template <typename CHARGESUM>
70 void elink2gbt(gsl::span<ElinkEncoder<BareFormat, CHARGESUM>> elinks, std::vector<uint64_t>& b64)
71 {
72  int n = elinks[0].len();
73 
74  for (int i = 0; i < n - 1; i += 2) {
75  uint64_t w0 = aggregate(elinks, 0, 64, i);
76  uint64_t w1 = aggregate(elinks, 64, 80, i);
77  b64.push_back(w0);
78  b64.push_back(w1);
79  }
80 }
