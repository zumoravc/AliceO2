43 BOOST_AUTO_TEST_CASE(BadChannelMap_test)
44 {
45  auto geo = o2::emcal::Geometry::GetInstanceFromRunNumber(300000);
46 
47  // Closure creating function testing whether a cell is good, bad
48  // or warm. The outer function takes two vectors, (with bad and warm cells),
49  // the inner function (used in comparisons) a cell ID.
50  auto refChannelCombined = [](const std::vector<unsigned short>& deadcells, const std::vector<unsigned short>& badcells, const std::vector<unsigned short>& warmcells) {
51  return [&deadcells, &badcells, &warmcells](unsigned short cellID) {
52  if (std::find(deadcells.begin(), deadcells.end(), cellID) != deadcells.end())
53  return BadChannelMap::MaskType_t::DEAD_CELL;
54  if (std::find(badcells.begin(), badcells.end(), cellID) != badcells.end())
55  return BadChannelMap::MaskType_t::BAD_CELL;
56  if (std::find(warmcells.begin(), warmcells.end(), cellID) != warmcells.end())
57  return BadChannelMap::MaskType_t::WARM_CELL;
58  return BadChannelMap::MaskType_t::GOOD_CELL;
59  };
60  };
61 
62  // Single channel test
63  //
64  // For each channel test set and update, and compare read value
65  // against set value
66  for (unsigned short c = 0; c < static_cast<unsigned short>(geo->GetNCells()); c++) {
67  BadChannelMap singletest;
68  singletest.addBadChannel(c, BadChannelMap::MaskType_t::DEAD_CELL);
69  BOOST_CHECK_EQUAL(singletest.getChannelStatus(c), BadChannelMap::MaskType_t::DEAD_CELL);
70  // update: BAD_CELL
71  singletest.addBadChannel(c, BadChannelMap::MaskType_t::BAD_CELL);
72  BOOST_CHECK_EQUAL(singletest.getChannelStatus(c), BadChannelMap::MaskType_t::BAD_CELL);
73  // update: WARN_CELL
74  singletest.addBadChannel(c, BadChannelMap::MaskType_t::WARM_CELL);
75  BOOST_CHECK_EQUAL(singletest.getChannelStatus(c), BadChannelMap::MaskType_t::WARM_CELL);
76  // update: GOOD_CELL (=erase)
77  singletest.addBadChannel(c, BadChannelMap::MaskType_t::GOOD_CELL);
78  BOOST_CHECK_EQUAL(singletest.getChannelStatus(c), BadChannelMap::MaskType_t::GOOD_CELL);
79  }
80 
81  // Pattern test
82  //
83  // Insert user definde pattern of bad and warm channels, and check for each channel
84  // whether the channel is masked or not. The original pattern has to be reconstructed.
85  // Test also handles checking good cells.
86  //
87  // Test data obtained from LHC18m block1
88  BadChannelMap patterntest;
89  std::vector<unsigned short> badcells = getBadChannelsLHC18m_block1(),
90  deadcells = getDeadChannelsLHC18m_block1(),
91  warmcells = getWarmChannelsLHC18m_block1();
92  for (auto bc : badcells) {
93  patterntest.addBadChannel(bc, BadChannelMap::MaskType_t::BAD_CELL);
94  }
95  for (auto wc : warmcells) {
96  patterntest.addBadChannel(wc, BadChannelMap::MaskType_t::WARM_CELL);
97  }
98  for (auto dc : deadcells) {
99  patterntest.addBadChannel(dc, BadChannelMap::MaskType_t::DEAD_CELL);
100  }
101  auto getRefChannelTypeLHC18m = refChannelCombined(deadcells, badcells, warmcells);
102  for (unsigned short c = 0; c < static_cast<unsigned short>(geo->GetNCells()); c++) {
103  BOOST_CHECK_EQUAL(patterntest.getChannelStatus(c), getRefChannelTypeLHC18m(c));
104  }
105 
106  // Combine test
107  // Second bad channel map is taken from LHC17o block 2
108  BadChannelMap lhc17omap;
109  std::vector<unsigned short> badLHC17o = getBadChannelsLHC17o_block2(),
110  warmLHC7o = getWarmChannelsLHC17o_block2(),
111  deadLHC17o = getDeadChannelsLHC17o_block2(),
112  combinedBad = combineMaps(badLHC17o, badcells),
113  combinedDead = combineMaps(deadLHC17o, deadcells),
114  combinedWarm = combineMaps(warmLHC7o, warmcells);
115  filterBad(combinedWarm, combinedBad);
116  filterBad(combinedWarm, combinedDead);
117  filterBad(combinedBad, combinedDead);
118  for (auto bc : badLHC17o)
119  lhc17omap.addBadChannel(bc, BadChannelMap::MaskType_t::BAD_CELL);
120  for (auto wc : warmLHC7o)
121  lhc17omap.addBadChannel(wc, BadChannelMap::MaskType_t::WARM_CELL);
122  for (auto dc : deadLHC17o)
123  lhc17omap.addBadChannel(dc, BadChannelMap::MaskType_t::DEAD_CELL);
124  BadChannelMap summed(lhc17omap);
125  summed += patterntest;
126  auto getRefChannelTypeCombined = refChannelCombined(combinedDead, combinedBad, combinedWarm);
127  for (unsigned short c = 0; c < static_cast<unsigned short>(geo->GetNCells()); c++) {
128  BOOST_CHECK_EQUAL(summed.getChannelStatus(c), getRefChannelTypeCombined(c));
129  }
130 
131  // Equal
132  //
133  // - Compare map for LHC17o with itself. The result must be true.
134  // - Compare map for LHC18m with map for LHC17o. The result must be false
135  BOOST_CHECK_EQUAL(lhc17omap == lhc17omap, true);
136  BOOST_CHECK_EQUAL(lhc17omap == patterntest, false);
137 }

143 std::vector<unsigned short> combineMaps(const std::vector<unsigned short>& list1, const std::vector<unsigned short> list2)
144 {
145  std::vector<unsigned short> combined;
146  for (auto l1 : list1)
147  combined.emplace_back(l1);
148  for (auto l2 : list2) {
149  auto found = std::find(combined.begin(), combined.end(), l2);
150  if (found == combined.end())
151  combined.emplace_back(l2);
152  }
153  std::sort(combined.begin(), combined.end(), std::less<>());
154  return combined;
155 }

157 void filterBad(std::vector<unsigned short> warmcells, const std::vector<unsigned short> badcells)
158 {
159  std::vector<unsigned short> todelete;
160  for (auto wc : warmcells)
161  if (std::find(badcells.begin(), badcells.end(), wc) != badcells.end())
162  todelete.emplace_back(wc);
163  for (auto td : todelete) {
164  auto it = std::find(warmcells.begin(), warmcells.end(), td);
165  warmcells.erase(it);
166  }
167 }
