326 void Chebyshev3D::prepareBoundaries(const Float_t* bmin, const Float_t* bmax)
327 {
328  // Set and check boundaries defined by user, prepare coefficients for their conversion to [-1:1] interval
329  for (int i = 3; i--;) {
330  mMinBoundaries[i] = bmin[i];
331  mMaxBoundaries[i] = bmax[i];
332  mBoundaryMappingScale[i] = bmax[i] - bmin[i];
333  if (mBoundaryMappingScale[i] <= 0) {
334  LOG(FATAL) << "Boundaries for " << i << "-th dimension are not increasing: "
335  << mMinBoundaries[i] << " " << mMaxBoundaries[i] << "\nStop\n";
336  }
337  mBoundaryMappingOffset[i] = bmin[i] + mBoundaryMappingScale[i] / 2.0;
338  mBoundaryMappingScale[i] = 2. / mBoundaryMappingScale[i];
339  }
340 }

424 #ifdef _INC_CREATION_Chebyshev3D_
425 Int_t Chebyshev3D::calculateChebyshevCoefficients(const Float_t* funval, int np, Float_t* outCoefs, Float_t prec)
426 {
427  // Calculate Chebyshev coeffs using precomputed function values at np roots.
428  // If prec>0, estimate the highest coeff number providing the needed precision
429  double sm; // do summations in double to minimize the roundoff error
430  for (int ic = 0; ic < np; ic++) { // compute coeffs
431  sm = 0;
432  for (int ir = 0; ir < np; ir++) {
433  float rt = TMath::Cos(ic * (ir + 0.5) * TMath::Pi() / np);
434  sm += funval[ir] * rt;
435  }
436  outCoefs[ic] = Float_t(sm * ((ic == 0) ? 1. / np : 2. / np));
437  }
438 
439  if (prec <= 0) {
440  return np;
441  }
442 
443  sm = 0;
444  int cfMax = 0;
445  for (cfMax = np; cfMax--;) {
446  sm += TMath::Abs(outCoefs[cfMax]);
447  if (sm >= prec) {
448  break;
449  }
450  }
451  if (++cfMax == 0) {
452  cfMax = 1;
453  }
454  return cfMax;
455 }
456 #endif

459 void Chebyshev3D::defineGrid(const Int_t* npoints)
460 {
461  // prepare the grid of Chebyshev roots in each dimension
462  const int kMinPoints = 1;
463  int ntot = 0;
464  mMaxCoefficients = 1;
465  for (int id = 3; id--;) {
466  mNumberOfPoints[id] = npoints[id];
467  if (mNumberOfPoints[id] < kMinPoints) {
468  LOG(ERROR) << "DefineGrid: at " << id << "-th dimension " << mNumberOfPoints[id] << " point is requested, at least " << kMinPoints << " is needed\nStop\n";
469  exit(1);
470  }
471  ntot += mNumberOfPoints[id];
472  mMaxCoefficients *= mNumberOfPoints[id];
473  }
474  printf("Computing Chebyshev nodes on [%2d/%2d/%2d] grid\n", npoints[0], npoints[1], npoints[2]);
475  if (mTemporaryChebyshevGrid) {
476  delete[] mTemporaryChebyshevGrid;
477  }
478  mTemporaryChebyshevGrid = new Float_t[ntot];
479 
480  int curp = 0;
481  for (int id = 3; id--;) {
482  int np = mNumberOfPoints[id];
483  mTemporaryChebyshevGridOffs[id] = curp;
484  for (int ip = 0; ip < np; ip++) {
485  Float_t x = TMath::Cos(TMath::Pi() * (ip + 0.5) / np);
486  mTemporaryChebyshevGrid[curp++] = mapToExternal(x, id);
487  }
488  }
489 }

504 #ifdef _INC_CREATION_Chebyshev3D_
505 Int_t Chebyshev3D::chebyshevFit(int dmOut)
506 {
507  // prepare paramaterization of 3D function for dmOut-th dimension
508  int maxDim = 0;
509  for (int i = 0; i < 3; i++) {
510  if (maxDim < mNumberOfPoints[i]) {
511  maxDim = mNumberOfPoints[i];
512  }
513  }
514  auto* fvals = new Float_t[mNumberOfPoints[0]];
515  auto* tmpCoef3D = new Float_t[mNumberOfPoints[0] * mNumberOfPoints[1] * mNumberOfPoints[2]];
516  auto* tmpCoef2D = new Float_t[mNumberOfPoints[0] * mNumberOfPoints[1]];
517  auto* tmpCoef1D = new Float_t[maxDim];
518 
519  // 1D Cheb.fit for 0-th dimension at current steps of remaining dimensions
520  int ncmax = 0;
521 
522  printf("Dim%d : 00.00%% Done", dmOut);
523  fflush(stdout);
524  Chebyshev3DCalc* cheb = getChebyshevCalc(dmOut);
525 
526  Float_t prec = cheb->getPrecision();
527  if (prec < sMinimumPrecision)
528  prec = mPrecision; // no specific precision for this dim.
529  Float_t rTiny = 0.1 * prec / Float_t(maxDim); // neglect coefficient below this threshold
530 
531  float ncals2count = mNumberOfPoints[2] * mNumberOfPoints[1] * mNumberOfPoints[0];
532  float ncals = 0;
533  float frac = 0;
534  float fracStep = 0.001;
535 
536  for (int id2 = mNumberOfPoints[2]; id2--;) {
537  mTemporaryCoefficient[2] = mTemporaryChebyshevGrid[mTemporaryChebyshevGridOffs[2] + id2];
538 
539  for (int id1 = mNumberOfPoints[1]; id1--;) {
540  mTemporaryCoefficient[1] = mTemporaryChebyshevGrid[mTemporaryChebyshevGridOffs[1] + id1];
541 
542  for (int id0 = mNumberOfPoints[0]; id0--;) {
543  mTemporaryCoefficient[0] = mTemporaryChebyshevGrid[mTemporaryChebyshevGridOffs[0] + id0];
544  evaluateUserFunction(); // compute function values at Chebyshev roots of 0-th dimension
545  fvals[id0] = mTemporaryUserResults[dmOut];
546  float fr = (++ncals) / ncals2count;
547  if (fr - frac >= fracStep) {
548  frac = fr;
549  printf("\b\b\b\b\b\b\b\b\b\b\b");
550  printf("%05.2f%% Done", fr * 100);
551  fflush(stdout);
552  }
553  }
554  int nc = calculateChebyshevCoefficients(fvals, mNumberOfPoints[0], tmpCoef1D, prec);
555  for (int id0 = mNumberOfPoints[0]; id0--;) {
556  tmpCoef2D[id1 + id0 * mNumberOfPoints[1]] = tmpCoef1D[id0];
557  }
558  if (ncmax < nc) {
559  ncmax = nc; // max coefs to be kept in dim0 to guarantee needed precision
560  }
561  }
562  // once each 1d slice of given 2d slice is parametrized, parametrize the Cheb.coeffs
563  for (int id0 = mNumberOfPoints[0]; id0--;) {
564  calculateChebyshevCoefficients(tmpCoef2D + id0 * mNumberOfPoints[1], mNumberOfPoints[1], tmpCoef1D, -1);
565  for (int id1 = mNumberOfPoints[1]; id1--;) {
566  tmpCoef3D[id2 + mNumberOfPoints[2] * (id1 + id0 * mNumberOfPoints[1])] = tmpCoef1D[id1];
567  }
568  }
569  }
570  // now fit the last dimensions Cheb.coefs
571  for (int id0 = mNumberOfPoints[0]; id0--;) {
572  for (int id1 = mNumberOfPoints[1]; id1--;) {
573  calculateChebyshevCoefficients(tmpCoef3D + mNumberOfPoints[2] * (id1 + id0 * mNumberOfPoints[1]),
574  mNumberOfPoints[2], tmpCoef1D, -1);
575  for (int id2 = mNumberOfPoints[2]; id2--;) {
576  tmpCoef3D[id2 + mNumberOfPoints[2] * (id1 + id0 * mNumberOfPoints[1])] = tmpCoef1D[id2]; // store on place
577  }
578  }
579  }

773 void Chebyshev3D::loadData(FILE* stream)
774 {

786 
787  Chebyshev3DCalc::readLine(buffs, stream); // N output dimensions
788  mOutputArrayDimension = buffs.Atoi();
789  if (mOutputArrayDimension < 1) {

795  Chebyshev3DCalc::readLine(buffs, stream); // Interpolation abs. precision

799  }
