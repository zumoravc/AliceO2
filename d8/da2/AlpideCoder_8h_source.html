107  template <class T>
108  static int decodeChip(ChipPixelData& chipData, T& buffer)
109  {
110  // read record for single non-empty chip, updating on change module and cycle.
111  // return number of records filled (>0), EOFFlag or Error
112  //
113  uint8_t dataC = 0, timestamp = 0;
114  uint16_t dataS = 0, region = 0;
115  //
116  int nRightCHits = 0; // counter for the hits in the right column of the current double column
117  std::uint16_t rightColHits[NRows]; // buffer for the accumulation of hits in the right column
118  std::uint16_t colDPrev = 0xffff; // previously processed double column (to dected change of the double column)
119 
120  uint32_t expectInp = ExpectChipHeader | ExpectChipEmpty; // data must always start with chip header or chip empty flag
121 
122  chipData.clear();
123 
124  while (buffer.next(dataC)) {
125  //
126  // ---------- chip info ?
127  uint8_t dataCM = dataC & (~MaskChipID);
128  //
129  if ((expectInp & ExpectChipEmpty) && dataCM == CHIPEMPTY) { // empty chip was expected
130  chipData.setChipID(dataC & MaskChipID); // here we set the chip ID within the module
131  if (!buffer.next(timestamp)) {
132  return unexpectedEOF("CHIP_EMPTY:Timestamp");
133  }
134  expectInp = ExpectChipHeader | ExpectChipEmpty;
135  continue;
136  }
137 
138  if ((expectInp & ExpectChipHeader) && dataCM == CHIPHEADER) { // chip header was expected
139  chipData.setChipID(dataC & MaskChipID); // here we set the chip ID within the module
140  if (!buffer.next(timestamp)) {
141  return unexpectedEOF("CHIP_HEADER");
142  }
143  expectInp = ExpectRegion; // now expect region info
144  continue;
145  }
146 
147  // region info ?
148  if ((expectInp & ExpectRegion) && (dataC & REGION) == REGION) { // chip header was seen, or hit data read
149  region = dataC & MaskRegion;
150  expectInp = ExpectData;
151  continue;
152  }
153 
154  if ((expectInp & ExpectChipTrailer) && dataCM == CHIPTRAILER) { // chip trailer was expected
155  expectInp = ExpectChipHeader | ExpectChipEmpty;
156  chipData.setROFlags(dataC & MaskROFlags);
157  // in case there are entries in the "right" columns buffer, add them to the container
158  if (nRightCHits) {
159  colDPrev++;
160  for (int ihr = 0; ihr < nRightCHits; ihr++) {
161  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
162  }
163  }
164  break;
165  }
166 
167  // hit info ?
168  if ((expectInp & ExpectData)) {
169  if (isData(dataC)) { // region header was seen, expect data
170  // note that here we are checking on the byte rather than the short, need complete to ushort
171  dataS = dataC << 8;
172  if (!buffer.next(dataC)) {
173  return unexpectedEOF("CHIPDATA");
174  }
175  dataS |= dataC;
176  // we are decoding the pixel addres, if this is a DATALONG, we will fetch the mask later
177  uint16_t dColID = (dataS & MaskEncoder) >> 10;
178  uint16_t pixID = dataS & MaskPixID;
179 
180  // convert data to usual row/pixel format
181  uint16_t row = pixID >> 1;
182  // abs id of left column in double column
183  uint16_t colD = (region * NDColInReg + dColID) << 1; // TODO consider <<4 instead of *NDColInReg?
184 
185  // if we start new double column, transfer the hits accumulated in the right column buffer of prev. double column
186  if (colD != colDPrev) {
187  colDPrev++;
188  for (int ihr = 0; ihr < nRightCHits; ihr++) {
189  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
190  }
191  colDPrev = colD;
192  nRightCHits = 0; // reset the buffer
193  }
194 
195  bool rightC = (row & 0x1) ? !(pixID & 0x1) : (pixID & 0x1); // true for right column / lalse for left
196 
197  // we want to have hits sorted in column/row, so the hits in right column of given double column
198  // are first collected in the temporary buffer
199  // real columnt id is col = colD + 1;
200  if (rightC) {
201  rightColHits[nRightCHits++] = row; // col = colD+1
202  } else {
203  chipData.getData().emplace_back(row, colD); // col = colD, left column hits are added directly to the container
204  }
205 
206  if ((dataS & (~MaskDColID)) == DATALONG) { // multiple hits ?
207  uint8_t hitsPattern = 0;
208  if (!buffer.next(hitsPattern)) {
209  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
210  }
211  for (int ip = 0; ip < HitMapSize; ip++) {
212  if (hitsPattern & (0x1 << ip)) {
213  uint16_t addr = pixID + ip + 1, rowE = addr >> 1;
214  rightC = ((rowE & 0x1) ? !(addr & 0x1) : (addr & 0x1)); // true for right column / lalse for left
215  // the real columnt is int colE = colD + rightC;
216  if (rightC) { // same as above
217  rightColHits[nRightCHits++] = rowE;
218  } else {
219  chipData.getData().emplace_back(rowE, colD + rightC); // left column hits are added directly to the container
220  }
221  }
222  }
223  }
224  } else {
225  LOG(ERROR) << "Expected DataShort or DataLong mask, got : " << dataS;
226  return Error;
227  }
228  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
229  continue; // end of DATA(SHORT or LONG) processing
230  }
231 
232  if (!dataC) {
233  buffer.clear(); // 0 padding reached (end of the cable data), no point in continuing
234  break;
235  }
236  std::stringstream stream;
237  stream << "Unknown word 0x" << std::hex << int(dataC) << " [mode = 0x" << int(expectInp) << "]";
238  return unexpectedEOF(stream.str().c_str()); // error
239  }
240 
241  return chipData.getData().size();
242  }

282  static uint16_t makeChipHeader(short chipID, short bc)
283  {
284  uint16_t v = CHIPHEADER | (MaskChipID & chipID);
285  v = (v << 8) | bc2TimeStamp(bc);

297  static uint16_t makeChipEmpty(short chipID, short bc)
298  {
299  uint16_t v = CHIPEMPTY | (MaskChipID & chipID);

305  static uint8_t makeRegion(short reg)
306  {
307  uint8_t v = REGION | (reg & MaskRegion);

312  static uint16_t makeDataShort(short encoder, short address)
313  {
314  uint16_t v = DATASHORT | (MaskEncoder & (encoder << 10)) | (address & MaskPixID);
315  return v;
316  }

318  // packs the address for data long

320  {
321  uint16_t v = DATALONG | (MaskEncoder & (encoder << 10)) | (address & MaskPixID);
