267 GPUdi() int TPCFastSpaceChargeCorrection::getCorrection(int slice, int row, float u, float v, float& dx, float& du, float& dv) const
268 {
269  const SplineType& spline = getSpline(slice, row);
270  const float* splineData = getSplineData(slice, row);
271  float su = 0, sv = 0;
272  mGeo.convUVtoScaledUV(slice, row, u, v, su, sv);
273  su *= spline.getGridU1().getUmax();
274  sv *= spline.getGridU2().getUmax();
275  float dxuv[3];
276  spline.interpolateU(splineData, su, sv, dxuv);
277  dx = dxuv[0];
278  du = dxuv[1];
279  dv = dxuv[2];
280  return 0;
281 }

283 GPUdi() void TPCFastSpaceChargeCorrection::getCorrectionInvCorrectedX(
284  int slice, int row, float cu, float cv, float& x) const
285 {
286  //const RowInfo& rowInfo = getRowInfo(row);
287  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
288  const Spline2D<float, 1, 0>& spline = reinterpret_cast<const Spline2D<float, 1, 0>&>(getSpline(slice, row));
289  const float* splineData = getSplineData(slice, row, 1);
290  float gridU = (cu - sliceRowInfo.CorrU0) * sliceRowInfo.scaleCorrUtoGrid;
291  float gridV = cv * sliceRowInfo.scaleCorrVtoGrid;
292  float dx = 0;
293  spline.interpolateU(splineData, gridU, gridV, &dx);
294  x = mGeo.getRowInfo(row).x + dx;
295 }

297 GPUdi() void TPCFastSpaceChargeCorrection::getCorrectionInvUV(
298  int slice, int row, float corrU, float corrV, float& nomU, float& nomV) const
299 {
300  //const RowInfo& rowInfo = getRowInfo(row);
301  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
302  const Spline2D<float, 2, 0>& spline = reinterpret_cast<const Spline2D<float, 2, 0>&>(getSpline(slice, row));
303  const float* splineData = getSplineData(slice, row, 2);
304  float gridU = (corrU - sliceRowInfo.CorrU0) * sliceRowInfo.scaleCorrUtoGrid;
305  float gridV = corrV * sliceRowInfo.scaleCorrVtoGrid;
306  float duv[2];
307  spline.interpolateU(splineData, gridU, gridV, duv);
308  nomU = corrU - duv[0];
309  nomV = corrV - duv[1];
310 }

312 GPUdi() float TPCFastSpaceChargeCorrection::getMaxDriftLength(int slice, int row, float pad) const
313 {
314  const RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
315  const float* c = area.maxDriftLengthCheb;
316  float x = -1.f + 2.f * pad / mGeo.getRowInfo(row).maxPad;
317  float y = c[0] + c[1] * x;
318  float f0 = 1.;
319  float f1 = x;
320  x *= 2.f;
321  for (int i = 2; i < 5; i++) {
322  double f = x * f1 - f0;
323  y += c[i] * f;
324  f0 = f1;
325  f1 = f;
326  }
327  return y;
328 }
