68 float CalibTimeSlewingParamTOF::evalTimeSlewing(int channel, float tot) const
69 {
70  int sector = channel / NCHANNELXSECTOR;
71  channel = channel % NCHANNELXSECTOR;
72 
73  if (sector >= NSECTORS)
74  return 0.; // something went wrong!
75 
76  int n = mChannelStart[sector][channel];
77  if (n < 0)
78  return 0.;
79 
80  int nstop = (*(mTimeSlewing[sector])).size();
81  if (channel < NCHANNELXSECTOR - 1)
82  nstop = mChannelStart[sector][channel + 1];
83 
84  if (n >= nstop)
85  return 0.; // something went wrong!
86 
87  while (n < nstop && tot > (*(mTimeSlewing[sector]))[n].first)
88  n++;
89  n--;
90 
91  if (n < 0) { // tot is lower than the first available value
92  return 0;
93  }
94 
95  if (n == nstop - 1)
96  return (*(mTimeSlewing[sector]))[n].second; // use the last value stored for that channel
97 
98  float w1 = tot - (*(mTimeSlewing[sector]))[n].first;
99  float w2 = (*(mTimeSlewing[sector]))[n + 1].first - tot;
100 
101  return ((*(mTimeSlewing[sector]))[n].second * w2 + (*(mTimeSlewing[sector]))[n + 1].second * w1) / ((*(mTimeSlewing[sector]))[n + 1].first - (*(mTimeSlewing[sector]))[n].first);
102 }

105 void CalibTimeSlewingParamTOF::addTimeSlewingInfo(int channel, float tot, float time)
106 {
107  // WE ARE ASSUMING THAT:
108  // channels have to be filled in increasing order (within the sector)
109  // tots have to be filled in increasing order (within the channel)
110  int sector = channel / NCHANNELXSECTOR;
111  channel = channel % NCHANNELXSECTOR;
112 
113  // printf("DBG: addTimeSlewinginfo sec=%i\n",sector);
114 
115  if (sector >= NSECTORS)
116  return; // something went wrong!
117 
118  int currentch = channel;
119  while (mChannelStart[sector][currentch] == -1 && currentch > -1) {
120  // printf("DBG: fill channel %i\n",currentch);
121  // set also all the previous ones which were not filled
122  mChannelStart[sector][currentch] = (*(mTimeSlewing[sector])).size();
123  currentch--;
124  }
125  // printf("DBG: emplace back (%f,%f)\n",tot,time);
126  (*(mTimeSlewing[sector])).emplace_back(tot, time);
127 }
