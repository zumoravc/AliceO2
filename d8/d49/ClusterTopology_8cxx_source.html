51 unsigned int ClusterTopology::hashFunction(const void* key, int len)
52 {
53  //
54  // Developed from https://github.com/rurban/smhasher , function MurMur2
55  //
56  // 'm' and 'r' are mixing constants generated offline.
57  const unsigned int m = 0x5bd1e995;
58  const int r = 24;
59  // Initialize the hash
60  unsigned int h = len ^ 0xdeadbeef;
61  // Mix 4 bytes at a time into the hash
62  const unsigned char* data = (const unsigned char*)key;
63  // int recIndex=0;
64  while (len >= 4) {
65  unsigned int k = *(unsigned int*)data;
66  k *= m;
67  k ^= k >> r;
68  k *= m;
69  h *= m;
70  h ^= k;
71  data += 4;
72  len -= 4;
73  }
74  // Handle the last few bytes of the input array
75  switch (len) {
76  case 3:
77  h ^= data[2] << 16;
78  case 2:
79  h ^= data[1] << 8;
80  case 1:
81  h ^= data[0];
82  h *= m;
83  };
84  // Do a few final mixes of the hash to ensure the last few
85  // bytes are well-incorporated.
86  h ^= h >> 13;
87  h *= m;
88  h ^= h >> 15;
89  return h;
90 }

92 unsigned long ClusterTopology::getCompleteHash(int nRow, int nCol,
93  const unsigned char patt[Cluster::kMaxPatternBytes])
94 {
95  unsigned char extended_pattern[ClusterPattern::kExtendedPatternBytes] = {0};
96  extended_pattern[0] = (unsigned char)nRow;
97  extended_pattern[1] = (unsigned char)nCol;
98  int nBits = nRow * nCol;
99  int nBytes = nBits / 8;
100  if (nBits % 8 != 0)
101  nBytes++;
102  memcpy(&extended_pattern[2], patt, nBytes);
103 
104  unsigned long partialHash = (unsigned long)hashFunction(extended_pattern, nBytes);
105  // The first four bytes are directly taken from partialHash
106  unsigned long completeHash = partialHash << 32;
107  // The last four bytes of the hash are the first 32 pixels of the topology.
108  // The bits reserved for the pattern that are not used are set to 0.
109  if (nBytes == 1) {
110  completeHash += ((((unsigned long)extended_pattern[2]) << 24));
111  } else if (nBytes == 2) {
112  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16));
113  } else if (nBytes == 3) {
114  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
115  (((unsigned long)extended_pattern[4]) << 8));
116  } else if (nBytes >= 4) {
117  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
118  (((unsigned long)extended_pattern[4]) << 8) + ((unsigned long)extended_pattern[5]));
119  } else {
120  LOG(ERROR) << "No fired pixels in small topology";
121  throw std::runtime_error("No fired pixels in small topology");
122  }
123  return completeHash;
124 }
