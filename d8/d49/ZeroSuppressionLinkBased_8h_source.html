67 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2>
68 struct Data {
69  static constexpr uint32_t ChannelsPerHalfWord = DataWordSizeBits / DataBitSizeT / 2;
70  static constexpr uint32_t DataBitSize = DataBitSizeT;
71  static constexpr uint32_t SignificantBits = SignificantBitsT;
72  static constexpr uint64_t BitMask = ((uint64_t(1) << DataBitSize) - 1);
73  static constexpr float FloatConversion = 1.f / float(1 << SignificantBits);
74 
75  uint64_t adcValues[2]{};
76 
78  void setADCValue(uint32_t pos, uint64_t value)
79  {
80  const uint32_t word = pos / ChannelsPerHalfWord;
81  const uint32_t posInWord = pos % ChannelsPerHalfWord;
82 
83  const uint64_t set = (value & BitMask) << (posInWord * DataBitSize);
84  const uint64_t mask = (0xFFFFFFFFFFFFFFFF ^ (BitMask << (posInWord * DataBitSize)));
85 
86  adcValues[word] &= mask;
87  adcValues[word] |= set;
88  }
89 
91  void setADCValueFloat(uint32_t pos, float value)
92  {
93  setADCValue(pos, uint64_t((value + 0.5f * FloatConversion) / FloatConversion));
94  }
95 
97  uint32_t getADCValue(uint32_t pos)
98  {
99  const uint32_t word = pos / ChannelsPerHalfWord;
100  const uint32_t posInWord = pos % ChannelsPerHalfWord;
101 
102  return (adcValues[word] >> (posInWord * DataBitSize)) & BitMask;
103  }
104 
106  float getADCValueFloat(uint32_t pos)
107  {
108  return float(getADCValue(pos)) * FloatConversion;
109  }
110 
112  void reset()
113  {
114  adcValues[0] = 0;
115  adcValues[1] = 0;
116  }
117 };

140 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2, bool HasHeaderT = true>
141 struct Container {
142  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
143 
144  ContainerT<DataBitSizeT, SignificantBitsT, HasHeaderT> cont;
145 
147  uint32_t getADCValue(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValue(word % ChannelsPerWord); }
148 
150  float getADCValueFloat(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValueFloat(word % ChannelsPerWord); }
151 
153  void setADCValue(uint32_t word, uint64_t value) { cont.data[word / ChannelsPerWord].setADCValue(word % ChannelsPerWord, value); }
154 
156  void setADCValueFloat(uint32_t word, float value) { cont.data[word / ChannelsPerWord].setADCValueFloat(word % ChannelsPerWord, value); }
157 
159  void reset()
160  {
161  for (int i = 0; i < cont.dataWords(); ++i) {
162  cont.data[i].reset();
163  }
164  }

168  {
