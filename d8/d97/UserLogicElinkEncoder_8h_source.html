83 template <typename CHARGESUM>
84 size_t ElinkEncoder<UserLogicFormat, CHARGESUM>::moveToBuffer(std::vector<uint64_t>& buffer, uint16_t gbtId)
85 {
86  if (mBuffer.empty()) {
87  return 0;
88  }
89 
90  int error{0}; // FIXME: what to do with error ?
91  uint16_t b9{0};
92 
93  b9 |= static_cast<uint64_t>(gbtId & 0x1F) << 9;
94  b9 |= static_cast<uint64_t>(mElinkId & 0x3F) << 3;
95  b9 |= static_cast<uint64_t>(error & 0x3);
96 
97  auto n = buffer.size();
98  impl::b10to64(mBuffer, buffer, b9);
99  clear();
100  return buffer.size() - n;
101 }
