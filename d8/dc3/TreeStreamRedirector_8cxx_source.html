128 void TreeStreamRedirector::FixLeafNameBug(TTree* tree)
129 {
130  // On the fly BUG FIX for name and titles of branches and Leave:
131  // renaming Leaves and changing title of branches to be the same as Branch name
132  // Explanation of FIX:
133  // In the friend tree Join logic it is assumed leaf names are used to find appropraiat primary/secondary keys
134  // For the standard queries however the branch names are used to identify data
135  // Hovewer in the Branch constructor it is not checked
136  // As a consequence - in case the name of the leave and and the name of branch is not the same + freind trees are
137  // sparse
138  // wrong joins ( unrelated pair of information) are used
139  // FIX:
140  // To be able to use friend trees with proper indexing (in case of sarse trees) branches and leaves has to be named
141  // consistently
142  // In this routine bnrach name is taken as a reference and branch title and leave name titles are renamed
143  // After the fix unit test code with pairs of sprse friend trees worked properly
144  // Side effects of fix:
145  //
146  if (!tree)
147  return;
148  TObjArray* brArray = tree->GetListOfBranches();
149  TObjArray* lArray = tree->GetListOfLeaves();
150  for (int i = 0; i < brArray->GetLast(); i++) {
151  TBranch* br = (TBranch*)brArray->At(i);
152  TString brTitle(br->GetTitle());
153  if (!brTitle.Contains(br->GetName())) {
154  int pos = brTitle.First("/");
155  TString leafName;
156  if (pos < brTitle.Length()) {
157  brTitle[pos] = 0;
158  leafName = TString::Format("%s", brTitle.Data()).Data();
159  TLeaf* leaf = (TLeaf*)lArray->FindObject(leafName);
160  if (leaf) {
161  leaf->SetName(br->GetName());
162  leaf->SetTitle(br->GetName());
163  br->SetTitle(TString::Format("%s/%s", br->GetName(), &(brTitle.Data()[pos + 1])).Data());
164  }
165  }
166  }
167  }
168 }
