135 GPUg() void trackleterKernel(
136  DeviceStoreVertexerHIP* store,
137  const TrackletingLayerOrder layerOrder,
138  const float phiCut)
139 {
140  const size_t nClustersMiddleLayer = store->getClusters()[1].size();
141  for (unsigned int currentClusterIndex = blockIdx.x * blockDim.x + threadIdx.x; currentClusterIndex < nClustersMiddleLayer; currentClusterIndex += blockDim.x * gridDim.x) {
142  if (currentClusterIndex < nClustersMiddleLayer) {
143  int storedTracklets{0};
144  const unsigned int stride{currentClusterIndex * store->getConfig().maxTrackletsPerCluster};
145  const Cluster& currentCluster = store->getClusters()[1][currentClusterIndex]; // assign-constructor may be a problem, check
146  const VertexerLayerName adjacentLayerIndex{layerOrder == TrackletingLayerOrder::fromInnermostToMiddleLayer ? VertexerLayerName::innermostLayer : VertexerLayerName::outerLayer};
147  const int4 selectedBinsRect{VertexerTraits::getBinsRect(currentCluster, static_cast<int>(adjacentLayerIndex), 0.f, 50.f, phiCut / 2)};
148  if (selectedBinsRect.x != 0 || selectedBinsRect.y != 0 || selectedBinsRect.z != 0 || selectedBinsRect.w != 0) {
149  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
150  if (phiBinsNum < 0) {
151  phiBinsNum += PhiBins;
152  }
153  const int nClustersAdjacentLayer = store->getClusters()[static_cast<int>(adjacentLayerIndex)].size();
154  for (size_t iPhiBin{static_cast<size_t>(selectedBinsRect.y)}, iPhiCount{0}; (int)iPhiCount < phiBinsNum; iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
155  const int firstBinIndex{index_table_utils::getBinIndex(selectedBinsRect.x, iPhiBin)};
156  const int firstRowClusterIndex{store->getIndexTable(adjacentLayerIndex)[firstBinIndex]};
157  const int maxRowClusterIndex{store->getIndexTable(adjacentLayerIndex)[firstBinIndex + selectedBinsRect.z - selectedBinsRect.x + 1]};
158  for (int iAdjacentCluster{firstRowClusterIndex}; iAdjacentCluster < maxRowClusterIndex && iAdjacentCluster < nClustersAdjacentLayer; ++iAdjacentCluster) {
159  const Cluster& adjacentCluster = store->getClusters()[static_cast<int>(adjacentLayerIndex)][iAdjacentCluster]; // assign-constructor may be a problem, check
160  if (gpu::GPUCommonMath::Abs(currentCluster.phiCoordinate - adjacentCluster.phiCoordinate) < phiCut) {
161  if (storedTracklets < store->getConfig().maxTrackletsPerCluster) {
162  if (layerOrder == TrackletingLayerOrder::fromInnermostToMiddleLayer) {
163  store->getDuplets01().emplace(stride + storedTracklets, iAdjacentCluster, currentClusterIndex, adjacentCluster, currentCluster);
164  } else {
165  store->getDuplets12().emplace(stride + storedTracklets, currentClusterIndex, iAdjacentCluster, currentCluster, adjacentCluster);
166  }
167  ++storedTracklets;
168  } else {
169  printf("debug: leaving tracklet behind\n");
170  }
171  }
172  }
173  }
174  }
175  store->getNFoundTracklets(layerOrder).emplace(currentClusterIndex, storedTracklets);
176  }
177  }
178 }

180 GPUg() void trackletSelectionKernel(
181  DeviceStoreVertexerHIP* store,
182  const unsigned char isInitRun = false,
183  const float tanLambdaCut = 0.025f,
184  const float phiCut = 0.002f)
185 {
186  const size_t nClustersMiddleLayer = store->getClusters()[1].size();
187  for (size_t currentClusterIndex = blockIdx.x * blockDim.x + threadIdx.x; currentClusterIndex < nClustersMiddleLayer; currentClusterIndex += blockDim.x * gridDim.x) {
188  const int stride{static_cast<int>(currentClusterIndex * store->getConfig().maxTrackletsPerCluster)};
189  int validTracklets{0};
190  for (int iTracklet12{0}; iTracklet12 < store->getNFoundTracklets(TrackletingLayerOrder::fromMiddleToOuterLayer)[currentClusterIndex]; ++iTracklet12) {
191  for (int iTracklet01{0}; iTracklet01 < store->getNFoundTracklets(TrackletingLayerOrder::fromInnermostToMiddleLayer)[currentClusterIndex] && validTracklets < store->getConfig().maxTrackletsPerCluster; ++iTracklet01) {
192  const float deltaTanLambda{gpu::GPUCommonMath::Abs(store->getDuplets01()[stride + iTracklet01].tanLambda - store->getDuplets12()[stride + iTracklet12].tanLambda)};
193  const float deltaPhi{gpu::GPUCommonMath::Abs(store->getDuplets01()[stride + iTracklet01].phiCoordinate - store->getDuplets12()[stride + iTracklet12].phiCoordinate)};
194  if (deltaTanLambda < tanLambdaCut && deltaPhi < phiCut && validTracklets != store->getConfig().maxTrackletsPerCluster) {
195  assert(store->getDuplets01()[stride + iTracklet01].secondClusterIndex == store->getDuplets12()[stride + iTracklet12].firstClusterIndex);
196  if (!isInitRun) {
197  store->getLines().emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, store->getDuplets01()[stride + iTracklet01], store->getClusters()[0].get(), store->getClusters()[1].get());
198 #ifdef _ALLOW_DEBUG_TREES_ITS_
199  store->getDupletIndices()[0].emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, stride + iTracklet01);
200  store->getDupletIndices()[1].emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, stride + iTracklet12);
201 #endif
202  }
203  ++validTracklets;
204  }
205  }
206  }
207  if (isInitRun) {
208  store->getNFoundLines().emplace(currentClusterIndex, validTracklets);
209  if (validTracklets >= store->getConfig().maxTrackletsPerCluster) {
210  printf("Warning: not enough space for tracklet selection, some lines will be left behind\n");
211  }
212  }
213  }
214  if (blockIdx.x * blockDim.x + threadIdx.x == 0) {
215  // first thread I want to write an empty line after last found, as adebug flag. Might delete later
216  store->getLines().emplace(store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1]);
217  }
218 }

220 GPUg() void computeCentroidsKernel(DeviceStoreVertexerHIP* store,
221  const float pairCut)
222 {
223  const unsigned int nLines = store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1];
224  const unsigned int maxIterations{nLines * (nLines - 1) / 2};
225  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; currentThreadIndex < maxIterations; currentThreadIndex += blockDim.x * gridDim.x) {
226  int iFirstLine = currentThreadIndex / nLines;
227  int iSecondLine = currentThreadIndex % nLines;
228  if (iSecondLine <= iFirstLine) {
229  iFirstLine = nLines - iFirstLine - 2;
230  iSecondLine = nLines - iSecondLine - 1;
231  }
232  if (Line::getDCA(store->getLines()[iFirstLine], store->getLines()[iSecondLine]) < pairCut) {
233  ClusterLinesHIP cluster{store->getLines()[iFirstLine], store->getLines()[iSecondLine]};
234  if (cluster.getVertex()[0] * cluster.getVertex()[0] + cluster.getVertex()[1] * cluster.getVertex()[1] < 1.98f * 1.98f) {
235  // printOnThread(0, "xCentr: %f, yCentr: %f \n", cluster.getVertex()[0], cluster.getVertex()[1]);
236  store->getXYCentroids().emplace(2 * currentThreadIndex, cluster.getVertex()[0]);
237  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, cluster.getVertex()[1]);
238  } else {
239  // writing some data anyway outside the histogram, they will not be put in the histogram, by construction.
240  store->getXYCentroids().emplace(2 * currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[0]);
241  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[1]);
242  }
243  } else {
244  // writing some data anyway outside the histogram, they will not be put in the histogram, by construction.
245  store->getXYCentroids().emplace(2 * currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[0]);
246  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[1]);
247  }
248  }
249 }

251 GPUg() void computeZCentroidsKernel(DeviceStoreVertexerHIP* store,
252  const float pairCut, const int binOpeningX, const int binOpeningY)
253 {
254  const unsigned int nLines = store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1];
255  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; currentThreadIndex < nLines; currentThreadIndex += blockDim.x * gridDim.x) {
256  // printOnThread(0, "Max X: %d Max Y %d \n", store->getTmpVertexPositionBins()[0].value, store->getTmpVertexPositionBins()[1].value);
257  if (store->getTmpVertexPositionBins()[0].value || store->getTmpVertexPositionBins()[1].value) {
258  float tmpX{store->getConfig().histConf.lowHistBoundariesXYZ[0] + store->getTmpVertexPositionBins()[0].key * store->getConfig().histConf.binSizeHistX + store->getConfig().histConf.binSizeHistX / 2};
259  int sumWX{store->getTmpVertexPositionBins()[0].value};
260  float wX{tmpX * store->getTmpVertexPositionBins()[0].value};
261  for (int iBin{gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[0].key - binOpeningX)}; iBin < gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[0].key + binOpeningX + 1, store->getConfig().histConf.nBinsXYZ[0] - 1); ++iBin) {
262  if (iBin != store->getTmpVertexPositionBins()[0].key) {
263  wX += (store->getConfig().histConf.lowHistBoundariesXYZ[0] + iBin * store->getConfig().histConf.binSizeHistX + store->getConfig().histConf.binSizeHistX / 2) * store->getHistogramXYZ()[0].get()[iBin];
264  sumWX += store->getHistogramXYZ()[0].get()[iBin];
265  }
266  }
267  float tmpY{store->getConfig().histConf.lowHistBoundariesXYZ[1] + store->getTmpVertexPositionBins()[1].key * store->getConfig().histConf.binSizeHistY + store->getConfig().histConf.binSizeHistY / 2};
268  int sumWY{store->getTmpVertexPositionBins()[1].value};
269  float wY{tmpY * store->getTmpVertexPositionBins()[1].value};
270  for (int iBin{gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[1].key - binOpeningY)}; iBin < gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[1].key + binOpeningY + 1, store->getConfig().histConf.nBinsXYZ[1] - 1); ++iBin) {
271  if (iBin != store->getTmpVertexPositionBins()[1].key) {
272  wY += (store->getConfig().histConf.lowHistBoundariesXYZ[1] + iBin * store->getConfig().histConf.binSizeHistY + store->getConfig().histConf.binSizeHistY / 2) * store->getHistogramXYZ()[1].get()[iBin];
273  sumWY += store->getHistogramXYZ()[1].get()[iBin];
274  }
275  }
276  store->getBeamPosition().emplace(0, wX / sumWX);
277  store->getBeamPosition().emplace(1, wY / sumWY);
278  float fakeBeamPoint1[3] = {store->getBeamPosition()[0], store->getBeamPosition()[1], -1}; // get two points laying at different z, to create line object
279  float fakeBeamPoint2[3] = {store->getBeamPosition()[0], store->getBeamPosition()[1], 1};
280  Line pseudoBeam{fakeBeamPoint1, fakeBeamPoint2};
281  if (Line::getDCA(store->getLines()[currentThreadIndex], pseudoBeam) < pairCut) {
282  ClusterLinesHIP cluster{store->getLines()[currentThreadIndex], pseudoBeam};
283  store->getZCentroids().emplace(currentThreadIndex, cluster.getVertex()[2]);
284  } else {
285  store->getZCentroids().emplace(currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[2]);
286  }
287  }
288  }
289 }

291 GPUg() void computeVertexKernel(DeviceStoreVertexerHIP* store, const int vertIndex, const int minContributors, const int binOpeningZ)
292 {
293  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; (int)currentThreadIndex < binOpeningZ; currentThreadIndex += blockDim.x * gridDim.x) {
294  if (currentThreadIndex == 0) {
295  if (store->getTmpVertexPositionBins()[2].value > 1 && (store->getTmpVertexPositionBins()[0].value || store->getTmpVertexPositionBins()[1].value)) {
296  float z{store->getConfig().histConf.lowHistBoundariesXYZ[2] + store->getTmpVertexPositionBins()[2].key * store->getConfig().histConf.binSizeHistZ + store->getConfig().histConf.binSizeHistZ / 2};
297  float ex{0.f};
298  float ey{0.f};
299  float ez{0.f};
300  int sumWZ{store->getTmpVertexPositionBins()[2].value};
301  float wZ{z * store->getTmpVertexPositionBins()[2].value};
302  for (int iBin{gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[2].key - binOpeningZ)}; iBin < gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[2].key + binOpeningZ + 1, store->getConfig().histConf.nBinsXYZ[2] - 1); ++iBin) {
303  if (iBin != store->getTmpVertexPositionBins()[2].key) {
304  wZ += (store->getConfig().histConf.lowHistBoundariesXYZ[2] + iBin * store->getConfig().histConf.binSizeHistZ + store->getConfig().histConf.binSizeHistZ / 2) * store->getHistogramXYZ()[2].get()[iBin];
305  sumWZ += store->getHistogramXYZ()[2].get()[iBin];
306  }
307  store->getHistogramXYZ()[2].get()[iBin] = 0;
308  }
309  if (sumWZ > minContributors || vertIndex == 0) {
310  store->getVertices().emplace(vertIndex, store->getBeamPosition()[0], store->getBeamPosition()[1], wZ / sumWZ, ex, ey, ez, sumWZ);
311  } else {
312  store->getVertices().emplace(vertIndex);
313  }
314  } else {
315  store->getVertices().emplace(vertIndex);
316  }
317  }
318  }

401 void VertexerTraitsHIP::computeVertices()
402 {
403  if (!mClusters[1].size()) {
404  std::cout << "\t\tno clusters on layer 1. Returning.\n";
405  return;
406  }
407  const dim3 threadsPerBlock{GPU::Utils::HostHIP::getBlockSize(mClusters[1].capacity())};
408  const dim3 blocksGrid{GPU::Utils::HostHIP::getBlocksGrid(threadsPerBlock, mClusters[1].capacity())};
409  size_t bufferSize = mStoreVertexerGPU.getConfig().tmpCUBBufferSize * sizeof(int);
410  int nLines = mStoreVertexerGPU.getNExclusiveFoundLines().getElementFromDevice(mClusters[1].size() - 1) + mStoreVertexerGPU.getNFoundLines().getElementFromDevice(mClusters[1].size() - 1);
411  int nCentroids{static_cast<int>(nLines * (nLines - 1) / 2)};
412  int* histogramXY[2] = {mStoreVertexerGPU.getHistogramXYZ()[0].get(), mStoreVertexerGPU.getHistogramXYZ()[1].get()};
413  float tmpArrayLow[2] = {mStoreVertexerGPU.getConfig().histConf.lowHistBoundariesXYZ[0], mStoreVertexerGPU.getConfig().histConf.lowHistBoundariesXYZ[1]};
414  float tmpArrayHigh[2] = {mStoreVertexerGPU.getConfig().histConf.highHistBoundariesXYZ[0], mStoreVertexerGPU.getConfig().histConf.highHistBoundariesXYZ[1]};
415  hipLaunchKernelGGL((GPU::computeCentroidsKernel), dim3(blocksGrid), dim3(threadsPerBlock), 0, 0, getDeviceContextPtr(),
416  mVrtParams.histPairCut);
417 
418  (void)hipcub::DeviceHistogram::MultiHistogramEven<2, 2>(reinterpret_cast<void*>(mStoreVertexerGPU.getCUBTmpBuffer().get()), // d_temp_storage
419  bufferSize, // temp_storage_bytes
420  mStoreVertexerGPU.getXYCentroids().get(), // d_samples
421  histogramXY, // d_histogram
422  mStoreVertexerGPU.getConfig().histConf.nBinsXYZ, // num_levels
423  tmpArrayLow, // lower_level
424  tmpArrayHigh, // fupper_level
425  nCentroids); // num_row_pixels
426 
427  (void)hipcub::DeviceReduce::ArgMax(reinterpret_cast<void*>(mStoreVertexerGPU.getCUBTmpBuffer().get()),
