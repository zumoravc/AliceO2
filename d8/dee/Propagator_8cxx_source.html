54 bool Propagator::PropagateToXBxByBz(o2::track::TrackParCov& track, float xToGo, float mass, float maxSnp, float maxStep,
55  int matCorr, o2::track::TrackLTIntegral* tofInfo, int signCorr) const
56 {
57  //----------------------------------------------------------------
58  //
59  // Propagates the track to the plane X=xk (cm)
60  // taking into account all the three components of the magnetic field
61  // and correcting for the crossed material.
62  //
63  // mass - mass used in propagation - used for energy loss correction (if <0 then q=2)
64  // maxStep - maximal step for propagation
65  // tofInfo - optional container for track length and PID-dependent TOF integration
66  //
67  // matCorr - material correction type, it is up to the user to make sure the pointer is attached (if LUT is requested)
68  //----------------------------------------------------------------
69  const float Epsilon = 0.00001;
70  auto dx = xToGo - track.getX();
71  int dir = dx > 0.f ? 1 : -1;
72  if (!signCorr) {
73  signCorr = -dir; // sign of eloss correction is not imposed
74  }
75 
76  std::array<float, 3> b;
77  while (std::abs(dx) > Epsilon) {
78  auto step = std::min(std::abs(dx), maxStep);
79  if (dir < 0) {
80  step = -step;
81  }
82  auto x = track.getX() + step;
83  auto xyz0 = track.getXYZGlo();
84  mField->Field(xyz0, b.data());
85 
86  if (!track.propagateTo(x, b)) {
87  return false;
88  }
89  if (maxSnp > 0 && std::abs(track.getSnp()) >= maxSnp) {
90  return false;
91  }
92  if (matCorr != USEMatCorrNONE) {
93  auto xyz1 = track.getXYZGlo();
94  auto mb = getMatBudget(matCorr, xyz0, xyz1);
95  if (!track.correctForMaterial(mb.meanX2X0, ((signCorr < 0) ? -mb.length : mb.length) * mb.meanRho, mass)) {
96  return false;
97  }
98 
99  if (tofInfo) {
100  tofInfo->addStep(mb.length, track); // fill L,ToF info using already calculated step length
101  tofInfo->addX2X0(mb.meanX2X0);
102  }
103  } else if (tofInfo) { // if tofInfo filling was requested w/o material correction, we need to calculate the step lenght
104  auto xyz1 = track.getXYZGlo();
105  Vector3D<float> stepV(xyz1.X() - xyz0.X(), xyz1.Y() - xyz0.Y(), xyz1.Z() - xyz0.Z());
106  tofInfo->addStep(stepV.R(), track);
107  }
108  dx = xToGo - track.getX();
109  }
110  return true;
111 }

173 bool Propagator::propagateToDCA(const Point3D<float>& vtx, o2::track::TrackParCov& track, float bZ,
174  float mass, float maxStep, int matCorr,
175  o2::track::TrackLTIntegral* tofInfo, int signCorr, float maxD) const
176 {
177  // propagate track to DCA to the vertex
178  float sn, cs, alp = track.getAlpha();
179  o2::utils::sincosf(alp, sn, cs);
180  float x = track.getX(), y = track.getY(), snp = track.getSnp(), csp = std::sqrt((1.f - snp) * (1.f + snp));
181  float xv = vtx.X() * cs + vtx.Y() * sn, yv = -vtx.X() * sn + vtx.Y() * cs, zv = vtx.Z();
182  x -= xv;
183  y -= yv;
184  //Estimate the impact parameter neglecting the track curvature
185  Double_t d = std::abs(x * snp - y * csp);
186  if (d > maxD) {
187  return false;
188  }
189  float crv = track.getCurvature(bZ);
190  float tgfv = -(crv * x - snp) / (crv * y + csp);
191  sn = tgfv / std::sqrt(1.f + tgfv * tgfv);
192  cs = std::sqrt((1. - sn) * (1. + sn));
193  cs = (std::abs(tgfv) > o2::constants::math::Almost0) ? sn / tgfv : o2::constants::math::Almost1;
194 
195  x = xv * cs + yv * sn;
196  yv = -xv * sn + yv * cs;
197  xv = x;
198 
199  o2::track::TrackParCov tmpT(track); // operate on the copy to recover after the failure
200  alp += std::asin(sn);
201  if (!tmpT.rotate(alp) || !propagateToX(tmpT, xv, bZ, mass, 0.85, maxStep, matCorr, tofInfo, signCorr)) {
202  LOG(ERROR) << "failed to propagate to alpha=" << alp << " X=" << xv << " for vertex "
203  << vtx.X() << ' ' << vtx.Y() << ' ' << vtx.Z() << " | Track is: ";
204  tmpT.print();
205  return false;
206  }
207  track = tmpT;
208  return true;
209 }

231 {
233 
234  if (TGeoGlobalMagField::Instance()->IsLocked()) {
235  if (TGeoGlobalMagField::Instance()->GetField()->TestBit(o2::field::MagneticField::kOverrideGRP)) {
