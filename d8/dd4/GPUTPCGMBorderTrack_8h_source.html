55  GPUd() static bool CheckChi2(float x1, float y1, float cx1, float cxy1, float cy1, float x2, float y2, float cx2, float cxy2, float cy2, float chi2cut)
56  {
57  //* Calculate Chi2/ndf deviation
58  float dx = x1 - x2;
59  float dy = y1 - y2;
60  float cx = cx1 + cx2;
61  float cxy = cxy1 + cxy2;
62  float cy = cy1 + cy2;
63  float det = cx * cy - cxy * cxy;
64  // printf("Res %f Det %f Cut %f %s - ", ( cy*dx - (cxy+cxy)*dy )*dx + cx*dy*dy, det, (det + det) * chi2cut, (CAMath::Abs(( cy*dx - (cxy+cxy)*dy )*dx + cx*dy*dy) < CAMath::Abs((det+det)*chi2cut)) ? "OK" : "Fail");
65  return (CAMath::Abs((cy * dx - (cxy + cxy) * dy) * dx + cx * dy * dy) < CAMath::Abs((det + det) * chi2cut));
66  }

80  GPUd() bool CheckChi2QPt(const GPUTPCGMBorderTrack& t, float chi2cut) const
81  {
82  float d = mP[4] - t.mP[4];
83  if (CAMath::Abs(d) > 0.3f && CAMath::Abs(d) > 0.5f * CAMath::Min(CAMath::Abs(mP[4]), CAMath::Abs(t.mP[4]))) {
84  return false; // Crude cut to avoid some bogus matches, TODO: recheck

93  GPUd() void LimitCov()
94  {
95  // TODO: Why are Cov entries so large?
96  for (int i = 0; i < 2; i++) {
97  if (mC[i] > 5.f) {
98  mC[i] = 5.f;
99  }
100  }

102  if (mC[i] > 0.5f) {
103  mC[i] = 0.5f;
104  }
105  }
106  float maxCov4 = CAMath::Max(0.5f, mP[4] * mP[4] * 0.25f);
107  if (mC[4] > maxCov4) {
108  mC[4] = maxCov4;
109  }
110  for (int i = 0; i < 2; i++) {
111  if (mD[i] > 0.5f) {
112  mD[i] = 0.5f;
113  }
114  }
115  if (mD[0] * mD[0] > mC[0] * mC[2]) {
116  mD[0] = 0.f;
117  }
118  if (mD[1] * mD[1] > mC[1] * mC[3]) {
119  mD[1] = 0.f;
