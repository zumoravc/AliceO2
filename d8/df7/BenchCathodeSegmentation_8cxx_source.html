24 std::vector<TestPoint> generateUniformTestPoints(int n, double xmin, double ymin, double xmax, double ymax)
25 {
26  std::random_device rd;
27  std::mt19937 mt(rd());
28  std::vector<TestPoint> testPoints;
29 
30  testPoints.resize(n);
31  std::uniform_real_distribution<double> distX{xmin, xmax};
32  std::uniform_real_distribution<double> distY{ymin, ymax};
33  std::generate(testPoints.begin(), testPoints.end(), [&distX, &distY, &mt] {
34  return TestPoint{distX(mt), distY(mt)};
35  });
36 
37  return testPoints;
38 }
