116 GPUd() int GPUTPCGMPropagator::RotateToAlpha(float newAlpha)
117 {
118  //
119  // Rotate the track coordinate system in XY to the angle newAlpha
120  // return value is error code (0==no error)
121  //
122 
123  float newCosAlpha = CAMath::Cos(newAlpha);
124  float newSinAlpha = CAMath::Sin(newAlpha);
125 
126  float cc = newCosAlpha * mCosAlpha + newSinAlpha * mSinAlpha; // cos(newAlpha - mAlpha);
127  float ss = newSinAlpha * mCosAlpha - newCosAlpha * mSinAlpha; //sin(newAlpha - mAlpha);
128 
129  GPUTPCGMPhysicalTrackModel t0 = mT0;
130 
131  float x0 = mT0.X();
132  float y0 = mT0.Y();
133  float px0 = mT0.Px();
134  float py0 = mT0.Py();
135  // float pt0 = mT0.GetPt();
136 
137  if (CAMath::Abs(mT->GetSinPhi()) >= mMaxSinPhi || CAMath::Abs(px0) < (1 - mMaxSinPhi)) {
138  return -1;
139  }
140 
141  // rotate t0 track
142  float px1 = px0 * cc + py0 * ss;
143  float py1 = -px0 * ss + py0 * cc;
144 
145  {
146  t0.X() = x0 * cc + y0 * ss;
147  t0.Y() = -x0 * ss + y0 * cc;
148  t0.Px() = px1;
149  t0.Py() = py1;
150  t0.UpdateValues();
151  }
152 
153  if (CAMath::Abs(py1) > mMaxSinPhi * mT0.GetPt() || CAMath::Abs(px1) < (1 - mMaxSinPhi)) {
154  return -1;
155  }
156 
157  // calculate X of rotated track:
158  float trackX = x0 * cc + ss * mT->Y();
159 
160  // transport t0 to trackX
161  float B[3];
162  GetBxByBz(newAlpha, t0.X(), t0.Y(), t0.Z(), B);
163  float dLp = 0;
164  if (t0.PropagateToXBxByBz(trackX, B[0], B[1], B[2], dLp)) {
165  return -1;
166  }
167 
168  if (CAMath::Abs(t0.SinPhi()) >= mMaxSinPhi) {
169  return -1;
170  }
171 
172  // now t0 is rotated and propagated, all checks are passed
173 
174  // Rotate track using mT0 for linearisation. After rotation X is not fixed, but has a covariance
175 
176  // Y Z Sin DzDs q/p
177  // Jacobian J0 = { { j0, 0, 0, 0, 0 }, // Y
178  // { 0, 1, 0, 0, 0 }, // Z
179  // { 0, 0, j1, 0, 0 }, // SinPhi
180  // { 0, 0, 0, 1, 0 }, // DzDs
181  // { 0, 0, 0, 0, 1 }, // q/p
182  // { j2, 0, 0, 0, 0 } }// X (rotated )
183 
184  float j0 = cc;
185  float j1 = px1 / px0;
186  float j2 = ss;
187  // float dy = mT->Y() - y0;
188  // float ds = mT->SinPhi() - mT0.SinPhi();
189 
190  mT->X() = trackX; // == x0*cc + ss*mT->Y() == t0.X() + j0*dy;
191  mT->Y() = -x0 * ss + cc * mT->Y(); //== t0.Y() + j0*dy;
192  // mT->SinPhi() = py1/pt0 + j1*ds; // == t0.SinPhi() + j1*ds; // use py1, since t0.SinPhi can have different sign
193  mT->SinPhi() = -CAMath::Sqrt(1.f - mT->SinPhi() * mT->SinPhi()) * ss + mT->SinPhi() * cc;
194 
195  // Rotate cov. matrix Cr = J0 x C x J0T. Cr has one more row+column for X:
196  float* c = mT->Cov();
197 
198  float c15 = c[0] * j0 * j2;
199  float c16 = c[1] * j2;
200  float c17 = c[3] * j1 * j2;
201  float c18 = c[6] * j2;
202  float c19 = c[10] * j2;
203  float c20 = c[0] * j2 * j2;
204 
205  c[0] *= j0 * j0;
206  c[3] *= j0;
207  c[10] *= j0;
208 
209  c[3] *= j1;
210  c[5] *= j1 * j1;
211  c[12] *= j1;
212 
213  if (!mFitInProjections && mT->NDF() > 0) {
214  c[1] *= j0;
215  c[6] *= j0;
216  c[4] *= j1;
217  c[8] *= j1;
218  }
219 
220  if (t0.SetDirectionAlongX()) { // change direction if Px < 0
221  mT->SinPhi() = -mT->SinPhi();
222  mT->DzDs() = -mT->DzDs();
223  mT->QPt() = -mT->QPt();
224  c[3] = -c[3]; // covariances with SinPhi
225  c[4] = -c[4];
226  c17 = -c17;
227  c[6] = -c[6]; // covariances with DzDs
228  c[7] = -c[7];
229  c18 = -c18;
230  c[10] = -c[10]; // covariances with QPt
231  c[11] = -c[11];
232  c19 = -c19;
233  }
234 
235  // Now fix the X coordinate: so to say, transport track T to fixed X = mT->X().
236  // only covariance changes. Use rotated and transported t0 for linearisation
237  float j3 = -t0.Py() / t0.Px();
238  float j4 = -t0.Pz() / t0.Px();
239  float j5 = t0.QPt() * B[2];
240 
241  // Y Z Sin DzDs q/p X
242  // Jacobian J1 = { { 1, 0, 0, 0, 0, j3 }, // Y
243  // { 0, 1, 0, 0, 0, j4 }, // Z
244  // { 0, 0, 1, 0, 0, j5 }, // SinPhi
245  // { 0, 0, 0, 1, 0, 0 }, // DzDs
246  // { 0, 0, 0, 0, 1, 0 } }; // q/p
247 
248  float h15 = c15 + c20 * j3;
249  float h16 = c16 + c20 * j4;
250  float h17 = c17 + c20 * j5;
251 
252  c[0] += j3 * (c15 + h15);
253 
254  c[2] += j4 * (c16 + h16);
255 
256  c[3] += c17 * j3 + h15 * j5;
257  c[5] += j5 * (c17 + h17);
258 
259  c[7] += c18 * j4;
260  // c[ 9] = c[ 9];
261 
262  c[10] += c19 * j3;
263  c[12] += c19 * j5;
264  // c[14] = c[14];
265 
266  if (!mFitInProjections && mT->NDF() > 0) {
267  c[1] += c16 * j3 + h15 * j4;
268  c[4] += c17 * j4 + h16 * j5;
269  c[6] += c18 * j3;
270  c[8] += c18 * j5;
271  c[11] += c19 * j4;
272  // c[13] = c[13];
273  }
274 
275  mAlpha = newAlpha;
276  mCosAlpha = newCosAlpha;
277  mSinAlpha = newSinAlpha;
278  mT0 = t0;
279 
280  return 0;
281 }

283 GPUd() int GPUTPCGMPropagator::PropagateToXAlpha(float posX, float posAlpha, bool inFlyDirection)
284 {
285  if (mPropagateBzOnly) {
286  return PropagateToXAlphaBz(posX, posAlpha, inFlyDirection);
287  }
288  if (CAMath::Abs(posAlpha - mAlpha) > 1.e-4f) {
289  if (RotateToAlpha(posAlpha) != 0) {
290  return -2;
291  }
292  }
293 
294  float B[3];
295  GetBxByBz(mT0.X(), mT0.Y(), mT0.Z(), B);
296 
297  // propagate mT0 to t0e
298 
299  GPUTPCGMPhysicalTrackModel t0e(mT0);
300  float dLp = 0;
301  if (t0e.PropagateToXBxByBz(posX, B[0], B[1], B[2], dLp) && t0e.PropagateToXBzLight(posX, B[2], dLp)) {
302  return 1;
303  }
304 
305  if (CAMath::Abs(t0e.SinPhi()) >= mMaxSinPhi) {
306  return -3;
307  }
308 
309  return FollowLinearization(t0e, B[2], dLp, inFlyDirection);
310 }

312 GPUd() int GPUTPCGMPropagator::PropagateToXAlphaBz(float posX, float posAlpha, bool inFlyDirection)
313 {
314  if (CAMath::Abs(posAlpha - mAlpha) > 1.e-4) {
315  if (RotateToAlpha(posAlpha) != 0) {
316  return -2;
317  }
318  }
319 
320  float Bz = GetBz(mT0.X(), mT0.Y(), mT0.Z());
321 
322  // propagate mT0 to t0e
323 
324  GPUTPCGMPhysicalTrackModel t0e(mT0);
325  float dLp = 0;
326  if (t0e.PropagateToXBzLight(posX, Bz, dLp))
327  return 1;
328 
329  if (CAMath::Abs(t0e.SinPhi()) >= mMaxSinPhi)
330  return -3;
331 
332  return FollowLinearization(t0e, Bz, dLp, inFlyDirection);
333 }

335 GPUd() int GPUTPCGMPropagator::FollowLinearization(const GPUTPCGMPhysicalTrackModel& GPUrestrict() t0e, float Bz, float dLp, bool inFlyDirection)
336 {
337  // t0e is alrerady extrapolated t0
338 
339  // propagate track and cov matrix with derivatives for (0,0,Bz) field
340 
341  float dS = dLp * t0e.GetPt();
342  float dL = CAMath::Abs(dLp * t0e.GetP());
343 
344  if (inFlyDirection) {
345  dL = -dL;
346  }
347 
348  float ey = mT0.SinPhi();
349  float ex = mT0.CosPhi();
350  float exi = mT0.SecPhi();
351  float ey1 = t0e.GetSinPhi();
352  float ex1 = t0e.GetCosPhi();
353  float ex1i = t0e.GetSecPhi();
354 
355  float k = -mT0.QPt() * Bz;
356  float dx = t0e.GetX() - mT0.X();
357  float kdx = k * dx;
358  float cc = ex + ex1;
359  float cci = 1.f / cc;
360 
361  float dxcci = dx * cci;
362  float hh = dxcci * ex1i * (1.f + ex * ex1 + ey * ey1);
363 
364  float j02 = exi * hh;
365  float j04 = -Bz * dxcci * hh;
366  float j13 = dS;
367  float j24 = -dx * Bz;
368 
369  float* p = mT->Par();
370 
371  float d0 = p[0] - mT0.Y();
372  float d1 = p[1] - mT0.Z();
373  float d2 = p[2] - mT0.SinPhi();
374  float d3 = p[3] - mT0.DzDs();
375  float d4 = p[4] - mT0.QPt();
376 
377  float newSinPhi = ey1 + d2 + j24 * d4;
378  if (mT->NDF() >= 15 && CAMath::Abs(newSinPhi) > GPUCA_MAX_SIN_PHI) {
379  return -4;
380  }
381 
382  if (mMatLUT) {
383  float xyz1[3] = {getGlobalX(mT0.GetX(), mT0.GetY()), getGlobalY(mT0.GetX(), mT0.GetY()), mT0.GetZ()};
384  float xyz2[3] = {getGlobalX(t0e.GetX(), t0e.GetY()), getGlobalY(t0e.GetX(), t0e.GetY()), t0e.GetZ()};
385  o2::base::MatBudget mat = getMatBudget(xyz1, xyz2);
386  if (mat.meanX2X0 > 1.e-8) {
387  SetMaterial(mat.length / mat.meanX2X0, mat.meanRho);
388  } else {
389  SetMaterialTPC();
390  }
391  }
392 
393  mT0 = t0e;
394  mT->X() = t0e.GetX();
395  p[0] = t0e.GetY() + d0 + j02 * d2 + j04 * d4;
396  p[1] = t0e.GetZ() + d1 + j13 * d3;
397  p[2] = newSinPhi;
398  p[3] = t0e.GetDzDs() + d3;
399  p[4] = t0e.GetQPt() + d4;
400 
401  float* c = mT->Cov();
402 
403  float c20 = c[3];
404  float c21 = c[4];
405  float c22 = c[5];
406 
407  float c30 = c[6];
408  float c31 = c[7];
409  float c32 = c[8];
410  float c33 = c[9];
411 
412  float c40 = c[10];
413  float c41 = c[11];
414  float c42 = c[12];
415  float c43 = c[13];
416  float c44 = c[14];
417 
418  if (mFitInProjections || mT->NDF() <= 0) {
419  float c20ph04c42 = c20 + j04 * c42;
420  float j02c22 = j02 * c22;
421  float j04c44 = j04 * c44;
422 
423  float n6 = c30 + j02 * c32 + j04 * c43;
424  float n7 = c31 + j13 * c33;
425  float n10 = c40 + j02 * c42 + j04c44;
426  float n11 = c41 + j13 * c43;
427  float n12 = c42 + j24 * c44;
428 
429  c[0] += j02 * j02c22 + j04 * j04c44 + 2.f * (j02 * c20ph04c42 + j04 * c40);
430  c[1] += j02 * c21 + j04 * c41 + j13 * n6;
431  c[2] += j13 * (c31 + n7);
432  c[3] = c20ph04c42 + j02c22 + j24 * n10;
433  c[4] = c21 + j13 * c32 + j24 * n11;
434  c[5] = c22 + j24 * (c42 + n12);
435  c[6] = n6;
436  c[7] = n7;
437  c[8] = c32 + c43 * j24;
438  c[10] = n10;
439  c[11] = n11;
440  c[12] = n12;
441  } else {
442  float c00 = c[0];
443  float c10 = c[1];
444  float c11 = c[2];
445 
446  float ss = ey + ey1;
447  float tg = ss * cci;
448  float xx = 1.f - 0.25f * kdx * kdx * (1.f + tg * tg);
449  if (xx < 1.e-8f) {
450  return -1;
451  }
452  xx = CAMath::Sqrt(xx);
453  float yy = CAMath::Sqrt(ss * ss + cc * cc);
454 
455  float j12 = dx * mT0.DzDs() * tg * (2.f + tg * (ey * exi + ey1 * ex1i)) / (xx * yy);
456  float j14 = 0;
457  if (CAMath::Abs(mT0.QPt()) > 1.e-6f) {
458  j14 = (2.f * xx * ex1i * dx / yy - dS) * mT0.DzDs() / mT0.QPt();
459  } else {
460  j14 = -mT0.DzDs() * Bz * dx * dx * exi * exi * exi * (0.5f * ey + (1.f / 3.f) * kdx * (1 + 2.f * ey * ey) * exi * exi);
461  }
462 
463  p[1] += j12 * d2 + j14 * d4;
464 
465  float h00 = c00 + c20 * j02 + c40 * j04;
466  // float h01 = c10 + c21*j02 + c41*j04;
467  float h02 = c20 + c22 * j02 + c42 * j04;
468  // float h03 = c30 + c32*j02 + c43*j04;
469  float h04 = c40 + c42 * j02 + c44 * j04;
470 
471  float h10 = c10 + c20 * j12 + c30 * j13 + c40 * j14;
472  float h11 = c11 + c21 * j12 + c31 * j13 + c41 * j14;
473  float h12 = c21 + c22 * j12 + c32 * j13 + c42 * j14;
474  float h13 = c31 + c32 * j12 + c33 * j13 + c43 * j14;
475  float h14 = c41 + c42 * j12 + c43 * j13 + c44 * j14;
476 
477  float h20 = c20 + c40 * j24;
478  float h21 = c21 + c41 * j24;
479  float h22 = c22 + c42 * j24;
480  float h23 = c32 + c43 * j24;
481  float h24 = c42 + c44 * j24;
482 
483  c[0] = h00 + h02 * j02 + h04 * j04;
484 
485  c[1] = h10 + h12 * j02 + h14 * j04;
486  c[2] = h11 + h12 * j12 + h13 * j13 + h14 * j14;
487 
488  c[3] = h20 + h22 * j02 + h24 * j04;
489  c[4] = h21 + h22 * j12 + h23 * j13 + h24 * j14;
490  c[5] = h22 + h24 * j24;
491 
492  c[6] = c30 + c32 * j02 + c43 * j04;
493  c[7] = c31 + c32 * j12 + c33 * j13 + c43 * j14;
494  c[8] = c32 + c43 * j24;
495  // c[ 9] = c33;
496 
497  c[10] = c40 + c42 * j02 + c44 * j04;
498  c[11] = c41 + c42 * j12 + c43 * j13 + c44 * j14;
499  c[12] = c42 + c44 * j24;
500  // c[13] = c43;
501  // c[14] = c44;
502  }
503 
504  float& mC22 = c[5];
505  float& mC33 = c[9];
506  float& mC40 = c[10];
507  float& mC41 = c[11];
508  float& mC42 = c[12];
509  float& mC43 = c[13];
510  float& mC44 = c[14];
511 
512  float dLmask = 0.f;
513  bool maskMS = (CAMath::Abs(dL) < mMaterial.DLMax);
514  if (maskMS) {
515  dLmask = dL;
516  }
517  float dLabs = CAMath::Abs(dLmask);
518 
519  // Energy Loss
520 
521  if (1 || !mToyMCEvents) {
522  // std::cout<<"APPLY ENERGY LOSS!!!"<<std::endl;
523  float corr = 1.f - mMaterial.EP2 * dLmask;
524  float corrInv = 1.f / corr;
525  mT0.Px() *= corrInv;
526  mT0.Py() *= corrInv;
527  mT0.Pz() *= corrInv;
528  mT0.Pt() *= corrInv;
529  mT0.P() *= corrInv;
530  mT0.QPt() *= corr;
531 
532  p[4] *= corr;
533 
534  mC40 *= corr;
535  mC41 *= corr;
536  mC42 *= corr;
537  mC43 *= corr;
538  mC44 = mC44 * corr * corr + dLabs * mMaterial.sigmadE2;
539  } else {
540  // std::cout<<"DONT APPLY ENERGY LOSS!!!"<<std::endl;
541  }
542 
543  // Multiple Scattering
544 
545  if (!mToyMCEvents) {
546  mC22 += dLabs * mMaterial.k22 * mT0.CosPhi() * mT0.CosPhi();
547  mC33 += dLabs * mMaterial.k33;
548  mC43 += dLabs * mMaterial.k43;
549  mC44 += dLabs * mMaterial.k44;
550  }
551  return 0;
552 }

554 GPUd() int GPUTPCGMPropagator::GetPropagatedYZ(float x, float& GPUrestrict() projY, float& GPUrestrict() projZ)
555 {
556  float bz = GetBz(mT->X(), mT->Y(), mT->Z());
557  float k = mT0.QPt() * bz;
558  float dx = x - mT->X();
559  float kdx = k * dx;
560  float ex = mT0.CosPhi();
561  float ey = mT0.SinPhi();
562  float ey1 = kdx + ey;
563  if (CAMath::Abs(ey1) > GPUCA_MAX_SIN_PHI) {
564  return 1;
565  }
566  float ss = ey + ey1;
567  float ex1 = CAMath::Sqrt(1.f - ey1 * ey1);
568  float cc = ex + ex1;
569  float dxcci = dx / cc;
570  float dy = dxcci * ss;

580  }

583  projZ = mT->Z() + dz;
584  return 0;

624  w0 = w0 * det;
625  w1 = -w1 * det;
626  w2 = w2 * det;
627  }

668 GPUd() int GPUTPCGMPropagator::InterpolateReject(float posY, float posZ, short clusterState, char rejectChi2, GPUTPCGMMergerTypes::InterpolationErrorHit* inter, float err2Y, float err2Z)
669 {
670  float* GPUrestrict() mC = mT->Cov();
671  float* GPUrestrict() mP = mT->Par();
672  if (rejectChi2 == 2) {
673  inter->posY = mP[0];
674  inter->posZ = mP[1];
675  inter->errorY = mC[0];
676  inter->errorZ = mC[2];
677  } else if (rejectChi2 == 3) {
678  float chiY, chiZ;
679  if (mFitInProjections || mT->NDF() <= 0) {
680  const float Iz0 = inter->posY - mP[0];
681  const float Iz1 = inter->posZ - mP[1];
682  const float Iw0 = 1.f / (mC[0] + inter->errorY);
683  const float Iw2 = 1.f / (mC[2] + inter->errorZ);
684  const float Ik00 = mC[0] * Iw0;
685  const float Ik11 = mC[2] * Iw2;
686  const float ImP0 = mP[0] + Ik00 * Iz0;
687  const float ImP1 = mP[1] + Ik11 * Iz1;
688  const float ImC0 = mC[0] - Ik00 * mC[0];
689  const float ImC2 = mC[2] - Ik11 * mC[2];
690  //printf("\t%21sInterpo ----- abde artaf%16s Y %8.3f, Z %8.3f (Errors %f <-- (%f, %f) %f <-- (%f, %f))\n", "", "", ImP0, ImP1, sqrtf(ImC0), sqrtf(mC[0]), sqrtf(inter->errorY), sqrtf(ImC2), sqrtf(mC[2]), sqrtf(inter->errorZ));
691  const float Jz0 = posY - ImP0;
692  const float Jz1 = posZ - ImP1;
693  const float Jw0 = 1.f / (ImC0 + err2Y);
694  const float Jw2 = 1.f / (ImC2 + err2Z);
695  chiY = Jw0 * Jz0 * Jz0;
696  chiZ = Jw2 * Jz1 * Jz1;
697  } else {
698  const float Iz0 = inter->posY - mP[0];
699  const float Iz1 = inter->posZ - mP[1];
700  float Iw0 = mC[2] + inter->errorZ;
701  float Iw2 = mC[0] + inter->errorY;
702  float Idet = CAMath::Max(1e-10f, Iw0 * Iw2 - mC[1] * mC[1]);
703  Idet = 1.f / Idet;
704  Iw0 *= Idet;
705  const float Iw1 = mC[1] * Idet;
706  Iw2 *= Idet;
707  const float Ik00 = mC[0] * Iw0 + mC[1] * Iw1;
708  const float Ik01 = mC[0] * Iw1 + mC[1] * Iw2;
709  const float Ik10 = mC[1] * Iw0 + mC[2] * Iw1;
710  const float Ik11 = mC[1] * Iw1 + mC[2] * Iw2;
711  const float ImP0 = mP[0] + Ik00 * Iz0 + Ik01 * Iz1;
712  const float ImP1 = mP[1] + Ik10 * Iz0 + Ik11 * Iz1;
713  const float ImC0 = mC[0] - Ik00 * mC[0] + Ik01 * mC[1];
714  const float ImC1 = mC[1] - Ik10 * mC[0] + Ik11 * mC[1];
715  const float ImC2 = mC[2] - Ik10 * mC[1] + Ik11 * mC[2];
716  const float Jz0 = posY - ImP0;
717  const float Jz1 = posZ - ImP1;
718  float Jw0 = ImC2 + err2Z;
719  float Jw2 = ImC0 + err2Y;
720  float Jdet = CAMath::Max(1e-10f, Jw0 * Jw2 - ImC1 * ImC1);
721  Jdet = 1.f / Jdet;
722  Jw0 *= Jdet;
723  const float Jw1 = ImC1 * Jdet;
724  Jw2 *= Jdet;
725  chiY = CAMath::Abs((Jw0 * Jz0 + Jw1 * Jz1) * Jz0);
726  chiZ = CAMath::Abs((Jw1 * Jz0 + Jw2 * Jz1) * Jz1);
727  }
728  if (RejectCluster(chiY, chiZ, clusterState)) { // TODO: Relative Pt resolution decreases slightly, why?
729  return 2;
730  }
731  }
732  return 0;
733 }

735 GPUd() int GPUTPCGMPropagator::Update(float posY, float posZ, short clusterState, bool rejectChi2, float err2Y, float err2Z)
736 {
737  float* GPUrestrict() mC = mT->Cov();
738  float* GPUrestrict() mP = mT->Par();
739 
740  const float d00 = mC[0], d01 = mC[1], d02 = mC[3], d03 = mC[6], d04 = mC[10];
741  const float d10 = mC[1], d11 = mC[2], d12 = mC[4], d13 = mC[7], d14 = mC[11];
742 
743  const float z0 = posY - mP[0];
744  const float z1 = posZ - mP[1];
745  float w0, w1, w2, chiY, chiZ;
746  if (mFitInProjections || mT->NDF() <= 0) {
747  w0 = 1.f / (err2Y + d00);
748  w1 = 0;
749  w2 = 1.f / (err2Z + d11);
750  chiY = w0 * z0 * z0;
751  chiZ = w2 * z1 * z1;
752  } else {
753  w0 = d11 + err2Z, w1 = d10, w2 = d00 + err2Y;
754  { // Invert symmetric matrix
755  float det = w0 * w2 - w1 * w1;
756  if (CAMath::Abs(det) < 1.e-10f) {
757  return -1;
758  }
759  det = 1.f / det;
760  w0 = w0 * det;
761  w1 = -w1 * det;
762  w2 = w2 * det;
763  }

765  chiZ = CAMath::Abs((w1 * z0 + w2 * z1) * z1);

788  mC[2] -= k11 * d11;

790  mC[5] -= k20 * d02;
791  mC[7] -= k31 * d11;
792  mC[9] -= k31 * d13;
793  mC[10] -= k00 * d04;
794  mC[12] -= k40 * d02;
795  mC[14] -= k40 * d04;
796  } else {
797  const float k00 = d00 * w0 + d01 * w1;
798  const float k01 = d00 * w1 + d10 * w2;
799  const float k10 = d01 * w0 + d11 * w1;
800  const float k11 = d01 * w1 + d11 * w2;
801  const float k20 = d02 * w0 + d12 * w1;
802  const float k21 = d02 * w1 + d12 * w2;
803  const float k30 = d03 * w0 + d13 * w1;
804  const float k31 = d03 * w1 + d13 * w2;
805  const float k40 = d04 * w0 + d14 * w1;
806  const float k41 = d04 * w1 + d14 * w2;
807 
808  mP[0] += k00 * z0 + k01 * z1;
809  mP[1] += k10 * z0 + k11 * z1;
810  mP[2] += k20 * z0 + k21 * z1;
811  mP[3] += k30 * z0 + k31 * z1;
812  mP[4] += k40 * z0 + k41 * z1;
813

815

842 GPUd() float GPUTPCGMPropagator::ApproximateBetheBloch(float beta2)
843 {
844  //------------------------------------------------------------------
845  // This is an approximation of the Bethe-Bloch formula with
846  // the density effect taken into account at beta*gamma > 3.5
847  // (the approximation is reasonable only for solid materials)
848  //------------------------------------------------------------------
849 
850  const float log0 = log(5940.f);
851  const float log1 = log(3.5f * 5940.f);
852 
853  bool bad = (beta2 >= .999f) || (beta2 < 1.e-8f);
854 
855  if (bad) {
856  beta2 = 0.5f;
857  }
858 
859  float a = beta2 / (1.f - beta2);
860  float b = 0.5f * log(a);
861  float d = 0.153e-3f / beta2;
862  float c = b - beta2;
863 
864  float ret = d * (log0 + b + c);
865  float case1 = d * (log1 + c);
866 
867  if (a > 3.5f * 3.5f) {
868  ret = case1;
869  }
870  if (bad) {
871  ret = 0.f;
872  }
873 
874  return ret;
875 }

877 GPUd() void GPUTPCGMPropagator::CalculateMaterialCorrection()
878 {
879  //*!
880 
881  const float mass = 0.13957f;
882 
883  float qpt = mT0.GetQPt();
884  if (CAMath::Abs(qpt) > 20) {
885  qpt = 20;
886  }
887 
888  float w2 = (1.f + mT0.GetDzDs() * mT0.GetDzDs()); //==(P/pt)2
889  float pti2 = qpt * qpt;
890  if (pti2 < 1.e-4f) {
891  pti2 = 1.e-4f;
892  }
893 
894  float mass2 = mass * mass;
895  float beta2 = w2 / (w2 + mass2 * pti2);
896 
897  float p2 = w2 / pti2; // impuls 2
898  float betheRho = ApproximateBetheBloch(p2 / mass2) * mMaterial.rho;
899  float E = CAMath::Sqrt(p2 + mass2);
900  float theta2 = (14.1f * 14.1f / 1.e6f) / (beta2 * p2) * mMaterial.radLenInv;
901 
902  mMaterial.EP2 = E / p2;
903 
904  // Approximate energy loss fluctuation (M.Ivanov)
905 
906  const float knst = 0.07f; // To be tuned.
907  mMaterial.sigmadE2 = knst * mMaterial.EP2 * qpt;
908  mMaterial.sigmadE2 = mMaterial.sigmadE2 * mMaterial.sigmadE2;
909 
910  mMaterial.k22 = theta2 * w2;
911  mMaterial.k33 = mMaterial.k22 * w2;
912  mMaterial.k43 = 0.f;
913  mMaterial.k44 = theta2 * mT0.GetDzDs() * mT0.GetDzDs() * pti2;
914 
915  float br = (betheRho > 1.e-8f) ? betheRho : 1.e-8f;
916  mMaterial.DLMax = 0.3f * E / br;
917  mMaterial.EP2 *= betheRho;
918  mMaterial.sigmadE2 = mMaterial.sigmadE2 * betheRho; // + mMaterial.fK44;
919 }
