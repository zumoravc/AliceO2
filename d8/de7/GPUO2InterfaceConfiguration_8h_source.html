50 // This defines an output region. Ptr points to a memory buffer, which should have a proper alignment.
51 // Since DPL does not respect the alignment of data types, we do not impose anything specic but just use a char data type, but it should be >= 64 bytes ideally.
52 // The size defines the maximum possible buffer size when GPUReconstruction is called, and returns the number of filled bytes when it returns.
53 // If ptr == nullptr, there is no region defined and GPUReconstruction will write its output to an internal buffer.
54 // If allocator is set, it is called as a callback to provide a ptr to the memory.
55 struct GPUInterfaceOutputRegion {
56  void* ptr = nullptr;
57  size_t size = 0;
58  std::function<void*(size_t)> allocator = nullptr;
59 };

93 // Structure with pointers to actual data for input and output
94 // Which ptr is used for input and which for output is defined in GPUO2InterfaceConfiguration::configWorkflow
95 // Inputs and outputs are mutually exclusive.
96 // Inputs which are nullptr are considered empty, and will not throw an error.
97 // Outputs, which point to std::[container] / MCTruthContainer, will be filled and no output
98 // is written if the ptr is a nullptr.
99 // Outputs, which point to other structures are set by GPUCATracking to the location of the output. The previous
100 // value of the pointer is overridden. GPUCATracking will try to place the output in the "void* outputBuffer"
101 // location if it is not a nullptr.
102 struct GPUO2InterfaceIOPtrs {
103  // Input: TPC clusters in cluster native format, or digits, or list of ZS pages - const as it can only be input
104  const o2::tpc::ClusterNativeAccess* clusters = nullptr;
105  const std::array<gsl::span<const o2::tpc::Digit>, o2::tpc::Constants::MAXSECTOR>* o2Digits = nullptr;
106  std::array<std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>>, o2::tpc::Constants::MAXSECTOR>* o2DigitsMC = nullptr;
107  const o2::gpu::GPUTrackingInOutZS* tpcZS = nullptr;
108 
109  // Input / Output for Merged TPC tracks, two ptrs, for the tracks themselves, and for the MC labels.
110  std::vector<o2::tpc::TrackTPC>* outputTracks = nullptr;
111  std::vector<uint32_t>* outputClusRefs = nullptr;
112  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* outputTracksMCTruth = nullptr;
113 
114  // Output for entropy-reduced clusters of TPC compression
115  const o2::tpc::CompressedClustersFlat* compressedClusters;
116 
117  // Hint for GPUCATracking to place its output in this buffer if possible.
118  // This enables to create the output directly in a shared memory segment of the framework.
119  // This allows further processing with zero-copy.
120  // So far this is only a hint, GPUCATracking will not always follow.
121  // If outputBuffer = nullptr, GPUCATracking will allocate the output internally and own the memory.
122  // TODO: Make this mandatory if outputBuffer != nullptr, and throw an error if outputBufferSize is too small.
123  void* outputBuffer = nullptr;
124  size_t outputBufferSize = 0;
125 };
