61 void doTest(const std::map<o2::InteractionRecord, std::vector<o2::mid::ColumnData>>& inData, const std::vector<o2::mid::ROFRecord>& rofRecords, const std::vector<o2::mid::ColumnData>& data, const o2::mid::EventType inEventType = o2::mid::EventType::Standard)
62 {
63  BOOST_REQUIRE(rofRecords.size() == inData.size());
64  auto inItMap = inData.begin();
65  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
66  BOOST_TEST(static_cast<int>(rofIt->eventType) == static_cast<int>(inEventType));
67  BOOST_TEST(rofIt->interactionRecord == inItMap->first);
68  BOOST_TEST(rofIt->nEntries == inItMap->second.size());
69  auto sortedIn = sortData(inItMap->second, 0, inItMap->second.size());
70  auto sortedOut = sortData(data, rofIt->firstEntry, rofIt->firstEntry + rofIt->nEntries);
71  BOOST_REQUIRE(sortedOut.size() == sortedIn.size());
72  for (size_t icol = 0; icol < sortedOut.size(); ++icol) {
73  BOOST_TEST(sortedOut[icol].deId == sortedIn[icol].deId);
74  BOOST_TEST(sortedOut[icol].columnId == sortedIn[icol].columnId);
75  BOOST_TEST(sortedOut[icol].getNonBendPattern() == sortedIn[icol].getNonBendPattern());
76  for (int iline = 0; iline < 4; ++iline) {
77  BOOST_TEST(sortedOut[icol].getBendPattern(iline) == sortedIn[icol].getBendPattern(iline));
78  }
79  }
80  ++inItMap;
81  }
82 }

84 std::tuple<std::vector<o2::mid::ColumnData>, std::vector<o2::mid::ROFRecord>> encodeDecode(std::map<o2::InteractionRecord, std::vector<o2::mid::ColumnData>> inData, o2::mid::EventType inEventType = o2::mid::EventType::Standard)
85 {
86  auto severity = fair::Logger::GetConsoleSeverity();
87  fair::Logger::SetConsoleSeverity(fair::Severity::WARNING);
88  std::string tmpFilename = "tmp_mid_raw.dat";
89  o2::mid::Encoder encoder;
90  encoder.init(tmpFilename.c_str());
91  for (auto& item : inData) {
92  encoder.process(item.second, item.first, inEventType);
93  }
94  encoder.finalize();
95 
96  o2::raw::RawFileReader rawReader;
97  rawReader.addFile(tmpFilename.c_str());
98  rawReader.init();
99  std::vector<char> buffer;
100  for (size_t itf = 0; itf < rawReader.getNTimeFrames(); ++itf) {
101  rawReader.setNextTFToRead(itf);
102  for (size_t ilink = 0; ilink < rawReader.getNLinks(); ++ilink) {
103  auto& link = rawReader.getLink(ilink);
104  auto tfsz = link.getNextTFSize();
105  if (!tfsz) {
106  continue;
107  }
108  std::vector<char> linkBuffer(tfsz);
109  link.readNextTF(linkBuffer.data());
110  buffer.insert(buffer.end(), linkBuffer.begin(), linkBuffer.end());
111  }
112  }
113  fair::Logger::SetConsoleSeverity(severity);
114 
115  std::remove(tmpFilename.c_str());
116 
117  o2::mid::Decoder<o2::mid::GBTUserLogicDecoder> decoder;
118  gsl::span<const uint8_t> data(reinterpret_cast<uint8_t*>(buffer.data()), buffer.size());
119  decoder.process(data);
120 
121  o2::mid::DecodedDataAggregator aggregator;
122  aggregator.process(decoder.getData(), decoder.getROFRecords());
123 
124  return std::make_tuple(aggregator.getData(), aggregator.getROFRecords());
125 }

127 BOOST_AUTO_TEST_CASE(ColumnDataConverter)
128 {
129  std::map<o2::InteractionRecord, std::vector<o2::mid::ColumnData>> inData;
130  o2::InteractionRecord ir(100, 0);
131  // Crate 5 link 0
132  inData[ir].emplace_back(getColData(2, 4, 0x1, 0xFFFF));
133 
134  ir.bc = 200;
135  inData[ir].emplace_back(getColData(3, 4, 0xFF00, 0xFF));
136  inData[ir].emplace_back(getColData(12, 4, 0xFF));
137 
138  std::vector<o2::mid::ROFRecord> rofs;
139  std::vector<o2::mid::LocalBoardRO> outData;
140  auto inEventType = o2::mid::EventType::Standard;
141  o2::mid::ColumnDataToLocalBoard converter;
142  for (auto& item : inData) {
143  converter.process(item.second);
144  auto firstEntry = outData.size();
145  for (auto& gbtItem : converter.getData()) {
146  auto crateId = o2::mid::crateparams::getCrateIdFromROId(gbtItem.first);
147  for (auto& loc : gbtItem.second) {
148  // The crate ID information is not encoded in the local board information,
149  // since it is not needed at this level (it is encoded in the feeId)
150  // However, when we put the info back together, we need to know the crate ID
151  // So we encode it in the local board ID in the output
152  outData.emplace_back(loc);
153  outData.back().boardId = o2::mid::crateparams::makeUniqueLocID(crateId, loc.boardId);
154  }
155  rofs.push_back({item.first, inEventType, firstEntry, outData.size() - firstEntry});
156  }
157  }
158 
159  o2::mid::DecodedDataAggregator aggregator;
160  aggregator.process(outData, rofs);
161 
162  doTest(inData, aggregator.getROFRecords(), aggregator.getData());
163 }

165 BOOST_AUTO_TEST_CASE(GBTUserLogicDecoder)
166 {
168 
169  std::map<uint16_t, std::vector<o2::mid::LocalBoardRO>> inData;
170  uint16_t bc = 100;
171  o2::mid::LocalBoardRO loc;
172  // Crate 5 link 0
173  loc.statusWord = o2::mid::raw::sSTARTBIT | o2::mid::raw::sCARDTYPE;
174  loc.triggerWord = 0;
175  loc.boardId = 2;
176  loc.firedChambers = 0x1;
177  loc.patternsBP[0] = 0xF;
178  loc.patternsNBP[0] = 0xF;
179  inData[bc].emplace_back(loc);
180 
181  bc = 200;
182  loc.patternsBP.fill(0);
183  loc.patternsNBP.fill(0);
184  loc.boardId = 5;
185  loc.firedChambers = 0x4;
186  loc.patternsBP[2] = 0xF0;
187  loc.patternsNBP[2] = 0x5;
188  inData[bc].emplace_back(loc);
189 
190  uint8_t crateId = 5;
191  uint8_t linkInCrate = 0;
192  uint16_t feeId = o2::mid::crateparams::makeROId(crateId, linkInCrate);
193  o2::mid::GBTUserLogicEncoder encoder;
194  encoder.setFeeId(feeId);
195  for (auto& item : inData) {
196  encoder.process(item.second, item.first);
197  }
198  o2::header::RAWDataHeader rdh;
199  auto memSize = encoder.getBufferSize() + 64;
200  rdh.word1 |= (memSize | (memSize << 16));
201  // Sets the feeId
202  rdh.word0 |= ((5 * 2) << 16);
203  o2::mid::GBTUserLogicDecoder decoder;
204  decoder.init(feeId);
205  decoder.process(encoder.getBuffer(), rdh);
206  BOOST_REQUIRE(decoder.getROFRecords().size() == inData.size());
207  auto inItMap = inData.begin();
208  for (auto rofIt = decoder.getROFRecords().begin(); rofIt != decoder.getROFRecords().end(); ++rofIt) {
209  BOOST_TEST(rofIt->interactionRecord.bc == inItMap->first);
210  BOOST_TEST(rofIt->nEntries == inItMap->second.size());
211  auto outLoc = decoder.getData().begin() + rofIt->firstEntry;
212  for (auto inLoc = inItMap->second.begin(); inLoc != inItMap->second.end(); ++inLoc) {
213  BOOST_TEST(inLoc->statusWord == outLoc->statusWord);
214  BOOST_TEST(inLoc->triggerWord == outLoc->triggerWord);
215  BOOST_TEST(o2::mid::crateparams::makeUniqueLocID(crateId, inLoc->boardId) == outLoc->boardId);
216  BOOST_TEST(inLoc->firedChambers == outLoc->firedChambers);
217  for (int ich = 0; ich < 4; ++ich) {
218  BOOST_TEST(inLoc->patternsBP[ich] == outLoc->patternsBP[ich]);
219  BOOST_TEST(inLoc->patternsNBP[ich] == outLoc->patternsNBP[ich]);
220  }
221  ++outLoc;
222  }
223  ++inItMap;
224  }
225 }
