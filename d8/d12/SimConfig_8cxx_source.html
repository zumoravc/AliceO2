65 bool SimConfig::resetFromParsedMap(boost::program_options::variables_map const& vm)
66 {
67  using o2::detectors::DetID;
68  mConfigData.mMCEngine = vm["mcEngine"].as<std::string>();
69  mConfigData.mActiveDetectors = vm["modules"].as<std::vector<std::string>>();
70  auto& active = mConfigData.mActiveDetectors;
71  if (active.size() == 1 && active[0] == "all") {
72  active.clear();
73  for (int d = DetID::First; d <= DetID::Last; ++d) {
74  active.emplace_back(DetID::getName(d));
75  }
76  // add passive components manually (make a PassiveDetID for them!)
77  active.emplace_back("HALL");
78  active.emplace_back("MAG");
79  active.emplace_back("DIPO");
80  active.emplace_back("COMP");
81  active.emplace_back("PIPE");
82  active.emplace_back("ABSO");
83  active.emplace_back("SHIL");
84  }
85  // now we take out detectors listed as skipped
86  auto& skipped = vm["skipModules"].as<std::vector<std::string>>();
87  for (auto& s : skipped) {
88  auto iter = std::find(active.begin(), active.end(), s);
89  if (iter != active.end()) {
90  // take it out
91  active.erase(iter);
92  }
93  }
94 
95  mConfigData.mGenerator = vm["generator"].as<std::string>();
96  mConfigData.mTrigger = vm["trigger"].as<std::string>();
97  mConfigData.mNEvents = vm["nEvents"].as<unsigned int>();
98  mConfigData.mExtKinFileName = vm["extKinFile"].as<std::string>();
99  mConfigData.mHepMCFileName = vm["HepMCFile"].as<std::string>();
100  mConfigData.mExtGenFileName = vm["extGenFile"].as<std::string>();
101  mConfigData.mExtGenFuncName = vm["extGenFunc"].as<std::string>();
102  mConfigData.mExtTrgFileName = vm["extTrgFile"].as<std::string>();
103  mConfigData.mExtTrgFuncName = vm["extTrgFunc"].as<std::string>();
104  mConfigData.mEmbedIntoFileName = vm["embedIntoFile"].as<std::string>();
105  mConfigData.mDecayConfig = vm["decayConfig"].as<std::string>();
106  mConfigData.mStartEvent = vm["startEvent"].as<unsigned int>();
107  mConfigData.mBMax = vm["bMax"].as<float>();
108  mConfigData.mIsMT = vm["isMT"].as<bool>();
109  mConfigData.mOutputPrefix = vm["outPrefix"].as<std::string>();
110  mConfigData.mLogSeverity = vm["logseverity"].as<std::string>();
111  mConfigData.mLogVerbosity = vm["logverbosity"].as<std::string>();
112  mConfigData.mKeyValueTokens = vm["configKeyValues"].as<std::string>();
113  mConfigData.mConfigFile = vm["configFile"].as<std::string>();
114  mConfigData.mPrimaryChunkSize = vm["chunkSize"].as<unsigned int>();
115  mConfigData.mInternalChunkSize = vm["chunkSizeI"].as<int>();
116  mConfigData.mStartSeed = vm["seed"].as<int>();
117  mConfigData.mSimWorkers = vm["nworkers"].as<int>();
118  mConfigData.mTimestamp = vm["timestamp"].as<long>();
119  mConfigData.mCCDBUrl = vm["CCDBUrl"].as<std::string>();
120  if (vm.count("noemptyevents")) {
121  mConfigData.mFilterNoHitEvents = true;
122  }
123  mConfigData.mField = vm["field"].as<int>();
124  return true;
125 }

127 bool SimConfig::resetFromArguments(int argc, char* argv[])
128 {
129  namespace bpo = boost::program_options;
130 
131  // Arguments parsing
132  bpo::variables_map vm;
133  bpo::options_description desc("Allowed options");
134  desc.add_options()("help,h", "Produce help message.");
135  initOptions(desc);
136 
137  try {
138  bpo::store(parse_command_line(argc, argv, desc), vm);
139 
140  // help
141  if (vm.count("help")) {
142  std::cout << desc << std::endl;
143  return false;
144  }
145 
146  bpo::notify(vm);
147  } catch (const bpo::error& e) {
148  std::cerr << e.what() << "\n\n";
149  std::cerr << "Error parsing command line arguments; Available options:\n";
150 
151  std::cerr << desc << std::endl;
152  return false;
153  }
154 
155  return resetFromParsedMap(vm);
156 }
