46 void CompressedDecodingTask::postData(ProcessingContext& pc)
47 {
48  mHasToBePosted = false;
49  mDecoder.FillWindows();
50 
51  // send output message
52  std::vector<o2::tof::Digit>* alldigits = mDecoder.getDigitPerTimeFrame();
53  std::vector<o2::tof::ReadoutWindowData>* row = mDecoder.getReadoutWindowData();
54 
55  int n_tof_window = row->size();
56  int n_orbits = n_tof_window / 3;
57  int digit_size = alldigits->size();
58 
59  LOG(INFO) << "TOF: N tof window decoded = " << n_tof_window << "(orbits = " << n_orbits << ") with " << digit_size << " digits";
60 
61  // add digits in the output snapshot
62  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe}, *alldigits);
63  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe}, *row);
64 
65  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
66 
67  LOG(INFO) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
68  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}, roMode);
69 
70  mDecoder.clear();
71 
72  LOG(INFO) << "TOF: TF = " << mNTF << " - Crate in " << mNCrateOpenTF;
73 
74  mNTF++;
75  mNCrateOpenTF = 0;
76  mNCrateCloseTF = 0;
77 }

79 void CompressedDecodingTask::run(ProcessingContext& pc)
80 {
81  LOG(INFO) << "CompressedDecoding run";
82  mTimer.Start(false);
83 
85  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
86  if (!iit.isValid())
87  continue;
88 
90  for (auto const& ref : iit) {
91 
92  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
93  auto payloadIn = ref.payload;
94  auto payloadInSize = headerIn->payloadSize;
95 
96  DecoderBase::setDecoderBuffer(payloadIn);
97  DecoderBase::setDecoderBufferSize(payloadInSize);
98  DecoderBase::run();
99  }
100  }
101 
102  if (mNCrateOpenTF == 72 && mNCrateOpenTF == mNCrateCloseTF)
103  mHasToBePosted = true;
104 
105  if (mHasToBePosted) {
106  postData(pc);
107  }
108  mTimer.Stop();
109 }

117 void CompressedDecodingTask::rdhHandler(const o2::header::RAWDataHeader* rdh)
118 {
119 
120  // rdh close
121  if (rdh->stop && rdh->heartbeatOrbit == o2::raw::HBFUtils::Instance().getNOrbitsPerTF() - 1 + mInitOrbit) {
122  mNCrateCloseTF++;
123  // printf("New TF close RDH %d\n", int(rdh->feeId));
124  return;
125  }
126 
127  // rdh open
128  if ((rdh->pageCnt == 0) && (rdh->triggerType & o2::trigger::TF)) {
129  mNCrateOpenTF++;
130  mInitOrbit = rdh->heartbeatOrbit;
131  // printf("New TF open RDH %d\n", int(rdh->feeId));
