66 framework::DataProcessorSpec MergerBuilder::buildSpec()
67 {
68  framework::DataProcessorSpec merger;
69 
70  merger.name = mergerIdString() + "-" + mName + std::to_string(mLayer) + "l-" + std::to_string(mId);
71 
72  merger.inputs = mInputSpecs;
73 
74  merger.outputs.push_back(mOutputSpec);
75  framework::DataAllocator::SubSpecificationType subSpec = DataSpecUtils::getOptionalSubSpec(mOutputSpec).value();
76  if (DataSpecUtils::validate(mOutputSpec) == false) {
77  // inner layer => generate output spec according to scheme
78  subSpec = mergerSubSpec(mLayer, mId);
79  merger.outputs[0] = OutputSpec{{mergerOutputBinding()},
80  mergerDataOrigin(),
81  mergerDataDescription(mName),
82  subSpec}; // it servers as a unique merger output ID
83  } else {
84  // last layer
85  merger.outputs[0].binding = {mergerOutputBinding()};
86  }
87 
88  if (mConfig.inputObjectTimespan.value == InputObjectsTimespan::LastDifference) {
89  merger.algorithm = framework::adaptFromTask<IntegratingMerger>(mConfig, subSpec);
90  } else {
91  merger.algorithm = framework::adaptFromTask<FullHistoryMerger>(mConfig, subSpec);
92  }
93 
94  merger.inputs.push_back({"timer-publish", "MRGR", mergerDataDescription("timer-" + mName), mergerSubSpec(mLayer, mId), framework::Lifetime::Timer});
95  merger.options.push_back({"period-timer-publish", framework::VariantType::Int, static_cast<int>(mConfig.publicationDecision.param * 1000000), {"timer period"}});
96 
97  return std::move(merger);
98 }
