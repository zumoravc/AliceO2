39 GPUd() void ClusterAccumulator::update(Charge splitCharge, Delta2 d)
40 {
41  mQtot += splitCharge;

43  mTimeMean += splitCharge * d.y;
44  mPadSigma += splitCharge * d.x * d.x;

76 GPUd() void ClusterAccumulator::finalize(const ChargePos& pos, Charge q, TPCTime timeOffset)
77 {
78  mQtot += q;
79 
80  mPadMean /= mQtot;
81  mTimeMean /= mQtot;
82  mPadSigma /= mQtot;
83  mTimeSigma /= mQtot;
84 
85  mPadSigma = CAMath::Sqrt(mPadSigma - mPadMean * mPadMean);
86  mTimeSigma = CAMath::Sqrt(mTimeSigma - mTimeMean * mTimeMean);
87 
88  mPadMean += pos.pad();
89  mTimeMean += timeOffset + pos.time();
90 
91 #if defined(CORRECT_EDGE_CLUSTERS)
92  if (CfUtils::isAtEdge(pos)) {
93  float s = (pos.pad() < 2) ? 1.f : -1.f;
94  bool c = s * (mPadMean - pos.pad()) > 0.f;
95  mPadMean = (c) ? pos.pad() : mPadMean;
96  }
97 #endif
98 }
