242 GPUdi() int GPUTPCGMTrackParam::initResetT0()
243 {

272 GPUdi() float GPUTPCGMTrackParam::GetMirroredY(float Bz) const
273 {
274  // get Y of the point which has the same X, but located on the other side of trajectory
275  float qptBz = GetQPt() * Bz;
276  float cosPhi2 = 1.f - GetSinPhi() * GetSinPhi();
277  if (CAMath::Abs(qptBz) < 1.e-8f) {
278  qptBz = 1.e-8f;
279  }
280  if (cosPhi2 < 0.f) {
281  cosPhi2 = 0.f;
282  }
283  return GetY() - 2.f * sqrt(cosPhi2) / qptBz;
284 }
