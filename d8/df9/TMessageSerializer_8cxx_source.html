19 void TMessageSerializer::loadSchema(gsl::span<o2::byte> buffer)
20 {
21  std::unique_ptr<TObject> obj = deserialize(buffer);
22 
23  TObjArray* pSchemas = dynamic_cast<TObjArray*>(obj.get());
24  if (!pSchemas) {
25  return;
26  }
27 
28  // TODO: this is a bit of a problem in general: non-owning ROOT containers should become
29  // owners at deserialize, otherwise there is a leak. Switch to a better container.
30  pSchemas->SetOwner(kTRUE);
31 
32  for (int i = 0; i < pSchemas->GetEntriesFast(); i++) {
33  TStreamerInfo* pSchema = dynamic_cast<TStreamerInfo*>(pSchemas->At(i));
34  if (!pSchema) {
35  continue;
36  }
37  int version = pSchema->GetClassVersion();
38  TClass* pClass = TClass::GetClass(pSchema->GetName());
39  if (!pClass) {
40  continue;
41  }
42  if (pClass->GetClassVersion() == version) {
43  continue;
44  }
45  TObjArray* pInfos = const_cast<TObjArray*>(pClass->GetStreamerInfos());
46  if (!pInfos) {
47  continue;
48  }
49  TVirtualStreamerInfo* pInfo = dynamic_cast<TVirtualStreamerInfo*>(pInfos->At(version));
50  if (pInfo) {
51  continue;
52  }
53  pSchema->SetClass(pClass);
54  pSchema->BuildOld();
55  pInfos->AddAtAndExpand(pSchema, version);
56  pSchemas->Remove(pSchema);
57  }
58 }

83 void TMessageSerializer::updateStreamers(const FairTMessage& message, StreamerList& streamers)
84 {
85  std::lock_guard<std::mutex> lock{TMessageSerializer::sStreamersLock};
86 
87  TIter nextStreamer(message.GetStreamerInfos()); // unfortunately ROOT uses TList* here
88  // this looks like we could use std::map here.
89  while (TVirtualStreamerInfo* in = static_cast<TVirtualStreamerInfo*>(nextStreamer())) {
90  auto found = std::find_if(streamers.begin(), streamers.end(), [&](const auto& old) {
91  return (old->GetName() == in->GetName() && old->GetClassVersion() == in->GetClassVersion());
92  });
93  if (found == streamers.end()) {
94  streamers.push_back(in);
95  }
96  }
97 }
