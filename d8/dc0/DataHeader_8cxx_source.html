35 bool o2::header::BaseHeader::sanityCheck(uint32_t expectedVersion) const
36 {
37  if (this->headerVersion != expectedVersion) {
38  std::string errmsg = "header of type " + this->description.as<std::string>() + " with invalid ";
39  errmsg += "version: " + std::to_string(this->headerVersion) + " (expected " + std::to_string(expectedVersion) + ")";
40  // for the moment we throw, there is no support for multiple versions of a particular header
41  // so we better spot non matching header stacks early, we migh change this later
42  throw std::runtime_error(errmsg);
43  return false;
44  }
45  return true;
46 }

135 void o2::header::hexDump(const char* desc, const void* voidaddr, size_t len, size_t max)
136 {
137  size_t i;
138  unsigned char buff[17]; // stores the ASCII data
139  memset(&buff[0], '\0', 17);
140  const byte* addr = reinterpret_cast<const byte*>(voidaddr);
141 
142  // Output description if given.
143  if (desc != nullptr)
144  printf("%s, ", desc);
145  printf("%zu bytes:", len);
146  if (max > 0 && len > max) {
147  len = max; //limit the output if requested
148  printf(" output limited to %zu bytes\n", len);
149  } else {
150  printf("\n");
151  }
152 
153  // In case of null pointer addr
154  if (addr == nullptr) {
155  printf(" nullptr, size: %zu\n", len);
156  return;
157  }
158 
159  // Process every byte in the data.
160  for (i = 0; i < len; i++) {
161  // Multiple of 16 means new line (with line offset).
162  if ((i % 16) == 0) {
163  // Just don't print ASCII for the zeroth line.
164  if (i != 0)
165  printf(" %s\n", buff);
166 
167  // Output the offset.
168  //printf (" %04x ", i);
169  printf(" %p ", &addr[i]);
170  }
171 
172  // Now the hex code for the specific character.
173  printf(" %02x", addr[i]);
174 
175  // And store a printable ASCII character for later.
176  if ((addr[i] < 0x20) || (addr[i] > 0x7e))
177  buff[i % 16] = '.';
178  else
179  buff[i % 16] = addr[i];
180  buff[(i % 16) + 1] = '\0';
181  fflush(stdout);
182  }
183 
184  // Pad out last line if not exactly 16 characters.
185  while ((i % 16) != 0) {
186  printf(" ");
187  fflush(stdout);
188  i++;
189  }
190 
191  // And print the final ASCII bit.
192  printf(" %s\n", buff);
193  fflush(stdout);
194 }
