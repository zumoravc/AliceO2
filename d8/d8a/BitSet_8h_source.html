55  // appends a bit at the current position (i.e. len-1)

57 
59  // appends the n first bits from a x-bits word.
60  // if n is < 0 it is computed for val (using log2(val)+1)
61  // otherwise it should be >= log2(val)+1 and <=x
62  // and the exact number of specified bits will be set
63  // (to 0 or 1).
64  // returns the number of bits actually added
65  int append(uint8_t val, int n = -1);
66  int append(uint16_t val, int n = -1);
67  int append(uint32_t val, int n = -1);
68  int append(uint64_t val, int n = -1);
70 
71  // count returns the number of bits set at 1
72  int count() const;
73 
74  // sets all the bits to false (i.e. resets)
75  void clear();
76

78  bool isEmpty() const { return len() == 0; }
79

108 
109  void setFast(int pos, bool val);

111  void setFromBytes(gsl::span<uint8_t> bytes);
112 
113  // setRangeFromString populates the bits at indice [a,b] (inclusive range)
114  // from the characters in the string: 0 to unset the bit (=false)
115  // or 1 to set the bit (=true).
116  // A string containing anything else than '0' or '1' is invalid and
117  // triggers an exception
118  void setRangeFromString(int a, int b, std::string_view s);
119 
121  // setRangeFromUint(a,b,uintX_t) populates the bits at indices [a,b] (inclusive range)

135  std::string stringLSBRight() const;
136 
137  // subset returns a subset of the bitset.
138  // subset is not a slice (i.e. not a reference, but a copy of the internals)
139  // [a,b] inclusive
140  BitSet subset(int a, int b) const;
141 
142  // uint8 converts the bit set into a 8-bits value, if possible.
143  // if b is negative, it is set to the bitset length
144  uint8_t uint8(int a, int b) const;
145 
146  // uint16 converts the bit set into a 16-bits value, if possible.
147  // if b is negative, it is set to the bitset length
148  uint16_t uint16(int a, int b) const;
149 
150  // uint32 converts the bit set into a 32-bits value, if possible.
151  // if b is negative, it is set to the bitset length
152  uint32_t uint32(int a, int b) const;
153 
154  // uint64 converts the bit set into a 64-bits value, if possible.
155  // if b is negative, it is set to the bitset length
156  uint64_t uint64(int a, int b) const;
157

163  static int nofInstances;
