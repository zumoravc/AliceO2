81 void DigitizerTask::FinishTask()
82 {
83  // finalize digitization, if needed, flash remaining digits
84  if (!mContinuous)
85  return;
86 
87  // is there QED backgroung provided? Fill QED slots up to the end of reserved ROFrames
88  if (mQEDBranch) {
89  Float_t digitROtime = 25; //ns
90  processQEDBackground(digitROtime);
91  }
92 
93  FairRootManager* mgr = FairRootManager::Instance();
94  mgr->SetLastFill(kTRUE);
95 
96  mDigitizer.finish();
97 }

99 void DigitizerTask::processQEDBackground(double tMax)
100 {
101  // process QED time-slots until provided collision time (in ns)
102 
103  double tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS;
104 
105  while (tQEDNext < tMax) {
106  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
107  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
108  if (++mLastQEDEntry >= mQEDBranch->GetEntries()) {
109  mLastQEDEntry = 0; // wrapp if needed
110  }
111  mQEDBranch->GetEntry(mLastQEDEntry);
112  mDigitizer.setEventTime(mLastQEDTimeNS);
113  mDigitizer.setEventID(mQEDSourceID);
114 
115  mDigitizer.process(mHitsArrayQED, mEventDigit);
116  }
