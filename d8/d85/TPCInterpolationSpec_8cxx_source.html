43 void TPCInterpolationDPL::run(ProcessingContext& pc)
44 {
45  mTimer.Start(false);
46  const auto tracksITS = pc.inputs().get<gsl::span<o2::its::TrackITS>>("trackITS");
47  const auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
48  const auto tracksITSTPC = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("match");
49  const auto tracksTPCClRefs = pc.inputs().get<gsl::span<o2::tpc::TPCClRefElem>>("trackTPCClRefs");
50  const auto trackMatchesTOF = pc.inputs().get<gsl::span<o2::dataformats::MatchInfoTOF>>("matchTOF"); // FIXME missing reader
51  const auto clustersTOF = pc.inputs().get<gsl::span<o2::tof::Cluster>>("clustersTOF");
52 
53  // TPC Cluster loading part is copied from TPCITSMatchingSpec.cxx
54  //---------------------------->> TPC Clusters loading >>------------------------------------------
55  int operation = 0;
56  uint64_t activeSectors = 0;
57  std::bitset<o2::tpc::Constants::MAXSECTOR> validSectors = 0;
58  std::map<int, DataRef> datarefs;
59  std::vector<InputSpec> filter = {
60  {"check", ConcreteDataTypeMatcher{"TPC", "CLUSTERNATIVE"}, Lifetime::Timeframe},
61  };
62  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
63  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
64  if (sectorHeader == nullptr) {
65  // FIXME: think about error policy
66  LOG(ERROR) << "sector header missing on header stack";
67  throw std::runtime_error("sector header missing on header stack");
68  }
69  const int& sector = sectorHeader->sector();
70  std::bitset<o2::tpc::Constants::MAXSECTOR> sectorMask(sectorHeader->sectorBits);
71  LOG(INFO) << "Reading TPC cluster data, sector mask is " << sectorMask;
72  if ((validSectors & sectorMask).any()) {
73  // have already data for this sector, this should not happen in the current
74  // sequential implementation, for parallel path merged at the tracker stage
75  // multiple buffers need to be handled
76  throw std::runtime_error("can only have one data set per sector");
77  }
78  activeSectors |= sectorHeader->activeSectors;
79  validSectors |= sectorMask;
80  datarefs[sector] = ref;
81  }
82 
83  auto printInputLog = [&validSectors, &activeSectors](auto& r, const char* comment, auto& s) {
84  LOG(INFO) << comment << " " << *(r.spec) << ", size " << DataRefUtils::getPayloadSize(r) //
85  << " for sector " << s //
86  << std::endl //
87  << " input status: " << validSectors //
88  << std::endl //
89  << " active sectors: " << std::bitset<o2::tpc::Constants::MAXSECTOR>(activeSectors);
90  };
91 
92  if (activeSectors == 0 || (activeSectors & validSectors.to_ulong()) != activeSectors) {
93  // not all sectors available
94  // Since we expect complete input, this should not happen (why does the bufferization considered for TPC CA tracker? Ask Matthias)
95  throw std::runtime_error("Did not receive TPC clusters data for all sectors");
96  }
97  //------------------------------------------------------------------------------
98  std::vector<gsl::span<const char>> clustersTPC;
99 
100  for (auto const& refentry : datarefs) {
101  auto& sector = refentry.first;
102  auto& ref = refentry.second;
103  clustersTPC.emplace_back(ref.payload, DataRefUtils::getPayloadSize(ref));
104  printInputLog(ref, "received", sector);
105  }
106 
107  // Just print TPC clusters status
108  {
109  // make human readable information from the bitfield
110  std::string bitInfo;
111  auto nActiveBits = validSectors.count();
112  if (((uint64_t)0x1 << nActiveBits) == validSectors.to_ulong() + 1) {
113  // sectors 0 to some upper bound are active
114  bitInfo = "0-" + std::to_string(nActiveBits - 1);
115  } else {
116  int rangeStart = -1;
117  int rangeEnd = -1;
118  for (size_t sector = 0; sector < validSectors.size(); sector++) {
119  if (validSectors.test(sector)) {
120  if (rangeStart < 0) {
121  if (rangeEnd >= 0) {
122  bitInfo += ",";
123  }
124  bitInfo += std::to_string(sector);
125  if (nActiveBits == 1) {
126  break;
127  }
128  rangeStart = sector;
129  }
130  rangeEnd = sector;
131  } else {
132  if (rangeStart >= 0 && rangeEnd > rangeStart) {
133  bitInfo += "-" + std::to_string(rangeEnd);
134  }
135  rangeStart = -1;
136  }
137  }
138  if (rangeStart >= 0 && rangeEnd > rangeStart) {
139  bitInfo += "-" + std::to_string(rangeEnd);
140  }
141  }
142  LOG(INFO) << "running matching for sector(s) " << bitInfo;
143  }
144 
145  o2::tpc::ClusterNativeAccess clusterIndex;
146  std::unique_ptr<o2::tpc::ClusterNative[]> clusterBuffer;
147  memset(&clusterIndex, 0, sizeof(clusterIndex));
148  o2::tpc::ClusterNativeHelper::Reader::fillIndex(clusterIndex, clusterBuffer, clustersTPC);
149  //----------------------------<< TPC Clusters loading <<------------------------------------------
150 
151  // pass input data to TrackInterpolation object
152  mInterpolation.setITSTracksInp(tracksITS);
153  mInterpolation.setTPCTracksInp(tracksTPC);
154  mInterpolation.setTPCTrackClusIdxInp(tracksTPCClRefs);
155  mInterpolation.setTPCClustersInp(&clusterIndex);
156  mInterpolation.setTOFMatchesInp(trackMatchesTOF);
157  mInterpolation.setITSTPCTrackMatchesInp(tracksITSTPC);
158  mInterpolation.setTOFClustersInp(clustersTOF);
159 
160  if (mUseMC) {
161  // possibly MC labels will be used to check filtering procedure performance before interpolation
162  // not yet implemented
163  }
164 
165  printf("TPC Interpolation Workflow initialized. Start processing...\n");
166 
167  mInterpolation.process();
168 
169  pc.outputs().snapshot(Output{"GLO", "TPCINT_TRK", 0, Lifetime::Timeframe}, mInterpolation.getReferenceTracks());
170  pc.outputs().snapshot(Output{"GLO", "TPCINT_RES", 0, Lifetime::Timeframe}, mInterpolation.getClusterResiduals());
171  mTimer.Stop();
172 }

180 DataProcessorSpec getTPCInterpolationSpec(bool useMC, const std::vector<int>& tpcClusLanes)
181 {
182  std::vector<InputSpec> inputs;
183  std::vector<OutputSpec> outputs;
184 
185  inputs.emplace_back("trackITS", "ITS", "TRACKS", 0, Lifetime::Timeframe);
186  inputs.emplace_back("trackTPC", "TPC", "TRACKS", 0, Lifetime::Timeframe);
187  inputs.emplace_back("trackTPCClRefs", "TPC", "CLUSREFS", 0, Lifetime::Timeframe);
188 
189  inputs.emplace_back("clusTPC", ConcreteDataTypeMatcher{"TPC", "CLUSTERNATIVE"}, Lifetime::Timeframe);
190 
191  inputs.emplace_back("match", "GLO", "TPCITS", 0, Lifetime::Timeframe);
192  inputs.emplace_back("matchTOF", "TOF", "MATCHINFOS", 0, Lifetime::Timeframe);
193  inputs.emplace_back("clustersTOF", "TOF", "CLUSTERS", 0, Lifetime::Timeframe);
194 
195  if (useMC) {
196  LOG(FATAL) << "MC usage must be disabled for this workflow, since it is not yet implemented";
197  // are the MC inputs from ITS-TPC matching and TOF matching duplicates? if trackITSMCTR == matchTOFMCITS one of them should be removed
198  inputs.emplace_back("trackITSMCTR", "GLO", "TPCITS_ITSMC", 0, Lifetime::Timeframe);
199  inputs.emplace_back("trackTPCMCTR", "GLO", "TPCITS_TPCMC", 0, Lifetime::Timeframe);
200  inputs.emplace_back("matchTOFMC", o2::header::gDataOriginTOF, "MATCHTOFINFOSMC", 0, Lifetime::Timeframe);
201  inputs.emplace_back("matchTOFMCTPC", o2::header::gDataOriginTOF, "MATCHTPCINFOSMC", 0, Lifetime::Timeframe);
202  inputs.emplace_back("matchTOFMCITS", o2::header::gDataOriginTOF, "MATCHITSINFOSMC", 0, Lifetime::Timeframe);
203  }
204 
205  outputs.emplace_back("GLO", "TPCINT_TRK", 0, Lifetime::Timeframe);
206  outputs.emplace_back("GLO", "TPCINT_RES", 0, Lifetime::Timeframe);
207 
208  return DataProcessorSpec{
209  "tpc-track-interpolation",
210  inputs,
211  outputs,
212  AlgorithmSpec{adaptFromTask<TPCInterpolationDPL>(useMC)},
213  Options{}};
214 }
