29 class ServiceRegistry
30 {
31  using ServicePtr = void*;
32  using ConstServicePtr = void const*;
34  constexpr static int MAX_DISTANCE = 8;
36  constexpr static int MAX_SERVICES = 256;
38  constexpr static int MAX_SERVICES_MASK = MAX_SERVICES - 1;
39 
40  public:
41  ServiceRegistry()
42  {
43  mServices.fill(std::make_pair(0L, nullptr));
44  mConstServices.fill(std::make_pair(0L, nullptr));
45  }
46  // Register a service for the given interface T
47  // with actual implementation C, i.e. C is derived from T.
48  // Only one instance of type C can be registered per type T.
49  // The fact we use a bare pointer indicates that the ownership
50  // of the service still belongs to whatever created it, and is
51  // not passed to the registry. It's therefore responsibility of
52  // the creator of the service to properly dispose it.
53  template <class I, class C>
54  void registerService(C* service)
55  {
56  // This only works for concrete implementations of the type T.
57  // We need type elision as we do not want to know all the services in
58  // advance
59  static_assert(std::is_base_of<I, C>::value == true,
60  "Registered service is not derived from declared interface");
61  auto typeHash = TypeIdHelpers::uniqueId<std::decay_t<I>>();
62  auto serviceId = typeHash & MAX_SERVICES_MASK;
63  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
64  if (mServices[i + serviceId].second == nullptr) {
65  mServices[i + serviceId].first = typeHash;
66  mServices[i + serviceId].second = reinterpret_cast<ServicePtr>(service);
67  return;
68  }
69  }
70  O2_BUILTIN_UNREACHABLE();
71  }
72 
73  template <class I, class C>
74  void registerService(C const* service)
75  {
76  // This only works for concrete implementations of the type T.
77  // We need type elision as we do not want to know all the services in
78  // advance
79  static_assert(std::is_base_of<I, C>::value == true,
80  "Registered service is not derived from declared interface");
81  auto typeHash = TypeIdHelpers::uniqueId<std::decay_t<I>>();
82  auto serviceId = typeHash & MAX_SERVICES_MASK;
83  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
84  if (mConstServices[i + serviceId].second == nullptr) {
85  mConstServices[i + serviceId].first = typeHash;
86  mConstServices[i + serviceId].second = reinterpret_cast<ConstServicePtr>(service);
87  return;
88  }
89  }
90  O2_BUILTIN_UNREACHABLE();
91  }

96  template <typename T>
97  std::enable_if_t<std::is_const_v<T> == false, T&> get() const
98  {
99  auto typeHash = TypeIdHelpers::uniqueId<std::decay_t<T>>();
100  auto serviceId = typeHash & MAX_SERVICES_MASK;
101  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
102  if (mServices[i + serviceId].first == typeHash) {
103  return *reinterpret_cast<T*>(mServices[i + serviceId].second);
104  }
105  }
106  throw std::runtime_error(std::string("Unable to find service of kind ") +
107  typeid(T).name() +
108  " did you register one as non-const reference?");
109  }

112  template <typename T>
113  std::enable_if_t<std::is_const_v<T> == false, bool> active() const
114  {
115  auto typeHash = TypeIdHelpers::uniqueId<std::decay_t<T>>();
116  auto serviceId = typeHash & MAX_SERVICES_MASK;
117  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
118  if (mServices[i + serviceId].first == typeHash) {
119  return mServices[i + serviceId].second != nullptr;
120  }
121  }
122  throw std::runtime_error(std::string("Unable to find service of kind ") +
123  typeid(T).name() +
124  " did you register one as non-const reference?");
125  }
129  template <typename T>
130  std::enable_if_t<std::is_const_v<T>, T&> get() const
131  {
132  auto typeHash = TypeIdHelpers::uniqueId<std::decay_t<T>>();
133  auto serviceId = typeHash & MAX_SERVICES_MASK;

137  }
138  }
139  throw std::runtime_error(std::string("Unable to find service of kind ") +
140  typeid(T).name() +
141  " did you register one as const reference?");
142  }

144  private:

147 };
