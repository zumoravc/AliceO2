110 void PreClusterFinder::loadDigits(gsl::span<const Digit> digits)
111 {
113 
114  for (const auto& digit : digits) {
115 
116  int deIndex = mDEIndices[digit.getDetID()];
117  assert(deIndex >= 0 && deIndex < SNDEs);
118 
119  DetectionElement& de(*(mDEs[deIndex]));
120 
121  uint16_t iPad = digit.getPadID();
122  int iPlane = (iPad < de.mapping->nPads[0]) ? 0 : 1;
123 
124  // register this digit
125  uint16_t iDigit = de.nFiredPads[0] + de.nFiredPads[1];
126  if (iDigit >= de.digits.size()) {
127  de.digits.push_back(&digit);
128  } else {
129  de.digits[iDigit] = &digit;
130  }
131  de.mapping->pads[iPad].iDigit = iDigit;
132  de.mapping->pads[iPad].useMe = true;
133 
134  // set this pad as fired
135  if (de.nFiredPads[iPlane] < de.firedPads[iPlane].size()) {
136  de.firedPads[iPlane][de.nFiredPads[iPlane]] = iPad;
137  } else {
138  de.firedPads[iPlane].push_back(iPad);
139  }
140  ++de.nFiredPads[iPlane];
141  }
142 }

153 void PreClusterFinder::getPreClusters(std::vector<o2::mch::PreCluster>& preClusters, std::vector<Digit>& digits)
154 {
158 
159  for (int iDE = 0, nDEs = SNDEs; iDE < nDEs; ++iDE) {
160 
161  DetectionElement& de(*(mDEs[iDE]));
162  if (de.nOrderedPads[1] == 0) {
163  continue;
164  }
165 
166  for (int iPlane = 0; iPlane < 2; ++iPlane) {
167  for (int iCluster = 0; iCluster < mNPreClusters[iDE][iPlane]; ++iCluster) {
168 
169  PreCluster* cluster = mPreClusters[iDE][iPlane][iCluster].get();
170  if (!cluster->storeMe) {
171  continue;
172  }
173 
174  // add this precluster
175  uint16_t firstDigit = digits.size();
176  uint16_t nDigits = cluster->lastPad - cluster->firstPad + 1;
177  preClusters.push_back({firstDigit, nDigits});
178 
179  // add the digits of this precluster
180  for (uint16_t iOrderedPad = cluster->firstPad; iOrderedPad <= cluster->lastPad; ++iOrderedPad) {
181  digits.emplace_back(*de.digits[de.mapping->pads[de.orderedPads[1][iOrderedPad]].iDigit]);
182  }
183  }
184  }
185  }
186 }

189 void PreClusterFinder::preClusterizeRecursive()
190 {
192 
193  PreCluster* cluster(nullptr);
194  uint16_t iPad(0);
195 
196  // loop over DEs
197  for (int iDE = 0; iDE < SNDEs; ++iDE) {
198 
199  DetectionElement& de(*(mDEs[iDE]));
200 
201  // loop over planes
202  for (int iPlane = 0; iPlane < 2; ++iPlane) {
203 
204  // loop over fired pads
205  for (int iFiredPad = 0; iFiredPad < de.nFiredPads[iPlane]; ++iFiredPad) {
206 
207  iPad = de.firedPads[iPlane][iFiredPad];
208 
209  if (de.mapping->pads[iPad].useMe) {
210 
211  // create the precluster if needed
212  if (mNPreClusters[iDE][iPlane] >= mPreClusters[iDE][iPlane].size()) {
213  mPreClusters[iDE][iPlane].push_back(std::make_unique<PreCluster>());
214  }
215 
216  // get the precluster
217  cluster = mPreClusters[iDE][iPlane][mNPreClusters[iDE][iPlane]].get();
218  ++mNPreClusters[iDE][iPlane];
219 
220  // reset its content
221  cluster->area[0][0] = 1.e6;
222  cluster->area[0][1] = -1.e6;
223  cluster->area[1][0] = 1.e6;
224  cluster->area[1][1] = -1.e6;
225  cluster->useMe = true;
226  cluster->storeMe = false;
227 
228  // add the pad and its fired neighbours recusively
229  cluster->firstPad = de.nOrderedPads[0];
230  addPad(de, iPad, *cluster);
231  }
232  }
233  }
234  }

238 void PreClusterFinder::addPad(DetectionElement& de, uint16_t iPad, PreCluster& cluster)
239 {
241 
242  Mapping::MpPad* pads(de.mapping->pads.get());
243 
244  // add the given pad
245  Mapping::MpPad& pad(pads[iPad]);
246  if (de.nOrderedPads[0] < de.orderedPads[0].size()) {
247  de.orderedPads[0][de.nOrderedPads[0]] = iPad;
248  } else {
249  de.orderedPads[0].push_back(iPad);
250  }
251  cluster.lastPad = de.nOrderedPads[0];
252  ++de.nOrderedPads[0];
253  if (pad.area[0][0] < cluster.area[0][0])
254  cluster.area[0][0] = pad.area[0][0];
255  if (pad.area[0][1] > cluster.area[0][1])
256  cluster.area[0][1] = pad.area[0][1];
257  if (pad.area[1][0] < cluster.area[1][0])
258  cluster.area[1][0] = pad.area[1][0];
259  if (pad.area[1][1] > cluster.area[1][1])
260  cluster.area[1][1] = pad.area[1][1];
261 
262  pad.useMe = false;
263 
264  // loop over its neighbours
265  for (int iNeighbour = 0; iNeighbour < pad.nNeighbours; ++iNeighbour) {
266 
267  if (pads[pad.neighbours[iNeighbour]].useMe) {
268 
269  // add the pad to the precluster
270  addPad(de, pad.neighbours[iNeighbour], cluster);
271  }
272  }
273 }

276 int PreClusterFinder::mergePreClusters()
277 {
280 
281  PreCluster* cluster(nullptr);
282  int nPreClusters(0);
283 
284  // loop over DEs
285  for (int iDE = 0; iDE < SNDEs; ++iDE) {
286 
287  DetectionElement& de(*(mDEs[iDE]));
288 
289  // loop over preclusters of one plane
290  for (int iCluster = 0; iCluster < mNPreClusters[iDE][0]; ++iCluster) {
291 
292  if (!mPreClusters[iDE][0][iCluster]->useMe) {
293  continue;
294  }
295 
296  cluster = mPreClusters[iDE][0][iCluster].get();
297  cluster->useMe = false;
298 
299  // look for overlapping preclusters in the other plane
300  PreCluster* mergedCluster(nullptr);
301  mergePreClusters(*cluster, mPreClusters[iDE], mNPreClusters[iDE], de, 1, mergedCluster);
302 
303  // add the current one
304  if (!mergedCluster) {
305  mergedCluster = usePreClusters(cluster, de);
306  } else {
307  mergePreClusters(*mergedCluster, *cluster, de);
308  }
309 
310  ++nPreClusters;
311  }
312 
313  // loop over preclusters of the other plane
314  for (int iCluster = 0; iCluster < mNPreClusters[iDE][1]; ++iCluster) {
315 
316  if (!mPreClusters[iDE][1][iCluster]->useMe) {
317  continue;
318  }
319 
320  // all remaining preclusters have to be stored
321  usePreClusters(mPreClusters[iDE][1][iCluster].get(), de);
322 
323  ++nPreClusters;
324  }
325  }
326 
327  return nPreClusters;
328 }

331 void PreClusterFinder::mergePreClusters(PreCluster& cluster, std::vector<std::unique_ptr<PreCluster>> preClusters[2],
332  int nPreClusters[2], DetectionElement& de, int iPlane,
333  PreCluster*& mergedCluster)
334 {
336 
337  // overlap precision in cm: positive(negative) = increase(decrease) precluster size
338  constexpr float overlapPrecision = -1.e-4;
339 
340  PreCluster* cluster2(nullptr);
341 
342  // loop over preclusters in the given plane
343  for (int iCluster = 0; iCluster < nPreClusters[iPlane]; ++iCluster) {
344 
345  if (!preClusters[iPlane][iCluster]->useMe) {
346  continue;
347  }
348 
349  cluster2 = preClusters[iPlane][iCluster].get();
350  if (Mapping::areOverlapping(cluster.area, cluster2->area, overlapPrecision) &&
351  areOverlapping(cluster, *cluster2, de, overlapPrecision)) {
352 
353  cluster2->useMe = false;
354 
355  // look for new overlapping preclusters in the other plane
356  mergePreClusters(*cluster2, preClusters, nPreClusters, de, (iPlane + 1) % 2, mergedCluster);
357 
358  // store overlapping preclusters and merge them
359  if (!mergedCluster) {
360  mergedCluster = usePreClusters(cluster2, de);
361  } else {
362  mergePreClusters(*mergedCluster, *cluster2, de);
363  }
364  }
365  }
366 }

369 PreClusterFinder::PreCluster* PreClusterFinder::usePreClusters(PreCluster* cluster, DetectionElement& de)
370 {
372 
373  uint16_t firstPad = de.nOrderedPads[1];
374 
375  // move the fired pads
376  for (int iOrderPad = cluster->firstPad; iOrderPad <= cluster->lastPad; ++iOrderPad) {
377 
378  if (de.nOrderedPads[1] < de.orderedPads[1].size()) {
379  de.orderedPads[1][de.nOrderedPads[1]] = de.orderedPads[0][iOrderPad];
380  } else {
381  de.orderedPads[1].push_back(de.orderedPads[0][iOrderPad]);
382  }
383 
384  ++de.nOrderedPads[1];
385  }
386 
387  cluster->firstPad = firstPad;
388  cluster->lastPad = de.nOrderedPads[1] - 1;
389 
390  cluster->storeMe = true;
391 
392  return cluster;
393 }

416 bool PreClusterFinder::areOverlapping(PreCluster& cluster1, PreCluster& cluster2, DetectionElement& de, float precision)
417 {
420 
421  // loop over all pads of the precluster1
422  for (int iOrderPad1 = cluster1.firstPad; iOrderPad1 <= cluster1.lastPad; ++iOrderPad1) {
423 
424  // loop over all pads of the precluster2
425  for (int iOrderPad2 = cluster2.firstPad; iOrderPad2 <= cluster2.lastPad; ++iOrderPad2) {
426 
427  if (Mapping::areOverlapping(de.mapping->pads[de.orderedPads[0][iOrderPad1]].area,
428  de.mapping->pads[de.orderedPads[0][iOrderPad2]].area, precision)) {
429  return true;
430  }
431  }
432  }
433 
434  return false;
435 }

438 void PreClusterFinder::createMapping()
439 {
441 
442  auto tStart = std::chrono::high_resolution_clock::now();
443 
444  std::vector<std::unique_ptr<Mapping::MpDE>> mpDEs = Mapping::createMapping();
445 
446  if (mpDEs.size() != SNDEs) {
447  throw runtime_error("invalid mapping");
448  }
449 
450  mDEIndices.reserve(SNDEs);
451 
452  for (int iDE = 0; iDE < SNDEs; ++iDE) {
453 
454  DetectionElement& de(*(mDEs[iDE]));
455 
456  de.mapping = std::move(mpDEs[iDE]);
457 
458  mDEIndices.emplace(de.mapping->uid, iDE);
459 
460  int initialSize = (de.mapping->nPads[0] / 10 + de.mapping->nPads[1] / 10); // 10 % occupancy
461 
462  de.digits.reserve(initialSize);
463  de.nOrderedPads[0] = 0;
464  de.orderedPads[0].reserve(initialSize);
465  de.nOrderedPads[1] = 0;
466  de.orderedPads[1].reserve(initialSize);
467 
468  for (int iPlane = 0; iPlane < 2; ++iPlane) {
469  de.nFiredPads[iPlane] = 0;
470  de.firedPads[iPlane].reserve(de.mapping->nPads[iPlane] / 10); // 10% occupancy
471  }
472  }
473 
474  auto tEnd = std::chrono::high_resolution_clock::now();
475  LOG(INFO) << "create mapping in: " << std::chrono::duration<double, std::milli>(tEnd - tStart).count() << " ms";
476 }
