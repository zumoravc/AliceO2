92 //
93 // DataCompression tools
94 //
95 // The Alphabet: for every parameter to be stored in a compressed format, there
96 // is a range of valid symbols referred to be an alphabet. The DataCompression
97 // framework requires an implementation of an alphabet which is then used with
98 // the probability model and codecs. The alphabet class needs to implement a
99 // check function (isValid), and a forward iterator together with begin and end
100 // function to hook it up to the framework.
101 //
102 // Default Alphabet implementations: most of the alphabets will describe a range
103 // of contiguous values, which is implemented by the class ContiguousAlphabet.
104 // Two specializations describe symbols in a range from 0 to a max value and
105 // within a certain bit range, respectively
106 // ZeroBoundContiguousAlphabet and BitRangeContiguousAlphabet
107 
108 /******************************************************************************
109  * @class ExampleAlphabet
110  * @brief An example for an Alphabet definition.
111  *
112  * A class definition for an alphabet. Note that the functions are only defined,
113  * not implemented. The class is required to support the following functions to
114  * hook it up onto the DataCompression framework:
115  * - isValid validity of a value
116  * - getIndex get index from value
117  * - getSymbol get symbol from index
118  * - forward iterator class to walk through elements of alphabet
119  * - begin start iteration over elements
120  * - end end marker for iteration
121  *
122  * The alphabet has to provide an index for the symbol range such that the
123  * framework can build a one to one relation between symbols and index values
124  * used for internal mapping of the symbols.
125  */

193 /******************************************************************************
194  * Definition of a contiguous alphabet
195  *
196  * The contiguous alphabet defines integer number elements of the specified
197  * type in a given range.
198  *
199  * TODO: the mpl string length is limited and can be configured with the
200  * define BOOST_MPL_LIMIT_STRING_SIZE (somehow the number of allowed
201  * template arguments is given by this number divided by four)
202  */
203 template <typename T, T _min, T _max, //
204  typename NameT = boost::mpl::string<'U', 'n', 'n', 'a', 'm', 'e', 'd'>::type //
205  >
206 class ContiguousAlphabet
207 {
208  public:
209  ContiguousAlphabet() = default;
210  ~ContiguousAlphabet() = default;
211 
212  using value_type = T;
213  using size_type = T;
214  using range = boost::mpl::range_c<T, _min, _max>;
215  using size = boost::mpl::plus<boost::mpl::size<range>, boost::mpl::int_<1>>;
216 
218  static bool isValid(value_type v) { return v >= _min && v <= _max; }
219 
225  static unsigned getIndex(value_type symbol)
226  {
227  int index = symbol;
228  if (_min < 0) {
229  index += -_min;
230  } else if (_min > 0) {
231  index -= _min;
232  }
233  return index;
234  }
235 
237  static value_type getSymbol(unsigned index) { return _min + index; }
238 
240  constexpr unsigned getIndexRange() { return _max - _min; }
241 
245  constexpr const char* getName() const { return boost::mpl::c_str<NameT>::value; }

247  template <typename ValueT>
248  using _iterator_base = std::iterator<std::forward_iterator_tag, ValueT>;
249 
251  template <typename ValueT>
252  class Iterator : public _iterator_base<ValueT>
253  {
254  public:
255  Iterator() : mValue(_max), mIsEnd(true) {}
256  Iterator(T value, bool isEnd) : mValue(value), mIsEnd(isEnd) {}
257  ~Iterator() = default;
258 
259  using self_type = Iterator;
260  using value_type = typename _iterator_base<ValueT>::value_type;
261  using reference = typename _iterator_base<ValueT>::reference;
262  using pointer = typename _iterator_base<ValueT>::pointer;
263 
264  // prefix increment
265  self_type& operator++()
266  {
267  if (mValue < _max) {
268  mValue++;
269  } else {
270  mIsEnd = true;
271  }
272  return *this;
273  }
274 
275  // postfix increment
276  self_type operator++(int /*unused*/)
277  {
278  self_type copy(*this);
279  ++*this;
280  return copy;
281  }
282 
283  // addition
284  self_type operator+(size_type n) const
285  {
286  self_type copy(*this);
287  if (!copy.mIsEnd) {
288  if ((n > _max) || (_max - n < mValue)) {
289  copy.mIsEnd = true;
290  copy.mValue = _max;
291  } else {
292  copy.mValue += n;
293  }
294  }
295  return copy;
296  }
297 
298  reference operator*() { return mValue; }
299  // pointer operator->() const {return &mValue;}
300  // reference operator[](size_type n) const;
301 
302  bool operator==(const self_type& other) { return mValue == other.mValue && mIsEnd == other.mIsEnd; }
303  bool operator!=(const self_type& other) { return not(*this == other); }
304 
305  private:
306  value_type mValue;
307  bool mIsEnd;
308  };

356 template <class Alphabet, typename WeightType = double>
357 class ProbabilityModel
358 {
359  public:
360  using alphabet_type = Alphabet;
361  using value_type = typename Alphabet::value_type;
362  using weight_type = WeightType;
363  using TableType = std::map<value_type, WeightType>;
364 
365  // TODO: check if this is the correct way for WeightType defaults
366  static const value_type _default0 = 0;
367  static const value_type _default1 = _default0 + 1;
368 
369  ProbabilityModel() : mProbabilityTable(), mTotalWeight(_default0) {}
370  ~ProbabilityModel() = default;
371 
372  constexpr const char* getName() const
373  {
374  Alphabet tmp;
375  return tmp.getName();
376  }
377 
378  int addWeight(value_type value, weight_type weight = _default1)
379  {
380  mProbabilityTable[value] += weight;
381  mTotalWeight += weight;
382  return 0;
383  }
384 
385  int initWeight(Alphabet& alphabet, WeightType weight = _default1)
386  {
387  mProbabilityTable.clear();
388  mTotalWeight = _default0;
389  for (auto i : alphabet) {
390  addWeight(i, weight);
391  }
392  return 0;
393  }
394 
395  WeightType normalize()
396  {
397  WeightType totalWeight = _default0;
398  // TODO: handle division by zero, although that should not occur at all
399  for (typename TableType::iterator i = mProbabilityTable.begin(); i != mProbabilityTable.end(); i++) {
400  totalWeight += i->second;
401  i->second /= mTotalWeight;
402  }
403  // TODO: verify total weight
404  // if (mTotalWeight - verifyTotalWeight > some small value)
405  mTotalWeight = totalWeight / mTotalWeight;
406  return totalWeight;
407  }
408 
409  // const reference only to avoid changes in the weight count
410  // without registering in the total weight as well
411  const WeightType& operator[](value_type v) const
412  {
413  typename TableType::const_iterator i = mProbabilityTable.find(v);
414  if (i != mProbabilityTable.end()) {
415  return i->second;
416  }
417  static WeightType dummy = _default0;
418  return dummy;
419  }
420 
421  typename TableType::const_iterator begin() const { return mProbabilityTable.begin(); }
422 
423  typename TableType::const_iterator end() const { return mProbabilityTable.end(); }
424 
425  typename TableType::iterator begin() { return mProbabilityTable.begin(); }
426 
427  typename TableType::iterator end() { return mProbabilityTable.end(); }
428 
429  void print() const {}

432  TableType mProbabilityTable;
433  WeightType mTotalWeight;
434 };
