43 void Encoder::hbTrigger(const InteractionRecord& ir)
44 {
46  if (mLastIR.isDummy()) {
47  // This is the first HB
48  for (uint16_t feeId = 0; feeId < crateparams::sNGBTs; ++feeId) {
49  mGBTEncoders[feeId].processTrigger(o2::constants::lhc::LHCMaxBunches, raw::sORB);
50  }
51  mLastIR = o2::raw::HBFUtils::Instance().getFirstIR();
52  return;
53  }
54 
55  std::vector<InteractionRecord> HBIRVec;
56  o2::raw::HBFUtils::Instance().fillHBIRvector(HBIRVec, mLastIR + 1, ir);
57  for (auto& hbIr : HBIRVec) {
58  for (uint16_t feeId = 0; feeId < crateparams::sNGBTs; ++feeId) {
59  flush(feeId, mLastIR);
60  mGBTEncoders[feeId].processTrigger(o2::constants::lhc::LHCMaxBunches, raw::sORB);
61  }
62  mLastIR = hbIr;
63  }
64  mLastIR = ir;
65 }

67 void Encoder::flush(uint16_t feeId, const InteractionRecord& ir)
68 {
70 
71  if (mGBTEncoders[feeId].getBufferSize() == 0) {
72  return;
73  }
74  size_t dataSize = mGBTEncoders[feeId].getBufferSize();
75  size_t resto = dataSize % o2::raw::RDHUtils::GBTWord;
76  if (dataSize % o2::raw::RDHUtils::GBTWord) {
77  dataSize += o2::raw::RDHUtils::GBTWord - resto;
78  }
79  std::vector<char> buf(dataSize);
80  memcpy(buf.data(), mGBTEncoders[feeId].getBuffer().data(), mGBTEncoders[feeId].getBufferSize());
81  mRawWriter.addData(feeId, mFEEIdConfig.getCRUId(mGBTIds[feeId]), mFEEIdConfig.getLinkId(mGBTIds[feeId]), mFEEIdConfig.getEndPointId(mGBTIds[feeId]), ir, buf);
82  mGBTEncoders[feeId].clear();
83 }
