73 void ClustererTask::run(const std::string inpName, const std::string outName)
74 {
75  // standalone execution
76  Init(); // create reader, clusterer
77 
78  if (mRawDataMode) {
79 
80  mReaderRaw->openInput(inpName);
81  mClusterer.process(1, *mReaderRaw.get(), &mFullClus, &mCompClus, &mPatterns, &mROFRecVec, nullptr);
82 
83  auto basename = outName.substr(0, outName.size() - sizeof("root"));
84  auto nFiles = int(mROFRecVec.size() / maxROframe);
85  int i = 0;
86  for (; i < nFiles; i++) {
87  writeTree(basename, i);
88  }
89  writeTree(basename, i); // The remainder
90 
91  } else {
92 
93  mReaderMC->openInput(inpName, o2::detectors::DetID("ITS"));
94 
95  TFile outFile(outName.data(), "new");
96  if (!outFile.IsOpen()) {
97  LOG(FATAL) << "Failed to open output file " << outName;
98  }
99 
100  TTree outTree("o2sim", "ITS Clusters");
101 
102  auto fullClusPtr = &mFullClus;
103  if (mClusterer.getWantFullClusters()) {
104  outTree.Branch("ITSCluster", &fullClusPtr);
105  } else {
106  LOG(INFO) << Class()->GetName() << " output of full clusters is not requested";
107  }
108 
109  auto compClusPtr = &mCompClus;
110  outTree.Branch("ITSClusterComp", &compClusPtr);
111 
112  auto rofRecVecPtr = &mROFRecVec;
113  outTree.Branch("ITSClustersROF", &rofRecVecPtr);
114 
115  auto clsLabelsPtr = &mClsLabels;
116  if (mUseMCTruth && mReaderMC->getDigitsMCTruth()) {
117  // digit labels are provided directly to clusterer
118  outTree.Branch("ITSClusterMCTruth", &clsLabelsPtr);
119  } else {
120  mUseMCTruth = false;
121  }
122  LOG(INFO) << Class()->GetName() << " | MCTruth: " << (mUseMCTruth ? "ON" : "OFF");
123 
124  outTree.Branch("ITSClusterPatt", &mPatterns);
125 
126  std::vector<o2::itsmft::MC2ROFRecord> mc2rof, *mc2rofPtr = &mc2rof;
127  if (mUseMCTruth) {
128  auto mc2rofOrig = mReaderMC->getMC2ROFRecords();
129  mc2rof.reserve(mc2rofOrig.size());
130  for (const auto& m2r : mc2rofOrig) { // clone from the span
131  mc2rof.push_back(m2r);
132  }
133  outTree.Branch("ITSClustersMC2ROF", mc2rofPtr);
134  }
135 
136  // loop over entries of the input tree
137  while (mReaderMC->readNextEntry()) {
138  mClusterer.process(1, *mReaderMC.get(), &mFullClus, &mCompClus, &mPatterns, &mROFRecVec, &mClsLabels);
139  }
140 
141  outTree.Fill();
142  outTree.Write();
143  }
144 
145  mClusterer.clear();
146 }

148 void ClustererTask::writeTree(std::string basename, int i)
149 {
150  auto name = basename + std::to_string(i) + ".root";
151  TFile outFile(name.data(), "new");
152  if (!outFile.IsOpen()) {
153  LOG(FATAL) << "Failed to open output file " << name;
154  }
155  TTree outTree("o2sim", "ITS Clusters");
156 
157  auto max = (i + 1) * maxROframe;
158  auto lastf = (max < mROFRecVec.size()) ? mROFRecVec.begin() + max : mROFRecVec.end();
159  std::vector<o2::itsmft::ROFRecord> rofRecBuffer(mROFRecVec.begin() + i * maxROframe, lastf);
160  std::vector<o2::itsmft::ROFRecord>* rofRecPtr = &rofRecBuffer;
161  outTree.Branch("ITSClustersROF", rofRecPtr);
162 
163  auto first = rofRecBuffer[0].getFirstEntry();
164  auto last = rofRecBuffer.back().getFirstEntry() + rofRecBuffer.back().getNEntries();
165 
166  std::vector<Cluster> fullClusBuffer, *fullClusPtr = &fullClusBuffer;
167  if (mClusterer.getWantFullClusters()) {
168  fullClusBuffer.assign(&mFullClus[first], &mFullClus[last]);
169  outTree.Branch("ITSCluster", &fullClusPtr);
170  } else {
171  LOG(INFO) << Class()->GetName() << " output of full clusters is not requested";
172  }
173 
174  std::vector<CompClusterExt> compClusBuffer, *compClusPtr = &compClusBuffer;
175  compClusBuffer.assign(&mCompClus[first], &mCompClus[last]);
176  outTree.Branch("ITSClusterComp", &compClusPtr);
177  outTree.Branch("ITSClusterPatt", &mPatterns);
178 
179  for (auto& rof : rofRecBuffer) {
180  rof.setFirstEntry(rof.getFirstEntry() - first);
181  }
182 
183  outTree.Fill();
184  outTree.Write();
185 }
