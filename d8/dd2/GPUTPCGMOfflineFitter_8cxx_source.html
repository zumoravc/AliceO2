74 void GPUTPCGMOfflineFitter::RefitTrack(GPUTPCGMMergedTrack& track, const GPUTPCGMPolynomialField* field, GPUTPCGMMergedTrackHit* clusters)
75 {
76  // copy of HLT RefitTrack() with calling of the offline fit utilities
77 
78  if (!track.OK()) {
79  return;
80  }
81 
82  int nTrackHits = track.NClusters();
83  cout << "call FitOffline .. " << endl;
84  bool ok = FitOffline(field, track, clusters + track.FirstClusterRef(), nTrackHits);
85  cout << ".. end of call FitOffline " << endl;
86 
87  GPUTPCGMTrackParam t = track.Param();
88  float Alpha = track.Alpha();
89 
90  if (fabsf(t.QPt()) < 1.e-4) {
91  t.QPt() = 1.e-4;
92  }
93 
94  track.SetOK(ok);
95  track.SetNClustersFitted(nTrackHits);
96  track.Param() = t;
97  track.Alpha() = Alpha;
98 
99  {
100  int ind = track.FirstClusterRef();
101  float alphaa = fCAParam.Alpha(clusters[ind].slice);
102  float xx = clusters[ind].fX;
103  float yy = clusters[ind].fY;
104  float zz = clusters[ind].fZ - track.Param().GetZOffset();
105  float sinA = CAMath::Sin(alphaa - track.Alpha());
106  float cosA = CAMath::Cos(alphaa - track.Alpha());
107  track.SetLastX(xx * cosA - yy * sinA);
108  track.SetLastY(xx * sinA + yy * cosA);
109  track.SetLastZ(zz);
110  }
111 }

133 bool GPUTPCGMOfflineFitter::FitOffline(const GPUTPCGMPolynomialField* field, GPUTPCGMMergedTrack& gmtrack, GPUTPCGMMergedTrackHit* clusters, int& N)
134 {
135  const float maxSinPhi = GPUCA_MAX_SIN_PHI;
136 
137  int maxN = N;
138  float covYYUpd = 0.;
139  float lastUpdateX = -1.;
140 
141  const bool rejectChi2ThisRound = 0;
142  const bool markNonFittedClusters = 0;
143  const double kDeg2Rad = 3.14159265358979323846 / 180.;
144  const float maxSinForUpdate = CAMath::Sin(70. * kDeg2Rad);
145 
146  bool ok = 1;
147 
148  AliTPCtracker::SetIteration(2);
149 
150  AliTPCseed seed;
151  gmtrack.Param().GetExtParam(seed, gmtrack.Alpha());
152 
153  AliTPCtracker::AddCovariance(&seed);
154 
155  N = 0;
156  lastUpdateX = -1;
157 
158  // find last leg
159  int ihitStart = 0;
160  for (int ihit = 0; ihit < maxN; ihit++) {
161  if (clusters[ihit].leg != clusters[ihitStart].leg) {
162  ihitStart = ihit;
163  }
164  }
165 
166  for (int ihit = ihitStart; ihit < maxN; ihit++) {
167  if (clusters[ihit].fState < 0) {
168  continue; // hit is excluded from fit
169  }
170  float xx = clusters[ihit].fX;
171  float yy = clusters[ihit].fY;
172  float zz = clusters[ihit].fZ;
173 
174  if (DOUBLE && ihit + 1 >= 0 && ihit + 1 < maxN && clusters[ihit].row == clusters[ihit + 1].row) {
175  float count = 1.;
176  do {
177  if (clusters[ihit].slice != clusters[ihit + 1].slice || clusters[ihit].leg != clusters[ihit + 1].leg || fabsf(clusters[ihit].fY - clusters[ihit + 1].fY) > 4. || fabsf(clusters[ihit].fZ - clusters[ihit + 1].fZ) > 4.) {
178  break;
179  }
180  ihit += 1;
181  xx += clusters[ihit].fX;
182  yy += clusters[ihit].fY;
183  zz += clusters[ihit].fZ;
184  count += 1.;
185  } while (ihit + 1 >= 0 && ihit + 1 < maxN && clusters[ihit].row == clusters[ihit + 1].row);
186  xx /= count;
187  yy /= count;
188  zz /= count;
189  }
190 
191  // Create AliTPCclusterMI for the hit
192 
193  AliTPCclusterMI cluster;
194  Int_t tpcindex = CreateTPCclusterMI(clusters[ihit], cluster);
195  if (tpcindex < 0) {
196  continue;
197  }
198  Double_t sy2 = 0, sz2 = 0;
199  AliTPCtracker::ErrY2Z2(&seed, &cluster, sy2, sz2);
200  cluster.SetSigmaY2(sy2);
201  cluster.SetSigmaZ2(sz2);
202  cluster.SetQ(10);
203  cluster.SetMax(10);
204 
205  Int_t iRow = clusters[ihit].row;
206 
207  if (iRow < AliHLTTPCGeometry::GetNRowLow()) {
208  AliTPCtracker::fSectors = AliTPCtracker::fInnerSec;
209  } else {
210  AliTPCtracker::fSectors = AliTPCtracker::fOuterSec;
211  }
212 
213  seed.SetClusterIndex2(iRow, tpcindex);
214  seed.SetClusterPointer(iRow, &cluster);
215  seed.SetCurrentClusterIndex1(tpcindex);
216 
217  int retVal;
218  float threshold = 3. + (lastUpdateX >= 0 ? (fabsf(seed.GetX() - lastUpdateX) / 2) : 0.);
219  if (N > 2 && (fabsf(yy - seed.GetY()) > threshold || fabsf(zz - seed.GetZ()) > threshold)) {
220  retVal = 2;
221  } else {
222  Int_t err = !(AliTPCtracker::FollowToNext(seed, iRow));
223 
224  const int err2 = N > 0 && CAMath::Abs(seed.GetSnp()) >= maxSinForUpdate;
225  if (err || err2) {
226  if (markNonFittedClusters) {
227  if (N > 0 && (fabsf(yy - seed.GetY()) > 3 || fabsf(zz - seed.GetZ()) > 3)) {
228  clusters[ihit].fState = -2;
229  } else if (err && err >= -3) {
230  clusters[ihit].fState = -1;
231  }
232  }
233  continue;
234  }
235 
236  // retVal = prop.Update( yy, zz, clusters[ihit].row, param, rejectChi2ThisRound);
237  retVal = 0;
238  }
239 
240  if (retVal == 0) // track is updated
241  {
242  lastUpdateX = seed.GetX();
243  covYYUpd = seed.GetCovariance()[0];
244  ihitStart = ihit;
245  N++;
246  } else if (retVal == 2) { // cluster far away form the track
247  if (markNonFittedClusters) {
248  clusters[ihit].fState = -2;
249  }
250  } else {
251  break; // bad chi2 for the whole track, stop the fit
252  }
253  } // end loop over clusters
254 
255  GPUTPCGMTrackParam t;
256  t.SetExtParam(seed);
257 
258  float Alpha = seed.GetAlpha();
259 
260  t.ConstrainSinPhi();
261 
262  bool ok1 = N >= GPUCA_TRACKLET_SELECTOR_MIN_HITS(t.GetQPt()) && t.CheckNumericalQuality(covYYUpd);
263  if (!ok1) {
264  return (false);
265  }
266 
267  // const float kDeg2Rad = 3.1415926535897 / 180.f;
268  const float kSectAngle = 2 * 3.1415926535897 / 18.f;
269 
270  if (fCAParam.GetTrackReferenceX() <= 500) {
271  GPUTPCGMPropagator prop;
272  prop.SetMaterialTPC();
273  prop.SetPolynomialField(field);
274  prop.SetMaxSinPhi(maxSinPhi);
275  prop.SetToyMCEventsFlag(fCAParam.ToyMCEventsFlag());
276 
277  for (int k = 0; k < 3; k++) // max 3 attempts
278  {
279  int err = prop.PropagateToXAlpha(fCAParam.GetTrackReferenceX(), Alpha, 0);
280  t.ConstrainSinPhi();
281  if (fabsf(t.GetY()) <= t.GetX() * tan(kSectAngle / 2.f)) {
282  break;
283  }
284  float dAngle = floor(atan2(t.GetY(), t.GetX()) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
285  Alpha += dAngle;
286  if (err || k == 2) {
287  t.Rotate(dAngle);
288  break;
289  }
290  }
291  } else if (fabsf(t.GetY()) > t.GetX() * tan(kSectAngle / 2.f)) {
292  float dAngle = floor(atan2(t.GetY(), t.GetX()) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
293  t.Rotate(dAngle);
294  Alpha += dAngle;
295  }
296  if (Alpha > 3.1415926535897) {
297  Alpha -= 2 * 3.1415926535897;
298  } else if (Alpha <= -3.1415926535897) {
299  Alpha += 2 * 3.1415926535897;
300  }
301 
302  gmtrack.Param() = t;
303  gmtrack.Alpha() = Alpha;
304 
305  return (ok);
306 }
