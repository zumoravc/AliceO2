52 TBranch* getOrMakeBranch(TTree& tree, std::string basename, int sector, T* ptr)
53 {
54  std::stringstream stream;
55  stream << basename << "_" << sector;
56  const auto brname = stream.str();
57  if (auto br = tree.GetBranch(brname.c_str())) {
58  br->SetAddress(static_cast<void*>(&ptr));
59  return br;
60  }
61  // otherwise make it
62  return tree.Branch(brname.c_str(), ptr);
63 }

68 DataProcessorSpec getTPCDigitRootWriterSpec(std::vector<int> const& laneConfiguration, bool mctruth)
69 {
70  auto initFunction = [](InitContext& ic) {
71  // get the option from the init context
72  auto filename = ic.options().get<std::string>("tpc-digit-outfile");
73  auto treename = ic.options().get<std::string>("treename");
74 
75  auto outputfile = std::make_shared<TFile>(filename.c_str(), "RECREATE");
76  auto outputtree = std::make_shared<TTree>(treename.c_str(), treename.c_str());
77 
78  // container for cached grouping of digits
79  // TODO: once InputRecord::ReturnType helper has been merged
80  // using DigitGroupRefInputContainer = InputRecord::ReturnType<std::vector<DigiGroupRef>>;
81  using DigitGroupRefInputContainer = decltype(std::declval<InputRecord>().get<std::vector<DigiGroupRef>>(DataRef{nullptr, nullptr, nullptr}));
82  auto trigP2Sect = std::make_shared<std::array<DigitGroupRefInputContainer, 36>>();
83 
84  // the callback to be set as hook at stop of processing for the framework
85  auto finishWriting = [outputfile, outputtree]() {
86  // check/verify number of entries (it should be same in all branches)
87 
88  // will return a TObjArray
89  const auto brlist = outputtree->GetListOfBranches();
90  int entries = -1; // init to -1 (as unitialized)
91  for (TObject* entry : *brlist) {
92  auto br = static_cast<TBranch*>(entry);
93  int brentries = br->GetEntries();
94  if (entries == -1) {
95  entries = brentries;
96  } else {
97  if (brentries != entries && !TString(br->GetName()).Contains("CommonMode")) {
98  LOG(WARNING) << "INCONSISTENT NUMBER OF ENTRIES IN BRANCH " << br->GetName() << ": " << entries << " vs " << brentries;
99  }
100  }
101  }
102  if (entries > 0) {
103  outputtree->SetEntries(entries);
104  }
105  outputtree->Write();
106  outputfile->Close();
107  };
108  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
109 
110  // set up the processing function
111  // using by-copy capture of the worker instance shared pointer
112  // the shared pointer makes sure to clean up the instance when the processing
113  // function gets out of scope
114  auto processingFct = [outputfile, outputtree, trigP2Sect](ProcessingContext& pc) {
115  // extracts the sector from header of an input
116  auto extractSector = [&pc](auto const& ref) {
117  auto sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
118  if (!sectorHeader) {
119  throw std::runtime_error("Missing sector header in TPC data");
120  }
121  // the TPCSectorHeader now allows to transport information for more than one sector,
122  // e.g. for transporting clusters in one single data block. The digitization is however
123  // only on sector level
124  if (sectorHeader->sector() >= TPCSectorHeader::NSectors) {
125  throw std::runtime_error("Digitizer can only work on single sectors");
126  }
127  return sectorHeader->sector();
128  };
129 
130  // read the trigger data first
131  {
132  std::vector<InputSpec> filter = {
133  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "DIGTRIGGERS"}, Lifetime::Timeframe},
134  };
135  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
136  auto sector = extractSector(ref);
137  auto const* dh = DataRefUtils::getHeader<DataHeader*>(ref);
138  LOG(INFO) << "HAVE TRIGGER DATA FOR SECTOR " << sector << " ON CHANNEL " << dh->subSpecification;
139  if (sector >= 0) {
140  auto triggers = pc.inputs().get<std::vector<DigiGroupRef>>(ref);
141  (*trigP2Sect.get())[sector] = std::move(triggers);
142  const auto& trigS = (*trigP2Sect.get())[sector];
143  LOG(INFO) << "GOT Triggers of sector " << sector << " | SIZE " << trigS.size();
144  }
145  }
146  }
147 
148  {
149  // probe which channel has data and of what kind
150  // a) probe for digits:
151  std::vector<InputSpec> filter = {
152  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "DIGITS"}, Lifetime::Timeframe},
153  };
154  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
155  auto sector = extractSector(ref);
156  auto const* dh = DataRefUtils::getHeader<DataHeader*>(ref);
157  LOG(INFO) << "HAVE DIGIT DATA FOR SECTOR " << sector << " ON CHANNEL " << dh->subSpecification;
158  if (sector >= 0) {
159  // the digits
160  auto digiData = pc.inputs().get<gsl::span<o2::tpc::Digit>>(ref);
161  LOG(INFO) << "DIGIT SIZE " << digiData.size();
162  const auto& trigS = (*trigP2Sect.get())[sector];
163  if (!trigS.size()) {
164  LOG(FATAL) << "Digits for sector " << sector << " are received w/o info on grouping in triggers";
165  } else { // check consistency of Ndigits with that of expected from the trigger
166  int nExp = trigS.back().getFirstEntry() + trigS.back().getEntries() - trigS.front().getFirstEntry();
167  if (nExp != digiData.size()) {
168  LOG(ERROR) << "Number of digits " << digiData.size() << " is inconsistent with expectation " << nExp
169  << " from digits grouping for sector " << sector;
170  }
171  }
172 
173  {
174  if (trigS.size() == 1) { // just 1 entry (continous mode?), use digits directly
175  // connect this to a particular branch
176  // the input data span is directly using the raw buffer, we need to copy to
177  // the object we want to write, maybe we can avoid this with some tricks
178  std::vector<o2::tpc::Digit> writeObj(digiData.begin(), digiData.end());
179  auto digP = &writeObj;
180  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigit", sector, digP);
181  br->Fill();
182  br->ResetAddress();
183  } else { // triggered mode (>1 entries will be written)
184  std::vector<o2::tpc::Digit> digGroup; // group of digits related to single trigger
185  auto digGroupPtr = &digGroup;
186  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigit", sector, digGroupPtr);
187  for (auto grp : trigS) {
188  digGroup.clear();
189  for (int i = 0; i < grp.getEntries(); i++) {
190  digGroup.emplace_back(digiData[grp.getFirstEntry() + i]); // fetch digits of given trigger
191  }
192  br->Fill();
193  }
194  br->ResetAddress();
195  }
196  }
197  }
198  } // end digit case
199  }
200 
201  {
202  // b) probe for labels
203  std::vector<InputSpec> filter = {
204  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "DIGITSMCTR"}, Lifetime::Timeframe},
205  };
206  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
207  auto sector = extractSector(ref);
208  auto const* dh = DataRefUtils::getHeader<DataHeader*>(ref);
209  LOG(INFO) << "HAVE LABEL DATA FOR SECTOR " << sector << " ON CHANNEL " << dh->subSpecification;
210  if (sector >= 0) {
211  // the labels
212  auto labeldata = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>(ref);
213  auto labeldataRaw = labeldata.get();
214  LOG(INFO) << "MCTRUTH ELEMENTS " << labeldataRaw->getIndexedSize()
215  << " WITH " << labeldataRaw->getNElements() << " LABELS";
216  const auto& trigS = (*trigP2Sect.get())[sector];
217  if (!trigS.size()) {
218  LOG(FATAL) << "MCTruth for sector " << sector << " are received w/o info on grouping in triggers";
219  } else {
220  int nExp = trigS.back().getFirstEntry() + trigS.back().getEntries() - trigS.front().getFirstEntry();
221  if (nExp != labeldataRaw->getIndexedSize()) {
222  LOG(ERROR) << "Number of indexed (label) slots " << labeldataRaw->getIndexedSize()
223  << " is inconsistent with expectation " << nExp
224  << " from digits grouping for sector " << sector;
225  }
226  }
227  {
228  if (trigS.size() == 1) { // just 1 entry (continous mode?), use labels directly
229  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigitMCTruth", sector, &labeldataRaw);
230  br->Fill();
231  br->ResetAddress();
232  } else {
233  o2::dataformats::MCTruthContainer<o2::MCCompLabel> lblGroup; // labels for group of digits related to single trigger
234  auto lblGroupPtr = &lblGroup;
235  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigitMCTruth", sector, &lblGroupPtr);
236  for (auto grp : trigS) {
237  lblGroup.clear();
238  for (int i = 0; i < grp.getEntries(); i++) {
239  auto lbls = labeldataRaw->getLabels(grp.getFirstEntry() + i);
240  lblGroup.addElements(i, lbls);
241  }
242  br->Fill();
243  }
244  }
245  }
246  }
247  } // end label case
248  }
249 
250  {
251  // c) probe for common mode:
252  std::vector<InputSpec> filter = {
253  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "COMMONMODE"}, Lifetime::Timeframe},
254  };
255  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
256  auto sector = extractSector(ref);
257  auto const* dh = DataRefUtils::getHeader<DataHeader*>(ref);
258  LOG(INFO) << "HAVE COMMON MODE DATA FOR SECTOR " << sector << " ON CHANNEL " << dh->subSpecification;
259  const auto& trigS = (*trigP2Sect.get())[sector];
260  if (sector >= 0) {
261  auto commonModeData = pc.inputs().get<std::vector<o2::tpc::CommonMode>>(ref);
262  LOG(INFO) << "COMMON MODE SIZE " << commonModeData.size();
263 
264  if (!trigS.size()) {
265  LOG(FATAL) << "CommonMode for sector " << sector << " are received w/o info on grouping in triggers";
266  }
267  {
268  auto digC = &commonModeData;
269  auto br = getOrMakeBranch(*outputtree.get(), "TPCCommonMode", sector, digC);
270  br->Fill();
271  br->ResetAddress();
272  }
273  }
274  } // end common mode case
275  }
276  };
277 
278  // return the actual processing function as a lambda function using variables
279  // of the init function
280  return processingFct;
281  };
282 
283  std::vector<InputSpec> inputs = {
284  {"digitinput", "TPC", "DIGITS", 0, Lifetime::Timeframe}, // digit input
285  {"triggerinput", "TPC", "DIGTRIGGERS", 0, Lifetime::Timeframe}, // groupping in triggers
286  {"commonmodeinput", "TPC", "COMMONMODE", 0, Lifetime::Timeframe},
287  };
288  if (mctruth) {
289  inputs.emplace_back("labelinput", "TPC", "DIGITSMCTR", 0, Lifetime::Timeframe);
290  }
291 
292  auto amendInput = [&laneConfiguration](InputSpec& spec, size_t index) {
293  spec.binding += std::to_string(laneConfiguration[index]);
294  DataSpecUtils::updateMatchingSubspec(spec, laneConfiguration[index]);
295  };
296  inputs = mergeInputs(inputs, laneConfiguration.size(), amendInput);
297 
298  return DataProcessorSpec{
299  "TPCDigitWriter",
300  inputs,
301  {}, // no output
302  AlgorithmSpec(initFunction),
303  Options{
304  {"tpc-digit-outfile", VariantType::String, "tpcdigits.root", {"Name of the input file"}},
305  {"treename", VariantType::String, "o2sim", {"Name of tree for tracks"}},
306  }};
307 }
