51 template <typename source_T>
52 std::unique_ptr<std::vector<source_T>> TPCEntropyDecoder::decodeEntry(source_T** destPPtr, const std::string& name, const EncodedClusters& ec, size_t msgLength)
53 {
54  // find which array we are dealing with
55  auto it = std::find(ec.NAMES.begin(), ec.NAMES.end(), name);
56  assert(it != ec.NAMES.end());
57  auto i = std::distance(ec.NAMES.begin(), it);
58 
59  // get references to the right data
60  auto& dict = *ec.dicts[i];
61  auto& buffer = *ec.buffers[i];
62  auto& md = ec.metadata->at(i);
63 
64  // allocate output buffer
65  auto outBuffer = std::make_unique<std::vector<source_T>>(msgLength);
66 
67  // decode
68  rans::SymbolStatistics stats(dict.begin(), dict.end(), md.min, md.max, msgLength);
69  rans::Decoder64<source_T> decoder(stats, md.probabilityBits);
70  decoder.process(outBuffer->begin(), buffer.begin(), msgLength);
71 
72  LOG(INFO) << "finished decoding entry " << name;
73  //return
74  *destPPtr = outBuffer->data();
75  return std::move(outBuffer);
76 }
