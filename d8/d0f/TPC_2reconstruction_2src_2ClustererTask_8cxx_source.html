38 InitStatus ClustererTask::Init()
39 {
40  LOG(DEBUG) << "Enter Initializer of ClustererTask";
41 
42  FairRootManager* mgr = FairRootManager::Instance();
43  if (!mgr) {
44  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ...";
45  return kERROR;
46  }
47 
48  if (mClusterSector < 0 || mClusterSector >= Sector::MAXSECTOR) {
49  LOG(ERROR) << "Sector ID " << mClusterSector << " is not supported. Exiting ...";
50  return kERROR;
51  }
52 
53  // Register input container
54  std::stringstream sectornamestr;
55  sectornamestr << "TPCDigit" << mClusterSector;
56  LOG(INFO) << "FETCHING DIGITS FOR SECTOR " << mClusterSector << "\n";
57  mDigitsArray = std::unique_ptr<const std::vector<Digit>>(
58  mgr->InitObjectAs<const std::vector<Digit>*>(sectornamestr.str().c_str()));
59  if (!mDigitsArray) {
60  LOG(ERROR) << "TPC points not registered in the FairRootManager. Exiting ...";
61  return kERROR;
62  }
63  std::stringstream mcsectornamestr;
64  mcsectornamestr << "TPCDigitMCTruth" << mClusterSector;
65  mDigitMCTruthArray = std::unique_ptr<const MCLabelContainer>(
66  mgr->InitObjectAs<const MCLabelContainer*>(mcsectornamestr.str().c_str()));
67  if (!mDigitMCTruthArray) {
68  LOG(ERROR) << "TPC MC Truth not registered in the FairRootManager. Exiting ...";
69  return kERROR;
70  }
71 
72  // Register output container
73  mHwClustersArray = std::make_unique<std::vector<OutputType>>();
74  // a trick to register the unique pointer with FairRootManager
75  static auto clusterArrayTmpPtr = mHwClustersArray.get();
76  mgr->RegisterAny(Form("TPCClusterHW%i", mClusterSector), clusterArrayTmpPtr, kTRUE);
77 
78  // Register MC Truth output container
79  mHwClustersMCTruthArray = std::make_unique<MCLabelContainer>();
80  // a trick to register the unique pointer with FairRootManager
81  static auto clusterMcTruthTmpPtr = mHwClustersMCTruthArray.get();
82  mgr->RegisterAny(Form("TPCClusterHWMCTruth%i", mClusterSector), clusterMcTruthTmpPtr, kTRUE);
83 
84  // create clusterer and pass output pointer
85  mHwClusterer = std::make_unique<HwClusterer>(mHwClustersArray.get(), mClusterSector, mHwClustersMCTruthArray.get());
86  mHwClusterer->setContinuousReadout(mIsContinuousReadout);
87 
88  // TODO: implement noise/pedestal objects
89  // mHwClusterer->setNoiseObject(...);
90  // mHwClusterer->setPedestalObject(...);
91 
92  return kSUCCESS;
93 }

96 void ClustererTask::Exec(Option_t* option)
97 {
98  LOG(DEBUG) << "Running clusterization on event " << mEventCount << " with " << mDigitsArray->size() << " digits.";
99 
100  if (mHwClustersArray)
101  mHwClustersArray->clear();
102  if (mHwClustersMCTruthArray)
103  mHwClustersMCTruthArray->clear();
104 
105  mHwClusterer->process(gsl::span<o2::tpc::Digit const>(mDigitsArray->data(), mDigitsArray->size()), mDigitMCTruthArray.get());
106  LOG(DEBUG) << "Hw clusterer delivered " << mHwClustersArray->size() << " cluster container";
107 
108  ++mEventCount;
109 }
