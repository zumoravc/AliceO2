57 o2::framework::WorkflowSpec getWorkflow(bool propagateMC,
58  bool enableDigitsPrinter,
59  std::string const& cfgInput,
60  std::string const& cfgOutput)
61 {
62  InputType inputType;
63 
64  try {
65  inputType = InputMap.at(cfgInput);
66  } catch (std::out_of_range&) {
67  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
68  }
69  std::vector<OutputType> outputTypes;
70  try {
71  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
72  } catch (std::out_of_range&) {
73  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
74  }
75  auto isEnabled = [&outputTypes](OutputType type) {
76  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
77  };

139  // check if the process is ready to quit
140  // this is decided upon the meta information in the PHOS block header, the operation is set
141  // value kNoPayload in case of no data or no operation
142  // see also PublisherSpec.cxx
143  // in this workflow, the EOD is sent after the last real data, and all inputs will receive EOD,
144  // so it is enough to check on the first occurence
145  // FIXME: this will be changed once DPL can propagate control events like EOD
146  auto checkReady = [](o2::framework::DataRef const& ref) {
147  auto const* phosheader = o2::framework::DataRefUtils::getHeader<o2::phos::PHOSBlockHeader*>(ref);
148  // sector number -1 indicates end-of-data
149  if (phosheader != nullptr) {
150  // indicate normal processing if not ready and skip if ready
151  if (!phosheader->mHasPayload) {
152  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::SkipProcessing, true);
153  }
154  }
155  return std::make_tuple(o2::framework::MakeRootTreeWriterSpec::TerminationCondition::Action::DoProcessing, false);
156  };
