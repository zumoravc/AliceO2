82 inline void Road::setPoint(const Float_t x, const Float_t y, const Float_t z, const Int_t layer, const Int_t clusterId, const MCCompLabel label, Bool_t& newPoint)
83 {
84  if (!newPoint) {
85  if (!mZ.empty()) {
86  mX[layer].pop_back();
87  mY[layer].pop_back();
88  mZ[layer].pop_back();
89  mClusterId[layer].pop_back();
90  mMCCompLabel[layer].pop_back();
91  }
92  } else { // end replace point
93  newPoint = kFALSE;
94  }
95  mX[layer].push_back(x);
96  mY[layer].push_back(y);
97  mZ[layer].push_back(z);
98  mClusterId[layer].push_back(clusterId);
99  mMCCompLabel[layer].emplace_back(label);
100  ++mNPoints;
101 }
