21 template <class Mapping>
22 void MC2RawEncoder<Mapping>::init()
23 {
24  if (mROMode == NotSet) {
25  LOG(FATAL) << "Readout Mode must be set explicitly via setContinuousReadout(bool)";
26  }
27  mWriter.setCarryOverCallBack(this);
28 
29  // limit RUs to convert to existing ones
30  mRUSWMax = (mRUSWMax < uint8_t(mMAP.getNRUs())) ? mRUSWMax : mMAP.getNRUs() - 1;
31  //
32  mNLinks = 0;
33  for (uint8_t ru = mRUSWMin; ru <= mRUSWMax; ru++) {
34  auto& ruData = getCreateRUDecode(ru);
35  int nLinks = 0;
36  for (int il = 0; il < RUDecodeData::MaxLinksPerRU; il++) {
37  auto* link = getGBTLink(ruData.links[il]);
38  if (link) {
39  auto subspec = o2::raw::RDHUtils::getSubSpec(link->cruID, link->idInCRU, link->endPointID, link->feeID);
40  if (!mWriter.isLinkRegistered(subspec)) {
41  LOGF(INFO, "RU%3d FEEId 0x%04x Link %02d of CRU=0x%94x will be writing to default sink %s",
42  int(ru), link->feeID, link->idInCRU, link->cruID, mDefaultSinkName);
43  mWriter.registerLink(link->feeID, link->cruID, link->idInCRU, link->endPointID, mDefaultSinkName);
44  }
45  nLinks++;
46  if (link->packetCounter < 0) {
47  link->packetCounter = 0; // reset only once
48  }
49  }
50  }
51  mNLinks += nLinks;
52  if (!nLinks) {
53  LOG(WARNING) << "No GBT links were defined for RU " << int(ru) << " defining automatically";
54  ruData.links[0] = addGBTLink();
55  auto* link = getGBTLink(ruData.links[0]);
56  link->lanes = mMAP.getCablesOnRUType(ruData.ruInfo->ruType);
57  link->idInCRU = 0;
58  link->idInRU = 0;
59  link->cruID = ru;
60  link->feeID = mMAP.RUSW2FEEId(ruData.ruInfo->idSW, 0);
61  link->endPointID = 0;
62  link->packetCounter = 0;
63  mWriter.registerLink(link->feeID, link->cruID, link->idInCRU, link->endPointID, mDefaultSinkName);
64  LOGF(INFO, "RU%3d FEEId 0x%04x Link %02d of CRU=0x%04x Lanes: %s -> %s", int(ru), link->feeID,
65  link->idInCRU, link->cruID, std::bitset<28>(link->lanes).to_string(), mDefaultSinkName);
66  mNLinks++;
67  }
68  }
69 
70  assert(mNLinks > 0);
71 }

81 template <class Mapping>
82 void MC2RawEncoder<Mapping>::digits2raw(gsl::span<const Digit> digits, const o2::InteractionRecord& bcData)
83 {
84  // Convert digits to raw data
85  // The digits in the span/vector must be in increasing chipID order
86  // Return the number of pages in the link with smallest amount of pages
87 
88  if (!mNLinks) {
89  init();
90  }
91  int nDigTot = digits.size();
92  mCurrIR = bcData;
93  // place digits into corresponding chip buffers
94  ChipPixelData* curChipData = nullptr;
95  ChipInfo chInfo;
96  UShort_t curChipID = 0xffff; // currently processed SW chip id
97  for (const auto& dig : digits) {
98  if (curChipID != dig.getChipIndex()) {
99  mMAP.getChipInfoSW(dig.getChipIndex(), chInfo);
100  if (chInfo.ru < mRUSWMin || chInfo.ru > mRUSWMax) { // ignore this chip?
101  continue;
102  }
103  auto* ru = getRUDecode(chInfo.ru);
104  curChipID = dig.getChipIndex();
105  curChipData = &ru->chipsData[ru->nChipsFired++];
106  curChipData->setChipID(chInfo.chOnRU->id); // set chip ID within the RU
107  }
108  curChipData->getData().emplace_back(&dig); // add new digit to the container
109  }
110 
111  // convert digits to alpide data in the per-cable buffers
112  for (int iru = int(mRUSWMin); iru <= int(mRUSWMax); iru++) {
113  auto& ru = *getRUDecode(iru);
114  uint16_t next2Proc = 0, nchTot = mMAP.getNChipsOnRUType(ru.ruInfo->ruType);
115  for (int ich = 0; ich < ru.nChipsFired; ich++) {
116  auto& chipData = ru.chipsData[ich];
117  convertEmptyChips(next2Proc, chipData.getChipID(), ru); // if needed store EmptyChip flags for the empty chips
118  next2Proc = chipData.getChipID() + 1;
119  convertChip(chipData, ru);
120  chipData.clear();
121  }
122  convertEmptyChips(next2Proc, nchTot, ru); // if needed store EmptyChip flags
123  fillGBTLinks(ru); // flush per-lane buffers to link buffers
124  }
125 }

128 template <class Mapping>
129 void MC2RawEncoder<Mapping>::convertChip(ChipPixelData& chipData, RUDecodeData& ru)
130 {
132  const auto& chip = *mMAP.getChipOnRUInfo(ru.ruInfo->ruType, chipData.getChipID());
133  ru.cableHWID[chip.cableSW] = chip.cableHW; // register the cable HW ID
134  auto& pixels = chipData.getData();
135  std::sort(pixels.begin(), pixels.end(),
136  [](auto lhs, auto rhs) {
137  return (lhs.getRow() < rhs.getRow()) ? true : ((lhs.getRow() > rhs.getRow()) ? false : (lhs.getCol() < rhs.getCol()));
138  });
139  ru.cableData[chip.cableSW].ensureFreeCapacity(40 * (2 + pixels.size())); // make sure buffer has enough capacity
140  mCoder.encodeChip(ru.cableData[chip.cableSW], chipData, chip.chipOnModuleHW, mCurrIR.bc);
141 }

144 template <class Mapping>
145 void MC2RawEncoder<Mapping>::convertEmptyChips(int fromChip, int uptoChip, RUDecodeData& ru)
146 {
147  // add empty chip words to respective cable's buffers for all chips of the current RU container
148  for (int chipIDSW = fromChip; chipIDSW < uptoChip; chipIDSW++) { // flag chips w/o data
149  const auto& chip = *mMAP.getChipOnRUInfo(ru.ruInfo->ruType, chipIDSW);
150  ru.cableHWID[chip.cableSW] = chip.cableHW; // register the cable HW ID
151  ru.cableData[chip.cableSW].ensureFreeCapacity(100);
152  mCoder.addEmptyChip(ru.cableData[chip.cableSW], chip.chipOnModuleHW, mCurrIR.bc);
153  }
154 }

157 template <class Mapping>
158 void MC2RawEncoder<Mapping>::fillGBTLinks(RUDecodeData& ru)
159 {
160  // fill data of the RU to links buffer, return the number of pages in the link with smallest amount of pages
161  constexpr uint8_t zero16[GBTPaddedWordLength] = {0}; // to speedup padding
162  ru.nCables = ru.ruInfo->nCables;
163 
164  // trigger word
165  GBTTrigger gbtTrigger;
166  gbtTrigger.bc = mCurrIR.bc;
167  gbtTrigger.orbit = mCurrIR.orbit;
168  gbtTrigger.internal = isContinuousReadout();
169  gbtTrigger.triggerType = 0; //TODO
170 
171  for (int il = 0; il < RUDecodeData::MaxLinksPerRU; il++) {
172  auto link = getGBTLink(ru.links[il]);
173  if (!link) {
174  continue;
175  }
176  // estimate real payload size in GBT words
177  int nPayLoadWordsNeeded = 0; // number of payload words filled to link buffer (RDH not included) for current IR
178  for (int icab = ru.nCables; icab--;) { // calculate number of GBT words per link
179  if ((link->lanes & (0x1 << icab))) {
180  int nb = ru.cableData[icab].getSize();
181  nPayLoadWordsNeeded += nb ? 1 + (nb - 1) / 9 : 0; // single GBT word carries at most 9 payload bytes
182  }
183  }
184  // reserve space for payload + trigger + header + trailer
185  link->data.ensureFreeCapacity((3 + nPayLoadWordsNeeded) * GBTPaddedWordLength);
186  link->data.addFast(gbtTrigger.getW8(), GBTPaddedWordLength); // write GBT trigger in the beginning of the buffer
187 
188  GBTDataHeader gbtHeader;
189  gbtHeader.packetIdx = 0;
190  gbtHeader.activeLanes = link->lanes;
191  link->data.addFast(gbtHeader.getW8(), GBTPaddedWordLength); // write GBT header
192 
193  // now loop over the lanes served by this link, writing each time at most 9 bytes, untill all lanes are copied
194  bool hasData = true;
195  while (hasData) {
196  hasData = false;
197  for (int icab = 0; icab < ru.nCables; icab++) {
198  if ((link->lanes & (0x1 << icab))) {
199  auto& cableData = ru.cableData[icab];
200  int nb = cableData.getUnusedSize();
201  if (!nb) {
202  continue; // write 80b word only if there is something to write
203  }
204  if (nb > 9) {
205  nb = 9;
206  }
207  int gbtWordStart = link->data.getSize(); // beginning of the current GBT word in the link
208  link->data.addFast(cableData.getPtr(), nb); // fill payload of cable
209  link->data.addFast(zero16, GBTPaddedWordLength - nb); // fill the rest of the GBT word by 0
210  link->data[gbtWordStart + 9] = mMAP.getGBTHeaderRUType(ru.ruInfo->ruType, ru.cableHWID[icab]); // set cable flag
211  cableData.setPtr(cableData.getPtr() + nb);
212  hasData = true;
213  } // storing data of single cable
214  } // loop over cables of this link
215  }; // loop until all links are w/o data
216 
217  // all payload was dumped, write final trailer
218  GBTDataTrailer gbtTrailer; // lanes will be set on closing the trigger
219  gbtTrailer.lanesStops = link->lanes;
220  gbtTrailer.packetDone = true;
221  link->data.addFast(gbtTrailer.getW8(), GBTPaddedWordLength); // write GBT trailer for the last packet
222  LOGF(DEBUG, "Filled %s with %d GBT words", link->describe(), nPayLoadWordsNeeded + 3);
223 
224  // flush to writer
225  mWriter.addData(link->feeID, link->cruID, link->idInCRU, link->endPointID, mCurrIR, gsl::span((char*)link->data.data(), link->data.getSize()));
226  link->data.clear();
227  //
228  } // loop over links of RU
229  ru.clear();
230 }

233 template <class Mapping>
234 RUDecodeData& MC2RawEncoder<Mapping>::getCreateRUDecode(int ruSW)
235 {
236  assert(ruSW < mMAP.getNRUs());
237  if (mRUEntry[ruSW] < 0) {
238  mRUEntry[ruSW] = mNRUs++;
239  mRUDecodeVec[mRUEntry[ruSW]].ruInfo = mMAP.getRUInfoSW(ruSW); // info on the stave/RU
240  mRUDecodeVec[mRUEntry[ruSW]].chipsData.resize(mMAP.getNChipsOnRUType(mMAP.getRUInfoSW(ruSW)->ruType));
241  LOG(INFO) << "Defining container for RU " << ruSW << " at slot " << mRUEntry[ruSW];
242  }
243  return mRUDecodeVec[mRUEntry[ruSW]];

247 template <class Mapping>
248 int MC2RawEncoder<Mapping>::carryOverMethod(const header::RDHAny* rdh, const gsl::span<char> data,
249  const char* ptr, int maxSize, int splitID,
250  std::vector<char>& trailer, std::vector<char>& header) const
251 {
252  // The RawFileWriter receives from the encoder the payload to format according to the CRU format
253  // In case this payload size does not fit into the CRU page (this may happen even if it is
254  // less than 8KB, since it might be added to already partially populated CRU page of the HBF)
255  // it will write on the page only part of the payload and carry over the rest on extra page(s).
256  // By default the RawFileWriter will simply chunk payload as is considers necessary, but some
257  // detectors want their CRU pages to be self-consistent and in case of payload splitting they
258  // add in the end of page to be closed and beginning of the new page to be opened
259  // (right after the RDH) detector-specific trailer and header respectively.
260  //
261  // The role of this method is to suggest to writer how to split the payload:
262  // If this method was set to the RawFileWriter using
263  // RawFileWriter::setCarryOverCallBack(pointer_on_the_converter_class);
264  // then the RawFileWriter will call it before splitting.
265  //
266  // It provides to carryOverMethod method the following info:
267  // rdh : RDH of the CRU page being written
268  // data : original payload received by the RawFileWriter
269  // ptr : pointer on the data in the payload which was not yet added to the link CRU pages
270  // maxSize : maximum size (multiple of 16 bytes) of the bloc starting at ptr which it can
271  // accomodate at the current CRU page (i.e. what it would write by default)
272  // splitID : number of times this payload was already split, 0 at 1st call
273  // trailer : what it wants to add in the end of the CRU page where the data starting from ptr
274  // will be added. The trailer is supplied as an empy vector, which carryOverMethod
275  // may populate, but its size must be multiple of 16 bytes.
276  // header : what it wants to add right after the RDH of the new CRU page before the rest of
277  // the payload (starting at ptr+actualSize) will be written
278  //
279  // The method must return actual size of the bloc which can be written (<=maxSize).
280  // If this method populates the trailer, it must ensure that it returns the actual size such that
281  // actualSize + trailer.size() <= maxSize
282  // In case returned actualSize == 0, current CRU page will be closed w/o adding anything, and new
283  // query of this method will be done on the new CRU page
284 
285  constexpr int TrigHeadSize = sizeof(GBTTrigger) + sizeof(GBTDataHeader);
286  constexpr int TotServiceSize = sizeof(GBTTrigger) + sizeof(GBTDataHeader) + sizeof(GBTDataTrailer);
287 
288  int offs = ptr - &data[0]; // offset wrt the head of the payload
289  // make sure ptr and end of the suggested block are within the payload
290  assert(offs >= 0 && size_t(offs + maxSize) <= data.size());
291 
292  if ((maxSize <= TotServiceSize)) { // we cannot split trigger+header
293  return 0; // suggest moving the whole payload to the new CRU page
294  }
295 
296  // this is where we would usually split: account for the trailer to add
297  int actualSize = maxSize - sizeof(GBTDataTrailer);
298 
299  char* trailPtr = &data[data.size() - sizeof(GBTDataTrailer)]; // pointer on the payload trailer
300  if (ptr + actualSize >= trailPtr) { // we need to split at least 1 GBT word before the trailer
301  actualSize = trailPtr - ptr - GBTPaddedWordLength;
302  }
303  // copy the GBTTrigger and GBTHeader from the head of the payload
304  header.resize(TrigHeadSize);
305  memcpy(header.data(), &data[0], TrigHeadSize);
306  GBTDataHeader& gbtHeader = *reinterpret_cast<GBTDataHeader*>(&header[sizeof(GBTTrigger)]); // 1st trigger then header are written
307  gbtHeader.packetIdx = splitID + 1; // update the ITS specific packets counter
308 
309  // copy the GBTTrailer from the end of the payload
310  trailer.resize(sizeof(GBTDataTrailer));
311  memcpy(trailer.data(), trailPtr, sizeof(GBTDataTrailer));
312  GBTDataTrailer& gbtTrailer = *reinterpret_cast<GBTDataTrailer*>(&trailer[0]);
313  gbtTrailer.packetDone = false; // intermediate trailers should not have done=true
314  gbtTrailer.lanesStops = 0; // intermediate trailers should not have lanes closed
315 
316  return actualSize;
317 }
