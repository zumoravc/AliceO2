38 SampaChannelHandler handlePacket(std::string& result)
39 {
40  return [&result](DsElecId dsId, uint8_t channel, SampaCluster sc) {
41  result += fmt::format("{}-ch-{}-ts-{}-q", asString(dsId), channel, sc.sampaTime);
42  if (sc.isClusterSum()) {
43  result += fmt::format("-{}", sc.chargeSum);
44  } else {
45  for (auto s : sc.samples) {
46  result += fmt::format("-{}", s);
47  }
48  }
49  result += "\n";
50  };
51 }

53 std::vector<std::byte> convertBuffer2PayloadBuffer(gsl::span<const std::byte> buffer,
54  std::optional<size_t> insertSync = std::nullopt)
55 {
56  // some gym to go from a buffer coming from the encoder,
57  // which holds (DataBlockHeader,Payload) pairs in a byte buffer
58  // to a buffer that holds just Payload information
59  // to be fed to the decoder
60 
61  // strip the headers
62  std::vector<std::byte> b8;
63  forEachDataBlockRef(
64  buffer, [&](const DataBlockRef& r) {
65  auto& b = r.block;
66  b8.insert(b8.end(), b.payload.begin(), b.payload.end());
67  });
68 
69  // convert to a 64-bits buffer to be able to insert sync if needed
70  std::vector<uint64_t> b64;
71  impl::copyBuffer(b8, b64);
72 
73  // insert a sync at the given position if required
74  if (insertSync.has_value() && insertSync.value() < b64.size()) {
75  uint64_t prefix = b64[0] & 0xFFFC00000000000F;
76  b64.insert(b64.begin() + insertSync.value(), prefix | sampaSyncWord);
77  }
78 
79  impl::dumpBuffer<o2::mch::raw::UserLogicFormat>(b64);
80 
81  // get back to byte buffer to return
82  std::vector<std::byte> bytes;
83  impl::copyBuffer(b64, bytes);
84  return bytes;
85 }

97 template <typename CHARGESUM>
98 std::string testPayloadDecode(DsElecId ds1,
99  int ch1,
100  const std::vector<SampaCluster>& clustersFirstChannel,
101  DsElecId ds2 = DsElecId{0, 0, 0},
102  int ch2 = 47,
103  const std::vector<SampaCluster>& clustersSecondChannel = {},
104  std::optional<size_t> insertSync = std::nullopt)
105 {
106  auto encoder = createPayloadEncoder<UserLogicFormat, CHARGESUM, true>();
107 
108  encoder->startHeartbeatFrame(0, 0);
109 
110  auto solar2feelink = createSolar2FeeLinkMapper<ElectronicMapperGenerated>();
111 
112  uint16_t feeId{0};
113 
114  auto f1 = solar2feelink(ds1.solarId());
115  if (!f1.has_value()) {
116  throw std::invalid_argument("invalid solarId for ds1");
117  }
118  if (!clustersSecondChannel.empty()) {
119  auto f2 = solar2feelink(ds2.solarId());
120  if (!f2.has_value()) {
121  throw std::invalid_argument("invalid solarId for ds2");
122  }
123  if (f2->feeId() != f1->feeId()) {
124  throw std::invalid_argument("this test is only meant to work with 2 solars in the same cru endpoint");
125  }
126  }
127 
128  feeId = f1->feeId();
129 
130  encoder->addChannelData(ds1, ch1, clustersFirstChannel);
131  if (!clustersSecondChannel.empty()) {
132  encoder->addChannelData(ds2, ch2, clustersSecondChannel);
133  }
134 
135  std::vector<std::byte> buffer;
136  encoder->moveToBuffer(buffer);
137  auto payloadBuffer = convertBuffer2PayloadBuffer(buffer, insertSync);
138 
139  return decodeBuffer<CHARGESUM>(feeId, payloadBuffer);
140 }
