70 bool ConfigParamsHelper::dpl2BoostOptions(const std::vector<ConfigParamSpec>& spec,
71  boost::program_options::options_description& options,
72  boost::program_options::options_description vetos)
73 {
74  bool haveOption = false;
75  for (const auto& configSpec : spec) {
76  // skip everything found in the veto definition
77  try {
78  if (vetos.find_nothrow(configSpec.name, false))
79  continue;
80  } catch (boost::program_options::ambiguous_option& e) {
81  for (auto& alternative : e.alternatives()) {
82  std::cerr << alternative << std::endl;
83  }
84  throw;
85  }
86 
87  haveOption = true;
88  std::stringstream defaultValue;
89  defaultValue << configSpec.defaultValue;
90  if (configSpec.type != VariantType::Bool) {
91  if (configSpec.defaultValue.type() != VariantType::Empty) {
92  options.add_options()(configSpec.name.c_str(),
93  bpo::value<std::string>()->default_value(defaultValue.str().c_str()),
94  configSpec.help.c_str());
95  } else {
96  options.add_options()(configSpec.name.c_str(),
97  bpo::value<std::string>(),
98  configSpec.help.c_str());
99  }
100  } else {
101  if (configSpec.defaultValue.type() != VariantType::Empty) {
102  options.add_options()(configSpec.name.c_str(),
103  bpo::value<bool>()->zero_tokens()->default_value(configSpec.defaultValue.get<bool>()),
104  configSpec.help.c_str());
105  } else {
106  options.add_options()(configSpec.name.c_str(),
107  bpo::value<bool>()->zero_tokens(),
108  configSpec.help.c_str());
109  }
110  }
111  }
112 
113  return haveOption;
114 }
