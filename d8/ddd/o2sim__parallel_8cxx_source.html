107 // monitores a certain incoming pipe and displays new information
108 void launchThreadMonitoringEvents(int pipefd, std::string text)
109 {
110  static std::vector<std::thread> threads;
111  auto lambda = [pipefd, text]() {
112  int eventcounter;
113  while (1) {
114  ssize_t count = read(pipefd, &eventcounter, sizeof(eventcounter));
115  if (count == -1) {
116  LOG(INFO) << "ERROR READING";
117  if (errno == EINTR) {
118  continue;
119  } else {
120  return;
121  }
122  } else if (count == 0) {
123  break;
124  } else {
125  LOG(INFO) << text.c_str() << eventcounter;
126  }
127  };
128  };
129  threads.push_back(std::thread(lambda));
130  threads.back().detach();
131 }

133 // helper executable to launch all the devices/processes
134 // for parallel simulation
135 int main(int argc, char* argv[])
136 {
137  signal(SIGINT, sighandler);
138  signal(SIGTERM, sighandler);
139  // we enable the forked version of the code by default
140  setenv("ALICE_SIMFORKINTERNAL", "ON", 1);
141 
142  TStopwatch timer;
143  timer.Start();
144  auto o2env = getenv("O2_ROOT");
145  if (!o2env) {
146  LOG(FATAL) << "O2_ROOT environment not defined";
147  }
148  std::string rootpath(o2env);
149  std::string installpath = rootpath + "/bin";
150 
151  // copy topology file to working dir and update ports
152  std::stringstream configss;
153  configss << rootpath << "/share/config/o2simtopology.json";
154  std::string localconfig = std::string("o2simtopology_") + std::to_string(getpid()) + std::string(".json");
155 
156  // need to add pid to channel urls to allow simultaneous deploys!
157  // open otiginal config and read the JSON config
158  FILE* fp = fopen(configss.str().c_str(), "r");
159  constexpr unsigned short usmax = std::numeric_limits<unsigned short>::max() - 1;
160  char Buffer[usmax];
161  rapidjson::FileReadStream is(fp, Buffer, sizeof(Buffer));
162  rapidjson::Document d;
163  d.ParseStream(is);
164  fclose(fp);
165  // find and manipulate URLS
166  std::string serveraddress;
167  std::string mergeraddress;
168  std::string s;
169 
170  auto& options = d["fairMQOptions"];
171  assert(options.IsObject());
172  for (auto option = options.MemberBegin(); option != options.MemberEnd();
173  ++option) {
174  s = option->name.GetString();
175  if (s == "devices") {
176  assert(option->value.IsArray());
177  auto devices = option->value.GetArray();
178  for (auto& device : devices) {
179  s = device["id"].GetString();
180  if (s == "primary-server") {
181  auto channels = device["channels"].GetArray();
182  auto sockets = (channels[0])["sockets"].GetArray();
183  auto& addressv = (sockets[0])["address"];
184  auto address = addressv.GetString();
185  serveraddress = address + std::to_string(getpid());
186  addressv.SetString(serveraddress.c_str(), d.GetAllocator());
187  }
188  if (s == "hitmerger") {
189  auto channels = device["channels"].GetArray();
190  for (auto& channel : channels) {
191  s = channel["name"].GetString();
192  // set server's address for merger
193  if (s == "primary-get") {
194  auto sockets = channel["sockets"].GetArray();
195  auto& addressv = (sockets[0])["address"];
196  addressv.SetString(serveraddress.c_str(), d.GetAllocator());
197  }
198  if (s == "simdata") {
199  auto sockets = channel["sockets"].GetArray();
200  auto& addressv = (sockets[0])["address"];
201  auto address = addressv.GetString();
202  mergeraddress = address + std::to_string(getpid());
203  addressv.SetString(mergeraddress.c_str(), d.GetAllocator());
204  }
205  }
206  }
207  }
208  //loop over devices again and set URLs for the workers
209  for (auto& device : devices) {
210  s = device["id"].GetString();
211  if (s == "worker") {
212  auto channels = device["channels"].GetArray();
213  for (auto& channel : channels) {
214  s = channel["name"].GetString();
215  if (s == "primary-get") {
216  auto sockets = channel["sockets"].GetArray();
217  auto& addressv = (sockets[0])["address"];
218  addressv.SetString(serveraddress.c_str(), d.GetAllocator());
219  }
220  if (s == "simdata") {
221  auto sockets = channel["sockets"].GetArray();
222  auto& addressv = (sockets[0])["address"];
223  addressv.SetString(mergeraddress.c_str(), d.GetAllocator());
224  }
225  }
226  }
227  }
228  }
229  }
230  // write the config copy with new name
231  fp = fopen(localconfig.c_str(), "w");
232  rapidjson::FileWriteStream os(fp, Buffer, sizeof(Buffer));
233  rapidjson::Writer<rapidjson::FileWriteStream> writer(os);
234  d.Accept(writer);
235  fclose(fp);
236 
237  auto& conf = o2::conf::SimConfig::Instance();
238  if (!conf.resetFromArguments(argc, argv)) {
239  return 1;
240  }
241  // in case of zero events asked (only setup geometry etc) we just call the non-distributed version
242  // (otherwise we would need to add more synchronization between the actors)
243  if (conf.getNEvents() <= 0) {
244  LOG(INFO) << "No events to be simulated; Switching to non-distributed mode";
245  const int Nargs = argc + 1;
246  std::string name("o2-sim-serial");
247  const char* arguments[Nargs];
248  arguments[0] = name.c_str();
249  for (int i = 1; i < argc; ++i) {
250  arguments[i] = argv[i];
251  }
252  arguments[argc] = nullptr;
253  std::string path = installpath + "/" + name;
254  auto r = execv(path.c_str(), (char* const*)arguments);
255  if (r != 0) {
256  perror(nullptr);
257  }
258  return r;
259  }
260 
261  // we create the global shared mem pool; just enough to serve
262  // n simulation workers
263  int nworkers = conf.getNSimWorkers();
264  setenv("ALICE_NSIMWORKERS", std::to_string(nworkers).c_str(), 1);
265  LOG(INFO) << "Running with " << nworkers << " sim workers ";
266 
267  o2::utils::ShmManager::Instance().createGlobalSegment(nworkers);
268 
269  // we can try to disable it here
270  if (getenv("ALICE_NOSIMSHM")) {
271  o2::utils::ShmManager::Instance().disable();
272  }
273 
274  std::vector<int> childpids;
275 
276  int pipe_serverdriver_fd[2];
277  if (pipe(pipe_serverdriver_fd) != 0) {
278  perror("problem in creating pipe");
279  }
280 
281  // the server
282  int pid = fork();
283  if (pid == 0) {
284  int fd = open(serverlogname, O_RDWR | O_CREAT, S_IRUSR | S_IWUSR);
285  setenv("ALICE_O2SIMSERVERTODRIVER_PIPE", std::to_string(pipe_serverdriver_fd[1]).c_str(), 1);
286 
287  dup2(fd, 1); // make stdout go to file
288  dup2(fd, 2); // make stderr go to file - you may choose to not do this
289  // or perhaps send stderr to another file
290  close(pipe_serverdriver_fd[0]);
291  close(fd); // fd no longer needed - the dup'ed handles are sufficient
292 
293  const std::string name("o2-sim-primary-server-device-runner");
294  const std::string path = installpath + "/" + name;
295  const std::string config = localconfig;
296 
297  // copy all arguments into a common vector
298  const int Nargs = argc + 7;
299  const char* arguments[Nargs];
300  arguments[0] = name.c_str();
301  arguments[1] = "--control";
302  arguments[2] = "static";
303  arguments[3] = "--id";
304  arguments[4] = "primary-server";
305  arguments[5] = "--mq-config";
306  arguments[6] = config.c_str();
307  for (int i = 1; i < argc; ++i) {
308  arguments[6 + i] = argv[i];
309  }
310  arguments[Nargs - 1] = nullptr;
311  for (int i = 0; i < Nargs; ++i) {
312  if (arguments[i]) {
313  std::cerr << arguments[i] << "\n";
314  }
315  }
316  std::cerr << "$$$$\n";
317  auto r = execv(path.c_str(), (char* const*)arguments);
318  LOG(INFO) << "Starting the server"
319  << "\n";
320  if (r != 0) {
321  perror(nullptr);
322  }
323  return r;
324  } else {
325  childpids.push_back(pid);
326  close(pipe_serverdriver_fd[1]);
327  std::cout << "Spawning particle server on PID " << pid << "; Redirect output to " << serverlogname << "\n";
328  launchThreadMonitoringEvents(pipe_serverdriver_fd[0], "DISTRIBUTING EVENT : ");
329  }
330 
331  auto internalfork = getenv("ALICE_SIMFORKINTERNAL");
332  if (internalfork) {
333  // forking will be done internally to profit from copy-on-write
334  nworkers = 1;
335  }
336  for (int id = 0; id < nworkers; ++id) {
337  // the workers
338  std::stringstream workerlogss;
339  workerlogss << workerlogname << id;
340 
341  // the workers
342  std::stringstream workerss;
343  workerss << "worker" << id;
344 
345  pid = fork();
346  if (pid == 0) {
347  int fd = open(workerlogss.str().c_str(), O_RDWR | O_CREAT, S_IRUSR | S_IWUSR);
348  dup2(fd, 1); // make stdout go to file
349  dup2(fd, 2); // make stderr go to file - you may choose to not do this
350  // or perhaps send stderr to another file
351  close(fd); // fd no longer needed - the dup'ed handles are sufficient
352 
353  const std::string name("o2-sim-device-runner");
354  const std::string path = installpath + "/" + name;
355  execl(path.c_str(), name.c_str(), "--control", "static", "--id", workerss.str().c_str(), "--config-key",
356  "worker", "--mq-config", localconfig.c_str(), "--severity", "info", (char*)nullptr);
357  return 0;
358  } else {
359  childpids.push_back(pid);
360  std::cout << "Spawning sim worker " << id << " on PID " << pid
361  << "; Redirect output to " << workerlogss.str() << "\n";
362  }
363  }
364 
365  // the hit merger
366  int pipe_mergerdriver_fd[2];
367  if (pipe(pipe_mergerdriver_fd) != 0) {
368  perror("problem in creating pipe");
369  }
370 
371  pid = fork();
372  if (pid == 0) {
373  int fd = open(mergerlogname, O_RDWR | O_CREAT, S_IRUSR | S_IWUSR);
374  dup2(fd, 1); // make stdout go to file
375  dup2(fd, 2); // make stderr go to file - you may choose to not do this
376  // or perhaps send stderr to another file
377  close(fd); // fd no longer needed - the dup'ed handles are sufficient
378  close(pipe_mergerdriver_fd[0]);
379  setenv("ALICE_O2SIMMERGERTODRIVER_PIPE", std::to_string(pipe_mergerdriver_fd[1]).c_str(), 1);
380 
381  const std::string name("o2-sim-hit-merger-runner");
382  const std::string path = installpath + "/" + name;
383  execl(path.c_str(), name.c_str(), "--control", "static", "--catch-signals", "0", "--id", "hitmerger", "--mq-config", localconfig.c_str(),
384  (char*)nullptr);
385  return 0;
386  } else {
387  std::cout << "Spawning hit merger on PID " << pid << "; Redirect output to " << mergerlogname << "\n";
388  childpids.push_back(pid);
389  close(pipe_mergerdriver_fd[1]);
390  launchThreadMonitoringEvents(pipe_mergerdriver_fd[0], "EVENT FINISHED : ");
391  }
392 
393  // wait on merger (which when exiting completes the workflow)
394  auto mergerpid = childpids.back();
395 
396  int status, cpid;
397  // wait just blocks and waits until any child returns; but we make sure to wait until merger is here
398  while ((cpid = wait(&status)) != mergerpid) {
399  if (WIFSIGNALED(status)) {
400  LOG(INFO) << "Process " << cpid << " EXITED WITH CODE " << WEXITSTATUS(status) << " SIGNALED "
401  << WIFSIGNALED(status) << " SIGNAL " << WTERMSIG(status);
402  }
403  // we bring down all processes if one of them aborts
404  if (WTERMSIG(status) == SIGABRT) {
405  for (auto p : childpids) {
406  kill(p, SIGABRT);
407  }
408  cleanup();
409  LOG(FATAL) << "ABORTING DUE TO ABORT IN COMPONENT";
410  }
411  }
412  // This marks the actual end of the computation (since results are available)
413  LOG(INFO) << "Merger process " << mergerpid << " returned";
414  LOG(INFO) << "Simulation process took " << timer.RealTime() << " s";
415 
416  // make sure the rest shuts down
417  for (auto p : childpids) {
418  if (p != mergerpid) {
419  LOG(DEBUG) << "SHUTTING DOWN CHILD PROCESS " << p;
420  kill(p, SIGTERM);
421  }
422  }
423 
424  LOG(DEBUG) << "ShmManager operation " << o2::utils::ShmManager::Instance().isOperational() << "\n";
425  cleanup();
426 
427  // do a quick check to see if simulation produced something reasonable
428  // (mainly useful for continuous integration / automated testing suite)
429  auto returncode = checkresult();
430  if (returncode == 0) {
431  LOG(INFO) << "SIMULATION RETURNED SUCCESFULLY";
432  }
433 
434  return returncode;
