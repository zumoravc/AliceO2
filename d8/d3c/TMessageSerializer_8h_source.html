109 template <typename T>
110 inline void TMessageSerializer::serialize(FairTMessage& tm, const T* input, //
111  const TClass* cl, CacheStreamers streamers, //
112  CompressionLevel compressionLevel)
113 {
114  if (streamers == CacheStreamers::yes) {
115  tm.EnableSchemaEvolution(true);
116  }
117 
118  if (compressionLevel >= 0) {
119  // if negative, skip to use ROOT default
120  tm.SetCompressionLevel(compressionLevel);
121  }
122 
123  // TODO: check what WriateObject and WriteObjectAny are doing
124  if (cl == nullptr) {
125  tm.WriteObject(input);
126  } else {
127  tm.WriteObjectAny(input, cl);
128  }
129 
130  if (streamers == CacheStreamers::yes) {
131  updateStreamers(tm, sStreamers);
132  }
133 }

135 template <typename T>
136 inline std::unique_ptr<T> TMessageSerializer::deserialize(gsl::span<o2::byte> buffer)
137 {
138  TClass* tgtClass = TClass::GetClass(typeid(T));
139  if (tgtClass == nullptr) {
140  std::string error("class is not ROOT-serializable: ");
141  error += typeid(T).name();
142  throw std::runtime_error(error);
143  }
144  // FIXME: we need to add consistency check for buffer data to be serialized
145  // at the moment, TMessage might simply crash if an invalid or inconsistent
146  // buffer is provided
147  FairTMessage tm(buffer);
148  TClass* serializedClass = tm.GetClass();
149  if (serializedClass == nullptr) {
150  std::string error("can not read class info from buffer");
151  throw std::runtime_error(error);
152  }
153  if (tgtClass != serializedClass && serializedClass->GetBaseClass(tgtClass) == nullptr) {
154  std::string error("can not convert serialized class ");
155  error += tm.GetClass()->GetName();
156  error += " into target class ";
157  error += tgtClass->GetName();
158  throw std::runtime_error(error);
159  }
160  return std::unique_ptr<T>(reinterpret_cast<T*>(tm.ReadObjectAny(serializedClass)));
161 }

175 inline void TMessageSerializer::Serialize(FairMQMessage& msg, const TObject* input,
176  TMessageSerializer::CacheStreamers streamers,
177  TMessageSerializer::CompressionLevel compressionLevel)
178 {
179  std::unique_ptr<FairTMessage> tm = std::make_unique<FairTMessage>(kMESS_OBJECT);
180 
181  serialize(*tm, input, input->Class(), streamers, compressionLevel);
182 
183  msg.Rebuild(tm->Buffer(), tm->BufferSize(), FairTMessage::free, tm.get());
184  tm.release();
185 }
