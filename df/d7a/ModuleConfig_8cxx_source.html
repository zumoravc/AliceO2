51 void Module::check() const
52 {
53  // make sure that the channel has <= 2 triggers
54  int ntr = 0;
55  for (int i = MaxChannels; i--;) {
56  ntr += trigChannel[i];
57  }
58  if (ntr > Module::MaxTriggChannels) {
59  print();
60  LOG(FATAL) << "ZDC Module can have at most " << Module::MaxTriggChannels << " trigger channels";
61  }
62 }
