725 inline void GBTFrame::updateSyncCheck(SyncArray& syncArray)
726 {
727  const auto offset = mPrevHWpos ^ 4;
728 
729  for (int s = 0; s < 5; s++)
730  for (int h = 0; h < 4; h++) {
731  const auto hPos = h + offset; // set position of last filled HW
732  // shift in a 1 if the halfword is 0x15
733  if (mFrameHalfWords[s][hPos] == 0x15)
734  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
735  // shift in a 0 if the halfword is 0xA
736  else if (mFrameHalfWords[s][hPos] == 0xA)
737  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
738  // otherwise reset the register to 0
739  else
740  mSyncCheckRegister[s] = 0;
741  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
742  // check the register content for the SYNC pattern
743  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
744  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
745  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
746  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
747  };
748  };
749 }

751 inline void GBTFrame::updateSyncCheck(bool verbose)
752 {
753  const auto offset = mPrevHWpos ^ 4;
754 
755  for (int s = 0; s < 5; s++) {
756  for (int h = 0; h < 4; h++) {

758  // shift in a 1 if the halfword is 0x15

761  // shift in a 0 if the halfword is 0xA

764  // otherwise reset the register to 0
765  else
766  mSyncCheckRegister[s] = 0;
767  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;

784 inline void GBTFrame::getFrameHalfWords()
785 {
786  constexpr adc_t P[5][4] = {{19, 18, 17, 16}, {39, 38, 37, 36}, {63, 62, 61, 60}, {83, 82, 81, 80}, {107, 106, 105, 104}};
787  // i = Stream, j = Halfword
788  for (int i = 0; i < 5; i++) {
789  for (int j = 0; j < 4; j++) {
790  mFrameHalfWords[i][j + mPrevHWpos] = bit(P[i][j], adc_t(4)) |
791  bit(adc_t(P[i][j] - 4), adc_t(3)) |
792  bit(adc_t(P[i][j] - 8), adc_t(2)) |
793  bit(adc_t(P[i][j] - 12), adc_t(1)) |
794  bit(adc_t(P[i][j] - 16), adc_t(0));
795  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
796  };
797  }
798  mPrevHWpos ^= 4; // toggle position of previous HW position
799 }

802 inline void GBTFrame::getAdcValues(ADCRawData& rawData)
803 {
804  // loop over all the 5 data streams
805  for (int s = 0; s < 5; s++) {
806  if (!mSyncPos[s].synched()) {
807  continue;
808  }
809 
810  // assemble the 2 ADC words from 4 half-words. Which halfwords are
811  // used depends on the position of the SYNC pattern.
812  const uint32_t pos = mSyncPos[s].getHalfWordPosition();
813  const uint32_t v0 = (mFrameHalfWords[s][(pos + 2 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 1 + mPrevHWpos) & 7];
814  const uint32_t v1 = (mFrameHalfWords[s][(pos + 4 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 3 + mPrevHWpos) & 7];
815 
816  // add the two ADC values for the current processed stream
817  rawData.add(s, v0, v1);
818  };

820 }
