80 void GeometryTGeo::Build(Int_t loadTrans)
81 {
82  if (isBuilt()) {
83  LOG(WARNING) << "Already built";
84  return; // already initialized
85  }
86 
87  if (!gGeoManager) {
88  // RSTODO: in future there will be a method to load matrices from the CDB
89  LOG(FATAL) << "Geometry is not loaded";
90  }
91 
92  mNumberOfHalves = extractNumberOfHalves();
93  if (!mNumberOfHalves) {
94  return;
95  }
96 
97  // LOG(INFO) << "Number of halves " << mNumberOfHalves;
98  mNumberOfDisks.resize(mNumberOfHalves);
99 
100  mTotalNumberOfSensors = 0;
101  for (Int_t i = 0; i < mNumberOfHalves; i++) {
102  mNumberOfDisks[i] = extractNumberOfDisks(i);
103  // LOG(INFO) << "Number of disks " << mNumberOfDisks[i] << " in half " << i;
104 
105  // use one half only
106  if (i == 0) {
107  mNumberOfLadders.resize(mNumberOfDisks[i]);
108  mNumberOfLaddersPerDisk.resize(mNumberOfDisks[i]);
109  mNumberOfSensorsPerDisk.resize(mNumberOfDisks[i]);
110  mLastSensorIndex.resize(mNumberOfDisks[i]);
111  mLadderIndex2Id.resize(mNumberOfDisks[i]);
112  mLadderId2Index.resize(mNumberOfDisks[i]);
113 
114  // loop over disks
115  for (Int_t j = 0; j < mNumberOfDisks[i]; j++) {
116  mNumberOfLadders[j].resize(MaxSensorsPerLadder + 1);
117  Int_t numberOfLadders = 0;
118  for (Int_t nSensor = MinSensorsPerLadder; nSensor <= MaxSensorsPerLadder; nSensor++) {
119  mNumberOfLadders[j][nSensor] = extractNumberOfLadders(i, j, nSensor);
120  // LOG(INFO) << "Number of ladders with " << nSensor << " sensors is " << mNumberOfLadders[j][nSensor] << " in
121  // disk " << j;
122 
123  numberOfLadders += mNumberOfLadders[j][nSensor];
124  mTotalNumberOfSensors += mNumberOfLadders[j][nSensor] * nSensor;
125  mNumberOfSensorsPerDisk[j] += nSensor * mNumberOfLadders[j][nSensor];
126 
127  } // nSensor
128  mLastSensorIndex[j] = mTotalNumberOfSensors - 1;
129  mNumberOfLaddersPerDisk[j] = numberOfLadders;
130 
131  mLadderIndex2Id[j].resize(numberOfLadders + 1);
132  mLadderId2Index[j].resize(numberOfLadders + 1);
133  Int_t nL = 0;
134  for (Int_t nSensor = MinSensorsPerLadder; nSensor <= MaxSensorsPerLadder; nSensor++) {
135  if (mNumberOfLadders[j][nSensor] == 0)
136  continue;
137  Int_t n = extractNumberOfLadders(i, j, nSensor, nL);
138  } // nSensor
139 
140  LOG(DEBUG) << "MFT: Disk " << j << " has " << mNumberOfSensorsPerDisk[j] << " sensors ";
141 
142  } // disk
143 
144  } // half = 0
145 
146  } // halves
147 
148  mTotalNumberOfSensors *= mNumberOfHalves;
149  LOG(DEBUG) << "MFT: Total number of sensors " << mTotalNumberOfSensors << " in " << mNumberOfHalves << " detector halves";
150 
151  mSensorIndexToLayer.resize(mTotalNumberOfSensors + 1);
152  mLayerMedianZ.resize(mNumberOfDisks[0] + 1);
153  Double_t zLay1[mNumberOfDisks[0]], zLay0[mNumberOfDisks[0]];
154  for (Int_t j = 0; j < mNumberOfDisks[0]; j++) {
155  zLay1[j] = +9999.;
156  zLay0[j] = -9999.;
157  }
158  for (Int_t i = 0; i < mTotalNumberOfSensors; i++) {
159  TGeoHMatrix* hm = extractMatrixSensor(i);
160  Double_t* trans = hm->GetTranslation();
161  Int_t disk = getDisk(i);
162  zLay1[disk] = std::min(zLay1[disk], trans[2]);
163  zLay0[disk] = std::max(zLay0[disk], trans[2]);
164  }
165  for (Int_t j = 0; j < mNumberOfDisks[0]; j++) {
166  mLayerMedianZ[j] = 0.5 * (zLay0[j] + zLay1[j]);
167  // LOG(INFO) << "Disk " << j << " has median z " << mLayerMedianZ[j];
168  }
169  for (Int_t i = 0; i < mTotalNumberOfSensors; i++) {
170  TGeoHMatrix* hm = extractMatrixSensor(i);
171  Double_t* trans = hm->GetTranslation();
172  Int_t disk = getDisk(i);
173  if (trans[2] > mLayerMedianZ[disk]) {
174  mSensorIndexToLayer[i] = 2 * disk;
175  } else {
176  mSensorIndexToLayer[i] = 2 * disk + 1;
177  }
178  // LOG(INFO) << "Sensor " << i << " is in layer " << mSensorIndexToLayer[i] << " translation z " << trans[2] <<
179  // FairLogger::endl;
180  }
181  /*
182  // checks
183  for (Int_t i = 0; i < mTotalNumberOfSensors; i++) {
184  Int_t ladder = getLadder(i);
185  Int_t disk = getDisk(i);
186  Int_t half = getHalf(i);
187  Int_t ladderID = mLadderIndex2Id[disk][ladder];
188  LOG(INFO) << "Index " << i << " half " << half << " disk " << disk << " ladder " << ladder << " geomap " << ladderID
189 ;
190  }
191  */
192  LOG(INFO) << "MFT GeometryTGeo::Build total number of sensors " << mTotalNumberOfSensors;
193  setSize(mTotalNumberOfSensors);
194 
195  fillMatrixCache(loadTrans);
196  /*
197  // checks
198  Int_t index;
199  for (Int_t iH = 0; iH < mNumberOfHalves; iH++) {
200  for (Int_t iD = 0; iD < mNumberOfDisks[iH]; iD++) {
201  for (Int_t iL = 0; iL < mNumberOfLaddersPerDisk[iD]; iL++) {
202  Int_t ladder = mLadderId2Index[iD][iL];
203  Int_t nS = extractNumberOfSensorsPerLadder(iH,iD,iL);
204  for (Int_t iS = 0; iS < nS; iS++) {
205  index = getSensorIndex(iH,iD,iL,iS);
206  LOG(INFO) << "Half " << iH << " disk " << iD << " ladder " << ladder << " ladderID " << iL << " sensor " << iS
207  << " index " << index;
208  } // sensor
209  } // ladder
210  } // disk
211  } // half
212  */
213 }

216 Int_t GeometryTGeo::extractNumberOfSensorsPerLadder(Int_t half, Int_t disk, Int_t ladder) const
217 {
218  Int_t numberOfSensors = 0;
219  Char_t laddername[30];
220  snprintf(laddername, 30, "%s_%d_%d_%d", getMFTLadderPattern(), half, disk, ladder);
221  TGeoVolume* volLadder = gGeoManager->GetVolume(laddername);
222  if (!volLadder) {

224  }
225  // Loop on all ladder nodes, count sensor volumes by checking names
226  Int_t nNodes = volLadder->GetNodes()->GetEntries();
227  for (int j = 0; j < nNodes; j++) {
228  // LOG(INFO) << "GeometryTGeo::extractNumberOfSensorsPerLadder " << half << " " << disk << " " << ladder << " " <<
229  // volLadder->GetNodes()->At(j)->GetName();
230  if (strstr(volLadder->GetNodes()->At(j)->GetName(), getMFTChipPattern())) {
231  numberOfSensors++;
232  }
233  }
234 
235  return numberOfSensors;
236 }

239 Int_t GeometryTGeo::extractNumberOfLadders(Int_t half, Int_t disk, Int_t nsensor) const
240 {
241  Int_t numberOfLadders = 0;
242  Char_t diskname[30];
243  snprintf(diskname, 30, "%s_%d_%d", getMFTDiskPattern(), half, disk);
244  TGeoVolume* volDisk = gGeoManager->GetVolume(diskname);
245  if (!volDisk) {
246  LOG(FATAL) << "can't find volume " << diskname;
247  }
248  // Loop on all disk nodes, count ladder volumes by checking names
249  TObjArray* nodes = volDisk->GetNodes();
250  Int_t nNodes = nodes->GetEntries();
251  Int_t ladderID = -1;
252  for (int j = 0; j < nNodes; j++) {
253  TGeoNode* nd = (TGeoNode*)nodes->At(j);
254  const Char_t* name = nd->GetName();
255  if (strstr(name, getMFTLadderPattern())) {
256  ladderID = extractVolumeCopy(name, Form("%s_%d_%d", getMFTLadderPattern(), half, disk));
257  if (nsensor == extractNumberOfSensorsPerLadder(half, disk, ladderID)) {
258  numberOfLadders++;
259  }
260  }
261  }
262 
263  return numberOfLadders;
264 }

326 {
327  Int_t numberOfHalves = 0;
328 
329  TGeoVolume* volMFT = gGeoManager->GetVolume(getMFTVolPattern());
330  if (!volMFT) {

332  }
333

335  TObjArray* nodes = volMFT->GetNodes();
336  int nNodes = nodes->GetEntriesFast();
337 
338  for (int j = 0; j < nNodes; j++) {
339  Int_t halfID = -1;
340  TGeoNode* nd = (TGeoNode*)nodes->At(j);
341  const Char_t* name = nd->GetName();
342 
343  if (strstr(name, getMFTHalfPattern())) {
344  numberOfHalves++;
345  if ((halfID = extractVolumeCopy(name, getMFTHalfPattern())) < 0) {
346  LOG(FATAL) << "Failed to extract half ID from the " << name;
347  exit(1);
348  }
349  }
350  }
351 
352  return numberOfHalves;
353 }

356 int GeometryTGeo::extractVolumeCopy(const char* name, const char* prefix) const
357 {

371 TGeoHMatrix* GeometryTGeo::extractMatrixSensor(Int_t index) const
372 {
373  Int_t half, disk, ladder, sensor, ladderID;
374  getSensorID(index, half, disk, ladder, sensor);
375  ladderID = mLadderIndex2Id[disk][ladder];
376  // LOG(INFO) << "extractMatrixSensor index " << index << " half " << half << " disk " << disk << " ladder " << ladder
377  // << " ladderID " << ladderID;
378 
379  TString path = Form("/cave_1/%s_0/", getMFTVolPattern());
380  path += Form("%s_%d_%d/%s_%d_%d_%d/%s_%d_%d_%d_%d/%s_%d_%d_%d_%d/%s_1", getMFTHalfPattern(), half, half,
381  getMFTDiskPattern(), half, disk, disk, getMFTLadderPattern(), half, disk, ladderID, ladderID,
382  getMFTChipPattern(), half, disk, ladderID, sensor, getMFTSensorPattern());
383  // LOG(INFO) << "Volume path is " << path.Data();
384 
385  static TGeoHMatrix matTmp;
386  gGeoManager->PushPath();
387 
388  if (!gGeoManager->cd(path.Data())) {
389  gGeoManager->PopPath();
390  LOG(ERROR) << "Error in cd-ing to " << path.Data();
391  return nullptr;
392  } // end if !gGeoManager
393 
394  matTmp = *gGeoManager->GetCurrentMatrix(); // matrix may change after cd
395 
396  // Restore the modeler state.
397  gGeoManager->PopPath();
398 
399  // account for the difference between sensitive layer and physical sensor ticknesses
400  static TGeoTranslation tra(0., 0.5 * (AlpideSegmentation::SensorLayerThickness - AlpideSegmentation::SensorLayerThicknessEff), 0.);
401 
402  matTmp *= tra;
403 
404  return &matTmp;
405 }

408 void GeometryTGeo::fillMatrixCache(Int_t mask)
409 {
410  // populate matrix cache for requested transformations
411  //
412  if (mSize < 1) {
413  LOG(WARNING) << "The method Build was not called yet";
414  Build(mask);
415  return;
416  }
417  // LOG(INFO) << "mask " << mask << " o2::utils::bit2Mask " << o2::utils::bit2Mask(o2::TransformType::L2G) <<
418  // FairLogger::endl;
419  // build matrices
420  if ((mask & o2::utils::bit2Mask(o2::TransformType::L2G)) && !getCacheL2G().isFilled()) {
421  LOG(INFO) << "Loading MFT L2G matrices from TGeo";
422  auto& cacheL2G = getCacheL2G();
423  cacheL2G.setSize(mSize);
424  for (Int_t i = 0; i < mSize; i++) {
425  TGeoHMatrix* hm = extractMatrixSensor(i);
426  cacheL2G.setMatrix(hm ? Mat3D(*hm) : Mat3D(), i);
427  }
428  }
429 
430  if ((mask & o2::utils::bit2Mask(o2::TransformType::T2L)) && !getCacheT2L().isFilled()) {
431  // matrices for Tracking to Local frame transformation
432  LOG(INFO) << "Loading MFT T2L matrices from TGeo";
433  auto& cacheT2L = getCacheT2L();
434  cacheT2L.setSize(mSize);
435  for (int i = 0; i < mSize; i++) {
436  TGeoHMatrix& hm = createT2LMatrix(i);
437  cacheT2L.setMatrix(Mat3D(hm), i);
438  }
439  }
440 
441  if ((mask & o2::utils::bit2Mask(o2::TransformType::T2G)) && !getCacheT2G().isFilled()) {
442  // matrices for Tracking to Global frame transformation
443  LOG(INFO) << "Loading MFT T2G matrices from TGeo";
444  auto& cacheT2G = getCacheT2G();
445  cacheT2G.setSize(mSize);
446  for (int i = 0; i < mSize; i++) {
447  TGeoHMatrix& mat = createT2LMatrix(i);
448  mat.MultiplyLeft(extractMatrixSensor(i));
449  cacheT2G.setMatrix(Mat3D(mat), i);
450  }
451  }
452 }

474 //__________________________________________________________________________
475 Bool_t GeometryTGeo::getSensorID(Int_t index, Int_t& half, Int_t& disk, Int_t& ladder, Int_t& sensor) const
476 {
477  if (index < 0 || index >= mTotalNumberOfSensors)
478  return kFALSE;
479 
480  half = index / (mTotalNumberOfSensors / mNumberOfHalves);
481  index = index % (mTotalNumberOfSensors / mNumberOfHalves);
482  disk = 0;
483  while (index > mLastSensorIndex[disk]) {
484  disk++;
485  }
486  index -= getFirstSensorIndex(disk);
487  Int_t nSensor = MinSensorsPerLadder;
488  Int_t nFirstSensorIndex = 0, nFirstSensorIndexSave = 0;
489  ladder = 0;
490  while (index > ((nFirstSensorIndex += nSensor * mNumberOfLadders[disk][nSensor]) - 1)) {
491  ladder += mNumberOfLadders[disk][nSensor];
492  nFirstSensorIndexSave = nFirstSensorIndex;
493  nSensor++;
494  }
495  index -= nFirstSensorIndexSave;
496  ladder += index / nSensor;
497  sensor = index % nSensor;
498 
499  return kTRUE;
500 }

546 //__________________________________________________________________________
547 Int_t GeometryTGeo::getSensorIndex(Int_t halfID, Int_t diskID, Int_t ladderID, Int_t sensorID) const
548 {
549  Int_t index = 0;
550  Int_t ladder = mLadderId2Index[diskID][ladderID];
551 
552  Int_t nL = 0;
553  Int_t nS = MinSensorsPerLadder;
554  while (ladder > ((nL += mNumberOfLadders[diskID][nS]) - 1)) {
555  index += nS * mNumberOfLadders[diskID][nS];
556  nS++;
557  }
558  ladder -= nL - mNumberOfLadders[diskID][nS];
559 
560  index += ladder * nS;
561  index += sensorID;
562  index += getFirstSensorIndex(diskID);
563  index += halfID * mTotalNumberOfSensors / 2;
564 
565  return index;
566 }
