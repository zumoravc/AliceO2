74 template <typename T>
75 int loadLightTable(T& table, int beta, int NRADBINS, std::string filename)
76 {
77  // Retrieve the light yield table
78  std::string data;
79  std::ifstream input(filename);
80  //std::cout << " ********* Reading data from light table " << filename << std::endl;
81  int radiusbin = 0;
82  int anglebin = 0;
83  int counter = 0;
84  float value;
85  if (input.is_open()) {
86  while (input >> value) {
87  counter++;
88  table[beta][anglebin][radiusbin] = value;
89  //printf(" %f ", value);
90  radiusbin++;
91  if (radiusbin % NRADBINS == 0) {
92  radiusbin = 0;
93  anglebin++;
94  //printf("\n");
95  }
96  }
97  LOG(DEBUG) << "Read " << counter << " values from ZDC data file " << filename;
98  input.close();
99  return counter;
100  } else {
101  LOG(ERROR) << "Could not open file " << filename;
102  return 0;
103  }
104 }

190 // determines detectorID and sectorID from volume and coordinates
191 void Detector::getDetIDandSecID(TString const& volname, Vector3D<float> const& x,
192  Vector3D<float>& xDet, int& detector, int& sector) const
193 {
194  if (volname.BeginsWith("ZN")) {
195  // for the neutron calorimeter
196 
197  if (x.Z() > 0) {
198  detector = ZNA;
199  xDet = x - Vector3D<float>(Geometry::ZNAPOSITION[0], Geometry::ZNAPOSITION[1], Geometry::ZNAPOSITION[2]);
200 
201  } else if (x.Z() < 0) {
202  detector = ZNC;
203  xDet = x - Vector3D<float>(Geometry::ZNCPOSITION[0], Geometry::ZNCPOSITION[1], Geometry::ZNCPOSITION[2]);
204  }
205  // now determine sector/tower
206  if (xDet.X() <= 0.) {
207  if (xDet.Y() <= 0.) {
208  sector = Ch1;
209  } else
210  sector = Ch3;
211  } else {
212  if (xDet.Y() <= 0.) {
213  sector = Ch2;
214  } else {
215  sector = Ch4;
216  }
217  }
218  return;
219 
220  } else if (volname.BeginsWith("ZP")) {
221  // proton calorimeter
222  if (x.Z() > 0) {
223  detector = ZPA; // (NB -> DIFFERENT FROM AliRoot!!!)
224  xDet = x - Vector3D<float>(Geometry::ZPAPOSITION[0], Geometry::ZPAPOSITION[1], Geometry::ZPAPOSITION[2]);
225  } else if (x.Z() < 0) {
226  detector = ZPC; // (NB -> DIFFERENT FROM AliRoot!!!)
227  xDet = x - Vector3D<float>(Geometry::ZPCPOSITION[0], Geometry::ZPCPOSITION[1], Geometry::ZPCPOSITION[2]);
228  }
229 
230  // determine sector/tower
231  if (xDet.X() >= Geometry::ZPDIMENSION[0]) {
232  xDet.SetX(Geometry::ZPDIMENSION[0] - 0.01);
233  } else if (xDet.X() <= -Geometry::ZPDIMENSION[0]) {
234  xDet.SetX(-Geometry::ZPDIMENSION[0] + 0.01);
235  }
236 
237  float xTow = 2. * xDet.X() / (Geometry::ZPDIMENSION[0]);
238  for (int i = 1; i <= 4; i++) {
239  if (xTow >= (i - 3) && xTow < (i - 2)) {
240  sector = i;
241  break;
242  }
243  }
244  return;
245 
246  } else if (volname.BeginsWith("ZE")) {
247  // electromagnetic calorimeter
248  detector = ZEM;
249  xDet = x - Vector3D<float>(Geometry::ZEMPOSITION[0], Geometry::ZEMPOSITION[1], Geometry::ZEMPOSITION[2]);
250  sector = (x.X() > 0.) ? Ch1 : Ch2;
251  return;
252  }
253 
254  assert(false);
255 }

285 Bool_t Detector::ProcessHits(FairVolume* v)
286 {
287  // Method called from MC stepping for the sensitive volumes
288  TString volname = fMC->CurrentVolName();
289  Float_t x[3] = {0., 0., 0.};
290  fMC->TrackPosition(x[0], x[1], x[2]);
291 
292  // determine detectorID and sectorID
293  int detector = -1;
294  int sector = -1;
295  Vector3D<float> xImp;
296  getDetIDandSecID(volname, Vector3D<float>(x[0], x[1], x[2]), xImp, detector, sector);
297 
298  auto stack = (o2::data::Stack*)fMC->GetStack();
299  int trackn = stack->GetCurrentTrackNumber();
300 
301  // find out if we are entering into the detector NEU or PRO for the first time
302  int volID, copy;
303  volID = fMC->CurrentVolID(copy);
304  //printf("\t ---> track %d in vol. %d %d (volID %d) mother %d \n",
305  //trackn, detector, sector, volID, stack->GetCurrentTrack()->GetMother(0));
306 
307  // If the particle is in a ZN or ZP fiber connected to the common PMT
308  // then the assigned sector is 0 (PMC) NB-> does not work for ZEM
309  if ((fMC->CurrentMedium() == mMediumPMCid) && (detector != ZEM))
310  sector = 0;
311  //printf("ProcessHits: x=(%f, %f, %f) \n",x[0], x[1], x[2]);
312  //printf("\tDET %d SEC %d -> XImpact=(%f, %f, %f)\n",detector,sector, xImp.X(), xImp.Y(), xImp.Z());
313 
314  if ((volID == mZNENVVolID || volID == mZPENVVolID || volID == mZEMVolID)) {
315  // there is nothing more to do here as we are not
316  // in the fiber volumes
317  return false;
318  }
319 
320  Float_t p[3] = {0., 0., 0.};
321  Float_t trackenergy = 0.;
322  fMC->TrackMomentum(p[0], p[1], p[2], trackenergy);
323  Float_t eDep = fMC->Edep();
324 
325  int pdgCode = fMC->TrackPid();
326  float lightoutput = 0.;
327  auto currentMediumid = fMC->CurrentMedium();
328  int nphe = 0;
329  if (((currentMediumid == mMediumPMCid) || (currentMediumid == mMediumPMQid))) {
330  if(eDep){
331  int ibeta = 0, iangle = 0, iradius = 0;
332  Bool_t isLightProduced = calculateTableIndexes(ibeta, iangle, iradius);
333  if (isLightProduced) {
334  int charge = 0;
335  if (pdgCode < 10000) {
336  charge = fMC->TrackCharge();
337  } else {
338  charge = TMath::Abs(pdgCode / 10000 - 100000);
339  }
340 
341  //look into the light tables if the particle is charged
342  if (TMath::Abs(charge) > 0) {
343  if (detector == 1 || detector == 4) {
344  iradius = std::min((int)Geometry::ZNFIBREDIAMETER, iradius);
345  lightoutput = charge * charge * mLightTableZN[ibeta][iangle][iradius];
346  //printf(" \t ZNtableEntry[%d %d %d] = %1.5f -> lightoutput %f\n", ibeta, iangle, iradius, mLightTableZN[ibeta][iangle][iradius], lightoutput);
347  } else {
348  iradius = std::min((int)Geometry::ZPFIBREDIAMETER, iradius);
349  lightoutput = charge * charge * mLightTableZP[ibeta][iangle][iradius];
350  //printf(" \t ZPtableEntry[%d %d %d] = %1.5f -> lightoutput %f\n", ibeta, iangle, iradius, mLightTableZP[ibeta][iangle][iradius], lightoutput);
351  }
352  if (lightoutput > 0) {
353  nphe = gRandom->Poisson(lightoutput);
354  //printf(" \t\t-> nphe %d \n", nphe);
355  }
356  }
357  }
358  }
359  }
360 
361  auto tof = 1.e09 * fMC->TrackTime(); //TOF in ns
362 
363  if (o2::zdc::ZDCSimParam::Instance().recordSpatialResponse) {
364  // some diagnostic; trying to really get the pixel fired
365  if (nphe > 0) {
366  if (detector == ZNA || detector == ZNC) {
367  mNeutronResponseImage.setDetectorID(detector);
368  mNeutronResponseImage.addPhoton(x[0], x[1], nphe);
369  mNeutronResponseImage.setHitTime(tof);
370  }
371  if (detector == ZPA || detector == ZPC) {
372  mProtonResponseImage.setDetectorID(detector);
373  mProtonResponseImage.addPhoton(x[0], x[1], nphe);
374  mProtonResponseImage.setHitTime(tof);
375  }
376  }
377  }
378 
379  // A new hit is created when there is nothing yet for this det + sector
380  if (mCurrentHitsIndices[detector - 1][sector] == -1) {
381  bool issecondary = trackn != stack->getCurrentPrimaryIndex();
382  //if(!issecondary) printf(" !!! primary track (index %d)\n",stack->getCurrentPrimaryIndex());
383 
384  mTotLightPMC = mTotLightPMQ = 0;
385  if (currentMediumid == mMediumPMCid) {
386  mTotLightPMC = nphe;
387  } else if (currentMediumid == mMediumPMQid) {
388  mTotLightPMQ = nphe;
389  }
390 
391  Vector3D<float> pos(x[0], x[1], x[2]);
392  Vector3D<float> mom(p[0], p[1], p[2]);
393  addHit(trackn, mLastPrincipalTrackEntered, issecondary, trackenergy, detector, sector,
394  pos, mom, tof, xImp, eDep, mTotLightPMC, mTotLightPMQ);
395  stack->addHit(GetDetId());
396  mCurrentHitsIndices[detector - 1][sector] = mHits->size() - 1;
397 
398  mXImpact = xImp;
399  //printf("### NEW HIT CREATED in vol %d %d for track %d (mother: %d) \t light %1.0f %1.0f\n",
400  //detector, sector, trackn, stack->GetCurrentTrack()->GetFirstMother(), mTotLightPMC, mTotLightPMQ);
401  return true;
402 
403  } else {
404  auto& curHit = (*mHits)[mCurrentHitsIndices[detector - 1][sector]];
405  // summing variables that needs to be updated (Eloss and light yield)
406  curHit.setNoNumContributingSteps(curHit.getNumContributingSteps() + 1);
407  int nPMC{0}, nPMQ{0};
408  if (currentMediumid == mMediumPMCid) {
409  mTotLightPMC += nphe;
410  nPMC = nphe;
411  } else if (currentMediumid == mMediumPMQid) {
412  mTotLightPMQ += nphe;
413  nPMQ = nphe;
414  }
415  float incenloss = curHit.GetEnergyLoss() + eDep;
416  if (nphe > 0) {
417  curHit.SetEnergyLoss(incenloss);
418  curHit.setPMCLightYield(curHit.getPMCLightYield() + nPMC);
419  curHit.setPMQLightYield(curHit.getPMQLightYield() + nPMQ);
420  //printf(" >>> Hit updated in vol %d %d for track %d (mother: %d) \t light %1.0f %1.0f \n",
421  //detector, sector, trackn, stack->GetCurrentTrack()->GetFirstMother(), curHit.getPMCLightYield(), curHit.getPMQLightYield());
422  }
423  return true;
424  }
425  return false;
426 }

428 // function to create hit structure from a SpatialResponseImage
429 // idea is to use this from a fast sim generating the response
430 bool Detector::createHitsFromImage(SpatialPhotonResponse const& image, int detector)
431 {
432  // one image will make one hit per sector
433  Vector3D<float> xImp(0., 0., 0.); // good value
434 
435  const int Nx = image.getNx();
436  const int Ny = image.getNy();
437  const auto& pixels = image.getImageData();
438 
439  // could be put inside the image class
440  auto determineSectorID = [Nx, Ny](int detector, int x, int y) {
441  if (detector == ZNA || detector == ZNC) {
442  if (x < Nx / 2) {
443  if (y < Ny / 2) {
444  return (int)Ch1;
445  } else {
446  return (int)Ch3;
447  }
448  } else {
449  if (y >= Ny / 2) {
450  return (int)Ch4;
451  } else {
452  return (int)Ch2;
453  }
454  }
455  }
456 
457  if (detector == ZPA || detector == ZPC) {
458  auto i = (int)(4.f * x / Nx);
459  return (int)(i + 1);
460  }
461  return -1;
462  };
463 
464  auto determineMediumID = [this](int detector, int x, int y) {
465  // it is a simple checkerboard pattern
466  return ((x + y) % 2 == 0) ? mMediumPMCid : mMediumPMQid;
467  };
468 
469  // loop over x = columns
470  for (int x = 0; x < Nx; ++x) {
471  // loop over y = rows
472  for (int y = 0; y < Ny; ++y) {
473  // get sector
474  int sector = determineSectorID(detector, x, y);
475  // get medium PMQ and PMC
476  int currentMediumid = determineMediumID(detector, x, y);
477  // LOG(INFO) << " x " << x << " y " << y << " sec " << sector << " medium " << currentMediumid;
478  int nphe = pixels[x][y];
479  float tof = 0.; // needs to be in nanoseconds ---> to be filled later on (should be meta-data of image or calculated otherwise)
480  float trackenergy = 0; // energy of the primary (need to fill good value)
481  createOrAddHit(detector,
482  sector,
483  currentMediumid,
484  0 /*issecondary ---> don't know in fast sim */,
485  nphe,
486  0 /* trackn */,
487  0 /* parent */,
488  tof,
489  trackenergy,
490  xImp,
491  0. /* eDep */, 0 /* x */, 0. /* y */, 0. /* z */, 0. /* px */, 0. /* py */, 0. /* pz */);
492  } // end loop over y
493  } // end loop over x
494  return true;
495 } // end function
