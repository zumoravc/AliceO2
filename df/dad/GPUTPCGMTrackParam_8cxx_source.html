309  }

311  nMissed2++;

315  }

318  }
319  }

325  }
326

328  dEdx.computedEdx(*dEdxOut, param);
329  }
330  Alpha = prop.GetAlpha();

332  for (int k = 0; k < 3; k++) // max 3 attempts
333  {
334  int err = prop.PropagateToXAlpha(param.rec.TrackReferenceX, Alpha, 0);
335  ConstrainSinPhi();

337  break;
338  }
339  float dAngle = floor(CAMath::ATan2(mP[0], mX) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
340  Alpha += dAngle;
341  if (err || k == 2) {
342  Rotate(dAngle);
343  ConstrainSinPhi();
344  break;
345  }
346  }
347  } else if (CAMath::Abs(mP[0]) > mX * CAMath::Tan(kSectAngle / 2.f)) {

349  Rotate(dAngle);
350  ConstrainSinPhi();
351  Alpha += dAngle;
352  }
353  if (Alpha > M_PI) {
354  Alpha -= 2 * M_PI;
355  } else if (Alpha <= -M_PI) {
356  Alpha += 2 * M_PI;
357  }
358 
359  return (ok);
360 }

362 GPUd() void GPUTPCGMTrackParam::MirrorTo(GPUTPCGMPropagator& GPUrestrict() prop, float toY, float toZ, bool inFlyDirection, const GPUParam& param, unsigned char row, unsigned char clusterState, bool mirrorParameters)
363 {
364  if (mirrorParameters) {
365  prop.Mirror(inFlyDirection);
366  }
367  float err2Y, err2Z;
368  prop.GetErr2(err2Y, err2Z, param, toZ, row, clusterState);
369  prop.Model().Y() = mP[0] = toY;
370  prop.Model().Z() = mP[1] = toZ;
371  if (mC[0] < err2Y) {
372  mC[0] = err2Y;
373  }
374  if (mC[2] < err2Z) {
375  mC[2] = err2Z;
376  }

378  mC[5] = mC[5] > 0 ? 0.1f : -0.1f;
379  }

381  mC[9] = 1.f;
382  }
383  mC[1] = mC[4] = mC[6] = mC[8] = mC[11] = mC[13] = 0;
384  prop.SetTrack(this, prop.GetAlpha());
385  mNDF = -3;
386  mChi2 = 0;

389 GPUd() int GPUTPCGMTrackParam::MergeDoubleRowClusters(int& ihit, int wayDirection, GPUTPCGMMergedTrackHit* GPUrestrict() clusters, const GPUTPCGMMerger* GPUrestrict() merger, GPUTPCGMPropagator& GPUrestrict() prop, float& GPUrestrict() xx, float& GPUrestrict() yy, float& GPUrestrict() zz, int maxN, float clAlpha, unsigned char& GPUrestrict() clusterState, bool rejectChi2)
390 {
391  if (ihit + wayDirection >= 0 && ihit + wayDirection < maxN && clusters[ihit].row == clusters[ihit + wayDirection].row && clusters[ihit].slice == clusters[ihit + wayDirection].slice && clusters[ihit].leg == clusters[ihit + wayDirection].leg) {
392  float maxDistY, maxDistZ;
393  prop.GetErr2(maxDistY, maxDistZ, merger->Param(), zz, clusters[ihit].row, 0);
394  maxDistY = (maxDistY + mC[0]) * 20.f;
395  maxDistZ = (maxDistZ + mC[2]) * 20.f;
396  int noReject = 0; // Cannot reject if simple estimation of y/z fails (extremely unlike case)
397  if (CAMath::Abs(clAlpha - prop.GetAlpha()) > 1.e-4f) {
398  noReject = prop.RotateToAlpha(clAlpha);
399  }
400  float projY = 0, projZ = 0;
401  if (noReject == 0) {
402  noReject |= prop.GetPropagatedYZ(xx, projY, projZ);
403  }
404  float count = 0.f;
405  xx = yy = zz = 0.f;
406  clusterState = 0;
407  while (true) {
408  float clx, cly, clz, clamp;
409  if (merger->Param().earlyTpcTransform) {
410  const float zOffset = (clusters[ihit].slice < 18) == (clusters[0].slice < 18) ? mTZOffset : -mTZOffset;
411  clx = clusters[ihit].x;
412  cly = clusters[ihit].y;
413  clz = clusters[ihit].z - zOffset;
414  clamp = clusters[ihit].amp;
415  } else {

652  }

700 {

702  return;
703  }
704  if (Merger->Param().rec.loopInterpolationInExtraPass && phase2 == false) {
705  StoreAttachMirror(Merger, slice, iRow, iTrack, false, 0, toY, 0, 0, 0, 0, prop.GetAlpha());

707  }
708  float X = mP[2] > 0 ? mP[0] : -mP[0];
709  float toX = mP[2] > 0 ? toY : -toY;

713  return;

723  return;

743  Y += dS * SinPhi + h4;
744  Z += dS * mP[3];

750  // printf("count %d: At X %f Y %f Z %f SinPhi %f\n", count, mP[2] > 0 ? -Y : Y, mP[2] > 0 ? X : -X, Z, SinPhi);

755  for (int j = iRow; j >= 0 && j < GPUCA_ROW_COUNT && found < 3; j += step) {
756  float rowX = mX + Merger->Param().tpcGeometry.Row2X(j) - myRowX;

758  // printf("Attempt row %d\n", j);

792  const float r2 = (r + CAMath::Sqrt(xp * xp + yp * yp)) / 2.f; // Improve the radius by taking into acount both points we know (00 and xy).
793  xp = mX + dxf * r2;

798  // printf("Tan %f %f (%f %f)\n", atana, atanb, mX - xp, mP[0] - yp);

811  float zMin = CAMath::Min(tz1, tz2);
812  if (zMin < 0 && zMin - mTZOffset < -250) {
813  deltaZ = zMin - mTZOffset + 250;
814  } else if (zMax > 0 && zMax - mTZOffset > 250) {
815  deltaZ = zMax - mTZOffset - 250;
816  }
817  if (zMin < 0 && zMax - mTZOffset > 0) {
818  deltaZ = zMax - mTZOffset;
819  } else if (zMax > 0 && zMin - mTZOffset < 0) {
820  deltaZ = zMin - mTZOffset;

823  mTZOffset += deltaZ;
824  mP[1] -= deltaZ;
825  } else {
