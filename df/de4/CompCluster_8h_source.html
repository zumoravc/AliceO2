26 class CompCluster
27 {
28  public:
29  static constexpr int NBitsRow = 9; // number of bits for the row
30  static constexpr int NBitsCol = 10; // number of bits for the column
31  static constexpr int NBitsPattID = 11; // number of bits for the pattern ID
32  private:
33  static constexpr UInt_t RowMask = (0x1 << NBitsRow) - 1;
34  static constexpr UInt_t ColMask = (0x1 << NBitsCol) - 1;
35  static constexpr UInt_t PattIDMask = (0x1 << NBitsPattID) - 1;
36  static constexpr UInt_t FlagBit = 0x1 << (NBitsRow + NBitsCol + NBitsPattID);
37  //
39  UInt_t mData;
40 
41  void sanityCheck();
42 
43  public:
44  static constexpr unsigned short InvalidPatternID = (0x1 << NBitsPattID) - 1; // All 11 bits of pattern ID are 1
45  CompCluster(UShort_t row = 0, UShort_t col = 0, UShort_t patt = 0)
46  {
47  set(row, col, patt);
48  }
49 
50  void set(UShort_t row, UShort_t col, UShort_t patt)
51  {
52  mData = (row & RowMask) | ((col & ColMask) << NBitsRow) | ((patt & PattIDMask) << (NBitsRow + NBitsCol));
53  }
54 
55  UShort_t getRow() const { return mData & RowMask; }
56  UShort_t getCol() const { return (mData >> NBitsRow) & ColMask; }
57  UShort_t getPatternID() const { return (mData >> (NBitsRow + NBitsCol)) & PattIDMask; }
58  bool getFlag() const { return (mData & FlagBit) == FlagBit; }
59 
60  void setRow(UShort_t r)
61  {
62  mData &= ~RowMask;
63  mData |= (r & RowMask);
64  }
65  void setCol(UShort_t c)
66  {
67  mData &= ~(ColMask << NBitsRow);
68  mData |= (c & ColMask) << NBitsRow;
69  };
70  void setPatternID(UShort_t p)
71  {
72  mData &= ~(PattIDMask << (NBitsRow + NBitsCol));
73  mData |= (p & PattIDMask) << (NBitsRow + NBitsCol);
74  }
75  void setFlag(bool v)
76  {
77  mData &= ~FlagBit;
78  if (v) {
79  mData |= FlagBit;
80  }
81  }
82 
83  void print() const;
84 
85  ClassDefNV(CompCluster, 2);
86 };
