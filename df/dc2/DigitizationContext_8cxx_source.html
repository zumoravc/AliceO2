89 bool DigitizationContext::checkVertexCompatibility(bool verbose) const
90 {
91  if (mMaxPartNumber == 1) {
92  return true;
93  }
94 
95  auto checkVertexPair = [](Point3D<double> const& p1, Point3D<double> const& p2) -> bool {
96  return (p2 - p1).Mag2() < 1E-6;
97  };
98 
99  std::vector<TChain*> kinematicschain;
100  std::vector<TBranch*> headerbranches;
101  std::vector<o2::dataformats::MCEventHeader*> headers;
102  std::vector<Point3D<double>> vertices;
103  initSimKinematicsChains(kinematicschain);
104  bool consistent = true;
105  if (kinematicschain.size() > 0) {
106  headerbranches.resize(kinematicschain.size(), nullptr);
107  headers.resize(kinematicschain.size(), nullptr);
108  // loop over all collisions in this context
109  int collisionID = 0;
110  for (auto& collision : getEventParts()) {
111  collisionID++;
112  vertices.clear();
113  for (auto& part : collision) {
114  const auto source = part.sourceID;
115  const auto entry = part.entryID;
116  auto chain = kinematicschain[source];
117  if (!headerbranches[source]) {
118  headerbranches[source] = chain->GetBranch("MCEventHeader.");
119  headerbranches[source]->SetAddress(&headers[source]);
120  }
121  // get the MCEventHeader to read out the vertex
122  headerbranches[source]->GetEntry(entry);
123  auto header = headers[source];
124  vertices.emplace_back(header->GetX(), header->GetY(), header->GetZ());
125  }
126  // analyse vertex matching
127  bool thiscollision = true;
128  const auto& p1 = vertices[0];
129  for (int j = 1; j < vertices.size(); ++j) {
130  const auto& p2 = vertices[j];
131  bool thischeck = checkVertexPair(p1, p2);
132  thiscollision &= thischeck;
133  }
134  if (verbose && !thiscollision) {
135  std::stringstream text;
136  text << "Found inconsistent vertices for digit collision ";
137  text << collisionID << " : ";
138  for (auto& p : vertices) {
139  text << p << " ";
140  }
141  LOG(ERROR) << text.str();
142  }
143  consistent &= thiscollision;
144  }
145  }
146  return consistent;
147 }
