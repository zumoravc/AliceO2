27 BOOST_AUTO_TEST_CASE(TestInputRecord)
28 {
29  // Create the routes we want for the InputRecord
30  InputSpec spec1{"x", "TPC", "CLUSTERS", 0, Lifetime::Timeframe};
31  InputSpec spec2{"y", "ITS", "CLUSTERS", 0, Lifetime::Timeframe};
32  InputSpec spec3{"z", "TST", "EMPTY", 0, Lifetime::Timeframe};
33 
34  size_t i = 0;
35  auto createRoute = [&i](const char* source, InputSpec& spec) {
36  return InputRoute{
37  spec,
38  i++,
39  source};
40  };
41 
43  std::vector<InputRoute> schema = {
44  createRoute("x_source", spec1),
45  createRoute("y_source", spec2),
46  createRoute("z_source", spec3)};
47  // First of all we test if an empty registry behaves as expected, raising a
48  // bunch of exceptions.
49  InputRecord emptyRecord(schema, {[](size_t) { return DataRef{nullptr, nullptr, nullptr}; }, 0});
50 
51  BOOST_CHECK_EXCEPTION(emptyRecord.get("x"), std::exception, any_exception);
52  BOOST_CHECK_EXCEPTION(emptyRecord.get("y"), std::exception, any_exception);
53  BOOST_CHECK_EXCEPTION(emptyRecord.get("z"), std::exception, any_exception);
54  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(0), std::exception, any_exception);
55  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(1), std::exception, any_exception);
56  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(2), std::exception, any_exception);
57  // Then we actually check with a real set of inputs.
58 
59  std::vector<void*> inputs;
60 
61  auto createMessage = [&inputs](DataHeader& dh, int value) {
62  DataProcessingHeader dph{0, 1};
63  Stack stack{dh, dph};
64  void* header = malloc(stack.size());
65  void* payload = malloc(sizeof(int));
66  memcpy(header, stack.data(), stack.size());
67  memcpy(payload, &value, sizeof(int));
68  inputs.emplace_back(header);
69  inputs.emplace_back(payload);
70  };
71 
72  auto createEmpty = [&inputs]() {
73  inputs.emplace_back(nullptr);
74  inputs.emplace_back(nullptr);
75  };
76 
77  DataHeader dh1;
78  dh1.dataDescription = "CLUSTERS";
79  dh1.dataOrigin = "TPC";
80  dh1.subSpecification = 0;
81  dh1.payloadSerializationMethod = o2::header::gSerializationMethodNone;
82  DataHeader dh2;
83  dh2.dataDescription = "CLUSTERS";
84  dh2.dataOrigin = "ITS";
85  dh2.subSpecification = 0;
86  dh2.payloadSerializationMethod = o2::header::gSerializationMethodNone;
87  createMessage(dh1, 1);
88  createMessage(dh2, 2);
89  createEmpty();
90  InputSpan span{[&inputs](size_t i) { return DataRef{nullptr, static_cast<char const*>(inputs[2 * i]), static_cast<char const*>(inputs[2 * i + 1])}; }, inputs.size() / 2};
91  InputRecord record{schema, std::move(span)};
92 
93  // Checking we can get the whole ref by name
94  BOOST_CHECK_NO_THROW(record.get("x"));
95  BOOST_CHECK_NO_THROW(record.get("y"));
96  BOOST_CHECK_NO_THROW(record.get("z"));
97  auto ref00 = record.get("x");
98  auto ref10 = record.get("y");
99  auto ref20 = record.get("z");
100  BOOST_CHECK_EXCEPTION(record.get("err"), std::exception, any_exception);
101 
102  // Or we can get it positionally
103  BOOST_CHECK_NO_THROW(record.get("x"));
104  auto ref01 = record.getByPos(0);
105  auto ref11 = record.getByPos(1);
106  BOOST_CHECK_EXCEPTION(record.getByPos(10), std::exception, any_exception);
107 
108  // This should be exactly the same pointers
109  BOOST_CHECK_EQUAL(ref00.header, ref01.header);
110  BOOST_CHECK_EQUAL(ref00.payload, ref01.payload);
111  BOOST_CHECK_EQUAL(ref10.header, ref11.header);
112  BOOST_CHECK_EQUAL(ref10.payload, ref11.payload);
113 
114  BOOST_CHECK_EQUAL(record.isValid("x"), true);
115  BOOST_CHECK_EQUAL(record.isValid("y"), true);
116  BOOST_CHECK_EQUAL(record.isValid("z"), false);
117 
118  BOOST_CHECK_EQUAL(record.isValid(0), true);
119  BOOST_CHECK_EQUAL(record.isValid(1), true);
120  BOOST_CHECK_EQUAL(record.isValid(2), false);
121  // This by default is a shortcut for
122  //
123  // *static_cast<int const *>(record.get("x").payload);
124  //
125  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
126  BOOST_CHECK_EQUAL(record.get<int>("y"), 2);
127  // A few more time just to make sure we are not stateful..
128  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
129  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
130 
131  // test the iterator
132  int position = 0;
133  for (auto input = record.begin(), end = record.end(); input != end; input++, position++) {
134  if (position == 0) {
135  BOOST_CHECK(input.matches("TPC") == true);
136  BOOST_CHECK(input.matches("TPC", "CLUSTERS") == true);
137  BOOST_CHECK(input.matches("ITS", "CLUSTERS") == false);
138  }
139  if (position == 1) {
140  BOOST_CHECK(input.matches("ITS") == true);
141  BOOST_CHECK(input.matches("ITS", "CLUSTERS") == true);
142  BOOST_CHECK(input.matches("TPC", "CLUSTERS") == false);
143  }
144  // check if invalid slots are filtered out by the iterator
145  BOOST_CHECK(position != 2);
146  }
147 
148  // the 2-level iterator to access inputs and their parts
149  // all inputs have 1 part, we check the first input
150  BOOST_CHECK(record.begin().size() == 1);
151  // the end-instance of the inputs has no parts
152  BOOST_CHECK(record.end().size() == 0);
153  // thus there is no element and begin == end
154  BOOST_CHECK(record.end().begin() == record.end().end());
155 }
