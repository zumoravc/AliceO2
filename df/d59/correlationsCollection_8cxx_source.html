43  void process(aod::Tracks const& tracks)
44  {
45  for (auto& track : tracks) {
46  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
47  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
48  float pt = fabs(1.0f / track.signed1Pt());
49 
50  etaphi(eta, phi, pt);
51  }
52  }
53 };

275  template <typename T>
276  bool conversionCuts(T const& track1, T const& track2)
277  {
278  // skip if like sign
279  if (track1.charge() * track2.charge() > 0)
280  return false;
281 
282  bool decision = false;
283 
284  if (conversionCut(track1, track2, Photon, cfgPairCutPhoton))
285  decision = true;
286  if (conversionCut(track1, track2, K0, cfgPairCutK0))
287  decision = true;
288  if (conversionCut(track1, track2, Lambda, cfgPairCutLambda))
289  decision = true;
290  if (conversionCut(track2, track1, Lambda, cfgPairCutLambda))
291  decision = true;
292  if (conversionCut(track1, track2, Phi, cfgPairCutPhi))
293  decision = true;
294  if (conversionCut(track1, track2, Rho, cfgPairCutRho))
295  decision = true;
296 
297  return decision;
298  }

300  template <typename T>
301  bool conversionCut(T const& track1, T const& track2, PairCuts conv, double cut)
302  {
303  //LOGF(info, "pt is %f %f", track1.pt2(), track2.pt2());
304 
305  if (cut < 0)
306  return false;
307 
308  double massD1, massD2, massM;
309 
310  switch (conv) {
311  case Photon:
312  massD1 = 0.51e-3;
313  massD2 = 0.51e-3;
314  massM = 0;
315  break;
316  case K0:
317  massD1 = 0.1396;
318  massD2 = 0.1396;
319  massM = 0.4976;
320  break;
321  case Lambda:
322  massD1 = 0.9383;
323  massD2 = 0.1396;
324  massM = 1.115;
325  break;
326  case Phi:
327  massD1 = 0.4937;
328  massD2 = 0.4937;
329  massM = 1.019;
330  break;
331  case Rho:
332  massD1 = 0.1396;
333  massD2 = 0.1396;
334  massM = 0.770;
335  break;
336  }
337 
338  auto massC = getInvMassSquaredFast(track1, massD1, track2, massD2);
339 
340  if (TMath::Abs(massC - massM * massM) > cut * 5)
341  return false;
342 
343  massC = getInvMassSquared(track1, massD1, track2, massD2);
344  qa.mControlConvResoncances->Fill(static_cast<int>(conv), massC - massM * massM);
345  if (massC > (massM - cut) * (massM - cut) && massC < (massM + cut) * (massM + cut))
346  return true;
347 
348  return false;
349  }

351  template <typename T>
352  double getInvMassSquared(T const& track1, double m0_1, T const& track2, double m0_2)
353  {
354  // calculate inv mass squared
355  // same can be achieved, but with more computing time with
356  /*TLorentzVector photon, p1, p2;
357  p1.SetPtEtaPhiM(triggerParticle->Pt(), triggerEta, triggerParticle->Phi(), 0.510e-3);
358  p2.SetPtEtaPhiM(particle->Pt(), eta[j], particle->Phi(), 0.510e-3);
359  photon = p1+p2;
360  photon.M()*/
361 
362  float tantheta1 = 1e10;
363 
364  if (track1.eta2() < -1e-10 || track1.eta2() > 1e-10) {
365  float expTmp = TMath::Exp(-track1.eta2());
366  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
367  }
368 
369  float tantheta2 = 1e10;
370  if (track2.eta2() < -1e-10 || track2.eta2() > 1e-10) {
371  float expTmp = TMath::Exp(-track2.eta2());
372  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
373  }
374 
375  float e1squ = m0_1 * m0_1 + track1.pt2() * track1.pt2() * (1.0 + 1.0 / tantheta1 / tantheta1);
376  float e2squ = m0_2 * m0_2 + track2.pt2() * track2.pt2() * (1.0 + 1.0 / tantheta2 / tantheta2);
377 
378  float mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (track1.pt2() * track2.pt2() * (TMath::Cos(track1.phi2() - track2.phi2()) + 1.0 / tantheta1 / tantheta2)));
379 
380  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
381 
382  return mass2;
383  }

385  template <typename T>
386  double getInvMassSquaredFast(T const& track1, double m0_1, T const& track2, double m0_2)
387  {
388  // calculate inv mass squared approximately
389 
390  const float eta1 = track1.eta2();
391  const float eta2 = track2.eta2();
392  const float phi1 = track1.phi2();
393  const float phi2 = track2.phi2();
394  const float pt1 = track1.pt2();
395  const float pt2 = track2.pt2();
396 
397  float tantheta1 = 1e10;
398 
399  if (eta1 < -1e-10 || eta1 > 1e-10) {
400  float expTmp = 1.0 - eta1 + eta1 * eta1 / 2 - eta1 * eta1 * eta1 / 6 + eta1 * eta1 * eta1 * eta1 / 24;
401  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
402  }
403 
404  float tantheta2 = 1e10;
405  if (eta2 < -1e-10 || eta2 > 1e-10) {
406  float expTmp = 1.0 - eta2 + eta2 * eta2 / 2 - eta2 * eta2 * eta2 / 6 + eta2 * eta2 * eta2 * eta2 / 24;
407  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
408  }
409 
410  float e1squ = m0_1 * m0_1 + pt1 * pt1 * (1.0 + 1.0 / tantheta1 / tantheta1);
411  float e2squ = m0_2 * m0_2 + pt2 * pt2 * (1.0 + 1.0 / tantheta2 / tantheta2);
412 
413  // fold onto 0...pi
414  float deltaPhi = TMath::Abs(phi1 - phi2);
415  while (deltaPhi > TMath::TwoPi())
416  deltaPhi -= TMath::TwoPi();
417  if (deltaPhi > TMath::Pi())
418  deltaPhi = TMath::TwoPi() - deltaPhi;
419 
420  float cosDeltaPhi = 0;
421  if (deltaPhi < TMath::Pi() / 3)
422  cosDeltaPhi = 1.0 - deltaPhi * deltaPhi / 2 + deltaPhi * deltaPhi * deltaPhi * deltaPhi / 24;
423  else if (deltaPhi < 2 * TMath::Pi() / 3)
424  cosDeltaPhi = -(deltaPhi - TMath::Pi() / 2) + 1.0 / 6 * TMath::Power((deltaPhi - TMath::Pi() / 2), 3);
425  else
426  cosDeltaPhi = -1.0 + 1.0 / 2.0 * (deltaPhi - TMath::Pi()) * (deltaPhi - TMath::Pi()) - 1.0 / 24.0 * TMath::Power(deltaPhi - TMath::Pi(), 4);
427 
428  double mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (pt1 * pt2 * (cosDeltaPhi + 1.0 / tantheta1 / tantheta2)));
429 
430  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
431 
432  return mass2;
433  }
