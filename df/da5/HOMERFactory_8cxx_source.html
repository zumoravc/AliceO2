92  return nullptr;

96  return nullptr;

100  if (mFctCreateReaderFromTCPPort != nullptr && (pReader = (((AliHLTHOMERReaderCreateFromTCPPort_t)mFctCreateReaderFromTCPPort)(hostname, port))) == nullptr) {

102  }

203 int HOMERFactory::LoadHOMERLibrary()
204 {
205  // delete a writer
206 
207  // the actual deletion function is inside the HOMER library
208  int iResult = -EBADF;
209  const char** library = &sLibraries[0];
210  int* refcount = &sLibRefCount[0];
211  do {
212  mHandle = dlopen(*library, RTLD_NOW);
213  if (mHandle) {
214  ++(*refcount);
215  mLoadedLib = *library;
216  iResult = 1;
217  break;
218  }
219  ++library;
220  ++refcount;
221  } while ((*library) != nullptr);
222 
223  if (iResult > 0 && *library != nullptr) {
224  // print compile info
225  using CompileInfo = void (*)(char*&, char*&);
226 
227  mFctCreateReaderFromTCPPort = (void (*)())dlsym(mHandle, ALIHLTHOMERREADER_CREATE_FROM_TCPPORT);
228  mFctCreateReaderFromTCPPorts = (void (*)())dlsym(mHandle, ALIHLTHOMERREADER_CREATE_FROM_TCPPORTS);
229  mFctCreateReaderFromBuffer = (void (*)())dlsym(mHandle, ALIHLTHOMERREADER_CREATE_FROM_BUFFER);
230  mFctDeleteReader = (void (*)())dlsym(mHandle, ALIHLTHOMERREADER_DELETE);
231  mFctCreateWriter = (void (*)())dlsym(mHandle, ALIHLTHOMERWRITER_CREATE);
232  mFctDeleteWriter = (void (*)())dlsym(mHandle, ALIHLTHOMERWRITER_DELETE);
233  if (mFctCreateReaderFromTCPPort == nullptr ||
234  mFctCreateReaderFromTCPPorts == nullptr ||
235  mFctCreateReaderFromBuffer == nullptr ||
236  mFctDeleteReader == nullptr ||
237  mFctCreateWriter == nullptr ||
238  mFctDeleteWriter == nullptr) {
239  iResult = -ENOSYS;
240  } else {
241  }
242  }
243  if (iResult < 0 || *library == nullptr) {
244  mFctCreateReaderFromTCPPort = nullptr;
245  mFctCreateReaderFromTCPPorts = nullptr;
246  mFctCreateReaderFromBuffer = nullptr;
247  mFctDeleteReader = nullptr;
248  mFctCreateWriter = nullptr;
249  mFctDeleteWriter = nullptr;
250  }
251 
252  return iResult;
253 }

255 int HOMERFactory::UnloadHOMERLibrary()
256 {
257  // unload HOMER library
258  int iResult = 0;
259 
260  if (mLoadedLib != nullptr) {
261  // Find the corresponding reference count.
262  const char** library = &sLibraries[0];
263  int* refcount = &sLibRefCount[0];
264  while (*library != nullptr) {
265  if (strcmp(*library, mLoadedLib) == 0)
266  break;
267  ++library;
268  ++refcount;
269  }
270 
271  // Decrease the reference count and remove the library if it is zero.
272  if (*refcount >= 0)
273  --(*refcount);
274  if (*refcount == 0) { /* Matthias 2014-05-08 that part of the code has been disabled for the ROOT-independent
275  version.
276  TODO: dlopen is maintaining an internal reference count, which also makes the
277  counters here obsolete
278 
279  // Check that the library we are trying to unload is actually the last library
280  // in the gSystem->GetLibraries() list. If not then we must abort the removal.
281  // This is because of a ROOT bug/feature/limitation. If we try unload the library
282  // then ROOT will also wipe all libraries in the gSystem->GetLibraries() list
283  // following the library we want to unload.
284  TString libstring = gSystem->GetLibraries();
285  TString token, lastlib;
286  Ssiz_t from = 0;
287  Int_t numOfLibs = 0, posOfLib = -1;
288  while (libstring.Tokenize(token, from, " "))
289  {
290  ++numOfLibs;
291  lastlib = token;
292  if (token.Contains(mLoadedLib)) posOfLib = numOfLibs;
293  }
294  if (numOfLibs == posOfLib)
295  {
296  gSystem->Unload(mLoadedLib);
297 
298  // Check that the library is gone, since Unload() does not return a status code.
299  libstring = gSystem->GetLibraries();
300  if (libstring.Contains(mLoadedLib)) iResult = -EBADF;
301  }
302  else
303  {
304  AliHLTLogging log;
305  log.LoggingVarargs(kHLTLogWarning, Class_Name(), FUNCTIONNAME(), __FILE__, __LINE__,
306  Form("ROOT limitation! Cannot properly cleanup and unload the shared"
307  " library '%s' since another library '%s' was loaded afterwards. Trying to"
308  " unload this library will remove the others and lead to serious memory faults.",
309  mLoadedLib, lastlib.Data()
310  ));
311  }
312  */
313  dlclose(mHandle);
314  }
315  }
316 
317  // Clear the function pointers.
318  mFctCreateReaderFromTCPPort = nullptr;
319  mFctCreateReaderFromTCPPorts = nullptr;
320  mFctCreateReaderFromBuffer = nullptr;
321  mFctDeleteReader = nullptr;
322  mFctCreateWriter = nullptr;
323  mFctDeleteWriter = nullptr;
324 
325  return iResult;
326 }
