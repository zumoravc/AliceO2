31 bool FreePortFinder::testPort(int port) const
32 {
33  struct sockaddr_in addr;
34  memset(&addr, 0, sizeof(addr));
35  addr.sin_family = AF_INET;
36  addr.sin_addr.s_addr = INADDR_ANY;
37  addr.sin_port = htons(port);
38  return bind(mSocket, (struct sockaddr*)&addr, sizeof(addr)) == 0;
39 }

41 void FreePortFinder::scan()
42 {
43  for (mPort = mInitialPort; mPort < mFinalPort; mPort += mStep) {
44  if (!testPort(mPort)) {
45  if (mVerbose) {
46  LOG(WARN) << "Port range [" << mPort << ", " << mPort + mStep
47  << "] already taken. Skipping";
48  }
49  continue;
50  }
51  if (mVerbose) {
52  LOG(INFO) << "Using port range [" << mPort << ", " << mPort + mStep << "]";
53  }
54  break;
55  }
56 }
