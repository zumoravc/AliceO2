72 int TopologyDictionary::readBinaryFile(string fname)
73 {
74  mVectorOfIDs.clear();
75  mCommonMap.clear();
76  for (auto& p : mSmallTopologiesLUT)
77  p = -1;
78  std::ifstream in(fname.data(), std::ios::in | std::ios::binary);
79  GroupStruct gr;
80  int groupID = 0;
81  if (!in.is_open()) {
82  LOG(ERROR) << "The file " << fname << " coud not be opened";
83  throw std::runtime_error("The file coud not be opened");
84  } else {
85  while (in.read(reinterpret_cast<char*>(&gr.mHash), sizeof(unsigned long))) {
86  in.read(reinterpret_cast<char*>(&gr.mErrX), sizeof(float));
87  in.read(reinterpret_cast<char*>(&gr.mErrZ), sizeof(float));
88  in.read(reinterpret_cast<char*>(&gr.mErr2X), sizeof(float));
89  in.read(reinterpret_cast<char*>(&gr.mErr2Z), sizeof(float));
90  in.read(reinterpret_cast<char*>(&gr.mXCOG), sizeof(float));
91  in.read(reinterpret_cast<char*>(&gr.mZCOG), sizeof(float));
92  in.read(reinterpret_cast<char*>(&gr.mNpixels), sizeof(int));
93  in.read(reinterpret_cast<char*>(&gr.mFrequency), sizeof(double));
94  in.read(reinterpret_cast<char*>(&gr.mIsGroup), sizeof(bool));
95  in.read(reinterpret_cast<char*>(&gr.mPattern.mBitmap), sizeof(unsigned char) * (ClusterPattern::kExtendedPatternBytes));
96  mVectorOfIDs.push_back(gr);
97  if (!gr.mIsGroup) {
98  mCommonMap.insert(std::make_pair(gr.mHash, groupID));
99  if (gr.mPattern.getUsedBytes() == 1)
100  mSmallTopologiesLUT[(gr.mPattern.getColumnSpan() - 1) * 255 + (int)gr.mPattern.mBitmap[2]] = groupID;
101  } else {
102  mGroupMap.insert(std::make_pair((int)(gr.mHash >> 32) & 0x00000000ffffffff, groupID));
103  }
104  groupID++;
105  }
106  }
107  in.close();
108  return 0;
109 }

126 {

129  locCl.SetX(locCl.X() + this->getXCOG(cl.getPatternID()));
