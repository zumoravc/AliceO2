66 void EPNReceiverDevice::DiscardIncompleteTimeframes()
67 {
68  auto it = mTimeframeBuffer.begin();
69 
70  while (it != mTimeframeBuffer.end()) {
71  if (duration_cast<milliseconds>(steady_clock::now() - (it->second).start).count() > mBufferTimeoutInMs) {
72  LOG(WARN) << "Timeframe #" << it->first << " incomplete after " << mBufferTimeoutInMs << " milliseconds, discarding";
73  mDiscardedSet.insert(it->first);
74  mTimeframeBuffer.erase(it++);
75  LOG(WARN) << "Number of discarded timeframes: " << mDiscardedSet.size();
76  } else {
77  // LOG(INFO) << "Timeframe #" << it->first << " within timeout, buffering...";
78  ++it;
79  }
80  }

83 void EPNReceiverDevice::Run()
84 {
85  uint16_t id = 0; // holds the timeframe id of the currently arrived sub-timeframe.
86 
87  FairMQChannel& ackOutChannel = fChannels.at(mAckChannelName).at(0);
88 
89  // Simple multi timeframe index
90  using TimeframeId = int;
91  using FlpId = int;
92  std::multimap<TimeframeId, IndexElement> index;
93  std::multimap<TimeframeId, FlpId> flpIds;
94 
95  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this)) {
96  FairMQParts subtimeframeParts;
97  if (Receive(subtimeframeParts, mInChannelName, 0, 100) <= 0)
98  continue;
99 
100  assert(subtimeframeParts.Size() >= 2);
101 
102  const auto* dh = o2::header::get<header::DataHeader*>(subtimeframeParts.At(0)->GetData());
103  assert(strncmp(dh->dataDescription.str, "SUBTIMEFRAMEMD", 16) == 0);
104  SubframeMetadata* sfm = reinterpret_cast<SubframeMetadata*>(subtimeframeParts.At(1)->GetData());
105  id = o2::data_flow::timeframeIdFromTimestamp(sfm->startTime, sfm->duration);
106  auto flpId = sfm->flpIndex;
107 
108  if (mDiscardedSet.find(id) == mDiscardedSet.end()) {
109  if (mTimeframeBuffer.find(id) == mTimeframeBuffer.end()) {
110  // if this is the first part with this ID, save the receive time.
111  mTimeframeBuffer[id].start = steady_clock::now();
112  }
113  flpIds.insert(std::make_pair(id, flpId));
114  LOG(INFO) << "Timeframe ID " << id << " for startTime " << sfm->startTime << "\n";
115  // If the received ID has not previously been discarded, store
116  // the data part in the buffer For the moment we just concatenate
117  // the subtimeframes and add an index for their description at
118  // the end. Given every second part is a data header we skip
119  // every two parts to populate the index. Moreover we know that
120  // the SubframeMetadata is always in the second part, so we can
121  // extract the flpId from there.
122  for (size_t i = 0; i < subtimeframeParts.Size(); ++i) {
123  if (i % 2 == 0) {
124  const auto* adh = o2::header::get<header::DataHeader*>(subtimeframeParts.At(i)->GetData());
125  auto ie = std::make_pair(*adh, index.count(id) * 2);
126  index.insert(std::make_pair(id, ie));
127  }
128  mTimeframeBuffer[id].parts.AddPart(move(subtimeframeParts.At(i)));
129  }
130  } else {
131  // if received ID has been previously discarded.
132  LOG(WARN) << "Received part from an already discarded timeframe with id " << id;
133  }
134 
135  if (flpIds.count(id) == mNumFLPs) {
136  LOG(INFO) << "Timeframe " << id << " complete. Publishing.\n";
137  o2::header::DataHeader tih;
138  std::vector<IndexElement> flattenedIndex;
139 
140  tih.dataDescription = o2::header::DataDescription("TIMEFRAMEINDEX");
141  tih.dataOrigin = o2::header::DataOrigin("EPN");
142  tih.subSpecification = 0;
143  tih.payloadSize = index.count(id) * sizeof(flattenedIndex.front());
144  void* indexData = malloc(tih.payloadSize);
145  auto indexRange = index.equal_range(id);
146  for (auto ie = indexRange.first; ie != indexRange.second; ++ie) {
147  flattenedIndex.push_back(ie->second);
148  }
149  memcpy(indexData, flattenedIndex.data(), tih.payloadSize);
150 
151  mTimeframeBuffer[id].parts.AddPart(NewSimpleMessage(tih));
152  mTimeframeBuffer[id].parts.AddPart(NewMessage(
153  indexData, tih.payloadSize,
154  [](void* data, void* hint) { free(data); }, nullptr));
155  // LOG(INFO) << "Collected all parts for timeframe #" << id;
156  // when all parts are collected send then to the output channel
157  Send(mTimeframeBuffer[id].parts, mOutChannelName);
158  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
159  index.erase(id);
160  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
161  flpIds.erase(id);
162 
163  if (mTestMode > 0) {
164  // Send an acknowledgement back to the sampler to measure the round trip time
165  unique_ptr<FairMQMessage> ack(NewMessage(sizeof(uint16_t)));
166  memcpy(ack->GetData(), &id, sizeof(uint16_t));
167 
168  if (ackOutChannel.Send(ack, 0) <= 0) {
169  LOG(ERROR) << "Could not send acknowledgement without blocking";
170  }
171  }
172 
173  mTimeframeBuffer.erase(id);
174  }
175 
176  // LOG(WARN) << "Buffer size: " << fTimeframeBuffer.size();
177 
178  // Check if any incomplete timeframes in the buffer are older than
179  // timeout period, and discard them if they are
180  // QUESTION: is this really what we want to do?
181  DiscardIncompleteTimeframes();
182  }
183 }
