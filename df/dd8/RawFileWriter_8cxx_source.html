36 void RawFileWriter::close()
37 {
38  // finalize all links
39  if (mFName2File.empty()) {
40  return;
41  }
42  if (!mFirstIRAdded.isDummy()) { // flushing and completing the last HBF makes sense only if data was added.
43  auto irmax = getIRMax();
44  for (auto& lnk : mSSpec2Link) {
45  lnk.second.close(irmax);
46  lnk.second.print();
47  }
48  }
49  //
50  // close all files
51  for (auto& flh : mFName2File) {
52  LOG(INFO) << "Closing output file " << flh.first;
53  fclose(flh.second.handler);
54  flh.second.handler = nullptr;
55  }
56  mFName2File.clear();
57 }

86 RawFileWriter::LinkData& RawFileWriter::registerLink(uint16_t fee, uint16_t cru, uint8_t link, uint8_t endpoint, std::string_view outFileNameV)
87 {
88  // register the GBT link and its output file
89  std::string outFileName{outFileNameV};
90  auto sspec = RDHUtils::getSubSpec(cru, link, endpoint, fee);
91  auto& linkData = mSSpec2Link[sspec];
92  auto& file = mFName2File[std::string(outFileName)];
93  if (!file.handler) {
94  if (!(file.handler = fopen(outFileName.c_str(), "wb"))) { // if file does not exist, create it
95  LOG(ERROR) << "Failed to open output file " << outFileName;
96  throw std::runtime_error(std::string("cannot open link output file ") + outFileName);
97  }
98  }
99  if (!linkData.fileName.empty()) { // this link was already declared and associated with a file
100  if (linkData.fileName == outFileName) {
101  LOGF(INFO, "Link 0x%ux was already declared with same output, do nothing", sspec);
102  return linkData;
103  } else {
104  LOGF(ERROR, "Link 0x%ux was already connected to different output file %s", sspec, linkData.fileName);
105  throw std::runtime_error("redifinition of the link output file");
106  }
107  }
108  linkData.fileName = outFileName;
109  linkData.subspec = sspec;
110  RDHUtils::setVersion(linkData.rdhCopy, mUseRDHVersion);
111  RDHUtils::setFEEID(linkData.rdhCopy, fee);
112  RDHUtils::setCRUID(linkData.rdhCopy, cru);
113  RDHUtils::setLinkID(linkData.rdhCopy, link);
114  RDHUtils::setEndPointID(linkData.rdhCopy, endpoint);
115  if (mUseRDHVersion >= 6) {
116  RDHUtils::setSourceID(linkData.rdhCopy, o2::header::DAQID::O2toDAQ(mOrigin));
117  }
118  linkData.writer = this;
119  linkData.updateIR = mHBFUtils.getFirstIR();
120  linkData.buffer.reserve(mSuperPageSize);
121  LOGF(INFO, "Registered %s with output to %s", linkData.describe(), outFileName);
122  return linkData;
123 }

195 void RawFileWriter::LinkData::addData(const IR& ir, const gsl::span<char> data, bool preformatted)
196 {
197  // add payload corresponding to IR
198  LOG(DEBUG) << "Adding " << data.size() << " bytes in IR " << ir << " to " << describe();
199  std::lock_guard<std::mutex> lock(mtx);
200  int dataSize = data.size();
201  if (ir >= updateIR) { // new IR exceeds or equal IR of next HBF to open, insert missed HBFs if needed
202  fillEmptyHBHs(ir, dataSize > 0);
203  }
204  // we are guaranteed to be under the valid RDH + possibly some data
205  if (!dataSize) {
206  return;
207  }
208  if (preformatted) { // in case detectors wants to add new CRU page of predefined size
209  addPreformattedCRUPage(data);
210  return;
211  }
212  const char* ptr = &data[0];
213  // in case particular detector CRU pages need to be self-consistent, when carrying-over
214  // large payload to new CRU page we may need to write optional trailer and header before
215  // and after the new RDH.
216  bool carryOver = false;
217  int splitID = 0;
218  std::vector<char> carryOverHeader;
219  while (dataSize > 0) {
220  if (carryOver) { // check if there is carry-over header to write in the buffer
221  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
222  // for sure after the carryOver we have space on the CRU page, no need to check
223  LOG(DEBUG) << "Adding carryOverHeader " << carryOverHeader.size()
224  << " bytes in IR " << ir << " to " << describe();
225  pushBack(carryOverHeader.data(), carryOverHeader.size());
226  carryOverHeader.clear();
227  carryOver = false;
228  }
229  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
230  int sizeLeftCRUPage = RDHUtils::MAXCRUPage - (int(buffer.size()) - lastRDHoffset);
231  int sizeLeft = sizeLeftCRUPage < sizeLeftSupPage ? sizeLeftCRUPage : sizeLeftSupPage;
232  if (!sizeLeft) { // this page is just over, open a new one
233  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
234  continue;
235  }
236  if (dataSize <= sizeLeft) { // add all remaining data
237  LOG(DEBUG) << "Adding payload " << dataSize << " bytes in IR " << ir << " (carryover=" << carryOver << " ) to " << describe();
238  pushBack(ptr, dataSize);
239  dataSize = 0;
240  } else { // need to carryOver payload, determine 1st wsize bytes to write starting from ptr
241  carryOver = true;
242  int sizeActual = sizeLeft;
243  std::vector<char> carryOverTrailer;
244  if (writer->carryOverFunc) {
245  sizeActual = writer->carryOverFunc(&rdhCopy, data, ptr, sizeLeft, splitID++, carryOverTrailer, carryOverHeader);
246  }
247  LOG(DEBUG) << "Adding carry-over " << splitID - 1 << " fitted payload " << sizeActual << " bytes in IR " << ir << " to " << describe();
248  if (sizeActual < 0 || sizeActual + carryOverTrailer.size() > sizeLeft) {
249  throw std::runtime_error(std::string("wrong carry-over data size provided by carryOverMethod") + std::to_string(sizeActual));
250  }
251  pushBack(ptr, sizeActual); // write payload fitting to this page
252  dataSize -= sizeActual;
253  ptr += sizeActual;
254  LOG(DEBUG) << "Adding carryOverTrailer " << carryOverTrailer.size() << " bytes in IR "
255  << ir << " to " << describe();
256  pushBack(carryOverTrailer.data(), carryOverTrailer.size());
257  }
258  }
259 }

261 //___________________________________________________________________________________
262 void RawFileWriter::LinkData::addPreformattedCRUPage(const gsl::span<char> data)
263 {
264  // add preformatted CRU page w/o any attempt of splitting
265 
266  // we are guaranteed to have a page with RDH open
267  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
268  if (sizeLeftSupPage < data.size()) { // we are not allowed to split this payload
269  flushSuperPage(true); // flush all but the last added RDH
270  }
271  if (data.size() > RDHUtils::MAXCRUPage - sizeof(RDHAny)) {
272  LOG(ERROR) << "Preformatted payload size of " << data.size() << " bytes for " << describe()
273  << " exceeds max. size " << RDHUtils::MAXCRUPage - sizeof(RDHAny);
274  throw std::runtime_error("preformatted payload exceeds max size");
275  }
276  if (int(buffer.size()) - lastRDHoffset > sizeof(RDHAny)) { // we must start from empty page
277  addHBFPage(); // start new CRU page
278  }
279  pushBack(&data[0], data.size());
280 }

282 //___________________________________________________________________________________
283 void RawFileWriter::LinkData::addHBFPage(bool stop)
284 {
286 
287  if (lastRDHoffset < 0) {
288  return; // no page was open
289  }
290  // finalize last RDH
291  auto& lastRDH = *getLastRDH();
292  int psize = buffer.size() - lastRDHoffset; // set the size for the previous header RDH
293  if (stop && psize == sizeof(RDHAny) && writer->emptyHBFFunc) { // we are closing an empty page, does detector want to add something?
294  std::vector<char> emtyHBFFiller; // working space for optional empty HBF filler
295  writer->emptyHBFFunc(&lastRDH, emtyHBFFiller);
296  if (emtyHBFFiller.size()) {
297  LOG(DEBUG) << "Adding empty HBF filler of size " << emtyHBFFiller.size() << " for " << describe();
298  pushBack(emtyHBFFiller.data(), emtyHBFFiller.size());
299  psize += emtyHBFFiller.size();
300  }
301  }
302  RDHUtils::setOffsetToNext(lastRDH, psize);
303  RDHUtils::setMemorySize(lastRDH, psize);
304 
305  rdhCopy = lastRDH;
306  bool add = true;
307  if (stop && !writer->mAddSeparateHBFStopPage) {
308  RDHUtils::setStop(lastRDH, stop);
309  add = false;
310  }
311  if (writer->mVerbosity > 2) {
312  RDHUtils::printRDH(lastRDH);
313  }
314  if (add) { // if we are in stopping HBF and new page is needed, add it
315  // check if the superpage reached the size where it hase to be flushed
316  int left = writer->mSuperPageSize - buffer.size();
317  if (left <= MarginToFlush) {
318  flushSuperPage();
319  }
320  RDHUtils::setPacketCounter(rdhCopy, packetCounter++);
321  RDHUtils::setPageCounter(rdhCopy, pageCnt++);
322  RDHUtils::setStop(rdhCopy, stop);
323  RDHUtils::setOffsetToNext(rdhCopy, sizeof(RDHAny));
324  RDHUtils::setMemorySize(rdhCopy, sizeof(RDHAny));
325  lastRDHoffset = pushBack(rdhCopy); // entry of the new RDH
326  }
327  if (stop) {
328  if (RDHUtils::getTriggerType(rdhCopy) & o2::trigger::TF) {
329  nTFWritten++;
330  }
331  if (writer->mVerbosity > 2 && add) {
332  RDHUtils::printRDH(rdhCopy);
333  }
334  lastRDHoffset = -1; // after closing, the previous RDH is not valid anymore
335  startOfRun = false; // signal that we are definitely not in the beginning of the run
336  }
337  //
338 }

340 //___________________________________________________________________________________
341 void RawFileWriter::LinkData::openHBFPage(const RDHAny& rdhn)
342 {
344  bool forceNewPage = false;
345  if (RDHUtils::getTriggerType(rdhn) & o2::trigger::TF) {
346  if (writer->mVerbosity > 0) {
347  LOGF(INFO, "Starting new TF for link FEEId 0x%04x", RDHUtils::getFEEID(rdhn));
348  }
349  if (writer->mStartTFOnNewSPage && nTFWritten) { // don't flush if 1st TF
350  forceNewPage = true;
351  }
352  }
353  int left = writer->mSuperPageSize - buffer.size();
354  if (forceNewPage || left <= MarginToFlush) {
355  flushSuperPage();
356  }
357  pageCnt = 0;
358  lastRDHoffset = pushBack(rdhn);
359  auto& newrdh = *getLastRDH(); // dress new RDH with correct counters
360  RDHUtils::setPacketCounter(newrdh, packetCounter++);
361  RDHUtils::setPageCounter(newrdh, pageCnt++);
362  RDHUtils::setStop(newrdh, 0);
363  RDHUtils::setMemorySize(newrdh, sizeof(RDHAny));
364  RDHUtils::setOffsetToNext(newrdh, sizeof(RDHAny));
365  if (startOfRun && writer->isReadOutModeSet()) {
366  auto trg = RDHUtils::getTriggerType(newrdh) | (writer->isContinuousReadout() ? o2::trigger::SOC : o2::trigger::SOT);
367  RDHUtils::setTriggerType(newrdh, trg);
368  }
369  rdhCopy = newrdh;
370 }

372 //___________________________________________________________________________________
373 void RawFileWriter::LinkData::flushSuperPage(bool keepLastPage)
374 {
375  // write link superpage data to file (if requested, only up to the last page)
376  size_t pgSize = (lastRDHoffset < 0 || !keepLastPage) ? buffer.size() : lastRDHoffset;
377  if (writer->mVerbosity) {
378  LOGF(INFO, "Flushing super page of %u bytes for %s", pgSize, describe());
379  }
380  writer->mFName2File.find(fileName)->second.write(buffer.data(), pgSize);
381  auto toMove = buffer.size() - pgSize;
382  if (toMove) { // is there something left in the buffer, move it to the beginning of the buffer
383  if (toMove > pgSize) {
384  memcpy(buffer.data(), &buffer[pgSize], toMove);
385  } else {
386  memmove(buffer.data(), &buffer[pgSize], toMove);
387  }
388  buffer.resize(toMove);
389  lastRDHoffset -= pgSize;
390  } else {
391  buffer.clear();
392  lastRDHoffset = -1;
393  }
394 }

396 //___________________________________________________________________________________
397 void RawFileWriter::LinkData::close(const IR& irf)
398 {
399  // finalize link data
400  // close open HBF, write empty HBFs until the end of the TF corresponding to irfin and detach from the stream
401  if (writer->mFName2File.find(fileName) == writer->mFName2File.end()) {
402  return; // already closed
403  }
404  auto irfin = irf;
405  if (irfin < updateIR) {
406  irfin = updateIR;
407  }
408  int tf = writer->mHBFUtils.getTF(irfin);
409  auto finalIR = writer->mHBFUtils.getIRTF(tf + 1) - 1; // last IR of the current TF
410  fillEmptyHBHs(finalIR, false);
411  closeHBFPage(); // close last HBF
412  flushSuperPage();
413 }

415 //___________________________________________________________________________________
416 void RawFileWriter::LinkData::fillEmptyHBHs(const IR& ir, bool dataAdded)
417 {
418  // fill HBFs from last processed one to requested ir
419  std::vector<o2::InteractionRecord> irw;
420  if (!writer->mHBFUtils.fillHBIRvector(irw, updateIR, ir)) {
421  return;
422  }
423  for (const auto& irdummy : irw) {
424  if (writer->mDontFillEmptyHBF && writer->mHBFUtils.getTFandHBinTF(irdummy).second != 0 && (!dataAdded || irdummy < ir)) {
425  // even if requested, we skip empty HBF filling only if
426  // 1) we are not at the new TF start
427  // 2) method was called from addData and the current IR is the one for which it was called
428  continue;
429  }
430  if (writer->mVerbosity > 2) {
431  LOG(INFO) << "Adding HBF " << irdummy << " for " << describe();
432  }
433  closeHBFPage(); // close current HBF: add RDH with stop and update counters
434  RDHUtils::setTriggerType(rdhCopy, 0); // reset to avoid any detector specific flags in the dummy HBFs
435  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, irdummy); // update HBF orbit/bc and trigger flags
436  openHBFPage(rdhCopy); // open new HBF
437  }
438  updateIR = irw.back() + o2::constants::lhc::LHCMaxBunches; // new HBF will be generated at >= this IR
439 }
