58 template <typename T>
59 TFitResultPtr fit(const size_t nBins, const T* arr, const T xMin, const T xMax, TF1& func, std::string_view option = "")
60 {
61  Foption_t fitOption;
62  ROOT::Fit::FitOptionsMake(ROOT::Fit::kHistogram, option.data(), fitOption);
63 
64  ROOT::Fit::DataRange range(xMin, xMax);
65  ROOT::Fit::DataOptions opt;
66  ROOT::Fit::BinData fitdata(opt, range);
67  fitdata.Initialize(nBins, 1);
68 
69  // create an empty TFitResult
70  std::shared_ptr<TFitResult> tfr(new TFitResult());
71  // create the fitter from an empty fit result
72  //std::shared_ptr<ROOT::Fit::Fitter> fitter(new ROOT::Fit::Fitter(std::static_pointer_cast<ROOT::Fit::FitResult>(tfr) ) );
73  ROOT::Fit::Fitter fitter(tfr);
74  //ROOT::Fit::FitConfig & fitConfig = fitter->Config();
75 
76  const double binWidth = double(xMax - xMin) / double(nBins);
77 
78  for (Int_t ibin = 0; ibin < nBins; ibin++) {
79  const double x = double(xMin) + double(ibin + 0.5) * binWidth;
80  const double y = double(arr[ibin]);
81  const double ey = std::sqrt(y);
82  fitdata.Add(x, y, ey);
83  }
84 
85  const int special = func.GetNumber();
86  const int npar = func.GetNpar();
87  bool linear = func.IsLinear();
88  if (special == 299 + npar) {
89  linear = kTRUE; // for polynomial functions
90  }
91  // do not use linear fitter in these case
92  if (fitOption.Bound || fitOption.Like || fitOption.Errors || fitOption.Gradient || fitOption.More || fitOption.User || fitOption.Integral || fitOption.Minuit) {
93  linear = kFALSE;
94  }
95 
96  if (special != 0 && !fitOption.Bound && !linear) {
97  if (special == 100) {
98  ROOT::Fit::InitGaus(fitdata, &func); // gaussian
99  } else if (special == 400) {
100  ROOT::Fit::InitGaus(fitdata, &func); // landau (use the same)
101  } else if (special == 200) {
102  ROOT::Fit::InitExpo(fitdata, &func); // exponential
103  }
104  }
105 
106  if ((linear || fitOption.Gradient)) {
107  fitter.SetFunction(ROOT::Math::WrappedMultiTF1(func));
108  } else {
109  fitter.SetFunction(static_cast<const ROOT::Math::IParamMultiFunction&>(ROOT::Math::WrappedMultiTF1(func)));
110  }
111 
112  // standard least square fit
113  const bool fitok = fitter.Fit(fitdata, fitOption.ExecPolicy);
114  if (!fitok) {
115  LOGP(warning, "bad fit");
116  }
117 
118  return TFitResultPtr(tfr);
119 }

142 template <typename T>
143 Double_t fitGaus(const size_t nBins, const T* arr, const T xMin, const T xMax, std::vector<T>& param)
144 {
145  static TLinearFitter fitter(3, "pol2");
146  static TMatrixD mat(3, 3);
147  static Double_t kTol = mat.GetTol();
148  fitter.StoreData(kFALSE);
149  fitter.ClearPoints();
150  TVectorD par(3);
151  TVectorD sigma(3);
152  TMatrixD A(3, 3);
153  TMatrixD b(3, 1);
154  T rms = TMath::RMS(nBins, arr);
155  T max = TMath::MaxElement(nBins, arr);
156  T binWidth = (xMax - xMin) / T(nBins);
157 
158  Float_t meanCOG = 0;
159  Float_t rms2COG = 0;
160  Float_t sumCOG = 0;
161 
162  Float_t entries = 0;
163  Int_t nfilled = 0;
164 
165  param.resize(4);
166  param[0] = 0.;
167  param[1] = 0.;
168  param[2] = 0.;
169  param[3] = 0.;
170 
171  for (Int_t i = 0; i < nBins; i++) {
172  entries += arr[i];
173  if (arr[i] > 0)
174  nfilled++;
175  }
176 
177  // TODO: Check why this is needed
178  if (max < 4)
179  return -4;
180  if (entries < 12)
181  return -4;
182 
183  if (rms < kTol)
184  return -4;
185 
186  param[3] = entries;
187 
188  Int_t npoints = 0;
189  for (Int_t ibin = 0; ibin < nBins; ibin++) {
190  Float_t entriesI = arr[ibin];
191  if (entriesI > 1) {
192  Double_t xcenter = xMin + (ibin + 0.5) * binWidth;
193  Double_t error = 1. / TMath::Sqrt(entriesI);
194  Double_t val = TMath::Log(Float_t(entriesI));
195  fitter.AddPoint(&xcenter, val, error);
196  if (npoints < 3) {
197  A(npoints, 0) = 1;
198  A(npoints, 1) = xcenter;
199  A(npoints, 2) = xcenter * xcenter;
200  b(npoints, 0) = val;
201  meanCOG += xcenter * entriesI;
202  rms2COG += xcenter * entriesI * xcenter;
203  sumCOG += entriesI;
204  }
205  npoints++;
206  }
207  }
208 
209  Double_t chi2 = 0;
210  if (npoints >= 3) {
211  if (npoints == 3) {
212  //analytic calculation of the parameters for three points
213  A.Invert();
214  TMatrixD res(1, 3);
215  res.Mult(A, b);
216  par[0] = res(0, 0);
217  par[1] = res(0, 1);
218  par[2] = res(0, 2);
219  chi2 = -3.;
220  } else {
221  // use fitter for more than three points
222  fitter.Eval();
223  fitter.GetParameters(par);
224  fitter.GetCovarianceMatrix(mat);
225  chi2 = fitter.GetChisquare() / Double_t(npoints);
226  }
227  if (TMath::Abs(par[1]) < kTol)
228  return -4;
229  if (TMath::Abs(par[2]) < kTol)
230  return -4;
231 
232  param[1] = T(par[1] / (-2. * par[2]));
233  param[2] = T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2])));
234  Double_t lnparam0 = par[0] + par[1] * param[1] + par[2] * param[1] * param[1];
235  if (lnparam0 > 307)
236  return -4;
237  param[0] = TMath::Exp(lnparam0);
238 
239  return chi2;
240  }
241 
242  if (npoints == 2) {
243  //use center of gravity for 2 points
244  meanCOG /= sumCOG;
245  rms2COG /= sumCOG;
246  param[0] = max;
247  param[1] = meanCOG;
248  param[2] = TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG));
249  chi2 = -2.;
250  }
251  if (npoints == 1) {
252  meanCOG /= sumCOG;
253  param[0] = max;
254  param[1] = meanCOG;
255  param[2] = binWidth / TMath::Sqrt(12);
256  chi2 = -1.;
257  }
258  return chi2;
259 }

278 template <typename T>
279 StatisticsData getStatisticsData(const T* arr, const size_t nBins, const double xMin, const double xMax)
280 {
281  double mean = 0;
282  double rms2 = 0;
283  double sum = 0;
284  size_t npoints = 0;
285 
286  double binWidth = (xMax - xMin) / (double)nBins;
287 
288  StatisticsData data;
289  // in case something went wrong the COG is the histogram lower limit
290  data.mCOG = xMin;
291 
292  for (size_t ibin = 0; ibin < nBins; ++ibin) {
293  double entriesI = (double)arr[ibin];
294  double xcenter = xMin + (ibin + 0.5) * binWidth; // +0.5 to shift to bin centre
295  if (entriesI > 0) {
296  mean += xcenter * entriesI;
297  rms2 += xcenter * entriesI * xcenter;
298  sum += entriesI;
299  ++npoints;
300  }
301  }
302  if (sum == 0)
303  return data;
304  mean /= sum;
305 
306  data.mCOG = mean;
307  // exception in case of only one bin is filled
308  // set the standard deviation to bin width over sqrt(12)
309  rms2 /= sum;
310  if (npoints == 1) {
311  data.mStdDev = binWidth / std::sqrt(12.);
312  } else {
313  data.mStdDev = std::sqrt(std::abs(rms2 - mean * mean));
314  }
315 
316  data.mSum = sum;
317 
318  return data;
319 }

326 template <typename T, typename R = double>
327 R median(std::vector<T> v)
328 {
329  if (v.empty()) {
330  return R{};
331  }
332  auto n = v.size() / 2;
333  nth_element(v.begin(), v.begin() + n, v.end());
334  auto med = R{v[n]};
335  if (!(v.size() & 1)) { //If the set size is even
336  auto max_it = max_element(v.begin(), v.begin() + n);
337  med = R{(*max_it + med) / 2.0};
338  }
339  return med;
340 }

376 template <typename T>
377 bool LTMUnbinned(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeep)
378 {
379  int nPoints = data.size();
380  std::vector<float> w(2 * nPoints);
381  int nKeep = nPoints * fracKeep;
382  if (nKeep > nPoints) {
383  nKeep = nPoints;
384  }
385  if (nKeep < 2) {
386  return false;
387  }
388  // sort in increasing order
389  SortData(data, index);
390  // build cumulants
391  double sum1 = 0.0;
392  double sum2 = 0.0;
393  for (int i = 0; i < nPoints; i++) {
394  double x = data[index[i]];
395  sum1 += x;
396  sum2 += x * x;
397  w[i] = sum1;
398  w[i + nPoints] = sum2;
399  }
400  double maxRMS = sum2 + 1e6;
401  params[0] = nKeep;
402  int limI = nPoints - nKeep + 1; // lowest possible bin to accept
403  for (int i = 0; i < limI; i++) {
404  const int limJ = i + nKeep - 1; // highest accepted bin
405  sum1 = static_cast<double>(w[limJ]) - static_cast<double>(i ? w[i - 1] : 0.);
406  sum2 = static_cast<double>(w[nPoints + limJ]) - static_cast<double>(i ? w[nPoints + i - 1] : 0.);
407  const double mean = sum1 / nKeep;
408  const double rms2 = sum2 / nKeep - mean * mean;
409  if (rms2 > maxRMS) {
410  continue;
411  }
412  maxRMS = rms2;
413  params[1] = mean;
414  params[2] = rms2;
415  params[5] = i;
416  params[6] = limJ;
417  }
418  //
419  if (params[2] < 0) {
420  LOG(error) << "Rounding error: RMS = " << params[2] << " < 0";
421  return false;
422  }
423  params[2] = std::sqrt(params[2]);
424  params[3] = params[2] / std::sqrt(params[0]); // error on mean
425  params[4] = params[3] / std::sqrt(2.0); // error on RMS
426  return true;
427 }

455 template <typename T>
456 bool LTMUnbinnedSig(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeepMin, float sigTgt, bool sorted = false)
457 {
458  int nPoints = data.size();
459  std::vector<double> wx(nPoints);
460  std::vector<double> wx2(nPoints);
461 
462  if (!sorted) {
463  // sort in increasing order
464  SortData(data, index);
465  } else {
466  // array is already sorted
467  std::iota(index.begin(), index.end(), 0);
468  }
469  // build cumulants
470  double sum1 = 0.0;
471  double sum2 = 0.0;
472  for (int i = 0; i < nPoints; i++) {
473  double x = data[index[i]];
474  sum1 += x;
475  sum2 += x * x;
476  wx[i] = sum1;
477  wx2[i] = sum2;
478  }
479  int keepMax = nPoints;
480  int keepMin = fracKeepMin * nPoints;
481  if (keepMin > keepMax) {
482  keepMin = keepMax;
483  }
484  float sigTgt2 = sigTgt * sigTgt;
485  //
486  while (true) {
487  double maxRMS = wx2.back() + 1e6;
488  int keepN = (keepMax + keepMin) / 2;
489  if (keepN < 2) {
490  return false;
491  }
492  params[0] = keepN;
493  int limI = nPoints - keepN + 1;
494  for (int i = 0; i < limI; ++i) {
495  const int limJ = i + keepN - 1;
496  sum1 = wx[limJ] - (i ? wx[i - 1] : 0.);
497  sum2 = wx2[limJ] - (i ? wx2[i - 1] : 0.);
498  const double mean = sum1 / keepN;
499  const double rms2 = sum2 / keepN - mean * mean;
500  if (rms2 > maxRMS) {
501  continue;
502  }
503  maxRMS = rms2;
504  params[1] = mean;
505  params[2] = rms2;
506  params[5] = i;
507  params[6] = limJ;
508  }
509  if (maxRMS < sigTgt2) {
510  keepMin = keepN;
511  } else {
512  keepMax = keepN;
513  }
514  if (keepMin >= keepMax - 1) {
515  break;
516  }
517  }
518  params[2] = std::sqrt(params[2]);
519  params[3] = params[2] / std::sqrt(params[0]); // error on mean
520  params[4] = params[3] / std::sqrt(2.0); // error on RMS
521  return true;
522 }
