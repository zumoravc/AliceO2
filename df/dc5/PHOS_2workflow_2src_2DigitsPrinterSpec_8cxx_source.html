28 void DigitsPrinterSpec::run(framework::ProcessingContext& pc)
29 {
30  // Get the PHOS block header and check whether it contains digits
31  LOG(DEBUG) << "[PHOSDigitsPrinter - process] called";
32  auto dataref = pc.inputs().get("digits");
33  auto const* phosheader = o2::framework::DataRefUtils::getHeader<o2::phos::PHOSBlockHeader*>(dataref);
34  if (!phosheader->mHasPayload) {
35  LOG(DEBUG) << "[PHOSDigitsPrinter - process] No more digits" << std::endl;
36  pc.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
37  return;
38  }
39 
40  auto digits = pc.inputs().get<std::vector<o2::phos::Digit>>("digits");
41  std::cout << "[PHOSDigitsPrinter - process] receiveed " << digits.size() << " digits ..." << std::endl;
42  if (digits.size()) {
43  for (const auto& d : digits) {
44  std::cout << "[PHOSDigitsPrinter - process] Channel(" << d.getAbsId() << ") energy: " << d.getAmplitude() << std::endl;
45  }
46  }
47 }
