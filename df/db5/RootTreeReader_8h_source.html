235  void setupInstance(TTree& tree)
236  {
237  // recursing through the tree structure by simply using method of the previous type,
238  // i.e. the base class method.
239  if constexpr (STAGE > 1) {
240  PrevT::setupInstance(tree);
241  }
242 
243  // right now we allow the same key to appear for multiple branches
244  mBranch = tree.GetBranch(mName.c_str());
245  if (mBranch) {
246  std::string sizebranchName = std::string(mName) + "Size";
247  auto sizebranch = tree.GetBranch(sizebranchName.c_str());
248  auto* classinfo = TClass::GetClass(mBranch->GetClassName());
249  if constexpr (not std::is_void<value_type>::value) {
250  // check if the configured type matches the stored type
251  auto* storedclass = TClass::GetClass(typeid(value_type));
252  if (classinfo != storedclass) {
253  throw std::runtime_error(std::string("Configured type ") +
254  (storedclass != nullptr ? storedclass->GetName() : typeid(value_type).name()) +
255  " does not match the stored data type " +
256  (classinfo != nullptr ? classinfo->GetName() : "") +
257  " in branch " + mName);
258  }
259  }
260  if (!sizebranch) {
261  if (classinfo == nullptr) {
262  throw std::runtime_error(std::string("can not find class description for branch ") + mName);
263  }
264  LOG(INFO) << "branch set up: " << mName;
265  } else {
266  if (classinfo == nullptr || classinfo != TClass::GetClass(typeid(BinaryDataStoreType))) {
267  throw std::runtime_error("mismatching class type, expecting std::vector<char> for binary branch");
268  }
269  mSizeBranch = sizebranch;
270  LOG(INFO) << "binary branch set up: " << mName;
271  }
272  mClassInfo = classinfo;
273  } else {
274  throw std::runtime_error(std::string("can not find branch ") + mName);
275  }
276  }

279  void process(ProcessingContext& context, int entry, std::function<o2::header::Stack()>& stackcreator)
280  {
281  // recursing through the tree structure by simply using method of the previous type,
282  // i.e. the base class method.
283  if constexpr (STAGE > 1) {
284  PrevT::process(context, entry, stackcreator);
285  }
286 
287  auto snapshot = [&context, &stackcreator](const KeyType& key, const auto& object) {
288  context.outputs().snapshot(Output{key.origin, key.description, key.subSpec, key.lifetime, std::move(stackcreator())}, object);
289  };
290 
291  char* data = nullptr;
292  mBranch->SetAddress(&data);
293  mBranch->GetEntry(entry);
294  if (mSizeBranch != nullptr) {
295  size_t datasize = 0;
296  mSizeBranch->SetAddress(&datasize);
297  mSizeBranch->GetEntry(entry);
298  auto* buffer = reinterpret_cast<BinaryDataStoreType*>(data);
299  if (buffer->size() == datasize) {
300  LOG(INFO) << "branch " << mName << ": publishing binary chunk of " << datasize << " bytes(s)";
301  snapshot(mKey, std::move(*buffer));
302  } else {
303  LOG(ERROR) << "branch " << mName << ": inconsitent size of binary chunk "
304  << buffer->size() << " vs " << datasize;
305  BinaryDataStoreType empty;
306  snapshot(mKey, empty);
307  }
308  } else {
309  if constexpr (std::is_void<value_type>::value == true) {
310  // the default branch configuration publishes the object ROOT serialized
311  snapshot(mKey, std::move(ROOTSerializedByClass(*data, mClassInfo)));
312  } else {
313  // if type is specified in the branch configuration, the allocator API decides
314  // upon serialization
315  snapshot(mKey, *reinterpret_cast<value_type*>(data));
316  }
317  }
318  auto* delfunc = mClassInfo->GetDelete();
319  if (delfunc) {
320  (*delfunc)(data);
321  }
322  mBranch->DropBaskets("all");
323  }
324 
325  private:
326  key_type mKey;
327  std::string mName;
328  TBranch* mBranch = nullptr;
329  TBranch* mSizeBranch = nullptr;
330  TClass* mClassInfo = nullptr;
331  };

386  bool next()
387  {
388  if ((mReadEntry + 1) >= mNEntries || mNEntries == 0) {
389  if (mPublishingMode == PublishingMode::Single) {
390  // stop here
391  if (mReadEntry < mNEntries) {
392  mReadEntry = mNEntries;
393  }
394  return false;
395  }
396  // start over in loop mode
397  mReadEntry = -1;
398  }
399  if (mMaxEntries > 0 && (mNofPublished + 1) >= mMaxEntries) {
400  if (mReadEntry < mNEntries) {
401  mReadEntry = mNEntries;
402  }
403  return false;
404  }
405  ++mReadEntry;
406  ++mNofPublished;
407  return true;
408  }

472  template <size_t skip, typename BASE, typename U, typename... Args>
473  std::unique_ptr<BranchConfigurationInterface> createBranchConfiguration(ConstructorArgs&& cargs, U def, Args&&... args)
474  {
475  if constexpr (skip > 0) {
476  return createBranchConfiguration<skip - 1, BASE>(std::move(cargs), std::forward<Args>(args)...);
477  } else if constexpr (std::is_same<U, const char*>::value) {
478  addFile(def);
479  } else if constexpr (std::is_same<U, int>::value) {
480  mMaxEntries = def;
481  } else if constexpr (std::is_same<U, PublishingMode>::value) {
482  mPublishingMode = def;
483  } else if constexpr (is_specialization<U, BranchDefinition>::value) {
484  cargs.emplace_back(key_type(def.key), def.name);
485  using type = BranchConfigurationElement<typename U::type, BASE>;
486  return std::move(createBranchConfiguration<0, type>(std::move(cargs), std::forward<Args>(args)...));
487  } else if constexpr (sizeof...(Args) > 0) {
488  const char* arg = getCharArg(std::forward<Args>(args)...);
489  if (arg != nullptr && *arg != 0) {
490  cargs.emplace_back(key_type(def), arg);
491  using type = BranchConfigurationElement<void, BASE>;
492  return std::move(createBranchConfiguration<1, type>(std::move(cargs), std::forward<Args>(args)...));
493  }
494  throw std::runtime_error("expecting valid branch name string after key");
495  } else {
496  static_assert(always_static_assert<U>::value, "argument mismatch, define branches either as argument pairs of key and branchname or using the BranchDefinition helper struct");
497  }
498  return createBranchConfiguration<0, BASE>(std::move(cargs), std::forward<Args>(args)...);
499  }

503  template <size_t skip, typename T>
504  std::unique_ptr<BranchConfigurationInterface> createBranchConfiguration(ConstructorArgs&& cargs)
505  {
506  static_assert(skip == 0);
507  return std::move(std::make_unique<T>(cargs));
508  }
