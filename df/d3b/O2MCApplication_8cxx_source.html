79 void O2MCApplicationBase::ConstructGeometry()
80 {
81  // fill the mapping
82  mModIdToName.clear();
83  for (int i = 0; i < fModules->GetEntries(); ++i) {
84  auto mod = static_cast<FairModule*>(fModules->At(i));
85  if (mod) {
86  mModIdToName[mod->GetModId()] = mod->GetName();
87  }
88  }
89 
90  FairMCApplication::ConstructGeometry();
91 
92  std::ofstream voltomodulefile("MCStepLoggerVolMap.dat");
93  // construct the volume name to module name mapping useful for StepAnalysis
94  auto vollist = gGeoManager->GetListOfVolumes();
95  for (int i = 0; i < vollist->GetEntries(); ++i) {
96  auto vol = static_cast<TGeoVolume*>(vollist->At(i));
97  auto iter = fModVolMap.find(vol->GetNumber());
98  voltomodulefile << vol->GetName() << ":" << mModIdToName[iter->second] << "\n";
99  }

191 // helper function to fetch data from FairRootManager branch and serialize it
192 // returns handle to container
193 template <typename T>
194 const T* attachBranch(std::string const& name, FairMQChannel& channel, FairMQParts& parts)
195 {
196  auto mgr = FairRootManager::Instance();
197  // check if branch is present
198  if (mgr->GetBranchId(name) == -1) {
199  LOG(ERROR) << "Branch " << name << " not found";
200  return nullptr;
201  }
202  auto data = mgr->InitObjectAs<const T*>(name.c_str());
203  if (data) {
204  o2::base::attachTMessage(*data, channel, parts);
205  }
206  return data;
207 }

216 void O2MCApplication::SendData()
217 {
218  FairMQParts simdataparts;
219 
220  // fill these parts ... the receiver has to unpack similary
221  // TODO: actually we could just loop over branches in FairRootManager at this moment?
222  mSubEventInfo->npersistenttracks = static_cast<o2::data::Stack*>(GetStack())->getMCTracks()->size();
223  mSubEventInfo->nprimarytracks = static_cast<o2::data::Stack*>(GetStack())->GetNprimary();
224  attachSubEventInfo(simdataparts, *mSubEventInfo);
225  auto tracks = attachBranch<std::vector<o2::MCTrack>>("MCTrack", *mSimDataChannel, simdataparts);
226  attachBranch<std::vector<o2::TrackReference>>("TrackRefs", *mSimDataChannel, simdataparts);
227  attachBranch<o2::dataformats::MCTruthContainer<o2::TrackReference>>("IndexedTrackRefs", *mSimDataChannel, simdataparts);
228  assert(tracks->size() == mSubEventInfo->npersistenttracks);
229 
230  for (auto det : listActiveDetectors) {
231  if (dynamic_cast<o2::base::Detector*>(det)) {
232  ((o2::base::Detector*)det)->attachHits(*mSimDataChannel, simdataparts);
233  }
234  }
235  LOG(INFO) << "sending message with " << simdataparts.Size() << " parts";
236  mSimDataChannel->Send(simdataparts);
237 }
