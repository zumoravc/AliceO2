43  void init(o2::framework::InitContext& ic) final
44  {
45  mCCDBpath = ic.options().get<std::string>("ccdb-path");
46  auto& mgr = CcdbManager::instance();
47  mgr.setURL(mCCDBpath);
48  mAPI.init(mgr.getURL());
49  }
50 
51  void run(o2::framework::ProcessingContext& pc) final
52  {
53  int nSlots = pc.inputs().getNofParts(0);
54  assert(pc.inputs().getNofParts(1) == nSlots);
55 
56  for (int isl = 0; isl < nSlots; isl++) {
57  const auto wrp = pc.inputs().get<CcdbObjectInfo*>("clbInfo", isl);
58  const auto pld = pc.inputs().get<gsl::span<char>>("clbPayload", isl); // this is actually an image of TMemFile
59 
60  LOG(INFO) << "Storing in ccdb " << wrp->getPath() << "/" << wrp->getFileName() << " of size " << pld.size()
61  << " Valid for " << wrp->getStartValidityTimestamp() << " : " << wrp->getEndValidityTimestamp();
62  mAPI.storeAsBinaryFile(&pld[0], pld.size(), wrp->getFileName(), wrp->getObjectType(), wrp->getPath(),
63  wrp->getMetaData(), wrp->getStartValidityTimestamp(), wrp->getEndValidityTimestamp());
64  }
65  }
66 
67  private:
68  CcdbApi mAPI;
69  std::string mCCDBpath = "http://ccdb-test.cern.ch:8080"; // CCDB path
70 };
