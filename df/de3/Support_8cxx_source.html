58 TGeoVolumeAssembly* Support::create(Int_t half, Int_t disk)
59 {
60  TGeoSubtraction* localSubtraction;
61  TGeoBBox* localBox;
62  TGeoTube* localTube;
63  TGeoUnion* localUnion;
64  TGeoTranslation* localTranslation;
65  TGeoCompositeShape* localCS = nullptr;
66 
67  //Info("Create",Form("Creating Support_H%d_D%d", half,disk),0,0);
68  mHalfDisk = new TGeoVolumeAssembly(Form("Support_H%d_D%d", half, disk));
69  auto* base = new TGeoTubeSeg(Form("Base_H%d_D%d", half, disk), 0, mSupRad[disk], mSupThickness / 2., mPhi0, mPhi1);
70 
71  // Adding fixationBoxes
72  localCS = serialBoolOperation(base, TGeoBBox(), mDiskFixBoxes[disk], TGeoUnion());
73 
74  // Cutting boxes
75  //Info("Create",Form("Cutting Boxes Support_H%d_D%d", half,disk),0,0);
76  // Using template function to remove boxes
77  localCS = serialBoolOperation(localCS, TGeoBBox(), mDiskBoxCuts[disk], TGeoSubtraction());
78 
79  // Adding raisedBoxes
80  //Info("Create",Form("Adding raised boxes Support_H%d_D%d", half,disk),0,0);
81  localCS = serialBoolOperation(localCS, TGeoBBox(), mDiskRaisedBoxes[disk], TGeoUnion());
82 
83  // ================= Holes ==================
84  //TODO: Holes pointing the y axis
85 
86  // ======= Creating big holes =========
87  //Info("Create",Form("Cutting Voids Support_H%d_D%d", half,disk),0,0);
88  for (auto iVoid = 0; iVoid < mNumberOfVoids[disk]; iVoid++) {
89  TGeoArb8* localArb;
90  localArb = new TGeoArb8(Form("sc_void_%d_H%d_D%d", iVoid, half, disk), mSupThickness + 10. * mT_delta);
91  for (auto iVertex = 0; iVertex < 4; iVertex++) {
92  Double_t* vertex = &mVoidVert[disk][iVoid][iVertex][0];
93  double y;
94  y = ((disk == 3) || (disk == 4)) ? mOuterCut[disk] - vertex[1] : vertex[1];
95  localArb->SetVertex(iVertex, vertex[0], y);
96  localArb->SetVertex(iVertex + 4, vertex[0], y); //Vertexes 4..7 = 0..3
97  }
98  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localArb, nullptr, TGeoSubtraction()));
99  }
100 
101  // ==== M2 6mm deep holes)
102  //Info("Create",Form("Cutting M2 6 mm deep holes Support_H%d_D%d", half,disk),0,0);
103  localTube = new TGeoTube(Form("sc_tube1_a_H%d_D%d", half, disk), 0, mRad_M2, mHeight_M2 + 6. * mT_delta);
104  for (auto iHole = 0; iHole < mNumberOfM2Holes[disk]; iHole++) {
105  localTranslation = new TGeoTranslation(-mM2Holes[disk][iHole][0],
106  mOuterCut[disk] - mM2Holes[disk][iHole][1],
107  mRaisedBoxHeight + mSupThickness / 2. - mHeight_M2);
108  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
109 
110  //For the backside
111  localTranslation = new TGeoTranslation(mM2Holes[disk][iHole][0],
112  mOuterCut[disk] - mM2Holes[disk][iHole][1],
113  -(mRaisedBoxHeight + mSupThickness / 2. - mHeight_M2));
114  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
115  }
116 
117  // ==== D2 H7 - 4 mm deep (on raisedBoxes)
118  //Info("Create",Form("Cutting D2 mm holes on raisedboxes Support_H%d_D%d", half,disk),0,0);
119  localTube = new TGeoTube(Form("sc_tube1_a_H%d_D%d", half, disk), 0, mRad_D2_h, mHeight_D2_h + 6. * mT_delta);
120 
121  for (auto iHole = 0; iHole < mNumberOfD2_hHoles[disk]; iHole++) {
122  localTranslation = new TGeoTranslation(-mD2_hHoles[disk][iHole][0],
123  mOuterCut[disk] - mD2_hHoles[disk][iHole][1],
124  mRaisedBoxHeight + mSupThickness / 2. - mHeight_D2_h);
125  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
126 
127  //For the backside
128  localTranslation = new TGeoTranslation(mD2_hHoles[disk][iHole][0],
129  mOuterCut[disk] - mD2_hHoles[disk][iHole][1],
130  -(mRaisedBoxHeight + mSupThickness / 2. - mHeight_D2_h));
131  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
132  }
133 
134  // ==== D6.5 H7 (6.5 mm diameter holes)
135  //Info("Create",Form("Cutting 6.5 holes Support_H%d_D%d", half,disk),0,0);
136  localTube = new TGeoTube(Form("D65tube_H%d_D%d", half, disk), 0, mD65, mSupThickness / 2. + 20. * mT_delta);
137  for (auto iHole = 0; iHole < mTwoHoles; iHole++) {
138  localTranslation = new TGeoTranslation(-mD65Holes[disk][iHole][0],
139  mOuterCut[disk] - mD65Holes[disk][iHole][1],
140  0.);
141  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
142  }
143 
144  // ==== D6 H7 (6 mm diameter holes)
145  //Info("Create",Form("Cutting 6 mm holes Support_H%d_D%d", half,disk),0,0);
146  localTube = new TGeoTube(Form("D6tube_H%d_D%d", half, disk), 0, mD6, mSupThickness / 2. + mT_delta);
147  for (auto iHole = 0; iHole < mTwoHoles; iHole++) {
148  localTranslation = new TGeoTranslation(-mD6Holes[disk][iHole][0],
149  mOuterCut[disk] - mD6Holes[disk][iHole][1],
150  0.);
151  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
152  }
153 
154  // ==== D8 H7 (8 mm diameter holes)
155  //Info("Create",Form("Cutting 8 mm holes Support_H%d_D%d", half,disk),0,0);
156  localTube = new TGeoTube(Form("D8tube_H%d_D%d", half, disk), 0, mD8, mSupThickness / 2. + mRaisedBoxHeight + 20 * mT_delta);
157  for (auto iHole = 0; iHole < mNumberOfD8_Holes[disk]; iHole++) {
158  localTranslation = new TGeoTranslation(-mD8Holes[disk][iHole][0],
159  mOuterCut[disk] - mD8Holes[disk][iHole][1],
160  0.);
161  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
162  }
163 
164  // ==== D3 H7 (3 mm diameter holes)
165  //Info("Create",Form("Cutting 3 mm holes Support_H%d_D%d", half,disk),0,0);
166  localTube = new TGeoTube(Form("D3tube_H%d_D%d", half, disk), 0, mD3, mSupThickness / 2. + mT_delta);
167  for (auto iHole = 0; iHole < mTwoHoles; iHole++) {
168  localTranslation = new TGeoTranslation(-mD3Holes[disk][iHole][0],
169  mOuterCut[disk] - mD3Holes[disk][iHole][1],
170  0.);
171  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
172  }
173 
174  // ==== M3 H7 (?? mm diameter holes)
175  //Info("Create",Form("Cutting M3 H7 holes Support_H%d_D%d", half,disk),0,0);
176  localTube = new TGeoTube(Form("M3tube_H%d_D%d", half, disk), 0, mM3, mSupThickness / 2. + mT_delta);
177  for (auto iHole = 0; iHole < mNumberOfM3Holes[disk]; iHole++) {
178  localTranslation = new TGeoTranslation(-mM3Holes[disk][iHole][0],
179  mOuterCut[disk] - mM3Holes[disk][iHole][1],
180  0.);
181  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
182  }
183 
184  // ==== D4.5 H9
185  //Info("Create",Form("Cutting 4.5 mm holes Support_H%d_D%d", half,disk),0,0);
186  localTube = new TGeoTube(Form("D45tube_H%d_D%d", half, disk), 0, mD45, mSupThickness / 2. + mT_delta);
187  for (auto iHole = 0; iHole < mTwoHoles; iHole++) {
188  localTranslation = new TGeoTranslation(-mD45Holes[disk][iHole][0],
189  mOuterCut[disk] - mD45Holes[disk][iHole][1],
190  0.);
191  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
192  }
193 
194  // ==== D2 H7 - 4 mm deep (on lower surface)
195  //Info("Create",Form("Cutting D2 holes Support_H%d_D%d", half,disk),0,0);
196  localTube = new TGeoTube(Form("D2tube_H%d_D%d", half, disk), 0, mD2, .4 / 2. + 6 * mT_delta);
197  for (auto iHole = 0; iHole < mTwoHoles; iHole++) {
198  localTranslation = new TGeoTranslation(-mD2Holes[disk][iHole][0],
199  mOuterCut[disk] - mD2Holes[disk][iHole][1],
200  mSupThickness / 2. - .4 / 2);
201  localCS = new TGeoCompositeShape(nullptr, compositeOperation(localCS, localTube, localTranslation, TGeoSubtraction()));
202  }
203 
204  // ======= Prepare support volume and add to HalfDisk =========
205 
206  auto* support_vol = new TGeoVolume(Form("Support_H%d_D%d", half, disk), localCS, mSupportMedium);
207 
208  auto* rot = new TGeoRotation("rot", 0, 0, 180);
209  mHalfDisk->AddNode(support_vol, 0, rot);
210  return mHalfDisk;
211 }
