129 Bool_t Detector::ProcessHits(FairVolume* v)
130 {
131 
132  // 1. Remember all particles first entered PHOS (active medium)
133  // 2. Collect all energy depositions in Cell by all secondaries from particle first entered PHOS
134 
135  // Check if this is first entered PHOS particle ("SuperParent")
136  TVirtualMCStack* stack = fMC->GetStack();
137  const Int_t partID = stack->GetCurrentTrackNumber();
138  Int_t superParent = -1;
139  Bool_t isNewPartile = false; // Create Hit even if zero energy deposition
140  if (partID != mCurrentTrackID) { // not same track as before, check: same SuperParent or new one?
141  auto itTr = mSuperParents.find(partID);
142  if (itTr == mSuperParents.end()) {
143  // Search parent
144  Int_t parentID = stack->GetCurrentTrack()->GetMother(0);
145  itTr = mSuperParents.find(parentID);
146  if (itTr == mSuperParents.end()) { // Neither track or its parent found: new SuperParent
147  mSuperParents[partID] = partID;
148  superParent = partID;
149  isNewPartile = true;
150  } else { // parent found, this track - not
151  superParent = itTr->second;
152  mSuperParents[partID] = superParent;
153  mCurrentTrackID = partID;
154  }
155  } else {
156  superParent = itTr->second;
157  mCurrentTrackID = partID;
158  }
159  } else {
160  superParent = mCurentSuperParent;
161  }
162 
163  Double_t lostenergy = fMC->Edep();
164  if (lostenergy < DBL_EPSILON && !isNewPartile)
165  return false; // do not create hits with zero energy deposition
166 
167  // if(strcmp(mc->CurrentVolName(),"PXTL")!=0) //Non need to check, alwais there...
168  // return false ; // We are not inside a PBWO crystal
169 
170  Int_t moduleNumber;
171  fMC->CurrentVolOffID(
172  11, moduleNumber); // 11: number of geom. levels between PXTL and PHOS module: get the PHOS module number ;
173  Int_t strip;
174  fMC->CurrentVolOffID(3, strip); // 3: number of geom levels between PXTL and strip: get strip number in PHOS module
175  Int_t cell;
176  fMC->CurrentVolOffID(2, cell); // 2: number of geom levels between PXTL and cell: get sell in strip number.
177  Int_t detID = Geometry::relToAbsId(moduleNumber, strip, cell);
178 
179  if (superParent == mCurentSuperParent && detID == mCurrentCellID && mCurrentHit) {
180  // continue with current hit
181  mCurrentHit->addEnergyLoss(lostenergy);
182  return true;
183  }
184 
185  // try to find existing Hit
186  if (!isNewPartile) {
187  for (Int_t itr = mHits->size() - 1; itr >= 0; itr--) {
188  Hit* h = &(mHits->at(itr));
189  if (h->GetTrackID() != superParent) // switched to another SuperParent, do not search further
190  break;
191  if (h->GetDetectorID() == detID) { // found correct hit
192  h->addEnergyLoss(lostenergy);
193  mCurentSuperParent = superParent;
194  mCurrentTrackID = partID;
195  mCurrentCellID = detID;
196  mCurrentHit = h;
197  return true;
198  }
199  }
200  }
201  // Create new Hit
202  Float_t posX = 0., posY = 0., posZ = 0., momX = 0, momY = 0., momZ = 0., energy = 0.;
203  fMC->TrackPosition(posX, posY, posZ);
204  fMC->TrackMomentum(momX, momY, momZ, energy);
205  Double_t estart = fMC->Etot();
206  Double_t time = fMC->TrackTime() * 1.e+9; // time in ns?? To be consistent with EMCAL
207 
208  mCurrentHit = addHit(superParent, detID, Point3D<float>(posX, posY, posZ), Vector3D<float>(momX, momY, momZ), estart,
209  time, lostenergy);
210  mCurentSuperParent = superParent;
211  mCurrentTrackID = partID;
212  mCurrentCellID = detID;
213 
214  return true;
215 }

441 void Detector::ConstructEMCGeometry()
442 {
443  // Create the PHOS-EMC geometry for GEANT
444  // Author: Dmitri Peressounko August 2001
445  // Adopted for O2 project 2017
446  // The used coordinate system:
447  // 1. in Module: X along longer side, Y out of beam, Z along shorter side (along beam)
448  // 2. In Strip the same: X along longer side, Y out of beam, Z along shorter side (along beam)
449 
450  phos::GeometryParams* geom = phos::GeometryParams::GetInstance();
451 
452  Float_t par[4] = {0};
453  Int_t ipar;
454 
455  // ======= Define the strip ===============
456  for (ipar = 0; ipar < 3; ipar++) {
457  par[ipar] = *(geom->getStripHalfSize() + ipar);
458  }
459  // --- define steel volume (cell of the strip unit)
460  fMC->Gsvolu("PSTR", "BOX ", getMediumID(ID_FE), par, 3); // Made of steel
461 
462  // --- define air cell in the steel strip
463  for (ipar = 0; ipar < 3; ipar++) {
464  par[ipar] = *(geom->getAirCellHalfSize() + ipar);
465  }
466  fMC->Gsvolu("PCEL", "BOX ", getMediumID(ID_AIR), par, 3);
467 
468  // --- define wrapped crystal and put it into steel cell
469  for (ipar = 0; ipar < 3; ipar++) {
470  par[ipar] = *(geom->getWrappedHalfSize() + ipar);
471  }
472  fMC->Gsvolu("PWRA", "BOX ", getMediumID(ID_TYVEK), par, 3);
473  const Float_t* pin = geom->getAPDHalfSize();
474  const Float_t* preamp = geom->getPreampHalfSize();
475  Float_t y = (geom->getAirGapLed() - 2 * pin[1] - 2 * preamp[1]) / 2;
476  fMC->Gspos("PWRA", 1, "PCEL", 0.0, y, 0.0, 0, "ONLY");
477 
478  // --- Define crystal and put it into wrapped crystall ---
479  for (ipar = 0; ipar < 3; ipar++) {
480  par[ipar] = *(geom->getCrystalHalfSize() + ipar);
481  }
482 
483  fMC->Gsvolu("PXTL", "BOX ", getMediumID(ID_PWO), par, 3);
484  // fMC->Gsvolu("PXTL", "BOX ", 1, par, 3) ;
485  fMC->Gspos("PXTL", 1, "PWRA", 0.0, 0.0, 0.0, 0, "ONLY");
486 
487  // --- define APD/PIN preamp and put it into AirCell
488  for (ipar = 0; ipar < 3; ipar++) {
489  par[ipar] = *(geom->getAPDHalfSize() + ipar);
490  }
491  fMC->Gsvolu("PPIN", "BOX ", getMediumID(ID_APD), par, 3);
492  const Float_t* crystal = geom->getCrystalHalfSize();
493  y = crystal[1] + geom->getAirGapLed() / 2 - preamp[1];
494  fMC->Gspos("PPIN", 1, "PCEL", 0.0, y, 0.0, 0, "ONLY");
495  for (ipar = 0; ipar < 3; ipar++) {
496  par[ipar] = *(geom->getPreampHalfSize() + ipar);
497  }
498  fMC->Gsvolu("PREA", "BOX ", getMediumID(ID_PRINTCIRC), par, 3); // Here I assumed preamp as a printed Circuit
499  y = crystal[1] + geom->getAirGapLed() / 2 + pin[1]; // May it should be changed
500  fMC->Gspos("PREA", 1, "PCEL", 0.0, y, 0.0, 0, "ONLY"); // to ceramics?
501 
502  // --- Fill strip with wrapped cristals in cells
503 
504  const Float_t* splate = geom->getSupportPlateHalfSize();
505  y = -splate[1];
506  const Float_t* acel = geom->getAirCellHalfSize();
507 
508  for (Int_t lev = 2, icel = 1; icel <= geom->getNCellsXInStrip() * geom->getNCellsZInStrip(); icel += 2, lev += 2) {
509  Float_t x = (2 * (lev / 2) - 1 - geom->getNCellsXInStrip()) * acel[0];
510  Float_t z = acel[2];
511 
512  fMC->Gspos("PCEL", icel, "PSTR", x, y, +z, 0, "ONLY");
513  fMC->Gspos("PCEL", icel + 1, "PSTR", x, y, -z, 0, "ONLY");
514  }
515 
516  // --- define the support plate, hole in it and position it in strip ----
517  for (ipar = 0; ipar < 3; ipar++) {
518  par[ipar] = *(geom->getSupportPlateHalfSize() + ipar);
519  }
520  fMC->Gsvolu("PSUP", "BOX ", getMediumID(ID_AL), par, 3);
521 
522  for (ipar = 0; ipar < 3; ipar++) {
523  par[ipar] = *(geom->getSupportPlateInHalfSize() + ipar);
