62 class CathodeSegmentation
63 {
64  public:
66  CathodeSegmentation(int detElemId, bool isBendingPlane)
67  : mImpl{mchCathodeSegmentationConstruct(detElemId, isBendingPlane)},
68  mDualSampaIds{},
69  mDetElemId{detElemId},
70  mIsBendingPlane{isBendingPlane}
71  {
72  if (!mImpl) {
73  throw std::runtime_error("Can not create segmentation for DE " + std::to_string(detElemId) +
74  (mIsBendingPlane ? " Bending" : " NonBending"));
75  }
76  std::vector<int> dpid;
77  auto addDualSampaId = [&dpid](int dualSampaId) { dpid.push_back(dualSampaId); };
78  auto callback = [](void* data, int dualSampaId) {
79  auto fn = static_cast<decltype(&addDualSampaId)>(data);
80  (*fn)(dualSampaId);
81  };
82  mchCathodeSegmentationForEachDualSampa(mImpl, callback, &addDualSampaId);
83  mDualSampaIds = dpid;
84  int n{0};
85  for (auto i = 0; i < nofDualSampas(); ++i) {
86  forEachPadInDualSampa(dualSampaId(i), [&n](int /*catPadIndex*/) { ++n; });
87  }
88  mNofPads = n;
89  }

279 inline std::string CathodeSegmentation::padAsString(int catPadIndex) const
280 {
281  if (!isValid(catPadIndex)) {
282  return "invalid pad with uid=" + std::to_string(catPadIndex);
283  }
284  std::stringstream s;
285 
286  s << "FEC " << std::setw(4) << padDualSampaId(catPadIndex)
287  << "CH " << std::setw(2) << padDualSampaChannel(catPadIndex)
288  << "X " << std::setw(7) << std::setprecision(2) << padPositionX(catPadIndex)
289  << "Y " << std::setw(7) << std::setprecision(2) << padPositionY(catPadIndex)
290  << "SX " << std::setw(7) << std::setprecision(2) << padSizeX(catPadIndex)
291  << "SY " << std::setw(7) << std::setprecision(2) << padSizeY(catPadIndex);
292 
293  return s.str();
294 }
