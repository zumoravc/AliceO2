32 void FDDRecPointWriter::run(ProcessingContext& pc)
33 {
34  if (mFinished) {
35  return;
36  }
37  // no MC infor treatment at the moment
38  auto recPoints = pc.inputs().get<const std::vector<o2::fdd::RecPoint>>("recpoints");
39  auto recPointsPtr = &recPoints;
40  LOG(INFO) << "FDDRecPointWriter pulled " << recPoints.size() << " RecPoints";
41 
42  TFile flOut(mOutputFileName.c_str(), "recreate");
43  if (flOut.IsZombie()) {
44  LOG(FATAL) << "Failed to create FDD RecPoints output file " << mOutputFileName;
45  }
46  TTree tree(mOutputTreeName.c_str(), "Tree with FDD RecPoints");
47  tree.Branch(mRPOutputBranchName.c_str(), &recPointsPtr);
48  tree.Fill();
49  tree.Write();
50 
51  mFinished = true;
