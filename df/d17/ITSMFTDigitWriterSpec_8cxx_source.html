41 {

43  using MCCont = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
44 
45  public:
46  void init(framework::InitContext& ic)
47  {
48  // std::string detStr = mID.getName();
49  std::string detStrL = mID.getName();
50  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
51 
52  mFileName = ic.options().get<std::string>((detStrL + "-digit-outfile").c_str());
53  mTreeNameDig = ic.options().get<std::string>("treename");
54 
55  LOG(INFO) << "Will store in " << mFileName << ":";
56  LOG(INFO) << "Tree " << mTreeNameDig << " with " << mID.getName() << " digits";
57 
58  mOutFile = std::make_unique<TFile>(mFileName.c_str(), "RECREATE");
59  if (!mOutFile || mOutFile->IsZombie()) {
60  LOG(ERROR) << "Failed to open " << mFileName << " output file";
61  } else {
62  LOG(INFO) << "Opened " << mFileName << " output file";
63  }
64  mOutTreeDig = std::make_unique<TTree>(mTreeNameDig.c_str(), "Digits tree");
65  }
66 
67  void run(framework::ProcessingContext& pc)
68  {
69  if (mFinished) {
70  return;
71  }
72  std::string detStr = mID.getName();
73  std::string detStrL = mID.getName();
74  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
75 
76  // retrieve the digits from the input
77  auto inDigits = pc.inputs().get<std::vector<o2::itsmft::Digit>>((detStr + "digits").c_str());
78  auto inROFs = pc.inputs().get<std::vector<o2::itsmft::ROFRecord>>((detStr + "digitsROF").c_str());
79 
80  if (mWithMCTruth) {
81  auto inLabels = pc.inputs().get<MCCont*>((detStr + "digitsMCTR").c_str());
82  auto labelsRaw = inLabels.get();
83 
84  auto inMC2ROFs = pc.inputs().get<std::vector<o2::itsmft::MC2ROFRecord>>((detStr + "digitsMC2ROF").c_str());
85  auto mc2rofP = &inMC2ROFs;
86  fillBranch(*mOutTreeDig.get(), (detStr + "DigitMCTruth").c_str(), &labelsRaw);
87  fillBranch(*mOutTreeDig.get(), (detStr + "DigitMC2ROF").c_str(), &mc2rofP);
88  }
89  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
90 
91  auto digitsP = &inDigits;
92  auto rofP = &inROFs;
93  fillBranch(*mOutTreeDig.get(), (detStr + "Digit").c_str(), &digitsP);
94  fillBranch(*mOutTreeDig.get(), (detStr + "DigitROF").c_str(), &rofP);
95 
96  mOutTreeDig->SetEntries(mOutTreeDig->GetEntries() + 1);
97 
98  mOutTreeDig->Write();
99  mOutTreeDig.reset(); // delete the trees before closing the file
100 
101  mOutFile->Close();
102  mFinished = true;
103  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
104  }
105 
106  protected:
107  ITSMFTDPLDigitWriter(bool mctruth = true) : mWithMCTruth(mctruth) {}
108  template <typename T>
109  void fillBranch(TTree& tree, std::string brname, T* ptr)
110  {
111  auto br = tree.GetBranch(brname.c_str());
112  if (br) {
113  br->SetAddress(static_cast<void*>(ptr));
114  } else {
115  // otherwise make it
116  br = tree.Branch(brname.c_str(), ptr);
117  }
118  // fill data and reset
119  br->Fill();
120  br->ResetAddress();
121  }
122 
123  std::string mFileName = "ditigs.root"; // output file name
124  std::string mTreeNameDig = "o2sim"; // tree name for digits
125 
126  bool mFinished = false;
127  o2::detectors::DetID mID;
128  o2::header::DataOrigin mOrigin = o2::header::gDataOriginInvalid;
129  std::vector<o2::itsmft::Digit> mDigits; // input digits
130  std::unique_ptr<TFile> mOutFile;
131  std::unique_ptr<TTree> mOutTreeDig; // output tree with digits
132  bool mWithMCTruth = true;
133 };

171 std::vector<InputSpec> makeInputs(std::string const& detStr, o2::header::DataOrigin detOrig, bool mctruth)
172 {
173  std::vector<InputSpec> inputs;
174  inputs.emplace_back(InputSpec{(detStr + "digits").c_str(), detOrig, "DIGITS", 0, Lifetime::Timeframe});
175  inputs.emplace_back(InputSpec{(detStr + "digitsROF").c_str(), detOrig, "DIGITSROF", 0, Lifetime::Timeframe});
176  if (mctruth) {
177  inputs.emplace_back(InputSpec{(detStr + "digitsMCTR").c_str(), detOrig, "DIGITSMCTR", 0, Lifetime::Timeframe});
178  inputs.emplace_back(InputSpec{(detStr + "digitsMC2ROF").c_str(), detOrig, "DIGITSMC2ROF", 0, Lifetime::Timeframe});
179  }
180  return inputs;
181 }
