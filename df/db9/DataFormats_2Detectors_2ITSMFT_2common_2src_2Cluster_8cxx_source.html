35 Bool_t Cluster::testPixel(UShort_t row, UShort_t col) const
36 {
37  // test if pixel at relative row,col is fired
38  int nbits = row * getPatternColSpan() + col;
39  if (nbits >= kMaxPatternBits)
40  return kFALSE;
41  int bytn = nbits >> 3; // 1/8
42  int bitn = 7 - (nbits % 8);
43  return (mPattern[bytn] & (0x1 << bitn)) != 0;

45 }

96 void Cluster::getDiffPattern(std::vector<std::pair<short, short>>& diffv, short colRef, short rowRef) const
97 {
98  // fill vector with incremental differences starting from colRed / rowRef
99  int nr = getPatternRowSpan();
100  int nc = getPatternColSpan();
101  diffv.clear();
102  for (short ir = 0; ir < nr; ir++) {
103  for (short ic0 = 0; ic0 < nc; ic0++) {
104  short ic = (ir & 0x1) ? nc - ic0 - 1 : ic0; // left-to-right for even rows, right-to-left for odd columns
105  if (testPixel(ir, ic)) {
106  short row = ir + mPatternRowMin, col = ic + mPatternColMin;
107  short drow = row - rowRef, dcol = col - colRef;
108  if (!dcol && !drow) {
109  continue;
110  }
111  diffv.emplace_back(dcol, drow);
112  colRef = col;
113  rowRef = row;
114  }
115  }
116  }
117  // flag end of the pattern
118  diffv.emplace_back(0, 0);
119 }
