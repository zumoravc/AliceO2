45 class PreClusterFinderTask
46 {
47  public:
48  //_________________________________________________________________________________________________
49  void init(framework::InitContext& ic)
50  {
52  LOG(INFO) << "initializing preclusterizer";
53 
54  mPreClusterFinder.init();
55 
56  auto stop = [this]() {
57  LOG(INFO) << "reset precluster finder duration = " << mTimeResetPreClusterFinder.count() << " ms";
58  LOG(INFO) << "load digits duration = " << mTimeLoadDigits.count() << " ms";
59  LOG(INFO) << "precluster finder duration = " << mTimePreClusterFinder.count() << " ms";
60  LOG(INFO) << "store precluster duration = " << mTimeStorePreClusters.count() << " ms";
62  auto tStart = std::chrono::high_resolution_clock::now();
63  this->mPreClusterFinder.deinit();
64  auto tEnd = std::chrono::high_resolution_clock::now();
65  LOG(INFO) << "deinitializing preclusterizer in: "
66  << std::chrono::duration<double, std::milli>(tEnd - tStart).count() << " ms";
67  };
68  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
69  }
70 
71  //_________________________________________________________________________________________________
72  void run(framework::ProcessingContext& pc)
73  {
75 
76  // prepare to receive new data
77  auto tStart = std::chrono::high_resolution_clock::now();
78  mPreClusterFinder.reset();
79  mPreClusters.clear();
80  mUsedDigits.clear();
81  auto tEnd = std::chrono::high_resolution_clock::now();
82  mTimeResetPreClusterFinder += tEnd - tStart;
83 
84  // get the input digits
85  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
86 
87  // load the digits to get the fired pads
88  tStart = std::chrono::high_resolution_clock::now();
89  mPreClusterFinder.loadDigits(digits);
90  tEnd = std::chrono::high_resolution_clock::now();
91  mTimeLoadDigits += tEnd - tStart;
92 
93  // preclusterize
94  tStart = std::chrono::high_resolution_clock::now();
95  int nPreClusters = mPreClusterFinder.run();
96  tEnd = std::chrono::high_resolution_clock::now();
97  mTimePreClusterFinder += tEnd - tStart;
98 
99  // get the preclusters and associated digits
100  tStart = std::chrono::high_resolution_clock::now();
101  mPreClusters.reserve(nPreClusters); // to avoid reallocation if
102  mUsedDigits.reserve(digits.size()); // the capacity is exceeded
103  mPreClusterFinder.getPreClusters(mPreClusters, mUsedDigits);
104  if (mUsedDigits.size() != digits.size()) {
105  throw runtime_error("some digits have been lost during the preclustering");
106  }
107  tEnd = std::chrono::high_resolution_clock::now();
108  mTimeStorePreClusters += tEnd - tStart;
109 
110  // send the output messages
111  pc.outputs().snapshot(Output{"MCH", "PRECLUSTERS", 0, Lifetime::Timeframe}, mPreClusters);
112  pc.outputs().snapshot(Output{"MCH", "PRECLUSTERDIGITS", 0, Lifetime::Timeframe}, mUsedDigits);
113  }
114 
115  private:
116  PreClusterFinder mPreClusterFinder{};
117  std::vector<PreCluster> mPreClusters{};
118  std::vector<Digit> mUsedDigits{};
119 
120  std::chrono::duration<double, std::milli> mTimeResetPreClusterFinder{};
121  std::chrono::duration<double, std::milli> mTimeLoadDigits{};
122  std::chrono::duration<double, std::milli> mTimePreClusterFinder{};
123  std::chrono::duration<double, std::milli> mTimeStorePreClusters{};
124 };
