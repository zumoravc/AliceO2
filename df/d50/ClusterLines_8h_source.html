135 inline float Line::getDistanceFromPoint(const Line& line, const std::array<float, 3>& point)
136 {
137  float DCASquared{0};
138  float cdelta{0};
139  for (int i{0}; i < 3; ++i) {
140  cdelta -= line.cosinesDirector[i] * (line.originPoint[i] - point[i]);
141  }
142  for (int i{0}; i < 3; ++i) {
143  DCASquared += (line.originPoint[i] - point[i] + line.cosinesDirector[i] * cdelta) *
144  (line.originPoint[i] - point[i] + line.cosinesDirector[i] * cdelta);
145  }
146  return gpu::CAMath::Sqrt(DCASquared);
147 }

162 GPUhdi() float Line::getDCA(const Line& firstLine, const Line& secondLine, const float precision)
163 {
164  float normalVector[3];
165  normalVector[0] = firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2] -
166  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1];
167  normalVector[1] = -firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2] +
168  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0];
169  normalVector[2] = firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1] -
170  firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0];
171 
172  float norm{0.f}, distance{0.f};
173  for (int i{0}; i < 3; ++i) {
174  norm += normalVector[i] * normalVector[i];
175  distance += (secondLine.originPoint[i] - firstLine.originPoint[i]) * normalVector[i];
176  }
177  if (norm > precision) {
178  return gpu::CAMath::Abs(distance / gpu::CAMath::Sqrt(norm));
179  } else {
180 #if defined(__CUDACC__) || defined(__HIPCC__)
181  float stdOriginPoint[3];
182  for (int i{0}; i < 3; ++i) {
183  stdOriginPoint[i] = secondLine.originPoint[1];
184  }
185 #else
186  std::array<float, 3> stdOriginPoint = {};
187  std::copy_n(secondLine.originPoint, 3, stdOriginPoint.begin());
188 #endif
189  return getDistanceFromPoint(firstLine, stdOriginPoint);
190  }
191 }

193 GPUhdi() void Line::getDCAComponents(const Line& line, const float point[3], float destArray[6])
194 {
195  float cdelta{0.};
196  for (int i{0}; i < 3; ++i)
197  cdelta -= line.cosinesDirector[i] * (line.originPoint[i] - point[i]);
198 
199  destArray[0] = line.originPoint[0] - point[0] + line.cosinesDirector[0] * cdelta;
200  destArray[3] = line.originPoint[1] - point[1] + line.cosinesDirector[1] * cdelta;
201  destArray[5] = line.originPoint[2] - point[2] + line.cosinesDirector[2] * cdelta;
202  destArray[1] = std::sqrt(destArray[0] * destArray[0] + destArray[3] * destArray[3]);
203  destArray[2] = std::sqrt(destArray[0] * destArray[0] + destArray[5] * destArray[5]);
204  destArray[4] = std::sqrt(destArray[3] * destArray[3] + destArray[5] * destArray[5]);
205 }
