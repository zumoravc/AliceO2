44 void* GPUTPCClusterFinder::SetPointersInput(void* mem)

108 unsigned int GPUTPCClusterFinder::getNSteps(size_t items) const
109 {
110  if (items == 0) {
111  return 0;
112  }
113  unsigned int c = 1;
114  size_t capacity = mScanWorkGroupSize;
115  while (items > capacity) {
116  capacity *= mScanWorkGroupSize;
117  c++;
118  }
119  return c;
120 }
