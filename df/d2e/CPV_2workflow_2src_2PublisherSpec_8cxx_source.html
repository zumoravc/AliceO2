36  auto initFunction = [config, propagateMC](o2::framework::InitContext& ic) {
37  // get the option from the init context
38  auto filename = ic.options().get<std::string>("infile");
39  auto treename = ic.options().get<std::string>("treename");
40  auto dtbrName = ic.options().get<std::string>(config.databranch.option.c_str()); // databranch name
41  auto dttrbrName = ic.options().get<std::string>(config.datatrbranch.option.c_str()); // datatrigrec name
42  auto mcbrName = ic.options().get<std::string>(config.mcbranch.option.c_str()); // mcbranch name
43  auto nofEvents = ic.options().get<int>("nevents");
44  // auto publishingMode = nofEvents == -1 ? o2::framework::RootTreeReader::PublishingMode::Single : o2::framework::RootTreeReader::PublishingMode::Loop;
45  auto publishingMode = o2::framework::RootTreeReader::PublishingMode::Single;
46 
47  auto processAttributes = std::make_shared<ProcessAttributes>();
48  {
49  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
50  processAttributes->finished = false;
51  processAttributes->datatype = config.databranch.defval;
52  auto dto = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.dataoutput);
53  auto dttro = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.datatroutput);
54  auto mco = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.mcoutput);
55  constexpr auto persistency = o2::framework::Lifetime::Timeframe;
56  o2::header::DataHeader::SubSpecificationType subSpec = 0;
57  if (propagateMC) {
58  processAttributes->reader = std::make_shared<o2::framework::RootTreeReader>(treename.c_str(), // tree name
59  filename.c_str(), // input file name
60  nofEvents, // number of entries to publish
61  publishingMode,
62  o2::framework::Output{dto.origin, dto.description, subSpec, persistency},
63  dtbrName.c_str(), // name of data branch
64  o2::framework::Output{dttro.origin, dttro.description, subSpec, persistency},
65  dttrbrName.c_str(), // name of data triggerrecords branch
66  o2::framework::Output{mco.origin, mco.description, subSpec, persistency},
67  mcbrName.c_str() // name of mc label branch
68  );
69  } else {
70  processAttributes->reader = std::make_shared<o2::framework::RootTreeReader>(treename.c_str(), // tree name
71  filename.c_str(), // input file name
72  nofEvents, // number of entries to publish
73  publishingMode,
74  o2::framework::Output{dto.origin, dto.description, subSpec, persistency},
75  dtbrName.c_str(), // name of data branch
76  o2::framework::Output{dttro.origin, dttro.description, subSpec, persistency},
77  dttrbrName.c_str() // name of data tr branch
78  );
79  }
80  }
81 
82  auto processFunction = [processAttributes, propagateMC](o2::framework::ProcessingContext& pc) {
83  if (processAttributes->finished) {
84  return;
85  }
86 
87  auto publish = [&processAttributes, &pc, propagateMC]() {
88  o2::cpv::CPVBlockHeader cpvheader(true);
89  if (processAttributes->reader->next()) {
90  (*processAttributes->reader)(pc, cpvheader);
91  } else {
92  processAttributes->reader.reset();
93  return false;
94  }
95  return true;
96  };
97 
98  bool active(true);
99  if (!publish()) {
100  active = false;
101  // Send dummy header with no payload option
102  o2::cpv::CPVBlockHeader dummyheader(false);
103  pc.outputs().snapshot(o2::framework::OutputRef{"output", 0, {dummyheader}}, 0);
104  pc.outputs().snapshot(o2::framework::OutputRef{"outputTR", 0, {dummyheader}}, 0);
105  if (propagateMC) {
106  pc.outputs().snapshot(o2::framework::OutputRef{"outputMC", 0, {dummyheader}}, 0);
107  }
108  }
109  if ((processAttributes->finished = (active == false)) && processAttributes->terminateOnEod) {
110  pc.services().get<o2::framework::ControlService>().endOfStream();
111  pc.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
112  }
113  };
114 
115  return processFunction;
116  };

118  auto createOutputSpecs = [&config, propagateMC]() {
119  std::vector<o2::framework::OutputSpec> outputSpecs;
120  auto dto = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.dataoutput);
121  auto dttro = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.datatroutput);
122  auto mco = o2::framework::DataSpecUtils::asConcreteDataTypeMatcher(config.mcoutput);
123  outputSpecs.emplace_back(o2::framework::OutputSpec{{"output"}, dto.origin, dto.description, 0, o2::framework::Lifetime::Timeframe});
124  outputSpecs.emplace_back(o2::framework::OutputSpec{{"outputTR"}, dttro.origin, dttro.description, 0, o2::framework::Lifetime::Timeframe});
125  if (propagateMC) {
126  outputSpecs.emplace_back(o2::framework::OutputSpec{{"outputMC"}, mco.origin, mco.description, 0, o2::framework::Lifetime::Timeframe});
127  }
128  return std::move(outputSpecs);
129  };
130 
131  auto& dtb = config.databranch;
132  auto& dttrb = config.datatrbranch;
133  auto& mcb = config.mcbranch;
134  return o2::framework::DataProcessorSpec{
135  config.processName.c_str(),
136  o2::framework::Inputs{}, // no inputs
137  {createOutputSpecs()},
138  o2::framework::AlgorithmSpec(initFunction),
139  o2::framework::Options{
140  {"infile", o2::framework::VariantType::String, "", {"Name of the input file"}},
141  {"treename", o2::framework::VariantType::String, config.defaultTreeName.c_str(), {"Name of input tree"}},
142  {dtb.option.c_str(), o2::framework::VariantType::String, dtb.defval.c_str(), {dtb.help.c_str()}},
143  {dttrb.option.c_str(), o2::framework::VariantType::String, dttrb.defval.c_str(), {dttrb.help.c_str()}},
144  {mcb.option.c_str(), o2::framework::VariantType::String, mcb.defval.c_str(), {mcb.help.c_str()}},
145  {"nevents", o2::framework::VariantType::Int, -1, {"number of events to run"}},
146  {"terminate-on-eod", o2::framework::VariantType::Bool, true, {"terminate on end-of-data"}},
147  }};
148 }
