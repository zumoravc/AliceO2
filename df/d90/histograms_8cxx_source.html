20 // This is a very simple example showing how to create an histogram
21 // FIXME: this should really inherit from AnalysisTask but

23 struct ATask {
24  OutputObj<TH1F> phiH{TH1F("phi", "phi", 100, 0., 2. * M_PI)};
25  OutputObj<TH1F> etaH{TH1F("eta", "eta", 102, -2.01, 2.01)};

36 struct BTask {
37  OutputObj<TH2F> etaphiH{TH2F("etaphi", "etaphi", 100, 0., 2. * M_PI, 102, -2.01, 2.01)};
38  // Create a configurable which can be used inside the process method.
39  Configurable<float> phiCut{"phiCut", 6.29f, "A cut on phi"};
40 
41  void process(aod::Tracks const& tracks)
42  {
43  for (auto& track : tracks) {
44  // FIXME: this is until we have configurables which
45  // can be used in expressions.
46  if (track.phi() < phiCut) {
47  etaphiH->Fill(track.phi(), track.eta());
48  }
49  }
50  }
51 };

53 struct CTask {
54  // needs to be initialized with a label or an obj
55  // when adding an object to OutputObj later, the object name will be
56  // *reset* to OutputObj label - needed for correct placement in the output file
57  OutputObj<TH1F> ptH{TH1F("pt", "pt", 100, -0.01, 10.01)};
58  OutputObj<TH1F> trZ{"trZ", OutputObjHandlingPolicy::QAObject};
59  Configurable<float> pTCut{"pTCut", 0.5f, "Lower pT limit"};
60 
61  void init(InitContext const&)
62  {
63  trZ.setObject(new TH1F("Z", "Z", 100, -10., 10.));
64  // other options:
65  // TH1F* t = new TH1F(); trZ.setObject(t); <- resets content!
66  // TH1F t(); trZ.setObject(t) <- makes a copy
67  // trZ.setObject({"Z","Z",100,-10.,10.}); <- creates new
68  }
69 
70  void process(aod::Tracks const& tracks)
71  {
72  for (auto& track : tracks) {
73  if (track.pt() < pTCut)
74  continue;
75  ptH->Fill(track.pt());
76  trZ->Fill(track.z());
77  }
78  }
79 };
