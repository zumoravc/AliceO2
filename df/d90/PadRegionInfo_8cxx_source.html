37  const float ks = mPadHeight / mPadWidth * tan(1.74532925199432948e-01); // tan(10deg)
38  // initialize number of pads per row

55 const PadPos PadRegionInfo::findPad(const float localX, const float localY, const Side side /*=Side::A*/) const
56 {
57  if (!isInRegion(localX))
58  return PadPos(255, 255);
59 
60  // the pad coordinate system is for pad-side view.
61  // on the A-Side one looks from the back-side, therefore
62  // the localY-sign must be changed
63  const float localYfactor = (side == Side::A) ? -1.f : 1.f;
64  const unsigned int row = std::floor((localX - mRadiusFirstRow) / mPadHeight);
65  if (row >= mNumberOfPadRows)
66  return PadPos(255, 255);
67 
68  const unsigned int npads = getPadsInRowRegion(row);
69  const float padfloat = (npads / 2 * mPadWidth - localYfactor * localY) / mPadWidth;
70  if (padfloat < 0)
71  return PadPos(255, 255);
72  const unsigned int pad = static_cast<unsigned int>(padfloat);
73 
74  if (pad >= npads)
75  return PadPos(255, 255);
76 
77  return PadPos(row, pad);
78 }
