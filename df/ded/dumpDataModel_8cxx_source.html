113  label, style.color, style.background, style.fontcolor, style.headerbgcolor, style.headerfontcolor, label);
114  if (pack_size(typename T::iterator::persistent_columns_t{}) -
115  pack_size(typename T::iterator::external_index_columns_t{})) {
116  dumpColumns(typename T::iterator::persistent_columns_t{}, style.color, style.background);
117  fmt::printf("%s", "HR");
118  }
119  if (pack_size(typename T::iterator::dynamic_columns_t{})) {

123  dumpIndexColumns(typename T::iterator::external_index_columns_t{}, style.indexcolor, style.indexbgcolor);
124  fmt::printf("%s", "</TABLE>\n>]\n");
125  if (index)
126  dumpIndex<T>(typename T::iterator::external_index_columns_t{});
127 }
