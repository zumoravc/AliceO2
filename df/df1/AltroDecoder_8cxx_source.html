58 template <class RawReader>
59 void AltroDecoder<RawReader>::readChannels()
60 {
61  mChannelsInitialized = false;
62  mChannels.clear();
63  int currentpos = 0;
64  auto& buffer = mRawReader.getPayload().getPayloadWords();
65  std::array<uint16_t, 1024> bunchwords;
66  while (currentpos < buffer.size() - mRCUTrailer.getTrailerSize()) {
67  auto currentword = buffer[currentpos++];
68  if (currentword >> 30 != 1) {
69  continue;
70  }
71  // starting a new channel
72  mChannels.emplace_back(currentword & 0xFFF, (currentword >> 16) & 0x3FF);
73  auto& currentchannel = mChannels.back();
74  currentchannel.setBadChannel((currentword >> 29) & 0x1);
75 
77  int numberofsamples = 0,
78  numberofwords = (currentchannel.getPayloadSize() + 2) / 3;
79  for (int iword = 0; iword < numberofwords; iword++) {
80  currentword = buffer[currentpos++];
81  if ((currentword >> 30) != 0) {
82  // AliceO2::InfoLogger::InfoLogger logger;
83  // logger << "Unexpected end of payload in altro channel payload! DDL=" << std::setw(3) << std::setfill(0) << mRawReader.getRawHeader().getLink()
84  // << ", Address=0x" << std::hex << current.getHardwareAddress() << ", word=0x" << currentword << std::dec;
85  currentpos--;
86  continue;
87  }
88  bunchwords[numberofsamples++] = (currentword >> 20) & 0x3FF;
89  bunchwords[numberofsamples++] = (currentword >> 10) & 0x3FF;
90  bunchwords[numberofsamples++] = currentword & 0x3FF;
91  }
92 
93  // decode bunches
94  int currentsample = 0;
95  while (currentsample < currentchannel.getPayloadSize()) {
96  int bunchlength = bunchwords[currentsample] - 2, // remove words for bunchlength and starttime
97  starttime = bunchwords[currentsample + 1];
98  auto& currentbunch = currentchannel.createBunch(bunchlength, starttime);
99  currentbunch.initFromRange(gsl::span<uint16_t>(&bunchwords[currentsample + 2], std::min(bunchlength, numberofsamples - currentsample - 2)));
100  currentsample += bunchlength + 2;
101  }
102  }
103  mChannelsInitialized = true;
104 }
