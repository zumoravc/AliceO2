102 template <typename CHARGESUM>
103 ElinkEncoder<BareFormat, CHARGESUM>::ElinkEncoder(uint8_t elinkId,
104  int phase)
105  : mElinkId(elinkId),
106  mBitSet{},
107  mNofSync{0},
108  mSyncIndex{0},
109  mNofBitSeen{0},
110  mLocalBunchCrossing{0},
111  mPhase{phase}
112 {
113  impl::assertIsInRange("elinkId", elinkId, 0, 39);
114 
115  // the phase is used to "simulate" a possible different timing alignment between elinks.
116 
117  if (phase < 0) {
118  mPhase = static_cast<int>(rand() % 20);
119  }
120 
121  for (int i = 0; i < mPhase; i++) {
122  // filling the phase with random bits
123  append(static_cast<bool>(rand() % 2));
124  }
125 }

127 template <typename CHARGESUM>
128 void ElinkEncoder<BareFormat, CHARGESUM>::addChannelData(uint8_t chId, const std::vector<SampaCluster>& data)
129 {
130  if (data.empty()) {
131  throw std::invalid_argument("cannot add empty data");
132  }
133  assertSync();
134  assertNotMixingClusters<CHARGESUM>(data);
135 
136  auto header = impl::buildSampaHeader(mElinkId, chId, data);
137 
138  append50(header.uint64());
139 
140  for (const auto& s : data) {
141  append(s);
142  }
143 }

200 template <typename CHARGESUM>
201 void ElinkEncoder<BareFormat, CHARGESUM>::assertSync()
202 {
203  bool firstSync = (mNofSync == 0);
204 
205  // if mSyncIndex is not zero it means
206  // we have a pending sync to finish to transmit
207  // (said otherwise we're not aligned to an expected 50bits mark)
208  bool pendingSync = (mSyncIndex != 0);
209 
210  if (firstSync || pendingSync) {
211  for (int i = mSyncIndex; i < 50; i++) {
212  append(sync.get(i));
213  }
214  mSyncIndex = 0;
215  if (firstSync) {
216  mNofSync++;
217  }
218  }
219 }

229 template <typename CHARGESUM>
230 void ElinkEncoder<BareFormat, CHARGESUM>::fillWithSync(int upto)
231 {
232  auto d = upto - len();
233  mSyncIndex = circularAppend(mBitSet, sync, mSyncIndex, d);
234  mNofSync += d / 50;
235  mNofBitSeen += d;
236 }
