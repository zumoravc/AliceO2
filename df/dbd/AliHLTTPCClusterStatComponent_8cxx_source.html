69 int AliHLTTPCClusterStatComponent::ProcessOption(TString option, TString value)
70 {
71  int iResult = 0;
72 
73  if (option.EqualTo("print-clusters")) {
74  fPrintClusters = 1;
75  } else if (option.EqualTo("aggregate")) {

78  fSort = 1;

147 void AliHLTTPCClusterStatComponent::TransformReverse(int slice, int row, float y, float z, float padtime[])
148 {
149  AliTPCcalibDB* calib = AliTPCcalibDB::Instance();
150  AliTPCParam* param = calib->GetParameters();
151 
152  float padWidth = 0;
153  // float padLength = 0;
154  float maxPad = 0;
155  float sign = slice < NSLICES / 2 ? 1 : -1;
156  float zwidth;
157 
158  int sector;
159  int sectorrow;
160  if (row < AliHLTTPCGeometry::GetNRowLow()) {
161  sector = slice;
162  sectorrow = row;
163  maxPad = param->GetNPadsLow(sectorrow);
164  // padLength = param->GetPadPitchLength(sector, sectorrow);
165  padWidth = param->GetPadPitchWidth(sector);
166  } else {
167  sector = slice + NSLICES;
168  sectorrow = row - AliHLTTPCGeometry::GetNRowLow();
169  maxPad = param->GetNPadsUp(sectorrow);
170  // padLength = param->GetPadPitchLength(sector, sectorrow);
171  padWidth = param->GetPadPitchWidth(sector);
172  }
173 
174  padtime[0] = y * sign / padWidth + 0.5 * maxPad;
175 
176  float xyzGlobal[2] = {param->GetPadRowRadii(sector, sectorrow), y};
177  AliHLTTPCGeometry::Local2Global(xyzGlobal, slice);
178 
179  float time = z * sign * 1024.f / 250.f;
180  padtime[1] = (1024.f - time);
181 }

183 void AliHLTTPCClusterStatComponent::TransformForward(int slice, int row, float pad, float time, float xyz[])
184 {
185  AliTPCcalibDB* calib = AliTPCcalibDB::Instance();
186  AliTPCParam* param = calib->GetParameters();
187 
188  float padWidth = 0;
189  // float padLength = 0;
190  float maxPad = 0;
191  float sign = slice < NSLICES / 2 ? 1 : -1;
192  float zwidth;
193 
194  int sector;
195  int sectorrow;
196  if (row < AliHLTTPCGeometry::GetNRowLow()) {
197  sector = slice;
198  sectorrow = row;
199  maxPad = param->GetNPadsLow(sectorrow);
200  // padLength = param->GetPadPitchLength(sector, sectorrow);
201  padWidth = param->GetPadPitchWidth(sector);
202  } else {
203  sector = slice + NSLICES;
204  sectorrow = row - AliHLTTPCGeometry::GetNRowLow();
205  maxPad = param->GetNPadsUp(sectorrow);
206  // padLength = param->GetPadPitchLength(sector, sectorrow);
207  padWidth = param->GetPadPitchWidth(sector);
208  }
209 
210  xyz[0] = param->GetPadRowRadii(sector, sectorrow);
211  xyz[1] = (pad - 0.5 * maxPad) * padWidth * sign;
212 
213  float xyzGlobal[2] = {xyz[0], xyz[1]};
214  AliHLTTPCGeometry::Local2Global(xyzGlobal, slice);
215 
216  xyz[2] = sign * (1024 - time) * 250.f / 1024.f;
217 }

219 static bool AliHLTTPCClusterStat_sorthelper(const AliHLTTPCRawCluster& a, const AliHLTTPCRawCluster& b)
220 {
221  if (a.GetPadRow() < b.GetPadRow()) {
222  return (true);
223  }
224  if (a.GetPadRow() > b.GetPadRow()) {
225  return (false);
226  }
227  if (a.GetPad() < b.GetPad()) {
228  return (true);
229  }
230  if (a.GetPad() > b.GetPad()) {
231  return (false);
232  }
233  if (a.GetTime() < b.GetTime()) {
234  return (true);
235  }
236  if (a.GetTime() > b.GetTime()) {
237  return (false);
238  }
239  return (false);
240 }

244  int iResult = 0;
245 
246  if (!IsDataEvent()) {
247  return iResult;
248  }
249

260  memset(clustersTrackIDArray, 0, NSLICES * NPATCHES * sizeof(void*));

273 
274  for (int ndx = 0; ndx < nBlocks; ndx++) {
275  const AliHLTComponentBlockData* iter = blocks + ndx;
276 
277  if (iter->fDataType == (AliHLTTPCDefinitions::fgkRawClustersDataType | kAliHLTDataOriginTPC)) {
278  int slice = AliHLTTPCDefinitions::GetMinSliceNr(iter->fSpecification);
279  int patch = AliHLTTPCDefinitions::GetMinPatchNr(iter->fSpecification);
280

282  }
283 
284  if (iter->fDataType == AliHLTTPCDefinitions::ClustersXYZDataType()) {
285  int slice = AliHLTTPCDefinitions::GetMinSliceNr(iter->fSpecification);
286  int patch = AliHLTTPCDefinitions::GetMinPatchNr(iter->fSpecification);
287 
288  clustersTransformedArray[slice][patch] = (AliHLTTPCClusterXYZData*)(iter->fPtr);
289  if (clustersTransformedArray[slice][patch]->fCount) {
290  clustersTrackIDArray[slice][patch] = new AliHLTTPCTrackHelperStruct[clustersTransformedArray[slice][patch]->fCount];
291  memset(clustersTrackIDArray[slice][patch], 0, clustersTransformedArray[slice][patch]->fCount * sizeof(AliHLTTPCTrackHelperStruct));
292  for (int i = 0; i < clustersTransformedArray[slice][patch]->fCount; i++) {

307  }
308  }

311  double residualBarrelTrackYabs = 0, residualBarrelTrackZabs = 0, residualExternalTrackYabs = 0, residualExternalTrackZabs = 0, residualBacktransformPadabs = 0, residualBacktransformTimeabs = 0;

316  if (fCompressionStudy) {
317  GPUTPCGMPropagator prop;
318  prop.SetMaxSinPhi(.999);
319  prop.SetMaterialTPC();
320  GPUTPCGMPolynomialField field;
321  int err = GPUTPCGMPolynomialFieldManager::GetPolynomialField(field);
322  if (err != 0) {
323  HLTError("Can not initialize polynomial magnetic field");
324  return -1;
325  }
326  prop.SetPolynomialField(&field);
327  for (unsigned i = 0; i < tracks->fCount; i++) {
328  const AliHLTExternalTrackParam* track = reinterpret_cast<const AliHLTExternalTrackParam*>(pCurrent);
329  if (track->fNPoints == 0) {
330  continue;
331  }
332 
333  AliHLTGlobalBarrelTrack btrack(*track);
334  btrack.CalculateHelixParams(bz);
335 
336  AliExternalTrackParam etrack(btrack);
337 
338  GPUTPCGMTrackParam ftrack;
339  float falpha;
340 
341  int hitsUsed = 0;
342  float averageCharge = 0;
343  float averageQMax = 0;
344  AliHLTTPCTrackHelperStruct* hitIndexCache[1024];
345  for (int ip = 0; ip < track->fNPoints; ip++) {
346  int clusterID = track->fPointIDs[ip];
347  int slice = AliHLTTPCGeometry::CluID2Slice(clusterID);
348  int patch = AliHLTTPCGeometry::CluID2Partition(clusterID);
349  int index = AliHLTTPCGeometry::CluID2Index(clusterID);
350 
351  if (clustersTrackIDArray[slice][patch][index].fID != -1) {
352  HLTDebug("Already assigned hit %d of track %d, skipping", ip, i);
353  continue;
354  }
355 
356  if (index > clustersArray[slice][patch]->fCount) {
357  HLTError("Cluster index out of range");
358  continue;
359  }
360 
361  AliHLTTPCRawCluster& cluster = clustersArray[slice][patch]->fClusters[index];
362  AliHLTTPCClusterXYZ& clusterTransformed = clustersTransformedArray[slice][patch]->fClusters[index];
363 
364  int padrow = AliHLTTPCGeometry::GetFirstRow(patch) + cluster.GetPadRow();
365  float x = AliHLTTPCGeometry::Row2X(padrow);
366  float y = 0.0;
367  float z = 0.0;
368 
369  float xyz[3];

371  {
372  TransformForward(slice, padrow, cluster.GetPad(), cluster.GetTime(), xyz);
373  } else { // Correct cluster coordinates using correct transformation
374  xyz[0] = x;
375  xyz[1] = clusterTransformed.fY;
376  xyz[2] = clusterTransformed.fZ;

380  if (alpha > NSLICES / 2) {
381  alpha -= NSLICES / 2;
382  }
383  if (alpha > NSLICES / 4) {
384  alpha -= NSLICES / 2;
385  }
386  alpha = (alpha + 0.5f) * M_PI / 9.f;
387  btrack.CalculateCrossingPoint(x, alpha /* Better use btrack.GetAlpha() ?? */, y, z);
388 
389  etrack.Propagate(alpha, x, bz);
390 
391  if (ip == 0) {
392  ftrack.Par()[0] = xyz[1];
393  ftrack.Par()[1] = xyz[2];
394  for (int k = 2; k < 5; k++) {
395  ftrack.Par()[k] = etrack.GetParameter()[k];

399

404  } else {

406  prop.PropagateToXAlpha(xyz[0], alpha, inFlyDirection);
407  }

413  residualExternalTrackZabs += fabsf(clusterTransformed.fZ - (float)etrack.GetZ());

415  residualBarrelTrackZ += clusterTransformed.fZ - z;
416  residualExternalTrackY += clusterTransformed.fY - etrack.GetY();
417  residualExternalTrackZ += clusterTransformed.fZ - etrack.GetZ();
418  residualFitTrackY += clusterTransformed.fY - ftrack.GetY();
419  residualFitTrackZ += clusterTransformed.fZ - ftrack.GetZ();
420  residualFitTrackYabs += fabsf(clusterTransformed.fY - ftrack.GetY());
421  residualFitTrackZabs += fabsf(clusterTransformed.fZ - ftrack.GetZ());

425  // clusterTransformed.fY - etrack.GetY(), clusterTransformed.fZ - etrack.GetZ(),
426  // clusterTransformed.fY - ftrack.GetY(), clusterTransformed.fZ - ftrack.GetZ());
427 
428  float padtime[2];
429  TransformReverse(slice, padrow, ftrack.GetY(), ftrack.GetZ(), padtime);
430 
431  // Check forward / backward transformation
432  /*float xyzChk[3];
433  TransformForward(slice, padrow, padtime[0], padtime[1], xyzChk);

437  // HLTImportant("Raw Cluster Residual %d (%d/%d) %d: %f %f (%f %f)", i, ip, track->fNPoints, padrow, cluster.GetPad() - padtime[0], cluster.GetTime() - padtime[1], clusterTransformed.fY - ftrack.GetY(), clusterTransformed.fZ - ftrack.GetZ());

439  break;
440  }
441 
442  if (ip != 0) {
443  clustersTrackIDArray[slice][patch][index].fResidualPad = cluster.GetPad() - padtime[0];
444  clustersTrackIDArray[slice][patch][index].fResidualTime = cluster.GetTime() - padtime[1];
445  clustersTrackIDArray[slice][patch][index].fFirstHit = 0;
446

448  residualTrackRawTime += cluster.GetTime() - padtime[1];

450  residualTrackRawTimeabs += fabsf(cluster.GetTime() - padtime[1]);
451  nClusterTracksRaw++;
452  } else {
453  clustersTrackIDArray[slice][patch][index].fResidualPad = cluster.GetPad();
454  clustersTrackIDArray[slice][patch][index].fResidualTime = cluster.GetTime();
455  clustersTrackIDArray[slice][patch][index].fFirstHit = 1;
456  }
457  clustersTrackIDArray[slice][patch][index].fID = i;
458  clustersTrackIDArray[slice][patch][index].fTrack = track;
459  if (hitsUsed >= 1024) {
460  HLTFatal("hitIndex cache exceeded");
461  }
462  hitIndexCache[hitsUsed] = &clustersTrackIDArray[slice][patch][index];
463  hitsUsed++;
464  averageCharge += cluster.GetCharge();
465  averageQMax += cluster.GetQMax();
466 
467  if (ip != 0) {
468  int rowType = padrow < 64 ? 0 : (padrow < 128 ? 2 : 1);
469  prop.Update(xyz[1], xyz[2], rowType, *mSliceParam, 0, 0, nullptr, false);
470  }
471  }
472  if (hitsUsed) {
473  averageCharge /= hitsUsed;
474  averageQMax /= hitsUsed;
475  }
476  for (int ip = 0; ip < hitsUsed; ip++) {
477  hitIndexCache[ip]->fAverageQMax = averageQMax;
478  hitIndexCache[ip]->fAverageQTot = averageCharge;
479  }
480  pCurrent += sizeof(AliHLTExternalTrackParam) + track->fNPoints * sizeof(UInt_t);
481  }
482  }
483 
484  for (unsigned int is = 0; is < NSLICES; is++) {
485  for (unsigned int ip = 0; ip < NPATCHES; ip++) {
486  AliHLTTPCRawClusterData* clusters = clustersArray[is][ip];
487  AliHLTTPCClusterXYZData* clustersTransformed = clustersTransformedArray[is][ip];
488  int firstRow = AliHLTTPCGeometry::GetFirstRow(ip);
489 
490  if (clusters == nullptr) {
491  HLTDebug("Clusters missing for slice %d patch %d\n", is, ip);
492  continue;
493  }
494  if (fCompressionStudy && (clustersTransformed == nullptr || clusters->fCount != clustersTransformed->fCount)) {
495  HLTError("Cluster cound not equal");
496  continue;
497  }
498 
499  AliHLTTPCRawCluster* sortedClusters;
500  if (fSort) {
501  if (fCompressionStudy) {
502  HLTFatal("Cannot sort when compressionstudy is enabled");
503  }
504  sortedClusters = new AliHLTTPCRawCluster[clusters->fCount];
505  memcpy(sortedClusters, clusters->fClusters, sizeof(AliHLTTPCRawCluster) * clusters->fCount);
506  std::sort(sortedClusters, sortedClusters + clusters->fCount, AliHLTTPCClusterStat_sorthelper);
507  }
508 
509  for (UInt_t iCluster = 0; iCluster < clusters->fCount; iCluster++) {
510  AliHLTTPCRawCluster& cluster = clusters->fClusters[iCluster];
511  AliHLTTPCClusterXYZ& clusterTransformed = clustersTransformed->fClusters[iCluster];
512  static AliHLTTPCTrackHelperStruct tmp;
513  AliHLTTPCTrackHelperStruct& clusterTrack = fCompressionStudy ? clustersTrackIDArray[is][ip][iCluster] : tmp;
514 
515  if (fCompressionStudy) {
516  int row = cluster.GetPadRow() + firstRow;
517 
518  float xyz[3];
519  TransformForward(is, row, cluster.GetPad(), cluster.GetTime(), xyz);
520 
521  /*float xyzOrig[3], xyzLocGlob[3];

525 
526  Double_t xx[] = {(double) sectorrow, cluster.GetPad(), cluster.GetTime()};
527  transform->Transform(xx, &sector, 0, 1);
528

534  xyzLocGlob[k] = yy[k];
535  }
536  }*/
