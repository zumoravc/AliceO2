69 void convertDigitsToZSfinal(std::string_view digitsFile, std::string_view outputPath, bool sectorBySector, uint32_t rdhV, bool stopPage, bool noPadding, bool createParentDir)
70 {
71 
72  // ===| open file and get tree |==============================================
73  std::unique_ptr<TFile> o2simDigits(TFile::Open(digitsFile.data()));
74  if (!o2simDigits || !o2simDigits->IsOpen() || o2simDigits->IsZombie()) {
75  LOGP(error, "Could not open file {}", digitsFile.data());
76  exit(1);
77  }
78  auto treeSim = (TTree*)o2simDigits->Get("o2sim");
79  if (!treeSim) {
80  LOGP(error, "Could not read digits tree from file {}", digitsFile.data());
81  exit(1);
82  }
83 
84  gROOT->cd();
85 
86  // ===| set up output directory |=============================================
87  std::string outDir{outputPath};
88  if (outDir.empty()) {
89  outDir = "./";
90  }
91  if (outDir.back() != '/') {
92  outDir += '/';
93  }
94 
95  if (gSystem->AccessPathName(outDir.data())) {
96  if (createParentDir) {
97  if (gSystem->mkdir(outDir.data(), kTRUE)) {
98  LOGP(error, "could not create output directory {}", outDir.data());
99  exit(1);
100  }
101  } else {
102  LOGP(error, "Requested output directory '{}' does not exists, consider removing '-n'", outDir.data());
103  exit(1);
104  }
105  }
106 
107  // ===| set up raw writer |===================================================
108  o2::raw::RawFileWriter writer{"TPC"}; // to set the RDHv6.sourceID if V6 is used
109  writer.useRDHVersion(rdhV);
110  writer.setAddSeparateHBFStopPage(stopPage);
111  const unsigned int defaultLink = rdh_utils::UserLogicLinkID;
112 
113  for (unsigned int i = 0; i < NSectors; i++) {
114  for (unsigned int j = 0; j < NEndpoints; j++) {
115  const unsigned int cruInSector = j / 2;
116  const unsigned int cruID = i * 10 + cruInSector;
117  const rdh_utils::FEEIDType feeid = rdh_utils::getFEEID(cruID, j & 1, defaultLink);
118  writer.registerLink(feeid, cruID, defaultLink, j & 1, fmt::format("{}cru{}_{}.raw", outDir, cruID, j & 1));
119  }
120  }
121 
122  // ===| set up branch addresses |=============================================
123  std::vector<Digit>* vDigitsPerSectorCollection[Sector::MAXSECTOR] = {nullptr}; // container that keeps Digits per sector
124 
125  treeSim->SetBranchStatus("*", 0);
126  treeSim->SetBranchStatus("TPCDigit_*", 1);
127 
128  ProcessAttributes attr;
129  attr.padding = !noPadding;
130 
131  for (int iSecBySec = 0; iSecBySec < Sector::MAXSECTOR; ++iSecBySec) {
132  treeSim->ResetBranchAddresses();
133  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
134  if (sectorBySector) {
135  iSec = iSecBySec;
136  }
137  vDigitsPerSectorCollection[iSec] = nullptr;
138  treeSim->SetBranchAddress(TString::Format("TPCDigit_%d", iSec), &vDigitsPerSectorCollection[iSec]);
139  if (sectorBySector) {
140  break;
141  }
142  }
143  for (Long64_t ievent = 0; ievent < treeSim->GetEntries(); ++ievent) {
144  DigitArray inputDigits;
145  if (sectorBySector) {
146  treeSim->GetBranch(TString::Format("TPCDigit_%d", iSecBySec))->GetEntry(ievent);
147  } else {
148  treeSim->GetEntry(ievent);
149  }
150 
151  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
152  if (sectorBySector) {
153  iSec = iSecBySec;
154  }
155  inputDigits[iSec] = *vDigitsPerSectorCollection[iSec]; //????
156  if (sectorBySector) {
157  break;
158  }
159  }
160  convert(inputDigits, &attr, writer);
161  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
162  delete vDigitsPerSectorCollection[iSec];
163  vDigitsPerSectorCollection[iSec] = nullptr;
164  }
165  }
166  if (!sectorBySector) {
167  break;
168  }
169  }
170  // for further use we write the configuration file for the output
171  writer.writeConfFile("TPC", "RAWDATA", fmt::format("{}tpcraw.cfg", outDir));
172 }

174 void convert(DigitArray& inputDigits, ProcessAttributes* processAttributes, o2::raw::RawFileWriter& writer)
175 {
176  auto& zsEncoder = processAttributes->zsEncoder;
177  const auto zsThreshold = processAttributes->zsThreshold;
178  const auto zs12bit = processAttributes->zs12bit;
179  GPUParam _GPUParam;
180  _GPUParam.SetDefaults(5.00668);
181  const GPUParam mGPUParam = _GPUParam;
182 
183  o2::InteractionRecord ir = o2::raw::HBFUtils::Instance().getFirstIR();
184  zsEncoder->RunZSEncoder<o2::tpc::Digit>(inputDigits, nullptr, nullptr, &writer, &ir, mGPUParam, zs12bit, false, zsThreshold, processAttributes->padding);
185 }
