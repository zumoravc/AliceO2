85 class ConcreteRawParser
86 {
87  public:
88  using buffer_type = unsigned char;
89  using header_type = HeaderType;
90  using self_type = ConcreteRawParser;
91  static constexpr size_t max_size = MAX_SIZE;
92 
93  ConcreteRawParser() = delete;
94 
97  template <typename T>
98  ConcreteRawParser(T const* buffer, size_t size)
99  : mRawBuffer(reinterpret_cast<buffer_type const*>(buffer)), mSize(size)
100  {
101  static_assert(sizeof(T) == sizeof(buffer_type), "buffer required to be byte-type");
102  if (size < sizeof(header_type)) {
103  std::runtime_error("buffer too small to fit at least the page header");
104  }
105  next();
106  }
107  // FIXME: add constructor for container type buffer
108 
112  ConcreteRawParser(ConcreteRawParser const& other) = default;
113 
115  header_type const& header() const
116  {
117  return *reinterpret_cast<header_type const*>(mPosition);
118  }
119 
121  size_t size() const
122  {
123  if (mPosition == mRawBuffer + mSize) {
124  return 0;
125  }
126  header_type const& h = header();
127  if (h.memorySize >= h.headerSize) {
128  return h.memorySize - h.headerSize;
129  }
130  return max_size - h.headerSize;
131  }
132 
134  buffer_type const* data() const
135  {
136  size_t size = this->size();
137  if (size == 0) {
138  return nullptr;
139  }
140  header_type const& h = header();
141  if (mPosition + size + h.headerSize > mRawBuffer + mSize) {
142  throw std::runtime_error("not enough data at position " + std::to_string(mPosition - mRawBuffer));
143  }
144  return mPosition + h.headerSize;
145  }
146 
148  buffer_type const* raw() const
149  {
150  if (mPosition < mRawBuffer + mSize) {
151  return mPosition;
152  }
153  return nullptr;
154  }
155 
157  size_t offset() const
158  {
159  if (mPosition < mRawBuffer + mSize) {
160  header_type const& h = header();
161  return h.headerSize;
162  }
163  return 0;
164  }

170  template <typename Processor>
171  void parse(Processor&& processor)
172  {
173  reset();
174  //auto deleter = [](buffer_type*) {};
175  do {
176  processor(data(), size());
177  //processor(std::unique_ptr<buffer_type, decltype(deleter)>(data(), deleter), size());
178  } while (next());
179  }
180 
182  bool next()
183  {
184  if (mPosition == nullptr) {
185  mPosition = mRawBuffer;
186  } else {
187  auto offset = header().offsetToNext;
188  if ((mPosition + offset + sizeof(header_type) > mRawBuffer + mSize) || (offset < sizeof(header_type))) {
189  // FIXME: should check if there is unread data left of if we just reached
190  // the end of the buffer by parsing all pages
191  mPosition = mRawBuffer + mSize;
192  return false;
193  }
194  mPosition += offset;
195  }
196  // FIXME: check page header validity: version, size
197  return true;
198  }
199 
201  void reset()
202  {
203  mPosition = mRawBuffer;
204  }
205 
209  void advance(int step)
210  {
211  if (step < 0) {
212  // set beyond the buffer
213  mPosition = mRawBuffer + mSize;
214  } else {
215  while ((step-- > 0) && next()) {
216  };
217  }
218  }
219 
222  template <typename T = self_type>
223  bool operator==(T const& other) const
224  {
225  if constexpr (std::is_same<T, self_type>::value == true) {
226  return mRawBuffer == other.mRawBuffer && mPosition == other.mPosition;
227  } else {
228  throw std::runtime_error(std::string("incompatible types for comparison ") + typeid(T).name() + "/" + typeid(self_type).name());
229  }
230  }
231 
232  void format(std::ostream& os, FormatSpec choice = FormatSpec::Entry, const char* delimiter = "\n") const
233  {
234  RDHFormatter<header_type>::apply(os, header(), choice, delimiter);
235  }
236 
237  private:
238  buffer_type const* mRawBuffer;
239  buffer_type const* mPosition = nullptr;
240  size_t mSize;

265 template <size_t PageSize, typename T>
266 ConcreteParserVariants<PageSize> create(T const* buffer, size_t size)
267 {
268  // we use v5 for checking the matching version
269  if (buffer == nullptr || size < sizeof(header::RAWDataHeaderV5)) {
270  throw std::runtime_error("can not create RawParser: invalid buffer");
271  }
272 
273  V5 const* v5 = reinterpret_cast<V5 const*>(buffer);
274  if (v5->version == 5) {
275  return ConcreteRawParser<V5, PageSize>(buffer, size);
276  } else if (v5->version == 6) {
277  return ConcreteRawParser<V6, PageSize>(buffer, size);
278  } else if (v5->version == 4) {
279  return ConcreteRawParser<V4, PageSize>(buffer, size);
280  //} else if (v5->version == 3) {
281  // return ConcreteRawParser<V3, PageSize>(buffer, size);
282  }
283  throw std::runtime_error("can not create RawParser: invalid version " + std::to_string(v5->version));
284 }

288 template <size_t N, typename T, typename P>
289 void walk_parse(T& instances, P&& processor, size_t index)
290 {
291  if constexpr (N > 0) {
292  if (index == N - 1) {
293  std::get<N - 1>(instances).parse(processor);
294  }
295 
296  walk_parse<N - 1>(instances, processor, index);
297  }

300 template <typename U, typename T, size_t N = std::variant_size_v<T>>
301 U const* get_if(T& instances)
302 {
303  if constexpr (N > 0) {
304  auto* parser = std::get_if<N - 1>(&instances);
305  if (parser) {
306  // we are in the active instance, return header if type matches
307  using parser_type = typename std::variant_alternative<N - 1, T>::type;
308  using header_type = typename parser_type::header_type;
309  if constexpr (std::is_same<U, header_type>::value == true) {
310  return &(parser->header());
311  }
312  } else {
313  // continue walking through instances until active one is found
314  return get_if<U, T, N - 1>(instances);
315  }
316  }
317  return nullptr;
318 }
