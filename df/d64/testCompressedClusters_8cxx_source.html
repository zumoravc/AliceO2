126 BOOST_AUTO_TEST_CASE(test_tpc_compressedclusters)
127 {
128  CompressedClustersROOT clusters;
129  ClustersData data;
130  fillClusters(clusters, data);
131  std::vector<char> buffer;
132  CompressedClustersHelpers::flattenTo(buffer, clusters);
133  CompressedClustersROOT restored;
134  CompressedClustersCounters& x = restored;
135  x = static_cast<CompressedClustersCounters&>(clusters);
136  BOOST_REQUIRE(restored.qTotA == nullptr);
137  CompressedClustersHelpers::restoreFrom(buffer, restored);
138 
139  // check one entry from each category
140  BOOST_CHECK(restored.nAttachedClusters == data.qMaxA.size());
141  BOOST_CHECK(memcmp(restored.qMaxA, data.qMaxA.data(), restored.nAttachedClusters * sizeof(decltype(data.qMaxA)::value_type)) == 0);
142  BOOST_CHECK(restored.nTracks == data.nTrackClusters.size());
143  BOOST_CHECK(memcmp(restored.nTrackClusters, data.nTrackClusters.data(), restored.nTracks * sizeof(decltype(data.nTrackClusters)::value_type)) == 0);
144  BOOST_CHECK(restored.nUnattachedClusters == data.qMaxU.size());
145  BOOST_CHECK(memcmp(restored.qMaxU, data.qMaxU.data(), restored.nUnattachedClusters * sizeof(decltype(data.qMaxU)::value_type)) == 0);
146  BOOST_CHECK(restored.nSliceRows == data.nSliceRowClusters.size());
147  BOOST_CHECK(memcmp(restored.nSliceRowClusters, data.nSliceRowClusters.data(), restored.nSliceRows * sizeof(decltype(data.nSliceRowClusters)::value_type)) == 0);
148 }
