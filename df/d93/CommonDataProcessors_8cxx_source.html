73 DataProcessorSpec CommonDataProcessors::getOutputObjSink(outputObjects const& objmap, outputTasks const& tskmap)
74 {
75  auto writerFunction = [objmap, tskmap](InitContext& ic) -> std::function<void(ProcessingContext&)> {
76  auto& callbacks = ic.services().get<CallbackService>();
77  auto inputObjects = std::make_shared<std::vector<std::pair<InputObjectRoute, InputObject>>>();
78 
79  auto endofdatacb = [inputObjects](EndOfStreamContext& context) {
80  LOG(DEBUG) << "Writing merged objects to file";
81  if (inputObjects->empty()) {
82  LOG(ERROR) << "Output object map is empty!";
83  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
84  return;
85  }
86  std::string currentDirectory = "";
87  std::string currentFile = "";
88  TFile* f[OutputObjHandlingPolicy::numPolicies];
89  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
90  f[i] = nullptr;
91  }
92  for (auto& [route, entry] : *inputObjects) {
93  auto file = ROOTfileNames.find(route.policy);
94  if (file != ROOTfileNames.end()) {
95  auto filename = file->second;
96  if (f[route.policy] == nullptr) {
97  f[route.policy] = TFile::Open(filename.c_str(), "RECREATE");
98  }
99  auto nextDirectory = route.directory;
100  if ((nextDirectory != currentDirectory) || (filename != currentFile)) {
101  if (!f[route.policy]->FindKey(nextDirectory.c_str())) {
102  f[route.policy]->mkdir(nextDirectory.c_str());
103  }
104  currentDirectory = nextDirectory;
105  currentFile = filename;
106  }
107  (f[route.policy]->GetDirectory(currentDirectory.c_str()))->WriteObjectAny(entry.obj, entry.kind, entry.name.c_str());
108  }
109  }
110  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
111  if (f[i] != nullptr) {
112  f[i]->Close();
113  }
114  }
115  LOG(DEBUG) << "All outputs merged in their respective target files";
116  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
117  };
118 
119  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
120  return [inputObjects, objmap, tskmap](ProcessingContext& pc) mutable -> void {
121  auto const& ref = pc.inputs().get("x");
122  if (!ref.header) {
123  LOG(ERROR) << "Header not found";
124  return;
125  }
126  if (!ref.payload) {
127  LOG(ERROR) << "Payload not found";
128  return;
129  }
130  auto datah = o2::header::get<o2::header::DataHeader*>(ref.header);
131  if (!datah) {
132  LOG(ERROR) << "No data header in stack";
133  return;
134  }
135 
136  auto objh = o2::header::get<o2::framework::OutputObjHeader*>(ref.header);
137  if (!objh) {
138  LOG(ERROR) << "No output object header in stack";
139  return;
140  }
141 
142  FairTMessage tm(const_cast<char*>(ref.payload), static_cast<int>(datah->payloadSize));
143  InputObject obj;
144  obj.kind = tm.GetClass();
145  if (obj.kind == nullptr) {
146  LOG(error) << "Cannot read class info from buffer.";
147  return;
148  }
149 
150  auto policy = objh->mPolicy;
151  auto hash = objh->mTaskHash;
152 
153  obj.obj = tm.ReadObjectAny(obj.kind);
154  TNamed* named = static_cast<TNamed*>(obj.obj);
155  obj.name = named->GetName();
156  auto hpos = std::find_if(tskmap.begin(), tskmap.end(), [&](auto&& x) { return x.first == hash; });
157  if (hpos == tskmap.end()) {
158  LOG(ERROR) << "No task found for hash " << hash;
159  return;
160  }
161  auto taskname = hpos->second;
162  auto opos = std::find_if(objmap.begin(), objmap.end(), [&](auto&& x) { return x.first == hash; });
163  if (opos == objmap.end()) {
164  LOG(ERROR) << "No object list found for task " << taskname << " (hash=" << hash << ")";
165  return;
166  }
167  auto objects = opos->second;
168  if (std::find(objects.begin(), objects.end(), obj.name) == objects.end()) {
169  LOG(ERROR) << "No object " << obj.name << " in map for task " << taskname;
170  return;
171  }
172  auto nameHash = compile_time_hash(obj.name.c_str());
173  InputObjectRoute key{obj.name, nameHash, taskname, hash, policy};
174  auto existing = std::find_if(inputObjects->begin(), inputObjects->end(), [&](auto&& x) { return (x.first.uniqueId == nameHash) && (x.first.taskHash == hash); });
175  if (existing == inputObjects->end()) {
176  inputObjects->push_back(std::make_pair(key, obj));
177  return;
178  }
179  auto merger = existing->second.kind->GetMerge();
180  if (!merger) {
181  LOG(error) << "Already one unmergeable object found for " << obj.name;
182  return;
183  }
184 
185  TList coll;
186  coll.Add(static_cast<TObject*>(obj.obj));
187  merger(existing->second.obj, &coll, nullptr);
188  };
189  };
190 
191  DataProcessorSpec spec{
192  "internal-dpl-global-analysis-file-sink",
193  {InputSpec("x", DataSpecUtils::dataDescriptorMatcherFrom(header::DataOrigin{"ATSK"}))},
194  Outputs{},
195  AlgorithmSpec(writerFunction),
196  {}};
197 
198  return spec;
199 }

202 DataProcessorSpec
203  CommonDataProcessors::getGlobalAODSink(std::vector<InputSpec> const& OutputInputs,
204  std::vector<bool> const& isdangling)
205 {
206 
207  auto writerFunction = [OutputInputs, isdangling](InitContext& ic) -> std::function<void(ProcessingContext&)> {
208  LOG(DEBUG) << "======== getGlobalAODSink::Init ==========";
209 
210  auto dod = std::make_shared<DataOutputDirector>();
211 
212  // analyze ic and take actions accordingly
213  // default values
214  std::string fnbase("AnalysisResults");
215  std::string filemode("RECREATE");
216  int ntfmerge = 1;
217 
218  // values from json
219  if (ic.options().isSet("json-file")) {
220  auto fnjson = ic.options().get<std::string>("json-file");
221  if (!fnjson.empty()) {
222  auto [fnb, fmo, ntfm] = dod->readJson(fnjson);
223  if (!fnb.empty()) {
224  fnbase = fnb;
225  }
226  if (!fmo.empty()) {
227  filemode = fmo;
228  }
229  if (ntfm > 0) {
230  ntfmerge = ntfm;
231  }
232  }
233  }
234 
235  // values from command line options, information from json is overwritten
236  if (ic.options().isSet("res-file")) {
237  fnbase = ic.options().get<std::string>("res-file");
238  }
239  if (ic.options().isSet("res-mode")) {
240  filemode = ic.options().get<std::string>("res-mode");
241  }
242  if (ic.options().isSet("ntfmerge")) {
243  auto ntfm = ic.options().get<int>("ntfmerge");
244  if (ntfm > 0) {
245  ntfmerge = ntfm;
246  }
247  }
248  // parse the keepString
249  if (ic.options().isSet("keep")) {
250  dod->reset();
251  auto keepString = ic.options().get<std::string>("keep");
252 
253  std::string d("dangling");
254  if (d.find(keepString) == 0) {
255 
256  // use the dangling outputs
257  std::vector<InputSpec> danglingOutputs;
258  for (auto ii = 0; ii < OutputInputs.size(); ii++) {
259  if (isdangling[ii]) {
260  danglingOutputs.emplace_back(OutputInputs[ii]);
261  }
262  }
263  dod->readSpecs(danglingOutputs);
264 
265  } else {
266 
267  // use the keep string
268  dod->readString(keepString);
269  }
270  }
271  dod->setFilenameBase(fnbase);
272 
273  // find out if any table needs to be saved
274  bool hasOutputsToWrite = false;
275  for (auto& outobj : OutputInputs) {
276  auto ds = dod->getDataOutputDescriptors(outobj);
277  if (ds.size() > 0) {
278  hasOutputsToWrite = true;
279  break;
280  }
281  }
282 
283  // if nothing needs to be saved then return a trivial functor
284  if (!hasOutputsToWrite) {
285  return [](ProcessingContext&) mutable -> void {
286  static bool once = false;
287  if (!once) {
288  LOG(INFO) << "No AODs to be saved.";
289  once = true;
290  }
291  };
292  }
293 
294  // end of data functor is called at the end of the data stream
295  auto endofdatacb = [dod](EndOfStreamContext& context) {
296  dod->closeDataFiles();
297 
298  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
299  };
300 
301  auto& callbacks = ic.services().get<CallbackService>();
302  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
303 
304  // this functor is called once per time frame
305  Int_t ntf = -1;
306  return std::move([ntf, ntfmerge, filemode, dod](ProcessingContext& pc) mutable -> void {
307  LOG(DEBUG) << "======== getGlobalAODSink::processing ==========";
308  LOG(DEBUG) << " processing data set with " << pc.inputs().size() << " entries";
309 
310  // return immediately if pc.inputs() is empty
311  auto ninputs = pc.inputs().size();
312  if (ninputs == 0) {
313  LOG(INFO) << "No inputs available!";
314  return;
315  }
316 
317  // increment the time frame counter ntf
318  ntf++;
319 
320  // loop over the DataRefs which are contained in pc.inputs()
321  for (const auto& ref : pc.inputs()) {
322 
323  // does this need to be saved?
324  auto dh = DataRefUtils::getHeader<header::DataHeader*>(ref);
325  auto ds = dod->getDataOutputDescriptors(*dh);
326 
327  if (ds.size() > 0) {
328 
329  // get the TableConsumer and corresponding arrow table
330  auto s = pc.inputs().get<TableConsumer>(ref.spec->binding);
331  auto table = s->asArrowTable();
332 
333  // loop over all DataOutputDescriptors
334  // a table can be saved in multiple ways
335  // e.g. different selections of columns to different files
336  for (auto d : ds) {
337  TableToTree ta2tr(table,
338  dod->getDataOutputFile(d, ntf, ntfmerge, filemode),
339  d->treename.c_str());
340 
341  if (d->colnames.size() > 0) {
342  for (auto cn : d->colnames) {
343  auto idx = table->schema()->GetFieldIndex(cn);
344  auto col = table->column(idx);
345  auto field = table->schema()->field(idx);
346  if (idx != -1) {
347  ta2tr.addBranch(col, field);
348  }
349  }
350  } else {
351  ta2tr.addAllBranches();
352  }
353  ta2tr.process();
354  }
355  }
356  }
357  });
358  }; // end of writerFunction
359 
360  DataProcessorSpec spec{
361  "internal-dpl-aod-writer",
362  OutputInputs,
363  Outputs{},
364  AlgorithmSpec(writerFunction),
365  {{"json-file", VariantType::String, {"Name of the json configuration file"}},
366  {"res-file", VariantType::String, {"Default name of the output file"}},
367  {"res-mode", VariantType::String, {"Creation mode of the result files: NEW, CREATE, RECREATE, UPDATE"}},
368  {"ntfmerge", VariantType::Int, {"Number of time frames to merge into one file"}},
369  {"keep", VariantType::String, {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION:treename:col1/col2/..:filename"}}}};
370 
371  return spec;
372 }

374 DataProcessorSpec
375  CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
376  std::vector<InputSpec>& unmatched)
377 {
378  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
379  auto filename = ic.options().get<std::string>("outfile");
380  auto keepString = ic.options().get<std::string>("keep");
381 
382  if (filename.empty()) {
383  throw std::runtime_error("output file missing");
384  }
385 
386  bool hasOutputsToWrite = false;
387  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
388  VariableContext context;
389  for (auto& spec : danglingOutputInputs) {
390  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(spec);
391  if (outputMatcher->match(concrete, context)) {
392  hasOutputsToWrite = true;
393  }
394  }
395  if (hasOutputsToWrite == false) {
396  return [](ProcessingContext&) mutable -> void {
397  static bool once = false;
398  if (!once) {
399  LOG(DEBUG) << "No dangling output to be dumped.";
400  once = true;
401  }
402  };
403  }
404  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
405  return [output, matcher = outputMatcher](ProcessingContext& pc) mutable -> void {
406  VariableContext matchingContext;
407  LOG(DEBUG) << "processing data set with " << pc.inputs().size() << " entries";
408  for (const auto& entry : pc.inputs()) {
409  LOG(DEBUG) << " " << *(entry.spec);
410  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
411  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
412  if (matcher->match(*header, matchingContext) == false) {
413  continue;
414  }
415  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
416  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
417  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
418  LOG(DEBUG) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
419  }
420  };
421  };
422 
423  std::vector<InputSpec> validBinaryInputs;
424  auto onlyTimeframe = [](InputSpec const& input) {
425  return input.lifetime == Lifetime::Timeframe;
426  };
427 
428  auto noTimeframe = [](InputSpec const& input) {
429  return input.lifetime != Lifetime::Timeframe;
430  };
431 
432  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
433  std::back_inserter(validBinaryInputs), onlyTimeframe);
434  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
435  std::back_inserter(unmatched), noTimeframe);
436 
437  DataProcessorSpec spec{
438  "internal-dpl-global-binary-file-sink",
439  validBinaryInputs,
440  Outputs{},
441  AlgorithmSpec(writerFunction),
442  {{"outfile", VariantType::String, "dpl-out.bin", {"Name of the output file"}},
443  {"keep", VariantType::String, "", {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile"}}}};
444 
445  return spec;
446 }
