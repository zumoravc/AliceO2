91 int AliHLTGPUDumpComponent::DoInit(int argc, const char** argv)
92 {
93  fSolenoidBz = GetBz();
94  fIsMC = TVirtualMC::GetMC();
95 
96  if (!AliGeomManager::GetGeometry()) {
97  AliGeomManager::LoadGeometry();
98  }
99  if (!AliGeomManager::GetGeometry()) {
100  HLTFatal("Can not initialise geometry");
101  }
102 
103  fCalib = AliTPCcalibDB::Instance();
104  if (!fCalib) {
105  HLTFatal("Calibration not found");
106  }
107  fCalib->SetRun(GetRunNo());
108  fCalib->UpdateRunInformations(GetRunNo());
109 
110  const AliMagF* field = (AliMagF*)TGeoGlobalMagField::Instance()->GetField();
111  fCalib->SetExBField(field);
112 
113  if (!fCalib->GetTransform()) {
114  HLTFatal("No TPC transformation found");
115  }
116 
117  AliGRPObject* pGRP = 0;
118  AliCDBEntry* entry = AliCDBManager::Instance()->Get("GRP/GRP/Data");
119  if (!entry) {
120  HLTFatal("No GRP object found in data base");
121  }
122  pGRP = dynamic_cast<AliGRPObject*>(entry->GetObject());
123 
124  if (!pGRP) {
125  HLTFatal("Unknown format of the GRP object in data base");
126  }
127 
128  AliRunInfo runInfo(pGRP->GetLHCState(), pGRP->GetBeamType(), pGRP->GetBeamEnergy(), pGRP->GetRunType(), pGRP->GetDetectorMask());
129  AliEventInfo evInfo;
130  evInfo.SetEventType(AliRawEventHeaderBase::kPhysicsEvent);
131 
132  entry = AliCDBManager::Instance()->Get("TPC/Calib/RecoParam");
133  if (!entry) {
134  HLTFatal("No TPC reco param entry found in data base");
135  }
136  TObject* recoParamObj = entry->GetObject();
137  if (dynamic_cast<TObjArray*>(recoParamObj)) {
138  TObjArray* copy = (TObjArray*)(static_cast<TObjArray*>(recoParamObj)->Clone());
139  fOfflineRecoParam.AddDetRecoParamArray(1, copy);
140  } else if (dynamic_cast<AliDetectorRecoParam*>(recoParamObj)) {
141  AliDetectorRecoParam* copy = (AliDetectorRecoParam*)static_cast<AliDetectorRecoParam*>(recoParamObj)->Clone();
142  fOfflineRecoParam.AddDetRecoParam(1, copy);
143  } else {
144  HLTFatal("Unknown format of the TPC Reco Param entry in the data base");
145  }
146 
147  fOfflineRecoParam.SetEventSpecie(&runInfo, evInfo, 0);
148  fRecParam = const_cast<AliTPCRecoParam*>(reinterpret_cast<const AliTPCRecoParam*>(fOfflineRecoParam.GetDetRecoParam(1)));
149  if (!fRecParam) {
150  HLTFatal("No TPC Reco Param entry found for the given event specification");
151  }
152  fCalib->GetTransform()->SetCurrentRecoParam(fRecParam);
153 
154  return 0;
155 }

161 int AliHLTGPUDumpComponent::DoEvent(const AliHLTComponentEventData& evtData, const AliHLTComponentBlockData* blocks, AliHLTComponentTriggerData& /*trigData*/, AliHLTUInt8_t* outputPtr, AliHLTUInt32_t& size, vector<AliHLTComponentBlockData>& outputBlocks)
162 {
163  if (GetFirstInputBlock(kAliHLTDataTypeSOR) || GetFirstInputBlock(kAliHLTDataTypeEOR)) {
164  return 0;
165  }
166 
167  if (evtData.fBlockCnt <= 0) {
168  HLTWarning("no blocks in event");
169  return 0;
170  }
171 
172  // Prepare everything for all slices
173  const AliHLTTPCClusterMCData* clusterLabels[NSLICES][NPATCHES] = {nullptr};
174  const AliHLTTPCClusterXYZData* clustersXYZ[NSLICES][NPATCHES] = {nullptr};
175  const AliHLTTPCRawClusterData* clustersRaw[NSLICES][NPATCHES] = {nullptr};
176  bool labelsPresent = false;
177  GPUTRDTrackletWord* TRDtracklets = nullptr;
178  GPUTRDTrackletLabels* TRDtrackletsMC = nullptr;
179  int nTRDTrackletsTotal = 0;
180  int nTRDTrackletsMCTotal = 0;
181 
182  for (unsigned long ndx = 0; ndx < evtData.fBlockCnt; ndx++) {
183  const AliHLTComponentBlockData& pBlock = blocks[ndx];
184  int slice = AliHLTTPCDefinitions::GetMinSliceNr(pBlock);
185  int patch = AliHLTTPCDefinitions::GetMinPatchNr(pBlock);
186  if (pBlock.fDataType == AliHLTTPCDefinitions::RawClustersDataType()) {
187  clustersRaw[slice][patch] = (const AliHLTTPCRawClusterData*)pBlock.fPtr;
188  } else if (pBlock.fDataType == AliHLTTPCDefinitions::ClustersXYZDataType()) {
189  clustersXYZ[slice][patch] = (const AliHLTTPCClusterXYZData*)pBlock.fPtr;
190  } else if (pBlock.fDataType == AliHLTTPCDefinitions::AliHLTDataTypeClusterMCInfo()) {
191  clusterLabels[slice][patch] = (const AliHLTTPCClusterMCData*)pBlock.fPtr;
192  labelsPresent = true;
193  } else if (pBlock.fDataType == AliHLTTRDDefinitions::fgkTRDTrackletDataType) {
194  TRDtracklets = reinterpret_cast<GPUTRDTrackletWord*>(pBlock.fPtr);
195  nTRDTrackletsTotal = pBlock.fSize / sizeof(GPUTRDTrackletWord);
196  } else if (pBlock.fDataType == (AliHLTTRDDefinitions::fgkTRDMCTrackletDataType)) {
197  TRDtrackletsMC = reinterpret_cast<GPUTRDTrackletLabels*>(pBlock.fPtr);
198  nTRDTrackletsMCTotal = pBlock.fSize / sizeof(GPUTRDTrackletLabels);
199  }
200  }
201 
202  std::vector<AliHLTTPCRawCluster> rawClusters[NSLICES];
203  std::vector<GPUTPCClusterData> clusterData[NSLICES];
204 
205  int nClustersTotal = 0;
206  for (int slice = 0; slice < NSLICES; slice++) {
207  int nClustersSliceTotal = 0;
208  clusterData[slice].clear();
209  rawClusters[slice].clear();
210  for (int patch = 0; patch < 6; patch++) {
211  if (clustersXYZ[slice][patch]) {
212  nClustersSliceTotal += clustersXYZ[slice][patch]->fCount;
213  }
214  }
215  GPUTPCClusterData cluster;
216  for (int patch = 0; patch < 6; patch++) {
217  if (clustersXYZ[slice][patch] != nullptr && clustersRaw[slice][patch] != nullptr) {
218  const AliHLTTPCClusterXYZData& clXYZ = *clustersXYZ[slice][patch];
219  const AliHLTTPCRawClusterData& clRaw = *clustersRaw[slice][patch];
220 
221  if (clXYZ.fCount != clRaw.fCount) {
222  HLTError("Number of entries in raw and xyz clusters are not mached %d vs %d", clXYZ.fCount, clRaw.fCount);
223  continue;
224  }
225 
226  const int firstRow = AliHLTTPCGeometry::GetFirstRow(patch);
227  for (int ic = 0; ic < clXYZ.fCount; ic++) {
228  const AliHLTTPCClusterXYZ& c = clXYZ.fClusters[ic];
229  const AliHLTTPCRawCluster& cRaw = clRaw.fClusters[ic];
230  if (fabsf(c.GetZ()) > 300) {
231  continue;
232  }
233  if (c.GetX() < 1.f) {
234  continue; // cluster xyz position was not calculated for whatever reason
235  }
236  cluster.id = AliHLTTPCGeometry::CreateClusterID(slice, patch, ic);
237  cluster.x = c.GetX();
238  cluster.y = c.GetY();
239  cluster.z = c.GetZ();
240  cluster.row = firstRow + cRaw.GetPadRow();
241  cluster.flags = cRaw.GetFlags();
242  if (cRaw.GetSigmaPad2() < kAlmost0 || cRaw.GetSigmaTime2() < kAlmost0) {
243  cluster.flags |= GPUTPCGMMergedTrackHit::flagSingle;
244  }
245  cluster.amp = cRaw.GetCharge();
246 #ifdef GPUCA_FULL_CLUSTERDATA
247  cluster.pad = cRaw.GetPad();
248  cluster.time = cRaw.GetTime();
249  cluster.ampMax = cRaw.GetQMax();
250  cluster.sigmaPad2 = cRaw.GetSigmaPad2();
251  cluster.sigmaTime2 = cRaw.GetSigmaTime2();
252 #endif
253  AliHLTTPCRawCluster tmp = cRaw;
254  tmp.fPadRow += firstRow;
255  clusterData[slice].emplace_back(cluster);
256  rawClusters[slice].emplace_back(tmp);
257  nClustersTotal++;
258  }
259  }
260  }
261  HLTDebug("Read %d->%d hits for slice %d", nClustersSliceTotal, (int)clusterData[slice].size(), slice);
262  }
263 
264  if (nClustersTotal < 100) {
265  return (0);
266  }
267  fChain->ClearIOPointers();
268 
269  for (int i = 0; i < NSLICES; i++) {
270  fChain->mIOPtrs.nClusterData[i] = clusterData[i].size();
271  fChain->mIOPtrs.clusterData[i] = clusterData[i].data();
272  fChain->mIOPtrs.nRawClusters[i] = rawClusters[i].size();
273  fChain->mIOPtrs.rawClusters[i] = rawClusters[i].data();
274  HLTDebug("Slice %d - Clusters %d", i, (int)clusterData[i].size());
275  }
276 
277  std::vector<AliHLTTPCClusterMCLabel> labels;
278  std::vector<GPUTPCMCInfo> mcInfo;
279 
280  if (labelsPresent) {
281  // Write cluster labels
282  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
283  GPUTPCClusterData* pCluster = clusterData[iSlice].data();
284  for (unsigned int iPatch = 0; iPatch < NPATCHES; iPatch++) {
285  if (clusterLabels[iSlice][iPatch] == nullptr || clustersXYZ[iSlice][iPatch] == nullptr || clusterLabels[iSlice][iPatch]->fCount != clustersXYZ[iSlice][iPatch]->fCount) {
286  continue;
287  }
288  const AliHLTTPCClusterXYZData& clXYZ = *clustersXYZ[iSlice][iPatch];
289  for (int ic = 0; ic < clXYZ.fCount; ic++) {
290  if (pCluster->id != AliHLTTPCGeometry::CreateClusterID(iSlice, iPatch, ic)) {
291  continue;
292  }
293  pCluster->id = labels.size();
294  labels.push_back(clusterLabels[iSlice][iPatch]->fLabels[ic]);
295  pCluster++;
296  }
297  }
298  }
299 
300  if (labels.size() != nClustersTotal) {
301  HLTFatal("Error getting cluster MC labels (%d labels, %d clusters)", (int)labels.size(), nClustersTotal);
302  return (-EINVAL);
303  }
304 
305  fChain->mIOPtrs.nMCLabelsTPC = labels.size();
306  fChain->mIOPtrs.mcLabelsTPC = labels.data();
307  HLTDebug("Number of mc labels %d", (int)labels.size());
308 
309  // Write MC tracks
310  bool OK = false;
311  do {
312  AliRunLoader* rl = AliRunLoader::Instance();
313  if (rl == nullptr) {
314  HLTFatal("error: RL");
315  break;
316  }
317 
318  rl->LoadKinematics();
319  rl->LoadTrackRefs();
320 
321  int nTracks = rl->GetHeader()->GetNtrack();
322  mcInfo.resize(nTracks);
323 
324  AliStack* stack = rl->Stack();
325  if (stack == nullptr) {
326  HLTFatal("error: stack");
327  break;
328  }
329  TTree* TR = rl->TreeTR();
330  if (TR == nullptr) {
331  HLTFatal("error: TR");
332  break;
333  }
334  TBranch* branch = TR->GetBranch("TrackReferences");
335  if (branch == nullptr) {
336  HLTFatal("error: branch");
337  break;
338  }
339 
340  int nPrimaries = stack->GetNprimary();
341 
342  std::vector<AliTrackReference*> trackRefs(nTracks, nullptr);
343  TClonesArray* tpcRefs = nullptr;
344  branch->SetAddress(&tpcRefs);
345  int nr = TR->GetEntries();
346  for (int r = 0; r < nr; r++) {
347  TR->GetEvent(r);
348  for (int i = 0; i < tpcRefs->GetEntriesFast(); i++) {
349  AliTrackReference* tpcRef = (AliTrackReference*)tpcRefs->UncheckedAt(i);
350  if (tpcRef->DetectorId() != AliTrackReference::kTPC) {
351  continue;
352  }
353  if (tpcRef->Label() < 0 || tpcRef->Label() >= nTracks) {
354  HLTFatal("Invalid reference %d / %d", tpcRef->Label(), nTracks);
355  continue;
356  }
357  if (trackRefs[tpcRef->Label()] != nullptr) {
358  continue;
359  }
360  trackRefs[tpcRef->Label()] = new AliTrackReference(*tpcRef);
361  }
362  }
363 
364  memset(mcInfo.data(), 0, nTracks * sizeof(mcInfo[0]));
365 
366  for (int i = 0; i < nTracks; i++) {
367  mcInfo[i].pid = -100;
368  TParticle* particle = (TParticle*)stack->Particle(i);
369  if (particle == nullptr) {
370  continue;
371  }
372  if (particle->GetPDG() == nullptr) {
373  continue;
374  }
375 
376  int charge = (int)particle->GetPDG()->Charge();
377  int prim = stack->IsPhysicalPrimary(i);
378  int hasPrimDaughter = particle->GetFirstDaughter() != -1 && particle->GetFirstDaughter() < nPrimaries;
379 
380  mcInfo[i].charge = charge;
381  mcInfo[i].prim = prim;
382  mcInfo[i].primDaughters = hasPrimDaughter;
383  mcInfo[i].genRadius = sqrt(particle->Vx() * particle->Vx() + particle->Vy() * particle->Vy() + particle->Vz() * particle->Vz());
384 
385  Int_t pid = -1;
386  if (TMath::Abs(particle->GetPdgCode()) == kElectron) {
387  pid = 0;
388  }
389  if (TMath::Abs(particle->GetPdgCode()) == kMuonMinus) {
390  pid = 1;
391  }
392  if (TMath::Abs(particle->GetPdgCode()) == kPiPlus) {
393  pid = 2;
394  }
395  if (TMath::Abs(particle->GetPdgCode()) == kKPlus) {
396  pid = 3;
397  }
398  if (TMath::Abs(particle->GetPdgCode()) == kProton) {
399  pid = 4;
400  }
401  mcInfo[i].pid = pid;
402 
403  AliTrackReference* ref = trackRefs[i];
404  if (ref) {
405  mcInfo[i].x = ref->X();
406  mcInfo[i].y = ref->Y();
407  mcInfo[i].z = ref->Z();
408  mcInfo[i].pX = ref->Px();
409  mcInfo[i].pY = ref->Py();
410  mcInfo[i].pZ = ref->Pz();
411  }
412 
413  // if (ref) HLTImportant("Particle %d: Charge %d, Prim %d, PrimDaughter %d, Pt %f %f ref %p\n", i, charge, prim, hasPrimDaughter, ref->Pt(), particle->Pt(), ref);
414  }
415  for (int i = 0; i < nTracks; i++) {
416  delete trackRefs[i];
417  }
418 
419  OK = true;
420  } while (false);
421 
422  if (!OK) {
423  HLTFatal("Error accessing MC data");
424  return (-EINVAL);
425  }
426 
427  fChain->mIOPtrs.nMCInfosTPC = mcInfo.size();
428  fChain->mIOPtrs.mcInfosTPC = mcInfo.data();
429  HLTDebug("Number of MC infos: %d", (int)mcInfo.size());
430  }
431  unsigned int clusterNum = 0;
432  for (unsigned int slice = 0; slice < NSLICES; slice++) {
433  for (int k = 0; k < fChain->mIOPtrs.nClusterData[slice]; k++) {
434  clusterData[slice][k].id = clusterNum++;
435  }
436  }
437 
438  fChain->mIOPtrs.nTRDTracklets = nTRDTrackletsTotal;
439  fChain->mIOPtrs.trdTracklets = TRDtracklets;
440  fChain->mIOPtrs.nTRDTrackletsMC = nTRDTrackletsMCTotal;
441  fChain->mIOPtrs.trdTrackletsMC = TRDtrackletsMC;
442  HLTDebug("Number of TRD tracklets: %d", (int)nTRDTrackletsTotal);
443 
444  static int nEvent = 0;
445  char filename[256];
446  std::ofstream out;
447 
448  if (nEvent == 0) {
449  std::unique_ptr<TPCFastTransform> fFastTransformIRS(new TPCFastTransform);
450  long TimeStamp = GetTimeStamp();
451  if (fIsMC && !fRecParam->GetUseCorrectionMap()) {
452  TimeStamp = 0;
453  }
454  if (fFastTransformManager->create(*fFastTransformIRS, fCalib->GetTransform(), TimeStamp)) {
455  HLTFatal("Initialisation of Fast Transformation failed with error %s", fFastTransformManager->getLastError());
456  }
457  fChain->SetTPCFastTransform(std::move(fFastTransformIRS));
458 
459  fRec->SetSettings(fSolenoidBz);
460  fRec->DumpSettings();
461  }
462 
463  sprintf(filename, GPUCA_EVDUMP_FILE ".%d.dump", nEvent++);
464  fChain->DumpData(filename);
465  return (0);
466 }
