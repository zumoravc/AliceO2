60 template <typename FORMAT, typename CHARGESUM>
61 class PageDecoderImpl
62 {
63  public:
64  PageDecoderImpl(SampaChannelHandler sampaChannelHandler, FeeLink2SolarMapper fee2solar) : mSampaChannelHandler{sampaChannelHandler},
65  mFee2SolarMapper(fee2solar)

69  void operator()(Page page)
70  {
71  const void* rdhP = reinterpret_cast<const void*>(page.data());
72  if (!o2::raw::RDHUtils::checkRDH(rdhP, true)) {
73  throw std::invalid_argument("page does not start with a valid RDH");
74  }
75 
76  auto feeId = o2::raw::RDHUtils::getFEEID(rdhP);
77  auto linkId = o2::raw::RDHUtils::getLinkID(rdhP);
78  FeeLinkId feeLinkId(feeId & CRUID_MASK, linkId);
79 
80  auto p = mPayloadDecoders.find(feeLinkId);
81  if (p == mPayloadDecoders.end()) {
82  mPayloadDecoders.emplace(feeLinkId, PayloadDecoderImpl<FORMAT, CHARGESUM>()(feeLinkId, mSampaChannelHandler, mFee2SolarMapper));
83  p = mPayloadDecoders.find(feeLinkId);
84  }
85 
86  uint32_t orbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdhP);
87  auto rdhSize = o2::raw::RDHUtils::getHeaderSize(rdhP);
88  auto payloadSize = o2::raw::RDHUtils::getMemorySize(rdhP) - rdhSize;
89  p->second.process(orbit, page.subspan(rdhSize, payloadSize));
90  }
91 
92  private:
93  SampaChannelHandler mSampaChannelHandler;
94  FeeLink2SolarMapper mFee2SolarMapper;
95  std::map<FeeLinkId, typename PayloadDecoderImpl<FORMAT, CHARGESUM>::type> mPayloadDecoders;
96 };

100 PageDecoder createPageDecoder(RawBuffer rdhBuffer, SampaChannelHandler channelHandler, FeeLink2SolarMapper fee2solar)
101 {
102  const void* rdhP = reinterpret_cast<const void*>(rdhBuffer.data());
103  bool ok = o2::raw::RDHUtils::checkRDH(rdhP, true);
104  if (!ok) {
105  throw std::invalid_argument("rdhBuffer does not point to a valid RDH !");
106  }
107  auto linkId = o2::raw::RDHUtils::getLinkID(rdhP);
108  auto feeId = o2::raw::RDHUtils::getFEEID(rdhP);
109  if (linkId == 15) {
110  if (feeId & impl::CHARGESUM_MASK) {
111  return impl::PageDecoderImpl<UserLogicFormat, ChargeSumMode>(channelHandler, fee2solar);
112  } else {
113  return impl::PageDecoderImpl<UserLogicFormat, SampleMode>(channelHandler, fee2solar);
114  }
115  } else {
116  if (feeId & impl::CHARGESUM_MASK) {
117  return impl::PageDecoderImpl<BareFormat, ChargeSumMode>(channelHandler, fee2solar);
118  } else {
119  return impl::PageDecoderImpl<BareFormat, SampleMode>(channelHandler, fee2solar);
120  }
121  }
122 }

130 PageParser createPageParser()
131 {
132  return [](RawBuffer buffer, PageDecoder pageDecoder) {
133  size_t pos{0};
134  const void* rdhP = reinterpret_cast<const void*>(buffer.data());
135  bool ok = o2::raw::RDHUtils::checkRDH(rdhP, true);
136  if (!ok) {
137  throw std::invalid_argument("buffer does not start with a valid RDH !");
138  }
139  auto rdhSize = o2::raw::RDHUtils::getHeaderSize(rdhP);
140  while (pos < buffer.size_bytes() - rdhSize) {
141  const void* rdhP = reinterpret_cast<const void*>(buffer.data() + pos);
142  bool ok = o2::raw::RDHUtils::checkRDH(rdhP, true);
143  if (!ok) {
144  throw std::invalid_argument(fmt::format("buffer at pos {} does not point to a valid RDH !", pos));
145  }
146  auto payloadSize = o2::raw::RDHUtils::getMemorySize(rdhP) - rdhSize;
147  pageDecoder(buffer.subspan(pos, rdhSize + payloadSize));
148  pos += o2::raw::RDHUtils::getOffsetToNext(rdhP);
149  }
150  };
151 }
