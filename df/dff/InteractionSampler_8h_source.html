78 inline void InteractionSampler::generateCollisionTimes(std::vector<o2::InteractionTimeRecord>& dest)
79 {
80  // fill vector with interaction records
81  dest.clear();
82  for (int i = dest.capacity(); i--;) {
83  dest.push_back(generateCollisionTime());
84  }
85 }

88 inline void InteractionSampler::nextCollidingBC()
89 {
90  // increment bunch ID till next colliding bunch
91  do {
92  if (mIR.bc >= mBCMax) { // did we exhaust full orbit?
93  mIR.bc = mBCMin;
94  if (mIR.orbit == o2::constants::lhc::MaxNOrbits) { // wrap orbit (should not happen in run3)
95  warnOrbitWrapped();
96  }
97  mIR.orbit++;
98  } else {
99  mIR.bc++;
100  }
101  } while (!mBCFilling.testBC(mIR.bc));
102 }

105 inline int InteractionSampler::genPoissonZT()
106 {
107  // generate 0-truncated poisson number
108  // https://en.wikipedia.org/wiki/Zero-truncated_Poisson_distribution
109  int k = 1;
110  if (mMuBCZTRed > 0) {
111  double t = mMuBCZTRed, u = gRandom->Rndm(), s = t;
112  while (s < u) {
113  s += t *= mMuBC / (++k);
114  }
115  } else { // need to generate explicitly since the prob of
116  while (!(k = gRandom->Poisson(mMuBC))) {
117  };
118  }
119  return k;
120 }
