48 void Initializer::setup(const Options options, EventManager::EDataSource defaultDataSource)
49 {
50  TEnv settings;
51  ConfigurationManager::getInstance().getConfig(settings);
52 
53  const bool fullscreen = settings.GetValue("fullscreen.mode", false); // hide left and bottom tabs
54  const string ocdbStorage = settings.GetValue("OCDB.default.path", "local://$ALICE_ROOT/OCDB"); // default path to OCDB
55  LOG(INFO) << "Initializer -- OCDB path:" << ocdbStorage;
56 
57  auto& eventManager = EventManager::getInstance();
58  eventManager.setDataSourceType(defaultDataSource);
59  eventManager.setCdbPath(ocdbStorage);
60 
61  eventManager.registerDetector(new DataReaderTPC(), new DataInterpreterTPC(), EVisualisationGroup::TPC);
62  eventManager.registerDetector(new DataReaderITS(), new DataInterpreterITS(), EVisualisationGroup::ITS);
63 
64  eventManager.setDataSourceType(EventManager::EDataSource::SourceOffline);
65  eventManager.Open();
66 
67  GeometryManager::getInstance().setR2Geometry(std::string(settings.GetValue("simple.geom.default", "R3")).compare("R2") == 0);
68 
69  setupGeometry();
70  gSystem->ProcessEvents();
71  gEve->Redraw3D(true);
72 
73  setupBackground();
74 
75  // Setup windows size, fullscreen and focus
76  TEveBrowser* browser = gEve->GetBrowser();
77  browser->GetTabRight()->SetTab(1);
78  browser->MoveResize(0, 0, gClient->GetDisplayWidth(), gClient->GetDisplayHeight() - 32);
79 
80  browser->StartEmbedding(TRootBrowser::kBottom);
81  EventManagerFrame* frame = new EventManagerFrame(eventManager);
82  browser->StopEmbedding("EventCtrl Balbinka");
83 
84  if (fullscreen) {
85  ((TGWindow*)gEve->GetBrowser()->GetTabLeft()->GetParent())->Resize(1, 0);
86  ((TGWindow*)gEve->GetBrowser()->GetTabBottom()->GetParent())->Resize(0, 1);
87  }
88  gEve->GetBrowser()->Layout();
89  gSystem->ProcessEvents();
90 
91  setupCamera();
92 
93  // Temporary:
94  // For the time being we draw single random event on startup.
95  // Later this will be triggered by button, and finally moved to configuration.
96  gEve->AddEvent(&EventManager::getInstance());
97 
98  frame->DoFirstEvent();
99 }

101 void Initializer::setupGeometry()
102 {
103  // read path to geometry files from config file
104  TEnv settings;
105  ConfigurationManager::getInstance().getConfig(settings);
106 
107  // get geometry from Geometry Manager and register in multiview
108  auto multiView = MultiView::getInstance();
109  //auto geometry_enabled = GeometryManager::getInstance().getR2Geometry()? R2Visualisation:R3Visualisation;
110  for (int iDet = 0; iDet < NvisualisationGroups; ++iDet) {
111  if (GeometryManager::getInstance().getR2Geometry())
112  if (!R2Visualisation[iDet])
113  continue;
114  if (!GeometryManager::getInstance().getR2Geometry())
115  if (!R3Visualisation[iDet])
116  continue;
117 
118  EVisualisationGroup det = static_cast<EVisualisationGroup>(iDet);
119  string detName = gVisualisationGroupName[det];
120  if (settings.GetValue((detName + ".draw").c_str(), false)) {
121  if (detName == "TPC" || detName == "MCH" || detName == "MTR" || detName == "MID" || detName == "MFT" || detName == "AD0" || detName == "FMD") { // don't load MUON+MFT and AD and standard TPC to R-Phi view
122 
123  multiView->drawGeometryForDetector(detName, true, false);
124  } else if (detName == "RPH") { // special TPC geom from R-Phi view
125 
126  multiView->drawGeometryForDetector(detName, false, true, false);
127  } else { // default
128  multiView->drawGeometryForDetector(detName);
129  }
130  }
131  }
132 }

134 void Initializer::setupCamera()
135 {
136  // move and rotate sub-views
137  TEnv settings;
138  ConfigurationManager::getInstance().getConfig(settings);
139 
140  // read settings from config file
141  const double angleHorizontal = settings.GetValue("camera.3D.rotation.horizontal", -0.4);
142  const double angleVertical = settings.GetValue("camera.3D.rotation.vertical", 1.0);
143 
144  double zoom[MultiView::NumberOfViews];
145  zoom[MultiView::View3d] = settings.GetValue("camera.3D.zoom", 1.0);
146  zoom[MultiView::ViewRphi] = settings.GetValue("camera.R-Phi.zoom", 1.0);
147  zoom[MultiView::ViewZrho] = settings.GetValue("camera.Rho-Z.zoom", 1.0);
148 
149  // get necessary elements of the multiview and set camera position
150  auto multiView = MultiView::getInstance();
151 
152  for (int viewIter = 0; viewIter < MultiView::NumberOfViews; ++viewIter) {
153  TGLViewer* glView = multiView->getView(static_cast<MultiView::EViews>(viewIter))->GetGLViewer();
154  glView->CurrentCamera().Reset();
155 
156  if (viewIter == 0) {
157  glView->CurrentCamera().RotateRad(angleHorizontal, angleVertical);
158  }
159  glView->CurrentCamera().Dolly(zoom[viewIter], kFALSE, kTRUE);
160  }
161 }
