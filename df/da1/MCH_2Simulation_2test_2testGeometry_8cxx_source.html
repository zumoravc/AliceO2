71 std::vector<TGeoNode*> getSlatNodes()
72 {
73  std::vector<TGeoNode*> slats;
74  for (auto chname : slatChamberNames) {
75  auto vol = gGeoManager->GetVolume(chname.c_str());
76  TIter next(vol->GetNodes());
77  while (TGeoNode* node = static_cast<TGeoNode*>(next())) {
78  if (strstr(node->GetName(), "support") == nullptr) {
79  slats.push_back(node);
80  }
81  }
82  }
83  return slats;
84 }
