133 GBTLink::ErrorType GBTLink::checkErrorsRDH(const RDH& rdh)
134 {
135  if (!RDHUtils::checkRDH(rdh, true)) {
136  statistics.errorCounts[GBTS::ErrNoRDHAtStart]++;
137  if (verbosity >= VerboseErrors) {
138  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrNoRDHAtStart];
139  }
140  errorBits |= 0x1 << int(GBTS::ErrNoRDHAtStart);
141  return Abort; // fatal error
142  }
143  if (format == OldFormat && RDHUtils::getVersion(rdh) > 4) {
144  if (verbosity >= VerboseErrors) {
145  LOG(ERROR) << "Requested old format requires data with RDH version 3 or 4, RDH version "
146  << RDHUtils::getVersion(rdh) << " is found";
147  }
148  return Abort;
149  }
150  if ((RDHUtils::getPacketCounter(rdh) > packetCounter + 1) && packetCounter >= 0) {
151  statistics.errorCounts[GBTS::ErrPacketCounterJump]++;
152  if (verbosity >= VerboseErrors) {
153  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrPacketCounterJump]
154  << " : jump from " << int(packetCounter) << " to " << int(RDHUtils::getPacketCounter(rdh));
155  }
156  errorBits |= 0x1 << int(GBTS::ErrPacketCounterJump);
157  return Warning;
158  }
159  packetCounter = RDHUtils::getPacketCounter(rdh);
160  return NoError;
161 }

165 GBTLink::ErrorType GBTLink::checkErrorsRDHStop(const RDH& rdh)
166 {
167  if (format == NewFormat && lastRDH && RDHUtils::getHeartBeatOrbit(*lastRDH) != RDHUtils::getHeartBeatOrbit(rdh) // new HB starts
168  && !RDHUtils::getStop(*lastRDH)) {
169  statistics.errorCounts[GBTS::ErrPageNotStopped]++;
170  if (verbosity >= VerboseErrors) {
171  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrPageNotStopped];
172  RDHUtils::printRDH(*lastRDH);
173  RDHUtils::printRDH(rdh);
174  }
175  errorBits |= 0x1 << int(GBTS::ErrPageNotStopped);
176  return Warning;
177  }
178  return NoError;

215 GBTLink::ErrorType GBTLink::checkErrorsHeaderWord(const GBTDataHeader* gbtH)
216 {
217  if (!gbtH->isDataHeader()) { // check header word
218  statistics.errorCounts[GBTS::ErrMissingGBTHeader]++;
219  gbtH->printX();
220  if (verbosity >= VerboseErrors) {
221  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrMissingGBTHeader];
222  }
223  errorBits |= 0x1 << int(GBTS::ErrMissingGBTHeader);
224  return Abort;
225  }
226  int cnt = RDHUtils::getPageCounter(*lastRDH);
227  // RSTODO: this makes sense only for old format, where every trigger has its RDH
228  if (format == OldFormat && gbtH->packetIdx != cnt) {
229  statistics.errorCounts[GBTS::ErrRDHvsGBTHPageCnt]++;
230  if (verbosity >= VerboseErrors) {
231  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrRDHvsGBTHPageCnt] << ": diff in GBT header "
232  << gbtH->packetIdx << " and RDH page " << cnt << " counters";
233  }
234  errorBits |= 0x1<<int(GBTS::ErrRDHvsGBTHPageCnt);
235  return Warning;
236  }
237  // RSTODO CHECK
238  if (lanesActive == lanesStop) { // all lanes received their stop, new page 0 expected
239  //if (cnt) { // makes sens for old format only
240  if (gbtH->packetIdx) {
241  statistics.errorCounts[GBTS::ErrNonZeroPageAfterStop]++;
242  if (verbosity >= VerboseErrors) {
243  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrNonZeroPageAfterStop]
244  << ": Non-0 page counter (" << cnt << ") while all lanes were stopped";
245  }
246  errorBits |= 0x1 << int(GBTS::ErrNonZeroPageAfterStop);
247  return Warning;
248  }
249  }
250  return NoError;
251 }

323 GBTLink::ErrorType GBTLink::checkErrorsLanesStops()
324 {
325  // make sure all lane stops for finished page are received
326  auto err = NoError;
327  if ((lanesActive & ~lanesStop)) {
328  if (RDHUtils::getTriggerType(*lastRDH) != o2::trigger::SOT) { // only SOT trigger allows unstopped lanes?
329  statistics.errorCounts[GBTS::ErrUnstoppedLanes]++;
330  std::bitset<32> active(lanesActive), stopped(lanesStop);
331  if (verbosity >= VerboseErrors) {
332  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrUnstoppedLanes]
333  << " | active: " << active << " stopped: " << stopped;
334  }
335  errorBits |= 0x1 << int(GBTS::ErrUnstoppedLanes);
336  }
337  err = Warning;
338  }
339  // make sure all active lanes (except those in time-out) have sent some data
340  if ((~lanesWithData & lanesActive) != lanesTimeOut) {
341  std::bitset<32> withData(lanesWithData), active(lanesActive), timeOut(lanesTimeOut);
342  statistics.errorCounts[GBTS::ErrNoDataForActiveLane]++;
343  if (verbosity >= VerboseErrors) {
344  LOG(ERROR) << describe() << ' ' << statistics.ErrNames[GBTS::ErrNoDataForActiveLane]
345  << " | with data: " << withData << " active: " << active << " timeOut: " << timeOut;
346  }
347  errorBits |= 0x1 << int(GBTS::ErrNoDataForActiveLane);
348  err = Warning;
349  }
350  return err;
351 }
